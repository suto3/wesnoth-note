  56 game_stats::game_stats(const display_context& board, const int viewing_team, int& selected_index)
  57         : board_(board)
  58         , viewing_team_(board_.teams()[viewing_team])
  59         , selected_index_(selected_index)
  60 {
  61         for(const auto& team : board_.teams()) {
  62                 team_data_.push_back(board_.calculate_team_data(team));
  63         }
  64 }

  66 unit_const_ptr game_stats::get_leader(const int side)
  67 {
  68         unit_map::const_iterator leader = board_.units().find_leader(side);
  69 
  70         if(leader != board_.units().end()) {
  71                 return leader.get_shared_ptr();
  72         }
  73 
  74         return nullptr;
  75 }

  77 static std::string controller_name(const team& t) {
  78         static const t_string names[3] = {_("controller^Human"), _("controller^Computer"), _("controller^Idle")};
  79         return "    <span color='#808080'>(" + names[t.controller().v] + ")</span>";
  80 }

  82 void game_stats::pre_show(window& window)
  83 {
  84         listbox& stats_list    = find_widget<listbox>(&window, "game_stats_list", false);
  85         listbox& settings_list = find_widget<listbox>(&window, "scenario_settings_list", false);
  86 
  87         for(const auto& team : board_.teams()) {
  88                 if(team.hidden()) {
  89                         continue;
  90                 }
  91 
  92                 std::map<std::string, string_map> row_data_stats;
  93                 string_map column_stats;
  94 
  95                 const bool known = viewing_team_.knows_about_team(team.side() - 1);
  96                 const bool enemy = viewing_team_.is_enemy(team.side());
  97 
  98                 const team_data& data = team_data_[team.side() - 1];
  99 
 100                 unit_const_ptr leader = get_leader(team.side());
 101 
 102                 std::string leader_name;
 103                 std::string leader_image;
 104 
 105                 if(leader) {
 106                         const bool fogged = viewing_team_.fogged(leader->get_location());
 107 
 108                         // Add leader image. If it's fogged show only a random leader image.
 109                         if(!fogged || known || game_config::debug) {
 110                                 leader_image = leader->absolute_image() + leader->image_mods();
 111                                 leader_name  = leader->name();
 112                         } else {
 113                                 leader_image = formatter() << "units/unknown-unit.png" << "~RC(magenta>" << team.color() << ")";
 114                                 leader_name  = _("Unknown");
 115                         }
 116 
 117                         if(resources::controller) {
 118                                 if(resources::controller->get_classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 119                                         leader_name = team.side_name();
 120                                 }
 121                         }
 122 
 123                         leader_name = "<span color='" + team::get_side_highlight_pango(team.side() - 1) + "'>" + leader_name + "</span>";
 124                 }
 125 
 126                 //
 127                 // Status list
 128                 //
 129                 column_stats["use_markup"] = "true";
 130 
 131                 column_stats["label"] = leader_image;
 132                 row_data_stats.emplace("team_leader_image", column_stats);
 133 
 134                 column_stats["label"] = leader_name + "\n" + controller_name(team);
 135                 row_data_stats.emplace("team_leader_name", column_stats);
 136 
 137                 column_stats["label"] = data.teamname.empty() ? team.team_name() : data.teamname;
 138                 row_data_stats.emplace("team_name", column_stats);
 139 
 140                 // Only fill in the rest of the info if the side is known...
 141                 if(known || game_config::debug) {
 142                         std::string gold_str;
 143                         if(game_config::debug || !enemy || !viewing_team_.uses_fog()) {
 144                                 gold_str = utils::half_signed_value(data.gold);
 145                         }
 146 
 147                         column_stats["label"] = data.gold < 0 ? "<span color='#ff0000'>" + gold_str + "</span>" : gold_str;
 148                         row_data_stats.emplace("team_gold", column_stats);
 149 
 150                         std::string village_count = std::to_string(data.villages);
 151                         if(!viewing_team_.uses_fog() && !viewing_team_.uses_shroud()) {
 152                                 village_count += "/" + std::to_string(board_.map().villages().size());
 153                         }
 154 
 155                         column_stats["label"] = village_count;
 156                         row_data_stats.emplace("team_villages", column_stats);
 157 
 158                         column_stats["label"] = std::to_string(data.units);
 159                         row_data_stats.emplace("team_units", column_stats);
 160 
 161                         column_stats["label"] = std::to_string(data.upkeep);
 162                         row_data_stats.emplace("team_upkeep", column_stats);
 163 
 164                         const std::string income = utils::signed_value(data.net_income);
 165                         column_stats["label"] = data.net_income < 0 ? "<span color='#ff0000'>" + income + "</span>" : income;
 166                         row_data_stats.emplace("team_income", column_stats);
 167                 }
 168 
 169                 stats_list.add_row(row_data_stats);
 170 
 171                 //
 172                 // Settings list
 173                 //
 174                 std::map<std::string, string_map> row_data_settings;
 175                 string_map column_settings;
 176 
 177                 column_settings["use_markup"] = "true";
 178 
 179                 column_settings["label"] = leader_image;
 180                 row_data_settings.emplace("team_leader_image", column_settings);
 181 
 182                 column_settings["label"] = leader_name + "\n" + controller_name(team);
 183                 row_data_settings.emplace("team_leader_name", column_settings);
 184 
 185                 column_settings["label"] = std::to_string(team.side());
 186                 row_data_settings.emplace("team_side", column_settings);
 187 
 188                 column_settings["label"] = std::to_string(team.start_gold());
 189                 row_data_settings.emplace("team_start_gold", column_settings);
 190 
 191                 column_settings["label"] = std::to_string(team.base_income());
 192                 row_data_settings.emplace("team_base_income", column_settings);
 193 
 194                 column_settings["label"] = std::to_string(team.village_gold());
 195                 row_data_settings.emplace("team_village_gold", column_settings);
 196 
 197                 column_settings["label"] = std::to_string(team.village_support());
 198                 row_data_settings.emplace("team_village_support", column_settings);
 199 
 200                 column_settings["label"] = team.uses_fog() ? _("yes") : _("no");
 201                 row_data_settings.emplace("team_fog", column_settings);
 202 
 203                 column_settings["label"] = team.uses_shroud() ? _("yes") : _("no");;
 204                 row_data_settings.emplace("team_shroud", column_settings);
 205 
 206                 settings_list.add_row(row_data_settings);
 207         }
 208 
 209         // Sorting options for the status list
 210         stats_list.register_sorting_option(0, [this](const int i) {
 211                 unit_const_ptr leader = get_leader(i + 1);
 212                 return leader ? leader->name().str() : ""; });
 213 
 214         stats_list.register_sorting_option(1, [this](const int i) { return board_.teams()[i].user_team_name().str(); });
 215         stats_list.register_sorting_option(2, [this](const int i) { return board_.teams()[i].gold(); });
 216         stats_list.register_sorting_option(3, [this](const int i) { return board_.teams()[i].villages(); });
 217         stats_list.register_sorting_option(4, [this](const int i) { return team_data_[i].units; });
 218         stats_list.register_sorting_option(5, [this](const int i) { return team_data_[i].upkeep; });
 219         stats_list.register_sorting_option(6, [this](const int i) { return team_data_[i].net_income; });
 220 
 221         // Sorting options for the settings list
 222         settings_list.register_sorting_option(0, [this](const int i) {
 223                 unit_const_ptr leader = get_leader(i + 1);
 224                 return leader ? leader->name().str() : ""; });
 225 
 226         settings_list.register_sorting_option(1, [this](const int i) { return board_.teams()[i].side(); });
 227         settings_list.register_sorting_option(2, [this](const int i) { return board_.teams()[i].start_gold(); });
 228         settings_list.register_sorting_option(3, [this](const int i) { return board_.teams()[i].base_income(); });
 229         settings_list.register_sorting_option(4, [this](const int i) { return board_.teams()[i].village_gold(); });
 230         settings_list.register_sorting_option(5, [this](const int i) { return board_.teams()[i].village_support(); });
 231         settings_list.register_sorting_option(6, [this](const int i) { return board_.teams()[i].uses_fog(); });
 232         settings_list.register_sorting_option(7, [this](const int i) { return board_.teams()[i].uses_shroud(); });
 233 
 234         //
 235         // Set up tab control
 236         //
 237         listbox& tab_bar = find_widget<listbox>(&window, "tab_bar", false);
 238 
 239         window.keyboard_capture(&tab_bar);
 240 
 241 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 242         connect_signal_notify_modified(*tab_bar,
 243                         std::bind(&game_stats::on_tab_select,
 244                                 *this, std::ref(window)));
 245 #else
 246         tab_bar.set_callback_value_change(
 247                         dialog_callback<game_stats, &game_stats::on_tab_select>);
 248 #endif
 249 
 250         on_tab_select(window);
 251 }

 253 void game_stats::on_tab_select(window& window)
 254 {
 255         const int i = find_widget<listbox>(&window, "tab_bar", false).get_selected_row();
 256 
 257         find_widget<stacked_widget>(&window, "pager", false).select_layer(i);
 258 
 259         // There are only two tabs, so this is simple
 260         find_widget<label>(&window, "title", false).set_label(
 261                 i == 0 ? _("Current Status") : _("Scenario Settings")
 262         );
 263 }

 265 void game_stats::post_show(window& window)
 266 {
 267         if(get_retval() == window::OK) {
 268                 const int selected_tab = find_widget<listbox>(&window, "tab_bar", false).get_selected_row();
 269 
 270                 const std::string list_id = selected_tab == 0 ? "game_stats_list" : "scenario_settings_list";
 271                 selected_index_ = find_widget<listbox>(&window, list_id, false).get_selected_row();
 272         }
 273 }
