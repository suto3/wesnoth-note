  76         /// This is a wrapper for a list of weak pointers to handlers. It allows forward
  77         /// iterations of the list, with each element returned as a shared pointer.
  78         /// (Weak pointers that fail to lock are silently removed from the list.) These
  79         /// iterations can be used recursively, even when the innermost iteration might
  80         /// erase arbitrary elements from the list.
  81         ///
  82         /// The interface is not the standard list interface because that would be
  83         /// inconvenient. The functionality implemented is that required by Wesnoth.
  84         class  handler_list
  85         {
  86                 /// The weak pointers that are used internally.
  87                 typedef std::weak_ptr<event_handler> internal_ptr;
  88                 /// The underlying list.
  89                 typedef utils::smart_list<internal_ptr> list_t;
  90 
  91         public: // types
  92                 /// Handler list iterators are rather limited. They can be constructed
  93                 /// from a reference iterator (not default constructed), incremented,
  94                 /// and dereferenced. Consecutive dereferences are not guaranteed to
  95                 /// return the same element (if the list mutates between them, the next
  96                 /// element might be returned). An increment guarantees that the next
  97                 /// dereference will differ from the previous (unless at the end of the
  98                 /// list). The end of the list is indicated by dereferencing to a null
  99                 /// pointer.
 100                 class iterator
 101                 {
 102                         /// The current element.
 103                         list_t::iterator iter_;
 104 
 105                 public:
 106                         /// Initialized constructor (to be called by handler_list).
 107                         explicit iterator(const list_t::iterator & base_iter) :
 108                                 iter_(base_iter)
 109                         {}
 110 
 111                         /// Increment.
 112                         iterator & operator++()            { ++iter_; return *this; }
 113                         /// Dereference.
 114                         handler_ptr operator*();
 115                 };
 116                 friend class iterator;
 117                 typedef iterator const_iterator;
 118 
 119         public: // functions
 120                 /// Default constructor.
 121                 /// Note: This explicit definition is required (by the more pedantic
 122                 ///       compilers) in order to declare a default-constructed, static,
 123                 ///       and const variable in event_handlers::get(), in handlers.cpp.
 124                 handler_list() : data_() {}
 125 
 126                 const_iterator begin() const           { return iterator(const_cast<list_t &>(data_).begin()); }
 127                 // The above const_cast is so the iterator can remove obsolete entries.
 128                 const_iterator end() const             { return iterator(const_cast<list_t &>(data_).end()); }
 129 
 130                 // push_front() is probably unneeded, but I'll leave the code here, just in case.
 131                 // (These lists must be maintained in index order, which means pushing to the back.)
 132                 void push_front(const handler_ptr & p) { data_.push_front(internal_ptr(p)); }
 133                 void push_back(const handler_ptr & p)  { data_.push_back(internal_ptr(p)); }
 134 
 135                 void clear()                           { data_.clear(); }
 136 
 137         private:
 138                 /// No implementation of operator=() since smart_list does not support it.
 139                 handler_list & operator=(const handler_list &);
 140 
 141                 /// The actual list.
 142                 list_t data_;
 143         };
