  83 void holder::init( side_number side )
  84 {
  85         if (side_context_ == nullptr) {
  86                 side_context_ = new side_context_impl(side,cfg_);
  87         } else {
  88                 side_context_->set_side(side);
  89         }
  90         if (readonly_context_ == nullptr){
  91                 readonly_context_ = new readonly_context_impl(*side_context_,cfg_);
  92                 readonly_context_->on_readonly_context_create();
  93         }
  94         if (readwrite_context_ == nullptr){
  95                 readwrite_context_ = new readwrite_context_impl(*readonly_context_,cfg_);
  96         }
  97         if (default_ai_context_ == nullptr){
  98                 default_ai_context_ = new default_ai_context_impl(*readwrite_context_,cfg_);
  99         }
 100         if (!this->ai_){
 101                 ai_ = std::shared_ptr<ai_composite>(new ai_composite(*default_ai_context_,cfg_));
 102         }
 103 
 104         if (this->ai_) {
 105                 ai_->on_create();
 106                 for (config &mod_ai : cfg_.child_range("modify_ai")) {
 107                         if (!mod_ai.has_attribute("side")) {
 108                                 mod_ai["side"] = side;
 109                         }
 110                         modify_ai(mod_ai);
 111                 }
 112                 cfg_.clear_children("modify_ai");
 113 
 114                 std::vector<engine_ptr> engines = ai_->get_engines();
 115                 for (std::vector<engine_ptr>::iterator it = engines.begin(); it != engines.end(); ++it)
 116                 {
 117                         (*it)->set_ai_context(&(ai_->get_ai_context()));
 118                 }
 119 
 120         } else {
 121                 ERR_AI_MANAGER << describe_ai()<<"AI lazy initialization error!" << std::endl;
 122         }
 123 
 124 }

 151 void holder::modify_ai(const config &cfg)
 152 {
 153         if (!this->ai_) {
 154                 // if not initialized, initialize now.
 155                 get_ai_ref();
 156         }
 157         const std::string &act = cfg["action"];
 158         LOG_AI_MOD << "side "<< side_ << "        "<<act<<"_ai_component \""<<cfg["path"]<<"\""<<std::endl;
 159         DBG_AI_MOD << std::endl << cfg << std::endl;
 160         DBG_AI_MOD << "side "<< side_ << " before "<<act<<"_ai_component"<<std::endl << to_config() << std::endl;
 161         bool res = false;
 162         if (act == "add") {
 163                 res = component_manager::add_component(&*this->ai_,cfg["path"],cfg);
 164         } else if (act == "change") {
 165                 res = component_manager::change_component(&*this->ai_,cfg["path"],cfg);
 166         } else if (act == "delete") {
 167                 res = component_manager::delete_component(&*this->ai_,cfg["path"]);
 168         } else {
 169                 ERR_AI_MOD << "modify_ai tag has invalid 'action' attribute " << act << std::endl;
 170         }
 171         DBG_AI_MOD << "side "<< side_ << "  after [modify_ai]"<<act<<std::endl << to_config() << std::endl;
 172         if (!res) {
 173                 LOG_AI_MOD << act << "_ai_component failed"<< std::endl;
 174         } else {
 175                 LOG_AI_MOD << act << "_ai_component success"<< std::endl;
 176         }
 177 
 178 }

 180 void holder::append_ai(const config& cfg)
 181 {
 182         if(!this->ai_) {
 183                 get_ai_ref();
 184         }
 185         for(const config& aspect : cfg.child_range("aspect")) {
 186                 const std::string& id = aspect["id"];
 187                 for(const config& facet : aspect.child_range("facet")) {
 188                         ai_->add_facet(id, facet);
 189                 }
 190         }
 191         for(const config& goal : cfg.child_range("goal")) {
 192                 ai_->add_goal(goal);
 193         }
 194         for(const config& stage : cfg.child_range("stage")) {
 195                 if(stage["name"] != "empty") {
 196                         ai_->add_stage(stage);
 197                 }
 198         }
 199 }

 468 const std::string manager::evaluate_command( side_number side, const std::string& str )
 469 {
 470         //insert new command into history
 471         history_.emplace_back(history_item_counter_++,str);
 472 
 473         //prune history - erase 1/2 of it if it grows too large
 474         if (history_.size()>MAX_HISTORY_SIZE){
 475                 history_.erase(history_.begin(),history_.begin()+MAX_HISTORY_SIZE/2);
 476                 LOG_AI_MANAGER << "AI MANAGER: pruned history" << std::endl;
 477         }
 478 
 479         if (!should_intercept(str)){
 480                 ai_composite& ai = get_active_ai_for_side(side);
 481                 raise_gamestate_changed();
 482                 return ai.evaluate(str);
 483         }
 484 
 485         return internal_evaluate_command(side,str);
 486 }

 489 bool manager::should_intercept( const std::string& str )
 490 {
 491         if (str.length()<1) {
 492                 return false;
 493         }
 494         if (str.at(0)=='!'){
 495                 return true;
 496         }
 497         if (str.at(0)=='?'){
 498                 return true;
 499         }
 500         return false;
 501 
 502 }

 504 std::deque< command_history_item > manager::history_;
 505 long manager::history_item_counter_ = 1;
 506 
 507 //this is stub code to allow testing of basic 'history', 'repeat-last-command', 'add/remove/replace ai' capabilities.
 508 //yes, it doesn't look nice. but it is usable.
 509 //to be refactored at earliest opportunity
 510 ///@todo 1.9 extract to separate class which will use fai or lua parser
 511 const std::string manager::internal_evaluate_command( side_number side, const std::string& str ){
 512         const int MAX_HISTORY_VISIBLE = 30;
 513 
 514         //repeat last command
 515         if (str=="!") {
 516                         //this command should not be recorded in history
 517                         if (!history_.empty()){
 518                                 history_.pop_back();
 519                                 history_item_counter_--;
 520                         }
 521 
 522                         if (history_.empty()){
 523                                 return "AI MANAGER: empty history";
 524                         }
 525                         return evaluate_command(side, history_.back().get_command());//no infinite loop since '!' commands are not present in history
 526         };
 527         //show last command
 528         if (str=="?") {
 529                 //this command should not be recorded in history
 530                 if (!history_.empty()){
 531                         history_.pop_back();
 532                         history_item_counter_--;
 533                 }
 534 
 535                 if (history_.empty()){
 536                         return "AI MANAGER: History is empty";
 537                 }
 538 
 539                 int n = std::min<int>( MAX_HISTORY_VISIBLE, history_.size() );
 540                 std::stringstream strstream;
 541                 strstream << "AI MANAGER: History - last "<< n <<" commands:\n";
 542                 std::deque< command_history_item >::reverse_iterator j = history_.rbegin();
 543 
 544                 for (int cmd_id=n; cmd_id>0; --cmd_id){
 545                         strstream << j->get_number() << "    :" << j->get_command() << '\n';
 546                         ++j;//this is *reverse* iterator
 547                 }
 548 
 549                 return strstream.str();
 550         };
 551 
 552 
 553         std::vector< std::string > cmd = utils::parenthetical_split(str, ' ',"'","'");
 554 
 555         if (cmd.size()==3){
 556                 //!add_ai side file
 557                 if (cmd.at(0)=="!add_ai"){
 558                         side = std::stoi(cmd.at(1));
 559                         std::string file = cmd.at(2);
 560                         if (add_ai_for_side_from_file(side,file,false)){
 561                                 return std::string("AI MANAGER: added [")+manager::get_active_ai_identifier_for_side(side)+std::string("] AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 562                         } else {
 563                                 return std::string("AI MANAGER: failed attempt to add AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 564                         }
 565                 }
 566                 //!replace_ai side file
 567                 if (cmd.at(0)=="!replace_ai"){
 568                         side = std::stoi(cmd.at(1));
 569                         std::string file = cmd.at(2);
 570                         if (add_ai_for_side_from_file(side,file,true)){
 571                                         return std::string("AI MANAGER: added [")+manager::get_active_ai_identifier_for_side(side)+std::string("] AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 572                         } else {
 573                                         return std::string("AI MANAGER: failed attempt to add AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 574                         }
 575                 }
 576 
 577         } else if (cmd.size()==2){
 578                 //!remove_ai side
 579                 if (cmd.at(0)=="!remove_ai"){
 580                         side = std::stoi(cmd.at(1));
 581                         remove_ai_for_side(side);
 582                         return std::string("AI MANAGER: made an attempt to remove AI for side ")+std::to_string(side);
 583                 }
 584                 if (cmd.at(0)=="!"){
 585                         //this command should not be recorded in history
 586                         if (!history_.empty()){
 587                                 history_.pop_back();
 588                                 history_item_counter_--;
 589                         }
 590 
 591                         int command = std::stoi(cmd.at(1));
 592                         std::deque< command_history_item >::reverse_iterator j = history_.rbegin();
 593                         //yes, the iterator could be precisely positioned (since command numbers go 1,2,3,4,..). will do it later.
 594                         while ( (j!=history_.rend()) && (j->get_number()!=command) ){
 595                                 ++j;// this is *reverse* iterator
 596                         }
 597                         if (j!=history_.rend()){
 598                                 return evaluate_command(side,j->get_command());//no infinite loop since '!' commands are not present in history
 599                         }
 600                         return "AI MANAGER: no command with requested number found";
 601                 }
 602         } else if (cmd.size()==1){
 603                 if (cmd.at(0)=="!help") {
 604                         return
 605                                 "known commands:\n"
 606                                 "!    - repeat last command (? and ! do not count)\n"
 607                                 "! NUMBER    - repeat numbered command\n"
 608                                 "?    - show a history list\n"
 609                                 "!add_ai TEAM FILE    - add a AI to side (0 - command AI, N - AI for side #N) from file\n"
 610                                 "!remove_ai TEAM    - remove AI from side (0 - command AI, N - AI for side #N)\n"
 611                                 "!replace_ai TEAM FILE    - replace AI of side (0 - command AI, N - AI for side #N) from file\n"
 612                                 "!help    - show this help message";
 613                 }
 614         }
 615 
 616 
 617         return "AI MANAGER: nothing to do";
 618 }
