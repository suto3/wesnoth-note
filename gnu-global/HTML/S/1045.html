  67 static bool check_side_number(const team &t, const std::string &str)
  68 {
  69                 std::vector<std::pair<int,int> > ranges = utils::parse_ranges(str);
  70                 int side_number = t.side();
  71 
  72                 std::vector<std::pair<int,int> >::const_iterator range, range_end = ranges.end();
  73                 for (range = ranges.begin(); range != range_end; ++range) {
  74                         if(side_number >= range->first && side_number <= range->second) {
  75                                 return true;
  76                         }
  77                 }
  78                 return false;

  81 bool side_filter::match_internal(const team &t) const
  82 {
  83         assert(fc_);
  84 
  85         if (cfg_.has_attribute("side_in")) {
  86                 if (!check_side_number(t,cfg_["side_in"])) {
  87                         return false;
  88                 }
  89         }
  90         if (cfg_.has_attribute("side")) {
  91                 if (!check_side_number(t,cfg_["side"])) {
  92                         return false;
  93                 }
  94         }
  95         if (!side_string_.empty()) {
  96                 if (!check_side_number(t,side_string_)) {
  97                         return false;
  98                 }
  99         }
 100 
 101         config::attribute_value cfg_team_name = cfg_["team_name"];
 102         if (!cfg_team_name.blank()) {
 103                 const std::string& that_team_name = cfg_team_name;
 104                 const std::string& this_team_name = t.team_name();
 105 
 106                 if(std::find(this_team_name.begin(), this_team_name.end(), ',') == this_team_name.end()) {
 107                         if(this_team_name != that_team_name) return false;
 108                 }
 109                 else {
 110                         const std::vector<std::string>& these_team_names = utils::split(this_team_name);
 111                         bool search_futile = true;
 112                         for(const std::string& this_single_team_name : these_team_names) {
 113                                 if(this_single_team_name == that_team_name) {
 114                                         search_futile = false;
 115                                         break;
 116                                 }
 117                         }
 118                         if(search_futile) return false;
 119                 }
 120         }
 121 
 122         //Allow filtering on units
 123         if(cfg_.has_child("has_unit")) {
 124                 const vconfig & ufilt_cfg = cfg_.child("has_unit");
 125                 if (!ufilter_)
 126                         ufilter_.reset(new unit_filter(ufilt_cfg, fc_, flat_));
 127                 bool found = false;
 128                 for(const unit &u : fc_->get_disp_context().units()) {
 129                         if (u.side() != t.side()) {
 130                                 continue;
 131                         }
 132                         if (ufilter_->matches(u)) {
 133                                 found = true;
 134                                 break;
 135                         }
 136                 }
 137                 if(!found && ufilt_cfg["search_recall_list"].to_bool(false)) {
 138                         for(const unit_const_ptr & u : t.recall_list()) {
 139                                 scoped_recall_unit this_unit("this_unit", t.save_id(),t.recall_list().find_index(u->id()));
 140                                 if(ufilter_->matches(*u)) {
 141                                         found = true;
 142                                         break;
 143                                 }
 144                         }
 145                 }
 146                 if (!found) {
 147                         return false;
 148                 }
 149         }
 150 
 151         const vconfig& enemy_of = cfg_.child("enemy_of");
 152         if(!enemy_of.null()) {
 153                 if (!enemy_filter_)
 154                         enemy_filter_.reset(new side_filter(enemy_of, fc_));
 155                 const std::vector<int>& teams = enemy_filter_->get_teams();
 156                 if(teams.empty()) return false;
 157                 for(const int side : teams) {
 158                         if(!(fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 159                                 return false;
 160                 }
 161         }
 162 
 163         const vconfig& allied_with = cfg_.child("allied_with");
 164         if(!allied_with.null()) {
 165                 if (!allied_filter_)
 166                         allied_filter_.reset(new side_filter(allied_with, fc_));
 167                 const std::vector<int>& teams = allied_filter_->get_teams();
 168                 if(teams.empty()) return false;
 169                 for(const int side : teams) {
 170                         if((fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 171                                 return false;
 172                 }
 173         }
 174 
 175         const vconfig& has_enemy = cfg_.child("has_enemy");
 176         if(!has_enemy.null()) {
 177                 if (!has_enemy_filter_)
 178                         has_enemy_filter_.reset(new side_filter(has_enemy, fc_));
 179                 const std::vector<int>& teams = has_enemy_filter_->get_teams();
 180                 bool found = false;
 181                 for(const int side : teams) {
 182                         if((fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 183                         {
 184                                 found = true;
 185                                 break;
 186                         }
 187                 }
 188                 if (!found) return false;
 189         }
 190 
 191         const vconfig& has_ally = cfg_.child("has_ally");
 192         if(!has_ally.null()) {
 193                 if (!has_ally_filter_)
 194                         has_ally_filter_.reset(new side_filter(has_ally, fc_));
 195                 const std::vector<int>& teams = has_ally_filter_->get_teams();
 196                 bool found = false;
 197                 for(const int side : teams) {
 198                         if(!(fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 199                         {
 200                                 found = true;
 201                                 break;
 202                         }
 203                 }
 204                 if (!found) return false;
 205         }
 206 
 207 
 208         const config::attribute_value cfg_controller = cfg_["controller"];
 209         if (!cfg_controller.blank())
 210         {
 211                 if (resources::controller->is_networked_mp() && synced_context::is_synced()) {
 212                         ERR_NG << "ignoring controller= in SSF due to danger of OOS errors" << std::endl;
 213                 }
 214                 else {
 215                         bool found = false;
 216                         for(const std::string& controller : utils::split(cfg_controller))
 217                         {
 218                                 if(t.controller().to_string() == controller) {
 219                                         found = true;
 220                                 }
 221                         }
 222                         if(!found) {
 223                                 return false;
 224                         }
 225                 }
 226         }
 227 
 228         if (cfg_.has_attribute("formula")) {
 229                 try {
 230                         const wfl::team_callable callable(t);
 231                         const wfl::formula form(cfg_["formula"]);
 232                         if(!form.evaluate(callable).as_bool()) {
 233                                 return false;
 234                         }
 235                         return true;
 236                 } catch(wfl::formula_error& e) {
 237                         lg::wml_error() << "Formula error in side filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 238                         // Formulae with syntax errors match nothing
 239                         return false;
 240                 }
 241         }
 242 
 243         return true;
 244 }

 252 bool side_filter::match(const team& t) const
 253 {
 254         bool matches = match_internal(t);
 255 
 256         //handle [and], [or], and [not] with in-order precedence
 257         vconfig::all_children_iterator cond = cfg_.ordered_begin();
 258         vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 259         while (cond != cond_end) {
 260                 const std::string& cond_name = cond.get_key();
 261                 const vconfig& cond_cfg = cond.get_child();
 262 
 263                 //handle [and]
 264                 if(cond_name == "and")
 265                 {
 266                         matches = matches && side_filter(cond_cfg, fc_, flat_).match(t);
 267                 }
 268                 //handle [or]
 269                 else if(cond_name == "or")
 270                 {
 271                         matches = matches || side_filter(cond_cfg, fc_, flat_).match(t);
 272                 }
 273                 //handle [not]
 274                 else if(cond_name == "not")
 275                 {
 276                         matches = matches && !side_filter(cond_cfg, fc_, flat_).match(t);
 277                 }
 278                         ++cond;
 279         }
 280         return matches;
 281 }
