  50 void controller_base::handle_event(const SDL_Event& event)
  51 {
  52         if(gui::in_dialog()) {
  53                 return;
  54         }
  55         static const hotkey::hotkey_command& quit_hotkey = hotkey::hotkey_command::get_command_by_command(hotkey::HOTKEY_QUIT_GAME);
  56 
  57         switch(event.type) {
  58         case SDL_TEXTINPUT:
  59                 if(have_keyboard_focus()) {
  60                         hotkey::key_event(event, get_hotkey_command_executor());
  61                 }
  62                 break;
  63         case SDL_KEYDOWN:
  64                 // Detect key press events, unless there something that has keyboard focus
  65                 // in which case the key press events should go only to it.
  66                 if(have_keyboard_focus()) {
  67                         if(event.key.keysym.sym == SDLK_ESCAPE) {
  68                                 hotkey::execute_command(quit_hotkey, get_hotkey_command_executor());
  69                                 break;
  70                         }
  71 
  72                         process_keydown_event(event);
  73                         hotkey::key_event(event, get_hotkey_command_executor());
  74                         process_keyup_event(event);
  75                 } else {
  76                         process_focus_keydown_event(event);
  77                 }
  78                 break;
  79         case SDL_KEYUP:
  80                 process_keyup_event(event);
  81                 hotkey::key_event(event, get_hotkey_command_executor());
  82                 break;
  83         case SDL_JOYBUTTONDOWN:
  84                 process_keydown_event(event);
  85                 hotkey::jbutton_event(event, get_hotkey_command_executor());
  86                 break;
  87         case SDL_JOYHATMOTION:
  88                 process_keydown_event(event);
  89                 hotkey::jhat_event(event, get_hotkey_command_executor());
  90                 break;
  91         case SDL_MOUSEMOTION:
  92                 // Ignore old mouse motion events in the event queue
  93                 SDL_Event new_event;
  94                 if(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {
  95                         while(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {};
  96                         get_mouse_handler_base().mouse_motion_event(new_event.motion, is_browsing());
  97                 } else {
  98                         get_mouse_handler_base().mouse_motion_event(event.motion, is_browsing());
  99                 }
 100                 break;
 101         case SDL_MOUSEBUTTONDOWN:
 102                 process_keydown_event(event);
 103                 get_mouse_handler_base().mouse_press(event.button, is_browsing());
 104                 if (get_mouse_handler_base().get_show_menu()){
 105                         show_menu(get_display().get_theme().context_menu()->items(),event.button.x,event.button.y,true, get_display());
 106                 }
 107                 hotkey::mbutton_event(event, get_hotkey_command_executor());
 108                 break;
 109         case SDL_MOUSEBUTTONUP:
 110                 get_mouse_handler_base().mouse_press(event.button, is_browsing());
 111                 if (get_mouse_handler_base().get_show_menu()){
 112                         show_menu(get_display().get_theme().context_menu()->items(),event.button.x,event.button.y,true, get_display());
 113                 }
 114                 break;
 115         case SDL_MOUSEWHEEL:
 116                 get_mouse_handler_base().mouse_wheel(event.wheel.x, event.wheel.y, is_browsing());
 117                 break;
 118         default:
 119                 break;
 120         }
 121 }

 140 bool controller_base::handle_scroll(int mousex, int mousey, int mouse_flags, double x_axis, double y_axis)
 141 {
 142         sdl::window* window = CVideo::get_singleton().get_window();
 143         bool mouse_in_window = (window != nullptr && (window->get_flags() & SDL_WINDOW_MOUSE_FOCUS) != 0)
 144                 || preferences::get("scroll_when_mouse_outside", true);
 145         int scroll_speed = preferences::scroll_speed();
 146         int dx = 0, dy = 0;
 147         int scroll_threshold = (preferences::mouse_scroll_enabled())
 148                 ? preferences::mouse_scroll_threshold() : 0;
 149         for (const theme::menu& m : get_display().get_theme().menus()) {
 150                 if (sdl::point_in_rect(mousex, mousey, m.get_location())) {
 151                         scroll_threshold = 0;
 152                 }
 153         }
 154 
 155         // apply keyboard scrolling
 156         dy -= scroll_up_ * scroll_speed;
 157         dy += scroll_down_ * scroll_speed;
 158         dx -= scroll_left_ * scroll_speed;
 159         dx += scroll_right_ * scroll_speed;
 160 
 161         // scroll if mouse is placed near the edge of the screen
 162         if (mouse_in_window) {
 163                 if (mousey < scroll_threshold) {
 164                         dy -= scroll_speed;
 165                 }
 166                 if (mousey > get_display().h() - scroll_threshold) {
 167                         dy += scroll_speed;
 168                 }
 169                 if (mousex < scroll_threshold) {
 170                         dx -= scroll_speed;
 171                 }
 172                 if (mousex > get_display().w() - scroll_threshold) {
 173                         dx += scroll_speed;
 174                 }
 175         }
 176 
 177         // scroll with middle-mouse if enabled
 178         if ((mouse_flags & SDL_BUTTON_MMASK) != 0 && preferences::middle_click_scrolls()) {
 179                 const SDL_Point original_loc = get_mouse_handler_base().get_scroll_start();
 180 
 181                 if (get_mouse_handler_base().scroll_started()) {
 182                         const SDL_Rect& rect = get_display().map_outside_area();
 183                         if (sdl::point_in_rect(mousex, mousey,rect) &&
 184                                 get_mouse_handler_base().scroll_started()) {
 185                                 // Scroll speed is proportional from the distance from the first
 186                                 // middle click and scrolling speed preference.
 187                                 const double speed = 0.04 * sqrt(static_cast<double>(scroll_speed));
 188                                 const double snap_dist = 16; // Snap to horizontal/vertical scrolling
 189                                 const double x_diff = (mousex - original_loc.x);
 190                                 const double y_diff = (mousey - original_loc.y);
 191 
 192                                 if (std::fabs(x_diff) > snap_dist || std::fabs(y_diff) <= snap_dist) dx += speed * x_diff;
 193                                 if (std::fabs(y_diff) > snap_dist || std::fabs(x_diff) <= snap_dist) dy += speed * y_diff;
 194                         }
 195                 }
 196                 else { // Event may fire mouse down out of order with respect to initial click
 197                         get_mouse_handler_base().set_scroll_start(mousex, mousey);
 198                 }
 199         }
 200 
 201         // scroll with joystick
 202         dx += round_double( x_axis * scroll_speed);
 203         dy += round_double( y_axis * scroll_speed);
 204 
 205         return get_display().scroll(dx, dy);
 206 }

 208 void controller_base::play_slice(bool is_delay_enabled)
 209 {
 210         CKey key;
 211 
 212         if (plugins_context *l = get_plugins_context()) {
 213                 l->play_slice();
 214         }
 215 
 216         events::pump();
 217         events::raise_process_event();
 218         events::raise_draw_event();
 219 
 220         // Update sound sources before scrolling
 221         if (soundsource::manager *l = get_soundsource_man()) {
 222                 l->update();
 223         }
 224 
 225         const theme::menu* const m = get_display().menu_pressed();
 226         if(m != nullptr) {
 227                 const SDL_Rect& menu_loc = m->location(get_display().screen_area());
 228                 show_menu(m->items(),menu_loc.x+1,menu_loc.y + menu_loc.h + 1,false, get_display());
 229 
 230                 return;
 231         }
 232         const theme::action* const a = get_display().action_pressed();
 233         if(a != nullptr) {
 234                 const SDL_Rect& action_loc = a->location(get_display().screen_area());
 235                 execute_action(a->items(), action_loc.x+1, action_loc.y + action_loc.h + 1,false);
 236 
 237                 return;
 238         }
 239         auto str_vec = additional_actions_pressed();
 240         if (!str_vec.empty()) {
 241                 execute_action(str_vec, 0, 0, false);
 242                 return;
 243         }
 244 
 245         bool was_scrolling = scrolling_;
 246 
 247         std::pair<double, double> values = joystick_manager_.get_scroll_axis_pair();
 248         const double joystickx = values.first;
 249         const double joysticky = values.second;
 250 
 251         int mousex, mousey;
 252         Uint8 mouse_flags = SDL_GetMouseState(&mousex, &mousey);
 253 
 254         /* TODO fendrin enable after an axis choosing mechanism is implemented
 255         std::pair<double, double> values = joystick_manager_.get_mouse_axis_pair();
 256         mousex += values.first * 10;
 257         mousey += values.second * 10;
 258         SDL_WarpMouse(mousex, mousey);
 259         */
 260         scrolling_ = handle_scroll(mousex, mousey, mouse_flags, joystickx, joysticky);
 261 
 262         map_location highlighted_hex = get_display().mouseover_hex();
 263 
 264         /* TODO fendrin enable when the relative cursor movement is implemented well enough
 265         const map_location& selected_hex = get_display().selected_hex();
 266 
 267         if (selected_hex != map_location::null_location()) {
 268                 if (joystick_manager_.next_highlighted_hex(highlighted_hex, selected_hex)) {
 269                         get_mouse_handler_base().mouse_motion(0,0, true, true, highlighted_hex);
 270                         get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 271                         scrolling_ = true;
 272                 }
 273         } else */
 274 
 275         if (joystick_manager_.update_highlighted_hex(highlighted_hex)
 276                         && get_display().get_map().on_board(highlighted_hex)) {
 277                 get_mouse_handler_base().mouse_motion(0,0, true, true, highlighted_hex);
 278                 get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 279                 scrolling_ = true;
 280                 }
 281 
 282         get_display().draw();
 283 
 284         // be nice when window is not visible
 285         // NOTE should be handled by display instead, to only disable drawing
 286         sdl::window* window = CVideo::get_singleton().get_window();
 287         if (window != nullptr && is_delay_enabled && (window->get_flags() & SDL_WINDOW_SHOWN) == 0) {
 288                 CVideo::delay(200);
 289         }
 290 
 291         if (!scrolling_ && was_scrolling) {
 292                 // scrolling ended, update the cursor and the brightened hex
 293                 get_mouse_handler_base().mouse_update(is_browsing(), highlighted_hex);
 294         }
 295 }

 297 void controller_base::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool context_menu, display& disp)
 298 {
 299         hotkey::command_executor* cmd_exec = get_hotkey_command_executor();
 300         if(!cmd_exec) {
 301                 return;
 302         }
 303 
 304         std::vector<config> items;
 305         for(const config& c : items_arg) {
 306                 const std::string& id = c["id"];
 307                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
 308 
 309                 if(cmd_exec->can_execute_command(command) && (!context_menu || in_context_menu(command.id))) {
 310                         items.emplace_back(config_of("id", id));
 311                 }
 312         }
 313 
 314         if(items.empty()) {
 315                 return;
 316         }
 317 
 318         cmd_exec->show_menu(items, xloc, yloc, context_menu, disp);
 319 }

 321 void controller_base::execute_action(const std::vector<std::string>& items_arg, int xloc, int yloc, bool context_menu)
 322 {
 323         hotkey::command_executor * cmd_exec = get_hotkey_command_executor();
 324         if (!cmd_exec) {
 325                 return;
 326         }
 327 
 328         std::vector<std::string> items;
 329         for (const std::string& item : items_arg) {
 330 
 331                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(item);
 332                 if(cmd_exec->can_execute_command(command))
 333                         items.push_back(item);
 334         }
 335 
 336         if(items.empty())
 337                 return;
 338         cmd_exec->execute_action(items, xloc, yloc, context_menu, get_display());
 339 }

 348 const config& controller_base::get_theme(const config& game_config, std::string theme_name)
 349 {
 350         if (theme_name.empty()) theme_name = preferences::theme();
 351 
 352         if (const config &c = game_config.find_child("theme", "id", theme_name))
 353                 return c;
 354 
 355         ERR_DP << "Theme '" << theme_name << "' not found. Trying the default theme." << std::endl;
 356 
 357         if (const config &c = game_config.find_child("theme", "id", "Default"))
 358                 return c;
 359 
 360         ERR_DP << "Default theme not found." << std::endl;
 361 
 362         static config empty;
 363         return empty;
 364 }
