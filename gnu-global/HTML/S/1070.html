  91 /**
  92  * Event-specific constructor.
  93  * This iteration will go through all event handlers matching the given name
  94  * (including those defined via menu items).
  95  * An empty @a event_name will automatically match nothing.
  96  */
  97 manager::iteration::iteration(const std::string & event_name, manager & man) :
  98         main_list_(man.event_handlers_->get(event_name)),
  99         var_list_(man.event_handlers_->get()),
 100         event_name_(event_name),
 101         end_(man.event_handlers_->size()),
 102         current_is_known_(false),
 103         main_is_current_(false),
 104         main_it_(main_list_.begin()),
 105         var_it_(event_name.empty() ? var_list_.end() : var_list_.begin()),
 106         gamedata_(resources::gamedata)
 107 {
 108 }

 110 /**
 111  * Increment
 112  * Incrementing guarantees that the next dereference will differ from the
 113  * previous derference (unless the iteration is exhausted). However, multiple
 114  * increments between dereferences are allowed to have the same effect as a
 115  * single increment.
 116  */
 117 manager::iteration & manager::iteration::operator++()
 118 {
 119         if ( !current_is_known_ )
 120                 // Either *this has never been dereferenced, or we already incremented
 121                 // since the last dereference. We are allowed to ignore this increment.
 122                 return *this;
 123 
 124         // Guarantee a different element next dereference.
 125         if ( main_is_current_ )
 126                 ++main_it_;
 127         else
 128                 ++var_it_; // (We'll check for a name match when we dereference.)
 129 
 130         // We no longer know which list is current.
 131         current_is_known_ = false;
 132 
 133         // Done.
 134         return *this;
 135 }
 136 
 137 /**
 138  * Dereference
 139  * Will return a null pointer when the end of the iteration is reached.
 140  */
 141 handler_ptr manager::iteration::operator*()
 142 {
 143         // Get the candidate for the current element from the main list.
 144         handler_ptr main_ptr = *main_it_;
 145         handler_vec::size_type main_index = ptr_index(main_ptr);
 146 
 147         // Get the candidate for the current element from the var list.
 148         handler_ptr var_ptr = *var_it_;
 149         // (Loop while var_ptr would be chosen over main_ptr, but the name does not match.)
 150         while ( var_ptr  &&  var_ptr->index() < main_index  &&
 151                 !var_ptr->matches_name(event_name_, gamedata_) )
 152                 var_ptr = *++var_it_;
 153         handler_vec::size_type var_index = ptr_index(var_ptr);
 154 
 155         // Which list? (Index ties go to the main list.)
 156         current_is_known_ = main_index < end_  ||  var_index < end_;
 157         main_is_current_ = main_index <= var_index;
 158 
 159         if ( !current_is_known_ )
 160                 return handler_ptr(); // End of list; return a null pointer.
 161         else
 162                 return main_is_current_ ? main_ptr : var_ptr;
 163 }

 166 void manager::add_events(const config::const_child_itors &cfgs, const std::string& type)
 167 {
 168         if(!type.empty()) {
 169                 if(std::find(unit_wml_ids_.begin(),unit_wml_ids_.end(),type) != unit_wml_ids_.end()) return;
 170                 unit_wml_ids_.insert(type);
 171         }
 172         for(const config &new_ev : cfgs) {
 173                 if(type.empty() && new_ev["id"].empty())
 174                 {
 175                         WRN_NG << "attempt to add an [event] with empty id=, ignoring " << std::endl;
 176                         continue;
 177                 }
 178                 add_event_handler(new_ev);
 179         }
 180 }

 182 void manager::write_events(config& cfg) const
 183 {
 184         for(const handler_ptr &eh : *event_handlers_) {
 185                 if ( !eh || eh->is_menu_item() ) {
 186                         continue;
 187                 }
 188                 cfg.add_child("event", eh->get_config());
 189         }
 190 
 191         cfg["unit_wml_ids"] = utils::join(unit_wml_ids_);
 192         wml_menu_items_.to_config(cfg);
 193 }
