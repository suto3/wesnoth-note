  81 static void set_weapon_info(twindow& window
  82                 , const std::vector<battle_context>& weapons
  83                 , const int best_weapon)
  84 {
  85         tlistbox& weapon_list =
  86                         find_widget<tlistbox>(&window, "weapon_list", false);
  87         window.keyboard_capture(&weapon_list);
  88 
  89         const config empty;
  90         attack_type no_weapon(empty);
  91 
  92         for (size_t i = 0; i < weapons.size(); ++i) {
  93                 const battle_context::unit_stats& attacker =
  94                                 weapons[i].get_attacker_stats();
  95 
  96                 const battle_context::unit_stats& defender =
  97                                 weapons[i].get_defender_stats();
  98 
  99                 const attack_type& attacker_weapon = attack_type(*attacker.weapon);
 100                 const attack_type& defender_weapon = attack_type(
 101                                 defender.weapon ? *defender.weapon : no_weapon);
 102 
 103                 std::map<std::string, string_map> data;
 104                 string_map item;
 105 
 106                 item["label"] = attacker_weapon.name();
 107                 data.insert(std::make_pair("attacker_weapon", item));
 108 
 109                 item["label"] = defender_weapon.name();
 110                 data.insert(std::make_pair("defender_weapon", item));
 111 
 112                 weapon_list.add_row(data);
 113 
 114         }
 115 
 116         assert(best_weapon < static_cast<int>(weapon_list.get_item_count()));
 117         weapon_list.select_row(best_weapon);
 118 }
