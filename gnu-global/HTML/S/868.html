  28 /// This is an iterator class that extends an existing iterator by overriding
  29 /// dereference. Access to the underlying iterator is controlled, promoting
  30 /// a black-box approach.
  31 ///
  32 /// The expected use of this class is as a typedef.
  33 /// If used in conjunction with a typedef of const_iterator_extend (with the
  34 /// same template parameters), you get conversion to your const iterator.
  35 ///
  36 /// Value     = Intended value_type (result of dereferencing).
  37 /// Container = The container whose iterator is being extended.
  38 /// Deref     = A class whose member
  39 ///                 static const Value& eval(const Container::const_iterator &)
  40 ///             can be used for dereferencing. (The const's might seem odd, but
  41 ///             they allow the same class to be used for const_iterator_extend.)
  42 ///             A const_cast will be used to convert the return value to Value&.
  43 /// Key       = A class that unlocks the underlying iterator. (This way the
  44 ///             underlying iterator is not exposed to everyone.) If Key is
  45 ///             accessible, then the underlying iterator can be obtained via
  46 ///             get(const Key &).
  47 template <typename Value, class Container, class Deref, class Key>
  48 class iterator_extend
  49 {
  50 public:
  51         // Handy shortcut.
  52         typedef typename Container::iterator  base_iter_type;
  53 
  54         // Types required of an iterator:
  55         typedef Value        value_type;
  56         typedef value_type * pointer;
  57         typedef value_type & reference;
  58         typedef typename base_iter_type::difference_type   difference_type;
  59         typedef typename base_iter_type::iterator_category iterator_category;
  60 
  61         /// Default constructor
  62         iterator_extend() : iter_() {}
  63         /// Initialized constructor
  64         explicit iterator_extend(const base_iter_type & iter) : iter_(iter) {}
  65 
  66 
  67         // Comparison:
  68         bool operator==(const iterator_extend & that) const { return iter_ == that.iter_; }
  69         bool operator!=(const iterator_extend & that) const { return iter_ != that.iter_; }
  70         // For random-access iterators:
  71         bool operator<(const iterator_extend & that)  const { return iter_ < that.iter_; }
  72         bool operator>(const iterator_extend & that)  const { return iter_ > that.iter_; }
  73         bool operator<=(const iterator_extend & that) const { return iter_ <= that.iter_; }
  74         bool operator>=(const iterator_extend & that) const { return iter_ >= that.iter_; }
  75 
  76         // Dereference:
  77         reference operator*() const { return const_cast<Value &>(Deref::eval(iter_)); }
  78         pointer  operator->() const { return &*(*this); }
  79 
  80         // Increment/decrement:
  81         iterator_extend & operator++() { ++iter_; return *this; }
  82         iterator_extend & operator--() { --iter_; return *this; }
  83         iterator_extend operator++(int) { return iterator_extend(iter_++); }
  84         iterator_extend operator--(int) { return iterator_extend(iter_--); }
  85         // Arithmetic (for random-access iterators):
  86         iterator_extend & operator+=(difference_type n) { iter_ += n; return *this; }
  87         iterator_extend & operator-=(difference_type n) { iter_ -= n; return *this; }
  88         iterator_extend operator+(difference_type n) const { return iterator_extend(iter_ + n); }
  89         iterator_extend operator-(difference_type n) const { return iterator_extend(iter_ - n); }
  90         difference_type operator-(const iterator_extend & that) const { return iter_ - that.iter_; }
  91         difference_type operator-(const const_iterator_extend<Value, Container, Deref, Key> & that) const
  92         { return -(that - const_iterator_extend<Value, Container, Deref, Key>(*this)); }
  93         reference operator[](difference_type n) const { return *(*this + n); }
  94 
  95         // Allow access to the underlying iterator to those with the key.
  96         const base_iter_type & get(const Key &) const { return iter_; }
  97 
  98 private:
  99         /// The underlying base iterator.
 100         base_iter_type iter_;
 101 };

 104 /// This is a const_iterator class that extends an existing const_iterator by
 105 /// overriding dereference. Access to the underlying iterator is controlled,
 106 /// promoting a black-box approach.
 107 ///
 108 /// The expected use of this class is as a typedef.
 109 /// If used in conjunction with a typedef of iterator_extend (with the same
 110 /// template parameters), you get conversion from your regular iterator.
 111 ///
 112 /// Value     = Intended value_type, minus "const".
 113 /// Container = The container whose const_iterator is being extended.
 114 /// Deref     = A class whose member
 115 ///                 static const Value& eval(const Container::const_iterator &)
 116 ///             can be used for dereferencing. (This same class can be used
 117 ///             for iterator_extend.)
 118 /// Key       = A class that unlocks the underlying const_iterator. (This way
 119 ///             the underlying const_iterator is not exposed to everyone.) If
 120 ///             Key is accessible, then the underlying const_iterator can be
 121 ///             obtained via get(const Key &).
 122 template <typename Value, class Container, class Deref, class Key>
 123 class const_iterator_extend
 124 {
 125 public:
 126         // Handy shortcut.
 127         typedef typename Container::const_iterator  base_iter_type;
 128 
 129         // Types required of an iterator:
 130         typedef const Value  value_type;
 131         typedef value_type * pointer;
 132         typedef value_type & reference;
 133         typedef typename base_iter_type::difference_type   difference_type;
 134         typedef typename base_iter_type::iterator_category iterator_category;
 135 
 136         /// Default constructor
 137         const_iterator_extend() : iter_() {}
 138         /// Initialized constructor
 139         explicit const_iterator_extend(const base_iter_type & iter) : iter_(iter) {}
 140         /// Conversion from iterator_extend (same parameters).
 141         const_iterator_extend(const iterator_extend<Value, Container, Deref, Key> & iter) :
 142                 iter_(iter.get(Key()))
 143         {}
 144 
 145         // Comparison:
 146         bool operator==(const const_iterator_extend & that) const { return iter_ == that.iter_; }
 147         bool operator!=(const const_iterator_extend & that) const { return iter_ != that.iter_; }
 148         // For random-access iterators:
 149         bool operator<(const const_iterator_extend & that)  const { return iter_ < that.iter_; }
 150         bool operator>(const const_iterator_extend & that)  const { return iter_ > that.iter_; }
 151         bool operator<=(const const_iterator_extend & that) const { return iter_ <= that.iter_; }
 152         bool operator>=(const const_iterator_extend & that) const { return iter_ >= that.iter_; }
 153 
 154         // Dereference:
 155         reference operator*() const { return Deref::eval(iter_); }
 156         pointer  operator->() const { return &*(*this); }
 157 
 158         // Increment/decrement:
 159         const_iterator_extend & operator++() { ++iter_; return *this; }
 160         const_iterator_extend & operator--() { --iter_; return *this; }
 161         const_iterator_extend operator++(int) { return const_iterator_extend(iter_++); }
 162         const_iterator_extend operator--(int) { return const_iterator_extend(iter_--); }
 163         // Arithmetic (for random-access iterators):
 164         const_iterator_extend & operator+=(difference_type n) { iter_ += n; return *this; }
 165         const_iterator_extend & operator-=(difference_type n) { iter_ -= n; return *this; }
 166         const_iterator_extend operator+(difference_type n) const { return const_iterator_extend(iter_ + n); }
 167         const_iterator_extend operator-(difference_type n) const { return const_iterator_extend(iter_ - n); }
 168         difference_type operator-(const const_iterator_extend & that) const { return iter_ - that.iter_; }
 169         reference operator[](difference_type n) const { return *(*this + n); }

 172         const base_iter_type & get(const Key &) const { return iter_; }

 174 private:
 175         /// The underlying base iterator.
 176         base_iter_type iter_;

 182 /*
 183  * An example of how to use these iterators:
 184 
 185 #include <vector>
 186 
 187 class int_vect {
 188         // We will store a vector of pointers to int.
 189         typedef std::vector<int *> pint_vector;
 190         pint_vector data_;
 191 
 192         // We need a struct to control access to underlying iterators.
 193         // Also, it is convenient to put evaluation in a private struct.
 194         struct key {
 195                 // Here we define the conversion from a pint_vector iterator to int.
 196                 static const int& eval(const pint_vector::const_iterator & iter)
 197                 { return **iter; }
 198         };
 199 
 200 public:
 201         // To the public, we will look like a container of ints.
 202         typedef utils::iterator_extend      <int, pint_vector, key, key> iterator;
 203         typedef utils::const_iterator_extend<int, pint_vector, key, key> const_iterator;
 204         // This gives us iterators that dereference to _int_, but otherwise
 205         // behave like iterators of _pint_vector_. The first _key_ defines
 206         // how to dereference, while the second _key_ prevents others from
 207         // accessing underlying iterators (because key is private).
 208 
 209 public:
 210         // That's the basic definition. Let's move on to some uses, in the guise
 211         // of giving others access to our data. But first, so this class is not
 212         // completely useless, maybe we should be able to add data to it.
 213         void push_back(int i)
 214         {
 215                 data_.push_back(nullptr); // (This two-step approach is for exception safety.)
 216                 data_.back() = new int(i);
 217         }
 218 
 219         // Now we give read-write access via iterators.
 220         iterator begin() { return iterator(data_.begin()); } // Simple creation of iterators.
 221         iterator end()   { return iterator(data_.end()); }
 222         // We won't bother with the const versions for now.
 223 
 224         // Almost done. To avoid leaking memory, we'll need a destructor.
 225         ~int_vect()
 226         {
 227                 // Free all the memory we allocated.
 228                 for ( iterator it = begin(); it != end(); ++it )
 229                         delete *it.get(key()); // Accessing the underlying iterator.
 230         }
 231 };
