 430     def create_network(self):
 431         """
 432         Assuming that each node which has been added to the tree only has a
 433         valid list of children in unit.child_ids, also fill in unit.parent_ids
 434         and update the unit.children shortcut.
 435         """
 436 
 437         # Complete the network
 438         for uid, u in list(self.lookup.items()):
 439             for cid in u.child_ids:
 440                 c = self.lookup.get(cid, None)
 441                 if not c: continue
 442                 u.children.append(c)
 443                 if not uid in c.parent_ids:
 444                     c.parent_ids.append(uid)
 445 
 446         # Put all roots into the forest
 447         for uid, u in list(self.lookup.items()):
 448             if not u.parent_ids:
 449                 self.trees[uid] = u
 450 
 451         # Sanity check because some argGRRxxx addons have units who advance to
 452         # themselves.
 453 
 454         def recurse(u, already):
 455             already2 = already.copy()
 456             for c in u.children[:]:
 457                 already2[c.id] = True
 458                 if c.id in already:
 459                     error_message(
 460                         ("Warning: Unit %s advances to unit %s in a loop.\n" %
 461                         (u.id, c.id)) +
 462                         ("    Removing advancement %s.\n" % c.id))
 463                     u.children.remove(c)
 464             for c in u.children:
 465                 recurse(c, already2)
 466         for u in list(self.trees.values()):
 467             already = {u.id : True}
 468             recurse(u, already)
 469 
 470     def update(self):
 471         self.create_network()
 472 
 473         self.breadth = sum([x.update_breadth() for x in list(self.trees.values())])
 474         return self.breadth
