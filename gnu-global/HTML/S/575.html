 118 void debug_layout_graph::set_level(const std::string& level)
 119 {
 120         if(level.empty()) {
 121                 level_ = ALL; /** @todo Should default to 0. */
 122                 return;
 123         }
 124 
 125         std::vector<std::string> params = utils::split(level);
 126 
 127         for(const auto & param : params)
 128         {
 129                 if(param == "all") {
 130                         level_ = ALL;
 131                         // No need to look further eventhough invalid items are now
 132                         // ignored.
 133                         return;
 134                 } else if(param == "size") {
 135                         level_ |= SIZE_INFO;
 136                 } else if(param == "state") {
 137                         level_ |= STATE_INFO;
 138                 } else {
 139                         // logging might not be up yet.
 140                         std::cerr << "Unknown level '" << param << "' is ignored.\n";
 141                 }
 142         }
 143 }

 145 void debug_layout_graph::set_domain(const std::string& domain)
 146 {
 147         if(domain.empty()) {
 148                 // return error and die
 149                 domain_ = ALL; /** @todo Should default to 0. */
 150                 return;
 151         }
 152 
 153         std::vector<std::string> params = utils::split(domain);
 154 
 155         for(const auto & param : params)
 156         {
 157                 if(param == "all") {
 158                         domain_ = ALL;
 159                         // No need to look further eventhough invalid items are now
 160                         // ignored.
 161                         return;
 162                 } else if(param == "show") {
 163                         domain_ |= SHOW;
 164                 } else if(param == "layout") {
 165                         domain_ |= LAYOUT;
 166                 } else {
 167                         // logging might not be up yet.
 168                         std::cerr << "Unknown domain '" << param << "' is ignored.\n";
 169                 }
 170         }

 173 void debug_layout_graph::generate_dot_file(const std::string& generator,
 174                                                                                         const unsigned domain)
 175 {
 176         // domain == 0 must also evaluate to true.
 177         if((domain_ & domain) != domain) {
 178                 return;
 179         }
 180 
 181         std::string id = window_->id();
 182         if(!id.empty()) {
 183                 id += '_';
 184         }
 185         const std::string filename = filename_base_ + id
 186                                                                  + std::to_string(++sequence_number_)
 187                                                                  + "-" + generator + ".dot";
 188 
 189         std::ofstream file(filename.c_str());
 190 
 191         file << "//Basic layout graph for window id '" << window_->id()
 192                  << "' using definition '" << window_->definition_ << "'.\n"
 193                  << "digraph window {\n"
 194                  << "\tnode [shape=record, style=filled, fillcolor=\"bisque\"];\n"
 195                  << "\trankdir=LR;\n";
 196 
 197         widget_generate_info(file, window_, "root");
 198 
 199         file << "}\n";
 200 }

 202 void debug_layout_graph::widget_generate_info(std::ostream& out,
 203                                                                                            const widget* widget,
 204                                                                                            const std::string& id,
 205                                                                                            const bool embedded) const
 206 {
 207         assert(!id.empty());
 208 
 209         out << "\t" << id
 210                 << " [label=<<table border=\"0\" cellborder=\"1\" cellspacing=\"0\">";
 211 
 212         widget_generate_basic_info(out, widget);
 213         if(level_ & STATE_INFO)
 214                 widget_generate_state_info(out, widget);
 215         if(level_ & SIZE_INFO)
 216                 widget_generate_size_info(out, widget);
 217 
 218         out << "</table>>";
 219         if(embedded) {
 220                 out << ", fillcolor=\"palegoldenrod\"";
 221         }
 222         out << "];\n";
 223 
 224         const grid* grid = dynamic_cast<const grid*>(widget);
 225         if(!grid) {
 226                 const container_base* container = dynamic_cast<const container_base*>(widget);
 227 
 228                 if(container) {
 229 
 230                         widget_generate_info(out, &container->get_grid(), id + "_G", true);
 231                         out << "\t" << id << " -> " << id << "_G"
 232                                 << " [label=\"(grid)\"];\n";
 233                 }
 234 
 235                 const scrollbar_container* scrollbar_container
 236                                 = dynamic_cast<const scrollbar_container*>(widget);
 237 
 238                 if(scrollbar_container) {
 239                         widget_generate_info(
 240                                         out, scrollbar_container->content_grid_, id + "_C", true);
 241                         out << "\t" << id << " -> " << id << "_C"
 242                                 << " [label=\"(content)\"];\n";
 243                 }
 244 
 245                 const listbox* listbox = dynamic_cast<const listbox*>(widget);
 246                 if(listbox) {
 247                         assert(listbox->generator_);
 248                 }
 249 
 250                 const generator_base* generator = dynamic_cast<const generator_base*>(widget);
 251 
 252                 if(generator) {
 253                         for(size_t i = 0; i < generator->get_item_count(); ++i) {
 254 
 255                                 const std::string child_id = id + "_I_"
 256                                                                                          + std::to_string(i);
 257 
 258                                 widget_generate_info(out, &generator->item(i), child_id, true);
 259 
 260                                 out << "\t" << id << " -> " << child_id
 261                                         << " [label=\"(item)\"];\n";
 262                         }
 263                 }
 264         }
 265         if(grid) {
 266                 grid_generate_info(out, grid, id);
 267         }
 268 }

 270 static std::string format_label(std::string label)
 271 {
 272         if(label.size() > 50) {
 273                 label = label.substr(0, 50) + "...";
 274         }
 275 
 276         // Replace characters that break the dot file/
 277         std::replace(label.begin(), label.end(), '>', '_');
 278 
 279         return label;
 280 }

 395 void debug_layout_graph::grid_generate_info(std::ostream& out,
 396                                                                                          const grid* grid,
 397                                                                                          const std::string& parent_id) const
 398 {
 399         assert(!parent_id.empty());
 400 
 401         // maybe change the order to links, child, widgets so the output of the
 402         // dot file might look better.
 403 
 404         out << "\n\n\t// The children of " << parent_id << ".\n";
 405 
 406         for(unsigned row = 0; row < grid->get_rows(); ++row) {
 407                 for(unsigned col = 0; col < grid->get_cols(); ++col) {
 408 
 409                         const widget* widget = grid->child(row, col).widget();
 410                         assert(widget);
 411 
 412                         widget_generate_info(
 413                                         out, widget, get_child_widget_id(parent_id, row, col));
 414                 }
 415         }
 416 
 417         out << "\n\t// The grid child data of " << parent_id << ".\n";
 418 
 419         for(unsigned row = 0; row < grid->get_rows(); ++row) {
 420                 for(unsigned col = 0; col < grid->get_cols(); ++col) {
 421 
 422                         child_generate_info(out,
 423                                                                 grid->child(row, col),
 424                                                                 get_child_id(parent_id, row, col));
 425                 }
 426         }
 427 
 428 
 429         out << "\n\t// The links of " << parent_id << ".\n";
 430 
 431         for(unsigned row = 0; row < grid->get_rows(); ++row) {
 432                 for(unsigned col = 0; col < grid->get_cols(); ++col) {
 433 
 434                         // grid -> child
 435                         out << "\t" << parent_id << " -> "
 436                                 << get_child_id(parent_id, row, col) << " [label=\"(" << row
 437                                 << ',' << col << ")\"];\n";
 438 
 439                         // child -> widget
 440                         out << "\t" << get_child_id(parent_id, row, col) << " -> "
 441                                 << get_child_widget_id(parent_id, row, col) << ";\n";
 442                 }
 443         }
 444 }

 446 void debug_layout_graph::child_generate_info(std::ostream& out,
 447                                                                                           const grid::child& child,
 448                                                                                           const std::string& id) const
 449 {
 450         assert(!id.empty());
 451 
 452         unsigned flags = child.get_flags();
 453 
 454         out << "\t" << id << " [style=\"\", label=<<table border=\"0\" "
 455                                                  "cellborder=\"1\" cellspacing=\"0\">\n";
 456         out << "<tr><td>\n"
 457                 << "vertical flag=";
 458 
 459         switch(flags & grid::VERTICAL_MASK) {
 460                 case grid::VERTICAL_GROW_SEND_TO_CLIENT:
 461                         out << "send to client";
 462                         break;
 463                 case grid::VERTICAL_ALIGN_TOP:
 464                         out << "align to top";
 465                         break;
 466                 case grid::VERTICAL_ALIGN_CENTER:
 467                         out << "center";
 468                         break;
 469                 case grid::VERTICAL_ALIGN_BOTTOM:
 470                         out << "align to bottom";
 471                         break;
 472                 default:
 473                         out << "unknown value("
 474                                 << ((flags & grid::VERTICAL_MASK) >> grid::VERTICAL_SHIFT)
 475                                 << ")";
 476         }
 477 
 478         out << "\n</td></tr>\n"
 479                 << "<tr><td>\n"
 480                 << "horizontal flag=";
 481 
 482         switch(flags & grid::HORIZONTAL_MASK) {
 483                 case grid::HORIZONTAL_GROW_SEND_TO_CLIENT:
 484                         out << "send to client";
 485                         break;
 486                 case grid::HORIZONTAL_ALIGN_LEFT:
 487                         out << "align to left";
 488                         break;
 489                 case grid::HORIZONTAL_ALIGN_CENTER:
 490                         out << "center";
 491                         break;
 492                 case grid::HORIZONTAL_ALIGN_RIGHT:
 493                         out << "align to right";
 494                         break;
 495                 default:
 496                         out << "unknown value("
 497                                 << ((flags & grid::HORIZONTAL_MASK) >> grid::HORIZONTAL_SHIFT)
 498                                 << ")";
 499         }
 500 
 501         out << "\n</td></tr>\n"
 502                 << "<tr><td>\n"
 503                 << "border location=";
 504 
 505         if((flags & grid::BORDER_ALL) == 0) {
 506                 out << "none";
 507         } else if((flags & grid::BORDER_ALL) == grid::BORDER_ALL) {
 508                 out << "all";
 509         } else {
 510                 std::string result;
 511                 if(flags & grid::BORDER_TOP)
 512                         result += "top, ";
 513                 if(flags & grid::BORDER_BOTTOM)
 514                         result += "bottom, ";
 515                 if(flags & grid::BORDER_LEFT)
 516                         result += "left, ";
 517                 if(flags & grid::BORDER_RIGHT)
 518                         result += "right, ";
 519 
 520                 if(!result.empty()) {
 521                         result.resize(result.size() - 2);
 522                 }
 523 
 524                 out << result;
 525         }
 526 
 527         out << "\n</td></tr>\n"
 528                 << "<tr><td>\n"
 529                 << "border_size=" << child.get_border_size() << "\n</td></tr>\n";
 530 
 531         out << "</table>>];\n";
 532 }

 534 std::string debug_layout_graph::get_type(const widget* widget) const
 535 {
 536         const styled_widget* control = dynamic_cast<const control*>(widget);
 537         if(control) {
 538                 return control->get_control_type();
 539         } else {
 540                 const grid* grid = dynamic_cast<const grid*>(widget);
 541                 const generator_base* generator = dynamic_cast<const generator_base*>(widget);
 542 
 543                 if(grid) {
 544                         return "grid";
 545                 } else if(generator) {
 546                         return "generator";
 547                 } else {
 548                         return "unknown";
 549                 }
 550         }
 551 }
