  25 //Our definition of map labels being obscured is if the tile is obscured,
  26 //or the tile below is obscured. This is because in the case where the tile
  27 //itself is visible, but the tile below is obscured, the bottom half of the
  28 //tile will still be shrouded, and the label being drawn looks weird.
  29 inline bool is_shrouded(const display* disp, const map_location& loc)
  30 {
  31         return disp->shrouded(loc) || disp->shrouded(loc.get_direction(map_location::SOUTH));
  32 }

 138 const terrain_label* map_labels::set_label(const map_location& loc,
 139                                            const t_string& text,
 140                                            const int creator,
 141                                            const std::string& team_name,
 142                                            const color_t color,
 143                                            const bool visible_in_fog,
 144                                            const bool visible_in_shroud,
 145                                            const bool immutable,
 146                                            const std::string& category,
 147                                            const t_string& tooltip )
 148 {
 149         terrain_label* res = nullptr;
 150 
 151         // See if there is already a label in this location for this team.
 152         // (We do not use get_label_private() here because we might need
 153         // the label_map as well as the terrain_label.)
 154         team_label_map::iterator current_label_map = labels_.find(team_name);
 155         label_map::iterator current_label;
 156 
 157         if ( current_label_map != labels_.end()
 158                         && (current_label = current_label_map->second.find(loc)) != current_label_map->second.end() )
 159         {
 160                 // Found old checking if need to erase it
 161                 if(text.str().empty())
 162                 {
 163                         // Erase the old label.
 164                         delete current_label->second;
 165                         current_label_map->second.erase(current_label);
 166 
 167                         // Restore the global label in the same spot, if any.
 168                         if ( terrain_label* global_label = get_label_private(loc, "") )
 169                                 global_label->recalculate();
 170                 }
 171                 else
 172                 {
 173                         current_label->second->update_info(text, creator, tooltip, team_name, color, visible_in_fog, visible_in_shroud, immutable, category);
 174                         res = current_label->second;
 175                 }
 176         }
 177         else if(!text.str().empty())
 178         {
 179                 // See if we will be replacing a global label.
 180                 terrain_label* global_label = get_label_private(loc, "");
 181 
 182                 // Add the new label.
 183                 res = new terrain_label(text,
 184                                 creator,
 185                                 team_name,
 186                                 loc,
 187                                 *this,
 188                                 color,
 189                                 visible_in_fog,
 190                                 visible_in_shroud,
 191                                 immutable,
 192                                 category,
 193                                 tooltip);
 194                 add_label(loc, res);
 195 
 196                 // Hide the old label.
 197                 if ( global_label != nullptr )
 198                         global_label->recalculate();
 199         }
 200         categories_dirty = true;
 201         return res;
 202 }

 211 {
 212         team_label_map::iterator i = labels_.find(team_name);
 213         if (i != labels_.end())
 214         {
 215                 clear_map(i->second, force);
 216         }

 221                 clear_map(i->second, force);

 226 void map_labels::clear_map(label_map &m, bool force)
 227 {
 228         label_map::iterator i = m.begin();
 229         while (i != m.end())
 230         {
 231                 if (!i->second->immutable() || force) {
 232                         delete i->second;
 233                         m.erase(i++);
 234                 } else ++i;
 235         }
 236         categories_dirty = true;
 237 }

 289 const std::vector<std::string>& map_labels::all_categories() const {
 290         if(categories_dirty) {
 291                 categories_dirty = false;
 292                 categories.clear();
 293                 categories.push_back("team");
 294                 for(size_t i = 1; i <= resources::gameboard->teams().size(); i++) {
 295                         categories.push_back("side:" + std::to_string(i));
 296                 }
 297                 std::set<std::string> unique_cats;
 298                 for (const team_label_map::value_type& m : labels_) {
 299                         for (const label_map::value_type& l : m.second) {
 300                                 if(l.second->category().empty()) continue;
 301                                 unique_cats.insert("cat:" + l.second->category());
 302                         }
 303                 }
 304                 std::copy(unique_cats.begin(), unique_cats.end(), std::back_inserter(categories));
 305         }
 306         return categories;
 307 }

 361 void terrain_label::read(const config &cfg)
 362 {
 363         const variable_set &vs = *resources::gamedata;
 364         loc_ = map_location(cfg, &vs);
 365         color_t color = font::LABEL_COLOR;
 366 
 367         std::string tmp_color = cfg["color"];
 368 
 369         text_ = cfg["text"];
 370         tooltip_ = cfg["tooltip"];
 371         team_name_ = cfg["team_name"].str();
 372         visible_in_fog_ = cfg["visible_in_fog"].to_bool(true);
 373         visible_in_shroud_ = cfg["visible_in_shroud"].to_bool();
 374         immutable_ = cfg["immutable"].to_bool(true);
 375         category_ = cfg["category"].str();
 376 
 377         int side = cfg["side"].to_int(-1);
 378         if(side >= 0) {
 379                 creator_ = side - 1;
 380         } else if(cfg["side"].str() == "current") {
 381                 config::attribute_value current_side = vs.get_variable_const("side_number");
 382                 if(!current_side.empty()) {
 383                         creator_ = current_side.to_int();
 384                 }
 385         }
 386 
 387         text_ = utils::interpolate_variables_into_tstring(text_, vs); // Not moved to rendering, as that would depend on variables at render-time
 388         team_name_ = utils::interpolate_variables_into_string(team_name_, vs);
 389         tmp_color = utils::interpolate_variables_into_string(tmp_color, vs);
 390 
 391         if(!tmp_color.empty()) {
 392                 try {
 393                         color = color_t::from_rgb_string(tmp_color);
 394                 } catch(std::invalid_argument&) {
 395                         // Prior to the color_t conversion, labels were written to savefiles with an alpha key, despite alpha not
 396                         // being accepted in color=. Because of this, this enables the loading of older saves without an exception
 397                         // throwing.
 398                         color = color_t::from_rgba_string(tmp_color);
 399                 }
 400         }
 401 
 402         color_ = color;
 403 }

 510 void terrain_label::calculate_shroud()
 511 {
 512         if (handle_) {
 513                 font::show_floating_label(handle_, !hidden());
 514         }
 515 
 516         if (tooltip_.empty() || hidden()) {
 517                 tooltips::remove_tooltip(tooltip_handle_);
 518                 tooltip_handle_ = 0;
 519                 return;
 520         }
 521 
 522         //tooltips::update_tooltip(tooltip_handle, get_rect(), tooltip_.str(), "", true);
 523 
 524         if (tooltip_handle_)
 525                 tooltips::update_tooltip(tooltip_handle_,get_rect(), tooltip_.str(), "", true);
 526         else
 527                 tooltip_handle_ = tooltips::add_tooltip(get_rect(), tooltip_.str());
 528 }

 530 SDL_Rect terrain_label::get_rect() const
 531 {
 532         SDL_Rect rect {0,0,0,0};
 533 
 534         display* disp = display::get_singleton();
 535         if(!disp) {
 536                 return rect;
 537         }
 538 
 539         int hex_size = disp->hex_size();
 540 
 541         rect.x = disp->get_location_x(loc_) + hex_size / 4;
 542         rect.y = disp->get_location_y(loc_);
 543         rect.h = disp->hex_size();
 544         rect.w = disp->hex_size() - hex_size/2;
 545 
 546         return rect;
 547 }

 549 void terrain_label::draw()
 550 {
 551         display* disp = display::get_singleton();
 552         if(!disp) {
 553                 return;
 554         }
 555 
 556         if (text_.empty() && tooltip_.empty()) {
 557                 return;
 558         }
 559         clear();
 560 
 561         if (!viewable(disp->get_disp_context())) {
 562                 return;
 563         }
 564         //Note: the y part of loc_nextx is not used at all.
 565         const map_location loc_nextx = loc_.get_direction(map_location::NORTH_EAST);
 566         const map_location loc_nexty = loc_.get_direction(map_location::SOUTH);
 567         const int xloc = (disp->get_location_x(loc_) +
 568                         disp->get_location_x(loc_nextx)*2)/3;
 569         const int yloc = disp->get_location_y(loc_nexty) - font::SIZE_NORMAL;
 570 
 571         // If a color is specified don't allow to override it with markup. (prevents faking map labels for example)
 572         // FIXME: @todo Better detect if it's team label and not provided by
 573         // the scenario.
 574         bool use_markup = color_ == font::LABEL_COLOR;
 575 
 576         font::floating_label flabel(text_.str());
 577         flabel.set_color(color_);
 578         flabel.set_position(xloc, yloc);
 579         flabel.set_clip_rect(disp->map_outside_area());
 580         flabel.set_width(font::SIZE_NORMAL * 13);
 581         flabel.set_height(font::SIZE_NORMAL * 4);
 582         flabel.set_scroll_mode(font::ANCHOR_LABEL_MAP);
 583         flabel.use_markup(use_markup);
 584 
 585         handle_ = font::add_floating_label(flabel);
 586 
 587         calculate_shroud();
 588 }

 590 /**
 591  * This is a lightweight test used to see if labels are revealed as a result
 592  * of unit actions (i.e. fog/shroud clearing). It should not contain any tests
 593  * that are invariant during unit movement (disregarding potential WML events);
 594  * those belong in visible().
 595  */
 596 bool terrain_label::hidden() const
 597 {
 598         display* disp = display::get_singleton();
 599         if(!disp) {
 600                 return false;
 601         }
 602 
 603         // Respect user's label preferences
 604         std::string category = "cat:" + category_;
 605         std::string creator = "side:" + std::to_string(creator_ + 1);
 606         const std::vector<std::string>& hidden_categories = disp->get_disp_context().hidden_label_categories();
 607 
 608         if(std::find(hidden_categories.begin(), hidden_categories.end(), category) != hidden_categories.end())
 609                 return true;
 610         if(creator_ >= 0 && std::find(hidden_categories.begin(), hidden_categories.end(), creator) != hidden_categories.end())
 611                 return true;
 612         if(!team_name().empty() && std::find(hidden_categories.begin(), hidden_categories.end(), "team") != hidden_categories.end())
 613                 return true;
 614 
 615         // Fog can hide some labels.
 616         if ( !visible_in_fog_ && is_fogged(disp, loc_) )
 617                 return true;
 618 
 619         // Shroud can hide some labels.
 620         if ( !visible_in_shroud_ && is_shrouded(disp, loc_) )
 621                 return true;
 622 
 623         return false;
 624 }

 626 /**
 627  * This is a test used to see if we should bother with the overhead of actually
 628  * creating a label. Conditions that can change during unit movement (disregarding
 629  * potential WML events) should not be listed here; they belong in hidden().
 630  */
 631 bool terrain_label::viewable(const display_context & dc) const
 632 {
 633         if ( !parent_->enabled() )
 634                 return false;
 635 
 636         // In the editor, all labels are viewable.
 637         if ( dc.teams().empty() )
 638                 return true;
 639 
 640         // Observers are not privvy to team labels.
 641         const bool can_see_team_labels = !dc.is_observer();
 642 
 643         // Global labels are shown unless covered by a team label.
 644         if ( team_name_.empty() )
 645                 return !can_see_team_labels || parent_->visible_global_label(loc_);
 646 
 647         // Team labels are only shown to members of the team.
 648         return can_see_team_labels  &&  parent_->team_name() == team_name_;
 649 }
