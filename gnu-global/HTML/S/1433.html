 127     /**
 128      * Initializes the preferences to their default values
 129      */
 130     public static void initializeToDefault( )
 131     {
 132         IPreferenceStore store = WesnothPlugin.getDefault( )
 133             .getPreferenceStore( );
 134         // general settings
 135         store.setDefault( WESNOTH_EXEC_PATH, "" ); //$NON-NLS-1$
 136         store.setDefault( WESNOTH_WORKING_DIR, "" ); //$NON-NLS-1$
 137         store.setDefault( WESNOTH_USER_DIR, "" ); //$NON-NLS-1$
 138         store.setDefault( WESNOTH_WMLTOOLS_DIR, "" ); //$NON-NLS-1$
 139         store.setDefault( PYTHON_PATH, "" ); //$NON-NLS-1$
 140 
 141         // wml tools
 142         store.setDefault( WMLINDENT_VERBOSE, true );
 143         store.setDefault( WMLINDENT_DRYRUN, true );
 144 
 145         store.setDefault( WMLLINT_DRYRUN, true );
 146         store.setDefault( WMLLINT_SPELL_CHECK, false );
 147         store.setDefault( WMLLINT_VERBOSE_LEVEL, 0 );
 148 
 149         store.setDefault( WMLSCOPE_VERBOSE_LEVEL, 0 );
 150         store.setDefault( WMLSCOPE_COLLISIONS, false );
 151 
 152         // upload manager
 153         store.setDefault( ADDON_MANAGER_PASSWORD, "" ); //$NON-NLS-1$
 154         store.setDefault( ADDON_MANAGER_VERBOSE, false );
 155         store.setDefault( ADDON_MANAGER_ADDRESS, "add-ons.wesnoth.org" ); //$NON-NLS-1$
 156         store.setDefault( ADDON_MANAGER_PORT, 15006 );
 157 
 158         // advanced
 159         store.setDefault( NO_TERRAIN_GFX, true );
 160         store.setDefault( WML_VALIDATION, false );
 161 
 162         // installs
 163         store.setDefault( INST_DEFAULT_INSTALL, "" ); //$NON-NLS-1$
 164         store.setDefault( INST_INSTALL_LIST, "" ); //$NON-NLS-1$
 165     }

 272     /**
 273      * Returns a new Paths object based on the specified install name
 274      * 
 275      * @param installName
 276      *        The install name used for the paths
 277      * @return A new Paths object
 278      */
 279     public static Paths getPaths( String installName )
 280     {
 281         // no null allowed -> fallback to ""
 282         if( installName == null ) {
 283             installName = ""; //$NON-NLS-1$
 284         }
 285 
 286         Paths paths = paths_.get( installName );
 287         if( paths == null ) {
 288             paths = new Paths( installName, getInstallPrefix( installName ) );
 289             paths_.put( installName, paths );
 290         }
 291 
 292         return paths;
 293     }
