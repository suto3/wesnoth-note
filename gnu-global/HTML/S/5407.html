   7 #define RECALL_SUPPORTER
   8     # Recall a walk-on to say lines. Prefer whoever was tagged with
   9     # role=Supporter previously; if that unit is dead, pick anyone from side 1.
  10     [if]
  11         [have_unit]
  12             side=1
  13             role=Supporter
  14             search_recall_list=yes
  15         [/have_unit]
  16         [then]
  17             # Recall a supporter if we have one to do so
  18             [recall]
  19                 role=Supporter
  20             [/recall]
  21         [/then]
  22         [elseif]
  23             # Else, make a new supporter
  24             [have_unit]
  25                 side=1
  26                 [not]
  27                     id=Tallin
  28                 [/not]
  29                 search_recall_list=yes
  30             [/have_unit]
  31             [then]
  32                 [role]
  33                     side=1
  34                     [not]
  35                         id=Tallin
  36                     [/not]
  37                     role=Supporter
  38                 [/role]
  39 
  40                 [recall]
  41                     role=Supporter
  42                 [/recall]
  43             [/then]
  44         [/elseif]
  45         [else]
  46             # If that fails too, make a brand new supporter
  47             [unit]
  48                 type=Peasant
  49                 side=1
  50                 role=Supporter
  51                 animate=yes
  52                 placement=leader
  53             [/unit]
  54         [/else]
  55     [/if]
  56 #enddef
