 104 void mouse_handler::mouse_motion(int x, int y, const bool browse, bool update, map_location new_hex)
 105 {
 106         // we ignore the position coming from event handler
 107         // because it's always a little obsolete and we don't need
 108         // to highlight all the hexes where the mouse passed.
 109         // Also, sometimes it seems to have one *very* obsolete
 110         // and isolated mouse motion event when using drag&drop
 111         SDL_GetMouseState(&x, &y); // <-- modify x and y
 112 
 113         if(mouse_handler_base::mouse_motion_default(x, y, update)) {
 114                 return;
 115         }
 116 
 117         game_board& board = pc_.gamestate().board_;
 118 
 119         if(new_hex == map_location::null_location()) {
 120                 new_hex = gui().hex_clicked_on(x, y);
 121         }
 122 
 123         if(new_hex != last_hex_) {
 124                 if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 125                         lk->mouse_over_hex_callback(new_hex);
 126                 }
 127 
 128                 update = true;
 129 
 130                 if(pc_.get_map_const().on_board(last_hex_)) {
 131                         // we store the previous hexes used to propose attack direction
 132                         previous_hex_ = last_hex_;
 133 
 134                         // the hex of the selected unit is also "free"
 135                         { // start planned unit map scope
 136                                 wb::future_map_if_active raii;
 137                                 if(last_hex_ == selected_hex_ || !find_unit(last_hex_)) {
 138                                         previous_free_hex_ = last_hex_;
 139                                 }
 140                         } // end planned unit map scope
 141                 }
 142 
 143                 last_hex_ = new_hex;
 144         }
 145 
 146         if(reachmap_invalid_) {
 147                 update = true;
 148         }
 149 
 150         if(!update) {
 151                 return;
 152         }
 153 
 154         if(reachmap_invalid_) {
 155                 reachmap_invalid_ = false;
 156 
 157                 if(!current_paths_.destinations.empty() && !show_partial_move_) {
 158                         bool selected_hex_has_unit;
 159                         { // start planned unit map scope
 160                                 wb::future_map_if_active planned_unit_map;
 161                                 selected_hex_has_unit = find_unit(selected_hex_).valid();
 162                         } // end planned unit map scope
 163 
 164                         if(selected_hex_.valid() && selected_hex_has_unit) {
 165                                 // reselect the unit without firing events (updates current_paths_)
 166                                 select_hex(selected_hex_, true);
 167                         }
 168 
 169                         // we do never deselect here, mainly because of canceled attack-move
 170                 }
 171         }
 172 
 173         // reset current_route_ and current_paths if not valid anymore
 174         // we do it before cursor selection, because it uses current_paths_
 175         if(!pc_.get_map_const().on_board(new_hex)) {
 176                 current_route_.steps.clear();
 177                 gui().set_route(nullptr);
 178                 pc_.get_whiteboard()->erase_temp_move();
 179         }
 180 
 181         if(unselected_paths_) {
 182                 unselected_paths_ = false;
 183                 current_paths_ = pathfind::paths();
 184                 gui().unhighlight_reach();
 185         } else if(over_route_) {
 186                 over_route_ = false;
 187                 current_route_.steps.clear();
 188                 gui().set_route(nullptr);
 189                 pc_.get_whiteboard()->erase_temp_move();
 190         }
 191 
 192         gui().highlight_hex(new_hex);
 193         pc_.get_whiteboard()->on_mouseover_change(new_hex);
 194 
 195         unit_map::iterator selected_unit;
 196         unit_map::iterator mouseover_unit;
 197         map_location attack_from;
 198 
 199         { // start planned unit map scope
 200                 wb::future_map_if_active planned_unit_map;
 201                 selected_unit = find_unit(selected_hex_);
 202                 mouseover_unit = find_unit(new_hex);
 203 
 204                 // we search if there is an attack possibility and where
 205                 attack_from = current_unit_attacks_from(new_hex);
 206 
 207                 // see if we should show the normal cursor, the movement cursor, or
 208                 // the attack cursor
 209                 // If the cursor is on WAIT, we don't change it and let the setter
 210                 // of this state end it
 211                 if(cursor::get() != cursor::WAIT) {
 212                         if(selected_unit && selected_unit->side() == side_num_ && !selected_unit->incapacitated() && !browse) {
 213                                 if(attack_from.valid()) {
 214                                         cursor::set(dragging_started_ ? cursor::ATTACK_DRAG : cursor::ATTACK);
 215                                 } else if(!mouseover_unit && current_paths_.destinations.contains(new_hex)) {
 216                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 217                                 } else {
 218                                         // selected unit can't attack or move there
 219                                         cursor::set(cursor::NORMAL);
 220                                 }
 221                         } else {
 222                                 // no selected unit or we can't move it
 223 
 224                                 if(selected_hex_.valid() && mouseover_unit && mouseover_unit->side() == side_num_) {
 225                                         // empty hex field selected and unit on our site under the cursor
 226                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 227                                 } else {
 228                                         cursor::set(cursor::NORMAL);
 229                                 }
 230                         }
 231                 }
 232         } // end planned unit map scope
 233 
 234         // show (or cancel) the attack direction indicator
 235         if(attack_from.valid() && (!browse || pc_.get_whiteboard()->is_active())) {
 236                 gui().set_attack_indicator(attack_from, new_hex);
 237         } else {
 238                 gui().clear_attack_indicator();
 239         }
 240 
 241         unit_ptr un; // will later point to unit at mouseover_hex_
 242 
 243         // the destination is the pointed hex or the adjacent hex
 244         // used to attack it
 245         map_location dest;
 246         unit_map::const_iterator dest_un;
 247 
 248         { // start planned unit map scope
 249                 wb::future_map_if_active raii;
 250                 if(attack_from.valid()) {
 251                         dest = attack_from;
 252                         dest_un = find_unit(dest);
 253                 } else {
 254                         dest = new_hex;
 255                         dest_un = find_unit(new_hex);
 256                 }
 257 
 258                 if(dest == selected_hex_ || dest_un) {
 259                         current_route_.steps.clear();
 260                         gui().set_route(nullptr);
 261                         pc_.get_whiteboard()->erase_temp_move();
 262                 } else if(!current_paths_.destinations.empty() && board.map().on_board(selected_hex_)
 263                                   && board.map().on_board(new_hex)) {
 264                         if(selected_unit && !selected_unit->incapacitated()) {
 265                                 // Show the route from selected unit to mouseover hex
 266                                 current_route_ = get_route(&*selected_unit, dest, viewing_team());
 267 
 268                                 pc_.get_whiteboard()->create_temp_move();
 269 
 270                                 if(!browse) {
 271                                         gui().set_route(&current_route_);
 272                                 }
 273                         }
 274                 }
 275 
 276                 if(board.map().on_board(selected_hex_) && !selected_unit && mouseover_unit.valid() && mouseover_unit) {
 277                         // Show the route from selected hex to mouseover unit
 278                         current_route_ = get_route(&*mouseover_unit, selected_hex_, viewing_team());
 279 
 280                         pc_.get_whiteboard()->create_temp_move();
 281 
 282                         if(!browse) {
 283                                 gui().set_route(&current_route_);
 284                         }
 285                 } else if(!selected_unit) {
 286                         current_route_.steps.clear();
 287                         gui().set_route(nullptr);
 288                         pc_.get_whiteboard()->erase_temp_move();
 289                 }
 290 
 291                 unit_map::iterator iter = mouseover_unit;
 292                 if(iter) {
 293                         un = iter.get_shared_ptr();
 294                 } else {
 295                         un.reset();
 296                 }
 297         } // end planned unit map scope
 298 
 299         if((!selected_hex_.valid()) && un && current_paths_.destinations.empty() && !gui().fogged(un->get_location())) {
 300                 /*
 301                  * Only process unit if toggler not preventing normal unit
 302                  * processing. This can happen e.g. if, after activating 'show
 303                  * [best possible] enemy movements' through the UI menu, the
 304                  * mouse cursor lands on a hex with unit in it.
 305                  */
 306                 if(!preventing_units_highlight_) {
 307                         if(un->side() == side_num_) {
 308                                 // unit is on our team, show path if the unit has one
 309                                 const map_location go_to = un->get_goto();
 310                                 if(board.map().on_board(go_to)) {
 311                                         pathfind::marked_route route;
 312                                         { // start planned unit map scope
 313                                                 wb::future_map_if_active raii;
 314                                                 route = get_route(un.get(), go_to, current_team());
 315                                         } // end planned unit map scope
 316                                         gui().set_route(&route);
 317                                 }
 318                                 over_route_ = true;
 319 
 320                                 wb::future_map_if_active raii;
 321                                 current_paths_ = pathfind::paths(*un, false, true, viewing_team(), path_turns_);
 322                         } else {
 323                                 // unit under cursor is not on our team
 324                                 // Note: planned unit map must be activated after this is done,
 325                                 // since the future state includes changes to units' movement.
 326                                 unit_movement_resetter move_reset(*un);
 327 
 328                                 wb::future_map_if_active raii;
 329                                 current_paths_ = pathfind::paths(*un, false, true, viewing_team(), path_turns_);
 330                         }
 331 
 332                         unselected_paths_ = true;
 333                         gui().highlight_reach(current_paths_);
 334 
 335                 }
 336         }
 337 
 338         if(!un && preventing_units_highlight_) {
 339                 // Cursor on empty hex, turn unit highlighting back on.
 340                 enable_units_highlight();
 341         }
 342 }

 382 map_location mouse_handler::current_unit_attacks_from(const map_location& loc) const
 383 {
 384         if(loc == selected_hex_) {
 385                 return map_location();
 386         }
 387 
 388         bool wb_active = pc_.get_whiteboard()->is_active();
 389 
 390         {
 391                 // Check the unit SOURCE of the attack
 392 
 393                 // Check that there's a selected unit
 394                 const unit_map::const_iterator source_unit = find_unit(selected_hex_);
 395 
 396                 bool source_eligible = source_unit.valid();
 397                 if(!source_eligible) {
 398                         return map_location();
 399                 }
 400 
 401                 // The selected unit must at least belong to the player currently controlling this client.
 402                 source_eligible &= source_unit->side() == gui_->viewing_side();
 403                 if(!source_eligible) {
 404                         return map_location();
 405                 }
 406 
 407                 // In addition:
 408                 // - If whiteboard is enabled, we allow planning attacks outside of player's turn
 409                 // - If whiteboard is disabled, it must be the turn of the player controlling this client
 410                 if(!wb_active) {
 411                         source_eligible &= gui_->viewing_side() == pc_.current_side();
 412                         if(!source_eligible) {
 413                                 return map_location();
 414                         }
 415                 }
 416 
 417                 // Unit must have attacks left
 418                 source_eligible &= source_unit->attacks_left() != 0;
 419                 if(!source_eligible) {
 420                         return map_location();
 421                 }
 422 
 423                 // Check the unit TARGET of the attack
 424 
 425                 team const& viewer = viewing_team();
 426 
 427                 // Check that there's a unit at the target location
 428                 const unit_map::const_iterator target_unit = find_unit(loc);
 429 
 430                 bool target_eligible = target_unit.valid();
 431                 if(!target_eligible) {
 432                         return map_location();
 433                 }
 434 
 435                 // The player controlling this client must be an enemy of the target unit's side
 436                 target_eligible &= viewer.is_enemy(target_unit->side());
 437                 if(!target_eligible) {
 438                         return map_location();
 439                 }
 440 
 441                 // Sanity check: source and target of the attack shouldn't be on the same team
 442                 assert(source_unit->side() != target_unit->side());
 443 
 444                 target_eligible &= !target_unit->incapacitated();
 445                 if(!target_eligible) {
 446                         return map_location();
 447                 }
 448         }
 449 
 450         const map_location::DIRECTION preferred = loc.get_relative_dir(previous_hex_);
 451         const map_location::DIRECTION second_preferred = loc.get_relative_dir(previous_free_hex_);
 452 
 453         int best_rating = 100; // smaller is better
 454 
 455         map_location res;
 456         map_location adj[6];
 457         get_adjacent_tiles(loc, adj);
 458 
 459         for(size_t n = 0; n != 6; ++n) {
 460                 if(pc_.gamestate().board_.map().on_board(adj[n]) == false) {
 461                         continue;
 462                 }
 463 
 464                 if(adj[n] != selected_hex_ && find_unit(adj[n])) {
 465                         continue;
 466                 }
 467 
 468                 if(current_paths_.destinations.contains(adj[n])) {
 469                         static const size_t NDIRECTIONS = map_location::NDIRECTIONS;
 470 
 471                         unsigned int difference = std::abs(int(preferred - n));
 472                         if(difference > NDIRECTIONS / 2) {
 473                                 difference = NDIRECTIONS - difference;
 474                         }
 475 
 476                         unsigned int second_difference = std::abs(int(second_preferred - n));
 477                         if(second_difference > NDIRECTIONS / 2) {
 478                                 second_difference = NDIRECTIONS - second_difference;
 479                         }
 480 
 481                         const int rating = difference * 2 + (second_difference > difference);
 482                         if(rating < best_rating || res.valid() == false) {
 483                                 best_rating = rating;
 484                                 res = adj[n];
 485                         }
 486                 }
 487         }
 488 
 489         return res;
 490 }

 492 pathfind::marked_route mouse_handler::get_route(const unit* un, map_location go_to, team& team) const
 493 {
 494         game_board& board = pc_.gamestate().board_;
 495 
 496         // The pathfinder will check unit visibility (fogged/stealthy).
 497         const pathfind::shortest_path_calculator calc(*un, team, board.teams(), board.map());
 498 
 499         pathfind::teleport_map allowed_teleports = pathfind::get_teleport_locations(*un, viewing_team());
 500 
 501         pathfind::plain_route route;
 502 
 503         route = pathfind::a_star_search(
 504                 un->get_location(), go_to, 10000.0, calc, board.map().w(), board.map().h(), &allowed_teleports);
 505 
 506         return mark_route(route);
 507 }

 524 void mouse_handler::select_or_action(bool browse)
 525 {
 526         if(!pc_.get_map_const().on_board(last_hex_)) {
 527                 tooltips::click(drag_from_x_, drag_from_y_);
 528                 return;
 529         }
 530 
 531         // Load whiteboard partial moves
 532         wb::future_map_if_active planned_unit_map;
 533 
 534         if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 535                 lk->select_hex_callback(last_hex_);
 536         }
 537 
 538         unit_map::iterator clicked_u = find_unit(last_hex_);
 539         unit_map::iterator selected_u = find_unit(selected_hex_);
 540 
 541         if(clicked_u && (!selected_u || selected_u->side() != side_num_ ||
 542           (clicked_u->side() == side_num_ && clicked_u->id() != selected_u->id()))
 543         ) {
 544                 select_hex(last_hex_, false);
 545         } else {
 546                 move_action(browse);
 547         }
 548 }

 550 void mouse_handler::move_action(bool browse)
 551 {
 552         // Lock whiteboard activation state to avoid problems due to
 553         // its changing while an animation takes place.
 554         wb::whiteboard_lock wb_lock = pc_.get_whiteboard()->get_activation_state_lock();
 555 
 556         // we use the last registered highlighted hex
 557         // since it's what update our global state
 558         map_location hex = last_hex_;
 559 
 560         // TODO
 561         //      // Clicks on border hexes mean to deselect.
 562         //      // (Check this before doing processing that might not be needed.)
 563         //      if ( !pc_.get_map_const().on_board(hex) ) {
 564         //              deselect_hex();
 565         //              return false;
 566         //      }
 567 
 568         unit_map::iterator u;
 569         unit_map::iterator clicked_u;
 570 
 571         map_location src;
 572         pathfind::paths orig_paths;
 573         map_location attack_from;
 574 
 575         { // start planned unit map scope
 576                 wb::future_map_if_active planned_unit_map;
 577                 u = find_unit(selected_hex_);
 578 
 579                 // if the unit is selected and then itself clicked on,
 580                 // any goto command is canceled
 581                 if(u && !browse && selected_hex_ == hex && u->side() == side_num_) {
 582                         u->set_goto(map_location());
 583                 }
 584 
 585                 clicked_u = find_unit(hex);
 586 
 587                 src = selected_hex_;
 588                 orig_paths = current_paths_;
 589                 attack_from = current_unit_attacks_from(hex);
 590         } // end planned unit map scope
 591 
 592         // see if we're trying to do a attack or move-and-attack
 593         if((!browse || pc_.get_whiteboard()->is_active()) && attack_from.valid()) {
 594                 // Ignore this command if commands are disabled.
 595                 if(commands_disabled) {
 596                         return;
 597                 }
 598 
 599                 if(((u.valid() && u->side() == side_num_) || pc_.get_whiteboard()->is_active()) && clicked_u.valid()) {
 600                         if(attack_from == selected_hex_) { // no move needed
 601                                 int choice = -1;
 602                                 {
 603                                         wb::future_map_if_active planned_unit_map; // start planned unit map scope
 604                                         choice = show_attack_dialog(attack_from, clicked_u->get_location());
 605                                 } // end planned unit map scope
 606 
 607                                 if(choice >= 0) {
 608                                         if(pc_.get_whiteboard()->is_active()) {
 609                                                 save_whiteboard_attack(attack_from, clicked_u->get_location(), choice);
 610                                         } else {
 611                                                 // clear current unit selection so that any other unit selected
 612                                                 // triggers a new selection
 613                                                 selected_hex_ = map_location();
 614 
 615                                                 attack_enemy(u->get_location(), clicked_u->get_location(), choice);
 616                                         }
 617                                 }
 618 
 619                                 return;
 620                         } else {
 621                                 int choice = -1; // for the attack dialog
 622 
 623                                 {
 624                                         wb::future_map_if_active planned_unit_map; // start planned unit map scope
 625                                         // we will now temporary move next to the enemy
 626                                         pathfind::paths::dest_vect::const_iterator itor = current_paths_.destinations.find(attack_from);
 627                                         if(itor == current_paths_.destinations.end()) {
 628                                                 // can't reach the attacking location
 629                                                 // not supposed to happen, so abort
 630                                                 return;
 631                                         }
 632 
 633                                         // block where we temporary move the unit
 634                                         {
 635                                                 temporary_unit_mover temp_mover(
 636                                                                 pc_.gamestate().board_.units_, src, attack_from, itor->move_left);
 637                                                 choice = show_attack_dialog(attack_from, clicked_u->get_location());
 638                                         }
 639 
 640                                         if(choice < 0) {
 641                                                 // user hit cancel, don't start move+attack
 642                                                 return;
 643                                         }
 644                                 } // end planned unit map scope
 645 
 646                                 if(pc_.get_whiteboard()->is_active()) {
 647                                         save_whiteboard_attack(attack_from, hex, choice);
 648                                 } else {
 649                                         bool not_interrupted = move_unit_along_current_route();
 650                                         bool alt_unit_selected = (selected_hex_ != src);
 651                                         src = selected_hex_;
 652                                         // clear current unit selection so that any other unit selected
 653                                         // triggers a new selection
 654                                         selected_hex_ = map_location();
 655 
 656                                         if(not_interrupted)
 657                                                 attack_enemy(attack_from, hex, choice); // Fight !!
 658 
 659                                         // TODO: Maybe store the attack choice so "press t to continue"
 660                                         //      can also continue the attack?
 661 
 662                                         if(alt_unit_selected && !selected_hex_.valid()) {
 663                                                 // reselect other unit if selected during movement animation
 664                                                 select_hex(src, browse);
 665                                         }
 666                                 }
 667 
 668                                 return;
 669                         }
 670                 }
 671         }
 672         // otherwise we're trying to move to a hex
 673         else if(
 674                 // The old use case: move selected unit to mouse hex field.
 675                 (
 676                         (!browse || pc_.get_whiteboard()->is_active())
 677                         && selected_hex_.valid()
 678                         && selected_hex_ != hex
 679                         && u.valid()
 680                         && (u->side() == side_num_ || pc_.get_whiteboard()->is_active())
 681                         && !clicked_u
 682                         && !current_route_.steps.empty()
 683                         && current_route_.steps.front() == selected_hex_
 684                 )
 685                 || // The new use case: move mouse unit to selected hex field.
 686                 (
 687                         (!browse || pc_.get_whiteboard()->is_active())
 688                         && selected_hex_.valid()
 689                         && selected_hex_ != hex
 690                         && clicked_u
 691                         && !current_route_.steps.empty()
 692                         && current_route_.steps.back() == selected_hex_
 693                         && !u
 694                         && clicked_u->side() == side_num_
 695                 )
 696         ) {
 697                 // Ignore this command if commands are disabled.
 698                 if(commands_disabled) {
 699                         return;
 700                 }
 701 
 702                 // If the whiteboard is active, it intercepts any unit movement.
 703                 if(pc_.get_whiteboard()->is_active()) {
 704                         // Deselect the current hex, and create planned move for whiteboard.
 705                         selected_hex_ = map_location();
 706 
 707                         gui().select_hex(map_location());
 708                         gui().clear_attack_indicator();
 709                         gui().set_route(nullptr);
 710 
 711                         show_partial_move_ = false;
 712 
 713                         gui().unhighlight_reach();
 714 
 715                         current_paths_ = pathfind::paths();
 716                         current_route_.steps.clear();
 717 
 718                         pc_.get_whiteboard()->save_temp_move();
 719 
 720                         // Otherwise proceed to normal unit movement
 721                 } else {
 722                         // Don't move if the unit already has actions
 723                         // from the whiteboard.
 724                         if(pc_.get_whiteboard()->unit_has_actions(u ? &*u : &*clicked_u)) {
 725                                 return;
 726                         }
 727 
 728                         move_unit_along_current_route();
 729 
 730                         // During the move, we may have selected another unit
 731                         // (but without triggering a select event (command was disabled)
 732                         // in that case reselect it now to fire the event (+ anim & sound)
 733                         if(selected_hex_ != src) {
 734                                 select_hex(selected_hex_, browse);
 735                         }
 736                 }
 737 
 738                 return;
 739         }
 740 }

 784                                         // end forced real unit map
 785                                 }
 786                         }
 787                 }
 788 
 789                 return;
 790         }
 791 
 792         if(selected_hex_.valid() && !unit) {
 793                 // Compute unit in range of the empty selected_hex field
 794 
 795                 gui_->unhighlight_reach();

 803                                 ++u) {

 808                                         pathfind::paths(*u, false, true, viewing_team(), path_turns_, false, false);

 810                                 if(path.destinations.find(hex) != path.destinations.end()) {
 811                                         reaching_unit_locations.destinations.insert(u->get_location());
 812                                         gui_->highlight_another_reach(clicked_location);
 813                                 }
 814                         }
 815                 }
 816 
 817                 gui_->highlight_another_reach(reaching_unit_locations);
 818         } else {
 819                 if(!pc_.gamestate().board_.units_.find(last_hex_)) {
 820                         unselected_reach_ = gui_->unhighlight_reach();
 821                 }
 822 
 823                 current_paths_ = pathfind::paths();
 824                 current_route_.steps.clear();
 825 
 826                 pc_.get_whiteboard()->on_deselect_hex();
 827         }

 835 /**
 836  * Moves a unit along the currently cached route.
 837  *
 838  * @returns  true if the end of the route was reached and no information was
 839  *           uncovered that would warrant interrupting a chain of actions;
 840  *           false otherwise.
 841  */
 842 bool mouse_handler::move_unit_along_current_route()
 843 {
 844         // Copy the current route to ensure it remains valid throughout the animation.
 845         const std::vector<map_location> steps = current_route_.steps;
 846 
 847         // do not show footsteps during movement
 848         gui().set_route(nullptr);
 849         gui().unhighlight_reach();
 850 
 851         // do not keep the hex highlighted that we started from
 852         selected_hex_ = map_location();
 853         gui().select_hex(map_location());
 854 
 855         bool interrupted = false;
 856         if(steps.size() > 1) {
 857                 size_t num_moves = move_unit_along_route(steps, interrupted);
 858 
 859                 interrupted = interrupted || num_moves + 1 < steps.size();
 860                 next_unit_ = steps[num_moves];
 861         }
 862 
 863         // invalid after the move
 864         current_paths_ = pathfind::paths();
 865         current_route_.steps.clear();
 866 
 867         return !interrupted;
 868 }

 870 /**
 871  * Moves a unit across the board for a player.
 872  * This is specifically for movement at the time it is initiated by a player,
 873  * whether via a mouse click or executing whiteboard actions. Continued moves
 874  * (including goto execution) can bypass this and call actions::move_unit() directly.
 875  * This function call may include time for an animation, so make sure the
 876  * provided route will remain unchanged (the caller should probably make a local
 877  * copy).
 878  *
 879  * @param[in]   steps           The route to be traveled. The unit to be moved is at the beginning of this route.
 880  * @param[out]  interrupted     This is set to true if information was uncovered that warrants interrupting a chain of
 881  * actions (and set to false otherwise).
 882  *
 883  * @returns The number of hexes entered. This can safely be used as an index
 884  *          into steps to get the location where movement ended, provided
 885  *          steps is not empty (the return value is guaranteed to be less
 886  *          than steps.size() ).
 887  */
 888 size_t mouse_handler::move_unit_along_route(const std::vector<map_location>& steps, bool& interrupted)
 889 {
 890         if(steps.empty()) {
 891                 interrupted = false;
 892                 return 0;
 893         }
 894 
 895         // Default return value.
 896         interrupted = true;
 897 
 898         // If this is a leader on a keep, ask permission to the whiteboard to move it
 899         // since otherwise it may cause planned recruits to be erased.
 900         if(pc_.get_map_const().is_keep(steps.front())) {
 901                 unit_map::const_iterator const u = pc_.gamestate().board_.units().find(steps.front());
 902 
 903                 if(u && u->can_recruit() && u->side() == gui().viewing_side()
 904                                 && !pc_.get_whiteboard()->allow_leader_to_move(*u)) {
 905                         gui2::show_transient_message(gui_->video(), "",
 906                                         _("You cannot move your leader away from the keep with some planned recruits or recalls left."));
 907                         return 0;
 908                 }
 909         }
 910 
 911         LOG_NG << "move unit along route  from " << steps.front() << " to " << steps.back() << "\n";
 912         size_t moves = actions::move_unit_and_record(steps, &pc_.get_undo_stack(), false, true, &interrupted);
 913 
 914         cursor::set(cursor::NORMAL);
 915         gui().invalidate_game_status();
 916 
 917         if(moves == 0)
 918                 return 0;
 919 
 920         if(interrupted && moves + 1 < steps.size()) {
 921                 // reselect the unit (for "press t to continue")
 922                 select_hex(steps[moves], false, false, false);
 923                 // the new discovery is more important than the new movement range
 924                 show_partial_move_ = true;
 925         }
 926 
 927         return moves;
 928 }

 960 int mouse_handler::fill_weapon_choices(
 961                 std::vector<battle_context>& bc_vector, unit_map::iterator attacker, unit_map::iterator defender)
 962 {
 963         int best = 0;
 964         for(unsigned int i = 0; i < attacker->attacks().size(); i++) {
 965                 // skip weapons with attack_weight=0
 966                 if(attacker->attacks()[i].attack_weight() > 0) {
 967                         battle_context bc(pc_.gamestate().board_.units_, attacker->get_location(), defender->get_location(), i);
 968 
 969                         if(!bc_vector.empty() && bc.better_attack(bc_vector[best], 0.5)) {
 970                                 // as some weapons can be hidden, i is not a valid index into the resulting vector
 971                                 best = bc_vector.size();
 972                         }
 973 
 974                         bc_vector.push_back(bc);
 975                 }
 976         }
 977 
 978         return best;
 979 }

 981 int mouse_handler::show_attack_dialog(const map_location& attacker_loc, const map_location& defender_loc)
 982 {
 983         game_board& board = pc_.gamestate().board_;
 984 
 985         unit_map::iterator attacker = board.units_.find(attacker_loc);
 986         unit_map::iterator defender = board.units_.find(defender_loc);
 987 
 988         if(!attacker || !defender) {
 989                 ERR_NG << "One fighter is missing, can't attack";
 990                 return -1; // abort, click will do nothing
 991         }
 992 
 993         std::vector<battle_context> bc_vector;
 994         const int best = fill_weapon_choices(bc_vector, attacker, defender);
 995 
 996         const bool all_disabled = std::all_of(bc_vector.begin(), bc_vector.end(),
 997                 [](battle_context& context) { return (*context.get_attacker_stats().weapon).get_special_bool("disable"); }
 998         );
 999 
1000         if((*attacker).attacks().empty() || all_disabled) {
1001                 gui2::show_transient_message(gui_->video(), "No Attacks", _("This unit has no usable weapons."));
1002 
1003                 return -1;
1004         }
1005 
1006         gui2::dialogs::unit_attack dlg(attacker, defender, bc_vector, best);
1007 
1008         dlg.show(gui_->video());
1009 
1010         if(dlg.get_retval() == gui2::window::OK) {
1011                 return dlg.get_selected_weapon();
1012         }
1013 
1014         return -1;
1015 }

1110 /**
1111  * Causes attackable hexes to be highlighted.
1112  *
1113  * This checks the hexes that the provided unit can attack. If there is a valid
1114  * target there, that location is inserted into current_paths_.destinations.
1115  */
1116 void mouse_handler::show_attack_options(const unit_map::const_iterator& u)
1117 {
1118         // Cannot attack if no attacks are left.
1119         if(u->attacks_left() == 0) {
1120                 return;
1121         }
1122 
1123         // Get the teams involved.
1124         const team& cur_team = current_team();
1125         const team& u_team = pc_.gamestate().board_.teams_[u->side() - 1];
1126 
1127         // Check each adjacent hex.
1128         map_location adj[6];
1129         get_adjacent_tiles(u->get_location(), adj);
1130 
1131         for(const map_location& loc : adj) {
1132                 // No attack option shown if no visible unit present.
1133                 // (Visible to current team, not necessarily the unit's team.)
1134                 if(!pc_.get_map_const().on_board(loc)) {
1135                         continue;
1136                 }
1137 
1138                 unit_map::const_iterator i = pc_.gamestate().board_.units().find(loc);
1139                 if(!i || !i->is_visible_to_team(cur_team, gui_->get_disp_context(), false)) {
1140                         continue;
1141                 }
1142 
1143                 const unit& target = *i;
1144 
1145                 // Can only attack non-petrified enemies.
1146                 if(u_team.is_enemy(target.side()) && !target.incapacitated()) {
1147                         current_paths_.destinations.insert(loc);
1148                 }
1149         }
1150 }

1152 bool mouse_handler::unit_in_cycle(unit_map::const_iterator it)
1153 {
1154         game_board& board = pc_.gamestate().board_;
1155 
1156         if(!it) {
1157                 return false;
1158         }
1159 
1160         if(it->side() != side_num_ || it->user_end_turn() || gui().fogged(it->get_location()) || !board.unit_can_move(*it)) {
1161                 return false;
1162         }
1163 
1164         if(current_team().is_enemy(int(gui().viewing_team() + 1)) && it->invisible(it->get_location(), gui().get_disp_context())) {
1165                 return false;
1166         }
1167 
1168         if(it->get_hidden()) {
1169                 return false;
1170         }
1171 
1172         return true;
1173 }

1175 void mouse_handler::cycle_units(const bool browse, const bool reverse)
1176 {
1177         game_board& board = pc_.gamestate().board_;
1178 
1179         if(board.units().begin() == board.units().end()) {
1180                 return;
1181         }
1182 
1183         unit_map::const_iterator it = find_unit(next_unit_);
1184         if(!it) {
1185                 it = board.units().begin();
1186         }
1187 
1188         const unit_map::const_iterator itx = it;
1189 
1190         do {
1191                 if(reverse) {
1192                         if(it == board.units().begin()) {
1193                                 it = board.units().end();
1194                         }
1195 
1196                         --it;
1197                 } else {
1198                         if(it == board.units().end()) {
1199                                 it = board.units().begin();
1200                         } else {
1201                                 ++it;
1202                         }
1203                 }
1204         } while(it != itx && !unit_in_cycle(it));
1205 
1206         if(unit_in_cycle(it)) {
1207                 gui().scroll_to_tile(it->get_location(), game_display::WARP);
1208 
1209                 select_hex(it->get_location(), browse);
1210                 //              mouse_update(browse);
1211         }
1212 }
