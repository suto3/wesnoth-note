  63 void scrollbar_base::scroll(const scroll_mode scroll)
  64 {
  65         switch(scroll) {
  66                 case BEGIN:
  67                         set_item_position(0);
  68                         break;
  69 
  70                 case ITEM_BACKWARDS:
  71                         if(item_position_) {
  72                                 set_item_position(item_position_ - 1);
  73                         }
  74                         break;
  75 
  76                 case HALF_JUMP_BACKWARDS:
  77                         set_item_position(item_position_ > (visible_items_ / 2)
  78                                                                           ? item_position_ - (visible_items_ / 2)
  79                                                                           : 0);
  80                         break;
  81 
  82                 case JUMP_BACKWARDS:
  83                         set_item_position(item_position_ > visible_items_
  84                                                                           ? item_position_ - visible_items_
  85                                                                           : 0);
  86                         break;
  87 
  88                 case END:
  89                         set_item_position(item_count_ - 1);
  90                         break;
  91 
  92                 case ITEM_FORWARD:
  93                         set_item_position(item_position_ + 1);
  94                         break;
  95 
  96                 case HALF_JUMP_FORWARD:
  97                         set_item_position(item_position_ + (visible_items_ / 2));
  98                         break;
  99 
 100                 case JUMP_FORWARD:
 101                         set_item_position(item_position_ + visible_items_);
 102                         break;
 103 
 104                 default:
 105                         assert(false);
 106         }
 107 
 108         /*
 109          * @note I'm not sure if this should be in @ref set_item_position, but right now it
 110          * causes weird viewport issues with listboxes. Selecting certain items makes the
 111          * viewport to jump to the top, likely having to do with @ref recalculate dispatching
 112          * an event (since it calls set_item_position). This seems like a safe spot for now.
 113          *
 114          * -- vultraz, 2017-09-13
 115          */
 116         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 117 }

 144 void scrollbar_base::set_item_position(const unsigned item_position)
 145 {
 146         // Set the value always execute since we update a part of the state.
 147         item_position_ = item_position > item_count_ - visible_items_
 148                                                          ? item_count_ - visible_items_
 149                                                          : item_position;
 150 
 151         item_position_ = (item_position_ + step_size_ - 1) / step_size_;
 152 
 153         if(all_items_visible()) {
 154                 item_position_ = 0;
 155         }
 156 
 157         // Determine the pixel offset of the item position.
 158         positioner_offset_
 159                         = static_cast<unsigned>(item_position_ * pixels_per_step_);
 160 
 161         update_canvas();
 162 
 163         assert(item_position_ <= item_count_ - visible_items_);
 164 
 165 #if 0
 166         /** See comment in @ref scroll for more info. */
 167         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 168 #endif
 169 }

 189 void scrollbar_base::recalculate()
 190 {
 191         // We can be called before the size has been set up in that case we can't do
 192         // the proper recalculation so stop before we die with an assert.
 193         if(!get_length()) {
 194                 return;
 195         }
 196 
 197         // Get the available size for the slider to move.
 198         const int available_length = get_length() - offset_before()
 199                                                                  - offset_after();
 200 
 201         assert(available_length > 0);
 202 
 203         // All visible.
 204         if(item_count_ <= visible_items_) {
 205                 positioner_offset_ = offset_before();
 206                 positioner_length_ = available_length;
 207                 recalculate_positioner();
 208                 item_position_ = 0;
 209                 update_canvas();
 210                 return;
 211         }
 212 
 213         /**
 214          * @todo In the MP lobby it can happen that a listbox has first zero items,
 215          * then gets filled and since there are no visible items the second assert
 216          * after this block will be triggered. Use this ugly hack to avoid that
 217          * case. (This hack also added the gui/widgets/window.hpp include.)
 218          */
 219         if(!visible_items_) {
 220                 window* window = get_window();
 221                 assert(window);
 222                 window->invalidate_layout();
 223                 ERR_GUI_G << LOG_HEADER
 224                                   << " Can't recalculate size, force a window layout phase.\n";
 225                 return;
 226         }
 227 
 228         assert(step_size_);
 229         assert(visible_items_);
 230 
 231         const unsigned steps = (item_count_ - visible_items_ - step_size_)
 232                                                    / step_size_;
 233 
 234         positioner_length_ = available_length * visible_items_ / item_count_;
 235         recalculate_positioner();
 236 
 237         // Make sure we can also show the last step, so add one more step.
 238         pixels_per_step_ = (available_length - positioner_length_)
 239                                            / static_cast<float>(steps + 1);
 240 
 241         set_item_position(item_position_ * step_size_);
 242 #if 0
 243         std::cerr << "Scrollbar recalculate overview:\n"
 244                 << "item_count_ " << item_count_
 245                 << " visible_items_ " << visible_items_
 246                 << " step_size_ " << step_size_
 247                 << " steps " << steps
 248                 << "\n"
 249                 << "minimum_positioner_length() " << minimum_positioner_length()
 250                 << " maximum_positioner_length() " << maximum_positioner_length()
 251                 << "\n"
 252                 << " positioner_length_ " << positioner_length_
 253                 << " positioner_offset_ " << positioner_offset_
 254                 << "\n"
 255                 << "available_length " << available_length
 256                 << " pixels_per_step_ " << pixels_per_step_
 257                 << ".\n\n";
 258 #endif
 259 }

 261 void scrollbar_base::recalculate_positioner()
 262 {
 263         const unsigned minimum = minimum_positioner_length();
 264         const unsigned maximum = maximum_positioner_length();
 265 
 266         if(minimum == maximum) {
 267                 positioner_length_ = maximum;
 268         } else if(maximum != 0 && positioner_length_ > maximum) {
 269                 positioner_length_ = maximum;
 270         } else if(positioner_length_ < minimum) {
 271                 positioner_length_ = minimum;
 272         }
 273 }

 275 void scrollbar_base::move_positioner(const int distance)
 276 {
 277         if(distance < 0 && -distance > static_cast<int>(positioner_offset_)) {
 278                 positioner_offset_ = 0;
 279         } else {
 280                 positioner_offset_ += distance;
 281         }
 282 
 283         const unsigned length = get_length() - offset_before() - offset_after()
 284                                                         - positioner_length_;
 285 
 286         if(positioner_offset_ > length) {
 287                 positioner_offset_ = length;
 288         }
 289 
 290         unsigned position
 291                         = static_cast<unsigned>(positioner_offset_ / pixels_per_step_);
 292 
 293         // Note due to floating point rounding the position might be outside the
 294         // available positions so set it back.
 295         if(position > item_count_ - visible_items_) {
 296                 position = item_count_ - visible_items_;
 297         }
 298 
 299         if(position != item_position_) {
 300                 item_position_ = position;
 301 
 302                 child_callback_positioner_moved();
 303 
 304                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 305 
 306                 // positioner_moved_notifier_.notify();
 307         }
 308 #if 0
 309         std::cerr << "Scrollbar move overview:\n"
 310                 << "item_count_ " << item_count_
 311                 << " visible_items_ " << visible_items_
 312                 << " step_size_ " << step_size_
 313                 << "\n"
 314                 << "minimum_positioner_length() " << minimum_positioner_length()
 315                 << " maximum_positioner_length() " << maximum_positioner_length()
 316                 << "\n"
 317                 << " positioner_length_ " << positioner_length_
 318                 << " positioner_offset_ " << positioner_offset_
 319                 << "\n"
 320                 << " pixels_per_step_ " << pixels_per_step_
 321                 << " item_position_ " << item_position_
 322                 << ".\n\n";
 323 #endif
 324         update_canvas();
 325 }

 337 void scrollbar_base::signal_handler_mouse_motion(const event::ui_event event,
 338                                                                                           bool& handled,
 339                                                                                           bool& halt,
 340                                                                                           const point& coordinate)
 341 {
 342         DBG_GUI_E << LOG_HEADER << ' ' << event << " at " << coordinate << ".\n";
 343 
 344         point mouse = coordinate;
 345         mouse.x -= get_x();
 346         mouse.y -= get_y();
 347 
 348         switch(state_) {
 349                 case ENABLED:
 350                         if(on_positioner(mouse)) {
 351                                 set_state(FOCUSED);
 352                         }
 353 
 354                         break;
 355 
 356                 case PRESSED: {
 357                         if(in_orthogonal_range(mouse)) {
 358                                 const int distance = get_length_difference(mouse_, mouse);
 359                                 mouse_ = mouse;
 360                                 move_positioner(distance);
 361                         }
 362 
 363                 } break;
 364 
 365                 case FOCUSED:
 366                         if(!on_positioner(mouse)) {
 367                                 set_state(ENABLED);
 368                         }
 369                         break;
 370 
 371                 case DISABLED:
 372                         // Shouldn't be possible, but seems to happen in the lobby
 373                         // if a resize layout happens during dragging.
 374                         halt = true;
 375                         break;
 376 
 377                 default:
 378                         assert(false);
 379         }
 380         handled = true;
 381 }

 395 void scrollbar_base::signal_handler_left_button_down(const event::ui_event event,
 396                                                                                                   bool& handled)
 397 {
 398         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 399 
 400         point mouse = get_mouse_position();
 401         mouse.x -= get_x();
 402         mouse.y -= get_y();
 403 
 404         if(on_positioner(mouse)) {
 405                 assert(get_window());
 406                 mouse_ = mouse;
 407                 get_window()->mouse_capture();
 408                 set_state(PRESSED);
 409         }
 410 
 411         const int bar = on_bar(mouse);
 412 
 413         if(bar == -1) {
 414                 scroll(HALF_JUMP_BACKWARDS);
 415                 // positioner_moved_notifier_.notify();
 416         } else if(bar == 1) {
 417                 scroll(HALF_JUMP_FORWARD);
 418                 // positioner_moved_notifier_.notify();
 419         } else {
 420                 assert(bar == 0);
 421         }
 422 
 423         handled = true;
 424 }

 426 void scrollbar_base::signal_handler_left_button_up(const event::ui_event event,
 427                                                                                                 bool& handled)
 428 {
 429         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 430 
 431         point mouse = get_mouse_position();
 432         mouse.x -= get_x();
 433         mouse.y -= get_y();
 434 
 435         if(state_ != PRESSED) {
 436                 return;
 437         }
 438 
 439         assert(get_window());
 440         get_window()->mouse_capture(false);
 441 
 442         if(on_positioner(mouse)) {
 443                 set_state(FOCUSED);
 444         } else {
 445                 set_state(ENABLED);
 446         }
 447 
 448         handled = true;
 449 }
