   2 wmliterator.py -- Python routines for navigating a Battle For Wesnoth WML tree
   3 Author: Sapient (Patrick Parker), 2007
   4 
   5 Purpose:
   6  The WmlIterator class can be used to analyze and search the structure of WML
   7  files non-invasively (i.e. preserving existing line structure), and its main
   8  use is to determine when a transformation of deprecated content needs to take
   9  place. (I wrote it was because wmllint was trying to do a lot of things with
  10  regular expressions which really required a more algorithmic approach. Also,
  11  wmllint was often inconsistent with correct handling of comments and values
  12  inside strings.)
  13 
  14 Limitations:
  15  The WmlIterator does not attempt to expand macros, only to recognize them as
  16  another level of nesting. Also, the handling of multiple assignment syntax
  17  is somewhat limited (for similar reasons). Adding seamless support for these
  18  would be ideal, but it presents a design challenge since the iteration is
  19  supposed to be non-invasive. Thus, the current behavior is considered good
  20  enough for now.
  21 """

 103 class WmlIterator(object):
 104     """Return an iterable WML navigation object.
 105     Initialize with a list of lines or a file; if the the line list is
 106     empty and the filename is specified, lines will be read from the file.
 107 
 108     Note: if changes are made to lines while iterating, this may produce
 109     unexpected results. In such case, seek() to the line number of a
 110     scope behind where changes were made.
 111 Important Attributes:
 112     lines - this is an internal list of all the physical lines
 113     scopes - this is an internal list of all open scopes (as iterators)
 114              note: when retrieving an iterator from this list, always
 115              use a copy to perform seek() or next(), and not the original
 116     element - the wml tag, key, or macro name for this logical line
 117               (in complex cases, this may be a tuple of elements...
 118               see parseElements for list of possible values)
 119     text - the exact text of this logical line, as it appears in the
 120            original source, and ending with a newline
 121            note: the logical line also includes multi-line quoted strings
 122     span - the number of physical lines in this logical line:
 123            always 1, unless text contains a multi-line quoted string
 124     lineno - a zero-based line index marking where this text begins
 125     """
 126     def __init__(self, lines=None, filename=None, begin=-1):
 127         "Initialize a new WmlIterator."
 128         self.fname = filename
 129         if lines is None:
 130             lines = []
 131             if filename:
 132                 try:
 133                     with codecs.open(self.fname, "r", "utf8") as ifp:
 134                         lines = ifp.readlines()
 135                 except Exception:
 136                     self.printError('error opening file')
 137         self.lines = lines
 138         self.reset()
 139         self.seek(begin)

 141     def parseQuotes(self, lines):
 142         """Return the line or multiline text if a quote spans multiple lines"""
 143         text = lines[self.lineno]
 144         span = 1
 145         begincomment = text.find('#')
 146         if begincomment < 0:
 147             begincomment = None
 148         beginquote = text[:begincomment].find('<<')
 149         while beginquote >= 0:
 150             endquote = -1
 151             beginofend = beginquote+2
 152             while endquote < 0:
 153                 endquote = text.find('>>', beginofend)
 154                 if endquote < 0:
 155                     if self.lineno + span >= len(lines):
 156                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 157                         return text, span
 158                     beginofend = len(text)
 159                     text += lines[self.lineno + span]
 160                     span += 1
 161             begincomment = text.find('#', endquote+2)
 162             if begincomment < 0:
 163                 begincomment = None
 164             beginquote = text[:begincomment].find('<<', endquote+2)
 165         beginquote = text[:begincomment].find('"')
 166         while beginquote >= 0:
 167             endquote = -1
 168             beginofend = beginquote+1
 169             while endquote < 0:
 170                 endquote = text.find('"', beginofend)
 171                 if endquote < 0:
 172                     if self.lineno + span >= len(lines):
 173                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 174                         return text, span
 175                     beginofend = len(text)
 176                     text += lines[self.lineno + span]
 177                     span += 1
 178             begincomment = text.find('#', endquote+1)
 179             if begincomment < 0:
 180                 begincomment = None
 181             beginquote = text[:begincomment].find('"', endquote+1)
 182         return text, span

 184     def closeScope(self, scopes, closerElement):
 185         """Close the most recently opened scope. Return false if not enough scopes.
 186         note: directives close all the way back to the last open directive
 187         non-directives cannot close a directive and will no-op in that case."""
 188         try:
 189             if isDirective(closerElement):
 190                 while not isDirective(scopes.pop()):
 191                     pass
 192             elif (closerElement==closeMacroType):
 193                 elem = ''
 194                 while not elem.startswith('{'):
 195                     closed = scopes.pop()
 196                     elem = closed
 197                     if isinstance(closed, WmlIterator):
 198                         elem = closed.element
 199                     if isDirective(elem):
 200                         self.printScopeError(closerElement)
 201                         scopes.append(closed) # to reduce additional errors (hopefully)
 202                         return True
 203             elif not isDirective(scopes[-1]):
 204                 closed = scopes.pop()
 205                 elem = closed
 206                 if isinstance(closed, WmlIterator):
 207                     elem = closed.element
 208                 if (elem.startswith('{') and closerElement != closeMacroType):
 209                     scopes.append(closed)
 210                 elif (isOpener(elem) and closerElement != '[/'+elem[1:]
 211                 and '+'+closerElement != elem[1]+'[/'+elem[2:]):
 212                     self.printError('reached', closerElement, 'at line', self.lineno+1, 'before closing scope', elem)
 213                     scopes.append(closed) # to reduce additional errors (hopefully)
 214             return True
 215         except IndexError:
 216             return False

 218     def parseElements(self, text):
 219         """Remove any closed scopes, return a list of element names
 220         and list of new unclosed scopes
 221     Element Types:
 222         tags: one of "[tag_name]" or "[/tag_name]"
 223             [tag_name] - opens a scope
 224             [/tag_name] - closes a scope
 225         keys: either "key=" or ("key1=", "key2=") for multi-assignment
 226             key= - does not affect the scope
 227             key1,key2= - multi-assignment returns multiple elements
 228         directives: one of "#ifdef", "#ifndef", "#ifhave", "#ifnhave", "#ifver", "#ifnver", "#else", "#endif", "#define", "#enddef"
 229             #ifdef - opens a scope
 230             #ifndef - opens a scope
 231             #ifhave - opens a scope
 232             #ifnhave - opens a scope
 233             #ifver - opens a scope
 234             #ifnver - opens a scope
 235             #else - closes a scope, also opens a new scope
 236             #endif - closes a scope
 237             #define - opens a scope
 238             #enddef - closes a scope
 239         macro calls: "{MACRO_NAME}"
 240             {MACRO_NAME - opens a scope
 241             } - closes a scope
 242         """
 243         elements = [] #(elementType, sortPos, scopeDelta)
 244         # first remove any lua strings
 245         beginquote = text.find('<<')
 246         while beginquote >= 0:
 247             endquote = text.find('>>', beginquote+2)
 248             if endquote < 0:
 249                 text = text[:beginquote]
 250                 beginquote = -1 #terminate loop
 251             else:
 252                 text = text[:beginquote] + text[endquote+2:]
 253                 beginquote = text.find('<<')
 254         # remove any quoted strings
 255         beginquote = text.find('"')
 256         while beginquote >= 0:
 257             endquote = text.find('"', beginquote+1)
 258             if endquote < 0:
 259                 text = text[:beginquote]
 260                 beginquote = -1 #terminate loop
 261             else:
 262                 text = text[:beginquote] + text[endquote+1:]
 263                 beginquote = text.find('"')
 264         # next remove any comments
 265         text = text.lstrip()
 266         commentSearch = 1
 267         if text.startswith('#ifdef'):
 268             return (['#ifdef'],)*2
 269         elif text.startswith('#ifndef'):
 270             return (['#ifndef'],)*2
 271         elif text.startswith('#ifhave'):
 272             return (['#ifhave'],)*2
 273         elif text.startswith('#ifnhave'):
 274             return (['#ifnhave'],)*2
 275         elif text.startswith('#ifver'):
 276             return (['#ifver'],)*2
 277         elif text.startswith('#ifnver'):
 278             return (['#ifnver'],)*2
 279         elif text.startswith('#else'):
 280             if not self.closeScope(self.scopes, '#else'):
 281                 self.printScopeError('#else')
 282             return (['#else'],)*2
 283         elif text.startswith('#endif'):
 284             if not self.closeScope(self.scopes, '#endif'):
 285                 self.printScopeError('#endif')
 286                 return ['#endif'], []
 287         elif text.startswith('#define'):
 288             return (['#define'],)*2
 289         elif text.find('#enddef') >= 0:
 290             elements.append(('#enddef', text.find('#enddef'), -1))
 291         elif text.startswith('#po:') or text.startswith('# po:'):
 292             elements.append(("#po", 0, 0))
 293         else:
 294             commentSearch = 0
 295         begincomment = text.find('#', commentSearch)
 296         if begincomment >= 0:
 297             text = text[:begincomment]
 298         #now find elements in a loop
 299         for m in tagPattern.finditer(text):
 300             delta = 1
 301             if isCloser(m.group(2)):
 302                 delta = -1
 303             elements.append((m.group(2), m.start(), delta))
 304         for m in keyPattern.finditer(text):
 305             for i, k in enumerate(keySplit.split(m.group(0))):
 306                 if k:
 307                     elements.append((k+'=', m.start()+i, 0))
 308         for m in macroOpenPattern.finditer(text):
 309             elements.append((m.group(1), m.start(), 1))
 310         for m in macroClosePattern.finditer(text):
 311             elements.append((closeMacroType, m.start(), -1))
 312         #sort by start position
 313         elements.sort(key=lambda x:x[1])
 314         resultElements = []
 315         openedScopes = []
 316         for elem, sortPos, scopeDelta in elements:
 317             while scopeDelta < 0:
 318                 if not(self.closeScope(openedScopes, elem)\
 319                         or self.closeScope(self.scopes, elem)):
 320                     self.printScopeError(elem)
 321                 scopeDelta += 1
 322             while scopeDelta > 0:
 323                 openedScopes.append(elem)
 324                 scopeDelta -= 1
 325             resultElements.append(elem)
 326         return resultElements, openedScopes

 354     def seek(self, lineno, clearEnd=True):
 355         """Move the iterator to a specific line number"""
 356         if clearEnd:
 357             self.endScope = None
 358         if lineno < self.lineno:
 359             for scope in reversed(self.scopes):
 360                 # if moving backwards, try to re-use a scope iterator
 361                 if scope.lineno <= lineno:
 362                     # copy the scope iterator's state to self
 363                     self.__dict__ = dict(scope.__dict__)
 364                     self.scopes = scope.scopes[:]
 365                     self.nextScopes = scope.nextScopes[:]
 366                     break
 367             else:
 368                 # moving backwards past all scopes forces a reset
 369                 self.reset()
 370         while self.lineno + self.span - 1 < lineno:
 371             next(self)
 372         return self

 405     def __next__(self):
 406         """Move the iterator to the next line number
 407         note: May raise StopIteration"""
 408         if not self.hasNext():
 409             if self.scopes:
 410                 self.printError("reached EOF with open scopes", self.scopes)
 411             raise StopIteration
 412         self.lineno = self.lineno + self.span
 413         self.text, self.span = self.parseQuotes(self.lines)
 414         self.scopes.extend(self.nextScopes)
 415         self.element, nextScopes = self.parseElements(self.text)
 416         self.nextScopes = []
 417         for elem in nextScopes:
 418         # remember scopes by storing a copy of the iterator
 419             copyItor = self.copy()
 420             copyItor.element = elem
 421             self.nextScopes.append(copyItor)
 422             copyItor.nextScopes.append(copyItor)
 423         if(len(self.element) == 1):
 424             # currently we only wish to handle simple single assignment syntax
 425             self.element = self.element[0]
 426         if self.endScope is not None and not self.scopes.count(self.endScope):
 427             raise StopIteration
 428         return self

 456     def printError(nav, *misc):
 457         """Print error associated with a given file; avoid printing duplicates"""
 458         if nav.fname:
 459             silenceValue = ' '.join(map(str, misc))
 460             if nav.fname not in silenceErrors:
 461                 print(nav.fname, file=sys.stderr)
 462                 silenceErrors[nav.fname] = set()
 463             elif silenceValue in silenceErrors[nav.fname]:
 464                 return # do not print a duplicate error for this file
 465             silenceErrors[nav.fname].add(silenceValue)
 466         print('wmliterator:', end=" ", file=sys.stderr)
 467         for item in misc:
 468             print(item, end=" ", file=sys.stderr)
 469         print("", file=sys.stderr) #terminate line
