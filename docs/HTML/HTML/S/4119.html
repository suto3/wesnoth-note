  65 def comma_split(csstring, list=None, strip="r"):
  66     "Split a comma-separated string, and append the entries to a list if specified."
  67     vallist = [x.lstrip() for x in csstring.split(",") if x.lstrip()]
  68     # strip=: utils::split will remove trailing whitespace from items in comma-
  69     # separated lists but the wml-tags.lua split function only removes leading
  70     # whitespace. So two flags are offered to change default behavior: one to
  71     # lstrip() only, the other to warn about trailing whitespace.
  72     if 'w' in strip:
  73         for item in vallist:
  74             if re.search('\s$', item):
  75                 print('Trailing whitespace may be problematic: "%s" in "%s"' % (item, csstring))
  76     if 'l' not in strip:
  77         vallist = [x.rstrip() for x in vallist]

  83 def parse_attribute(line):
  84     "Parse a WML key-value pair from a line."
  85     if '=' not in line or line.find("#") > -1 and line.find("#") < line.find("="):
  86         return None
  87     where = line.find("=")
  88     leader = line[:where]
  89     after = line[where+1:]
  90     after = after.lstrip()
  91     if re.search("\s#", after):
  92         where = len(re.split("\s+#", after)[0])
  93         value = after[:where]
  94         comment = after[where:]
  95     else:
  96         value = after.rstrip()
  97         comment = ""
  98     # Return four fields: stripped key, part of line before value,
  99     # value, trailing whitespace and comment.
 100     return (leader.strip(), leader+"=", string_strip(value), comment)

 222     for i in range(start, len(line)):

 245                 arg += line[i]
 246         elif line[i] == "(":
 247             parendepth += 1
 248         elif line[i] == ")":
 249             parendepth -= 1
 250         elif not line[i-1].isspace() and \
 251              line[i].isspace() and \
 252              brackdepth == 1 and \

 255             if arg.startswith('"') and arg.endswith('"'):

 257             args.append(arg)
 258             arg = ""
 259         elif not line[i].isspace() or parendepth > 0:

 312 def actualtype(a):
 313     if a is None:
 314         return None
 315     # Deduce the type of the actual
 316     if a.isdigit() or a.startswith("-") and a[1:].isdigit():
 317         atype = "numeric"
 318     elif re.match(r"0\.[0-9]+\Z", a):
 319         atype = "percentage"
 320     elif re.match(r"-?[0-9]+,-?[0-9]+\Z", a):
 321         atype = "position"
 322     elif re.match(r"([0-9]+\-[0-9]+,?|[0-9]+,?)+\Z", a):
 323         atype = "span"
 324     elif a in ("melee", "ranged"):
 325         atype = "range"
 326     elif a in ("lawful", "neutral", "chaotic", "liminal"):
 327         atype = "alignment"
 328     elif a.startswith("{") or a.endswith("}") or a.startswith("$"):
 329         atype = None # Can't tell -- it's a macro expansion
 330     elif re.match(image_reference, a) or a == "unit_image":
 331         atype = "image"
 332     elif re.match(r"(\*|[A-Z][a-z]+)\^([A-Z][a-z\\|/]+\Z)?", a):
 333         atype = "terrain_code"
 334     elif a.endswith(".wav") or a.endswith(".ogg"):
 335         atype = "sound"
 336     elif a.startswith('"') and a.endswith('"') or (a.startswith("_") and a[1] not in string.ascii_lowercase):
 337         atype = "stringliteral"
 338     elif "=" in a:
 339         atype = "filter"
 340     elif re.match(r"[A-Z][a-z][a-z]?\Z", a):
 341         atype = "shortname"
 342     elif a == "":
 343         atype = "empty"
 344     elif not ' ' in a:
 345         atype = "name"
 346     else:
 347         atype = "string"
 348     return atype

 356         # formal one.  Then we have a fallback rule checking for type
 357         # equality or wildcarding.

 365             pass

 367             pass

 369             pass
 370         elif atype in ("shortname", "name", "stringliteral") and ftype == "string":
 371             pass
 372         elif atype in ("shortname", "name", "string", "stringliteral", "empty") and ftype == "optional_string":
 373             pass
 374         elif atype in ("shortname",) and ftype == "terrain_code":
 375             pass
 376         elif atype in ("numeric", "position", "span", "empty") and ftype == "alliance":
 377             pass
 378         elif atype in ("terrain_code", "shortname", "name") and ftype == "terrain_pattern":
 379             pass
 380         elif atype in ("string", "shortname", "name") and ftype == "types":
 381             pass
 382         elif atype in ("numeric", "percentage") and ftype == "percentage":
 383             pass

 567                                         % (here, name, defn), file=sys.stderr)

 569                         self.xref[name] = []

 571                     state = "outside"
 572                 elif state == "macro_header" and line.strip() and line.strip()[0] != "#":

 574                 if state == "macro_header":

 576                     if "wmlscope" not in line and "wmllint:" not in line:

 578                 if state in ("macro_header", "macro_body"):
 579                     here.hash.update(line.encode("utf8"))

 581                     tokens = line.split()
 582                     name = tokens[1]
 583                     if name in self.xref and self.xref[name]:
 584                         self.xref[name][-1].undef = n+1
 585                     else:

 587                               % (Reference(namespace, filename, n+1), name))
 588                 if state == 'outside':
 589                     if '[unit_type]' in line:
 590                         latch_unit = True
 591                     elif '[/unit_type]' in line:
 592                         latch_unit = False
 593                     elif '[base_unit]' in line:
 594                         in_base_unit = True
 595                     elif '[/base_unit]' in line:
 596                         in_base_unit = False
 597                     elif '[theme]' in line:
 598                         in_theme = True
 599                     elif '[/theme]' in line:
 600                         in_theme = False
 601                     elif latch_unit and not in_base_unit and not in_theme and "id" in line:

 603                         if m and m.group(1) == "id":
 604                             uid = m.group(2)
 605                             if uid not in self.unit_ids:
 606                                 self.unit_ids[uid] = []
 607                             self.unit_ids[uid].append(Reference(namespace, filename, n+1))

 744                                 if len(candidates) > 1:
 745                                     print("%s: more than one resource matching %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 746                             if not key:

 816 class Translation(dict):
 817     "Parses a po file to create a translation dictionary."
 818     def __init__(self, textdomain, isocode, topdir=""):
 819         self.textdomain = textdomain
 820         self.isocode = isocode
 821         self.gettext = {}
 822         if self.isocode != "C":
 823             isocode2 = isocode[:isocode.rfind("_")]
 824             for code in [isocode, isocode2]:
 825                 fn = "po/%s/%s.po" % (textdomain, code)
 826                 if topdir: fn = os.path.join(topdir, fn)
 827                 try:
 828                     f = file(fn)
 829                     break
 830                 except IOError:
 831                     pass
 832             else:
 833                 raise TranslationError(textdomain, self.isocode)
 834 
 835             expect = False
 836             fuzzy = "#, fuzzy\n"
 837             gettext = f.read().decode("utf8")
 838             matches = re.compile("""(msgid|msgstr)((\s*".*?")+)""").finditer(gettext)
 839             msgid = ""
 840             for match in matches:
 841                 text = "".join(re.compile('"(.*?)"').findall(match.group(2)))
 842                 if match.group(1) == "msgid":
 843                     msgid = text.replace("\\n", "\n")
 844                     expect = gettext[match.start(1) - len(fuzzy):match.start(1)] != fuzzy
 845                 elif expect:
 846                     self.gettext[msgid] = text.replace("\\n", "\n")
 847     def get(self, key, dflt):
 848         if self.isocode == "C":
 849             if key:
 850                 return key[key.find("^") + 1:]
 851             return "?"
 852         else:
 853             t = self.gettext.get(key, dflt)
 854             if not t:
 855                 if key:
 856                     return key[key.find("^") + 1:]
 857                 return "?"
 858             return t
 859     def __getitem__(self, key):
 860         if self.isocode == "C":
 861             return key
 862         else:
 863             return self.gettext[key]
 864     def __contains__(self, key):
 865         if self.isocode == "C":
 866             return True
 867         else:

 870 class Translations:
 871     "Wraps around Translation to support multiple languages and domains."
 872     def __init__(self, topdir = ""):
 873         self.translations = {}
 874         self.topdir = topdir
 875     def get(self, textdomain, isocode, key, default):
 876         t = (textdomain, isocode)
 877         if not t in self.translations:
 878             try:
 879                 self.translations[t] = Translation(textdomain, isocode, self.topdir)
 880             except TranslationError as e:
 881                 print(str(e), file=sys.stderr)
 882                 self.translations[t] = Translation(textdomain, "C", self.topdir)
 883         result = self.translations[t].get(key, default)
