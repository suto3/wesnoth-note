 215     # Quenoth Youths have a non-null advances_to= in order to prevent them from
 216     # having purple xp bars (until all advancements have been gained); these
 217     # events make sure that they've nullified before advancement (to prevent the
 218     # unit type from appearing in the advancement menu) and restored afterwards.
 219     [event]
 220         name=pre advance
 221         first_time_only=no
 222 
 223         [filter]
 224             type=Quenoth Youth,Quenoth Youth 2,Quenoth Youth 3
 225         [/filter]
 226 
 227         {VARIABLE unit.advances_to null}
 228 
 229         [unstore_unit]
 230             variable=unit
 231             find_vacant=no
 232             advance=no
 233         [/unstore_unit]
 234     [/event]
 235     [event]
 236         name=post advance
 237         first_time_only=no
 238 
 239         [filter]
 240             type=Quenoth Youth 3
 241 
 242             [not]
 243                 [filter_wml]
 244                     [modifications]
 245                         [advancement]
 246                             exclude_amla=warrior_2_1,warrior_2_2
 247                         [/advancement]
 248                         [advancement]
 249                             exclude_amla=hunter_2_1,hunter_2_2
 250                         [/advancement]
 251                         [advancement]
 252                             exclude_amla=leader_2_1,leader_2_2
 253                         [/advancement]
 254                     [/modifications]
 255                 [/filter_wml]
 256             [/not]
 257         [/filter]
 258 
 259         {VARIABLE unit.advances_to "Quenoth Youth 3"}
 260 
 261         [unstore_unit]
 262             variable=unit
 263             find_vacant=no
 264             advance=no
 265         [/unstore_unit]
 266     [/event]

 268     # On recruit, this switches the higher-recruit-cost units to the normal
 269     # types, so that the different cost variations don't need to be taken into
 270     # account for example in unit filters.
 271 #define DISCARD_RECRUIT_COST_HELPER_UNITS TYPE
 272     [event]
 273         name=recruit
 274         first_time_only=no
 275 
 276         [filter]
 277             race=elf
 278         [/filter]
 279 
 280         [filter_condition]
 281             [variable]
 282                 name=unit.type
 283                 contains={TYPE}
 284             [/variable]
 285 
 286             [variable]
 287                 name=unit.type
 288                 not_equals={TYPE}
 289             [/variable]
 290         [/filter_condition]
 291 
 292         {VARIABLE unit.type {TYPE}}
 293 
 294         [unstore_unit]
 295             variable=unit
 296             find_vacant=no
 297         [/unstore_unit]
 298     [/event]
 299 #enddef
