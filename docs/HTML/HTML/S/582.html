 105 void button::signal_handler_left_button_down(const event::ui_event event,
 106                                                                                           bool& handled)
 107 {
 108         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 109 
 110         window* window = get_window();
 111         if(window) {
 112                 window->mouse_capture();
 113         }
 114 
 115         set_state(PRESSED);
 116         handled = true;
 117 }

 127 
 128 void button::signal_handler_left_button_click(const event::ui_event event,
 129                                                                                            bool& handled)
 130 {
 131         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 132 
 133         sound::play_UI_sound(settings::sound_button_click);
 134 
 135         // If a button has a retval do the default handling.
 136         if(retval_ != 0) {
 137                 window* window = get_window();
 138                 if(window) {
 139                         window->set_retval(retval_);
 140                         return;
 141                 }
 142         }
 143 
 144         handled = true;
 145 }

 188         // Note the order should be the same as the enum state_t in button.hpp.
