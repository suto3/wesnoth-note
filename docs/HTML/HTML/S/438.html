  57 void statistics_dialog::pre_show(window& window)
  58 {
  59         //
  60         // Set title
  61         //
  62         label& title = find_widget<label>(&window, "title", false);
  63         title.set_label((formatter() << title.get_label() << " (" << current_team_.side_name() << ")").str());
  64 
  65         //
  66         // Set up scenario menu
  67         //
  68         std::vector<config> menu_items;
  69         for(const auto& scenario : scenarios_) {
  70                 menu_items.emplace_back(config {"label", *scenario.first});
  71         }
  72 
  73         menu_button& scenario_menu = find_widget<menu_button>(&window, "scenario_menu", false);
  74 
  75         scenario_menu.set_values(menu_items, scenario_index_);
  76         scenario_menu.connect_click_handler(std::bind(&statistics_dialog::on_scenario_select, this, std::ref(window)));
  77 
  78         //
  79         // Set up tab toggle
  80         //
  81         listbox& tab_bar = find_widget<listbox>(&window, "tab_bar", false);
  82         tab_bar.select_row(use_campaign);
  83 
  84         connect_signal_notify_modified(tab_bar,
  85                 std::bind(&statistics_dialog::on_tab_select, this, std::ref(window)));
  86 
  87         //
  88         // Set up primary stats list
  89         //
  90         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
  91 
  92         connect_signal_notify_modified(stat_list,
  93                 std::bind(&statistics_dialog::on_primary_list_select, this, std::ref(window)));
  94 
  95         update_lists(window);
  96 }

  99 {
 100         return use_campaign ? campaign_ : *scenarios_[scenario_index_].second;

 103 void statistics_dialog::add_stat_row(window& window, const std::string& type, const statistics::stats::str_int_map& value, const bool has_cost)
 104 {
 105         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
 106 
 107         std::map<std::string, string_map> data;
 108         string_map item;
 109 
 110         item["label"] = type;
 111         data.emplace("stat_type", item);
 112 
 113         item["label"] = std::to_string(statistics::sum_str_int_map(value));
 114         data.emplace("stat_detail", item);
 115 
 116         item["label"] = has_cost ? std::to_string(statistics::sum_cost_str_int_map(value)) : font::unicode_em_dash;
 117         data.emplace("stat_cost", item);
 118 
 119         stat_list.add_row(data);
 120 
 121         main_stat_table_.push_back(&value);
 122 }

 124 void statistics_dialog::add_damage_row(
 125                 window& window,
 126                 const std::string& type,
 127                 const long long& damage,
 128                 const long long& expected,
 129                 const long long& turn_damage,
 130                 const long long& turn_expected,
 131                 const bool show_this_turn)
 132 {
 133         listbox& damage_list = find_widget<listbox>(&window, "stats_list_damage", false);
 134 
 135         std::map<std::string, string_map> data;
 136         string_map item;
 137 
 138         item["label"] = type;
 139         data.emplace("damage_type", item);
 140 
 141         const int shift = statistics::stats::decimal_shift;
 142 
 143         const long long dsa = shift * damage      - expected;
 144         const long long dst = shift * turn_damage - turn_expected;
 145 
 146         std::ostringstream str;
 147         str << damage << " / "
 148                 << (expected * 10 + shift / 2) / shift * 0.1
 149                 << "    " // TODO: should probably make this two columns
 150                 << ((dsa < 0) ^ (expected < 0) ? "" : "+")
 151                 << (expected == 0 ? 0 : 100 * dsa / expected) << '%';
 152 
 153         item["label"] = str.str();
 154         data.emplace("damage_overall", item);
 155 
 156         str.str("");
 157 
 158         if(show_this_turn) {
 159                 str << turn_damage << " / "
 160                         << (turn_expected * 10 + shift / 2) / shift / 10
 161                         << "    " // TODO: should probably make this two columns
 162                         << ((dst < 0) ^ (turn_expected < 0) ? "" : "+")
 163                         << (turn_expected == 0 ? 0 : 100 * dst / turn_expected) << '%';
 164 
 165                 item["label"] = str.str();
 166                 data.emplace("damage_this_turn", item);
 167         }
 168 
 169         damage_list.add_row(data);
 170 }

 172 void statistics_dialog::update_lists(window& window)
 173 {
 174         //
 175         // Update primary stats list
 176         //
 177         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
 178 
 179         stat_list.clear();
 180         main_stat_table_.clear();
 181 
 182         const statistics::stats& stats = current_stats();
 183 
 184         add_stat_row(window, _("Recruits"),     stats.recruits);
 185         add_stat_row(window, _("Recalls"),      stats.recalls);
 186         add_stat_row(window, _("Advancements"), stats.advanced_to, false);
 187         add_stat_row(window, _("Losses"),       stats.deaths);
 188         add_stat_row(window, _("Kills"),        stats.killed);
 189 
 190         // Update unit count list
 191         on_primary_list_select(window);
 192 
 193         //
 194         // Update damage stats list
 195         //
 196         const bool show_this_turn = use_campaign || scenario_index_ + 1 == scenarios_.size();
 197 
 198         listbox& damage_list = find_widget<listbox>(&window, "stats_list_damage", false);
 199 
 200         damage_list.clear();
 201 
 202         add_damage_row(window, _("Inflicted"),
 203                 stats.damage_inflicted,
 204                 stats.expected_damage_inflicted,
 205                 stats.turn_damage_inflicted,
 206                 stats.turn_expected_damage_inflicted,
 207                 show_this_turn
 208         );
 209 
 210         add_damage_row(window, _("Taken"),
 211                 stats.damage_taken,
 212                 stats.expected_damage_taken,
 213                 stats.turn_damage_taken,
 214                 stats.turn_expected_damage_taken,
 215                 show_this_turn
 216         );
 217 }

 219 void statistics_dialog::on_tab_select(window& window)
 220 {
 221         const bool is_campaign_tab = find_widget<listbox>(&window, "tab_bar", false).get_selected_row() == 1;
 222 
 223         if(use_campaign != is_campaign_tab) {
 224                 use_campaign = is_campaign_tab;
 225 
 226                 update_lists(window);
 227         }
 228 }

 230 void statistics_dialog::on_scenario_select(window& window)
 231 {
 232         const size_t new_index = find_widget<menu_button>(&window, "scenario_menu", false).get_value();
 233 
 234         if(scenario_index_ != new_index) {
 235                 scenario_index_ = new_index;
 236 
 237                 update_lists(window);
 238         }
 239 }

 241 void statistics_dialog::on_primary_list_select(window& window)
 242 {
 243         const int selected_row = find_widget<listbox>(&window, "stats_list_main", false).get_selected_row();
 244         if(selected_row == -1) {
 245                 return;
 246         }
 247 
 248         listbox& unit_list = find_widget<listbox>(&window, "stats_list_units", false);
 249 
 250         unit_list.clear();
 251 
 252         for(const auto& i : *main_stat_table_[selected_row]) {
 253                 const unit_type* type = unit_types.find(i.first);
 254                 if(!type) {
 255                         continue;
 256                 }
 257 
 258                 std::map<std::string, string_map> data;
 259                 string_map item;
 260 
 261                 item["label"] = (formatter() << type->image() << "~RC(" << type->flag_rgb() << ">" << current_team_.color() << ")").str();
 262                 data.emplace("unit_image", item);
 263 
 264                 item["label"] = type->type_name();
 265                 data.emplace("unit_name", item);
 266 
 267                 item["label"] = (formatter() << i.second << font::unicode_multiplication_sign).str();
 268                 data.emplace("unit_count", item);
 269 
 270                 unit_list.add_row(data);
 271         }
