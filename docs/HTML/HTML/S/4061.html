 265     def add_units(self, campaign):
 266         """
 267         We assume each unit, in mainline and all addons, has one unique id. So
 268         we reference them everywhere by this id, and here can add them all to
 269         one big collection.
 270         """
 271         addunits = self.parser.get_all(tag="units")
 272         addunits += self.parser.get_all(tag="+units")
 273         if not addunits:
 274             return 0

 277             res = []
 278             for units in addunits:
 279                 res += units.get_all(tag=oftype)
 280             return res

 283         newunits = getall("unit_type") + getall("unit")
 284         for unit in newunits:
 285             uid = unit.get_text_val("id")

 288             if unit.get_text_val("do_not_list", "no") not in ["no", "false"] or \
 289                unit.get_text_val("hide_help", "no") not in ["no", "false"]:
 290                 unit.hidden = True

 297                 # with the same id and if one is different try
 298                 # to do something clever like rename it...
 299             else:
 300                 self.unit_lookup[uid] = unit
 301             if not hasattr(unit, "campaigns"):
 302                 unit.campaigns = []
 303             unit.campaigns.append(campaign)
 304 
 305         # Find all races.
 306         newraces = getall("race")
 307         for race in newraces:
 308             rid = race.get_text_val("id")
 309             if rid is None:
 310                 rid = race.get_text_val("name")
 311             self.race_lookup[rid] = race

 313         # Find all movetypes.
 314         newmovetypes = getall("movetype")
 315         for movetype in newmovetypes:
 316             mtname = movetype.get_text_val("name")
 317             self.movetype_lookup[mtname] = movetype
 318 
 319         # Store race/movetype/faction of each unit for easier access later.
 320         for unit in newunits:
 321             uid = unit.get_text_val("id")
 322             race = self.get_unit_value(unit, "race")
 323             try:
 324                 unit.race = self.race_lookup[race]
 325                 unit.rid = unit.race.get_text_val("id", "none")
 326             except KeyError:
 327                 unit.race = None
 328                 unit.rid = "none"

 330                     race, unit.get_text_val("id")))

 332             try:
 333                 unit.movetype = self.movetype_lookup[movetype]
 334             except KeyError:
 335                 unit.movetype = None

 443     def create_network(self):
 444         """
 445         Assuming that each node which has been added to the tree only has a
 446         valid list of children in unit.child_ids, also fill in unit.parent_ids
 447         and update the unit.children shortcut.
 448         """
 449 
 450         # Complete the network
 451         for uid, u in list(self.lookup.items()):
 452             for cid in u.child_ids:
 453                 c = self.lookup.get(cid, None)
 454                 if not c:
 455                     continue
 456                 u.children.append(c)
 457                 if not uid in c.parent_ids:
 458                     c.parent_ids.append(uid)
 459 
 460         # Put all roots into the forest
 461         for uid, u in list(self.lookup.items()):
 462             if not u.parent_ids:
 463                 self.trees[uid] = u
 464 
 465         # Sanity check because some argGRRxxx addons have units who advance to
 466         # themselves.
 467 
 468         def recurse(u, already):
 469             already2 = already.copy()
 470             for c in u.children[:]:
 471                 already2[c.id] = True
 472                 if c.id in already:
 473                     error_message(
 474                         ("Warning: Unit %s advances to unit %s in a loop.\n" %
 475                          (u.id, c.id)) +
 476                         ("    Removing advancement %s.\n" % c.id))
 477                     u.children.remove(c)
 478             for c in u.children:
 479                 recurse(c, already2)
 480         for u in list(self.trees.values()):
 481             already = {u.id : True}
 482             recurse(u, already)
 483 
 484     def update(self):
 485         self.create_network()
 486         self.breadth = sum([x.update_breadth() for x in list(self.trees.values())])
 487         return self.breadth
 488 
 489     def get_children(self, uid):

 497 class UnitNode:

 503         self.children = []
 504         self.id = unit.get_text_val("id")
 505         self.child_ids = []
 506         self.parent_ids = []
 507         self.child_ids.extend(unit.advance)

 509     def update_breadth(self):
 510         if not self.children:
 511             self.breadth = 1
 512         else:
 513             self.breadth = sum([x.update_breadth() for x in self.children])
