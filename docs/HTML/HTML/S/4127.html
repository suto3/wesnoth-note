   3 """
   4 wmliterator.py -- Python routines for navigating a Battle For Wesnoth WML tree
   5 Author: Sapient (Patrick Parker), 2007
   6 
   7 Purpose:
   8  The WmlIterator class can be used to analyze and search the structure of WML
   9  files non-invasively (i.e. preserving existing line structure), and its main
  10  use is to determine when a transformation of deprecated content needs to take
  11  place. (I wrote it was because wmllint was trying to do a lot of things with
  12  regular expressions which really required a more algorithmic approach. Also,
  13  wmllint was often inconsistent with correct handling of comments and values
  14  inside strings.)
  15 
  16 Limitations:
  17  The WmlIterator does not attempt to expand macros, only to recognize them as
  18  another level of nesting. Also, the handling of multiple assignment syntax
  19  is somewhat limited (for similar reasons). Adding seamless support for these
  20  would be ideal, but it presents a design challenge since the iteration is
  21  supposed to be non-invasive. Thus, the current behavior is considered good
  22  enough for now.
  23 """

 105 class WmlIterator(object):
 106     """Return an iterable WML navigation object.
 107     Initialize with a list of lines or a file; if the the line list is
 108     empty and the filename is specified, lines will be read from the file.
 109 
 110     Note: if changes are made to lines while iterating, this may produce
 111     unexpected results. In such case, seek() to the line number of a
 112     scope behind where changes were made.
 113 Important Attributes:
 114     lines - this is an internal list of all the physical lines
 115     scopes - this is an internal list of all open scopes (as iterators)
 116              note: when retrieving an iterator from this list, always
 117              use a copy to perform seek() or next(), and not the original
 118     element - the wml tag, key, or macro name for this logical line
 119               (in complex cases, this may be a tuple of elements...
 120               see parseElements for list of possible values)
 121     text - the exact text of this logical line, as it appears in the
 122            original source, and ending with a newline
 123            note: the logical line also includes multi-line quoted strings
 124     span - the number of physical lines in this logical line:
 125            always 1, unless text contains a multi-line quoted string
 126     lineno - a zero-based line index marking where this text begins
 127     """
 128     def __init__(self, lines=None, filename=None, begin=-1):
 129         "Initialize a new WmlIterator."
 130         self.fname = filename
 131         if lines is None:
 132             lines = []
 133             if filename:
 134                 try:
 135                     with codecs.open(self.fname, "r", "utf8") as ifp:
 136                         lines = ifp.readlines()
 137                 except Exception:
 138                     self.printError('error opening file')
 139         self.lines = lines
 140         self.reset()
 141         self.seek(begin)

 143     def parseQuotes(self, lines):
 144         """Return the line or multiline text if a quote spans multiple lines"""
 145         text = lines[self.lineno]
 146         span = 1
 147         begincomment = text.find('#')
 148         if begincomment < 0:
 149             begincomment = None
 150         beginquote = text[:begincomment].find('<<')
 151         while beginquote >= 0:
 152             endquote = -1
 153             beginofend = beginquote+2
 154             while endquote < 0:
 155                 endquote = text.find('>>', beginofend)
 156                 if endquote < 0:
 157                     if self.lineno + span >= len(lines):
 158                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 159                         return text, span
 160                     beginofend = len(text)
 161                     text += lines[self.lineno + span]
 162                     span += 1
 163             begincomment = text.find('#', endquote+2)
 164             if begincomment < 0:
 165                 begincomment = None
 166             beginquote = text[:begincomment].find('<<', endquote+2)
 167         beginquote = text[:begincomment].find('"')
 168         while beginquote >= 0:
 169             endquote = -1
 170             beginofend = beginquote+1
 171             while endquote < 0:
 172                 endquote = text.find('"', beginofend)
 173                 if endquote < 0:
 174                     if self.lineno + span >= len(lines):
 175                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 176                         return text, span
 177                     beginofend = len(text)
 178                     text += lines[self.lineno + span]
 179                     span += 1
 180             begincomment = text.find('#', endquote+1)
 181             if begincomment < 0:
 182                 begincomment = None
 183             beginquote = text[:begincomment].find('"', endquote+1)
 184         return text, span

 186     def closeScope(self, scopes, closerElement):
 187         """Close the most recently opened scope. Return false if not enough scopes.
 188         note: directives close all the way back to the last open directive
 189         non-directives cannot close a directive and will no-op in that case."""
 190         try:
 191             if isDirective(closerElement):
 192                 while not isDirective(scopes.pop()):
 193                     pass
 194             elif (closerElement==closeMacroType):
 195                 elem = ''
 196                 while not elem.startswith('{'):
 197                     closed = scopes.pop()
 198                     elem = closed
 199                     if isinstance(closed, WmlIterator):
 200                         elem = closed.element
 201                     if isDirective(elem):
 202                         self.printScopeError(closerElement)
 203                         scopes.append(closed) # to reduce additional errors (hopefully)
 204                         return True
 205             elif not isDirective(scopes[-1]):
 206                 closed = scopes.pop()
 207                 elem = closed
 208                 if isinstance(closed, WmlIterator):
 209                     elem = closed.element
 210                 if (elem.startswith('{') and closerElement != closeMacroType):
 211                     scopes.append(closed)
 212                 elif (isOpener(elem) and closerElement != '[/'+elem[1:]
 213                 and '+'+closerElement != elem[1]+'[/'+elem[2:]):
 214                     self.printError('reached', closerElement, 'at line', self.lineno+1, 'before closing scope', elem)
 215                     scopes.append(closed) # to reduce additional errors (hopefully)
 216             return True
 217         except IndexError:
 218             return False

 220     def parseElements(self, text):
 221         """Remove any closed scopes, return a list of element names
 222         and list of new unclosed scopes
 223     Element Types:
 224         tags: one of "[tag_name]" or "[/tag_name]"
 225             [tag_name] - opens a scope
 226             [/tag_name] - closes a scope
 227         keys: either "key=" or ("key1=", "key2=") for multi-assignment
 228             key= - does not affect the scope
 229             key1,key2= - multi-assignment returns multiple elements
 230         directives: one of "#ifdef", "#ifndef", "#ifhave", "#ifnhave", "#ifver", "#ifnver", "#else", "#endif", "#define", "#enddef"
 231             #ifdef - opens a scope
 232             #ifndef - opens a scope
 233             #ifhave - opens a scope
 234             #ifnhave - opens a scope
 235             #ifver - opens a scope
 236             #ifnver - opens a scope
 237             #else - closes a scope, also opens a new scope
 238             #endif - closes a scope
 239             #define - opens a scope
 240             #enddef - closes a scope
 241         macro calls: "{MACRO_NAME}"
 242             {MACRO_NAME - opens a scope
 243             } - closes a scope
 244         """
 245         elements = [] #(elementType, sortPos, scopeDelta)
 246         # first remove any lua strings
 247         beginquote = text.find('<<')
 248         while beginquote >= 0:
 249             endquote = text.find('>>', beginquote+2)
 250             if endquote < 0:
 251                 text = text[:beginquote]
 252                 beginquote = -1 #terminate loop
 253             else:
 254                 text = text[:beginquote] + text[endquote+2:]
 255                 beginquote = text.find('<<')
 256         # remove any quoted strings
 257         beginquote = text.find('"')
 258         while beginquote >= 0:
 259             endquote = text.find('"', beginquote+1)
 260             if endquote < 0:
 261                 text = text[:beginquote]
 262                 beginquote = -1 #terminate loop
 263             else:
 264                 text = text[:beginquote] + text[endquote+1:]
 265                 beginquote = text.find('"')
 266         # next remove any comments
 267         text = text.lstrip()
 268         commentSearch = 1
 269         if text.startswith('#ifdef'):
 270             return (['#ifdef'],)*2
 271         elif text.startswith('#ifndef'):
 272             return (['#ifndef'],)*2
 273         elif text.startswith('#ifhave'):
 274             return (['#ifhave'],)*2
 275         elif text.startswith('#ifnhave'):
 276             return (['#ifnhave'],)*2
 277         elif text.startswith('#ifver'):
 278             return (['#ifver'],)*2
 279         elif text.startswith('#ifnver'):
 280             return (['#ifnver'],)*2
 281         elif text.startswith('#else'):
 282             if not self.closeScope(self.scopes, '#else'):
 283                 self.printScopeError('#else')
 284             return (['#else'],)*2
 285         elif text.startswith('#endif'):
 286             if not self.closeScope(self.scopes, '#endif'):
 287                 self.printScopeError('#endif')
 288                 return ['#endif'], []
 289         elif text.startswith('#define'):
 290             return (['#define'],)*2
 291         elif text.find('#enddef') >= 0:
 292             elements.append(('#enddef', text.find('#enddef'), -1))
 293         elif text.startswith('#po:') or text.startswith('# po:'):
 294             elements.append(("#po", 0, 0))
 295         else:
 296             commentSearch = 0
 297         begincomment = text.find('#', commentSearch)
 298         if begincomment >= 0:
 299             text = text[:begincomment]
 300         #now find elements in a loop
 301         for m in tagPattern.finditer(text):
 302             delta = 1
 303             if isCloser(m.group(2)):
 304                 delta = -1
 305             elements.append((m.group(2), m.start(), delta))
 306         for m in keyPattern.finditer(text):
 307             for i, k in enumerate(keySplit.split(m.group(0))):
 308                 if k:
 309                     elements.append((k+'=', m.start()+i, 0))
 310         for m in macroOpenPattern.finditer(text):
 311             elements.append((m.group(1), m.start(), 1))
 312         for m in macroClosePattern.finditer(text):
 313             elements.append((closeMacroType, m.start(), -1))
 314         #sort by start position
 315         elements.sort(key=lambda x:x[1])
 316         resultElements = []
 317         openedScopes = []
 318         for elem, sortPos, scopeDelta in elements:
 319             while scopeDelta < 0:
 320                 if not(self.closeScope(openedScopes, elem)\
 321                         or self.closeScope(self.scopes, elem)):
 322                     self.printScopeError(elem)
 323                 scopeDelta += 1
 324             while scopeDelta > 0:
 325                 openedScopes.append(elem)
 326                 scopeDelta -= 1
 327             resultElements.append(elem)
 328         return resultElements, openedScopes

 356     def seek(self, lineno, clearEnd=True):
 357         """Move the iterator to a specific line number"""
 358         if clearEnd:
 359             self.endScope = None
 360         if lineno < self.lineno:
 361             for scope in reversed(self.scopes):
 362                 # if moving backwards, try to re-use a scope iterator
 363                 if scope.lineno <= lineno:
 364                     # copy the scope iterator's state to self
 365                     self.__dict__ = dict(scope.__dict__)
 366                     self.scopes = scope.scopes[:]
 367                     self.nextScopes = scope.nextScopes[:]
 368                     break
 369             else:
 370                 # moving backwards past all scopes forces a reset
 371                 self.reset()
 372         while self.lineno + self.span - 1 < lineno:
 373             next(self)
 374         return self

 407     def __next__(self):
 408         """Move the iterator to the next line number
 409         note: May raise StopIteration"""
 410         if not self.hasNext():
 411             if self.scopes:
 412                 self.printError("reached EOF with open scopes", self.scopes)
 413             raise StopIteration
 414         self.lineno = self.lineno + self.span
 415         self.text, self.span = self.parseQuotes(self.lines)
 416         self.scopes.extend(self.nextScopes)
 417         self.element, nextScopes = self.parseElements(self.text)
 418         self.nextScopes = []
 419         for elem in nextScopes:
 420         # remember scopes by storing a copy of the iterator
 421             copyItor = self.copy()
 422             copyItor.element = elem
 423             self.nextScopes.append(copyItor)
 424             copyItor.nextScopes.append(copyItor)
 425         if(len(self.element) == 1):
 426             # currently we only wish to handle simple single assignment syntax
 427             self.element = self.element[0]
 428         if self.endScope is not None and not self.scopes.count(self.endScope):
 429             raise StopIteration
 430         return self

 458     def printError(nav, *misc):
 459         """Print error associated with a given file; avoid printing duplicates"""
 460         if nav.fname:
 461             silenceValue = ' '.join(map(str, misc))
 462             if nav.fname not in silenceErrors:
 463                 print(nav.fname, file=sys.stderr)
 464                 silenceErrors[nav.fname] = set()
 465             elif silenceValue in silenceErrors[nav.fname]:
 466                 return # do not print a duplicate error for this file
 467             silenceErrors[nav.fname].add(silenceValue)
 468         print('wmliterator:', end=" ", file=sys.stderr)
 469         for item in misc:
 470             print(item, end=" ", file=sys.stderr)
