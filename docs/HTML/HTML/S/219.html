 108 bool addons_client::upload_addon(const std::string& id, std::string& response_message, config& cfg)
 109 {
 110         LOG_ADDONS << "preparing to upload " << id << '\n';
 111 
 112         response_message.clear();
 113 
 114         utils::string_map i18n_symbols;
 115         i18n_symbols["addon_title"] = cfg["title"];
 116         if(i18n_symbols["addon_title"].empty()) {
 117                 i18n_symbols["addon_title"] = make_addon_title(id);
 118         }
 119 
 120         if(!addon_name_legal(id)){
 121                 i18n_symbols["addon_id"] = id;
 122                 this->last_error_ =
 123                         vgettext("The add-on <i>$addon_title</i> has an invalid id '$addon_id' "
 124                                 "and cannot be published.", i18n_symbols);
 125                 return false;
 126         }
 127 
 128         std::string passphrase = cfg["passphrase"];
 129         // generate a random passphrase and write it to disk
 130         // if the .pbl file doesn't provide one already
 131         if(passphrase.empty()) {
 132                 passphrase.resize(8);
 133                 for(size_t n = 0; n != 8; ++n) {
 134                         passphrase[n] = 'a' + (rand()%26);
 135                 }
 136                 cfg["passphrase"] = passphrase;
 137                 set_addon_pbl_info(id, cfg);
 138 
 139                 LOG_ADDONS << "automatically generated an initial passphrase for " << id << '\n';
 140         }
 141 
 142         cfg["name"] = id;
 143 
 144         config addon_data;
 145         archive_addon(id, addon_data);
 146 
 147         std::vector<std::string> badnames;
 148         if(!check_names_legal(addon_data, &badnames)){
 149                 this->last_error_ =
 150                         vgettext("The add-on <i>$addon_title</i> has an invalid file or directory "
 151                                 "name and cannot be published.", i18n_symbols);
 152                 this->last_error_data_ = utils::join(badnames, "\n");
 153                 return false;
 154         }
 155 
 156         config request_buf, response_buf;
 157         request_buf.add_child("upload", cfg).add_child("data", addon_data);
 158 
 159         LOG_ADDONS << "sending " << id << '\n';
 160 
 161         this->send_request(request_buf, response_buf);
 162         this->wait_for_transfer_done(vgettext("Sending add-on <i>$addon_title</i>...", i18n_symbols
 163         ), true);
 164 
 165         if(const config& message_cfg = response_buf.child("message")) {
 166                 response_message = message_cfg["message"].str();
 167                 LOG_ADDONS << "server response: " << response_message << '\n';
 168         }
 169 
 170         return !this->update_last_error(response_buf);
 171 
 172 }

 227 bool addons_client::install_addon(config& archive_cfg, const addon_info& info)
 228 {
 229         const cursor::setter cursor_setter(cursor::WAIT);
 230 
 231         utils::string_map i18n_symbols;
 232         i18n_symbols["addon_title"] = info.title;
 233 
 234         if(!check_names_legal(archive_cfg)) {
 235                 gui2::show_error_message(v_,
 236                         vgettext("The add-on <i>$addon_title</i> has an invalid file or directory "
 237                                 "name and cannot be installed.", i18n_symbols));
 238                 return false;
 239         }
 240 
 241         // Add local version information before unpacking
 242 
 243         config* maindir = &archive_cfg.find_child("dir", "name", info.id);
 244         if(!*maindir) {
 245                 LOG_ADDONS << "downloaded add-on '" << info.id << "' is missing its directory in the archive; creating it\n";
 246                 maindir = &archive_cfg.add_child("dir");
 247                 (*maindir)["name"] = info.id;
 248         }
 249 
 250         LOG_ADDONS << "generating version info for add-on '" << info.id << "'\n";
 251 
 252         std::ostringstream info_contents;
 253         config wml;
 254 
 255         info_contents <<
 256                 "#\n"
 257                 "# File automatically generated by Wesnoth to keep track\n"
 258                 "# of version information on installed add-ons. DO NOT EDIT!\n"
 259                 "#\n";
 260 
 261         info.write_minimal(wml.add_child("info"));
 262         write(info_contents, wml);
 263 
 264         config file;
 265         file["name"] = "_info.cfg";
 266         file["contents"] = info_contents.str();
 267 
 268         maindir->add_child("file", file);
 269 
 270         LOG_ADDONS << "unpacking " << info.id << '\n';
 271 
 272         // Remove any previously installed versions
 273         if(!remove_local_addon(info.id)) {
 274                 WRN_ADDONS << "failed to uninstall previous version of " << info.id << "; the add-on may not work properly!" << std::endl;
 275         }
 276 
 277         unarchive_addon(archive_cfg);
 278         LOG_ADDONS << "unpacking finished\n";
 279 
 280         return true;
 281 }

 283 bool addons_client::try_fetch_addon(const addon_info & addon)
 284 {
 285         config archive;
 286 
 287         if(!(
 288                 download_addon(archive, addon.id, addon.title, !is_addon_installed(addon.id)) &&
 289                 install_addon(archive, addon)
 290                 )) {
 291                 const std::string& server_error = get_last_server_error();
 292                 if(!server_error.empty()) {
 293                         gui2::show_error_message(v_,
 294                                 _("The server responded with an error:") + "\n" + server_error);
 295                 }
 296                 return false;
 297         } else {

 302 addons_client::install_result addons_client::do_resolve_addon_dependencies(const addons_list& addons, const addon_info& addon)
 303 {
 304         install_result result;
 305         result.outcome = install_outcome::success;
 306         result.wml_changed = false;
 307 
 308         std::unique_ptr<cursor::setter> cursor_setter(new cursor::setter(cursor::WAIT));
 309 
 310         // TODO: We don't currently check for the need to upgrade. I'll probably
 311         // work on that when implementing dependency tiers later.
 312 
 313         const std::set<std::string>& deps = addon.resolve_dependencies(addons);
 314 
 315         std::vector<std::string> missing_deps;
 316         std::vector<std::string> broken_deps;
 317 
 318         for(const std::string& dep : deps) {
 319                 if(!is_addon_installed(dep)) {
 320                         if(addons.find(dep) != addons.end()) {
 321                                 missing_deps.push_back(dep);
 322                         } else {
 323                                 broken_deps.push_back(dep);
 324                         }
 325                 }
 326         }
 327 
 328         cursor_setter.reset();
 329 
 330         if(!broken_deps.empty()) {
 331                 std::string broken_deps_report;
 332 
 333                 broken_deps_report = _n(
 334                         "The selected add-on has the following dependency, which is not currently installed or available from the server. Do you wish to continue?",
 335                         "The selected add-on has the following dependencies, which are not currently installed or available from the server. Do you wish to continue?",
 336                         broken_deps.size());
 337                 broken_deps_report += "\n";
 338 
 339                 for(const std::string& broken_dep_id : broken_deps) {
 340                         broken_deps_report += "\n    " + font::unicode_bullet + " " + make_addon_title(broken_dep_id);
 341                 }
 342 
 343                 if(gui2::show_message(v_, _("Broken Dependencies"), broken_deps_report, gui2::dialogs::message::yes_no_buttons) != gui2::window::OK) {
 344                         result.outcome = install_outcome::abort;
 345                         return result; // canceled by user
 346                 }
 347         }
 348 
 349         if(missing_deps.empty()) {
 350                 // No dependencies to install, carry on.
 351                 return result;
 352         }
 353 
 354         {
 355                 addons_list options;
 356                 for(const std::string& dep : missing_deps) {
 357                         const addon_info& missing_addon = addons.at(dep);
 358                         options[dep] = missing_addon;
 359                 }
 360 
 361                 gui2::dialogs::install_dependencies dlg(options);
 362                 bool cont = dlg.show(v_);
 363                 if(!cont) {
 364                         return result; // the user has chosen to continue without installing anything.
 365                 }
 366         }
 367 
 368         //
 369         // Install dependencies now.
 370         //
 371 
 372         std::vector<std::string> failed_titles;
 373 
 374         for(const std::string& dep : missing_deps) {
 375                 const addon_info& missing_addon = addons.at(dep);
 376 
 377                 if(!try_fetch_addon(missing_addon)) {
 378                         failed_titles.push_back(missing_addon.title);
 379                 } else {
 380                         result.wml_changed = true;
 381                 }
 382         }
 383 
 384         if(!failed_titles.empty()) {
 385                 const std::string& failed_deps_report = _n(
 386                         "The following dependency could not be installed. Do you still wish to continue?",
 387                         "The following dependencies could not be installed. Do you still wish to continue?",
 388                         failed_titles.size()) + std::string("\n\n") + utils::bullet_list(failed_titles);
 389 
 390                 result.outcome = gui2::show_message(v_, _("Dependencies Installation Failed"), failed_deps_report, gui2::dialogs::message::yes_no_buttons) == gui2::window::OK ? install_outcome::success : install_outcome::abort; // If the user cancels, return abort. Otherwise, return success, since the user chose to ignore the failure.
 391                 return result;
 392         }
 393 
 394         return result;
 395 }

 397 bool addons_client::do_check_before_overwriting_addon(const addon_info& addon)
 398 {
 399         const std::string& addon_id = addon.id;
 400 
 401         const bool pbl = have_addon_pbl_info(addon_id);
 402         const bool vcs = have_addon_in_vcs_tree(addon_id);
 403 
 404         if(!pbl && !vcs) {
 405                 return true;
 406         }
 407 
 408         utils::string_map symbols;
 409         symbols["addon"] = addon.title;
 410         std::string text;
 411         std::vector<std::string> extra_items;
 412 
 413         text = vgettext("The add-on '$addon|' is already installed and contains additional information that will be permanently lost if you continue:", symbols);
 414         text += "\n\n";
 415 
 416         if(pbl) {
 417                 extra_items.push_back(_("Publishing information file (.pbl)"));
 418         }
 419 
 420         if(vcs) {
 421                 extra_items.push_back(_("Version control system (VCS) information"));
 422         }
 423 
 424         text += utils::bullet_list(extra_items) + "\n\n";
 425         text += _("Do you really wish to continue?");
 426 
 427         return gui2::show_message(v_, _("Confirm"), text, gui2::dialogs::message::yes_no_buttons) == gui2::window::OK;
 428 }

 430 addons_client::install_result addons_client::install_addon_with_checks(const addons_list& addons, const addon_info& addon)
 431 {
 432         if(!(do_check_before_overwriting_addon(addon))) {
 433                 // Just do nothing and leave.
 434                 install_result result;
 435                 result.outcome = install_outcome::abort;
 436                 result.wml_changed = false;
 437 
 438                 return result;
 439         }
 440 
 441         // Resolve any dependencies
 442         install_result res = do_resolve_addon_dependencies(addons, addon);
 443         if(res.outcome != install_outcome::success) { // this function only returns SUCCESS and ABORT as outcomes
 444                 return res; // user aborted
 445         }
 446 
 447         if(!try_fetch_addon(addon)) {
 448                 res.outcome = install_outcome::failure;
 449                 return res; //wml_changed should have whatever value was obtained in resolving dependencies
 450         } else {
 451                 res.wml_changed = true;
 452                 return res; //we successfully installed something, so now the wml was definitely changed
 453         }
 454 }

 456 bool addons_client::update_last_error(config& response_cfg)
 457 {
 458         if(config const &error = response_cfg.child("error")) {
 459                 this->last_error_ = error["message"].str();
 460                 this->last_error_data_ = error["extra_data"].str();
 461                 ERR_ADDONS << "server error: " << error << '\n';
 462                 return true;
 463         } else {
 464                 this->last_error_.clear();
 465                 this->last_error_data_.clear();
 466                 return false;
 467         }

 513 void addons_client::wait_for_transfer_done(const std::string& status_message, bool track_upload)
 514 {
 515         check_connected();
 516         std::unique_ptr<network_transmission::connection_data> cd;
 517         if(track_upload)
 518                 cd.reset(new write_addon_connection_data{ *conn_ });
 519         else
 520                 cd.reset(new read_addon_connection_data{ *conn_ });
 521         if(!stat_) {
 522                 stat_.reset(new network_transmission(*cd, _("Add-ons Manager"), status_message));
 523         } else {
 524                 stat_->set_subtitle(status_message);
 525                 stat_->set_connection_data(*cd);
 526         }
 527 
 528         if(!stat_->show(v_)) {
 529                 // Notify the caller chain that the user aborted the operation.
 530                 throw user_exit();
 531         }
 532 }
