  73 struct unit_filter_adjacent : public unit_filter_base

  76                 : child_(cfg)
  77                 , cfg_(cfg)
  78         {
  79         }

  81         virtual bool matches(const unit_filter_args& args) const override
  82         {
  83                 const unit_map& units = args.fc->get_disp_context().units();
  84                 map_location adjacent[6];
  85                 get_adjacent_tiles(args.loc, adjacent);
  86                 int match_count=0;
  87 
  88                 config::attribute_value i_adjacent = cfg_["adjacent"];
  89                 std::vector<map_location::DIRECTION> dirs;
  90                 if (i_adjacent.empty()) {
  91                         dirs = map_location::default_dirs();
  92                 } else {
  93                         dirs = map_location::parse_directions(i_adjacent);
  94                 }
  95                 for (map_location::DIRECTION dir : dirs) {
  96                         unit_map::const_iterator unit_itor = units.find(adjacent[dir]);
  97                         if (unit_itor == units.end() || !child_.matches(unit_filter_args{*unit_itor, unit_itor->get_location(), &args.u, args.fc, args.use_flat_tod} )) {
  98                                 continue;
  99                         }
 100                         auto is_enemy = cfg_["is_enemy"];
 101                         if (!is_enemy.empty() && is_enemy.to_bool() != args.fc->get_disp_context().get_team(args.u.side()).is_enemy(unit_itor->side())) {
 102                                 continue;
 103                         }
 104                         ++match_count;
 105                 }
 106 
 107                 static std::vector<std::pair<int,int> > default_counts = utils::parse_ranges("1-6");
 108                 config::attribute_value i_count = cfg_["count"];
 109                 return in_ranges(match_count, !i_count.blank() ? utils::parse_ranges(i_count) : default_counts);
 110         }

 183 bool unit_filter_compound::matches(const unit_filter_args& args) const
 184 {
 185         bool res;
 186 
 187         if(args.loc.valid()) {
 188                 scoped_xy_unit auto_store("this_unit", args.loc, args.fc->get_disp_context().units());
 189                 if (args.u2) {
 190                         const map_location& loc2 = args.u2->get_location();
 191                         scoped_xy_unit u2_auto_store("other_unit", loc2, args.fc->get_disp_context().units());
 192                         res = filter_impl(args);
 193                 } else {
 194                         res = filter_impl(args);
 195                 }
 196         } else {
 197                 // If loc is invalid, then this is a recall list unit (already been scoped)
 198                 res = filter_impl(args);
 199         }
 200 
 201         // Handle [and], [or], and [not] with in-order precedence
 202         for(const auto & filter : cond_children_) {
 203                 bool child_res = filter.second.matches(args);
 204 
 205                 switch (filter.first.v) {
 206                 case CONDITIONAL_TYPE::AND:
 207                         res = res && child_res;
 208                         break;
 209                 case CONDITIONAL_TYPE::OR:
 210                         res = res || child_res;
 211                         break;
 212                 case CONDITIONAL_TYPE::NOT:
 213                         res = res && !child_res;
 214                         break;
 215                 }
 216         }
 217         return res;
 218 }

 220 bool unit_filter_compound::filter_impl(const unit_filter_args& args) const
 221 {
 222         for(const auto & filter : children_) {
 223                 if (!filter->matches(args)) {
 224                         return false;
 225                 }
 226         }
 227         return true;

 631                                         if (fwml.all_children_count() == 1 && fwml.attribute_count() == 1 && ci.front().key == "variables") {
 632                                                 return args.u.variables().matches(ci.front().cfg);
 633                                         } else {
 634                                                 config ucfg;
 635                                                 args.u.write(ucfg);
 636                                                 return ucfg.matches(fwml);
 637                                         }
 638                                 });
 639                         }
 640                         else if (child.first == "filter_vision") {
 641                                 create_child(child.second, [](const vconfig& c, const unit_filter_args& args) {
 642                                         std::set<int> viewers;
 643                                         side_filter ssf(c, args.fc);
 644                                         std::vector<int> sides = ssf.get_teams();
 645                                         viewers.insert(sides.begin(), sides.end());
 646 
 647                                         for (const int viewer : viewers) {
 648                                                 bool fogged = args.fc->get_disp_context().get_team(viewer).fogged(args.loc);
 649                                                 bool hiding = args.u.invisible(args.loc, args.fc->get_disp_context()) && args.fc->get_disp_context().get_team(viewer).is_enemy(args.u.side());
 650                                                 bool unit_hidden = fogged || hiding;
 651                                                 if (c["visible"].to_bool(true) != unit_hidden) {
 652                                                         return true;
 653                                                 }
 654                                         }
 655                                         return false;
 656                                 });
 657                         }
 658                         else if (child.first == "filter_adjacent") {
 659                                 children_.emplace_back(new unit_filter_adjacent(child.second));
 660                         }
 661                         else if (child.first == "filter_location") {
 662                                 create_child(child.second, [](const vconfig& c, const unit_filter_args& args) {
 663                                         return terrain_filter(c, args.fc, args.use_flat_tod).match(args.loc);
 664                                 });
 665                         }
 666                         else if (child.first == "filter_side") {
 667                                 create_child(child.second, [](const vconfig& c, const unit_filter_args& args) {
 668                                         return side_filter(c, args.fc).match(args.u.side());
 669                                 });
 670                         }
 671                         else if (child.first == "has_attack") {
 672                                 create_child(child.second, [](const vconfig& c, const unit_filter_args& args) {
 673                                         for(const attack_type& a : args.u.attacks()) {
 674                                                 if(a.matches_filter(c.get_parsed_config())) {
 675                                                         return true;
 676                                                 }
 677                                         }
 678                                         return false;
 679                                 });
 680                         }
 681                         else {
 682                                 std::stringstream errmsg;
 683                                 errmsg << "encountered a child [" << child.first << "] of a standard unit filter, it is being ignored";
 684                                 DBG_CF << errmsg.str() << std::endl;
 685                         }
 686 
 687                 }
