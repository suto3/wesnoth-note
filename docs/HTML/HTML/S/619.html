  36 tree_view_node::tree_view_node(
  37                 const std::string& id,
  38                 tree_view_node* parent_node,
  39                 tree_view& parent_tree_view,
  40                 const std::map<std::string /* widget id */, string_map>& data)
  41         : widget()
  42         , parent_node_(parent_node)
  43         , tree_view_(&parent_tree_view)
  44         , grid_()
  45         , children_()
  46         , toggle_(nullptr)
  47         , label_(nullptr)
  48         , unfolded_(false)
  49 {
  50         grid_.set_parent(this);
  51         set_parent(&parent_tree_view);
  52 
  53         if(id == "root") {
  54                 unfolded_ = true;
  55                 return;
  56         }
  57 
  58         for(const auto& node_definition : get_tree_view().get_node_definitions()) {
  59                 if(node_definition.id != id) {
  60                         continue;
  61                 }
  62 
  63                 node_definition.builder->build(&grid_);
  64                 init_grid(&grid_, data);
  65 
  66                 if(parent_node_ && parent_node_->toggle_) {
  67                         dynamic_cast<widget&>(*parent_node_->toggle_).set_visible(widget::visibility::visible);
  68                 }
  69 
  70                 if(node_definition.unfolded) {
  71                         unfolded_ = true;
  72                 }
  73 
  74                 widget* toggle_widget = grid_.find("tree_view_node_toggle", false);
  75                 toggle_ = dynamic_cast<selectable_item*>(toggle_widget);
  76 
  77                 if(toggle_) {
  78                         toggle_widget->set_visible(widget::visibility::hidden);
  79 
  80                         toggle_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  81                                         &tree_view_node::signal_handler_left_button_click, this, _2));
  82                         toggle_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  83                                         &tree_view_node::signal_handler_left_button_click, this, _2),
  84                                         event::dispatcher::back_post_child);
  85 
  86                         if(unfolded_) {
  87                                 toggle_->set_value(1);
  88                         }
  89                 }
  90 
  91                 widget* label_widget = grid_.find("tree_view_node_label", false);
  92                 label_ = dynamic_cast<selectable_item*>(label_widget);
  93 
  94                 if(label_) {
  95                         label_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  96                                         &tree_view_node::signal_handler_label_left_button_click, this, _2, _3, _4),
  97                                         event::dispatcher::front_child);

 102                         if(!get_tree_view().selected_item_) {

 130 tree_view_node& tree_view_node::add_child(
 131                 const std::string& id,
 132                 const std::map<std::string /* widget id */, string_map>& data,
 133                 const int index)
 134 {
 135         auto itor = children_.end();
 136 
 137         if(static_cast<size_t>(index) < children_.size()) {
 138                 itor = children_.begin() + index;
 139         }
 140 
 141         itor = children_.emplace(itor, new tree_view_node(id, this, get_tree_view(), data));
 142 
 143         if(is_folded() /*|| is_root_node()*/) {
 144                 return **itor;
 145         }
 146 
 147         if(get_tree_view().get_size() == point()) {
 148                 return **itor;
 149         }
 150 
 151         assert(get_tree_view().content_grid());
 152         const point current_size = get_tree_view().content_grid()->get_size();
 153 
 154         // Calculate width modification.
 155         // This increases tree width if the width of the new node is greater than the current width.
 156         point best_size = (*itor)->get_best_size();
 157         best_size.x += get_indentation_level() * get_tree_view().indentation_step_size_;
 158 
 159         const int width_modification = best_size.x > current_size.x
 160                 ? best_size.x - current_size.x
 161                 : 0;
 162 
 163         // Calculate height modification.
 164         // For this, we only increase height if the best size of the tree (that is, the size with the new node)
 165         // is larger than its current size. This prevents the scrollbar being reserved even when there's obviously
 166         // enough visual space.
 167 
 168         // Throw away cached best size to force a recomputation.
 169         get_tree_view().layout_initialize(false);
 170 
 171         const point tree_best_size = get_tree_view().get_best_size();
 172 
 173         const int height_modification = tree_best_size.y > current_size.y && get_tree_view().layout_size() == point()
 174                 ? tree_best_size.y - current_size.y
 175                 : 0;
 176 
 177         assert(height_modification >= 0);
 178 
 179         // Request new size.
 180         get_tree_view().resize_content(width_modification, height_modification, -1, (*itor)->calculate_ypos());
 181 
 182         return **itor;
 183 }

 247 void tree_view_node::fold_internal()
 248 {
 249         const point current_size(get_current_size().x, get_unfolded_size().y);
 250         const point new_size = get_folded_size();
 251 
 252         const int width_modification = std::max(0, new_size.x - current_size.x);
 253         const int height_modification = new_size.y - current_size.y;
 254         assert(height_modification <= 0);
 255 
 256         get_tree_view().resize_content(width_modification, height_modification, -1, calculate_ypos());
 257         unfolded_ = false;
 258 }

 294 struct tree_view_node_implementation

 296 private:
 297         template <class W, class It>
 298         static W* find_at_aux(It begin,
 299                                                   It end,
 300                                                   const point& coordinate,
 301                                                   const bool must_be_active)
 302         {
 303                 for(It it = begin; it != end; ++it) {
 304                         if(W* widget = (*it)->find_at(coordinate, must_be_active)) {
 305                                 return widget;
 306                         }
 307                 }
 308                 return nullptr;
 309         }

 313         static W* find_at(utils::const_clone_ref<tree_view_node, W> tree_view_node,
 314                                           const point& coordinate,
 315                                           const bool must_be_active)
 316         {
 317                 if(W* widget = tree_view_node.grid_.find_at(coordinate, must_be_active)) {
 318                         return widget;

 321                 if(tree_view_node.is_folded()) {
 322                         return nullptr;
 323                 }
 324 
 325                 return find_at_aux<W>(tree_view_node.children_.begin(),
 326                                                           tree_view_node.children_.end(),
 327                                                           coordinate,
 328                                                           must_be_active);
 329         }

 411 point tree_view_node::get_current_size(bool assume_visible) const
 412 {
 413         if(!assume_visible && parent_node_ && parent_node_->is_folded()) {
 414                 return point();
 415         }
 416 
 417         point size = get_folded_size();
 418         if(is_folded()) {
 419                 return size;
 420         }
 421

 423                 if(node->grid_.get_visible() == widget::visibility::invisible) {
 424                         continue;
 425                 }

 430                 size.x = std::max(size.x, node_size.x);
 431         }

 433         return size;

 466 point tree_view_node::calculate_best_size(const int indentation_level,
 467                                                                                         const unsigned indentation_step_size) const
 468 {
 469         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 470 
 471         point best_size = grid_.get_best_size();
 472         if(indentation_level > 0) {
 473                 best_size.x += indentation_level * indentation_step_size;
 474         }
 475 
 476         DBG_GUI_L << LOG_HEADER << " own grid best size " << best_size << ".\n";
 477 
 478         for(const auto& node : children_) {
 479                 if(node->grid_.get_visible() == widget::visibility::invisible) {
 480                         continue;
 481                 }
 482 
 483                 const point node_size = node->calculate_best_size(indentation_level + 1, indentation_step_size);
 484 
 485                 if(!is_folded()) {
 486                         best_size.y += node_size.y;
 487                 }
 488 
 489                 best_size.x = std::max(best_size.x, node_size.x);
 490         }
 491 
 492         DBG_GUI_L << LOG_HEADER << " result " << best_size << ".\n";
 493         return best_size;
 494 }

 513 unsigned tree_view_node::place(const unsigned indentation_step_size,
 514                                                                 point origin,
 515                                                                 unsigned width)
 516 {
 517         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 518         DBG_GUI_L << LOG_HEADER << " origin " << origin << ".\n";
 519 
 520         const unsigned offset = origin.y;
 521         point best_size = grid_.get_best_size();
 522         best_size.x = width;
 523 
 524         grid_.place(origin, best_size);
 525 
 526         if(!is_root_node()) {
 527                 origin.x += indentation_step_size;
 528                 assert(width >= indentation_step_size);
 529                 width -= indentation_step_size;
 530         }
 531 
 532         origin.y += best_size.y;
 533 
 534         if(is_folded()) {
 535                 DBG_GUI_L << LOG_HEADER << " folded node done.\n";
 536                 return origin.y - offset;
 537         }
 538 
 539         DBG_GUI_L << LOG_HEADER << " set children.\n";
 540         for(auto & node : children_) {
 541                 origin.y += node->place(indentation_step_size, origin, width);
 542         }
 543 
 544         // Inherited.
 545         widget::set_size(point(width, origin.y - offset));
 546 
 547         DBG_GUI_L << LOG_HEADER << " result " << (origin.y - offset) << ".\n";
 548         return origin.y - offset;
 549 }

 603 void tree_view_node::signal_handler_label_left_button_click(
 604                 const event::ui_event event, bool& handled, bool& halt)
 605 {
 606         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 607 
 608         assert(label_);
 609 
 610         // Normally, this is only an event hook and not full handling; however, if
 611         // the currently selected item was selected, we halt the event to prevent
 612         // deselection (which would leave no items selected).
 613         if(label_->get_value()) {
 614                 halt = handled = true;
 615                 return;
 616         }
 617 
 618         // Select the new item if a different one was selected
 619         if(get_tree_view().selected_item_ && get_tree_view().selected_item_->label_) {
 620                 get_tree_view().selected_item_->label_->set_value(false);
 621         }
 622 
 623         get_tree_view().selected_item_ = this;
 624 
 625         if(get_tree_view().selection_change_callback_) {
 626                 get_tree_view().selection_change_callback_(get_tree_view());
 627         }

 677 std::vector<int> tree_view_node::describe_path()
 678 {
 679         if(is_root_node()) {
 680                 return std::vector<int>();
 681         }
 682 
 683         std::vector<int> res = parent_node_->describe_path();
 684         for(size_t i = 0; i < parent_node_->count_children(); ++i) {
 685                 if(parent_node_->children_[i].get() == this) {
 686                         res.push_back(i);
 687                         return res;
 688                 }
 689         }
 690 
 691         assert(!"tree_view_node was not found in parent nodes children");
 692         throw "assertion ignored"; //To silence 'no return value in this codepath' warning.
 693 }

 695 int tree_view_node::calculate_ypos()
 696 {
 697         if(!parent_node_) {
 698                 return 0;
 699         }
 700 
 701         int res = parent_node_->calculate_ypos();
 702         for(const auto& node : parent_node_->children_) {
 703                 if(node.get() == this) {
 704                         break;
 705                 }
 706 
 707                 res += node->get_current_size(true).y;
 708         }
 709 
 710         return res;
 711 }

 713 tree_view_node* tree_view_node::get_last_visible_parent_node()
 714 {
 715         if(!parent_node_) {
 716                 return this;
 717         }
 718 
 719         tree_view_node* res = parent_node_->get_last_visible_parent_node();
 720         return res == parent_node_ && !res->is_folded() ? this : res;

 723 tree_view_node* tree_view_node::get_node_above()
 724 {
 725         assert(!is_root_node());
 726 
 727         tree_view_node* cur = nullptr;
 728         for(size_t i = 0; i < parent_node_->count_children(); ++i) {
 729                 if(parent_node_->children_[i].get() == this) {
 730                         if(i == 0) {
 731                                 return parent_node_->is_root_node() ? nullptr : parent_node_;
 732                         } else {
 733                                 cur = parent_node_->children_[i - 1].get();
 734                                 break;
 735                         }
 736                 }
 737         }
 738 
 739         while(cur && !cur->is_folded() && cur->count_children() > 0) {
 740                 cur = &cur->get_child_at(cur->count_children() - 1);
 741         }
 742 
 743         if (!cur)
 744                 throw std::domain_error("tree_view_node::get_node_above(): Cannot determine which node is this line, or which node is the line above this one, if any.");
 745 
 746         return cur;
 747 }

 749 tree_view_node* tree_view_node::get_node_below()
 750 {
 751         assert(!is_root_node());
 752         if(!is_folded() && count_children() > 0) {
 753                 return &get_child_at(0);
 754         }
 755 
 756         tree_view_node* cur = this;
 757         while(cur->parent_node_ != nullptr) {
 758                 tree_view_node& parent = *cur->parent_node_;
 759 
 760                 for(size_t i = 0; i < parent.count_children(); ++i) {
 761                         if(parent.children_[i].get() == cur) {
 762                                 if(i < parent.count_children() - 1) {
 763                                         return parent.children_[i + 1].get();
 764                                 } else {
 765                                         cur = &parent;
 766                                 }
 767                                 break;
 768                         }
 769                 }
 770         }
 771 
 772         return nullptr;
 773 }

 793 void tree_view_node::select_node(bool expand_parents)
 794 {
 795         if(!label_ || label_->get_value_bool()) {
 796                 return;
 797         }
 798 
 799         if(expand_parents) {
 800                 tree_view_node* root = &get_tree_view().get_root_node();
 801                 for(tree_view_node* cur = parent_node_; cur != root; cur = cur->parent_node_) {
 802                         cur->unfold();
 803                 }
 804         }
 805 
 806         if(get_tree_view().selected_item_ && get_tree_view().selected_item_->label_) {
 807                 get_tree_view().selected_item_->label_->set_value(false);
 808         }
 809         get_tree_view().selected_item_ = this;
 810 
 811         if(get_tree_view().selection_change_callback_) {
 812                 get_tree_view().selection_change_callback_(get_tree_view());
 813         }
 814         label_->set_value_bool(true);
 815 }
