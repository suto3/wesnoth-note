  86 void listbox::remove_row(const unsigned row, unsigned count)
  87 {
  88         assert(generator_);
  89 
  90         if(row >= get_item_count()) {
  91                 return;
  92         }
  93 
  94         if(!count || count + row > get_item_count()) {
  95                 count = get_item_count() - row;
  96         }
  97 
  98         int height_reduced = 0;
  99         int width_reduced = 0;
 100 
 101         // TODO: Fix this for horizontal listboxes
 102         // Note the we have to use content_grid_ and cannot use "_list_grid" which is what generator_ uses.
 103         int row_pos_y = is_horizontal_ ? -1 : generator_->item(row).get_y() - content_grid_->get_y();
 104         int row_pos_x = is_horizontal_ ? -1 : 0;
 105 
 106         for(; count; --count) {
 107                 if(generator_->item(row).get_visible() != visibility::invisible) {
 108                         if(is_horizontal_) {
 109                                 width_reduced += generator_->item(row).get_width();
 110                         } else {
 111                                 height_reduced += generator_->item(row).get_height();
 112                         }
 113                 }
 114 
 115                 generator_->delete_item(row);
 116         }
 117 
 118         if((height_reduced != 0 || width_reduced != 0) && get_item_count() != 0) {
 119                 resize_content(-width_reduced, -height_reduced, row_pos_x, row_pos_y);
 120         } else {
 121                 update_content_size();
 122         }
 123 }

 143 void listbox::set_row_shown(const unsigned row, const bool shown)
 144 {
 145         assert(generator_);
 146 
 147         window* window = get_window();
 148         assert(window);
 149 
 150         const int selected_row = get_selected_row();
 151 
 152         bool resize_needed = false;
 153 
 154         // Local scope for invalidate_layout_blocker
 155         {
 156                 window::invalidate_layout_blocker invalidate_layout_blocker(*window);
 157 
 158                 generator_->set_item_shown(row, shown);
 159 
 160                 point best_size = generator_->calculate_best_size();
 161                 generator_->place(generator_->get_origin(), {std::max(best_size.x, content_visible_area().w), best_size.y});
 162 
 163                 resize_needed = !content_resize_request();
 164         }
 165 
 166         if(resize_needed) {
 167                 window->invalidate_layout();
 168         } else {
 169                 content_grid_->set_visible_rectangle(content_visible_area());
 170                 set_is_dirty(true);
 171         }
 172 
 173         if(selected_row != get_selected_row()) {
 174                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 175         }
 176 }

 346 void listbox::place(const point& origin, const point& size)
 347 {
 348         boost::optional<unsigned> vertical_scrollbar_position, horizontal_scrollbar_position;
 349 
 350         // Check if this is the first time placing the list box
 351         if(get_origin() != point {-1, -1}) {
 352                 vertical_scrollbar_position = get_vertical_scrollbar_item_position();
 353                 horizontal_scrollbar_position = get_horizontal_scrollbar_item_position();
 354         }
 355 
 356         // Inherited.
 357         scrollbar_container::place(origin, size);
 358 
 359         const int selected_item = generator_->get_selected_item();
 360         if(vertical_scrollbar_position && horizontal_scrollbar_position) {
 361                 LOG_GUI_L << LOG_HEADER << " restoring scroll position" << std::endl;
 362 
 363                 set_vertical_scrollbar_item_position(*vertical_scrollbar_position);
 364                 set_horizontal_scrollbar_item_position(*horizontal_scrollbar_position);
 365         } else if(selected_item != -1) {
 366                 LOG_GUI_L << LOG_HEADER << " making the initially selected item visible" << std::endl;
 367 
 368                 const SDL_Rect& visible = content_visible_area();
 369                 SDL_Rect rect = generator_->item(selected_item).get_rectangle();
 370 
 371                 rect.x = visible.x;
 372                 rect.w = visible.w;
 373 
 374                 show_content_rect(rect);
 375         }
 376 }

 382 void listbox::resize_content(const int width_modification,
 383                 const int height_modification,
 384                 const int width_modification_pos,
 385                 const int height_modification_pos)
 386 {
 387         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size() << " width_modification "
 388                           << width_modification << " height_modification " << height_modification << ".\n";
 389 
 390         if(content_resize_request(
 391                 width_modification, height_modification, width_modification_pos, height_modification_pos))
 392         {
 393                 // Calculate new size.
 394                 point size = content_grid()->get_size();
 395                 size.x += width_modification;
 396                 size.y += height_modification;
 397 
 398                 // Set new size.
 399                 content_grid()->set_size(size);
 400 
 401                 // Set status.
 402                 need_layout_ = true;
 403 
 404                 // If the content grows assume it "overwrites" the old content.
 405                 if(width_modification < 0 || height_modification < 0) {
 406                         set_is_dirty(true);
 407                 }
 408 
 409                 DBG_GUI_L << LOG_HEADER << " succeeded.\n";
 410         } else {
 411                 DBG_GUI_L << LOG_HEADER << " failed.\n";
 412         }
 413 }

 415 void listbox::resize_content(const widget& row)
 416 {
 417         if(row.get_visible() == visibility::invisible) {
 418                 return;
 419         }
 420 
 421         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size() << " row size " << row.get_best_size()
 422                           << ".\n";
 423 
 424         const point content = content_grid()->get_size();
 425         point size = row.get_best_size();
 426 
 427         if(size.x < content.x) {
 428                 size.x = 0;
 429         } else {
 430                 size.x -= content.x;
 431         }
 432 
 433         resize_content(size.x, size.y);
 434 }

 451 void listbox::update_visible_area_on_key_event(const KEY_SCROLL_DIRECTION direction)
 452 {
 453         const SDL_Rect& visible = content_visible_area();
 454         SDL_Rect rect = generator_->item(generator_->get_selected_item()).get_rectangle();
 455 
 456         // When scrolling make sure the new items are visible...
 457         if(direction == KEY_VERTICAL) {
 458                 // ...but leave the horizontal scrollbar position.
 459                 rect.x = visible.x;
 460                 rect.w = visible.w;
 461         } else {
 462                 // ...but leave the vertical scrollbar position.
 463                 rect.y = visible.y;
 464                 rect.h = visible.h;
 465         }
 466 
 467         show_content_rect(rect);
 468 
 469         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 470 }

 528 void listbox::finalize(builder_grid_const_ptr header,
 529                 builder_grid_const_ptr footer,
 530                 const std::vector<std::map<std::string, string_map>>& list_data)
 531 {
 532         // "Inherited."
 533         scrollbar_container::finalize_setup();
 534 
 535         assert(generator_);
 536 
 537         if(header) {
 538                 swap_grid(&get_grid(), content_grid(), header->build(), "_header_grid");
 539         }
 540 
 541         grid& p = find_widget<grid>(this, "_header_grid", false);
 542 
 543         for(unsigned i = 0, max = std::max(p.get_cols(), p.get_rows()); i < max; ++i) {
 544                 //
 545                 // TODO: I had to change this to case to a toggle_button in order to use a signal handler.
 546                 // Should probably look into a way to make it more general like it was before (used to be
 547                 // cast to selectable_item).
 548                 //
 549                 // - vultraz, 2017-08-23
 550                 //
 551                 if(toggle_button* selectable = find_widget<toggle_button>(&p, "sort_" + std::to_string(i), false, false)) {
 552                         // Register callback to sort the list.
 553                         connect_signal_notify_modified(*selectable, std::bind(&listbox::order_by_column, this, i, _1));
 554 
 555                         if(orders_.size() < max) {
 556                                 orders_.resize(max);
 557                         }
 558 
 559                         orders_[i].first = selectable;
 560                 }
 561         }
 562 
 563         if(footer) {
 564                 swap_grid(&get_grid(), content_grid(), footer->build(), "_footer_grid");
 565         }
 566 
 567         generator_->create_items(-1, list_builder_, list_data, std::bind(&listbox::list_item_clicked, this, _1));
 568         swap_grid(nullptr, content_grid(), generator_, "_list_grid");
 569 }

 571 void listbox::order_by_column(unsigned column, widget& widget)
 572 {
 573         selectable_item& selectable = dynamic_cast<selectable_item&>(widget);
 574         if(column >= orders_.size()) {
 575                 return;
 576         }
 577 
 578         for(auto& pair : orders_) {
 579                 if(pair.first != nullptr && pair.first != &selectable) {
 580                         pair.first->set_value(SORT_NONE);
 581                 }
 582         }
 583 
 584         if(selectable.get_value() > orders_[column].second.size()) {
 585                 return;
 586         }
 587 
 588         if(selectable.get_value() == SORT_NONE) {
 589                 order_by(std::less<unsigned>());
 590         } else {
 591                 order_by(orders_[column].second[selectable.get_value() - 1]);
 592         }
 593 }

 612 void listbox::set_active_sorting_option(const order_pair& sort_by, const bool select_first)
 613 {
 614         // TODO: should this be moved to a public header_grid() getter function?
 615         grid& header_grid = find_widget<grid>(this, "_header_grid", false);
 616 
 617         selectable_item& w = find_widget<selectable_item>(&header_grid, "sort_" + std::to_string(sort_by.first), false);
 618 
 619         // Set the sorting toggle widgets' value (in this case, its state) to the given sorting
 620         // order. This is necessary since the widget's value is used to determine the order in
 621         // @ref order_by_column in lieu of a direction being passed directly.
 622         w.set_value(static_cast<int>(sort_by.second));
 623 
 624         order_by_column(sort_by.first, dynamic_cast<widget&>(w));
 625 
 626         if(select_first && generator_->get_item_count() > 0) {
 627                 select_row_at(0);
 628         }
 629 }

 882 static std::vector<std::map<std::string, string_map>> parse_list_data(const config& data, const unsigned int req_cols)
 883 {
 884         std::vector<std::map<std::string, string_map>> list_data;
 885 
 886         for(const auto& row : data.child_range("row")) {
 887                 auto cols = row.child_range("column");
 888 
 889                 VALIDATE(static_cast<unsigned>(cols.size()) == req_cols,
 890                         _("'list_data' must have the same number of columns as the 'list_definition'.")
 891                 );
 892 
 893                 for(const auto& c : cols) {
 894                         list_data.emplace_back();
 895 
 896                         for(const auto& i : c.attribute_range()) {
 897                                 list_data.back()[""][i.first] = i.second;
 898                         }
 899 
 900                         for(const auto& w : c.child_range("widget")) {
 901                                 VALIDATE(w.has_attribute("id"), missing_mandatory_wml_key("[list_data][row][column][widget]", "id"));
 902 
 903                                 for(const auto& i : w.attribute_range()) {
 904                                         list_data.back()[w["id"]][i.first] = i.second;
 905                                 }
 906                         }
 907                 }
 908         }
 909 
 910         return list_data;
 911 }

 929                 footer = std::make_shared<builder_grid>(f);

 940 
 941         if(cfg.has_child("list_data")) {
