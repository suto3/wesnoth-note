 172 def htmlescape(text, quote=True):
 173     """Escape any HTML special characters in the given string."""
 174     if text is None:
 175         return text
 176     return html.escape(text, quote)
 177 
 178 def urlencode(text):
 179     """
 180     Encode the given string to ensure it only contains valid URL characters
 181     (also known as percent-encoding).
 182     """
 183     if text is None:
 184         return text
 185     return urllib.parse.quote(text, encoding='utf-8')

 187 def output(path, url, datadir, data):
 188     """Write the HTML index of add-ons into the specified directory."""
 189     try:
 190         os.mkdir(path)
 191     except OSError:
 192         pass
 193 
 194     outfile = open(path + "/index.html", "w")
 195 
 196     def w(line):
 197         outfile.write(line + "\n")
 198 
 199     am_dir = os.path.dirname(__file__) + "/"
 200     root_dir = datadir + "/" if datadir is not None else am_dir + "../../../"
 201     images_to_tc = []
 202 
 203     # Copy required HTML assets into the destination dir.
 204     for filename in HTML_RESOURCES:
 205         Popen(["cp", "-u", am_dir + filename, path])
 206 
 207     server_name = os.path.basename(path)
 208     if server_name == "1.9":
 209         # 1.9 became the 1.10 add-ons server. Reflect that here.
 210         server_name = "1.10"
 211     elif server_name == "trunk":
 212         server_name = "Testing (Trunk)"
 213 
 214     w(WESMERE_HEADER % {
 215         "css_version": WESMERE_CSS_VERSION,
 216         "css_prefix": WESMERE_CSS_PREFIX,
 217         "server_name": server_name,
 218     })
 219     if url:
 220         w(WESMERE_DOWNLOAD_HELP)
 221 
 222     w('<table class="tablesorter" id="campaigns">\n<thead>\n<tr>')
 223     table_headers = [
 224         ("type", "Type"),
 225         ("icon", "Icon"),
 226         ("name", "Addon"),
 227         ("size", "Size"),
 228         ("stats", "Traffic"),
 229         ("date", "Date"),
 230         ("locales", "Translations")
 231     ]
 232     for header_class, header_label in table_headers:
 233         w('<th class="addon-%s">%s&nbsp;&nbsp;&nbsp;</th>' % (header_class, header_label))
 234     w('</tr>\n</thead>\n<tbody>')
 235 
 236     addons = data.get_all(tag="campaigns")[0]
 237     for addon in addons.get_all(tag="campaign"):
 238         v = addon.get_text_val
 239 
 240         addon_id = v("name") # Escaped as part of a path composition later on.
 241         title = htmlescape(v("title", "unknown"))
 242         size = float(v("size", "0")) # bytes
 243         display_size = size / (1024 * 1024) # MiB
 244         addon_type = htmlescape(v("type", "none"))
 245         version = htmlescape(v("version", "unknown"))
 246         author = htmlescape(v("author", "unknown"))
 247         feedback_url = v("feedback_url", None) # Escaped by a function call.
 248 
 249         icon = htmlescape(v("icon", ""))
 250         description = htmlescape(v('description', '(no description)'))
 251         imgurl = ""
 252 
 253         downloads = int(v("downloads", "0"))
 254         uploads = int(v("uploads", "0"))
 255         timestamp = int(v("timestamp", "0"))
 256         display_ts = time.strftime("%b %d %Y", time.localtime(timestamp))
 257 
 258         translations = addon.get_all(tag="translation")
 259         languages = [x.get_text_val("language") for x in translations]
 260 
 261         if icon:
 262             icon = icon.strip()
 263             tilde = icon.find("~")
 264             if tilde >= 0:
 265                 icon = icon[:tilde]
 266             if "\\" in icon:
 267                 icon = icon.replace("\\", "/")
 268             try:
 269                 os.mkdir(path + "/icons")
 270             except OSError:
 271                 pass
 272             if "." not in icon:
 273                 icon += ".png"
 274             src = root_dir + icon
 275             imgurl = "icons/" + os.path.basename(icon)
 276             if not os.path.exists(src):
 277                 src = root_dir + "data/core/images/" + icon
 278             if not os.path.exists(src):
 279                 src = root_dir + "images/" + icon
 280             if not os.path.exists(src):
 281                 src = glob.glob(root_dir + "data/campaigns/*/images/" + icon)
 282                 if src:
 283                     src = src[0]
 284                 if not src or not os.path.exists(src):
 285                     sys.stderr.write("Cannot find icon " + icon + "\n")
 286                     src = root_dir + "images/misc/missing-image.png"
 287                     imgurl = "icons/missing-image.png"
 288             images_to_tc.append((src, path + "/" + imgurl))
 289 
 290         w('<tr>')
 291 
 292         w('<td class="addon-type">')
 293         if addon_type in ADDON_TYPES_INFO:
 294             w('%(short)s<div class="type-tooltip"><b>%(long)s</b><br/>%(help)s</div>' \
 295                 % ADDON_TYPES_INFO[addon_type])
 296         else:
 297             w(addon_type)
 298         w('</td>')
 299 
 300         w(('<td class="addon-icon"><img alt="" src="%s"/>'
 301            '<div class="desc-tooltip"><b>%s</b><pre>%s</pre></div></td>') % (
 302                imgurl, title, description))
 303 
 304         def make_icon_button(url, label, icon):
 305             w(('<a href="{0}" title="{1}">'
 306                '<i class="fa fa-fw fa-2x fa-{2}" aria-hidden="true"></i>'
 307                '<span class="sr-only">{1}</span></a>').format(
 308                     htmlescape(url), htmlescape(label), icon))
 309 
 310         w('<td class="addon"><span hidden>%s</span>' % title)
 311         if url or feedback_url:
 312             w('<span class="addon-download">')
 313             if feedback_url:
 314                 make_icon_button(feedback_url, "Forum topic", "comment")
 315             if url:
 316                 link = url.rstrip("/") + "/" + urlencode(addon_id) + ".tar.bz2"
 317                 make_icon_button(link, "Download", "download")
 318             w('</span>')
 319         w(('<b>%s</b><br/>'
 320            '<span class="addon-meta"><span class="addon-meta-label">Version:</span> %s<br/>'
 321            '<span class="addon-meta-label">Author:</span> %s</span></td>') % (
 322                title, version, author))
 323 
 324         w("<td><span hidden>%d</span><b>%.2f</b>&nbsp;MiB</td>" % (size, display_size))
 325 
 326         w("<td><b>%d</b> down<br/>%s up</td>" % (downloads, uploads))
 327 
 328         w('<td><span hidden>%d</span>%s</td>' % (timestamp, display_ts))
 329 
 330         w("<td>%s</td>" % htmlescape(", ".join(languages), quote=False))
 331 
 332         w("</tr>")
 333 
 334     w('</tbody>\n</table>')
 335     w(WESMERE_FOOTER)
 336 
 337     sys.stderr.write("Done outputting html, now generating %d TC'ed images\n" % len(images_to_tc))
 338     for pair in images_to_tc:
 339         # wait() to ensure only one process exists at any time
 340         Popen([os.path.join(am_dir, "../unit_tree/TeamColorizer"), pair[0], pair[1]]).wait()
