  85 bool mouse_handler_base::mouse_motion_default(int x, int y, bool /*update*/)
  86 {
  87         tooltips::process(x, y);
  88 
  89         if(simple_warp_) {
  90                 return true;
  91         }
  92 
  93         if(minimap_scrolling_) {
  94                 // if the game is run in a window, we could miss a LMB/MMB up event
  95                 // if it occurs outside our window.
  96                 // thus, we need to check if the LMB/MMB is still down
  97                 minimap_scrolling_ = ((SDL_GetMouseState(nullptr, nullptr) & (SDL_BUTTON(1) | SDL_BUTTON(2))) != 0);
  98 
  99                 if(minimap_scrolling_) {
 100                         const map_location& loc = gui().minimap_location_on(x, y);
 101                         if(loc.valid()) {
 102                                 if(loc != last_hex_) {
 103                                         last_hex_ = loc;
 104                                         gui().scroll_to_tile(loc, display::WARP, false);
 105                                 }
 106                         } else {
 107                                 // clicking outside of the minimap will end minimap scrolling
 108                                 minimap_scrolling_ = false;
 109                         }
 110                 }
 111 
 112                 if(minimap_scrolling_) {
 113                         return true;
 114                 }
 115         }
 116 
 117         // Fire the drag & drop only after minimal drag distance
 118         // While we check the mouse buttons state, we also grab fresh position data.
 119         int mx = drag_from_x_; // some default value to prevent unlikely SDL bug
 120         int my = drag_from_y_;
 121 
 122         if(is_dragging() && !dragging_started_) {
 123                 if((dragging_left_  && (SDL_GetMouseState(&mx, &my) & SDL_BUTTON_LEFT)  != 0) ||
 124                    (dragging_right_ && (SDL_GetMouseState(&mx, &my) & SDL_BUTTON_RIGHT) != 0))
 125                 {
 126                         const double drag_distance =
 127                                 std::pow(static_cast<double>(drag_from_x_ - mx), 2) +
 128                                 std::pow(static_cast<double>(drag_from_y_ - my), 2);
 129 
 130                         if(drag_distance > drag_threshold() * drag_threshold()) {
 131                                 dragging_started_ = true;
 132                                 cursor::set_dragging(true);
 133                         }
 134                 }
 135         }
 136 
 137         return false;
 138 }

 140 void mouse_handler_base::mouse_press(const SDL_MouseButtonEvent& event, const bool browse)
 141 {
 142         if(is_middle_click(event) && !preferences::middle_click_scrolls()) {
 143                 simple_warp_ = true;
 144         }
 145 
 146         show_menu_ = false;
 147         map_location loc = gui().hex_clicked_on(event.x, event.y);
 148         mouse_update(browse, loc);
 149 
 150         if(is_left_click(event)) {
 151                 if(event.state == SDL_PRESSED) {
 152                         cancel_dragging();
 153                         init_dragging(dragging_left_);
 154                         left_click(event.x, event.y, browse);
 155                 } else if(event.state == SDL_RELEASED) {
 156                         minimap_scrolling_ = false;
 157                         clear_dragging(event, browse);
 158                         left_mouse_up(event.x, event.y, browse);
 159                 }
 160         } else if(is_right_click(event)) {
 161                 if(event.state == SDL_PRESSED) {
 162                         cancel_dragging();
 163                         init_dragging(dragging_right_);
 164                         right_click(event.x, event.y, browse);
 165                 } else if(event.state == SDL_RELEASED) {
 166                         minimap_scrolling_ = false;
 167                         clear_dragging(event, browse);
 168                         right_mouse_up(event.x, event.y, browse);
 169                 }
 170         } else if(is_middle_click(event)) {
 171                 if(event.state == SDL_PRESSED) {
 172                         set_scroll_start(event.x, event.y);
 173                         scroll_started_ = true;
 174 
 175                         map_location minimap_loc = gui().minimap_location_on(event.x, event.y);
 176                         minimap_scrolling_ = false;
 177                         if(minimap_loc.valid()) {
 178                                 simple_warp_ = false;
 179                                 minimap_scrolling_ = true;
 180                                 last_hex_ = minimap_loc;
 181                                 gui().scroll_to_tile(minimap_loc, display::WARP, false);
 182                         } else if(simple_warp_) {
 183                                 // middle click not on minimap, check gamemap instead
 184                                 if(loc.valid()) {
 185                                         last_hex_ = loc;
 186                                         gui().scroll_to_tile(loc, display::WARP, false);
 187                                 }
 188                         }
 189                 } else if(event.state == SDL_RELEASED) {
 190                         minimap_scrolling_ = false;
 191                         simple_warp_ = false;
 192                         scroll_started_ = false;
 193                 }
 194         }
 195 
 196         if(!dragging_left_ && !dragging_right_ && dragging_started_) {
 197                 dragging_started_ = false;
 198                 cursor::set_dragging(false);
 199         }
 200 
 201         mouse_update(browse, loc);
 202 }

 219 bool mouse_handler_base::left_click(int x, int y, const bool /*browse*/)
 220 {
 221         if(gui().view_locked()) {
 222                 return false;
 223         }
 224 
 225         // clicked on a hex on the minimap? then initiate minimap scrolling
 226         const map_location& loc = gui().minimap_location_on(x, y);
 227         minimap_scrolling_ = false;
 228         if(loc.valid()) {
 229                 minimap_scrolling_ = true;
 230                 last_hex_ = loc;
 231                 gui().scroll_to_tile(loc, display::WARP, false);
 232                 return true;
 233         }
 234 
 235         return false;
 236 }

 243 void mouse_handler_base::mouse_wheel(int scrollx, int scrolly, bool browse)
 244 {
 245         int x, y;
 246         SDL_GetMouseState(&x, &y);
 247 
 248         int movex = scrollx * preferences::scroll_speed();
 249         int movey = scrolly * preferences::scroll_speed();
 250 
 251         // Don't scroll map if cursor is not in gamemap area
 252         if(!sdl::point_in_rect(x, y, gui().map_area())) {
 253                 return;
 254         }
 255 
 256         if(movex != 0 || movey != 0) {
 257                 CKey pressed;
 258                 // Alt + mousewheel do an 90° rotation on the scroll direction
 259                 if(pressed[SDLK_LALT] || pressed[SDLK_RALT]) {
 260                         gui().scroll(-movey, -movex);
 261                 } else {
 262                         gui().scroll(-movex, -movey);
 263                 }
 264         }
 265 
 266         if(scrollx < 0) {
 267                 mouse_wheel_left(x, y, browse);
 268         } else if(scrollx > 0) {
 269                 mouse_wheel_right(x, y, browse);
 270         }
 271 
 272         if(scrolly < 0) {
 273                 mouse_wheel_down(x, y, browse);
 274         } else if(scrolly > 0) {
 275                 mouse_wheel_up(x, y, browse);
 276         }
 277 }

 279 void mouse_handler_base::right_mouse_up(int x, int y, const bool browse)
 280 {
 281         if(!right_click_show_menu(x, y, browse)) {
 282                 return;
 283         }
 284 
 285         gui().draw(); // redraw highlight (and maybe some more)
 286 
 287         const theme::menu* const m = gui().get_theme().context_menu();
 288         if(m != nullptr) {
 289                 show_menu_ = true;
 290         } else {
 291                 WRN_DP << "no context menu found..." << std::endl;
 292         }
 293 }

 310 void mouse_handler_base::clear_dragging(const SDL_MouseButtonEvent& event, bool browse)
 311 {
 312         // we reset dragging info before calling functions
 313         // because they may take time to return, and we
 314         // could have started other drag&drop before that
 315         cursor::set_dragging(false);
 316 
 317         if(dragging_started_) {
 318                 dragging_started_ = false;
 319                 if(dragging_left_) {
 320                         dragging_left_ = false;
 321                         left_drag_end(event.x, event.y, browse);
 322                 }
 323 
 324                 if(dragging_right_) {
 325                         dragging_right_ = false;
 326                         right_drag_end(event.x, event.y, browse);
 327                 }
 328         } else {
 329                 dragging_left_ = false;
 330                 dragging_right_ = false;
 331         }
 332 }
