   2 ### The list of answers will appear in console when you run wesnoth, after debug: messages.
   3 ### Copy the lines to a text file ("foo") and use the following sed script to clean them,
   4 ###
   5 ### sed -n 's/^.*: \([ [:digit:]]*\) ,/\1/p' <foo | tr '\n' ','
   6 ###
   7 ### so you get a nice comma separated list. Then paste it in data/test/macros/pathfind_answers.cfg,
   8 ### and in this file change GENERATE_ANSWERS to 0 again.

  11     0

  14 #define IF_GEN_ANS T E
  15     [if]
  16         {VARIABLE_CONDITIONAL GEN_ANS equals 1}
  17         [then]
  18             {T}
  19         [/then]
  20         [else]
  21             {E}
  22         [/else]
  23     [/if]
  24 #enddef

  26 #define SPLIT_COMMA_LIST LIST VAR
  27     [set_variables]
  28         name={VAR}
  29         [split]
  30             list={LIST}
  31             key=val
  32             separator=","
  33             remove_empty=no
  34         [/split]
  35     [/set_variables]
  36 #enddef

  38 #define CHECK_ANSWER X
  39     {IF_GEN_ANS (
  40         {VARIABLE observed_answer ${X}}
  41         [wml_message]
  42             message="$observed_answer ,"
  43             logger=debug
  44             to_chat=false
  45         [/wml_message]
  46     ) (
  47         {VARIABLE current_answer $answers[$answer_counter|].val|}
  48         {VARIABLE observed_answer ${X}}
  49         [wml_message]
  50             message="$current_answer , $observed_answer"
  51             logger=debug
  52             to_chat=false
  53         [/wml_message]
  54         {ASSERT ({VARIABLE_CONDITIONAL observed_answer equals $current_answer}) }
  55     )}
  56     {VARIABLE_OP answer_counter add 1}
  57 #enddef
  58 
  59 #define INIT_ANSWERS LIST
  60     {VARIABLE answer_counter 0}
  61     {VARIABLE answers_list ("{{LIST}}")}
  62     {SPLIT_COMMA_LIST $answers_list answers}
  63 
  64     [wml_message]
  65         message="Parsed: $answers.length| answers"
  66         logger=debug
  67         to_chat=false
  68     [/wml_message]
  69 #enddef

  82     {VARIABLE nhexes $path.hexes}
