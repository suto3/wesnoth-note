  26 local function ensure_config(cfg)
  27         if type(cfg) == 'table' then
  28                 return true
  29         end
  30         if type(cfg) == 'userdata' then
  31                 if getmetatable(cfg) == 'wml object' then return true end
  32                 error("Expected a table or wml object but got " .. getmetatable(cfg), 3)
  33         else
  34                 error("Expected a table or wml object but got " .. type(cfg), 3)
  35         end
  36         return false
  37 end

  39 --! Returns the first subtag of @a cfg with the given @a name.
  40 --! If @a id is not nil, the "id" attribute of the subtag has to match too.
  41 --! The function also returns the index of the subtag in the array.
  42 function wml.get_child(cfg, name, id)
  43         ensure_config(cfg)
  44         for i,v in ipairs(cfg) do
  45                 if v[1] == name then
  46                         local w = v[2]
  47                         if not id or w.id == id then return w, i end
  48                 end
  49         end
  50 end

  77 --! Returns an iterator over all the subtags of @a cfg with the given @a name.
  78 function wml.child_range(cfg, tag)
  79         ensure_config(cfg)
  80         local iter, state, i = ipairs(cfg)
  81         local function f(s)
  82                 local c
  83                 repeat
  84                         i,c = iter(s,i)
  85                         if not c then return end
  86                 until c[1] == tag
  87                 return c[2]
  88         end
  89         return f, state
  90 end

 151 -- Note: When using version (for level 2 or 3 deprecation), specify the first version
 152 -- in which the feature could be removed... NOT the version at which it was deprecated.
 153 function wesnoth.deprecation_message(elem_name, level, version, detail)
 154         local message_params = {elem = elem_name}
 155         local logger
 156         local message
 157         if level == 1 then
 158                 logger = function(msg) wesnoth.log("info", msg) end
 159                 message = wesnoth.format(_"$elem has been deprecated indefinitely.", message_params)
 160         elseif level == 2 then
 161                 logger = function(msg) wesnoth.log("warn", msg) end
 162                 if wesnoth.compare_versions(game_config.version, "<", version) then
 163                         message_params.version = version
 164                         message = wesnoth.format(_"$elem has been deprecated and may be removed in version $version.", message_params)
 165                 else
 166                         message = wesnoth.format(_"$elem has been deprecated and may be removed at any time.", message_params)
 167                 end
 168         elseif level == 3 then
 169                 logger = function(msg) wesnoth.log("err", msg) end
 170                 message = wesnoth.format(_"$elem has been deprecated and will be removed in the next version.", message_params)
 171         elseif level == 4 then
 172                 logger = error
 173                 message = wesnoth.format(_"$elem has been deprecated and removed.", message_params)
 174         else
 175                 error(_"Invalid deprecation level (should be 1-4)")
 176         end
 177         if #detail > 0 then
 178                 logger(message .. "\n  " .. detail)
 179         else
 180                 logger(message)
 181         end
 182 end

 184 -- Usage: module.something = wesnoth.deprecate(module.something, "something", ...)
 185 function wesnoth.deprecate_api(elem_name, replacement, level, version, elem, detail_msg)
 186         local message = detail_msg or ''
 187         if replacement then
 188                 message = message .. " " .. wesnoth.format(_"(Note: You should use $replacement instead in new code)", {replacement = replacement})
 189         end
 190         if type(level) ~= "number" or level < 1 or level > 4 then
 191                 error("Invalid deprecation level! Must be 1-4.")
 192         end
 193         local msg_shown = false
 194         if type(elem) == "function" then
 195                 return function(...)
 196                         if not msg_shown then
 197                                 msg_shown = true
 198                                 wesnoth.deprecation_message(elem_name, level, version, message)
 199                         end
 200                         return elem(...)
 201                 end
 202         elseif type(elem) == "table" or type(elem) == "userdata" then
 203                 local mt = {
 204                         __index = function(key)
 205                                 if not msg_shown then
 206                                         msg_shown = true
 207                                         wesnoth.deprecation_message(elem_name, level, version, message)
 208                                 end
 209                                 return elem[key]
 210                         end,
 211                         __newindex = function(key, val)
 212                                 if not msg_shown then
 213                                         msg_shown = true
 214                                         wesnoth.deprecation_message(elem_name, level, version, message)
 215                                 end
 216                                 elem[key] = val
 217                         end,
 218                 }
 219                 return setmetatable({}, mt)
 220         else
 221                 wesnoth.log('warn', "Attempted to deprecate something that is not a table or function: " ..
 222                         elem_name .. " -> " .. replacement .. ", where " .. elem_name .. " = " .. tostring(elem))
 223         end
 224         return elem
 225 end

 294         local function resolve_variable_context(ctx, err_hint)
 295                 if ctx == nil then
 296                         return {get = wesnoth.get_variable, set = wesnoth.set_variable}
 297                 elseif type(ctx) == 'number' and ctx > 0 and ctx <= #wesnoth.sides then
 298                         return resolve_variable_context(wesnoth.sides[ctx])
 299                 elseif type(ctx) == 'string' then
 300                         -- TODO: Treat it as a namespace for a global (persistent) variable
 301                         -- (Need Lua API for accessing them first, though.)
 302                 elseif getmetatable(ctx) == "unit" then
 303                         return {
 304                                 get = function(path) return ctx.variables[path] end,
 305                                 set = function(path, val) ctx.variables[path] = val end,
 306                         }
 307                 elseif getmetatable(ctx) == "side" then
 308                         return {
 309                                 get = function(path) return wesnoth.get_side_variable(ctx.side, path) end,
 310                                 set = function(path, val) wesnoth.set_side_variable(ctx.side, path, val) end,
 311                         }
 312                 elseif getmetatable(ctx) == "unit variables" or getmetatable(ctx) == "side variables" then
 313                         return {
 314                                 get = function(path) return ctx[path] end,
 315                                 set = function(path, val) ctx[path] = val end,
 316                         }
 317                 end
 318                 error(string.format("Invalid context for %s: expected nil, side, or unit", err_hint), 3)
 319         end

 321         --! Fetches all the WML container variables with name @a var.

 323         function wml.variable.get_array(var, context)
 324                 context = resolve_variable_context(context, "get_variable_array")
 325                 local result = {}
 326                 for i = 1, context.get(var .. ".length") do
 327                         result[i] = context.get(string.format("%s[%d]", var, i - 1))

 332         --! Puts all the elements of table @a t inside a WML container with name @a var.

 335                 context.set(var)
