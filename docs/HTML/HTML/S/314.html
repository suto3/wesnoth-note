 275                 if (ec) {

 288 static bool create_directory_if_missing(const path &dirpath)
 289 {
 290         error_code ec;
 291         bfs::file_status fs = bfs::status(dirpath, ec);
 292         if (error_except_not_found(ec)) {
 293                 ERR_FS << "Failed to retrieve file status for " << dirpath.string() << ": " << ec.message() << '\n';
 294                 return false;
 295         } else if (bfs::is_directory(fs)) {
 296                 DBG_FS << "directory " << dirpath.string() << " exists, not creating\n";
 297                 return true;
 298         } else if (bfs::exists(fs)) {
 299                 ERR_FS << "cannot create directory " << dirpath.string() << "; file exists\n";
 300                 return false;
 301         }
 302 
 303         bool created = bfs::create_directory(dirpath, ec);
 304         if (ec) {
 305                 ERR_FS << "Failed to create directory " << dirpath.string() << ": " << ec.message() << '\n';
 306         }
 307         return created;
 308 }
 309 static bool create_directory_if_missing_recursive(const path& dirpath)
 310 {
 311         DBG_FS << "creating recursive directory: " << dirpath.string() << '\n';
 312 
 313         if (dirpath.empty())
 314                 return false;
 315         error_code ec;
 316         bfs::file_status fs = bfs::status(dirpath);
 317         if (error_except_not_found(ec)) {
 318                 ERR_FS << "Failed to retrieve file status for " << dirpath.string() << ": " << ec.message() << '\n';
 319                 return false;
 320         } else if (bfs::is_directory(fs)) {
 321                 return true;
 322         } else if (bfs::exists(fs)) {
 323                 return false;
 324         }
 325 
 326         if (!dirpath.has_parent_path() || create_directory_if_missing_recursive(dirpath.parent_path())) {
 327                 return create_directory_if_missing(dirpath);
 328         } else {
 329                 ERR_FS << "Could not create parents to " << dirpath.string() << '\n';
 330                 return false;
 331         }

 334 void get_files_in_dir(const std::string &dir,
 335                       std::vector<std::string>* files,
 336                       std::vector<std::string>* dirs,
 337                       file_name_option mode,
 338                       file_filter_option filter,
 339                       file_reorder_option reorder,
 340                       file_tree_checksum* checksum) {
 341         if(path(dir).is_relative() && !game_config::path.empty()) {
 342                 path absolute_dir(game_config::path);
 343                 absolute_dir /= dir;
 344                 if(is_directory_internal(absolute_dir)) {
 345                         get_files_in_dir(absolute_dir.string(), files, dirs, mode, filter, reorder, checksum);
 346                         return;
 347                 }
 348         }
 349 
 350         const path dirpath(dir);
 351 
 352         if (reorder == DO_REORDER) {
 353                 LOG_FS << "searching for _main.cfg in directory " << dir << '\n';
 354                 const path maincfg = dirpath / maincfg_filename;
 355 
 356                 if (file_exists(maincfg)) {
 357                         LOG_FS << "_main.cfg found : " << maincfg << '\n';
 358                         push_if_exists(files, maincfg, mode == ENTIRE_FILE_PATH);
 359                         return;
 360                 }
 361         }
 362 
 363         error_code ec;
 364         bfs::directory_iterator di(dirpath, ec);
 365         bfs::directory_iterator end;
 366         if (ec) {
 367                 // Probably not a directory, let the caller deal with it.
 368                 return;
 369         }
 370         for(; di != end; ++di) {
 371                 bfs::file_status st = di->status(ec);
 372                 if (ec) {
 373                         LOG_FS << "Failed to get file status of " << di->path().string() << ": " << ec.message() << '\n';
 374                         continue;
 375                 }
 376                 if (st.type() == bfs::regular_file) {
 377                         {
 378                                 std::string basename = di->path().filename().string();
 379                                 if (filter == SKIP_PBL_FILES && looks_like_pbl(basename))
 380                                         continue;
 381                                 if(!basename.empty() && basename[0] == '.' )
 382                                         continue;
 383                         }
 384                         push_if_exists(files, di->path(), mode == ENTIRE_FILE_PATH);
 385 
 386                         if (checksum != nullptr) {
 387                                 std::time_t mtime = bfs::last_write_time(di->path(), ec);
 388                                 if (ec) {
 389                                         LOG_FS << "Failed to read modification time of " << di->path().string() << ": " << ec.message() << '\n';
 390                                 } else if (mtime > checksum->modified) {
 391                                         checksum->modified = mtime;
 392                                 }
 393 
 394                                 uintmax_t size = bfs::file_size(di->path(), ec);
 395                                 if (ec) {
 396                                         LOG_FS << "Failed to read filesize of " << di->path().string() << ": " << ec.message() << '\n';
 397                                 } else {
 398                                         checksum->sum_size += size;
 399                                 }
 400                                 checksum->nfiles++;
 401                         }
 402                 } else if (st.type() == bfs::directory_file) {
 403                         std::string basename = di->path().filename().string();
 404 
 405                         if(!basename.empty() && basename[0] == '.' )
 406                                 continue;
 407                         if (filter == SKIP_MEDIA_DIR
 408                                 && (basename == "images" || basename == "sounds"))
 409                                 continue;
 410 
 411                         const path inner_main(di->path() / maincfg_filename);
 412                         bfs::file_status main_st = bfs::status(inner_main, ec);
 413                         if (error_except_not_found(ec)) {
 414                                 LOG_FS << "Failed to get file status of " << inner_main.string() << ": " << ec.message() << '\n';
 415                         } else if (reorder == DO_REORDER && main_st.type() == bfs::regular_file) {
 416                                 LOG_FS << "_main.cfg found : " << (mode == ENTIRE_FILE_PATH ? inner_main.string() : inner_main.filename().string()) << '\n';
 417                                 push_if_exists(files, inner_main, mode == ENTIRE_FILE_PATH);
 418                         } else {
 419                                 push_if_exists(dirs, di->path(), mode == ENTIRE_FILE_PATH);
 420                         }
 421                 }
 422         }
 423 
 424         if (files != nullptr)
 425                 std::sort(files->begin(),files->end());
 426 
 427         if (dirs != nullptr)
 428                 std::sort(dirs->begin(),dirs->end());
 429 
 430         if (files != nullptr && reorder == DO_REORDER) {
 431                 // move finalcfg_filename, if present, to the end of the vector
 432                 for (unsigned int i = 0; i < files->size(); i++) {
 433                         if (ends_with((*files)[i], "/" + finalcfg_filename)) {
 434                                 files->push_back((*files)[i]);
 435                                 files->erase(files->begin()+i);
 436                                 break;
 437                         }
 438                 }
 439                 // move initialcfg_filename, if present, to the beginning of the vector
 440                 int foundit = -1;
 441                 for (unsigned int i = 0; i < files->size(); i++)
 442                         if (ends_with((*files)[i], "/" + initialcfg_filename)) {
 443                                 foundit = i;
 444                                 break;
 445                         }
 446                 if (foundit > 0) {
 447                         std::string initialcfg = (*files)[foundit];
 448                         for (unsigned int i = foundit; i > 0; i--)
 449                                 (*files)[i] = (*files)[i-1];
 450                         (*files)[0] = initialcfg;
 451                 }
 452         }
 453 }

 460 std::string get_next_filename(const std::string& name, const std::string& extension)
 461 {
 462         std::string next_filename;
 463         int counter = 0;
 464 
 465         do {
 466                 std::stringstream filename;
 467 
 468                 filename << name;
 469                 filename.width(3);
 470                 filename.fill('0');
 471                 filename.setf(std::ios_base::right);
 472                 filename << counter << extension;
 473                 counter++;
 474                 next_filename = filename.str();
 475         } while(file_exists(next_filename) && counter < 1000);
 476         return next_filename;
 477 }

 537 {
 538 #ifdef PREFERENCES_DIR
 539         if (newprefdir.empty()) newprefdir = PREFERENCES_DIR;
 540 #endif
 541 
 542 #ifdef _WIN32

 544                 //allow absolute path override
 545                 user_data_dir = newprefdir;

 547                 // Custom directory relative to workdir (for portable installs, etc.)
 548                 user_data_dir = get_cwd() + "/" + newprefdir;
 549         } else {

 551                         newprefdir = "Wesnoth" + get_version_path_suffix();

 558                                                                            SHGFP_TYPE_CURRENT,

 568                         user_data_dir = path(get_cwd()) / newprefdir;
 569                 } else {

 571                         create_directory_if_missing(games_path);

 574                 }
 575         }
 576 
 577 #else /*_WIN32*/
 578

 580 
 581 #ifdef _X11

 589                                 goto other;
 590                         }
 591                         user_data_dir = home_str;
 592                         user_data_dir /= ".local/share";
 593                 } else user_data_dir = xdg_data;
 594                 user_data_dir /= "wesnoth";
 595                 user_data_dir /= get_version_path_suffix();
 596         } else {
 597                 other:
 598                 path home = home_str ? home_str : ".";
 599 
 600                 if (newprefdir[0] == '/')
 601                         user_data_dir = newprefdir;
 602                 else
 603                         user_data_dir = home / newprefdir;
 604         }
 605 #else
 606         if (newprefdir.empty()) newprefdir = backupprefdir;
 607 
 608         const char* home_str = getenv("HOME");
 609         path home = home_str ? home_str : ".";
 610 
 611         if (newprefdir[0] == '/')
 612                 user_data_dir = newprefdir;
 613         else
 614                 user_data_dir = home / newprefdir;
 615 #endif
 616 
 617 #endif /*_WIN32*/

 797 filesystem::scoped_istream istream_file(const std::string &fname, bool treat_failure_as_error)
 798 {
 799         LOG_FS << "Streaming " << fname << " for reading.\n";
 800         if (fname.empty()) {
 801                 ERR_FS << "Trying to open file with empty name.\n";
 802                 filesystem::scoped_istream s(new bfs::ifstream());
 803                 s->clear(std::ios_base::failbit);
 804                 return s;
 805         }
 806 
 807         //mingw doesn't  support std::basic_ifstream::basic_ifstream(const wchar_t* fname)
 808         //that why boost::filesystem::fstream.hpp doesnt work with mingw.
 809         try
 810         {
 811                 boost::iostreams::file_descriptor_source fd(bfs::path(fname), std::ios_base::binary);
 812                 //TODO: has this still use ?
 813                 if (!fd.is_open() && treat_failure_as_error) {
 814                         ERR_FS << "Could not open '" << fname << "' for reading.\n";
 815                 } else if (!is_filename_case_correct(fname, fd)) {
 816                         ERR_FS << "Not opening '" << fname << "' due to case mismatch.\n";
 817                         filesystem::scoped_istream s(new bfs::ifstream());
 818                         s->clear(std::ios_base::failbit);
 819                         return s;
 820                 }
 821                 return filesystem::scoped_istream(new boost::iostreams::stream<boost::iostreams::file_descriptor_source>(fd, 4096, 0));
 822         }
 823         catch(const std::exception ex)
 824         {
 825                 if(treat_failure_as_error)
 826                 {
 827                         ERR_FS << "Could not open '" << fname << "' for reading.\n";
 828                 }
 829                 filesystem::scoped_istream s(new bfs::ifstream());
 830                 s->clear(std::ios_base::failbit);
 831                 return s;
 832         }

 835 filesystem::scoped_ostream ostream_file(const std::string& fname, bool create_directory)
 836 {
 837         LOG_FS << "streaming " << fname << " for writing.\n";
 838 #if 1
 839         try
 840         {
 841                 boost::iostreams::file_descriptor_sink fd(bfs::path(fname), std::ios_base::binary);
 842                 return filesystem::scoped_ostream(new boost::iostreams::stream<boost::iostreams::file_descriptor_sink>(fd, 4096, 0));

 859 void write_file(const std::string& fname, const std::string& data)
 860 {
 861         scoped_ostream os = ostream_file(fname);
 862         os->exceptions(std::ios_base::goodbit);
 863 
 864         const size_t block_size = 4096;
 865         char buf[block_size];
 866 
 867         for(size_t i = 0; i < data.size(); i += block_size) {
 868                 const size_t bytes = std::min<size_t>(block_size,data.size() - i);
 869                 std::copy(data.begin() + i, data.begin() + i + bytes,buf);
 870 
 871                 os->write(buf, bytes);
 872                 if (os->bad())
 873                         throw io_exception("Error writing to file: '" + fname + "'");
 874         }
 875 }

 997 bool is_root(const std::string& path)
 998 {
 999 #ifndef _WIN32
1000         error_code ec;
1001         const bfs::path& p = bfs::canonical(path, ec);
1002         return ec ? false : !p.has_parent_path();
1003 #else
1004         //
1005         // Boost.Filesystem is completely unreliable when it comes to detecting
1006         // whether a path refers to a drive's root directory on Windows, so we are
1007         // forced to take an alternative approach here. Instead of hand-parsing
1008         // strings we'll just call a graphical shell service.
1009         //
1010         // There are several poorly-documented ways to refer to a drive in Windows by
1011         // escaping the filesystem namespace using \\.\, \\?\, and \??\. We're just
1012         // going to ignore those here, which may yield unexpected results in places
1013         // such as the file dialog. This function really shouldn't be used for
1014         // security validation anyway, and there are virtually infinite ways to name
1015         // a drive's root using the NT object namespace so it's pretty pointless to
1016         // try to catch those there.
1017         //
1018         // (And no, shlwapi.dll's PathIsRoot() doesn't recognize \\.\C:\, \\?\C:\, or
1019         // \??\C:\ as roots either.)
1020         //
1021         // More generally, do NOT use this code in security-sensitive applications.
1022         //
1023         // See also: <https://googleprojectzero.blogspot.com/2016/02/the-definitive-guide-on-win32-to-nt.html>
1024         //
1025         const std::wstring& wpath = bfs::path{path}.make_preferred().wstring();
1026         return PathIsRootW(wpath.c_str()) == TRUE;
1027 #endif
1028 }

1160 /**
1161  * Returns a vector with all possible paths to a given type of binary,
1162  * e.g. 'images', 'sounds', etc,
1163  */
1164 const std::vector<std::string>& get_binary_paths(const std::string& type)
1165 {
1166         const paths_map::const_iterator itor = binary_paths_cache.find(type);
1167         if(itor != binary_paths_cache.end()) {
1168                 return itor->second;
1169         }
1170 
1171         if (type.find("..") != std::string::npos) {
1172                 // Not an assertion, as language.cpp is passing user data as type.
1173                 ERR_FS << "Invalid WML type '" << type << "' for binary paths\n";
1174                 static std::vector<std::string> dummy;
1175                 return dummy;
1176         }
1177 
1178         std::vector<std::string>& res = binary_paths_cache[type];
1179 
1180         init_binary_paths();
1181 
1182         for(const std::string &path : binary_paths)
1183         {
1184                 res.push_back(get_user_data_dir() + "/" + path + type + "/");
1185 
1186                 if(!game_config::path.empty()) {
1187                         res.push_back(game_config::path + "/" + path + type + "/");
1188                 }
1189         }
1190 
1191         // not found in "/type" directory, try main directory
1192         res.push_back(get_user_data_dir() + "/");
1193 
1194         if(!game_config::path.empty())
1195                 res.push_back(game_config::path+"/");
1196

1200 std::string get_binary_file_location(const std::string& type, const std::string& filename)
1201 {
1202         // We define ".." as "remove everything before" this is needed becasue
1203         // on the one hand allowing ".." would be a security risk but
1204         // especialy for terrains the c++ engine puts a hardcoded "terrain/" before filename
1205         // and there would be no way to "escape" from "terrain/" otherwise. This is not the
1206         // best solution but we cannot remove it without another solution (subtypes maybe?).
1207 
1208         {
1209                 std::string::size_type pos = filename.rfind("../");
1210                 if(pos != std::string::npos) {
1211                         return get_binary_file_location(type, filename.substr(pos + 3));
1212                 }
1213         }
1214 
1215         if (!is_legal_file(filename))
1216                 return std::string();
1217 
1218         for(const std::string &bp : get_binary_paths(type))
1219         {
1220                 path bpath(bp);
1221                 bpath /= filename;
1222                 DBG_FS << "  checking '" << bp << "'\n";
1223                 if (file_exists(bpath)) {
1224                         DBG_FS << "  found at '" << bpath.string() << "'\n";
1225                         return bpath.string();
1226                 }
1227         }
1228 
1229         DBG_FS << "  not found\n";
1230         return std::string();
1231 }

1288 {
1289         path::iterator fi = full.begin()
1290                      , fe = full.end()
1291                      , pi = prefix.begin()
1292                      , pe = prefix.end();
1293         while (fi != fe && pi != pe && *fi == *pi) {
1294                 ++fi;
1295                 ++pi;

1298         if (pi == pe)
1299                 while (fi != fe) {
1300                         rest /= *fi;
