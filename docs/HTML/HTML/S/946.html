  84 {
  85         ++(chunk_usage[mcp]);

  88 static void decrement_chunk_usage(Mix_Chunk* mcp)
  89 {
  90         if(mcp == nullptr) {
  91                 return;
  92         }
  93 
  94         std::map<Mix_Chunk*, int>::iterator this_usage = chunk_usage.find(mcp);
  95         assert(this_usage != chunk_usage.end());
  96         if(--(this_usage->second) == 0) {
  97                 Mix_FreeChunk(mcp);
  98                 chunk_usage.erase(this_usage);
  99         }

 251 static bool track_ok(const std::string& id)
 252 {
 253         LOG_AUDIO << "Considering " << id << "\n";
 254 
 255         // If they committed changes to list, we forget previous plays, but
 256         // still *never* repeat same track twice if we have an option.
 257         if(id == current_track->file_path()) {
 258                 return false;
 259         }
 260 
 261         if(current_track_list.size() <= 3) {
 262                 return true;
 263         }
 264 
 265         // Timothy Pinkham says:
 266         // 1) can't be repeated without 2 other pieces have already played
 267         // since A was played.
 268         // 2) cannot play more than 2 times without every other piece
 269         // having played at least 1 time.
 270 
 271         // Dammit, if our musicians keep coming up with algorithms, I'll
 272         // be out of a job!
 273         unsigned int num_played = 0;
 274         std::set<std::string> played;
 275         std::vector<std::string>::reverse_iterator i;
 276 
 277         for(i = played_before.rbegin(); i != played_before.rend(); ++i) {
 278                 if(*i == id) {
 279                         ++num_played;
 280                         if(num_played == 2) {
 281                                 break;
 282                         }
 283                 } else {
 284                         played.insert(*i);
 285                 }
 286         }
 287 
 288         // If we've played this twice, must have played every other track.
 289         if(num_played == 2 && played.size() != current_track_list.size() - 1) {
 290                 LOG_AUDIO << "Played twice with only " << played.size() << " tracks between\n";
 291                 return false;
 292         }
 293 
 294         // Check previous previous track not same.
 295         i = played_before.rbegin();
 296         if(i != played_before.rend()) {
 297                 ++i;
 298                 if(i != played_before.rend()) {
 299                         if(*i == id) {
 300                                 LOG_AUDIO << "Played just before previous\n";
 301                                 return false;
 302                         }
 303                 }
 304         }
 305 
 306         return true;
 307 }

 309 static std::shared_ptr<sound::music_track> choose_track()
 310 {
 311         assert(!current_track_list.empty());
 312 
 313         if(current_track_index >= current_track_list.size()) {
 314                 current_track_index = 0;
 315         }
 316 
 317         if(current_track_list[current_track_index]->shuffle()) {
 318                 unsigned int track = 0;
 319 
 320                 if(current_track_list.size() > 1) {
 321                         do {
 322                                 track = rand() % current_track_list.size();
 323                         } while(!track_ok(current_track_list[track]->file_path()));
 324                 }
 325 
 326                 current_track_index = track;
 327         }
 328 
 329         DBG_AUDIO << "Next track will be " << current_track_list[current_track_index]->file_path() << "\n";
 330         played_before.push_back(current_track_list[current_track_index]->file_path());
 331         return current_track_list[current_track_index];
 332 }

 334 static std::string pick_one(const std::string& files)
 335 {
 336         std::vector<std::string> ids = utils::square_parenthetical_split(files, ',', "[", "]");
 337 
 338         if(ids.empty()) {
 339                 return "";
 340         }
 341 
 342         if(ids.size() == 1) {
 343                 return ids[0];
 344         }
 345 
 346         // We avoid returning same choice twice if we can avoid it.
 347         static std::map<std::string, unsigned int> prev_choices;
 348         unsigned int choice;
 349 
 350         if(prev_choices.find(files) != prev_choices.end()) {
 351                 choice = rand() % (ids.size() - 1);
 352                 if(choice >= prev_choices[files]) {
 353                         ++choice;
 354                 }
 355 
 356                 prev_choices[files] = choice;
 357         } else {
 358                 choice = rand() % ids.size();
 359                 prev_choices.emplace(files, choice);
 360         }
 361 
 362         return ids[choice];
 363 }

 391 bool init_sound()
 392 {
 393         LOG_AUDIO << "Initializing audio...\n";
 394         if(SDL_WasInit(SDL_INIT_AUDIO) == 0) {
 395                 if(SDL_InitSubSystem(SDL_INIT_AUDIO) == -1) {
 396                         return false;
 397                 }
 398         }
 399 
 400         if(!mix_ok) {
 401                 if(Mix_OpenAudio(preferences::sample_rate(), MIX_DEFAULT_FORMAT, 2, preferences::sound_buffer_size()) == -1) {
 402                         mix_ok = false;
 403                         ERR_AUDIO << "Could not initialize audio: " << Mix_GetError() << std::endl;
 404                         return false;
 405                 }
 406 
 407                 mix_ok = true;
 408                 Mix_AllocateChannels(n_of_channels);
 409                 Mix_ReserveChannels(n_reserved_channels);
 410 
 411                 channel_chunks.clear();
 412                 channel_chunks.resize(n_of_channels, nullptr);
 413                 channel_ids.resize(n_of_channels, -1);
 414 
 415                 Mix_GroupChannel(bell_channel, SOUND_BELL);
 416                 Mix_GroupChannel(timer_channel, SOUND_TIMER);
 417                 Mix_GroupChannels(source_channel_start, source_channel_last, SOUND_SOURCES);
 418                 Mix_GroupChannels(UI_sound_channel_start, UI_sound_channel_last, SOUND_UI);
 419                 Mix_GroupChannels(n_reserved_channels, n_of_channels - 1, SOUND_FX);
 420 
 421                 set_sound_volume(preferences::sound_volume());
 422                 set_UI_volume(preferences::UI_volume());
 423                 set_music_volume(preferences::music_volume());
 424                 set_bell_volume(preferences::bell_volume());
 425 
 426                 Mix_ChannelFinished(channel_finished_hook);
 427 
 428                 LOG_AUDIO << "Audio initialized.\n";
 429 
 430                 DBG_AUDIO << "Channel layout: " << n_of_channels << " channels (" << n_reserved_channels << " reserved)\n"
 431                           << "    " << bell_channel << " - bell\n"
 432                           << "    " << timer_channel << " - timer\n"
 433                           << "    " << source_channel_start << ".." << source_channel_last << " - sound sources\n"
 434                           << "    " << UI_sound_channel_start << ".." << UI_sound_channel_last << " - UI\n"
 435                           << "    " << UI_sound_channel_last + 1 << ".." << n_of_channels - 1 << " - sound effects\n";
 436 
 437                 play_music();
 438         }
 439 
 440         return true;
 441 }

 443 void close_sound()
 444 {
 445         int frequency, channels;
 446         Uint16 format;
 447 
 448         if(mix_ok) {
 449                 stop_bell();
 450                 stop_UI_sound();
 451                 stop_sound();
 452                 sound_cache.clear();
 453                 stop_music();
 454                 mix_ok = false;
 455 
 456                 int numtimesopened = Mix_QuerySpec(&frequency, &format, &channels);
 457                 if(numtimesopened == 0) {
 458                         ERR_AUDIO << "Error closing audio device: " << Mix_GetError() << std::endl;
 459                 }
 460 
 461                 while(numtimesopened) {
 462                         Mix_CloseAudio();
 463                         --numtimesopened;
 464                 }
 465         }
 466 
 467         if(SDL_WasInit(SDL_INIT_AUDIO) != 0) {
 468                 SDL_QuitSubSystem(SDL_INIT_AUDIO);
 469         }
 470 
 471         LOG_AUDIO << "Audio device released.\n";
 472 }

 589 static void play_new_music()
 590 {
 591         music_start_time = 0; // reset status: no start time
 592         want_new_music = true;
 593 
 594         if(!preferences::music_on() || !mix_ok || !current_track->valid()) {
 595                 return;
 596         }
 597 
 598         const std::string& filename = current_track->file_path();
 599 
 600         auto itor = music_cache.find(filename);
 601         if(itor == music_cache.end()) {
 602                 LOG_AUDIO << "attempting to insert track '" << filename << "' into cache\n";
 603 
 604                 filesystem::rwops_ptr rwops = filesystem::make_read_RWops(filename);
 605                 // SDL takes ownership of rwops
 606                 const std::shared_ptr<Mix_Music> music(Mix_LoadMUSType_RW(rwops.release(), MUS_NONE, true), &Mix_FreeMusic);
 607 
 608                 if(music == nullptr) {
 609                         ERR_AUDIO << "Could not load music file '" << filename << "': " << Mix_GetError() << "\n";
 610                         return;
 611                 }
 612 
 613                 itor = music_cache.emplace(filename, music).first;
 614         }
 615 
 616         LOG_AUDIO << "Playing track '" << filename << "'\n";
 617         int fading_time = current_track->ms_before();
 618         if(no_fading) {
 619                 fading_time = 0;
 620         }
 621 
 622         const int res = Mix_FadeInMusic(itor->second.get(), 1, fading_time);
 623         if(res < 0) {
 624                 ERR_AUDIO << "Could not play music: " << Mix_GetError() << " " << filename << " " << std::endl;
 625         }
 626 
 627         want_new_music = false;
 628 }

 630 void play_music_repeatedly(const std::string& id)
 631 {
 632         // Can happen if scenario doesn't specify.
 633         if(id.empty()) {
 634                 return;
 635         }
 636 
 637         current_track_list.clear();
 638         current_track_list.emplace_back(new music_track(id));
 639 
 640         std::shared_ptr<music_track> last_track = current_track;
 641         current_track = current_track_list.back();
 642         current_track_index = 0;
 643 
 644         // If we're already playing it, don't interrupt.
 645         if(!last_track || *last_track != *current_track) {
 646                 play_music();
 647         }
 648 
 649         last_track.reset();
 650 }

 652 void play_music_config(const config& music_node, int i)
 653 {
 654         //
 655         // FIXME: there is a memory leak somewhere in this function, seemingly related to the shared_ptrs
 656         // stored in current_track_list.
 657         //
 658         // vultraz 5/8/2017
 659         //
 660 
 661         music_track track(music_node);
 662 
 663         if(!track.valid() && !track.id().empty()) {
 664                 ERR_AUDIO << "cannot open track '" << track.id() << "'; disabled in this playlist." << std::endl;
 665         }
 666 
 667         // If they say play once, we don't alter playlist.
 668         if(track.play_once()) {
 669                 set_previous_track(current_track);
 670                 current_track = std::make_shared<music_track>(track);
 671                 current_track_index = current_track_list.size();
 672                 play_music();
 673                 return;
 674         }
 675 
 676         // Clear play list unless they specify append.
 677         if(!track.append()) {
 678                 current_track_list.clear();
 679         }
 680 
 681         if(!track.valid()) {
 682                 return;
 683         }
 684 
 685         auto iter = find_track(track);
 686         // Avoid 2 tracks with the same name, since that can cause an infinite loop
 687         // in choose_track(), 2 tracks with the same name will always return the
 688         // current track and track_ok() doesn't allow that.
 689         if(iter == current_track_list.end()) {
 690                 if(i < 0 || static_cast<size_t>(i) >= current_track_list.size()) {
 691                         current_track_list.emplace_back(new music_track(track));
 692                         iter = current_track_list.end() - 1;
 693                 } else {
 694                         iter = current_track_list.emplace(current_track_list.begin() + 1, new music_track(track));
 695                         if(current_track_index >= static_cast<size_t>(i)) {
 696                                 current_track_index++;
 697                         }
 698                 }
 699         } else {
 700                 ERR_AUDIO << "tried to add duplicate track '" << track.file_path() << "'" << std::endl;
 701         }
 702 
 703         // They can tell us to start playing this list immediately.
 704         if(track.immediate()) {
 705                 set_previous_track(current_track);
 706                 current_track = *iter;
 707                 current_track_index = iter - current_track_list.begin();
 708                 play_music();
 709         } else if(!track.append()) { // Make sure the current track is finished
 710                 current_track->set_play_once(true);
 711         }
 712 }

 714 void music_thinker::process(events::pump_info& info)
 715 {
 716         if(preferences::music_on()) {
 717                 if(!music_start_time && !current_track_list.empty() && !Mix_PlayingMusic()) {
 718                         // Pick next track, add ending time to its start time.
 719                         set_previous_track(current_track);
 720                         current_track = choose_track();
 721                         music_start_time = info.ticks();
 722                         no_fading = true;
 723                         fadingout_time = 0;
 724                 }
 725 
 726                 if(music_start_time && info.ticks(&music_refresh, music_refresh_rate) >= music_start_time - fadingout_time) {
 727                         want_new_music = true;
 728                 }
 729 
 730                 if(want_new_music) {
 731                         if(Mix_PlayingMusic()) {
 732                                 Mix_FadeOutMusic(fadingout_time);
 733                         }
 734 
 735                         unload_music = false;
 736                         play_new_music();
 737                 }
 738         }
 739 
 740         if(unload_music) {
 741                 // The custom shared_ptr deleter (Mix_FreeMusic) will handle the freeing of each track.
 742                 music_cache.clear();
 743 
 744                 Mix_HookMusicFinished(nullptr);
 745 
 746                 unload_music = false;
 747         }
 748 }

 769 void commit_music_changes()
 770 {
 771         played_before.clear();
 772 
 773         // Play-once is OK if still playing.
 774         if(current_track->play_once()) {
 775                 return;
 776         }
 777 
 778         // If current track no longer on playlist, change it.
 779         for(auto m : current_track_list) {
 780                 if(*current_track == *m) {
 781                         return;
 782                 }
 783         }
 784 
 785         // Victory empties playlist: if next scenario doesn't specify one...
 786         if(current_track_list.empty()) {
 787                 return;
 788         }
 789 
 790         // FIXME: we don't pause ms_before on this first track.  Should we?
 791         set_previous_track(current_track);
 792         current_track = choose_track();
 793         play_music();
 794 }

 806 void reposition_sound(int id, unsigned int distance)
 807 {
 808         audio_lock lock;
 809         for(unsigned ch = 0; ch < channel_ids.size(); ++ch) {
 810                 if(channel_ids[ch] != id) {
 811                         continue;
 812                 }
 813 
 814                 if(distance >= DISTANCE_SILENT) {
 815                         // Don't call Mix_FadeOutChannel if the channel's volume is set to
 816                         // zero. It doesn't do anything in that case and the channel will
 817                         // resume playing as soon as its volume is reset to a non-zero
 818                         // value, which results in issues like sound sources deleted while
 819                         // their volume is zero coming back to life and escaping Wesnoth's
 820                         // sound source management code.
 821                         if(Mix_Volume(ch, -1) == 0) {
 822                                 Mix_HaltChannel(ch);
 823                         } else {
 824                                 Mix_FadeOutChannel(ch, 100);
 825                         }
 826                 } else {
 827                         Mix_SetDistance(ch, distance);
 828                 }
 829         }
 830 }

 844 {
 845 };

 847 static Mix_Chunk* load_chunk(const std::string& file, channel_group group)
 848 {
 849         sound_cache_iterator it_bgn, it_end;
 850         sound_cache_iterator it;
 851 
 852         sound_cache_chunk temp_chunk(file); // search the sound cache on this key
 853         it_bgn = sound_cache.begin(), it_end = sound_cache.end();
 854         it = std::find(it_bgn, it_end, temp_chunk);
 855 
 856         if(it != it_end) {
 857                 if(it->group != group) {
 858                         // cached item has been used in multiple sound groups
 859                         it->group = NULL_CHANNEL;
 860                 }
 861 
 862                 // splice the most recently used chunk to the front of the cache
 863                 sound_cache.splice(it_bgn, sound_cache, it);
 864         } else {
 865                 // remove the least recently used chunk from cache if it's full
 866                 bool cache_full = (sound_cache.size() == max_cached_chunks);
 867                 while(cache_full && it != it_bgn) {
 868                         // make sure this chunk is not being played before freeing it
 869                         std::vector<Mix_Chunk*>::iterator ch_end = channel_chunks.end();
 870                         if(std::find(channel_chunks.begin(), ch_end, (--it)->get_data()) == ch_end) {
 871                                 sound_cache.erase(it);
 872                                 cache_full = false;
 873                         }
 874                 }
 875 
 876                 if(cache_full) {
 877                         LOG_AUDIO << "Maximum sound cache size reached and all are busy, skipping.\n";
 878                         throw chunk_load_exception();
 879                 }
 880 
 881                 temp_chunk.group = group;
 882                 const std::string& filename = filesystem::get_binary_file_location("sounds", file);
 883 
 884                 if(!filename.empty()) {
 885                         filesystem::rwops_ptr rwops = filesystem::make_read_RWops(filename);
 886                         temp_chunk.set_data(Mix_LoadWAV_RW(rwops.release(), true)); // SDL takes ownership of rwops
 887                 } else {
 888                         ERR_AUDIO << "Could not load sound file '" << file << "'." << std::endl;
 889                         throw chunk_load_exception();
 890                 }
 891 
 892                 if(temp_chunk.get_data() == nullptr) {
 893                         ERR_AUDIO << "Could not load sound file '" << filename << "': " << Mix_GetError() << "\n";
 894                         throw chunk_load_exception();
 895                 }
 896 
 897                 sound_cache.push_front(temp_chunk);
 898         }
 899 
 900         return sound_cache.begin()->get_data();
 901 }

 903 static void play_sound_internal(const std::string& files,
 904                 channel_group group,
 905                 unsigned int repeats = 0,
 906                 unsigned int distance = 0,
 907                 int id = -1,
 908                 int loop_ticks = 0,
 909                 int fadein_ticks = 0)
 910 {
 911         if(files.empty() || distance >= DISTANCE_SILENT || !mix_ok) {
 912                 return;
 913         }
 914 
 915         audio_lock lock;
 916 
 917         // find a free channel in the desired group
 918         int channel = Mix_GroupAvailable(group);
 919         if(channel == -1) {
 920                 LOG_AUDIO << "All channels dedicated to sound group(" << group << ") are busy, skipping.\n";
 921                 return;
 922         }
 923 
 924         Mix_Chunk* chunk;
 925         std::string file = pick_one(files);
 926 
 927         try {
 928                 chunk = load_chunk(file, group);
 929                 assert(chunk);
 930         } catch(const chunk_load_exception&) {
 931                 return;
 932         }
 933 
 934         /*
 935          * This check prevents SDL_Mixer from blowing up on Windows when UI sound is played
 936          * in response to toggling the checkbox which disables sound.
 937          */
 938         if(group != SOUND_UI) {
 939                 Mix_SetDistance(channel, distance);
 940         }
 941 
 942         int res;
 943         if(loop_ticks > 0) {
 944                 if(fadein_ticks > 0) {
 945                         res = Mix_FadeInChannelTimed(channel, chunk, -1, fadein_ticks, loop_ticks);
 946                 } else {
 947                         res = Mix_PlayChannel(channel, chunk, -1);
 948                 }
 949 
 950                 if(res >= 0) {
 951                         Mix_ExpireChannel(channel, loop_ticks);
 952                 }
 953         } else {
 954                 if(fadein_ticks > 0) {
 955                         res = Mix_FadeInChannel(channel, chunk, repeats, fadein_ticks);
 956                 } else {
 957                         res = Mix_PlayChannel(channel, chunk, repeats);
 958                 }
 959         }
 960 
 961         if(res < 0) {
 962                 ERR_AUDIO << "error playing sound effect: " << Mix_GetError() << std::endl;
 963                 // still keep it in the sound cache, in case we want to try again later
 964                 return;
 965         }
 966 
 967         channel_ids[channel] = id;
 968 
 969         // reserve the channel's chunk from being freed, since it is playing
 970         channel_chunks[res] = chunk;
 971 }
