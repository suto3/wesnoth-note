 102 void scenario::set_sides()
 103 {
 104         if(map_.get() != nullptr) {
 105                 // If there are fewer sides in the configuration than there are
 106                 // starting positions, then generate the additional sides
 107                 const int map_positions = map_->num_valid_starting_positions();
 108 
 109                 for(int pos = data_.child_count("side"); pos < map_positions; ++pos) {
 110                         config& side = data_.add_child("side");
 111                         side["side"] = pos + 1;
 112                         side["team_name"] = "Team " + std::to_string(pos + 1);
 113                         side["canrecruit"] = true;
 114                         side["controller"] = "human";
 115                 }
 116 
 117                 num_players_ = 0;
 118                 for(const config& scenario : data_.child_range("side")) {
 119                         if(scenario["allow_player"].to_bool(true)) {
 120                                 ++num_players_;
 121                         }
 122                 }
 123         }
 124 }

 270                 game_config_manager::get()->game_config(), type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER, video_));

 272         // TODO: the editor dir is already configurable, is the preferences value
 273         filesystem::get_files_in_dir(filesystem::get_user_data_dir() + "/editor/maps", &user_map_names_,

 288                 if(game_config_manager::get()->game_config().find_child("modification", "id", str)) {
 289                         state_.mp_settings().active_mods.push_back(str);
 290                 }
 291         }
 292 
 293         if(current_level_type_ != level::TYPE::CAMPAIGN &&
 294                 current_level_type_ != level::TYPE::SP_CAMPAIGN) {
 295                 dependency_manager_->try_modifications(state_.mp_settings().active_mods, true);
 296         }
 297 
 298         reset_level_filters();
 299 }

 301 void create_engine::init_generated_level_data()
 302 {
 303         DBG_MP << "initializing generated level data\n";
 304 
 305         //DBG_MP << "current data:\n";
 306         //DBG_MP << current_level().data().debug();
 307 
 308         random_map * cur_lev = dynamic_cast<random_map *> (&current_level());
 309 
 310         if(!cur_lev) {
 311                 WRN_MP << "Tried to initialized generated level data on a level that wasn't a random map\n";
 312                 return;
 313         }
 314 
 315         try {
 316                 if(!cur_lev->generate_whole_scenario())
 317                 {
 318                         DBG_MP << "** replacing map ** \n";
 319 
 320                         config data = cur_lev->data();
 321 
 322                         data["map_data"] = generator_->create_map();
 323 
 324                         cur_lev->set_data(data);
 325 
 326                 } else { //scenario generation
 327 
 328                         DBG_MP << "** replacing scenario ** \n";
 329 
 330                         config data = generator_->create_scenario();
 331 
 332                         // Set the scenario to have placing of sides
 333                         // based on the terrain they prefer
 334                         if(!data.has_attribute("modify_placing")) {
 335                                 data["modify_placing"] = true;
 336                         }
 337 
 338                         const std::string& description = cur_lev->data()["description"];
 339                         data["description"] = description;
 340 
 341                         cur_lev->set_data(data);
 342                 }
 343         } catch (mapgen_exception & e) {
 344                 config data = cur_lev->data();
 345 
 346                 data["error_message"] = e.what();
 347 
 348                 cur_lev->set_data(data);
 349         }
 350 
 351         //DBG_MP << "final data:\n";
 352         //DBG_MP << current_level().data().debug();
 353 }

 410 std::string create_engine::select_campaign_difficulty(int set_value)
 411 {
 412         // Verify the existence of difficulties
 413         std::vector<std::string> difficulties;
 414 
 415         for(const config& d : current_level().data().child_range("difficulty")) {
 416                 difficulties.push_back(d["define"]);
 417         }
 418 
 419         if(difficulties.empty()) {
 420                 difficulties = utils::split(current_level().data()["difficulties"]);
 421         }
 422 
 423         // No difficulties found. Exit
 424         if(difficulties.empty()) {
 425                 return "";
 426         }
 427 
 428         // One difficulty found. Use it
 429         if(difficulties.size() == 1) {
 430                 return difficulties[0];
 431         }
 432 
 433         // A specific difficulty value was passed
 434         // Use a minimalistic interface to get the specified define
 435         if(set_value != -1) {
 436                 if(set_value > static_cast<int>(difficulties.size())) {
 437                         std::cerr << "incorrect difficulty number: [" <<
 438                                 set_value << "]. maximum is [" << difficulties.size() << "].\n";
 439                         return "FAIL";
 440                 } else if(set_value < 1) {
 441                         std::cerr << "incorrect difficulty number: [" <<
 442                                 set_value << "]. minimum is [1].\n";
 443                         return "FAIL";
 444                 } else {
 445                         return difficulties[set_value - 1];
 446                 }
 447         }
 448 
 449         // If not, let the user pick one from the prompt
 450         // We don't pass the difficulties vector here because additional data is required
 451         // to constrict the dialog
 452         gui2::dialogs::campaign_difficulty dlg(current_level().data());
 453         dlg.show(video_);
 454 
 455         selected_campaign_difficulty_ = dlg.selected_difficulty();
 456 
 457         return selected_campaign_difficulty_;
 458 }

 513 void create_engine::set_current_level(const size_t index)
 514 {
 515         try {
 516                 current_level_index_ = type_map_.at(current_level_type_.v).games_filtered.at(index);
 517         } catch (std::out_of_range&) {
 518                 current_level_index_ = 0u;
 519         }
 520 
 521         if(current_level_type_ == level::TYPE::RANDOM_MAP) {
 522                 random_map* current_random_map = dynamic_cast<random_map*>(&current_level());
 523 
 524                 // If dynamic cast has failed then we somehow have gotten all the pointers mixed together.
 525                 assert(current_random_map);
 526 
 527                 generator_.reset(current_random_map->create_map_generator());
 528         } else {
 529                 generator_.reset(nullptr);
 530         }
 531 
 532         if(current_level_type_ != level::TYPE::CAMPAIGN &&
 533                 current_level_type_ != level::TYPE::SP_CAMPAIGN) {
 534 
 535                 dependency_manager_->try_scenario(current_level().id());
 536         }

 656 void create_engine::init_all_levels()
 657 {
 658         if(const config& generic_multiplayer = game_config_manager::get()->game_config().child("generic_multiplayer")) {
 659                 config gen_mp_data = generic_multiplayer;
 660 
 661                 // User maps.
 662                 int dep_index_offset = 0;
 663                 for(size_t i = 0; i < user_map_names_.size(); i++)
 664                 {
 665                         config user_map_data = gen_mp_data;
 666                         user_map_data["map_data"] = filesystem::read_map(user_map_names_[i]);
 667 
 668                         // Check if a file is actually a map.
 669                         // Note that invalid maps should be displayed in order to
 670                         // show error messages in the GUI.
 671                         bool add_map = true;
 672                         std::unique_ptr<gamemap> map;
 673                         try {
 674                                 map.reset(new gamemap(game_config_manager::get()->terrain_types(), user_map_data["map_data"]));
 675                         } catch (incorrect_map_format_error& e) {
 676                                 user_map_data["description"] = _("Map could not be loaded: ") + e.message;
 677 
 678                                 ERR_CF << "map could not be loaded: " << e.message << '\n';
 679                         } catch (wml_exception&) {
 680                                 add_map = false;
 681                                 dep_index_offset++;
 682                         }
 683 
 684                         if(add_map) {
 685                                 user_map_ptr new_user_map(new user_map(user_map_data, user_map_names_[i], map.get()));
 686 
 687                                 type_map_[level::TYPE::USER_MAP].games.push_back(new_user_map);
 688 
 689                                 // Since user maps are treated as scenarios, some dependency info is required
 690                                 config depinfo;
 691                                 depinfo["id"] = user_map_names_[i];
 692                                 depinfo["name"] = user_map_names_[i];
 693                                 dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset);
 694                         }
 695                 }
 696 
 697                 // User made scenarios.
 698                 dep_index_offset = 0;
 699                 for(size_t i = 0; i < user_scenario_names_.size(); i++)
 700                 {
 701                         config data;
 702                         try {
 703                                 read(data, *preprocess_file(filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]));
 704                         } catch (config::error & e) {
 705                                 ERR_CF << "Caught a config error while parsing user made (editor) scenarios:\n" << e.message << std::endl;
 706                                 ERR_CF << "Skipping file: " << (filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]) << std::endl;
 707                                 continue;
 708                         }
 709 
 710                         scenario_ptr new_scenario(new scenario(data));
 711                         if(new_scenario->id().empty()) continue;
 712 
 713                         type_map_[level::TYPE::USER_SCENARIO].games.push_back(new_scenario);
 714 
 715                         // Since user scenarios are treated as scenarios, some dependency info is required
 716                         config depinfo;
 717                         depinfo["id"] = data["id"];
 718                         depinfo["name"] = data["name"];
 719                         dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset++);
 720                 }
 721         }
 722 
 723         // Stand-alone scenarios.
 724         for(const config& data : game_config_manager::get()->game_config().child_range("multiplayer"))
 725         {
 726                 if(!data["allow_new_game"].to_bool(true))
 727                         continue;
 728 
 729                 if(!data["campaign_id"].empty())
 730                         continue;
 731 
 732                 if(data.has_attribute("map_generation") || data.has_attribute("scenario_generation")) {
 733                         random_map_ptr new_random_map(new random_map(data));
 734 
 735                         type_map_[level::TYPE::RANDOM_MAP].games.push_back(new_random_map);
 736                 } else {
 737                         scenario_ptr new_scenario(new scenario(data));
 738 
 739                         type_map_[level::TYPE::SCENARIO].games.push_back(new_scenario);
 740                 }
 741         }
 742 
 743         // Campaigns.
 744         for(const config& data : game_config_manager::get()->game_config().child_range("campaign"))
 745         {
 746                 const std::string& type = data["type"];
 747                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 748 
 749                 if(type == "mp" || (type == "hybrid" && mp)) {
 750                         campaign_ptr new_campaign(new campaign(data));
 751 
 752                         type_map_[level::TYPE::CAMPAIGN].games.push_back(new_campaign);
 753                 }
 754 
 755                 if(type == "sp" || type.empty() || (type == "hybrid" && !mp)) {
 756                         campaign_ptr new_sp_campaign(new campaign(data));
 757                         new_sp_campaign->mark_if_completed();
 758 
 759                         type_map_[level::TYPE::SP_CAMPAIGN].games.push_back(new_sp_campaign);
 760                 }
 761         }
 762 
 763         auto& sp_campaigns = type_map_[level::TYPE::SP_CAMPAIGN].games;
 764 
 765         // Sort sp campaigns by rank.
 766         std::stable_sort(sp_campaigns.begin(), sp_campaigns.end(),
 767         [](const create_engine::level_ptr& a, const create_engine::level_ptr& b) {
 768                         return a->data()["rank"].to_int(1000) < b->data()["rank"].to_int(1000);
 769                 }
 770     );
 771 }

 773 void create_engine::init_extras(const MP_EXTRA extra_type)
 774 {
 775         std::vector<extras_metadata_ptr>& extras = get_extras_by_type(extra_type);
 776         const std::string extra_name = (extra_type == ERA) ? "era" : "modification";
 777 
 778         ng::depcheck::component_availability default_availabilty = (extra_type == ERA)
 779                 ? ng::depcheck::component_availability::MP
 780                 : ng::depcheck::component_availability::HYBRID;
 781 
 782         for(const config& extra : game_config_manager::get()->game_config().child_range(extra_name))
 783         {
 784                 ng::depcheck::component_availability type = extra["type"].to_enum(default_availabilty);
 785                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 786 
 787                 if((type != ng::depcheck::component_availability::MP || mp) && (type != ng::depcheck::component_availability::SP || !mp) )
 788                 {
 789                         extras_metadata_ptr new_extras_metadata(new extras_metadata());
 790                         new_extras_metadata->id = extra["id"].str();
 791                         new_extras_metadata->name = extra["name"].str();
 792                         new_extras_metadata->description = extra["description"].str();
 793                         new_extras_metadata->cfg = &extra;
 794 
 795                         extras.push_back(new_extras_metadata);
 796                 }
 797         }
 798 }
