  50 /**
  51  * The builder does the following steps in order to create and ensure
  52  * a correct PDL (Project Dependency Lits)
  53  * 
  54  * 1) remove REMOVED files from the PDL
  55  * 2) parse ADDED or CHANGED files, to check if new directory/file includes
  56  * happened
  57  * 3) add new dependencies to the list
  58  */

  78     @SuppressWarnings( "rawtypes" )
  79     @Override
  80     protected IProject[] build( int kind, Map args, IProgressMonitor monitor )
  81         throws CoreException
  82     {
  83         project_ = getProject( );
  84         if( WesnothInstallsUtils.setupInstallForResource( project_ ) == false ) {
  85             return null;
  86         }
  87 
  88         if( projectCache_ == null ) {
  89             projectCache_ = ProjectUtils.getCacheForProject( project_ );
  90         }
  91 
  92         Logger.getInstance( ).log(
  93             "Building project " + getProject( ).getName( ) + " ..." ); //$NON-NLS-1$
  94         monitor.beginTask(
  95             String.format( Messages.WesnothProjectBuilder_1,
  96                 project_.getName( ) ), 100 );
  97 
  98         String installName = WesnothInstallsUtils
  99             .getInstallNameForResource( project_ );
 100         Paths paths = Preferences.getPaths( installName );
 101 
 102         monitor.subTask( Messages.WesnothProjectBuilder_3 );
 103         if( WorkspaceUtils.checkPathsAreSet( installName, true ) == false ) {
 104             return null;
 105         }
 106         monitor.worked( 5 );
 107 
 108         // create the temporary directory used by the plugin if not created
 109         monitor.subTask( Messages.WesnothProjectBuilder_6 );
 110         WorkspaceUtils.getTemporaryFolder( );
 111         monitor.worked( 2 );
 112 
 113 
 114         if( runAntJob( paths, monitor ) == false ) {
 115             return null;
 116         }
 117 
 118         boolean readDefines = false;
 119 
 120         monitor.subTask( "Build started..." );
 121 
 122         if( kind == FULL_BUILD ) {
 123             readDefines = fullBuild( monitor );
 124         }
 125         else {
 126             IResourceDelta delta = getDelta( project_ );
 127             if( delta == null ) {
 128                 readDefines = fullBuild( monitor );
 129             }
 130             else {
 131                 readDefines = incrementalBuild( delta, monitor );
 132             }
 133         }
 134 
 135         if( readDefines ) {
 136             // we read the defines at the end of the build
 137             // to speed up things (and only if we had any .cfg files processed)
 138             projectCache_.readDefines( true );
 139             projectCache_.saveCache( );
 140 
 141             monitor.worked( 10 );
 142         }
 143         monitor.done( );
 144 
 145         Logger.getInstance( ).log(
 146             "Done building project " + getProject( ).getName( ) );
 147         return null;
 148     }

 150     /**
 151      * Does a full build on this project
 152      * 
 153      * @param monitor
 154      *        The monitor used to signal progress
 155      * @return True if there were config files processed
 156      * @throws CoreException
 157      */
 158     protected boolean fullBuild( final IProgressMonitor monitor )
 159         throws CoreException
 160     {
 161         // clean all project cache
 162         projectCache_.clear( );
 163         PreprocessorUtils.getInstance( ).clearTimestampsForPath(
 164             project_.getLocation( ).toOSString( ) );
 165 
 166         // force creating the dependency list
 167         projectCache_.getDependencyList( ).createDependencyList( true );
 168         boolean foundCfg = false;
 169 
 170         DependencyListNode node = null;
 171 
 172         node = projectCache_.getDependencyList( ).getNode(
 173             DependencyListBuilder.ROOT_NODE_KEY );
 174 
 175         if( node != null ) {
 176 
 177             do {
 178                 foundCfg = true;
 179 
 180                 // process the node
 181                 checkResource( node.getFile( ), monitor );
 182                 node = node.getNext( );
 183             } while( node != null );
 184         }
 185 
 186         // now (re-) preprocess the _main.cfg
 187         checkResource( project_.getFile( "_main.cfg" ), monitor );
 188 
 189         return foundCfg;
 190     }

 192     /**
 193      * Does an incremental build on this project
 194      * 
 195      * @param delta
 196      *        The delta which contains the project modifications
 197      * @param monitor
 198      *        The monitor used to signal progress
 199      * @return True if there were config files processed
 200      * @throws CoreException
 201      */
 202     protected boolean incrementalBuild( IResourceDelta delta,
 203         IProgressMonitor monitor ) throws CoreException
 204     {
 205         // Create the dependency list only if it's not created.
 206         projectCache_.getDependencyList( ).createDependencyList( false );
 207         monitor.worked( 10 );
 208 
 209         boolean foundCfg = false;
 210 
 211         // TODO: unprocessed files should be reprocessed on each build
 212         // until they get so.
 213 
 214         DependencyListBuilder list = projectCache_.getDependencyList( );
 215         Queue< IResourceDelta > deltasQueue = new LinkedBlockingDeque< IResourceDelta >( );
 216         List< DependencyListNode > nodesToProcess = new ArrayList< DependencyListNode >( );
 217 
 218         // gather the list of configs modified
 219         deltasQueue.add( delta );
 220 
 221         while( ! deltasQueue.isEmpty( ) ) {
 222             IResourceDelta deltaItem = deltasQueue.poll( );
 223             IResource resource = deltaItem.getResource( );
 224 
 225             // process just config files
 226             if( ResourceUtils.isConfigFile( resource ) ) {
 227                 IFile file = ( IFile ) resource;
 228                 int deltaKind = deltaItem.getKind( );
 229 
 230                 if( deltaKind == IResourceDelta.REMOVED ) {
 231                     projectCache_.getDependencyList( ).removeNode( file );
 232                     projectCache_.getWMLConfigs( ).remove(
 233                         file.getProjectRelativePath( ).toString( ) );
 234 
 235                 }
 236                 else if( deltaKind == IResourceDelta.ADDED ) {
 237                     DependencyListNode newNode = list.addNode( file );
 238                     if( newNode == null ) {
 239                         Logger.getInstance( ).logError(
 240                             "Couldn't create a new" + "PDL node for file: "
 241                                 + file.getFullPath( ).toString( ) );
 242                     }
 243                     else {
 244                         nodesToProcess.add( newNode );
 245                     }
 246                 }
 247                 else if( deltaKind == IResourceDelta.CHANGED ) {
 248                     DependencyListNode node = list.getNode( file );
 249                     if( node == null ) {
 250                         Logger.getInstance( ).logError(
 251                             "Couldn't find file "
 252                                 + file.getFullPath( ).toString( )
 253                                 + " in PDL!." );
 254                     }
 255                     else {
 256                         nodesToProcess.add( node );
 257                         list.updateNode( node );
 258                     }
 259                 }
 260                 else {
 261                     Logger.getInstance( ).log(
 262                         "unknown delta kind: " + deltaKind );
 263                 }
 264             }
 265 
 266             // skip core library files
 267             if( resource instanceof IContainer
 268                 && WesnothProjectsExplorer.CORE_LIBRARY_NAME
 269                     .equals( resource.getName( ) ) ) {
 270                 continue;
 271             }
 272 
 273             deltasQueue
 274                 .addAll( Arrays.asList( deltaItem.getAffectedChildren( ) ) );
 275         }
 276 
 277         // sort the list by index (ascending)
 278         Collections.sort( nodesToProcess,
 279             new Comparator< DependencyListNode >( ) {
 280 
 281                 @Override
 282                 public int compare( DependencyListNode o1,
 283                     DependencyListNode o2 )
 284                 {
 285                     if( o1.getIndex( ) < o2.getIndex( ) ) {
 286                         return - 1;
 287                     }
 288                     else if( o1.getIndex( ) == o2.getIndex( ) ) {
 289                         return 0;
 290                     }
 291 
 292                     return 1;
 293                 }
 294             } );
 295 
 296         foundCfg = ( ! nodesToProcess.isEmpty( ) );
 297 
 298         // process nodes
 299         for( DependencyListNode node: nodesToProcess ) {
 300             checkResource( node.getFile( ), monitor );
 301         }
 302 
 303         // now (re-) preprocess the _main.cfg
 304         checkResource( project_.getFile( "_main.cfg" ), monitor );
 305 
 306         return foundCfg;
 307     }

 309     /**
 310      * Runs the ant job that copies the project in user add-ons directory
 311      * 
 312      * @param paths
 313      *        The paths instance which contains the paths to wesnoth
 314      *        utilities
 315      * @param monitor
 316      *        The monitor used to signal progress
 317      * @return true or false whether the job completed successfully
 318      */
 319     private boolean runAntJob( Paths paths, IProgressMonitor monitor )
 320     {
 321         String buildXMLPath = project_.getLocation( ).toOSString( )
 322             + "/build.xml";
 323         // check for 'build.xml' existance
 324         if( new File( buildXMLPath ).exists( ) == true ) {
 325             // run the ant job to copy the whole project
 326             // in the user add-ons directory (incremental)
 327             monitor.subTask( Messages.WesnothProjectBuilder_8 );
 328             Map< String, String > properties = new HashMap< String, String >( );
 329             properties.put( "wesnoth.user.dir", paths.getUserDir( ) ); //$NON-NLS-1$
 330             Logger.getInstance( ).logTool( "Ant result:" ); //$NON-NLS-1$
 331 
 332             String result = AntUtils.runAnt( buildXMLPath, properties, true );
 333             Logger.getInstance( ).logTool( result );
 334             monitor.worked( 10 );
 335 
 336             if( result == null ) {
 337                 Logger.getInstance( ).log( "error running the ant job", //$NON-NLS-1$
 338                     Messages.WesnothProjectBuilder_13 );
 339                 return false;
 340             }
 341         }
 342 
 343         monitor.worked( 2 );
 344         return true;
 345     }

 347     protected boolean checkResource( IResource resource,
 348         IProgressMonitor monitor )
 349     {
 350         monitor.worked( 5 );
 351         if( resource.exists( ) == false || monitor.isCanceled( ) ) {
 352             return false;
 353         }
 354 
 355         // config files
 356         if( ResourceUtils.isConfigFile( resource ) ) {
 357             IFile file = ( IFile ) resource;
 358             String filePath = file.getProjectRelativePath( ).toString( );
 359             String macrosFilePath = PreprocessorUtils.getInstance( )
 360                 .getMacrosLocation( file );
 361 
 362             Logger.getInstance( ).log( "Resource: " + filePath ); //$NON-NLS-1$
 363 
 364             try {
 365 
 366                 // preprocess the file only if there's no _main.cfg to gather
 367                 // everything, so we preprocess individually each file
 368                 if( ! project_.getFile( "_main.cfg" ).exists( ) ||
 369                     project_.getFile( "_main.cfg" ).equals( resource ) ) {
 370                     monitor.subTask( String.format(
 371                         Messages.WesnothProjectBuilder_19, filePath ) );
 372 
 373                     // we use a single _MACROS_.cfg file for each project
 374                     PreprocessorUtils.getInstance( ).preprocessFile( file,
 375                         macrosFilePath, new ArrayList< String >( ) );
 376                     monitor.worked( 5 );
 377                 }
 378 
 379                 // process the AST ( Abstract Syntax Tree ) to get info for the
 380                 // file
 381                 monitor.subTask( String.format(
 382                     Messages.WesnothProjectBuilder_22, filePath ) );
 383 
 384                 WMLConfig config = projectCache_.getWMLConfig( filePath );
 385                 SimpleWMLParser parser = new SimpleWMLParser( file, config,
 386                     projectCache_ );
 387                 parser.parse( );
 388                 projectCache_.getDefines( ).putAll( parser.getDefines( ) );
 389 
 390                 monitor.worked( 10 );
 391 
 392             } catch( Exception e ) {
 393                 Logger.getInstance( ).logException( e );
 394             }
 395         }
 396         return true;
 397     }

 399     @SuppressWarnings( "unused" )
 400     private void runWMLLint( String installName, IProgressMonitor monitor,
 401         IFile file )
 402     {
 403         monitor.subTask( String.format( "Running WMLlint on file %s ...",
 404             file.getName( ) ) );
 405         ExternalToolInvoker tool = WMLTools.runWMLLint( installName, file
 406             .getLocation( ).toOSString( ), false, false );
 407         tool.waitForTool( );
 408 
 409         try {
 410             file.deleteMarkers( Constants.MARKER_WMLLINT, false,
 411                 IResource.DEPTH_INFINITE );
 412         } catch( CoreException e ) {
 413             Logger.getInstance( ).logException( e );
 414         }
 415 
 416         String[] output = StringUtils.getLines( tool.getErrorContent( ) );
 417         for( String line: output ) {
 418             WMLTools.parseAndAddMarkers( line, Constants.MARKER_WMLLINT );
 419         }
 420 
 421         monitor.worked( 20 );
 422     }
