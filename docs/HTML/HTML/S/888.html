  48 /**
  49  * Undoes this action.
  50  * @return true on success; false on an error.
  51  */
  52 bool recall_action::undo(int side)
  53 {
  54         game_display & gui = *resources::screen;
  55         unit_map &   units = resources::gameboard->units();
  56         team &current_team = resources::gameboard->get_team(side);
  57 
  58         const map_location & recall_loc = route.front();
  59         unit_map::iterator un_it = units.find(recall_loc);
  60         if ( un_it == units.end() ) {
  61                 return false;
  62         }
  63 
  64         unit_ptr un = un_it.get_shared_ptr();
  65         if (!un) {
  66                 return false;
  67         }
  68 
  69         statistics::un_recall_unit(*un);
  70         int cost = statistics::un_recall_unit_cost(*un);
  71         if (cost < 0) {
  72                 current_team.spend_gold(-current_team.recall_cost());
  73         }
  74         else {
  75                 current_team.spend_gold(-cost);
  76         }
  77 
  78         current_team.recall_list().add(un);
  79         // invalidate before erasing allow us
  80         // to also do the overlapped hexes
  81         gui.invalidate(recall_loc);
  82         units.erase(recall_loc);
  83         this->return_village();
  84         execute_undo_umc_wml();
  85         return true;
  86 }
