  52   static const char EOT = '\x4';
  53   // ASCII 4 is EOT character
  54   // Used to separate msg context from msgid in the hashmap, in MO files
  55   // We use the same formatting system, just for consistency.
  56   // c.f. https://www.gnu.org/software/gettext/manual/html_node/MO-Files.html

  58   static std::string form_context_index(const std::string & msgctxt, const std::string & id) {
  59     return msgctxt + EOT + id;
  60   }

  62   static std::string form_index(const po_message & msg) {
  63     return msg.context ? form_context_index(*msg.context, msg.id) : msg.id;

  67   /***
  68    * Helper for interacting with hashmap results
  69    * get(msg) gets the *singular* string from the message. if it's a plural message, look in singular_index_.
  70    * if it's not a plural message, then there is only one string. also, the po header is never a plural message
  71    */
  72   const std::string & get(const po_message & msg) const {
  73     if (msg.strings().size() == 1) { return msg.strings()[0]; }
  74     return msg.strings()[singular_index_];
  75   }

  77   const std::string & get(const po_message & msg, uint plural) const {
  78     uint idx = (plural == 1 ? singular_index_ : pf_function_object_(plural));
  79     return msg.strings()[idx];
  80   }

  82   /***
  83    * Emplace a message into the hashmap
  84    */
  85   void insert_message(po_message && msg) {
  86     if (!msg.strings().size()) { return; }
  87     // don't allow messages with ZERO translations into the catalog, this will cause segfaults later.
  88     // should perhaps throw an exception here
  89 
  90     if (!msg.strings()[0].size()) { return; }
  91     // if the (first) translated string is "", it is untranslated and message does not enter catalog
  92 
  93     if (msg.strings().size() > 1 && msg.strings().size() != metadata_.num_plural_forms) {
  94       if (warning_channel_) {
  95         warning_channel_("Ignoring a message with an incorrect number of plural forms: plural = " + std::to_string(msg.strings().size()) + " msgid = '" + msg.id + "'");
  96       }
  97       return;
  98     }
  99 
 100     std::string index = form_index(msg);
 101     // adjust the id based on context if necessary
 102 
 103     auto result = hashmap_.emplace(std::move(index), std::move(msg));
 104 
 105     // Issue a warning if emplace failed, rather than silently overwrite.
 106     if (!result.second) {
 107       if (warning_channel_) {
 108         std::string warning = "Overwriting a message: msgid = <<<" + msg.id + ">>>";
 109         if (msg.context) { warning += " msgctxt = <<<" + *msg.context + ">>>"; }
 110         warning_channel_(warning);
 111       }
 112       result.first->second = std::move(msg);
 113     }
 114   }

 133   template <typename Iterator>
 134   catalog(spirit::line_pos_iterator<Iterator> & it, spirit::line_pos_iterator<Iterator> & end, warning_channel_type warn_channel = warning_channel_type(), pf_compiler compiler = pf_compiler())
 135     : metadata_()
 136     , pf_function_object_()
 137     , warning_channel_(warn_channel)
 138     , hashmap_()
 139   {
 140     typedef spirit::line_pos_iterator<Iterator> iterator_type;
 141     po_grammar<iterator_type> grammar;
 142 
 143     po_message msg;
 144     std::size_t line_no = 0;
 145 
 146     // Parse header first
 147     {
 148       // must be able to parse first message
 149       qi::parse(it, end, grammar.skipped_block); // first parse any comments
 150       if (!qi::parse(it, end, grammar, msg)) {   // now parse the main grammar target
 151         int err_line = it.position();
 152         SPIRIT_PO_CATALOG_FAIL("Failed to parse po header, stopped at line " + std::to_string(err_line) + ": " + iterator_context(it, end));
 153       }
 154 
 155       // first message must have empty MSGID (po format says so)
 156       if (msg.id.size()) {
 157         SPIRIT_PO_CATALOG_FAIL("Failed to parse po header, first msgid must be empty string \"\", found: " + msg.id);
 158       }
 159 
 160       // Now parse the header string itself
 161       if (msg.strings().size()) {
 162         std::string maybe_error = metadata_.parse_header(msg.strings()[0]);
 163         if (maybe_error.size()) {
 164           SPIRIT_PO_CATALOG_FAIL("Failed to parse po header: " + maybe_error);
 165         }
 166       }
 167 
 168       if (!metadata_.num_plural_forms) {
 169         SPIRIT_PO_CATALOG_FAIL("Invalid metadata in po header, found num_plurals = 0");
 170       }
 171 
 172       // Try to compile the plural forms function string
 173       pf_function_object_ = compiler(metadata_.plural_forms_function_string);
 174       if (!pf_function_object_) {
 175         SPIRIT_PO_CATALOG_FAIL(("Failed to read plural forms function. "
 176                                 "Input: '" + metadata_.plural_forms_function_string + "', "
 177                                 "error message: " + pf_function_object_.error()));
 178       } 
 179 
 180       // Cache the 'singular' form index since it is most common
 181       singular_index_ = pf_function_object_(1);
 182       if (singular_index_ >= metadata_.num_plural_forms) {
 183         SPIRIT_PO_CATALOG_FAIL(("Invalid plural forms function. "
 184                                 "On input n = 1, returned plural = " + std::to_string(singular_index_) + ", "
 185                                 "while num_plurals = " + std::to_string(metadata_.num_plural_forms)));
 186       }
 187 
 188       msg.line_no = line_no;
 189       insert_message(std::move(msg)); // for compatibility, need to insert the header message at msgid ""
 190     }
 191 
 192     // Now parse non-fuzzy messages
 193     while (it != end) {
 194       // this parse rule cannot fail, it can be a zero length match
 195       qi::parse(it, end, grammar.ignored_comments);
 196 
 197       bool fuzzy = false;
 198       // this parse rule cannot fail, it can be a zero length match
 199       qi::parse(it, end, grammar.message_preamble, fuzzy);
 200 
 201       // check if we exhausted the file by comments
 202       if (it != end) {
 203         msg = po_message{};
 204         msg.strings().reserve(metadata_.num_plural_forms); // try to prevent frequent vector reallocations
 205         line_no = it.position();
 206         // actually parse a message
 207         if (!qi::parse(it, end, grammar, msg)) {
 208           int err_line = it.position();
 209           SPIRIT_PO_CATALOG_FAIL(("Failed to parse po file, "
 210                                   "started at " + std::to_string(line_no) + ": , stopped at " + std::to_string(err_line) + ":\n"
 211                                   + iterator_context(it, end)));
 212         }
 213         // cannot overwrite header
 214         if (!msg.id.size()) {
 215           int err_line = it.position();
 216           SPIRIT_PO_CATALOG_FAIL(("Malformed po file: Cannot overwrite the header entry later in the po file."
 217                                   "Started at " + std::to_string(line_no) + ": , stopped at " + std::to_string(err_line) + ":\n" 
 218                                   + iterator_context(it, end)));
 219         }
 220         msg.line_no = line_no;
 221         // only insert it if it wasn't marked fuzzy
 222         if (!fuzzy) { insert_message(std::move(msg)); }
 223       }
 224     }
 225 
 226 #ifdef SPIRIT_PO_DEBUG
 227     // validate resulting hashmap
 228     for (const auto & p : hashmap_) {
 229       if (!p.second.strings().size()) { SPIRIT_PO_CATALOG_FAIL(("Internal catalog error: found a message id with no strings, msgid='" + p.first + "'")); }
 230       if (p.second.strings().size() != 1 && p.second.strings().size() != metadata_.num_plural_forms) {
 231         SPIRIT_PO_CATALOG_FAIL(("Internal catalog error: found a message id with wrong number of strings, msgid='" + p.first + "' num msgstr = " + std::to_string(p.second.strings().size()) + ", catalog num_plural_forms = " + std::to_string(metadata_.num_plural_forms) + "\nWhole message: " + debug_string(p.second) ));
 232       }
 233     }
 234 #endif // SPIRIT_PO_DEBUG
 235   }

 270   /***
 271    * Lookup strings from the catalog
 272    *
 273    * When using string literals as the parameters, these versions are safe and
 274    * are maximally efficient.
 275    * (The returned pointer is either the input pointer, having static storage
 276    * duration, or has lifetime as long as the catalog.)
 277    *
 278    * Chosen to behave in the same manner as corresponding gettext functions.
 279    */
 280   const char * gettext(const char * msgid) const {
 281     auto it = hashmap_.find(msgid);
 282     if (it != hashmap_.end()) {
 283       return get(it->second).c_str();
 284     } else {
 285       return msgid;
 286     }
 287   }

 289   const char * ngettext(const char * msgid, const char * msgid_plural, uint plural) const {

 401   /***
 402    * Merge a different catalog into this one
 403    */
 404   template <typename H, typename P>
 405   void merge(catalog<H, P> && other) {
 406     std::string maybe_error = metadata_.check_compatibility(other.metadata_);
 407     if (maybe_error.size()) {
 408       SPIRIT_PO_CATALOG_FAIL(("Cannot merge catalogs: " + maybe_error));
 409     }
 410     for (auto & p : other.hashmap_) {
 411       if (p.first.size()) { // don't copy over the header, keep our original header
 412         insert_message(std::move(p.second));
 413       }
 414     }
 415   }
