 194     private void updatePageIsComplete( )
 195     {
 196         IResource container = ResourcesPlugin.getWorkspace( ).getRoot( )
 197             .findMember( new Path( getDirectory( ) ) );
 198         setPageComplete( false );
 199         String fileName = getFileName( );
 200 
 201         if( getDirectory( ).isEmpty( ) ) {
 202             setErrorMessage( Messages.FactionPage0_13 );
 203             return;
 204         }
 205 
 206         if( container == null || ! container.exists( )
 207             || ! ( container instanceof IContainer ) ) {
 208             setErrorMessage( Messages.FactionPage0_14 );
 209             return;
 210         }
 211 
 212         if( fileName.isEmpty( ) ) {
 213             setErrorMessage( Messages.FactionPage0_15 );
 214             return;
 215         }
 216 
 217         if( fileName.replace( '\\', '/' ).indexOf( '/', 1 ) > 0 ) {
 218             setErrorMessage( Messages.FactionPage0_16 );
 219             return;
 220         }
 221 
 222         int dotLoc = fileName.lastIndexOf( '.' );
 223         if( dotLoc == - 1
 224             || fileName.substring( dotLoc + 1 ).equalsIgnoreCase( "cfg" ) == false ) //$NON-NLS-1$
 225         {
 226             setErrorMessage( Messages.FactionPage0_18 );
 227             return;
 228         }
 229 
 230         if( getFactionId( ).isEmpty( ) ) {
 231             setErrorMessage( Messages.FactionPage0_19 );
 232             return;
 233         }
 234 
 235         if( getFactionName( ).isEmpty( ) ) {
 236             setErrorMessage( Messages.FactionPage0_20 );
 237             return;
 238         }
 239 
 240         setErrorMessage( null );
 241         setPageComplete( true );
 242     }
