   1 #textdomain wesnoth-utbs
   2 
   3 # event for the teaching ability
   4 #define ABILITY_TEACHING_EVENT
   5     [event]
   6         name=turn refresh,victory
   7         id=garak_teaching
   8         first_time_only=no
   9 
  10         [lua]
  11             # wmllint: markcheck off
  12             code=<<
  13                 local _ = wesnoth.textdomain "wesnoth-utbs"
  14 
  15                 local teachers = wesnoth.get_units { ability = "teaching", side = wesnoth.current.side }
  16 
  17                 for i, teacher in ipairs(teachers) do
  18                     local students = wesnoth.get_units { side = wesnoth.current.side, { "filter_adjacent", { id = teacher.id } } }
  19 
  20                     if #students > 0 then -- don't divide by zero
  21                         wesnoth.scroll_to_tile(teacher.x, teacher.y)
  22 
  23                         local function cf(u1, u2)
  24                             -- criteria: the unit with lower level comes first
  25                             -- if level is equal, the unit with less experience comes first
  26                             if u1.level < u2.level then return true end
  27                             if u1.level > u2.level then return false end
  28                             if u1.experience < u2.experience then return true end
  29                             return false
  30                         end
  31 
  32                         table.sort(students, cf)
  33 
  34                         -- now we need to divide Garak's xp between all the units
  35                         -- we can't just round the result of the division, because this will mean
  36                         -- either not redistributing all the XP points, or generating them out of thin air
  37                         -- so we perform both a division and a modulo
  38                         -- # (length) has a higher precedence than *, / and %
  39                         local div = math.floor(teacher.experience / #students)
  40                         local mod = teacher.experience % #students
  41 
  42                         for j, student in ipairs(students) do
  43                             -- the modulo operation is used to redistribute every remaining XP
  44                             -- so at the start of each turn Garak goes back to 0 XP
  45                             -- concrete example: we have 9 XP and 4 units
  46                             -- the first unit will receive 3 XP, while the other units 2
  47                             local xp_to_add = div
  48                             if j <= mod then
  49                                 xp_to_add = xp_to_add + 1
  50                             end
  51 
  52                             -- having a message that tells you that the unit gained 0 XP is simply obnoxious
  53                             if xp_to_add > 0 then
  54                                 student.experience = student.experience + xp_to_add
  55                                 -- performance-wise, string concatenation is slower than string interpolation
  56                                 -- but it's way more readable than two nested interpolations...
  57                                 wesnoth.float_label(student.x, student.y, "<span color='cyan'>" .. tostring(_ "+%d XP"):format(xp_to_add) .. "</span>")
  58                                 if student.experience >= student.max_experience then
  59                                     -- advance unit, animate and fire events
  60                                     wesnoth.advance_unit(student, true, true)
  61                                 end
  62                             end
  63                         end
  64                     end
  65 
  66                     -- in case that Garak has no "students", his experience goes wasted instead of being redistributed...
  67                     teacher.experience = 0
  68                 end
  69             >>
  70             # wmllint: markcheck off
  71         [/lua]
  72     [/event]
  73 #enddef
