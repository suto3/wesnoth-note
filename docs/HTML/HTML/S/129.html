 123 void parser::operator()()
 124 {
 125         cfg_.clear();
 126         elements.emplace(&cfg_, "");
 127 
 128         do {
 129                 tok_.next_token();
 130 
 131                 switch(tok_.current_token().type) {
 132                 case token::LF:
 133                         continue;
 134 
 135                 case '[':
 136                         parse_element();
 137                         break;
 138 
 139                 case token::STRING:
 140                         parse_variable();
 141                         break;
 142 
 143                 default:
 144                         if(static_cast<unsigned char>(tok_.current_token().value[0]) == 0xEF &&
 145                            static_cast<unsigned char>(tok_.next_token().value[0])    == 0xBB &&
 146                            static_cast<unsigned char>(tok_.next_token().value[0])    == 0xBF
 147                         ) {
 148                                 utils::string_map i18n_symbols;
 149                                 std::stringstream ss;
 150                                 ss << tok_.get_start_line() << " " << tok_.get_file();
 151                                 ERR_CF << lineno_string(i18n_symbols, ss.str(), "Skipping over a utf8 BOM at $pos") << '\n';
 152                         } else {
 153                                 error(_("Unexpected characters at line start"));
 154                         }
 155 
 156                         break;
 157 
 158                 case token::END:
 159                         break;
 160                 }
 161         } while(tok_.current_token().type != token::END);
 162 
 163         // The main element should be there. If it is not, this is a parser error.
 164         assert(!elements.empty());
 165 
 166         if(elements.size() != 1) {
 167                 utils::string_map i18n_symbols;
 168                 i18n_symbols["tag"] = elements.top().name;
 169 
 170                 std::stringstream ss;
 171                 ss << elements.top().start_line << " " << elements.top().file;
 172 
 173                 error(lineno_string(i18n_symbols, ss.str(),
 174                         _("Missing closing tag for tag [$tag]"),
 175                         _("expected at $pos")),
 176                         _("opened at $pos")
 177                 );
 178         }
 179 }

 181 void parser::parse_element()
 182 {
 183         tok_.next_token();
 184 
 185         std::string elname;
 186         config* current_element = nullptr;
 187 
 188         switch(tok_.current_token().type) {
 189         case token::STRING: // [element]
 190                 elname = tok_.current_token().value;
 191 
 192                 if(tok_.next_token().type != ']') {
 193                         error(_("Unterminated [element] tag"));
 194                 }
 195 
 196                 // Add the element
 197                 current_element = &(elements.top().cfg->add_child(elname));
 198                 elements.emplace(current_element, elname, tok_.get_start_line(), tok_.get_file());
 199 
 200                 if(validator_) {
 201                         validator_->open_tag(elname, tok_.get_start_line(), tok_.get_file());
 202                 }
 203 
 204                 break;
 205 
 206         case '+': // [+element]
 207                 if(tok_.next_token().type != token::STRING) {
 208                         error(_("Invalid tag name"));
 209                 }
 210 
 211                 elname = tok_.current_token().value;
 212 
 213                 if(tok_.next_token().type != ']') {
 214                         error(_("Unterminated [+element] tag"));
 215                 }
 216 
 217                 // Find the last child of the current element whose name is element
 218                 if(config& c = elements.top().cfg->child(elname, -1)) {
 219                         current_element = &c;
 220 
 221                         if(validator_) {
 222                                 validator_->open_tag(elname, tok_.get_start_line(), tok_.get_file(), true);
 223                         }
 224                 } else {
 225                         current_element = &elements.top().cfg->add_child(elname);
 226 
 227                         if(validator_) {
 228                                 validator_->open_tag(elname, tok_.get_start_line(), tok_.get_file());
 229                         }
 230                 }
 231 
 232                 elements.emplace(current_element, elname, tok_.get_start_line(), tok_.get_file());
 233                 break;
 234 
 235         case '/': // [/element]
 236                 if(tok_.next_token().type != token::STRING) {
 237                         error(_("Invalid closing tag name"));
 238                 }
 239 
 240                 elname = tok_.current_token().value;
 241 
 242                 if(tok_.next_token().type != ']') {
 243                         error(_("Unterminated closing tag"));
 244                 }
 245 
 246                 if(elements.size() <= 1) {
 247                         error(_("Unexpected closing tag"));
 248                 }
 249 
 250                 if(elname != elements.top().name) {
 251                         utils::string_map i18n_symbols;
 252                         i18n_symbols["tag1"] = elements.top().name;
 253                         i18n_symbols["tag2"] = elname;
 254 
 255                         std::stringstream ss;
 256                         ss << elements.top().start_line << " " << elements.top().file;
 257 
 258                         error(lineno_string(i18n_symbols, ss.str(),
 259                                 _("Found invalid closing tag [/$tag2] for tag [$tag1]"),
 260                                 _("opened at $pos")),
 261                                 _("closed at $pos")
 262                         );
 263                 }
 264 
 265                 if(validator_) {
 266                         element& el = elements.top();
 267                         validator_->validate(*el.cfg, el.name, el.start_line, el.file);
 268                         validator_->close_tag();
 269                 }
 270 
 271                 elements.pop();
 272                 break;
 273 
 274         default:
 275                 error(_("Invalid tag name"));
 276         }
 277 }

 279 void parser::parse_variable()
 280 {
 281         config& cfg = *elements.top().cfg;
 282         std::vector<std::string> variables;
 283         variables.emplace_back();
 284 
 285         while(tok_.current_token().type != '=') {
 286                 switch(tok_.current_token().type) {
 287                 case token::STRING:
 288                         if(!variables.back().empty()) {
 289                                 variables.back() += ' ';
 290                         }
 291 
 292                         variables.back() += tok_.current_token().value;
 293                         break;
 294 
 295                 case ',':
 296                         if(variables.back().empty()) {
 297                                 error(_("Empty variable name"));
 298                         } else {
 299                                 variables.emplace_back();
 300                         }
 301 
 302                         break;
 303 
 304                 default:
 305                         error(_("Unexpected characters after variable name (expected , or =)"));
 306                         break;
 307                 }
 308 
 309                 tok_.next_token();
 310         }
 311 
 312         if(variables.back().empty()) {
 313                 error(_("Empty variable name"));
 314         }
 315 
 316         t_string_base buffer;
 317 
 318         std::vector<std::string>::const_iterator curvar = variables.begin();
 319 
 320         bool ignore_next_newlines = false, previous_string = false;
 321 
 322         while(1) {
 323                 tok_.next_token();
 324                 assert(curvar != variables.end());
 325 
 326                 switch(tok_.current_token().type) {
 327                 case ',':
 328                         if((curvar + 1) != variables.end()) {
 329                                 if(buffer.translatable()) {
 330                                         cfg[*curvar] = t_string(buffer);
 331                                 } else {
 332                                         cfg[*curvar] = buffer.value();
 333                                 }
 334 
 335                                 if(validator_) {
 336                                         validator_->validate_key(cfg, *curvar, buffer.value(), tok_.get_start_line(), tok_.get_file());
 337                                 }
 338 
 339                                 buffer = t_string_base();
 340                                 ++curvar;
 341                         } else {
 342                                 buffer += ",";
 343                         }
 344 
 345                         break;
 346 
 347                 case '_':
 348                         tok_.next_token();
 349 
 350                         switch(tok_.current_token().type) {
 351                         case token::UNTERMINATED_QSTRING:
 352                                 error(_("Unterminated quoted string"));
 353                                 break;
 354 
 355                         case token::QSTRING:
 356                                 buffer += t_string_base(tok_.current_token().value, tok_.textdomain());
 357                                 break;
 358 
 359                         default:
 360                                 buffer += "_";
 361                                 buffer += tok_.current_token().value;
 362                                 break;
 363 
 364                         case token::END:
 365                         case token::LF:
 366                                 buffer += "_";
 367                                 goto finish;
 368                         }
 369 
 370                         break;
 371 
 372                 case '+':
 373                         ignore_next_newlines = true;
 374                         continue;
 375 
 376                 case token::STRING:
 377                         if(previous_string) {
 378                                 buffer += " ";
 379                         }
 380 
 381                         FALLTHROUGH;
 382 
 383                 default:
 384                         buffer += tok_.current_token().value;
 385                         break;
 386 
 387                 case token::QSTRING:
 388                         buffer += tok_.current_token().value;
 389                         break;
 390 
 391                 case token::UNTERMINATED_QSTRING:
 392                         error(_("Unterminated quoted string"));
 393                         break;
 394 
 395                 case token::LF:
 396                         if(ignore_next_newlines) {
 397                                 continue;
 398                         }
 399 
 400                         FALLTHROUGH;
 401 
 402                 case token::END:
 403                         goto finish;
 404                 }
 405 
 406                 previous_string = tok_.current_token().type == token::STRING;
 407                 ignore_next_newlines = false;
 408         }
 409 
 410 finish:
 411 
 412         if(buffer.translatable()) {
 413                 cfg[*curvar] = t_string(buffer);
 414         } else {
 415                 cfg[*curvar] = buffer.value();
 416         }
 417 
 418         if(validator_) {
 419                 validator_->validate_key(cfg, *curvar, buffer.value(), tok_.get_start_line(), tok_.get_file());
 420         }
 421 
 422         while(++curvar != variables.end()) {
 423                 cfg[*curvar] = "";
 424         }
 425 }

 430 std::string parser::lineno_string(utils::string_map& i18n_symbols,
 431                 const std::string& lineno,

 433                 const std::string& hint_string,
 434                 const std::string& debug_string)
 435 {
 436         i18n_symbols["pos"] = ::lineno_string(lineno);
 437         std::string result = error_string;

 440                 result += '\n' + hint_string;
 441         }
 442 
 443         if(!debug_string.empty()) {
 444                 result += '\n' + debug_string;
 445         }

 447         for(utils::string_map::value_type& var : i18n_symbols) {
 448                 boost::algorithm::replace_all(result, std::string("$") + var.first, std::string(var.second));
 449         }
 450 
 451         return result;
 452 }

 454 void parser::error(const std::string& error_type, const std::string& pos_format)
 455 {
 456         std::string hint_string = pos_format;
 457 
 458         if(hint_string.empty()) {
 459                 hint_string = _("at $pos");
 460         }
 461 
 462         utils::string_map i18n_symbols;
 463         i18n_symbols["error"] = error_type;
 464 
 465         std::stringstream ss;
 466         ss << tok_.get_start_line() << " " << tok_.get_file();
 467 
 468 #ifdef DEBUG_TOKENIZER
 469         i18n_symbols["value"] = tok_.current_token().value;
 470         i18n_symbols["previous_value"] = tok_.previous_token().value;
 471 
 472         const std::string& tok_state = _("Value: '$value' Previous: '$previous_value'");
 473 #else
 474         const std::string& tok_state = "";
 475 #endif
 476 
 477         const std::string& message = lineno_string(i18n_symbols, ss.str(), "$error", hint_string, tok_state);
 478 
 479         throw config::error(message);

 487 /**
 488  * Copies a string fragment and converts it to a suitable format for WML.
 489  * (I.e., quotes are doubled.)
 490  */
 491 std::string escaped_string(const std::string::const_iterator& begin, const std::string::const_iterator& end)
 492 {
 493         std::string res;
 494         std::string::const_iterator iter = begin;
 495 
 496         while(iter != end) {
 497                 const char c = *iter;
 498                 res.append(c == '"' ? 2 : 1, c);
 499                 ++iter;
 500         }
 501 
 502         return res;
 503 }

 506  * Copies a string and converts it to a suitable format for WML.
 507  * (I.e., quotes are doubled.)
 508  */
 509 inline std::string escaped_string(const std::string& value)

 511         return escaped_string(value.begin(), value.end());

 564 /**
 565  * Writes all the parts of a translatable string.
 566  *
 567  * @note If the first part is translatable and in the wrong textdomain,
 568  *       the textdomain change has to happen before the attribute name.
 569  *       That is the reason for not outputting the key beforehand and
 570  *       letting this function do it.
 571  */
 572 void write_key_val_visitor::operator()(t_string const& value) const
 573 {
 574         bool first = true;
 575 
 576         for(t_string::walker w(value); !w.eos(); w.next()) {
 577                 if(!first) {
 578                         out_ << " +\n";
 579                 }
 580 
 581                 if(w.translatable() && w.textdomain() != textdomain_) {
 582                         textdomain_ = w.textdomain();
 583                         out_ << "#textdomain " << textdomain_ << '\n';
 584                 }
 585 
 586                 indent();
 587 
 588                 if(first) {
 589                         out_ << key_ << '=';
 590                 } else {
 591                         out_ << '\t';
 592                 }
 593 
 594                 if(w.translatable()) {
 595                         out_ << '_';
 596                 }
 597 
 598                 out_ << '"' << escaped_string(w.begin(), w.end()) << '"';
 599                 first = false;
 600         }
 601 
 602         out_ << '\n';
 603 }

 623 template<typename decompressor>
 624 void read_compressed(config& cfg, std::istream& file, abstract_validator* validator)
 625 {
 626         // An empty gzip file seems to confuse boost on MSVC, so return early if this is the case.
 627         if(file.peek() == EOF) {
 628                 return;
 629         }
 630 
 631         boost::iostreams::filtering_stream<boost::iostreams::input> filter;
 632         filter.push(decompressor());
 633         filter.push(file);
 634 
 635         /* This causes gzip_error (and the corresponding bz2 error, std::ios_base::failure) to be
 636          * thrown here. save_index_class::data expects that and config_cache::read_cache and other
 637          * functions are also capable of catching.
 638          *
 639          * Note that parser(cuff, filter,validator)(); -> tokenizer::tokenizer can throw exceptions
 640          * too (meaning this function already threw these exceptions before this patch).
 641          *
 642          * We try to fix https://svn.boost.org/trac/boost/ticket/5237 by not creating empty gz files.
 643          */
 644         filter.exceptions(filter.exceptions() | std::ios_base::badbit);
 645 
 646         /*
 647          * It sometimes seems the file is not empty but still has no real data.
 648          * Filter that case here. The previous test might be no longer required but keep it for now.
 649          *
 650          * On msvc filter.peek() != EOF does not imply filter.good().
 651          * We never create empty compressed gzip files because boosts gzip fails at doing that, but
 652          * empty compressed bz2 files are possible.
 653          */
 654         if(filter.peek() == EOF) {
 655                 LOG_CF << "Empty compressed file or error at reading a compressed file.";
 656                 return;
 657         }
 658 
 659         if(!filter.good()) {
 660                 LOG_CF << " filter.peek() != EOF but !filter.good()."
 661                        << "This indicates a malformed gz stream and can make Wesnoth crash.";
 662         }
 663 
 664         parser(cfg, filter, validator)();
 665 }

 698 static void write_internal(config const& cfg, std::ostream& out, std::string& textdomain, size_t tab = 0)
 699 {
 700         if(tab > max_recursion_levels) {
 701                 throw config::error("Too many recursion levels in config write");
 702         }
 703 
 704         for(const config::attribute& i : cfg.attribute_range()) {
 705                 if(!config::valid_id(i.first)) {
 706                         ERR_CF << "Config contains invalid attribute name '" << i.first << "', skipping...\n";
 707                         continue;
 708                 }
 709 
 710                 write_key_val(out, i.first, i.second, tab, textdomain);
 711         }
 712 
 713         for(const config::any_child& item : cfg.all_children_range()) {
 714                 if(!config::valid_id(item.key)) {
 715                         ERR_CF << "Config contains invalid tag name '" << item.key << "', skipping...\n";
 716                         continue;
 717                 }
 718 
 719                 write_open_child(out, item.key, tab);
 720                 write_internal(item.cfg, out, textdomain, tab + 1);
 721                 write_close_child(out, item.key, tab);
 722         }
 723 }

 725 static void write_internal(configr_of const& cfg, std::ostream& out, std::string& textdomain, size_t tab = 0)
 726 {
 727         if(tab > max_recursion_levels) {
 728                 throw config::error("Too many recursion levels in config write");
 729         }
 730 
 731         if(cfg.data_) {
 732                 write_internal(*cfg.data_, out, textdomain, tab);
 733         }
 734 
 735         for(const auto& pair : cfg.subtags_) {
 736                 assert(pair.first && pair.second);
 737 
 738                 if(!config::valid_id(*pair.first)) {
 739                         ERR_CF << "Config contains invalid tag name '" << *pair.first << "', skipping...\n";
 740                         continue;
 741                 }
 742 
 743                 write_open_child(out, *pair.first, tab);
 744                 write_internal(*pair.second, out, textdomain, tab + 1);
 745                 write_close_child(out, *pair.first, tab);
 746         }
 747 }
