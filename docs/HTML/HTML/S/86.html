  99 static int placing_score(const config& side, const gamemap& map, const map_location& pos)
 100 {
 101         int positions = 0, liked = 0;
 102         const t_translation::ter_list terrain = t_translation::read_list(side["terrain_liked"]);
 103 
 104         for(int i = -8; i != 8; ++i) {
 105                 for(int j = -8; j != +8; ++j) {
 106                         const map_location pos2  = pos.plus(i, j);
 107                         if(map.on_board(pos2)) {
 108                                 ++positions;
 109                                 if(std::count(terrain.begin(),terrain.end(),map[pos2])) {
 110                                         ++liked;
 111                                 }
 112                         }
 113                 }
 114         }
 115 
 116         return (100*liked)/positions;
 117 }

 135 void game_state::place_sides_in_preferred_locations(const config& level)
 136 {
 137         std::vector<placing_info> placings;
 138 
 139         int num_pos = board_.map().num_valid_starting_positions();
 140 
 141         int side_num = 1;
 142         for(const config &side : level.child_range("side"))
 143         {
 144                 for(int p = 1; p <= num_pos; ++p) {
 145                         const map_location& pos = board_.map().starting_position(p);
 146                         int score = placing_score(side, board_.map(), pos);
 147                         placing_info obj;
 148                         obj.side = side_num;
 149                         obj.score = score;
 150                         obj.pos = pos;
 151                         placings.push_back(obj);
 152                 }
 153                 ++side_num;
 154         }
 155 
 156         std::stable_sort(placings.begin(),placings.end());
 157         std::set<int> placed;
 158         std::set<map_location> positions_taken;
 159 
 160         for (std::vector<placing_info>::const_iterator i = placings.begin(); i != placings.end() && int(placed.size()) != side_num - 1; ++i) {
 161                 if(placed.count(i->side) == 0 && positions_taken.count(i->pos) == 0) {
 162                         placed.insert(i->side);
 163                         positions_taken.insert(i->pos);
 164                         board_.map_->set_starting_position(i->side,i->pos);
 165                         LOG_NG << "placing side " << i->side << " at " << i->pos << std::endl;
 166                 }
 167         }
 168 }

 170 void game_state::init(const config& level, play_controller & pc)
 171 {
 172         events_manager_->read_scenario(level);
 173         gui2::dialogs::loading_screen::progress("init teams");
 174         if (level["modify_placing"].to_bool()) {
 175                 LOG_NG << "modifying placing..." << std::endl;
 176                 place_sides_in_preferred_locations(level);
 177         }
 178 
 179         LOG_NG << "initialized time of day regions... "    << (SDL_GetTicks() - pc.ticks()) << std::endl;
 180         for (const config &t : level.child_range("time_area")) {
 181                 tod_manager_.add_time_area(board_.map(),t);
 182         }
 183 
 184         LOG_NG << "initialized teams... "    << (SDL_GetTicks() - pc.ticks()) << std::endl;
 185 
 186         board_.teams_.resize(level.child_count("side"));
 187 
 188         std::vector<team_builder_ptr> team_builders;
 189 
 190         int team_num = 0;
 191         for (const config &side : level.child_range("side"))
 192         {
 193                 if (first_human_team_ == -1) {
 194                         const std::string &controller = side["controller"];
 195                         if (controller == "human" && side["is_local"].to_bool(true)) {
 196                                 first_human_team_ = team_num;
 197                         }
 198                 }
 199                 ++team_num;
 200                 team_builder_ptr tb_ptr = create_team_builder(side,
 201                         board_.teams_, level, board_, team_num);
 202                 build_team_stage_one(tb_ptr);
 203                 team_builders.push_back(tb_ptr);
 204         }
 205         //Initilize the lua kernel before the units are created.
 206         lua_kernel_->initialize(level);
 207 
 208         {
 209                 //sync traits of start units and the random start time.
 210                 randomness::set_random_determinstic deterministic(gamedata_.rng());
 211 
 212                 tod_manager_.resolve_random(*randomness::generator);
 213 
 214                 for(team_builder_ptr tb_ptr : team_builders)
 215                 {
 216                         build_team_stage_two(tb_ptr);
 217                 }
 218                 for(size_t i = 0; i < board_.teams_.size(); i++) {
 219                         // Labels from players in your ignore list default to hidden
 220                         if(preferences::is_ignored(board_.teams_[i].current_player())) {
 221                                 std::string label_cat = "side:" + std::to_string(i + 1);
 222                                 board_.hidden_label_categories_ref().push_back(label_cat);
 223                         }
 224                 }
 225         }
 226 }

 233 void game_state::write(config& cfg) const
 234 {
 235         cfg["init_side_done"] = init_side_done_;
 236         if(gamedata_.phase() == game_data::PLAY) {
 237                 cfg["playing_team"] = player_number_ - 1;
 238         }
 239         cfg["server_request_number"] = server_request_number_;
 240         //Call the lua save_game functions
 241         lua_kernel_->save_game(cfg);
 242 
 243         //Write the game events.
 244         events_manager_->write_events(cfg);
 245 
 246         //Write the map, unit_map, and teams info
 247         board_.write_config(cfg);
 248 
 249         //Write the tod manager, and time areas
 250         cfg.merge_with(tod_manager_.to_config());
 251 
 252         //write out the current state of the map
 253         cfg.merge_with(pathfind_manager_->to_config());
 254 
 255         //Write the game data, including wml vars
 256         gamedata_.write_snapshot(cfg);
 257 
 258         // Preserve the undo stack so that fog/shroud clearing is kept accurate.
 259         undo_stack_->write(cfg.add_child("undo_stack"));
 260 
 261         if(end_level_data_.get_ptr() != nullptr) {
 262                 end_level_data_->write(cfg.add_child("end_level_data"));
 263         }
 264 }

 266 namespace {
 267         struct castle_cost_calculator : pathfind::cost_calculator
 268         {
 269                 castle_cost_calculator(const gamemap& map, const team & view_team) :
 270                         map_(map),
 271                         viewer_(view_team),
 272                         use_shroud_(view_team.uses_shroud())
 273                 {}
 274 
 275                 virtual double cost(const map_location& loc, const double) const
 276                 {
 277                         if(!map_.is_castle(loc))
 278                                 return 10000;
 279 
 280                         if ( use_shroud_ && viewer_.shrouded(loc) )
 281                                 return 10000;
 282 
 283                         return 1;
 284                 }
 285 
 286         private:
 287                 const gamemap& map_;
 288                 const team& viewer_;
 289                 const bool use_shroud_; // Allows faster checks when shroud is disabled.
 290         };

 294 /**
 295  * Checks to see if a leader at @a leader_loc could recruit somewhere.
 296  * This takes into account terrain, shroud (for side @a side), and the presence
 297  * of visible units.
 298  * The behavior for an invalid @a side is subject to change for future needs.
 299  */
 300 bool game_state::can_recruit_from(const map_location& leader_loc, int side) const
 301 {
 302         const gamemap& map = board_.map();
 303 
 304         if( !map.is_keep(leader_loc) )
 305                 return false;
 306 
 307         if ( side < 1  ||  board_.teams().size() < static_cast<size_t>(side) ) {
 308                 // Invalid side specified.
 309                 // Currently this cannot happen, but it could conceivably be used in
 310                 // the future to request that shroud and visibility be ignored. Until
 311                 // that comes to pass, just return.
 312                 return false;
 313         }
 314 
 315         return pathfind::find_vacant_tile(leader_loc, pathfind::VACANT_CASTLE, nullptr,
 316                                           &(board_.teams())[side-1])
 317                != map_location::null_location();
 318 }

 326 /**
 327  * Checks to see if a leader at @a leader_loc could recruit on @a recruit_loc.
 328  * This takes into account terrain, shroud (for side @a side), and whether or
 329  * not there is already a visible unit at recruit_loc.
 330  * The behavior for an invalid @a side is subject to change for future needs.
 331  */
 332 bool game_state::can_recruit_on(const map_location& leader_loc, const map_location& recruit_loc, int side) const
 333 {
 334         const gamemap& map = board_.map();
 335 
 336         if( !map.is_castle(recruit_loc) )
 337                 return false;
 338 
 339         if( !map.is_keep(leader_loc) )
 340                 return false;
 341 
 342         if ( side < 1  ||  board_.teams().size() < static_cast<size_t>(side) ) {
 343                 // Invalid side specified.
 344                 // Currently this cannot happen, but it could conceivably be used in
 345                 // the future to request that shroud and visibility be ignored. Until
 346                 // that comes to pass, just return.
 347                 return false;
 348         }
 349         const team & view_team = board_.get_team(side);
 350 
 351         if ( view_team.shrouded(recruit_loc) )
 352                 return false;
 353 
 354         if ( board_.has_visible_unit(recruit_loc, view_team) )
 355                 return false;
 356 
 357         castle_cost_calculator calc(map, view_team);
 358         // The limit computed in the third argument is more than enough for
 359         // any convex castle on the map. Strictly speaking it could be
 360         // reduced to sqrt(map.w()**2 + map.h()**2).
 361         pathfind::plain_route rt =
 362                 pathfind::a_star_search(leader_loc, recruit_loc, map.w()+map.h(), calc,
 363                                         map.w(), map.h());
 364         return !rt.steps.empty();
 365 }

 372 bool game_state::side_can_recruit_on(int side, map_location hex) const
 373 {
 374         unit_map::const_iterator leader = board_.units().find(hex);
 375         if ( leader != board_.units().end() ) {
 376                 return leader->can_recruit() && leader->side() == side && can_recruit_from(*leader);
 377         } else {
 378                 // Look for a leader who can recruit on last_hex.
 379                 for ( leader = board_.units().begin(); leader != board_.units().end(); ++leader) {
 380                         if ( leader->can_recruit() && leader->side() == side && can_recruit_on(*leader, hex) ) {
 381                                 return true;
 382                         }
 383                 }
 384         }
 385         // No leader found who can recruit at last_hex.
 386         return false;
 387 }
