   1 #!/bin/sh
   2 #
   3 #  Check the format of PNGs
   4 #
   5 #  With --verbose switch, it reports every file and its non-RGBA format individually
   6 #
   7 
   8 rgb=0
   9 rgba=0
  10 gray=0
  11 graya=0
  12 index=0
  13 indexa=0
  14 other=0
  15 filelist=""
  16 verbose="no"
  17 
  18 while [ "${1}" != "" ]; do
  19     if [ "${1}" = "--verbose" ] || [ "${1}" = "-v" ]; then
  20         verbose="yes"
  21         shift
  22     elif [ -d "${1}" ]; then
  23         filelist="${filelist} `find ${1} -iname "*.png"`"
  24         shift
  25     elif [ -f "${1}" ]; then
  26         filelist="${filelist} ${1}"
  27         shift
  28     else
  29         echo "Argument ${1} is not a known switch nor is it a directory or file. Exiting."
  30         exit 1
  31     fi
  32 done

  34 report_file()
  35 {
  36     if [ "$verbose" = "yes" ]; then
  37         echo "File $1 is in format $2"
  38     fi
  39 }
  40 
  41 for i in $filelist; do
  42     result=`pngcheck $i`
  43     if [ $? -ne 0 ]; then
  44         echo "Failure executing pngcheck. Exiting."
  45         exit 1
  46     fi
  47     if echo $result|grep 'RGB+alpha'>/dev/null
  48     then
  49         rgba=$(($rgba+1))
  50     elif echo $result|grep 'RGB'>/dev/null
  51     then
  52         rgb=$(($rgb+1))
  53         report_file $i 'RGB'
  54     elif echo $result|grep 'grayscale+alpha'>/dev/null
  55     then
  56         graya=$(($graya+1))
  57         report_file $i 'grayscale+alpha'
  58     elif echo $result|grep 'grayscale'>/dev/null
  59     then
  60         grey=$(($grey+1))
  61         report_file $i 'grayscale'
  62     elif echo $result|grep 'colormap+trns'>/dev/null
  63     then
  64         indexa=$(($indexa+1))
  65         report_file $i 'palette+trns (alpha)'
  66     elif echo $result|grep 'colormap'>/dev/null
  67     then
  68         index=$(($index+1))
  69         report_file $i 'palette'
  70     else
  71         other=$(($other+1))
  72         report_file $i "`echo $result | sed -e 's/^.*(//;s/).*$//;'`"
  73     echo $result
  74     fi
  75 done
  76 echo "RGBA: $rgba RGB: $rgb Gray+A: $graya Gray: $gray Index+A: $indexa Index: $index Other: $other"
