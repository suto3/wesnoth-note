  87 static void copy_persistent(const config& src, config& dst)
  88 {
  89         static const std::set<std::string> attrs {
  90                         "description",
  91                         "name",
  92                         "victory_when_enemies_defeated",
  93                         "remove_from_carryover_on_defeat",
  94                         "disallow_recall",
  95                         "experience_modifier",
  96                         "require_scenario"};
  97 
  98         static const std::set<std::string> tags {
  99                         "terrain_graphics",
 100                         "lua"};
 101 
 102         for (const std::string& attr : attrs)
 103         {
 104                 dst[attr] = src[attr];
 105         }
 106 
 107         for (const std::string& tag : tags)
 108         {
 109                 dst.append_children(src, tag);
 110         }
 111 }

 265                         gui_->set_team(gamestate().first_human_team_);

 267                 else if(is_observer()) {
 268                         // Find first team that is allowed to be observed.

 273                                 if (!t.get_disallow_observers())

 277                         }
 278                 }

 282                 //loadscreen_manager->reset();
 283                 gamestate().gamedata_.set_phase(game_data::PRELOAD);
 284                 gamestate().lua_kernel_->load_game(level);
 285 
 286                 plugins_context_.reset(new plugins_context("Game"));
 287                 plugins_context_->set_callback("save_game", [this](const config& cfg) { save_game_auto(cfg["filename"]); }, true);
 288                 plugins_context_->set_callback("save_replay", [this](const config& cfg) { save_replay_auto(cfg["filename"]); }, true);
 289                 plugins_context_->set_callback("quit", throw_end_level(), false);
 290                 plugins_context_->set_accessor_string("scenario_name", [this](config) { return get_scenario_name(); });
 291         });

 350 void play_controller::fire_prestart()
 351 {
 352         // pre-start events must be executed before any GUI operation,
 353         // as those may cause the display to be refreshed.
 354         update_locker lock_display(gui_->video());
 355         gamestate().gamedata_.set_phase(game_data::PRESTART);
 356 
 357         // Fire these right before prestart events, to catch only the units sides
 358         // have started with.
 359         for (const unit& u : gamestate().board_.units()) {
 360                 pump().fire("unit_placed", map_location(u.get_location()));
 361         }
 362 
 363         pump().fire("prestart");
 364         // prestart event may modify start turn with WML, reflect any changes.
 365         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 366 }

 368 void play_controller::fire_start()
 369 {
 370         gamestate().gamedata_.set_phase(game_data::START);
 371         pump().fire("start");
 372         // start event may modify start turn with WML, reflect any changes.
 373         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 374         check_objectives();
 375         // prestart and start events may modify the initial gold amount,
 376         // reflect any changes.
 377         for (team& tm : gamestate().board_.teams_)
 378         {
 379                 tm.set_start_gold(tm.gold());
 380         }
 381         gamestate_->init_side_done() = false;
 382         gamestate().gamedata_.set_phase(game_data::PLAY);
 383 }

 407 void play_controller::maybe_do_init_side()
 408 {
 409         //
 410         // We do side init only if not done yet for a local side when we are not replaying.
 411         // For all other sides it is recorded in replay and replay handler has to handle
 412         // calling do_init_side() functions.
 413         //
 414         if (gamestate_->init_side_done()) {
 415                 // We already executed do_init_side this can for example happe if we reload a game,
 416                 // but also if we changed control of a side during it's turn
 417                 return;
 418         }
 419         if (!current_team().is_local()) {
 420                 // We are in a mp game and execute do_init_side as soon as we receive [init_side] from the current player
 421                 // (see replay.cpp)
 422                 return;
 423         }
 424 
 425         if (is_replay()) {
 426                 // We are in a replay and execute do_init_side as soon as we reach the next [init_side] in the replay data
 427                 // (see replay.cpp) 
 428                 return;
 429         }
 430 
 431         resources::recorder->init_side();
 432         do_init_side();
 433 }

 435 void play_controller::do_init_side()
 436 {
 437         set_scontext_synced sync;
 438         log_scope("player turn");
 439         // In case we might end up calling sync:network during the side turn events,
 440         // and we don't want do_init_side to be called when a player drops.
 441         gamestate_->init_side_done() = true;
 442         init_side_done_now_ = true;
 443 
 444         const std::string turn_num = std::to_string(turn());
 445         const std::string side_num = std::to_string(current_side());
 446 
 447         gamestate().gamedata_.get_variable("side_number") = current_side();
 448 
 449         // We might have skipped some sides because they were empty so it is not enough to check for side_num==1
 450         if(!gamestate().tod_manager_.has_turn_event_fired())
 451         {
 452                 pump().fire("turn_" + turn_num);
 453                 pump().fire("new_turn");
 454                 gamestate().tod_manager_.turn_event_fired();
 455         }
 456 
 457         pump().fire("side_turn");
 458         pump().fire("side_" + side_num + "_turn");
 459         pump().fire("side_turn_" + turn_num);
 460         pump().fire("side_" + side_num + "_turn_" + turn_num);
 461 
 462         // We want to work out if units for this player should get healed,
 463         // and the player should get income now.
 464         // Healing/income happen if it's not the first turn of processing,
 465         // or if we are loading a game.
 466         if (turn() > 1) {
 467                 gamestate().board_.new_turn(current_side());
 468                 current_team().new_turn();
 469 
 470                 // If the expense is less than the number of villages owned
 471                 // times the village support capacity,
 472                 // then we don't have to pay anything at all
 473                 int expense = gamestate().board_.side_upkeep(current_side()) -
 474                         current_team().support();
 475                 if(expense > 0) {
 476                         current_team().spend_gold(expense);
 477                 }
 478 
 479                 calculate_healing(current_side(), !is_skipping_replay());
 480         }
 481 
 482         // Prepare the undo stack.
 483         undo_stack().new_side_turn(current_side());
 484 
 485         pump().fire("turn_refresh");
 486         pump().fire("side_" + side_num + "_turn_refresh");
 487         pump().fire("turn_" + turn_num + "_refresh");
 488         pump().fire("side_" + side_num + "_turn_" + turn_num + "_refresh");
 489 
 490         // Make sure vision is accurate.
 491         actions::clear_shroud(current_side(), true);
 492         init_side_end();
 493         check_victory();
 494         sync.do_final_checkup();
 495 }

 497 void play_controller::init_side_end()
 498 {
 499         const time_of_day& tod = gamestate().tod_manager_.get_time_of_day();
 500 
 501         if (current_side() == 1 || !init_side_done_now_)
 502                 sound::play_sound(tod.sounds, sound::SOUND_SOURCES);
 503 
 504         if (!is_skipping_replay()){
 505                 gui_->invalidate_all();
 506         }
 507 
 508         if (!is_skipping_replay() && current_team().get_scroll_to_leader() && !map_start_.valid()){
 509                 gui_->scroll_to_leader(current_side(), game_display::ONSCREEN,false);
 510         }
 511         map_start_ = map_location();
 512         whiteboard_manager_->on_init_side();
 513 }

 533 void play_controller::finish_side_turn()
 534 {
 535         whiteboard_manager_->on_finish_side_turn(current_side());
 536 
 537         { //Block for set_scontext_synced
 538                 set_scontext_synced sync(1);
 539                 // Ending the turn commits all moves.
 540                 undo_stack().clear();
 541                 gamestate().board_.end_turn(current_side());
 542                 const std::string turn_num = std::to_string(turn());
 543                 const std::string side_num = std::to_string(current_side());
 544 
 545                 // Clear shroud, in case units had been slowed for the turn.
 546                 actions::clear_shroud(current_side());
 547 
 548                 pump().fire("side_turn_end");
 549                 pump().fire("side_"+ side_num + "_turn_end");
 550                 pump().fire("side_turn_" + turn_num + "_end");
 551                 pump().fire("side_" + side_num + "_turn_" + turn_num + "_end");
 552                 // This is where we refog, after all of a side's events are done.
 553                 actions::recalculate_fog(current_side());
 554                 check_victory();
 555                 sync.do_final_checkup();
 556         }
 557 
 558         mouse_handler_.deselect_hex();
 559         resources::gameboard->unit_id_manager().reset_fake();
 560         gamestate_->init_side_done() = false;
 561 }

 693 /// @returns: the number n in [min, min+mod ) so that (n - num) is a multiple of mod.
 694 static int modulo(int num, int mod, int min)
 695 {
 696         assert(mod > 0);
 697         int n = (num - min) % mod;
 698         if (n < 0)
 699                 n += mod;
 700         //n is now in [0, mod)
 701         n = n + min;
 702         return n;
 703         // the folowing properties are easy to verify:
 704         //  1) For all m: modulo(num, mod, min) == modulo(num + mod*m, mod, min)
 705         //  2) For all 0 <= m < mod: modulo(min + m, mod, min) == min + m
 706 }

 782 void play_controller::process_keyup_event(const SDL_Event& event)
 783 {
 784         // If the user has pressed 1 through 9, we want to show
 785         // how far the unit can move in that many turns
 786         if(event.key.keysym.sym >= '1' && event.key.keysym.sym <= '9') {
 787                 const int new_path_turns = (event.type == SDL_KEYDOWN) ?
 788                                            event.key.keysym.sym - '1' : 0;
 789 
 790                 if(new_path_turns != mouse_handler_.get_path_turns()) {
 791                         mouse_handler_.set_path_turns(new_path_turns);
 792 
 793                         const unit_map::iterator u = mouse_handler_.selected_unit();
 794 
 795                         if(u.valid()) {
 796                                 // if it's not the unit's turn, we reset its moves
 797                                 unit_movement_resetter move_reset(*u, u->side() != current_side());
 798 
 799                                 mouse_handler_.set_current_paths(pathfind::paths(*u, false,
 800                                                        true, gamestate().board_.teams_[gui_->viewing_team()],
 801                                                        mouse_handler_.get_path_turns()));
 802 
 803                                 gui_->highlight_reach(mouse_handler_.current_paths());
 804                         } else {
 805                                 mouse_handler_.select_hex(mouse_handler_.get_selected_hex(), false, false, false);
 806                         }
 807 
 808                 }
 809         } else if (event.key.keysym.sym == SDLK_TAB) {
 810                 CKey keys;
 811                 if (!keys[SDLK_TAB]) {
 812                         whiteboard_manager_->set_invert_behavior(false);
 813                 }
 814         }

 918 void play_controller::check_victory()
 919 {
 920         if(linger_)
 921         {
 922                 return;
 923         }
 924 
 925         if (is_regular_game_end()) {
 926                 return;
 927         }
 928         bool continue_level, found_player, found_network_player, invalidate_all;
 929         std::set<unsigned> not_defeated;
 930 
 931         gamestate().board_.check_victory(continue_level, found_player, found_network_player, invalidate_all, not_defeated, remove_from_carryover_on_defeat_);
 932 
 933         if (invalidate_all) {
 934                 gui_->invalidate_all();
 935         }
 936 
 937         if (continue_level) {
 938                 return ;
 939         }
 940 
 941         if (found_player || found_network_player) {
 942                 pump().fire("enemies_defeated");
 943                 if (is_regular_game_end()) {
 944                         return;
 945                 }
 946         }
 947 
 948         DBG_EE << "victory_when_enemies_defeated: " << victory_when_enemies_defeated_ << std::endl;
 949         DBG_EE << "found_player: " << found_player << std::endl;
 950         DBG_EE << "found_network_player: " << found_network_player << std::endl;
 951 
 952         if (!victory_when_enemies_defeated_ && (found_player || found_network_player)) {
 953                 // This level has asked not to be ended by this condition.
 954                 return;
 955         }
 956 
 957         if (gui_->video().non_interactive()) {
 958                 LOG_AIT << "winner: ";
 959                 for (unsigned l : not_defeated) {
 960                         std::string ai = ai::manager::get_active_ai_identifier_for_side(l);
 961                         if (ai.empty()) ai = "default ai";
 962                         LOG_AIT << l << " (using " << ai << ") ";
 963                 }
 964                 LOG_AIT << std::endl;
 965                 ai_testing::log_victory(not_defeated);
 966         }
 967 
 968         DBG_EE << "throwing end level exception..." << std::endl;
 969         //Also proceed to the next scenario when another player survived.
 970         end_level_data el_data;
 971         el_data.proceed_to_next_level = found_player || found_network_player;
 972         el_data.is_victory = found_player;
 973         set_end_level_data(el_data);
 974 }

1136 void play_controller::play_side()
1137 {
1138         //check for team-specific items in the scenario
1139         gui_->parse_team_overlays();
1140         do {

1143                         save_blocker blocker;
1144                         maybe_do_init_side();
1145                         if(is_regular_game_end()) {
1146                                 return;
1147                         }
1148                 }
1149                 // This flag can be set by derived classes (in overridden functions).
1150                 player_type_changed_ = false;

1166 void play_controller::play_turn()
1167 {
1168         whiteboard_manager_->on_gamestate_change();
1169         gui_->new_turn();
1170         gui_->invalidate_game_status();
1171         events::raise_draw_event();
1172 
1173         LOG_NG << "turn: " << turn() << "\n";
1174 
1175         if(gui_->video().non_interactive()) {
1176                 LOG_AIT << "Turn " << turn() << ":" << std::endl;
1177         }
1178 
1179         for (; gamestate_->player_number_ <= int(gamestate().board_.teams().size()); ++gamestate_->player_number_)
1180         {
1181                 // If a side is empty skip over it.
1182                 if (current_team().is_empty()) {
1183                         continue;
1184                 }
1185                 init_side_begin();
1186                 if(gamestate_->init_side_done()) {
1187                         // This is the case in a reloaded game where the side was initialized before saving the game.
1188                         init_side_end();
1189                 }
1190 
1191                 ai_testing::log_turn_start(current_side());
1192                 play_side();
1193                 if(is_regular_game_end()) {
1194                         return;
1195                 }
1196                 finish_side_turn();
1197                 if(is_regular_game_end()) {
1198                         return;
1199                 }
1200                 if(gui_->video().non_interactive()) {
1201                         LOG_AIT << " Player " << current_side() << ": " <<
1202                                 current_team().villages().size() << " Villages" <<
1203                                 std::endl;
1204                         ai_testing::log_turn_end(current_side());
1205                 }
1206         }
1207         // If the loop exits due to the last team having been processed.
1208         gamestate_->player_number_ = gamestate().board_.teams().size();
1209 
1210         finish_turn();
1211 
1212         // Time has run out
1213         check_time_over();
1214 }

1216 void play_controller::check_time_over()
1217 {
1218         const bool time_left = gamestate().tod_manager_.next_turn(&gamestate().gamedata_);
1219 
1220         if(!time_left) {
1221                 LOG_NG << "firing time over event...\n";
1222                 set_scontext_synced_base sync;
