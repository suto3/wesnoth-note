  30 def expand_square_braces(path):
  31     """Expands the square brackets notation to normal file names.
  32 Yields the expanded file names, or the original path itself
  33 if no expansion could be performed"""
  34     if "[" not in path: # clearly, no expansion can be done in this case
  35         yield path
  36     else:
  37         substitutions = [] # a matrix that will hold all the substitutions to be performed
  38 
  39         for i, match in enumerate(re.finditer(r"\[(.*?)\]", path)): # stop on the first closed square braces, to allow multiple substitutions
  40             substitutions.append([]) # a new list which will host all the substitutions for the current expansion
  41             for token in match.group(1).split(","):
  42                 match_mult = re.match("(.+)\*(\d+)", token) # repeat syntax, eg. [melee*3]
  43                 if match_mult:
  44                     substitutions[i].extend([Substitution(match_mult.group(1), match.start(0), match.end(0))] *
  45                                          int(match_mult.group(2)))
  46                     continue
  47                 match_range = re.match("(\d+)~(\d+)", token) # range syntax, eg [1~4]
  48                 if match_range:
  49                     before, after = int(match_range.group(1)), int(match_range.group(2))
  50                     incr = 1 if before <= after else -1 # to allow iterating in reversed order, eg. [4~1]
  51                     substitutions[i].extend([Substitution(str(n), match.start(0), match.end(0)) for n in range(before, after + incr, incr)])
  52                     continue
  53                 substitutions[i].append(Substitution(token, match.start(0), match.end(0))) # no operator found
  54 
  55         # the purpose is to have all the subs "increasing" simultaneously
  56         # in other words, if a string has two square braces blocks, like [1~3] and [melee*2,ranged]
  57         # we want to expand them as (1,melee), (2,melee), (3,ranged)
  58         # for this reason we need to transpose the matrix
  59         substitutions = zip(*substitutions)
  60 
  61         for sub_array in substitutions:
  62             new_string = path
  63             for sub in reversed(sub_array): # to avoid creating "holes" in the strings
  64                 yield new_string[:sub.start] + sub.sub + new_string[sub.end:] # these are the expanded strings

 105 def comma_split(csstring, list=None, strip="r"):
 106     "Split a comma-separated string, and append the entries to a list if specified."
 107     vallist = [x.lstrip() for x in csstring.split(",") if x.lstrip()]
 108     # strip=: utils::split will remove trailing whitespace from items in comma-
 109     # separated lists but the wml-tags.lua split function only removes leading
 110     # whitespace. So two flags are offered to change default behavior: one to
 111     # lstrip() only, the other to warn about trailing whitespace.
 112     if 'w' in strip:
 113         for item in vallist:
 114             if re.search('\s$', item):
 115                 print('Trailing whitespace may be problematic: "%s" in "%s"' % (item, csstring))
 116     if 'l' not in strip:
 117         vallist = [x.rstrip() for x in vallist]
 118     if list is not None:
 119         list.extend(vallist)
 120     else:
 121         return vallist

 123 def parse_attribute(line):
 124     "Parse a WML key-value pair from a line."
 125     if '=' not in line or line.find("#") > -1 and line.find("#") < line.find("="):
 126         return None
 127     where = line.find("=")
 128     leader = line[:where]
 129     after = line[where+1:]
 130     after = after.lstrip()
 131     if re.search("\s#", after):
 132         where = len(re.split("\s+#", after)[0])
 133         value = after[:where]
 134         comment = after[where:]
 135     else:
 136         value = after.rstrip()
 137         comment = ""
 138     # Return four fields: stripped key, part of line before value,
 139     # value, trailing whitespace and comment.
 140     return (leader.strip(), leader+"=", string_strip(value), comment)

 257 def parse_macroref(start, line):
 258     brackdepth = parendepth = 0
 259     instring = False
 260     args = []
 261     arg = ""
 262     for i in range(start, len(line)):
 263         if instring:
 264             if line[i] == '"':
 265                 instring = False
 266             arg += line[i]
 267         elif line[i] == '"':
 268             instring = not instring
 269             arg += line[i]
 270         elif line[i] == "{":
 271             if brackdepth > 0:
 272                 arg += line[i]
 273             brackdepth += 1
 274         elif line[i] == "}":
 275             brackdepth -= 1
 276             if brackdepth == 0:
 277                 if not line[i-1].isspace():
 278                     arg = arg.strip()
 279                     if arg.startswith('"') and arg.endswith('"'):
 280                         arg = arg[1:-1].strip()
 281                     args.append(arg)
 282                     arg = ""
 283                 break
 284             else:
 285                 arg += line[i]
 286         elif line[i] == "(":
 287             parendepth += 1
 288         elif line[i] == ")":
 289             parendepth -= 1
 290         elif not line[i-1].isspace() and \
 291              line[i].isspace() and \
 292              brackdepth == 1 and \
 293              parendepth == 0:
 294             arg = arg.strip()
 295             if arg.startswith('"') and arg.endswith('"'):
 296                 arg = arg[1:-1].strip()
 297             args.append(arg)
 298             arg = ""
 299         elif not line[i].isspace() or parendepth > 0:
 300             arg += line[i]
 301     return (args, brackdepth, parendepth)

 303 def formaltype(f):
 304     # Deduce the expected type of the formal
 305     if f.startswith("_"):
 306         f = f[1:]
 307     if f == "SIDE" or f.endswith("_SIDE") or re.match("SIDE[0-9]", f):
 308         ftype = "side"
 309     elif f in ("SIDE", "X", "Y", "RED", "GREEN", "BLUE", "TURN", "PROB", "LAYER", "TIME", "DURATION") or f.endswith("NUMBER") or f.endswith("AMOUNT") or f.endswith("COST") or f.endswith("RADIUS") or f.endswith("_X") or f.endswith("_Y") or f.endswith("_INCREMENT") or f.endswith("_FACTOR") or f.endswith("_TIME") or f.endswith("_SIZE"):
 310         ftype = "numeric"
 311     elif f.endswith("PERCENTAGE"):
 312         ftype = "percentage"
 313     elif f in ("POSITION",) or f.endswith("_POSITION") or f == "BASE":
 314         ftype = "position"
 315     elif f.endswith("_SPAN"):
 316         ftype = "span"
 317     elif f == "SIDES" or f.endswith("_SIDES"):
 318         ftype = "alliance"
 319     elif f in ("RANGE",):
 320         ftype = "range"
 321     elif f in ("ALIGN",):
 322         ftype = "alignment"
 323     elif f in ("TYPES"):
 324         ftype = "types"
 325     elif f.startswith("ADJACENT") or f.startswith("TERRAINLIST") or f == "RESTRICTING":
 326         ftype = "terrain_pattern"
 327     elif f.startswith("TERRAIN") or f.endswith("TERRAIN"):
 328         ftype = "terrain_code"
 329     elif f in ("NAME", "NAMESPACE", "VAR", "IMAGESTEM", "ID", "FLAG", "BUILDER") or f.endswith("_NAME") or f.endswith("_ID") or f.endswith("_VAR") or f.endswith("_OVERLAY"):
 330         ftype = "name"
 331     elif f in ("ID_STRING", "NAME_STRING", "DESCRIPTION", "IPF"):
 332         ftype = "optional_string"
 333     elif f in ("STRING", "TYPE", "TEXT") or f.endswith("_STRING") or f.endswith("_TYPE") or f.endswith("_TEXT"):
 334         ftype = "string"
 335     elif f.endswith("IMAGE") or f == "PROFILE":
 336         ftype = "image"
 337     elif f.endswith("MUSIC",) or f.endswith("SOUND"):
 338         ftype = "sound"
 339     elif f.endswith("FILTER",):
 340         ftype = "filter"
 341     elif f == "WML" or f.endswith("_WML"):
 342         ftype = "wml"
 343     elif f in ("AFFIX", "POSTFIX", "ROTATION") or f.endswith("AFFIX"):
 344         ftype = "affix"
 345     # The regexp case avoids complaints about some wacky terrain macros.
 346     elif f.endswith("VALUE") or re.match("[ARS][0-9]", f):
 347         ftype = "any"
 348     else:
 349         ftype = None
 350     return ftype

 352 def actualtype(a):
 353     if a is None:
 354         return None
 355     # Deduce the type of the actual
 356     if a.isdigit() or a.startswith("-") and a[1:].isdigit():
 357         atype = "numeric"
 358     elif re.match(r"0\.[0-9]+\Z", a):
 359         atype = "percentage"
 360     elif re.match(r"-?[0-9]+,-?[0-9]+\Z", a):
 361         atype = "position"
 362     elif re.match(r"([0-9]+\-[0-9]+,?|[0-9]+,?)+\Z", a):
 363         atype = "span"
 364     elif a in ("melee", "ranged"):
 365         atype = "range"
 366     elif a in ("lawful", "neutral", "chaotic", "liminal"):
 367         atype = "alignment"
 368     elif a.startswith("{") or a.endswith("}") or a.startswith("$"):
 369         atype = None # Can't tell -- it's a macro expansion
 370     elif re.match(image_reference, a) or a == "unit_image":
 371         atype = "image"
 372     elif re.match(r"(\*|[A-Z][a-z]+)\^([A-Z][a-z\\|/]+\Z)?", a):
 373         atype = "terrain_code"
 374     elif a.endswith(".wav") or a.endswith(".ogg"):
 375         atype = "sound"
 376     elif a.startswith('"') and a.endswith('"') or (a.startswith("_") and a[1] not in string.ascii_lowercase):
 377         atype = "stringliteral"
 378     elif "=" in a:
 379         atype = "filter"
 380     elif re.match(r"[A-Z][a-z][a-z]?\Z", a):
 381         atype = "shortname"
 382     elif a == "":
 383         atype = "empty"
 384     elif not ' ' in a:
 385         atype = "name"
 386     else:
 387         atype = "string"
 388     return atype

 390 def argmatch(formals, actuals):
 391     if len(formals) != len(actuals):
 392         return False
 393     for (f, a) in zip(formals, actuals):
 394         # Here's the compatibility logic.  First, we catch the situations
 395         # in which a more restricted actual type matches a more general
 396         # formal one.  Then we have a fallback rule checking for type
 397         # equality or wildcarding.
 398         ftype = formaltype(f)
 399         atype = actualtype(a)
 400         if ftype == "any":
 401             pass
 402         elif (atype == "numeric" or a == "global") and ftype == "side":
 403             pass
 404         elif atype in ("filter", "empty") and ftype == "wml":
 405             pass
 406         elif atype in ("numeric", "position") and ftype == "span":
 407             pass
 408         elif atype in ("shortname", "name", "empty", "stringliteral") and ftype == "affix":
 409             pass
 410         elif atype in ("shortname", "name", "stringliteral") and ftype == "string":
 411             pass
 412         elif atype in ("shortname", "name", "string", "stringliteral", "empty") and ftype == "optional_string":
 413             pass
 414         elif atype in ("shortname",) and ftype == "terrain_code":
 415             pass
 416         elif atype in ("numeric", "position", "span", "empty") and ftype == "alliance":
 417             pass
 418         elif atype in ("terrain_code", "shortname", "name") and ftype == "terrain_pattern":
 419             pass
 420         elif atype in ("string", "shortname", "name") and ftype == "types":
 421             pass
 422         elif atype in ("numeric", "percentage") and ftype == "percentage":
 423             pass
 424         elif atype == "range" and ftype == "name":
 425             pass
 426         elif atype != ftype and ftype is not None and atype is not None:
 427             return False
 428     return True

 434 class Reference:
 435     "Describes a location by file and line."
 436     def __init__(self, namespace, filename, lineno=None, docstring=None, args=None):
 437         self.namespace = namespace
 438         self.filename = filename
 439         self.lineno = lineno
 440         self.docstring = docstring
 441         self.args = args
 442         self.references = collections.defaultdict(list)
 443         self.undef = None
 444 
 445     def append(self, fn, n, a=None):
 446         self.references[fn].append((n, a))
 447 
 448     def dump_references(self):
 449         "Dump all known references to this definition."
 450         for (file, refs) in self.references.items():
 451             print("    %s: %s" % (file, repr([x[0] for x in refs])[1:-1]))
 452 
 453     def __eq__(self, other):
 454         return self.filename == other.filename and self.lineno == other.lineno
 455 
 456     def __gt__(self, other):
 457         # Major sort by file, minor by line number.  This presumes that the
 458         # files correspond to coherent topics and gives us control of the
 459         # sequence.
 460         if self.filename == other.filename:
 461             return self.lineno > other.lineno
 462         else:
 463             return self.filename > other.filename

 465     def mismatches(self):
 466         copy = Reference(self.namespace, self.filename, self.lineno, self.docstring, self.args)
 467         copy.undef = self.undef
 468         for filename in self.references:
 469             mis = [(ln,a) for (ln,a) in self.references[filename] if a is not None and not argmatch(self.args, a)]
 470             if mis:
 471                 copy.references[filename] = mis
 472         return copy
 473     def __str__(self):
 474         if self.lineno:
 475             return '"%s", line %d' % (self.filename, self.lineno)
 476         else:
 477             return self.filename

 480 class CrossRef:
 481     macro_reference = re.compile(r"\{([A-Z_][A-Za-z0-9_:]*)(?!\.)\b")
 482     file_reference = re.compile(r"[A-Za-z0-9{}.][A-Za-z0-9_/+{}.@\-\[\],~\*]*(" + "|".join(resource_extensions) + ")(?=(~.*)?)")
 483     tag_parse = re.compile("\s*([a-z_]+)\s*=(.*)")
 484     def mark_matching_resources(self, pattern, fn, n):
 485         "Mark all definitions matching a specified pattern with a reference."
 486         pattern = pattern.replace("+", r"\+")
 487         pattern = os.sep + pattern + "$"
 488         if os.sep == "\\":
 489             pattern = pattern.replace("\\", "\\\\")
 490         try:
 491             pattern = re.compile(pattern)
 492         except sre_constants.error:
 493             print("wmlscope: confused by %s" % pattern, file=sys.stderr)
 494             return None
 495         key = None
 496         for trial in self.fileref:
 497             if pattern.search(trial) and self.visible_from(trial, fn, n):
 498                 key = trial
 499                 self.fileref[key].append(fn, n)
 500         return key
 501     def visible_from(self, defn, fn, n):
 502         "Is specified definition visible from the specified file and line?"
 503         if isinstance(defn, str):
 504             defn = self.fileref[defn]
 505         if defn.undef is not None:
 506             # Local macros are only visible in the file where they were defined
 507             return defn.filename == fn and n >= defn.lineno and n <= defn.undef
 508         if self.exports(defn.namespace):
 509             # Macros and resources in subtrees with export=yes are global
 510             return True
 511         elif not self.filelist.neighbors(defn.filename, fn):
 512             # Otherwise, must be in the same subtree.
 513             return False
 514         else:
 515             # If the two files are in the same subtree, assume visibility.
 516             # This doesn't match the actual preprocessor semantics.
 517             # It means any macro without an undef is visible anywhere in the
 518             # same argument directory.
 519             #
 520             # We can't do better than this without a lot of hairy graph-
 521             # coloring logic to simulate include path interpretation.
 522             # If that logic ever gets built, it will go here.
 523             return True
 524     def scan_for_definitions(self, namespace, filename):
 525         ignoreflag = False
 526         conditionalsflag = False
 527         with codecs.open(filename, "r", "utf8") as dfp:
 528             state = "outside"
 529             latch_unit = in_base_unit = in_theme = False
 530             for (n, line) in enumerate(dfp):
 531                 if self.warnlevel > 1:
 532                     print(repr(line)[1:-1])
 533                 if line.strip().startswith("#textdomain"):
 534                     continue
 535                 m = re.search("# *wmlscope: warnlevel ([0-9]*)", line)
 536                 if m:
 537                     self.warnlevel = int(m.group(1))
 538                     print('"%s", line %d: warnlevel set to %d (definition-gathering pass)' \
 539                          % (filename, n+1, self.warnlevel))
 540                     continue
 541                 m = re.search("# *wmlscope: set *([^=]*)=(.*)", line)
 542                 if m:
 543                     prop = m.group(1).strip()
 544                     value = m.group(2).strip()
 545                     if namespace not in self.properties:
 546                         self.properties[namespace] = {}
 547                     self.properties[namespace][prop] = value
 548                 m = re.search("# *wmlscope: prune (.*)", line)
 549                 if m:
 550                     name = m.group(1)
 551                     if self.warnlevel >= 2:
 552                         print('"%s", line %d: pruning definitions of %s' \
 553                               % (filename, n+1, name ))
 554                     if name not in self.xref:
 555                         print("wmlscope: can't prune undefined macro %s" % name, file=sys.stderr)
 556                     else:
 557                         self.xref[name] = self.xref[name][:1]
 558                     continue
 559                 if "# wmlscope: start conditionals" in line:
 560                     if self.warnlevel > 1:
 561                         print('"%s", line %d: starting conditionals' \
 562                               % (filename, n+1))
 563                     conditionalsflag = True
 564                 elif "# wmlscope: stop conditionals" in line:
 565                     if self.warnlevel > 1:
 566                         print('"%s", line %d: stopping conditionals' \
 567                               % (filename, n+1))
 568                     conditionalsflag = False
 569                 if "# wmlscope: start ignoring" in line:
 570                     if self.warnlevel > 1:
 571                         print('"%s", line %d: starting ignoring (definition pass)' \
 572                               % (filename, n+1))
 573                     ignoreflag = True
 574                 elif "# wmlscope: stop ignoring" in line:
 575                     if self.warnlevel > 1:
 576                         print('"%s", line %d: stopping ignoring (definition pass)' \
 577                               % (filename, n+1))
 578                     ignoreflag = False
 579                 elif ignoreflag:
 580                     continue
 581                 if line.strip().startswith("#define"):
 582                     tokens = line.split()
 583                     if len(tokens) < 2:
 584                         print('"%s", line %d: malformed #define' \
 585                               % (filename, n+1), file=sys.stderr)
 586                     else:
 587                         name = tokens[1]
 588                         here = Reference(namespace, filename, n+1, line, args=tokens[2:])
 589                         here.hash = hashlib.md5()
 590                         here.docstring = line.lstrip()[8:] # Strip off #define_
 591                         state = "macro_header"
 592                     continue
 593                 elif state != 'outside' and line.strip().endswith("#enddef"):
 594                     here.hash.update(line.encode("utf8"))
 595                     here.hash = here.hash.digest()
 596                     if name in self.xref:
 597                         for defn in self.xref[name]:
 598                             if not self.visible_from(defn, filename, n+1):
 599                                 continue
 600                             elif conditionalsflag:
 601                                 continue
 602                             elif defn.hash != here.hash:
 603                                 print("%s: overrides different %s definition at %s" \
 604                                         % (here, name, defn), file=sys.stderr)
 605                             elif self.warnlevel > 0:
 606                                 print("%s: duplicates %s definition at %s" \
 607                                         % (here, name, defn), file=sys.stderr)
 608                     if name not in self.xref:
 609                         self.xref[name] = []
 610                     self.xref[name].append(here)
 611                     state = "outside"
 612                 elif state == "macro_header" and line.strip() and line.strip()[0] != "#":
 613                     state = "macro_body"
 614                 if state == "macro_header":
 615                     # Ignore macro header commends that are pragmas
 616                     if "wmlscope" not in line and "wmllint:" not in line:
 617                         here.docstring += line.lstrip()[1:]
 618                 if state in ("macro_header", "macro_body"):
 619                     here.hash.update(line.encode("utf8"))
 620                 elif line.strip().startswith("#undef"):
 621                     tokens = line.split()
 622                     name = tokens[1]
 623                     if name in self.xref and self.xref[name]:
 624                         self.xref[name][-1].undef = n+1
 625                     else:
 626                         print("%s: unbalanced #undef on %s" \
 627                               % (Reference(namespace, filename, n+1), name))
 628                 if state == 'outside':
 629                     if '[unit_type]' in line:
 630                         latch_unit = True
 631                     elif '[/unit_type]' in line:
 632                         latch_unit = False
 633                     elif '[base_unit]' in line:
 634                         in_base_unit = True
 635                     elif '[/base_unit]' in line:
 636                         in_base_unit = False
 637                     elif '[theme]' in line:
 638                         in_theme = True
 639                     elif '[/theme]' in line:
 640                         in_theme = False
 641                     elif latch_unit and not in_base_unit and not in_theme and "id" in line:
 642                         m = CrossRef.tag_parse.search(line)
 643                         if m and m.group(1) == "id":
 644                             uid = m.group(2)
 645                             if uid not in self.unit_ids:
 646                                 self.unit_ids[uid] = []
 647                             self.unit_ids[uid].append(Reference(namespace, filename, n+1))
 648                             latch_unit= False
 649     def __init__(self, dirpath=[], exclude="", warnlevel=0, progress=False):
 650         "Build cross-reference object from the specified filelist."
 651         self.filelist = Forest(dirpath, exclude)
 652         self.dirpath = [x for x in dirpath if not re.search(exclude, x)]
 653         self.warnlevel = warnlevel
 654         self.xref = {}
 655         self.fileref = {}
 656         self.noxref = False
 657         self.properties = {}
 658         self.unit_ids = {}
 659         all_in = []
 660         if self.warnlevel >=2 or progress:
 661             print("*** Beginning definition-gathering pass...")
 662         for (namespace, filename) in self.filelist.generator():
 663             all_in.append((namespace, filename))
 664             if self.warnlevel > 1:
 665                 print(filename + ":")
 666             if progress:
 667                 print(filename)
 668             if isresource(filename):
 669                 self.fileref[filename] = Reference(namespace, filename)
 670             elif iswml(filename):
 671                 # It's a WML file, scan for macro definitions
 672                 self.scan_for_definitions(namespace, filename)
 673             elif filename.endswith(".def"):
 674                 # It's a list of names to be considered defined
 675                 self.noxref = True
 676                 with codecs.open(filename, "r", "utf8") as dfp:
 677                     for line in dfp:
 678                         self.xref[line.strip()] = True
 679         # Next, decorate definitions with all references from the filelist.
 680         self.unresolved = []
 681         self.missing = []
 682         formals = []
 683         state = "outside"
 684         if self.warnlevel >=2 or progress:
 685             print("*** Beginning reference-gathering pass...")
 686         for (ns, fn) in all_in:
 687             if progress:
 688                 print(fn)
 689             if iswml(fn):
 690                 with codecs.open(fn, "r", "utf8") as rfp:
 691                     attack_name = None
 692                     beneath = 0
 693                     ignoreflag = False
 694                     for (n, line) in enumerate(rfp):
 695                         if line.strip().startswith("#define"):
 696                             formals = line.strip().split()[2:]
 697                         elif line.startswith("#enddef"):
 698                             formals = []
 699                         comment = ""
 700                         if '#' in line:
 701                             if "# wmlscope: start ignoring" in line:
 702                                 if self.warnlevel > 1:
 703                                     print('"%s", line %d: starting ignoring (reference pass)' \
 704                                           % (fn, n+1))
 705                                 ignoreflag = True
 706                             elif "# wmlscope: stop ignoring" in line:
 707                                 if self.warnlevel > 1:
 708                                     print('"%s", line %d: stopping ignoring (reference pass)' \
 709                                           % (fn, n+1))
 710                                 ignoreflag = False
 711                             m = re.search("# *wmlscope: self.warnlevel ([0-9]*)", line)
 712                             if m:
 713                                 self.warnlevel = int(m.group(1))
 714                                 print('"%s", line %d: self.warnlevel set to %d (reference-gathering pass)' \
 715                                      % (fn, n+1, self.warnlevel))
 716                                 continue
 717                             fields = line.split('#')
 718                             line = fields[0]
 719                             if len(fields) > 1:
 720                                 comment = fields[1]
 721                         if ignoreflag or not line:
 722                             continue
 723                         # Find references to macros
 724                         for match in re.finditer(CrossRef.macro_reference, line):
 725                             name = match.group(1)
 726                             candidates = []
 727                             if self.warnlevel >=2:
 728                                 print('"%s", line %d: seeking definition of %s' \
 729                                       % (fn, n+1, name))
 730                             if name in formals:
 731                                 continue
 732                             elif name in self.xref:
 733                                 # Count the number of actual arguments.
 734                                 # Set args to None if the call doesn't
 735                                 # close on this line
 736                                 (args, brackdepth, parendepth) = parse_macroref(match.start(0), line)
 737                                 if brackdepth > 0 or parendepth > 0:
 738                                     args = None
 739                                 else:
 740                                     args.pop(0)
 741                                 #if args:
 742                                 #    print('"%s", line %d: args of %s is %s' \
 743                                 #          % (fn, n+1, name, args))
 744                                 # Figure out which macros might resolve this
 745                                 for defn in self.xref[name]:
 746                                     if self.visible_from(defn, fn, n+1):
 747                                         defn.append(fn, n+1, args)
 748                                         candidates.append(str(defn))
 749                                 if len(candidates) > 1:
 750                                     print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns, fn, n), name, "; ".join(candidates)))
 751                             if len(candidates) == 0:
 752                                 self.unresolved.append((name,Reference(ns,fn,n+1)))
 753                         # Don't be fooled by HTML image references in help strings.
 754                         if "<img>" in line:
 755                             continue
 756                         # Find references to resource files
 757                         for match in re.finditer(CrossRef.file_reference, line):
 758                             for name in expand_square_braces(match.group(0)):
 759                                 # Catches maps that look like macro names.
 760                                 if (name.endswith(".map") or name.endswith(".mask")) and name[0] == '{':
 761                                     name = name[1:]
 762                                 if os.sep == "\\":
 763                                     name = name.replace("/", "\\")
 764                                 key = None
 765                                 # If name is already in our resource list, it's easy.
 766                                 if name in self.fileref and self.visible_from(name, fn, n):
 767                                     self.fileref[name].append(fn, n+1)
 768                                     continue
 769                                 # If the name contains substitutable parts, count
 770                                 # it as a reference to everything the substitutions
 771                                 # could potentially match.
 772                                 elif '{' in name or '@' in name:
 773                                     pattern = re.sub(r"(\{[^}]*\}|@R0|@V)", '.*', name)
 774                                     key = self.mark_matching_resources(pattern, fn,n+1)
 775                                     if key:
 776                                         self.fileref[key].append(fn, n+1)
 777                                 else:
 778                                     candidates = []
 779                                     for trial in self.fileref:
 780                                         if trial.endswith(os.sep + name) and self.visible_from(trial, fn, n):
 781                                             key = trial
 782                                             self.fileref[trial].append(fn, n+1)
 783                                             candidates.append(trial)
 784                                     if len(candidates) > 1:
 785                                         print("%s: more than one resource matching %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 786                                 if not key:
 787                                     self.missing.append((name, Reference(ns,fn,n+1)))
 788                         # Notice implicit references through attacks
 789                         if state == "outside":
 790                             have_icon = False
 791                             if "[attack]" in line:
 792                                 beneath = 0
 793                                 attack_name = default_icon = None
 794                                 have_icon = False
 795                             elif "name=" in line and not "no-icon" in comment:
 796                                 attack_name = line[line.find("name=")+5:].strip()
 797                                 default_icon = os.path.join("attacks", attack_name  + ".png")
 798                             elif "icon=" in line and beneath == 0:
 799                                 have_icon = True
 800                             elif "[/attack]" in line:
 801                                 if attack_name and not have_icon:
 802                                     candidates = []
 803                                     key = None
 804                                     for trial in self.fileref:
 805                                         if trial.endswith(os.sep + default_icon) and self.visible_from(trial, fn, n):
 806                                             key = trial
 807                                             self.fileref[trial].append(fn, n+1)
 808                                             candidates.append(trial)
 809                                     if len(candidates) > 1:
 810                                         print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 811                                 if not key:
 812                                     self.missing.append((default_icon, Reference(ns,fn,n+1)))
 813                             elif line.strip().startswith("[/"):
 814                                 beneath -= 1
 815                             elif line.strip().startswith("["):
 816                                 beneath += 1
 817         # Check whether each namespace has a defined export property
 818         for namespace in self.dirpath:
 819             if namespace not in self.properties or "export" not in self.properties[namespace]:
 820                 print("warning: %s has no export property" % namespace)

 857 class Translation(dict):
 858     "Parses a po file to create a translation dictionary."
 859     def __init__(self, textdomain, isocode, topdir=""):
 860         self.textdomain = textdomain
 861         self.isocode = isocode
 862         self.gettext = {}
 863         if self.isocode != "C":
 864             isocode2 = isocode[:isocode.rfind("_")]
 865             for code in [isocode, isocode2]:
 866                 fn = "po/%s/%s.po" % (textdomain, code)
 867                 if topdir: fn = os.path.join(topdir, fn)
 868                 try:
 869                     f = file(fn)
 870                     break
 871                 except IOError:
 872                     pass
 873             else:
 874                 raise TranslationError(textdomain, self.isocode)
 875 
 876             expect = False
 877             fuzzy = "#, fuzzy\n"
 878             gettext = f.read().decode("utf8")
 879             matches = re.compile("""(msgid|msgstr)((\s*".*?")+)""").finditer(gettext)
 880             msgid = ""
 881             for match in matches:
 882                 text = "".join(re.compile('"(.*?)"').findall(match.group(2)))
 883                 if match.group(1) == "msgid":
 884                     msgid = text.replace("\\n", "\n")
 885                     expect = gettext[match.start(1) - len(fuzzy):match.start(1)] != fuzzy
 886                 elif expect:

 888     def get(self, key, dflt):
 889         if self.isocode == "C":
 890             if key:
 891                 return key[key.find("^") + 1:]
 892             return "?"
 893         else:
 894             t = self.gettext.get(key, dflt)
 895             if not t:
 896                 if key:
 897                     return key[key.find("^") + 1:]
 898                 return "?"
 899             return t
 900     def __getitem__(self, key):
 901         if self.isocode == "C":
 902             return key
 903         else:
 904             return self.gettext[key]
 905     def __contains__(self, key):
 906         if self.isocode == "C":

 908         else:

 911 class Translations:
 912     "Wraps around Translation to support multiple languages and domains."
 913     def __init__(self, topdir = ""):
 914         self.translations = {}
 915         self.topdir = topdir
 916     def get(self, textdomain, isocode, key, default):
 917         t = (textdomain, isocode)
 918         if not t in self.translations:
 919             try:
 920                 self.translations[t] = Translation(textdomain, isocode, self.topdir)
 921             except TranslationError as e:

 923                 self.translations[t] = Translation(textdomain, "C", self.topdir)
