   4 function wesnoth.micro_ais.protect_unit(cfg)

   6         local CA_parms = {

   9                 { ca_id = 'attack', location = 'ca_protect_unit_attack.lua', score = 95000 },
  10                 { ca_id = 'move', location = 'ca_protect_unit_move.lua', score = 94999 }
  11         }
  12 
  13         local unit_ids = {}

  17                 end
  18                 if not u.goal_x then
  19                         H.wml_error("Protect Unit Micro AI missing goal_x key in [unit] tag")

  22                         H.wml_error("Protect Unit Micro AI missing goal_y key in [unit] tag")
  23                 end
  24                 table.insert(unit_ids, u.id)
  25         end
  26 
  27         -- Optional key disable_move_leader_to_keep: needs to be dealt with
  28         -- separately as it affects a default CA
  29         if cfg.disable_move_leader_to_keep then
  30                 wesnoth.delete_ai_component(cfg.side, "stage[main_loop].candidate_action[move_leader_to_keep]")
  31         end
  32 
  33         -- attacks aspects also needs to be set separately
  34         local aspect_parms = {
  35                 {
  36                         aspect = "attacks",
  37                         facet = {
  38                                 name = "ai_default_rca::aspect_attacks",
  39                                 id = "mai_protect_" .. (cfg.ca_id or "default") .. "_dont_attack",
  40                                 invalidate_on_gamestate_change = "yes",
  41                                 { "filter_own", {
  42                                         { "not", {
  43                                                 id = unit_ids
  44                                         } }
  45                                 } }
  46                         }
  47                 }
  48         }
  49 
  50         if (cfg.action == "delete") then

  53                 -- This works even if it was not removed, it simply overwrites the existing CA
  54                 wesnoth.add_ai_component(side, "stage[main_loop].candidate_action",
  55                         {
  56                                 id="move_leader_to_keep",
  57                                 engine="cpp",
  58                                 name="ai_default_rca::move_leader_to_keep_phase",
  59                                 max_score=160000,
  60                                 score=160000
  61                         }
  62                 )
  63         else
  64                 MAIH.add_aspects(cfg.side, aspect_parms)
  65         end
  66     return {"[unit]"}, {}, CA_parms
  67 end
