  22 function ca_herding_sheep_move:execution(cfg)
  23     -- We simply move the first sheep first, the order does not matter
  24     local sheep = get_next_sheep(cfg)
  25 
  26     local reach_map = AH.get_reachable_unocc(sheep)
  27     local dogs_filter = H.get_child(cfg, "filter")
  28     -- Exclude those that are next to a dog
  29     reach_map:iter( function(x, y, v)
  30         for xa, ya in H.adjacent_tiles(x, y) do
  31             local dog = wesnoth.get_unit(xa, ya)
  32             if dog and (wesnoth.match_unit(dog, dogs_filter)) then
  33                 reach_map:remove(x, y)
  34             end
  35         end
  36     end)
  37 
  38     -- Choose one of the possible locations  at random (or the current location, if no move possible)
  39     local x, y = sheep.x, sheep.y
  40     if (reach_map:size() > 0) then
  41         x, y = reach_map:random()
  42     end
  43 
  44     -- If this move remains within herding area or dogs have no moves left, or sheep doesn't move
  45     -- make it a full move, otherwise partial move
  46     local herding_area = herding_area(cfg)
  47     local dogs = AH.get_units_with_moves {
  48         side = wesnoth.current.side,
  49         { "and", dogs_filter }
  50     }
  51     if herding_area:get(x, y) or (not dogs[1]) or ((x == sheep.x) and (y == sheep.y)) then
  52         AH.movefull_stopunit(ai, sheep, x, y)
  53     else
  54         AH.checked_move(ai, sheep, x, y)
  55     end
  56 end
