  24

  27     for _,unit in ipairs(protected_units) do wesnoth.extract_unit(unit) end

  34 
  35     -- Now put the protected units back out there

  37 
  38     -- We move the weakest (fewest HP unit) first
  39     local unit = AH.choose(protected_units, function(u) return - u.hitpoints end)
  40     local goal = {}
  41     for u in H.child_range(cfg, "unit") do
  42         if (unit.id == u.id) then goal = { u.goal_x, u.goal_y } end
  43     end
  44 
  45     local reach_map = AH.get_reachable_unocc(unit)

  47 
  48     local terrain_defense_map = LS.create()
  49     reach_map:iter(function(x, y, data)
  50         terrain_defense_map:insert(x, y, 100 - wesnoth.unit_defense(unit, wesnoth.get_terrain(x, y)))
  51     end)
  52 
  53     local goal_distance_map = LS.create()
  54     reach_map:iter(function(x, y, data)
  55         goal_distance_map:insert(x, y, wesnoth.map.distance_between(x, y, goal[1], goal[2]))
  56     end)
  57 
  58     -- Configuration parameters (no option to change these enabled at the moment)
  59     local enemy_weight = data.PU_enemy_weight or 100.
  60     local my_unit_weight = data.PU_my_unit_weight or 1.
  61     local distance_weight = data.PU_distance_weight or 3.
  62     local terrain_weight = data.PU_terrain_weight or 0.1
  63 
  64     -- If there are no enemies left, only distance to goal matters
  65     -- This is to avoid rare situations where moving toward goal rating is canceled by rating for moving away from own units
  66     if (not enemy_units[1]) then
  67         enemy_weight = 0
  68         my_unit_weight = 0
  69         distance_weight = 3
  70         terrain_weight = 0
  71     end
  72 
  73     local max_rating, best_hex = -9e99
  74     for ind,_ in pairs(reach_map.values) do
  75         local rating =
  76             (attack_map.values[ind] or 0) * my_unit_weight
  77             - (enemy_attack_map.values[ind] or 0) * enemy_weight
  78 
  79         rating = rating - goal_distance_map.values[ind] * distance_weight
  80 
  81         rating = rating + terrain_defense_map.values[ind] * terrain_weight
  82 
  83         rating = rating + (enemy_attack_map.values[ind] or 0) / 10.
  84 
  85         if (rating > max_rating) then
  86             max_rating, best_hex = rating, ind
  87         end
  88     end
