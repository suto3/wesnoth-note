  19 # Centers the text horizontally.
  20 #
  21 # If the text is too wide it returns the left side.
  22 # Rounding happens to the left side.
  23 #define GUI__TEXT_HORIZONTALLY_CENTRED
  24         "(max((width - text_width) / 2, 0))"
  25 #enddef
  26 
  27 # Centers the text vertically.
  28 #
  29 # If the text is too high it returns the top side.
  30 # Rounding happens to the bottom side.
  31 #define GUI__TEXT_VERTICALLY_CENTRED
  32         "(max((height - text_height - 2) / 2, 0))"
  33 #enddef
  34

 157 # Forces a widget to be of a certain size.
 158 # Depending of the flags of the parent widget the item will grow
 159 #
 160 # WIDTH                           Minimum width for the widget.
 161 # HEIGHT                          Minimum height for the widget.
 162 # WIDGET                          The widget to add into the item.
 163 #
 164 # eg instead of
 165 #       [label]
 166 #       # Note Foo has trailing spaces to force extra width.
 167 #               label = "
 168 #               Foo
 169 #               "
 170 #       [label]
 171 # you can do:
 172 #
 173 # {GUI_FORCE_WIDGET_MINIMUM_SIZE 150 75 (
 174 #       [label]
 175 #               label = "Foo"
 176 #       [label]
 177 # )}
 178 #
 179 #define GUI_FORCE_WIDGET_MINIMUM_SIZE WIDTH HEIGHT WIDGET
 180         [stacked_widget]
 181                 definition = "default"
 182 
 183                 [layer]
 184 
 185                         [row]
 186                                 grow_factor = 1
 187 
 188                                 [column]
 189                                         grow_factor = 1
 190                                         horizontal_grow = true
 191                                         vertical_grow = true
 192 
 193                                         [spacer]
 194                                                 definition = "default"
 195                                                 width = "{WIDTH}"
 196                                                 height = "{HEIGHT}"
 197                                         [/spacer]
 198 
 199                                 [/column]
 200 
 201                         [/row]
 202 
 203                 [/layer]
 204 
 205                 [layer]
 206 
 207                         [row]
 208                                 grow_factor = 1
 209 
 210                                 [column]
 211                                         grow_factor = 1
 212                                         horizontal_grow = true
 213                                         vertical_grow = true
 214 
 215                                         {WIDGET}
 216                                 [/column]
 217 
 218                         [/row]
 219 
 220                 [/layer]
 221 
 222         [/stacked_widget]
 223 #enddef

 225 # Forces a widget to be of a certain size and blocks it from growing.
 226 #
 227 # WIDTH                      Width for the widget.
 228 # HEIGHT                     Height for the widget.
 229 # WIDGET                     The widget to add to the item.
 230 #
 231 #define GUI_FORCE_WIDGET_SIZE WIDTH HEIGHT WIDGET
 232         [size_lock]
 233                 definition = "default"
 234                 width = "{WIDTH}"
 235                 height = "{HEIGHT}"
 236 
 237                 [widget]
 238                         {WIDGET}
 239                 [/widget]
 240         [/size_lock]
 241 #enddef
