 129 void list_view::remove_row(const unsigned row, unsigned count)
 130 {
 131         assert(generator_);
 132 
 133         if(row >= get_item_count()) {
 134                 return;
 135         }
 136 
 137         if(!count || count > get_item_count()) {
 138                 count = get_item_count();
 139         }
 140 
 141         unsigned height_reduced = 0;
 142         for(; count; --count) {
 143                 if(generator_->item(row).get_visible() != visibility::invisible) {
 144                         height_reduced += generator_->item(row).get_height();
 145                 }
 146                 generator_->delete_item(row);
 147         }
 148 
 149         if(height_reduced != 0) {
 150                 // resize_content(0, -height_reduced);
 151         }
 152 }

 173 void list_view::set_row_shown(const unsigned row, const bool shown)
 174 {
 175         assert(generator_);
 176 
 177         window* window = get_window();
 178         assert(window);
 179 
 180         const int selected_row = get_selected_row();
 181 
 182         bool resize_needed = false;
 183         {
 184                 window::invalidate_layout_blocker invalidate_layout_blocker(*window);
 185 
 186                 generator_->set_item_shown(row, shown);
 187                 generator_->place(generator_->get_origin(),
 188                                                   generator_->calculate_best_size());
 189                 // resize_needed = !content_resize_request();
 190         }
 191 
 192         if(resize_needed) {
 193                 window->invalidate_layout();
 194         } else {
 195                 // get_grid().set_visible_rectangle(content_visible_rectangle());
 196                 set_is_dirty(true);
 197         }
 198 
 199         if(selected_row != get_selected_row()) {
 200                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 201         }

 390 void list_view::signal_handler_pre_child_left_button_click(
 391                 grid* grid, const event::ui_event event, bool& handled, bool& halt)
 392 {
 393         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 394 
 395         assert(grid);
 396         assert(generator_);
 397 
 398         for(size_t i = 0; i < generator_->get_item_count(); ++i) {
 399                 if(&generator_->item(i) == grid) {
 400 
 401                         /**
 402                          * @todo Here we should check whether the panel can be toggled.
 403                          *
 404                          * NO set halt + handled
 405                          * YES do nothing
 406                          *
 407                          * Then a post to the widget, which if done sets the proper state
 408                          * in the list.
 409                          *
 410                          * For now we simply assume an item can only be selected and not
 411                          * deselected (which is true at the moment).
 412                          */
 413                         if(generator_->is_selected(i)) {
 414                                 halt = true;
 415                                 handled = true;
 416                         }
 417                         return;
 418                 }
 419         }
 420         assert(false);

 439 void list_view::signal_handler_sdl_key_down(const event::ui_event event,
 440                                                                                 bool& handled,
 441                                                                                 const SDL_Keycode key,
 442                                                                                 SDL_Keymod modifier)
 443 {
 444         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 445 
 446         if(handled) {
 447                 return;
 448         }
 449 
 450         switch(key) {
 451                 case SDLK_UP:
 452                         generator_->handle_key_up_arrow(modifier, handled);
 453                         break;
 454                 case SDLK_DOWN:
 455                         generator_->handle_key_down_arrow(modifier, handled);
 456                         break;
 457                 case SDLK_LEFT:
 458                         generator_->handle_key_left_arrow(modifier, handled);
 459                         break;
 460                 case SDLK_RIGHT:
 461                         generator_->handle_key_right_arrow(modifier, handled);
 462                         break;
 463                 default:
 464                         ;
 465                         /* Do nothing. */
 466         }
 467         if(handled) {
 468                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 469         }
 470 }
