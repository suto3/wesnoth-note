  63 // The hashing code is basically taken from forum_auth.cpp
  64 bool fuh::login(const std::string& name, const std::string& password, const std::string& seed) {
  65 
  66         // Retrieve users' password as hash
  67 
  68         std::string hash;
  69 
  70         try {
  71                 hash = get_hash(name);
  72         } catch (error& e) {
  73                 ERR_UH << "Could not retrieve hash for user '" << name << "' :" << e.message << std::endl;
  74                 return false;
  75         }
  76 
  77         // Check hash prefix, if different than $H$ hash is invalid
  78         if(!utils::md5::is_valid_hash(hash)) {
  79                 ERR_UH << "Invalid hash for user '" << name << "'" << std::endl;
  80                 return false;
  81         }
  82 
  83         std::string valid_hash = utils::md5(hash.substr(12,34), seed).base64_digest();
  84 
  85         if(password == valid_hash) return true;
  86 
  87         return false;
  88 }

  90 std::string fuh::create_pepper(const std::string& name) {
  91 
  92         // Some double security, this should never be needed
  93         if(!(user_exists(name))) {
  94                 return "";
  95         }
  96 
  97         std::string hash;
  98 
  99         try {
 100                 hash = get_hash(name);
 101         } catch (error& e) {
 102                 ERR_UH << "Could not retrieve hash for user '" << name << "' :" << e.message << std::endl;
 103                 return "";
 104         }
 105 
 106         if(!utils::md5::is_valid_hash(hash)) return "";
 107 
 108         return hash.substr(0,12);
 109 }

 166 std::string fuh::user_info(const std::string& name) {
 167         if(!user_exists(name)) {
 168                 throw error("No user with the name '" + name + "' exists.");
 169         }
 170 
 171         time_t reg_date = get_registrationdate(name);
 172         time_t ll_date = get_lastlogin(name);
 173 
 174         std::string reg_string = ctime(&reg_date);
 175         std::string ll_string;
 176 
 177         if(ll_date) {
 178                 ll_string = ctime(&ll_date);
 179         } else {
 180                 ll_string = "Never\n";
 181         }
 182 
 183         std::stringstream info;
 184         info << "Name: " << name << "\n"
 185                  << "Registered: " << reg_string
 186                  << "Last login: " << ll_string;
 187         if(!user_is_active(name)) {
 188                 info << "This account is currently inactive.\n";
 189         }
 190 
 191         return info.str();
 192 }

 249 template<typename T, typename... Args>
 250 inline T fuh::prepared_statement(const std::string& sql, Args&&... args)
 251 {
 252         try {
 253                 return ::prepared_statement<T>(conn, sql, std::forward<Args>(args)...);
 254         } catch (sql_error&) {
 255                 WRN_UH << "caught sql error, trying to reconnect and retry..." << std::endl;
 256                 //Try to reconnect and execute query again
 257                 if(!mysql_real_connect(conn, db_host_.c_str(),  db_user_.c_str(), db_password_.c_str(), db_name_.c_str(), 0, nullptr, 0)) {
 258                         ERR_UH << "Could not connect to database: " << mysql_errno(conn) << ": " << mysql_error(conn) << std::endl;
 259                         throw sql_error("Error querying database.");
 260                 }
 261         }
 262         return ::prepared_statement<T>(conn, sql, std::forward<Args>(args)...);
 263 }

 272 template<typename T>

 280 template<typename T>
 281 void fuh::write_detail(const std::string& name, const std::string& detail, T&& value) {
 282         try {
 283                 // Check if we do already have a row for this user in the extra table
 284                 if(!extra_row_exists(name)) {
 285                         // If not create the row
 286                         prepared_statement<void>("INSERT INTO `" + db_extra_table_ + "` VALUES(?,?,'0')", name, std::forward<T>(value));
 287                 }
 288                 prepared_statement<void>("UPDATE `" + db_extra_table_ + "` SET " + detail + "=? WHERE UPPER(username)=UPPER(?)", std::forward<T>(value), name);
 289         } catch (sql_error& e) {
 290                 ERR_UH << "Could not set detail for user '" << name << "': " << e.message << std::endl;
 291         }
 292 }
