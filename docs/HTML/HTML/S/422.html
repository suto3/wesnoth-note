  87 static std::string format_level_string(const int level)
  88 {
  89         std::string lvl = std::to_string(level);
  90 
  91         if(level < 1) {
  92                 return "<span color='#969696'>" + lvl + "</span>";
  93         } else if(level == 1) {
  94                 return lvl;
  95         } else if(level == 2) {
  96                 return "<b>" + lvl + "</b>";
  97         } else {
  98                 return"<b><span color='#ffffff'>" + lvl + "</span></b>";
  99         }
 100 }

 102 static std::string format_cost_string(int unit_recall_cost, const int team_recall_cost)
 103 {
 104         std::stringstream str;
 105 
 106         if(unit_recall_cost < 0) {
 107                 unit_recall_cost = team_recall_cost;
 108         }
 109 
 110         if(unit_recall_cost > team_recall_cost) {
 111                 str << "<span color='#ff0000'>" << unit_recall_cost << "</span>";
 112         } else if(unit_recall_cost == team_recall_cost) {
 113                 str << unit_recall_cost;
 114         } else if(unit_recall_cost < team_recall_cost) {
 115                 str << "<span color='#00ff00'>" << unit_recall_cost << "</span>";
 116         }
 117 
 118         return str.str();
 119 }

 121 static std::string get_title_suffix(int side_num)
 122 {
 123         if(!resources::gameboard) {
 124                 return "";
 125         }
 126 
 127         unit_map& units = resources::gameboard->units();
 128 
 129         int controlled_recruiters = 0;
 130         for(const auto& team : resources::gameboard->teams()) {
 131                 if(team.is_local_human() && !team.recruits().empty() && units.find_leader(team.side()) !=units.end()) {
 132                         ++controlled_recruiters;
 133                 }
 134         }
 135 
 136         std::stringstream msg;
 137         if(controlled_recruiters >= 2) {
 138                 unit_map::const_iterator leader = resources::gameboard->units().find_leader(side_num);
 139                 if(leader != resources::gameboard->units().end() && !leader->name().empty()) {
 140                         msg << " (" << leader->name(); msg << ")";
 141                 }
 142         }
 143 
 144         return msg.str();
 145 }

 147 bool unit_recall::unit_recall_default_compare(const unit_const_ptr first, const unit_const_ptr second)
 148 {
 149         if (first->level() > second->level()) return true;
 150         if (first->level() < second->level()) return false;
 151         if (first->experience_to_advance() < second->experience_to_advance()) return true;
 152         return false;
 153 }

 155 void unit_recall::pre_show(window& window)
 156 {
 157         label& title = find_widget<label>(&window, "title", true);
 158         title.set_label(title.get_label() + get_title_suffix(team_.side()));
 159 
 160         text_box* filter
 161                         = find_widget<text_box>(&window, "filter_box", false, true);
 162 
 163         filter->set_text_changed_callback(
 164                         std::bind(&unit_recall::filter_text_changed, this, _1, _2));
 165 
 166         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 167 
 168         connect_signal_notify_modified(list, std::bind(&unit_recall::list_item_clicked, this, std::ref(window)));
 169 
 170         list.clear();
 171 
 172         window.keyboard_capture(filter);
 173         window.add_to_keyboard_chain(&list);
 174 
 175         connect_signal_mouse_left_click(
 176                 find_widget<button>(&window, "dismiss", false),
 177                 std::bind(&unit_recall::dismiss_unit, this, std::ref(window)));
 178 
 179         connect_signal_mouse_left_click(
 180                 find_widget<button>(&window, "show_help", false),
 181                 std::bind(&unit_recall::show_help, this, std::ref(window)));
 182 
 183         for(const unit_const_ptr& unit : recall_list_) {
 184                 std::map<std::string, string_map> row_data;
 185                 string_map column;
 186 
 187                 std::string mods = unit->image_mods();
 188 
 189                 if(unit->can_recruit()) {
 190                         mods += "~BLIT(" + unit::leader_crown() + ")";
 191                 }
 192 
 193                 for(const std::string& overlay : unit->overlays()) {
 194                         mods += "~BLIT(" + overlay + ")";
 195                 }
 196 
 197                 column["use_markup"] = "true";
 198 
 199                 column["label"] = unit->absolute_image() + mods;
 200                 row_data.emplace("unit_image", column);
 201 
 202                 column["label"] = unit->type_name();
 203                 row_data.emplace("unit_type", column);
 204 
 205                 column["label"] = format_cost_string(unit->recall_cost(), team_.recall_cost());
 206                 row_data.emplace("unit_recall_cost", column);
 207 
 208                 const std::string& name = !unit->name().empty() ? unit->name().str() : font::unicode_en_dash;
 209                 column["label"] = name;
 210                 row_data.emplace("unit_name", column);
 211 
 212                 column["label"] = format_level_string(unit->level());
 213                 row_data.emplace("unit_level", column);
 214 
 215                 std::stringstream exp_str;
 216                 exp_str << font::span_color(unit->xp_color()) << unit->experience() << "/"
 217                         << (unit->can_advance() ? std::to_string(unit->max_experience()) : font::unicode_en_dash) << "</span>";
 218 
 219                 column["label"] = exp_str.str();
 220                 row_data.emplace("unit_experience", column);
 221 
 222                 // Since the table widgets use heavy formatting, we save a bare copy
 223                 // of certain options to filter on.
 224                 std::string filter_text = unit->type_name() + " " + name + " " + std::to_string(unit->level());
 225 
 226                 std::string traits;
 227                 for(const std::string& trait : unit->trait_names()) {
 228                         traits += (traits.empty() ? "" : "\n") + trait;
 229                         filter_text += " " + trait;
 230                 }
 231 
 232                 column["label"] = !traits.empty() ? traits : font::unicode_en_dash;
 233                 row_data.emplace("unit_traits", column);
 234 
 235                 list.add_row(row_data);
 236                 filter_options_.push_back(filter_text);
 237         }
 238 
 239         list.register_sorting_option(0, [this](const int i) { return recall_list_[i]->type_name().str(); });
 240         list.register_sorting_option(1, [this](const int i) { return recall_list_[i]->name().str(); });
 241         list.set_column_order(2, {{
 242                         [this](int lhs, int rhs) { return unit_recall_default_compare(recall_list_[rhs], recall_list_[lhs]); },
 243                         [this](int lhs, int rhs) { return unit_recall_default_compare(recall_list_[lhs], recall_list_[rhs]); }
 244         }});
 245         list.register_sorting_option(3, [this](const int i) { return recall_list_[i]->experience(); });
 246         list.register_sorting_option(4, [this](const int i) {
 247                 return !recall_list_[i]->trait_names().empty() ? recall_list_[i]->trait_names().front().str() : "";
 248         });
 249 
 250         list.set_active_sorting_option(sort_last.first >= 0 ? sort_last : sort_default);
 251 
 252         list_item_clicked(window);
 253 }

 255 void unit_recall::dismiss_unit(window& window)
 256 {
 257         LOG_DP << "Recall list units:\n"; dump_recall_list_to_console(recall_list_);
 258 
 259         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 260         const int index = list.get_selected_row();
 261 
 262         const unit& u = *recall_list_[index].get();
 263 
 264         // If the unit is of level > 1, or is close to advancing, we warn the player about it
 265         std::stringstream message;
 266         if(u.loyal()) {
 267                 message << _("This unit is loyal and requires no upkeep.") << " " << (u.gender() == unit_race::MALE
 268                          ? _("Do you really want to dismiss him?")
 269                          : _("Do you really want to dismiss her?"));
 270 
 271         } else if(u.level() > 1) {
 272                 message << _("This unit is an experienced one, having advanced levels.") << " " << (u.gender() == unit_race::MALE
 273                          ? _("Do you really want to dismiss him?")
 274                          : _("Do you really want to dismiss her?"));
 275 
 276         } else if(u.experience() > u.max_experience()/2) {
 277                 message << _("This unit is close to advancing a level.") << " " << (u.gender() == unit_race::MALE
 278                          ? _("Do you really want to dismiss him?")
 279                          : _("Do you really want to dismiss her?"));
 280         }
 281 
 282         if(!message.str().empty()) {
 283                 const int res = gui2::show_message(window.video(), _("Dismiss Unit"), message.str(), message::yes_no_buttons);
 284 
 285                 if(res != gui2::window::OK) {
 286                         return;
 287                 }
 288         }
 289 
 290         recall_list_.erase(recall_list_.begin() + index);
 291 
 292         // Remove the entry from the dialog list
 293         list.remove_row(index);
 294         list_item_clicked(window);
 295 
 296         // Remove the entry from the filter list
 297         filter_options_.erase(filter_options_.begin() + index);
 298         assert(filter_options_.size() == list.get_item_count());
 299 
 300         LOG_DP << "Dismissing a unit, side = " << u.side() << ", id = '" << u.id() << "'\n";
 301         LOG_DP << "That side's recall list:\n";
 302         dump_recall_list_to_console(team_.recall_list());
 303 
 304         // Find the unit in the recall list.
 305         unit_ptr dismissed_unit = team_.recall_list().find_if_matches_id(u.id());
 306         assert(dismissed_unit);
 307 
 308         // Record the dismissal, then delete the unit.
 309         synced_context::run_and_throw("disband", replay_helper::get_disband(dismissed_unit->id()));
 310 
 311         // Close the dialog if all units are dismissed
 312         if(list.get_item_count() == 0) {
 313                 window.set_retval(window::CANCEL);
 314         }
 315 }

 345 void unit_recall::filter_text_changed(text_box_base* textbox, const std::string& text)
 346 {
 347         window& window = *textbox->get_window();
 348 
 349         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 350 
 351         const std::vector<std::string> words = utils::split(text, ' ');
 352 
 353         if(words == last_words_)
 354                 return;
 355         last_words_ = words;
 356 
 357         boost::dynamic_bitset<> show_items;
 358         show_items.resize(list.get_item_count(), true);
 359 
 360         if(!text.empty()) {
 361                 for(unsigned int i = 0; i < list.get_item_count(); i++) {
 362                         bool found = false;
 363 
 364                         for(const auto & word : words) {
 365                                 found = std::search(filter_options_[i].begin(),
 366                                                         filter_options_[i].end(),
 367                                                         word.begin(),
 368                                                         word.end(),
 369                                                         chars_equal_insensitive)
 370                                                 != filter_options_[i].end();
 371 
 372                                 if(!found) {
 373                                         // one word doesn't match, we don't reach words.end()
 374                                         break;
 375                                 }
 376                         }
 377 
 378                         show_items[i] = found;
 379                 }
 380         }
 381 
 382         list.set_row_shown(show_items);
 383 }
