 286 bool game_launcher::init_joystick()
 287 {
 288         if (!preferences::joystick_support_enabled())
 289                 return false;
 290 
 291         if(SDL_WasInit(SDL_INIT_JOYSTICK) == 0)
 292                 if(SDL_InitSubSystem(SDL_INIT_JOYSTICK) == -1)
 293                         return false;
 294 
 295         int joysticks = SDL_NumJoysticks();
 296         if (joysticks == 0) return false;
 297 
 298         SDL_JoystickEventState(SDL_ENABLE);
 299 
 300         bool joystick_found = false;
 301         for (int i = 0; i<joysticks; i++)  {
 302 
 303                 if (SDL_JoystickOpen(i))
 304                         joystick_found = true;
 305         }
 306         return joystick_found;
 307 }

 309 bool game_launcher::init_language()
 310 {
 311         if(!::load_language_list())
 312                 return false;
 313 
 314         language_def locale;
 315         if(cmdline_opts_.language) {
 316                 std::vector<language_def> langs = get_languages();
 317                 for(const language_def & def : langs) {
 318                         if(def.localename == *cmdline_opts_.language) {
 319                                 locale = def;
 320                                 break;
 321                         }
 322                 }
 323                 if(locale.localename.empty()) {
 324                         std::cerr << "Language symbol '" << *cmdline_opts_.language << "' not found.\n";
 325                         return false;
 326                 }
 327         } else {
 328                 locale = get_locale();
 329         }
 330         ::set_language(locale);
 331 
 332         return true;
 333 }

 364 bool game_launcher::init_lua_script()
 365 {
 366         bool error = false;
 367 
 368         std::cerr << "Checking lua scripts... ";
 369 
 370         if (cmdline_opts_.script_unsafe_mode) {
 371                 plugins_manager::get()->get_kernel_base()->load_package(); //load the "package" package, so that scripts can get what packages they want
 372         }
 373 
 374         // get the application lua kernel, load and execute script file, if script file is present
 375         if (cmdline_opts_.script_file)
 376         {
 377                 filesystem::scoped_istream sf = filesystem::istream_file(*cmdline_opts_.script_file);
 378 
 379                 if (!sf->fail()) {
 380                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 381                         jump_to_multiplayer_ = false;
 382                         jump_to_editor_ = false;
 383                         jump_to_campaign_.jump_ = false;
 384 
 385                         std::string full_script((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 386 
 387                         std::cerr << "\nRunning lua script: " << *cmdline_opts_.script_file << std::endl;
 388 
 389                         plugins_manager::get()->get_kernel_base()->run(full_script.c_str());
 390                 } else {
 391                         std::cerr << "Encountered failure when opening script '" << *cmdline_opts_.script_file << "'\n";
 392                         error = true;
 393                 }
 394         }
 395 
 396         if (cmdline_opts_.plugin_file)
 397         {
 398                 std::string filename = *cmdline_opts_.plugin_file;
 399 
 400                 std::cerr << "Loading a plugin file'" << filename << "'...\n";
 401 
 402                 filesystem::scoped_istream sf = filesystem::istream_file(filename);
 403 
 404                 try {
 405                         if (sf->fail()) {
 406                                 throw std::runtime_error("failed to open plugin file");
 407                         }
 408 
 409                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 410                         jump_to_multiplayer_ = false;
 411                         jump_to_editor_ = false;
 412                         jump_to_campaign_.jump_ = false;
 413 
 414                         std::string full_plugin((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 415 
 416                         plugins_manager & pm = *plugins_manager::get();
 417 
 418                         size_t i = pm.add_plugin(filename, full_plugin);
 419 
 420                         for (size_t j = 0 ; j < pm.size(); ++j) {
 421                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 422                         }
 423 
 424                         std::cerr << "Starting a plugin...\n";
 425                         pm.start_plugin(i);
 426 
 427                         for (size_t j = 0 ; j < pm.size(); ++j) {
 428                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 429                         }
 430 
 431                         plugins_context pc("init");
 432 
 433                         for (size_t repeat = 0; repeat < 5; ++repeat) {
 434                                 std::cerr << "Playing a slice...\n";
 435                                 pc.play_slice();
 436 
 437                                 for (size_t j = 0 ; j < pm.size(); ++j) {
 438                                         std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 439                                 }
 440                         }
 441 
 442                         return true;
 443                 } catch (std::exception & e) {
 444                         gui2::show_error_message(video(), std::string("When loading a plugin, error:\n") + e.what());
 445                         error = true;
 446                 }
 447         }
 448 
 449         if (!error) {
 450                 std::cerr << "ok\n";
 451         }
 452 
 453         return !error;
 454 }

 470 bool game_launcher::play_test()
 471 {
 472         static bool first_time = true;
 473 
 474         if(!cmdline_opts_.test) {

 476         }

 478                 return false;
 479 
 480         first_time = false;

 483         state_.classification().campaign_define = "TEST";
 484 
 485         game_config_manager::get()->
 486                 load_game_config_for_game(state_.classification());
 487

 493                 return true;
 494         }
 495 
 496         return false;

 499 // Same as play_test except that we return the results of play_game.
 500 int game_launcher::unit_test()
 501 {
 502         static bool first_time_unit = true;
 503 
 504         if(!cmdline_opts_.unit_test) {
 505                 return 0;
 506         }
 507         if(!first_time_unit)
 508                 return 0;
 509 
 510         first_time_unit = false;
 511 
 512         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::TEST;
 513         state_.classification().campaign_define = "TEST";
 514         state_.set_carryover_sides_start(
 515                 config {"next_scenario", test_scenario_}
 516         );
 517 
 518 
 519         game_config_manager::get()->
 520                 load_game_config_for_game(state_.classification());
 521 
 522         try {
 523                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 524                 LEVEL_RESULT res = ccontroller.play_game();
 525                 if (!(res == LEVEL_RESULT::VICTORY) || lg::broke_strict()) {
 526                         return 1;
 527                 }
 528         } catch(wml_exception& e) {
 529                 std::cerr << "Caught WML Exception:" << e.dev_message << std::endl;
 530                 return 1;
 531         }
 532 
 533         savegame::clean_saves(state_.classification().label);
 534 
 535         if (cmdline_opts_.noreplaycheck)
 536                 return 0; //we passed, huzzah!
 537 
 538         savegame::replay_savegame save(state_, compression::NONE);
 539         save.save_game_automatic(video(), false, "unit_test_replay"); //false means don't check for overwrite
 540 
 541         load_data_.reset(new savegame::load_game_metadata{ "unit_test_replay" , "", true, true, false });
 542 
 543         if (!load_game()) {
 544                 std::cerr << "Failed to load the replay!" << std::endl;
 545                 return 3; //failed to load replay
 546         }
 547 
 548         try {
 549                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 550                 LEVEL_RESULT res = ccontroller.play_replay();
 551                 if (!(res == LEVEL_RESULT::VICTORY)) {
 552                         std::cerr << "Observed failure on replay" << std::endl;
 553                         return 4;
 554                 }
 555         } catch(wml_exception& e) {
 556                 std::cerr << "WML Exception while playing replay: " << e.dev_message << std::endl;
 557                 return 4; //failed with an error during the replay
 558         }
 559 
 560         return 0; //we passed, huzzah!
 561 }

 578 bool game_launcher::play_render_image_mode()
 579 {
 580         if(!cmdline_opts_.render_image) {
 581                 return true;
 582         }
 583 
 584         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 585         DBG_GENERAL << "Current campaign type: " << state_.classification().campaign_type << std::endl;
 586 
 587         try {
 588                 game_config_manager::get()->
 589                         load_game_config_for_game(state_.classification());
 590         } catch(config::error& e) {
 591                 std::cerr << "Error loading game config: " << e.what() << std::endl;
 592                 return false;
 593         }
 594 
 595         // A default output filename
 596         std::string outfile = "wesnoth_image.bmp";
 597 
 598         // If a output path was given as an argument, use that instead
 599         if (cmdline_opts_.render_image_dst) {
 600                 outfile = *cmdline_opts_.render_image_dst;
 601         }
 602 
 603         if (!image::save_image(*cmdline_opts_.render_image, outfile)) {
 604                 exit(1);
 605         }
 606 
 607         return false;
 608 }

 615 bool game_launcher::load_game()
 616 {
 617         assert(game_config_manager::get());
 618 
 619         DBG_GENERAL << "Current campaign type: " << state_.classification().campaign_type << std::endl;
 620 
 621         savegame::loadgame load(video(), game_config_manager::get()->game_config(), state_);
 622         if (load_data_) {
 623                 std::unique_ptr<savegame::load_game_metadata> load_data = std::move(load_data_);
 624                 load.data() = std::move(*load_data);
 625         }
 626 
 627         try {
 628                 if(!load.load_game()) {
 629                         return false;
 630                 }
 631 
 632                 try {
 633                         game_config_manager::get()->
 634                                 load_game_config_for_game(state_.classification());
 635                 } catch(config::error&) {
 636                         return false;
 637                 }
 638 
 639                 load.set_gamestate();
 640 
 641         } catch(config::error& e) {
 642                 if(e.message.empty()) {
 643                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 644                 }
 645                 else {
 646                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 647                 }
 648                 return false;
 649         } catch(wml_exception& e) {
 650                 e.show(video());
 651                 return false;
 652         } catch(filesystem::io_exception& e) {
 653                 if(e.message.empty()) {
 654                         gui2::show_error_message(video(), _("File I/O Error while reading the game"));
 655                 } else {
 656                         gui2::show_error_message(video(), _("File I/O Error while reading the game: '") + e.message + '\'');
 657                 }
 658                 return false;
 659         } catch(game::error& e) {
 660                 if(e.message.empty()) {
 661                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 662                 }
 663                 else {
 664                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 665                 }
 666                 return false;
 667         }
 668 
 669         play_replay_ = load.data().show_replay;
 670         LOG_CONFIG << "is middle game savefile: " << (state_.is_mid_game_save() ? "yes" : "no") << "\n";
 671         LOG_CONFIG << "show replay: " << (play_replay_ ? "yes" : "no") << "\n";
 672         // in case load.data().show_replay && !state_.is_mid_game_save()
 673         // there won't be any turns to replay, but the
 674         // user gets to watch the intro sequence again ...
 675 
 676         if(state_.is_mid_game_save() && load.data().show_replay)
 677         {
 678                 statistics::clear_current_scenario();
 679         }
 680 
 681         if(state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 682                 state_.unify_controllers();
 683         }
 684 
 685         if (load.data().cancel_orders) {
 686                 state_.cancel_orders();
 687         }
 688 
 689         return true;
 690 }

 769 void game_launcher::start_wesnothd()
 770 {
 771         const std::string wesnothd_program =
 772                 preferences::get_mp_server_program_name().empty() ?
 773                 filesystem::get_program_invocation("wesnothd") : preferences::get_mp_server_program_name();
 774 
 775         std::string config = filesystem::get_user_config_dir() + "/lan_server.cfg";
 776         if (!filesystem::file_exists(config)) {
 777                 // copy file if it isn't created yet
 778                 filesystem::write_file(config, filesystem::read_file(filesystem::get_wml_location("lan_server.cfg")));
 779         }
 780 
 781 #ifndef _WIN32
 782         std::string command = "\"" + wesnothd_program +"\" -c \"" + config + "\" -d -t 2 -T 5";
 783 #else
 784         // start wesnoth as background job
 785         std::string command = "cmd /C start \"wesnoth server\" /B \"" + wesnothd_program + "\" -c \"" + config + "\" -t 2 -T 5";
 786         // Make sure wesnothd's console output is visible on the console window by
 787         // disabling SDL's stdio redirection code for this and future child
 788         // processes. No need to bother cleaning this up because it's only
 789         // meaningful to SDL applications during pre-main initialization.
 790         SetEnvironmentVariableA("SDL_STDIO_REDIRECT", "0");
 791 #endif
 792         LOG_GENERAL << "Starting wesnothd: "<< command << "\n";
 793         if (std::system(command.c_str()) == 0) {
 794                 // Give server a moment to start up
 795                 SDL_Delay(50);
 796                 return;
 797         }
 798         preferences::set_mp_server_program_name("");
 799 
 800         // Couldn't start server so throw error
 801         WRN_GENERAL << "Failed to run server start script" << std::endl;
 802         throw game::mp_server_error("Starting MP server failed!");
 803 }

 805 bool game_launcher::play_multiplayer(mp_selection res)
 806 {
 807         state_.clear();
 808         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 809 
 810         try {
 811                 if (res == MP_HOST)
 812                 {
 813                         try {
 814                                 start_wesnothd();
 815                         } catch(game::mp_server_error&)
 816                         {
 817                                 preferences::show_wesnothd_server_search(video());
 818 
 819                                 try {
 820                                         start_wesnothd();
 821                                 } catch(game::mp_server_error&)
 822                                 {
 823                                         return false;
 824                                 }
 825                         }
 826 
 827 
 828                 }
 829                 //create_engine already calls game_config_manager::get()->load_config but maybe its better to have MULTIPLAYER defined while we are in the lobby.
 830                 game_config_manager::get()->load_game_config_for_create(true);
 831 
 832                 events::discard_input(); // prevent the "keylogger" effect
 833                 cursor::set(cursor::NORMAL);
 834 
 835                 if(res == MP_LOCAL) {
 836                         mp::start_local_game(video(),
 837                             game_config_manager::get()->game_config(), state_);
 838                 } else {
 839                         mp::start_client(video(), game_config_manager::get()->game_config(),
 840                                 state_, multiplayer_server_);
 841                         multiplayer_server_.clear();
 842                 }
 843 
 844         } catch(game::mp_server_error& e) {
 845                 gui2::show_error_message(video(), _("Error while starting server: ") + e.message);
 846         } catch(game::load_game_failed& e) {
 847                 gui2::show_error_message(video(), _("The game could not be loaded: ") + e.message);
 848         } catch(game::game_error& e) {
 849                 gui2::show_error_message(video(), _("Error while playing the game: ") + e.message);
 850         } catch (mapgen_exception& e) {
 851                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 852         } catch(wesnothd_error& e) {
 853                 if(!e.message.empty()) {
 854                         ERR_NET << "caught network error: " << e.message << std::endl;
 855                         gui2::show_transient_message(video()
 856                                         , ""
 857                                         , translation::gettext(e.message.c_str()));
 858                 } else {
 859                         ERR_NET << "caught network error" << std::endl;
 860                 }
 861         } catch(config::error& e) {
 862                 if(!e.message.empty()) {
 863                         ERR_CONFIG << "caught config::error: " << e.message << std::endl;
 864                         gui2::show_transient_message(video(), "", e.message);
 865                 } else {
 866                         ERR_CONFIG << "caught config::error" << std::endl;
 867                 }
 868         } catch(incorrect_map_format_error& e) {
 869                 gui2::show_error_message(video(), _("The game map could not be loaded: ") + e.message);
 870         } catch (savegame::load_game_exception & e) {
 871                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 872                 //this will make it so next time through the title screen loop, this game is loaded
 873         } catch(wml_exception& e) {
 874                 e.show(video());
 875         } catch (game::error & e) {
 876                 std::cerr << "caught game::error...\n";
 877                 gui2::show_error_message(video(), _("Error: ") + e.message);
 878         }
 879 
 880         return false;
 881 }

 925 void game_launcher::launch_game(RELOAD_GAME_DATA reload)
 926 {
 927         assert(!load_data_);
 928         if(play_replay_)
 929         {
 930                 play_replay();
 931                 return;
 932         }
 933 
 934         gui2::dialogs::loading_screen::display(video(), [this, reload]() {
 935 
 936                 gui2::dialogs::loading_screen::progress("load data");
 937                 if(reload == RELOAD_DATA) {
 938                         try {
 939                                 game_config_manager::get()->
 940                                         load_game_config_for_game(state_.classification());
 941                         } catch(config::error&) {
 942                                 return;
 943                         }
 944                 }
 945         });
 946 
 947         try {
 948                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types());
 949                 LEVEL_RESULT result = ccontroller.play_game();
 950                 // don't show The End for multiplayer scenario
 951                 // change this if MP campaigns are implemented
 952                 if(result == LEVEL_RESULT::VICTORY && !state_.classification().is_normal_mp_game()) {
 953                         preferences::add_completed_campaign(state_.classification().campaign, state_.classification().difficulty);
 954 
 955                         gui2::dialogs::outro::display(state_.classification().end_text, state_.classification().end_text_duration, video());
 956 
 957                         if(state_.classification().end_credits) {
 958                                 gui2::dialogs::end_credits::display(video(), state_.classification().campaign);
 959                         }
 960                 }
 961         } catch (savegame::load_game_exception &e) {
 962                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 963                 //this will make it so next time through the title screen loop, this game is loaded
 964         } catch(wml_exception& e) {
 965                 e.show(video());
 966         } catch(mapgen_exception& e) {
 967                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 968         }
 969 }

 985 editor::EXIT_STATUS game_launcher::start_editor(const std::string& filename)
 986 {
 987         while(true){
 988                 game_config_manager::get()->load_game_config_for_editor();
 989 
 990                 ::init_textdomains(game_config_manager::get()->game_config());
 991 
 992                 editor::EXIT_STATUS res = editor::start(
 993                     game_config_manager::get()->game_config(), video(), filename);
 994 
 995                 if(res != editor::EXIT_RELOAD_DATA)
 996                         return res;
 997 
 998                 game_config_manager::get()->reload_changed_game_config();
 999                 image::flush_cache();
1000         }
1001         return editor::EXIT_ERROR; // not supposed to happen
1002 }
