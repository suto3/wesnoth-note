 171 {
 172         if(!can_activate())
 173         {
 174                 active_ = false;
 175                 LOG_WB << "Whiteboard can't be activated now.\n";
 176         }

 178         {
 179                 active_ = active;

 183                 {

 185                                 resources::undo_stack->clear();
 186                         validate_viewer_actions();
 187                         LOG_WB << "Whiteboard activated! " << *viewer_actions() << "\n";

 191                 }
 192         }

 195 void manager::set_invert_behavior(bool invert)
 196 {
 197         //any more than one reference means a lock on whiteboard state was requested
 198         if(!activation_state_lock_.unique())
 199                 return;
 200 
 201         bool block_whiteboard_activation = false;
 202         if(!can_activate())
 203         {
 204                  block_whiteboard_activation = true;
 205         }
 206 
 207         if (invert)
 208         {
 209                 if (!inverted_behavior_)
 210                 {
 211                         if (active_)
 212                         {
 213                                 DBG_WB << "Whiteboard deactivated temporarily.\n";
 214                                 inverted_behavior_ = true;
 215                                 set_active(false);
 216                         }
 217                         else if (!block_whiteboard_activation)
 218                         {
 219                                 DBG_WB << "Whiteboard activated temporarily.\n";
 220                                 inverted_behavior_ = true;
 221                                 set_active(true);
 222                         }
 223                 }
 224         }
 225         else
 226         {
 227                 if (inverted_behavior_)
 228                 {
 229                         if (active_)
 230                         {
 231                                 DBG_WB << "Whiteboard set back to deactivated status.\n";
 232                                 inverted_behavior_ = false;
 233                                 set_active(false);
 234                         }
 235                         else if (!block_whiteboard_activation)
 236                         {
 237                                 DBG_WB << "Whiteboard set back to activated status.\n";
 238                                 inverted_behavior_ = false;
 239                                 set_active(true);
 240                         }
 241                 }
 242         }
 243 }

 261 bool manager::allow_leader_to_move(unit const& leader) const
 262 {
 263         if(!has_actions())
 264                 return true;
 265 
 266         //Look for another leader on another keep in the same castle
 267         { wb::future_map future; // start planned unit map scope
 268                 if(!has_planned_unit_map()) {
 269                         WRN_WB << "Unable to build future map to determine whether leader's allowed to move." << std::endl;
 270                 }
 271                 if(find_backup_leader(leader))
 272                         return true;
 273         } // end planned unit map scope
 274 
 275         if(viewer_actions()->empty()) {
 276                 return true;
 277         }
 278 
 279         //Look for planned recruits that depend on this leader
 280         for(action_const_ptr action : *viewer_actions())
 281         {
 282                 recruit_const_ptr recruit = std::dynamic_pointer_cast<class recruit const>(action);
 283                 recall_const_ptr recall = std::dynamic_pointer_cast<class recall const>(action);
 284                 if(recruit || recall)
 285                 {
 286                         map_location const target_hex = recruit?recruit->get_recruit_hex():recall->get_recall_hex();
 287                         if (dynamic_cast<game_state&>(*resources::filter_con).can_recruit_on(leader, target_hex))
 288                                 return false;
 289                 }
 290         }
 291         return true;
 292 }

 326 void manager::post_delete_action(action_ptr action)
 327 {
 328         // The fake unit representing the destination of a chain of planned moves should have the regular animation.
 329         // If the last remaining action of the unit that owned this move is a move as well,
 330         // adjust its appearance accordingly.
 331 
 332         side_actions_ptr side_actions = resources::gameboard->teams().at(action->team_index()).get_side_actions();
 333 
 334         unit_ptr actor = action->get_unit();
 335         if(actor) { // The unit might have died following the execution of an attack
 336                 side_actions::iterator action_it = side_actions->find_last_action_of(*actor);
 337                 if(action_it != side_actions->end()) {
 338                         move_ptr move = std::dynamic_pointer_cast<class move>(*action_it);
 339                         if(move && move->get_fake_unit()) {
 340                                 move->get_fake_unit()->anim_comp().set_standing(true);
 341                         }
 342                 }
 343         }
 344 }

 354 void manager::update_plan_hiding(size_t team_index)
 355 {
 356         //We don't control the "viewing" side ... we're probably an observer
 357         if(!resources::gameboard->teams().at(team_index).is_local_human())
 358                 hide_all_plans();
 359         else // normal circumstance
 360         {
 361                 for(team& t : resources::gameboard->teams())
 362                 {
 363                         //make sure only appropriate teams are hidden
 364                         if(!t.is_network_human())
 365                                 team_plans_hidden_[t.side()-1] = false;
 366 
 367                         if(t.is_enemy(team_index+1) || team_plans_hidden_[t.side()-1])
 368                                 t.get_side_actions()->hide();
 369                         else
 370                                 t.get_side_actions()->show();
 371                 }
 372         }
 373         validate_viewer_actions();
 374 }

 384 void manager::on_change_controller(int side, const team& t)
 385 {
 386         wb::side_actions& sa = *t.get_side_actions();
 387         if(t.is_local_human()) // we own this side now
 388         {
 389                 //tell everyone to clear this side's actions -- we're starting anew
 390                 resources::whiteboard->queue_net_cmd(sa.team_index(),sa.make_net_cmd_clear());
 391                 sa.clear();
 392                 //refresh the hidden_ attribute of every team's side_actions
 393                 update_plan_hiding();
 394         }
 395         else if(t.is_local_ai() || t.is_network_ai()) // no one owns this side anymore
 396                 sa.clear(); // clear its plans away -- the ai doesn't plan ... yet
 397         else if(t.is_network()) // Another client is taking control of the side
 398         {
 399                 if(side==viewer_side()) // They're taking OUR side away!
 400                         hide_all_plans(); // give up knowledge of everyone's plans, in case we became an observer
 401 
 402                 //tell them our plans -- they may not have received them up to this point
 403                 size_t num_teams = resources::gameboard->teams().size();
 404                 for(size_t i=0; i<num_teams; ++i)
 405                 {
 406                         team& local_team = resources::gameboard->teams().at(i);
 407                         if(local_team.is_local_human() && !local_team.is_enemy(side))
 408                                 resources::whiteboard->queue_net_cmd(i,local_team.get_side_actions()->make_net_cmd_refresh());
 409                 }
 410         }
 411 }

 441 //helper fcn
 442 static void draw_numbers(map_location const& hex, side_actions::numbers_t numbers)
 443 {
 444         std::vector<int>& numbers_to_draw = numbers.numbers_to_draw;
 445         std::vector<size_t>& team_numbers = numbers.team_numbers;
 446         int& main_number = numbers.main_number;
 447         std::set<size_t>& secondary_numbers = numbers.secondary_numbers;
 448 
 449         const double x_offset_base = 0.0;
 450         const double y_offset_base = 0.2;
 451         //position 0,0 in the hex is the upper left corner
 452         //0.8 = horizontal coord., close to the right side of the hex
 453         const double x_origin = 0.8 - numbers_to_draw.size() * x_offset_base;
 454         //0.5 = halfway in the hex vertically
 455         const double y_origin = 0.5 - numbers_to_draw.size() * (y_offset_base / 2);
 456         double x_offset = 0, y_offset = 0;
 457 
 458         size_t size = numbers_to_draw.size();
 459         for(size_t i=0; i<size; ++i)
 460         {
 461                 int number = numbers_to_draw[i];
 462 
 463                 std::string number_text = std::to_string(number);
 464                 size_t font_size;
 465                 if (int(i) == main_number) font_size = 19;
 466                 else if (secondary_numbers.find(i)!=secondary_numbers.end()) font_size = 17;
 467                 else font_size = 15;
 468 
 469                 color_t color = team::get_side_color(static_cast<int>(team_numbers[i]+1));
 470                 const double x_in_hex = x_origin + x_offset;
 471                 const double y_in_hex = y_origin + y_offset;
 472                 resources::screen->draw_text_in_hex(hex, display::LAYER_ACTIONS_NUMBERING,
 473                                 number_text, font_size, color, x_in_hex, y_in_hex);
 474                 x_offset += x_offset_base;
 475                 y_offset += y_offset_base;
 476         }
 477 }

 482         //Helper struct that finds all units teams whose planned actions are currently visible
 483         //Only used by manager::pre_draw().
 484         //Note that this structure is used as a functor.
 485         struct move_owners_finder: public visitor
 486         {
 487 
 488         public:
 489                 move_owners_finder(): move_owners_() { }
 490 
 491                 void operator()(action* action) {
 492                         action->accept(*this);
 493                 }
 494 
 495                 std::set<size_t> const& get_units_owning_moves() {
 496                         return move_owners_;
 497                 }
 498 
 499                 virtual void visit(move_ptr move) {
 500                         if(size_t id = move->get_unit_id()) {
 501                                 move_owners_.insert(id);
 502                         }
 503                 }
 504 
 505                 virtual void visit(attack_ptr attack) {
 506                         //also add attacks if they have an associated move
 507                         if(attack->get_route().steps.size() >= 2) {
 508                                 if(size_t id = attack->get_unit_id()) {
 509                                         move_owners_.insert(id);
 510                                 }
 511                         }
 512                 }
 513                 virtual void visit(recruit_ptr){}
 514                 virtual void visit(recall_ptr){}
 515                 virtual void visit(suppose_dead_ptr){}
 516 
 517         private:
 518                 std::set<size_t> move_owners_;
 519         };
 520 }

 522 void manager::pre_draw()
 523 {
 524         if (can_modify_game_state() && has_actions() && unit_map_lock_.unique()) {
 525                 move_owners_finder move_finder;
 526                 for_each_action(std::ref(move_finder));
 527                 units_owning_moves_ = move_finder.get_units_owning_moves();
 528 
 529                 for (size_t unit_id : units_owning_moves_) {
 530                         unit_map::iterator unit_iter = resources::gameboard->units().find(unit_id);
 531                         assert(unit_iter.valid());
 532                         ghost_owner_unit(&*unit_iter);
 533                 }
 534         }
 535 }
 536 
 537 void manager::post_draw()
 538 {
 539         for (size_t unit_id : units_owning_moves_)
 540         {
 541                 unit_map::iterator unit_iter = resources::gameboard->units().find(unit_id);
 542                 if (unit_iter.valid()) {
 543                         unghost_owner_unit(&*unit_iter);
 544                 }
 545         }
 546         units_owning_moves_.clear();
 547 }

 549 void manager::draw_hex(const map_location& hex)
 550 {
 551         /**
 552          * IMPORTANT: none of the code in this method can call anything which would
 553          * cause a hex to be invalidated (i.e. by calling in turn any variant of display::invalidate()).
 554          * Doing so messes up the iterator currently going over the list of invalidated hexes to draw.
 555          */
 556 
 557         if (!wait_for_side_init_ && has_actions())
 558         {
 559                 //call draw() for all actions
 560                 for_each_action(std::bind(&action::draw_hex, std::placeholders::_1, hex));
 561 
 562                 //Info about the action numbers to be displayed on screen.
 563                 side_actions::numbers_t numbers;
 564                 for (team& t : resources::gameboard->teams())
 565                 {
 566                         side_actions& sa = *t.get_side_actions();
 567                         if(!sa.hidden())
 568                                 sa.get_numbers(hex,numbers);
 569                 }
 570                 draw_numbers(hex,numbers); // helper fcn
 571         }
 572 
 573 }

 607 void manager::send_network_data()
 608 {
 609         size_t size = net_buffer_.size();
 610         for(size_t team_index=0; team_index<size; ++team_index)
 611         {
 612                 config& buf_cfg = net_buffer_[team_index];
 613 
 614                 if(buf_cfg.empty())
 615                         continue;
 616 
 617                 config packet;

 650 void manager::create_temp_move()
 651 {
 652         route_.reset();
 653 
 654         /*
 655          * CHECK PRE-CONDITIONS
 656          * (This section has multiple return paths.)
 657          */
 658 
 659         if ( !active_ || !can_modify_game_state() )
 660                 return;
 661 
 662         pathfind::marked_route const& route =
 663                         resources::controller->get_mouse_handler_base().get_current_route();
 664 
 665         if (route.steps.empty() || route.steps.size() < 2) return;
 666 
 667         unit* temp_moved_unit =
 668                         future_visible_unit(resources::controller->get_mouse_handler_base().get_selected_hex(), viewer_side());
 669         if (!temp_moved_unit) temp_moved_unit =
 670                         future_visible_unit(resources::controller->get_mouse_handler_base().get_last_hex(), viewer_side());
 671         if (!temp_moved_unit) return;
 672         if (temp_moved_unit->side() != resources::screen->viewing_side()) return;
 673 
 674         /*
 675          * DONE CHECKING PRE-CONDITIONS, CREATE THE TEMP MOVE
 676          * (This section has only one return path.)
 677          */
 678 
 679         temp_move_unit_underlying_id_ = temp_moved_unit->underlying_id();
 680 
 681         //@todo: May be appropriate to replace these separate components by a temporary
 682         //      wb::move object
 683 
 684         route_.reset(new pathfind::marked_route(route));
 685         //NOTE: route_->steps.back() = dst, and route_->steps.front() = src
 686 
 687         size_t turn = 0;
 688         std::vector<map_location>::iterator prev_itor = route.steps.begin();
 689         std::vector<map_location>::iterator curr_itor = prev_itor;
 690         std::vector<map_location>::iterator end_itor  = route.steps.end();
 691         for(; curr_itor!=end_itor; ++curr_itor)
 692         {
 693                 const map_location& hex = *curr_itor;
 694 
 695                 //search for end-of-turn marks
 696                 pathfind::marked_route::mark_map::const_iterator w =
 697                                 route.marks.find(hex);
 698                 if(w != route.marks.end() && w->second.turns > 0)
 699                 {
 700                         turn = w->second.turns-1;
 701 
 702                         if(turn >= move_arrows_.size())
 703                                 move_arrows_.resize(turn+1);
 704                         if(turn >= fake_units_.size())
 705                                 fake_units_.resize(turn+1);
 706 
 707                         arrow_ptr& move_arrow = move_arrows_[turn];
 708                         fake_unit_ptr& fake_unit = fake_units_[turn];
 709 
 710                         if(!move_arrow)
 711                         {
 712                                 // Create temp arrow
 713                                 move_arrow.reset(new arrow());
 714                                 move_arrow->set_color(team::get_side_color_index(
 715                                                 viewer_side()));
 716                                 move_arrow->set_style(arrow::STYLE_HIGHLIGHTED);
 717                         }
 718 
 719                         arrow_path_t path(prev_itor,curr_itor+1);
 720                         move_arrow->set_path(path);
 721 
 722                         if(path.size() >= 2)
 723                         {
 724                                 // Bug #20299 demonstrates a situation where an incorrect fake/ghosted unit can be used.
 725                                 // So before assuming that a pre-existing fake_unit can be re-used, check that its ID matches the unit being moved.
 726                                 if(!fake_unit || fake_unit.get_unit_ptr()->id() != temp_moved_unit->id())
 727                                 {
 728                                         // Create temp ghost unit
 729                                         fake_unit = fake_unit_ptr(unit_ptr (new unit(*temp_moved_unit)), resources::fake_units);
 730                                         fake_unit->anim_comp().set_ghosted(true);
 731                                 }
 732 
 733                                 unit_display::move_unit(path, fake_unit.get_unit_ptr(), false); //get facing right
 734                                 fake_unit->anim_comp().invalidate(*game_display::get_singleton());
 735                                 fake_unit->set_location(*curr_itor);
 736                                 fake_unit->anim_comp().set_ghosted(true);
 737                         }
 738                         else //zero-hex path -- don't bother drawing a fake unit
 739                                 fake_unit.reset();
 740 
 741                         prev_itor = curr_itor;
 742                 }
 743         }
 744         //in case path shortens on next step and one ghosted unit has to be removed
 745         int ind = fake_units_.size() - 1;
 746         fake_units_[ind]->anim_comp().invalidate(*game_display::get_singleton());
 747         //toss out old arrows and fake units
 748         move_arrows_.resize(turn+1);
 749         fake_units_.resize(turn+1);
 750 }

 756                 if(tmp) {
 757                         tmp->anim_comp().invalidate(*game_display::get_singleton());
 758                 }

 765 void manager::save_temp_move()
 766 {
 767         if (has_temp_move() && !executing_actions_ && !resources::controller->is_linger_mode())
 768         {
 769                 side_actions& sa = *viewer_actions();
 770                 unit* u = future_visible_unit(route_->steps.front());
 771                 assert(u);
 772                 size_t first_turn = sa.get_turn_num_of(*u);
 773 
 774                 validate_viewer_actions();
 775 
 776                 assert(move_arrows_.size() == fake_units_.size());
 777                 size_t size = move_arrows_.size();
 778                 for(size_t i=0; i<size; ++i)
 779                 {
 780                         arrow_ptr move_arrow = move_arrows_[i];
 781                         if(!arrow::valid_path(move_arrow->get_path()))
 782                                 continue;
 783 
 784                         size_t turn = first_turn + i;
 785                         fake_unit_ptr fake_unit = fake_units_[i];
 786 
 787                         //@todo Using a marked_route here is wrong, since right now it's not marked
 788                         //either switch over to a plain route for planned moves, or mark it correctly
 789                         pathfind::marked_route route;
 790                         route.steps = move_arrow->get_path();
 791                         route.move_cost = path_cost(route.steps,*u);
 792 
 793                         sa.queue_move(turn,*u,route,move_arrow,fake_unit);
 794                 }
 795                 erase_temp_move();
 796 
 797                 LOG_WB << *viewer_actions() << "\n";
 798                 print_help_once();
 799         }
 800 }

 807 void manager::save_temp_attack(const map_location& attacker_loc, const map_location& defender_loc, int weapon_choice)
 808 {
 809         if (active_ && !executing_actions_ && !resources::controller->is_linger_mode())
 810         {
 811                 assert(weapon_choice >= 0);
 812 
 813                 arrow_ptr move_arrow;
 814                 fake_unit_ptr fake_unit;
 815                 map_location source_hex;
 816 
 817                 if (route_ && !route_->steps.empty())
 818                 {
 819                         //attack-move
 820                         assert(move_arrows_.size() == 1);
 821                         assert(fake_units_.size() == 1);
 822                         move_arrow = move_arrows_.front();
 823                         fake_unit = fake_units_.front();
 824 
 825                         assert(route_->steps.back() == attacker_loc);
 826                         source_hex = route_->steps.front();
 827 
 828                         fake_unit->anim_comp().set_disabled_ghosted(true);
 829                 }
 830                 else
 831                 {
 832                         //simple attack
 833                         move_arrow.reset(new arrow);
 834                         source_hex = attacker_loc;
 835                         route_.reset(new pathfind::marked_route);
 836                         // We'll pass as parameter a one-hex route with no marks.
 837                         route_->steps.push_back(attacker_loc);
 838                 }
 839 
 840                 unit* attacking_unit = future_visible_unit(source_hex);
 841                 assert(attacking_unit);
 842 
 843                 validate_viewer_actions();
 844 
 845                 side_actions& sa = *viewer_actions();
 846                 sa.queue_attack(sa.get_turn_num_of(*attacking_unit),*attacking_unit,defender_loc,weapon_choice,*route_,move_arrow,fake_unit);
 847 
 848                 print_help_once();
 849 
 850                 resources::screen->invalidate(defender_loc);
 851                 resources::screen->invalidate(attacker_loc);
 852                 erase_temp_move();
 853                 LOG_WB << *viewer_actions() << "\n";
 854         }

 921 void manager::contextual_execute()
 922 {
 923         validate_viewer_actions();
 924         if (can_enable_execution_hotkeys())
 925         {
 926                 erase_temp_move();
 927 
 928                 //For exception-safety, this struct sets executing_actions_ to false on destruction.
 929                 variable_finalizer<bool> finally(executing_actions_, false);
 930 
 931                 action_ptr action;
 932                 side_actions::iterator it = viewer_actions()->end();
 933                 unit const* selected_unit = future_visible_unit(resources::controller->get_mouse_handler_base().get_selected_hex(), viewer_side());
 934                 if (selected_unit &&
 935                                 (it = viewer_actions()->find_first_action_of(*selected_unit)) != viewer_actions()->end())
 936                 {
 937                         executing_actions_ = true;
 938                         viewer_actions()->execute(it);
 939                 }
 940                 else if (highlighter_ && (action = highlighter_->get_execute_target()) &&
 941                                  (it = viewer_actions()->get_position_of(action)) != viewer_actions()->end())
 942                 {
 943                         executing_actions_ = true;
 944                         viewer_actions()->execute(it);
 945                 }
 946                 else //we already check above for viewer_actions()->empty()
 947                 {
 948                         executing_actions_ = true;
 949                         viewer_actions()->execute_next();
 950                 }
 951         } //Finalizer struct sets executing_actions_ to false
 952 }

 960 bool manager::execute_all_actions()
 961 {
 962         //exception-safety: finalizers set variables to false on destruction
 963         //i.e. when method exits naturally or exception is thrown
 964         variable_finalizer<bool> finalize_executing_actions(executing_actions_, false);
 965         variable_finalizer<bool> finalize_executing_all_actions(executing_all_actions_, false);
 966 
 967         validate_viewer_actions();
 968         if(viewer_actions()->empty() || viewer_actions()->turn_size(0) == 0)
 969         {
 970                 //No actions to execute, job done.
 971                 return true;
 972         }
 973 
 974         assert(can_enable_execution_hotkeys());
 975 
 976         erase_temp_move();
 977 
 978         // Build unit map once to ensure spent gold and other calculations are refreshed
 979         set_planned_unit_map();
 980         assert(has_planned_unit_map());
 981         set_real_unit_map();
 982 
 983         executing_actions_ = true;
 984         executing_all_actions_ = true;
 985 
 986         side_actions_ptr sa = viewer_actions();
 987 
 988         if (resources::whiteboard->has_planned_unit_map())
 989         {
 990                 ERR_WB << "Modifying action queue while temp modifiers are applied!!!" << std::endl;
 991         }
 992 
 993         //LOG_WB << "Before executing all actions, " << *sa << "\n";
 994 
 995         while (sa->turn_begin(0) != sa->turn_end(0))
 996         {
 997                 bool action_successful = sa->execute(sa->begin());
 998 
 999                 // Interrupt on incomplete action
1000                 if (!action_successful)
1001                 {
1002                         return false;
1003                 }
1004         }
1005         return true;
1006 }

1115 
1116         switch(selection)
1117         {
1118         case 0:
1119                 for(team* t : allies) {
1120                         team_plans_hidden_[t->side()-1]=false;

1123         case 1:
1124                 for(team* t : allies) {
1125                         team_plans_hidden_[t->side()-1]=true;
1126                 }
1127                 break;
1128         default:
1129                 if(selection > 1)
1130                 {
1131                         size_t t_index = allies[selection-2]->side()-1;
1132                         //toggle ...
1133                         bool hidden = team_plans_hidden_[t_index];
1134                         team_plans_hidden_[t_index] = !hidden;

1136                 break;
1137         }
