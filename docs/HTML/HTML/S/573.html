 116 void chatbox::send_message_button_callback()
 117 {
 118         const std::string& input = chat_input_->get_value();
 119         if(input.empty()) {
 120                 return;
 121         }
 122 
 123         if(input[0] == '/') {
 124                 // TODO: refactor do_speak so it uses context information about
 125                 //      opened window, so e.g. /ignore in a whisper session ignores
 126                 //      the other party without having to specify it's nick.
 127                 chat_handler::do_speak(input);
 128         } else {
 129                 lobby_chat_window& t = open_windows_[active_window_];
 130 
 131                 if(t.whisper) {
 132                         send_whisper(t.name, input);
 133                         add_whisper_sent(t.name, input);
 134                 } else {
 135                         send_chat_room_message(t.name, input);
 136                         add_chat_room_message_sent(t.name, input);
 137                 }
 138         }
 139 
 140         chat_input_->save_to_history();
 141         chat_input_->set_value("");
 142 }

 145 void chatbox::chat_input_keypress_callback(bool& handled, bool& halt, const SDL_Keycode key)
 146 {
 147         if(key == SDLK_RETURN || key == SDLK_KP_ENTER) {
 148                 send_message_button_callback();
 149                 handled = true;
 150                 halt = true;
 151         } else if (key == SDLK_TAB) {
 152                 std::string text = chat_input_->get_value();
 153 
 154                 std::vector<std::string> matches;
 155                 for(const auto& ui : lobby_info_->users()) {
 156                         if(ui.name != preferences::login()) {
 157                                 matches.push_back(ui.name);
 158                         }
 159                 }
 160 
 161                 const bool line_start = utils::word_completion(text, matches);
 162 
 163                 if(matches.empty()) {
 164                         return;
 165                 }
 166 
 167                 if(matches.size() == 1) {
 168                         text.append(line_start ? ": " : " ");
 169                 } else {
 170                         std::string completion_list = utils::join(matches, " ");
 171                         append_to_chatbox(completion_list);
 172                 }
 173 
 174                 chat_input_->set_value(text);
 175 
 176                 handled = true;
 177                 halt = true;
 178         }
 179 }

 265 void chatbox::add_whisper_received(const std::string& sender, const std::string& message)
 266 {
 267         bool can_go_to_active = !preferences::whisper_friends_only() || preferences::is_friend(sender);
 268         bool can_open_new = preferences::auto_open_whisper_windows() && can_go_to_active;
 269 
 270         lobby_info_->get_whisper_log(sender).add_message(sender, message);
 271 
 272         if(whisper_window_open(sender, can_open_new)) {
 273                 if(whisper_window_active(sender)) {
 274                         add_active_window_message(sender, message);
 275 
 276                         do_notify(mp::NOTIFY_WHISPER, sender, message);
 277                 } else {
 278                         add_whisper_window_whisper(sender, message);
 279                         increment_waiting_whsipers(sender);
 280 
 281                         do_notify(mp::NOTIFY_WHISPER_OTHER_WINDOW, sender, message);
 282                 }
 283         } else if(can_go_to_active) {
 284                 add_active_window_whisper(sender, message);
 285                 do_notify(mp::NOTIFY_WHISPER, sender, message);
 286         } else {
 287                 LOG_LB << "Ignoring whisper from " << sender << "\n";
 288         }

 291 void chatbox::add_chat_room_message_sent(const std::string& room,
 292         const std::string& message)
 293 {
 294         lobby_chat_window* t = room_window_open(room, false);
 295         if(!t) {
 296                 LOG_LB << "Cannot add sent message to ui for room " << room << ", player not in the room\n";
 297                 return;
 298         }
 299 
 300         // Do not open room window here. The player should be in the room before sending messages
 301         mp::room_info* ri = lobby_info_->get_room(room);
 302         assert(ri);
 303 
 304         if(!room_window_active(room)) {
 305                 switch_to_window(t);
 306         }
 307 
 308         ri->log().add_message(preferences::login(), message);
 309         add_active_window_message(preferences::login(), message, true);

 312 void chatbox::add_chat_room_message_received(const std::string& room,
 313         const std::string& speaker,
 314         const std::string& message)
 315 {
 316         mp::room_info* ri = lobby_info_->get_room(room);
 317         if(!ri) {
 318                 LOG_LB << "Discarding message to room " << room << " from " << speaker << " (room not open)\n";
 319                 return;
 320         }
 321 
 322         mp::notify_mode notify_mode = mp::NOTIFY_NONE;
 323         ri->log().add_message(speaker, message);
 324 
 325         if(room_window_active(room)) {
 326                 add_active_window_message(speaker, message);
 327                 notify_mode = mp::NOTIFY_MESSAGE;
 328         } else {
 329                 add_room_window_message(room, speaker, message);
 330                 increment_waiting_messages(room);
 331                 notify_mode = mp::NOTIFY_MESSAGE_OTHER_WINDOW;
 332         }
 333 
 334         if(speaker == "server") {
 335                 notify_mode = mp::NOTIFY_SERVER_MESSAGE;
 336         } else if (utils::word_match(message, preferences::login())) {
 337                 notify_mode = mp::NOTIFY_OWN_NICK;
 338         } else if (preferences::is_friend(speaker)) {
 339                 notify_mode = mp::NOTIFY_FRIEND_MESSAGE;
 340         }
 341 
 342         do_notify(notify_mode, speaker, message);
 343 }

 367 lobby_chat_window* chatbox::search_create_window(const std::string& name,
 368         const bool whisper,
 369         const bool open_new,
 370         const bool allow_close)
 371 {
 372         for(auto& t : open_windows_) {
 373                 if(t.name == name && t.whisper == whisper) {
 374                         return &t;
 375                 }
 376         }
 377 
 378         if(!open_new) {
 379                 return nullptr;
 380         }
 381 
 382         open_windows_.push_back(lobby_chat_window(name, whisper));
 383 
 384         std::map<std::string, string_map> data;
 385         string_map item;
 386 
 387         item["use_markup"] = "true";
 388 
 389         if(whisper) {
 390                 item["label"] = vgettext("Whisper session with <i>“$name”</i> started. "
 391                         "If you do not want to receive messages from this user, type <i>/ignore $name</i>\n", {{"name", name}});
 392                 data.emplace("log_text", item);
 393         } else {
 394                 item["label"] = vgettext("Room <i>“$name”</i> joined", {{"name", name}});
 395                 data.emplace("log_text", item);
 396 
 397                 lobby_info_->open_room(name);
 398         }
 399 
 400         chat_log_container_->add_page(data);
 401 
 402         data.clear();
 403 
 404         item["label"] = whisper ? font::escape_text("<" + name + ">") : name;
 405         data.emplace("room", item);
 406 
 407         grid& row_grid = roomlistbox_->add_row(data);
 408 
 409         button& close_button = find_widget<button>(&row_grid, "close_window", false);
 410         connect_signal_mouse_left_click(close_button,
 411                 std::bind(&chatbox::close_window_button_callback, this, open_windows_.back(), _3, _4));
 412 
 413         if(!allow_close) {
 414                 close_button.set_visible(widget::visibility::hidden);
 415         }
 416 
 417         return &open_windows_.back();
 418 }

 492 void chatbox::close_window(size_t idx)
 493 {
 494         const lobby_chat_window& t = open_windows_[idx];
 495         bool active_changed = idx == active_window_;
 496         DBG_LB << "Close window " << idx << " - " << t.name << "\n";
 497 
 498         if((t.name == "lobby" && t.whisper == false) || open_windows_.size() == 1) {
 499                 return;
 500         }
 501 
 502         if(t.whisper == false) {
 503                 // closing a room window -- send a part to the server
 504                 ::config data, msg;
 505                 msg["room"] = t.name;
 506                 msg["player"] = preferences::login();
 507                 data.add_child("room_part", msg);
 508                 send_to_server(data);
 509         }
 510 
 511         if(active_window_ == open_windows_.size() - 1) {
 512                 active_window_--;
 513         }
 514 
 515         if(t.whisper) {
 516                 lobby_info_->get_whisper_log(t.name).clear();
 517         } else {
 518                 lobby_info_->close_room(t.name);
 519         }
 520 
 521         open_windows_.erase(open_windows_.begin() + idx);
 522 
 523         roomlistbox_->remove_row(idx);
 524         roomlistbox_->select_row(active_window_);
 525 
 526         chat_log_container_->remove_page(idx);
 527         chat_log_container_->select_page(active_window_);
 528 
 529         if(active_changed) {
 530                 active_window_changed();
 531         }
 532 }

 567 void chatbox::process_room_join(const ::config& data)
 568 {
 569         const std::string& room = data["room"];
 570         const std::string& player = data["player"];
 571 
 572         DBG_LB << "room join: " << room << " " << player << "\n";
 573 
 574         mp::room_info* r = lobby_info_->get_room(room);
 575         if(r) {
 576                 if(player == preferences::login()) {
 577                         if(const auto& members = data.child("members")) {
 578                                 r->process_room_members(members);
 579                         }
 580                 } else {
 581                         r->add_member(player);
 582 
 583                         /* TODO: add/use preference */
 584                         add_room_window_message(room, "server", vgettext("$player has entered the room", {{"player", player}}));
 585                 }
 586 
 587                 if(r == active_window_room()) {
 588                         active_window_changed_callback_();
 589                 }
 590         } else {
 591                 if(player == preferences::login()) {
 592                         lobby_chat_window* t = room_window_open(room, true);
 593                         lobby_info_->open_room(room);
 594                         r = lobby_info_->get_room(room);
 595                         assert(r);
 596                         if(const auto& members = data.child("members")) {
 597                                 r->process_room_members(members);
 598                         }
 599                         switch_to_window(t);
 600 
 601                         const std::string& topic = data["topic"];
 602                         if(!topic.empty()) {
 603                                 add_chat_room_message_received("room", "server", room + ": " + topic);
 604                         }
 605                 } else {
 606                         LOG_LB << "Discarding join info for a room the player is not in\n";
 607                 }
 608         }
 609 }

 612 {
 613         // TODO: close room window when the part message is sent
 614         const std::string& room = data["room"];
 615         const std::string& player = data["player"];
 616 
 617         DBG_LB << "Room part: " << room << " " << player << "\n";

 619         if(mp::room_info* r = lobby_info_->get_room(room)) {
 620                 r->remove_member(player);
 621 
 622                 /* TODO: add/use preference */
 623                 add_room_window_message(room, "server", vgettext("$player has left the room", {{"player", player}}));
 624                 if(active_window_room() == r) {
 625                         active_window_changed_callback_();
 626                 }
 627         } else {
 628                 LOG_LB << "Discarding part info for a room the player is not in\n";
 629         }
 630 }

 632 void chatbox::process_room_query_response(const ::config& data)
 633 {
 634         const std::string& room = data["room"];
 635         const std::string& message = data["message"];
 636 
 637         DBG_LB << "room query response: " << room << " " << message << "\n";
 638 
 639         if(room.empty()) {
 640                 if(!message.empty()) {
 641                         add_active_window_message("server", message);
 642                 }
 643 
 644                 if(const ::config& rooms = data.child("rooms")) {
 645                         // TODO: this should really open a nice join room dialog instead
 646                         std::stringstream ss;
 647                         ss << "Rooms:";
 648 
 649                         for(const auto & r : rooms.child_range("room")) {
 650                                 ss << " " << r["name"];
 651                         }
 652 
 653                         add_active_window_message("server", ss.str());
 654                 }
 655         } else {
 656                 if(room_window_open(room, false)) {
 657                         if(!message.empty()) {
 658                                 add_chat_room_message_received(room, "server", message);
 659                         }
 660 
 661                         if(const ::config& members = data.child("members")) {
 662                                 mp::room_info* r = lobby_info_->get_room(room);
 663                                 assert(r);
 664                                 r->process_room_members(members);
 665                                 if(r == active_window_room()) {
 666                                         active_window_changed_callback_();
 667                                 }
 668                         }
 669                 } else {
 670                         if(!message.empty()) {
 671                                 add_active_window_message("server", room + ": " + message);
 672                         }
 673                 }
 674         }
 675 }

 677 void chatbox::process_message(const ::config& data, bool whisper /*= false*/)
 678 {
 679         std::string sender = data["sender"];
 680         DBG_LB << "process message from " << sender << " " << (whisper ? "(w)" : "")
 681                 << ", len " << data["message"].str().size() << '\n';
 682 
 683         if(preferences::is_ignored(sender)) {
 684                 return;
 685         }
 686 
 687         const std::string& message = data["message"];
 688         preferences::parse_admin_authentication(sender, message);
 689 
 690         if(whisper) {
 691                 add_whisper_received(sender, message);
 692         } else {
 693                 std::string room = data["room"];
 694                 if(room.empty()) {
 695                         LOG_LB << "Message without a room from " << sender << ", assuming lobby\n";
 696                         room = "lobby";
 697                 }
 698 
 699                 add_chat_room_message_received(room, sender, message);
 700         }
 701 
 702         // Notify plugins about the message
 703         ::config plugin_data = data;
 704         plugin_data["whisper"] = whisper;
 705         plugins_manager::get()->notify_event("chat", plugin_data);
 706 }

 708 bool chatbox::process_network_data(const ::config& data)
 709 {
 710         if(const ::config& message = data.child("message")) {
 711                 process_message(message);
 712         } else if(const ::config& whisper = data.child("whisper")) {
 713                 process_message(whisper, true);
 714         } else if(const ::config& room_join = data.child("room_join")) {
 715                 process_room_join(room_join);
 716         } else if(const ::config& room_part = data.child("room_part")) {
 717                 process_room_part(room_part);
 718         } else if(const ::config& room_query_response = data.child("room_query_response")) {
 719                 process_room_query_response(room_query_response);
 720         }
 721 
 722         return false;
 723 }
