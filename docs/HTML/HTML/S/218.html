  31         void resolve_deps_recursive(const addons_list& addons, const std::string& base_id, std::set<std::string>& dest)
  32         {
  33                 addons_list::const_iterator it = addons.find(base_id);
  34                 if(it == addons.end()) {
  35                         LOG_AC << "resolve_deps_recursive(): " << base_id << " not in add-ons list\n";
  36                         return;
  37                 }
  38 
  39                 const std::vector<std::string>& base_deps = it->second.depends;
  40 
  41                 if(base_deps.empty()) {
  42                         return;
  43                 }
  44 
  45                 for(const std::string& dep : base_deps) {
  46                         if(base_id == dep) {
  47                                 LOG_AC << dep << " depends upon itself; breaking circular dependency\n";
  48                                 continue;
  49                         } else if(dest.find(dep) != dest.end()) {
  50                                 LOG_AC << dep << " already in dependency tree; breaking circular dependency\n";
  51                                 continue;
  52                         }
  53 
  54                         dest.insert(dep);
  55 
  56                         resolve_deps_recursive(addons, dep, dest);
  57                 }
  58         }
  59 }

  62 {

  72         this->type = get_addon_type(cfg["type"].str());
  73 
  74         const config::const_child_itors& locales_as_configs = cfg.child_range("translation");
  75 
  76         for(const config& locale : locales_as_configs) {
  77                 this->locales.push_back(locale["language"].str());
  78         }
  79 
  80         this->core = cfg["core"].str();
  81         this->depends = utils::split(cfg["dependencies"].str());
  82         this->feedback_url = cfg["feedback_url"].str();
  83 
  84         this->updated = cfg["timestamp"].to_time_t();
  85         this->created = cfg["original_timestamp"].to_time_t();

 132 std::string addon_info::display_icon() const
 133 {
 134         std::string ret = icon;
 135 
 136         if(ret.empty()) {
 137                 ERR_AC << "add-on '" << id << "' doesn't have an icon path set" << std::endl;
 138         } else if(!image::exists(ret)) {
 139                 ERR_AC << "add-on '" << id << "' has an icon which cannot be found: '" << ret << "'" << std::endl;
 140         } else if(ret.find("units/") != std::string::npos && ret.find_first_of('~') == std::string::npos) {
 141                 // HACK: prevent magenta icons, because they look awful
 142                 LOG_AC << "add-on '" << id << "' uses a unit baseframe as icon without TC/RC specifications\n";
 143                 ret += "~RC(magenta>red)";
 144         }
 145 
 146         return ret;
 147 }

 181 std::set<std::string> addon_info::resolve_dependencies(const addons_list& addons) const
 182 {
 183         std::set<std::string> deps;
 184         resolve_deps_recursive(addons, this->id, deps);
 185 
 186         if(deps.find(this->id) != deps.end()) {
 187                 LOG_AC << this->id << " depends upon itself; breaking circular dependency\n";
 188                 deps.erase(this->id);
 189         }
 190 
 191         return deps;
 192 }

 194 void read_addons_list(const config& cfg, addons_list& dest)
 195 {
 196         dest.clear();
 197 
 198         unsigned order = 0;
 199 
 200         /** @todo FIXME: get rid of this legacy "campaign"/"campaigns" silliness
 201          */
 202         const config::const_child_itors &addon_cfgs = cfg.child_range("campaign");
 203         for(const config& addon_cfg : addon_cfgs) {
 204                 const std::string& id = addon_cfg["name"].str();
 205                 if(dest.find(id) != dest.end()) {
 206                         ERR_AC << "add-ons list has multiple entries for '" << id << "', not good; ignoring them" << std::endl;
 207                         continue;
 208                 }
 209                 dest[id].read(addon_cfg);
 210                 dest[id].order = order++;
 211         }
 212 }
