  31 /**
  32  * Registers a window.
  33  *
  34  * This function registers a window. The registration is used to validate
  35  * whether the config for the window exists when starting Wesnoth.
  36  *
  37  * @note Most of the time you want to call @ref REGISTER_DIALOG instead of this
  38  * function. It also directly adds the code for the dialog's id function.
  39  *
  40  * @param id                      Id of the window, multiple dialogs can use
  41  *                                the same window so the id doesn't need to be
  42  *                                unique.
  43  */
  44 #define REGISTER_WINDOW(id)                                                    \
  45         namespace                                                                  \
  46         {                                                                          \
  47                                                                                \
  48         namespace ns_##id                                                          \
  49         {                                                                          \
  50                                                                                \
  51                 struct register_helper                                                \
  52                 {                                                                      \
  53                         register_helper()                                                 \
  54                         {                                                                  \
  55                                 register_window(#id);                                          \
  56                         }                                                                  \
  57                 };                                                                     \
  58                                                                                \
  59                 struct register_helper register_helper;                                \
  60         }                                                                          \
  61         }

  63 /**
  64  * Registers a window for a dialog.
  65  *
  66  * Call this function to register a window. In the header of the class it adds
  67  * the following code:
  68  *@code
  69  *  // Inherited from modal_dialog, implemented by REGISTER_DIALOG.
  70  *      virtual const std::string& id() const;
  71  *@endcode
  72  * Then use this macro in the implementation, inside the gui2 namespace.
  73  *
  74  * @note When the @p id is "foo" and the type tfoo it's easier to use
  75  * REGISTER_DIALOG(foo).
  76  *
  77  * @param type                    Class type of the window to register.
  78  * @param id                      Id of the window, multiple dialogs can use
  79  *                                the same window so the id doesn't need to be
  80  *                                unique.
  81  */
  82 #define REGISTER_DIALOG2(type, id)                                             \
  83         REGISTER_WINDOW(id) const std::string& type::window_id() const             \
  84         {                                                                          \
  85                 static const std::string result(#id);                                  \
  86                 return result;                                                         \
  87         }
  88 
  89 /**
  90  * Wrapper for REGISTER_DIALOG2.
  91  *
  92  * "Calls" REGISTER_DIALOG2(window_id, window_id)
  93  */
  94 #define REGISTER_DIALOG(window_id) REGISTER_DIALOG2(window_id, window_id)

  96 /**
  97  * Abstract base class for all modal dialogs.
  98  *
  99  * A dialog shows a certain window instance to the user. The subclasses of this
 100  * class will hold the parameters used for a certain window, eg a server
 101  * connection dialog will hold the name of the selected server as parameter that
 102  * way the caller doesn't need to know about the 'contents' of the window.
 103  *
 104  * @par Usage
 105  *
 106  * Simple dialogs that are shown to query user information it is recommended to
 107  * add a static member called @p execute. The parameters to the function are:
 108  * - references to in + out parameters by reference
 109  * - references to the in parameters
 110  * - the parameters for @ref modal_dialog::show.
 111  *
 112  * The 'in + out parameters' are used as initial value and final value when the
 113  * OK button is pressed. The 'in parameters' are just extra parameters for
 114  * showing.
 115  *
 116  * When a function only has 'in parameters' it should return a void value and
 117  * the function should be called @p display, if it has 'in + out parameters' it
 118  * must return a bool value. This value indicates whether or not the OK button
 119  * was pressed to close the dialog. See @ref editor_new_map::execute for an
 120  * example.
 121  */
