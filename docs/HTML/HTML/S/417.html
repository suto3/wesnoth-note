 215 {
 216         for(const auto& v : map) {
 217                 const std::string& key = v.first;
 218                 const string_map& strmap = v.second;

 220                 widget* w = grid->find(key, false);
 221                 if(!w) {
 222                         continue;
 223                 }
 224 
 225                 styled_widget* c = dynamic_cast<styled_widget*>(w);
 226                 if(!c) {
 227                         continue;
 228                 }
 229 
 230                 for(const auto & vv : strmap) {
 231                         if(vv.first == "label") {
 232                                 c->set_label(vv.second);
 233                         } else if(vv.first == "tooltip") {
 234                                 c->set_tooltip(vv.second);
 235                         }
 236                 }
 237         }

 254 void mp_lobby::update_gamelist()
 255 {
 256         SCOPE_LB;
 257         gamelistbox_->clear();
 258         gamelist_id_at_row_.clear();
 259         lobby_info_.make_games_vector();
 260 
 261         int select_row = -1;
 262         for(unsigned i = 0; i < lobby_info_.games().size(); ++i) {
 263                 const mp::game_info& game = *lobby_info_.games()[i];
 264 
 265                 if(game.id == selected_game_id_) {
 266                         select_row = i;
 267                 }
 268 
 269                 gamelist_id_at_row_.push_back(game.id);
 270                 LOG_LB << "Adding game to listbox (1)" << game.id << "\n";
 271                 grid* grid = &gamelistbox_->add_row(make_game_row_data(game));
 272 
 273                 adjust_game_row_contents(game, gamelistbox_->get_item_count() - 1, grid);
 274         }
 275 
 276         if(select_row >= 0 && select_row != gamelistbox_->get_selected_row()) {
 277                 gamelistbox_->select_row(select_row);
 278         }
 279 
 280         update_selected_game();
 281         gamelist_dirty_ = false;
 282         last_gamelist_update_ = SDL_GetTicks();
 283         lobby_info_.sync_games_display_status();
 284         lobby_info_.apply_game_filter();
 285         update_gamelist_header();
 286         gamelistbox_->set_row_shown(lobby_info_.games_visibility());
 287 }

 289 void mp_lobby::update_gamelist_diff()
 290 {
 291         SCOPE_LB;
 292         lobby_info_.make_games_vector();
 293         int select_row = -1;
 294         unsigned list_i = 0;
 295         int list_rows_deleted = 0;
 296 
 297         std::vector<int> next_gamelist_id_at_row;
 298         for(unsigned i = 0; i < lobby_info_.games().size(); ++i) {
 299                 const mp::game_info& game = *lobby_info_.games()[i];
 300 
 301                 if(game.display_status == mp::game_info::NEW) {
 302                         LOG_LB << "Adding game to listbox " << game.id << "\n";
 303 
 304                         if(list_i != gamelistbox_->get_item_count()) {
 305                                 gamelistbox_->add_row(make_game_row_data(game), list_i);
 306                                 DBG_LB << "Added a game listbox row not at the end" << list_i
 307                                            << " " << gamelistbox_->get_item_count() << "\n";
 308                                 list_rows_deleted--;
 309                         } else {
 310                                 gamelistbox_->add_row(make_game_row_data(game));
 311                         }
 312 
 313                         grid* grid = gamelistbox_->get_row_grid(gamelistbox_->get_item_count() - 1);
 314                         adjust_game_row_contents(game, gamelistbox_->get_item_count() - 1, grid);
 315 
 316                         list_i++;
 317                         next_gamelist_id_at_row.push_back(game.id);
 318                 } else {
 319                         if(list_i >= gamelistbox_->get_item_count()) {
 320                                 ERR_LB << "Ran out of listbox items -- triggering a full refresh\n";
 321                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 322                                 return;
 323                         }
 324 
 325                         if(list_i + list_rows_deleted >= gamelist_id_at_row_.size()) {
 326                                 ERR_LB << "gamelist_id_at_row_ overflow! " << list_i << " + "
 327                                            << list_rows_deleted
 328                                            << " >= " << gamelist_id_at_row_.size()
 329                                            << " -- triggering a full refresh\n";
 330                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 331                                 return;
 332                         }
 333 
 334                         int listbox_game_id = gamelist_id_at_row_[list_i + list_rows_deleted];
 335                         if(game.id != listbox_game_id) {
 336                                 ERR_LB << "Listbox game id does not match expected id "
 337                                            << listbox_game_id << " " << game.id << " (row " << list_i << ")\n";
 338                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 339                                 return;
 340                         }
 341 
 342                         if(game.display_status == mp::game_info::UPDATED) {
 343                                 LOG_LB << "Modifying game in listbox " << game.id << " (row " << list_i << ")\n";
 344                                 grid* grid = gamelistbox_->get_row_grid(list_i);
 345                                 modify_grid_with_data(grid, make_game_row_data(game));
 346                                 adjust_game_row_contents(game, list_i, grid, false);
 347                                 ++list_i;
 348                                 next_gamelist_id_at_row.push_back(game.id);
 349                         } else if(game.display_status == mp::game_info::DELETED) {
 350                                 LOG_LB << "Deleting game from listbox " << game.id << " (row "
 351                                            << list_i << ")\n";
 352                                 gamelistbox_->remove_row(list_i);
 353                                 ++list_rows_deleted;
 354                         } else {
 355                                 // clean
 356                                 LOG_LB << "Clean game in listbox " << game.id << " (row " << list_i << ")\n";
 357                                 next_gamelist_id_at_row.push_back(game.id);
 358                                 ++list_i;
 359                         }
 360                 }
 361         }
 362 
 363         for(unsigned i = 0; i < next_gamelist_id_at_row.size(); ++i) {
 364                 if(next_gamelist_id_at_row[i] == selected_game_id_) {
 365                         select_row = i;
 366                 }
 367         }
 368 
 369         next_gamelist_id_at_row.swap(gamelist_id_at_row_);
 370         if(select_row >= static_cast<int>(gamelistbox_->get_item_count())) {
 371                 ERR_LB << "Would select a row beyond the listbox" << select_row << " "
 372                            << gamelistbox_->get_item_count() << "\n";
 373                 select_row = gamelistbox_->get_item_count() - 1;
 374         }
 375 
 376         if(select_row >= 0 && select_row != gamelistbox_->get_selected_row()) {
 377                 gamelistbox_->select_row(select_row);
 378         }
 379 
 380         update_selected_game();
 381         gamelist_dirty_ = false;
 382         last_gamelist_update_ = SDL_GetTicks();
 383         lobby_info_.sync_games_display_status();
 384         lobby_info_.apply_game_filter();
 385         update_gamelist_header();
 386         gamelistbox_->set_row_shown(lobby_info_.games_visibility());
 387 }

 401 std::map<std::string, string_map> mp_lobby::make_game_row_data(const mp::game_info& game)
 402 {
 403         std::map<std::string, string_map> data;
 404 
 405         std::string color_string;
 406         if(game.vacant_slots > 0) {
 407                 color_string = (game.reloaded || game.started) ? "yellow" : "green";
 408         } else {
 409                 color_string = game.observers ? "#ddd" : "red";
 410         }
 411 
 412         if(!game.have_era && (game.vacant_slots > 0 || game.observers)) {
 413                 color_string = "#444";
 414         }
 415 
 416         add_label_data(data, "status", colorize(game.status, color_string));
 417         add_label_data(data, "name",   colorize(game.name, color_string));
 418 
 419         add_label_data(data, "era",             colorize(game.era, "#a69275"));
 420         add_label_data(data, "era_short",       game.era_short);
 421         add_label_data(data, "mods",            colorize(game.mod_info, "#a69275"));
 422         add_label_data(data, "map_info",        game.map_info);
 423         add_label_data(data, "scenario",        game.scenario);
 424         add_label_data(data, "map_size_text",   game.map_size_info);
 425         add_label_data(data, "time_limit",      game.time_limit);
 426         add_label_data(data, "gold_text",       game.gold);
 427         add_label_data(data, "xp_text",         game.xp);
 428         add_label_data(data, "vision_text",     game.vision);
 429         add_label_data(data, "time_limit_text", game.time_limit);
 430         add_label_data(data, "status",          game.status);
 431 
 432         if(game.observers) {
 433                 add_label_data(data, "observer_icon", "misc/eye.png");
 434                 add_tooltip_data(data, "observer_icon", _("Observers allowed"));
 435         } else {
 436                 add_label_data(data, "observer_icon", "misc/no_observer.png");
 437                 add_tooltip_data(data, "observer_icon", _("Observers not allowed"));
 438         }
 439 
 440         std::string vision_icon;
 441         if(game.fog) {
 442                 vision_icon = game.shroud ? "misc/vision-fog-shroud.png" : "misc/vision-fog.png";
 443         } else {
 444                 vision_icon = game.shroud ? "misc/vision-shroud.png" : "misc/vision-none.png";
 445         }
 446 
 447         add_label_data(data, "vision_icon", vision_icon);
 448         add_tooltip_data(data, "vision_icon", game.vision);
 449         return data;
 450 }

 452 void mp_lobby::adjust_game_row_contents(const mp::game_info& game,
 453                                                                                    int idx,

 455 {
 456         find_widget<styled_widget>(grid, "name", false).set_use_markup(true);

 468         set_visible_if_exists(grid, "shuffle_sides_icon", game.shuffle_sides);

 474         set_visible_if_exists(grid, "registered_only",    game.registered_users_only);

 480         }
 481 
 482         if(!add_callbacks) {
 483                 return;
 484         }
 485 
 486         if(button* join_button = dynamic_cast<button*>(grid->find("join", false))) {
 487                 connect_signal_mouse_left_click(
 488                                 *join_button,
 489                                 std::bind(&mp_lobby::join_global_button_callback,
 490                                                         this,
 491                                                         std::ref(*get_window())));
 492                 join_button->set_active(game.can_join());
 493         }
 494 
 495         if(button* observe_button = dynamic_cast<button*>(grid->find("observe", false))) {
 496                 connect_signal_mouse_left_click(
 497                                 *observe_button,
 498                                 std::bind(&mp_lobby::observe_global_button_callback,
 499                                                         this,
 500                                                         std::ref(*get_window())));
 501                 observe_button->set_active(game.can_observe());
 502         }
 503 
 504 }

 518 void mp_lobby::update_playerlist()
 519 {
 520         if(delay_playerlist_update_) {
 521                 return;
 522         }
 523 
 524         SCOPE_LB;
 525         DBG_LB << "Playerlist update: " << lobby_info_.users().size() << "\n";
 526         lobby_info_.update_user_statuses(selected_game_id_, chatbox_->active_window_room());
 527         lobby_info_.sort_users(player_list_.sort_by_name->get_value_bool(),
 528                                                    player_list_.sort_by_relation->get_value_bool());
 529 
 530         bool lobby = false;
 531         if(mp::room_info* ri = chatbox_->active_window_room()) {
 532                 if(ri->name() == "lobby") {
 533                         lobby = true;
 534                 }
 535         }
 536 
 537         assert(player_list_.active_game.tree);
 538         assert(player_list_.active_room.tree);
 539         assert(player_list_.other_games.tree);
 540         assert(player_list_.other_rooms.tree);
 541 
 542         player_list_.active_game.tree->clear();
 543         player_list_.active_room.tree->clear();
 544         player_list_.other_games.tree->clear();
 545         player_list_.other_rooms.tree->clear();
 546

 803 void mp_lobby::network_handler()
 804 {
 805         try {
 806                 config data;
 807                 if (wesnothd_connection_.receive_data(data)) {
 808                         process_network_data(data);
 809                 }
 810         } catch (wesnothd_error& e) {
 811                 LOG_LB << "caught wesnothd_error in network_handler: " << e.message << "\n";
 812                 throw;
 813         }
 814 
 815         if(gamelist_dirty_ && !delay_gamelist_update_ && (SDL_GetTicks() - last_gamelist_update_ > game_config::lobby_refresh)) {
 816                 if(gamelist_diff_update_) {
 817                         update_gamelist_diff();
 818                 } else {
 819                         update_gamelist();
 820                         gamelist_diff_update_ = true;
 821                 }
 822         }
 823 
 824         if(player_list_dirty_) {
 825                 update_gamelist_filter();
 826                 update_playerlist();
 827         }
 828 }

 831 {

 837                 process_gamelist_diff(gamelist_diff);
 838         }

 840         chatbox_->process_network_data(data);

 864                         do_notify(mp::NOTIFY_LOBBY_QUIT);

 893 static bool handle_addon_requirements_gui(CVideo& v, const std::vector<mp::game_info::required_addon>& reqs, mp::game_info::ADDON_REQ addon_outcome)
 894 {
 895         if(addon_outcome == mp::game_info::CANNOT_SATISFY) {
 896                 std::string e_title = _("Incompatible User-made Content.");
 897                 std::string err_msg = _("This game cannot be joined because the host has out-of-date add-ons that are incompatible with your version. You might wish to suggest that the host's add-ons be updated.");
 898 
 899                 err_msg +="\n\n";
 900                 err_msg += _("Details:");
 901                 err_msg += "\n";
 902 
 903                 for(const mp::game_info::required_addon & a : reqs) {
 904                         if (a.outcome == mp::game_info::CANNOT_SATISFY) {
 905                                 err_msg += font::unicode_bullet + " " + a.message + "\n";
 906                         }
 907                 }
 908                 gui2::show_message(v, e_title, err_msg, message::auto_close);
 909 
 910                 return false;
 911         } else if(addon_outcome == mp::game_info::NEED_DOWNLOAD) {
 912                 std::string e_title = _("Missing User-made Content.");
 913                 std::string err_msg = _("This game requires one or more user-made addons to be installed or updated in order to join.\nDo you want to try to install them?");
 914 
 915                 err_msg +="\n\n";
 916                 err_msg += _("Details:");
 917                 err_msg += "\n";
 918 
 919                 std::vector<std::string> needs_download;
 920                 for(const mp::game_info::required_addon & a : reqs) {
 921                         if(a.outcome == mp::game_info::NEED_DOWNLOAD) {
 922                                 err_msg += font::unicode_bullet + " " + a.message + "\n";
 923 
 924                                 needs_download.push_back(a.addon_id);
 925                         }
 926                 }
 927 
 928                 assert(needs_download.size() > 0);
 929 
 930                 if(gui2::show_message(v, e_title, err_msg, message::yes_no_buttons, true) == gui2::window::OK) {
 931                         // Begin download session
 932                         ad_hoc_addon_fetch_session(v, needs_download);
 933 
 934                         return true;
 935                 }
 936         }
 937 
 938         return false;
 939 }

 941 bool mp_lobby::do_game_join(int idx, bool observe)
 942 {
 943         if(idx < 0 || idx >= static_cast<int>(lobby_info_.games().size())) {
 944                 ERR_LB << "Requested join/observe of a game with index out of range: "
 945                            << idx << ", games size is " << lobby_info_.games().size() << "\n";
 946                 return false;
 947         }
 948         const mp::game_info& game = *lobby_info_.games()[idx];
 949         if(observe) {
 950                 if(!game.can_observe()) {
 951                         ERR_LB << "Requested observe of a game with observers disabled" << std::endl;
 952                         return false;
 953                 }
 954         } else {
 955                 if(!game.can_join()) {
 956                         ERR_LB << "Requested join to a game with no vacant slots" << std::endl;
 957                         return false;
 958                 }
 959         }
 960 
 961         // Prompt user to download this game's required addons if its requirements have not been met
 962         if(game.addons_outcome != mp::game_info::SATISFIED) {
 963                 if(game.required_addons.empty()) {
 964                         gui2::show_error_message(get_window()->video(), _("Something is wrong with the addon version check database supporting the multiplayer lobby. Please report this at http://bugs.wesnoth.org."));
 965                         return false;
 966                 }
 967 
 968                 if(!handle_addon_requirements_gui(get_window()->video(), game.required_addons, game.addons_outcome)) {
 969                         return false;
 970                 } else {
 971                         // Addons have been downloaded, so the game_config and installed addons list need to be reloaded.
 972                         // The lobby is closed and reopened.
 973                         get_window()->set_retval(RELOAD_CONFIG);
 974                         return false;
 975                 }
 976         }
 977 
 978         config response;
 979         config& join = response.add_child("join");
 980         join["id"] = std::to_string(game.id);
 981         join["observe"] = observe;
 982         if(join && !observe && game.password_required) {
 983                 std::string password;
 984                 if(!gui2::dialogs::mp_join_game_password_prompt::execute(password, get_window()->video())) {
 985                         return false;
 986                 }
 987 
 988                 join["password"] = password;
 989         }
 990 
 991         wesnothd_connection_.send_data(response);
 992         joined_game_id_ = game.id;
 993         return true;
 994 }
