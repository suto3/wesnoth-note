  60 unsigned toggle_button::num_states() const
  61 {
  62         std::div_t res = std::div(this->config()->state.size(), COUNT);
  63         assert(res.rem == 0);
  64         assert(res.quot > 0);
  65         return res.quot;
  66 }

 113 void toggle_button::set_value(const unsigned selected)
 114 {
 115         if(selected == get_value()) {
 116                 return;
 117         }
 118         state_num_ = selected % num_states();
 119         set_is_dirty(true);
 120 
 121         // Check for get_window() is here to prevent the callback from
 122         // being called when the initial value is set.
 123         if(!get_window()) {
 124                 return;
 125         }
 126 
 127         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 128 }

 130 void toggle_button::set_retval(const int retval)

 164 void toggle_button::signal_handler_left_button_click(const event::ui_event event,
 165                                                                                                           bool& handled)
 166 {
 167         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 168 
 169         sound::play_UI_sound(settings::sound_toggle_button_click);
 170 
 171         set_value(get_value() + 1);
 172 
 173         handled = true;
 174 }
 175 
 176 void toggle_button::signal_handler_left_button_double_click(
 177                 const event::ui_event event, bool& handled)
 178 {
 179         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 180 
 181         if(retval_ == 0) {
 182                 return;
 183         }
 184 
 185         window* window = get_window();
 186         assert(window);
 187 
 188         window->set_retval(retval_);
 189 
 190         handled = true;
 191 }

 212  * * state_enabled, the button is enabled and not selected.

 217  * * state_disabled_selected, the button is disabled and selected.
 218  * * state_focused_selected, the mouse is over the button and selected.
 219  * @begin{parent}{name="gui/"}

 222  * @begin{tag}{name="state_enabled"}{min=0}{max=1}{super="generic/state"}

 225  * @end{tag}{name="state_disabled"}
 226  * @begin{tag}{name="state_focused"}{min=0}{max=1}{super="generic/state"}
 227  * @end{tag}{name="state_focused"}

 238 toggle_button_definition::resolution::resolution(const config& cfg)
 239         : resolution_definition(cfg)
 240 {
 241         // Note the order should be the same as the enum state_t in
 242         // toggle_button.hpp.
 243         for(const auto& c : cfg.child_range("state"))
 244         {
 245                 state.emplace_back(c.child("enabled"));
 246                 state.emplace_back(c.child("disabled"));
 247                 state.emplace_back(c.child("focused"));
 248         }
 249 }
