 125 bool affects_side(const config& cfg, const std::vector<team>& teams, size_t side, size_t other_side)
 126 {
 127         if (side == other_side)
 128                 return cfg["affect_allies"].to_bool(true);
 129         if (teams[side - 1].is_enemy(other_side))
 130                 return cfg["affect_enemies"].to_bool();
 131         else
 132                 return cfg["affect_allies"].to_bool();

 138 bool unit::get_ability_bool(const std::string& tag_name, const map_location& loc, const display_context& dc) const
 139 {
 140         for (const config &i : this->abilities_.child_range(tag_name)) {
 141                 if (ability_active(tag_name, i, loc) &&
 142                         ability_affects_self(tag_name, i, loc))
 143                 {
 144                         return true;
 145                 }
 146         }
 147 
 148         assert(resources::units);
 149         const unit_map& units = *resources::units;
 150 
 151         map_location adjacent[6];
 152         get_adjacent_tiles(loc,adjacent);
 153         for(int i = 0; i != 6; ++i) {
 154                 const unit_map::const_iterator it = units.find(adjacent[i]);
 155                 if (it == units.end() || it->incapacitated())
 156                         continue;
 157                 // Abilities may be tested at locations other than the unit's current
 158                 // location. This is intentional to allow for less messing with the unit
 159                 // map during calculations, particularly with regards to movement.
 160                 // Thus, we need to make sure the adjacent unit (*it) is not actually
 161                 // ourself.
 162                 if ( &*it == this )
 163                         continue;
 164                 for (const config &j : it->abilities_.child_range(tag_name)) {
 165                         if (affects_side(j, dc.teams(), side(), it->side()) &&
 166                             it->ability_active(tag_name, j, adjacent[i]) &&
 167                             ability_affects_adjacent(tag_name,  j, i, loc, *it))
 168                         {
 169                                 return true;
 170                         }
 171                 }
 172         }
 173 
 174 
 175         return false;
 176 }

 304 bool unit::ability_active(const std::string& ability,const config& cfg,const map_location& loc) const
 305 {
 306         bool illuminates = ability == "illuminates";
 307 
 308         if (const config &afilter = cfg.child("filter"))
 309                 if ( !unit_filter(vconfig(afilter), resources::filter_con).set_use_flat_tod(illuminates).matches(*this, loc) )
 310                         return false;
 311 
 312         map_location adjacent[6];
 313         get_adjacent_tiles(loc,adjacent);
 314 
 315         assert(resources::units);
 316         const unit_map& units = *resources::units;
 317 
 318         for (const config &i : cfg.child_range("filter_adjacent"))
 319         {
 320                 size_t count = 0;
 321                 unit_filter ufilt(vconfig(i), resources::filter_con);
 322                 ufilt.set_use_flat_tod(illuminates);
 323                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 324                 for (const map_location::DIRECTION index : dirs)
 325                 {
 326                         if (index == map_location::NDIRECTIONS)
 327                                 continue;
 328                         unit_map::const_iterator unit = units.find(adjacent[index]);
 329                         if (unit == units.end())
 330                                 return false;
 331                         if (!ufilt(*unit, *this))
 332                                 return false;
 333                         if (i.has_attribute("is_enemy")) {
 334                                 const display_context& dc = resources::filter_con->get_disp_context();
 335                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(side_)) {
 336                                         continue;
 337                                 }
 338                         }
 339                         count++;
 340                 }
 341                 if (i["count"].empty() && count != dirs.size()) {
 342                         return false;
 343                 }
 344                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 345                         return false;
 346                 }
 347         }
 348 
 349         for (const config &i : cfg.child_range("filter_adjacent_location"))
 350         {
 351                 size_t count = 0;
 352                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 353                 adj_filter.flatten(illuminates);
 354 
 355                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 356                 for (const map_location::DIRECTION index : dirs)
 357                 {
 358                         if (index == map_location::NDIRECTIONS) {
 359                                 continue;
 360                         }
 361                         if(!adj_filter.match(adjacent[index])) {
 362                                 return false;
 363                         }
 364                         count++;
 365                 }
 366                 if (i["count"].empty() && count != dirs.size()) {
 367                         return false;
 368                 }
 369                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 370                         return false;
 371                 }
 372         }
 373         return true;
 374 }

 376 bool unit::ability_affects_adjacent(const std::string& ability, const config& cfg,int dir,const map_location& loc,const unit& from) const
 377 {
 378         bool illuminates = ability == "illuminates";
 379 
 380         assert(dir >=0 && dir <= 5);
 381         map_location::DIRECTION direction = static_cast<map_location::DIRECTION>(dir);
 382 
 383         for (const config &i : cfg.child_range("affect_adjacent"))
 384         {
 385                 if (i.has_attribute("adjacent")) { //key adjacent defined
 386                         std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 387                         if (std::find(dirs.begin(), dirs.end(), direction) == dirs.end()) {
 388                                 continue;
 389                         }
 390                 }
 391                 const config &filter = i.child("filter");
 392                 if (!filter || //filter tag given
 393                         unit_filter(vconfig(filter), resources::filter_con).set_use_flat_tod(illuminates).matches(*this, loc, from) ) {
 394                         return true;
 395                 }
 396         }
 397         return false;
 398 }

 416 template<typename T, typename TFuncFormula>
 417 class get_ability_value_visitor : public boost::static_visitor<T>
 418 {
 419 public:
 420         // Constructor stores the default value.
 421         get_ability_value_visitor(T def, const TFuncFormula& formula_handler) : def_(def), formula_handler_(formula_handler) {}
 422 
 423         T operator()(boost::blank const &) const { return def_; }
 424         T operator()(bool)                 const { return def_; }
 425         T operator()(int i)                const { return static_cast<T>(i); }
 426         T operator()(unsigned long long u) const { return static_cast<T>(u); }
 427         T operator()(double d)             const { return static_cast<T>(d); }
 428         T operator()(t_string const &)     const { return def_; }
 429         T operator()(const std::string& s) const
 430         {
 431                 if(s.size() >= 2 && s[0] == '(') {
 432                         return formula_handler_(s);
 433                 }
 434                 return lexical_cast_default<T>(s, def_);
 435         }
 436 
 437 private:
 438         const T def_;
 439         const TFuncFormula& formula_handler_;
 440 };
 441 template<typename T, typename TFuncFormula>
 442 get_ability_value_visitor<T, TFuncFormula> make_get_ability_value_visitor(T def, const TFuncFormula& formula_handler)
 443 {
 444         return get_ability_value_visitor<T, TFuncFormula>(def, formula_handler);
 445 }
 446 template<typename T, typename TFuncFormula>
 447 T get_single_ability_value(const config::attribute_value& v, T def, const map_location& sender_loc, const map_location& receiver_loc, const TFuncFormula& formula_handler)
 448 {
 449         return v.apply_visitor(make_get_ability_value_visitor(def, [&](const std::string& s) {
 450 
 451                         try {
 452                                 assert(resources::units);
 453                                 auto u_itor = resources::units->find(sender_loc);
 454 
 455                                 if(u_itor == resources::units->end()) {
 456                                         return def;
 457                                 }
 458                                 wfl::map_formula_callable callable(std::make_shared<wfl::unit_callable>(*u_itor));
 459                                 u_itor = resources::units->find(receiver_loc);
 460                                 if(u_itor != resources::units->end()) {
 461                                         callable.add("other", wfl::variant(std::make_shared<wfl::unit_callable>(*u_itor)));
 462                                 }
 463                                 return formula_handler(wfl::formula(s, new wfl::gamestate_function_symbol_table), callable);
 464                         } catch(wfl::formula_error& e) {
 465                                 lg::wml_error() << "Formula error in ability or weapon special: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 466                                 return def;
 467                         }
 468         }));
 469 }

 472 template<typename TComp>
 473 std::pair<int,map_location> unit_ability_list::get_extremum(const std::string& key, int def, const TComp& comp) const
 474 {
 475         if ( cfgs_.empty() ) {
 476                 return std::make_pair(def, map_location());
 477         }
 478         // The returned location is the best non-cumulative one, if any,
 479         // the best absolute cumulative one otherwise.
 480         map_location best_loc;
 481         bool only_cumulative = true;
 482         int abs_max = 0;
 483         int flat = 0;
 484         int stack = 0;
 485         for (unit_ability const &p : cfgs_)
 486         {
 487                 int value = get_single_ability_value((*p.first)[key], def, p.second, loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
 488                         return formula.evaluate(callable).as_int();
 489                 });
 490 
 491                 if ((*p.first)["cumulative"].to_bool()) {
 492                         stack += value;
 493                         if (value < 0) value = -value;
 494                         if (only_cumulative && !comp(value, abs_max)) {
 495                                 abs_max = value;
 496                                 best_loc = p.second;
 497                         }
 498                 } else if (only_cumulative || comp(flat, value)) {
 499                         only_cumulative = false;
 500                         flat = value;
 501                         best_loc = p.second;
 502                 }
 503         }
 504         return std::make_pair(flat + stack, best_loc);
 505 }

 540         /**
 541          * Gets the children of @parent (which should be the specials for an
 542          * attack_type) and places the ones whose tag or id= matches @a id into
 543          * @a result.
 544          * If @a just_peeking is set to true, then @a result is not touched;
 545          * instead the return value is used to indicate if any matching children
 546          * were found.
 547          *
 548          * @returns  true if @a just_peeking is true and a match was found;
 549          *           false otherwise.
 550          */
 551         bool get_special_children(std::vector<const config*>& result, const config& parent,
 552                                    const std::string& id, bool just_peeking=false) {
 553                 for (const config::any_child &sp : parent.all_children_range())
 554                 {
 555                         if (sp.key == id || sp.cfg["id"] == id) {
 556                                 if(just_peeking) {
 557                                         return true; // peek succeeded; done
 558                                 } else {
 559                                         result.push_back(&sp.cfg);
 560                                 }
 561                         }
 562                 }
 563                 return false;
 564         }
 565 }

 567 /**
 568  * Returns whether or not @a *this has a special with a tag or id equal to
 569  * @a special. If @a simple_check is set to true, then the check is merely
 570  * for being present. Otherwise (the default), the check is for a special
 571  * active in the current context (see set_specials_context), including
 572  * specials obtained from the opponent's attack.
 573  */
 574 bool attack_type::get_special_bool(const std::string& special, bool simple_check) const
 575 {
 576         {
 577                 std::vector<const config*> list;
 578                 if ( get_special_children(list, specials_, special, simple_check) ) {
 579                         return true;
 580                 }
 581                 // If we make it to here, then either list.empty() or !simple_check.
 582                 // So if the list is not empty, then this is not a simple check and
 583                 // we need to check each special in the list to see if any are active.
 584                 for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 585                         if ( special_active(**i, AFFECT_SELF) ) {
 586                                 return true;
 587                         }
 588                 }
 589         }
 590         // Skip checking the opponent's attack?
 591         if ( simple_check || !other_attack_ ) {
 592                 return false;
 593         }
 594 
 595         std::vector<const config*> list;
 596         get_special_children(list, other_attack_->specials_, special);
 597         for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 598                 if ( other_attack_->special_active(**i, AFFECT_OTHER) ) {
 599                         return true;
 600                 }
 601         }
 602         return false;
 603 }

 610 {
 611         //log_scope("get_specials");
 612         unit_ability_list res(self_loc_);
 613         for (const config &i : specials_.child_range(special)) {
 614                 if ( special_active(i, AFFECT_SELF) )
 615                         res.push_back(unit_ability(&i, self_loc_));
 616         }
 617         if (!other_attack_) return res;
 618         for (const config &i : other_attack_->specials_.child_range(special)) {
 619                 if ( other_attack_->special_active(i, AFFECT_OTHER) )
 620                         res.push_back(unit_ability(&i, other_loc_));
 621         }
 622         return res;
 623 }

 625 /**
 626  * Returns a vector of names and descriptions for the specials of *this.
 627  * Each std::pair in the vector has first = name and second = description.
 628  *
 629  * This uses either the active or inactive name/description for each special,
 630  * based on the current context (see set_specials_context), provided
 631  * @a active_list is not nullptr. Otherwise specials are assumed active.
 632  * If the appropriate name is empty, the special is skipped.
 633  */
 634 std::vector<std::pair<t_string, t_string> > attack_type::special_tooltips(
 635         boost::dynamic_bitset<>* active_list) const
 636 {
 637         //log_scope("special_tooltips");
 638         std::vector<std::pair<t_string, t_string> > res;
 639         if ( active_list )
 640                 active_list->clear();
 641 
 642         for (const config::any_child &sp : specials_.all_children_range())
 643         {
 644                 if ( !active_list || special_active(sp.cfg, AFFECT_EITHER) ) {
 645                         const t_string &name = sp.cfg["name"];
 646                         if (!name.empty()) {
 647                                 res.emplace_back(name, sp.cfg["description"].t_str() );
 648                                 if ( active_list )
 649                                         active_list->push_back(true);
 650                         }
 651                 } else {
 652                         t_string const &name = default_value(sp.cfg, "name_inactive", "name").t_str();
 653                         if (!name.empty()) {
 654                                 res.emplace_back(name, default_value(sp.cfg, "description_inactive", "description").t_str() );
 655                                 active_list->push_back(false);
 656                         }
 657                 }
 658         }
 659         return res;
 660 }

 731 /**
 732  * Calculates the number of attacks this weapon has, considering specials.
 733  * This returns two numbers because of the swarm special. The actual number of
 734  * attacks depends on the unit's health and should be:
 735  *   min_attacks + (max_attacks - min_attacks) * (current hp) / (max hp)
 736  * c.f. swarm_blows()
 737  */
 738 void attack_type::modified_attacks(bool is_backstab, unsigned & min_attacks,
 739                                    unsigned & max_attacks) const
 740 {
 741         // Apply [attacks].
 742         unit_abilities::effect attacks_effect(get_specials("attacks"),
 743                                               num_attacks(), is_backstab);
 744         int attacks_value = attacks_effect.get_composite_value();
 745         if ( attacks_value < 0 ) {
 746                 attacks_value = num_attacks();
 747                 ERR_NG << "negative number of strikes after applying weapon specials" << std::endl;
 748         }
 749 
 750         // Apply [swarm].
 751         unit_ability_list swarm_specials = get_specials("swarm");
 752         if ( !swarm_specials.empty() ) {
 753                 min_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_min").first);
 754                 max_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_max", attacks_value).first);
 755         } else {
 756                 min_attacks = max_attacks = attacks_value;
 757         }
 758 }

 777          * @param[in]  is_attacker  whether or not the unit with the special is the attacker

 784                         return false;

 788                         return true;
 789                 if ( is_attacker  &&  apply_to == "defender" )

 805                 if ( apply_to.empty() )
 806                         return true;

 808                         return true;
 809                 if ( apply_to == "self" )
 810                         return true;
 811                 if ( is_attacker  &&  apply_to == "attacker" )

 818         /**
 819          * Determines if a unit/weapon combination matches the specified child
 820          * (normally a [filter_*] child) of the provided filter.
 821          * @param[in]  un_it      The unit to filter.
 822          * @param[in]  loc        The presumed location of @a un_it.
 823          * @param[in]  weapon     The attack_type to filter.
 824          * @param[in]  filter     The filter containing the child filter to use.
 825          * @param[in]  child_tag  The tag of the child filter to use.
 826          */
 827         static bool special_unit_matches(const unit_map::const_iterator & un_it,
 828                                                                          const unit_map::const_iterator & u2,
 829                                              const map_location & loc,
 830                                              const_attack_ptr weapon,
 831                                              const config & filter,
 832                                                                          const bool for_listing,
 833                                              const std::string & child_tag)
 834         {
 835                 if (for_listing && !loc.valid())
 836                         // The special's context was set to ignore this unit, so assume we pass.
 837                         // (This is used by reports.cpp to show active specials when the
 838                         // opponent is not known. From a player's perspective, the special
 839                         // is active, in that it can be used, even though the player might
 840                         // need to select an appropriate opponent.)
 841                         return true;
 842 
 843                 const config & filter_child = filter.child(child_tag);
 844                 if ( !filter_child )
 845                         // The special does not filter on this unit, so we pass.
 846                         return true;
 847 
 848                 // If the primary unit doesn't exist, there's nothing to match
 849                 if (!un_it.valid()) {
 850                         return false;
 851                 }
 852 
 853                 unit_filter ufilt(vconfig(filter_child), resources::filter_con);
 854 
 855                 // If the other unit doesn't exist, try matching without it
 856                 if (!u2.valid()) {
 857                         return ufilt.matches(*un_it, loc);
 858                 }
 859 
 860                 // Check for a unit match.
 861                 if (!ufilt.matches(*un_it, loc, *u2)) {
 862                         return false;
 863                 }
 864 
 865                 // Check for a weapon match.
 866                 if ( const config & filter_weapon = filter_child.child("filter_weapon") ) {
 867                         if ( !weapon || !weapon->matches_filter(filter_weapon) )
 868                                 return false;
 869                 }
 870 
 871                 // Passed.
 872                 return true;
 873         }

 877 /**
 878  * Returns whether or not the given special is active for the specified unit,
 879  * based on the current context (see set_specials_context).
 880  * @param[in] special           a weapon special WML structure
 881  * @param[in] whom              specifies which combatant we care about
 882  * @param[in] include_backstab  false if backstab specials should not be active
 883  *                              (usually true since backstab is usually accounted
 884  *                              for elsewhere)
 885  */
 886 bool attack_type::special_active(const config& special, AFFECTS whom,
 887                                  bool include_backstab) const
 888 {
 889         //log_scope("special_active");
 890 
 891         // Backstab check
 892         if ( !include_backstab )
 893                 if ( special["backstab"].to_bool() )
 894                         return false;
 895 
 896         // Does this affect the specified unit?
 897         if ( whom == AFFECT_SELF ) {
 898                 if ( !special_affects_self(special, is_attacker_) )
 899                         return false;
 900         }
 901         if ( whom == AFFECT_OTHER ) {
 902                 if ( !special_affects_opponent(special, is_attacker_) )
 903                         return false;
 904         }
 905 
 906         // Is this active on attack/defense?
 907         const std::string & active_on = special["active_on"];
 908         if ( !active_on.empty() ) {
 909                 if ( is_attacker_  &&  active_on != "offense" )
 910                         return false;
 911                 if ( !is_attacker_  &&  active_on != "defense" )
 912                         return false;
 913         }
 914 
 915         // Get the units involved.
 916         assert(resources::units);
 917         const unit_map& units = *resources::units;
 918 
 919         unit_map::const_iterator self  = units.find(self_loc_);
 920         unit_map::const_iterator other = units.find(other_loc_);
 921 
 922         // Make sure they're facing each other.
 923         temporary_facing self_facing(self, self_loc_.get_relative_dir(other_loc_));
 924         temporary_facing other_facing(other, other_loc_.get_relative_dir(self_loc_));
 925 
 926         // Translate our context into terms of "attacker" and "defender".
 927         unit_map::const_iterator & att = is_attacker_ ? self : other;
 928         unit_map::const_iterator & def = is_attacker_ ? other : self;
 929         const map_location & att_loc   = is_attacker_ ? self_loc_ : other_loc_;
 930         const map_location & def_loc   = is_attacker_ ? other_loc_ : self_loc_;
 931         const_attack_ptr att_weapon = is_attacker_ ? shared_from_this() : other_attack_;
 932         const_attack_ptr def_weapon = is_attacker_ ? other_attack_ : shared_from_this();
 933 
 934         // Filter the units involved.
 935         if (!special_unit_matches(self, other, self_loc_, shared_from_this(), special, is_for_listing_, "filter_self"))
 936                 return false;
 937         if (!special_unit_matches(other, self, other_loc_, other_attack_, special, is_for_listing_, "filter_opponent"))
 938                 return false;
 939         if (!special_unit_matches(att, def, att_loc, att_weapon, special, is_for_listing_, "filter_attacker"))
 940                 return false;
 941         if (!special_unit_matches(def, att, def_loc, def_weapon, special, is_for_listing_, "filter_defender"))
 942                 return false;
 943 
 944         map_location adjacent[6];
 945         get_adjacent_tiles(self_loc_, adjacent);
 946 
 947         // Filter the adjacent units.
 948         for (const config &i : special.child_range("filter_adjacent"))
 949         {
 950                 size_t count = 0;
 951                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 952                 unit_filter filter(vconfig(i), resources::filter_con);
 953                 for (const map_location::DIRECTION index : dirs)
 954                 {
 955                         if (index == map_location::NDIRECTIONS)
 956                                 continue;
 957                         unit_map::const_iterator unit = units.find(adjacent[index]);
 958                         if (unit == units.end() || !filter.matches(*unit, adjacent[index], *self))
 959                                 return false;
 960                         if (i.has_attribute("is_enemy")) {
 961                                 const display_context& dc = resources::filter_con->get_disp_context();
 962                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(self->side())) {
 963                                         continue;
 964                                 }
 965                         }
 966                         count++;
 967                 }
 968                 if (i["count"].empty() && count != dirs.size()) {
 969                         return false;
 970                 }
 971                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 972                         return false;
 973                 }
 974         }
 975 
 976         // Filter the adjacent locations.
 977         for (const config &i : special.child_range("filter_adjacent_location"))
 978         {
 979                 size_t count = 0;
 980                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 981                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 982                 for (const map_location::DIRECTION index : dirs)
 983                 {
 984                         if (index == map_location::NDIRECTIONS)
 985                                 continue;
 986                         if(!adj_filter.match(adjacent[index])) {
 987                                 return false;
 988                         }
 989                         count++;
 990                 }
 991                 if (i["count"].empty() && count != dirs.size()) {
 992                         return false;
 993                 }
 994                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 995                         return false;
 996                 }
 997         }
 998 
 999         return true;
1000 }

1015 bool filter_base_matches(const config& cfg, int def)
1016 {
1017         if (const config &apply_filter = cfg.child("filter_base_value")) {
1018                 config::attribute_value cond_eq = apply_filter["equals"];
1019                 config::attribute_value cond_ne = apply_filter["not_equals"];
1020                 config::attribute_value cond_lt = apply_filter["less_than"];
1021                 config::attribute_value cond_gt = apply_filter["greater_than"];
1022                 config::attribute_value cond_ge = apply_filter["greater_than_equal_to"];
1023                 config::attribute_value cond_le = apply_filter["less_than_equal_to"];
1024                 return  (cond_eq.empty() || def == cond_eq.to_int()) &&
1025                         (cond_ne.empty() || def != cond_ne.to_int()) &&
1026                         (cond_lt.empty() || def <  cond_lt.to_int()) &&
1027                         (cond_gt.empty() || def >  cond_gt.to_int()) &&
1028                         (cond_ge.empty() || def >= cond_ge.to_int()) &&
1029                         (cond_le.empty() || def <= cond_le.to_int());
1030         }
1031         return true;
1032 }
1033 
1034 effect::effect(const unit_ability_list& list, int def, bool backstab) :
1035         effect_list_(),
1036         composite_value_(0)
1037 {
1038 
1039         int value_set = def;
1040         bool value_is_set = false;
1041         std::map<std::string,individual_effect> values_add;
1042         std::map<std::string,individual_effect> values_mul;
1043         std::map<std::string,individual_effect> values_div;
1044 
1045         individual_effect set_effect;
1046 
1047         for (const unit_ability & ability : list) {
1048                 const config& cfg = *ability.first;
1049                 const std::string& effect_id = cfg[cfg["id"].empty() ? "name" : "id"];
1050 
1051                 if (!cfg["backstab"].blank()) {
1052                         lg::wml_error() << "The backstab= key in weapon specials is deprecated; use [filter_adjacent] instead\n";
1053                 }
1054 
1055                 if (!backstab && cfg["backstab"].to_bool())
1056                         continue;
1057                 if (!filter_base_matches(cfg, def))
1058                         continue;
1059 
1060                 if (const config::attribute_value *v = cfg.get("value")) {
1061                         int value = get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1062                                 callable.add("base_value", wfl::variant(def));
1063                                 return formula.evaluate(callable).as_int();
1064                         });
1065 
1066                         bool cumulative = cfg["cumulative"].to_bool();
1067                         if (!value_is_set && !cumulative) {
1068                                 value_set = value;
1069                                 set_effect.set(SET, value, ability.first, ability.second);
1070                         } else {
1071                                 if (cumulative) value_set = std::max<int>(value_set, def);
1072                                 if (value > value_set) {
1073                                         value_set = value;
1074                                         set_effect.set(SET, value, ability.first, ability.second);
1075                                 }
1076                         }
1077                         value_is_set = true;
1078                 }
1079 
1080                 if (const config::attribute_value *v = cfg.get("add")) {
1081                         int add = get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1082                                 callable.add("base_value", wfl::variant(def));
1083                                 return formula.evaluate(callable).as_int();
1084                         });
1085                         std::map<std::string,individual_effect>::iterator add_effect = values_add.find(effect_id);
1086                         if(add_effect == values_add.end() || add > add_effect->second.value) {
1087                                 values_add[effect_id].set(ADD, add, ability.first, ability.second);
1088                         }
1089                 }
1090                 if (const config::attribute_value *v = cfg.get("sub")) {
1091                         int sub = - get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1092                                 callable.add("base_value", wfl::variant(def));
1093                                 return formula.evaluate(callable).as_int();
1094                         });
1095                         std::map<std::string,individual_effect>::iterator sub_effect = values_add.find(effect_id);
1096                         if(sub_effect == values_add.end() || sub > sub_effect->second.value) {
1097                                 values_add[effect_id].set(ADD, sub, ability.first, ability.second);
1098                         }
1099                 }
1100                 if (const config::attribute_value *v = cfg.get("multiply")) {
1101                         int multiply = static_cast<int>(get_single_ability_value(*v, static_cast<double>(def), ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1102                                 callable.add("base_value", wfl::variant(def));
1103                                 return formula.evaluate(callable).as_decimal() / 1000.0 ;
1104                         }) * 100);
1105                         std::map<std::string,individual_effect>::iterator mul_effect = values_mul.find(effect_id);
1106                         if(mul_effect == values_mul.end() || multiply > mul_effect->second.value) {
1107                                 values_mul[effect_id].set(MUL, multiply, ability.first, ability.second);
1108                         }
1109                 }
1110                 if (const config::attribute_value *v = cfg.get("divide")) {
1111                         int divide = static_cast<int>(get_single_ability_value(*v, static_cast<double>(def), ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1112                                 callable.add("base_value", wfl::variant(def));
1113                                 return formula.evaluate(callable).as_decimal() / 1000.0 ;
1114                         }) * 100);
1115 
1116                         if (divide == 0) {
1117                                 ERR_NG << "division by zero with divide= in ability/weapon special " << effect_id << std::endl;
1118                         }
1119                         else {
1120                                 std::map<std::string,individual_effect>::iterator div_effect = values_div.find(effect_id);
1121                                 if(div_effect == values_div.end() || divide > div_effect->second.value) {
1122                                         values_div[effect_id].set(DIV, divide, ability.first, ability.second);
1123                                 }
1124                         }
1125                 }
1126         }
1127 
1128         if(value_is_set && set_effect.type != NOT_USED) {
1129                 effect_list_.push_back(set_effect);
1130         }
1131 
1132         /* Do multiplication with floating point values rather than integers
1133          * We want two places of precision for each multiplier
1134          * Using integers multiplied by 100 to keep precision causes overflow
1135          *   after 3-4 abilities for 32-bit values and ~8 for 64-bit
1136          * Avoiding the overflow by dividing after each step introduces rounding errors
1137          *   that may vary depending on the order effects are applied
1138          * As the final values are likely <1000 (always true for mainline), loss of less significant digits is not an issue
1139          */
1140         double multiplier = 1.0;
1141         double divisor = 1.0;
1142         std::map<std::string,individual_effect>::const_iterator e, e_end;
1143         for (e = values_mul.begin(), e_end = values_mul.end(); e != e_end; ++e) {
1144                 multiplier *= e->second.value/100.0;
1145                 effect_list_.push_back(e->second);
1146         }
1147         for (e = values_div.begin(), e_end = values_div.end(); e != e_end; ++e) {
1148                 divisor *= e->second.value/100.0;
1149                 effect_list_.push_back(e->second);
1150         }
1151         int addition = 0;
1152         for (e = values_add.begin(), e_end = values_add.end(); e != e_end; ++e) {
1153                 addition += e->second.value;
1154                 effect_list_.push_back(e->second);
1155         }
1156 
1157         composite_value_ = int((value_set + addition) * multiplier / divisor);
1158 }
