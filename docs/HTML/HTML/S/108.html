 241 bool play_controller::hotkey_handler::execute_command(const hotkey::hotkey_command& cmd, int index, bool press)
 242 {
 243         hotkey::HOTKEY_COMMAND command = cmd.id;
 244         if(index >= 0) {
 245                 unsigned i = static_cast<unsigned>(index);
 246                 if(i < savenames_.size() && !savenames_[i].empty()) {
 247                         // Load the game by throwing load_game_exception
 248                         load_autosave(savenames_[i]);
 249 
 250                 } else if ( i < wml_commands_.size()  &&  wml_commands_[i] ) {
 251                         wml_commands_[i]->fire_event(mouse_handler_.get_last_hex(), gamestate().gamedata_);
 252                         return true;
 253                 }
 254         }
 255         int prefixlen = wml_menu_hotkey_prefix.length();
 256         if(press && command == hotkey::HOTKEY_WML && cmd.command.compare(0, prefixlen, wml_menu_hotkey_prefix) == 0)
 257         {
 258                 std::string name = cmd.command.substr(prefixlen);
 259                 const map_location& hex = mouse_handler_.get_last_hex();
 260 
 261                 return gamestate().get_wml_menu_items().fire_item(name, hex, gamestate().gamedata_, gamestate(), gamestate().board_.units_);
 262         }
 263         return command_executor::execute_command(cmd, index, press);
 264 }

 316         case hotkey::HOTKEY_NULL:
 317         case hotkey::HOTKEY_SAVE_REPLAY:
 318         case hotkey::HOTKEY_LABEL_SETTINGS:
 319         case hotkey::LUA_CONSOLE:
 320         case hotkey::HOTKEY_SCROLL_UP:
 321         case hotkey::HOTKEY_SCROLL_DOWN:
 322         case hotkey::HOTKEY_SCROLL_LEFT:
 323         case hotkey::HOTKEY_SCROLL_RIGHT:
 324                 return true;
 325 
 326         // Commands that have some preconditions:
 327         case hotkey::HOTKEY_SAVE_GAME:
 328                 return !events::commands_disabled;
 329 
 330         case hotkey::HOTKEY_SHOW_ENEMY_MOVES:
 331         case hotkey::HOTKEY_BEST_ENEMY_MOVES:
 332                 return !linger() && play_controller_.enemies_visible();
 333 
 334         case hotkey::HOTKEY_LOAD_GAME:
 335                 return !play_controller_.is_networked_mp(); // Can only load games if not in a network game
 336 
 337         case hotkey::HOTKEY_CHAT_LOG:
 338                 return true;
 339 
 340         case hotkey::HOTKEY_REDO:
 341                 return play_controller_.can_redo();
 342         case hotkey::HOTKEY_UNDO:
 343                 return play_controller_.can_undo();
 344 
 345         case hotkey::HOTKEY_UNIT_DESCRIPTION:
 346                 return menu_handler_.current_unit().valid();
 347 
 348         case hotkey::HOTKEY_TERRAIN_DESCRIPTION:
 349                 return mouse_handler_.get_last_hex().valid();
 350 
 351         case hotkey::HOTKEY_RENAME_UNIT:
 352                 return !events::commands_disabled &&
 353                         menu_handler_.current_unit().valid() &&
 354                         !(menu_handler_.current_unit()->unrenamable()) &&
 355                         menu_handler_.current_unit()->side() == gui()->viewing_side() &&
 356                         play_controller_.get_teams_const()[menu_handler_.current_unit()->side() - 1].is_local_human();
 357 
 358         default:
 359                 return false;
 360         }
 361 }

 363 template<typename T>
 364 static void trim_items(std::vector<T>& newitems)
 365 {
 366         if(newitems.size() > 5) {
 367                 std::vector<T> subitems;
 368                 subitems.push_back(std::move(newitems[0]));
 369                 subitems.push_back(std::move(newitems[1]));
 370                 subitems.push_back(std::move(newitems[newitems.size() / 3]));
 371                 subitems.push_back(std::move(newitems[newitems.size() * 2 / 3]));
 372                 subitems.push_back(std::move(newitems.back()));
 373                 newitems = subitems;
 374         }
 375 }

 377 void play_controller::hotkey_handler::expand_autosaves(std::vector<config>& items, int i)
 378 {
 379         const compression::format comp_format = preferences::save_compression_format();
 380 
 381         savenames_.resize(i);
 382 
 383         auto pos = items.erase(items.begin() + i);
 384         std::vector<config> newitems;
 385         std::vector<std::string> newsaves;
 386 
 387         const std::string& start_name = saved_game_.classification().label;
 388 
 389         for(unsigned int turn = play_controller_.turn(); turn != 0; turn--) {
 390                 const std::string name = formatter() << start_name << "-" << _("Auto-Save") << turn;
 391 
 392                 if(savegame::save_game_exists(name, comp_format)) {
 393                         newsaves.emplace_back(name + compression::format_extension(comp_format));
 394                         newitems.emplace_back(config {"label", _("Back to Turn ") + std::to_string(turn)});
 395                 }
 396         }
 397 
 398         if(savegame::save_game_exists(start_name, comp_format)) {
 399                 newsaves.emplace_back(start_name + compression::format_extension(comp_format));
 400                 newitems.emplace_back(config {"label", _("Back to Start")});
 401         }
 402 
 403         // Make sure list doesn't get too long: keep top two, midpoint and bottom.
 404         trim_items(newitems);
 405         trim_items(newsaves);
 406 
 407         items.insert(pos, newitems.begin(), newitems.end());
 408         savenames_.insert(savenames_.end(), newsaves.begin(), newsaves.end());
 409 }

 411 void play_controller::hotkey_handler::expand_wml_commands(std::vector<config>& items, int i)
 412 {
 413         // Pad the commands with null pointers (keeps the indices of items and wml_commands_ synced).
 414         wml_commands_.resize(i);
 415 
 416         auto pos = items.erase(items.begin() + i);
 417         std::vector<config> newitems;
 418 
 419         gamestate().get_wml_menu_items().get_items(mouse_handler_.get_last_hex(), wml_commands_, newitems,
 420                 gamestate(), gamestate().gamedata_, gamestate().board_.units_);
 421 
 422         // Replace this placeholder entry with available menu items.
 423         items.insert(pos, newitems.begin(), newitems.end());
 424 }

 426 void play_controller::hotkey_handler::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool context_menu, display& disp)
 427 {
 428         if(context_menu) {
 429                 last_context_menu_x_ = xloc;
 430                 last_context_menu_y_ = yloc;
 431         }
 432 
 433         std::vector<config> items;
 434         for(const auto& item : items_arg) {
 435                 const std::string& id = item["id"];
 436                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
 437 
 438                 if(id == "wml" || (can_execute_command(command) && (!context_menu || in_context_menu(command.id)))) {
 439                         items.emplace_back(config {"id", id});
 440                 }
 441         }
 442 
 443         // Add special non-hotkey items to the menu and remember their indices
 444         // Iterate in reverse to avoid also iterating over the new inserted items
 445         savenames_.clear();
 446         wml_commands_.clear();
 447 
 448         for(int i = items.size() - 1; i >= 0; i--) {
 449                 if(items[i]["id"] == "AUTOSAVES") {
 450                         expand_autosaves(items, i);
 451                 } else if(items[i]["id"] == "wml") {
 452                         expand_wml_commands(items, i);
 453                 }
 454         }
 455 
 456         if(items.empty()) {
 457                 return;
 458         }
 459 
 460         command_executor::show_menu(items, xloc, yloc, context_menu, disp);
 461 }

 463 bool play_controller::hotkey_handler::in_context_menu(hotkey::HOTKEY_COMMAND command) const
 464 {
 465         switch(command) {
 466         // Only display these if the mouse is over a castle or keep tile
 467         case hotkey::HOTKEY_RECRUIT:
 468         case hotkey::HOTKEY_REPEAT_RECRUIT:
 469         case hotkey::HOTKEY_RECALL: {
 470                 // last_hex_ is set by mouse_events::mouse_motion
 471                 const map_location & last_hex = mouse_handler_.get_last_hex();
 472                 const int viewing_side = gui()->viewing_side();
 473 
 474                 // A quick check to save us having to create the future map and
 475                 // possibly loop through all units.
 476                 if ( !play_controller_.get_map_const().is_keep(last_hex)  &&
 477                      !play_controller_.get_map_const().is_castle(last_hex) )
 478                         return false;
 479 
 480                 wb::future_map future; /* lasts until method returns. */
 481 
 482                 return gamestate().side_can_recruit_on(viewing_side, last_hex);
 483         }
 484         default:
 485                 return true;
 486         }
 487 }

 500 hotkey::ACTION_STATE play_controller::hotkey_handler::get_action_state(hotkey::HOTKEY_COMMAND command, int /*index*/) const
 501 {
 502         switch(command) {
 503 
 504         case hotkey::HOTKEY_MINIMAP_DRAW_VILLAGES:
 505                 return (preferences::minimap_draw_villages()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 506         case hotkey::HOTKEY_MINIMAP_CODING_UNIT:
 507                 return (preferences::minimap_movement_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 508         case hotkey::HOTKEY_MINIMAP_CODING_TERRAIN:
 509                 return (preferences::minimap_terrain_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 510         case hotkey::HOTKEY_MINIMAP_DRAW_UNITS:
 511                 return (preferences::minimap_draw_units()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 512         case hotkey::HOTKEY_MINIMAP_DRAW_TERRAIN:
 513                 return (preferences::minimap_draw_terrain()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 514         case hotkey::HOTKEY_ZOOM_DEFAULT:
 515                 return (gui()->get_zoom_factor() == 1.0) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 516         case hotkey::HOTKEY_DELAY_SHROUD:
 517                 return viewing_team().auto_shroud_updates() ? hotkey::ACTION_OFF : hotkey::ACTION_ON;
 518         default:
 519                 return hotkey::ACTION_STATELESS;
 520         }
 521 }
