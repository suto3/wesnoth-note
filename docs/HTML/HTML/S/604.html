  56 point slider::calculate_best_size() const
  57 {
  58         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
  59 
  60         // Inherited.
  61         point result = styled_widget::calculate_best_size();
  62 
  63         if(best_slider_length_ != 0) {
  64                 // Override length.
  65                 const auto conf = cast_config_to<slider_definition>();
  66                 assert(conf);
  67 
  68                 result.x = conf->left_offset + best_slider_length_ + conf->right_offset;
  69         }
  70 
  71         DBG_GUI_L << LOG_HEADER << " best_slider_length " << best_slider_length_ << " result " << result << ".\n";
  72         return result;
  73 }

  75 void slider::set_value(const int value)
  76 {
  77         if(value == get_value()) {
  78                 return;
  79         }
  80 
  81         if(value < minimum_value_) {
  82                 set_value(minimum_value_);
  83         } else if(value > get_maximum_value()) {
  84                 set_value(get_maximum_value());
  85         } else {
  86                 set_item_position(value - minimum_value_);
  87         }
  88 
  89         fire(event::NOTIFY_MODIFIED, *this, nullptr);
  90 }

  92 void slider::set_minimum_value(const int minimum_value)
  93 {
  94         if(minimum_value == minimum_value_) {
  95                 return;
  96         }
  97 
  98         /** @todo maybe make it a VALIDATE. */
  99         assert(minimum_value <= get_maximum_value());
 100 
 101         const int value = get_value();
 102         const int maximum_value = get_maximum_value();
 103         minimum_value_ = minimum_value;
 104 
 105         // The number of items needs to include the begin and end so distance step
 106         // size.
 107         set_item_count(maximum_value - minimum_value_ + get_step_size());
 108 
 109         if(value < minimum_value_) {
 110                 set_item_position(0);
 111         }
 112 }

 114 void slider::set_maximum_value(const int maximum_value)
 115 {
 116         if(maximum_value == get_maximum_value()) {
 117                 return;
 118         }
 119 
 120         /** @todo maybe make it a VALIDATE. */
 121         assert(minimum_value_ <= maximum_value);
 122 
 123         const int value = get_value();
 124 
 125         // The number of items needs to include the begin and end so distance + step
 126         // size.
 127         set_item_count(maximum_value - minimum_value_ + get_step_size());
 128 
 129         if(value > maximum_value) {
 130                 set_item_position(get_maximum_value());
 131         }
 132 }

 134 t_string slider::get_value_label() const
 135 {
 136         if(value_label_generator_) {

 180 bool slider::on_positioner(const point& coordinate) const
 181 {
 182         // Note we assume the positioner is over the entire height of the widget.
 183         return
 184                 coordinate.x >= static_cast<int>(get_positioner_offset()) &&
 185                 coordinate.x <  static_cast<int>(get_positioner_offset() + get_positioner_length()) &&
 186                 coordinate.y > 0 &&
 187                 coordinate.y <  static_cast<int>(get_height());
 188 }
 189 
 190 int slider::on_bar(const point& coordinate) const
 191 {
 192         const unsigned x = static_cast<size_t>(coordinate.x);
 193         const unsigned y = static_cast<size_t>(coordinate.y);
 194 
 195         // Not on the widget, leave.
 196         if(x > get_width() || y > get_height()) {
 197                 return 0;
 198         }
 199 
 200         // we also assume the bar is over the entire height of the widget.
 201         if(x < get_positioner_offset()) {
 202                 return -1;
 203         } else if(x > get_positioner_offset() + get_positioner_length()) {
 204                 return 1;
 205         }
 206 
 207         return 0;
 208 }

 453 builder_slider::builder_slider(const config& cfg)
 454         : implementation::builder_styled_widget(cfg)
 455         , best_slider_length_(cfg["best_slider_length"])
 456         , minimum_value_(cfg["minimum_value"])
 457         , maximum_value_(cfg["maximum_value"])
 458         , step_size_(cfg["step_size"].to_unsigned(1))
 459         , value_(cfg["value"])
 460         , minimum_value_label_(cfg["minimum_value_label"].t_str())
 461         , maximum_value_label_(cfg["maximum_value_label"].t_str())
 462         , value_labels_()
 463 {
 464         const config& labels = cfg.child("value_labels");
 465         if(!labels) {
 466                 return;
 467         }
 468 
 469         for(const auto& label : labels.child_range("value")) {
 470                 value_labels_.push_back(label["label"]);
 471         }
 472 }

 474 widget* builder_slider::build() const
 475 {
 476         slider* widget = new slider(*this);
 477 
 478         widget->set_best_slider_length(best_slider_length_);
 479         widget->set_maximum_value(maximum_value_);
 480         widget->set_minimum_value(minimum_value_);
 481         widget->set_step_size(step_size_);
 482         widget->set_value(value_);
 483 
 484         widget->finalize_setup();
 485 
 486         if(!value_labels_.empty()) {
 487                 VALIDATE(value_labels_.size() == widget->get_item_count(),
 488                                 _("The number of value_labels and values don't match."));
 489 
 490                 widget->set_value_labels(value_labels_);
 491 
 492         } else {
 493                 widget->set_minimum_value_label(minimum_value_label_);
 494                 widget->set_maximum_value_label(maximum_value_label_);
 495         }
 496 
 497         DBG_GUI_G << "Window builder: placed slider '" << id << "' with definition '" << definition << "'.\n";
 498 
 499         return widget;
 500 }
