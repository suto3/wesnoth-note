  47 /**
  48  * Creates a bound status label that will reflect the label state of a widget. The initial label
  49  * value is set here, and then again any time the widget is modified. A function is also returned
  50  * that can be called to update the label manually.
  51  *
  52  * This relies on hooking into the NOTIFY_MODIFIED event, so can only be used with widgets that fire
  53  * that event.
  54  *
  55  * @param find_in                  The containing widget (usually a window or grid) in which to find
  56  *                                 the source and status label widgets.
  57  * @param source_id                The ID of the source widget.
  58  * @param value_getter             Functor to process the value of the source widget.
  59  * @param label_id                 The ID of the status label widget.
  60  *
  61  * @returns                        The callback function used to update the status label's value.
  62  */
  63 template<typename W>
  64 std::function<void()> bind_status_label(
  65                 widget* find_in,
  66                 const std::string& source_id,
  67                 const std::function<std::string(W&)> value_getter = default_status_value_getter<W>,
  68                 const std::string& label_id = "")
  69 {
  70         // If no label ID is provided, use the format source ID + '_label'.
  71         const std::string label_id_ = label_id.empty() ? source_id + "_label" : label_id;
  72 
  73         // Find the source value widget.
  74         W& source = find_widget<W>(find_in, source_id, false);
  75 
  76         // Find the target status label.
  77         styled_widget& label = find_widget<styled_widget>(find_in, label_id_, false);
  78 
  79         const auto update_label = [&, value_getter]() {
  80                 const std::string value = value_getter(source);
  81 
  82                 label.set_label(value);
  83         };
  84 
  85         // Bind the callback.
  86         connect_signal_notify_modified(source, std::bind(update_label));
  87 
  88         // Set initial value.
  89         update_label();
  90 
  91         return update_label;
  92 }
