 112 typedef std::pair<Sint64, std::ios_base::seekdir> offset_dir;
 113 static offset_dir translate_seekdir(Sint64 offset, int whence) {
 114         switch(whence){
 115         case RW_SEEK_SET:
 116                 return std::make_pair(std::max<Sint64>(0, offset), std::ios_base::beg);
 117         case RW_SEEK_CUR:
 118                 return std::make_pair(offset, std::ios_base::cur);
 119         case RW_SEEK_END:
 120                 return std::make_pair(std::min<Sint64>(0, offset), std::ios_base::end);
 121         default:
 122                 assert(false);
 123                 throw "assertion ignored";
 124         }
 125 }
 126 static Sint64 SDLCALL ifs_seek(struct SDL_RWops *context, Sint64 offset, int whence) {
 127         std::ios_base::seekdir seekdir;
 128         std::tie(offset, seekdir) = translate_seekdir(offset, whence);
 129 
 130         std::istream *ifs = static_cast<std::istream*>(context->hidden.unknown.data1);
 131         const std::ios_base::iostate saved_state = ifs->rdstate();
 132 
 133         ifs->seekg(offset, seekdir);
 134 
 135         if(saved_state != ifs->rdstate() && offset < 0) {
 136                 ifs->clear(saved_state);
 137                 ifs->seekg(0, std::ios_base::beg);
 138         }
 139 
 140         std::streamsize pos = ifs->tellg();
 141         return static_cast<int>(pos);
 142 }

 161 static size_t SDLCALL ifs_read(struct SDL_RWops *context, void *ptr, size_t size, size_t maxnum) {
 162         std::istream *ifs = static_cast<std::istream*>(context->hidden.unknown.data1);
 163 
 164         // This seems overly simplistic, but it's the same as mem_read's implementation
 165         ifs->read(static_cast<char*>(ptr), maxnum * size);
 166         std::streamsize num = ifs->good() ? maxnum : ifs->gcount() / size;
 167 
 168         // EOF sticks unless we clear it. Bad is an actual I/O error
 169         if(!ifs->bad())
 170                 ifs->clear();
 171 
 172         return static_cast<int>(num);
 173 }

 183 static size_t SDLCALL ofs_write(struct SDL_RWops *context, const void *ptr, size_t size, size_t num) {
 184         std::ostream *ofs = static_cast<std::ostream*>(context->hidden.unknown.data1);
 185 
 186         const std::streampos before = ofs->tellp();
 187         ofs->write(static_cast<const char*>(ptr), num * size);
 188         const std::streampos after = ofs->tellp();
 189         const std::streamoff bytes_written = after - before;
 190         const int num_written = bytes_written / size;
 191 
 192         return num_written;
 193 }
