  79 void tod_manager::resolve_random(randomness::rng& r)
  80 {
  81         //process the random_start_time string, which can be boolean yes/no true/false or a
  82         //comma-separated string of integers >= 1 referring to the times_ array indices
  83         std::vector<std::string> output_strings = utils::split(random_tod_.str());
  84         std::vector<int> output;
  85 
  86         try
  87         {
  88                 std::transform(output_strings.begin(), output_strings.end(), std::back_inserter(output),
  89                         [](const std::string& str)
  90                 {
  91                         return std::stoi(str);
  92                 });
  93         }
  94         catch (std::invalid_argument)
  95         {
  96                 // This happens if the random_start_time string is a boolean.
  97                 // Simply ignore the exception.
  98         }
  99 
 100         // Remove non-positive times
 101         output.erase(
 102                 std::remove_if(
 103                         output.begin(),
 104                         output.end(),
 105                         [](int time){ return time <= 0; }),
 106                 output.end());
 107 
 108         if(!output.empty())
 109         {
 110                 int chosen = output[r.next_random() % output.size()];
 111                 currentTime_ = calculate_current_time(times_.size(), turn_, chosen, true);
 112                 r.next_random();
 113         }
 114         else if (random_tod_.to_bool(false))
 115         {
 116                 currentTime_ = calculate_current_time(times_.size(), turn_, r.next_random(), true);
 117         }
 118         random_tod_ = false;
 119 }
 120 config tod_manager::to_config() const
 121 {
 122         config cfg;
 123         cfg["turn_at"] = turn_;
 124         cfg["turns"] = num_turns_;
 125         cfg["current_time"] = currentTime_;
 126         cfg["random_start_time"] = random_tod_;
 127         cfg["it_is_a_new_turn"] = !has_turn_event_fired_;
 128         std::vector<time_of_day>::const_iterator t;
 129         for(t = times_.begin(); t != times_.end(); ++t) {
 130                 t->write(cfg.add_child("time"));
 131         }
 132         for(std::vector<area_time_of_day>::const_iterator i = areas_.begin(); i != areas_.end(); ++i) {
 133                 config& area = cfg.add_child("time_area");
 134                 // if no ranges, then use hexes to generate ranges
 135                 if(i->xsrc.empty() && i->ysrc.empty()) {
 136                         write_location_range(i->hexes, area);
 137                 } else {
 138                         area["x"] = i->xsrc;
 139                         area["y"] = i->ysrc;
 140                 }
 141                 for(t = i->times.begin(); t != i->times.end(); ++t) {
 142                         t->write(area.add_child("time"));
 143                 }
 144                 area["current_time"] = i->currentTime;
 145                 if (!i->id.empty())
 146                         area["id"] = i->id;
 147         }
 148         return cfg;
 149 }

 189 const time_of_day& tod_manager::get_time_of_day(const map_location& loc, int n_turn) const
 190 {
 191         if(n_turn == 0)
 192                 n_turn = turn_;
 193 
 194         if ( loc != map_location::null_location() )
 195         {
 196                 for ( std::vector<area_time_of_day>::const_reverse_iterator
 197                       i = areas_.rbegin(), i_end = areas_.rend(); i != i_end; ++i )
 198                 {
 199                         if (i->hexes.find(loc) != i->hexes.end())
 200                                 return get_time_of_day_turn(i->times, n_turn, i->currentTime);
 201                 }
 202         }
 203 
 204         return get_time_of_day_turn(times_, n_turn, currentTime_);

 207 const time_of_day tod_manager::get_illuminated_time_of_day(const unit_map & units, const gamemap & map, const map_location& loc, int for_turn) const
 208 {
 209         // get ToD ignoring illumination
 210         time_of_day tod = get_time_of_day(loc, for_turn);
 211 
 212         if ( map.on_board_with_border(loc) )
 213         {
 214                 // Now add terrain illumination.
 215                 const int terrain_light = map.get_terrain_info(loc).light_bonus(tod.lawful_bonus);
 216 
 217                 std::vector<int> mod_list;
 218                 std::vector<int> max_list;
 219                 std::vector<int> min_list;
 220                 int most_add = 0;
 221                 int most_sub = 0;
 222 
 223                 // Find the "illuminates" effects from units that can affect loc.
 224                 map_location locs[7];
 225                 locs[0] = loc;
 226                 get_adjacent_tiles(loc,locs+1);
 227                 for ( size_t i = 0; i != 7; ++i ) {
 228                         const unit_map::const_iterator itor = units.find(locs[i]);
 229                         if (itor != units.end() &&
 230                             itor->get_ability_bool("illuminates", *resources::gameboard) &&
 231                             !itor->incapacitated())
 232                         {
 233                                 unit_ability_list illum = itor->get_abilities("illuminates");
 234                                 unit_abilities::effect illum_effect(illum, terrain_light, false);
 235                                 const int unit_mod = illum_effect.get_composite_value();
 236 
 237                                 // Record this value.
 238                                 mod_list.push_back(unit_mod);
 239                                 max_list.push_back(illum.highest("max_value").first);
 240                                 min_list.push_back(illum.lowest("min_value").first);
 241                                 if ( unit_mod > most_add )
 242                                         most_add = unit_mod;
 243                                 else if ( unit_mod < most_sub )
 244                                         most_sub = unit_mod;
 245                         }
 246                 }
 247                 const bool net_darker = most_add < -most_sub;
 248 
 249                 // Apply each unit's effect, tracking the best result.
 250                 int best_result = terrain_light;
 251                 const int base_light = terrain_light + (net_darker ? most_add : most_sub);
 252                 for ( size_t i = 0; i != mod_list.size(); ++i ) {
 253                         int result =
 254                                 bounded_add(base_light, mod_list[i], max_list[i], min_list[i]);
 255 
 256                         if ( net_darker  &&  result < best_result )
 257                                 best_result = result;
 258                         else if ( !net_darker  &&  result > best_result )
 259                                 best_result = result;
 260                 }
 261 
 262                 // Update the object we will return.
 263                 tod.bonus_modified = best_result - tod.lawful_bonus;
 264                 tod.lawful_bonus = best_result;
 265         }
 266 
 267         return tod;
 268 }

 344 void tod_manager::add_time_area(const gamemap & map, const config& cfg)

 370 void tod_manager::remove_time_area(const std::string& area_id)
 371 {
 372         if(area_id.empty()) {
 373                 areas_.clear();
 374         } else {
 375                 // search for all time areas that match the id.
 376                 std::vector<area_time_of_day>::iterator i = areas_.begin();
 377                 while(i != areas_.end()) {
 378                         if((*i).id == area_id) {
 379                                 i = areas_.erase(i);
 380                         } else {
 381                                 ++i;
 382                         }
 383                 }
 384         }
 385         has_tod_bonus_changed_ = true;
 386 }

 436 void tod_manager::set_turn(const int num, game_data* vars, const bool increase_limit_if_needed)
 437 {
 438         has_tod_bonus_changed_ = false;
 439         const int new_turn = std::max<int>(num, 1);
 440         LOG_NG << "changing current turn number from " << turn_ << " to " << new_turn << '\n';
 441         // Correct ToD
 442         set_new_current_times(new_turn);
 443 
 444         if(increase_limit_if_needed && (new_turn > num_turns_) && num_turns_ != -1) {
 445                 set_number_of_turns(new_turn);
 446         }
 447         turn_ = new_turn;
 448         if (vars)
 449                 vars->get_variable("turn_number") = new_turn;
 450 }

 469 int tod_manager::calculate_current_time(
 470         const int number_of_times,
 471         const int for_turn_number,
 472         const int current_time,
 473         const bool only_to_allowed_range) const
 474 {
 475         if (number_of_times == 0) return 0;
 476         int new_current_time = 0;
 477         if(only_to_allowed_range) new_current_time = current_time % number_of_times;
 478         else new_current_time = (current_time + for_turn_number - turn_) % number_of_times;
 479         while(new_current_time < 0) { new_current_time += number_of_times; }
 480         return new_current_time;
 481 }
