 116 // Helper function to refresh resolution list
 117 void preferences_dialog::set_resolution_list(menu_button& res_list, CVideo& video)
 118 {
 119         resolutions_ = video.get_available_resolutions(true);
 120 
 121         std::vector<config> options;
 122         for(const auto& res : resolutions_)
 123         {
 124                 config option;
 125                 option["label"] = formatter() << res.first << font::unicode_multiplication_sign << res.second;
 126 
 127                 const int div = boost::math::gcd(res.first, res.second);
 128                 const int ratio[2] {res.first/div, res.second/div};
 129                 if(ratio[0] <= 10 || ratio[1] <= 10) {
 130                         option["details"] = formatter() << "<span color='#777777'>(" << ratio[0] << ':' << ratio[1] << ")</span>";
 131                 }
 132 
 133                 options.push_back(option);
 134         }
 135 
 136         const unsigned current_res = std::find(resolutions_.begin(), resolutions_.end(),
 137                 video.current_resolution()) - resolutions_.begin();
 138 
 139         res_list.set_values(options, current_res);
 140 }

 142 std::map<std::string, string_map> preferences_dialog::get_friends_list_row_data(const acquaintance& entry)
 143 {
 144         std::map<std::string, string_map> data;
 145         string_map item;
 146 
 147         std::string image = "friend.png";
 148         std::string descriptor = _("friend");
 149         std::string notes;
 150 
 151         if(entry.get_status() == "ignore") {
 152                 image = "ignore.png";
 153                 descriptor = _("ignored");
 154         }
 155 
 156         if(!entry.get_notes().empty()) {
 157                 notes = " <small>(" + entry.get_notes() + ")</small>";
 158         }
 159 
 160         item["use_markup"] = "true";
 161 
 162         item["label"] = "misc/status-" + image;
 163         data.emplace("friend_icon", item);
 164 
 165         item["label"] = entry.get_nick() + notes;
 166         data.emplace("friend_name", item);
 167 
 168         item["label"] = "<small>" + descriptor + "</small>";
 169         data.emplace("friend_status", item);
 170 
 171         return data;
 172 }

 174 void preferences_dialog::on_friends_list_select(listbox& list, text_box& textbox)
 175 {
 176         const int num_friends = get_acquaintances().size();
 177         const int sel = list.get_selected_row();
 178 
 179         if(sel < 0 || sel >= num_friends) {
 180                 return;
 181         }
 182 
 183         std::map<std::string, acquaintance>::const_iterator who = get_acquaintances().begin();
 184         std::advance(who, sel);
 185 
 186         textbox.set_value(who->second.get_nick() + " " + who->second.get_notes());
 187 }

 203 void preferences_dialog::add_friend_list_entry(const bool is_friend, text_box& textbox, window& window)
 204 {
 205         std::string username = textbox.text();
 206         if(username.empty()) {
 207                 gui2::show_transient_message(window.video(), "", _("No username specified"), "", false, false, true);
 208                 return;
 209         }
 210 
 211         std::string reason;
 212 
 213         size_t pos = username.find_first_of(' ');
 214         if(pos != std::string::npos) {
 215                 reason = username.substr(pos + 1);
 216                 username = username.substr(0, pos);
 217         }
 218 
 219         acquaintance* entry = add_acquaintance(username, (is_friend ? "friend": "ignore"), reason);
 220         if(!entry) {
 221                 gui2::show_transient_message(window.video(), _("Error"), _("Invalid username"), "", false, false, true);
 222                 return;
 223         }
 224 
 225         textbox.clear();
 226 
 227         listbox& list = find_widget<listbox>(&window, "friends_list", false);
 228         list.add_row(get_friends_list_row_data(*entry));
 229 
 230         update_friends_list_controls(window, list);
 231 }

 233 void preferences_dialog::remove_friend_list_entry(listbox& friends_list, text_box& textbox, window& window)
 234 {
 235         const int selected_row = std::max(0, friends_list.get_selected_row());
 236 
 237         std::map<std::string, acquaintance>::const_iterator who = get_acquaintances().begin();
 238         std::advance(who, selected_row);
 239 
 240         const std::string to_remove = !textbox.text().empty() ? textbox.text() : who->second.get_nick();
 241 
 242         if(to_remove.empty()) {
 243                 gui2::show_transient_message(window.video(), "", _("No username specified"), "", false, false, true);
 244                 return;
 245         }
 246 
 247         if(!remove_acquaintance(to_remove)) {
 248                 gui2::show_transient_error_message(window.video(), _("Not on friends or ignore lists"));
 249                 return;
 250         }
 251 
 252         textbox.clear();
 253 
 254         listbox& list = find_widget<listbox>(&window, "friends_list", false);
 255         list.remove_row(selected_row);
 256 
 257         update_friends_list_controls(window, list);
 258 }

 739 listbox& preferences_dialog::setup_hotkey_list(window& window)
 740 {
 741         const std::string& default_icon = "misc/empty.png~CROP(0,0,15,15)";
 742 
 743         std::map<std::string, string_map> row_data;

 746         t_string& row_action = row_data["lbl_desc"]["label"];

 791 void preferences_dialog::add_hotkey_callback(listbox& hotkeys)
 792 {
 793         CVideo& video = hotkeys.get_window()->video();
 794         int row_number = hotkeys.get_selected_row();
 795         const hotkey::hotkey_command& hotkey_item = *visible_hotkeys_[row_number];
 796 
 797         gui2::dialogs::hotkey_bind bind_dlg(hotkey_item.command);
 798         bind_dlg.show(video);
 799 
 800         hotkey::hotkey_ptr newhk = bind_dlg.get_new_binding();
 801         hotkey::hotkey_ptr oldhk;
 802 
 803         // only if not cancelled.
 804         if(newhk.get() == nullptr) {
 805                 return;
 806         }
 807 
 808         for(const hotkey::hotkey_ptr& hk : hotkey::get_hotkeys()) {
 809                 if(!hk->is_disabled() && newhk->bindings_equal(hk)) {
 810                         oldhk = hk;
 811                 }
 812         }
 813 
 814         hotkey::scope_changer scope_restorer;
 815         hotkey::set_active_scopes(hotkey_item.scope);
 816 
 817         if(oldhk && oldhk->get_command() == hotkey_item.command) {
 818                 return;
 819         }
 820 
 821         if(oldhk && oldhk->get_command() != "null") {
 822                 const std::string text = vgettext("<b>$hotkey_sequence|</b> is in use by <b>$old_hotkey_action|</b>.\nDo you wish to reassign it to <b>$new_hotkey_action|</b>?", {
 823                         {"hotkey_sequence",   oldhk->get_name()},
 824                         {"old_hotkey_action", hotkey::get_description(oldhk->get_command())},
 825                         {"new_hotkey_action", hotkey::get_description(newhk->get_command())}
 826                 });
 827 
 828                 const int res = gui2::show_message(video, _("Reassign Hotkey"), text, gui2::dialogs::message::yes_no_buttons, true);
 829                 if(res != gui2::window::OK) {
 830                         return;
 831                 }
 832         }
 833 
 834         hotkey::add_hotkey(newhk);
 835 
 836         // We need to recalculate all hotkey names in because we might have removed a hotkey from another command.
 837         for(size_t i = 0; i < hotkeys.get_item_count(); ++i) {
 838                 const hotkey::hotkey_command& hotkey_item_row = *visible_hotkeys_[i];
 839                 find_widget<label>(hotkeys.get_row_grid(i), "lbl_hotkey", false).set_label(hotkey::get_names(hotkey_item_row.command));
 840         }
 841 }

 865 void preferences_dialog::hotkey_type_filter_callback(window& window) const
 866 {
 867         const multimenu_button& hotkey_menu = find_widget<const multimenu_button>(&window, "hotkey_category_menu", false);
 868 
 869         boost::dynamic_bitset<> toggle_states = hotkey_menu.get_toggle_states();
 870         boost::dynamic_bitset<> res(visible_hotkeys_.size());
 871 
 872         if(!toggle_states.none()) {
 873                 for(size_t h = 0; h < visible_hotkeys_.size(); ++h) {
 874                         int index = 0;
 875 
 876                         for(size_t i = 0; i < cat_names_.size(); ++i) {
 877                                 hotkey::HOTKEY_CATEGORY cat = hotkey::HOTKEY_CATEGORY(i);
 878 
 879                                 if(visible_hotkeys_[h]->category == cat) {
 880                                         index = i;
 881                                         break;
 882                                 }
 883                         }
 884 
 885                         res[h] = toggle_states[index];
 886                 }
 887         } else {
 888                 // Nothing selected. It means that *all* categories are shown.
 889                 res = ~res;
 890         }
 891 
 892         find_widget<listbox>(&window, "list_hotkeys", false).set_row_shown(res);
 893 }

 900                 adv_preferences_cfg_[selected_row]["type"].str());

 910                         gui2::dialogs::select_orb_colors::display(window.video());

 912                         WRN_GUI_L << "Invalid or unimplemented custom advanced prefs option: " << selected_field << "\n";
 913                 }
 914 
 915                 // Add more options here as needed
 916         }
 917 
 918         const bool has_description = !adv_preferences_cfg_[selected_row]["description"].empty();
 919 
 920         if(has_description || (selected_type != ADVANCED_PREF_TYPE::SPECIAL && selected_type != ADVANCED_PREF_TYPE::TOGGLE)) {

 942 static int index_in_pager_range(const int& first, const stacked_widget& pager)
 943 {
 944         // Ensure the specified index is between 0 and one less than the max
 945         // number of pager layers (since get_layer_count returns one-past-end).
 946         return std::min<int>(std::max(0, first), pager.get_layer_count() - 1);
 947 }

 949 void preferences_dialog::pre_show(window& window)
 950 {
 951         set_always_save_fields(true);
 952 
 953         //
 954         // Status labels
 955         // These need to be set here in pre_show, once the fields are initialized. For some reason, this
 956         // is not the case for those in Advanced
 957         //
 958 
 959         gui2::bind_status_label<slider>(&window, "max_saves_slider", [](slider& s)->std::string {
 960                 return s.get_value() == INFINITE_AUTO_SAVES ? _("") : s.get_value_label().str();
 961         });
 962 
 963         gui2::bind_status_label<slider>(&window, "turbo_slider");
 964 
 965         gui2::bind_status_label<slider>(&window, "scaling_slider",   [](slider& s)->std::string {
 966                 return s.get_value_label() + "%";
 967         });
 968 
 969         listbox& selector = find_widget<listbox>(&window, "selector", false);
 970         stacked_widget& pager = find_widget<stacked_widget>(&window, "pager", false);
 971 
 972         pager.set_find_in_all_layers(true);
 973 
 974         connect_signal_notify_modified(selector,
 975                 std::bind(&preferences_dialog::on_page_select, this, std::ref(window)));
 976 
 977         window.keyboard_capture(&selector);
 978 
 979         VALIDATE(selector.get_item_count() == pager.get_layer_count(),
 980                 "The preferences pager and its selector listbox do not have the same number of items.");
 981 
 982         const int main_index = index_in_pager_range(initial_index_.first, pager);
 983 
 984         // Loops through each pager layer and checks if it has both a tab bar
 985         // and stack. If so, it initializes the options for the former and
 986         // selects the specified layer of the latter.
 987         for(unsigned int i = 0; i < pager.get_layer_count(); ++i) {
 988                 listbox* tab_selector = find_widget<listbox>(
 989                         pager.get_layer_grid(i), "tab_selector", false, false);
 990 
 991                 stacked_widget* tab_pager = find_widget<stacked_widget>(
 992                         pager.get_layer_grid(i), "tab_pager", false, false);
 993 
 994                 if(tab_pager && tab_selector) {
 995                         const int ii = static_cast<int>(i);
 996                         const int tab_index = index_in_pager_range(initial_index_.second, *tab_pager);
 997                         const int to_select = (ii == main_index ? tab_index : 0);
 998 
 999                         // Initialize tabs for this page
1000                         initialize_tabs(window, *tab_selector);
1001 
1002                         tab_selector->select_row(to_select);
1003                         tab_pager->select_layer(to_select);
1004                 }
1005         }
1006 
1007         // Finally, select the initial main page
1008         selector.select_row(main_index);
1009         pager.select_layer(main_index);
1010 }
