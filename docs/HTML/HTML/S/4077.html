 391 (defun wesnoth-find-next (type)
 392   "Find the next element of TYPE.
 393 TYPE is a symbol representing an element type, or a list of
 394 element types to find."
 395   (let ((element (wesnoth-next-element)))
 396     (while (and element (if (listp type)
 397                             (not (member (car element) type))
 398                           (not (eq (car element) type))))
 399       (setq element (wesnoth-next-element)))
 400     (when (if (listp type)
 401               (member (car element) type)
 402             (eq (car element) type))
 403       element)))

 405 (defun wesnoth-next-element ()
 406   "Move to the next element in the buffer.
 407 Return non-nil when an element is found.  Otherwise, return nil."
 408   (interactive)
 409   (save-match-data
 410     (and (or (eolp) (looking-at "[}\t ]"))
 411          (search-forward-regexp "[^}
 412 \t ]" (point-max) t)
 413          (forward-char -1)))
 414   (let ((details (wesnoth-element-type (point))))
 415     (save-match-data
 416       (when (nth 2 details)
 417         (goto-char (nth 2 details))
 418         (while (and (nth 3 (parse-partial-sexp
 419                             (save-excursion (search-backward-regexp
 420                                              (wesnoth-element t) (point-min) t)
 421                                             (point))
 422                             (point)))
 423                     (search-forward "\"" (point-max) t)))))
 424     details))

 473 (defun wesnoth-element-type (point)
 474   "Return details regarding the element at POINT.
 475 A list is returned, the elements of the list represent the
 476 following, respectively: A symbol representing the type of
 477 element; the position of the start of the element and the
 478 position of the end of the element.  POINT must be at the start
 479 of the element."

 481          '(("{\\(.*?[/\]\\)+}" . nil)  ;; pathnames
 482            ("{\\(\\w\\|[{_@~]\\)+" . macro)
 483            ("\\[\\+?[^/]+?\\]" . tag-opening)

 955 If COMPLETEP is non-nil, do not move forward a line when scanning
