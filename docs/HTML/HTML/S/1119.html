  35 static bool contains_ignore_case(const std::string& str1, const std::string& str2)
  36 {
  37         if(str2.size() > str1.size()) {
  38                 return false;
  39         }
  40 
  41         for(size_t i = 0; i < str1.size() - str2.size() + 1; i++) {
  42                 bool ok = true;
  43                 for(size_t j = 0; j < str2.size(); j++) {
  44                         if(std::tolower(str1[i + j]) != std::tolower(str2[j])) {
  45                                 ok = false;
  46                                 break;
  47                         }
  48                 }
  49 
  50                 if(ok) {
  51                         return true;
  52                 }
  53         }
  54 
  55         return false;
  56 }

 424         struct type_list

 426                 explicit type_list() : games(), games_filtered() {}
 427 
 428                 std::vector<level_ptr> games;
 429                 std::vector<size_t> games_filtered;
 430 
 431                 void apply_filter(const int player_count, const std::string& name_filter)
 432                 {
 433                         games_filtered.clear();
 434 
 435                         for(size_t i = 0; i < games.size(); i++) {
 436                                 const bool num_players_match = player_count == 1 || games[i]->player_count_filter(player_count);
 437 
 438                                 if(contains_ignore_case(games[i]->name(), name_filter) && num_players_match) {
 439                                         games_filtered.push_back(i);
 440                                 }
 441                         }
 442                 }
 443 
 444                 void reset_filter()
 445                 {
 446                         games_filtered.resize(games.size());
 447                         std::iota(games_filtered.begin(), games_filtered.end(), 0);
 448                 }
 449         };
