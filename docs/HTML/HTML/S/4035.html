 121 #define ON_SIGHTING ID SIGHTING_SIDE SIGHTED_FILTER ACTION_WML
 122     # NOTE: As of version 1.11, this macro is unnecessary, since sighted
 123     # events now work as intended.
 124     #
 125     # This is intended to be used when a regular sighted event can't be relied
 126     # on (that'd be most of the time). This event triggers when a unit matching
 127     # SIGHTED_FILTER is first seen by a unit of side SIGHTING_SIDE. This works
 128     # whether the sighting happens due to moving, recruiting, recalling,
 129     # attacking (in case DSU was on before the attack) or advancing. If none of
 130     # those actions manage to "catch" the moment of sighting, then it will occur
 131     # when the current side ends their turn.
 132     #
 133     # The 'unit' and 'second_unit' message speaker shortcuts as well as the
 134     # auto-stored $unit and $second_unit should work normally: the sighting unit
 135     # is the primary unit and the sighted unit is the secondary unit.
 136     #
 137     # You need to provide a unique ID if you use several of these events in the
 138     # same scenario. Otherwise it can be left empty.
 139     #
 140     # Only works if SIGHTING_SIDE has fog and/or shroud.
 141     #
 142     # Example:
 143     #! {ON_SIGHTING () 1 side=2,3 (
 144     #!     [message]
 145     #!         speaker=unit
 146     #!         message= _ "I can see a side $second_unit.side $second_unit.language_name|!"
 147     #!     [/message]
 148     #!     [message]
 149     #!         speaker=second_unit
 150     #!         message= _ "Oh my! I was seen by a side $unit.side $unit.language_name|!"
 151     #!     [/message]
 152     #! )}
 153     #!
 154     [event]
 155         name=SIGHTED_{ID}_trigger_action_WML
 156 
 157         {ACTION_WML}
 158     [/event]
 159 
 160     [event]
 161         name=moveto,recruit,recall,attack,post advance,side turn
 162         first_time_only=no
 163 
 164         [filter]
 165             side={SIGHTING_SIDE}
 166 
 167             [or]
 168                 {SIGHTED_FILTER}
 169             [/or]
 170         [/filter]
 171 
 172         [if]
 173             [variable]
 174                 name=SIGHTED_{ID}_finished
 175                 not_equals=yes
 176             [/variable]
 177 
 178             [then]
 179                 [store_side]
 180                     side={SIGHTING_SIDE}
 181                     variable=stored_sighting_side
 182                 [/store_side]
 183 
 184                 [if]
 185                     [have_unit]
 186                         {SIGHTED_FILTER}
 187 
 188                         [filter_vision]
 189                             side={SIGHTING_SIDE}
 190                         [/filter_vision]
 191                     [/have_unit]
 192 
 193                     [and]
 194                         [variable]
 195                             name=stored_sighting_side.fog
 196                             boolean_equals=yes
 197                         [/variable]
 198 
 199                         [or]
 200                             [variable]
 201                                 name=stored_sighting_side.shroud
 202                                 boolean_equals=yes
 203                             [/variable]
 204                         [/or]
 205                     [/and]
 206 
 207                     [then]
 208                         [fire_event]
 209                             name=SIGHTED_{ID}_trigger_action_WML
 210 
 211                             [primary_unit]
 212                                 side={SIGHTING_SIDE}
 213 
 214                                 [filter_location]
 215                                     [filter]
 216                                         {SIGHTED_FILTER}
 217 
 218                                         [not]
 219                                             side={SIGHTING_SIDE}
 220                                         [/not]
 221                                     [/filter]
 222 
 223                                     radius="$($this_unit.max_moves + 1)"
 224                                 [/filter_location]
 225                             [/primary_unit]
 226 
 227                             [secondary_unit]
 228                                 {SIGHTED_FILTER}
 229 
 230                                 [filter_vision]
 231                                     side={SIGHTING_SIDE}
 232                                 [/filter_vision]
 233                             [/secondary_unit]
 234                         [/fire_event]
 235 
 236                         [set_variable]
 237                             name=SIGHTED_{ID}_finished
 238                             value=yes
 239                         [/set_variable]
 240                     [/then]
 241 
 242                     [else]
 243                         [allow_undo]
 244                         [/allow_undo]
 245                     [/else]
 246                 [/if]
 247 
 248                 [clear_variable]
 249                     name=stored_sighting_side
 250                 [/clear_variable]
 251             [/then]
 252 
 253             [else]
 254                 [allow_undo]
 255                 [/allow_undo]
 256             [/else]
 257         [/if]
 258     [/event]
 259 
 260     [event]
 261         name=victory
 262 
 263         [clear_variable]
 264             name=SIGHTED_{ID}_finished
 265         [/clear_variable]
 266     [/event]
 267 #enddef

 321 #define FOREACH ARRAY_VALUE VAR

 323 {VARIABLE {VAR} 0}
 324 [while]
 325     [variable]
 326     name={VAR}
 327     less_than=${ARRAY_VALUE}.length
 328     [/variable]
 329     [do]
 330 #enddef

 332 #define NEXT VAR
 333 # Macro to end a WML clause that iterates over an array.
 334     [set_variable]
 335     name={VAR}
 336     add=1
 337     [/set_variable]
 338     [/do]
 339 [/while]
 340 {CLEAR_VARIABLE {VAR}}
 341 #enddef
