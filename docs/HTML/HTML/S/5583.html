   8 function helper.get_sides(cfg)

  10                 local i = s.i
  11                 while i < #wesnoth.sides do
  12                         i = i + 1

  14                                 s.i = i

 140 --! Displays a WML message box with attributes from table @attr and options
 141 --! from table @options.
 142 --! @return the index of the selected option.
 143 --! @code
 144 --! local result = helper.get_user_choice({ speaker = "narrator" },
 145 --!     { "Choice 1", "Choice 2" })
 146 --! @endcode
 147 function helper.get_user_choice(attr, options)
 148         local result = 0
 149         function wesnoth.__user_choice_helper(i)
 150                 result = i
 151         end
 152         local msg = {}
 153         for k,v in pairs(attr) do
 154                 msg[k] = attr[k]
 155         end
 156         for k,v in ipairs(options) do
 157                 table.insert(msg, { "option", { message = v,
 158                         { "command", { { "lua", {
 159                                 code = string.format("wesnoth.__user_choice_helper(%d)", k)
 160                         }}}}}})
 161         end
 162         wml_actions.message(msg)
 163         wesnoth.__user_choice_helper = nil
 164         return result
 165 end

 172 --! Returns an iterator over adjacent locations that can be used in a for-in loop.
 173 -- Not deprecated because, unlike wesnoth.map.get_adjacent_tiles,
 174 -- this verifies that the locations are on the map.
 175 function helper.adjacent_tiles(x, y, with_borders)
 176         local x1,y1,x2,y2,b = 1,1,wesnoth.get_map_size()
 177         if with_borders then
 178                 x1 = x1 - b
 179                 y1 = y1 - b
 180                 x2 = x2 + b
 181                 y2 = y2 + b
 182         end
 183         local adj = {wesnoth.map.get_adjacent_tiles(x, y)}
 184         local i = 0
 185         return function()
 186                 while i < #adj do
 187                         i = i + 1
 188                         local u, v = adj[i][1], adj[i][2]
 189                         if u >= x1 and u <= x2 and v >= y1 and v <= y2 then
 190                                 return u, v
 191                         end
 192                 end
 193                 return nil
 194         end
 195 end

 197 function helper.rand (possible_values, random_func)
 198         random_func = random_func or wesnoth.random
 199         assert(type(possible_values) == "table" or type(possible_values) == "string", string.format("helper.rand expects a string or table as parameter, got %s instead", type(possible_values)))
 200 
 201         local items = {}
 202         local num_choices = 0
 203 
 204         if type(possible_values) == "string" then
 205                 -- split on commas
 206                 for word in possible_values:gmatch("[^,]+") do
 207                         -- does the word contain two dots? If yes, that's a range
 208                         local dots_start, dots_end = word:find("%.%.")
 209                         if dots_start then
 210                                 -- split on the dots if so and cast as numbers
 211                                 local low = tonumber(word:sub(1, dots_start-1))
 212                                 local high = tonumber(word:sub(dots_end+1))
 213                                 -- perhaps someone passed a string as part of the range, intercept the issue
 214                                 if not (low and high) then
 215                                         wesnoth.message("Malformed range: " .. possible_values)
 216                                         table.insert(items, word)
 217                                         num_choices = num_choices + 1
 218                                 else
 219                                         if low > high then
 220                                                 -- low is greater than high, swap them
 221                                                 low, high = high, low
 222                                         end
 223 
 224                                         -- if both ends represent the same number, then just use that number
 225                                         if low == high then
 226                                                 table.insert(items, low)
 227                                                 num_choices = num_choices + 1
 228                                         else
 229                                                 -- insert a table representing the range
 230                                                 table.insert(items, {low, high})
 231                                                 -- how many items does the range contain? Increase difference by 1 because we include both ends
 232                                                 num_choices = num_choices + (high - low) + 1
 233                                         end
 234                                 end
 235                         else
 236                                 -- handle as a string
 237                                 table.insert(items, word)
 238                                 num_choices = num_choices + 1
 239                         end
 240                 end
 241         else
 242                 num_choices = #possible_values
 243                 items = possible_values
 244                 -- We need to parse ranges separately anyway
 245                 for i, val in ipairs(possible_values) do
 246                         if type(val) == "table" then
 247                                 assert(#val == 2 and type(val[1]) == "number" and type(val[2]) == "number", "Malformed range for helper.rand")
 248                                 if val[1] > val[2] then
 249                                         val = {val[2], val[1]}
 250                                 end
 251                                 num_choices = num_choices + (val[2] - val[1])
 252                         end
 253                 end
 254         end
 255 
 256         local idx = random_func(1, num_choices)
 257 
 258         for i, item in ipairs(items) do
 259                 if type(item) == "table" then -- that's a range
 260                         local elems = item[2] - item[1] + 1 -- amount of elements in the range, both ends included
 261                         if elems >= idx then
 262                                 return item[1] + elems - idx
 263                         else
 264                                 idx = idx - elems
 265                         end
 266                 else -- that's a single element
 267                         idx = idx - 1
 268                         if idx == 0 then
 269                                 return item
 270                         end
 271                 end
 272         end
 273 
 274         return nil
 275 end

 288 function helper.round( number )
 289         -- code converted from util.hpp, round_portable function
 290         -- round half away from zero method
 291         if number >= 0 then
 292                 number = math.floor( number + 0.5 )
 293         else
 294                 number = math.ceil ( number - 0.5 )
 295         end

 300 function helper.shuffle( t, random_func)
 301         random_func = random_func or wesnoth.random
 302         -- since tables are passed by reference, this is an in-place shuffle
 303         -- it uses the Fisher-Yates algorithm, also known as Knuth shuffle
 304         assert( type( t ) == "table", string.format( "helper.shuffle expects a table as parameter, got %s instead", type( t ) ) )
 305         local length = #t
 306         for index = length, 2, -1 do
 307                 local random = random_func( 1, index )
 308                 t[index], t[random] = t[random], t[index]
 309         end
 310 end
