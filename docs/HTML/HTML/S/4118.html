   3 """
   4 wmliterator.py -- Python routines for navigating a Battle For Wesnoth WML tree
   5 Author: Sapient (Patrick Parker), 2007
   6 
   7 Purpose:
   8  The WmlIterator class can be used to analyze and search the structure of WML
   9  files non-invasively (i.e. preserving existing line structure), and its main
  10  use is to determine when a transformation of deprecated content needs to take
  11  place. (I wrote it was because wmllint was trying to do a lot of things with
  12  regular expressions which really required a more algorithmic approach. Also,
  13  wmllint was often inconsistent with correct handling of comments and values
  14  inside strings.)
  15 
  16 Limitations:
  17  The WmlIterator does not attempt to expand macros, only to recognize them as
  18  another level of nesting. Also, the handling of multiple assignment syntax
  19  is somewhat limited (for similar reasons). Adding seamless support for these
  20  would be ideal, but it presents a design challenge since the iteration is
  21  supposed to be non-invasive. Thus, the current behavior is considered good
  22  enough for now.
  23 """

 110 class WmlIterator(object):
 111     """Return an iterable WML navigation object.
 112     Initialize with a list of lines or a file; if the the line list is
 113     empty and the filename is specified, lines will be read from the file.
 114 
 115     Note: if changes are made to lines while iterating, this may produce
 116     unexpected results. In such case, seek() to the line number of a
 117     scope behind where changes were made.
 118 Important Attributes:
 119     lines - this is an internal list of all the physical lines
 120     scopes - this is an internal list of all open scopes (as iterators)
 121              note: when retrieving an iterator from this list, always
 122              use a copy to perform seek() or next(), and not the original
 123     element - the wml tag, key, or macro name for this logical line
 124               (in complex cases, this may be a tuple of elements...
 125               see parseElements for list of possible values)
 126     text - the exact text of this logical line, as it appears in the
 127            original source, and ending with a newline
 128            note: the logical line also includes multi-line quoted strings
 129     span - the number of physical lines in this logical line:
 130            always 1, unless text contains a multi-line quoted string
 131     lineno - a zero-based line index marking where this text begins
 132     """
 133     def __init__(self, lines=None, filename=None, begin=-1):
 134         "Initialize a new WmlIterator."
 135         self.fname = filename
 136         if lines is None:
 137             lines = []
 138             if filename:
 139                 try:
 140                     with codecs.open(self.fname, "r", "utf8") as ifp:
 141                         lines = ifp.readlines()
 142                 except Exception:
 143                     self.printError('error opening file')
 144         self.lines = lines
 145         self.reset()
 146         self.seek(begin)

 148     def parseQuotes(self, lines):
 149         """Return the line or multiline text if a quote spans multiple lines"""
 150         text = lines[self.lineno]
 151         span = 1
 152         begincomment = text.find('#')
 153         if begincomment < 0:
 154             begincomment = None
 155         beginquote = text[:begincomment].find('<<')
 156         while beginquote >= 0:
 157             endquote = -1
 158             beginofend = beginquote+2
 159             while endquote < 0:
 160                 endquote = text.find('>>', beginofend)
 161                 if endquote < 0:
 162                     if self.lineno + span >= len(lines):
 163                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 164                         return text, span
 165                     beginofend = len(text)
 166                     text += lines[self.lineno + span]
 167                     span += 1
 168             begincomment = text.find('#', endquote+2)
 169             if begincomment < 0:
 170                 begincomment = None
 171             beginquote = text[:begincomment].find('<<', endquote+2)
 172         beginquote = text[:begincomment].find('"')
 173         while beginquote >= 0:
 174             endquote = -1
 175             beginofend = beginquote+1
 176             while endquote < 0:
 177                 endquote = text.find('"', beginofend)
 178                 if endquote < 0:
 179                     if self.lineno + span >= len(lines):
 180                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 181                         return text, span
 182                     beginofend = len(text)
 183                     text += lines[self.lineno + span]
 184                     span += 1
 185             begincomment = text.find('#', endquote+1)
 186             if begincomment < 0:
 187                 begincomment = None
 188             beginquote = text[:begincomment].find('"', endquote+1)
 189         return text, span

 191     def closeScope(self, scopes, closerElement):
 192         """Close the most recently opened scope. Return false if not enough scopes.
 193         note: directives close all the way back to the last open directive
 194         non-directives cannot close a directive and will no-op in that case."""
 195         try:
 196             if isDirective(closerElement):
 197                 while not isDirective(scopes.pop()):
 198                     pass
 199             elif (closerElement==closeMacroType):
 200                 elem = ''
 201                 while not elem.startswith('{'):
 202                     closed = scopes.pop()
 203                     elem = closed
 204                     if isinstance(closed, WmlIterator):
 205                         elem = closed.element
 206                     if isDirective(elem):
 207                         self.printScopeError(closerElement)
 208                         scopes.append(closed) # to reduce additional errors (hopefully)
 209                         return True
 210             elif not isDirective(scopes[-1]):
 211                 closed = scopes.pop()
 212                 elem = closed
 213                 if isinstance(closed, WmlIterator):
 214                     elem = closed.element
 215                 if (elem.startswith('{') and closerElement != closeMacroType):
 216                     scopes.append(closed)
 217                 elif (isOpener(elem) and closerElement != '[/'+elem[1:]
 218                 and '+'+closerElement != elem[1]+'[/'+elem[2:]):
 219                     self.printError('reached', closerElement, 'at line', self.lineno+1, 'before closing scope', elem)
 220                     scopes.append(closed) # to reduce additional errors (hopefully)
 221             return True
 222         except IndexError:
 223             return False

 225     def parseElements(self, text):
 226         """Remove any closed scopes, return a list of element names
 227         and list of new unclosed scopes
 228     Element Types:
 229         tags: one of "[tag_name]" or "[/tag_name]"
 230             [tag_name] - opens a scope
 231             [/tag_name] - closes a scope
 232         keys: either "key=" or ("key1=", "key2=") for multi-assignment
 233             key= - does not affect the scope
 234             key1,key2= - multi-assignment returns multiple elements
 235         directives: one of "#ifdef", "#ifndef", "#ifhave", "#ifnhave", "#ifver", "#ifnver", "#else", "#endif", "#define", "#enddef"
 236             #ifdef - opens a scope
 237             #ifndef - opens a scope
 238             #ifhave - opens a scope
 239             #ifnhave - opens a scope
 240             #ifver - opens a scope
 241             #ifnver - opens a scope
 242             #else - closes a scope, also opens a new scope
 243             #endif - closes a scope
 244             #define - opens a scope
 245             #enddef - closes a scope
 246         macro calls: "{MACRO_NAME}"
 247             {MACRO_NAME - opens a scope
 248             } - closes a scope
 249         """
 250         elements = [] #(elementType, sortPos, scopeDelta)
 251         # first remove any lua strings
 252         beginquote = text.find('<<')
 253         while beginquote >= 0:
 254             endquote = text.find('>>', beginquote+2)
 255             if endquote < 0:
 256                 text = text[:beginquote]
 257                 beginquote = -1 #terminate loop
 258             else:
 259                 text = text[:beginquote] + text[endquote+2:]
 260                 beginquote = text.find('<<')
 261         # remove any quoted strings
 262         beginquote = text.find('"')
 263         while beginquote >= 0:
 264             endquote = text.find('"', beginquote+1)
 265             if endquote < 0:
 266                 text = text[:beginquote]
 267                 beginquote = -1 #terminate loop
 268             else:
 269                 text = text[:beginquote] + text[endquote+1:]
 270                 beginquote = text.find('"')
 271         # next remove any comments
 272         text = text.lstrip()
 273         commentSearch = 1
 274         if text.startswith('#ifdef'):
 275             return (['#ifdef'],)*2
 276         elif text.startswith('#ifndef'):
 277             return (['#ifndef'],)*2
 278         elif text.startswith('#ifhave'):
 279             return (['#ifhave'],)*2
 280         elif text.startswith('#ifnhave'):
 281             return (['#ifnhave'],)*2
 282         elif text.startswith('#ifver'):
 283             return (['#ifver'],)*2
 284         elif text.startswith('#ifnver'):
 285             return (['#ifnver'],)*2
 286         elif text.startswith('#else'):
 287             if not self.closeScope(self.scopes, '#else'):
 288                 self.printScopeError('#else')
 289             return (['#else'],)*2
 290         elif text.startswith('#endif'):
 291             if not self.closeScope(self.scopes, '#endif'):
 292                 self.printScopeError('#endif')
 293                 return ['#endif'], []
 294         elif text.startswith('#define'):
 295             return (['#define'],)*2
 296         elif text.find('#enddef') >= 0:
 297             elements.append(('#enddef', text.find('#enddef'), -1))
 298         elif text.startswith('#po:') or text.startswith('# po:'):
 299             elements.append(("#po", 0, 0))
 300         else:
 301             commentSearch = 0
 302         begincomment = text.find('#', commentSearch)
 303         if begincomment >= 0:
 304             text = text[:begincomment]
 305         #now find elements in a loop
 306         for m in tagPattern.finditer(text):
 307             delta = 1
 308             if isCloser(m.group(2)):
 309                 delta = -1
 310             elements.append((m.group(2), m.start(), delta))
 311         for m in keyPattern.finditer(text):
 312             for i, k in enumerate(keySplit.split(m.group(0))):
 313                 if k:
 314                     elements.append((k+'=', m.start()+i, 0))
 315         for m in macroOpenPattern.finditer(text):
 316             elements.append((m.group(1), m.start(), 1))
 317         for m in macroClosePattern.finditer(text):
 318             elements.append((closeMacroType, m.start(), -1))
 319         #sort by start position
 320         elements.sort(key=lambda x:x[1])
 321         resultElements = []
 322         openedScopes = []
 323         for elem, sortPos, scopeDelta in elements:
 324             while scopeDelta < 0:
 325                 if not(self.closeScope(openedScopes, elem)\
 326                         or self.closeScope(self.scopes, elem)):
 327                     self.printScopeError(elem)
 328                 scopeDelta += 1
 329             while scopeDelta > 0:
 330                 openedScopes.append(elem)
 331                 scopeDelta -= 1
 332             resultElements.append(elem)
 333         return resultElements, openedScopes

 361     def seek(self, lineno, clearEnd=True):
 362         """Move the iterator to a specific line number"""
 363         if clearEnd:
 364             self.endScope = None
 365         if lineno < self.lineno:
 366             for scope in reversed(self.scopes):
 367                 # if moving backwards, try to re-use a scope iterator
 368                 if scope.lineno <= lineno:
 369                     # copy the scope iterator's state to self
 370                     self.__dict__ = dict(scope.__dict__)
 371                     self.scopes = scope.scopes[:]
 372                     self.nextScopes = scope.nextScopes[:]
 373                     break
 374             else:
 375                 # moving backwards past all scopes forces a reset
 376                 self.reset()
 377         while self.lineno + self.span - 1 < lineno:
 378             self.next()
 379         return self

 412     def next(self):
 413         """Move the iterator to the next line number
 414         note: May raise StopIteration"""
 415         if not self.hasNext():
 416             if self.scopes:
 417                 self.printError("reached EOF with open scopes", self.scopes)
 418             raise StopIteration
 419         self.lineno = self.lineno + self.span
 420         self.text, self.span = self.parseQuotes(self.lines)
 421         self.scopes.extend(self.nextScopes)
 422         self.element, nextScopes = self.parseElements(self.text)
 423         self.nextScopes = []
 424         for elem in nextScopes:
 425         # remember scopes by storing a copy of the iterator
 426             copyItor = self.copy()
 427             copyItor.element = elem
 428             self.nextScopes.append(copyItor)
 429             copyItor.nextScopes.append(copyItor)
 430         if(len(self.element) == 1):
 431             # currently we only wish to handle simple single assignment syntax
 432             self.element = self.element[0]
 433         if self.endScope is not None and not self.scopes.count(self.endScope):
 434             raise StopIteration
 435         return self

 463     def printError(nav, *misc):
 464         """Print error associated with a given file; avoid printing duplicates"""
 465         if nav.fname:
 466             silenceValue = ' '.join(map(str, misc))
 467             if nav.fname not in silenceErrors:
