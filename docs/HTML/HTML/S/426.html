  61 static std::string format_level_string(const int level)
  62 {
  63         std::string lvl = std::to_string(level);
  64 
  65         if(level < 1) {
  66                 return "<span color='#969696'>" + lvl + "</span>";
  67         } else if(level == 1) {
  68                 return lvl;
  69         } else if(level == 2) {
  70                 return "<b>" + lvl + "</b>";
  71         } else { // level must be > 2
  72                 return"<b><span color='#ffffff'>" + lvl + "</span></b>";
  73         }
  74 
  75 }

  77 static std::string format_if_leader(unit_const_ptr u, const std::string str)
  78 {
  79         return (*u).can_recruit() ? "<span color='#cdad00'>" + str + "</span>" : str;
  80 }

  82 static std::string format_movement_string(unit_const_ptr u)
  83 {
  84         const int moves_left = (*u).movement_left();
  85         const int moves_max  = (*u).total_movement();
  86 
  87         std::string color = "#00ff00";
  88 
  89         if(moves_left == 0) {
  90                 color = "#ff0000";
  91         } else if(moves_left < moves_max) {
  92                 color = "#ffff00";
  93         }
  94 
  95         return formatter() << "<span color='" << color << "'>" << moves_left << "/" << moves_max << "</span>";
  96 }

  98 void unit_list::pre_show(window& window)
  99 {
 100         listbox& list = find_widget<listbox>(&window, "units_list", false);
 101 
 102         connect_signal_notify_modified(list, std::bind(&unit_list::list_item_clicked, this, std::ref(window)));
 103 
 104         list.clear();
 105 
 106         window.keyboard_capture(&list);
 107 
 108         for(const unit_const_ptr& unit : unit_list_) {
 109                 std::map<std::string, string_map> row_data;
 110                 string_map column;
 111 
 112                 column["use_markup"] = "true";
 113 
 114                 column["label"] = format_if_leader(unit, unit->type_name());
 115                 row_data.emplace("unit_type", column);
 116 
 117                 const std::string& name = !unit->name().empty() ? format_if_leader(unit, unit->name().str()) : font::unicode_en_dash;
 118                 column["label"] = name;
 119                 row_data.emplace("unit_name", column);
 120 
 121                 column["label"] = format_movement_string(unit);
 122                 row_data.emplace("unit_moves", column);
 123 
 124                 std::stringstream hp_str;
 125                 hp_str << font::span_color(unit->hp_color()) << unit->hitpoints() << "/" << unit->max_hitpoints() << "</span>";
 126 
 127                 column["label"] = hp_str.str();
 128                 row_data.emplace("unit_hp", column);
 129 
 130                 column["label"] = format_level_string(unit->level());
 131                 row_data.emplace("unit_level", column);
 132 
 133                 std::stringstream exp_str;
 134                 exp_str << font::span_color(unit->xp_color()) << unit->experience() << "/"
 135                         << (unit->can_advance() ? std::to_string(unit->max_experience()) : font::unicode_en_dash) << "</span>";
 136 
 137                 column["label"] = exp_str.str();
 138                 row_data.emplace("unit_experience", column);
 139 
 140                 column["label"] = utils::join(unit->trait_names(), ", ");
 141                 row_data.emplace("unit_traits", column);
 142 
 143                 grid* row_grid = &list.add_row(row_data);
 144 
 145                 // NOTE: this needs to be done *after* the row is added
 146                 // TODO: show custom statuses
 147                 if(!unit->get_state(unit::STATE_PETRIFIED)) {
 148                         find_widget<image>(row_grid, "unit_status_petrified", false).set_visible(widget::visibility::invisible);
 149                 }
 150 
 151                 if(!unit->get_state(unit::STATE_POISONED)) {
 152                         find_widget<image>(row_grid, "unit_status_poisoned", false).set_visible(widget::visibility::invisible);
 153                 }
 154 
 155                 if(!unit->get_state(unit::STATE_SLOWED)) {
 156                         find_widget<image>(row_grid, "unit_status_slowed", false).set_visible(widget::visibility::invisible);
 157                 }
 158 
 159                 if(!unit->invisible(unit->get_location(), *resources::gameboard, false)) {
 160                         find_widget<image>(row_grid, "unit_status_invisible", false).set_visible(widget::visibility::invisible);
 161                 }
 162         }
 163 
 164         list.register_sorting_option(0, [this](const int i) { return unit_list_[i]->type_name().str(); });
 165         list.register_sorting_option(1, [this](const int i) { return unit_list_[i]->name().str(); });
 166         list.register_sorting_option(2, [this](const int i) { return unit_list_[i]->movement_left(); });
 167         list.register_sorting_option(3, [this](const int i) { return unit_list_[i]->hitpoints(); });
 168         list.register_sorting_option(4, [this](const int i) { return unit_list_[i]->level(); });
 169         list.register_sorting_option(5, [this](const int i) { return unit_list_[i]->experience(); });
 170         list.register_sorting_option(6, [this](const int i) {
 171                 return !unit_list_[i]->trait_names().empty() ? unit_list_[i]->trait_names().front().str() : ""; });
 172 
 173         list_item_clicked(window);
 174 }

 176 void unit_list::list_item_clicked(window& window)
 177 {
 178         const int selected_row
 179                 = find_widget<listbox>(&window, "units_list", false).get_selected_row();
 180 
 181         if(selected_row == -1) {
 182                 return;
 183         }
 184 
 185         find_widget<unit_preview_pane>(&window, "unit_details", false)
 186                 .set_displayed_unit(*unit_list_[selected_row].get());
 187 }

 189 void unit_list::post_show(window& window)
 190 {
 191         if(get_retval() == window::OK) {
 192                 const int selected_row = find_widget<listbox>(&window, "units_list", false).get_selected_row();
 193 
 194                 scroll_to_ = unit_list_[selected_row]->get_location();
 195         }

 198 void show_unit_list(display& gui)
 199 {
 200         unit_ptr_vector unit_list;
 201         map_location scroll_to;
 202 
 203         const unit_map& units = gui.get_units();
 204         for(unit_map::const_iterator i = units.begin(); i != units.end(); ++i) {
 205                 if(i->side() != gui.viewing_side()) {
 206                         continue;
 207                 }
 208 
 209                 unit_list.push_back(i.get_shared_ptr());
 210         }
 211 
 212         if(unit_list::execute(unit_list, scroll_to, gui.video())) {
 213                 gui.scroll_to_tile(scroll_to, display::WARP);
 214                 gui.select_hex(scroll_to);
 215         }
 216 }
