 246 class Translation:
 247     def __init__(self, localedir, langcode):
 248         self.catalog = {}
 249         self.localedir = localedir
 250         self.langcode = langcode
 251         class Dummy:
 252             def gettext(self, msgid):
 253                 if not msgid:
 254                     return ""
 255                 caret = msgid.find("^")
 256                 if caret < 0:
 257                     return msgid
 258                 return msgid[caret + 1:]
 259         self.dummy = Dummy()
 260 
 261     def translate(self, string, textdomain):
 262         if textdomain not in self.catalog:
 263             try:
 264                 self.catalog[textdomain] = gettext.translation(
 265                     textdomain, self.localedir, [self.langcode])
 266                 self.catalog[textdomain].add_fallback(self.dummy)
 267             except IOError:
 268                 self.catalog[textdomain] = self.dummy
 269             except AttributeError:
 270                 self.catalog[textdomain] = self.dummy
 271             except IndexError:
 272                 # not sure why, but this happens within the
 273                 # gettext.translation call sometimes
 274                 self.catalog[textdomain] = self.dummy
 275             except ValueError:
 276                 self.catalog[textdomain] = self.dummy
 277         return self.catalog[textdomain].gettext(string)

 361     def analyze_units(self, grouper, add_parents):
 362         """
 363         This takes all units belonging to a campaign, then groups them either
 364         by race or faction, and creates an advancements tree out of it.
 365         """
 366 
 367         # Build an advancement tree forest of all units.
 368         forest = self.forest = helpers.UnitForest()
 369         units_added = {}
 370         for uid, u in list(self.wesnoth.unit_lookup.items()):
 371             if u.hidden:
 372                 continue
 373             if grouper.unitfilter(u):
 374                 forest.add_node(helpers.UnitNode(u))
 375                 units_added[uid] = u
 376 
 377         #print("    %d/%d units" % (len(units_added), len(self.wesnoth.unit_lookup)))
 378 
 379         # Always add any child units, even if they have been filtered out..
 380         while units_added:
 381             new_units_added = {}
 382             for uid, u in list(units_added.items()):
 383                 for auid in u.advance:
 384                     if not auid in forest.lookup:
 385                         try:
 386                             au = self.wesnoth.unit_lookup[auid]
 387                         except KeyError:
 388                             error_message(
 389                                 "Warning: Unit %s not found as advancement of %s\n" %
 390                                 (auid, repr(uid)))
 391                             continue
 392                         forest.add_node(helpers.UnitNode(au))
 393                         new_units_added[auid] = au
 394             units_added = new_units_added
 395 
 396         if add_parents:
 397             # Also add parent units
 398             added = True
 399             while added:
 400                 added = False
 401                 for uid, u in list(self.wesnoth.unit_lookup.items()):
 402                     if uid in forest.lookup:
 403                         continue
 404                     for auid in u.advance:
 405                         if auid in forest.lookup:
 406                             forest.add_node(helpers.UnitNode(u))
 407                             added = True
 408                             break
 409 
 410         forest.update()
 411 
 412         # Partition trees by race/faction of first unit.
 413         groups = {}
 414         breadth = 0
 415 
 416         for tree in list(forest.trees.values()):
 417             u = tree.unit
 418             ugroups = grouper.groups(u)
 419 
 420             for group in ugroups:
 421                 groups[group] = groups.get(group, []) + [tree]
 422                 breadth += tree.breadth
 423 
 424         thelist = list(groups.keys())
 425         thelist.sort(key=lambda x: grouper.group_name(x))
 426 
 427         rows_count = breadth + len(thelist)
 428         # Create empty grid.
 429         rows = []
 430         for j in range(rows_count):
 431             column = []
 432             for i in range(6):
 433                 column.append((1, 1, None))
 434             rows.append(column)
 435 
 436         # Sort advancement trees by name of first unit and place into the grid.
 437         def by_name(t):
 438             x = T(t.unit, "name")
 439             return "" if x is None else x
 440 
 441         def grid_place(nodes, x):
 442             nodes.sort(key=by_name)
 443             for node in nodes:
 444                 level = max(0, min(5, node.unit.level))
 445                 rows[x][level] = (1, node.breadth, node)
 446                 for i in range(1, node.breadth):
 447                     rows[x + i][level] = (0, 0, node)
 448                 grid_place(node.children, x)
 449                 x += node.breadth
 450             return x
 451 
 452         x = 0
 453         for group in thelist:
 454             node = helpers.GroupNode(group)
 455             node.name = grouper.group_name(group)
 456 
 457             rows[x][0] = (6, 1, node)
 458             for i in range(1, 6):
 459                 rows[x][i] = (0, 0, None)
 460             nodes = groups[group]
 461             x += 1
 462             x = grid_place(nodes, x)
 463 
 464         self.unitgrid = rows
 465 
 466         return len(forest.lookup)

 737     def get_abilities(self, u):

 739         already = {}
 740         for abilities in u.get_all(tag="abilities"):
 741             try:

 744                 c = []
 745             for ability in c:
 746                 try:

 749                     error_message("Error: Ignoring ability " + ability.debug())
 750                     continue

 752                     continue

 760                 if name:
 761                     anames.append(name)

 764     def get_recursive_attacks(self, this_unit):
 765 
 766         def copy_attributes(copy_from, copy_to):
 767             for c in copy_from.data:
 768                 if isinstance(c, wmlparser3.AttributeNode):
 769                     copy_to.data.append(c)
 770 
 771         # Use attacks of base_units as base, if we have one.
 772         base_unit = self.wesnoth.get_base_unit(this_unit)
 773         attacks = []
 774         if base_unit:
 775             attacks = copy.deepcopy(self.get_recursive_attacks(base_unit))
 776 
 777         base_attacks_count = len(attacks)
 778         for i, attack in enumerate(this_unit.get_all(tag="attack")):
 779             # Attack merging is order based.
 780             if i < base_attacks_count:
 781                 copy_attributes(attack, attacks[i])
 782             else:
 783                 attacks.append(attack)
 784 
 785         return attacks

1423 def html_postprocess_file(filename, isocode, batchlist):
1424     print("postprocessing " + repr(filename))
1425     chtml = ""
1426     ehtml = ""
1427 
1428     cids = [[], []]
1429     for addon in batchlist:
1430         for campaign in addon.get("campaigns", []):
1431             if campaign["units"] == "?" or campaign["units"] <= 0:
1432                 continue
1433             lang = isocode if addon["name"] == "mainline" else "en_US"
1434             c = addon["name"], campaign["id"], campaign["translations"].get(
1435                 lang, campaign["name"]), lang
1436             if addon["name"] == "mainline":
1437                 cids[0].append(c)
1438             else:
1439                 cids[1].append(c)
1440     for i in range(2):
1441         campaigns = cids[i]
1442         campaigns.sort(key=lambda x: "A" if x[1] == "mainline" else "B" + x[2])
1443 
1444         for campaign in campaigns:
1445             addon, cname, campname, lang = campaign
1446             url = cleanurl("../../%s/%s/%s.html" % (addon, lang, cname))
1447             chtml += '<li><a title="%s" href="%s" role="menuitem">%s</a></li>\n' % (
1448                 cleantext(campname), url, cleantext(campname, quote=False))
1449         if i == 0 and cids[1]:
1450             chtml += '</ul><ul>'
1451 
1452     eids = [[], []]
1453     for addon in batchlist:
1454         for era in addon.get("eras", []):
1455             if era["units"] == "?" or era["units"] <= 0:
1456                 continue
1457             lang = isocode if addon["name"] == "mainline" else "en_US"
1458             e = addon["name"], era["id"], era["translations"].get(
1459                 lang, era["name"]), lang
1460             if addon["name"] == "mainline":
1461                 eids[0].append(e)
1462             else:
1463                 eids[1].append(e)
1464     for i in range(2):
1465         eras = eids[i]
1466         eras.sort(key=lambda x: x[2])
1467 
1468         for era in eras:
1469             addon, eid, eraname, lang = era
1470             url = cleanurl("../../%s/%s/%s.html" % (addon, lang, eid))
1471             ehtml += '<li><a title="%s" href="%s" role="menuitem">%s</a></li>\n' % (
1472                 cleantext(eraname), url, cleantext(eraname, quote=False))
1473         if i == 0 and eids[1]:
1474             ehtml += '</ul><ul>'
1475 
1476     f = open(filename, "r+b")
1477     html = f.read().decode("utf8")
1478     html = html.replace(PRE_PLACEHOLDER_CAMPAIGNS, chtml)
1479     html = html.replace(PRE_PLACEHOLDER_ERAS, ehtml)
1480     f.seek(0)
1481     f.write(html.encode("utf8"))
1482     f.close()
