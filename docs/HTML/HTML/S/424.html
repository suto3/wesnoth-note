  67 inline std::string concat_path(const std::string& a, const std::string& b)
  68 {
  69         //
  70         // As of Boost 1.61, normalize_path() displays unusual behavior when passing
  71         // it paths with extra path separators (e.g. //opt becomes
  72         // //opt/home/shadowm/src/wesnoth, where the extraneous sequence is probably
  73         // the current working dir), so avoid leaving those around.
  74         //
  75         // TODO: Maybe handle this corner case in filesystem::normalize_path()
  76         //       instead, really.
  77         //
  78         if((a.empty() || !fs::is_path_sep(a.back())) && (b.empty() || !fs::is_path_sep(b.front()))) {
  79                 return a + fs::path_separator() + b;
  80         } else {
  81                 return a + b;
  82         }
  83 }

  85 inline std::string filesystem_root()
  86 {
  87         // TODO: Multiple drives support (may require cooperation from the caller).
  88         return std::string(1, fs::path_separator());
  89 }

  91 inline void isort_dir_entries(std::vector<std::string>& entries)
  92 {
  93         // Yes, this uses Wesnoth's locale and not the filesystem/OS locale. Yes, this
  94         // isn't ideal. No, we don't really need to worry about it. It's just a
  95         // cosmetic procedure anyway.
  96         std::sort(entries.begin(), entries.end(),
  97                           [](const std::string& a, const std::string& b) { return translation::icompare(a, b) < 0; });
  98 }

 127 std::string file_dialog::path() const
 128 {
 129         const std::string& dir_norm = fs::normalize_path(current_dir_, true);
 130 
 131         if(current_entry_.empty() || current_entry_ == CURRENT_DIR) {
 132                 return dir_norm;
 133         } else if(current_entry_ == PARENT_DIR) {
 134                 return fs::directory_name(dir_norm);
 135         }
 136 
 137         return concat_path(dir_norm, current_entry_);
 138 }

 140 file_dialog& file_dialog::set_path(const std::string& value)
 141 {
 142         if(value.empty()) {
 143                 current_dir_ = filesystem_root();
 144         }
 145 
 146         const std::string& norm = fs::normalize_path(value, true);
 147 
 148         if(fs::is_directory(norm)) {
 149                 current_dir_ = norm;
 150         } else {
 151                 current_dir_ = fs::nearest_extant_parent(norm);
 152                 if(current_dir_.empty()) {
 153                         current_dir_ = filesystem_root();
 154                 }
 155                 // The file may or may not exist. We'll find out eventually when setting up
 156                 // the dialog.
 157                 current_entry_ = fs::base_name(norm);
 158         }
 159 
 160         return *this;
 161 }

 268 bool file_dialog::on_exit(window& window)
 269 {
 270         if(window.get_retval() == FILE_DIALOG_ITEM_RETVAL) {
 271                 // Attempting to exit by double clicking items -- only proceeds if the item
 272                 // was a file.
 273                 if(process_fileview_submit(window)) {
 274                         window.set_retval(window::OK, false);
 275                         return true;
 276                 } else {
 277                         return false;
 278                 }
 279         }
 280 
 281 
 282         if(window.get_retval() == window::OK) {
 283                 // Attempting to exit by pressing Enter/clicking OK -- only proceeds if the
 284                 // textbox was not altered by the user to point to a different directory.
 285                 return process_textbox_submit(window);
 286         }
 287 
 288         return true;
 289 }

 311 bool file_dialog::process_submit_common(window& window, const std::string& name)
 312 {
 313         const auto stype = register_new_selection(name);
 314 
 315         //DBG_FILEDLG << "current_dir_=" << current_dir_ << "  current_entry_=" << current_entry_ << '\n';
 316 
 317         if(is_selection_type_acceptable(stype)) {
 318                 return save_mode_ ? confirm_overwrite(window, stype) : true;
 319         }
 320 
 321         switch(stype) {
 322                 case SELECTION_IS_DIR:
 323                         // TODO: Adapt for implementing directory selection mode.
 324                         sync_bookmarks_bar(window);
 325                         refresh_fileview(window);
 326                         break;
 327                 case SELECTION_PARENT_NOT_FOUND:
 328                         // We get here in save mode or not. Use the file creation language only in
 329                         // save mode.
 330                         if(save_mode_) {
 331                                 show_transient_error_message(window.video(), vgettext("The file or folder $path cannot be created.", {{"path", name}}));
 332                                 break;
 333                         }
 334                         FALLTHROUGH;
 335                 case SELECTION_NOT_FOUND:
 336                         // We only get here if we aren't in save mode.
 337                         show_transient_error_message(window.video(), vgettext("The file or folder $path does not exist.", {{"path", name}}));
 338                         break;
 339                 case SELECTION_IS_FILE:
 340                         // TODO: Adapt for implementing directory selection mode.
 341                 default:
 342                         assert(false && "Unimplemented selection mode or semantics");
 343         }
 344 
 345         return false;
 346 }

 362 std::string file_dialog::get_filelist_selection(listbox& filelist)
 363 {
 364         const int row = filelist.get_selected_row();
 365 
 366         if(row == -1) {
 367                 // Shouldn't happen...
 368                 return "";
 369         }
 370 
 371         const bool i_am_root = fs::is_root(current_dir_);
 372 
 373         if(row == 0 && !i_am_root) {
 374                 return PARENT_DIR;
 375         } else {
 376                 size_t n = i_am_root ? row : row - 1;
 377 
 378                 if(n < dir_subdirs_.size()) {
 379                         return dir_subdirs_[n];
 380                 } else {
 381                         n -= dir_subdirs_.size();
 382 
 383                         if(n < dir_files_.size()) {
 384                                 return dir_files_[n];
 385                         } else {
 386                                 assert(false && "File list selection is out of range!");
 387                         }
 388                 }
 389         }
 390 
 391         return "";
 392 }

 394 file_dialog::SELECTION_TYPE file_dialog::register_new_selection(const std::string& name)
 395 {
 396         std::string new_path, new_parent;
 397 
 398         if(fs::is_relative(name)) {
 399                 // On Windows, \ represents a path relative to the root of the process'
 400                 // current working drive specified by the current working dir, so we get
 401                 // here. This makes it the only platform where is_relative() and is_root()
 402                 // aren't mutually exclusive.
 403                 if(fs::is_root(name)) {
 404                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is relative to a root resource\n";
 405                         // Using the browsed dir's root drive instead of the cwd's makes the most
 406                         // sense for users.
 407                         new_parent = fs::root_name(current_dir_);
 408                         new_path = fs::normalize_path(concat_path(new_parent, name), true, true);
 409                 } else {
 410                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' seems relative\n";
 411                         new_parent = current_dir_;
 412                         new_path = fs::normalize_path(concat_path(current_dir_, name), true, true);
 413                 }
 414         } else {
 415                 DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' seems absolute\n";
 416                 new_parent = fs::directory_name(name);
 417                 new_path = fs::normalize_path(name, true, true);
 418                 DBG_FILEDLG << "register_new_selection(): new selection is " << new_path << '\n';
 419         }
 420 
 421         if(!new_path.empty()) {
 422                 if(fs::is_directory(new_path)) {
 423                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is a directory: " << new_path << '\n';
 424                         current_dir_ = new_path;
 425                         current_entry_.clear();
 426                         return SELECTION_IS_DIR;
 427                 } else if(fs::file_exists(new_path)) {
 428                         // FIXME: Perhaps redundant since the three-params call to normalize_path()
 429                         //        above necessarily validates existence.
 430                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is a file, symbolic link, or special: " << new_path << '\n';
 431                         current_dir_ = fs::directory_name(new_path);
 432                         current_entry_ = fs::base_name(new_path);
 433                         return SELECTION_IS_FILE;
 434                 }
 435         }
 436 
 437         // The path does not exist, at least not entirely. See if the parent does
 438         // (in save mode non-existent files are accepted as long as the parent dir
 439         // exists).
 440         const std::string& absolute_parent = fs::normalize_path(new_parent, true, true);
 441         if(!absolute_parent.empty()) {
 442                 DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' does not exist or is not accessible, but parent exists\n";
 443                 current_dir_ = absolute_parent;
 444                 current_entry_ = fs::base_name(name);
 445                 return SELECTION_NOT_FOUND;
 446         }
 447 
 448         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' does not exist or is not accessible\n";
 449         return SELECTION_PARENT_NOT_FOUND;
 450 }

 452 void file_dialog::set_input_text(text_box& t, const std::string& value)
 453 {
 454         if(value.empty()) {
 455                 clear_input_text(t);
 456                 return;
 457         }
 458 
 459         t.set_value(value);
 460 
 461         const size_t vallen = t.get_length();
 462         const size_t extlen = utf8::size(extension_);
 463 
 464         if(save_mode_ && extlen && vallen > extlen) {
 465                 // Highlight everything but the extension if it matches
 466                 if(value.substr(vallen - extlen) == extension_) {
 467                         t.set_selection(0, vallen - extlen);
 468                 }
 469         }
 470 }

 482 void file_dialog::refresh_fileview(window& window)
 483 {
 484         cursor::setter cur{cursor::WAIT};
 485 
 486         dir_files_.clear();
 487         dir_subdirs_.clear();
 488 
 489         // TODO: Need to detect and handle cases where we don't have search permission
 490         //       on current_dir_, otherwise things may get weird.
 491         filesystem::get_files_in_dir(current_dir_, &dir_files_, &dir_subdirs_, filesystem::FILE_NAME_ONLY);
 492         isort_dir_entries(dir_files_);
 493         isort_dir_entries(dir_subdirs_);
 494 
 495         //
 496         // Clear and refill the filelist box.
 497         //
 498 
 499         listbox& filelist = find_widget<listbox>(&window, "filelist", false);
 500         button& rm_button = find_widget<button>(&window, "delete_file", false);
 501 
 502         filelist.clear();
 503 
 504         // Parent entry
 505         if(!fs::is_root(current_dir_)) {
 506                 // label_parent may not necessarily be always ".." in the future, so push
 507                 // with check_selection = false and check the selection ourselves here.
 508                 push_fileview_row(filelist, label_parent, icon_parent, false);
 509                 if(current_entry_ == PARENT_DIR || current_entry_.empty()) {
 510                         filelist.select_row(0, true);
 511                         rm_button.set_active(false);
 512                 } else {
 513                         rm_button.set_active(true);
 514                 }
 515         }
 516 
 517         for(const auto& dir : dir_subdirs_) {
 518                 push_fileview_row(filelist, dir, icon_dir);
 519         }
 520 
 521         for(const auto& file : dir_files_) {
 522                 push_fileview_row(filelist, file, icon_file);
 523         }
 524 
 525         find_widget<styled_widget>(&window, "current_dir", false).set_label(current_dir_);
 526         set_input_text(find_widget<text_box>(&window, "filename", false), current_entry_);
 527 
 528         on_row_selected(window);
 529 }

 531 void file_dialog::push_fileview_row(listbox& filelist, const std::string& name, const std::string& icon, bool check_selection)
 532 {
 533         // TODO: Hopefully some day GUI2 will allow us to make labels be ellipsized
 534         //       dynamically at layout/rendering time.
 535         std::string label = name;
 536         utils::ellipsis_truncate(label, FILE_DIALOG_MAX_ENTRY_LENGTH);
 537 
 538         std::map<std::string, string_map> data;
 539         data["icon"]["label"] = icon;
 540         data["file"]["label"] = label;
 541 
 542         grid& last_grid = filelist.add_row(data);
 543         const unsigned last_pos = filelist.get_item_count() - 1;
 544 
 545         //
 546         // Crummy hack around the lack of an option to hook into row double click
 547         // events for all rows using the GUI2 listbox API. Assign a special retval to
 548         // each row that triggers a special check during dialog exit.
 549         //
 550         find_widget<toggle_panel>(&last_grid, "item_panel", false)
 551                         .set_retval(FILE_DIALOG_ITEM_RETVAL);
 552 
 553         if(check_selection && name == current_entry_) {
 554                 filelist.select_row(last_pos, true);
 555         }
 556 }

 558 void file_dialog::sync_bookmarks_bar(window& window)
 559 {
 560         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 561 
 562         // Internal state has normalized path delimiters but dot entries aren't
 563         // resolved after callers call set_path(), so compare against a canonical
 564         // version. The bookmark paths are already canonical, though.
 565         const std::string& canon_current_dir = fs::normalize_path(current_dir_, true, true);
 566 
 567         // Go backwards so we can match user-defined bookmarks first (otherwise it may
 568         // become impossible for the user to delete them if they match any of the
 569         // predefined paths).
 570         auto it = std::find(bookmark_paths_.rbegin(), bookmark_paths_.rend(), canon_current_dir);
 571 
 572         if(it == bookmark_paths_.rend()) {
 573                 if(current_bookmark_ >= 0) {
 574                         bookmarks_bar.select_row(unsigned(current_bookmark_), false);
 575                 }
 576                 current_bookmark_ = -1;
 577         } else {
 578                 const int new_selection = int(std::distance(bookmark_paths_.begin(), it.base()) - 1);
 579                 if(new_selection != current_bookmark_) {
 580                         assert(unsigned(new_selection) < bookmarks_bar.get_item_count());
 581                         if(current_bookmark_ >= 0) {
 582                                 bookmarks_bar.select_row(unsigned(current_bookmark_), false);
 583                         }
 584                         bookmarks_bar.select_row(unsigned(new_selection), true);
 585                         current_bookmark_ = new_selection;
 586                 }
 587         }
 588 
 589         // Update bookmark edit controls.
 590         button& del_button = find_widget<button>(&window, "remove_bookmark", false);
 591 
 592         if(user_bookmarks_begin_ == -1) {
 593                 del_button.set_active(false);
 594         } else {
 595                 del_button.set_active(current_bookmark_ >= user_bookmarks_begin_);
 596         }
 597 }

 599 void file_dialog::on_row_selected(window& window)
 600 {
 601         listbox& filelist = find_widget<listbox>(&window, "filelist", false);
 602         text_box& file_textbox = find_widget<text_box>(&window, "filename", false);
 603         button& rm_button = find_widget<button>(&window, "delete_file", false);
 604 
 605         // Don't use register_new_selection() here, we don't want any parsing to be
 606         // performed at this point.
 607         current_entry_ = get_filelist_selection(filelist);
 608 
 609         // Clear the textbox when selecting ..
 610         if(current_entry_ != PARENT_DIR) {
 611                 set_input_text(file_textbox, current_entry_);
 612                 rm_button.set_active(true);
 613         } else {
 614                 clear_input_text(file_textbox);
 615                 rm_button.set_active(false);
 616         }
 617 
 618         // Need to do this every time so that input can still be sent to the
 619         // textbox without clicking on it.
 620         window.keyboard_capture(&file_textbox);
 621 }

 623 void file_dialog::on_bookmark_selected(window& window)
 624 {
 625         // Don't let us steal the focus from the primary widgets.
 626         text_box& file_textbox = find_widget<text_box>(&window, "filename", false);
 627         window.keyboard_capture(&file_textbox);
 628 
 629         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 630         const int new_selection = bookmarks_bar.get_selected_row();
 631 
 632         if(new_selection < 0) {
 633                 if(current_bookmark_ >= 0) {
 634                         // Don't allow the user to unselect the selected bookmark. That wouldn't
 635                         // make any sense.
 636                         bookmarks_bar.select_row(unsigned(current_bookmark_));
 637                 }
 638 
 639                 return;
 640         }
 641 
 642         assert(unsigned(new_selection) < bookmark_paths_.size());
 643         current_bookmark_ = new_selection;
 644         set_path(bookmark_paths_[new_selection]);
 645         refresh_fileview(window);
 646 
 647         // Update bookmark edit controls.
 648         button& del_button = find_widget<button>(&window, "remove_bookmark", false);
 649         del_button.set_active(user_bookmarks_begin_ >= 0
 650                                                   && current_bookmark_ >= user_bookmarks_begin_);
 651 }

 653 void file_dialog::on_bookmark_add_cmd(window& window)
 654 {
 655         const std::string& default_label = fs::base_name(current_dir_);
 656 
 657         std::string label = default_label;
 658 
 659         const bool confirm = bookmark_create::execute(label, window.video());
 660         if(!confirm) {
 661                 return;
 662         }
 663 
 664         if(label.empty()) {
 665                 label = default_label;
 666         }
 667 
 668         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 669 
 670         desktop::add_user_bookmark(label, current_dir_);
 671         bookmark_paths_.push_back(current_dir_);
 672         const unsigned top_bookmark = bookmark_paths_.size() - 1;
 673 
 674         if(user_bookmarks_begin_ == -1) {
 675                 user_bookmarks_begin_ = top_bookmark;
 676         }
 677 
 678         std::map<std::string, string_map> data;
 679         data["bookmark"]["label"] = label;
 680         bookmarks_bar.add_row(data);
 681 
 682         current_bookmark_ = -1;
 683 
 684         sync_bookmarks_bar(window);
 685 }

 687 void file_dialog::on_bookmark_del_cmd(window& window)
 688 {
 689         assert(user_bookmarks_begin_ >= 0
 690                    && current_bookmark_ >= 0
 691                    && current_bookmark_ >= user_bookmarks_begin_
 692                    && current_bookmark_ < int(bookmark_paths_.size()));
 693 
 694         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 695         desktop::remove_user_bookmark(current_bookmark_ - user_bookmarks_begin_);
 696         bookmark_paths_.erase(bookmark_paths_.begin() + current_bookmark_);
 697         bookmarks_bar.remove_row(current_bookmark_);
 698 
 699         current_bookmark_ = -1;
 700 
 701         sync_bookmarks_bar(window);
 702 }

 704 void file_dialog::on_dir_create_cmd(window& window)
 705 {
 706         std::string new_dir_name;
 707 
 708         if(folder_create::execute(new_dir_name, window.video())) {
 709                 const std::string& new_path = concat_path(current_dir_, new_dir_name);
 710 
 711                 if(!fs::make_directory(new_path)) {
 712                         show_transient_error_message(window.video(),
 713                                         vgettext("Could not create a new folder at $path|. Make sure you have the appropriate permissions to write to this location.",
 714                                         {{"path", new_path}}));
 715                 } else {
 716                         refresh_fileview(window);
 717                 }
 718         }

 721 void file_dialog::on_file_delete_cmd(window& window)
 722 {
 723         if(current_entry_.empty()) {
 724                 return;
 725         }
 726 
 727         const std::string& selection = concat_path(current_dir_, current_entry_);
 728         const bool is_dir = fs::is_directory(selection);
 729 
 730         const std::string& message = (is_dir
 731                         ? _("The following folder and its contents will be permanently deleted:")
 732                         : _("The following file will be permanently deleted:"))
 733                         + "\n\n" + selection + "\n\n" + _("Do you wish to continue?");
 734 
 735         if(gui2::show_message(window.video(), _("Confirm"), message, message::yes_no_buttons) == gui2::window::CANCEL) {
 736                 return;
 737         }
 738 
 739         const bool result = is_dir
 740                         ? fs::delete_directory(selection)
 741                         : fs::delete_file(selection);
 742 
 743         if(!result) {
 744                 show_transient_error_message(window.video(),
 745                                 vgettext("Could not delete $path|. Make sure you have the appropriate permissions to write to this location.",
 746                                                  {{"path", selection}}));
 747         } else {
 748                 refresh_fileview(window);
 749         }
 750 }
