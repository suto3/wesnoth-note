  32 function wml_actions.sync_variable(cfg)
  33         local names = cfg.name or helper.wml_error "[sync_variable] missing required name= attribute."
  34         local result = wesnoth.synchronize_choice(
  35                 function()
  36                         local res = {}
  37                         for name_raw in utils.split(names) do
  38                                 local name = utils.trim(name_raw)
  39                                 local variable_type = string.sub(name, string.len(name)) == "]" and "indexed" or ( wesnoth.get_variable(name .. ".length") > 0 and "array" or "attribute")
  40                                 local variable_info = { name = name, type = variable_type }
  41                                 table.insert(res, { "variable", variable_info })
  42                                 if variable_type == "indexed" then
  43                                         table.insert(variable_info, { "value", wesnoth.get_variable(name) } )
  44                                 elseif variable_type == "array" then
  45                                         for i = 1, wesnoth.get_variable(name .. ".length") do
  46                                                 table.insert(variable_info, { "value",  wesnoth.get_variable(string.format("%s[%d]", name, i - 1)) } )
  47                                         end
  48                                 else
  49                                         variable_info.value = wesnoth.get_variable(name)
  50                                 end
  51                         end
  52                         return res
  53                 end
  54         )
  55         for variable in helper.child_range(result, "variable") do
  56                 local name = variable.name
  57 
  58                 if variable.type == "indexed" then
  59                         wesnoth.set_variable(name, variable[1][2])
  60                 elseif variable.type == "array" then
  61                         for index, cfg_pair in ipairs(variable) do
  62                                 wesnoth.set_variable( string.format("%s[%d]", name, index - 1), cfg_pair[2])
  63                         end
  64                 else
  65                         wesnoth.set_variable(name, variable.value)
  66                 end
  67         end
  68 end

  70 function wml_actions.chat(cfg)
  71         local side_list = wesnoth.get_sides(cfg)
  72         local speaker = tostring(cfg.speaker or "WML")
  73         local message = tostring(cfg.message or
  74                 helper.wml_error "[chat] missing required message= attribute."
  75         )
  76 
  77         for index, side in ipairs(side_list) do
  78                 if side.controller == "human" and side.is_local then
  79                         wesnoth.message(speaker, message)
  80                         break
  81                 end
  82         end
  83 
  84         local observable = cfg.observable ~= false
  85 
  86         if observable then
  87                 local all_sides = wesnoth.get_sides()
  88                 local has_human_side = false
  89                 for index, side in ipairs(all_sides) do
  90                         if side.controller == "human" and side.is_local then
  91                                 has_human_side = true
  92                                 break
  93                         end
  94                 end
  95 
  96                 if not has_human_side then
  97                         wesnoth.message(speaker, message)
  98                 end
  99         end

 163         elseif cfg.name and cfg.name ~= "" then wesnoth.fire_event(cfg.name, x1, y1, x2, y2, w1, w2)

 259 function wml_actions.unit_worth(cfg)
 260         local u = wesnoth.get_units(cfg)[1] or
 261                 helper.wml_error "[unit_worth]'s filter didn't match any unit"
 262         local ut = wesnoth.unit_types[u.type]
 263         local hp = u.hitpoints / u.max_hitpoints
 264         local xp = u.experience / u.max_experience
 265         local best_adv = ut.cost
 266         for w in utils.split(ut.__cfg.advances_to) do
 267                 local uta = wesnoth.unit_types[w]
 268                 if uta and uta.cost > best_adv then best_adv = uta.cost end
 269         end
 270         wesnoth.set_variable("cost", ut.cost)
 271         wesnoth.set_variable("next_cost", best_adv)
 272         wesnoth.set_variable("health", math.floor(hp * 100))
 273         wesnoth.set_variable("experience", math.floor(xp * 100))
 274         wesnoth.set_variable("recall_cost", ut.recall_cost)
 275         wesnoth.set_variable("unit_worth", math.floor(math.max(ut.cost * hp, best_adv * xp)))
 276 end

 367 function wml_actions.remove_unit_overlay(cfg)
 368         local img = cfg.image or helper.wml_error( "[remove_unit_overlay] missing required image= attribute" )
 369 
 370         -- Loop through all matching units.
 371         for i,u in ipairs(wesnoth.get_units(cfg)) do
 372                 local ucfg = u.__cfg
 373                 local t = utils.parenthetical_split(ucfg.overlays)
 374                 -- Remove the specified image from the overlays.
 375                 for i = #t,1,-1 do
 376                         if t[i] == img then table.remove(t, i) end
 377                 end
 378                 -- Reassemble the list of remaining overlays.
 379                 ucfg.overlays = table.concat(t, ',')
 380                 wesnoth.put_unit(ucfg)
 381         end
 382 end

 389 function wml_actions.store_unit(cfg)
 390         local filter = helper.get_child(cfg, "filter") or
 391                 helper.wml_error "[store_unit] missing required [filter] tag"
 392         local kill_units = cfg.kill
 393 
 394         --cache the needed units here, since the filter might reference the to-be-cleared variable(s)
 395         local units = wesnoth.get_units(filter)
 396         local recall_units = wesnoth.get_recall_units(filter)
 397 
 398         local writer = utils.vwriter.init(cfg, "unit")
 399 
 400         for i,u in ipairs(units) do
 401                 utils.vwriter.write(writer, u.__cfg)
 402                 if kill_units then wesnoth.erase_unit(u) end
 403         end
 404 
 405         if (not filter.x or filter.x == "recall") and (not filter.y or filter.y == "recall") then
 406                 for i,u in ipairs(recall_units) do
 407                         local ucfg = u.__cfg
 408                         ucfg.x = "recall"
 409                         ucfg.y = "recall"
 410                         utils.vwriter.write(writer, ucfg)
 411                         if kill_units then wesnoth.erase_unit(u) end
 412                 end
 413         end
 414 end

 437 function wml_actions.store_reachable_locations(cfg)
 438         local unit_filter = helper.get_child(cfg, "filter") or
 439                 helper.wml_error "[store_reachable_locations] missing required [filter] tag"
 440         local location_filter = helper.get_child(cfg, "filter_location")
 441         local range = cfg.range or "movement"
 442         local moves = cfg.moves or "current"
 443         local variable = cfg.variable or helper.wml_error "[store_reachable_locations] missing required variable= key"
 444         local reach_param = { viewing_side = cfg.viewing_side or 0 }
 445         if range == "vision" then
 446                 moves = "max"
 447                 reach_param.ignore_units = true
 448         end
 449 
 450         local reach = location_set.create()
 451 
 452         for i,unit in ipairs(wesnoth.get_units(unit_filter)) do
 453                 local unit_reach
 454                 if moves == "max" then
 455                         local saved_moves = unit.moves
 456                         unit.moves = unit.max_moves
 457                         unit_reach = location_set.of_pairs(wesnoth.find_reach(unit, reach_param))
 458                         unit.moves = saved_moves
 459                 else
 460                         unit_reach = location_set.of_pairs(wesnoth.find_reach(unit, reach_param))
 461                 end
 462 
 463                 if range == "vision" or range == "attack" then
 464                         unit_reach:iter(function(x, y)
 465                                 reach:insert(x, y)
 466                                 for u,v in helper.adjacent_tiles(x, y) do
 467                                         reach:insert(u, v)
 468                                 end
 469                         end)
 470                 else
 471                         reach:union(unit_reach)
 472                 end
 473         end
 474 
 475         if location_filter then
 476                 reach = reach:filter(function(x, y)
 477                         return wesnoth.match_location(x, y, location_filter)
 478                 end)
 479         end
 480         reach:to_wml_var(variable)
 481 end

 565 function wml_actions.transform_unit(cfg)
 566         local transform_to = cfg.transform_to
 567 
 568         for index, unit in ipairs(wesnoth.get_units(cfg)) do
 569 
 570                 if transform_to then
 571                         wesnoth.transform_unit( unit, transform_to )
 572                 else
 573                         local hitpoints = unit.hitpoints
 574                         local experience = unit.experience
 575                         local recall_cost = unit.recall_cost
 576                         local status = helper.get_child( unit.__cfg, "status" )
 577 
 578                         unit.experience = unit.max_experience
 579                         wesnoth.advance_unit(unit, false, false)
 580 
 581                         unit.hitpoints = hitpoints
 582                         unit.experience = experience
 583                         recall_cost = unit.recall_cost
 584 
 585                         for key, value in pairs(status) do unit.status[key] = value end
 586                         if unit.status.unpoisonable then unit.status.poisoned = nil end
 587                 end
 588         end
 589 
 590         wml_actions.redraw {}
 591 end

 620         local exec = cfg.execution

 641                         engine = "lua",

 824 local function on_board(x, y)
 825         if type(x) ~= "number" or type(y) ~= "number" then
 826                 return false
 827         end
 828         local w, h = wesnoth.get_map_size()
 829         return x >= 1 and y >= 1 and x <= w and y <= h
 830 end

 832 wml_actions.unstore_unit = function(cfg)
 833         local variable = cfg.variable or helper.wml_error("[unstore_unit] missing required 'variable' attribute")
 834         local unit_cfg = wesnoth.get_variable(variable) or helper.wml_error("[unstore_unit]: variable '" .. variable .. "' doesn't exist")
 835         if type(unit_cfg) ~= "table" or unit_cfg.type == nil then
 836                 helper.wml_error("[unstore_unit]: variable '" .. variable .. "' doesn't contain unit data")
 837         end
 838         local unit = wesnoth.create_unit(unit_cfg)
 839         local advance = cfg.advance ~= false
 840         local animate = cfg.animate ~= false
 841         local check_passability = cfg.check_passability ~= false or nil
 842         local x = cfg.x or unit.x or -1
 843         local y = cfg.y or unit.y or -1
 844         wesnoth.add_known_unit(unit.type)
 845         if on_board(x, y) then
 846                 if cfg.find_vacant then
 847                         x,y = wesnoth.find_vacant_tile(x, y, check_passability and unit)
 848                 end
 849                 unit:to_map(x, y, cfg.fire_event)
 850                 local text = nil
 851                 if unit_cfg.gender == "female" then
 852                         text = cfg.female_text or cfg.text
 853                 else
 854                         text = cfg.male_text or cfg.text
 855                 end
 856                 local color = cfg.color
 857                 if color == nil and cfg.red and cfg.blue and cfg.green then
 858                         color = cfg.red .. "," .. cfg.green .. "," .. cfg.blue
 859                 end
 860                 if text ~= nil and not wesnoth.is_skipping_messages() then
 861                         wesnoth.float_label(x, y, text, color)
 862                 end
 863                 if advance then
 864                         wesnoth.advance_unit(unit, animate, cfg.fire_event)
 865                 end
 866         else
 867                 unit:to_recall()
 868         end
 869 end
