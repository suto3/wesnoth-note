 212     def _rmtree(self, directory, exclude):
 213         logging.debug("Deleting tree {0}, except for {1}".format(self.name, ",".join(exclude)))
 214 
 215         # Ensure the os calls all happen in the right directory
 216         # not needed for _execute, as that does the cwd manipulation itself
 217         # so only the os.chdir and os.path.isdir here need it
 218         # Another option would be to os.path.join with self.get_dir
 219         os.chdir(self.get_dir())
 220         for entry in os.listdir(directory):
 221             if entry in exclude:
 222                 continue
 223             if entry == ".git":
 224                 continue
 225 
 226             relpath = os.path.join(directory, entry)
 227 
 228             if os.path.isdir(relpath):
 229                 self._rmtree(relpath, exclude)
 230                 # git rm removes directories that it empties
 231                 if os.path.exists(relpath):
 232                     self._execute(["rmdir", "--ignore-fail-on-non-empty", relpath])
 233             else:
 234                 self._execute(["git", "rm", relpath], check_error=True)

 236     def _copytree(self, src, dst, ignore=None):
 237         """Recursively copy a directory tree using copy2().
 238 
 239         Based on shutil.copytree
 240         """
 241         names = os.listdir(src)
 242         if ignore is not None:
 243             ignored_names = ignore(src, names)
 244         else:
 245             ignored_names = set()
 246 
 247         if not os.path.exists(dst):
 248             os.makedirs(dst)
 249         errors = []
 250         for name in names:
 251             if name in ignored_names:
 252                 continue
 253             srcname = os.path.join(src, name)
 254             dstname = os.path.join(dst, name)
 255             try:
 256                 if os.path.isdir(srcname):
 257                     self._copytree(srcname, dstname, ignore)
 258                 else:
 259                     shutil.copy2(srcname, dstname)
 260                 # XXX What about devices, sockets etc.?
 261             except (IOError, os.error) as why:
 262                 errors.append((srcname, dstname, str(why)))
 263             # catch the Error from the recursive copytree so that we can
 264             # continue with other files
 265             except Error as err:
 266                 errors.extend(err.args[0])
 267         try:
 268             shutil.copystat(src, dst)
 269         except OSError as why:
 270             if shutil.WindowsError is not None and isinstance(why, shutil.WindowsError):
 271                 # Copying file access times may fail on Windows
 272                 pass
 273             else:
 274                 errors.extend((src, dst, str(why)))
 275         if errors:
 276             raise AddonError(self.name, "Errors attempting to sync:\n{0}".format("\n".join(errors)))

 325

 468     def _github_api_request(self, url, data=None, method=None, authenticate=False):
 469         logging.debug("Making github API request {0}".format(url))
 470 
 471         request = urllib.request.Request(url)
 472         if method:
 473             request.get_method = lambda: method
 474 
 475         if data == "":
 476             # Workaround for PUTs requiring data, even if you have nothing to pass
 477             request.add_data(data)
 478         elif data:
 479             request.add_data(json.dumps(data))
 480 
 481         # Manually adding authentication data
 482         # Basic works in curl, but urllib2 doesn't
 483         # probably because github's API doesn't send a www-authenticate header
 484         if authenticate or self._github_have_authorization():
 485             from base64 import encodestring
 486             auth = self._github_authorization()
 487             if ":" in auth:
 488                 # username:password
 489                 base64string = encodestring(auth).replace('\n', '')
 490                 request.add_header("Authorization", "Basic {0}".format(base64string))
 491             else:
 492                 # token
 493                 request.add_header("Authorization", "Bearer {0}".format(auth))
 494 
 495         try:
 496             response = urllib.request.urlopen(request)
 497         except IOError as e:
 498             raise Error("GitHub API failure: " + str(e))
 499         if response.code == 204:
 500             # 204 = No content
 501             return None
 502 
 503         json_parsed = json.load(response)
 504 
 505         link_headers = response.info().getallmatchingheaders("Link")
 506         if link_headers:
 507             logging.debug("Found a Link header in response, analyzing...")
 508             link_header = link_headers[0].lstrip("Link:")
 509             links_raw = link_header.split(",")
 510             links_split_raw = [link.split(";") for link in links_raw]
 511             links_split_proc = [(l[1].strip().lstrip('rel="').rstrip('"'), l[0].strip().lstrip("<").rstrip(">")) for l in links_split_raw]
 512             links_dict = dict((k, v) for (k, v) in links_split_proc)
 513             if "next" in links_dict:
 514                 logging.debug("Link with rel=\"next\" found, recursing to deal with pagination")
 515                 rest = self._github_api_request(links_dict["next"], data, method, authenticate)
 516                 json_parsed += rest
 517 
 518         return json_parsed

 528     def _execute(self, command, cwd=None, check_error=False):
 529         #TODO: have an errorcheck that actually checks the returncode?
 530         """Executes a command.
 531 
 532         command: The command to execute.
 533         cwd: Directory to execute the command from.
 534         check_error: Whether to raise an exception if there's stderr output.
 535         Returns stdout, stderr.
 536         Raises libgithub.Error if check_error and len(err).
 537         """
 538         logging.debug("execute command = '%s'", command)
 539 
 540         p = subprocess.Popen(command, stdout=subprocess.PIPE, stderr=subprocess.PIPE, close_fds=True, cwd=cwd)
 541         out = ""
 542         err = ""
 543         while(p.poll() == None):
 544             out += p.stdout.read()
 545             err += p.stderr.read()
 546 
 547         out += p.stdout.read()
 548         err += p.stderr.read()
 549 
 550         logging.debug("===== stdout ====\n%s\n===== stdout ====", out)
 551         logging.debug("===== stderr ====\n%s\n===== stderr ====", err)
 552 
 553         if check_error and len(err):
 554             raise Error("Failure executing command '{0}': {1}".format(" ".join(command), err))
 555 
 556         return _execresult(out, err, p.returncode)
