 184 void styled_widget::request_reduce_width(const unsigned maximum_width)
 185 {
 186         assert(config_);
 187 
 188         if(!label_.empty() && can_wrap()) {
 189 
 190                 point size = get_best_text_size(
 191                                 point(), point(maximum_width - config_->text_extra_width, 0));
 192 
 193                 size.x += config_->text_extra_width;
 194                 size.y += config_->text_extra_height;
 195 
 196                 set_layout_size(size);
 197 
 198                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 199                                   << "' maximum_width " << maximum_width << " result " << size
 200                                   << ".\n";
 201 
 202         } else if(label_.empty() || text_can_shrink()) {
 203                 point size = get_best_size();
 204                 point min_size = get_config_minimum_size();
 205                 size.x = std::min(size.x, std::max<int>(maximum_width, min_size.x));
 206                 set_layout_size(size);
 207 
 208                 DBG_GUI_L << LOG_HEADER << " styled_widget " << id()
 209                           << " maximum_width " << maximum_width << " result " << size
 210                           << ".\n";
 211         } else {
 212                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 213                                   << "' failed; either no label or wrapping not allowed.\n";
 214         }
 215 }

 233 point styled_widget::calculate_best_size() const
 234 {
 235         assert(config_);
 236         if(label_.empty()) {
 237                 DBG_GUI_L << LOG_HEADER << " empty label return default.\n";
 238                 return get_config_default_size();
 239         }
 240 
 241         const point minimum = get_config_default_size();
 242         const point maximum = get_config_maximum_size();
 243 
 244         /**
 245          * @todo The value send should subtract the border size
 246          * and read it after calculation to get the proper result.
 247          */
 248         point result = get_best_text_size(minimum, maximum);
 249         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 250                           << "' result " << result << ".\n";
 251         return result;

 254 void styled_widget::place(const point& origin, const point& size)
 255 {
 256         // resize canvasses
 257         for(auto & canvas : canvas_)
 258         {
 259                 canvas.set_width(size.x);
 260                 canvas.set_height(size.y);
 261         }
 262 
 263         // Note we assume that the best size has been queried but otherwise it
 264         // should return false.
 265         if(renderer_.is_truncated() && use_tooltip_on_label_overflow_
 266            && tooltip_.empty()) {
 267 
 268                 set_tooltip(label_);
 269         }
 270 
 271         // inherited
 272         widget::place(origin, size);
 273 
 274         // update the state of the canvas after the sizes have been set.
 275         update_canvas();
 276 }

 357 void styled_widget::update_canvas()
 358 {
 359         const int max_width = get_text_maximum_width();
 360         const int max_height = get_text_maximum_height();
 361 
 362         // set label in canvases
 363         for(auto & canvas : canvas_)
 364         {
 365                 canvas.set_variable("text", wfl::variant(label_));
 366                 canvas.set_variable("text_markup", wfl::variant(use_markup_));
 367                 canvas.set_variable("text_link_aware", wfl::variant(get_link_aware()));
 368 
 369                 // Possible TODO: consider making a formula_callable for colors.
 370                 color_t link_color = get_link_color();
 371                 std::vector<wfl::variant> link_color_as_list {
 372                         wfl::variant(link_color.r),
 373                         wfl::variant(link_color.g),
 374                         wfl::variant(link_color.b),
 375                         wfl::variant(link_color.a)
 376                 };
 377 
 378                 canvas.set_variable("text_link_color", wfl::variant(link_color_as_list));
 379                 canvas.set_variable("text_alignment",
 380                                                         wfl::variant(encode_text_alignment(text_alignment_)));
 381                 canvas.set_variable("text_maximum_width", wfl::variant(max_width));
 382                 canvas.set_variable("text_maximum_height", wfl::variant(max_height));
 383                 canvas.set_variable("text_wrap_mode", wfl::variant(get_text_ellipse_mode()));
 384                 canvas.set_variable("text_characters_per_line",
 385                                                         wfl::variant(get_characters_per_line()));
 386         }
 387 }

 424 void styled_widget::definition_load_configuration(const std::string& control_type)
 425 {
 426         assert(!config_);
 427 
 428         set_config(get_control(control_type, definition_));
 429 
 430         /**
 431          * Fill in each canvas from the widget state definitons.
 432          *
 433          * Most widgets have a single canvas. However, some widgets such as toggle_panel
 434          * and toggle_button have a variable canvas count determined by their definitions.
 435          */
 436         for(const auto& widget_state : config_->state) {
 437                 canvas_.emplace_back(std::move(widget_state.canvas_));
 438         }
 439 
 440         update_canvas();
 441 }

 443 point styled_widget::get_best_text_size(point minimum_size,
 444                                                                         point maximum_size) const
 445 {
 446         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 447 
 448         assert(!label_.empty());
 449 
 450         const point border(config_->text_extra_width, config_->text_extra_height);
 451         point size = minimum_size - border;
 452 
 453         renderer_.set_link_aware(get_link_aware())
 454                         .set_link_color(get_link_color());
 455 
 456         renderer_.set_text(label_, use_markup_);
 457 
 458         renderer_.set_family_class(config_->text_font_family);
 459         renderer_.set_font_size(config_->text_font_size);
 460         renderer_.set_font_style(config_->text_font_style);
 461         renderer_.set_alignment(text_alignment_);
 462 
 463         // Try with the minimum wanted size.
 464         const int maximum_width = text_maximum_width_ != 0 ? text_maximum_width_
 465                                                                                                            : maximum_size.x;
 466 
 467         renderer_.set_maximum_width(maximum_width);
 468 
 469         renderer_.set_ellipse_mode(get_text_ellipse_mode());
 470 
 471         renderer_.set_characters_per_line(get_characters_per_line());
 472         if(get_characters_per_line() != 0 && !can_wrap()) {
 473                 WRN_GUI_L
 474                 << LOG_HEADER << " Limited the number of characters per line, "
 475                 << "but wrapping is not set, output may not be as expected.\n";
 476         }
 477 
 478         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 479                           << "' status: "
 480                           << " minimum_size " << minimum_size << " maximum_size "
 481                           << maximum_size << " text_maximum_width_ " << text_maximum_width_
 482                           << " can_wrap " << can_wrap() << " characters_per_line "
 483                           << get_characters_per_line() << " truncated "
 484                           << renderer_.is_truncated() << " renderer size "
 485                           << renderer_.get_size() << ".\n";
 486 
 487         // If doesn't fit try the maximum.
 488         if(renderer_.is_truncated() && !can_wrap()) {
 489                 // FIXME if maximum size is defined we should look at that
 490                 // but also we don't adjust for the extra text space yet!!!
 491                 maximum_size = point(config_->max_width, config_->max_height);
 492                 renderer_.set_maximum_width(maximum_size.x ? maximum_size.x - border.x
 493                                                                                                    : -1);
 494         }
 495 
 496         size = renderer_.get_size() + border;
 497 
 498         if(size.x < minimum_size.x) {
 499                 size.x = minimum_size.x;
 500         }
 501 
 502         if(size.y < minimum_size.y) {
 503                 size.y = minimum_size.y;
 504         }
 505 
 506         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 507                           << "' result " << size << ".\n";
 508         return size;
 509 }

 511 void styled_widget::signal_handler_show_tooltip(const event::ui_event event,
 512                                                                                    bool& handled,
 513                                                                                    const point& location)
 514 {
 515         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 516 
 517         if(!tooltip_.empty()) {
 518                 std::string tip = tooltip_;
 519                 if(!help_message_.empty()) {
 520                         utils::string_map symbols;
 521                         symbols["hotkey"] = hotkey::get_names(
 522                                         hotkey::hotkey_command::get_command_by_command(
 523                                                         hotkey::GLOBAL__HELPTIP).command);
 524 
 525                         tip = tooltip_ + utils::interpolate_variables_into_string(
 526                                                                          settings::has_helptip_message, &symbols);
 527                 }
 528 
 529                 event::message_show_tooltip message(tip, location, get_rectangle());
 530                 handled = fire(event::MESSAGE_SHOW_TOOLTIP, *this, message);
 531         }
 532 }

 657         , use_tooltip_on_label_overflow(true)
 658         , use_markup(cfg["use_markup"].to_bool(false))
 659 {

 661                 definition = "default";
 662         }

 664         VALIDATE_WITH_DEV_MESSAGE(

 666                         _("Found a widget with a helptip and without a tooltip."),
 667                         formatter() << "id '" << id << "' label '" << label_string
 668                                                  << "' helptip '" << help << "'.");
 669 
 670 
 671         DBG_GUI_P << "Window builder: found styled_widget with id '" << id
 672                           << "' and definition '" << definition << "'.\n";
 673 }
