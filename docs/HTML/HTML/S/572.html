  77 void multi_page::remove_page(const unsigned page, unsigned count)
  78 {
  79         assert(generator_);
  80 
  81         if(page >= get_page_count()) {
  82                 return;
  83         }
  84 
  85         if(!count || count > get_page_count()) {
  86                 count = get_page_count();
  87         }
  88 
  89         for(; count; --count) {
  90                 generator_->delete_item(page);
  91         }
  92 }

 249 builder_multi_page::builder_multi_page(const config& cfg)
 250         : implementation::builder_styled_widget(cfg), builders(), data()
 251 {
 252         for (const config& page : cfg.child_range("page_definition"))
 253         {
 254                 auto builder = std::make_shared<builder_grid>(page);
 255                 assert(builder);
 256                 builders[page["id"]] = builder;
 257         }
 258         VALIDATE(!builders.empty(), _("No page defined."));
 259 
 260         /** @todo This part is untested. */
 261         const config& d = cfg.child("page_data");
 262         if(!d) {
 263                 return;
 264         }
 265 
 266         auto builder = builders.begin()->second;
 267         for(const auto & row : d.child_range("row"))
 268         {
 269                 unsigned col = 0;
 270 
 271                 for(const auto & column : row.child_range("column"))
 272                 {
 273                         data.emplace_back();
 274                         for(const auto & i : column.attribute_range())
 275                         {
 276                                 data.back()[i.first] = i.second;
 277                         }
 278                         ++col;
 279                 }
 280 
 281                 VALIDATE(col == builder->cols,
 282                                  _("'list_data' must have "
 283                                    "the same number of columns as the 'list_definition'."));
 284         }
 285 }

 287 widget* builder_multi_page::build() const
 288 {
 289         multi_page* widget = new multi_page(*this);
 290 
 291         widget->set_page_builders(builders);
 292 
 293         DBG_GUI_G << "Window builder: placed multi_page '" << id
 294                           << "' with definition '" << definition << "'.\n";
 295 
 296         const auto conf = widget->cast_config_to<multi_page_definition>();
 297         assert(conf);
 298 
 299         widget->init_grid(conf->grid);
 300 
 301         widget->finalize(data);
 302 
 303         return widget;
 304 }
