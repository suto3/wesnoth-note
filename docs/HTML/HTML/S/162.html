  40 void wesnothd::player::set_status(wesnothd::player::STATUS status)
  41 {
  42         status_ = status;
  43         switch (status)
  44         {
  45                 case wesnothd::player::LOBBY:
  46                         cfg_.set_attr("status", "lobby");

  48                 case wesnothd::player::PLAYING:
  49                         cfg_.set_attr("status", "playing");
  50                         break;
  51                 case wesnothd::player::OBSERVING:
  52                         cfg_.set_attr("status", "observing");
  53                         break;

  55                         cfg_.set_attr("status", "unknown");
  56         }

  59 // keep 'available' and game name ('location') for backward compatibility

  61                                       const std::string& location)
  62 {
  63         if (game_id == 0) {
  64                 cfg_.set_attr("available", "yes");
  65                 set_status(LOBBY);
  66         } else {
  67                 cfg_.set_attr("available", "no");
  68         }
  69         cfg_.set_attr_dup("game_id", lexical_cast<std::string>(game_id).c_str());
  70         cfg_.set_attr_dup("location", location.c_str());
  71 }

  79 bool wesnothd::player::is_message_flooding()
  80 {
  81         const time_t now = time(nullptr);
  82         if (flood_start_ == 0) {
  83                 flood_start_ = now;
  84                 return false;
  85         }
  86 
  87         ++messages_since_flood_start_;
  88 
  89         if (now - flood_start_ > TimePeriod) {
  90                 messages_since_flood_start_ = 0;
  91                 flood_start_ = now;
  92         } else if (messages_since_flood_start_ >= MaxMessages) {
  93                 return true;
  94         }
  95         return false;
  96 }
