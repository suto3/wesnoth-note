 171 def htmlescape(text, quote=True):
 172     """Escape any HTML special characters in the given string."""
 173     if text is None:
 174         return text
 175     return html.escape(text, quote)
 176 
 177 def urlencode(text):
 178     """
 179     Encode the given string to ensure it only contains valid URL characters
 180     (also known as percent-encoding).
 181     """
 182     if text is None:
 183         return text
 184     return urllib.parse.quote(text, encoding='utf-8')

 186 def output(path, url, data):
 187     """Write the HTML index of add-ons into the specified directory."""
 188     try:
 189         os.mkdir(path)
 190     except OSError:
 191         pass
 192 
 193     outfile = open(path + "/index.html", "w")
 194 
 195     def w(line):
 196         outfile.write(line + "\n")
 197 
 198     am_dir = os.path.dirname(__file__) + "/"
 199     root_dir = am_dir + "../../../"
 200     images_to_tc = []
 201 
 202     # Copy required HTML assets into the destination dir.
 203     for filename in HTML_RESOURCES:
 204         Popen(["cp", "-u", am_dir + filename, path])
 205 
 206     server_name = os.path.basename(path)
 207     if server_name == "1.9":
 208         # 1.9 became the 1.10 add-ons server. Reflect that here.
 209         server_name = "1.10"
 210     elif server_name == "trunk":
 211         server_name = "Testing (Trunk)"
 212 
 213     w(WESMERE_HEADER % {
 214         "css_version": WESMERE_CSS_VERSION,
 215         "css_prefix": WESMERE_CSS_PREFIX,
 216         "server_name": server_name,
 217     })
 218     if url:
 219         w(WESMERE_DOWNLOAD_HELP)
 220 
 221     w('<table class="tablesorter" id="campaigns">\n<thead>\n<tr>')
 222     table_headers = [
 223         ("type", "Type"),
 224         ("icon", "Icon"),
 225         ("name", "Addon"),
 226         ("size", "Size"),
 227         ("stats", "Traffic"),
 228         ("date", "Date"),
 229         ("locales", "Translations")
 230     ]
 231     for header_class, header_label in table_headers:
 232         w('<th class="addon-%s">%s&nbsp;&nbsp;&nbsp;</th>' % (header_class, header_label))
 233     w('</tr>\n</thead>\n<tbody>')
 234 
 235     addons = data.get_all(tag="campaigns")[0]
 236     for addon in addons.get_all(tag="campaign"):
 237         v = addon.get_text_val
 238 
 239         addon_id = v("name") # Escaped as part of a path composition later on.
 240         title = htmlescape(v("title", "unknown"))
 241         size = float(v("size", "0")) # bytes
 242         display_size = size / (1024 * 1024) # MiB
 243         addon_type = htmlescape(v("type", "none"))
 244         version = htmlescape(v("version", "unknown"))
 245         author = htmlescape(v("author", "unknown"))
 246         feedback_url = v("feedback_url", None) # Escaped by a function call.
 247 
 248         icon = htmlescape(v("icon", ""))
 249         description = htmlescape(v('description', '(no description)'))
 250         imgurl = ""
 251 
 252         downloads = int(v("downloads", "0"))
 253         uploads = int(v("uploads", "0"))
 254         timestamp = int(v("timestamp", "0"))
 255         display_ts = time.strftime("%b %d %Y", time.localtime(timestamp))
 256 
 257         translations = addon.get_all(tag="translation")
 258         languages = [x.get_text_val("language") for x in translations]
 259 
 260         if icon:
 261             icon = icon.strip()
 262             tilde = icon.find("~")
 263             if tilde >= 0:
 264                 icon = icon[:tilde]
 265             if "\\" in icon:
 266                 icon = icon.replace("\\", "/")
 267             try:
 268                 os.mkdir(path + "/icons")
 269             except OSError:
 270                 pass
 271             if "." not in icon:
 272                 icon += ".png"
 273             src = root_dir + icon
 274             imgurl = "icons/" + os.path.basename(icon)
 275             if not os.path.exists(src):
 276                 src = root_dir + "data/core/images/" + icon
 277             if not os.path.exists(src):
 278                 src = root_dir + "images/" + icon
 279             if not os.path.exists(src):
 280                 src = glob.glob(root_dir + "data/campaigns/*/images/" + icon)
 281                 if src:
 282                     src = src[0]
 283                 if not src or not os.path.exists(src):
 284                     sys.stderr.write("Cannot find icon " + icon + "\n")
 285                     src = root_dir + "images/misc/missing-image.png"
 286                     imgurl = "icons/missing-image.png"
 287             images_to_tc.append((src, path + "/" + imgurl))
 288 
 289         w('<tr>')
 290 
 291         w('<td class="addon-type">')
 292         if addon_type in ADDON_TYPES_INFO:
 293             w('%(short)s<div class="type-tooltip"><b>%(long)s</b><br/>%(help)s</div>' \
 294                 % ADDON_TYPES_INFO[addon_type])
 295         else:
 296             w(addon_type)
 297         w('</td>')
 298 
 299         w(('<td class="addon-icon"><img alt="" src="%s"/>'
 300            '<div class="desc-tooltip"><b>%s</b><pre>%s</pre></div></td>') % (
 301                imgurl, title, description))
 302 
 303         def make_icon_button(url, label, icon):
 304             w(('<a href="{0}" title="{1}">'
 305                '<i class="fa fa-fw fa-2x fa-{2}" aria-hidden="true"></i>'
 306                '<span class="sr-only">{1}</span></a>').format(
 307                     htmlescape(url), htmlescape(label), icon))
 308 
 309         w('<td class="addon"><span hidden>%s</span>' % title)
 310         if url or feedback_url:
 311             w('<span class="addon-download">')
 312             if feedback_url:
 313                 make_icon_button(feedback_url, "Forum topic", "comment")
 314             if url:
 315                 link = url.rstrip("/") + "/" + urlencode(addon_id) + ".tar.bz2"
 316                 make_icon_button(link, "Download", "download")
 317             w('</span>')
 318         w(('<b>%s</b><br/>'
 319            '<span class="addon-meta"><span class="addon-meta-label">Version:</span> %s<br/>'
 320            '<span class="addon-meta-label">Author:</span> %s</span></td>') % (
 321                title, version, author))
 322 
 323         w("<td><span hidden>%d</span><b>%.2f</b>&nbsp;MiB</td>" % (size, display_size))
 324 
 325         w("<td><b>%d</b> down<br/>%s up</td>" % (downloads, uploads))
 326 
 327         w('<td><span hidden>%d</span>%s</td>' % (timestamp, display_ts))
 328 
 329         w("<td>%s</td>" % htmlescape(", ".join(languages), quote=False))
 330 
 331         w("</tr>")
 332 
 333     w('</tbody>\n</table>')
 334     w(WESMERE_FOOTER)
 335 
 336     sys.stderr.write("Done outputting html, now generating %d TC'ed images\n" % len(images_to_tc))
 337     for pair in images_to_tc:
 338         # wait() to ensure only one process exists at any time
 339         Popen([os.path.join(am_dir, "../unit_tree/TeamColorizer"), pair[0], pair[1]]).wait()
