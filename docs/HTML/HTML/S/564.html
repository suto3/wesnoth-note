  80 unsigned toggle_panel::num_states() const
  81 {
  82         std::div_t res = std::div(this->config()->state.size(), COUNT);
  83         assert(res.rem == 0);
  84         assert(res.quot > 0);
  85         return res.quot;

 140 SDL_Rect toggle_panel::get_client_rect() const
 141 {
 142         const auto conf = cast_config_to<toggle_panel_definition>();
 143         assert(conf);
 144 
 145         SDL_Rect result = get_rectangle();
 146         result.x += conf->left_border;
 147         result.y += conf->top_border;
 148         result.w -= conf->left_border + conf->right_border;
 149         result.h -= conf->top_border + conf->bottom_border;
 150 
 151         return result;
 152 }

 323  * @begin{parent}{name="gui/"}
 324  * @begin{tag}{name="oggle_panel_definition"}{min=0}{max=-1}{super="generic/widget_definition"}
 325  * The definition of a toggle panel. A toggle panel is like a toggle button, but
 326  * instead of being a button it's a panel. This means it can hold multiple child
 327  * items.
 328  *
 329  * @begin{tag}{name="resolution"}{min=0}{max=-1}{super=generic/widget_definition/resolution}
 330  * The resolution for a toggle panel also contains the following keys:
 331  * @begin{table}{config}
 332  *     top_border & unsigned & 0 &     The size which isn't used for the client
 333  *                                   area. $
 334  *     bottom_border & unsigned & 0 &  The size which isn't used for the client
 335  *                                   area. $
 336  *     left_border & unsigned & 0 &    The size which isn't used for the client
 337  *                                   area. $
 338  *     right_border & unsigned & 0 &   The size which isn't used for the client
 339  *                                   area. $
 340  * @end{table}
 341  *
 342  * The following states exist:
 343  * * state_enabled, the panel is enabled and not selected.
 344  * * state_disabled, the panel is disabled and not selected.
 345  * * state_focused, the mouse is over the panel and not selected.
 346  *
 347  * * state_enabled_selected, the panel is enabled and selected.
 348  * * state_disabled_selected, the panel is disabled and selected.
 349  * * state_focused_selected, the mouse is over the panel and selected.
 350  * @begin{tag}{name="state_enabled"}{min=0}{max=1}{super="generic/state"}
 351  * @end{tag}{name="state_enabled"}
 352  * @begin{tag}{name="state_disabled"}{min=0}{max=1}{super="generic/state"}
 353  * @end{tag}{name="state_disabled"}
 354  * @begin{tag}{name="state_focused"}{min=0}{max=1}{super="generic/state"}
 355  * @end{tag}{name="state_focused"}
 356  * @begin{tag}{name="state_enabled_selected"}{min=0}{max=1}{super="generic/state"}
 357  * @end{tag}{name="state_enabled_selected"}
 358  * @begin{tag}{name="state_disabled_selected"}{min=0}{max=1}{super="generic/state"}
 359  * @end{tag}{name="state_disabled_selected"}
 360  * @begin{tag}{name="state_focused_selected"}{min=0}{max=1}{super="generic/state"}
 361  * @end{tag}{name="state_focused_selected"}
 362  * @end{tag}{name="resolution"}
 363  * @end{tag}{name="oggle_panel_definition"}
 364  * @end{parent}{name="gui/"}
 365  */
 366 toggle_panel_definition::resolution::resolution(const config& cfg)
 367         : resolution_definition(cfg)
 368         , top_border(cfg["top_border"])
 369         , bottom_border(cfg["bottom_border"])
 370         , left_border(cfg["left_border"])
 371         , right_border(cfg["right_border"])
 372 {
 373         // Note the order should be the same as the enum state_t in toggle_panel.hpp.
 374         for(const auto& c : cfg.child_range("state"))
 375         {
 376                 state.emplace_back(c.child("enabled"));
 377                 state.emplace_back(c.child("disabled"));
 378                 state.emplace_back(c.child("focused"));
 379         }
