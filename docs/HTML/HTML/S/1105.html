 775 namespace detail {
 776         template<typename... T>
 777         struct config_construct_unpacker;
 778 
 779         template<>
 780         struct config_construct_unpacker<>
 781         {
 782                 void visit(config&) {}
 783         };
 784 
 785         template<typename K, typename V, typename... Rest>
 786         struct config_construct_unpacker<K, V, Rest...>
 787         {
 788                 template<typename K2 = K, typename V2 = V>
 789                 void visit(config& cfg, K2&& key, V2&& val, Rest... fwd)
 790                 {
 791                         cfg.insert(std::forward<K>(key), std::forward<V>(val));
 792                         config_construct_unpacker<Rest...> unpack;
 793                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 794                 }
 795         };
 796 
 797         template<typename T, typename... Rest>
 798         struct config_construct_unpacker<T, config, Rest...>
 799         {
 800                 template<typename T2 = T, typename C = config>
 801                 void visit(config& cfg, T2&& tag, C&& child, Rest... fwd)
 802                 {
 803                         cfg.add_child(std::forward<T>(tag), std::forward<config>(child));
 804                         config_construct_unpacker<Rest...> unpack;
 805                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 806                 }
 807         };
 808 
 809         template<typename T, typename... Rest>
 810         struct config_construct_unpacker<T, config&, Rest...>
 811         {
 812                 template<typename T2 = T>
 813                 void visit(config& cfg, T2&& tag, config& child, Rest... fwd)
 814                 {
 815                         cfg.add_child(std::forward<T>(tag), std::forward<config>(child));
 816                         config_construct_unpacker<Rest...> unpack;
 817                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 818                 }
 819         };
 820 }
 821 
 822 template<typename... T>
 823 inline config::config(config_key_type first, T&&... args)
 824 {
 825         detail::config_construct_unpacker<config_key_type, T...> unpack;
 826         unpack.visit(*this, first, std::forward<T>(args)...);
 827 }
