  70 int tree_view::remove_node(tree_view_node* node)
  71 {
  72         assert(node && node != root_node_ && node->parent_node_);
  73         const point node_size = node->get_size();
  74 
  75         tree_view_node::node_children_vector& siblings = node->parent_node_->children_;
  76 
  77         auto node_itor = std::find_if(siblings.begin(), siblings.end(),
  78                 [node](const std::unique_ptr<tree_view_node>& c) { return c.get() == node; }
  79         );
  80 
  81         assert(node_itor != siblings.end());
  82 
  83         const int position = node_itor - siblings.begin();
  84 
  85         siblings.erase(node_itor);
  86 
  87         if(get_size() != point()) {
  88                 // Don't shrink the width, need to think about a good algorithm to do so.
  89                 resize_content(0, -node_size.y);
  90         }
  91 
  92         return position;
  93 }

 127 void tree_view::resize_content(const int width_modification,
 128                                                                 const int height_modification,
 129                                                 const int width__modification_pos,
 130                                                 const int height_modification_pos)
 131 {
 132         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 133                           << " width_modification " << width_modification
 134                           << " height_modification " << height_modification << ".\n";
 135 
 136         if(content_resize_request(width_modification, height_modification, width__modification_pos, height_modification_pos)) {
 137 
 138                 // Calculate new size.
 139                 point size = content_grid()->get_size();
 140                 size.x += width_modification;
 141                 size.y += height_modification;
 142 
 143                 // Set new size.
 144                 content_grid()->set_size(size);
 145 
 146                 // Set status.
 147                 need_layout_ = true;
 148                 // If the content grows assume it "overwrites" the old content.
 149                 if(width_modification < 0 || height_modification < 0) {
 150                         set_is_dirty(true);
 151                 }
 152                 horizontal_scrollbar_moved();
 153                 DBG_GUI_L << LOG_HEADER << " succeeded.\n";
 154         } else {
 155                 DBG_GUI_L << LOG_HEADER << " failed.\n";
 156         }
 157 }

 159 void tree_view::layout_children(const bool force)
 160 {
 161         assert(root_node_ && content_grid());
 162 
 163         if(need_layout_ || force) {
 164                 root_node_->place(indentation_step_size_,
 165                                                   get_origin(),
 166                                                   content_grid()->get_size().x);
 167                 root_node_->set_visible_rectangle(content_visible_area_);
 168 
 169                 need_layout_ = false;
 170                 horizontal_scrollbar_moved();
 171         }
 172 }

 195 template<tree_view_node* (tree_view_node::*func) ()>
 196 tree_view_node* tree_view::get_next_node()
 197 {
 198         tree_view_node* selected = selected_item();
 199         if(!selected) {
 200                 return nullptr;
 201         }
 202         tree_view_node* visible = selected->get_last_visible_parent_node();
 203         if(visible != selected) {
 204                 return visible;
 205         }
 206         return (selected->*func)();
 207 }

 209 template<tree_view_node* (tree_view_node::*func) ()>
 210 bool tree_view::handle_up_down_arrow()
 211 {
 212         if(tree_view_node* next = get_next_node<func>())
 213         {
 214                 next->select_node();
 215                 SDL_Rect visible = content_visible_area();
 216                 SDL_Rect rect = next->get_grid().get_rectangle();
 217                 visible.y = rect.y;// - content_grid()->get_y();
 218                 visible.h = rect.h;
 219                 show_content_rect(visible);
 220                 return true;
 221         }
 222         return false;
 223 }

 246 void tree_view::handle_key_left_arrow(SDL_Keymod modifier, bool& handled)
 247 {
 248         tree_view_node* selected = selected_item();
 249         if(!selected || selected->is_folded()) {
 250                 scrollbar_container::handle_key_left_arrow(modifier, handled);
 251                 return;
 252         }
 253         selected->fold();
 254         handled = true;
 255 }
 256 
 257 void tree_view::handle_key_right_arrow(SDL_Keymod modifier, bool& handled)
 258 {
 259         tree_view_node* selected = selected_item();
 260         if(!selected || !selected->is_folded()) {
 261                 scrollbar_container::handle_key_right_arrow(modifier, handled);
 262                 return;
 263         }
 264         selected->unfold();
 265         handled = true;
 266 }

 374 builder_tree_view::builder_tree_view(const config& cfg)
 375         : builder_styled_widget(cfg)
 376         , vertical_scrollbar_mode(
 377                           get_scrollbar_mode(cfg["vertical_scrollbar_mode"]))
 378         , horizontal_scrollbar_mode(
 379                           get_scrollbar_mode(cfg["horizontal_scrollbar_mode"]))
 380         , indentation_step_size(cfg["indentation_step_size"])
 381         , nodes()
 382 {
 383 
 384         for(const auto & node : cfg.child_range("node"))
 385         {
 386                 nodes.emplace_back(node);
 387         }
 388 
 389         VALIDATE(!nodes.empty(), _("No nodes defined for a tree view."));
