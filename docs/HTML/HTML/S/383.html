 404 template<typename widget>
 405 void mp_create_game::on_filter_change(window& window, const std::string& id)
 406 {
 407         create_engine_.apply_level_filter(find_widget<widget>(&window, id, false).get_value());
 408 
 409         listbox& game_list = find_widget<listbox>(&window, "games_list", false);
 410 
 411         boost::dynamic_bitset<> filtered(game_list.get_item_count());
 412         for(const size_t i : create_engine_.get_filtered_level_indices(create_engine_.current_level_type())) {
 413                 filtered[i] = true;
 414         }
 415 
 416         game_list.set_row_shown(filtered);
 417 
 418         on_game_select(window);
 419 }

 421 void mp_create_game::on_game_select(window& window)
 422 {
 423         const int selected_game = find_widget<listbox>(&window, "games_list", false).get_selected_row();
 424 
 425         if(selected_game == selected_game_index_) {
 426                 return;
 427         }
 428 
 429         // Convert the absolute-index get_selected_row to a relatve index for the create_engine to handle
 430         selected_game_index_ = convert_to_game_filtered_index(selected_game);
 431 
 432         create_engine_.set_current_level(selected_game_index_);
 433         update_details(window);
 434 
 435         // General settings
 436         stacked_widget& stack = find_widget<stacked_widget>(&window, "pager", false);
 437 
 438         const bool can_select_era = create_engine_.current_level().allow_era_choice();
 439 
 440         menu_button& era_combo = find_widget<menu_button>(stack.get_layer_grid(TAB_GENERAL), "eras", false);
 441         if(!can_select_era) {
 442                 era_combo.set_label(_("No eras available for this game."));
 443         } else {
 444                 era_combo.set_selected(era_combo.get_value());
 445         }
 446 
 447         era_combo.set_active(can_select_era);
 448 
 449         // Custom options
 450         options_manager_->update_game_options();
 451 
 452         // Game settings
 453         update_map_settings(window);
 454 }

 456 void mp_create_game::on_tab_select(window& window)
 457 {
 458         const int i = find_widget<listbox>(&window, "tab_bar", false).get_selected_row();
 459         find_widget<stacked_widget>(&window, "pager", false).select_layer(i);
 460 
 461         /* HACK: the GUI2 field functions always internally save the correct value when set_widget_value is called
 462          *       - ie, when on_game_select calls update_map_settings, the correct values will be stored in the field,
 463          *       but if the settings tab isn't selected the widgets will not display the correct values. This forces
 464          *       an update when we switch to that tab so the widgets correctly display their values.
 465          *
 466          *       A possible better fix would be storing a pointer to the widget in question in the field object. It
 467          *       seems widgets will still correctly update even if they are not on the currently selected page if a
 468          *       pointer already exists.
 469          *
 470          *       Another possible fix would be allowing stacked_widget to return widgets on any page, not just on the
 471          *       currently visible one.
 472          */
 473         if(i == TAB_SETTINGS) {
 474                 update_map_settings(window);
 475         }
 476 }

 515 void mp_create_game::display_games_of_type(window& window, ng::level::TYPE type, const std::string& level)
 516 {
 517         create_engine_.set_current_level_type(type);
 518 
 519         listbox& list = find_widget<listbox>(&window, "games_list", false);
 520 
 521         list.clear();
 522 
 523         for(const auto& game : create_engine_.get_levels_by_type_unfiltered(type)) {
 524                 if(!game->can_launch_game()) {
 525                         continue;
 526                 }
 527 
 528                 std::map<std::string, string_map> data;
 529                 string_map item;
 530 
 531                 if(type == ng::level::TYPE::CAMPAIGN || type == ng::level::TYPE::SP_CAMPAIGN) {
 532                         item["label"] = game->icon();
 533                         data.emplace("game_icon", item);
 534                 }
 535 
 536                 item["label"] = game->name();
 537                 data.emplace("game_name", item);
 538 
 539                 list.add_row(data);
 540         }
 541 
 542         if(!level.empty() && !list.get_rows_shown().empty()) {
 543                 // Recalculate which rows should be visible
 544                 on_filter_change<slider>(window, "num_players");
 545                 on_filter_change<text_box>(window, "game_filter");
 546 
 547                 int level_index = create_engine_.find_level_by_id(level);
 548                 if(level_index >= 0 && size_t(level_index) < list.get_item_count()) {
 549                         list.select_row(level_index);
 550                 }
 551         }
 552 
 553         const bool is_random_map = type == ng::level::TYPE::RANDOM_MAP;
 554 
 555         find_widget<button>(&window, "random_map_regenerate", false).set_active(is_random_map);
 556         find_widget<button>(&window, "random_map_settings", false).set_active(is_random_map);
 557 
 558         // Override the last selection so on_game_select selects the new level
 559         selected_game_index_ = -1;
 560 
 561         on_game_select(window);
 562 }

 584 void mp_create_game::update_details(window& win)
 585 {
 586         styled_widget& players = find_widget<styled_widget>(&win, "map_num_players", false);
 587         styled_widget& map_size = find_widget<styled_widget>(&win, "map_size", false);
 588 
 589         if(create_engine_.current_level_type() == ng::level::TYPE::RANDOM_MAP) {
 590                 // If the current random map doesn't have data, generate it
 591                 if(create_engine_.generator_assigned() && create_engine_.current_level().data()["map_data"].empty()) {
 592                         create_engine_.init_generated_level_data();
 593                 }
 594 
 595                 find_widget<button>(&win, "random_map_settings", false).set_active(create_engine_.generator_has_settings());
 596         }
 597 
 598         create_engine_.current_level().set_metadata();
 599 
 600         // Reset the config_engine with new values
 601         config_engine_.reset(new ng::configure_engine(create_engine_.get_state()));
 602         config_engine_->update_initial_cfg(create_engine_.current_level().data());
 603         config_engine_->set_default_values();
 604 
 605         // Set the title, with newlines replaced. Newlines are sometimes found in SP Campaign names
 606         std::string title = create_engine_.current_level().name();
 607         boost::replace_all(title, "\n", " " + font::unicode_em_dash + " ");
 608         find_widget<styled_widget>(&win, "game_title", false).set_label(title);
 609 
 610         show_description(win, create_engine_.current_level().description());
 611 
 612         switch(create_engine_.current_level_type().v) {
 613                 case ng::level::TYPE::SCENARIO:
 614                 case ng::level::TYPE::USER_MAP:
 615                 case ng::level::TYPE::USER_SCENARIO:
 616                 case ng::level::TYPE::RANDOM_MAP: {
 617                         ng::scenario* current_scenario = dynamic_cast<ng::scenario*>(&create_engine_.current_level());
 618 
 619                         assert(current_scenario);
 620 
 621                         create_engine_.get_state().classification().campaign = "";
 622 
 623                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(0);
 624                         find_widget<minimap>(&win, "minimap", false).set_map_data(current_scenario->data()["map_data"]);
 625 
 626                         players.set_label(std::to_string(current_scenario->num_players()));
 627                         map_size.set_label(current_scenario->map_size());
 628 
 629                         break;
 630                 }
 631                 case ng::level::TYPE::CAMPAIGN:
 632                 case ng::level::TYPE::SP_CAMPAIGN: {
 633                         ng::campaign* current_campaign = dynamic_cast<ng::campaign*>(&create_engine_.current_level());
 634 
 635                         assert(current_campaign);
 636 
 637                         create_engine_.get_state().classification().campaign = current_campaign->data()["id"].str();
 638 
 639                         const std::string img = formatter() << current_campaign->data()["image"] << "~SCALE_INTO(265,265)";
 640 
 641                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(1);
 642                         find_widget<image>(&win, "campaign_image", false).set_image(img);
 643 
 644                         std::stringstream players_str;
 645                         players_str << current_campaign->min_players();
 646 
 647                         if(current_campaign->max_players() != current_campaign->min_players()) {
 648                                 players_str << " to " << current_campaign->max_players();
 649                         }
 650 
 651                         players.set_label(players_str.str());
 652                         map_size.set_label(font::unicode_em_dash);
 653 
 654                         break;
 655                 }
 656         }
 657 }

 700 void mp_create_game::load_game_callback(window& window)
 701 {
 702         savegame::loadgame load(window.video(), cfg_, create_engine_.get_state());
 703 
 704         if(!load.load_multiplayer_game()) {
 705                 return;
 706         }
 707 
 708         if(load.data().cancel_orders) {
 709                 create_engine_.get_state().cancel_orders();
 710         }
 711 
 712         window.set_retval(LOAD_GAME);
 713 }

 715 bool mp_create_game::dialog_exit_hook(window& window)
 716 {
 717         if(!create_engine_.current_level_has_side_data()) {
 718                 gui2::show_transient_error_message(window.video(), _("The selected game has no sides!"));
 719                 return false;
 720         }
 721 
 722         if(create_engine_.current_level_type() != ng::level::TYPE::CAMPAIGN &&
 723            create_engine_.current_level_type() != ng::level::TYPE::SP_CAMPAIGN
 724         ) {
 725                 return true;
 726         }
 727 
 728         return create_engine_.select_campaign_difficulty() != "CANCEL";

 731 void mp_create_game::post_show(window& window)
 732 {
 733         plugins_context_.reset();
 734 
 735         // Show all tabs so that find_widget works correctly
 736         find_widget<stacked_widget>(&window, "pager", false).select_layer(-1);
 737 
 738         if(get_retval() == LOAD_GAME) {
 739                 create_engine_.prepare_for_saved_game();
 740                 return;
 741         }
 742 
 743         if(get_retval() == window::OK) {
 744                 prefs::set_modifications(create_engine_.active_mods());
 745                 prefs::set_level_type(create_engine_.current_level_type().v);
 746                 prefs::set_level(create_engine_.current_level().id());
 747                 prefs::set_era(create_engine_.current_extra(ng::create_engine::ERA).id);
 748 
 749                 create_engine_.prepare_for_era_and_mods();
 750 
 751                 if(create_engine_.current_level_type() == ng::level::TYPE::CAMPAIGN ||
 752                         create_engine_.current_level_type() == ng::level::TYPE::SP_CAMPAIGN) {
 753                         create_engine_.prepare_for_campaign();
 754                 } else if(create_engine_.current_level_type() == ng::level::TYPE::SCENARIO) {
 755                         create_engine_.prepare_for_scenario();
 756                 } else {
 757                         // This means define= doesn't work for randomly generated scenarios
 758                         create_engine_.prepare_for_other();
 759                 }
 760 
 761                 create_engine_.get_parameters();
 762 
 763                 create_engine_.prepare_for_new_level();
 764 
 765                 std::vector<const config*> entry_points;
 766                 std::vector<std::string> entry_point_titles;
 767 
 768                 const auto& tagname = create_engine_.get_state().classification().get_tagname();
 769 
 770                 if(tagname == "scenario") {
 771                         const std::string first_scenario = create_engine_.current_level().data()["first_scenario"];
 772                         for(const config& scenario : game_config_manager::get()->game_config().child_range(tagname)) {
 773                                 const bool is_first = scenario["id"] == first_scenario;
 774                                 if(scenario["allow_new_game"].to_bool(false) || is_first || game_config::debug ) {
 775                                         const std::string& title = !scenario["new_game_title"].empty()
 776                                                 ? scenario["new_game_title"]
 777                                                 : scenario["name"];
 778 
 779                                         entry_points.insert(is_first ? entry_points.begin() : entry_points.end(), &scenario);
 780                                         entry_point_titles.insert(is_first ? entry_point_titles.begin() : entry_point_titles.end(), title);
 781                                 }
 782                         }
 783                 }
 784 
 785                 if(entry_points.size() > 1) {
 786                         gui2::dialogs::simple_item_selector dlg(_("Choose Starting Scenario"), _("Select at which point to begin this campaign."), entry_point_titles);
 787 
 788                         dlg.set_single_button(true);
 789                         dlg.show(window.video());
 790 
 791                         const config& scenario = *entry_points[dlg.selected_index()];
 792 
 793                         create_engine_.get_state().mp_settings().hash = scenario.hash();
 794                         create_engine_.get_state().set_scenario(scenario);
 795                 }
 796 
 797                 config_engine_->set_use_map_settings(use_map_settings_->get_widget_value(window));
 798 
 799                 if(!config_engine_->force_lock_settings()) {
 800                         // Max slider value (in this case, 100) means 'unlimited turns', so pass the value -1
 801                         const int num_turns = turns_->get_widget_value(window);
 802                         config_engine_->set_num_turns(num_turns < ::settings::turns_max ? num_turns : - 1);
 803                         config_engine_->set_village_gold(gold_->get_widget_value(window));
 804                         config_engine_->set_village_support(support_->get_widget_value(window));
 805                         config_engine_->set_xp_modifier(experience_->get_widget_value(window));
 806                         config_engine_->set_random_start_time(start_time_->get_widget_value(window));
 807                         config_engine_->set_fog_game(fog_->get_widget_value(window));
 808                         config_engine_->set_shroud_game(shroud_->get_widget_value(window));
 809 
 810                         config_engine_->write_parameters();
 811                 }
 812 
 813                 config_engine_->set_mp_countdown(time_limit_->get_widget_value(window));
 814                 config_engine_->set_mp_countdown_init_time(init_turn_limit_->get_widget_value(window));
 815                 config_engine_->set_mp_countdown_turn_bonus(turn_bonus_->get_widget_value(window));
 816                 config_engine_->set_mp_countdown_reservoir_time(reservoir_->get_widget_value(window));
 817                 config_engine_->set_mp_countdown_action_bonus(action_bonus_->get_widget_value(window));
 818 
 819                 config_engine_->set_allow_observers(observers_->get_widget_value(window));
 820                 config_engine_->set_registered_users_only(registered_users_->get_widget_value(window));
 821                 config_engine_->set_oos_debug(strict_sync_->get_widget_value(window));
 822                 config_engine_->set_shuffle_sides(shuffle_sides_->get_widget_value(window));
 823 
 824                 config_engine_->set_random_faction_mode(rfm_types_[selected_rfm_index_]);
 825 
 826                 // Since we don't have a field handling this option, we need to save the value manually
 827                 prefs::set_random_faction_mode(mp_game_settings::RANDOM_FACTION_MODE::enum_to_string(rfm_types_[selected_rfm_index_]));
 828 
 829                 // Save custom option settings
 830                 const config options_config = options_manager_->get_options_config();
 831 
 832                 config_engine_->set_options(options_config);
 833                 prefs::set_options(options_config);
 834 
 835                 // Set game name
 836                 const std::string name = find_widget<text_box>(&window, "game_name", false).get_value();
 837                 if(!name.empty() && (name != ng::configure_engine::game_name_default())) {
 838                         config_engine_->set_game_name(name);
 839                 }
 840 
 841                 // Set game password
 842                 const std::string password = find_widget<text_box>(&window, "game_password", false).get_value();
 843                 if(!password.empty()) {
 844                         config_engine_->set_game_password(password);
 845                 }
 846         }
 847 }
