  36 # This dehydrates the given units. However, if the unit is adjacent to a
  37 # friendly healer, the dehydration will be delayed instead: the unit suffers no
  38 # ill effects, but it cannot receive any healing on this turn either. Basically,
  39 # dehydration works just like poison.
  40 #
  41 # Each turn of dehydration lowers the unitâ€™s attack damage by 1 point and
  42 # decreases hitpoints by -$dehydration_loss.
  43 
  44 #define APPLY_DEHYDRATION_EFFECT FILTER
  45     [store_unit]
  46         [filter]
  47             {FILTER}
  48         [/filter]
  49 
  50         kill=no
  51         variable=dehydrating_units
  52     [/store_unit]
  53 
  54     [for]
  55         array=dehydrating_units
  56         [do]
  57             {VARIABLE dehydrating_units[$i].resting no}
  58 
  59             [if]
  60                 [have_unit]
  61                     x,y=$dehydrating_units[$i].x,$dehydrating_units[$i].y
  62 
  63                     [filter_adjacent]
  64                         is_enemy=no
  65                         [and]
  66                             ability=healing
  67                             [or]
  68                                 ability=curing
  69                             [/or]
  70                         [/and]
  71                     [/filter_adjacent]
  72                 [/have_unit]
  73 
  74                 [then]
  75                     {VARIABLE dehydrating_units[$i].status.dehydration_slowed_by_healer yes}
  76 
  77                     [unstore_unit]
  78                         variable=dehydrating_units[$i]
  79                         find_vacant=no
  80                     [/unstore_unit]
  81                 [/then]
  82 
  83                 [else]
  84                     [for]
  85                         array=dehydrating_units[$i].attack
  86                         variable=j
  87                         [do]
  88                             # if dehydrating for the first time, store the original attack stats
  89                             [if]
  90                                 [variable]
  91                                     name=dehydrating_units[$i].status.dehydrated
  92                                     not_equals=yes
  93                                 [/variable]
  94 
  95                                 [then]
  96                                     {VARIABLE dehydrating_units[$i].variables.original_attack[$j].damage $dehydrating_units[$i].attack[$j].damage}
  97                                 [/then]
  98                             [/if]
  99 
 100                             # the attack's damage drops by 1 point, but cannot drop to zero
 101                             [if]
 102                                 [variable]
 103                                     name=dehydrating_units[$i].attack[$j].damage
 104                                     greater_than=1
 105                                 [/variable]
 106 
 107                                 [then]
 108                                     {VARIABLE_OP dehydrating_units[$i].attack[$j].damage sub 1}
 109                                 [/then]
 110                             [/if]
 111                         [/do]
 112                     [/for]
 113 
 114                     # unitâ€™s hitpoints drop by $dehydration_loss, but cannot kill it
 115                     [if]
 116                         [variable]
 117                             name=dehydrating_units[$i].hitpoints
 118                             greater_than=$dehydration_loss
 119                         [/variable]
 120 
 121                         [then]
 122                             {VARIABLE_OP dehydrating_units[$i].hitpoints sub $dehydration_loss}
 123                         [/then]
 124 
 125                         [else]
 126                             {VARIABLE dehydrating_units[$i].hitpoints 1}
 127                         [/else]
 128                     [/if]
 129 
 130                     {VARIABLE dehydrating_units[$i].status.dehydrated yes}
 131 
 132                     [unstore_unit]
 133                         variable=dehydrating_units[$i]
 134                         find_vacant=no
 135                         text= _ "thirst"
 136                         {COLOR_HARM}
 137                     [/unstore_unit]
 138 
 139                     [unit_overlay]
 140                         x,y=$dehydrating_units[$i].x,$dehydrating_units[$i].y
 141                         image=misc/dehydration-icon.png
 142                     [/unit_overlay]
 143                 [/else]
 144             [/if]
 145         [/do]
 146     [/for]
 147 
 148     {CLEAR_VARIABLE dehydrating_units}
 149 #enddef

 151 # This rehydrates the given units. The units gain back their full attack damage.
 152 #define APPLY_HYDRATION_EFFECT FILTER
 153     [store_unit]
 154         [filter]
 155             {FILTER}
 156         [/filter]
 157 
 158         kill=no
 159         variable=hydrating_units
 160     [/store_unit]
 161 
 162     [for]
 163         array=hydrating_units
 164         [do]
 165             [if]
 166                 [variable]
 167                     name=hydrating_units[$i].status.dehydrated
 168                     equals=yes
 169                 [/variable]
 170 
 171                 [then]
 172                     [for]
 173                         array=hydrating_units[$i].attack
 174                         variable=j
 175                         [do]
 176                             {VARIABLE hydrating_units[$i].attack[$j].damage $hydrating_units[$i].variables.original_attack[$j].damage}
 177                         [/do]
 178                     [/for]
 179                 [/then]
 180             [/if]
 181 
 182             {VARIABLE hydrating_units[$i].status.dehydrated no}
 183 
 184             [unstore_unit]
 185                 variable=hydrating_units[$i]
 186                 find_vacant=no
 187                 text= _ "refreshed"
 188                 {COLOR_HEAL}
 189             [/unstore_unit]
 190 
 191             [remove_unit_overlay]
 192                 x,y=$hydrating_units[$i].x,$hydrating_units[$i].y
 193                 image=misc/dehydration-icon.png
 194             [/remove_unit_overlay]
 195         [/do]
 196     [/for]
 197 
 198     {CLEAR_VARIABLE hydrating_units}
 199 #enddef
