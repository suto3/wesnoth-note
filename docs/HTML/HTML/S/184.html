 120 static gui2::widget* find_widget(lua_State* L, int i, bool readonly)
 121 {
 122         if(!scoped_dialog::current) {
 123                 luaL_error(L, "no visible dialog");
 124                 error_oob_call_dtors:
 125                 luaL_argerror(L, i, "out of bounds");
 126                 error_not_str_call_dtors:
 127                 luaW_type_error(L, i, "string");
 128                 error_no_wgt_call_dtors:
 129                 luaL_argerror(L, i, "widget not found");
 130                 return nullptr;
 131         }
 132 
 133         gui2::widget* w = scoped_dialog::current->window.get();
 134         for(; !lua_isnoneornil(L, i); ++i)
 135         {
 136 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 137                 if(gui2::list_view* list = dynamic_cast<gui2::list_view*>(w))
 138 #else
 139                 if(gui2::listbox* list = dynamic_cast<gui2::listbox*>(w))
 140 #endif
 141                 {
 142                         int v = lua_tointeger(L, i);
 143                         if(v < 1) {
 144                                 goto error_oob_call_dtors;
 145                         }
 146                         int n = list->get_item_count();
 147                         if(v > n) {
 148                                 if(readonly) {
 149                                         goto error_oob_call_dtors;
 150                                 }
 151                                 utils::string_map dummy;
 152                                 for(; n < v; ++n) {
 153                                         list->add_row(dummy);
 154                                 }
 155                         }
 156                         w = list->get_row_grid(v - 1);
 157                 } else if(gui2::multi_page* multi_page = dynamic_cast<gui2::multi_page*>(w)) {
 158                         int v = lua_tointeger(L, i);
 159                         if(v < 1) {
 160                                 goto error_oob_call_dtors;
 161                         }
 162                         int n = multi_page->get_page_count();
 163                         if(v > n) {
 164                                 if(readonly) {
 165                                         goto error_oob_call_dtors;
 166                                 }
 167                                 utils::string_map dummy;
 168                                 for(; n < v; ++n) {
 169                                         multi_page->add_page(dummy);
 170                                 }
 171                         }
 172                         w = &multi_page->page_grid(v - 1);
 173                 } else if(gui2::tree_view* tree_view = dynamic_cast<gui2::tree_view*>(w)) {
 174                         gui2::tree_view_node& tvn = tree_view->get_root_node();
 175                         if(lua_isnumber(L, i)) {
 176                                 int v = lua_tointeger(L, i);
 177                                 if(v < 1) {
 178                                         goto error_oob_call_dtors;
 179                                 }
 180                                 int n = tvn.count_children();
 181                                 if(v > n) {
 182                                         goto error_oob_call_dtors;
 183                                 }
 184                                 w = &tvn.get_child_at(v - 1);
 185 
 186                         } else {
 187                                 std::string m = luaL_checkstring(L, i);
 188                                 w = tvn.find(m, false);
 189                         }
 190                 } else if(gui2::tree_view_node* tree_view_node = dynamic_cast<gui2::tree_view_node*>(w)) {
 191                         if(lua_isnumber(L, i)) {
 192                                 int v = lua_tointeger(L, i);
 193                                 if(v < 1) {
 194                                         goto error_oob_call_dtors;
 195                                 }
 196                                 int n = tree_view_node->count_children();
 197                                 if(v > n) {
 198                                         goto error_oob_call_dtors;
 199                                 }
 200                                 w = &tree_view_node->get_child_at(v - 1);
 201 
 202                         } else {
 203                                 std::string m = luaL_checkstring(L, i);
 204                                 w = tree_view_node->find(m, false);
 205                         }
 206                 } else if(gui2::stacked_widget* stacked_widget = dynamic_cast<gui2::stacked_widget*>(w)) {
 207                         if(lua_isnumber(L, i)) {
 208                                 int v = lua_tointeger(L, i);
 209                                 if(v < 1) {
 210                                         goto error_oob_call_dtors;
 211                                 }
 212                                 int n = stacked_widget->get_layer_count();
 213                                 if(v > n) {
 214                                         goto error_oob_call_dtors;
 215                                 }
 216                                 w = stacked_widget->get_layer_grid(v - 1);
 217                         } else {
 218                                 std::string m = luaL_checkstring(L, i);
 219                                 w = stacked_widget->find(m, false);
 220                         }
 221                 } else {
 222                         char const *m = lua_tostring(L, i);
 223                         if(!m) {
 224                                 goto error_not_str_call_dtors;
 225                         }
 226                         w = w->find(m, false);
 227                 }
 228                 if(!w) {
 229                         goto error_no_wgt_call_dtors;
 230                 }
 231         }
 232 
 233         return w;
 234 }

 241  * - Arg 2: function called at pre-show.
 242  * - Arg 3: function called at post-show.

 247         config def_cfg = luaW_checkconfig(L, 1);
 248

 250         scoped_dialog w(L, gui2::build(video, &def));

 252         if(!lua_isnoneornil(L, 2)) {
 253                 lua_pushvalue(L, 2);
 254                 lua_call(L, 0, 0);

 256

 262         }

 268 /**
 269  * Displays a message window
 270  * - Arg 1: Table describing the window
 271  * - Arg 2: List of options (nil or empty table - no options)
 272  * - Arg 3: Text input specifications (nil or empty table - no text input)
 273  * - Ret 1: option chosen (if no options: 0 if there's text input, -2 if escape pressed, else -1)
 274  * - Ret 2: string entered (empty if none, nil if no text input)
 275  */
 276 int show_message_dialog(lua_State* L, CVideo& video)
 277 {
 278         config txt_cfg;
 279         const bool has_input = !lua_isnoneornil(L, 3) && luaW_toconfig(L, 3, txt_cfg) && !txt_cfg.empty();
 280 
 281         gui2::dialogs::wml_message_input input;
 282         input.caption = txt_cfg["label"].str();
 283         input.text = txt_cfg["text"].str();
 284         input.maximum_length = txt_cfg["max_length"].to_int(256);
 285         input.text_input_was_specified = has_input;
 286 
 287         gui2::dialogs::wml_message_options options;
 288         if(!lua_isnoneornil(L, 2)) {
 289                 luaL_checktype(L, 2, LUA_TTABLE);
 290                 size_t n = lua_rawlen(L, 2);
 291                 for(size_t i = 1; i <= n; i++) {
 292                         lua_rawgeti(L, 2, i);
 293                         t_string short_opt;
 294                         config opt;
 295                         if(luaW_totstring(L, -1, short_opt)) {
 296                                 // Note: Although this currently uses the legacy_menu_item class
 297                                 // for the deprecated syntax, this branch should still be retained
 298                                 // when the deprecated syntax is removed, as a simpler method
 299                                 // of specifying options when only a single string is needed.
 300                                 const std::string& opt_str = short_opt;
 301                                 gui2::legacy_menu_item item(opt_str);
 302                                 opt["image"] = item.icon();
 303                                 opt["label"] = item.label();
 304                                 opt["description"] = item.description();
 305                                 opt["default"] = item.is_default();
 306                                 if(!opt["image"].blank() || !opt["description"].blank() || !opt["default"].blank()) {
 307                                         // The exact error message depends on whether & or = was actually present
 308                                         if(opt_str.find_first_of('=') == std::string::npos) {
 309                                                 // They just used a simple message, so the other error would be misleading
 310                                                 ERR_LUA << "[option]message= is deprecated, use label= instead.\n";
 311                                         } else {
 312                                                 ERR_LUA << "The &image=col1=col2 syntax is deprecated, use new DescriptionWML instead.\n";
 313                                         }
 314                                 }
 315                         } else if(!luaW_toconfig(L, -1, opt)) {
 316                                 std::ostringstream error;
 317                                 error << "expected array of config and/or translatable strings, but index ";
 318                                 error << i << " was a " << lua_typename(L, lua_type(L, -1));
 319                                 return luaL_argerror(L, 2, error.str().c_str());
 320                         }
 321                         gui2::dialogs::wml_message_option option(opt["label"], opt["description"], opt["image"]);
 322                         if(opt["default"].to_bool(false)) {
 323                                 options.chosen_option = i - 1;
 324                         }
 325                         options.option_list.push_back(option);
 326                         lua_pop(L, 1);
 327                 }
 328                 lua_getfield(L, 2, "default");
 329                 if(lua_isnumber(L, -1)) {
 330                         int i = lua_tointeger(L, -1);
 331                         if(i < 1 || size_t(i) > n) {
 332                                 std::ostringstream error;
 333                                 error << "default= key in options list is not a valid option index (1-" << n << ")";
 334                                 return luaL_argerror(L, 2, error.str().c_str());
 335                         }
 336                         options.chosen_option = i - 1;
 337                 }
 338                 lua_pop(L, 1);
 339         }
 340 
 341         const config& def_cfg = luaW_checkconfig(L, 1);
 342         const std::string& title = def_cfg["title"];
 343         const std::string& message = def_cfg["message"];
 344 
 345         using portrait = gui2::dialogs::wml_message_portrait;
 346         std::unique_ptr<portrait> left;
 347         std::unique_ptr<portrait> right;
 348         const bool is_double = def_cfg.has_attribute("second_portrait");
 349         const bool left_side = def_cfg["left_side"].to_bool(true);
 350         if(is_double || left_side) {
 351                 left.reset(new portrait {def_cfg["portrait"], def_cfg["mirror"].to_bool(false)});
 352         } else {
 353                 // This means right side only.
 354                 right.reset(new portrait {def_cfg["portrait"], def_cfg["mirror"].to_bool(false)});
 355         }
 356         if(is_double) {
 357                 right.reset(new portrait {def_cfg["second_portrait"], def_cfg["second_mirror"].to_bool(false)});
 358         }
 359 
 360         int dlg_result = gui2::dialogs::show_wml_message(video, title, message, left.get(), right.get(), options, input);
 361 
 362         if(!has_input && options.option_list.empty()) {
 363                 lua_pushinteger(L, dlg_result);
 364         } else {
 365                 lua_pushinteger(L, options.chosen_option + 1);
 366         }
 367 
 368         if(has_input) {
 369                 lua_pushlstring(L, input.text.c_str(), input.text.length());
 370         } else {
 371                 lua_pushnil(L);
 372         }
 373 
 374         return 2;
 375 }

 404 /**
 405  * Displays a popup menu at the current mouse position
 406  * Best used from a [set_menu_item], to show a submenu
 407  * - Arg 1: Configs defining each item, with keys icon, image/label, second_label, tooltip
 408  * - Args 2, 3: Initial selection (integer); whether to parse markup (boolean)
 409  */

 411         std::vector<config> items = lua_check<std::vector<config>>(L, 1);
 412         SDL_Rect pos {1,1,1,1};
 413         SDL_GetMouseState(&pos.x, &pos.y);
 414 
 415         int initial = -1;
 416         bool markup = false;
 417         if(lua_isnumber(L, 2)) {
 418                 initial = lua_tointeger(L, 2) - 1;
 419                 markup = luaW_toboolean(L, 3);
 420         } else if(lua_isnumber(L, 3)) {

 426         menu.show(video);
 427         lua_pushinteger(L, menu.selected_item() + 1);
 428         return 1;

 431 /**
 432  * Displays a simple message box.
 433  */
 434 int show_message_box(lua_State* L, CVideo& video) {
 435         const std::string title = luaL_checkstring(L, 1), message = luaL_checkstring(L, 2);
 436         std::string button = luaL_optstring(L, 3, "ok");
 437         std::transform(button.begin(), button.end(), button.begin(), [](char c) { return std::tolower(c); });
 438         bool markup = lua_isnoneornil(L, 3) ? luaW_toboolean(L, 3) : luaW_toboolean(L, 4);
 439         using button_style = gui2::dialogs::message::button_style;
 440         boost::optional<button_style> style;
 441         if(button.empty()) {
 442                 style = button_style::auto_close;
 443         } else if(button == "ok") {
 444                 style = button_style::ok_button;
 445         } else if(button == "close") {
 446                 style = button_style::close_button;
 447         } else if(button == "ok_cancel") {
 448                 style = button_style::ok_cancel_buttons;
 449         } else if(button == "cancel") {
 450                 style = button_style::cancel_button;
 451         } else if(button == "yes_no") {
 452                 style = button_style::yes_no_buttons;
 453         }
 454         if(style) {
 455                 int result = gui2::show_message(video, title, message, *style, markup, markup);
 456                 if(style == button_style::ok_cancel_buttons || style == button_style::yes_no_buttons) {
 457                         lua_pushboolean(L, result == gui2::window::OK);
 458                         return 1;
 459                 }
 460         } else {
 461                 gui2::show_message(video, title, message, button, false, markup, markup);
 462         }
 463         return 0;
 464 }

 466 /**
 467  * Sets the value of a widget on the current dialog.
 468  * - Arg 1: scalar.
 469  * - Args 2..n: path of strings and integers.
 470  */
 471 int intf_set_dialog_value(lua_State* L)
 472 {
 473         gui2::widget *w = find_widget(L, 2, false);
 474 
 475 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 476         if(gui2::list_view* list = dynamic_cast<gui2::list_view*>(w))
 477 #else
 478         if(gui2::listbox* list = dynamic_cast<gui2::listbox*>(w))
 479 #endif
 480         {
 481                 if(lua_istable(L, 1)) {
 482                         // Do two passes in case has_minimum is true
 483                         // Probably not the best way to do it, but should work in the majority of cases.
 484                         std::vector<int> selected_vec = lua_check<std::vector<int>>(L, 1);
 485                         std::set<int> selected(selected_vec.begin(), selected_vec.end());
 486                         for(unsigned i = 0; i < list->get_item_count(); i++) {
 487                                 list->select_row(i, selected.count(i + 1) > 0);
 488                         }
 489                         for(unsigned i = 0; i < list->get_item_count(); i++) {
 490                                 list->select_row(i, selected.count(i + 1) > 0);
 491                         }
 492                 } else {
 493                         int v = luaL_checkinteger(L, 1);
 494                         int n = list->get_item_count();
 495                         if(1 <= v && v <= n) {
 496                                 list->select_row(v - 1);
 497                         } else {
 498                                 return luaL_argerror(L, 1, "out of bounds");
 499                         }
 500                 }
 501         } else if(gui2::multi_page* multi_page = dynamic_cast<gui2::multi_page*>(w)) {
 502                 int v = luaL_checkinteger(L, 1);
 503                 int n = multi_page->get_page_count();
 504                 if(1 <= v && v <= n) {
 505                         multi_page->select_page(v - 1);
 506                 } else {
 507                         return luaL_argerror(L, 1, "out of bounds");
 508                 }
 509         } else if (gui2::selectable_item* selectable = dynamic_cast<gui2::selectable_item*>(w)) {
 510                 if(selectable->num_states() == 2) {
 511                         selectable->set_value_bool(luaW_toboolean(L, 1));
 512                 } else {
 513                         selectable->set_value(luaL_checkinteger(L, 1) -1);
 514                 }
 515         } else if (gui2::text_box* text_box = dynamic_cast<gui2::text_box*>(w)) {
 516                 const t_string& text = luaW_checktstring(L, 1);
 517                 text_box->set_value(text.str());
 518         } else if (gui2::slider* slider = dynamic_cast<gui2::slider*>(w)) {
 519                 const int v = luaL_checkinteger(L, 1);
 520                 const int m = slider->get_minimum_value();
 521                 const int n = slider->get_maximum_value();
 522                 if(m <= v && v <= n) {
 523                         slider->set_value(v);
 524                 } else {
 525                         return luaL_argerror(L, 1, "out of bounds");
 526                 }
 527         } else if(gui2::progress_bar* progress_bar = dynamic_cast<gui2::progress_bar*>(w)) {
 528                 const int v = luaL_checkinteger(L, 1);
 529                 if(0 <= v && v <= 100) {
 530                         progress_bar->set_percentage(v);
 531                 } else {
 532                         return luaL_argerror(L, 1, "out of bounds");
 533                 }
 534         } else if(gui2::stacked_widget* stacked_widget = dynamic_cast<gui2::stacked_widget*>(w)) {
 535                 if(lua_istable(L, 1)) {
 536                         boost::dynamic_bitset<> states;
 537                         states.resize(stacked_widget->get_layer_count());
 538                         for(unsigned i : lua_check<std::vector<unsigned>>(L, 1)) {
 539                                 if(i > 0 && i <= stacked_widget->get_layer_count()) {
 540                                         states[i] = true;
 541                                 }
 542                         }
 543                         stacked_widget->select_layers(states);
 544                 } else {
 545                         const int v = luaL_checkinteger(L, 1);
 546                         const int n = stacked_widget->get_layer_count();
 547                         if(v >= 0 && v <= n) {
 548                                 stacked_widget->select_layer(v - 1);
 549                         }
 550                 }
 551         } else if(gui2::unit_preview_pane* unit_preview_pane = dynamic_cast<gui2::unit_preview_pane*>(w)) {
 552                 if(const unit_type* ut = luaW_tounittype(L, 1)) {
 553                         unit_preview_pane->set_displayed_type(*ut);
 554                 } else if(unit* u = luaW_tounit(L, 1)) {
 555                         unit_preview_pane->set_displayed_unit(*u);
 556                 } else {
 557                         return luaW_type_error(L, 1, "unit or unit type");
 558                 }
 559         } else if(gui2::tree_view_node* node = dynamic_cast<gui2::tree_view_node*>(w)) {
 560                 const bool unfolded = luaW_toboolean(L, 1);
 561                 if(unfolded) {
 562                         node->unfold();
 563                 } else {
 564                         node->fold();
 565                 }
 566         } else if(gui2::multimenu_button* menu = dynamic_cast<gui2::multimenu_button*>(w)) {
 567                 if(lua_istable(L, 1)) {
 568                         boost::dynamic_bitset<> states;
 569                         states.resize(menu->num_options());
 570                         for(unsigned i : lua_check<std::vector<unsigned>>(L, 1)) {
 571                                 if(i > 0 && i <= menu->num_options()) {
 572                                         states[i] = true;
 573                                 }
 574                         }
 575                         menu->select_options(states);
 576                 } else {
 577                         int v = luaL_checkinteger(L, 1);
 578                         int n = menu->num_options();
 579                         if(1 <= v && v <= n) {
 580                                 menu->select_option(v - 1);
 581                         } else {
 582                                 return luaL_argerror(L, 1, "out of bounds");
 583                         }
 584                 }
 585         } else {
 586                 t_string v = luaW_checktstring(L, 1);
 587                 gui2::styled_widget* c = dynamic_cast<gui2::styled_widget*>(w);
 588                 if(!c) {
 589                         return luaL_argerror(L, lua_gettop(L), "unsupported widget");
 590                 }
 591                 c->set_label(v);
 592         }
 593 
 594         return 0;

 597 /**
 598  * Gets the value of a widget on the current dialog.
 599  * - Args 1..n: path of strings and integers.
 600  * - Ret 1: scalar.
 601  */
 602 int intf_get_dialog_value(lua_State* L)
 603 {
 604         gui2::widget *w = find_widget(L, 1, true);
 605         int num_rets = 1;
 606 
 607 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 608         if(gui2::list_view* list = dynamic_cast<gui2::list_view*>(w))
 609 #else
 610         if(gui2::listbox* list = dynamic_cast<gui2::listbox*>(w))
 611 #endif
 612         {
 613                 lua_pushinteger(L, list->get_selected_row() + 1);
 614                 num_rets = 2;
 615                 lua_newtable(L);
 616                 int count = 0;
 617                 for(unsigned i = 0; i < list->get_item_count(); i++) {
 618                         if(list->row_selected(i)) {
 619                                 count++;
 620                                 lua_pushnumber(L, i + 1);
 621                                 lua_rawseti(L, -2, count);
 622                         }
 623                 }
 624         } else if(gui2::multi_page* multi_page = dynamic_cast<gui2::multi_page*>(w)) {
 625                 lua_pushinteger(L, multi_page->get_selected_page() + 1);
 626         } else if(gui2::selectable_item* selectable = dynamic_cast<gui2::selectable_item*>(w)) {
 627                 if(selectable->num_states() == 2) {
 628                         lua_pushboolean(L, selectable->get_value_bool());
 629                 } else {
 630                         lua_pushinteger(L, selectable->get_value() + 1);
 631                 }
 632         } else if(gui2::text_box* text_box = dynamic_cast<gui2::text_box*>(w)) {
 633                 lua_pushstring(L, text_box->get_value().c_str());
 634         } else if(gui2::slider* slider = dynamic_cast<gui2::slider*>(w)) {
 635                 lua_pushinteger(L, slider->get_value());
 636         } else if(gui2::progress_bar* progress_bar = dynamic_cast<gui2::progress_bar*>(w)) {
 637                 lua_pushinteger(L, progress_bar->get_percentage());
 638         } else if(gui2::tree_view* tree_view = dynamic_cast<gui2::tree_view*>(w)) {
 639                 std::vector<int> path = tree_view->selected_item()->describe_path();
 640                 lua_createtable(L, path.size(), 0);
 641                 for(size_t i = 0; i < path.size(); ++i) {
 642                         lua_pushinteger(L, path[i] + 1);
 643                         lua_rawseti(L, -2, i + 1);
 644                 }
 645         } else if(gui2::stacked_widget* stacked_widget = dynamic_cast<gui2::stacked_widget*>(w)) {
 646                 lua_pushinteger(L, stacked_widget->current_layer());
 647                 num_rets = 2;
 648                 lua_newtable(L);
 649                 int count = 0;
 650                 for(unsigned i = 0; i < stacked_widget->get_layer_count(); i++) {
 651                         if(stacked_widget->layer_selected(i)) {
 652                                 count++;
 653                                 lua_pushnumber(L, i + 1);
 654                                 lua_rawseti(L, -2, count);
 655                         }
 656                 }
 657         } else if(gui2::multimenu_button* menu = dynamic_cast<gui2::multimenu_button*>(w)) {
 658                 auto selected = menu->get_toggle_states();
 659                 int count = 0;
 660                 lua_newtable(L);
 661                 for(unsigned i = 0; i < selected.size(); i++) {
 662                         if(selected[i]) {
 663                                 count++;
 664                                 lua_pushnumber(L, i + 1);
 665                                 lua_rawseti(L, -2, count);
 666                         }
 667                 }
 668         } else {
 669                 return luaL_argerror(L, lua_gettop(L), "unsupported widget");
 670         }
 671 
 672         return num_rets;
 673 }

 723 namespace { // helpers of intf_set_dialog_callback()
 724         void dialog_callback(gui2::widget& w)
 725         {
 726                 int cb;
 727                 {
 728                         scoped_dialog::callback_map &m = scoped_dialog::current->callbacks;
 729                         scoped_dialog::callback_map::const_iterator i = m.find(&w);
 730                         if(i == m.end()) {
 731                                 return;
 732                         }
 733                         cb = i->second;
 734                 }
 735                 lua_State *L = scoped_dialog::current->L;
 736                 lua_pushstring(L, dlgclbkKey);
 737                 lua_rawget(L, LUA_REGISTRYINDEX);
 738                 lua_rawgeti(L, -1, cb);
 739                 lua_remove(L, -2);
 740                 lua_call(L, 0, 0);
 741         }

 743         /** Helper struct for intf_set_dialog_callback. */
 744         struct dialog_callback_wrapper
 745         {
 746                 void forward(gui2::widget* widget)
 747                 {
 748                         assert(widget);
 749                         dialog_callback(*widget);
 750                 }
 751         };

 754 /**
 755  * Sets a callback on a widget of the current dialog.
 756  * - Arg 1: function.
 757  * - Args 2..n: path of strings and integers.
 758  */
 759 int intf_set_dialog_callback(lua_State* L)
 760 {
 761         gui2::widget* w = find_widget(L, 2, true);
 762 
 763         scoped_dialog::callback_map &m = scoped_dialog::current->callbacks;
 764         scoped_dialog::callback_map::iterator i = m.find(w);
 765         if(i != m.end()) {
 766                 lua_pushstring(L, dlgclbkKey);
 767                 lua_rawget(L, LUA_REGISTRYINDEX);
 768                 lua_pushnil(L);
 769                 lua_rawseti(L, -2, i->second);
 770                 lua_pop(L, 1);
 771                 m.erase(i);
 772         }
 773 
 774         if(lua_isnil(L, 1)) {
 775                 return 0;
 776         }
 777 
 778         if(gui2::clickable_item* c = dynamic_cast<gui2::clickable_item*>(w)) {
 779                 static dialog_callback_wrapper wrapper;
 780                 c->connect_click_handler(std::bind(&dialog_callback_wrapper::forward, wrapper, w));
 781         } else if(gui2::selectable_item* s = dynamic_cast<gui2::selectable_item*>(w)) {
 782                 connect_signal_notify_modified(dynamic_cast<gui2::widget&>(*s), std::bind(dialog_callback, _1));
 783         }
 784 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 785         else if(gui2::list_view* l = dynamic_cast<gui2::list_view*>(w)) {
 786                 static dialog_callback_wrapper wrapper;
 787                 connect_signal_notify_modified(*l
 788                                 , std::bind(
 789                                           &dialog_callback_wrapper::forward
 790                                         , wrapper
 791                                         , w));
 792         }
 793 #else
 794         else if(gui2::listbox* l = dynamic_cast<gui2::listbox*>(w)) {
 795                 static dialog_callback_wrapper wrapper;
 796                 connect_signal_notify_modified(*l, std::bind(&dialog_callback_wrapper::forward, wrapper, w));
 797         }
 798 #endif
 799         else if(gui2::tree_view* tv = dynamic_cast<gui2::tree_view*>(w)) {
 800                 tv->set_selection_change_callback(&dialog_callback);
 801         } else {
 802                 return luaL_argerror(L, lua_gettop(L), "unsupported widget");
 803         }
 804 
 805         lua_pushstring(L, dlgclbkKey);
 806         lua_rawget(L, LUA_REGISTRYINDEX);
 807         int n = lua_rawlen(L, -1) + 1;
 808         m[w] = n;
 809         lua_pushvalue(L, 1);
 810         lua_rawseti(L, -2, n);
 811         lua_pop(L, 1);
 812 
 813         return 0;
 814 }

 889 /**
 890  * Sets the visiblity of a widget in the current dialog.
 891  * - Arg 1: boolean.
 892  * - Args 2..n: path of strings and integers.
 893  */
 894 int intf_set_dialog_visible(lua_State* L)
 895 {
 896         typedef gui2::styled_widget::visibility visibility;
 897 
 898         visibility flag = visibility::visible;
 899 
 900         switch(lua_type(L, 1)) {
 901                 case LUA_TBOOLEAN:
 902                         flag = luaW_toboolean(L, 1)
 903                                         ? visibility::visible
 904                                         : visibility::invisible;
 905                         break;
 906                 case LUA_TSTRING:
 907                         {
 908                                 const std::string& str = lua_tostring(L, 1);
 909                                 if(str == "visible") {
 910                                         flag = visibility::visible;
 911                                 } else if(str == "hidden") {
 912                                         flag = visibility::hidden;
 913                                 } else if(str == "invisible") {
 914                                         flag = visibility::invisible;
 915                                 } else {
 916                                         return luaL_argerror(L, 1, "string must be one of: visible, hidden, invisible");
 917                                 }
 918                         }
 919                         break;
 920                 default:
 921                         return luaW_type_error(L, 1, "boolean or string");
 922         }
 923 
 924         gui2::widget* w = find_widget(L, 2, true);
 925         gui2::styled_widget* c = dynamic_cast<gui2::styled_widget*>(w);
 926         if(!c) {
 927                 return luaL_argerror(L, lua_gettop(L), "unsupported widget");
 928         }
 929 
 930         c->set_visible(flag);
 931 
 932         if(flag == visibility::hidden) {
 933                 // HACK: this is needed to force the widget to be repainted immediately
 934                 //       to get rid of its ghost image.
 935                 scoped_dialog::current->window->invalidate_layout();
 936         }
 937 
 938         return 0;
 939 }
