  48 void controller_base::handle_event(const SDL_Event& event)
  49 {
  50         if(gui::in_dialog()) {
  51                 return;
  52         }
  53 
  54         static const hotkey::hotkey_command& quit_hotkey
  55                         = hotkey::hotkey_command::get_command_by_command(hotkey::HOTKEY_QUIT_GAME);
  56 
  57         events::mouse_handler_base& mh_base = get_mouse_handler_base();
  58 
  59         switch(event.type) {
  60         case SDL_TEXTINPUT:
  61                 if(have_keyboard_focus()) {
  62                         hotkey::key_event(event, get_hotkey_command_executor());
  63                 }
  64                 break;
  65 
  66         case SDL_TEXTEDITING:
  67                 if(have_keyboard_focus()) {
  68                         SDL_Event evt = event;
  69                         evt.type = SDL_TEXTINPUT;
  70                         hotkey::key_event(evt, get_hotkey_command_executor());
  71                         SDL_StopTextInput();
  72                         SDL_StartTextInput();
  73                 }
  74                 break;
  75 
  76         case SDL_KEYDOWN:
  77                 // Detect key press events, unless there something that has keyboard focus
  78                 // in which case the key press events should go only to it.
  79                 if(have_keyboard_focus()) {
  80                         if(event.key.keysym.sym == SDLK_ESCAPE) {
  81                                 hotkey::execute_command(quit_hotkey, get_hotkey_command_executor());
  82                                 break;
  83                         }
  84 
  85                         process_keydown_event(event);
  86                         hotkey::key_event(event, get_hotkey_command_executor());
  87                         process_keyup_event(event);
  88                 } else {
  89                         process_focus_keydown_event(event);
  90                 }
  91                 break;
  92 
  93         case SDL_KEYUP:
  94                 process_keyup_event(event);
  95                 hotkey::key_event(event, get_hotkey_command_executor());
  96                 break;
  97 
  98         case SDL_JOYBUTTONDOWN:
  99                 process_keydown_event(event);
 100                 hotkey::jbutton_event(event, get_hotkey_command_executor());
 101                 break;
 102 
 103         case SDL_JOYHATMOTION:
 104                 process_keydown_event(event);
 105                 hotkey::jhat_event(event, get_hotkey_command_executor());
 106                 break;
 107 
 108         case SDL_MOUSEMOTION:
 109                 // Ignore old mouse motion events in the event queue
 110                 SDL_Event new_event;
 111                 if(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {
 112                         while(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {
 113                         };
 114                         mh_base.mouse_motion_event(new_event.motion, is_browsing());
 115                 } else {
 116                         mh_base.mouse_motion_event(event.motion, is_browsing());
 117                 }
 118                 break;
 119 
 120         case SDL_MOUSEBUTTONDOWN:
 121                 process_keydown_event(event);
 122                 mh_base.mouse_press(event.button, is_browsing());
 123                 hotkey::mbutton_event(event, get_hotkey_command_executor());
 124                 break;
 125 
 126         case SDL_MOUSEBUTTONUP:
 127                 mh_base.mouse_press(event.button, is_browsing());
 128                 if(mh_base.get_show_menu()) {
 129                         show_menu(get_display().get_theme().context_menu()->items(), event.button.x, event.button.y, true,
 130                                         get_display());
 131                 }
 132                 break;
 133 
 134         case SDL_MOUSEWHEEL:
 135                 mh_base.mouse_wheel(-event.wheel.x, event.wheel.y, is_browsing());
 136                 break;
 137 
 138         default:
 139                 break;
 140         }
 141 }

 148 bool controller_base::handle_scroll(int mousex, int mousey, int mouse_flags, double x_axis, double y_axis)
 149 {
 150         sdl::window* window = CVideo::get_singleton().get_window();
 151 
 152         const bool mouse_in_window =
 153                 (window != nullptr && (window->get_flags() & SDL_WINDOW_MOUSE_FOCUS) != 0)
 154                 || preferences::get("scroll_when_mouse_outside", true);
 155 
 156         int scroll_speed = preferences::scroll_speed();
 157         int dx = 0, dy = 0;
 158 
 159         int scroll_threshold = preferences::mouse_scroll_enabled()
 160                 ? preferences::mouse_scroll_threshold()
 161                 : 0;
 162 
 163         for(const theme::menu& m : get_display().get_theme().menus()) {
 164                 if(sdl::point_in_rect(mousex, mousey, m.get_location())) {
 165                         scroll_threshold = 0;
 166                 }
 167         }
 168 
 169         // Apply keyboard scrolling
 170         dy -= scroll_up_    * scroll_speed;
 171         dy += scroll_down_  * scroll_speed;
 172         dx -= scroll_left_  * scroll_speed;
 173         dx += scroll_right_ * scroll_speed;
 174 
 175         // Scroll if mouse is placed near the edge of the screen
 176         if(mouse_in_window) {
 177                 if(mousey < scroll_threshold) {
 178                         dy -= scroll_speed;
 179                 }
 180 
 181                 if(mousey > get_display().h() - scroll_threshold) {
 182                         dy += scroll_speed;
 183                 }
 184 
 185                 if(mousex < scroll_threshold) {
 186                         dx -= scroll_speed;
 187                 }
 188 
 189                 if(mousex > get_display().w() - scroll_threshold) {
 190                         dx += scroll_speed;
 191                 }
 192         }
 193 
 194         events::mouse_handler_base& mh_base = get_mouse_handler_base();
 195 
 196         // Scroll with middle-mouse if enabled
 197         if((mouse_flags & SDL_BUTTON_MMASK) != 0 && preferences::middle_click_scrolls()) {
 198                 const SDL_Point original_loc = mh_base.get_scroll_start();
 199 
 200                 if(mh_base.scroll_started()) {
 201                         const SDL_Rect& rect = get_display().map_outside_area();
 202 
 203                         if(sdl::point_in_rect(mousex, mousey, rect) && mh_base.scroll_started()) {
 204                                 // Scroll speed is proportional from the distance from the first
 205                                 // middle click and scrolling speed preference.
 206                                 const double speed = 0.04 * sqrt(static_cast<double>(scroll_speed));
 207                                 const double snap_dist = 16; // Snap to horizontal/vertical scrolling
 208                                 const double x_diff = (mousex - original_loc.x);
 209                                 const double y_diff = (mousey - original_loc.y);
 210 
 211                                 if(std::fabs(x_diff) > snap_dist || std::fabs(y_diff) <= snap_dist) {
 212                                         dx += speed * x_diff;
 213                                 }
 214 
 215                                 if(std::fabs(y_diff) > snap_dist || std::fabs(x_diff) <= snap_dist) {
 216                                         dy += speed * y_diff;
 217                                 }
 218                         }
 219                 } else { // Event may fire mouse down out of order with respect to initial click
 220                         mh_base.set_scroll_start(mousex, mousey);
 221                 }
 222         }
 223 
 224         // scroll with joystick
 225         dx += round_double(x_axis * scroll_speed);
 226         dy += round_double(y_axis * scroll_speed);
 227 
 228         return get_display().scroll(dx, dy);
 229 }

 231 void controller_base::play_slice(bool is_delay_enabled)
 232 {
 233         CKey key;
 234 
 235         if(plugins_context* l = get_plugins_context()) {
 236                 l->play_slice();
 237         }
 238 
 239         events::pump();
 240         events::raise_process_event();
 241         events::raise_draw_event();
 242 
 243         // Update sound sources before scrolling
 244         if(soundsource::manager* l = get_soundsource_man()) {
 245                 l->update();
 246         }
 247 
 248         const theme::menu* const m = get_display().menu_pressed();
 249         if(m != nullptr) {
 250                 const SDL_Rect& menu_loc = m->location(get_display().screen_area());
 251                 show_menu(m->items(), menu_loc.x + 1, menu_loc.y + menu_loc.h + 1, false, get_display());
 252 
 253                 return;
 254         }
 255 
 256         const theme::action* const a = get_display().action_pressed();
 257         if(a != nullptr) {
 258                 const SDL_Rect& action_loc = a->location(get_display().screen_area());
 259                 execute_action(a->items(), action_loc.x + 1, action_loc.y + action_loc.h + 1, false);
 260 
 261                 return;
 262         }
 263 
 264         auto str_vec = additional_actions_pressed();
 265         if(!str_vec.empty()) {
 266                 execute_action(str_vec, 0, 0, false);
 267                 return;
 268         }
 269 
 270         bool was_scrolling = scrolling_;
 271 
 272         std::pair<double, double> values = joystick_manager_.get_scroll_axis_pair();
 273         const double joystickx = values.first;
 274         const double joysticky = values.second;
 275 
 276         int mousex, mousey;
 277         Uint8 mouse_flags = SDL_GetMouseState(&mousex, &mousey);
 278 
 279         // TODO enable after an axis choosing mechanism is implemented
 280 #if 0
 281         std::pair<double, double> values = joystick_manager_.get_mouse_axis_pair();
 282         mousex += values.first * 10;
 283         mousey += values.second * 10;
 284         SDL_WarpMouse(mousex, mousey);
 285 #endif
 286 
 287         scrolling_ = handle_scroll(mousex, mousey, mouse_flags, joystickx, joysticky);
 288 
 289         map_location highlighted_hex = get_display().mouseover_hex();
 290 
 291         // TODO: enable when the relative cursor movement is implemented well enough
 292 #if 0
 293         const map_location& selected_hex = get_display().selected_hex();
 294 
 295         if (selected_hex != map_location::null_location()) {
 296                 if (joystick_manager_.next_highlighted_hex(highlighted_hex, selected_hex)) {
 297                         get_mouse_handler_base().mouse_motion(0,0, true, true, highlighted_hex);
 298                         get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 299                         scrolling_ = true;
 300                 }
 301         } else
 302 #endif
 303 
 304         if(joystick_manager_.update_highlighted_hex(highlighted_hex) && get_display().get_map().on_board(highlighted_hex)) {
 305                 get_mouse_handler_base().mouse_motion(0, 0, true, true, highlighted_hex);
 306                 get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 307                 scrolling_ = true;
 308         }
 309 
 310         // be nice when window is not visible   // NOTE should be handled by display instead, to only disable drawing
 311         sdl::window* window = CVideo::get_singleton().get_window();
 312         if(window != nullptr && is_delay_enabled && (window->get_flags() & SDL_WINDOW_SHOWN) == 0) {
 313                 CVideo::delay(200);
 314         }
 315 
 316         // Scrolling ended, update the cursor and the brightened hex
 317         if(!scrolling_ && was_scrolling) {
 318                 get_mouse_handler_base().mouse_update(is_browsing(), highlighted_hex);
 319         }
 320 }

 322 void controller_base::show_menu(
 323                 const std::vector<config>& items_arg, int xloc, int yloc, bool context_menu, display& disp)
 324 {
 325         hotkey::command_executor* cmd_exec = get_hotkey_command_executor();
 326         if(!cmd_exec) {
 327                 return;
 328         }
 329 
 330         std::vector<config> items;
 331         for(const config& c : items_arg) {
 332                 const std::string& id = c["id"];
 333                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
 334 
 335                 if(cmd_exec->can_execute_command(command) && (!context_menu || in_context_menu(command.id))) {
 336                         items.emplace_back(config{"id", id});
 337                 }
 338         }
 339 
 340         if(items.empty()) {
 341                 return;
 342         }
 343 
 344         cmd_exec->show_menu(items, xloc, yloc, context_menu, disp);
 345 }

 347 void controller_base::execute_action(const std::vector<std::string>& items_arg, int xloc, int yloc, bool context_menu)
 348 {
 349         hotkey::command_executor* cmd_exec = get_hotkey_command_executor();
 350         if(!cmd_exec) {
 351                 return;
 352         }
 353 
 354         std::vector<std::string> items;
 355         for(const std::string& item : items_arg) {
 356                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(item);
 357                 if(cmd_exec->can_execute_command(command)) {
 358                         items.push_back(item);
 359                 }
 360         }
 361 
 362         if(items.empty()) {
 363                 return;
 364         }
 365 
 366         cmd_exec->execute_action(items, xloc, yloc, context_menu, get_display());
 367 }

 374 const config& controller_base::get_theme(const config& game_config, std::string theme_name)
 375 {
 376         if(theme_name.empty()) {
 377                 theme_name = preferences::theme();
 378         }
 379 
 380         if(const config& c = game_config.find_child("theme", "id", theme_name)) {
 381                 return c;
 382         }
 383 
 384         ERR_DP << "Theme '" << theme_name << "' not found. Trying the default theme." << std::endl;
 385 
 386         if(const config& c = game_config.find_child("theme", "id", "Default")) {
 387                 return c;
 388         }
 389 
 390         ERR_DP << "Default theme not found." << std::endl;
 391 
 392         static config empty;
 393         return empty;
 394 }
