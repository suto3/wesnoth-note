  14 /**
  15  *      Some information about savefiles:
  16  *
  17  *      A savefile can contain:
  18  *
  19  *  - General information (toplevel atributes, [multiplayer])
  20  *        This is present in all savefiles
  21  *
  22  *  - [statistics]
  23  *    This is present in all savefiles but it's not handled by playcampaign/play_controller/saved_game.
  24  *    It's handled by savegame.cpp
  25  *
  26  *  - [snapshot]
  27  *    If a savegame was saved during a scenario this contains a snapshot of the game at the point when
  28  *    it was saved.
  29  *
  30  *  - [carryover_sides_start]
  31  *    At start-of-scenario saves this contains data from the previous scenario that was preserved.
  32  *
  33  *  - [carryover_sides]
  34  *    In savefile made during the game, this tag contains data from [carryover_sides_start] that was not
  35  *    used in the current scenario but should be saved for a next scenario
  36  *
  37  *  - [replay_start]
  38  *    A snapshot made very early to replay the game from.
  39  *
  40  *  - [replay]
  41  *    A record of game actions that was made between the creation of [replay_start] and [snapshot].
  42  *
  43  *
  44  * The following types of savegames are known:
  45  *
  46  * - Start of scenario savefiles
  47  *   These files only contain general information, statistics, and [carryover_sides_start]. When these
  48  *   saves are loaded, the scenario data is loaded form the game config using the next_scenario attribute
  49  *   from [carryover_sides_start].
  50  *
  51  * - Expanded Start of scenario savefiles
  52  *   Similar to normal Start-of-scenario savefiles, but the also contain a [scenario] that contains the
  53  *   scenario data. This type is only used internally and usually doesn't get written to the disk.
  54  *
  55  * - In-game savefile
  56  *   These files contain general information, statistics, [snapshot], [replay], [replay_start], [snapshot],
  57  *   and [carryover_sides]. These files don't contain a [carryover_sides_start] because both starting points
  58  *   ([replay_start] and [snapshot]) were made after [carryover_sides_start] was merged into the scenario.
  59  *
  60  * - Replay savefiles
  61  *   Like a in-game save made during linger mode, but without the [snapshot].
  62  */

 177 void saved_game::set_defaults()
 178 {
 179         const bool is_loaded_game = this->starting_pos_type_ != STARTINGPOS_SCENARIO;
 180         const bool is_multiplayer_tag = classification().get_tagname() == "multiplayer";
 181 
 182         static const std::vector<std::string> team_defaults {
 183                 "carryover_percentage",
 184                 "carryover_add",
 185         };
 186 
 187         for(config& side : starting_pos_.child_range("side")) {
 188                 // Set save_id default value directly after loading to its default to prevent different default behaviour in
 189                 // mp_connect code and sp code.
 190                 if(side["save_id"].empty()) {
 191                         side["save_id"] = side["id"];
 192                 }
 193 
 194                 if(!is_multiplayer_tag && side["side_name"].blank()) {
 195                         side["side_name"] = side["name"];
 196                 }
 197 
 198                 if(!is_loaded_game && !side["current_player"].empty()) {
 199                         ERR_NG << "Removed invalid 'current_player' attribute from [side] while loading a scenario. Consider using "
 200                                           "'side_name' instead\n";
 201 
 202                         side["current_player"] = config::attribute_value();
 203                 }
 204 
 205                 // Set some team specific values to their defaults specified in scenario
 206                 for(const std::string& att_name : team_defaults) {
 207                         const config::attribute_value* scenario_value = starting_pos_.get(att_name);
 208                         config::attribute_value& team_value = side[att_name];
 209 
 210                         if(scenario_value && team_value.empty()) {
 211                                 team_value = *scenario_value;
 212                         }
 213                 }
 214         }
 215 }

 217 void saved_game::expand_scenario()
 218 {
 219         if(this->starting_pos_type_ == STARTINGPOS_NONE && !has_carryover_expanded_) {
 220                 game_config_manager::get()->load_game_config_for_game(this->classification());
 221 
 222                 const config& game_config = game_config_manager::get()->game_config();
 223                 const config& scenario =
 224                         game_config.find_child(classification().get_tagname(), "id", carryover_["next_scenario"]);
 225 
 226                 if(scenario) {
 227                         this->starting_pos_type_ = STARTINGPOS_SCENARIO;
 228                         this->starting_pos_ = scenario;
 229 
 230                         // A hash has to be generated using an unmodified scenario data.
 231                         mp_settings_.hash = scenario.hash();
 232 
 233                         // Add addon_id information if it exists.
 234                         if(!scenario["addon_id"].empty() && scenario["require_scenario"].to_bool(false)) {
 235                                 mp_settings_.update_addon_requirements(config {
 236                                         "id", scenario["addon_id"],
 237                                         "version", scenario["addon_version"],
 238                                         "min_version", scenario["addon_min_version"]
 239                                 });
 240                         }
 241 
 242                         update_label();
 243                         set_defaults();
 244                 } else {
 245                         this->starting_pos_type_ = STARTINGPOS_INVALID;
 246                         this->starting_pos_.clear();
 247                 }
 248         }
 249 }

 274 void saved_game::load_mod(const std::string& type, const std::string& id)
 275 {
 276         if(const config& cfg = game_config_manager::get()->game_config().find_child(type, "id", id)) {
 277                 // Note the addon_id if this mod is required to play the game in mp.
 278                 std::string require_attr = "require_" + type;
 279 
 280                 // By default, eras have "require_era = true", and mods have "require_modification = false".
 281                 bool require_default = (type == "era");
 282 
 283                 if(!cfg["addon_id"].empty() && cfg[require_attr].to_bool(require_default)) {
 284                         mp_settings_.update_addon_requirements(config{
 285                                 "id", cfg["addon_id"], "version", cfg["addon_version"], "min_version", cfg["addon_min_version"]});
 286                 }
 287 
 288                 // Copy events
 289                 for(const config& modevent : cfg.child_range("event")) {
 290                         if(modevent["enable_if"].empty()
 291                                 || variable_to_bool(carryover_.child_or_empty("variables"), modevent["enable_if"])
 292                         ) {
 293                                 this->starting_pos_.add_child("event", modevent);
 294                         }
 295                 }
 296 
 297                 // Copy lua
 298                 for(const config& modlua : cfg.child_range("lua")) {
 299                         this->starting_pos_.add_child("lua", modlua);
 300                 }
 301 
 302                 // Copy load_resource
 303                 for(const config& load_resource : cfg.child_range("load_resource")) {
 304                         this->starting_pos_.add_child("load_resource", load_resource);
 305                 }
 306         } else {
 307                 // TODO: A user message instead?
 308                 ERR_NG << "Couldn't find [" << type << "] with id=" << id << std::endl;
 309         }
 310 }
 311 
 312 // Gets the ids of the mp_era and modifications which were set to be active, then fetches these configs from the
 313 // game_config and copies their [event] and [lua] to the starting_pos_.
 314 // At this time, also collect the addon_id attributes which appeared in them and put this list in the addon_ids
 315 // attribute of the mp_settings.
 316 void saved_game::expand_mp_events()
 317 {
 318         expand_scenario();
 319 
 320         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !this->starting_pos_["has_mod_events"].to_bool(false)) {
 321                 std::vector<modevents_entry> mods;
 322                 std::set<std::string> loaded_resources;
 323 
 324                 std::transform(mp_settings_.active_mods.begin(), mp_settings_.active_mods.end(), std::back_inserter(mods),
 325                         [](const std::string& id) { return modevents_entry("modification", id); }
 326                 );
 327 
 328                 // We don't want the error message below if there is no era (= if this is a sp game).
 329                 if(!mp_settings_.mp_era .empty()) {
 330                         mods.emplace_back("era", mp_settings_.mp_era);
 331                 }
 332 
 333                 if(!classification_.campaign.empty()) {
 334                         mods.emplace_back("campaign", classification_.campaign);
 335                 }
 336 
 337                 // In the first iteration mod contains no [resource]s in all other iterations, mods contains only [resource]s.
 338                 do {
 339                         for(modevents_entry& mod : mods) {
 340                                 load_mod(mod.type, mod.id);
 341                         }
 342 
 343                         mods.clear();
 344 
 345                         for(const config& cfg : starting_pos_.child_range("load_resource")) {
 346                                 if(loaded_resources.find(cfg["id"].str()) == loaded_resources.end()) {
 347                                         mods.emplace_back("resource", cfg["id"].str());
 348 
 349                                         loaded_resources.insert(cfg["id"].str());
 350                                 }
 351                         }
 352 
 353                         starting_pos_.clear_children("load_resource");
 354                 } while(!mods.empty());
 355 
 356                 this->starting_pos_["has_mod_events"] = true;
 357         }
 358 }

 360 void saved_game::expand_mp_options()
 361 {
 362         if(starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_) {
 363                 std::vector<modevents_entry> mods;
 364 
 365                 std::transform(mp_settings_.active_mods.begin(), mp_settings_.active_mods.end(), std::back_inserter(mods),
 366                         [](const std::string& id) { return modevents_entry("modification", id); }
 367                 );
 368 
 369                 mods.emplace_back("era", mp_settings_.mp_era);
 370                 mods.emplace_back("multiplayer", get_scenario_id());
 371                 mods.emplace_back("campaign", classification().campaign);
 372 
 373                 config& variables = carryover_.child_or_add("variables");
 374 
 375                 for(modevents_entry& mod : mods) {
 376                         if(const config& cfg = this->mp_settings().options.find_child(mod.type, "id", mod.id)) {
 377                                 // Parse old [option] tag range syntax.
 378                                 for(const config& option : cfg.child_range("option")) {
 379                                         try {
 380                                                 variable_access_create(option["id"], variables).as_scalar() = option["value"];
 381                                         } catch(const invalid_variablename_exception&) {
 382                                                 ERR_NG << "variable " << option["id"] << "cannot be set to " << option["value"] << std::endl;
 383                                         }
 384                                 }
 385 
 386                                 // Parse new [options] id = value syntax.
 387                                 for(const auto& option : cfg.child_or_empty("options").attribute_range()) {
 388                                         try {
 389                                                 variable_access_create(option.first, variables).as_scalar() = option.second;
 390                                         } catch(const invalid_variablename_exception&) {
 391                                                 ERR_NG << "variable " << option.first << "cannot be set to " << option.second << std::endl;
 392                                         }
 393                                 }
 394                         } else {
 395                                 LOG_NG << "Couldn't find [" << mod.type << "] with id=" << mod.id << " for [option]s" << std::endl;
 396                         }
 397                 }
 398         }
 399 }

 401 void saved_game::expand_random_scenario()
 402 {
 403         expand_scenario();
 404 
 405         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO) {
 406                 // If the entire scenario should be randomly generated
 407                 if(!starting_pos_["scenario_generation"].empty()) {
 408                         LOG_NG << "randomly generating scenario...\n";
 409                         const cursor::setter cursor_setter(cursor::WAIT);
 410 
 411                         config scenario_new =
 412                                 random_generate_scenario(starting_pos_["scenario_generation"], starting_pos_.child("generator"));
 413 
 414                         // Preserve "story" form the scenario toplevel.
 415                         for(config& story : starting_pos_.child_range("story")) {
 416                                 scenario_new.add_child("story", story);
 417                         }
 418 
 419                         scenario_new["id"] = starting_pos_["id"];
 420                         starting_pos_ = scenario_new;
 421 
 422                         update_label();
 423                         set_defaults();
 424                 }
 425 
 426                 // If no map_data is provided, try to load the specified file directly
 427                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_file"].empty()) {
 428                         starting_pos_["map_data"] = filesystem::read_map(starting_pos_["map_file"]);
 429                 }
 430 
 431                 // If the map should be randomly generated
 432                 // We dont want that we accidentally to this twice so we check for starting_pos_["map_data"].empty()
 433                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_generation"].empty()) {
 434                         LOG_NG << "randomly generating map...\n";
 435                         const cursor::setter cursor_setter(cursor::WAIT);
 436 
 437                         starting_pos_["map_data"] =
 438                                 random_generate_map(starting_pos_["map_generation"], starting_pos_.child("generator"));
 439                 }
 440         }
 441 }

 443 void saved_game::expand_carryover()
 444 {
 445         expand_scenario();
 446         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_) {
 447                 carryover_info sides(carryover_);
 448 
 449                 sides.transfer_to(get_starting_pos());
 450                 for(config& side_cfg : get_starting_pos().child_range("side")) {
 451                         sides.transfer_all_to(side_cfg);
 452                 }
 453 
 454                 carryover_ = sides.to_config();
 455                 has_carryover_expanded_ = true;
 456         }
 457 }

 493 const config& saved_game::get_replay_starting_pos()
 494 {
 495         if(!replay_start_.empty()) {
 496                 return replay_start_;
 497         }
 498 
 499         if(!has_carryover_expanded_) {
 500                 // Try to load the scenario form game config or from [scenario] if there is no [replay_start]
 501                 expand_scenario();
 502                 expand_carryover();
 503         }
 504 
 505         if(starting_pos_type_ == STARTINGPOS_SCENARIO) {
 506                 return starting_pos_;
 507         }
 508 
 509         return this->replay_start_.child("some_non_existet_invalid");
 510 }

 512 void saved_game::convert_to_start_save()
 513 {
 514         assert(starting_pos_type_ == STARTINGPOS_SNAPSHOT);
 515 
 516         carryover_info sides(starting_pos_, true);
 517 
 518         sides.merge_old_carryover(carryover_info(carryover_));
 519         sides.rng().rotate_random();
 520 
 521         carryover_ = sides.to_config();
 522 
 523         has_carryover_expanded_ = false;
 524 
 525         replay_data_ = replay_recorder_base();
 526         replay_start_.clear();
 527 
 528         remove_snapshot();
 529 }

 531 config saved_game::to_config() const
 532 {
 533         // TODO: remove this code duplication with write_... functions.
 534         config r = classification_.to_config();

 537                 r.add_child("replay_start", replay_start_);

 549         r.add_child("multiplayer", mp_settings_.to_config());
 550 
 551         return r;

 584 void saved_game::cancel_orders()
 585 {
 586         for(config& side : this->starting_pos_.child_range("side")) {
 587                 // for humans "goto_x/y" is used for multi-turn-moves
 588                 // for the ai "goto_x/y" is a way for wml to order the ai to move a unit to a certain place.
 589                 // we want to cancel human order but not to break wml.
 590                 if(side["controller"] != "human" && side["controller"] != "network") {
 591                         continue;
 592                 }
 593 
 594                 for(config& unit : side.child_range("unit")) {
 595                         unit["goto_x"] = -999;
 596                         unit["goto_y"] = -999;
 597                 }
 598         }
 599 }

 601 void saved_game::unify_controllers()
 602 {
 603         for(config& side : this->starting_pos_.child_range("side")) {

 605                         side["controller"] = "human";
 606                 }
 607 
 608                 if(side["controller"] == "network_ai") {
 609                         side["controller"] = "ai";

 620 void saved_game::swap(saved_game& other)
 621 {
 622         carryover_.swap(other.carryover_);
 623 
 624         std::swap(classification_, other.classification_);
 625         std::swap(has_carryover_expanded_, other.has_carryover_expanded_);

 628         replay_data_.swap(other.replay_data_);
 629         replay_start_.swap(other.replay_start_);
 630         starting_pos_.swap(other.starting_pos_);
 631 
 632         std::swap(starting_pos_type_, other.starting_pos_type_);
 633 }

 635 void saved_game::set_data(config& cfg)
 636 {
 637         log_scope("read_game");
 638 
 639         if(config& caryover_sides = cfg.child("carryover_sides")) {
 640                 carryover_.swap(caryover_sides);
 641                 has_carryover_expanded_ = true;
 642         } else if(config& caryover_sides_start = cfg.child("carryover_sides_start")) {
 643                 carryover_.swap(caryover_sides_start);
 644                 has_carryover_expanded_ = false;
 645         } else {
 646                 carryover_.clear();
 647                 has_carryover_expanded_ = false;
 648         }
 649 
 650         if(config& replay_start = cfg.child("replay_start")) {
 651                 replay_start_.swap(replay_start);
 652         } else {
 653                 replay_start_.clear();
 654         }
 655 
 656         replay_data_ = replay_recorder_base();
 657 
 658         // Serversided replays can contain multiple [replay]
 659         for(config& replay : cfg.child_range("replay")) {
 660                 replay_data_.append_config(replay);
 661         }
 662 
 663         replay_data_.set_to_end();
 664 
 665         if(config& snapshot = cfg.child("snapshot")) {
 666                 this->starting_pos_type_ = STARTINGPOS_SNAPSHOT;
 667                 this->starting_pos_.swap(snapshot);
 668         } else if(config& scenario = cfg.child("scenario")) {
 669                 this->starting_pos_type_ = STARTINGPOS_SCENARIO;
 670                 this->starting_pos_.swap(scenario);
 671         } else {
 672                 this->starting_pos_type_ = STARTINGPOS_NONE;
 673                 this->starting_pos_.clear();
 674         }
 675 
 676         LOG_NG << "scenario: '" << carryover_["next_scenario"].str() << "'\n";
 677 
 678         if(const config& stats = cfg.child("statistics")) {
 679                 statistics::fresh_stats();
 680                 statistics::read_stats(stats);
 681         }
 682 
 683         classification_ = game_classification(cfg);
 684         mp_settings_ = mp_game_settings(cfg.child_or_empty("multiplayer"));
 685 
 686         cfg.clear();
 687 }

 689 void saved_game::clear()
 690 {
 691         carryover_.clear();
 692         classification_ = game_classification();
 693         has_carryover_expanded_ = false;
 694         mp_settings_ = mp_game_settings();
 695         replay_data_.swap(replay_recorder_base());
 696         replay_start_.clear();
 697         starting_pos_.clear();
 698         starting_pos_type_ = STARTINGPOS_NONE;
 699 }
