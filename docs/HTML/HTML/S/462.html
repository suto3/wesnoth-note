  81 loading_screen::loading_screen(std::function<void()> f)
  82         : timer_id_(0)
  83         , animation_counter_(0)
  84         , work_(f)
  85         , worker_()
  86         , cursor_setter_()
  87         , progress_stage_label_(nullptr)
  88         , animation_label_(nullptr)
  89         , current_stage_(nullptr)
  90         , visible_stages_()
  91         , animation_stages_()
  92         , current_visible_stage_()
  93         , is_worker_running_(false)
  94 {
  95         for(const auto& pair : stages) {
  96                 visible_stages_[pair.first] = t_string(pair.second, "wesnoth-lib") + "...";
  97         }
  98 
  99         for(int i = 0; i != 20; ++i) {
 100                 std::string s(20, ' ');
 101                 s[i] = '.';
 102                 animation_stages_.push_back(s);
 103         }
 104 
 105         current_visible_stage_ = visible_stages_.end();
 106         current_load = this;
 107 }

 109 void loading_screen::pre_show(window& window)
 110 {
 111         if(work_) {
 112                 worker_.reset(new boost::thread([this]() {
 113                         is_worker_running_ = true;
 114 
 115                         try {
 116                                 work_();
 117                         } catch(...) {
 118                                 // TODO: guard this with a mutex.
 119                                 exception_ = std::current_exception();
 120                         }
 121 
 122                         is_worker_running_ = false;
 123                 }));
 124         }
 125 
 126         timer_id_ = add_timer(100, std::bind(&loading_screen::timer_callback, this, std::ref(window)), true);
 127         cursor_setter_.reset(new cursor::setter(cursor::WAIT));
 128         progress_stage_label_ = find_widget<label>(&window, "status", false, true);
 129         animation_label_ = find_widget<label>(&window, "test_animation", false, true);
 130 
 131         window.set_enter_disabled(true);
 132         window.set_escape_disabled(true);
 133 }
 134 
 135 void loading_screen::post_show(window& /*window*/)
 136 {
 137         worker_.reset();
 138         clear_timer();
 139         cursor_setter_.reset();
 140 }

 142 void loading_screen::progress(const char* stage)
 143 {
 144         if(!current_load) {
 145                 return;
 146         }
 147 
 148         if(stage) {

 160 void loading_screen::timer_callback(window& window)
 161 {
 162         if(!work_ || !worker_ || worker_->timed_join(boost::posix_time::milliseconds(0))) {
 163                 if(exception_) {
 164                         clear_timer();
 165                         std::rethrow_exception(exception_);
 166                 }
 167 
 168                 window.close();
 169         }
 170 
 171         if(!work_) {
 172                 return;
 173         }
 174 
 175         const char* stage = current_stage_
 176 #if defined(_MSC_VER) && _MSC_VER < 1900
 177                 ;
 178 #else
 179                 .load(std::memory_order_acquire);
 180 #endif
 181 
 182         if(stage && (current_visible_stage_ == visible_stages_.end() || stage != current_visible_stage_->first)) {
 183                 auto iter = visible_stages_.find(stage);
 184                 if(iter == visible_stages_.end()) {
 185                         WRN_LS << "Stage ID '" << stage << "' missing description." << std::endl;
 186                         return;
 187                 }
 188 
 189                 current_visible_stage_ = iter;
 190                 progress_stage_label_->set_label(iter->second);
 191         }
 192 
 193         ++animation_counter_;
 194         if(animation_counter_ % 2 == 0) {
 195                 animation_label_->set_label(animation_stages_[(animation_counter_ / 2) % animation_stages_.size()]);
 196         }
 197 }

 199 loading_screen::~loading_screen()
 200 {
 201         /* If the worker thread is running, exit the application to prevent memory corruption.
 202          * TODO: this is still not optimal. The main problem is that this code assumes that this
 203          * happened because the window was closed, which is not necessarily the case (other
 204          * possibilities might be a 'dialog doesn't fit on screen' exception caused by resizing
 205          * the window).
 206          *
 207          * Another approach might be to add exit points (boost::this_thread::interruption_point())
 208          * to the worker functions (filesystem.cpp, config parsing code, etc.) and then use that
 209          * to end the thread faster.
 210          */
 211         if(is_worker_running_) {
 212 #if defined(_MSC_VER) && _MSC_VER <= 1800
 213                 HANDLE process = GetCurrentProcess();
 214                 TerminateProcess(process, 0u);
 215 #elif defined(_LIBCPP_VERSION) || defined(__MINGW32__)
 216                 std::_Exit(0);
 217 #else
 218                 std::quick_exit(0);
 219 #endif
 220         }
 221 
 222         clear_timer();
 223         current_load = nullptr;
 224 }

 226 void loading_screen::display(CVideo& video, std::function<void()> f)
 227 {
 228         const bool use_loadingscreen_animation = !preferences::disable_loadingscreen_animation();
 229 
 230         if(current_load || video.faked()) {
 231                 f();
 232         } else if(use_loadingscreen_animation) {
 233                 loading_screen(f).show(video);
 234         } else {
 235                 loading_screen(std::function<void()>()).show(video);
 236                 f();
 237         }
 238 }

 240 void loading_screen::clear_timer()
 241 {
 242         if(timer_id_ != 0) {
 243                 remove_timer(timer_id_);
 244                 timer_id_ = 0;
 245         }
