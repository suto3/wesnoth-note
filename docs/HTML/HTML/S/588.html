  92 void text_box_base::set_maximum_length(const size_t maximum_length)
  93 {
  94         if(maximum_length == 0) {
  95                 return;
  96         }
  97 
  98         const bool need_update = text_.get_length() > maximum_length;
  99 
 100         text_.set_maximum_length(maximum_length);
 101 
 102         if(need_update) {
 103                 if(selection_start_ > maximum_length) {
 104                         selection_start_ = maximum_length;
 105                         selection_length_ = 0;
 106                 } else if(selection_start_ + selection_length_ > maximum_length) {
 107                         selection_length_ = maximum_length - selection_start_;
 108                 }
 109                 update_canvas();
 110                 set_is_dirty(true);
 111         }
 112 }

 127 void text_box_base::set_cursor(const size_t offset, const bool select)
 128 {
 129         reset_cursor_state();
 130 
 131         if(select) {
 132 
 133                 if(selection_start_ == offset) {
 134                         selection_length_ = 0;
 135                 } else {
 136                         selection_length_ = -static_cast<int>(selection_start_ - offset);
 137                 }
 138 
 139 #ifdef __unix__
 140                 // selecting copies on UNIX systems.
 141                 copy_selection(true);
 142 #endif
 143                 update_canvas();
 144                 set_is_dirty(true);
 145 
 146         } else {
 147                 assert(offset <= text_.get_length());
 148                 selection_start_ = offset;
 149                 selection_length_ = 0;
 150 
 151                 update_canvas();
 152                 set_is_dirty(true);
 153         }
 154 }

 178 void text_box_base::copy_selection(const bool mouse)
 179 {
 180         if(selection_length_ == 0) {
 181                 return;
 182         }
 183 
 184         unsigned end, start = selection_start_;
 185         const utf8::string txt = text_.text();
 186 
 187         if(selection_length_ > 0) {
 188                 end = utf8::index(txt, start + selection_length_);
 189                 start = utf8::index(txt, start);
 190         } else {
 191                 // inverse selection: selection_start_ is in fact the end
 192                 end = utf8::index(txt, start);
 193                 start = utf8::index(txt, start + selection_length_);
 194         }
 195         desktop::clipboard::copy_to_clipboard(txt.substr(start, end - start), mouse);
 196 }

 230 void text_box_base::set_selection(size_t start, int length)
 231 {
 232         const size_t text_size = text_.get_length();
 233 
 234         if(start >= text_size) {
 235                 start = text_size;
 236         }
 237 
 238         if(length == 0) {
 239                 set_cursor(start, false);
 240                 return;
 241         }
 242 
 243         // The text pos/size type differs in both signedness and size with the
 244         // selection length. Such is life.
 245         const int sel_start = std::min<size_t>(start, std::numeric_limits<int>::max());
 246         const int sel_max_length = std::min<size_t>(text_size - start, std::numeric_limits<int>::max());
 247 
 248         const bool backwards = length < 0;
 249 
 250         if(backwards && -length > sel_start) {
 251                 length = -sel_start;
 252         } else if(!backwards && length > sel_max_length) {
 253                 length = sel_max_length;
 254         }
 255 
 256         set_selection_start(start);
 257         set_selection_length(length);
 258 
 259         update_canvas();
 260 }

 270 void text_box_base::toggle_cursor_timer(bool enable)
 271 {
 272         if(!cursor_blink_rate_ms_) {
 273                 return;
 274         }
 275 
 276         if(cursor_timer_) {
 277                 remove_timer(cursor_timer_);
 278         }
 279 
 280         cursor_timer_ = enable
 281                         ? add_timer(cursor_blink_rate_ms_, std::bind(&text_box_base::cursor_timer_callback, this), true)
 282                         : 0;
 283 }

 285 void text_box_base::cursor_timer_callback()
 286 {
 287         unsigned was_alpha = cursor_alpha_;
 288         switch(state_) {
 289                 case DISABLED:
 290                         cursor_alpha_ = 0;
 291                         return;
 292                 case ENABLED:
 293                         cursor_alpha_ = 255;
 294                         return;
 295                 default:
 296                         if(get_window() != open_window_stack.back()) {
 297                                 cursor_alpha_ = 0;
 298                         } else {
 299                                 cursor_alpha_ = (~cursor_alpha_) & 0xFF;
 300                         }
 301         }
 302 
 303         if(was_alpha == cursor_alpha_) {
 304                 return;
 305         }
 306 
 307         for(auto& tmp : get_canvases()) {
 308                 tmp.set_variable("cursor_alpha", wfl::variant(cursor_alpha_));
 309         }
 310 
 311         set_is_dirty(true);
 312 }

 404 void text_box_base::handle_commit(bool& handled, const utf8::string& unicode)
 405 {
 406         DBG_GUI_E << LOG_SCOPE_HEADER << '\n';
 407 
 408         if(unicode.size() > 1 || unicode[0] != 0) {
 409                 handled = true;
 410                 if(ime_in_progress_) {
 411                         selection_start_ = ime_start_point_;
 412                         selection_length_ = ime_length_;
 413                         ime_in_progress_ = false;
 414                         ime_length_ = 0;
 415                 }
 416                 insert_char(unicode);
 417                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 418 
 419                 if(text_changed_callback_) {
 420                         text_changed_callback_(this, this->text());
 421                 }
 422         }
 423 }

 425 void text_box_base::handle_editing(bool& handled, const utf8::string& unicode, int32_t start, int32_t len)
 426 {
 427         if(unicode.size() > 1 || unicode[0] != 0) {
 428                 handled = true;
 429                 size_t new_len = utf8::size(unicode);
 430                 if(!ime_in_progress_) {
 431                         ime_in_progress_ = true;
 432                         delete_selection();
 433                         ime_start_point_ = selection_start_;
 434                         text_cached_ = text_.text();
 435                         SDL_Rect rect = get_rectangle();
 436                         if(new_len > 0) {
 437                                 rect.x += get_cursor_position(ime_start_point_).x;
 438                                 rect.w = get_cursor_position(ime_start_point_ + new_len).x - rect.x;
 439                         } else {
 440                                 rect.x += get_cursor_position(ime_start_point_ + new_len).x;
 441                                 rect.w = get_cursor_position(ime_start_point_).x - rect.x;
 442                         }
 443                         SDL_SetTextInputRect(&rect);
 444                 }
 445                 ime_cursor_ = start;
 446                 ime_length_ = new_len;
 447                 std::string new_text(text_cached_);
 448                 utf8::insert(new_text, ime_start_point_, unicode);
 449                 text_.set_text(new_text, false);
 450 
 451                 // Update status
 452                 set_cursor(ime_start_point_ + ime_cursor_, false);
 453                 if(len > 0) {
 454                         set_cursor(ime_start_point_ + ime_cursor_ + len, true);
 455                 }
 456                 update_canvas();
 457                 set_is_dirty(true);
 458         }
 459 }

 471 void text_box_base::signal_handler_sdl_key_down(const event::ui_event event,
 472                                                                                  bool& handled,
 473                                                                                  const SDL_Keycode key,
 474                                                                                  SDL_Keymod modifier)
 475 {
 476 
 477         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 478 
 479 // For copy/paste we use a different key on the MAC. Other ctrl modifiers won't
 480 // be modifed seems not to be required when I read the comment in
 481 // widgets/textbox.cpp:516. Would be nice if somebody on a MAC would test it.
 482 #ifdef __APPLE__
 483         const unsigned copypaste_modifier = KMOD_LGUI | KMOD_RGUI;
 484 #else
 485         const unsigned copypaste_modifier = KMOD_CTRL;
 486 #endif
 487 
 488         switch(key) {
 489 
 490                 case SDLK_LEFT:
 491                         handle_key_left_arrow(modifier, handled);
 492                         break;
 493 
 494                 case SDLK_RIGHT:
 495                         handle_key_right_arrow(modifier, handled);
 496                         break;
 497 
 498                 case SDLK_UP:
 499                         handle_key_up_arrow(modifier, handled);
 500                         break;
 501 
 502                 case SDLK_DOWN:
 503                         handle_key_down_arrow(modifier, handled);
 504                         break;
 505 
 506                 case SDLK_PAGEUP:
 507                         handle_key_page_up(modifier, handled);
 508                         break;
 509 
 510                 case SDLK_PAGEDOWN:
 511                         handle_key_page_down(modifier, handled);
 512                         break;
 513 
 514                 case SDLK_a:
 515                         if(!(modifier & KMOD_CTRL)) {
 516                                 return;
 517                         }
 518 
 519                         select_all();
 520                         break;
 521 
 522                 case SDLK_HOME:
 523                         handle_key_home(modifier, handled);
 524                         break;
 525 
 526                 case SDLK_END:
 527                         handle_key_end(modifier, handled);
 528                         break;
 529 
 530                 case SDLK_BACKSPACE:
 531                         handle_key_backspace(modifier, handled);
 532                         break;
 533 
 534                 case SDLK_DELETE:
 535                         handle_key_delete(modifier, handled);
 536                         break;
 537 
 538                 case SDLK_c:
 539                         if(!(modifier & copypaste_modifier)) {
 540                                 return;
 541                         }
 542 
 543                         // atm we don't care whether there is something to copy or paste
 544                         // if nothing is there we still don't want to be chained.
 545                         copy_selection(false);
 546                         handled = true;
 547                         break;
 548 
 549                 case SDLK_x:
 550                         if(!(modifier & copypaste_modifier)) {
 551                                 return;
 552                         }
 553 
 554                         copy_selection(false);
 555                         delete_selection();
 556                         handled = true;
 557                         break;
 558 
 559                 case SDLK_v:
 560                         if(!(modifier & copypaste_modifier)) {
 561                                 return;
 562                         }
 563 
 564                         paste_selection(false);
 565                         handled = true;
 566                         break;
 567 
 568                 case SDLK_RETURN:
 569                 case SDLK_KP_ENTER:
 570                         if(!ime_in_progress_ || (modifier & (KMOD_CTRL | KMOD_ALT | KMOD_GUI | KMOD_SHIFT))) {
 571                                 return;
 572                         }
 573                         // The IME will handle it, we just need to make sure nothing else handles it too.
 574                         handled = true;
 575                         break;
 576                 
 577                 case SDLK_ESCAPE:
 578                         if(!ime_in_progress_ || (modifier & (KMOD_CTRL | KMOD_ALT | KMOD_GUI | KMOD_SHIFT))) {
 579                                 return;
 580                         }
 581                         interrupt_composition();
 582                         handled = true;
 583                         break;
 584                 
 585                 default:
 586                         // Don't call the text changed callback if nothing happened.
 587                         return;
 588         }
 589 
 590         if(text_changed_callback_) {
 591                 text_changed_callback_(this, this->text());
 592         }
 593 }
