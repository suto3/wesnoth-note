 176         std::vector<const unit *> all_matches_on_map(const map_location* loc = nullptr, const unit* other_unit = nullptr) const
 177         {
 178                 std::vector<const unit *> ret;
 179                 int max_matches = max_matches_;
 180                 
 181                 for (const unit & u : fc_->get_disp_context().units()) {
 182                         if (impl_.matches(unit_filter_impl::unit_filter_args{u, loc ? *loc : u.get_location(), other_unit, fc_, use_flat_tod_})) {
 183                                 if(max_matches == 0) {
 184                                         return ret;
 185                                 }
 186                                 --max_matches;
 187                                 ret.push_back(&u);
 188                         }
 189                 }
 190                 return ret;
 191         }

 205         unit_const_ptr first_match_on_map() const {
 206                 const unit_map & units = fc_->get_disp_context().units();
 207                 for(unit_map::const_iterator u = units.begin(); u != units.end(); u++) {
 208                         if (matches(*u, u->get_location())) {
 209                                 return u.get_shared_ptr();
 210                         }
 211                 }
 212                 return unit_const_ptr();
 213         }
