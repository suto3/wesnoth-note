  25 /**
  26  * Our definition of map labels being obscured is if the tile is obscured,
  27  * or the tile below is obscured. This is because in the case where the tile
  28  * itself is visible, but the tile below is obscured, the bottom half of the
  29  * tile will still be shrouded, and the label being drawn looks weird.
  30  */
  31 inline bool is_shrouded(const display* disp, const map_location& loc)
  32 {
  33         return disp->shrouded(loc) || disp->shrouded(loc.get_direction(map_location::SOUTH));
  34 }

  41 inline bool is_fogged(const display* disp, const map_location& loc)

 146 const terrain_label* map_labels::set_label(const map_location& loc,
 147                 const t_string& text,
 148                 const int creator,
 149                 const std::string& team_name,
 150                 const color_t color,
 151                 const bool visible_in_fog,
 152                 const bool visible_in_shroud,
 153                 const bool immutable,
 154                 const std::string& category,
 155                 const t_string& tooltip)
 156 {
 157         terrain_label* res = nullptr;
 158 
 159         // See if there is already a label in this location for this team.
 160         // (We do not use get_label_private() here because we might need
 161         // the label_map as well as the terrain_label.)
 162         team_label_map::iterator current_label_map = labels_.find(team_name);
 163         label_map::iterator current_label;
 164 
 165         if(current_label_map != labels_.end() &&
 166                 (current_label = current_label_map->second.find(loc)) != current_label_map->second.end())
 167         {
 168                 // Found old checking if need to erase it
 169                 if(text.str().empty()) {
 170                         // Erase the old label.
 171                         current_label_map->second.erase(current_label);
 172 
 173                         // Restore the global label in the same spot, if any.
 174                         if(terrain_label* global_label = get_label_private(loc, "")) {
 175                                 global_label->recalculate();
 176                         }
 177                 } else {
 178                         current_label->second.update_info(
 179                                 text, creator, tooltip, team_name, color, visible_in_fog, visible_in_shroud, immutable, category);
 180 
 181                         res = &current_label->second;
 182                 }
 183         } else if(!text.str().empty()) {
 184                 // See if we will be replacing a global label.
 185                 terrain_label* global_label = get_label_private(loc, "");
 186 
 187                 // Add the new label.
 188                 res = add_label(
 189                         *this, text, creator, team_name, loc, color, visible_in_fog, visible_in_shroud, immutable, category, tooltip);
 190 
 191                 // Hide the old label.
 192                 if(global_label != nullptr) {
 193                         global_label->recalculate();
 194                 }
 195         }
 196 
 197         categories_dirty = true;
 198         return res;
 199 }

 225 void map_labels::clear_map(label_map& m, bool force)
 226 {
 227         label_map::iterator i = m.begin();
 228         while(i != m.end()) {
 229                 if(!i->second.immutable() || force) {
 230                         m.erase(i++);
 231                 } else {
 232                         ++i;
 233                 }
 234         }
 235 
 236         categories_dirty = true;
 237 }

 281 const std::vector<std::string>& map_labels::all_categories() const
 282 {
 283         if(categories_dirty) {
 284                 categories_dirty = false;
 285                 categories.clear();
 286                 categories.push_back("team");
 287 
 288                 for(size_t i = 1; i <= resources::gameboard->teams().size(); i++) {
 289                         categories.push_back("side:" + std::to_string(i));
 290                 }
 291 
 292                 std::set<std::string> unique_cats;
 293                 for(const auto& m : labels_) {
 294                         for(const auto& l : m.second) {
 295                                 if(l.second.category().empty()) {
 296                                         continue;
 297                                 }
 298 
 299                                 unique_cats.insert("cat:" + l.second.category());
 300                         }
 301                 }
 302 
 303                 std::copy(unique_cats.begin(), unique_cats.end(), std::back_inserter(categories));
 304         }
 305 
 306         return categories;
 307 }

 379 void terrain_label::read(const config& cfg)
 380 {
 381         const variable_set& vs = *resources::gamedata;
 382 
 383         loc_ = map_location(cfg, &vs);
 384         color_t color = font::LABEL_COLOR;
 385 
 386         std::string tmp_color = cfg["color"];
 387 
 388         text_ = cfg["text"];
 389         tooltip_ = cfg["tooltip"];
 390         team_name_ = cfg["team_name"].str();
 391         visible_in_fog_ = cfg["visible_in_fog"].to_bool(true);
 392         visible_in_shroud_ = cfg["visible_in_shroud"].to_bool();
 393         immutable_ = cfg["immutable"].to_bool(true);
 394         category_ = cfg["category"].str();
 395 
 396         int side = cfg["side"].to_int(-1);
 397         if(side >= 0) {
 398                 creator_ = side - 1;
 399         } else if(cfg["side"].str() == "current") {
 400                 config::attribute_value current_side = vs.get_variable_const("side_number");
 401                 if(!current_side.empty()) {
 402                         creator_ = current_side.to_int();
 403                 }
 404         }
 405 
 406         // Not moved to rendering, as that would depend on variables at render-time
 407         text_ = utils::interpolate_variables_into_tstring(text_, vs);
 408 
 409         team_name_ = utils::interpolate_variables_into_string(team_name_, vs);
 410         tmp_color = utils::interpolate_variables_into_string(tmp_color, vs);
 411 
 412         if(!tmp_color.empty()) {
 413                 try {
 414                         color = color_t::from_rgb_string(tmp_color);
 415                 } catch(std::invalid_argument&) {
 416                         // Prior to the color_t conversion, labels were written to savefiles with an alpha key, despite alpha not
 417                         // being accepted in color=. Because of this, this enables the loading of older saves without an exception
 418                         // throwing.
 419                         color = color_t::from_rgba_string(tmp_color);
 420                 }
 421         }
 422 
 423         color_ = color;
 424 }

 479 void terrain_label::calculate_shroud()
 480 {
 481         if(handle_) {
 482                 font::show_floating_label(handle_, !hidden());
 483         }
 484 
 485         if(tooltip_.empty() || hidden()) {
 486                 tooltips::remove_tooltip(tooltip_handle_);
 487                 tooltip_handle_ = 0;
 488                 return;
 489         }
 490 
 491         // tooltips::update_tooltip(tooltip_handle, get_rect(), tooltip_.str(), "", true);
 492 
 493         if(tooltip_handle_) {
 494                 tooltips::update_tooltip(tooltip_handle_, get_rect(), tooltip_.str(), "", true);
 495         } else {
 496                 tooltip_handle_ = tooltips::add_tooltip(get_rect(), tooltip_.str());
 497         }
 498 }

 500 SDL_Rect terrain_label::get_rect() const
 501 {
 502         SDL_Rect rect {0, 0, 0, 0};
 503 
 504         display* disp = display::get_singleton();
 505         if(!disp) {
 506                 return rect;
 507         }
 508 
 509         int hex_size = disp->hex_size();
 510 
 511         rect.x = disp->get_location_x(loc_) + hex_size / 4;
 512         rect.y = disp->get_location_y(loc_);
 513         rect.h = disp->hex_size();
 514         rect.w = disp->hex_size() - hex_size / 2;
 515 
 516         return rect;
 517 }

 519 void terrain_label::draw()
 520 {
 521         display* disp = display::get_singleton();
 522         if(!disp) {
 523                 return;
 524         }
 525 
 526         if(text_.empty() && tooltip_.empty()) {
 527                 return;
 528         }
 529 
 530         clear();
 531 
 532         if(!viewable(disp->get_disp_context())) {
 533                 return;
 534         }
 535 
 536         // Note: the y part of loc_nextx is not used at all.
 537         const map_location loc_nextx = loc_.get_direction(map_location::NORTH_EAST);
 538         const map_location loc_nexty = loc_.get_direction(map_location::SOUTH);
 539         const int xloc = (disp->get_location_x(loc_) + disp->get_location_x(loc_nextx) * 2) / 3;
 540         const int yloc = disp->get_location_y(loc_nexty) - font::SIZE_NORMAL;
 541 
 542         // If a color is specified don't allow to override it with markup. (prevents faking map labels for example)
 543         // FIXME: @todo Better detect if it's team label and not provided by the scenario.
 544         bool use_markup = color_ == font::LABEL_COLOR;
 545 
 546         font::floating_label flabel(text_.str());
 547         flabel.set_color(color_);
 548         flabel.set_position(xloc, yloc);
 549         flabel.set_clip_rect(disp->map_outside_area());
 550         flabel.set_width(font::SIZE_NORMAL * 13);
 551         flabel.set_height(font::SIZE_NORMAL * 4);
 552         flabel.set_scroll_mode(font::ANCHOR_LABEL_MAP);
 553         flabel.use_markup(use_markup);
 554 
 555         handle_ = font::add_floating_label(flabel);
 556 
 557         calculate_shroud();
 558 }

 560 /**
 561  * This is a lightweight test used to see if labels are revealed as a result
 562  * of unit actions (i.e. fog/shroud clearing). It should not contain any tests
 563  * that are invariant during unit movement (disregarding potential WML events);
 564  * those belong in visible().
 565  */
 566 bool terrain_label::hidden() const
 567 {
 568         display* disp = display::get_singleton();
 569         if(!disp) {
 570                 return false;
 571         }
 572 
 573         // Respect user's label preferences
 574         std::string category = "cat:" + category_;
 575         std::string creator = "side:" + std::to_string(creator_ + 1);
 576         const std::vector<std::string>& hidden_categories = disp->get_disp_context().hidden_label_categories();
 577 
 578         if(std::find(hidden_categories.begin(), hidden_categories.end(), category) != hidden_categories.end()) {
 579                 return true;
 580         }
 581 
 582         if(creator_ >= 0 &&
 583                 std::find(hidden_categories.begin(), hidden_categories.end(), creator) != hidden_categories.end())
 584         {
 585                 return true;
 586         }
 587 
 588         if(!team_name().empty() &&
 589                 std::find(hidden_categories.begin(), hidden_categories.end(), "team") != hidden_categories.end())
 590         {
 591                 return true;
 592         }
 593 
 594         // Fog can hide some labels.
 595         if(!visible_in_fog_ && is_fogged(disp, loc_)) {
 596                 return true;
 597         }
 598 
 599         // Shroud can hide some labels.
 600         if(!visible_in_shroud_ && is_shrouded(disp, loc_)) {
 601                 return true;
 602         }
 603 
 604         return false;
 605 }

 607 /**
 608  * This is a test used to see if we should bother with the overhead of actually
 609  * creating a label. Conditions that can change during unit movement (disregarding
 610  * potential WML events) should not be listed here; they belong in hidden().
 611  */
 612 bool terrain_label::viewable(const display_context& dc) const
 613 {
 614         if(!parent_->enabled()) {
 615                 return false;
 616         }
 617 
 618         // In the editor, all labels are viewable.
 619         if(dc.teams().empty()) {
 620                 return true;
 621         }
 622 
 623         // Observers are not privvy to team labels.
 624         const bool can_see_team_labels = !dc.is_observer();
 625 
 626         // Global labels are shown unless covered by a team label.
 627         if(team_name_.empty()) {
 628                 return !can_see_team_labels || parent_->visible_global_label(loc_);
 629         }
 630 
 631         // Team labels are only shown to members of the team.
 632         return can_see_team_labels && parent_->team_name() == team_name_;
 633 }
