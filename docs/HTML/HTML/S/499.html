 163 /**
 164  * Shows a message to the user.
 165  *
 166  * Normally the dialog won't have a button only when the text doesn't fit in
 167  * the dialog and a scrollbar is used the button will be shown.
 168  *
 169  * @param video               The video which contains the surface to draw upon.
 170  * @param title               The title of the dialog.
 171  * @param message             The message to show in the dialog.
 172  * @param button_caption      The caption of the close button.
 173  * @param auto_close          When true the window will hide the ok button
 174  *                            when the message doesn't need a scrollbar to
 175  *                            show itself.
 176  * @param message_use_markup  Use markup for the message?
 177  * @param title_use_markup    Use markup for the title?
 178  */

 185                                   const bool title_use_markup = false);
