  58 struct animation_cursor
  59 {
  60         animation_cursor(const config& cfg)
  61                 : itors(cfg.all_children_range()), branches(1), parent(nullptr)
  62         {
  63                 branches.back().attributes.merge_attributes(cfg);
  64         }
  65 
  66         animation_cursor(const config&cfg, animation_cursor *p)
  67                 : itors(cfg.all_children_range()), branches(p->branches), parent(p)
  68         {
  69                 // If similar 'if' condition in parent branches, we need to
  70                 // cull the branches where there are partial matches.
  71                 // Hence the need to check if the condition has come up before.
  72                 // Also, the attributes are merged here between branches.
  73                 bool previously_hits_set      = false;
  74                 bool previously_direction_set = false;
  75                 bool previously_terrain_set   = false;
  76                 bool previously_value_set     = false;
  77                 bool previously_value_2nd_set = false;
  78 
  79                 const std::string s_cfg_hits      = cfg["hits"];
  80                 const std::string s_cfg_direction = cfg["direction"];
  81                 const std::string s_cfg_terrain   = cfg["terrain_types"];
  82                 const std::string s_cfg_value     = cfg["value"];
  83                 const std::string s_cfg_value_2nd = cfg["value_2nd"];
  84 
  85                 for(const auto& branch : branches) {
  86                         const std::string s_branch_hits      = branch.attributes["hits"];
  87                         const std::string s_branch_direction = branch.attributes["direction"];
  88                         const std::string s_branch_terrain   = branch.attributes["terrain_types"];
  89                         const std::string s_branch_value     = branch.attributes["value"];
  90                         const std::string s_branch_value_2nd = branch.attributes["value_second"];
  91 
  92                         if(!s_branch_hits.empty() && s_branch_hits == s_cfg_hits) {
  93                                 previously_hits_set = true;
  94                         }
  95 
  96                         if(!s_branch_direction.empty() && s_branch_direction == s_cfg_direction) {
  97                                 previously_direction_set = true;
  98                         }
  99 
 100                         if(!s_branch_terrain.empty() && s_branch_terrain == s_cfg_terrain) {
 101                                 previously_terrain_set = true;
 102                         }
 103 
 104                         if(!s_branch_value.empty() && s_branch_value == s_cfg_value) {
 105                                 previously_value_set = true;
 106                         }
 107 
 108                         if(!s_branch_value_2nd.empty() && s_branch_value_2nd == s_cfg_value_2nd) {
 109                                 previously_value_2nd_set = true;
 110                         }
 111                 }
 112 
 113                 // Merge all frames that have new matches and prune any impossible
 114                 // matches, e.g. hits='yes' and hits='no'
 115                 for(auto iter = branches.begin(); iter != branches.end(); /* nothing */) {
 116                         const std::string s_branch_hits      = (*iter).attributes["hits"];
 117                         const std::string s_branch_direction = (*iter).attributes["direction"];
 118                         const std::string s_branch_terrain   = (*iter).attributes["terrain_types"];
 119                         const std::string s_branch_value     = (*iter).attributes["value"];
 120                         const std::string s_branch_value_2nd = (*iter).attributes["value_second"];
 121 
 122                         const bool hits_match      = (previously_hits_set && s_branch_hits != s_cfg_hits);
 123                         const bool direction_match = (previously_direction_set && s_branch_direction != s_cfg_direction);
 124                         const bool terrain_match   = (previously_terrain_set && s_branch_terrain != s_cfg_terrain);
 125                         const bool value_match     = (previously_value_set && s_branch_value != s_cfg_value);
 126                         const bool value_2nd_match = (previously_value_2nd_set && s_branch_value_2nd != s_cfg_value_2nd);
 127 
 128                         if((!previously_hits_set      || hits_match) &&
 129                            (!previously_direction_set || direction_match) &&
 130                            (!previously_terrain_set   || terrain_match) &&
 131                            (!previously_value_set     || value_match)  &&
 132                            (!previously_value_2nd_set || value_2nd_match) &&
 133                             (hits_match || direction_match || terrain_match || value_match || value_2nd_match))
 134                         {
 135                                 branches.erase(iter++);
 136                         } else {
 137                                 (*iter).attributes.merge_attributes(cfg);
 138                                 ++iter;
 139                         }
 140                 }
 141 
 142                 // Then we prune all parent branches with similar matches as they
 143                 // now will not have the full frame list
 144                 for(auto iter = parent->branches.begin(); iter != parent->branches.end(); /* nothing */) {
 145                         const std::string s_branch_hits      = (*iter).attributes["hits"];
 146                         const std::string s_branch_direction = (*iter).attributes["direction"];
 147                         const std::string s_branch_terrain   = (*iter).attributes["terrain_types"];
 148                         const std::string s_branch_value     = (*iter).attributes["value"];
 149                         const std::string s_branch_value_2nd = (*iter).attributes["value_second"];
 150 
 151                         const bool hits_match      = (previously_hits_set && s_branch_hits == s_cfg_hits);
 152                         const bool direction_match = (previously_direction_set && s_branch_direction == s_cfg_direction);
 153                         const bool terrain_match   = (previously_terrain_set && s_branch_terrain == s_cfg_terrain);
 154                         const bool value_match     = (previously_value_set && s_branch_value == s_cfg_value);
 155                         const bool value_2nd_match = (previously_value_2nd_set && s_branch_value_2nd == s_cfg_value_2nd);
 156 
 157                         if((!previously_hits_set        || hits_match) &&
 158                              (!previously_direction_set || direction_match) &&
 159                              (!previously_terrain_set   || terrain_match) &&
 160                              (!previously_value_set     || value_match) &&
 161                              (!previously_value_2nd_set || value_2nd_match) &&
 162                              (hits_match || direction_match || terrain_match || value_match || value_2nd_match))
 163                         {
 164                                 parent->branches.erase(iter++);
 165                         } else {
 166                                 ++iter;
 167                         }
 168                 }
 169         }

 174         animation_cursor* parent;

 177 static void prepare_single_animation(const config& anim_cfg, animation_branches& expanded_anims)
 178 {
 179         /* The anim_cursor holds the current parsing through the config and the branches hold the data
 180          * that will be interpreted as the actual animation. The branches store the config attributes
 181          * for each block and the children of those branches make up all the 'frame', 'missile_frame',
 182          * etc. individually (so 2 instances of 'frame' would be stored as 2 children).
 183          */
 184         std::list<animation_cursor> anim_cursors;
 185         anim_cursors.push_back(animation_cursor(anim_cfg));
 186 
 187         while(!anim_cursors.empty()) {
 188                 animation_cursor& ac = anim_cursors.back();
 189 
 190                 // Reached end of sub-tag config block
 191                 if(ac.itors.empty()) {
 192                         if(!ac.parent) break;
 193 
 194                         // Merge all the current branches into the parent.
 195                         ac.parent->branches.splice(ac.parent->branches.end(), ac.branches);
 196                         anim_cursors.pop_back();
 197                         continue;
 198                 }
 199 
 200                 if(ac.itors.front().key != "if") {
 201                         // Append current config object to all the branches in scope.
 202                         for(animation_branch &ab : ac.branches) {
 203                                 ab.children.push_back(ac.itors.begin());
 204                         }
 205 
 206                         ac.itors.pop_front();
 207                         continue;
 208                 }
 209 
 210                 int count = 0;
 211                 do {
 212                         // Copies the current branches to each cursor created for the conditional clauses.
 213                         // Merge attributes of the clause into them.
 214                         anim_cursors.push_back(animation_cursor(ac.itors.front().cfg, &ac));
 215                         ac.itors.pop_front();
 216                         ++count;
 217                 } while (!ac.itors.empty() && ac.itors.front().key == "else");
 218 
 219                 if(count > 1) {
 220                         // When else statements present, clear all branches before 'if'
 221                         ac.branches.clear();
 222                 }
 223         }
 224 
 225 #if 0
 226         // Debug aid
 227         for(animation_branch& ab : anim_cursors.back().branches) {
 228                 std::cout << "--branch--\n" << ab.attributes;
 229                 for(config::all_children_iterator &ci : ab.children) {
 230                         std::cout << "--branchcfg--\n" << ci->cfg;
 231                 }
 232                 std::cout << "\n";
 233         }
 234 #endif
 235 
 236         // Create the config object describing each branch.
 237         assert(anim_cursors.size() == 1);
 238         animation_cursor& ac = anim_cursors.back();
 239         expanded_anims.splice(expanded_anims.end(), ac.branches, ac.branches.begin(), ac.branches.end());
 240 }

 376 int unit_animation::matches(const display& disp, const map_location& loc, const map_location& second_loc,
 377                 const unit* my_unit, const std::string& event, const int value, hit_type hit, const_attack_ptr attack,
 378                 const_attack_ptr second_attack, int value2) const
 379 {
 380         int result = base_score_;
 381 
 382         if(!event.empty() && !event_.empty()) {
 383                 if(std::find(event_.begin(), event_.end(), event) == event_.end()) {
 384                         return MATCH_FAIL;
 385                 }
 386 
 387                 result++;
 388         }
 389 
 390         if(!terrain_types_.empty()) {
 391                 if(!t_translation::terrain_matches(disp.get_map().get_terrain(loc), terrain_types_)) {
 392                         return MATCH_FAIL;
 393                 }
 394 
 395                 result++;
 396         }
 397 
 398         if(!value_.empty()) {
 399                 if(std::find(value_.begin(), value_.end(), value) == value_.end()) {
 400                         return MATCH_FAIL;
 401                 }
 402 
 403                 result++;
 404         }
 405 
 406         if(my_unit) {
 407                 if(!directions_.empty()) {
 408                         if(std::find(directions_.begin(), directions_.end(), my_unit->facing()) == directions_.end()) {
 409                                 return MATCH_FAIL;
 410                         }
 411 
 412                         result++;
 413                 }
 414 
 415                 for(const auto& filter : unit_filter_) {
 416                         unit_filter f(vconfig(filter), &disp);
 417                         if(!f(*my_unit, loc)) return MATCH_FAIL;
 418                         ++result;
 419                 }
 420 
 421                 if(!secondary_unit_filter_.empty()) {
 422                         unit_map::const_iterator unit = disp.get_units().find(second_loc);
 423                         if(!unit.valid()) {
 424                                 return MATCH_FAIL;
 425                         }
 426 
 427                         for(const config& c : secondary_unit_filter_) {
 428                                 unit_filter f(vconfig(c), &disp);
 429                                 if(!f(*unit, second_loc)) return MATCH_FAIL;
 430                                 result++;
 431                         }
 432                 }
 433         } else if(!unit_filter_.empty()) {
 434                 return MATCH_FAIL;
 435         }
 436 
 437         if(frequency_ && !(rand()%frequency_)) {
 438                 return MATCH_FAIL;
 439         }
 440 
 441         if(!hits_.empty()) {
 442                 if(std::find(hits_.begin(),hits_.end(),hit) == hits_.end()) {
 443                         return MATCH_FAIL;
 444                 }
 445 
 446                 result ++;
 447         }
 448 
 449         if(!value2_.empty()) {
 450                 if(std::find(value2_.begin(),value2_.end(),value2) == value2_.end()) {
 451                         return MATCH_FAIL;
 452                 }
 453 
 454                 result ++;
 455         }
 456 
 457         if(!attack) {
 458                 if(!primary_attack_filter_.empty()) {
 459                         return MATCH_FAIL;
 460                 }
 461         }
 462 
 463         for(const auto& iter : primary_attack_filter_) {
 464                 if(!attack->matches_filter(iter)) return MATCH_FAIL;
 465                 result++;
 466         }
 467 
 468         if(!second_attack) {
 469                 if(!secondary_attack_filter_.empty()) {
 470                         return MATCH_FAIL;
 471                 }
 472         }
 473 
 474         for(const auto& iter : secondary_attack_filter_) {
 475                 if(!second_attack->matches_filter(iter)) return MATCH_FAIL;
 476                 result++;
 477         }
 478 
 479         return result;
 480 }

1105 bool unit_animation::invalidate(frame_parameters& value)
1106 {
1107         if(invalidated_) return false;
1108 
1109         display* disp = display::get_singleton();
1110         const bool complete_redraw = disp->tile_nearly_on_screen(src_) || disp->tile_nearly_on_screen(dst_);
1111 
1112         if(overlaped_hex_.empty()) {
1113                 if(complete_redraw) {
1114                         value.primary_frame = true;
1115                         overlaped_hex_ = unit_anim_.get_overlaped_hex(value, src_, dst_);
1116                         value.primary_frame = false;
1117 
1118                         for(auto& anim : sub_anims_) {
1119                                 std::set<map_location> tmp = anim.second.get_overlaped_hex(value, src_, dst_);
1120                                 overlaped_hex_.insert(tmp.begin(), tmp.end());
1121                         }
1122                 } else {
1123                         // Offscreen animations only invalidate their own hex, no propagation,
1124                         // but we still need this to play sounds
1125                         overlaped_hex_.insert(src_);
1126                 }
1127         }
1128 
1129         if(complete_redraw) {
1130                 if( need_update()) {
1131                         disp->invalidate(overlaped_hex_);
1132                         invalidated_ = true;
1133                         return true;
1134                 } else {
1135                         invalidated_ = disp->propagate_invalidation(overlaped_hex_);
1136                         return invalidated_;
1137                 }
1138         } else {
1139                 if(need_minimal_update()) {
1140                         disp->invalidate(overlaped_hex_);
1141                         invalidated_ = true;
1142                         return true;
1143                 } else {
1144                         return false;
1145                 }
1146         }
1147 }

1246 void unit_animation::particle::redraw(const frame_parameters& value,const map_location& src, const map_location& dst, halo::manager& halo_man)
1247 {
1248         const unit_frame& current_frame = get_current_frame();
1249         const int animation_time = get_animation_time();
1250         const frame_parameters default_val = parameters_.parameters(animation_time - get_begin_time());
1251 
1252         // Everything is relative to the first frame in an attack/defense/etc. block.
1253         // so we need to check if this particular frame is due to be shown at this time
1254         bool in_scope_of_frame = (animation_time >= get_current_frame_begin_time() ? true: false);
1255         if(animation_time > get_current_frame_end_time()) in_scope_of_frame = false;
1256 
1257         // Sometimes even if the frame is not due to be shown, a frame image still must be shown.
1258         // i.e. in a defense animation that is shorter than an attack animation.
1259         // the halos should not persist though and use the 'in_scope_of_frame' variable.
1260 
1261         // For sound frames we want the first time variable set only after the frame has started.
1262         if(get_current_frame_begin_time() != last_frame_begin_time_ && animation_time >= get_current_frame_begin_time()) {
1263                 last_frame_begin_time_ = get_current_frame_begin_time();
1264                 current_frame.redraw(get_current_frame_time(), true, in_scope_of_frame, src, dst, halo_id_, halo_man, default_val, value);
1265         } else {
1266                 current_frame.redraw(get_current_frame_time(), false, in_scope_of_frame, src, dst, halo_id_, halo_man, default_val, value);
1267         }
1268 }
