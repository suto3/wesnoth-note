 105         struct filter_transform
 106         {
 107                 explicit filter_transform(const std::vector<std::string>& filtertext) : filtertext_(filtertext) {}
 108                 bool operator()(const config& cfg) const
 109                 {
 110                         for(const auto& filter : filtertext_)
 111                         {
 112                                 bool found = false;
 113                                 for(const auto& attribute : cfg.attribute_range())
 114                                 {
 115                                         std::string val = attribute.second.str();
 116                                         if(std::search(val.begin(),
 117                                                 val.end(),
 118                                                 filter.begin(),
 119                                                 filter.end(),
 120                                                 chars_equal_insensitive)
 121                                                 != val.end())
 122                                         {
 123                                                 found = true;
 124                                                 break;
 125                                         }
 126                                 }
 127                                 if(!found) {
 128                                         return false;
 129                                 }
 130                         }
 131                         return true;
 132                 }
 133                 const std::vector<std::string> filtertext_;
 134         };

 136         /**
 137          * Retrieves an element from the given associative container or dies in some
 138          * way.
 139          *
 140          * It fails an @a assert() check or throws an exception if the requested element
 141          * does not exist.
 142          *
 143          * @return An element from the container that is guranteed to have existed
 144          *         before running this function.
 145          */
 146         template <typename MapT>
 147         typename MapT::mapped_type const& const_at(typename MapT::key_type const& key,
 148                                                                                    MapT const& map)
 149         {
 150                 typename MapT::const_iterator it = map.find(key);
 151                 if(it == map.end()) {
 152                         assert(it != map.end());
 153                         throw std::out_of_range(
 154                                         "const_at()"); // Shouldn't get here without disabling assert()
 155                 }
 156                 return it->second;
 157         }

 159         std::string make_display_dependencies(
 160                 const std::string& addon_id,
 161                 const addons_list& addons_list,
 162                 const addons_tracking_list& addon_states)
 163         {
 164                 const addon_info& addon = const_at(addon_id, addons_list);
 165                 std::string str;
 166 
 167                 const std::set<std::string>& deps = addon.resolve_dependencies(addons_list);
 168 
 169                 for(const auto& dep_id : deps) {
 170                         addon_info dep;
 171                         addon_tracking_info depstate;
 172 
 173                         addons_list::const_iterator ali = addons_list.find(dep_id);
 174                         addons_tracking_list::const_iterator tli = addon_states.find(dep_id);
 175 
 176                         if(ali == addons_list.end()) {
 177                                 dep.id = dep_id; // Build dummy addon_info.
 178                         } else {
 179                                 dep = ali->second;
 180                         }
 181 
 182                         if(tli == addon_states.end()) {
 183                                 depstate = get_addon_tracking_info(dep);
 184                         } else {
 185                                 depstate = tli->second;
 186                         }
 187 
 188                         if(!str.empty()) {
 189                                 str += ", ";
 190                         }
 191 
 192                         str += addon_list::colorize_addon_state_string(dep.display_title(), depstate.state);
 193                 }
 194 
 195                 return str;
 196         }

 253 static std::string describe_status_verbose(const addon_tracking_info& state)
 254 {
 255         std::string s;
 256 
 257         utils::string_map i18n_symbols {{"local_version", state.installed_version.str()}};
 258 
 259         switch(state.state) {
 260                 case ADDON_NONE:
 261                         s = !state.can_publish
 262                                 ? _("addon_state^Not installed")
 263                                 : _("addon_state^Published, not installed");
 264                         break;
 265                 case ADDON_INSTALLED:
 266                         s = !state.can_publish
 267                                 ? _("addon_state^Installed")
 268                                 : _("addon_state^Published");
 269                         break;
 270                 case ADDON_NOT_TRACKED:
 271                         s = !state.can_publish
 272                                 ? _("addon_state^Installed, not tracking local version")
 273                                 // Published add-ons often don't have local status information,
 274                                 // hence untracked. This should be considered normal.
 275                                 : _("addon_state^Published, not tracking local version");
 276                         break;
 277                 case ADDON_INSTALLED_UPGRADABLE: {
 278                         const std::string vstr = !state.can_publish
 279                                 ? _("addon_state^Installed ($local_version|), upgradable")
 280                                 : _("addon_state^Published ($local_version| installed), upgradable");
 281 
 282                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 283                 } break;
 284                 case ADDON_INSTALLED_OUTDATED: {
 285                         const std::string vstr = !state.can_publish
 286                                 ? _("addon_state^Installed ($local_version|), outdated on server")
 287                                 : _("addon_state^Published ($local_version| installed), outdated on server");
 288 
 289                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 290                 } break;
 291                 case ADDON_INSTALLED_LOCAL_ONLY:
 292                         s = !state.can_publish
 293                                 ? _("addon_state^Installed, not ready to publish")
 294                                 : _("addon_state^Ready to publish");
 295                         break;
 296                 case ADDON_INSTALLED_BROKEN:
 297                         s = !state.can_publish
 298                                 ? _("addon_state^Installed, broken")
 299                                 : _("addon_state^Published, broken");
 300                         break;
 301                 default:
 302                         s = _("addon_state^Unknown");
 303         }
 304 
 305         return addon_list::colorize_addon_state_string(s, state.state);
 306 }

 445 void addon_manager::load_addon_list(window& window)
 446 {
 447         if(need_wml_cache_refresh_) {
 448                 refresh_addon_version_info_cache();
 449         }
 450 
 451         read_addons_list(cfg_, addons_);
 452 
 453         std::vector<std::string> publishable_addons = available_addons();
 454 
 455         for(std::string id : publishable_addons) {
 456                 if(addons_.find(id) == addons_.end()) {
 457                         // Get a config from the addon's pbl file
 458                         // Note that the name= key is necessary or stuff breaks, since the filter code uses this key
 459                         // to match add-ons in the config list. It also fills in addon_info's id field.
 460                         config pbl_cfg = get_addon_pbl_info(id);
 461                         pbl_cfg["name"] = id;
 462 
 463                         // Add the add-on to the list.
 464                         addon_info addon(pbl_cfg);
 465                         addon.local_only = true;
 466                         addons_[id] = addon;
 467 
 468                         // Add the addon to the config entry
 469                         cfg_.add_child("campaign", pbl_cfg);
 470                 }
 471         }
 472 
 473         if(addons_.empty()) {
 474                 show_transient_message(window.video(), _("No Add-ons Available"), _("There are no add-ons available for download from this server."));
 475                 window.close();
 476         }
 477 
 478         addon_list& list = find_widget<addon_list>(&window, "addons", false);
 479         list.set_addons(addons_);
 480 
 481         bool has_upgradable_addons = false;
 482         for(const auto& a : addons_) {
 483                 tracking_info_[a.first] = get_addon_tracking_info(a.second);
 484 
 485                 if(tracking_info_[a.first].state == ADDON_INSTALLED_UPGRADABLE) {
 486                         has_upgradable_addons = true;
 487                 }
 488         }
 489 
 490         find_widget<button>(&window, "update_all", false).set_active(has_upgradable_addons);
 491 
 492         apply_filters(window);
 493 }

 504 boost::dynamic_bitset<> addon_manager::get_name_filter_visibility(const window& window) const
 505 {
 506         const text_box& name_filter = find_widget<const text_box>(&window, "filter", false);
 507         const std::string& text = name_filter.get_value();
 508 
 509         filter_transform filter(utils::split(text, ' '));
 510         boost::dynamic_bitset<> res;
 511 
 512         const config::const_child_itors& addon_cfgs = cfg_.child_range("campaign");
 513 
 514         for(const auto& a : addons_)
 515         {
 516                 const config& addon_cfg = *std::find_if(addon_cfgs.begin(), addon_cfgs.end(),
 517                         [&a](const config& cfg)
 518                 {
 519                         return cfg["name"] == a.first;
 520                 });
 521 
 522                 res.push_back(filter(addon_cfg));
 523         }
 524 
 525         return res;
 526 }

 528 boost::dynamic_bitset<> addon_manager::get_status_filter_visibility(const window& window) const
 529 {
 530         const menu_button& status_filter = find_widget<const menu_button>(&window, "install_status_filter", false);
 531         const ADDON_STATUS_FILTER selection = status_filter_types_[status_filter.get_value()].first;
 532 
 533         boost::dynamic_bitset<> res;
 534         for(const auto& a : addons_) {
 535                 const addon_tracking_info& info = tracking_info_.at(a.second.id);
 536 
 537                 res.push_back(
 538                         (selection == FILTER_ALL) ||
 539                         (selection == FILTER_INSTALLED     && is_installed_addon_status(info.state)) ||
 540                         (selection == FILTER_UPGRADABLE    && info.state == ADDON_INSTALLED_UPGRADABLE) ||
 541                         (selection == FILTER_PUBLISHABLE   && info.can_publish == true) ||
 542                         (selection == FILTER_NOT_INSTALLED && info.state == ADDON_NONE)
 543                 );
 544         }
 545 
 546         return res;
 547 }

 549 boost::dynamic_bitset<> addon_manager::get_type_filter_visibility(const window& window) const
 550 {
 551         const multimenu_button& type_filter = find_widget<const multimenu_button>(&window, "type_filter", false);
 552 
 553         boost::dynamic_bitset<> toggle_states = type_filter.get_toggle_states();
 554         if(toggle_states.none()) {
 555                 // Nothing selected. It means that *all* add-ons are shown.
 556                 boost::dynamic_bitset<> res_flipped(addons_.size());
 557                 return ~res_flipped;
 558         } else {
 559                 boost::dynamic_bitset<> res;
 560 
 561                 for(const auto& a : addons_) {
 562                         int index = std::find_if(type_filter_types_.begin(), type_filter_types_.end(),
 563                                 [&a](const std::pair<ADDON_TYPE, std::string>& entry) {
 564                                         return entry.first == a.second.type;
 565                                 }) - type_filter_types_.begin();
 566                         res.push_back(toggle_states[index]);
 567                 }
 568 
 569                 return res;
 570         }
 571 }

 583 void addon_manager::execute_action_on_selected_addon(window& window)

 585         // Explicitly return to the main page if we're in low-res mode so the list is visible.
 586         if(stacked_widget* stk = find_widget<stacked_widget>(&window, "main_stack", false, false)) {
 587                 stk->select_layer(0);
 588         }
 589 
 590         addon_list& addons = find_widget<addon_list>(&window, "addons", false);
 591         const addon_info* addon = addons.get_selected_addon();
 592 
 593         if(addon == nullptr) {
 594                 return;
 595         }
 596 
 597         try {
 598                 (this->*fptr)(*addon, window);
 599         } catch(const addons_client::user_exit&) {
 600                 // User canceled the op.
 601         }
 602 }

 604 void addon_manager::install_addon(const addon_info& addon, window& window)
 605 {
 606         addons_client::install_result result = client_.install_addon_with_checks(addons_, addon);
 607 
 608         // Take note if any wml_changes occurred
 609         need_wml_cache_refresh_ |= result.wml_changed;
 610 
 611         if(result.outcome != addons_client::install_outcome::abort) {
 612                 reload_list_and_reselect_item(addon.id, window);
 613         }
 614 }

 616 void addon_manager::uninstall_addon(const addon_info& addon, window& window)
 617 {
 618         if(have_addon_pbl_info(addon.id) || have_addon_in_vcs_tree(addon.id)) {
 619                 show_error_message(window.video(),
 620                         _("The following add-on appears to have publishing or version control information stored locally, and will not be removed:") + " " +
 621                                 addon.display_title());
 622                 return;
 623         }
 624 
 625         bool success = remove_local_addon(addon.id);
 626 
 627         if(!success) {
 628                 show_error_message(window.video(), _("The following add-on could not be deleted properly:") + " " + addon.display_title());
 629         } else {
 630                 need_wml_cache_refresh_ = true;
 631 
 632                 reload_list_and_reselect_item(addon.id, window);
 633         }
 634 }

 660 /** Performs all backend and UI actions for publishing the specified add-on. */
 661 void addon_manager::publish_addon(const addon_info& addon, window& window)
 662 {
 663         std::string server_msg;
 664 
 665         const std::string addon_id = addon.id;
 666         config cfg = get_addon_pbl_info(addon_id);
 667 
 668         const version_info& version_to_publish = cfg["version"].str();
 669 
 670         if(version_to_publish <= tracking_info_[addon_id].remote_version) {
 671                 const int res = gui2::show_message(window.video(), _("Warning"),
 672                         _("The remote version of this add-on is greater or equal to the version being uploaded. Do you really wish to continue?"),
 673                         gui2::dialogs::message::yes_no_buttons);
 674 
 675                 if(res != gui2::window::OK) {
 676                         return;
 677                 }
 678         }
 679 
 680         if(!::image::exists(cfg["icon"].str())) {
 681                 gui2::show_error_message(window.video(), _("Invalid icon path. Make sure the path points to a valid image."));
 682         } else if(!client_.request_distribution_terms(server_msg)) {
 683                 gui2::show_error_message(window.video(),
 684                         _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 685         } else if(gui2::show_message(window.video(), _("Terms"), server_msg, gui2::dialogs::message::ok_cancel_buttons, true) == gui2::window::OK) {
 686                 if(!client_.upload_addon(addon_id, server_msg, cfg)) {
 687                         gui2::show_error_message(window.video(),
 688                                 _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 689                 } else {
 690                         gui2::show_transient_message(window.video(), _("Response"), server_msg);
 691                         fetch_addons_list(window);
 692                         reload_list_and_reselect_item(addon_id, window);
 693                 }
 694         }
 695 }

 697 /** Performs all backend and UI actions for taking down the specified add-on. */
 698 void addon_manager::delete_addon(const addon_info& addon, window& window)
 699 {
 700         const std::string addon_id = addon.id;
 701         const std::string& text = vgettext(
 702                 "Deleting '$addon|' will permanently erase its download and upload counts on the add-ons server. Do you really wish to continue?",
 703                 {{"addon", make_addon_title(addon_id)}} // FIXME: need the real title!
 704         );
 705 
 706         const int res = gui2::show_message(
 707                 window.video(), _("Confirm"), text, gui2::dialogs::message::yes_no_buttons);
 708 
 709         if(res != gui2::window::OK) {
 710                 return;
 711         }
 712 
 713         std::string server_msg;
 714         if(!client_.delete_remote_addon(addon_id, server_msg)) {
 715                 gui2::show_error_message(window.video(), _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 716         } else {
 717                 // FIXME: translation needed!
 718                 gui2::show_transient_message(window.video(), _("Response"), server_msg);
 719                 fetch_addons_list(window);
 720                 reload_list_and_reselect_item(addon_id, window);
 721         }
 722 }

 724 /** Called when the player double-clicks an add-on. */
 725 void addon_manager::execute_default_action(const addon_info& addon, window& window)
 726 {
 727         switch(tracking_info_[addon.id].state) {
 728                 case ADDON_NONE:
 729                         install_addon(addon, window);
 730                         break;
 731                 case ADDON_INSTALLED:
 732                         if(!tracking_info_[addon.id].can_publish) {
 733                                 utils::string_map symbols{ { "addon", addon.display_title() } };
 734                                 int res = gui2::show_message(window.video(), _("Uninstall add-on"),
 735                                         vgettext("Do you want to uninstall '$addon|'?", symbols),
 736                                         gui2::dialogs::message::ok_cancel_buttons);
 737                                 if(res == gui2::window::OK) {
 738                                         uninstall_addon(addon, window);
 739                                 }
 740                         }
 741                         break;
 742                 case ADDON_INSTALLED_UPGRADABLE:
 743                         update_addon(addon, window);
 744                         break;
 745                 case ADDON_INSTALLED_LOCAL_ONLY:
 746                 case ADDON_INSTALLED_OUTDATED:
 747                         publish_addon(addon, window);
 748                         break;
 749                 default:
 750                         break;
 751         }
 752 }

 770 static std::string format_addon_time(time_t time)
 771 {
 772         if(time) {
 773                 std::ostringstream ss;
 774 
 775                 const char* format = preferences::use_twelve_hour_clock_format()
 776                         ? "%Y-%m-%d %I:%M %p"
 777                         : "%Y-%m-%d %H:%M";
 778 
 779                 ss << utils::put_time(std::localtime(&time), format);
 780 
 781                 return ss.str();
 782         }
 783 
 784         return font::unicode_em_dash;
 785 }

 787 void addon_manager::on_addon_select(window& window)
 788 {
 789         const addon_info* info = find_widget<addon_list>(&window, "addons", false).get_selected_addon();
 790 
 791         if(info == nullptr) {
 792                 return;
 793         }
 794 
 795         widget* parent = &window;
 796         if(stacked_widget* stk = find_widget<stacked_widget>(&window, "main_stack", false, false)) {
 797                 parent = stk->get_layer_grid(1);
 798         }
 799 
 800         find_widget<drawing>(parent, "image", false).set_label(info->display_icon());
 801 
 802         find_widget<styled_widget>(parent, "title", false).set_label(info->display_title());
 803         find_widget<styled_widget>(parent, "description", false).set_label(info->description);
 804         find_widget<styled_widget>(parent, "version", false).set_label(info->version.str());
 805         find_widget<styled_widget>(parent, "author", false).set_label(info->author);
 806         find_widget<styled_widget>(parent, "type", false).set_label(info->display_type());
 807 
 808         styled_widget& status = find_widget<styled_widget>(parent, "status", false);
 809         status.set_label(describe_status_verbose(tracking_info_[info->id]));
 810         status.set_use_markup(true);
 811 
 812         find_widget<styled_widget>(parent, "size", false).set_label(size_display_string(info->size));
 813         find_widget<styled_widget>(parent, "downloads", false).set_label(std::to_string(info->downloads));
 814         find_widget<styled_widget>(parent, "created", false).set_label(format_addon_time(info->created));
 815         find_widget<styled_widget>(parent, "updated", false).set_label(format_addon_time(info->updated));
 816 
 817         find_widget<styled_widget>(parent, "dependencies", false).set_label(!info->depends.empty()
 818                 ? make_display_dependencies(info->id, addons_, tracking_info_)
 819                 : _("None"));
 820 
 821         std::string languages;
 822 
 823         for(const auto& lc : info->locales) {
 824                 const std::string& langlabel = langcode_to_string(lc);
 825                 if(!langlabel.empty()) {
 826                         if(!languages.empty()) {
 827                                 languages += ", ";
 828                         }
 829                         languages += langlabel;
 830                 }
 831         }
 832 
 833         find_widget<styled_widget>(parent, "translations", false).set_label(!languages.empty() ? languages : _("None"));
 834 
 835         const std::string& feedback_url = info->feedback_url;
 836 
 837         if(!feedback_url.empty()) {
 838                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(1);
 839                 find_widget<text_box>(parent, "url", false).set_value(feedback_url);
 840         } else {
 841                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(0);
 842         }
 843 
 844         bool installed = is_installed_addon_status(tracking_info_[info->id].state);
 845         bool updatable = tracking_info_[info->id].state == ADDON_INSTALLED_UPGRADABLE;
 846 
 847         stacked_widget& action_stack = find_widget<stacked_widget>(parent, "action_stack", false);
 848 
 849         if(!tracking_info_[info->id].can_publish) {
 850                 action_stack.select_layer(0);
 851 
 852                 stacked_widget& install_update_stack = find_widget<stacked_widget>(parent, "install_update_stack", false);
 853                 install_update_stack.select_layer(updatable ? 1 : 0);
 854 
 855                 if(!updatable) {
 856                         find_widget<button>(parent, "install", false).set_active(!installed);
 857                 } else {
 858                         find_widget<button>(parent, "update", false).set_active(true);
 859                 }
 860 
 861                 find_widget<button>(parent, "uninstall", false).set_active(installed);
 862         } else {
 863                 action_stack.select_layer(1);
 864 
 865                 // Always enable the publish button, but disable the delete button if not yet published.
 866                 find_widget<button>(parent, "publish", false).set_active(true);
 867                 find_widget<button>(parent, "delete", false).set_active(!info->local_only);
 868         }
 869 }
