  92 static void make_add_diff(const simple_wml::node& src, const char* gamelist,
  93                                                   const char* type,
  94                                                   simple_wml::document& out, int index=-1)
  95 {
  96         if (!out.child("gamelist_diff")) {
  97                 out.root().add_child("gamelist_diff");
  98         }
  99 
 100         simple_wml::node* top = out.child("gamelist_diff");
 101         if(gamelist) {
 102                 top = &top->add_child("change_child");
 103                 top->set_attr_int("index", 0);
 104                 top = &top->add_child("gamelist");
 105         }
 106 
 107         simple_wml::node& insert = top->add_child("insert_child");
 108         const simple_wml::node::child_list& children = src.children(type);
 109         assert(!children.empty());
 110         if(index < 0) {
 111                 index = children.size() - 1;
 112         }
 113 
 114         assert(index < static_cast<int>(children.size()));
 115         insert.set_attr_int("index", index);
 116         children[index]->copy_into(insert.add_child(type));
 117 }

 119 static bool make_delete_diff(const simple_wml::node& src,
 120                                                          const char* gamelist,
 121                                                          const char* type,
 122                                                          const simple_wml::node* remove,
 123                                                          simple_wml::document& out)
 124 {
 125         if (!out.child("gamelist_diff")) {
 126                 out.root().add_child("gamelist_diff");
 127         }
 128 
 129         simple_wml::node* top = out.child("gamelist_diff");
 130         if(gamelist) {
 131                 top = &top->add_child("change_child");
 132                 top->set_attr_int("index", 0);
 133                 top = &top->add_child("gamelist");
 134         }
 135 
 136         const simple_wml::node::child_list& children = src.children(type);
 137         const simple_wml::node::child_list::const_iterator itor =
 138                         std::find(children.begin(), children.end(), remove);
 139         if(itor == children.end()) {
 140                 return false;
 141         }
 142         const int index = itor - children.begin();
 143         simple_wml::node& del = top->add_child("delete_child");
 144         del.set_attr_int("index", index);
 145         del.add_child(type);
 146         return true;
 147 }

 149 static bool make_change_diff(const simple_wml::node& src,
 150                                                          const char* gamelist,
 151                                                          const char* type,
 152                                                          const simple_wml::node* item,
 153                                                          simple_wml::document& out)
 154 {
 155         if (!out.child("gamelist_diff")) {
 156                 out.root().add_child("gamelist_diff");
 157         }
 158 
 159         simple_wml::node* top = out.child("gamelist_diff");
 160         if(gamelist) {
 161                 top = &top->add_child("change_child");
 162                 top->set_attr_int("index", 0);
 163                 top = &top->add_child("gamelist");
 164         }
 165         const simple_wml::node::child_list& children = src.children(type);
 166         const simple_wml::node::child_list::const_iterator itor =
 167                         std::find(children.begin(), children.end(), item);
 168         if(itor == children.end()) {
 169                 return false;
 170         }
 171 
 172         simple_wml::node& diff = *top;
 173         simple_wml::node& del = diff.add_child("delete_child");
 174         const int index = itor - children.begin();
 175         del.set_attr_int("index", index);
 176         del.add_child(type);
 177 
 178         //inserts will be processed first by the client, so insert at index+1,
 179         //and then when the delete is processed we'll slide into the right position
 180         simple_wml::node& insert = diff.add_child("insert_child");
 181         insert.set_attr_int("index", index + 1);
 182         children[index]->copy_into(insert.add_child(type));
 183         return true;
 184 }

 371 config server::read_config() const {
 372         config configuration;
 373         if (config_file_.empty()) return configuration;
 374         try {
 375                 filesystem::scoped_istream stream = preprocess_file(config_file_);
 376                 read(configuration, *stream);
 377                 LOG_SERVER << "Server configuration from file: '" << config_file_

 380                 ERR_CONFIG << "ERROR: Could not read configuration file: '"
 381                                    << config_file_ << "': '" << e.message << "'.\n";
 382         }
 383         return configuration;

 524 void server::read_version(socket_ptr socket, std::shared_ptr<simple_wml::document> doc)
 525 {
 526         if (const simple_wml::node* const version = doc->child("version")) {
 527                 const simple_wml::string_span& version_str_span = (*version)["version"];
 528                 const std::string version_str(version_str_span.begin(),
 529                                                                           version_str_span.end());
 530                 std::vector<std::string>::const_iterator accepted_it;
 531                 // Check if it is an accepted version.
 532                 accepted_it = std::find_if(accepted_versions_.begin(), accepted_versions_.end(),
 533                         std::bind(&utils::wildcard_string_match, version_str, _1));
 534                 if(accepted_it != accepted_versions_.end()) {
 535                         LOG_SERVER << client_address(socket)
 536                                 << "\tplayer joined using accepted version " << version_str
 537                                 << ":\ttelling them to log in.\n";
 538                         async_send_doc(socket, login_response_, std::bind(&server::login, this, _1));
 539                         return;
 540                 }
 541 
 542                 simple_wml::document response;
 543 
 544                 // Check if it is a redirected version
 545                 for(const auto& redirect_version : redirected_versions_) {
 546                         if(utils::wildcard_string_match(version_str, redirect_version.first)) {
 547                                 LOG_SERVER << client_address(socket)
 548                                                    << "\tplayer joined using version " << version_str
 549                                                    << ":\tredirecting them to " << redirect_version.second["host"]
 550                                                    << ":" << redirect_version.second["port"] << "\n";
 551                                 simple_wml::node& redirect = response.root().add_child("redirect");
 552                                 for(const auto& attr : redirect_version.second.attribute_range()) {
 553                                         redirect.set_attr(attr.first.c_str(), attr.second.str().c_str());
 554                                 }
 555                                 send_to_player(socket, response);
 556                                 return;
 557                         }
 558                 }
 559 
 560                 LOG_SERVER << client_address(socket)
 561                                    << "\tplayer joined using unknown version " << version_str
 562                                    << ":\trejecting them\n";
 563 
 564                 // For compatibility with older clients
 565                 response.set_attr("version", accepted_versions_.begin()->c_str());
 566 
 567                 simple_wml::node& reject = response.root().add_child("reject");
 568                 reject.set_attr_dup("accepted_versions", utils::join(accepted_versions_).c_str());
 569                 send_to_player(socket, response);
 570         } else {
 571                 LOG_SERVER << client_address(socket)
 572                                    << "\tclient didn't send its version: rejecting\n";
 573         }

 583 void server::handle_login(socket_ptr socket, std::shared_ptr<simple_wml::document> doc)
 584 {
 585         if(const simple_wml::node* const login = doc->child("login")) {
 586                 // Check if the username is valid (all alpha-numeric plus underscore and hyphen)
 587                 std::string username = (*login)["username"].to_string();
 588                 if (!utils::isvalid_username(username)) {
 589                         async_send_error(socket, "The nickname '" + username + "' contains invalid "
 590                                                                          "characters. Only alpha-numeric characters, underscores and hyphens"
 591                                                                          "are allowed.", MP_INVALID_CHARS_IN_NAME_ERROR);
 592                         server::login(socket);
 593                         return;
 594                 }
 595                 if (username.size() > 20) {
 596                         async_send_error(socket, "The nickname '" + username + "' is too long. Nicks must be 20 characters or less.",
 597                                                          MP_NAME_TOO_LONG_ERROR);
 598                         server::login(socket);
 599                         return;
 600                 }
 601                 // Check if the username is allowed.
 602                 for (std::vector<std::string>::const_iterator d_it = disallowed_names_.begin();
 603                          d_it != disallowed_names_.end(); ++d_it)
 604                 {
 605                         if (utils::wildcard_string_match(utf8::lowercase(username),
 606                                                                                          utf8::lowercase(*d_it)))
 607                         {
 608                                 async_send_error(socket, "The nickname '" + username + "' is reserved and cannot be used by players",
 609                                                                  MP_NAME_RESERVED_ERROR);
 610                                 server::login(socket);
 611                                 return;
 612                         }
 613                 }
 614 
 615                 // If this is a request for password reminder
 616                 if(user_handler_) {
 617                         std::string password_reminder = (*login)["password_reminder"].to_string();
 618                         if(password_reminder == "yes") {
 619                                 try {
 620                                         user_handler_->password_reminder(username);
 621                                         async_send_error(socket, "Your password reminder email has been sent.");
 622                                 } catch (user_handler::error& e) {
 623                                         async_send_error(socket, "There was an error sending your password reminder email. The error message was: " +
 624                                                                          e.message);
 625                                 }
 626                                 return;
 627                         }
 628                 }
 629 
 630                 // Check the username isn't already taken
 631                 auto p = player_connections_.get<name_t>().find(username);
 632                 bool name_taken = p != player_connections_.get<name_t>().end();
 633 
 634                 // Check for password
 635 
 636                 // Current login procedure  for registered nicks is:
 637                 // - Client asks to log in with a particular nick
 638                 // - Server sends client random salt plus some info
 639                 //      generated from the original hash that is required to
 640                 //      regenerate the hash
 641                 // - Client generates hash for the user provided password
 642                 //      and mixes it with the received random salt
 643                 // - Server received salted hash, salts the valid hash with
 644                 //      the same salt it sent to the client and compares the results
 645 
 646                 bool registered = false;
 647                 if(user_handler_) {
 648                         std::string password = (*login)["password"].to_string();
 649                         const bool exists = user_handler_->user_exists(username);
 650                         // This name is registered but the account is not active
 651                         if(exists && !user_handler_->user_is_active(username)) {
 652                                 async_send_warning(socket, "The nickname '" + username + "' is inactive. You cannot claim ownership of this "
 653                                                                                                                                                  "nickname until you activate your account via email or ask an administrator to do it for you.", MP_NAME_INACTIVE_WARNING);
 654                                 //registered = false;
 655                         }
 656                         else if(exists) {
 657                                 // This name is registered and no password provided
 658                                 if(password.empty()) {
 659                                         if(!name_taken) {
 660                                                 send_password_request(socket, "The nickname '" + username +"' is registered on this server.",
 661                                                                                           username, MP_PASSWORD_REQUEST);
 662                                         } else {
 663                                                 send_password_request(socket, "The nickname '" + username + "' is registered on this server."
 664                                                                                                           "\n\nWARNING: There is already a client using this username, "
 665                                                                                                           "logging in will cause that client to be kicked!",
 666                                                                                           username, MP_PASSWORD_REQUEST_FOR_LOGGED_IN_NAME, true);
 667                                         }
 668                                         return;
 669                                 }
 670 
 671                                 // A password (or hashed password) was provided, however
 672                                 // there is no seed
 673                                 if(seeds_[reinterpret_cast<long int>(socket.get())].empty()) {
 674                                         send_password_request(socket, "Please try again.", username, MP_NO_SEED_ERROR);
 675                                         return;
 676                                 }
 677                                 // This name is registered and an incorrect password provided
 678                                 else if(!(user_handler_->login(username, password, seeds_[reinterpret_cast<unsigned long>(socket.get())]))) {
 679                                         const time_t now = time(nullptr);
 680 
 681                                         // Reset the random seed
 682                                         seeds_.erase(reinterpret_cast<unsigned long>(socket.get()));
 683 
 684                                         login_log login_ip = login_log(client_address(socket), 0, now);
 685                                         std::deque<login_log>::iterator i = std::find(failed_logins_.begin(), failed_logins_.end(), login_ip);
 686                                         if(i == failed_logins_.end()) {
 687                                                 failed_logins_.push_back(login_ip);
 688                                                 i = --failed_logins_.end();
 689 
 690                                                 // Remove oldest entry if maximum size is exceeded
 691                                                 if(failed_logins_.size() > failed_login_buffer_size_)
 692                                                         failed_logins_.pop_front();
 693 
 694                                         }
 695 
 696                                         if (i->first_attempt + failed_login_ban_ < now) {
 697                                                 // Clear and move to the beginning
 698                                                 failed_logins_.erase(i);
 699                                                 failed_logins_.push_back(login_ip);
 700                                                 i = --failed_logins_.end();
 701                                         }
 702 
 703                                         i->attempts++;
 704 
 705                                         if (i->attempts > failed_login_limit_) {
 706                                                 LOG_SERVER << ban_manager_.ban(login_ip.ip, now + failed_login_ban_, "Maximum login attempts exceeded", "automatic", "", username);
 707                                                 async_send_error(socket, "You have made too many failed login attempts.", MP_TOO_MANY_ATTEMPTS_ERROR);
 708                                         } else {
 709                                                 send_password_request(socket, "The password you provided for the nickname '" + username +
 710                                                                                           "' was incorrect.", username, MP_INCORRECT_PASSWORD_ERROR);
 711                                         }
 712 
 713                                         // Log the failure
 714                                         LOG_SERVER << client_address(socket) << "\t"
 715                                                            << "Login attempt with incorrect password for nickname '" << username << "'.\n";
 716                                         return;
 717                                 }
 718                                 // This name exists and the password was neither empty nor incorrect
 719                                 registered = true;
 720                                 // Reset the random seed
 721                                 seeds_.erase(reinterpret_cast<long int>(socket.get()));
 722                                 user_handler_->user_logged_in(username);
 723                         }
 724                 }
 725 
 726                 // If we disallow unregistered users and this user is not registered send an error
 727                 if(user_handler_ && !registered && deny_unregistered_login_) {
 728                         async_send_error(socket, "The nickname '" + username + "' is not registered. "
 729                                                                          "This server disallows unregistered nicknames.", MP_NAME_UNREGISTERED_ERROR);
 730                         return;
 731                 }
 732 
 733                 if(name_taken) {
 734                         if(registered) {
 735                                 // If there is already a client using this username kick it
 736                                 process_command("kick " + p->info().name() + " autokick by registered user", username);
 737                         } else {
 738                                 async_send_error(socket, "The nickname '" + username + "' is already taken.", MP_NAME_TAKEN_ERROR);
 739                                 server::login(socket);
 740                                 return;
 741                         }
 742                 }
 743 
 744                 simple_wml::node& player_cfg = games_and_users_list_.root().add_child("user");
 745                 async_send_doc(socket, join_lobby_response_,
 746                         std::bind(&server::add_player, this, _1,
 747                                 wesnothd::player(username, player_cfg, registered,
 748                                 default_max_messages_, default_time_period_, false/*selective_ping*/,
 749                                 user_handler_ && user_handler_->user_is_moderator(username))
 750                         )
 751                 );
 752                 LOG_SERVER << client_address(socket) << "\t" << username
 753                                    << "\thas logged on" << (registered ? " to a registered account" : "") << "\n";
 754 
 755                 if(user_handler_ && user_handler_->user_is_moderator(username)) {
 756                         LOG_SERVER << "Admin automatically recognized: IP: "
 757                                            << client_address(socket) << "\tnick: "
 758                                            << username << std::endl;
 759                         // This string is parsed by the client!
 760                         send_server_message(socket, "You are now recognized as an administrator. "
 761                                                                                 "If you no longer want to be automatically authenticated use '/query signout'.");
 762                 }
 763 
 764                 // Log the IP
 765                 connection_log ip_name = connection_log(username, client_address(socket), 0);
 766                 if (std::find(ip_log_.begin(), ip_log_.end(), ip_name) == ip_log_.end()) {
 767                         ip_log_.push_back(ip_name);
 768                         // Remove the oldest entry if the size of the IP log exceeds the maximum size
 769                         if(ip_log_.size() > max_ip_log_size_) ip_log_.pop_front();
 770                 }
 771         } else {
 772                 async_send_error(socket, "You must login first.", MP_MUST_LOGIN);
 773         }
 774 }

 776 void server::send_password_request(socket_ptr socket, const std::string& msg,
 777                                                                    const std::string& user, const char* error_code, bool force_confirmation)
 778 {
 779         std::string salt = user_handler_->create_salt();
 780         std::string pepper = user_handler_->create_pepper(user);
 781         std::string spices = pepper + salt;
 782         if(user_handler_->use_phpbb_encryption() && pepper.empty()) {
 783                 async_send_error(socket, "Even though your nickname is registered on this server you "
 784                                                                  "cannot log in due to an error in the hashing algorithm. "
 785                                                                  "Logging into your forum account on https://forums.wesnoth.org "
 786                                                                  "may fix this problem.");
 787                 login(socket);
 788                 return;
 789         }
 790 
 791         seeds_[reinterpret_cast<long int>(socket.get())] = salt;
 792 
 793         simple_wml::document doc;
 794         simple_wml::node& e = doc.root().add_child("error");
 795         e.set_attr_dup("message", msg.c_str());
 796         e.set_attr("password_request", "yes");
 797         e.set_attr("phpbb_encryption", user_handler_->use_phpbb_encryption() ? "yes" : "no");
 798         e.set_attr_dup("salt", spices.c_str());
 799         e.set_attr("force_confirmation", force_confirmation ? "yes" : "no");
 800         if(*error_code != '\0') {
 801                 e.set_attr("error_code", error_code);
 802         }
 803 
 804         async_send_doc(socket, doc,
 805                 std::bind(&server::login, this, _1)
 806         );
 807 }

 899 void server::handle_query(socket_ptr socket, simple_wml::node& query)
 900 {
 901         auto iter = player_connections_.find(socket);
 902         if(iter == player_connections_.end())
 903                 return;
 904 
 905         wesnothd::player& player = iter->info();
 906 
 907         const std::string command(query["type"].to_string());
 908         std::ostringstream response;
 909         const std::string& query_help_msg = "Available commands are: adminmsg <msg>, help, games, metrics,"
 910                                                                   " motd, netstats [all], requests, sample, stats, status, wml.";
 911         // Commands a player may issue.
 912         if (command == "status") {
 913                 response << process_command(command + " " + player.name(), player.name());
 914         } else if (command.compare(0, 8, "adminmsg") == 0 || command.compare(0, 6, "report") == 0
 915                            || command == "games"
 916                            || command == "metrics"
 917                            || command == "motd"
 918                            || command == "netstats"
 919                            || command == "netstats all"
 920                            || command == "requests"
 921                            || command == "sample"
 922                            || command == "stats"
 923                            || command == "status " + player.name()
 924                            || command == "wml")
 925         {
 926                 response << process_command(command, player.name());
 927         } else if (player.is_moderator()) {
 928                 if (command == "signout") {
 929                         LOG_SERVER << "Admin signed out: IP: "
 930                                            << client_address(socket) << "\tnick: "
 931                                            << player.name() << std::endl;
 932                         player.set_moderator(false);
 933                         // This string is parsed by the client!
 934                         response << "You are no longer recognized as an administrator.";
 935                         if(user_handler_) {
 936                                 user_handler_->set_is_moderator(player.name(), false);
 937                         }
 938                 } else {
 939                         LOG_SERVER << "Admin Command: type: " << command
 940                                            << "\tIP: "<< client_address(socket)
 941                                            << "\tnick: "<< player.name() << std::endl;
 942                         response << process_command(command, player.name());
 943                         LOG_SERVER << response.str() << std::endl;
 944                 }
 945         } else if (command == "help" || command.empty()) {
 946                 response << query_help_msg;
 947         } else if (command == "admin" || command.compare(0, 6, "admin ") == 0) {
 948                 if (admin_passwd_.empty()) {
 949                         send_server_message(socket, "No password set.");
 950                         return;
 951                 }
 952                 std::string passwd;
 953                 if (command.size() >= 6) passwd = command.substr(6);
 954                 if (passwd == admin_passwd_) {
 955                         LOG_SERVER << "New Admin recognized: IP: "
 956                                            << client_address(socket) << "\tnick: "
 957                                            << player.name() << std::endl;
 958                         player.set_moderator(true);
 959                         // This string is parsed by the client!
 960                         response << "You are now recognized as an administrator.";
 961                         if (user_handler_) {
 962                                 user_handler_->set_is_moderator(player.name(), true);
 963                         }
 964                 } else {
 965                         WRN_SERVER << "FAILED Admin attempt with password: '" << passwd << "'\tIP: "
 966                                            << client_address(socket) << "\tnick: "
 967                                            << player.name() << std::endl;
 968                         response << "Error: wrong password";
 969                 }
 970         } else {
 971                 response << "Error: unrecognized query: '" << command << "'\n" << query_help_msg;
 972         }
 973         send_server_message(socket, response.str());
 974 }

 976 void server::handle_nickserv(socket_ptr socket, simple_wml::node& nickserv)
 977 {
 978         // Check if this server allows nick registration at all
 979         if(!user_handler_) {
 980                 send_server_message(socket, "This server does not allow username registration.");
 981                 return;
 982         }
 983 
 984         if(nickserv.child("register")) {
 985                 try {
 986                         (user_handler_->add_user(player_connections_.find(socket)->name(), (*nickserv.child("register"))["mail"].to_string(),
 987                                         (*nickserv.child("register"))["password"].to_string()));
 988 
 989                         std::stringstream msg;
 990                         msg << "Your username has been registered." <<
 991                                    // Warn that providing an email address might be a good idea
 992                                    ((*nickserv.child("register"))["mail"].empty() ?
 993                                            " It is recommended that you provide an email address for password recovery." : "");
 994                         send_server_message(socket, msg.str());
 995 
 996                         // Mark the player as registered and send the other clients
 997                         // an update to dislpay this change
 998                         player_connections_.find(socket)->info().mark_registered();
 999 
1000                         simple_wml::document diff;
1001                         make_change_diff(games_and_users_list_.root(), nullptr,
1002                                                          "user", player_connections_.find(socket)->info().config_address(), diff);
1003                         send_to_lobby(diff);
1004 
1005                 } catch (user_handler::error& e) {
1006                         send_server_message(socket, "There was an error registering your username. The error message was: "
1007                                                                 + e.message);
1008                 }
1009                 return;
1010         }
1011 
1012         // A user requested to update his password or mail
1013         if(nickserv.child("set")) {
1014                 if(!(user_handler_->user_exists(player_connections_.find(socket)->name()))) {
1015                         send_server_message(socket, "You are not registered. Please register first.");
1016                         return;
1017                 }
1018 
1019                 const simple_wml::node& set = *(nickserv.child("set"));
1020 
1021                 try {
1022                         user_handler_->set_user_detail(player_connections_.find(socket)->name(), set["detail"].to_string(), set["value"].to_string());
1023 
1024                         send_server_message(socket, "Your details have been updated.");
1025 
1026                 } catch (user_handler::error& e) {
1027                         send_server_message(socket, "There was an error updating your details. The error message was: "
1028                                                                 + e.message);
1029                 }
1030 
1031                 return;
1032         }
1033 
1034         // A user requested information about another user
1035         if(nickserv.child("details")) {
1036                 send_server_message(socket, "Valid details for this server are: " +
1037                                                         user_handler_->get_valid_details());
1038                 return;
1039         }
1040 
1041         // A user requested a list of which details can be set
1042         if(nickserv.child("info")) {
1043                 try {
1044                         std::string res = user_handler_->user_info((*nickserv.child("info"))["name"].to_string());
1045                         send_server_message(socket, res);
1046                 } catch (user_handler::error& e) {
1047                         send_server_message(socket, "There was an error looking up the details of the user '" +
1048                                                                 (*nickserv.child("info"))["name"].to_string() + "'. " +" The error message was: "
1049                                         + e.message);
1050                 }
1051                 return;
1052         }
1053 
1054         // A user requested to delete his nick
1055         if(nickserv.child("drop")) {
1056                 if(!(user_handler_->user_exists(player_connections_.find(socket)->name()))) {
1057                         send_server_message(socket, "You are not registered.");
1058                         return;
1059                 }
1060 
1061                 // With the current policy of dissallowing to log in with a
1062                 // registerd username without the password we should never get
1063                 // to call this
1064                 if(!(player_connections_.find(socket)->info().registered())) {
1065                         send_server_message(socket, "You are not logged in.");
1066                         return;
1067                 }
1068 
1069                 try {
1070                         user_handler_->remove_user(player_connections_.find(socket)->name());
1071                         send_server_message(socket, "Your username has been dropped.");
1072 
1073                         // Mark the player as not registered and send the other clients
1074                         // an update to dislpay this change
1075                         player_connections_.find(socket)->info().mark_registered(false);
1076 
1077                         simple_wml::document diff;
1078                         make_change_diff(games_and_users_list_.root(), nullptr,
1079                                                          "user", player_connections_.find(socket)->info().config_address(), diff);
1080                         send_to_lobby(diff);
1081                 } catch (user_handler::error& e) {
1082                         send_server_message(socket, "There was an error dropping your username. The error message was: "
1083                                                                 + e.message);
1084                 }
1085                 return;
1086         }
1087 }

1096 void server::handle_create_game(socket_ptr socket, simple_wml::node& create_game)

1231                 player_connections_.find(socket),

1971 std::string server::process_command(std::string query, std::string issuer_name) {
1972         boost::trim(query);
1973 
1974         if (issuer_name == "*socket*" && query.at(0) == '+') {
1975                 // The first argument might be "+<issuer>: ".
1976                 // In that case we use +<issuer>+ as the issuer_name.
1977                 // (Mostly used for communication with IRC.)
1978                 std::string::iterator issuer_end =
1979                                 std::find(query.begin(), query.end(), ':');
1980                 std::string issuer(query.begin() + 1, issuer_end);
1981                 if (!issuer.empty()) {
1982                         issuer_name = "+" + issuer + "+";
1983                         query = std::string(issuer_end + 1, query.end());
1984                         boost::trim(query);
1985                 }
1986         }
1987 
1988         const std::string::iterator i = std::find(query.begin(), query.end(), ' ');
1989 
1990         try {
1991 
1992                 const std::string command = utf8::lowercase(std::string(query.begin(), i));
1993                 std::string parameters = (i == query.end() ? "" : std::string(i + 1, query.end()));
1994                 boost::trim(parameters);
1995 
1996                 std::ostringstream out;
1997                 std::map<std::string, server::cmd_handler>::iterator handler_itor = cmd_handlers_.find(command);
1998                 if(handler_itor == cmd_handlers_.end()) {
1999                         out << "Command '" << command << "' is not recognized.\n" << help_msg;
2000                 } else {
2001                         const cmd_handler &handler = handler_itor->second;
2002                         try {
2003                                 handler(issuer_name, query, parameters, &out);
2004                         } catch (std::bad_function_call & ex) {
2005                                 ERR_SERVER << "While handling a command '" << command << "', caught a std::bad_function_call exception.\n";

2007                                 out << "An internal server error occurred (std::bad_function_call) while executing '" << command << "'\n";
2008                         }
2009                 }
2010

2012 
2013         } catch ( utf8::invalid_utf8_exception & e ) {
2014                 std::string msg = "While handling a command, caught an invalid utf8 exception: ";
