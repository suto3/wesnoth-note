 213     def _rmtree(self, directory, exclude):
 214         logging.debug("Deleting tree {0}, except for {1}".format(self.name, ",".join(exclude)))
 215 
 216         # Ensure the os calls all happen in the right directory
 217         # not needed for _execute, as that does the cwd manipulation itself
 218         # so only the os.chdir and os.path.isdir here need it
 219         # Another option would be to os.path.join with self.get_dir
 220         os.chdir(self.get_dir())
 221         for entry in os.listdir(directory):
 222             if entry in exclude:
 223                 continue
 224             if entry == ".git":
 225                 continue
 226 
 227             relpath = os.path.join(directory, entry)
 228 
 229             if os.path.isdir(relpath):

 233                     self._execute(["rmdir", "--ignore-fail-on-non-empty", relpath])

 237     def _copytree(self, src, dst, ignore=None):
 238         """Recursively copy a directory tree using copy2().
 239 
 240         Based on shutil.copytree
 241         """
 242         names = os.listdir(src)
 243         if ignore is not None:
 244             ignored_names = ignore(src, names)
 245         else:
 246             ignored_names = set()
 247 
 248         if not os.path.exists(dst):
 249             os.makedirs(dst)
 250         errors = []
 251         for name in names:
 252             if name in ignored_names:
 253                 continue
 254             srcname = os.path.join(src, name)
 255             dstname = os.path.join(dst, name)
 256             try:
 257                 if os.path.isdir(srcname):
 258                     self._copytree(srcname, dstname, ignore)
 259                 else:
 260                     shutil.copy2(srcname, dstname)
 261                 # XXX What about devices, sockets etc.?
 262             except (IOError, os.error) as why:
 263                 errors.append((srcname, dstname, str(why)))
 264             # catch the Error from the recursive copytree so that we can
 265             # continue with other files
 266             except Error as err:
 267                 errors.extend(err.args[0])
 268         try:
 269             shutil.copystat(src, dst)
 270         except OSError as why:
 271             if shutil.WindowsError is not None and isinstance(why, shutil.WindowsError):
 272                 # Copying file access times may fail on Windows
 273                 pass
 274             else:
 275                 errors.extend((src, dst, str(why)))
 276         if errors:
 277             raise AddonError(self.name, "Errors attempting to sync:\n{0}".format("\n".join(errors)))

 469     def _github_api_request(self, url, data=None, method=None, authenticate=False):
 470         logging.debug("Making github API request {0}".format(url))
 471 
 472         request = urllib.request.Request(url)
 473         if method:
 474             request.get_method = lambda: method
 475 
 476         if data == "":
 477             # Workaround for PUTs requiring data, even if you have nothing to pass
 478             request.add_data(data)
 479         elif data:
 480             request.add_data(json.dumps(data))
 481 
 482         # Manually adding authentication data
 483         # Basic works in curl, but urllib2 doesn't
 484         # probably because github's API doesn't send a www-authenticate header
 485         if authenticate or self._github_have_authorization():
 486             from base64 import encodestring
 487             auth = self._github_authorization()
 488             if ":" in auth:
 489                 # username:password
 490                 base64string = encodestring(auth).replace('\n', '')
 491                 request.add_header("Authorization", "Basic {0}".format(base64string))
 492             else:
 493                 # token
 494                 request.add_header("Authorization", "Bearer {0}".format(auth))
 495 
 496         try:
 497             response = urllib.request.urlopen(request)
 498         except IOError as e:
 499             raise Error("GitHub API failure: " + str(e))
 500         if response.code == 204:
 501             # 204 = No content
 502             return None
 503 
 504         json_parsed = json.load(response)
 505 
 506         link_headers = response.info().getallmatchingheaders("Link")
 507         if link_headers:
 508             logging.debug("Found a Link header in response, analyzing...")
 509             link_header = link_headers[0].lstrip("Link:")
 510             links_raw = link_header.split(",")
 511             links_split_raw = [link.split(";") for link in links_raw]
 512             links_split_proc = [(l[1].strip().lstrip('rel="').rstrip('"'), l[0].strip().lstrip("<").rstrip(">")) for l in links_split_raw]
 513             links_dict = dict((k, v) for (k, v) in links_split_proc)
 514             if "next" in links_dict:
 515                 logging.debug("Link with rel=\"next\" found, recursing to deal with pagination")
 516                 rest = self._github_api_request(links_dict["next"], data, method, authenticate)
 517                 json_parsed += rest
 518 
 519         return json_parsed

 529     def _execute(self, command, cwd=None, check_error=False):
 530         #TODO: have an errorcheck that actually checks the returncode?
 531         """Executes a command.
 532 
 533         command: The command to execute.
 534         cwd: Directory to execute the command from.
 535         check_error: Whether to raise an exception if there's stderr output.
 536         Returns stdout, stderr.
 537         Raises libgithub.Error if check_error and len(err).
 538         """
 539         logging.debug("execute command = '%s'", command)
 540 
 541         p = subprocess.Popen(command, stdout=subprocess.PIPE, stderr=subprocess.PIPE, close_fds=True, cwd=cwd)
 542         out = ""
 543         err = ""
 544         while(p.poll() == None):
 545             out += p.stdout.read()
 546             err += p.stderr.read()
 547 
 548         out += p.stdout.read()
 549         err += p.stderr.read()
 550 
 551         logging.debug("===== stdout ====\n%s\n===== stdout ====", out)
 552         logging.debug("===== stderr ====\n%s\n===== stderr ====", err)
 553 
 554         if check_error and len(err):
 555             raise Error("Failure executing command '{0}': {1}".format(" ".join(command), err))
 556 
 557         return _execresult(out, err, p.returncode)
