 158 /**
 159  * Load data into an empty unit_type (build to FULL).
 160  */
 161 void unit_type::build_full(
 162                 const movement_type_map& mv_types, const race_map& races, const config::const_child_itors& traits)
 163 {
 164         // Don't build twice.
 165         if(FULL <= build_status_) {
 166                 return;
 167         }
 168 
 169         // Make sure we are built to the preceding build level.
 170         build_help_index(mv_types, races, traits);
 171 
 172         for(int i = 0; i < 2; ++i) {
 173                 if(gender_types_[i]) {
 174                         gender_types_[i]->build_full(mv_types, races, traits);
 175                 }
 176         }
 177 
 178         if(race_ != &unit_race::null_race) {
 179                 if(undead_variation_.empty()) {
 180                         undead_variation_ = race_->undead_variation();
 181                 }
 182         }
 183 
 184         zoc_ = cfg_["zoc"].to_bool(level_ > 0);
 185 
 186         game_config::add_color_info(cfg_);
 187 
 188         hp_bar_scaling_ = cfg_["hp_bar_scaling"].to_double(game_config::hp_bar_scaling);
 189         xp_bar_scaling_ = cfg_["xp_bar_scaling"].to_double(game_config::xp_bar_scaling);
 190 
 191         // Propagate the build to the variations.
 192         for(variations_map::value_type& variation : variations_) {
 193                 variation.second.build_full(mv_types, races, traits);
 194         }
 195 
 196         // Deprecation messages, only seen when unit is parsed for the first time.
 197 
 198         build_status_ = FULL;
 199 }

 201 /**
 202  * Partially load data into an empty unit_type (build to HELP_INDEXED).
 203  */
 204 void unit_type::build_help_index(
 205                 const movement_type_map& mv_types, const race_map& races, const config::const_child_itors& traits)
 206 {
 207         // Don't build twice.
 208         if(HELP_INDEXED <= build_status_) {
 209                 return;
 210         }
 211 
 212         // Make sure we are built to the preceding build level.
 213         build_created(mv_types, races, traits);
 214 
 215         type_name_ = cfg_["name"];
 216         description_ = cfg_["description"];
 217         hitpoints_ = cfg_["hitpoints"].to_int(1);
 218         level_ = cfg_["level"];
 219         recall_cost_ = cfg_["recall_cost"].to_int(-1);
 220         movement_ = cfg_["movement"].to_int(1);
 221         vision_ = cfg_["vision"].to_int(-1);
 222         jamming_ = cfg_["jamming"].to_int(0);
 223         max_attacks_ = cfg_["attacks"].to_int(1);
 224         cost_ = cfg_["cost"].to_int(1);
 225         usage_ = cfg_["usage"].str();
 226         undead_variation_ = cfg_["undead_variation"].str();
 227         image_ = cfg_["image"].str();
 228         icon_ = cfg_["image_icon"].str();
 229         small_profile_ = cfg_["small_profile"].str();
 230         profile_ = cfg_["profile"].str();
 231 
 232         adjust_profile(profile_);
 233 
 234         alignment_ = unit_type::ALIGNMENT::NEUTRAL;
 235         alignment_.parse(cfg_["alignment"].str());
 236 
 237         for(int i = 0; i < 2; ++i) {
 238                 if(gender_types_[i]) {
 239                         gender_types_[i]->build_help_index(mv_types, races, traits);
 240                 }
 241         }
 242 
 243         const race_map::const_iterator race_it = races.find(cfg_["race"]);
 244         if(race_it != races.end()) {
 245                 race_ = &race_it->second;
 246         } else {
 247                 race_ = &unit_race::null_race;
 248         }
 249 
 250         // if num_traits is not defined, we use the num_traits from race
 251         num_traits_ = cfg_["num_traits"].to_int(race_->num_traits());
 252 
 253         const std::vector<std::string> genders = utils::split(cfg_["gender"]);
 254         for(std::vector<std::string>::const_iterator g = genders.begin(); g != genders.end(); ++g) {
 255                 genders_.push_back(string_gender(*g));
 256         }
 257 
 258         // For simplicity in other parts of the code, we must have at least one gender.
 259         if(genders_.empty()) {
 260                 genders_.push_back(unit_race::MALE);
 261         }
 262 
 263         if(const config& abil_cfg = cfg_.child("abilities")) {
 264                 for(const config::any_child& ab : abil_cfg.all_children_range()) {
 265                         const config::attribute_value& name = ab.cfg["name"];
 266 
 267                         if(!name.empty()) {
 268                                 abilities_.push_back(name.t_str());
 269                                 ability_tooltips_.push_back(ab.cfg["description"].t_str());
 270                         }
 271                 }
 272         }
 273 
 274         for(const config& adv : cfg_.child_range("advancement")) {
 275                 for(const config& effect : adv.child_range("effect")) {
 276                         const config& abil_cfg = effect.child("abilities");
 277 
 278                         if(!abil_cfg || effect["apply_to"] != "new_ability") {
 279                                 continue;
 280                         }
 281 
 282                         for(const config::any_child& ab : abil_cfg.all_children_range()) {
 283                                 const config::attribute_value& name = ab.cfg["name"];
 284 
 285                                 if(!name.empty()) {
 286                                         adv_abilities_.push_back(name.t_str());
 287                                         adv_ability_tooltips_.push_back(ab.cfg["description"].t_str());
 288                                 }
 289                         }
 290                 }
 291         }
 292 
 293         // Set the movement type.
 294         const std::string move_type = cfg_["movement_type"];
 295         const movement_type_map::const_iterator find_it = mv_types.find(move_type);
 296 
 297         if(find_it != mv_types.end()) {
 298                 DBG_UT << "inheriting from movement_type '" << move_type << "'\n";
 299                 movement_type_ = find_it->second;
 300         } else if(!move_type.empty()) {
 301                 DBG_UT << "movement_type '" << move_type << "' not found\n";
 302         }
 303 
 304         // Override parts of the movement type with what is in our config.
 305         movement_type_.merge(cfg_);
 306 
 307         for(const config& t : traits) {
 308                 possible_traits_.add_child("trait", t);
 309         }
 310 
 311         if(race_ != &unit_race::null_race) {
 312                 if(!race_->uses_global_traits()) {
 313                         possible_traits_.clear();
 314                 }
 315 
 316                 if(cfg_["ignore_race_traits"].to_bool()) {
 317                         possible_traits_.clear();
 318                 } else {
 319                         for(const config& t : race_->additional_traits()) {
 320                                 if(alignment_ != unit_type::ALIGNMENT::NEUTRAL || t["id"] != "fearless")
 321                                         possible_traits_.add_child("trait", t);
 322                         }
 323                 }
 324 
 325                 if(undead_variation_.empty()) {
 326                         undead_variation_ = race_->undead_variation();
 327                 }
 328         }
 329 
 330         // Insert any traits that are just for this unit type
 331         for(const config& trait : cfg_.child_range("trait")) {
 332                 possible_traits_.add_child("trait", trait);
 333         }
 334 
 335         for(const config& var_cfg : cfg_.child_range("variation")) {
 336                 const std::string& var_id = var_cfg["variation_id"].empty()
 337                         ? var_cfg["variation_name"]
 338                         : var_cfg["variation_id"];
 339 
 340                 variations_map::iterator ut;
 341                 bool success;
 342                 std::tie(ut, success) = variations_.emplace(var_id, unit_type(var_cfg, id_));
 343 
 344                 if(success) {
 345                         ut->second.debug_id_ = debug_id_ + " [" + var_id + "]";
 346                         ut->second.base_id_ = base_id_; // In case this is not id_.
 347                         ut->second.build_help_index(mv_types, races, traits);
 348                 } else {
 349                         ERR_CF << "Skipping duplicate unit variation ID: " << var_id << "\n";
 350                 }
 351         }
 352 
 353         hide_help_ = cfg_["hide_help"].to_bool();
 354 
 355         build_status_ = HELP_INDEXED;
 356 }

 358 /**
 359  * Load the most needed data into an empty unit_type (build to CREATE).
 360  * This creates the gender-specific types (if needed) and also defines how much
 361  * experience is needed to advance as well as what this advances to.
 362  */
 363 void unit_type::build_created(
 364                 const movement_type_map& mv_types, const race_map& races, const config::const_child_itors& traits)
 365 {
 366         // Don't build twice.
 367         if(CREATED <= build_status_) {
 368                 return;
 369         }
 370 
 371         // There is no preceding build level (other than being constructed).
 372 
 373         // These should still be nullptr from the constructor.
 374         assert(!gender_types_[0]);
 375         assert(!gender_types_[1]);
 376 
 377         if(const config& male_cfg = cfg_.child("male")) {
 378                 gender_types_[0].reset(new unit_type(male_cfg, id_));
 379                 gender_types_[0]->debug_id_ = debug_id_ + " (male)";
 380         }
 381 
 382         if(const config& female_cfg = cfg_.child("female")) {
 383                 gender_types_[1].reset(new unit_type(female_cfg, id_));
 384                 gender_types_[1]->debug_id_ = debug_id_ + " (female)";
 385         }
 386 
 387         for(unsigned i = 0; i < gender_types_.size(); ++i) {
 388                 if(gender_types_[i]) {
 389                         gender_types_[i]->build_created(mv_types, races, traits);
 390                 }
 391         }
 392 
 393         const std::string& advances_to_val = cfg_["advances_to"];
 394         if(advances_to_val != "null" && !advances_to_val.empty()) {
 395                 advances_to_ = utils::split(advances_to_val);
 396         }
 397 
 398         DBG_UT << "unit_type '" << log_id() << "' advances to : " << advances_to_val << "\n";
 399 
 400         experience_needed_ = cfg_["experience"].to_int(500);
 401 
 402         build_status_ = CREATED;
 403 }

 405 /**
 406  * Performs a build of this to the indicated stage.
 407  */
 408 void unit_type::build(BUILD_STATUS status,
 409                 const movement_type_map& movement_types,
 410                 const race_map& races,
 411                 const config::const_child_itors& traits)
 412 {
 413         DBG_UT << "Building unit type " << log_id() << ", level " << status << '\n';
 414 
 415         switch(status) {
 416         case NOT_BUILT:
 417                 // Already done in the constructor.
 418                 return;
 419 
 420         case CREATED:
 421                 // Build the basic data.
 422                 build_created(movement_types, races, traits);
 423                 return;
 424 
 425         case VARIATIONS: // Implemented as part of HELP_INDEXED
 426         case HELP_INDEXED:
 427                 // Build the data needed to feed the help index.
 428                 build_help_index(movement_types, races, traits);
 429                 return;
 430 
 431         case FULL:
 432                 build_full(movement_types, races, traits);
 433                 return;
 434 
 435         default:
 436                 ERR_UT << "Build of unit_type to unrecognized status (" << status << ") requested." << std::endl;
 437                 // Build as much as possible.
 438                 build_full(movement_types, races, traits);
 439                 return;
 440         }
 441 }

 590 void unit_type::add_advancement(const unit_type& to_unit, int xp)
 591 {
 592         const std::string& to_id = to_unit.id_;
 593 
 594         // Add extra advancement path to this unit type
 595         LOG_CONFIG << "adding advancement from " << log_id() << " to " << to_unit.log_id() << "\n";
 596         if(std::find(advances_to_.begin(), advances_to_.end(), to_id) == advances_to_.end()) {
 597                 advances_to_.push_back(to_id);
 598         } else {
 599                 LOG_CONFIG << "advancement from " << log_id() << " to " << to_unit.log_id() << " already known, ignoring.\n";
 600                 return;
 601         }
 602 
 603         if(xp > 0) {
 604                 // xp is 0 in case experience= wasn't given.
 605                 if(!in_advancefrom_) {
 606                         // This function is called for and only for an [advancefrom] tag in a unit_type referencing this unit_type.
 607                         in_advancefrom_ = true;
 608                         experience_needed_ = xp;
 609 
 610                         DBG_UT << "Changing experience_needed from " << experience_needed_ << " to " << xp
 611                                    << " due to (first) [advancefrom] of " << to_unit.log_id() << "\n";
 612                 } else if(experience_needed_ > xp) {
 613                         experience_needed_ = xp;
 614                         DBG_UT << "Lowering experience_needed from " << experience_needed_ << " to " << xp
 615                                    << " due to (multiple, lower) [advancefrom] of " << to_unit.log_id() << "\n";
 616                 } else {
 617                         DBG_UT << "Ignoring experience_needed change from " << experience_needed_ << " to " << xp
 618                                    << " due to (multiple, higher) [advancefrom] of " << to_unit.log_id() << "\n";
 619                 }
 620         }
 621 
 622         // Add advancements to gendered subtypes, if supported by to_unit
 623         for(int gender = 0; gender <= 1; ++gender) {
 624                 if(!gender_types_[gender]) {
 625                         continue;
 626                 }
 627 
 628                 if(!to_unit.gender_types_[gender]) {
 629                         WRN_CF << to_unit.log_id() << " does not support gender " << gender << std::endl;
 630                         continue;
 631                 }
 632 
 633                 LOG_CONFIG << "gendered advancement " << gender << ": ";
 634                 gender_types_[gender]->add_advancement(*(to_unit.gender_types_[gender]), xp);
 635         }
 636 
 637         if(cfg_.has_child("variation")) {
 638                 // Make sure the variations are created.
 639                 unit_types.build_unit_type(*this, VARIATIONS);
 640 
 641                 // Add advancements to variation subtypes.
 642                 // Since these are still a rare and special-purpose feature,
 643                 // we assume that the unit designer knows what they're doing,
 644                 // and don't block advancements that would remove a variation.
 645                 for(variations_map::iterator v = variations_.begin(); v != variations_.end(); ++v) {
 646                         LOG_CONFIG << "variation advancement: ";
 647                         v->second.add_advancement(to_unit, xp);
 648                 }
 649         }
 650 }

 691 // This function is only meant to return the likely state a given status
 692 // for a new recruit of this type. It should not be used to check if
 693 // a particular unit has it, use get_state(status_name) for that.
 694 bool unit_type::musthave_status(const std::string& status_name) const
 695 {
 696         // Statuses default to absent.
 697         bool current_status = false;
 698 
 699         // Look at all of the "musthave" traits to see if the
 700         // status gets changed. In the unlikely event it gets changed
 701         // multiple times, we want to try to do it in the same order
 702         // that unit::apply_modifications does things.
 703         for(const config& mod : possible_traits()) {
 704                 if(mod["availability"] != "musthave") {
 705                         continue;
 706                 }
 707 
 708                 for(const config& effect : mod.child_range("effect")) {
 709                         // See if the effect only applies to
 710                         // certain unit types But don't worry
 711                         // about gender checks, since we don't
 712                         // know what the gender of the
 713                         // hypothetical recruit is.
 714                         const std::string& ut = effect["unit_type"];
 715 
 716                         if(!ut.empty()) {
 717                                 const std::vector<std::string>& types = utils::split(ut);
 718 
 719                                 if(std::find(types.begin(), types.end(), id()) == types.end()) {
 720                                         continue;
 721                                 }
 722                         }
 723 
 724                         // We're only interested in status changes.
 725                         if(effect["apply_to"] != "status") {
 726                                 continue;
 727                         }
 728 
 729                         if(effect["add"] == status_name) {
 730                                 current_status = true;
 731                         }
 732 
 733                         if(effect["remove"] == status_name) {
 734                                 current_status = false;
 735                         }
 736                 }
 737         }
 738 
 739         return current_status;
 740 }

 841 int unit_type::resistance_against(const std::string& damage_name, bool attacker) const
 842 {
 843         int resistance = movement_type_.resistance_against(damage_name);
 844         unit_ability_list resistance_abilities;
 845 
 846         if(const config& abilities = cfg_.child("abilities")) {
 847                 for(const config& cfg : abilities.child_range("resistance")) {
 848                         if(!cfg["affect_self"].to_bool(true)) {
 849                                 continue;
 850                         }
 851 
 852                         if(!resistance_filter_matches(cfg, attacker, damage_name, 100 - resistance)) {
 853                                 continue;
 854                         }
 855 
 856                         resistance_abilities.push_back(unit_ability(&cfg, map_location::null_location()));
 857                 }
 858         }
 859 
 860         if(!resistance_abilities.empty()) {
 861                 unit_abilities::effect resist_effect(resistance_abilities, 100 - resistance, false);
 862 
 863                 resistance = 100 - std::min<int>(
 864                         resist_effect.get_composite_value(),
 865                         resistance_abilities.highest("max_value").first
 866                 );
 867         }
 868 
 869         return resistance;
 870 }

 872 bool unit_type::resistance_filter_matches(
 873                 const config& cfg, bool attacker, const std::string& damage_name, int res) const
 874 {
 875         if(!(cfg["active_on"].empty() ||
 876                 (attacker  && cfg["active_on"] == "offense") ||
 877                 (!attacker && cfg["active_on"] == "defense"))
 878         ) {
 879                 return false;
 880         }
 881 
 882         const std::string& apply_to = cfg["apply_to"];
 883 
 884         if(!apply_to.empty()) {
 885                 if(damage_name != apply_to) {
 886                         if(apply_to.find(',') != std::string::npos && apply_to.find(damage_name) != std::string::npos) {
 887                                 const std::vector<std::string>& vals = utils::split(apply_to);
 888 
 889                                 if(std::find(vals.begin(), vals.end(), damage_name) == vals.end()) {
 890                                         return false;
 891                                 }
 892                         } else {
 893                                 return false;
 894                         }
 895                 }
 896         }
 897 
 898         if(!unit_abilities::filter_base_matches(cfg, res)) {
 899                 return false;
 900         }
 901 
 902         return true;
 903 }

 989 /**
 990  * Modifies the provided config by merging all base units into it.
 991  * The @a base_tree parameter is used solely for detecting and reporting
 992  * cycles of base units; it is no longer needed to prevent infinite loops.
 993  */
 994 void apply_base_unit(config& ut_cfg, config& all_types, std::vector<std::string>& base_tree)
 995 {
 996         // Get a list of base units to apply.
 997         std::vector<std::string> base_ids;
 998         for(config& base : ut_cfg.child_range("base_unit")) {
 999                 base_ids.push_back(base["id"]);
1000         }
1001 
1002         // Nothing to do.
1003         if(base_ids.empty()) {
1004                 return;
1005         }
1006 
1007         // Store the base ids for the help system.
1008         ut_cfg["base_ids"] = utils::join(base_ids);
1009 
1010         // Clear the base units (otherwise they could interfere with the merge).
1011         // This has the side-effect of breaking cycles, hence base_tree is
1012         // merely for error detection, not error recovery.
1013         ut_cfg.clear_children("base_unit");
1014 
1015         // Merge the base units, in order.
1016         for(const std::string& base_id : base_ids) {
1017                 // Detect recursion so the WML author is made aware of an error.
1018                 if(std::find(base_tree.begin(), base_tree.end(), base_id) != base_tree.end()) {
1019                         throw_base_unit_recursion_error(base_tree, base_id);
1020                 }
1021 
1022                 // Find the base unit.
1023                 config& base_cfg = find_unit_type_config(base_id, all_types);
1024 
1025                 // Make sure the base unit has had its base units accounted for.
1026                 base_tree.push_back(base_id);
1027 
1028                 apply_base_unit(base_cfg, all_types, base_tree);
1029 
1030                 base_tree.pop_back();
1031 
1032                 // Merge the base unit "under" our config.
1033                 ut_cfg.inherit_from(base_cfg);
1034         }
1035 }

1076 const boost::regex fai_identifier("[a-zA-Z_]+");
1077 
1078 template<typename MoveT>
1079 void patch_movetype(
1080                 MoveT& mt, const std::string& new_key, const std::string& formula_str, int default_val, bool replace)
1081 {
1082         config temp_cfg, original_cfg;
1083         mt.write(original_cfg);
1084 
1085         if(!replace && !original_cfg[new_key].blank()) {
1086                 // Don't replace if the key already exists in the config (even if empty).
1087                 return;
1088         }
1089 
1090         gui2::typed_formula<int> formula(formula_str);
1091         wfl::map_formula_callable original;
1092 
1093         boost::sregex_iterator m(formula_str.begin(), formula_str.end(), fai_identifier);
1094         for(const boost::sregex_iterator::value_type& p : std::make_pair(m, boost::sregex_iterator())) {
1095                 const std::string var_name = p.str();
1096 
1097                 wfl::variant val(original_cfg[var_name].to_int(default_val));
1098                 original.add(var_name, val);
1099         }
1100 
1101         temp_cfg[new_key] = formula(original);
1102         mt.merge(temp_cfg, true);
1103 }

1106 /**
1107  * Resets all data based on the provided config.
1108  * This includes some processing of the config, such as expanding base units.
1109  * A pointer to the config is stored, so the config must be persistent.
1110  */
1111 void unit_type_data::set_config(config& cfg)
1112 {
1113         DBG_UT << "unit_type_data::set_config, name: " << cfg["name"] << "\n";
1114 
1115         clear();
1116         unit_cfg_ = &cfg;
1117 
1118         for(const config& mt : cfg.child_range("movetype")) {
1119                 movement_types_.emplace(mt["name"].str(), movetype(mt));
1120 
1121                 gui2::dialogs::loading_screen::progress();
1122         }
1123 
1124         for(const config& r : cfg.child_range("race")) {
1125                 const unit_race race(r);
1126                 races_.emplace(race.id(), race);
1127 
1128                 gui2::dialogs::loading_screen::progress();
1129         }
1130 
1131         // Movetype resistance patching
1132         for(const config& r : cfg.child_range("resistance_defaults")) {
1133                 const std::string& dmg_type = r["id"];
1134                 config temp_cfg;
1135 
1136                 for(const config::attribute& attr : r.attribute_range()) {
1137                         const std::string& mt = attr.first;
1138

1140                                 continue;
1141                         }

1147                         for(movement_type_map::value_type& mt : movement_types_) {
1148                                 // Don't apply a default if a value is explicitly specified.
1149                                 if(r.has_attribute(mt.first)) {

1154                         }
1155                 }

1157

1159         for(const config& terrain : cfg.child_range("terrain_defaults")) {
1160                 const std::string& ter_type = terrain["id"];
1161                 config temp_cfg;
1162 
1163                 static const std::string terrain_info_tags[] {"movement", "vision", "jamming", "defense"};
1164

1166                         if(!terrain.has_child(tag)) {

1168                         }

1171

1173                                 const std::string& mt = attr.first;
1174 
1175                                 if(mt == "default" || movement_types_.find(mt) == movement_types_.end()) {
1176                                         continue;
1177                                 }
1178 
1179                                 if(tag == "defense") {
1180                                         patch_movetype(movement_types_[mt].get_defense(),  ter_type, attr.second, 100, true);
1181                                 } else if(tag == "vision") {
1182                                         patch_movetype(movement_types_[mt].get_vision(),   ter_type, attr.second, 99, true);
1183                                 } else if(tag == "movement") {
1184                                         patch_movetype(movement_types_[mt].get_movement(), ter_type, attr.second, 99, true);
1185                                 } else if(tag == "jamming") {
1186                                         patch_movetype(movement_types_[mt].get_jamming(),  ter_type, attr.second, 99, true);
1187                                 }
1188                         }
1189 
1190                         if(info.has_attribute("default")) {
1191                                 for(movement_type_map::value_type& mt : movement_types_) {
1192                                         // Don't apply a default if a value is explicitly specified.
1193                                         if(info.has_attribute(mt.first)) {
1194                                                 continue;
1195                                         }
1196 
1197                                         if(tag == "defense") {
1198                                                 patch_movetype(mt.second.get_defense(), ter_type,  info["default"], 100, false);
1199                                         } else if(tag == "vision") {
1200                                                 patch_movetype(mt.second.get_vision(), ter_type,   info["default"], 99, false);
1201                                         } else if(tag == "movement") {
1202                                                 patch_movetype(mt.second.get_movement(), ter_type, info["default"], 99, false);
1203                                         } else if(tag == "jamming") {
1204                                                 patch_movetype(mt.second.get_jamming(), ter_type,  info["default"], 99, false);
1205                                         }
1206                                 }
1207                         }
1208                 }
1209         }
1210 
1211         // Apply base units.
1212         for(config& ut : cfg.child_range("unit_type")) {
1213                 if(ut.has_child("base_unit")) {
1214                         // Derived units must specify a new id.
1215                         // (An error message will be emitted later if id is empty.)
1216                         const std::string id = ut["id"];
1217                         if(!id.empty()) {
1218                                 std::vector<std::string> base_tree(1, id);
1219                                 apply_base_unit(ut, cfg, base_tree);
1220 
1221                                 gui2::dialogs::loading_screen::progress();
1222                         }
1223                 }
1224         }
1225 
1226         // Handle inheritance and recording of unit types.
1227         for(config& ut : cfg.child_range("unit_type")) {
1228                 std::string id = ut["id"];
1229 
1230                 // Every type is required to have an id.
1231                 if(id.empty()) {
1232                         ERR_CF << "[unit_type] with empty id=, ignoring:\n" << ut.debug();
1233                         continue;
1234                 }
1235 
1236                 // Complete the gender-specific children of the config.
1237                 if(config& male_cfg = ut.child("male")) {
1238                         fill_unit_sub_type(male_cfg, ut, true);
1239                         handle_variations(male_cfg);
1240                 }
1241 
1242                 if(config& female_cfg = ut.child("female")) {
1243                         fill_unit_sub_type(female_cfg, ut, true);
1244                         handle_variations(female_cfg);
1245                 }
1246 
1247                 // Complete the variation-defining children of the config.
1248                 handle_variations(ut);
1249 
1250                 // Record this unit type.
1251                 if(types_.emplace(id, unit_type(ut)).second) {
1252                         LOG_CONFIG << "added " << id << " to unit_type list (unit_type_data.unit_types)\n";
1253                 } else {
1254                         ERR_CF << "Multiple [unit_type]s with id=" << id << " encountered." << std::endl;
1255                 }
1256 
1257                 gui2::dialogs::loading_screen::progress();
1258         }
1259 
1260         // Build all unit types. (This was not done within the loop for performance.)
1261         build_all(unit_type::CREATED);
1262 
1263         // Suppress some unit types (presumably used as base units) from the help.
1264         if(const config& hide_help = cfg.child("hide_help")) {
1265                 hide_help_all_ = hide_help["all"].to_bool();
1266                 read_hide_help(hide_help);
1267         }
1268 }

1319 void unit_type_data::build_all(unit_type::BUILD_STATUS status)
1320 {
1321         // Nothing to do if already built to the requested level.
1322         if(status <= build_status_) {
1323                 return;
1324         }
1325 
1326         assert(unit_cfg_ != nullptr);
1327 
1328         for(unit_type_map::iterator u = types_.begin(), u_end = types_.end(); u != u_end; ++u) {
1329                 build_unit_type(u->second, status);
1330 
1331                 gui2::dialogs::loading_screen::progress();
1332         }
1333 
1334         // Handle [advancefrom] (once) after building to (at least) the CREATED level.
1335         // (Currently, this could be simply a test for build_status_ == NOT_BUILT,
1336         // but to guard against future changes, use a more thorough test.)
1337         if(build_status_ < unit_type::CREATED && unit_type::CREATED <= status) {
1338                 for(unit_type_map::iterator u = types_.begin(), u_end = types_.end(); u != u_end; ++u) {
1339                         add_advancement(u->second);
1340                 }
1341         }
1342 
1343         build_status_ = status;
1344 }

1346 void unit_type_data::read_hide_help(const config& cfg)
1347 {
1348         if(!cfg) {
1349                 return;
1350         }
1351 
1352         hide_help_race_.emplace_back();
1353         hide_help_type_.emplace_back();
1354 
1355         std::vector<std::string> races = utils::split(cfg["race"]);
1356         hide_help_race_.back().insert(races.begin(), races.end());
1357 
1358         std::vector<std::string> types = utils::split(cfg["type"]);
1359         hide_help_type_.back().insert(types.begin(), types.end());
1360 
1361         std::vector<std::string> trees = utils::split(cfg["type_adv_tree"]);
1362         hide_help_type_.back().insert(trees.begin(), trees.end());
1363 
1364         for(const std::string& t_id : trees) {
1365                 unit_type_map::iterator ut = types_.find(t_id);
1366 
1367                 if(ut != types_.end()) {
1368                         std::set<std::string> adv_tree = ut->second.advancement_tree();
1369                         hide_help_type_.back().insert(adv_tree.begin(), adv_tree.end());
1370                 }
1371         }
1372 
1373         // We recursively call all the imbricated [not] tags
1374         read_hide_help(cfg.child("not"));
1375 }

1377 bool unit_type_data::hide_help(const std::string& type, const std::string& race) const
1378 {
1379         bool res = hide_help_all_;
1380         int lvl = hide_help_all_ ? 1 : 0; // first level is covered by 'all=yes'
1381 
1382         // supposed to be equal but let's be cautious
1383         int lvl_nb = std::min(hide_help_race_.size(), hide_help_type_.size());
1384 
1385         for(; lvl < lvl_nb; ++lvl) {
1386                 if(hide_help_race_[lvl].count(race) || hide_help_type_[lvl].count(type)) {
1387                         res = !res; // each level is a [not]
1388                 }
1389         }
1390 
1391         return res;
1392 }

1394 void unit_type_data::add_advancement(unit_type& to_unit) const
1395 {
1396         const config& cfg = to_unit.get_cfg();
1397 
1398         for(const config& af : cfg.child_range("advancefrom")) {
1399                 const std::string& from = af["unit"];
1400                 int xp = af["experience"];
1401 
1402                 unit_type_data::unit_type_map::iterator from_unit = types_.find(from);
1403 
1404                 if(from_unit == types_.end()) {
1405                         std::ostringstream msg;
1406                         msg << "unit type '" << from << "' not found when resolving [advancefrom] tag for '" << to_unit.log_id()
1407                                 << "'";
1408                         throw config::error(msg.str());
1409                 }
1410 
1411                 // Fix up advance_from references
1412                 from_unit->second.add_advancement(to_unit, xp);
1413 
1414                 DBG_UT << "Added advancement ([advancefrom]) from " << from << " to " << to_unit.log_id() << "\n";
1415         }
1416 }

1424 void unit_type::check_id(std::string& id)
1425 {
1426         assert(!id.empty());
1427 
1428         // We don't allow leading whitepaces.
1429         if(id[0] == ' ') {
1430                 throw error("Found unit type id with a leading whitespace \"" + id + "\"");
1431         }
1432 
1433         bool gave_warning = false;
1434 
1435         for(size_t pos = 0; pos < id.size(); ++pos) {
1436                 const char c = id[pos];
1437                 const bool valid =
1438                         c == '_' ||
1439                         c == ' ' ||
1440                         (c >= 'a' && c <= 'z') ||
1441                         (c >= 'A' && c <= 'Z') ||
1442                         (c >= '0' && c <= '9');
1443 
1444                 if(!valid) {
1445                         if(!gave_warning) {
1446                                 ERR_UT << "Found unit type id with invalid chracters: \"" << id << "\"\n";
1447                                 gave_warning = true;
1448                         }
1449 
1450                         id[pos] = '_';
1451                 }
1452         }
1453 }

1457 void adjust_profile(std::string& profile)
1458 {
1459         // Create a temp copy
1460         std::string temp = profile;
1461 
1462         static const std::string path_adjust = "/transparent";
1463         const std::string::size_type offset = profile.find_last_of('/', profile.find('~'));
1464 
1465         // If the path already refers to /transparent...
1466         if(profile.find(path_adjust) != std::string::npos && offset != std::string::npos) {
1467                 if(!image::locator(profile).file_exists()) {
1468                         profile.replace(profile.find(path_adjust), path_adjust.length(), "");
1469                 }
1470 
1471                 return;
1472         }
1473 
1474         // else, check for the file with /transparent appended...
1475         offset != std::string::npos ? temp.insert(offset, path_adjust) : temp = path_adjust + temp;
1476 
1477         // and use that path if it exists.
1478         if(image::locator(temp).file_exists()) {
1479                 profile = temp;
1480         }
1481 }
