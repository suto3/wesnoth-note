  43 static unsigned int sdl_get_mods()
  44 {
  45         unsigned int mods;
  46         mods = SDL_GetModState();
  47 
  48         mods &= ~KMOD_NUM;
  49         mods &= ~KMOD_CAPS;
  50         mods &= ~KMOD_MODE;
  51 
  52         // save the matching for checking right vs left keys
  53         if (mods & KMOD_SHIFT)
  54                 mods |= KMOD_SHIFT;
  55 
  56         if (mods & KMOD_CTRL)
  57                 mods |= KMOD_CTRL;
  58 
  59         if (mods & KMOD_ALT)
  60                 mods |= KMOD_ALT;
  61 
  62         if (mods & KMOD_GUI)
  63         mods |= KMOD_GUI;
  64 
  65         return mods;
  66 }

 103 bool hotkey_base::bindings_equal(hotkey_ptr other)
 104 {
 105         if (other == hotkey_ptr()) {
 106                 return false;
 107         }
 108 
 109         hk_scopes scopematch = hotkey::get_hotkey_command(get_command()).scope
 110                         & hotkey::get_hotkey_command(other->get_command()).scope;
 111 
 112         if (scopematch.none()) {
 113                 return false;
 114         }
 115 
 116         return mod_ == other->mod_ && bindings_equal_helper(other);
 117 }

 142 hotkey_ptr create_hotkey(const std::string &id, const SDL_Event &event)
 143 {
 144         hotkey_ptr base = hotkey_ptr(new hotkey_void);
 145         const hotkey_command& command = get_hotkey_command(id);
 146         unsigned mods = sdl_get_mods();
 147 
 148         switch (event.type) {
 149         case SDL_KEYUP: {
 150                 if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI || CKey::is_uncomposable(event.key) || command.toggle) {
 151                         hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 152                         base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 153                         SDL_Keycode code;
 154                         code = event.key.keysym.sym;
 155                         keyboard->set_keycode(code);
 156                         keyboard->set_text(SDL_GetKeyName(event.key.keysym.sym));
 157                 }
 158         }
 159                 break;
 160         case SDL_TEXTINPUT: {
 161                 if(command.toggle) {
 162                         return nullptr;
 163                 }
 164                 hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 165                 base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 166                 std::string text =  std::string(event.text.text);
 167                 keyboard->set_text(text);
 168                 if(text == ":" || text == "`") {
 169                         mods = mods & ~KMOD_SHIFT;
 170                 }
 171         }
 172                 break;
 173         case SDL_MOUSEBUTTONUP: {
 174                 hotkey_mouse_ptr mouse(new hotkey_mouse());
 175                 base = std::dynamic_pointer_cast<hotkey_base>(mouse);
 176                 mouse->set_button(event.button.button);
 177                 break;
 178         }
 179         default:
 180                 ERR_G<< "Trying to bind an unknown event type:" << event.type << "\n";
 181                 break;
 182         }
 183 
 184         base->set_mods(mods);
 185         base->set_command(id);
 186         base->unset_default();
 187 
 188         return base;
 189 }

 255 bool hotkey_mouse::matches_helper(const SDL_Event &event) const
 256 {
 257         if (event.type != SDL_MOUSEBUTTONUP && event.type != SDL_MOUSEBUTTONDOWN) {
 258                 return false;
 259         }
 260 
 261         unsigned int mods = sdl_get_mods();
 262         if ((mods != mod_)) {
 263                 return false;
 264         }
 265 
 266         if (event.button.button != button_) {
 267                 return false;
 268         }
 269 
 270         return true;
 271 }

 291 bool hotkey_keyboard::matches_helper(const SDL_Event &event) const
 292 {
 293         unsigned int mods = sdl_get_mods();
 294         const hotkey_command& command = get_hotkey_command(get_command());
 295 
 296         if ((event.type == SDL_KEYDOWN || event.type == SDL_KEYUP) &&
 297                         (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI ||
 298                         command.toggle || CKey::is_uncomposable(event.key))) {
 299                 return event.key.keysym.sym == keycode_ && mods == mod_;
 300         }
 301 
 302         if (event.type == SDL_TEXTINPUT && !command.toggle) {
 303                 std::string text = std::string(event.text.text);
 304                 boost::algorithm::to_lower(text);
 305                 if(text == ":" || text == "`") {
 306                         mods = mods & ~KMOD_SHIFT;
 307                 }
 308                 return text_ == text && utf8::size(utf8::string(event.text.text)) == 1 && mods == mod_;
 309         }
 310 
 311         return false;
 312 }

 361 void add_hotkey(const hotkey_ptr item)
 362 {
 363 
 364         if (item == hotkey_ptr()) {
 365                 return;
 366         }
 367 
 368         scope_changer scope_ch;
 369         set_active_scopes(hotkey::get_hotkey_command(item->get_command()).scope);
 370 
 371         if (!hotkeys_.empty()) {
 372                 hotkeys_.erase(
 373                         std::remove_if(hotkeys_.begin(), hotkeys_.end(), [item](const hotkey::hotkey_ptr& hk) { return hk->bindings_equal(item); }),
 374                         hotkeys_.end());
 375         }
 376 
 377         hotkeys_.push_back(item);
 378 
 379 }

 409 void load_hotkeys(const config& cfg, bool set_as_default)
 410 {
 411         for (const config &hk : cfg.child_range("hotkey")) {
 412 
 413                 hotkey_ptr item = load_from_config(hk);
 414                 if (!set_as_default) {
 415                         item->unset_default();
 416                 }
 417 
 418                 if (!item->null()) {
 419                         add_hotkey(item);
 420                 }
 421         }
 422 
 423         if (set_as_default) {
 424                 default_hotkey_cfg_ = cfg;
 425         }
 426 }

 456 std::string get_names(const std::string& id)
 457 {
 458         // Names are used in places like the hot-key preferences menu
 459         std::vector<std::string> names;
 460         for (const hotkey::hotkey_ptr item : hotkeys_) {
 461                 if (item->get_command() == id && !item->null() && !item->is_disabled()) {
 462                         names.push_back(item->get_name());
 463                 }
 464         }
 465 
 466         // These are hard-coded, non-rebindable hotkeys
 467         if (id == "quit") {
 468                 names.push_back("escape");
 469         }
 470         else if (id == "quit-to-desktop") {
 471 #ifdef __APPLE__
 472                 names.push_back("cmd+q");
 473 #else
 474                 names.push_back("alt+F4");
 475 #endif
 476         }
 477 
 478         return boost::algorithm::join(names, ", ");
 479 }

 481 bool is_hotkeyable_event(const SDL_Event &event) {
 482 
 483         if (event.type == SDL_JOYBUTTONUP ||
 484                         event.type == SDL_JOYHATMOTION ||
 485                         event.type == SDL_MOUSEBUTTONUP) {
 486                 return true;
 487         }
 488 
 489         unsigned mods = sdl_get_mods();
 490 
 491         if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI) {
 492                 return event.type == SDL_KEYUP;
 493         } else {
 494                 return event.type == SDL_TEXTINPUT || event.type  == SDL_KEYUP;
 495         }
 496 }
