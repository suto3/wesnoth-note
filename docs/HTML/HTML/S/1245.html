  73 void highlighter::set_mouseover_hex(const map_location& hex)
  74 {
  75         clear();
  76 
  77         if(!hex.valid()) {
  78                 return;
  79         }
  80 
  81         real_map ensure_real_map;
  82         mouseover_hex_ = hex;
  83         //if we're right over a unit, just highlight all of this unit's actions
  84         unit_map::iterator it = get_unit_map().find(hex);
  85         if(it != get_unit_map().end()) {
  86                 selection_candidate_ = it.get_shared_ptr();
  87 
  88                 if(resources::gameboard->get_team(it->side()).get_side_actions()->unit_has_actions(*it)) {
  89                         owner_unit_ = it.get_shared_ptr();
  90                 }
  91 
  92                 //commented code below is to also select the first action of this unit as
  93                 //the main highlight; it doesn't fit too well in the UI
  94 //              side_actions::iterator action_it = side_actions_->find_first_action_of(*it);
  95 //              if(action_it != side_actions_->end()) {
  96 //                      main_highlight_ = *action_it;
  97 //              }
  98         }
  99 
 100         //Set the execution/deletion/bump targets.
 101         if(owner_unit_) {
 102                 side_actions::iterator itor = side_actions_->find_first_action_of(*owner_unit_);
 103                 if(itor != side_actions_->end()) {
 104                         selected_action_ = *itor;
 105                 }
 106         }
 107 
 108         //Overwrite the above selected_action_ if we find a better one
 109         if(side_actions_->empty()) {
 110                 return;
 111         }
 112         for(action_ptr act : boost::adaptors::reverse(*side_actions_)) {
 113                 /**@todo "is_numbering_hex" is not the "correct" criterion by which to
 114                  * select the hightlighted/selected action. It's just convenient for me
 115                  * to use at the moment since it happens to coincide with the "correct"
 116                  * criterion, which is to use find_main_highlight.*/
 117                 if(act->is_numbering_hex(hex)) {
 118                         selected_action_ = act;
 119                         break;
 120                 }
 121         }
 122 }

 133 void highlighter::highlight()
 134 {
 135         //Find main action to highlight if any, as well as owner unit
 136         find_main_highlight();
 137 
 138         if(action_ptr main = main_highlight_.lock()) {
 139                 //Highlight main highlight
 140                 highlight_main_visitor hm_visitor(*this);
 141                 main->accept(hm_visitor);
 142         }
 143 
 144         if(owner_unit_) {
 145                 //Find secondary actions to highlight
 146                 find_secondary_highlights();
 147 
 148                 //Make sure owner unit is the only one displayed in its hex
 149                 resources::screen->add_exclusive_draw(owner_unit_->get_location(), *owner_unit_);
 150                 exclusive_display_hexes_.insert(owner_unit_->get_location());
 151 
 152                 if(!secondary_highlights_.empty()) {
 153                         //Highlight secondary highlights
 154                         highlight_secondary_visitor hs_visitor(*this);
 155                         for(weak_action_ptr weak : secondary_highlights_) {
 156                                 if(action_ptr action = weak.lock()) {
 157                                         action->accept(hs_visitor);
 158                                 }
 159                         }
 160                 }
 161         }
 162 }

 164 void highlighter::unhighlight()
 165 {
 166         unhighlight_visitor uh_visitor(*this);
 167 
 168         //unhighlight main highlight
 169         if(action_ptr main = main_highlight_.lock()) {
 170                 main->accept(uh_visitor);
 171         }
 172 
 173         //unhighlight secondary highlights
 174         for(weak_action_ptr weak : secondary_highlights_) {
 175                 if(action_ptr action = weak.lock()) {
 176                         action->accept(uh_visitor);
 177                 }
 178         }
 179 
 180         //unhide other units if needed
 181         for(map_location hex : exclusive_display_hexes_) {
 182                 resources::screen->remove_exclusive_draw(hex);
 183         }
 184         exclusive_display_hexes_.clear();
 185 }

 187 void highlighter::last_action_redraw(move_ptr move)
 188 {
 189         //Last action with a fake unit always gets normal appearance
 190         if(move->get_fake_unit()) {
 191                 side_actions& sa = *resources::gameboard->teams().at(move->team_index()).get_side_actions().get();
 192 
 193                 // Units with planned actions may have been killed in the previous turn before all actions were completed.
 194                 // In these cases, remove these planned actions for any invalid units and do not redraw anything.
 195                 if (move->get_unit() == nullptr)
 196                 {
 197                         // Note: the planned actions seem to only get removed from the screen when
 198                         // a redraw is triggered by the mouse cursor moving over them.
 199                         for (side_actions::iterator iterator = sa.begin(); iterator < sa.end(); iterator++)
 200                         {
 201                                 if (iterator->get()->get_unit() == nullptr)
 202                                         sa.remove_action (iterator);
 203                         }
 204 
 205                         return;
 206                 }
 207 
 208                 side_actions::iterator last_action = sa.find_last_action_of(*(move->get_unit()));
 209                 side_actions::iterator second_to_last_action = last_action != sa.end() && last_action != sa.begin() ? last_action - 1 : sa.end();
 210 
 211                 bool this_is_last_action = last_action != sa.end() && move == *last_action;
 212                 bool last_action_has_fake_unit = last_action != sa.end() && (*last_action)->get_fake_unit();
 213                 bool this_is_second_to_last_action = (second_to_last_action != sa.end() && move == *second_to_last_action);
 214 
 215                 if(this_is_last_action || (this_is_second_to_last_action && !last_action_has_fake_unit)) {
 216                         move->get_fake_unit()->anim_comp().set_standing(true);
 217                 }
 218         }
 219 }

 221 void highlighter::find_main_highlight()
 222 {
 223         // Even if we already found an owner_unit_ in the mouseover hex,
 224         // action destination hexes usually take priority over that
 225         assert(main_highlight_.expired());
 226         //@todo re-enable the following assert once I find out what happends to
 227         // viewing side assignments after victory
 228         //assert(side_actions_->team_index() == resources::screen->viewing_team());
 229 
 230         main_highlight_ = find_action_at(mouseover_hex_);
 231         if(action_ptr main = main_highlight_.lock()) {
 232                 owner_unit_ = main->get_unit();
 233         }
 234 }

 236 void highlighter::find_secondary_highlights()
 237 {
 238         assert(owner_unit_);
 239         assert(secondary_highlights_.empty());
 240 
 241         if(owner_unit_ == nullptr) {
 242                 return;
 243         }
 244 
 245         // List all the actions of owner_unit_
 246         std::deque<action_ptr> actions = find_actions_of(*owner_unit_);
 247 
 248         // Remove main_highlight_ if present
 249         actions.erase(std::remove(actions.begin(), actions.end(), main_highlight_.lock()), actions.end());
 250 
 251         // Copy in secondary_highlights_
 252         std::copy(actions.begin(), actions.end(), std::back_inserter(secondary_highlights_));
 253 }

 287 void highlighter::highlight_main_visitor::visit(move_ptr move)
 288 {
 289         if(move->get_arrow()) {
 290                 move->set_arrow_brightness(move::ARROW_BRIGHTNESS_FOCUS);
 291         }
 292         if(move->get_fake_unit()) {
 293                 ///@todo find some highlight animation
 294                 move->get_fake_unit()->anim_comp().set_ghosted(true);
 295                 //Make sure the fake unit is the only one displayed in its hex
 296                 resources::screen->add_exclusive_draw(move->get_fake_unit()->get_location(), *move->get_fake_unit());
 297                 highlighter_.exclusive_display_hexes_.insert(move->get_fake_unit()->get_location());
 298 
 299                 highlighter_.last_action_redraw(move);
 300         }
 301 }
