 160 void widget::hide(bool value)
 161 {
 162         if (value) {
 163                 if ((state_ == DIRTY || state_ == DRAWN) && !hidden_override_)
 164                         bg_restore();
 165                 state_ = HIDDEN;
 166         } else if (state_ == HIDDEN) {
 167                 state_ = DRAWN;
 168                 if (!hidden_override_) {
 169                         bg_update();
 170                         set_dirty(true);
 171                 }
 172         }
 173 }
 174 
 175 void widget::hide_override(bool value) {
 176         if (hidden_override_ != value) {
 177                 hidden_override_ = value;
 178                 if (state_ == DIRTY || state_ == DRAWN) {
 179                         if (value) {
 180                                 bg_restore();
 181                         } else {
 182                                 bg_update();
 183                                 set_dirty(true);
 184                         }
 185                 }
 186         }
 187 }

 277 void widget::draw()
 278 {
 279         if (hidden() || !dirty())
 280                 return;
 281 
 282         bg_restore();
 283 
 284         clip_rect_setter clipper(video().getSurface(), &clip_rect_, clip_);
 285 
 286         draw_contents();
 287 
 288         set_dirty(false);
 289 }

 317 void widget::process_help_string(int mousex, int mousey)
 318 {
 319         if (!hidden() && sdl::point_in_rect(mousex, mousey, rect_)) {
 320                 if(help_string_ == 0 && !help_text_.empty()) {
 321                         //std::cerr << "setting help string to '" << help_text_ << "'\n";
 322                         help_string_ = video().set_help_string(help_text_);
 323                 }
 324         } else if(help_string_ > 0) {
 325                 video().clear_help_string(help_string_);
 326                 help_string_ = 0;
 327         }
 328 }
 329 
 330 void widget::process_tooltip_string(int mousex, int mousey)
 331 {
 332         if (!hidden() && sdl::point_in_rect(mousex, mousey, rect_)) {
 333                 if (!tooltip_text_.empty())
 334                         tooltips::add_tooltip(rect_, tooltip_text_ );
 335         }
 336 }

 345 void widget::handle_window_event(SDL_Event const &event) {
 346         if (event.type == SDL_WINDOWEVENT) {
 347                 switch (event.window.event) {
 348                 case SDL_WINDOWEVENT_RESIZED:
 349                 case SDL_WINDOWEVENT_RESTORED:
 350                 case SDL_WINDOWEVENT_SHOWN:
 351                 case SDL_WINDOWEVENT_EXPOSED:
 352                         set_dirty();
 353                 }
 354         }
