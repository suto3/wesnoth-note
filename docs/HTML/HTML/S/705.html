 107 static void add_widget(gui2::grid& grid
 108                 , gui2::widget* widget
 109                 , const std::string& id
 110                 , const unsigned row
 111                 , const unsigned column)
 112 {
 113         BOOST_REQUIRE_NE(widget, static_cast<gui2::widget*>(nullptr));
 114 
 115         widget->set_id(id);
 116         grid.set_child(widget
 117                         , row
 118                         , column
 119                         , gui2::grid::VERTICAL_GROW_SEND_TO_CLIENT
 120                                 | gui2::grid::HORIZONTAL_GROW_SEND_TO_CLIENT
 121                         , 0);
 122 }

 124 template<class T>
 125 static void test_control(T&& control)
 126 {
 127         {
 128                 gui2::iteration::iterator< gui2::iteration::policy::order::top_down<
 129                                 true
 130                                 , true
 131                                 , true> >
 132                         iterator(control);
 133 
 134                 /***** INITIAL STATE *****/
 135 
 136                 BOOST_CHECK_EQUAL(iterator.at_end(), false);
 137 
 138                 BOOST_CHECK_EQUAL(&*iterator, &control);
 139 
 140                 /***** POST END *****/
 141 
 142                 BOOST_CHECK_EQUAL(iterator.next(), false);
 143 
 144                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 145 
 146         }
 147         {
 148                 gui2::iteration::iterator< gui2::iteration::policy::order::top_down<
 149                                 false
 150                                 , true
 151                                 , true> >
 152                         iterator(control);
 153 
 154                 /***** INITIAL STATE *****/
 155 
 156                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 157         }
 158 
 159         {
 160                 gui2::iteration::iterator<gui2::iteration::policy::order::bottom_up<true, true, true> > iterator(control);
 161                 BOOST_CHECK_EQUAL(iterator.at_end(), false);
 162         }
 163         {
 164                 gui2::iteration::iterator<gui2::iteration::policy::order::bottom_up<false, false, false> > iterator(control);
 165                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 166         }
 167 }
