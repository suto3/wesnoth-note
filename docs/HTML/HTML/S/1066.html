 107         fake_unit_ptr create_fake_unit(const vconfig& cfg)
 108         {
 109                 std::string type = cfg["type"];
 110                 std::string variation = cfg["variation"];
 111                 std::string img_mods = cfg["image_mods"];
 112 
 113                 size_t side_num = cfg["side"].to_int(1);
 114                 if (!resources::gameboard->has_team(side_num)) {
 115                         side_num = 1;
 116                 }
 117 
 118                 unit_race::GENDER gender = string_gender(cfg["gender"]);
 119                 const unit_type *ut = unit_types.find(type);
 120                 if (!ut) return fake_unit_ptr();
 121                 fake_unit_ptr fake = fake_unit_ptr(unit_ptr(new unit(*ut, side_num, false, gender)));
 122 
 123                 if(!variation.empty()) {
 124                         config mod;
 125                         config &effect = mod.add_child("effect");
 126                         effect["apply_to"] = "variation";
 127                         effect["name"] = variation;
 128                         fake->add_modification("variation",mod);
 129                 }
 130 
 131                 if(!img_mods.empty()) {
 132                         config mod;
 133                         config &effect = mod.add_child("effect");
 134                         effect["apply_to"] = "image_mod";
 135                         effect["add"] = img_mods;
 136                         fake->add_modification("image_mod",mod);
 137                 }
 138 
 139                 return fake;
 140         }

 142         std::vector<map_location> fake_unit_path(const unit& fake_unit, const std::vector<std::string>& xvals, const std::vector<std::string>& yvals)
 143         {
 144                 const gamemap *game_map = & resources::gameboard->map();
 145                 std::vector<map_location> path;
 146                 map_location src;
 147                 map_location dst;
 148                 for(size_t i = 0; i != std::min(xvals.size(),yvals.size()); ++i) {
 149                         if(i==0){
 150                                 try {
 151                                         src.set_wml_x(std::stoi(xvals[i]));
 152                                         src.set_wml_y(std::stoi(yvals[i]));
 153                                 } catch(std::invalid_argument) {
 154                                         ERR_CF << "Invalid move_unit_fake source: " << xvals[i] << ", " << yvals[i] << '\n';
 155                                         continue;
 156                                 }
 157                                 if (!game_map->on_board(src)) {
 158                                         ERR_CF << "Invalid move_unit_fake source: " << src << '\n';
 159                                         break;
 160                                 }
 161                                 path.push_back(src);
 162                                 continue;
 163                         }
 164                         pathfind::shortest_path_calculator calc(fake_unit,
 165                                         resources::gameboard->get_team(fake_unit.side()),
 166                                         resources::gameboard->teams(),
 167                                         *game_map);
 168 
 169                         try {
 170                                 dst.set_wml_x(std::stoi(xvals[i]));
 171                                 dst.set_wml_y(std::stoi(yvals[i]));
 172                         } catch(std::invalid_argument) {
 173                                 ERR_CF << "Invalid move_unit_fake destination: " << xvals[i] << ", " << yvals[i] << '\n';
 174                         }
 175                         if (!game_map->on_board(dst)) {
 176                                 ERR_CF << "Invalid move_unit_fake destination: " << dst << '\n';
 177                                 break;
 178                         }
 179 
 180                         pathfind::plain_route route = pathfind::a_star_search(src, dst, 10000, calc,
 181                                 game_map->w(), game_map->h());
 182 
 183                         if (route.steps.empty()) {
 184                                 WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring complexities" << std::endl;
 185                                 pathfind::emergency_path_calculator emergency_calc(fake_unit, *game_map);
 186 
 187                                 route = pathfind::a_star_search(src, dst, 10000, emergency_calc,
 188                                                 game_map->w(), game_map->h());
 189                                 if(route.steps.empty()) {
 190                                         // This would occur when trying to do a MUF of a unit
 191                                         // over locations which are unreachable to it (infinite movement
 192                                         // costs). This really cannot fail.
 193                                         WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring terrain" << std::endl;
 194                                         pathfind::dummy_path_calculator dummy_calc(fake_unit, *game_map);
 195                                         route = a_star_search(src, dst, 10000, dummy_calc, game_map->w(), game_map->h());
 196                                         assert(!route.steps.empty());
 197                                 }
 198                         }
 199                         // we add this section to the end of the complete path
 200                         // skipping section's head because already included
 201                         // by the previous iteration
 202                         path.insert(path.end(),
 203                                         route.steps.begin()+1, route.steps.end());
 204 
 205                         src = dst;
 206                 }
 207                 return path;
 208         }

 271 // This tag exposes part of the code path used to handle [command]'s in replays
 272 // This allows to perform scripting in WML that will use the same code path as player actions, for example.
 273 WML_HANDLER_FUNCTION(do_command,, cfg)
 274 {
 275         // Doing this in a whiteboard applied context will cause bugs
 276         // Note that even though game_events::wml_event_pump() will always apply the real unit map
 277         // It is still possible get a wml commands to run in a whiteboard applied context
 278         // With the theme_items lua callbacks
 279         if(resources::whiteboard->has_planned_unit_map())
 280         {
 281                 ERR_NG << "[do_command] called while whiteboard is applied, ignoring" << std::endl;
 282                 return;
 283         }
 284 
 285         static const std::set<std::string> allowed_tags {"attack", "move", "recruit", "recall", "disband", "fire_event", "lua_ai"};
 286 
 287         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 288         const bool is_unsynced_too_early = resources::gamedata->phase() != game_data::PLAY;
 289         const bool is_unsynced = synced_context::get_synced_state() == synced_context::UNSYNCED;
 290         if(is_too_early)
 291         {
 292                 ERR_NG << "[do_command] called too early, only allowed at START or later" << std::endl;
 293                 return;
 294         }
 295         if(is_unsynced && resources::controller->is_lingering())
 296         {
 297                 ERR_NG << "[do_command] cannot be used in linger mode" << std::endl;
 298                 return;
 299         }
 300         if(is_unsynced && resources::controller->gamestate().init_side_done())
 301         {
 302                 ERR_NG << "[do_command] cannot be used before the turn has started" << std::endl;
 303                 return;
 304         }
 305         if(is_unsynced && is_unsynced_too_early)
 306         {
 307                 ERR_NG << "[do_command] called too early" << std::endl;
 308                 return;
 309         }
 310         if(is_unsynced && events::commands_disabled)
 311         {
 312                 ERR_NG << "[do_command] cannot invoke synced commands while commands are blocked" << std::endl;
 313                 return;
 314         }
 315         if(is_unsynced && !resources::controller->current_team().is_local())
 316         {
 317                 ERR_NG << "[do_command] can only be used from clients that control the currently playing side" << std::endl;
 318                 return;
 319         }
 320         for(vconfig::all_children_iterator i = cfg.ordered_begin(); i != cfg.ordered_end(); ++i)
 321         {
 322                 if(allowed_tags.find( i.get_key()) == allowed_tags.end()) {
 323                         ERR_NG << "unsupported tag [" << i.get_key() << "] in [do_command]" << std::endl;
 324                         std::stringstream o;
 325                         std::copy(allowed_tags.begin(), allowed_tags.end(), std::ostream_iterator<std::string>(o, " "));
 326                         ERR_NG << "allowed tags: " << o.str() << std::endl;
 327                         continue;
 328                 }
 329                 // TODO: afaik run_in_synced_context_if_not_already thows exceptions when the executed action end the scenario or the turn.
 330                 //       This could cause problems, specially when its unclear whether that excetion is caught by lua or not...
 331 
 332                 //Note that this fires related events and everthing else that also happen normally.
 333                 //have to watch out with the undo stack, therefore forbid [auto_shroud] and [update_shroud] here...
 334                 synced_context::run_in_synced_context_if_not_already(
 335                         /*commandname*/ i.get_key(),
 336                         /*data*/ i.get_child().get_parsed_config(),
 337                         /*use_undo*/ true,
 338                         /*show*/ true,
 339                         /*error_handler*/ &on_replay_error
 340                 );
 341         }
 342 }

 351 WML_HANDLER_FUNCTION(modify_turns,, cfg)
 352 {
 353         config::attribute_value value = cfg["value"];
 354         std::string add = cfg["add"];
 355         config::attribute_value current = cfg["current"];
 356         tod_manager& tod_man = *resources::tod_manager;
 357         if(!add.empty()) {
 358                 tod_man.modify_turns_by_wml(add);
 359         } else if(!value.empty()) {
 360                 tod_man.set_number_of_turns_by_wml(value.to_int(-1));
 361         }
 362         // change current turn only after applying mods
 363         if(!current.empty()) {
 364                 const unsigned int current_turn_number = tod_man.turn();
 365                 int new_turn_number = current.to_int(current_turn_number);
 366                 const unsigned int new_turn_number_u = static_cast<unsigned int>(new_turn_number);
 367                 if(new_turn_number_u < 1 || (new_turn_number > tod_man.number_of_turns() && tod_man.number_of_turns() != -1)) {
 368                         ERR_NG << "attempted to change current turn number to one out of range (" << new_turn_number << ")" << std::endl;
 369                 } else if(new_turn_number_u != current_turn_number) {
 370                         tod_man.set_turn_by_wml(new_turn_number_u, resources::gamedata);
 371                         resources::screen->new_turn();
 372                 }
 373         }
 374 }

 400 WML_HANDLER_FUNCTION(move_units_fake,, cfg)
 401 {
 402         events::command_disabler command_disabler;
 403         LOG_NG << "Processing [move_units_fake]\n";
 404 
 405         const vconfig::child_list unit_cfgs = cfg.get_children("fake_unit");
 406         size_t num_units = unit_cfgs.size();
 407         std::vector<fake_unit_ptr > units;
 408         units.reserve(num_units);
 409         std::vector<std::vector<map_location> > paths;
 410         paths.reserve(num_units);
 411 
 412         LOG_NG << "Moving " << num_units << " units\n";
 413 
 414         size_t longest_path = 0;
 415 
 416         for (const vconfig& config : unit_cfgs) {
 417                 const std::vector<std::string> xvals = utils::split(config["x"]);
 418                 const std::vector<std::string> yvals = utils::split(config["y"]);
 419                 int skip_steps = config["skip_steps"];
 420                 fake_unit_ptr u = create_fake_unit(config);
 421                 units.push_back(u);
 422                 paths.push_back(fake_unit_path(*u, xvals, yvals));
 423                 if(skip_steps > 0)
 424                         paths.back().insert(paths.back().begin(), skip_steps, paths.back().front());
 425                 longest_path = std::max(longest_path, paths.back().size());
 426                 DBG_NG << "Path " << paths.size() - 1 << " has length " << paths.back().size() << '\n';
 427 
 428                 u->set_location(paths.back().front());
 429                 u.place_on_fake_unit_manager(resources::fake_units);
 430         }
 431 
 432         LOG_NG << "Units placed, longest path is " << longest_path << " long\n";
 433 
 434         std::vector<map_location> path_step(2);
 435         path_step.resize(2);
 436         for(size_t step = 1; step < longest_path; ++step) {
 437                 DBG_NG << "Doing step " << step << "...\n";
 438                 for(size_t un = 0; un < num_units; ++un) {
 439                         if(step >= paths[un].size() || paths[un][step - 1] == paths[un][step])
 440                                 continue;
 441                         DBG_NG << "Moving unit " << un << ", doing step " << step << '\n';
 442                         path_step[0] = paths[un][step - 1];
 443                         path_step[1] = paths[un][step];
 444                         unit_display::move_unit(path_step, units[un].get_unit_ptr());
 445                         units[un]->set_location(path_step[1]);
 446                         units[un]->anim_comp().set_standing();
 447                 }
 448         }
 449 
 450         LOG_NG << "Units moved\n";
 451 }

 453 /// If we should recall units that match a certain description.
 454 // If you change attributes specific to [recall] (that is, not a Standard Unit Filter)
 455 // be sure to update data/lua/wml_tag, auto_recall feature for [role] to reflect your changes.
 456 WML_HANDLER_FUNCTION(recall,, cfg)
 457 {
 458         events::command_disabler command_disabler;
 459         LOG_NG << "recalling unit...\n";
 460         config temp_config(cfg.get_config());
 461         // Prevent the recall unit filter from using the location as a criterion
 462 
 463         /**
 464          * @todo FIXME: we should design the WML to avoid these types of
 465          * collisions; filters should be named consistently and always have a
 466          * distinct scope.
 467          */
 468         temp_config["x"] = "recall";
 469         temp_config["y"] = "recall";
 470         vconfig unit_filter_cfg(temp_config);
 471         const vconfig & leader_filter = cfg.child("secondary_unit");
 472 
 473         for(int index = 0; index < int(resources::gameboard->teams().size()); ++index) {
 474                 LOG_NG << "for side " << index + 1 << "...\n";
 475                 const std::string player_id = resources::gameboard->teams()[index].save_id();
 476 
 477                 if(resources::gameboard->teams()[index].recall_list().size() < 1) {
 478                         DBG_NG << "recall list is empty when trying to recall!\n"
 479                                    << "player_id: " << player_id << " side: " << index+1 << "\n";
 480                         continue;
 481                 }
 482 
 483                 recall_list_manager & avail = resources::gameboard->teams()[index].recall_list();
 484                 std::vector<unit_map::unit_iterator> leaders = resources::gameboard->units().find_leaders(index + 1);
 485 
 486                 const unit_filter ufilt(unit_filter_cfg, resources::filter_con);
 487                 const unit_filter lfilt(leader_filter, resources::filter_con); // Note that if leader_filter is null, this correctly gives a null filter that matches all units.
 488                 for(std::vector<unit_ptr>::iterator u = avail.begin(); u != avail.end(); ++u) {
 489                         DBG_NG << "checking unit against filter...\n";
 490                         scoped_recall_unit auto_store("this_unit", player_id, u - avail.begin());
 491                         if (ufilt(*(*u), map_location())) {
 492                                 DBG_NG << (*u)->id() << " matched the filter...\n";
 493                                 const unit_ptr to_recruit = *u;
 494                                 const unit* pass_check = to_recruit.get();
 495                                 if(!cfg["check_passability"].to_bool(true)) pass_check = nullptr;
 496                                 const map_location cfg_loc = cfg_to_loc(cfg);
 497 
 498                                 /// @todo fendrin: comment this monster
 499                                 for (unit_map::const_unit_iterator leader : leaders) {
 500                                         DBG_NG << "...considering " + leader->id() + " as the recalling leader...\n";
 501                                         map_location loc = cfg_loc;
 502                                         if ( lfilt(*leader)  &&
 503                                              unit_filter(vconfig(leader->recall_filter()), resources::filter_con).matches( *(*u),map_location() ) ) {
 504                                                 DBG_NG << "...matched the leader filter and is able to recall the unit.\n";
 505                                                 if(!resources::gameboard->map().on_board(loc))
 506                                                         loc = leader->get_location();
 507                                                 if(pass_check || (resources::gameboard->units().count(loc) > 0))
 508                                                         loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 509                                                 if(resources::gameboard->map().on_board(loc)) {
 510                                                         DBG_NG << "...valid location for the recall found. Recalling.\n";
 511                                                         avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 512                                                         actions::place_recruit(to_recruit, loc, leader->get_location(), 0, true,
 513                                                                                map_location::parse_direction(cfg["facing"]),
 514                                                                                cfg["show"].to_bool(true), cfg["fire_event"].to_bool(false),
 515                                                                                true, true);
 516                                                         return;
 517                                                 }
 518                                         }
 519                                 }
 520                                 if (resources::gameboard->map().on_board(cfg_loc)) {
 521                                         map_location loc = cfg_loc;
 522                                         if(pass_check || (resources::gameboard->units().count(loc) > 0))
 523                                                 loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 524                                         // Check if we still have a valid location
 525                                         if (resources::gameboard->map().on_board(loc)) {
 526                                                 DBG_NG << "No usable leader found, but found usable location. Recalling.\n";
 527                                                 avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 528                                                 map_location null_location = map_location::null_location();
 529                                                 actions::place_recruit(to_recruit, loc, null_location, 0, true,
 530                                                                        map_location::parse_direction(cfg["facing"]),
 531                                                                        cfg["show"].to_bool(true), cfg["fire_event"].to_bool(false),
 532                                                                        true, true);
 533                                                 return;
 534                                         }
 535                                 }
 536                         }
 537                 }
 538         }
 539         LOG_WML << "A [recall] tag with the following content failed:\n" << cfg.get_config().debug();
 540 }

 543         struct map_choice : public mp_sync::user_choice
 544         {
 545                 map_choice(const std::string& filename) : filename_(filename) {}
 546                 std::string filename_;
 547                 virtual config query_user(int /*side*/) const
 548                 {
 549                         //Do a regex check for the file format to prevent sending aribitary files to other clients.
 550                         //Note: this allows only the new format.
 551                         static const std::string s_simple_terrain = R"""([A-Za-z\\|/]{1,4})""";
 552                         static const std::string s_terrain = s_simple_terrain + R"""((\^)""" + s_simple_terrain + ")?";
 553                         static const std::string s_sep = "(, |\\n)";
 554                         static const std::string s_prefix = R"""((\d+ )?)""";
 555                         static const std::string s_all = "(" + s_prefix + s_terrain + s_sep + ")+";
 556                         static const boost::regex r_all(s_all);
 557 
 558                         const std::string& mapfile = filesystem::get_wml_location(filename_);
 559                         std::string res = "";
 560                         if(filesystem::file_exists(mapfile)) {
 561                                 res = filesystem::read_file(mapfile);
 562                         }
 563                         config retv;
 564                         if(boost::regex_match(res, r_all))
 565                         {
 566                                 retv["map_data"] = res;
 567                         }
 568                         return retv;
 569                 }
 570                 virtual config random_choice(int /*side*/) const
 571                 {
 572                         return config();
 573                 }
 574                 virtual std::string description() const
 575                 {
 576                         return "Map Data";
 577                 }

 579         };

 582 /// Experimental map replace
 583 /// @todo Finish experimenting.
 584 WML_HANDLER_FUNCTION(replace_map,, cfg)
 585 {
 586         /*
 587          * When a hex changes from a village terrain to a non-village terrain, and
 588          * a team owned that village it loses that village. When a hex changes from
 589          * a non-village terrain to a village terrain and there is a unit on that
 590          * hex it does not automatically capture the village. The reason for not
 591          * capturing villages it that there are too many choices to make; should a
 592          * unit loose its movement points, should capture events be fired. It is
 593          * easier to do this as wanted by the author in WML.
 594          */
 595 
 596         const gamemap * game_map = & resources::gameboard->map();
 597         gamemap map(*game_map);
 598 
 599         try {
 600                 if(!cfg["map_file"].empty()) {
 601                         config file_cfg = mp_sync::get_user_choice("map_data", map_choice(cfg["map_file"].str()));
 602                         map.read(file_cfg["map_data"].str(), false);
 603                 } else {
 604                         map.read(cfg["map"], false);
 605                 }
 606         } catch(incorrect_map_format_error&) {
 607                 const std::string log_map_name = cfg["map"].empty() ? cfg["file"] : std::string("from inline data");
 608                 lg::wml_error() << "replace_map: Unable to load map " << log_map_name << std::endl;
 609                 return;
 610         } catch(wml_exception& e) {
 611                 e.show(resources::screen->video());
 612                 return;
 613         }
 614 
 615         if (map.total_width() > game_map->total_width()
 616         || map.total_height() > game_map->total_height()) {
 617                 if (!cfg["expand"].to_bool()) {
 618                         lg::wml_error() << "replace_map: Map dimension(s) increase but expand is not set" << std::endl;
 619                         return;
 620                 }
 621         }
 622 
 623         if (map.total_width() < game_map->total_width()
 624         || map.total_height() < game_map->total_height()) {
 625                 if (!cfg["shrink"].to_bool()) {
 626                         lg::wml_error() << "replace_map: Map dimension(s) decrease but shrink is not set" << std::endl;
 627                         return;
 628                 }
 629         }
 630 
 631         boost::optional<std::string> errmsg = resources::gameboard->replace_map(map);
 632 
 633         if (errmsg) {
 634                 lg::wml_error() << *errmsg << std::endl;
 635         }
 636 
 637         resources::screen->reload_map();
 638         resources::screen->needs_rebuild(true);
 639         ai::manager::raise_map_changed();
 640 }

 650 WML_HANDLER_FUNCTION(set_variables,, cfg)
 651 {
 652         const t_string& name = cfg["name"];
 653         variable_access_create dest = resources::gamedata->get_variable_access_write(name);
 654         if(name.empty()) {
 655                 ERR_NG << "trying to set a variable with an empty name:\n" << cfg.get_config().debug();
 656                 return;
 657         }
 658 
 659         std::vector<config> data;
 660         if(cfg.has_attribute("to_variable"))
 661         {
 662                 try
 663                 {
 664                         variable_access_const tovar = resources::gamedata->get_variable_access_read(cfg["to_variable"]);
 665                         for (const config& c : tovar.as_array())
 666                         {
 667                                 data.push_back(c);
 668                         }
 669                 }
 670                 catch(const invalid_variablename_exception&)
 671                 {
 672                         ERR_NG << "Cannot do [set_variables] with invalid to_variable variable: " << cfg["to_variable"] << " with " << cfg.get_config().debug() << std::endl;
 673                 }
 674         } else {
 675                 typedef std::pair<std::string, vconfig> vchild;
 676                 for (const vchild& p : cfg.all_ordered()) {
 677                         if(p.first == "value") {
 678                                 data.push_back(p.second.get_parsed_config());
 679                         } else if(p.first == "literal") {
 680                                 data.push_back(p.second.get_config());
 681                         } else if(p.first == "split") {
 682                                 const vconfig & split_element = p.second;
 683 
 684                                 std::string split_string=split_element["list"];
 685                                 std::string separator_string=split_element["separator"];
 686                                 std::string key_name=split_element["key"];
 687                                 if(key_name.empty())
 688                                 {
 689                                         key_name="value";
 690                                 }
 691 
 692                                 bool remove_empty = split_element["remove_empty"].to_bool();
 693 
 694                                 char* separator = separator_string.empty() ? nullptr : &separator_string[0];
 695 
 696                                 std::vector<std::string> split_vector;
 697 
 698                                 //if no separator is specified, explode the string
 699                                 if(separator == nullptr)
 700                                 {
 701                                         for(std::string::iterator i=split_string.begin(); i!=split_string.end(); ++i)
 702                                         {
 703                                                 split_vector.push_back(std::string(1, *i));
 704                                         }
 705                                 }
 706                                 else {
 707                                         split_vector=utils::split(split_string, *separator, remove_empty ? utils::REMOVE_EMPTY | utils::STRIP_SPACES : utils::STRIP_SPACES);
 708                                 }
 709 
 710                                 for(std::vector<std::string>::iterator i=split_vector.begin(); i!=split_vector.end(); ++i)
 711                                 {
 712                                         data.emplace_back(config {key_name, *i});
 713                                 }
 714                         }
 715                 }
 716         }
 717         try
 718         {
 719                 const std::string& mode = cfg["mode"];
 720                 if(mode == "merge")
 721                 {
 722                         if(dest.explicit_index() && data.size() > 1)
 723                         {
 724                                 //merge children into one
 725                                 config merged_children;
 726                                 for (const config &ch : data) {
 727                                         merged_children.append(ch);
 728                                 }
 729                                 data = {merged_children};
 730                         }
 731                         dest.merge_array(data);
 732                 }
 733                 else if(mode == "insert")
 734                 {
 735                         dest.insert_array(data);
 736                 }
 737                 else if(mode == "append")
 738                 {
 739                         dest.append_array(data);
 740                 }
 741                 else /*default if(mode == "replace")*/
 742                 {
 743                         dest.replace_array(data);
 744                 }
 745         }
 746         catch(const invalid_variablename_exception&)
 747         {
 748                 ERR_NG << "Cannot do [set_variables] with invalid destination variable: " << name << " with " << cfg.get_config().debug() << std::endl;
 749         }
 750 }

 752 /// Store the relative direction from one hex to another in a WML variable.
 753 /// This is mainly useful as a diagnostic tool, but could be useful
 754 /// for some kind of scenario.
 755 WML_HANDLER_FUNCTION(store_relative_direction,, cfg)
 756 {
 757         if (!cfg.child("source")) {
 758                 WRN_NG << "No source in [store_relative_direction]" << std::endl;
 759                 return;
 760         }
 761         if (!cfg.child("destination")) {
 762                 WRN_NG << "No destination in [store_relative_direction]" << std::endl;
 763                 return;
 764         }
 765         if (!cfg.has_attribute("variable")) {
 766                 WRN_NG << "No variable in [store_relative_direction]" << std::endl;
 767                 return;
 768         }
 769 
 770         const map_location src = cfg_to_loc(cfg.child("source"));
 771         const map_location dst = cfg_to_loc(cfg.child("destination"));
 772 
 773         std::string variable = cfg["variable"];
 774         map_location::RELATIVE_DIR_MODE mode = static_cast<map_location::RELATIVE_DIR_MODE> (cfg["mode"].to_int(0));
 775         try
 776         {
 777                 variable_access_create store = resources::gamedata->get_variable_access_write(variable);
 778 
 779                 store.as_scalar() = map_location::write_direction(src.get_relative_dir(dst,mode));
 780         }
 781         catch(const invalid_variablename_exception&)
 782         {
 783                 ERR_NG << "Cannot do [store_relative_direction] with invalid destination variable: " << variable << " with " << cfg.get_config().debug() << std::endl;
 784         }

 862                                 mask_map.read(filesystem::read_file(maskfile), false);

 886 {
 887         const bool remove = cfg["remove"].to_bool(false);
 888         const bool delay = cfg["delayed_variable_substitution"].to_bool(true);

 890                 const std::vector<std::string> ids = utils::split(cfg["id"]);

 892                         resources::tunnels->remove(id);

 900                 pathfind::teleport_group tunnel(delay ? cfg : vconfig(cfg.get_parsed_config()), false);
 901                 resources::tunnels->add(tunnel);

 905                         resources::tunnels->add(tunnel);
 906                 }

 910 /// If we should spawn a new unit on the map somewhere
 911 WML_HANDLER_FUNCTION(unit,, cfg)
 912 {
 913         events::command_disabler command_disabler;
 914         config parsed_cfg = cfg.get_parsed_config();
 915 
 916         config::attribute_value to_variable = cfg["to_variable"];
 917         if (!to_variable.blank())
 918         {
 919                 parsed_cfg.remove_attribute("to_variable");
 920                 unit new_unit(parsed_cfg, true, &cfg);
 921                 try
 922                 {
 923                         config &var = resources::gamedata->get_variable_cfg(to_variable);
 924                         var.clear();
 925                         new_unit.write(var);
 926                         if (const config::attribute_value *v = parsed_cfg.get("x")) var["x"] = *v;
 927                         if (const config::attribute_value *v = parsed_cfg.get("y")) var["y"] = *v;
 928                 }
 929                 catch(const invalid_variablename_exception&)
 930                 {
 931                         ERR_NG << "Cannot do [unit] with invalid to_variable:  " << to_variable << " with " << cfg.get_config().debug() << std::endl;
 932                 }
 933                 return;
 934 
 935         }
 936 
 937         int side = parsed_cfg["side"].to_int(1);
 938 
 939 
 940         if ((side<1)||(side > static_cast<int>(resources::gameboard->teams().size()))) {
 941                 ERR_NG << "wrong side in [unit] tag - no such side: "<<side<<" ( number of teams :"<<resources::gameboard->teams().size()<<")"<<std::endl;
 942                 DBG_NG << parsed_cfg.debug();
 943                 return;
 944         }
 945         team &tm = resources::gameboard->get_team(side);
 946 
 947         unit_creator uc(tm,resources::gameboard->map().starting_position(side));
 948 
 949         uc
 950                 .allow_add_to_recall(true)
 951                 .allow_discover(true)
 952                 .allow_get_village(true)
 953                 .allow_invalidate(true)
 954                 .allow_rename_side(true)
 955                 .allow_show(true);
 956 
 957         uc.add_unit(parsed_cfg, &cfg);
 958 
 959 }
