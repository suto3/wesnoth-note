 267     def add_units(self, campaign):
 268         """
 269         We assume each unit, in mainline and all addons, has one unique id. So
 270         we reference them everywhere by this id, and here can add them all to
 271         one big collection.
 272         """
 273         addunits = self.parser.get_all(tag="units")
 274         addunits += self.parser.get_all(tag="+units")
 275         if not addunits:
 276             return 0
 277 
 278         def getall(oftype):
 279             res = []
 280             for units in addunits:
 281                 res += units.get_all(tag=oftype)
 282             return res
 283 
 284         # Find all unit types.
 285         newunits = getall("unit_type") + getall("unit")
 286         for unit in newunits:
 287             uid = unit.get_text_val("id")
 288             unit.id = uid
 289 
 290             if unit.get_text_val("do_not_list", "no") not in ["no", "false"] or \
 291                unit.get_text_val("hide_help", "no") not in ["no", "false"]:
 292                 unit.hidden = True
 293             else:
 294                 unit.hidden = False
 295 
 296             if uid in self.unit_lookup:
 297                 unit = self.unit_lookup[uid]
 298                 # TODO: We might want to compare the two units
 299                 # with the same id and if one is different try
 300                 # to do something clever like rename it...
 301             else:
 302                 self.unit_lookup[uid] = unit
 303             if not hasattr(unit, "campaigns"):
 304                 unit.campaigns = []
 305             unit.campaigns.append(campaign)
 306 
 307         # Find all races.
 308         newraces = getall("race")
 309         for race in newraces:
 310             rid = race.get_text_val("id")
 311             if rid is None:
 312                 rid = race.get_text_val("name")
 313             self.race_lookup[rid] = race
 314 
 315         # Find all movetypes.
 316         newmovetypes = getall("movetype")
 317         for movetype in newmovetypes:
 318             mtname = movetype.get_text_val("name")
 319             self.movetype_lookup[mtname] = movetype
 320 
 321         # Store race/movetype/faction of each unit for easier access later.
 322         for unit in newunits:
 323             uid = unit.get_text_val("id")
 324             race = self.get_unit_value(unit, "race")
 325             try:
 326                 unit.race = self.race_lookup[race]
 327                 unit.rid = unit.race.get_text_val("id", "none")
 328             except KeyError:
 329                 unit.race = None
 330                 unit.rid = "none"
 331                 error_message("Warning: No race \"%s\" found (%s).\n" % (
 332                     race, unit.get_text_val("id")))
 333             movetype = self.get_unit_value(unit, "movement_type")
 334             try:
 335                 unit.movetype = self.movetype_lookup[movetype]
 336             except KeyError:
 337                 unit.movetype = None
 338 
 339             unit.advance = []
 340             advanceto = unit.get_text_val("advances_to")
 341             # Add backwards compatibility for 1.4
 342             if not advanceto:
 343                 advanceto = unit.get_text_val("advanceto")
 344             if advanceto and advanceto != "null":
 345                 for advance in advanceto.split(","):
 346                     auid = advance.strip()
 347                     if auid: unit.advance.append(auid)

 351             except TypeError:
 352                 level = 0
 353             except ValueError:
 354                 level = 0

 445     def create_network(self):
 446         """
 447         Assuming that each node which has been added to the tree only has a
 448         valid list of children in unit.child_ids, also fill in unit.parent_ids
 449         and update the unit.children shortcut.
 450         """
 451 
 452         # Complete the network
 453         for uid, u in list(self.lookup.items()):
 454             for cid in u.child_ids:
 455                 c = self.lookup.get(cid, None)
 456                 if not c:
 457                     continue
 458                 u.children.append(c)
 459                 if not uid in c.parent_ids:
 460                     c.parent_ids.append(uid)
 461 
 462         # Put all roots into the forest
 463         for uid, u in list(self.lookup.items()):
 464             if not u.parent_ids:
 465                 self.trees[uid] = u
 466 
 467         # Sanity check because some argGRRxxx addons have units who advance to
 468         # themselves.
 469 
 470         def recurse(u, already):
 471             already2 = already.copy()
 472             for c in u.children[:]:
 473                 already2[c.id] = True
 474                 if c.id in already:
 475                     error_message(
 476                         ("Warning: Unit %s advances to unit %s in a loop.\n" %
 477                          (u.id, c.id)) +
 478                         ("    Removing advancement %s.\n" % c.id))
 479                     u.children.remove(c)
 480             for c in u.children:
 481                 recurse(c, already2)
 482         for u in list(self.trees.values()):
 483             already = {u.id : True}
 484             recurse(u, already)
 485 
 486     def update(self):
 487         self.create_network()
 488         self.breadth = sum([x.update_breadth() for x in list(self.trees.values())])
 489         return self.breadth
 490

 499 class UnitNode:

 511     def update_breadth(self):
