 102 void chat_command_handler::do_ignore()
 103 {
 104         if (get_arg(1).empty()) {
 105                 do_display();
 106         }
 107         else {
 108                 utils::string_map symbols;
 109                 symbols["nick"] = get_arg(1);
 110 
 111                 if (preferences::add_acquaintance(get_arg(1), "ignore", get_data(2))) {
 112                         print(_("ignores list"), VGETTEXT("Added to ignore list: $nick", symbols));
 113                         chat_handler_.user_relation_changed(get_arg(1));
 114                 }
 115                 else {
 116                         command_failed(VGETTEXT("Invalid username: $nick", symbols));
 117                 }
 118         }
 119 }

 162 void chat_command_handler::do_register() {
 163         config data;
 164         config& nickserv = data.add_child("nickserv");
 165 
 166         if (get_data(1).empty()) return command_failed_need_arg(1);
 167 
 168         config &reg = nickserv.add_child("register");
 169         reg["password"] = get_arg(1);
 170         if (!get_data(2).empty()) {
 171                 reg["mail"] = get_arg(2);
 172         }
 173         std::string msg;
 174         if (get_data(2).empty()) {
 175                 msg = _("registering with password *** and no email address");
 176         }
 177         else {
 178                 utils::string_map symbols;
 179                 symbols["email"] = get_data(2);
 180                 msg = VGETTEXT("registering with password *** and "
 181                         "email address $email", symbols);
 182         }
 183         print(_("nick registration"), msg);
 184 
 185         chat_handler_.send_to_server(data);
 186 }
