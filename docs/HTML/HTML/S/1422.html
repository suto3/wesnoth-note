 209     /**
 210      * Declares an IDE-specific workbench image.
 211      * 
 212      * @param symbolicName
 213      *        the symbolic name of the image
 214      * @param path
 215      *        the path of the image file; this path is relative to the base
 216      *        of the IDE plug-in
 217      * @param shared
 218      *        <code>true</code> if this is a shared image, and
 219      *        <code>false</code> if this is not a shared image
 220      * @see org.eclipse.ui.internal.ide.IDEWorkbenchAdvisor#declareImage
 221      */
 222     private void declareWorkbenchImage( Bundle ideBundle, String symbolicName,
 223         String path, boolean shared )
 224     {
 225         URL url = FileLocator.find( ideBundle, new Path( path ), null );
 226         ImageDescriptor desc = ImageDescriptor.createFromURL( url );
 227         getWorkbenchConfigurer( ).declareImage( symbolicName, desc, shared );
 228     }
 229 
 230     @Override
 231     public void initialize( IWorkbenchConfigurer configurer )
 232     {
 233         configurer.setSaveAndRestore( true );
 234         declareWorkbenchImages( );
 235     }
