 193         # Here we set up an array that holds the possible ambusher unit types.
 194         # We do this instead of just using a random type when creating the
 195         # ambusher so that the even distribution of different types is
 196         # guaranteed.

 198         {VARIABLE random_ambusher_type_table[0].type "Bone Shooter"}

 214         {VARIABLE random_ambusher_type_i 0}
 215 
 216         # With this macro we place a set number of ambushers (from the above list) to a
 217         # given rectangular area. We do this in blocks so that we can guarantee that
 218         # every major area has at least some ambushers: if we randomized the ambush
 219         # locations across all possible locations at once, the RNG could theoretically
 220         # clump all the ambushers into a single corner of the map.
 221 
 222 #define UNDEAD_AMBUSH_AREA SIDE X_SPAN Y_SPAN NUMBER
 223     [random_placement]
 224         variable=ambush_location
 225         num_items={NUMBER}
 226         [filter_location]
 227             x={X_SPAN}
 228             y={Y_SPAN}
 229             terrain=Ss
 230 
 231             [filter_adjacent_location]
 232                 [not]
 233                     terrain=Ss
 234                 [/not]
 235 
 236                 [not]
 237                     [filter]
 238                     [/filter]
 239                 [/not]
 240             [/filter_adjacent_location]
 241         [/filter_location]
 242         [command]
 243             {VARIABLE random_ambusher_type_i ("$(($random_ambusher_type_i + 1) % $random_ambusher_type_table.length)")}
 244             [unit]
 245                 x,y=$ambush_location.x,$ambush_location.y
 246                 type=$random_ambusher_type_table[$random_ambusher_type_i].type
 247                 side={SIDE}
 248                 role=ambusher
 249                 random_traits=yes
 250                 upkeep=loyal
 251             [/unit]
 252 
 253             [object]
 254                 silent=yes
 255 
 256                 [filter]
 257                     x,y=$ambush_location.x,$ambush_location.y
 258                 [/filter]
 259 
 260                 [effect]
 261                     apply_to=new_ability
 262 
 263                     [abilities]
 264                         [hides]
 265                             id=undead_ambush
 266                             affect_self=yes
 267 
 268                             [filter_self]
 269                                 role=ambusher
 270 
 271                                 [filter_location]
 272                                     terrain=Ss
 273                                 [/filter_location]
 274                             [/filter_self]
 275                         [/hides]
 276                     [/abilities]
 277                 [/effect]
 278             [/object]
 279         [/command]
 280     [/random_placement]
 281 #enddef
