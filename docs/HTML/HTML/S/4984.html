  23 # Create a new ghost-type "advisor" if Malin doesn't already have one
  24 #define CREATE_ADVISOR
  25     [if]
  26         [have_unit]
  27             side=1
  28             role=advisor
  29             search_recall_list=yes
  30         [/have_unit]
  31         [then]
  32             # Recall an advisor if we have one to do so
  33             [recall]
  34                 role=advisor
  35             [/recall]
  36         [/then]
  37         [elseif]
  38             # Else, make a new advisor from a ghost unit
  39             [have_unit]
  40                 side=1
  41                 type=Spectre,Nightgaunt,Wraith,Shadow,Ghost
  42                 search_recall_list=yes
  43             [/have_unit]
  44             [then]
  45                 [role]
  46                     role=advisor
  47                     side=1
  48                     type=Spectre,Nightgaunt,Wraith,Shadow,Ghost
  49                 [/role]
  50 
  51                 [recall]
  52                     role=advisor
  53                 [/recall]
  54             [/then]
  55         [/elseif]
  56         [else]
  57             # If that fails too, make a brand new ghost advisor
  58             [unit]
  59                 type=Ghost
  60                 side=1
  61                 role=advisor
  62                 animate=yes
  63                 placement=leader
  64             [/unit]
  65         [/else]
  66     [/if]
  67 #enddef
