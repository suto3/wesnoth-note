   3 #define MINIMUM CONTAINER VAR
   4     # Get the minimum value of CONTAINER.VAR
   5     # Result is returned in variable 'minimum'
   6     # Index is returned in min_index
   7     # It only works for values smaller than 99999
   8 
   9     # Seed for the minimum variable
  10     {VARIABLE minimum 99999}
  11 
  12     # Go through container variable
  13     [for]
  14         array={CONTAINER}
  15         variable=i_min
  16         [do]
  17             {IF_VAR {CONTAINER}[$i_min].{VAR} less_than $minimum (
  18                 [then]
  19                     {VARIABLE minimum ${CONTAINER}[$i_min].{VAR}}
  20                     {VARIABLE min_index $i_min}
  21                 [/then]
  22             )}
  23             #{DEBUG "$i_min: $minimum (${CONTAINER}[$i_min].{VAR})"}
  24         [/do]
  25     [/for]
  26 #enddef

  28 #define LURKER_MOVES SIDE ENEMY_SIDES
  29     # The events that move the Saurian Skirmishers each turn
  30     # This is not left to the AI, because they can only stop on swamps, but move across other terrain
  31     # This means they cannot be used on a human player side either
  32 
  33     # Lurkers have no moves left each turn, instead, they are moved "manually" here
  34     [event]
  35         name=side {SIDE} turn refresh
  36         first_time_only=no
  37 
  38         # No moves each turn
  39         {MODIFY_UNIT (side,type={SIDE},Saurian Skirmisher) moves 0}
  40 
  41         # Store all the Lurkers of the side
  42         [store_unit]
  43             [filter]
  44                 side={SIDE}
  45                 type=Saurian Skirmisher
  46             [/filter]
  47             variable=stored_lurkers
  48         [/store_unit]
  49 
  50         # For each Lurker, we do:
  51         [for]
  52             array=stored_lurkers
  53             variable=i_l
  54             [do]
  55                 #{DEBUG "Lurker $i_l"}
  56 
  57                 # Store reachable swamp locations next to an enemy
  58                 # This should include the current location
  59                 [store_reachable_locations]
  60                     [filter]
  61                         id=$stored_lurkers[$i_l].id
  62                     [/filter]
  63                     [filter_location]
  64                         terrain=S*  # swamp
  65                         [not]  # unoccupied by other unit
  66                             [filter]
  67                                 [not]
  68                                     id=$stored_lurkers[$i_l].id
  69                                 [/not]
  70                             [/filter]
  71                         [/not]
  72                         [filter_adjacent_location]  # next to enemy
  73                             [filter]
  74                                 side={ENEMY_SIDES}
  75                             [/filter]
  76                         [/filter_adjacent_location]
  77                     [/filter_location]
  78                     moves=max
  79                     variable=stored_locs
  80                 [/store_reachable_locations]
  81                 #{DEBUG "  reachable with enemy adjacent: $stored_locs.length"}
  82 
  83                 # Now find all those enemies and store
  84                 # Doesn't matter if some enemies are stored several times
  85                 [for]
  86                     array=stored_locs
  87                     variable=i_e
  88                     [do]
  89                         [store_unit]
  90                             [filter]
  91                                 side={ENEMY_SIDES}
  92                                 [filter_location]
  93                                     [filter_adjacent_location]
  94                                         x,y=$stored_locs[$i_e].x,$stored_locs[$i_e].y
  95                                     [/filter_adjacent_location]
  96                                 [/filter_location]
  97                             [/filter]
  98                             variable=adj_enemies
  99                             mode=append
 100                         [/store_unit]
 101                         #{DEBUG "  $i_e: enemies adjacent to current location: $adj_enemies.length"}
 102                     [/do]
 103                 [/for]
 104 
 105                 {IF_VAR adj_enemies.length equals 0 (
 106                     [then] # if there is no reachable enemy
 107                         # Simply pick a random reachable swamp hex (Lurkers are pretty stupid)
 108                         # (there is always at least the hex the unit is on)
 109                         [store_reachable_locations]
 110                             [filter]
 111                                 id=$stored_lurkers[$i_l].id
 112                             [/filter]
 113                             [filter_location]
 114                                 terrain=S*  # swamp
 115                                 [not]  # unoccupied by other unit
 116                                     [filter]
 117                                         [not]
 118                                             id=$stored_lurkers[$i_l].id
 119                                         [/not]
 120                                     [/filter]
 121                                 [/not]
 122                             [/filter_location]
 123                             moves=max
 124                             variable=stored_locs
 125                         [/store_reachable_locations]
 126                         #{DEBUG "Reachable hexes: $stored_locs.length"}
 127                     [/then]
 128 
 129                     [else] # if there are reachable enemies
 130                         # We simply find the enemies with minimum hitpoints (Lurkers are pretty stupid)
 131                         {MINIMUM adj_enemies hitpoints}
 132                         # Unfortunately, now we need to store the reachable location again
 133                         [store_reachable_locations]
 134                             [filter]
 135                                 id=$stored_lurkers[$i_l].id
 136                             [/filter]
 137                             [filter_location]
 138                                 terrain=S*  # swamp
 139                                 [not]  # unoccupied by other unit
 140                                     [filter]
 141                                         [not]
 142                                             id=$stored_lurkers[$i_l].id
 143                                         [/not]
 144                                     [/filter]
 145                                 [/not]
 146                                 [filter_adjacent_location]
 147                                     [filter]
 148                                         x,y=$adj_enemies[$min_index].x,$adj_enemies[$min_index].y
 149                                     [/filter]
 150                                 [/filter_adjacent_location]
 151                             [/filter_location]
 152                             moves=max
 153                             variable=stored_locs
 154                         [/store_reachable_locations]
 155                         #{DEBUG "  final possible reachable locations: $stored_locs.length"}
 156                     [/else]
 157                 )}
 158 
 159                 # Move unit to one of those at random
 160                 {RANDOM "0..$($stored_locs.length-1)"}
 161                 #{DEBUG "$stored_lurkers[$i_l].x $stored_lurkers[$i_l].y : $random"}
 162                 [if]  # only if different from current position
 163                     {VARIABLE_CONDITIONAL stored_locs[$random].x not_equals $stored_lurkers[$i_l].x}
 164                     [or]
 165                         {VARIABLE_CONDITIONAL stored_locs[$random].y not_equals $stored_lurkers[$i_l].y}
 166                     [/or]
 167                     [then]
 168                         {MOVE_UNIT id=$stored_lurkers[$i_l].id $stored_locs[$random].x $stored_locs[$random].y}
 169                     [/then]
 170                 [/if]
 171 
 172                 # Need to clear adj_enemies inside the loop, as mode=append
 173                 {CLEAR_VARIABLE adj_enemies,minimum,min_index,random,stored_locs}
 174             [/do]
 175         [/for]
 176         {CLEAR_VARIABLE stored_lurkers}
 177 
 178         # Attack is then left to the AI
 179     [/event]
 180 #enddef

 450         {MESSAGE Pekzs "" "" _"In this scenario we demonstrate the Lurker Micro AI. A lurker is a unit that is capable of moving across most terrains, but that only stops on and attacks from specific terrain. It might also have the ability to hide on this terrain (which is the reason why this is called the Lurker AI).
 451 
 452 Lurkers move individually without any strategy and always attack the weakest enemy within their reach. If no enemy is in reach, the lurker does a random move instead - or it just sits and waits (lurks)."}
