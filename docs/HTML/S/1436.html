  63      * 
  64      * @param source
  65      *        The source file.
  66      * @param target
  67      *        The target file.
  68      * 
  69      * @return True if the copy was successfull, false otherwise.

  71      */
  72     public static boolean copyTo( File source, File target )
  73     {
  74         if( source == null || target == null ) {
  75             return false;
  76         }
  77 
  78         try {
  79             InputStream in = new FileInputStream( source );
  80             OutputStream out = new FileOutputStream( target );
  81 
  82             // Transfer bytes from in to out
  83             byte[] buf = new byte[1024];
  84             int len;
  85             while( ( len = in.read( buf ) ) > 0 ) {
  86                 out.write( buf, 0, len );
  87             }
  88             in.close( );
  89             out.close( );
  90 
  91             return true;
  92         } catch( IOException e ) {
  93             Logger.getInstance( ).logException( e );
  94             return false;
  95         }
  96     }

  98     /**
  99      * Gets the contents of the specified file as string.
 100      * 
 101      * @param file
 102      *        The file
 103      * @param skipEmptyLines
 104      *        True to not take in account empty lines when creating the result
 105      * @param skipCommentLines
 106      *        True to not take in account comment lines ( WML Style: # ) when
 107      *        creating the result
 108      * @return The string contents of the file.
 109      */
 110     public static String getFileContents( File file, boolean skipEmptyLines,
 111         boolean skipCommentLines )
 112     {
 113         if( ! file.exists( ) || ! file.isFile( ) ) {
 114             return ""; //$NON-NLS-1$
 115         }
 116 
 117         StringBuilder contentsString = new StringBuilder( );
 118         BufferedReader reader = null;
 119         try {
 120             String line = ""; //$NON-NLS-1$
 121             reader = new BufferedReader( new InputStreamReader(
 122                 new FileInputStream( file ) ) );
 123             while( ( line = reader.readLine( ) ) != null ) {
 124                 if( skipEmptyLines && line.isEmpty( ) ) {
 125                     continue;
 126                 }
 127 
 128                 if( skipCommentLines && StringUtils.startsWith( line, "#" ) ) {
 129                     continue;
 130                 }
 131 
 132                 contentsString.append( line + "\n" ); //$NON-NLS-1$
 133             }
 134         } catch( IOException e ) {
 135             Logger.getInstance( ).logException( e );
 136         } finally {
 137             try {
 138                 if( reader != null ) {
 139                     reader.close( );
 140                 }
 141             } catch( Exception e ) {
 142                 Logger.getInstance( ).logException( e );
 143             }
 144         }
 145         return contentsString.toString( );
 146     }

 458     /**
 459      * Returns "_main.cfg" file
 460      * from the specified resource or null if it isn't any
 461      * It will start searching upwards starting from curren
 462      * resource's directory, until it finds a '_main.cfg' but it will
 463      * stop when encounters a project
 464      * 
 465      * @param resource
 466      *        The resource where to search for '_main.cfg'
 467      * @return An {@link IFile} instance or null if there is no _main.cfg
 468      *         in that resource's context.
 469      */
 470     public static IFile getMainConfigLocation( IResource resource )
 471     {
 472         if( resource == null ) {
 473             return null;
 474         }
 475 
 476         IFile targetResource = null;
 477         if( resource instanceof IProject ) {
 478             IProject project = ( IProject ) resource;
 479             if( project.getFile( "_main.cfg" ).exists( ) ) {
 480                 targetResource = project.getFile( "_main.cfg" ); //$NON-NLS-1$
 481             }
 482         }
 483 
 484         if( targetResource == null && resource instanceof IFolder ) {
 485             IFolder folder = ( IFolder ) resource;
 486             if( folder.getFile( new Path( "_main.cfg" ) ).exists( ) ) {
 487                 targetResource = folder.getFile( new Path( "_main.cfg" ) ); //$NON-NLS-1$
 488             }
 489         }
 490 
 491         if( targetResource == null && resource instanceof IFile ) {
 492             if( resource.getName( ).equals( "_main.cfg" ) ) {
 493                 targetResource = ( IFile ) resource;
 494             }
 495             else {
 496                 IProject project = resource.getProject( );
 497                 if( project.getFile( "_main.cfg" ).exists( ) ) {
 498                     targetResource = project.getFile( "_main.cfg" ); //$NON-NLS-1$
 499                 }
 500                 else {
 501                     // this might be the case of "user addon's" project
 502                     // we're going to the first subdirectory under the project
 503                     IContainer container = resource.getParent( );
 504                     if( container != null ) {
 505                         while( container.getParent( ) != null
 506                             && container.getParent( ) != resource
 507                                 .getProject( ) ) {
 508                             container = container.getParent( );
 509                         }
 510                         IFile file = project.getFile( container
 511                             .getProjectRelativePath( ).toOSString( )
 512                             + "/_main.cfg" ); //$NON-NLS-1$
 513                         if( file.exists( ) ) {
 514                             targetResource = file;
 515                         }
 516                     }
 517                 }
 518             }
 519         }
 520         if( targetResource == null ) {
 521             return null;
 522         }
 523         return targetResource;
 524     }

 574     /**
 575      * Parses the current line of the file and add the marker (if any) on the
 576      * file.
 577      * Current used format: "sourcefile", line x: error message
 578      * 
 579      * @param line
 580      *        the line to parse
 581      * @param type
 582      *        the created marker or null if there was none
 583      * @return The {@link IMarker} instance that was added, or null
 584      *         if there was an error.
 585      */
 586     public static IMarker addMarkerForLine( String line, String type )
 587     {
 588         // error template, is the one used by GCC:
 589         // "sourcefile", line <linenumber>: error message
 590         try {
 591             final String pivot = ", line "; //$NON-NLS-1$
 592             int pivotIndex = line.indexOf( pivot, 2 );
 593 
 594             String sourceFile = line.substring( 1, pivotIndex - 1 );
 595             int lineNumber = Integer.parseInt( line.substring( pivotIndex
 596                 + pivot.length( ),
 597                 line.indexOf( ":", pivotIndex + pivot.length( ) + 1 ) ) ); //$NON-NLS-1$
 598             String message = line.substring( line.indexOf( " ", pivotIndex + pivot.length( ) + 1 ) ); //$NON-NLS-1$
 599 
 600             // Get the file
 601             IFile file = ResourcesPlugin.getWorkspace( ).getRoot( )
 602                 .getFileForLocation( new Path( sourceFile ) );
 603             if( file.exists( ) == false ) {
 604                 return null;
 605             }
 606 
 607             IMarker marker = file.createMarker( type );
 608             marker.setAttribute( IMarker.SOURCE_ID, sourceFile );
 609             marker.setAttribute( IMarker.MESSAGE, message );
 610             marker.setAttribute( IMarker.LINE_NUMBER, lineNumber );
 611 
 612             // wmllint should be info or warning
 613             if( type.equals( Constants.MARKER_WMLLINT ) ) {
 614                 marker.setAttribute( IMarker.SEVERITY, IMarker.SEVERITY_INFO );
 615             }
 616             else {
 617                 marker.setAttribute( IMarker.SEVERITY, IMarker.SEVERITY_ERROR );
 618             }
 619             return marker;
 620         } catch( Exception e ) {
 621             Logger.getInstance( ).logException( e );
 622             return null;
 623         }
 624     }

 643     /**
 644      * Gets the set of included files or folders from this file
 645      * 
 646      * @param file
 647      *        The file to get the containers from
 648      * @return A set of containers represented by their Path as string
 649      */
 650     public static Set< String > getMacroIncludes( IFile file )
 651     {
 652         IProject project = file.getProject( );
 653         WMLRoot root = WMLUtils.getWMLRoot( file );
 654         // nothing to do
 655         if( root == null ) {
 656             return new LinkedHashSet< String >( 0 );
 657         }
 658 
 659         EList< WMLMacroCall > macroCalls = new BasicEList< WMLMacroCall >( );
 660 
 661         // iterate to find macro calls
 662         TreeIterator< EObject > treeItor = root.eAllContents( );
 663 
 664         while( treeItor.hasNext( ) ) {
 665             EObject object = treeItor.next( );
 666             if( object instanceof WMLMacroCall ) {
 667                 macroCalls.add( ( WMLMacroCall ) object );
 668             }
 669         }
 670 
 671         // now check what macros are really an inclusion macro
 672         Set< String > includesToAdd = new LinkedHashSet< String >( );
 673 
 674         for( WMLMacroCall macro: macroCalls ) {
 675             String text = WMLUtils.toString( macro );
 676             /**
 677              * To include a folder/file the macro should be the following
 678              * forms:
 679              * - {campaigns/... }
 680              * - {~add-ons/... }
 681              */
 682             if( ! ( text.startsWith( "{campaigns" ) ) && //$NON-NLS-1$
 683                 ! ( text.startsWith( "{~add-ons" ) ) ) {
 684                 continue;
 685             }
 686 
 687             // check if the macro includes directories local
 688             // to this project
 689             String projectPath = project.getLocation( ).toOSString( );
 690 
 691             if( projectPath.contains( WMLUtils.toString( macro.getParameters( )
 692                 .get( 1 ) ) ) ) {
 693                 String subString = text.replace( "}", "" )
 694                     .replaceFirst( "\\{campaigns/", "" )
 695                     .replaceFirst( "\\{~add-ons/", "" );
 696                 includesToAdd.add( subString.substring( subString
 697                     .indexOf( '/' ) ) );
 698             }
 699         }
 700 
 701         return includesToAdd;
 702     }

 720             // not interested
 721         }
 722 
 723         return index;
 724     }

 726     /**
 727      * This is a WML files comparator, based on the WML parsing rules.
 728      * 
 729      * @see <a href="http://wiki.wesnoth.org/PreprocessorRef"> Preprocessor
 730      *      Reference </a>
 731      */
 732     public static class WMLFilesComparator implements Comparator< IResource >,
 733         Serializable
 734     {
 735 
 736         private static final long serialVersionUID = 1045365969430128101L;
 737 
 738         @Override
 739         public int compare( IResource o1, IResource o2 )
 740         {
 741             return ResourceUtils.wmlFileNameCompare( o1.getName( ),
 742                 o2.getName( ) );
 743         }
 744     }

 746     /**
 747      * Compares 2 filenames to get the wml file order
 748      * 
 749      * @param fileName1
 750      *        The first filename
 751      * @param fileName2
 752      *        The second filename
 753      * @return -1, 0 or 1 if the fileName1 is lower, equal or greater than
 754      *         fileName2
 755      */
 756     public static int wmlFileNameCompare( String fileName1, String fileName2 )
 757     {
 758         // _initial.cfg is always the "lowest"
 759         if( fileName1.equals( "_initial.cfg" )
 760             && ! ( fileName2.equals( "_initial.cfg" ) ) ) {
 761             return - 1;
 762         }
 763 
 764         if( fileName2.equals( "_initial.cfg" )
 765             && ! ( fileName1.equals( "_initial.cfg" ) ) ) {
 766             return 1;
 767         }
 768 
 769         // _final.cfg is always the "highest"
 770         if( fileName1.equals( "_final.cfg" )
 771             && ! ( fileName2.equals( "_final.cfg" ) ) ) {
 772             return 1;
 773         }
 774 
 775         if( fileName2.equals( "_final.cfg" )
 776             && ! ( fileName1.equals( "_final.cfg" ) ) ) {
 777             return - 1;
 778         }
 779 
 780         return fileName1.compareTo( fileName2 );
 781     }
