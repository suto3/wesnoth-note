 106 void game_load::pre_show(window& window)
 107 {
 108         find_widget<minimap>(&window, "minimap", false).set_config(&cache_config_);
 109 
 110         text_box* filter = find_widget<text_box>(&window, "txtFilter", false, true);
 111 
 112         filter->set_text_changed_callback(
 113                         std::bind(&game_load::filter_text_changed, this, _1, _2));
 114 
 115         listbox& list = find_widget<listbox>(&window, "savegame_list", false);
 116 
 117 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 118         connect_signal_notify_modified(list,
 119                         std::bind(&game_load::display_savegame, *this, std::ref(window)));
 120 #else
 121         list.set_callback_value_change(
 122                         dialog_callback<game_load, &game_load::display_savegame>);
 123 #endif
 124 
 125         window.keyboard_capture(filter);
 126         window.add_to_keyboard_chain(&list);
 127 
 128         list.clear();
 129 
 130         for(const auto& game : games_) {
 131                 std::map<std::string, string_map> data;
 132                 string_map item;
 133 
 134                 std::string name = game.name();
 135                 utils::ellipsis_truncate(name, 40);
 136                 item["label"] = name;
 137                 data.emplace("filename", item);
 138 
 139                 item["label"] = game.format_time_summary();
 140                 data.emplace("date", item);
 141 
 142                 list.add_row(data);
 143         }
 144 
 145         list.register_sorting_option(0, [this](const int i) { return games_[i].name(); });
 146         list.register_sorting_option(1, [this](const int i) { return games_[i].modified(); });
 147 
 148         connect_signal_mouse_left_click(
 149                         find_widget<button>(&window, "delete", false),
 150                         std::bind(&game_load::delete_button_callback,
 151                                         this, std::ref(window)));
 152 
 153         display_savegame(window);
 154 }

 156 void game_load::display_savegame(window& window)
 157 {
 158         const int selected_row =
 159                 find_widget<listbox>(&window, "savegame_list", false).get_selected_row();
 160 
 161         if(selected_row == -1) {
 162                 return;
 163         }
 164 
 165         savegame::save_info& game = games_[selected_row];
 166         filename_ = game.name();
 167         summary_  = game.summary();
 168 
 169         find_widget<minimap>(&window, "minimap", false)
 170                         .set_map_data(summary_["map_data"]);
 171 
 172         find_widget<label>(&window, "lblScenario", false)
 173                         .set_label(summary_["label"]);
 174 
 175         listbox& leader_list = find_widget<listbox>(&window, "leader_list", false);
 176 
 177         leader_list.clear();
 178 
 179         for(const auto& leader : summary_.child_range("leader")) {
 180                 std::map<std::string, string_map> data;
 181                 string_map item;
 182 
 183                 // First, we evaluate whether the leader image as provided exists.
 184                 // If not, we try getting a binary-path independent path. If that still doesn't
 185                 // work, we fallback on unknown-unit.png.
 186                 std::string leader_image = leader["leader_image"].str();
 187                 if(!::image::exists(leader_image)) {
 188                         leader_image = filesystem::get_independent_image_path(leader_image);
 189                 }
 190 
 191                 if(leader_image.empty()) {
 192                         leader_image = "units/unknown-unit.png" + leader["leader_image_tc_modifier"].str();
 193                 }
 194 
 195                 item["label"] = leader_image;
 196                 data.emplace("imgLeader", item);
 197 
 198                 item["label"] = leader["leader_name"];
 199                 data.emplace("leader_name", item);
 200 
 201                 item["label"] = leader["gold"];
 202                 data.emplace("leader_gold", item);
 203 
 204                 item["label"] = leader["units"];
 205                 data.emplace("leader_troops", item);
 206 
 207                 item["label"] = leader["recall_units"];
 208                 data.emplace("leader_reserves", item);
 209 
 210                 leader_list.add_row(data);
 211         }
 212 
 213         std::stringstream str;
 214         str << game.format_time_local() << "\n";
 215         evaluate_summary_string(str, summary_);
 216 
 217         find_widget<label>(&window, "lblSummary", false).set_label(str.str());
 218 
 219         toggle_button& replay_toggle            = dynamic_cast<toggle_button&>(*show_replay_->get_widget());
 220         toggle_button& cancel_orders_toggle     = dynamic_cast<toggle_button&>(*cancel_orders_->get_widget());
 221         toggle_button& change_difficulty_toggle = dynamic_cast<toggle_button&>(*change_difficulty_->get_widget());
 222 
 223         const bool is_replay = savegame::loadgame::is_replay_save(summary_);
 224         const bool is_scenario_start = summary_["turn"].empty();
 225 
 226         // Always toggle show_replay on if the save is a replay
 227         replay_toggle.set_value(is_replay);
 228         replay_toggle.set_active(!is_replay && !is_scenario_start);
 229 
 230         // Cancel orders doesnt make sense on replay saves or start-of-scenario saves
 231         cancel_orders_toggle.set_active(!is_replay && !is_scenario_start);
 232 
 233         // Changing difficulty doesn't make sense on non-start-of-scenario saves
 234         change_difficulty_toggle.set_active(!is_replay && is_scenario_start);
 235 }

 237 void game_load::filter_text_changed(text_box_base* textbox, const std::string& text)
 238 {
 239         window& window = *textbox->get_window();
 240 
 241         listbox& list = find_widget<listbox>(&window, "savegame_list", false);
 242 
 243         const std::vector<std::string> words = utils::split(text, ' ');
 244 
 245         if(words == last_words_)
 246                 return;
 247         last_words_ = words;
 248 
 249         boost::dynamic_bitset<> show_items;
 250         show_items.resize(list.get_item_count(), true);
 251 
 252         if(!text.empty()) {
 253                 for(unsigned int i = 0; i < list.get_item_count(); i++) {
 254                         grid* row = list.get_row_grid(i);
 255 
 256                         grid::iterator it = row->begin();
 257                         label& filename_label = find_widget<label>(*it, "filename", false);
 258 
 259                         bool found = false;
 260                         for(const auto & word : words)
 261                         {
 262                                 found = std::search(filename_label.get_label().str().begin(),
 263                                                                         filename_label.get_label().str().end(),
 264                                                                         word.begin(),
 265                                                                         word.end(),
 266                                                                         chars_equal_insensitive)
 267                                                 != filename_label.get_label().str().end();
 268 
 269                                 if(!found) {
 270                                         // one word doesn't match, we don't reach words.end()
 271                                         break;
 272                                 }
 273                         }
 274 
 275                         show_items[i] = found;
 276                 }
 277         }
 278 
 279         list.set_row_shown(show_items);
 280 
 281         const bool any_shown = list.any_rows_shown();
 282 
 283         // Disable Load button if no games are available
 284         find_widget<button>(&window, "ok", false).set_active(any_shown);
 285 
 286         // Diable 'Enter' loading if no games are available
 287         window.set_enter_disabled(!any_shown);
 288 }

 290 void game_load::evaluate_summary_string(std::stringstream& str, const config& cfg_summary)
 291 {
 292         if(cfg_summary["corrupt"].to_bool()) {
 293                 str << "\n<span color='#f00'>" << _("(Invalid)") << "</span>";
 294 
 295                 return;
 296         }
 297 
 298         const std::string& campaign_type = cfg_summary["campaign_type"];
 299 
 300         try {
 301                 switch(game_classification::CAMPAIGN_TYPE::string_to_enum(campaign_type).v) {
 302                         case game_classification::CAMPAIGN_TYPE::SCENARIO: {
 303                                 const std::string campaign_id = cfg_summary["campaign"];
 304 
 305                                 const config* campaign = nullptr;
 306                                 if(!campaign_id.empty()) {
 307                                         if(const config& c = cache_config_.find_child("campaign", "id", campaign_id)) {
 308                                                 campaign = &c;
 309                                         }
 310                                 }
 311 
 312                                 utils::string_map symbols;
 313                                 if(campaign != nullptr) {
 314                                         symbols["campaign_name"] = (*campaign)["name"];
 315                                 } else {
 316                                         // Fallback to nontranslatable campaign id.
 317                                         symbols["campaign_name"] = "(" + campaign_id + ")";
 318                                 }
 319 
 320                                 str << vgettext("Campaign: $campaign_name", symbols);
 321 
 322                                 // Display internal id for debug purposes if we didn't above
 323                                 if(game_config::debug && (campaign != nullptr)) {
 324                                         str << '\n' << "(" << campaign_id << ")";
 325                                 }
 326                                 break;
 327                         }
 328                         case game_classification::CAMPAIGN_TYPE::MULTIPLAYER:
 329                                 str << _("Multiplayer");
 330                                 break;
 331                         case game_classification::CAMPAIGN_TYPE::TUTORIAL:
 332                                 str << _("Tutorial");
 333                                 break;
 334                         case game_classification::CAMPAIGN_TYPE::TEST:
 335                                 str << _("Test scenario");
 336                                 break;
 337                 }
 338         } catch(bad_enum_cast&) {
 339                 str << campaign_type;
 340         }
 341 
 342         str << "\n";
 343 
 344         if(savegame::loadgame::is_replay_save(cfg_summary)) {
 345                 str << _("Replay");
 346         } else if(!cfg_summary["turn"].empty()) {
 347                 str << _("Turn") << " " << cfg_summary["turn"];
 348         } else {
 349                 str << _("Scenario start");
 350         }
 351 
 352         str << "\n" << _("Difficulty: ")
 353                 << string_table[cfg_summary["difficulty"]];
 354 
 355         if(!cfg_summary["version"].empty()) {
 356                 str << "\n" << _("Version: ") << cfg_summary["version"];
 357         }
 358 }

 360 void game_load::delete_button_callback(window& window)
 361 {
 362         listbox& list = find_widget<listbox>(&window, "savegame_list", false);
 363 
 364         const size_t index = size_t(list.get_selected_row());
 365         if(index < games_.size()) {
 366 
 367                 // See if we should ask the user for deletion confirmation
 368                 if(preferences::ask_delete_saves()) {
 369                         if(!gui2::dialogs::game_delete::execute(window.video())) {
 370                                 return;
 371                         }
 372                 }
 373 
 374                 // Delete the file
 375                 savegame::delete_game(games_[index].name());
 376 
 377                 // Remove it from the list of saves
 378                 games_.erase(games_.begin() + index);
 379 
 380                 list.remove_row(index);
 381 
 382                 // Close the dialog if there are no more saves
 383                 if(list.get_item_count() == 0) {
 384                         window.set_retval(window::CANCEL);
 385                 }
 386 
 387                 display_savegame(window);
 388         }
 389 }
