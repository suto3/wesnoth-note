 102 void scenario::set_sides()
 103 {
 104         if(map_.get() != nullptr) {
 105                 // If there are fewer sides in the configuration than there are
 106                 // starting positions, then generate the additional sides
 107                 const int map_positions = map_->num_valid_starting_positions();
 108 
 109                 for(int pos = data_.child_count("side"); pos < map_positions; ++pos) {
 110                         config& side = data_.add_child("side");
 111                         side["side"] = pos + 1;
 112                         side["team_name"] = "Team " + std::to_string(pos + 1);
 113                         side["canrecruit"] = true;
 114                         side["controller"] = "human";
 115                 }
 116 
 117                 num_players_ = 0;
 118                 for(const config& scenario : data_.child_range("side")) {
 119                         if(scenario["allow_player"].to_bool(true)) {
 120                                 ++num_players_;
 121                         }
 122                 }
 123         }
 124 }

 242 
 243         // Restore game config for multiplayer.
 244         game_classification::CAMPAIGN_TYPE type = state_.classification().campaign_type;
 245 
 246         bool connect = state_.mp_settings().show_connect;
 247 
 248         state_ = saved_game();
 249         state_.classification().campaign_type = type;
 250         state_.mp_settings().show_connect = connect;
 251 
 252         game_config_manager::get()->load_game_config_for_create(type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER);
 253 
 254         // Initialize dependency_manager_ after refreshing game config.
 255         dependency_manager_.reset(new depcheck::manager(
 256                 game_config_manager::get()->game_config(), type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER, video_));
 257 
 258         // TODO: the editor dir is already configurable, is the preferences value
 259         filesystem::get_files_in_dir(filesystem::get_user_data_dir() + "/editor/maps", &user_map_names_,
 260                 nullptr, filesystem::FILE_NAME_ONLY);
 261 
 262         filesystem::get_files_in_dir(filesystem::get_user_data_dir() + "/editor/scenarios", &user_scenario_names_,
 263                 nullptr, filesystem::FILE_NAME_ONLY);
 264 
 265         DBG_MP << "initializing all levels, eras and mods\n";
 266 
 267         init_all_levels();
 268         init_extras(ERA);
 269         init_extras(MOD);
 270 
 271         state_.mp_settings().saved_game = false;
 272 
 273         for(const std::string& str : preferences::modifications(state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER)) {
 274                 if(game_config_manager::get()->game_config().find_child("modification", "id", str)) {
 275                         state_.mp_settings().active_mods.push_back(str);
 276                 }
 277         }
 278 
 279         if(current_level_type_ != level::TYPE::CAMPAIGN &&
 280                 current_level_type_ != level::TYPE::SP_CAMPAIGN) {
 281                 dependency_manager_->try_modifications(state_.mp_settings().active_mods, true);
 282         }
 283 
 284         reset_level_filters();
 285 }

 287 void create_engine::init_generated_level_data()
 288 {
 289         DBG_MP << "initializing generated level data\n";
 290 
 291         //DBG_MP << "current data:\n";
 292         //DBG_MP << current_level().data().debug();
 293 
 294         random_map * cur_lev = dynamic_cast<random_map *> (&current_level());
 295 
 296         if(!cur_lev) {
 297                 WRN_MP << "Tried to initialized generated level data on a level that wasn't a random map\n";
 298                 return;
 299         }
 300 
 301         try {
 302                 if(!cur_lev->generate_whole_scenario())
 303                 {
 304                         DBG_MP << "** replacing map ** \n";
 305 
 306                         config data = cur_lev->data();
 307 
 308                         data["map_data"] = generator_->create_map();
 309 
 310                         cur_lev->set_data(data);
 311 
 312                 } else { //scenario generation
 313 
 314                         DBG_MP << "** replacing scenario ** \n";
 315 
 316                         config data = generator_->create_scenario();
 317 
 318                         // Set the scenario to have placing of sides
 319                         // based on the terrain they prefer
 320                         if(!data.has_attribute("modify_placing")) {
 321                                 data["modify_placing"] = "true";
 322                         }
 323 
 324                         const std::string& description = cur_lev->data()["description"];
 325                         data["description"] = description;
 326 
 327                         cur_lev->set_data(data);
 328                 }
 329         } catch (mapgen_exception & e) {
 330                 config data = cur_lev->data();
 331 
 332                 data["error_message"] = e.what();
 333 
 334                 cur_lev->set_data(data);
 335         }
 336 
 337         //DBG_MP << "final data:\n";
 338         //DBG_MP << current_level().data().debug();
 339 }

 396 std::string create_engine::select_campaign_difficulty(int set_value)
 397 {
 398         // Verify the existence of difficulties
 399         std::vector<std::string> difficulties;
 400 
 401         for(const config& d : current_level().data().child_range("difficulty")) {
 402                 difficulties.push_back(d["define"]);
 403         }
 404 
 405         if(difficulties.empty()) {
 406                 difficulties = utils::split(current_level().data()["difficulties"]);
 407         }
 408 
 409         // No difficulties found. Exit
 410         if(difficulties.empty()) {
 411                 return "";
 412         }
 413 
 414         // One difficulty found. Use it
 415         if(difficulties.size() == 1) {
 416                 return difficulties[0];
 417         }
 418 
 419         // A specific difficulty value was passed
 420         // Use a minimalistic interface to get the specified define
 421         if(set_value != -1) {
 422                 if(set_value > static_cast<int>(difficulties.size())) {
 423                         std::cerr << "incorrect difficulty number: [" <<
 424                                 set_value << "]. maximum is [" << difficulties.size() << "].\n";
 425                         return "FAIL";
 426                 } else if(set_value < 1) {
 427                         std::cerr << "incorrect difficulty number: [" <<
 428                                 set_value << "]. minimum is [1].\n";
 429                         return "FAIL";
 430                 } else {
 431                         return difficulties[set_value - 1];
 432                 }
 433         }
 434 
 435         // If not, let the user pick one from the prompt
 436         // We don't pass the difficulties vector here because additional data is required
 437         // to constrict the dialog
 438         gui2::dialogs::campaign_difficulty dlg(current_level().data());
 439         dlg.show(video_);
 440 
 441         selected_campaign_difficulty_ = dlg.selected_difficulty();
 442 
 443         return selected_campaign_difficulty_;
 444 }

 510 void create_engine::set_current_level(const size_t index)
 511 {
 512         try {
 513                 current_level_index_ = type_map_.at(current_level_type_.v).games_filtered.at(index);
 514         } catch (std::out_of_range&) {
 515                 current_level_index_ = 0u;
 516         }
 517 
 518         if(current_level_type_ == level::TYPE::RANDOM_MAP) {
 519                 random_map* current_random_map = dynamic_cast<random_map*>(&current_level());
 520 
 521                 // If dynamic cast has failed then we somehow have gotten all the pointers mixed together.
 522                 assert(current_random_map);
 523 
 524                 generator_.reset(current_random_map->create_map_generator());
 525         } else {
 526                 generator_.reset(nullptr);
 527         }
 528 
 529         if(current_level_type_ != level::TYPE::CAMPAIGN &&
 530                 current_level_type_ != level::TYPE::SP_CAMPAIGN) {
 531 
 532                 dependency_manager_->try_scenario(current_level().id());
 533         }

 658                 // User maps.
 659                 int dep_index_offset = 0;
 660                 for(size_t i = 0; i < user_map_names_.size(); i++)
 661                 {
 662                         config user_map_data = gen_mp_data;
 663                         user_map_data["map_data"] = filesystem::read_map(user_map_names_[i]);
 664 
 665                         // Check if a file is actually a map.
 666                         // Note that invalid maps should be displayed in order to
 667                         // show error messages in the GUI.
 668                         bool add_map = true;
 669                         std::unique_ptr<gamemap> map;
 670                         try {
 671                                 map.reset(new gamemap(game_config_manager::get()->terrain_types(), user_map_data["map_data"]));
 672                         } catch (incorrect_map_format_error& e) {
 673                                 user_map_data["description"] = _("Map could not be loaded: ") + e.message;
 674 
 675                                 ERR_CF << "map could not be loaded: " << e.message << '\n';
 676                         } catch (wml_exception&) {
 677                                 add_map = false;
 678                                 dep_index_offset++;
 679                         }
 680 
 681                         if(add_map) {
 682                                 user_map_ptr new_user_map(new user_map(user_map_data, user_map_names_[i], map.get()));
 683 
 684                                 type_map_[level::TYPE::USER_MAP].games.push_back(new_user_map);
 685 
 686                                 // Since user maps are treated as scenarios, some dependency info is required
 687                                 config depinfo;
 688                                 depinfo["id"] = user_map_names_[i];
 689                                 depinfo["name"] = user_map_names_[i];
 690                                 dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset);
 691                         }
 692                 }
 693 
 694                 // User made scenarios.
 695                 dep_index_offset = 0;
 696                 for(size_t i = 0; i < user_scenario_names_.size(); i++)
 697                 {
 698                         config data;
 699                         try {
 700                                 read(data, *preprocess_file(filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]));
 701                         } catch (config::error & e) {
 702                                 ERR_CF << "Caught a config error while parsing user made (editor) scenarios:\n" << e.message << std::endl;
 703                                 ERR_CF << "Skipping file: " << (filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]) << std::endl;
 704                                 continue;
 705                         }
 706 
 707                         scenario_ptr new_scenario(new scenario(data));
 708                         if(new_scenario->id().empty()) continue;
 709 
 710                         type_map_[level::TYPE::USER_SCENARIO].games.push_back(new_scenario);
 711 
 712                         // Since user scenarios are treated as scenarios, some dependency info is required
 713                         config depinfo;
 714                         depinfo["id"] = data["id"];
 715                         depinfo["name"] = data["name"];
 716                         dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset++);
 717                 }
 718         }
 719 
 720         // Stand-alone scenarios.
 721         for(const config& data : game_config_manager::get()->game_config().child_range("multiplayer"))
 722         {
 723                 if(!data["allow_new_game"].to_bool(true))
 724                         continue;
 725 
 726                 if(!data["campaign_id"].empty())
 727                         continue;
 728 
 729                 if(data.has_attribute("map_generation") || data.has_attribute("scenario_generation")) {
 730                         random_map_ptr new_random_map(new random_map(data));
 731 
 732                         type_map_[level::TYPE::RANDOM_MAP].games.push_back(new_random_map);
 733                 } else {
 734                         scenario_ptr new_scenario(new scenario(data));
 735 
 736                         type_map_[level::TYPE::SCENARIO].games.push_back(new_scenario);
 737                 }
 738         }
 739 
 740         // Campaigns.
 741         for(const config& data : game_config_manager::get()->game_config().child_range("campaign"))
 742         {
 743                 const std::string& type = data["type"];
 744                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 745 
 746                 if(type == "mp" || (type == "hybrid" && mp)) {
 747                         campaign_ptr new_campaign(new campaign(data));
 748 
 749                         type_map_[level::TYPE::CAMPAIGN].games.push_back(new_campaign);
 750                 }
 751 
 752                 if(type == "sp" || type.empty() || (type == "hybrid" && !mp)) {
 753                         campaign_ptr new_sp_campaign(new campaign(data));
 754                         new_sp_campaign->mark_if_completed();
 755 
 756                         type_map_[level::TYPE::SP_CAMPAIGN].games.push_back(new_sp_campaign);
 757                 }
 758         }
 759 
 760         auto& sp_campaigns = type_map_[level::TYPE::SP_CAMPAIGN].games;
 761 
 762         // Sort sp campaigns by rank.
 763         std::stable_sort(sp_campaigns.begin(), sp_campaigns.end(),
 764         [](const create_engine::level_ptr& a, const create_engine::level_ptr& b) {
 765                         return a->data()["rank"].to_int(1000) < b->data()["rank"].to_int(1000);
 766                 }
 767     );
 768 }

 770 void create_engine::init_extras(const MP_EXTRA extra_type)
 771 {
 772         std::vector<extras_metadata_ptr>& extras = get_extras_by_type(extra_type);
 773         const std::string extra_name = (extra_type == ERA) ? "era" : "modification";
 774 
 775         ng::depcheck::component_availabilty default_availabilty = (extra_type == ERA)
 776                 ? ng::depcheck::component_availabilty::MP
 777                 : ng::depcheck::component_availabilty::HYBRID;
 778 
 779         for(const config& extra : game_config_manager::get()->game_config().child_range(extra_name))
 780         {
 781                 ng::depcheck::component_availabilty type = extra["type"].to_enum(default_availabilty);
 782                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 783 
 784                 if((type != ng::depcheck::component_availabilty::MP || mp) && (type != ng::depcheck::component_availabilty::SP || !mp) )
 785                 {
 786                         extras_metadata_ptr new_extras_metadata(new extras_metadata());
 787                         new_extras_metadata->id = extra["id"].str();
 788                         new_extras_metadata->name = extra["name"].str();
 789                         new_extras_metadata->description = extra["description"].str();
 790                         new_extras_metadata->cfg = &extra;
 791 
 792                         extras.push_back(new_extras_metadata);
 793                 }
 794         }
 795 }
