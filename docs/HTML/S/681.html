 399 bool locator::value::operator==(const value& a) const
 400 {
 401         if(a.type_ != type_) {
 402                 return false;
 403         } else if(type_ == FILE) {
 404                 return filename_ == a.filename_;
 405         } else if(type_ == SUB_FILE) {
 406                 return filename_ == a.filename_ && loc_ == a.loc_ && modifications_ == a.modifications_
 407                            && center_x_ == a.center_x_ && center_y_ == a.center_y_;
 408         }
 409 
 410         return false;
 411 }
 412 
 413 bool locator::value::operator<(const value& a) const
 414 {
 415         if(type_ != a.type_) {
 416                 return type_ < a.type_;
 417         } else if(type_ == FILE) {
 418                 return filename_ < a.filename_;
 419         } else if(type_ == SUB_FILE) {
 420                 if(filename_ != a.filename_)
 421                         return filename_ < a.filename_;
 422                 if(loc_ != a.loc_)
 423                         return loc_ < a.loc_;
 424                 if(center_x_ != a.center_x_)
 425                         return center_x_ < a.center_x_;
 426                 if(center_y_ != a.center_y_)
 427                         return center_y_ < a.center_y_;
 428                 return (modifications_ < a.modifications_);
 429         }
 430 
 431         return false;
 432 }

 434 // Check if localized file is up-to-date according to l10n track index.
 435 // Make sure only that the image is not explicitly recorded as fuzzy,
 436 // in order to be able to use non-tracked images (e.g. from UMC).
 437 static std::set<std::string> fuzzy_localized_files;
 438 static bool localized_file_uptodate(const std::string& loc_file)
 439 {
 440         if(fuzzy_localized_files.empty()) {
 441                 // First call, parse track index to collect fuzzy files by path.
 442                 std::string fsep = "\xC2\xA6"; // UTF-8 for "broken bar"
 443                 std::string trackpath = filesystem::get_binary_file_location("", "l10n-track");
 444                 std::string contents = filesystem::read_file(trackpath);
 445 
 446                 for(const std::string& line : utils::split(contents, '\n')) {
 447                         size_t p1 = line.find(fsep);
 448                         if(p1 == std::string::npos) {
 449                                 continue;
 450                         }
 451 
 452                         std::string state = line.substr(0, p1);
 453                         boost::trim(state);
 454                         if(state == "fuzzy") {
 455                                 size_t p2 = line.find(fsep, p1 + fsep.length());
 456                                 if(p2 == std::string::npos) {
 457                                         continue;
 458                                 }
 459 
 460                                 std::string relpath = line.substr(p1 + fsep.length(), p2 - p1 - fsep.length());
 461                                 fuzzy_localized_files.insert(game_config::path + '/' + relpath);
 462                         }
 463                 }
 464 
 465                 fuzzy_localized_files.insert(""); // make sure not empty any more
 466         }
 467 
 468         return fuzzy_localized_files.count(loc_file) == 0;
 469 }

 471 // Return path to localized counterpart of the given file, if any, or empty string.
 472 // Localized counterpart may also be requested to have a suffix to base name.
 473 static std::string get_localized_path(const std::string& file, const std::string& suff = "")
 474 {
 475         std::string dir = filesystem::directory_name(file);
 476         std::string base = filesystem::base_name(file);
 477 
 478         const size_t pos_ext = base.rfind(".");
 479 
 480         std::string loc_base;
 481         if(pos_ext != std::string::npos) {
 482                 loc_base = base.substr(0, pos_ext) + suff + base.substr(pos_ext);
 483         } else {
 484                 loc_base = base + suff;
 485         }
 486 
 487         // TRANSLATORS: This is the language code which will be used
 488         // to store and fetch localized non-textual resources, such as images,
 489         // when they exist. Normally it is just the code of the PO file itself,
 490         // e.g. "de" of de.po for German. But it can also be a comma-separated
 491         // list of language codes by priority, when the localized resource
 492         // found for first of those languages will be used. This is useful when
 493         // two languages share sufficient commonality, that they can use each
 494         // other's resources rather than duplicating them. For example,
 495         // Swedish (sv) and Danish (da) are such, so Swedish translator could
 496         // translate this message as "sv,da", while Danish as "da,sv".
 497         std::vector<std::string> langs = utils::split(_("language code for localized resources^en_US"));
 498 
 499         // In case even the original image is split into base and overlay,
 500         // add en_US with lowest priority, since the message above will
 501         // not have it when translated.
 502         langs.push_back("en_US");
 503         for(const std::string& lang : langs) {
 504                 std::string loc_file = dir + "/" + "l10n" + "/" + lang + "/" + loc_base;
 505                 if(filesystem::file_exists(loc_file) && localized_file_uptodate(loc_file)) {
 506                         return loc_file;
 507                 }
 508         }
 509 
 510         return "";
 511 }

 527 static surface load_image_file(const image::locator& loc)
 528 {
 529         surface res;
 530 
 531         std::string location = filesystem::get_binary_file_location("images", loc.get_filename());
 532 
 533         {
 534                 if(!location.empty()) {
 535                         // Check if there is a localized image.
 536                         const std::string loc_location = get_localized_path(location);
 537                         if(!loc_location.empty()) {
 538                                 location = loc_location;
 539                         }
 540                         SDL_RWops* rwops = filesystem::load_RWops(location);
 541                         res = IMG_Load_RW(rwops, true); // SDL takes ownership of rwops
 542                         // If there was no standalone localized image, check if there is an overlay.
 543                         if(!res.null() && loc_location.empty()) {
 544                                 const std::string ovr_location = get_localized_path(location, "--overlay");
 545                                 if(!ovr_location.empty()) {
 546                                         add_localized_overlay(ovr_location, res);
 547                                 }
 548                         }
 549                 }
 550         }
 551 
 552         if(res.null() && !loc.get_filename().empty()) {
 553                 ERR_DP << "could not open image '" << loc.get_filename() << "'" << std::endl;
 554                 if(game_config::debug && loc.get_filename() != game_config::images::missing)
 555                         return get_image(game_config::images::missing, UNSCALED);
 556         }
 557 
 558         return res;
 559 }

 561 static surface load_image_sub_file(const image::locator& loc)
 562 {
 563         surface surf = get_image(loc.get_filename(), UNSCALED);
 564         if(surf == nullptr) {
 565                 return nullptr;
 566         }
 567 
 568         modification_queue mods = modification::decode(loc.get_modifications());
 569 
 570         while(!mods.empty()) {
 571                 modification* mod = mods.top();
 572 
 573                 try {
 574                         surf = (*mod)(surf);
 575                 } catch(const image::modification::imod_exception& e) {
 576                         ERR_CFG << "Failed to apply a modification to an image:\n"
 577                                         << "Image: " << loc.get_filename() << ".\n"
 578                                         << "Modifications: " << loc.get_modifications() << ".\n"
 579                                         << "Error: " << e.message;
 580                 }
 581 
 582                 // NOTE: do this *after* applying the mod or you'll get crashes!
 583                 mods.pop();
 584         }
 585 
 586         if(loc.get_loc().valid()) {
 587                 SDL_Rect srcrect = sdl::create_rect(
 588                         ((tile_size * 3) / 4)                           *  loc.get_loc().x,
 589                           tile_size * loc.get_loc().y + (tile_size / 2) * (loc.get_loc().x % 2),
 590                           tile_size,
 591                           tile_size
 592                 );
 593 
 594                 if(loc.get_center_x() >= 0 && loc.get_center_y() >= 0) {
 595                         srcrect.x += surf->w / 2 - loc.get_center_x();
 596                         srcrect.y += surf->h / 2 - loc.get_center_y();
 597                 }
 598 
 599                 // cut and hex mask, but also check and cache if empty result
 600                 surface cut(cut_surface(surf, srcrect));
 601                 bool is_empty = false;
 602                 surf = mask_surface(cut, get_hexmask(), &is_empty);
 603 
 604                 // discard empty images to free memory
 605                 if(is_empty) {
 606                         // Safe because those images are only used by terrain rendering
 607                         // and it filters them out.
 608                         // A safer and more general way would be to keep only one copy of it
 609                         surf = nullptr;
 610                 }
 611 
 612                 loc.add_to_cache(is_empty_hex_, is_empty);
 613         }
 614 
 615         return surf;
 616 }

 636 static surface apply_light(surface surf, const light_string& ls)
 637 {
 638         // atomic lightmap operation are handled directly (important to end recursion)
 639         if(ls.size() == 4) {
 640                 // if no lightmap (first char = -1) then we need the initial value
 641                 //(before the halving done for lightmap)
 642                 int m = ls[0] == -1 ? 2 : 1;
 643                 return adjust_surface_color(surf, ls[1] * m, ls[2] * m, ls[3] * m);
 644         }
 645 
 646         // check if the lightmap is already cached or need to be generated
 647         surface lightmap = nullptr;
 648         auto i = lightmaps_.find(ls);
 649         if(i != lightmaps_.end()) {
 650                 lightmap = i->second;
 651         } else {
 652                 // build all the paths for lightmap sources
 653                 static const std::string p = "terrain/light/light";
 654                 static const std::string lm_img[19] {
 655                         p + ".png",
 656                         p + "-concave-2-tr.png", p + "-concave-2-r.png", p + "-concave-2-br.png",
 657                         p + "-concave-2-bl.png", p + "-concave-2-l.png", p + "-concave-2-tl.png",
 658                         p + "-convex-br-bl.png", p + "-convex-bl-l.png", p + "-convex-l-tl.png",
 659                         p + "-convex-tl-tr.png", p + "-convex-tr-r.png", p + "-convex-r-br.png",
 660                         p + "-convex-l-bl.png",  p + "-convex-tl-l.png", p + "-convex-tr-tl.png",
 661                         p + "-convex-r-tr.png",  p + "-convex-br-r.png", p + "-convex-bl-br.png"
 662                 };
 663 
 664                 // decompose into atomic lightmap operations (4 chars)
 665                 for(size_t c = 0; c + 3 < ls.size(); c += 4) {
 666                         light_string sls = ls.substr(c, 4);
 667 
 668                         // get the corresponding image and apply the lightmap operation to it
 669                         // This allows to also cache lightmap parts.
 670                         // note that we avoid infinite recursion by using only atomic operation
 671                         surface lts = image::get_lighted_image(lm_img[sls[0]], sls, HEXED);
 672 
 673                         // first image will be the base where we blit the others
 674                         if(lightmap == nullptr) {
 675                                 // copy the cached image to avoid modifying the cache
 676                                 lightmap = make_neutral_surface(lts);
 677                         } else {
 678                                 sdl_blit(lts, nullptr, lightmap, nullptr);
 679                         }
 680                 }
 681 
 682                 // cache the result
 683                 lightmaps_[ls] = lightmap;
 684         }
 685 
 686         // apply the final lightmap
 687         return light_surface(surf, lightmap);
 688 }

 802 {

 826                 scaling_function result = &scale_xbrz_helper<scale_surface_nn>;
 827                 return result;
 828         }
 829         default:
 830                 assert(false && "I don't know how to implement this scaling algorithm");
 831                 throw 42;
 832         }

 884 /// translate type to a simpler one when possible
 885 static TYPE simplify_type(const image::locator& i_locator, TYPE type)
 886 {
 887         switch(type) {
 888         case SCALED_TO_ZOOM:
 889                 if(zoom == tile_size) {
 890                         type = UNSCALED;
 891                 }
 892 
 893                 break;
 894         case BRIGHTENED:
 895                 if(ftofxp(game_config::hex_brightening) == ftofxp(1.0)) {
 896                         type = TOD_COLORED;
 897                 }
 898 
 899                 break;
 900         default:
 901                 break;
 902         }
 903 
 904         if(type == TOD_COLORED) {
 905                 if(red_adjust == 0 && green_adjust == 0 && blue_adjust == 0) {
 906                         type = SCALED_TO_HEX;
 907                 }
 908         }
 909 
 910         if(type == SCALED_TO_HEX) {
 911                 if(zoom == tile_size) {
 912                         type = HEXED;
 913                 }
 914         }
 915 
 916         if(type == HEXED) {
 917                 // check if the image is already hex-cut by the location system
 918                 if(i_locator.get_loc().valid()) {
 919                         type = UNSCALED;
 920                 }
 921         }
 922 
 923         return type;
 924 }

 926 surface get_image(const image::locator& i_locator, TYPE type)
 927 {
 928         surface res;
 929 
 930         if(i_locator.is_void()) {
 931                 return res;
 932         }
 933 
 934         type = simplify_type(i_locator, type);
 935 
 936         image_cache* imap;
 937         // select associated cache
 938         switch(type) {
 939         case UNSCALED:
 940                 imap = &images_;
 941                 break;
 942         case TOD_COLORED:
 943                 imap = &tod_colored_images_;
 944                 break;
 945         case SCALED_TO_ZOOM:
 946                 imap = &scaled_to_zoom_;
 947                 break;
 948         case HEXED:
 949                 imap = &hexed_images_;
 950                 break;
 951         case SCALED_TO_HEX:
 952                 imap = &scaled_to_hex_images_;
 953                 break;
 954         case BRIGHTENED:
 955                 imap = &brightened_images_;
 956                 break;
 957         default:
 958                 return res;
 959         }
 960 
 961         // return the image if already cached
 962         bool tmp;
 963 #ifdef _OPENMP
 964 #pragma omp critical(image_cache)
 965 #endif //_OPENMP
 966         tmp = i_locator.in_cache(*imap);
 967 
 968         if(tmp) {
 969                 surface result;
 970 #ifdef _OPENMP
 971 #pragma omp critical(image_cache)
 972 #endif //_OPENMP
 973                 result = i_locator.locate_in_cache(*imap);
 974                 return result;
 975         }
 976 
 977         // not cached, generate it
 978         switch(type) {
 979         case UNSCALED:
 980                 // If type is unscaled, directly load the image from the disk.
 981                 res = load_from_disk(i_locator);
 982                 break;
 983         case TOD_COLORED:
 984                 res = get_tod_colored(i_locator);
 985                 break;
 986         case SCALED_TO_ZOOM:
 987                 res = get_scaled_to_zoom(i_locator);
 988                 break;
 989         case HEXED:
 990                 res = get_hexed(i_locator);
 991                 break;
 992         case SCALED_TO_HEX:
 993                 res = get_scaled_to_hex(i_locator);
 994                 break;
 995         case BRIGHTENED:
 996                 res = get_brightened(i_locator);
 997                 break;
 998         default:
 999                 return res;
1000         }
1001 
1002 #ifdef _OPENMP
1003 #pragma omp critical(image_cache)
1004 #endif //_OPENMP
1005         i_locator.add_to_cache(*imap, res);
1006 
1007         return res;
1008 }

1010 surface get_lighted_image(const image::locator& i_locator, const light_string& ls, TYPE type)
1011 {
1012         surface res;
1013         if(i_locator.is_void()) {
1014                 return res;
1015         }
1016 
1017         if(type == SCALED_TO_HEX && zoom == tile_size) {
1018                 type = HEXED;
1019         }
1020 
1021         // select associated cache
1022         lit_cache* imap = &lit_images_;
1023         if(type == SCALED_TO_HEX) {
1024                 imap = &lit_scaled_images_;
1025         }
1026 
1027         // if no light variants yet, need to add an empty map
1028         if(!i_locator.in_cache(*imap)) {
1029                 i_locator.add_to_cache(*imap, lit_variants());
1030         }
1031 
1032         // need access to add it if not found
1033         { // enclose reference pointing to data stored in a changing vector
1034                 const lit_variants& lvar = i_locator.locate_in_cache(*imap);
1035                 auto lvi = lvar.find(ls);
1036                 if(lvi != lvar.end()) {
1037                         return lvi->second;
1038                 }
1039         }
1040 
1041         // not cached yet, generate it
1042         switch(type) {
1043         case HEXED:
1044                 res = get_image(i_locator, HEXED);
1045                 res = apply_light(res, ls);
1046                 break;
1047         case SCALED_TO_HEX:
1048                 // we light before scaling to reuse the unscaled cache
1049                 res = get_lighted_image(i_locator, ls, HEXED);
1050                 res = scale_surface(res, zoom, zoom);
1051                 break;
1052         default:
1053                 break;
1054         }
1055 
1056         // record the lighted surface in the corresponding variants cache
1057         i_locator.access_in_cache(*imap)[ls] = res;
1058 
1059         return res;
1060 }

1113 surface reverse_image(const surface& surf)
1114 {
1115         if(surf == nullptr) {
1116                 return surface(nullptr);
1117         }
1118 
1119         const auto itor = reversed_images_.find(surf);
1120         if(itor != reversed_images_.end()) {
1121                 // sdl_add_ref(itor->second);
1122                 return itor->second;
1123         }
1124 
1125         const surface rev(flip_surface(surf));
1126         if(rev == nullptr) {
1127                 return surface(nullptr);
1128         }
1129 
1130         reversed_images_.emplace(surf, rev);
1131         // sdl_add_ref(rev);
1132         return rev;
1133 }

1152                 cache = !filesystem::get_binary_file_location("images", i_locator.get_filename()).empty();
1153         }

1158 static void precache_file_existence_internal(const std::string& dir, const std::string& subdir)
1159 {
1160         const std::string checked_dir = dir + "/" + subdir;
1161         if(precached_dirs.find(checked_dir) != precached_dirs.end()) {
1162                 return;
1163         }
1164 
1165         precached_dirs.insert(checked_dir);
1166 
1167         if(!filesystem::is_directory(checked_dir)) {
1168                 return;
1169         }
1170 
1171         std::vector<std::string> files_found;
1172         std::vector<std::string> dirs_found;
1173         filesystem::get_files_in_dir(checked_dir, &files_found, &dirs_found, filesystem::FILE_NAME_ONLY,
1174                         filesystem::NO_FILTER, filesystem::DONT_REORDER);
1175 
1176         for(const auto& f : files_found) {
1177                 image_existence_map[subdir + f] = true;
1178         }
1179 
1180         for(const auto& d : dirs_found) {
1181                 precache_file_existence_internal(dir, subdir + d + "/");
1182         }
1183 }
