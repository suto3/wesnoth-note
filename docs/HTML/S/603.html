  47 /**
  48  * Creates a bound status label that will reflect the label state of a widget. The initial label
  49  * value is set here, and then again any time the widget is modified. A function is also returned
  50  * that can be called to update the label manually.
  51  *
  52  * This relies on hooking into the NOTIFY_MODIFIED event, so can only be used with widgets that fire
  53  * that event.
  54  */
  55 template<typename W>
  56 std::function<void()> bind_status_label(widget& find_in, const std::string& id,
  57                 const std::function<std::string(W&)> value_getter = default_value_getter<W>,
  58                 const std::string& label_id = "")
  59 {
  60         const std::string label_id_ = label_id.empty() ? id + "_label" : label_id;
  61 
  62         W& source = find_widget<W>(&find_in, id, false);
  63         styled_widget& label = find_widget<styled_widget>(&find_in, label_id_, false);
  64 
  65         const auto update_label = [&, value_getter]() {
  66                 const std::string value = value_getter(source);
  67 
  68                 label.set_label(value);
  69         };
  70 
  71         // Bind the callback
  72         connect_signal_notify_modified(source, std::bind(update_label));
  73 
  74         // Set initial value
  75         update_label();
  76 
  77         return update_label;
  78 }
