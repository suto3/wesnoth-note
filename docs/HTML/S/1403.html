 207     /**
 208      * Checks the mandatory fields and updates the isPageComplete status
 209      */
 210     private void updatePageIsComplete( )
 211     {
 212         setPageComplete( false );
 213 
 214         IResource container = ResourcesPlugin.getWorkspace( ).getRoot( )
 215             .findMember( new Path( getProjectName( ) ) );
 216         String fileName = getFileName( );
 217         String warningMessage = ""; //$NON-NLS-1$
 218 
 219         if( getProjectName( ).isEmpty( ) ) {
 220             setErrorMessage( Messages.ScenarioPage0_14 );
 221             return;
 222         }
 223         if( container == null
 224             || ( container.getType( ) & ( IResource.PROJECT | IResource.FOLDER ) ) == 0 ) {
 225             setErrorMessage( Messages.ScenarioPage0_15 );
 226             return;
 227         }
 228 
 229         if( ! container.isAccessible( ) ) {
 230             setErrorMessage( Messages.ScenarioPage0_16 );
 231             return;
 232         }
 233 
 234         if( ! getProjectName( ).contains( "scenarios" ) ) //$NON-NLS-1$
 235         {
 236             warningMessage += Messages.ScenarioPage0_18;
 237         }
 238 
 239         if( fileName.isEmpty( ) ) {
 240             setErrorMessage( Messages.ScenarioPage0_19 );
 241             return;
 242         }
 243         if( fileName.replace( '\\', '/' ).indexOf( '/', 1 ) > 0 ) {
 244             setErrorMessage( Messages.ScenarioPage0_20 );
 245             return;
 246         }
 247 
 248         int dotLoc = fileName.lastIndexOf( '.' );
 249         if( dotLoc == - 1
 250             || ! ( fileName.substring( dotLoc + 1 )
 251                 .equalsIgnoreCase( "cfg" ) ) ) //$NON-NLS-1$
 252         {
 253             setErrorMessage( Messages.ScenarioPage0_22 );
 254             return;
 255         }
 256 
 257         if( txtScenarioId_.getText( ).isEmpty( ) ) {
 258             setErrorMessage( Messages.ScenarioPage0_23 );
 259             return;
 260         }
 261 
 262         setErrorMessage( null );
 263         setMessage( warningMessage.isEmpty( ) ? null: warningMessage, WARNING );
 264         setPageComplete( true );
 265     }

 267     private void updateMapPath( )
 268     {
 269         if( txtMapData_ == null ) {
 270             return;
 271         }
 272 
 273         if( rawMapPath_ == null ) {
 274             txtMapData_.setText( "" ); //$NON-NLS-1$
 275             return;
 276         }
 277 
 278         // make the map path to be relative to the ~addons directory
 279 
 280         String homePath = txtProject_.getText( );
 281         if( ! homePath.isEmpty( ) && homePath.charAt( 0 ) == '/' ) {
 282             homePath = homePath.substring( 1 );
 283         }
 284 
 285         txtMapData_.setText( String.format( "{~add-ons/%s/maps/%s}", //$NON-NLS-1$
 286             homePath, new File( rawMapPath_ ).getName( ) ) );
 287     }

 289     private void handleBrowseMap( )
 290     {
 291         FileDialog dialog = new FileDialog( getShell( ) );
 292         dialog.setText( Messages.ScenarioPage0_26 );
 293         if( container_ != null ) {
 294             dialog.setFilterPath( container_.getLocation( ).toOSString( ) );
 295         }
 296         dialog.setFilterExtensions( new String[] { "*.map" } ); //$NON-NLS-1$
 297         rawMapPath_ = dialog.open( );
 298         updateMapPath( );
 299     }
