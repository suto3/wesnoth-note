  49 bool playmp_controller::hotkey_handler::can_execute_command(const hotkey::hotkey_command& cmd, int index) const
  50 {
  51         hotkey::HOTKEY_COMMAND command = cmd.id;
  52         bool res = true;
  53         switch (command){
  54                 case hotkey::HOTKEY_ENDTURN:
  55                         if  (linger())
  56                         {
  57                                 bool has_next_scenario = !gamestate().gamedata_.next_scenario().empty() &&
  58                                         gamestate().gamedata_.next_scenario() != "null";
  59                                 return playmp_controller_.is_host() || !has_next_scenario;
  60                         }
  61                         else
  62                         {
  63                                 return playsingle_controller::hotkey_handler::can_execute_command(cmd, index);
  64                         }
  65                 case hotkey::HOTKEY_SPEAK:
  66                 case hotkey::HOTKEY_SPEAK_ALLY:
  67                 case hotkey::HOTKEY_SPEAK_ALL:
  68                         res = playmp_controller_.is_networked_mp();
  69                         break;
  70                 case hotkey::HOTKEY_START_NETWORK:
  71                 case hotkey::HOTKEY_STOP_NETWORK:
  72                         res = is_observer();
  73                         break;
  74             default:
  75                         return playsingle_controller::hotkey_handler::can_execute_command(cmd, index);
  76         }
  77         return res;
  78 }
