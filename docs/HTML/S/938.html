  83 static void increment_chunk_usage(Mix_Chunk* mcp)
  84 {
  85         ++(chunk_usage[mcp]);
  86 }
  87 
  88 static void decrement_chunk_usage(Mix_Chunk* mcp)
  89 {
  90         if(mcp == nullptr) {
  91                 return;
  92         }
  93 
  94         std::map<Mix_Chunk*, int>::iterator this_usage = chunk_usage.find(mcp);
  95         assert(this_usage != chunk_usage.end());
  96         if(--(this_usage->second) == 0) {
  97                 Mix_FreeChunk(mcp);
  98                 chunk_usage.erase(this_usage);
  99         }
 100 }

 242 static bool track_ok(const std::string& id)
 243 {
 244         LOG_AUDIO << "Considering " << id << "\n";
 245 
 246         // If they committed changes to list, we forget previous plays, but
 247         // still *never* repeat same track twice if we have an option.
 248         if(id == current_track->file_path()) {
 249                 return false;
 250         }
 251 
 252         if(current_track_list.size() <= 3) {
 253                 return true;
 254         }
 255 
 256         // Timothy Pinkham says:
 257         // 1) can't be repeated without 2 other pieces have already played
 258         // since A was played.
 259         // 2) cannot play more than 2 times without every other piece
 260         // having played at least 1 time.
 261 
 262         // Dammit, if our musicians keep coming up with algorithms, I'll
 263         // be out of a job!
 264         unsigned int num_played = 0;
 265         std::set<std::string> played;
 266         std::vector<std::string>::reverse_iterator i;
 267 
 268         for(i = played_before.rbegin(); i != played_before.rend(); ++i) {
 269                 if(*i == id) {
 270                         ++num_played;
 271                         if(num_played == 2) {
 272                                 break;
 273                         }
 274                 } else {
 275                         played.insert(*i);
 276                 }
 277         }
 278 
 279         // If we've played this twice, must have played every other track.
 280         if(num_played == 2 && played.size() != current_track_list.size() - 1) {
 281                 LOG_AUDIO << "Played twice with only " << played.size() << " tracks between\n";
 282                 return false;
 283         }
 284 
 285         // Check previous previous track not same.
 286         i = played_before.rbegin();
 287         if(i != played_before.rend()) {
 288                 ++i;
 289                 if(i != played_before.rend()) {
 290                         if(*i == id) {
 291                                 LOG_AUDIO << "Played just before previous\n";
 292                                 return false;
 293                         }
 294                 }
 295         }
 296 
 297         return true;
 298 }

 300 static std::shared_ptr<sound::music_track> choose_track()
 301 {
 302         assert(!current_track_list.empty());
 303 
 304         if(current_track_index >= current_track_list.size()) {
 305                 current_track_index = 0;
 306         }
 307 
 308         if(current_track_list[current_track_index]->shuffle()) {
 309                 unsigned int track = 0;
 310 
 311                 if(current_track_list.size() > 1) {
 312                         do {
 313                                 track = rand() % current_track_list.size();
 314                         } while(!track_ok(current_track_list[track]->file_path()));
 315                 }
 316 
 317                 current_track_index = track;
 318         }
 319 
 320         DBG_AUDIO << "Next track will be " << current_track_list[current_track_index]->file_path() << "\n";
 321         played_before.push_back(current_track_list[current_track_index]->file_path());
 322         return current_track_list[current_track_index];
 323 }

 325 static std::string pick_one(const std::string& files)
 326 {
 327         std::vector<std::string> ids = utils::square_parenthetical_split(files, ',', "[", "]");
 328 
 329         if(ids.empty()) {
 330                 return "";
 331         }
 332 
 333         if(ids.size() == 1) {
 334                 return ids[0];
 335         }
 336 
 337         // We avoid returning same choice twice if we can avoid it.
 338         static std::map<std::string, unsigned int> prev_choices;
 339         unsigned int choice;
 340 
 341         if(prev_choices.find(files) != prev_choices.end()) {
 342                 choice = rand() % (ids.size() - 1);
 343                 if(choice >= prev_choices[files]) {
 344                         ++choice;
 345                 }
 346 
 347                 prev_choices[files] = choice;
 348         } else {
 349                 choice = rand() % ids.size();
 350                 prev_choices.emplace(files, choice);
 351         }
 352 
 353         return ids[choice];
 354 }

 382 bool init_sound()
 383 {
 384         LOG_AUDIO << "Initializing audio...\n";
 385         if(SDL_WasInit(SDL_INIT_AUDIO) == 0) {
 386                 if(SDL_InitSubSystem(SDL_INIT_AUDIO) == -1) {
 387                         return false;
 388                 }
 389         }
 390 
 391         if(!mix_ok) {
 392                 if(Mix_OpenAudio(preferences::sample_rate(), MIX_DEFAULT_FORMAT, 2, preferences::sound_buffer_size()) == -1) {
 393                         mix_ok = false;
 394                         ERR_AUDIO << "Could not initialize audio: " << Mix_GetError() << std::endl;
 395                         return false;
 396                 }
 397 
 398                 mix_ok = true;
 399                 Mix_AllocateChannels(n_of_channels);
 400                 Mix_ReserveChannels(n_reserved_channels);
 401 
 402                 channel_chunks.clear();
 403                 channel_chunks.resize(n_of_channels, nullptr);
 404                 channel_ids.resize(n_of_channels, -1);
 405 
 406                 Mix_GroupChannel(bell_channel, SOUND_BELL);
 407                 Mix_GroupChannel(timer_channel, SOUND_TIMER);
 408                 Mix_GroupChannels(source_channel_start, source_channel_last, SOUND_SOURCES);
 409                 Mix_GroupChannels(UI_sound_channel_start, UI_sound_channel_last, SOUND_UI);
 410                 Mix_GroupChannels(n_reserved_channels, n_of_channels - 1, SOUND_FX);
 411 
 412                 set_sound_volume(preferences::sound_volume());
 413                 set_UI_volume(preferences::UI_volume());
 414                 set_music_volume(preferences::music_volume());
 415                 set_bell_volume(preferences::bell_volume());
 416 
 417                 Mix_ChannelFinished(channel_finished_hook);
 418 
 419                 LOG_AUDIO << "Audio initialized.\n";
 420 
 421                 DBG_AUDIO << "Channel layout: " << n_of_channels << " channels (" << n_reserved_channels << " reserved)\n"
 422                           << "    " << bell_channel << " - bell\n"
 423                           << "    " << timer_channel << " - timer\n"
 424                           << "    " << source_channel_start << ".." << source_channel_last << " - sound sources\n"
 425                           << "    " << UI_sound_channel_start << ".." << UI_sound_channel_last << " - UI\n"
 426                           << "    " << UI_sound_channel_last + 1 << ".." << n_of_channels - 1 << " - sound effects\n";
 427 
 428                 play_music();
 429         }
 430 
 431         return true;
 432 }

 434 void close_sound()
 435 {
 436         int frequency, channels;
 437         Uint16 format;
 438 
 439         if(mix_ok) {
 440                 stop_bell();
 441                 stop_UI_sound();
 442                 stop_sound();
 443                 sound_cache.clear();
 444                 stop_music();
 445                 mix_ok = false;
 446 
 447                 int numtimesopened = Mix_QuerySpec(&frequency, &format, &channels);
 448                 if(numtimesopened == 0) {
 449                         ERR_AUDIO << "Error closing audio device: " << Mix_GetError() << std::endl;
 450                 }
 451 
 452                 while(numtimesopened) {
 453                         Mix_CloseAudio();
 454                         --numtimesopened;
 455                 }
 456         }
 457 
 458         if(SDL_WasInit(SDL_INIT_AUDIO) != 0) {
 459                 SDL_QuitSubSystem(SDL_INIT_AUDIO);
 460         }
 461 
 462         LOG_AUDIO << "Audio device released.\n";
 463 }

 578 static void play_new_music()
 579 {
 580         music_start_time = 0; // reset status: no start time
 581         want_new_music = true;
 582 
 583         if(!preferences::music_on() || !mix_ok || !current_track->valid()) {
 584                 return;
 585         }
 586 
 587         const std::string& filename = current_track->file_path();
 588 
 589         auto itor = music_cache.find(filename);
 590         if(itor == music_cache.end()) {
 591                 LOG_AUDIO << "attempting to insert track '" << filename << "' into cache\n";
 592 
 593                 SDL_RWops* rwops = filesystem::load_RWops(filename);
 594                 // SDL takes ownership of rwops
 595                 const std::shared_ptr<Mix_Music> music(Mix_LoadMUSType_RW(rwops, MUS_NONE, true), &Mix_FreeMusic);
 596 
 597                 if(music == nullptr) {
 598                         ERR_AUDIO << "Could not load music file '" << filename << "': " << Mix_GetError() << "\n";
 599                         return;
 600                 }
 601 
 602                 itor = music_cache.emplace(filename, music).first;
 603         }
 604 
 605         LOG_AUDIO << "Playing track '" << filename << "'\n";
 606         int fading_time = current_track->ms_before();
 607         if(no_fading) {
 608                 fading_time = 0;
 609         }
 610 
 611         const int res = Mix_FadeInMusic(itor->second.get(), 1, fading_time);
 612         if(res < 0) {
 613                 ERR_AUDIO << "Could not play music: " << Mix_GetError() << " " << filename << " " << std::endl;
 614         }
 615 
 616         want_new_music = false;
 617 }

 619 void play_music_repeatedly(const std::string& id)
 620 {
 621         // Can happen if scenario doesn't specify.
 622         if(id.empty()) {
 623                 return;
 624         }
 625 
 626         current_track_list.clear();
 627         current_track_list.emplace_back(new music_track(id));
 628 
 629         std::shared_ptr<music_track> last_track = current_track;
 630         current_track = current_track_list.back();
 631         current_track_index = 0;
 632 
 633         // If we're already playing it, don't interrupt.
 634         if(!last_track || *last_track != *current_track) {
 635                 play_music();
 636         }
 637 
 638         last_track.reset();
 639 }

 641 void play_music_config(const config& music_node, int i)
 642 {
 643         //
 644         // FIXME: there is a memory leak somewhere in this function, seemingly related to the shared_ptrs
 645         // stored in current_track_list.
 646         //
 647         // vultraz 5/8/2017
 648         //
 649 
 650         music_track track(music_node);
 651 
 652         if(!track.valid() && !track.id().empty()) {
 653                 ERR_AUDIO << "cannot open track '" << track.id() << "'; disabled in this playlist." << std::endl;
 654         }
 655 
 656         // If they say play once, we don't alter playlist.
 657         if(track.play_once()) {
 658                 current_track = std::make_shared<music_track>(track);
 659                 current_track_index = current_track_list.size();
 660                 play_music();
 661                 return;
 662         }
 663 
 664         // Clear play list unless they specify append.
 665         if(!track.append()) {
 666                 current_track_list.clear();
 667         }
 668 
 669         if(!track.valid()) {
 670                 return;
 671         }
 672 
 673         auto iter = find_track(track);
 674         // Avoid 2 tracks with the same name, since that can cause an infinite loop
 675         // in choose_track(), 2 tracks with the same name will always return the
 676         // current track and track_ok() doesn't allow that.
 677         if(iter == current_track_list.end()) {
 678                 if(i < 0 || static_cast<size_t>(i) >= current_track_list.size()) {
 679                         current_track_list.emplace_back(new music_track(track));
 680                         iter = current_track_list.end() - 1;
 681                 } else {
 682                         iter = current_track_list.emplace(current_track_list.begin() + 1, new music_track(track));
 683                         if(current_track_index >= static_cast<size_t>(i)) {
 684                                 current_track_index++;
 685                         }
 686                 }
 687         } else {
 688                 ERR_AUDIO << "tried to add duplicate track '" << track.file_path() << "'" << std::endl;
 689         }
 690 
 691         // They can tell us to start playing this list immediately.
 692         if(track.immediate()) {
 693                 current_track = *iter;
 694                 current_track_index = iter - current_track_list.begin();
 695                 play_music();
 696         } else if(!track.append()) { // Make sure the current track is finished
 697                 current_track->set_play_once(true);
 698         }
 699 }

 701 void music_thinker::process(events::pump_info& info)
 702 {
 703         if(preferences::music_on()) {
 704                 if(!music_start_time && !current_track_list.empty() && !Mix_PlayingMusic()) {
 705                         // Pick next track, add ending time to its start time.
 706                         current_track = choose_track();
 707                         music_start_time = info.ticks();
 708                         no_fading = true;
 709                         fadingout_time = 0;
 710                 }
 711 
 712                 if(music_start_time && info.ticks(&music_refresh, music_refresh_rate) >= music_start_time - fadingout_time) {
 713                         want_new_music = true;
 714                 }
 715 
 716                 if(want_new_music) {
 717                         if(Mix_PlayingMusic()) {
 718                                 Mix_FadeOutMusic(fadingout_time);
 719                         }
 720 
 721                         unload_music = false;
 722                         play_new_music();
 723                 }
 724         }
 725 
 726         if(unload_music) {
 727                 // The custom shared_ptr deleter (Mix_FreeMusic) will handle the freeing of each track.
 728                 music_cache.clear();
 729 
 730                 Mix_HookMusicFinished(nullptr);
 731 
 732                 unload_music = false;
 733         }
 734 }

 755 void commit_music_changes()
 756 {
 757         played_before.clear();
 758 
 759         // Play-once is OK if still playing.
 760         if(current_track->play_once()) {
 761                 return;
 762         }
 763 
 764         // If current track no longer on playlist, change it.
 765         for(auto m : current_track_list) {
 766                 if(*current_track == *m) {
 767                         return;
 768                 }
 769         }
 770 
 771         // Victory empties playlist: if next scenario doesn't specify one...
 772         if(current_track_list.empty()) {
 773                 return;
 774         }
 775 
 776         // FIXME: we don't pause ms_before on this first track.  Should we?
 777         current_track = choose_track();
 778         play_music();
 779 }

 791 void reposition_sound(int id, unsigned int distance)
 792 {
 793         audio_lock lock;
 794         for(unsigned ch = 0; ch < channel_ids.size(); ++ch) {
 795                 if(channel_ids[ch] != id) {
 796                         continue;
 797                 }
 798 
 799                 if(distance >= DISTANCE_SILENT) {
 800                         // Don't call Mix_FadeOutChannel if the channel's volume is set to
 801                         // zero. It doesn't do anything in that case and the channel will
 802                         // resume playing as soon as its volume is reset to a non-zero
 803                         // value, which results in issues like sound sources deleted while
 804                         // their volume is zero coming back to life and escaping Wesnoth's
 805                         // sound source management code.
 806                         if(Mix_Volume(ch, -1) == 0) {
 807                                 Mix_HaltChannel(ch);
 808                         } else {
 809                                 Mix_FadeOutChannel(ch, 100);
 810                         }
 811                 } else {
 812                         Mix_SetDistance(ch, distance);
 813                 }
 814         }
 815 }

 828 struct chunk_load_exception
 829 {
 830 };
 831 
 832 static Mix_Chunk* load_chunk(const std::string& file, channel_group group)
 833 {
 834         sound_cache_iterator it_bgn, it_end;
 835         sound_cache_iterator it;
 836 
 837         sound_cache_chunk temp_chunk(file); // search the sound cache on this key
 838         it_bgn = sound_cache.begin(), it_end = sound_cache.end();
 839         it = std::find(it_bgn, it_end, temp_chunk);
 840 
 841         if(it != it_end) {
 842                 if(it->group != group) {
 843                         // cached item has been used in multiple sound groups
 844                         it->group = NULL_CHANNEL;
 845                 }
 846 
 847                 // splice the most recently used chunk to the front of the cache
 848                 sound_cache.splice(it_bgn, sound_cache, it);
 849         } else {
 850                 // remove the least recently used chunk from cache if it's full
 851                 bool cache_full = (sound_cache.size() == max_cached_chunks);
 852                 while(cache_full && it != it_bgn) {
 853                         // make sure this chunk is not being played before freeing it
 854                         std::vector<Mix_Chunk*>::iterator ch_end = channel_chunks.end();
 855                         if(std::find(channel_chunks.begin(), ch_end, (--it)->get_data()) == ch_end) {
 856                                 sound_cache.erase(it);
 857                                 cache_full = false;
 858                         }
 859                 }
 860 
 861                 if(cache_full) {
 862                         LOG_AUDIO << "Maximum sound cache size reached and all are busy, skipping.\n";
 863                         throw chunk_load_exception();
 864                 }
 865 
 866                 temp_chunk.group = group;
 867                 const std::string& filename = filesystem::get_binary_file_location("sounds", file);
 868 
 869                 if(!filename.empty()) {
 870                         SDL_RWops* rwops = filesystem::load_RWops(filename);
 871                         temp_chunk.set_data(Mix_LoadWAV_RW(rwops, true)); // SDL takes ownership of rwops
 872                 } else {
 873                         ERR_AUDIO << "Could not load sound file '" << file << "'." << std::endl;
 874                         throw chunk_load_exception();
 875                 }
 876 
 877                 if(temp_chunk.get_data() == nullptr) {
 878                         ERR_AUDIO << "Could not load sound file '" << filename << "': " << Mix_GetError() << "\n";
 879                         throw chunk_load_exception();
 880                 }
 881 
 882                 sound_cache.push_front(temp_chunk);
 883         }
 884 
 885         return sound_cache.begin()->get_data();
 886 }

 888 static void play_sound_internal(const std::string& files,
 889                 channel_group group,
 890                 unsigned int repeats = 0,
 891                 unsigned int distance = 0,
 892                 int id = -1,
 893                 int loop_ticks = 0,
 894                 int fadein_ticks = 0)
 895 {
 896         if(files.empty() || distance >= DISTANCE_SILENT || !mix_ok) {
 897                 return;
 898         }
 899 
 900         audio_lock lock;
 901 
 902         // find a free channel in the desired group
 903         int channel = Mix_GroupAvailable(group);
 904         if(channel == -1) {
 905                 LOG_AUDIO << "All channels dedicated to sound group(" << group << ") are busy, skipping.\n";
 906                 return;
 907         }
 908 
 909         Mix_Chunk* chunk;
 910         std::string file = pick_one(files);
 911 
 912         try {
 913                 chunk = load_chunk(file, group);
 914                 assert(chunk);
 915         } catch(const chunk_load_exception&) {
 916                 return;
 917         }
 918 
 919         /*
 920          * This check prevents SDL_Mixer from blowing up on Windows when UI sound is played
 921          * in response to toggling the checkbox which disables sound.
 922          */
 923         if(group != SOUND_UI) {
 924                 Mix_SetDistance(channel, distance);
 925         }
 926 
 927         int res;
 928         if(loop_ticks > 0) {
 929                 if(fadein_ticks > 0) {
 930                         res = Mix_FadeInChannelTimed(channel, chunk, -1, fadein_ticks, loop_ticks);
 931                 } else {
 932                         res = Mix_PlayChannel(channel, chunk, -1);
 933                 }
 934 
 935                 if(res >= 0) {
 936                         Mix_ExpireChannel(channel, loop_ticks);
 937                 }
 938         } else {
 939                 if(fadein_ticks > 0) {
 940                         res = Mix_FadeInChannel(channel, chunk, repeats, fadein_ticks);
 941                 } else {
 942                         res = Mix_PlayChannel(channel, chunk, repeats);
 943                 }
 944         }
 945 
 946         if(res < 0) {
 947                 ERR_AUDIO << "error playing sound effect: " << Mix_GetError() << std::endl;
 948                 // still keep it in the sound cache, in case we want to try again later
 949                 return;
 950         }
 951 
 952         channel_ids[channel] = id;
 953 
 954         // reserve the channel's chunk from being freed, since it is playing
 955         channel_chunks[res] = chunk;
 956 }
