  12 function retreat_functions.min_hp(unit)
  13     -- The minimum hp to retreat is a function of level and terrain defense
  14     -- We want to stay longer on good terrain and leave early on very bad terrain
  15     local hp_per_level = wesnoth.unit_defense(unit, wesnoth.get_terrain(unit.x, unit.y))/15
  16     local level = unit.level
  17 
  18     -- Leaders are considered to be higher level because of their value
  19     if unit.canrecruit then level = level+2 end
  20 
  21     local min_hp = hp_per_level*(level+2)
  22 
  23     -- Account for poison damage on next turn
  24     if unit.status.poisoned then min_hp = min_hp + 8 end
  25 
  26     -- Make sure that units are actually injured
  27     if min_hp > unit.max_hitpoints - 4 then
  28         min_hp = unit.max_hitpoints - 4
  29     end
  30 
  31     return min_hp
  32 end

  34 -- Given a set of units, return one from the set that should retreat and the location to retreat to
  35 -- Return nil if no unit needs to retreat
  36 function retreat_functions.retreat_injured_units(units)
  37     -- Split units into those that regenerate and those that do not
  38     local regen, non_regen = {}, {}
  39     for i,u in ipairs(units) do
  40         if u.hitpoints < retreat_functions.min_hp(u) then
  41             if wesnoth.unit_ability(u, 'regenerate') then
  42                 table.insert(regen, u)
  43             else
  44                 table.insert(non_regen, u)
  45             end
  46         end
  47     end
  48 
  49     -- First we retreat non-regenerating units to healing terrain, if they can get to a safe location
  50     local unit_nr, loc_nr, threat_nr
  51     if non_regen[1] then
  52         unit_nr, loc_nr, threat_nr = retreat_functions.get_retreat_injured_units(non_regen, false)
  53         if unit_nr and (threat_nr == 0) then
  54             return unit_nr, loc_nr, threat_nr
  55         end
  56     end
  57 
  58     -- Then we retreat regenerating units to terrain with high defense, if they can get to a safe location
  59     local unit_r, loc_r, threat_r
  60     if regen[1] then
  61         unit_r, loc_r, threat_r = retreat_functions.get_retreat_injured_units(regen, true)
  62         if unit_r and (threat_r == 0) then
  63             return unit_r, loc_r, threat_r
  64         end
  65     end
  66 
  67     -- The we retreat those that cannot get to a safe location (non-regenerating units first again)
  68     if unit_nr then
  69         return unit_nr, loc_nr, threat_nr
  70     end
  71     if unit_r then
  72         return unit_r, loc_r, threat_r
  73     end
  74 end

  76 function retreat_functions.get_healing_locations()
  77     local possible_healers = AH.get_live_units {
  78         { "filter_side", {{"allied_with", {side = wesnoth.current.side} }} }
  79     }
  80 
  81     local healing_locs = LS.create()
  82     for i,u in ipairs(possible_healers) do
  83         -- Only consider healers that cannot move this turn
  84         if u.moves == 0 or u.side ~= wesnoth.current.side then
  85             local heal_amount = 0
  86             local cure = 0
  87             local abilities = H.get_child(u.__cfg, "abilities") or {}
  88             for ability in H.child_range(abilities, "heals") do
  89                 heal_amount = ability.value
  90                 if ability.poison == "slowed" then
  91                     cure = 1
  92                 elseif ability.poison == "cured" then
  93                     cure = 2
  94                 end
  95             end
  96             if heal_amount + cure > 0 then
  97                 for x, y in H.adjacent_tiles(u.x, u.y) do
  98                     local old_values = healing_locs:get(x, y) or {0, 0}
  99                     local best_heal = math.max(old_values[0] or heal_amount)
 100                     local best_cure = math.max(old_values[1] or cure)
 101                     healing_locs:insert(u.x, u.y, {best_heal, best_cure})
 102                 end
 103             end
 104         end
 105     end
 106 
 107     return healing_locs
 108 end

 110 function retreat_functions.get_retreat_injured_units(healees, regenerates)
 111     -- Only retreat to safe locations
 112     local enemies = AH.get_attackable_enemies()
 113     local enemy_attack_map = BC.get_attack_map(enemies)
 114 
 115     local healing_locs = retreat_functions.get_healing_locations()
 116 
 117     local max_rating, best_loc, best_unit = -9e99, nil, nil
 118     for i,u in ipairs(healees) do
 119         local possible_locations = wesnoth.find_reach(u)
 120         -- TODO: avoid ally's villages (may be preferable to lower rating so they will
 121         -- be used if unit is very injured)
 122         if not regenerates then
 123             -- Unit cannot self heal, make the terrain do it for us if possible
 124             local location_subset = {}
 125             for j,loc in ipairs(possible_locations) do
 126                 local heal_amount = wesnoth.get_terrain_info(wesnoth.get_terrain(loc[1], loc[2])).healing or 0
 127                 if heal_amount == true then
 128                     -- handle deprecated syntax
 129                     -- TODO: remove this when removed from game
 130                     heal_amount = 8
 131                 end
 132                 local curing = 0
 133                 if heal_amount > 0 then
 134                     curing = 2
 135                 end
 136                 local healer_values = healing_locs:get(loc[1], loc[2]) or {0, 0}
 137                 heal_amount = math.max(heal_amount, healer_values[1])
 138                 curing = math.max(curing, healer_values[2])
 139                 table.insert(location_subset, {loc[1], loc[2], heal_amount, curing})
 140             end
 141 
 142             possible_locations = location_subset
 143         end
 144 
 145         local base_rating = - u.hitpoints + u.max_hitpoints / 2.
 146         if u.status.poisoned then base_rating = base_rating + 8 end
 147         if u.status.slowed then base_rating = base_rating + 4 end
 148         base_rating = base_rating * 1000
 149 
 150         for j,loc in ipairs(possible_locations) do
 151             local unit_in_way = wesnoth.get_unit(loc[1], loc[2])
 152             if (not AH.is_visible_unit(wesnoth.current.side, unit_in_way))
 153                 or ((unit_in_way.moves > 0) and (unit_in_way.side == wesnoth.current.side))
 154             then
 155                 local rating = base_rating
 156                 local heal_score = 0
 157                 if regenerates then
 158                     heal_score = math.min(8, u.max_hitpoints - u.hitpoints)
 159                 else
 160                     if u.status.poisoned then
 161                         if loc[4] > 0 then
 162                             heal_score = math.min(8, u.hitpoints - 1)
 163                             if loc[4] == 2 then
 164                                 -- This value is arbitrary, it just represents the ability to heal on the turn after
 165                                 heal_score = heal_score + 1
 166                             end
 167                         end
 168                     else
 169                         heal_score = math.min(loc[3], u.max_hitpoints - u.hitpoints)
 170                     end
 171                 end
 172 
 173                 -- Huge penalty for each enemy that can reach location,
 174                 -- this is the single most important point (and non-linear)
 175                 local enemy_count = enemy_attack_map.units:get(loc[1], loc[2]) or 0
 176                 rating = rating - enemy_count * 100000
 177 
 178                 -- Penalty based on terrain defense for unit
 179                 rating = rating - wesnoth.unit_defense(u, wesnoth.get_terrain(loc[1], loc[2]))/10
 180 
 181                 if (loc[1] == u.x) and (loc[2] == u.y) and (not u.status.poisoned) then
 182                     if enemy_count == 0 then
 183                         -- Bonus if we can rest heal
 184                         -- TODO: Always apply bonus if unit has healthy trait
 185                         heal_score = heal_score + 2
 186                     end
 187                 elseif unit_in_way then
 188                     -- Penalty if a unit has to move out of the way
 189                     -- (based on hp of moving unit)
 190                     rating = rating + unit_in_way.hitpoints - unit_in_way.max_hitpoints
 191                 end
 192 
 193                 rating = rating + heal_score^2
 194 
 195                 if (rating > max_rating) then
 196                     max_rating, best_loc, best_unit = rating, loc, u
 197                 end
 198             end
 199         end
 200     end
 201 
 202     return best_unit, best_loc, enemy_attack_map.units:get(best_loc[1], best_loc[2]) or 0
 203 end
