  39 use_target() {
  40         useq $1 && echo $2
  41 }
  42 
  43 use_variable() {
  44         echo ${2:-$1}=$(useq $1 && echo ${3:-yes} || echo ${4:-no})

  51 src_prepare() {
  52         if use server ; then
  53                 sed \
  54                         -e "s:GAMES_BINDIR:${GAMES_BINDIR}:" \
  55                         -e "s:GAMES_STATEDIR:${GAMES_STATEDIR}:" \
  56                         -e "s/GAMES_USER_DED/${GAMES_USER_DED}/" \
  57                         -e "s/GAMES_GROUP/${GAMES_GROUP}/" "${FILESDIR}"/wesnothd.rc \
  58                         > "${T}"/wesnothd \
  59                         || die "sed failed"
  60         fi
  61 }
  62 
  63 src_configure() {
  64         true
  65 }
  66 
  67 src_compile() {
  68         local myconf="build=release desktop_entry=yes"
  69 
  70         local SCONSOPTS=`echo ${MAKEOPTS} | \
  71                 sed s/-j$/-j256/\;s/-j[[:space:]]/-j256\ / |
  72                 sed s/--jobs$/--jobs=256/\;s/--jobs[[:space:]]/--jobs=256\ /`
  73 
  74         einfo "running scons with ${SCONSOPTS}"
  75 
  76         filter-flags -ftracer -fomit-frame-pointer
  77         if [[ $(gcc-major-version) -eq 3 ]] ; then
  78                 filter-flags -fstack-protector
  79                 append-flags -fno-stack-protector
  80         fi
  81         if use server ; then
  82                 myconf="${myconf} server_uid=${GAMES_USER_DED}"
  83                 myconf="${myconf} server_gid=${GAMES_GROUP}"
  84         fi
  85 
  86         if has ccache $FEATURES; then
  87                 myconf="${myconf} ccache=yes"
  88         else
  89                 myconf="${myconf} ccache=no"
  90         fi
  91 
  92         scons $myconf \
  93                 ${SCONSOPTS/-l[0-9]} \
  94                 $(use_target client wesnoth) \
  95                 $(use_target server wesnothd) \
  96                 $(use_target server campaignd) \
  97                 $(use_target tools cutter) \
  98                 $(use_target tools exploder) \
  99                 $(use_variable editor) \
 100                 $(use_variable tinygui gui tiny normal) \
 101                 $(use_variable lite lowmem) \
 102                 $(use_variable nls) \
 103                 localedirname=/usr/share/locale \
 104                 prefix=/usr/games \
 105                 prefsdir=.wesnoth-1.5 \
 106                 icondir=/usr/share/icons \
 107                 desktopdir=/usr/share/applications \
 108                 docdir=/usr/share/doc/${PF} \
 109                 default_targets=none ||
 110                 die "scons build failed"
 111 }
 112 
 113 src_install() {
 114         scons install destdir=${D} --implicit-deps-unchanged ||
 115         die "scons install failed"
 116         dodoc changelog
 117         if use server; then
 118                 keepdir "${GAMES_STATEDIR}/run/wesnothd"
 119                 doinitd "${T}"/wesnothd || die "doinitd failed"
 120         fi
 121         prepgamesdirs
 122 }
