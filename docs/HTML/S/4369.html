   7 -- This is a collection of Lua functions used for custom AI development.
   8 -- Note that this is still work in progress with significant changes occurring
   9 -- frequently. Backward compatibility cannot be guaranteed at this time in
  10 -- development releases, but it is of course easily possible to copy a function
  11 -- from a previous release directly into an add-on if it is needed there.
  12 --
  13 -- Invisible units ('viewing_side' parameter):
  14 -- With their default settings, the ai_helper functions use the vision a player of
  15 -- the respective side would see, that is, they assume no knowledge of invisible
  16 -- units. This can be influenced with the 'viewing_side' parameter, which works
  17 -- in the same way as it does in wesnoth.find_reach() and wesnoth.find_path():
  18 --   - If set to a valid side number, vision for that side is used
  19 --   - If set to an invalid side number (e.g. 0), all units on the map are seen
  20 --   - If omitted and a function takes a a parameter linked to a specific side,
  21 --     such as a side number or a unit, as input, vision of that side is used. In
  22 --     this case, viewing_side is passed as part of the optional @cfg configuration
  23 --     table and can be passed from function to function.
  24 --   - If omitted and the function takes no such input, viewing_side is made a
  25 --     required parameter in order to avoid mismatches between the default values
  26 --     of different functions.
  27 --
  28 -- Path finding:
  29 -- All ai_helper functions disregard shroud for path finding (while still ignoring
  30 -- hidden units correctly) as of Wesnoth 1.13.7. This is consistent with default
  31 -- Wesnoth AI behavior and ensures that Lua AIs, including the Micro AIs, can be
  32 -- used for AI sides with shroud=yes. It is accomplished by using
  33 -- ai_helper.find_path_with_shroud() instead of wesnoth.find_path().

  87     -- @cfg: table with optional configuration parameters:

  90     --   - keys: (array) if the value to be displayed is a subelement of the LS data,

  92     --     set keys = { 3 }, if it's data.arg[3], set keys = { 'arg', 3 }

  95     local factor = cfg.factor or 1
  96 
  97     ai_helper.clear_labels()
  98 
  99     map:iter(function(x, y, data)
 100         local out
 101         if cfg.show_coords then
 102             out = x .. ',' .. y
 103         else
 104             if cfg.keys then
 105                 for _,key in ipairs(cfg.keys) do data = data[key] end
 106             end
 107             if (type(data) == 'string') then
 108                 out = data
 109             else
 110                 out = tonumber(data) or 'nan'
 111             end
 112         end
 113 
 114         if (type(out) == 'number') then out = out * factor end
 115         W.label { x = x, y = y, text = out }
 116     end)

 157         -- E_FAILED_TELEPORT = 2006,

 276 function ai_helper.robust_move_and_attack(ai, src, dst, target_loc, cfg)
 277     -- Perform a move and/or attack with an AI unit in a way that is robust against
 278     -- unexpected outcomes such as being ambushed or changes caused by WML events.
 279     -- As much as possible, this function also tries to ensure that the gamestate
 280     -- is changed in case an action turns out to be impossible due to such an
 281     -- unexpected outcome.
 282     --
 283     -- Required input parameters:
 284     -- @ai: the Lua ai table
 285     -- @src: current coordinates of the AI unit to be used
 286     -- @dst: coordinates to which the unit should move. This does not have to be
 287     --   different from @src. In fact, the unit does not even need to have moves
 288     --   left, as long as an attack is specified in the latter case. If another
 289     --   AI unit is at @dst, it is moved out of the way.
 290     --
 291     -- Optional parameters:
 292     -- @target_loc: coordinates of the enemy unit to be attacked. If not given, no
 293     --   attack is attempted.
 294     -- @cfg: table with optional configuration parameters:
 295     --   partial_move: By default, this function performs full moves. If this
 296     --     parameter is true, a partial move is done instead.
 297     --   weapon: The number (starting at 1) of the attack weapon to be used.
 298     --     If omitted, the best weapon is automatically selected.
 299     --   all optional parameters for ai_helper.move_unit_out_of_way()
 300 
 301     -- Notes:
 302     -- - @src, @dst and @target_loc can be any table (including proxy units) that contains
 303     --   the coordinates of the respective locations using either indices .x/.y or [1]/[2].
 304     --   If both are given, .x/.y takes precedence over [1]/[2].
 305     -- - This function only safeguards AI moves against outcomes that the AI cannot know
 306     --   about, such as hidden units and WML events. It is assumed that the potential
 307     --   move was tested for general feasibility (units are on AI side and have moves
 308     --   left, terrain is passable, etc.) beforehand. If that is not done, it might
 309     --   lead to very undesirable behavior, incl. the CA being blacklisted or even the
 310     --   entire AI turn being ended.
 311 
 312     local src_x, src_y = src.x or src[1], src.y or src[2] -- this works with units or locations
 313     local dst_x, dst_y = dst.x or dst[1], dst.y or dst[2]
 314 
 315     local unit = wesnoth.get_unit(src_x, src_y)
 316     if (not unit) then
 317         return ai_helper.dummy_check_action(false, false, 'robust_move_and_attack::NO_UNIT')
 318     end
 319 
 320     -- Getting target at beginning also, in case events mess up things along the way
 321     local target, target_x, target_y
 322     if target_loc then
 323         target_x, target_y = target_loc.x or target_loc[1], target_loc.y or target_loc[2]
 324         target = wesnoth.get_unit(target_x, target_y)
 325 
 326         if (not target) then
 327             return ai_helper.dummy_check_action(false, false, 'robust_move_and_attack::NO_TARGET')
 328         end
 329     end
 330 
 331     local gamestate_changed = false
 332     local move_result
 333     if (unit.moves > 0) then
 334         if (src_x == dst_x) and (src_y == dst_y) then
 335             move_result = ai.stopunit_moves(unit)
 336 
 337             -- The only possible failure modes are non-recoverable (such as E_NOT_OWN_UNIT)
 338             if (not move_result.ok) then return move_result end
 339 
 340             if (not unit) or (not unit.valid) then
 341                 return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_DISAPPEARED')
 342             end
 343 
 344             gamestate_changed = true
 345         else
 346             local unit_old_moves = unit.moves
 347 
 348             local unit_in_way = wesnoth.get_unit(dst_x, dst_y)
 349             if unit_in_way and (unit_in_way.side == wesnoth.current.side) and (unit_in_way.moves > 0) then
 350                 local uiw_old_moves = unit_in_way.moves
 351                 ai_helper.move_unit_out_of_way(ai, unit_in_way, cfg)
 352 
 353                 if (not unit_in_way) or (not unit_in_way.valid) then
 354                     return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_IN_WAY_DISAPPEARED')
 355                 end
 356 
 357                 -- Failed move out of way: abandon remaining actions
 358                 if (unit_in_way.x == dst_x) and (unit_in_way.y == dst_y) then
 359                     if (unit_in_way.moves == uiw_old_moves) then
 360                         -- Forcing a gamestate change, if necessary
 361                         ai.stopunit_moves(unit_in_way)
 362                     end
 363                     return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_IN_WAY_EMPTY_MOVE')
 364                 end
 365 
 366                 -- Check whether dst hex is free now (an event could have done something funny)
 367                 local unit_in_way = wesnoth.get_unit(dst_x, dst_y)
 368                 if unit_in_way then
 369                     return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::ANOTHER_UNIT_IN_WAY')
 370                 end
 371 
 372                 gamestate_changed = true
 373             end
 374 
 375             if (not unit) or (not unit.valid) or (unit.x ~= src_x) or (unit.y ~= src_y) then
 376                 return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_DISAPPEARED')
 377             end
 378 
 379             local check_result = ai.check_move(unit, dst_x, dst_y)
 380             if (not check_result.ok) then
 381                 if (not ai_helper.is_incomplete_or_empty_move(check_result)) then
 382                     if (not gamestate_changed) then
 383                         ai.stopunit_moves(unit)
 384                     end
 385                     return check_result
 386                 end
 387             end
 388 
 389             if cfg and cfg.partial_move then
 390                 move_result = ai.move(unit, dst_x, dst_y)
 391             else
 392                 move_result = ai.move_full(unit, dst_x, dst_y)
 393             end
 394             if (not move_result.ok) then return move_result end
 395 
 396             if (not unit) or (not unit.valid) then
 397                 return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_DISAPPEARED')
 398             end
 399 
 400             -- Failed move: abandon rest of actions
 401             if (unit.x == src_x) and (unit.y == src_y) then
 402                 if (not gamestate_changed) and (unit.moves == unit_old_moves) then
 403                     -- Forcing a gamestate change, if necessary
 404                     ai.stopunit_moves(unit)
 405                 end
 406                 return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNPLANNED_EMPTY_MOVE')
 407             end
 408 
 409             gamestate_changed = true
 410         end
 411     end
 412 
 413     -- Tests after the move, before continuing to attack, to ensure WML events
 414     -- did not do something funny
 415     if (not unit) or (not unit.valid) then
 416         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_DISAPPEARED')
 417     end
 418     if (unit.x ~= dst_x) or (unit.y ~= dst_y) then
 419         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_NOT_AT_DESTINATION')
 420     end
 421 
 422     -- In case all went well and there's no attack to be done
 423     if (not target_x) then return move_result end
 424 
 425     if (not target) or (not target.valid) then
 426         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::TARGET_DISAPPEARED')
 427     end
 428     if (target.x ~= target_x) or (target.y ~= target_y) then
 429         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::TARGET_MOVED')
 430     end
 431 
 432     local weapon = cfg and cfg.weapon
 433     local old_attacks_left = unit.attacks_left
 434 
 435     local check_result = ai.check_attack(unit, target, weapon)
 436     if (not check_result.ok) then
 437         if (not gamestate_changed) then
 438             ai.stopunit_all(unit)
 439         end
 440         return check_result
 441     end
 442 
 443     move_result = ai.attack(unit, target, weapon)
 444     -- This should not happen, given that we just checked, but just in case
 445     if (not move_result.ok) then return move_result end
 446 
 447     if (not unit) or (not unit.valid) then
 448         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::UNIT_DISAPPEARED')
 449     end
 450 
 451     if (unit.attacks_left == old_attacks_left) and (not gamestate_changed) then
 452         ai.stopunit_all(unit)
 453         return ai_helper.dummy_check_action(true, false, 'robust_move_and_attack::NO_ATTACK')
 454     end
 455 
 456     return move_result
 457 end

 485 function ai_helper.serialize(input)
 486     -- Convert @input to a string in a format corresponding to the type of @input
 487     -- The string is all put into one line
 488 
 489     local str = ''
 490     if (type(input) == "number") or (type(input) == "boolean") then
 491         str = tostring(input)
 492     elseif type(input) == "string" then
 493         str = string.format("%q", input)
 494     elseif type(input) == "table" then
 495         str = str .. "{ "
 496         for k,v in pairs(input) do
 497             str = str .. "[" .. ai_helper.serialize(k)  .. "] = "
 498             str = str .. ai_helper.serialize(v)
 499             str = str .. ", "
 500         end
 501         str = str .. "}"
 502     else
 503         error("cannot serialize a " .. type(input), 2)
 504     end
 505 
 506     return str
 507 end

 509 function ai_helper.split(str, sep)
 510     -- Split string @str into a table using the delimiter @sep (default: ',')
 511 
 512     local sep, fields = sep or ",", {}
 513     local pattern = string.format("([^%s]+)", sep)

 540 function ai_helper.get_angle(from_hex, to_hex)
 541     -- Returns the angle of the direction from @from_hex to @to_hex
 542     -- Angle is in radians and goes from -pi to pi. 0 is toward east.
 543     -- Input hex tables can be of form { x, y } or { x = x, y = y }, which
 544     -- means that it is also possible to pass a unit table
 545     local x1, y1 = from_hex.x or from_hex[1], from_hex.y or from_hex[2]
 546     local x2, y2 = to_hex.x or to_hex[1], to_hex.y or to_hex[2]
 547 
 548     local _, y1cart =  ai_helper.cartesian_coords(x1, y1)
 549     local _, y2cart =  ai_helper.cartesian_coords(x2, y2)
 550 
 551     return math.atan(y2cart - y1cart, x2 - x1)
 552 end

 554 function ai_helper.get_direction_index(from_hex, to_hex, n, center_on_east)
 555     -- Returns an integer index for the direction from @from_hex to @to_hex
 556     -- with the full circle divided into @n slices
 557     -- 1 is always to the east, with indices increasing clockwise
 558     -- Input hex tables can be of form { x, y } or { x = x, y = y }, which
 559     -- means that it is also possible to pass a unit table
 560     --
 561     -- Optional input:
 562     -- @center_on_east (false): boolean. By default, the eastern direction is the
 563     -- northern border of the first slice. If this parameter is set, east will
 564     -- instead be the center direction of the first slice
 565 
 566     local d_east = 0
 567     if center_on_east then d_east = 0.5 end
 568 
 569     local angle = ai_helper.get_angle(from_hex, to_hex)
 570     local index = math.floor((angle / math.pi * n/2 + d_east) % n ) + 1
 571 
 572     return index
 573 end

 590 function ai_helper.find_opposite_hex_adjacent(hex, center_hex)
 591     -- Find the hex that is opposite of @hex with respect to @center_hex
 592     -- Both input hexes are of format { x, y }
 593     -- Output: {opp_x, opp_y} -- or nil if @hex and @center_hex are not adjacent
 594     --   (or no opposite hex is found, e.g. for hexes on border)
 595 
 596     -- If the two input hexes are not adjacent, return nil
 597     if (M.distance_between(hex[1], hex[2], center_hex[1], center_hex[2]) ~= 1) then return nil end
 598 
 599     -- Finding the opposite x position is easy
 600     local opp_x = center_hex[1] + (center_hex[1] - hex[1])
 601 
 602     -- y is slightly more tricky, because of the hexagonal shape, but there's a trick
 603     -- that saves us from having to build in a lot of if statements
 604     -- Among the adjacent hexes, it is the one with the correct x, and y _different_ from hex[2]
 605     for xa,ya in H.adjacent_tiles(center_hex[1], center_hex[2]) do
 606         if (xa == opp_x) and (ya ~= hex[2]) then return { xa, ya } end
 607     end
 608 
 609     return nil
 610 end

 612 function ai_helper.find_opposite_hex(hex, center_hex)
 613     -- Find the hex that is opposite of @hex with respect to @center_hex
 614     -- Using "square coordinate" method by JaMiT
 615     -- Note: this also works for non-adjacent hexes, but might return hexes that are not on the map!
 616     -- Both input hexes are of format { x, y }
 617     -- Output: { opp_x, opp_y }
 618 
 619     -- Finding the opposite x position is easy
 620     local opp_x = center_hex[1] + (center_hex[1] - hex[1])
 621 
 622     -- Going to "square geometry" for y coordinate
 623     local y_sq = hex[2] * 2 - (hex[1] % 2)
 624     local yc_sq = center_hex[2] * 2 - (center_hex[1] % 2)
 625 
 626     -- Now the same equation as for x can be used for y
 627     local opp_y = yc_sq + (yc_sq - y_sq)
 628     opp_y = math.floor((opp_y + 1) / 2)
 629 
 630     return {opp_x, opp_y}
 631 end

 642 function ai_helper.get_closest_location(hex, location_filter, unit)
 643     -- Get the location closest to @hex (in format { x, y })
 644     -- that matches @location_filter (in WML table format)
 645     -- @unit can be passed as an optional third parameter, in which case the
 646     -- terrain needs to be passable for that unit
 647     -- Returns nil if no terrain matching the filter was found
 648 
 649     -- Find the maximum distance from 'hex' that's possible on the map
 650     local max_distance = 0
 651     local width, height = wesnoth.get_map_size()
 652     local to_top_left = M.distance_between(hex[1], hex[2], 0, 0)
 653     if (to_top_left > max_distance) then max_distance = to_top_left end
 654     local to_top_right = M.distance_between(hex[1], hex[2], width+1, 0)
 655     if (to_top_right > max_distance) then max_distance = to_top_right end
 656     local to_bottom_left = M.distance_between(hex[1], hex[2], 0, height+1)
 657     if (to_bottom_left > max_distance) then max_distance = to_bottom_left end
 658     local to_bottom_right = M.distance_between(hex[1], hex[2], width+1, height+1)
 659     if (to_bottom_right > max_distance) then max_distance = to_bottom_right end
 660 
 661     local radius = 0
 662     while (radius <= max_distance) do
 663         local loc_filter = {}
 664         if (radius == 0) then
 665             loc_filter = {
 666                 { "and", { x = hex[1], y = hex[2], radius = radius } },
 667                 { "and", location_filter }
 668             }
 669         else
 670             loc_filter = {
 671                 { "and", { x = hex[1], y = hex[2], radius = radius } },
 672                 { "not", { x = hex[1], y = hex[2], radius = radius - 1 } },
 673                 { "and", location_filter }
 674             }
 675         end
 676 
 677         local locs = wesnoth.get_locations(loc_filter)
 678 
 679         if unit then
 680             for _,loc in ipairs(locs) do
 681                 local movecost = wesnoth.unit_movement_cost(unit, wesnoth.get_terrain(loc[1], loc[2]))
 682                 if (movecost <= unit.max_moves) then return loc end
 683             end
 684         else
 685             if locs[1] then return locs[1] end
 686         end
 687 
 688         radius = radius + 1
 689     end
 690 
 691     return nil
 692 end

 694 function ai_helper.get_passable_locations(location_filter, unit)
 695     -- Finds all locations matching @location_filter that are passable for
 696     -- @unit. This also excludes hexes on the map border.
 697     -- @unit is optional: if omitted, all hexes matching the filter, but
 698     -- excluding border hexes are returned
 699 
 700     -- All hexes that are not on the map border
 701     local width, height = wesnoth.get_map_size()
 702     local all_locs = wesnoth.get_locations{
 703         x = '1-' .. width,
 704         y = '1-' .. height,
 705         { "and", location_filter }
 706     }
 707 
 708     -- If @unit is provided, exclude terrain that's impassable for the unit
 709     if unit then
 710         local locs = {}
 711         for _,loc in ipairs(all_locs) do
 712             local movecost = wesnoth.unit_movement_cost(unit, wesnoth.get_terrain(loc[1], loc[2]))
 713             if (movecost <= unit.max_moves) then table.insert(locs, loc) end
 714         end
 715         return locs
 716     end
 717 
 718     return all_locs

 721 function ai_helper.distance_map(units, map)
 722     -- Get the distance map DM for all units in @units (as a location set)
 723     -- DM = sum ( distance_from_unit )
 724     -- This is done for all elements of @map (a locations set), or for the entire map if @map is not given
 725 
 726     local DM = LS.create()
 727 
 728     if map then
 729         map:iter(function(x, y, data)
 730             local dist = 0
 731             for _,unit in ipairs(units) do
 732                 dist = dist + M.distance_between(unit.x, unit.y, x, y)
 733             end
 734             DM:insert(x, y, dist)
 735         end)
 736     else
 737         local width, height = wesnoth.get_map_size()
 738         for x = 1,width do
 739             for y = 1,height do
 740                 local dist = 0
 741                 for _,unit in ipairs(units) do
 742                     dist = dist + M.distance_between(unit.x, unit.y, x, y)
 743                 end
 744                 DM:insert(x, y, dist)
 745             end
 746         end
 747     end
 748 
 749     return DM
 750 end

 797 function ai_helper.xyoff(x, y, ori, hex)
 798     -- Finds hexes at a certain offset from @x,@y
 799     -- @ori: direction/orientation: north (0), ne (1), se (2), s (3), sw (4), nw (5)
 800     -- @hex: string for the hex to be queried. Possible values:
 801     --   's': self, 'u': up, 'lu': left up, 'ld': left down, 'ru': right up, 'rd': right down
 802     --   This is all relative "looking" in the direction of 'ori'
 803     -- returns x,y for the queried hex
 804 
 805     -- Unlike Lua default, we count 'ori' from 0 (north) to 5 (nw), so that modulo operator can be used
 806     ori = ori % 6
 807 
 808     if (hex == 's') then return x, y end
 809 
 810     -- This is all done with ifs, to keep it as fast as possible
 811     if (ori == 0)  then -- "north"
 812         if (hex == 'u') then return x, y-1 end
 813         if (hex == 'd') then return x, y+1 end
 814         local dy = 0
 815         if (x % 2) == 1 then dy=1 end
 816         if (hex == 'lu') then return x-1, y-dy end
 817         if (hex == 'ld') then return x-1, y+1-dy end
 818         if (hex == 'ru') then return x+1, y-dy end
 819         if (hex == 'rd') then return x+1, y+1-dy end
 820     end
 821 
 822     if (ori == 1)  then -- "north-east"
 823         local dy = 0
 824         if (x % 2) == 1 then dy=1 end
 825         if (hex == 'u') then return x+1, y-dy end
 826         if (hex == 'd') then return x-1, y+1-dy end
 827         if (hex == 'lu') then return x, y-1 end
 828         if (hex == 'ld') then return x-1, y-dy end
 829         if (hex == 'ru') then return x+1, y+1-dy end
 830         if (hex == 'rd') then return x, y+1 end
 831     end
 832 
 833     if (ori == 2)  then -- "south-east"
 834         local dy = 0
 835         if (x % 2) == 1 then dy=1 end
 836         if (hex == 'u') then return x+1, y+1-dy end
 837         if (hex == 'd') then return x-1, y-dy end
 838         if (hex == 'lu') then return x+1, y-dy end
 839         if (hex == 'ld') then return x, y-1 end
 840         if (hex == 'ru') then return x, y+1 end
 841         if (hex == 'rd') then return x-1, y+1-dy end
 842     end
 843 
 844     if (ori == 3)  then -- "south"
 845         if (hex == 'u') then return x, y+1 end
 846         if (hex == 'd') then return x, y-1 end
 847         local dy = 0
 848         if (x % 2) == 1 then dy=1 end
 849         if (hex == 'lu') then return x+1, y+1-dy end
 850         if (hex == 'ld') then return x+1, y-dy end
 851         if (hex == 'ru') then return x-1, y+1-dy end
 852         if (hex == 'rd') then return x-1, y-dy end
 853     end
 854 
 855     if (ori == 4)  then -- "south-west"
 856         local dy = 0
 857         if (x % 2) == 1 then dy=1 end
 858         if (hex == 'u') then return x-1, y+1-dy end
 859         if (hex == 'd') then return x+1, y-dy end
 860         if (hex == 'lu') then return x, y+1 end
 861         if (hex == 'ld') then return x+1, y+1-dy end
 862         if (hex == 'ru') then return x-1, y-dy end
 863         if (hex == 'rd') then return x, y-1 end
 864     end
 865 
 866     if (ori == 5)  then -- "north-west"
 867         local dy = 0
 868         if (x % 2) == 1 then dy=1 end
 869         if (hex == 'u') then return x-1, y-dy end
 870         if (hex == 'd') then return x+1, y+1-dy end
 871         if (hex == 'lu') then return x-1, y+1-dy end
 872         if (hex == 'ld') then return x, y+1 end
 873         if (hex == 'ru') then return x, y-1 end
 874         if (hex == 'rd') then return x+1, y-dy end
 875     end
 876 
 877     return
 878 end

1231 function ai_helper.next_hop(unit, x, y, cfg)
1232     -- Finds the next "hop" of @unit on its way to (@x,@y)
1233     -- Returns coordinates of the endpoint of the hop (or nil if no path to
1234     -- (x,y) is found for the unit), and movement cost to get there.
1235     -- Only unoccupied hexes are considered
1236     -- @cfg: standard extra options for wesnoth.find_path()
1237     --   including:
1238     --     viewing_side: see comments at beginning of this file. Defaults to side of @unit
1239     --   plus:
1240     --     ignore_own_units: if set to true, then own units that can move out of the way are ignored
1241 
1242     local path, cost = ai_helper.find_path_with_shroud(unit, x, y, cfg)
1243 
1244     if cost >= ai_helper.no_path then return nil, cost end
1245 
1246     -- If none of the hexes are unoccupied, use current position as default
1247     local next_hop, nh_cost = { unit.x, unit.y }, 0
1248 
1249     -- Go through loop to find reachable, unoccupied hex along the path
1250     -- Start at second index, as first is just the unit position itself
1251     for i = 2,#path do
1252         local sub_path, sub_cost = ai_helper.find_path_with_shroud(unit, path[i][1], path[i][2], cfg)
1253 
1254         if sub_cost <= unit.moves then
1255             -- Check for unit in way only if cfg.ignore_units is not set
1256             local unit_in_way
1257             if (not cfg) or (not cfg.ignore_units) then
1258                 unit_in_way = wesnoth.get_unit(path[i][1], path[i][2])
1259 
1260                 -- If ignore_own_units is set, ignore own side units that can move out of the way
1261                 if cfg and cfg.ignore_own_units then
1262                     if unit_in_way and (unit_in_way.side == unit.side) then
1263                         local reach = ai_helper.get_reachable_unocc(unit_in_way, cfg)
1264                         if (reach:size() > 1) then unit_in_way = nil end
1265                     end
1266                 end
1267             end
1268 
1269             if not unit_in_way then
1270                 next_hop, nh_cost = path[i], sub_cost
1271             end
1272         else
1273             break
1274         end
1275     end
1276 
1277     return next_hop, nh_cost
1278 end

1280 function ai_helper.can_reach(unit, x, y, cfg)
1281     -- Returns true if hex (@x, @y) is unoccupied (by a visible unit), or
1282     -- at most occupied by unit on same side as @unit that can move away
1283     -- (can be modified with options below)
1284     --
1285     -- @cfg: table with optional configuration parameters:
1286     --   moves = 'max' use max_moves instead of current moves
1287     --   ignore_units: if true, ignore both own and enemy units
1288     --   exclude_occupied: if true, exclude hex if there's a unit there, irrespective of value of 'ignore_units'
1289     --   viewing_side: see comments at beginning of this file. Defaults to side of @unit
1290 
1291     cfg = cfg or {}
1292     local viewing_side = cfg.viewing_side or unit.side
1293 
1294     -- Is there a unit at the goal hex?
1295     local unit_in_way = wesnoth.get_unit(x, y)
1296     if (cfg.exclude_occupied)
1297       and unit_in_way and ai_helper.is_visible_unit(viewing_side, unit_in_way)
1298     then
1299         return false
1300     end
1301 
1302     -- Otherwise, if 'ignore_units' is not set, return false if there's a unit of other side,
1303     -- or a unit of own side that cannot move away (this might be slow, don't know)
1304     if (not cfg.ignore_units) then
1305         -- If there's a unit at the goal that's not on own side (even ally), return false
1306         if unit_in_way
1307             and (unit_in_way.side ~= unit.side)
1308             and ai_helper.is_visible_unit(viewing_side, unit_in_way)
1309         then
1310             return false
1311         end
1312 
1313         -- If the unit in the way is on 'unit's' side and cannot move away, also return false
1314         if unit_in_way and (unit_in_way.side == unit.side) then
1315             -- need to pass the cfg here so that it works for enemy units (generally with no moves left) also
1316             local move_away = ai_helper.get_reachable_unocc(unit_in_way, cfg)
1317             if (move_away:size() <= 1) then return false end
1318         end
1319     end
1320 
1321     -- After all that, test whether our unit can actually get there
1322     local old_moves = unit.moves
1323     if (cfg.moves == 'max') then unit.moves = unit.max_moves end
1324 
1325     local can_reach = false
1326     local path, cost = ai_helper.find_path_with_shroud(unit, x, y, cfg)
1327     if (cost <= unit.moves) then can_reach = true end
1328 
1329     unit.moves = old_moves
1330 
1331     return can_reach
1332 end

1368 function ai_helper.find_path_with_shroud(unit, x, y, cfg)
1369     -- Same as wesnoth.find_path, just that it works under shroud as well while still
1370     -- ignoring invisible units. It does this by using viewing_side=0 and taking
1371     -- invisible units off the map for the path finding process.
1372     --
1373     -- Notes on some of the optional parameters that can be passed in @cfg:
1374     --  - viewing_side: If not given, use side of the unit (not the current side!)
1375     --    for determining which units are hidden and need to be extracted, as that
1376     --    is what the path_finder code uses. If set to an invalid side, we can use
1377     --    default path finding as shroud is ignored then anyway.
1378     --  - ignore_units: if true, hidden units do not need to be extracted because
1379     --    all units are ignored anyway
1380 
1381     local viewing_side = (cfg and cfg.viewing_side) or unit.side
1382 
1383     local path, cost
1384     if wesnoth.sides[viewing_side] and wesnoth.sides[viewing_side].shroud then
1385         local extracted_units = {}
1386         if (not cfg) or (not cfg.ignore_units) then
1387             local all_units = wesnoth.get_units()
1388             for _,u in ipairs(all_units) do
1389                 if (u.side ~= viewing_side)
1390                     and (not ai_helper.is_visible_unit(viewing_side, u))
1391                 then
1392                     wesnoth.extract_unit(u)
1393                     table.insert(extracted_units, u)
1394                 end
1395             end
1396         end
1397 
1398         local cfg_copy = {}
1399         if cfg then cfg_copy = ai_helper.table_copy(cfg) end
1400         cfg_copy.viewing_side = 0
1401         path, cost = wesnoth.find_path(unit, x, y, cfg_copy)
1402 
1403         for _,extracted_unit in ipairs(extracted_units) do
1404             wesnoth.put_unit(extracted_unit)
1405         end
1406     else
1407         path, cost = wesnoth.find_path(unit, x, y, cfg)
1408     end
1409 
1410     return path, cost
1411 end

1413 function ai_helper.find_best_move(units, rating_function, cfg)

1419     --    labels: if set, put labels with ratings onto map
1420     --    no_random: if set, do not add random value between 0.0001 and 0.0099 to each hex

1422     --
1423     -- OUTPUTS:
1424     --  best_hex: format { x, y }
1425     --  best_unit: unit for which this rating function produced the maximum value

1427     -- If no valid moves were found, best_unit and best_hex are empty arrays

1459 function ai_helper.move_unit_out_of_way(ai, unit, cfg)
1460     -- Move @unit to the best close location.
1461     -- Main rating is the moves the unit still has left after that
1462     -- Other, configurable, parameters are given to function in @cfg:
1463     --   dx, dy: the direction in which moving out of the way is preferred
1464     --   labels: if set, display labels of the rating for each hex the unit can reach
1465     --   viewing_side: see comments at beginning of this file. Defaults to side of @unit.
1466     --   all other optional parameters to wesnoth.find_reach()
1467 
1468     cfg = cfg or {}
1469     local viewing_side = cfg.viewing_side or unit.side
1470 
1471     local dx, dy
1472     if cfg.dx and cfg.dy then
1473         local r = math.sqrt(cfg.dx * cfg.dx + cfg.dy * cfg.dy)
1474         dx, dy = cfg.dx / r, cfg.dy / r
1475     end
1476 
1477     local reach = wesnoth.find_reach(unit, cfg)
1478     local reach_map = LS.create()
1479 
1480     local max_rating, best_hex = -9e99
1481     for _,loc in ipairs(reach) do
1482         local unit_in_way = wesnoth.get_unit(loc[1], loc[2])
1483         if (not unit_in_way)       -- also excludes current hex
1484             or (not ai_helper.is_visible_unit(viewing_side, unit_in_way))
1485         then
1486             local rating = loc[3]  -- also disfavors hexes next to visible enemy units for which loc[3] = 0
1487 
1488             if dx then
1489                 rating = rating + (loc[1] - unit.x) * dx * 0.01
1490                 rating = rating + (loc[2] - unit.y) * dy * 0.01
1491             end
1492 
1493             if cfg.labels then reach_map:insert(loc[1], loc[2], rating) end
1494 
1495             if (rating > max_rating) then
1496                 max_rating, best_hex = rating, { loc[1], loc[2] }
1497             end
1498         end
1499     end
1500     if cfg.labels then ai_helper.put_labels(reach_map) end
1501 
1502     if best_hex then
1503         ai_helper.checked_move(ai, unit, best_hex[1], best_hex[2])
1504     end
1505 end

1566 function ai_helper.get_attacks(units, cfg)

1568     -- excluded unless option @cfg.viewing_side=0 is used.
1569     --
1570     -- This includes a variety of configurable options, passed in the @cfg table
1571     -- @cfg: table with optional configuration parameters:
1572     --   moves: "current" (default for units on current side) or "max" (always used for units on other sides)
1573     --   include_occupied (false): if set, also include hexes occupied by own-side units that can move away
1574     --   simulate_combat (false): if set, also simulate the combat and return result (this is slow; only set if needed)
1575     --   viewing_side: see comments at beginning of this file. Defaults to side of @units
1576     --   all other optional parameters to wesnoth.find_reach()
1577     --
1578     -- Returns {} if no attacks can be done, otherwise table with fields:
1579     --   dst: { x = x, y = y } of attack position
1580     --   src: { x = x, y = y } of attacking unit (don't use id, could be ambiguous)
1581     --   target: { x = x, y = y } of defending unit
1582     --   att_stats, def_stats: as returned by wesnoth.simulate_combat (if cfg.simulate_combat is set)
1583     --   attack_hex_occupied: boolean storing whether an own unit that can move away is on the attack hex
1584 
1585     cfg = cfg or {}
1586 
1587     local attacks = {}
1588     if (not units[1]) then return attacks end
1589 
1590     local side = units[1].side  -- all units need to be on same side
1591     local viewing_side = cfg and cfg.viewing_side or side
1592 
1593     -- 'moves' can be either "current" or "max"
1594     -- For unit on current side: use "current" by default, or override by cfg.moves
1595     local moves = cfg.moves or "current"
1596     -- For unit on any other side, only moves="max" makes sense
1597     if (side ~= wesnoth.current.side) then moves = "max" end
1598 
1599     local old_moves = {}
1600     if (moves == "max") then
1601         for i,unit in ipairs(units) do
1602             old_moves[i] = unit.moves
1603             unit.moves = unit.max_moves
1604         end
1605     end
1606 
1607     -- Note: the remainder is optimized for speed, so we only get_units once,
1608     -- do not use WML filters, etc.
1609     local all_units = wesnoth.get_units()
1610 
1611     local enemy_map, my_unit_map, other_unit_map = LS.create(), LS.create(), LS.create()
1612     for i,unit in ipairs(all_units) do
1613         -- The value of all the location sets is the index of the
1614         -- unit in the all_units array
1615         if ai_helper.is_attackable_enemy(unit, side, cfg) then
1616             enemy_map:insert(unit.x, unit.y, i)
1617         end
1618 
1619         if (unit.side == side) then
1620             my_unit_map:insert(unit.x, unit.y, i)
1621         else
1622             if ai_helper.is_visible_unit(viewing_side, unit) then
1623                 other_unit_map:insert(unit.x, unit.y, i)
1624             end
1625         end
1626     end
1627 
1628     local attack_hex_map = LS.create()
1629     enemy_map:iter(function(e_x, e_y, i)
1630         for xa,ya in H.adjacent_tiles(e_x, e_y) do
1631             -- If there's no unit of another side on this hex, include it
1632             -- as possible attack location (this includes hexes occupied
1633             -- by own units at this time)
1634             if (not other_unit_map:get(xa, ya)) then
1635                 local target_table = attack_hex_map:get(xa, ya) or {}
1636                 table.insert(target_table, { x = e_x, y = e_y, i = i })
1637                 attack_hex_map:insert(xa, ya, target_table)
1638             end
1639         end
1640     end)
1641 
1642     -- The following is done so that we at most need to do find_reach() once per unit
1643     -- It is needed for all units in @units and for testing whether units can move out of the way
1644     local reaches = LS.create()
1645 
1646     for _,unit in ipairs(units) do
1647         local reach
1648         if reaches:get(unit.x, unit.y) then
1649             reach = reaches:get(unit.x, unit.y)
1650         else
1651             reach = wesnoth.find_reach(unit, cfg)
1652             reaches:insert(unit.x, unit.y, reach)
1653         end
1654 
1655         for _,loc in ipairs(reach) do
1656             if attack_hex_map:get(loc[1], loc[2]) then
1657                 local add_target = true
1658                 local attack_hex_occupied = false
1659 
1660                 -- If another unit of same side is on this hex:
1661                 if my_unit_map:get(loc[1], loc[2]) and ((loc[1] ~= unit.x) or (loc[2] ~= unit.y)) then
1662                     attack_hex_occupied = true
1663                     add_target = false
1664 
1665                     if cfg.include_occupied then -- Test whether it can move out of the way
1666                         local unit_in_way = all_units[my_unit_map:get(loc[1], loc[2])]
1667                         local uiw_reach
1668                         if reaches:get(unit_in_way.x, unit_in_way.y) then
1669                             uiw_reach = reaches:get(unit_in_way.x, unit_in_way.y)
1670                         else
1671                             uiw_reach = wesnoth.find_reach(unit_in_way, cfg)
1672                             reaches:insert(unit_in_way.x, unit_in_way.y, uiw_reach)
1673                         end
1674 
1675                         -- Check whether the unit to move out of the way has an unoccupied hex to move to.
1676                         -- We do not deal with cases where a unit can move out of the way for a
1677                         -- unit that is moving out of the way of the initial unit (etc.).
1678                         for _,uiw_loc in ipairs(uiw_reach) do
1679                             -- Unit in the way of the unit in the way
1680                             local uiw_uiw = wesnoth.get_unit(uiw_loc[1], uiw_loc[2])
1681                             if (not uiw_uiw) or (not ai_helper.is_visible_unit(viewing_side, uiw_uiw)) then
1682                                 add_target = true
1683                                 break
1684                             end
1685                         end
1686                     end
1687                 end
1688 
1689                 if add_target then
1690                     for _,target in ipairs(attack_hex_map:get(loc[1], loc[2])) do
1691                         local att_stats, def_stats
1692                         if cfg.simulate_combat then
1693                             local unit_dst = wesnoth.copy_unit(unit)
1694                             unit_dst.x, unit_dst.y = loc[1], loc[2]
1695 
1696                             local enemy = all_units[target.i]
1697                             att_stats, def_stats = wesnoth.simulate_combat(unit_dst, enemy)
1698                         end
1699 
1700                         table.insert(attacks, {
1701                             src = { x = unit.x, y = unit.y },
1702                             dst = { x = loc[1], y = loc[2] },
1703                             target = { x = target.x, y = target.y },
1704                             att_stats = att_stats,
1705                             def_stats = def_stats,
1706                             attack_hex_occupied = attack_hex_occupied
1707                         })
1708                     end
1709                 end
1710             end
1711         end
1712     end
1713 
1714     if (moves == "max") then
1715         for i,unit in ipairs(units) do
1716             unit.moves = old_moves[i]
1717         end
1718     end
1719 
1720     return attacks
1721 end

1723 function ai_helper.add_next_attack_combo_level(combos, attacks)
1724     -- This is called from ai_helper.get_attack_combos_full() and
1725     -- builds up the combos for the next recursion level.
1726     -- It also calls the next recursion level, if possible
1727     -- Important: function needs to make a copy of the input array, otherwise original is changed
1728 
1729     -- Set up the array, if this is the first recursion level
1730     if (not combos) then combos = {} end
1731 
1732     -- Array to hold combinations for this recursion level only
1733     local combos_this_level = {}
1734 
1735     for _,attack in ipairs(attacks) do
1736         local dst = attack.dst.y + attack.dst.x * 1000.  -- attack hex (src)
1737         local src = attack.src.y + attack.src.x * 1000.  -- attacker hex (dst)
1738         if (not combos[1]) then  -- if this is the first recursion level, set up new combos for this level
1739             local move = {}
1740             move[dst] = src
1741             table.insert(combos_this_level, move)
1742         else
1743             -- Otherwise, we need to go through the already existing elements in 'combos'
1744             -- to see if either hex, or attacker is already used; and then add new attack to each
1745             for _,combo in ipairs(combos) do
1746                 local this_combo = {}  -- needed because tables are pointers, need to create a separate one
1747                 local add_combo = true
1748                 for d,s in pairs(combo) do
1749                     if (d == dst) or (s == src) then
1750                         add_combo = false
1751                         break
1752                     end
1753                     this_combo[d] = s  -- insert individual moves to a combo
1754                 end
1755                 if add_combo then  -- and add it into the array, if it contains only unique moves
1756                     this_combo[dst] = src
1757                     table.insert(combos_this_level, this_combo)
1758                 end
1759             end
1760         end
1761     end
1762 
1763     local combos_next_level = {}
1764     if combos_this_level[1] then  -- If moves were found for this level, also find those for the next level
1765         combos_next_level = ai_helper.add_next_attack_combo_level(combos_this_level, attacks)
1766     end
1767 
1768     -- Finally, combine this level and next level combos
1769     combos_this_level = ai_helper.array_merge(combos_this_level, combos_next_level)
1770     return combos_this_level
1771 end

1773 function ai_helper.get_attack_combos_full(units, enemy, cfg)
1774     -- Calculate attack combination result by @units on @enemy
1775     -- All combinations of all units are taken into account, as well as their order
1776     -- This can result in a _very_ large number of possible combinations
1777     -- Use ai_helper.get_attack_combos() instead if order does not matter
1778     --
1779     -- Optional inputs:
1780     --   @cfg: Configuration table to be passed on to ai_helper.get_attacks()
1781     --
1782     -- Return value:
1783     --   1. Attack combinations in form { dst = src }
1784 
1785     local all_attacks = ai_helper.get_attacks(units, cfg)
1786 
1787     -- Eliminate those that are not on @enemy
1788     local attacks = {}
1789     for _,attack in ipairs(all_attacks) do
1790         if (attack.target.x == enemy.x) and (attack.target.y == enemy.y) then
1791             table.insert(attacks, attack)
1792         end
1793     end
1794     if (not attacks[1]) then return {} end
1795 
1796     -- This recursive function does all the work:
1797     local combos = ai_helper.add_next_attack_combo_level(combos, attacks)
1798 
1799     return combos
1800 end

1802 function ai_helper.get_attack_combos(units, enemy, cfg)
1803     -- Calculate attack combination result by @units on @enemy
1804     -- All the unit/hex combinations are considered, but without specifying the order of the
1805     -- attacks. Use ai_helper.get_attack_combos_full() if order matters.
1806     --
1807     -- Optional inputs:
1808     -- @cfg: Configuration table to be passed on to wesnoth.find_path()
1809     --
1810     -- Return values:
1811     --   1. Attack combinations in form { dst = src }
1812     --   2. All the attacks indexed by [dst][src]
1813 
1814     -- We don't need the full attacks here, just the coordinates,
1815     -- so for speed reasons, we do not use ai_helper.get_attacks()
1816 
1817     -- For units on the current side, we need to make sure that
1818     -- there isn't a unit in the way that cannot move any more
1819     -- TODO: generalize it so that it works not only for units with moves=0, but blocked units etc.
1820     local blocked_hexes = LS.create()
1821     if units[1] and (units[1].side == wesnoth.current.side) then
1822         local all_units = wesnoth.get_units { side = wesnoth.current.side }
1823         for _,unit in ipairs(all_units) do
1824             if (unit.moves == 0) then
1825                 blocked_hexes:insert(unit.x, unit.y)
1826             end
1827         end
1828     end
1829 
1830     -- For sides other than the current, we always use max_moves,
1831     -- for the current side we always use current moves
1832     local old_moves = {}
1833     for i,unit in ipairs(units) do
1834         if (unit.side ~= wesnoth.current.side) then
1835             old_moves[i] = unit.moves
1836             unit.moves = unit.max_moves
1837         end
1838     end
1839 
1840     -- Find which units in @units can get to hexes next to the enemy
1841     local attacks_dst_src = {}
1842     local found_attacks = false
1843     for xa,ya in H.adjacent_tiles(enemy.x, enemy.y) do
1844         -- Make sure the hex is not occupied by unit that cannot move out of the way
1845 
1846         local dst = xa * 1000 + ya
1847 
1848         for _,unit in ipairs(units) do
1849             if ((unit.x == xa) and (unit.y == ya)) or (not blocked_hexes:get(xa, ya)) then
1850 
1851                 -- wesnoth.map.distance_between() is much faster than wesnoth.find_path()
1852                 --> pre-filter using the former
1853                 local cost = M.distance_between(unit.x, unit.y, xa, ya)
1854 
1855                 -- If the distance is <= the unit's MP, then see if it can actually get there
1856                 -- This also means that only short paths have to be evaluated (in most situations)
1857                 if (cost <= unit.moves) then
1858                     local path  -- since cost is already defined outside this block
1859                     path, cost = ai_helper.find_path_with_shroud(unit, xa, ya, cfg)
1860                 end
1861 
1862                 if (cost <= unit.moves) then
1863                     -- for attack by no unit on this hex
1864                     if (not attacks_dst_src[dst]) then
1865                         attacks_dst_src[dst] = { 0, unit.x * 1000 + unit.y }
1866                         found_attacks = true  -- since attacks_dst_src is not a simple array, this is easier
1867                     else
1868                         table.insert(attacks_dst_src[dst], unit.x * 1000 + unit.y )
1869                     end
1870                 end
1871             end
1872         end
1873     end
1874 
1875     for i,unit in ipairs(units) do
1876         if (unit.side ~= wesnoth.current.side) then
1877             unit.moves = old_moves[i]
1878         end
1879     end
1880 
1881     if (not found_attacks) then return {}, {} end
1882 
1883     -- Now we set up an array of all attack combinations
1884     -- at this time, this includes all the 'no unit attacks this hex' elements
1885     -- which have a value of 0 for 'src'
1886     -- They need to be kept in this part, so that we get the combos that do not
1887     -- use the maximum amount of units possible. They will be eliminated below.
1888     local attack_array = {}
1889     -- For all values of 'dst'
1890     for dst,ads in pairs(attacks_dst_src) do
1891         local org_array = ai_helper.table_copy(attack_array)
1892         attack_array = {}
1893 
1894         -- Go through all the values of 'src'
1895         for _,src in ipairs(ads) do
1896             -- If the array does not exist, set it up
1897             if (not org_array[1]) then
1898                 local tmp = {}
1899                 tmp[dst] = src
1900                 table.insert(attack_array, tmp)
1901             else  -- otherwise, add the new dst-src pair to each element of the existing array
1902                 for _,org in ipairs(org_array) do
1903                     -- but only do so if that 'src' value does not exist already
1904                     -- except for 0's those all need to be kept
1905                     local add_attack = true
1906                     for _,s in pairs(org) do
1907                         if (s == src) and (src ~=0) then
1908                             add_attack = false
1909                             break
1910                         end
1911                     end
1912                     -- Finally, add it to the array
1913                     if add_attack then
1914                         local tmp = ai_helper.table_copy(org)
1915                         tmp[dst] = src
1916                         table.insert(attack_array, tmp)
1917                     end
1918                 end
1919             end
1920         end
1921     end
1922 
1923     -- Now eliminate all the 0s
1924     -- Also eliminate the combo that has no attacks on any hex (all zeros)
1925     local i_empty = 0
1926     for i,att in ipairs(attack_array) do
1927         local count = 0
1928         for dst,src in pairs(att) do
1929             if (src == 0) then
1930                 att[dst] = nil
1931             else
1932                 count = count + 1
1933             end
1934         end
1935         if (count == 0) then i_empty = i end
1936     end
1937 
1938     -- This last step eliminates the "empty attack combo" (the one with all zeros)
1939     -- Since this is only one, it's okay to use table.remove (even though it's slow)
1940     table.remove(attack_array, i_empty)
1941 
1942     return attack_array
1943 end

1945 function ai_helper.get_unit_time_of_day_bonus(alignment, lawful_bonus)
1946     local multiplier = 1
1947     if (lawful_bonus ~= 0) then
1948         if (alignment == 'lawful') then
1949             multiplier = (1 + lawful_bonus / 100.)
1950         elseif (alignment == 'chaotic') then
1951             multiplier = (1 - lawful_bonus / 100.)
1952         elseif (alignment == 'liminal') then
1953             multipler = (1 - math.abs(lawful_bonus) / 100.)
1954         end
1955     end
1956     return multiplier
