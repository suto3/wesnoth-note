 108         int count_of_pages() const
 109         {
 110                 int size = chat_log_history.size();
 111                 return (size % COUNT_PER_PAGE == 0) ? (size / COUNT_PER_PAGE)
 112                                                                                         : (size / COUNT_PER_PAGE) + 1;
 113         }

 115         void stream_log(std::ostringstream& s,
 116                                         int first,
 117                                         int last,
 118                                         bool raw = false)
 119         {
 120                 if(first >= last) {
 121                         return;
 122                 }
 123 
 124                 const std::string& lcfilter = utf8::lowercase(filter->get_value());
 125                 LOG_CHAT_LOG << "entering chat_log::model::stream_log\n";
 126 
 127                 for(const auto & t : make_pair(chat_log_history.begin() + first,
 128                                                                                   chat_log_history.begin() + last))
 129                 {
 130                         const std::string& timestamp
 131                                         = preferences::get_chat_timestamp(t.time());
 132 
 133                         if(!lcfilter.empty()) {
 134                                 const std::string& lcsample = utf8::lowercase(timestamp)
 135                                                                                           + utf8::lowercase(t.nick())
 136                                                                                           + utf8::lowercase(t.text());
 137 
 138                                 if(lcsample.find(lcfilter) == std::string::npos) {
 139                                         continue;
 140                                 }
 141                         }
 142 
 143                         const std::string me_prefix = "/me";
 144                         const bool is_me = t.text().compare(0, me_prefix.size(),
 145                                                                                                 me_prefix) == 0;
 146 
 147                         std::string nick_prefix, nick_suffix;
 148 
 149                         if(!raw) {
 150                                 nick_prefix = "<span color=\"" + t.color() + "\">";
 151                                 nick_suffix = "</span> ";
 152                         } else {
 153                                 nick_suffix = " ";
 154                         }
 155 
 156                         const std::string lbracket = raw ? "<" : "&lt;";
 157                         const std::string rbracket = raw ? ">" : "&gt;";
 158 
 159                         //
 160                         // Chat line format:
 161                         //
 162                         // is_me == true:  "<[TS] nick message text here>\n"
 163                         // is_me == false: "<[TS] nick> message text here\n"
 164                         //
 165 
 166                         s << nick_prefix << lbracket;
 167 
 168                         if(raw) {
 169                                 s << timestamp
 170                                   << t.nick();
 171                         } else {
 172                                 s << font::escape_text(timestamp)
 173                                   << font::escape_text(t.nick());
 174                         }
 175 
 176                         if(is_me) {
 177                                 if(!raw) {
 178                                         s << font::escape_text(t.text().substr(3));
 179                                 } else {
 180                                         s << t.text().substr(3);
 181                                 }
 182                                 s << rbracket << nick_suffix;
 183                         } else {
 184                                 // <[TS] nick> message text here
 185                                 s << rbracket << nick_suffix;
 186                                 if(!raw) {
 187                                         s << font::escape_text(t.text());
 188                                 } else {
 189                                         s << t.text();
 190                                 }
 191                         }
 192 
 193                         s << '\n';
 194                 }
 195         }
 196 
 197         void populate_chat_message_list(int first, int last)
 198         {
 199                 std::ostringstream s;
 200                 stream_log(s, first, last);
 201                 msg_label->set_label(s.str());
 202 
 203                 // It makes sense to always scroll to the bottom, since the newest messages are there.
 204                 // The only time this might not be desired is tabbing forward through the pages, since
 205                 // one might want to continue reading the conversation in order.
 206                 //
 207                 // TODO: look into implementing the above suggestion
 208                 dynamic_cast<scroll_label*>(msg_label)->scroll_vertical_scrollbar(scrollbar_base::END);
 209         }

 211         void chat_message_list_to_clipboard(int first, int last)
 212         {
 213                 std::ostringstream s;
 214                 stream_log(s, first, last, true);
 215                 desktop::clipboard::copy_to_clipboard(s.str(), false);
 216         }

 219 // The controller acts upon the model. It retrieves data from repositories,
 220 // persists it, manipulates it, and determines how it will be displayed in the
 221 // view.
 222 class chat_log::controller
 223 {
 224 public:
 225         controller(model& m) : model_(m)
 226         {
 227                 LOG_CHAT_LOG << "Entering chat_log::controller" << std::endl;
 228                 LOG_CHAT_LOG << "Exiting chat_log::controller" << std::endl;
 229         }
 230 
 231         void next_page()
 232         {
 233                 LOG_CHAT_LOG << "Entering chat_log::controller::next_page"
 234                                          << std::endl;
 235                 if(model_.page >= model_.count_of_pages() - 1) {
 236                         return;
 237                 }
 238                 model_.page++;
 239                 LOG_CHAT_LOG << "Set page to " << model_.page + 1 << std::endl;
 240                 update_view_from_model();
 241                 LOG_CHAT_LOG << "Exiting chat_log::controller::next_page" << std::endl;
 242         }
 243 
 244         void previous_page()
 245         {
 246                 LOG_CHAT_LOG << "Entering chat_log::controller::previous_page"
 247                                          << std::endl;
 248                 if(model_.page == 0) {
 249                         return;
 250                 }
 251                 model_.page--;
 252                 LOG_CHAT_LOG << "Set page to " << model_.page + 1 << std::endl;
 253                 update_view_from_model();
 254                 LOG_CHAT_LOG << "Exiting chat_log::controller::previous_page"
 255                                          << std::endl;
 256         }
 257 
 258         void filter()
 259         {
 260                 LOG_CHAT_LOG << "Entering chat_log::controller::filter" << std::endl;
 261                 update_view_from_model();
 262                 LOG_CHAT_LOG << "Exiting chat_log::controller::filter" << std::endl;
 263         }
 264 
 265         void handle_page_number_changed()
 266         {
 267                 LOG_CHAT_LOG
 268                 << "Entering chat_log::controller::handle_page_number_changed"
 269                 << std::endl;
 270                 model_.page = model_.page_number->get_value() - 1;
 271                 LOG_CHAT_LOG << "Set page to " << model_.page + 1 << std::endl;
 272                 update_view_from_model();
 273                 LOG_CHAT_LOG
 274                 << "Exiting chat_log::controller::handle_page_number_changed"
 275                 << std::endl;
 276         }
 277 
 278         std::pair<int, int> calculate_log_line_range()
 279         {
 280                 const int log_size = model_.chat_log_history.size();
 281                 const int page_size = model_.COUNT_PER_PAGE;
 282 
 283                 const int page = model_.page;
 284                 const int count_of_pages = std::max(1, model_.count_of_pages());
 285 
 286                 LOG_CHAT_LOG << "Page: " << page + 1 << " of " << count_of_pages
 287                              << '\n';
 288 
 289                 const int first = page * page_size;
 290                 const int last = page < (count_of_pages - 1)
 291                                                  ? first + page_size
 292                                                  : log_size;
 293 
 294                 LOG_CHAT_LOG << "First " << first << ", last " << last << '\n';
 295 
 296                 return std::make_pair(first, last);
 297         }
 298 
 299         void update_view_from_model(bool select_last_page = false)
 300         {
 301                 LOG_CHAT_LOG << "Entering chat_log::controller::update_view_from_model"
 302                                          << std::endl;
 303                 model_.msg_label->set_use_markup(true);
 304                 int size = model_.chat_log_history.size();
 305                 LOG_CHAT_LOG << "Number of chat messages: " << size << std::endl;
 306                 // determine count of pages
 307                 const int count_of_pages = std::max(1, model_.count_of_pages());
 308                 if(select_last_page) {
 309                         model_.page = count_of_pages - 1;
 310                 }
 311                 // get page
 312                 const int page = model_.page;
 313                 // determine first and last
 314                 const std::pair<int, int>& range = calculate_log_line_range();
 315                 const int first = range.first;
 316                 const int last = range.second;
 317                 // determine has previous, determine has next
 318                 bool has_next = page + 1 < count_of_pages;
 319                 bool has_previous = page > 0;
 320                 model_.previous_page->set_active(has_previous);
 321                 model_.next_page->set_active(has_next);
 322                 model_.populate_chat_message_list(first, last);
 323                 model_.page_number->set_minimum_value(1);
 324                 model_.page_number->set_maximum_value(count_of_pages);
 325                 model_.page_number->set_active(count_of_pages > 1);
 326                 LOG_CHAT_LOG << "Maximum value of page number slider: "
 327                                          << count_of_pages << std::endl;
 328                 model_.page_number->set_value(page + 1);
 329 
 330                 std::ostringstream cur_page_text;
 331                 cur_page_text << (page + 1) << '/' << std::max(1, count_of_pages);
 332                 model_.page_label->set_label(cur_page_text.str());
 333 
 334                 LOG_CHAT_LOG << "Exiting chat_log::controller::update_view_from_model"
 335                                          << std::endl;
 336         }
 337 
 338         void handle_copy_button_clicked()
 339         {
 340                 const std::pair<int, int>& range = calculate_log_line_range();
 341                 model_.chat_message_list_to_clipboard(range.first, range.second);
 342         }

 349 // The view is an interface that displays data (the model) and routes user
 350 // commands to the controller to act upon that data.
 351 class chat_log::view
 352 {
 353 public:
 354         view(const vconfig& cfg, const replay& r) : model_(cfg, r), controller_(model_)
 355         {
 356         }
 357 
 358         void pre_show()
 359         {
 360                 LOG_CHAT_LOG << "Entering chat_log::view::pre_show" << std::endl;
 361                 controller_.update_view_from_model(true);
 362                 LOG_CHAT_LOG << "Exiting chat_log::view::pre_show" << std::endl;
 363         }
 364 
 365         void handle_page_number_changed()
 366         {
 367                 controller_.handle_page_number_changed();
 368         }
 369 
 370         void next_page()
 371         {
 372                 controller_.next_page();
 373         }
 374 
 375         void previous_page()
 376         {
 377                 controller_.previous_page();
 378         }
 379 
 380         void filter()
 381         {
 382                 controller_.filter();
 383         }
 384 
 385         void handle_copy_button_clicked(window& /*window*/)
 386         {
 387                 controller_.handle_copy_button_clicked();
 388         }
 389 
 390         void bind(window& window)
 391         {
 392                 LOG_CHAT_LOG << "Entering chat_log::view::bind" << std::endl;
 393                 model_.msg_label = &find_widget<styled_widget>(&window, "msg", false);
 394                 model_.page_number
 395                                 = &find_widget<slider>(&window, "page_number", false);
 396                 connect_signal_notify_modified(
 397                                 *model_.page_number,
 398                                 std::bind(&view::handle_page_number_changed, this));
 399 
 400                 model_.previous_page
 401                                 = &find_widget<button>(&window, "previous_page", false);
 402                 model_.previous_page->connect_click_handler(
 403                                 std::bind(&view::previous_page, this));
 404 
 405                 model_.next_page = &find_widget<button>(&window, "next_page", false);
 406                 model_.next_page->connect_click_handler(
 407                                 std::bind(&view::next_page, this));
 408 
 409                 model_.filter = &find_widget<text_box>(&window, "filter", false);
 410                 model_.filter->set_text_changed_callback(
 411                                 std::bind(&view::filter, this));
 412                 window.keyboard_capture(model_.filter);
 413 
 414                 model_.copy_button = &find_widget<button>(&window, "copy", false);
 415                 connect_signal_mouse_left_click(
 416                                 *model_.copy_button,
 417                                 std::bind(&view::handle_copy_button_clicked,
 418                                                         this,
 419                                                         std::ref(window)));
 420                 if (!desktop::clipboard::available()) {
 421                         model_.copy_button->set_active(false);
 422                         model_.copy_button->set_tooltip(_("Clipboard support not found, contact your packager"));
 423                 }
 424 
 425                 model_.page_label = &find_widget<styled_widget>(&window, "page_label", false);
 426 
 427                 LOG_CHAT_LOG << "Exiting chat_log::view::bind" << std::endl;
 428         }
 429 
 430 private:
 431         model model_;
 432         controller controller_;
 433 };
