  67 mp_login::mp_login(const std::string& label, const bool focus_password)
  68 {
  69         register_label("login_label", false, label);
  70         register_text("user_name", true,
  71                 &preferences::login,
  72                 &preferences::set_login,
  73                 !focus_password);
  74 
  75         register_text("password",  true,
  76                 &preferences::password,
  77                 nullptr /* The password box returns '*' as value. */,
  78                 focus_password);
  79 
  80         register_bool("remember_password", false,
  81                 &preferences::remember_password,
  82                 &preferences::set_remember_password);
  83 }
  84 
  85 void mp_login::pre_show(window& win)
  86 {
  87         if(button* btn = find_widget<button>(&win, "password_reminder", false, false)) {
  88 
  89                 btn->set_retval(1);
  90         }
  91 
  92         if(button* btn = find_widget<button>(&win, "change_username", false, false)) {
  93 
  94                 btn->set_retval(2);
  95         }
  96 
  97         win.add_to_tab_order(find_widget<text_box>(&win, "user_name", false, false));
  98         win.add_to_tab_order(find_widget<text_box>(&win, "password", false, false));
  99 }
 100 
 101 void mp_login::post_show(window& win)
 102 {
 103         if(get_retval() == window::OK) {
 104                 preferences::set_password(find_widget<password_box>(&win, "password", false).get_real_value());
 105         }
 106 }
