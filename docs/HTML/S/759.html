 116         /// Stores a set of defense levels.
 117         class terrain_defense
 118         {
 119                 static const terrain_info::parameters params_min_;
 120                 static const terrain_info::parameters params_max_;
 121 
 122         public:
 123                 terrain_defense() : min_(params_min_), max_(params_max_) {}
 124                 explicit terrain_defense(const config & cfg) :
 125                         min_(cfg, params_min_), max_(cfg, params_max_)
 126                 {}
 127 
 128                 /// Returns the defense associated with the given terrain.
 129                 int defense(const t_translation::terrain_code & terrain) const
 130                 { return std::max(min_.value(terrain), max_.value(terrain)); }
 131                 /// Returns whether there is a defense cap associated to this terrain.
 132                 bool capped(const t_translation::terrain_code & terrain) const
 133                 { return min_.value(terrain) != 0; }
 134                 /// Merges the given config over the existing costs.
 135                 /// (Not overwriting implies adding.)
 136                 void merge(const config & new_data, bool overwrite)
 137                 { min_.merge(new_data, overwrite);  max_.merge(new_data, overwrite); }
 138                 /// Writes our data to a config, as a child if @a child_name is specified.
 139                 /// (No child is created if there is no data.)
 140                 void write(config & cfg, const std::string & child_name="") const
 141                 { max_.write(cfg, child_name, false); }
 142 
 143         private:
 144                 // There will be duplication of the config here, but it is a small
 145                 // config, and the duplication allows greater code sharing.
 146                 terrain_info min_;
 147                 terrain_info max_;
 148         };
