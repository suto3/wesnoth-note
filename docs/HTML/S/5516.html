  32 function wml_actions.sync_variable(cfg)
  33         local names = cfg.name or helper.wml_error "[sync_variable] missing required name= attribute."
  34         local result = wesnoth.synchronize_choice(
  35                 function()
  36                         local res = {}
  37                         for name_raw in utils.split(names) do
  38                                 local name = utils.trim(name_raw)
  39                                 local variable_type = string.sub(name, string.len(name)) == "]" and "indexed" or ( wesnoth.get_variable(name .. ".length") > 0 and "array" or "attribute")
  40                                 local variable_info = { name = name, type = variable_type }
  41                                 table.insert(res, { "variable", variable_info })
  42                                 if variable_type == "indexed" then
  43                                         table.insert(variable_info, { "value", wesnoth.get_variable(name) } )
  44                                 elseif variable_type == "array" then
  45                                         for i = 1, wesnoth.get_variable(name .. ".length") do
  46                                                 table.insert(variable_info, { "value",  wesnoth.get_variable(string.format("%s[%d]", name, i - 1)) } )
  47                                         end
  48                                 else
  49                                         variable_info.value = wesnoth.get_variable(name)
  50                                 end
  51                         end
  52                         return res
  53                 end
  54         )
  55         for variable in helper.child_range(result, "variable") do
  56                 local name = variable.name
  57 
  58                 if variable.type == "indexed" then
  59                         wesnoth.set_variable(name, variable[1][2])
  60                 elseif variable.type == "array" then
  61                         for index, cfg_pair in ipairs(variable) do
  62                                 wesnoth.set_variable( string.format("%s[%d]", name, index - 1), cfg_pair[2])
  63                         end
  64                 else
  65                         wesnoth.set_variable(name, variable.value)
  66                 end
  67         end
  68 end

 242 function wml_actions.unit_worth(cfg)
 243         local u = wesnoth.get_units(cfg)[1] or
 244                 helper.wml_error "[unit_worth]'s filter didn't match any unit"
 245         local ut = wesnoth.unit_types[u.type]
 246         local hp = u.hitpoints / u.max_hitpoints
 247         local xp = u.experience / u.max_experience
 248         local best_adv = ut.cost
 249         for w in utils.split(ut.__cfg.advances_to) do
 250                 local uta = wesnoth.unit_types[w]
 251                 if uta and uta.cost > best_adv then best_adv = uta.cost end
 252         end
 253         wesnoth.set_variable("cost", ut.cost)
 254         wesnoth.set_variable("next_cost", best_adv)
 255         wesnoth.set_variable("health", math.floor(hp * 100))
 256         wesnoth.set_variable("experience", math.floor(xp * 100))
 257         wesnoth.set_variable("recall_cost", ut.recall_cost)
 258         wesnoth.set_variable("unit_worth", math.floor(math.max(ut.cost * hp, best_adv * xp)))
 259 end

 350 function wml_actions.remove_unit_overlay(cfg)
 351         local img = cfg.image or helper.wml_error( "[remove_unit_overlay] missing required image= attribute" )
 352 
 353         -- Loop through all matching units.
 354         for i,u in ipairs(wesnoth.get_units(cfg)) do
 355                 local ucfg = u.__cfg
 356                 local t = utils.parenthetical_split(ucfg.overlays)
 357                 -- Remove the specified image from the overlays.
 358                 for i = #t,1,-1 do
 359                         if t[i] == img then table.remove(t, i) end
 360                 end
 361                 -- Reassemble the list of remaining overlays.
 362                 ucfg.overlays = table.concat(t, ',')
 363                 wesnoth.put_unit(ucfg)
 364         end
 365 end

 372 function wml_actions.store_unit(cfg)
 373         local filter = helper.get_child(cfg, "filter") or
 374                 helper.wml_error "[store_unit] missing required [filter] tag"
 375         local kill_units = cfg.kill
 376 
 377         --cache the needed units here, since the filter might reference the to-be-cleared variable(s)
 378         local units = wesnoth.get_units(filter)
 379         local recall_units = wesnoth.get_recall_units(filter)
 380 
 381         local writer = utils.vwriter.init(cfg, "unit")
 382 
 383         for i,u in ipairs(units) do
 384                 utils.vwriter.write(writer, u.__cfg)
 385                 if kill_units then wesnoth.erase_unit(u) end
 386         end
 387 
 388         if (not filter.x or filter.x == "recall") and (not filter.y or filter.y == "recall") then
 389                 for i,u in ipairs(recall_units) do
 390                         local ucfg = u.__cfg
 391                         ucfg.x = "recall"
 392                         ucfg.y = "recall"
 393                         utils.vwriter.write(writer, ucfg)
 394                         if kill_units then wesnoth.erase_unit(u) end
 395                 end
 396         end
 397 end

 420 function wml_actions.store_reachable_locations(cfg)
 421         local unit_filter = helper.get_child(cfg, "filter") or
 422                 helper.wml_error "[store_reachable_locations] missing required [filter] tag"
 423         local location_filter = helper.get_child(cfg, "filter_location")
 424         local range = cfg.range or "movement"
 425         local moves = cfg.moves or "current"
 426         local variable = cfg.variable or helper.wml_error "[store_reachable_locations] missing required variable= key"
 427         local reach_param = { viewing_side = cfg.viewing_side or 0 }
 428         if range == "vision" then
 429                 moves = "max"
 430                 reach_param.ignore_units = true
 431         end
 432 
 433         local reach = location_set.create()
 434 
 435         for i,unit in ipairs(wesnoth.get_units(unit_filter)) do
 436                 local unit_reach
 437                 if moves == "max" then
 438                         local saved_moves = unit.moves
 439                         unit.moves = unit.max_moves
 440                         unit_reach = location_set.of_pairs(wesnoth.find_reach(unit, reach_param))
 441                         unit.moves = saved_moves
 442                 else
 443                         unit_reach = location_set.of_pairs(wesnoth.find_reach(unit, reach_param))
 444                 end
 445 
 446                 if range == "vision" or range == "attack" then
 447                         unit_reach:iter(function(x, y)
 448                                 reach:insert(x, y)
 449                                 for u,v in helper.adjacent_tiles(x, y) do
 450                                         reach:insert(u, v)
 451                                 end
 452                         end)
 453                 else
 454                         reach:union(unit_reach)
 455                 end
 456         end
 457 
 458         if location_filter then
 459                 reach = reach:filter(function(x, y)
 460                         return wesnoth.match_location(x, y, location_filter)
 461                 end)
 462         end
 463         reach:to_wml_var(variable)
 464 end

 548 function wml_actions.transform_unit(cfg)
 549         local transform_to = cfg.transform_to
 550 
 551         for index, unit in ipairs(wesnoth.get_units(cfg)) do
 552 
 553                 if transform_to then
 554                         wesnoth.transform_unit( unit, transform_to )
 555                 else
 556                         local hitpoints = unit.hitpoints
 557                         local experience = unit.experience
 558                         local recall_cost = unit.recall_cost
 559                         local status = helper.get_child( unit.__cfg, "status" )
 560 
 561                         unit.experience = unit.max_experience
 562                         wesnoth.advance_unit(unit, false, false)
 563 
 564                         unit.hitpoints = hitpoints
 565                         unit.experience = experience
 566                         recall_cost = unit.recall_cost
 567 
 568                         for key, value in pairs(status) do unit.status[key] = value end
 569                         if unit.status.unpoisonable then unit.status.poisoned = nil end
 570                 end
 571         end
 572 
 573         wml_actions.redraw {}
 574 end

 593 function wml_actions.add_ai_behavior(cfg)

 595                 helper.wml_error("[add_ai_behavior]: no unit specified")
 596 
 597         local side = cfg.side or
 598                 helper.wml_error("[add_ai_behavior]: no side attribute given")
 599

 602         local eval = cfg.evaluation
 603         local exec = cfg.execution
 604         local id = cfg.bca_id or ("bca-" .. unit.__cfg.underlying_id)
 605 
 606         local ux = unit.x -- @note: did I get it right that coordinates in C++ differ by 1 from thos in-game(and in Lua)?
 607         local uy = unit.y
 608 
 609         if not (eval and exec) then
 610                 helper.wml_error("[add_ai_behavior]: invalid execution/evaluation handler(s)")
 611         end
 612 
 613         local path = "stage[" .. loop_id .. "].candidate_action[" .. id .. "]" -- bca: behavior candidate action
 614 
 615         wesnoth.wml_actions.modify_ai {
 616                 action = "add",
 617                 engine = "lua",
 618                 path = path,
 619                 side = side,
 620 
 621                 T.candidate_action {
 622                         id = id,
 623                         name = id,
 624                         engine = "lua",
 625                         sticky = sticky,
 626                         unit_x = ux,
 627                         unit_y = uy,
 628                         evaluation = eval,
 629                         execution = exec
 630                 },
 631         }

 807 local function on_board(x, y)
 808         if type(x) ~= "number" or type(y) ~= "number" then
 809                 return false
 810         end
 811         local w, h = wesnoth.get_map_size()
 812         return x >= 1 and y >= 1 and x <= w and y <= h
 813 end

 815 wml_actions.unstore_unit = function(cfg)
 816         local variable = cfg.variable or helper.wml_error("[unstore_unit] missing required 'variable' attribute")
 817         local unit_cfg = wesnoth.get_variable(variable)
 818         local unit = wesnoth.create_unit(unit_cfg)
 819         local advance = cfg.advance ~= false
 820         local animate = cfg.animate ~= false
 821         local check_passability = cfg.check_passability ~= false or nil
 822         local x = cfg.x or unit.x or -1
 823         local y = cfg.y or unit.y or -1
 824         wesnoth.add_known_unit(unit.type)
 825         if on_board(x, y) then
 826                 if cfg.find_vacant then
 827                         x,y = wesnoth.find_vacant_tile(x, y, check_passability and unit)
 828                 end
 829                 unit:to_map(x, y, cfg.fire_event)
 830                 local text = nil
 831                 if unit_cfg.gender == "female" then
 832                         text = cfg.female_text or cfg.text
 833                 else
 834                         text = cfg.male_text or cfg.text
 835                 end
 836                 local color = cfg.color
 837                 if color == nil and cfg.red and cfg.blue and cfg.green then
 838                         color = cfg.red .. "," .. cfg.green .. "," .. cfg.blue
 839                 end
 840                 if text ~= nil and not wesnoth.is_skipping_messages() then
 841                         wesnoth.float_label(x, y, text, color)
 842                 end
 843                 if advance then
 844                         wesnoth.advance_unit(unit, animate, cfg.fire_event)
 845                 end
 846         else
 847                 unit:to_recall()
 848         end
 849 end
