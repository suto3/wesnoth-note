  42 static unsigned int sdl_get_mods()
  43 {
  44         unsigned int mods;
  45         mods = SDL_GetModState();
  46 
  47         mods &= ~KMOD_NUM;
  48         mods &= ~KMOD_CAPS;
  49         mods &= ~KMOD_MODE;
  50 
  51         // save the matching for checking right vs left keys
  52         if (mods & KMOD_SHIFT)
  53                 mods |= KMOD_SHIFT;
  54 
  55         if (mods & KMOD_CTRL)
  56                 mods |= KMOD_CTRL;
  57 
  58         if (mods & KMOD_ALT)
  59                 mods |= KMOD_ALT;
  60 
  61         if (mods & KMOD_GUI)
  62         mods |= KMOD_GUI;
  63 
  64         return mods;
  65 }

 102 bool hotkey_base::bindings_equal(hotkey_ptr other)
 103 {
 104         if (other == hotkey_ptr()) {
 105                 return false;
 106         }
 107 
 108         hk_scopes scopematch = hotkey::get_hotkey_command(get_command()).scope
 109                         & hotkey::get_hotkey_command(other->get_command()).scope;
 110 
 111         if (scopematch.none()) {
 112                 return false;
 113         }
 114 
 115         return mod_ == other->mod_ && bindings_equal_helper(other);
 116 }

 141 hotkey_ptr create_hotkey(const std::string &id, const SDL_Event &event)
 142 {
 143         hotkey_ptr base = hotkey_ptr(new hotkey_void);
 144         unsigned mods = sdl_get_mods();
 145 
 146         switch (event.type) {
 147         case SDL_KEYDOWN:
 148         case SDL_KEYUP: {
 149                 if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI || CKey::is_uncomposable(event.key)) {
 150                         hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 151                         base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 152                         SDL_Keycode code;
 153                         code = event.key.keysym.sym;
 154                         keyboard->set_keycode(code);
 155                         keyboard->set_text(SDL_GetKeyName(event.key.keysym.sym));
 156                 }
 157         }
 158                 break;
 159         case SDL_TEXTINPUT: {
 160                 hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 161                 base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 162                 std::string text =  std::string(event.text.text);
 163                 keyboard->set_text(text);
 164                 if (text == ":") {
 165                         mods = mods & ~KMOD_SHIFT;
 166                 }
 167         }
 168                 break;
 169         case SDL_MOUSEBUTTONDOWN:
 170         case SDL_MOUSEBUTTONUP: {
 171                 hotkey_mouse_ptr mouse(new hotkey_mouse());
 172                 base = std::dynamic_pointer_cast<hotkey_base>(mouse);
 173                 mouse->set_button(event.button.button);
 174                 break;
 175         }
 176         default:
 177                 ERR_G<< "Trying to bind an unknown event type:" << event.type << "\n";
 178                 break;
 179         }
 180 
 181         base->set_mods(mods);
 182         base->set_command(id);
 183         base->unset_default();
 184 
 185         return base;
 186 }

 252 bool hotkey_mouse::matches_helper(const SDL_Event &event) const
 253 {
 254         if (event.type != SDL_MOUSEBUTTONUP && event.type != SDL_MOUSEBUTTONDOWN) {
 255                 return false;
 256         }
 257 
 258         unsigned int mods = sdl_get_mods();
 259         if ((mods != mod_)) {
 260                 return false;
 261         }
 262 
 263         if (event.button.button != button_) {
 264                 return false;
 265         }
 266 
 267         return true;
 268 }

 288 bool hotkey_keyboard::matches_helper(const SDL_Event &event) const
 289 {
 290         unsigned int mods = sdl_get_mods();
 291 
 292         if ((event.type == SDL_KEYDOWN || event.type == SDL_KEYUP) &&
 293                                 (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI ||
 294                           CKey::is_uncomposable(event.key))) {
 295                 return event.key.keysym.sym == keycode_ && mods == mod_;
 296         }
 297 
 298         if (event.type == SDL_TEXTINPUT) {
 299                 std::string text = std::string(event.text.text);
 300                 boost::algorithm::to_lower(text);
 301                 if (text == ":") {
 302                         mods = mods & ~KMOD_SHIFT;
 303                 }
 304                 return text_ == text && mods == mod_;
 305         }
 306 
 307         return false;
 308 }

 357 void add_hotkey(const hotkey_ptr item)
 358 {
 359 
 360         if (item == hotkey_ptr()) {
 361                 return;
 362         }
 363 
 364         scope_changer scope_ch;
 365         set_active_scopes(hotkey::get_hotkey_command(item->get_command()).scope);
 366 
 367         if (!hotkeys_.empty()) {
 368                 hotkeys_.erase(
 369                         std::remove_if(hotkeys_.begin(), hotkeys_.end(), [item](const hotkey::hotkey_ptr& hk) { return hk->bindings_equal(item); }),
 370                         hotkeys_.end());
 371         }
 372 
 373         hotkeys_.push_back(item);
 374 
 375 }

 405 void load_hotkeys(const config& cfg, bool set_as_default)
 406 {
 407         for (const config &hk : cfg.child_range("hotkey")) {
 408 
 409                 hotkey_ptr item = load_from_config(hk);
 410                 if (!set_as_default) {
 411                         item->unset_default();
 412                 }
 413 
 414                 if (!item->null()) {
 415                         add_hotkey(item);
 416                 }
 417         }
 418 
 419         if (set_as_default) {
 420                 default_hotkey_cfg_ = cfg;
 421         }
 422 }

 452 std::string get_names(const std::string& id)
 453 {
 454         // Names are used in places like the hot-key preferences menu
 455         std::vector<std::string> names;
 456         for (const hotkey::hotkey_ptr item : hotkeys_) {
 457                 if (item->get_command() == id && !item->null() && !item->is_disabled()) {
 458                         names.push_back(item->get_name());
 459                 }
 460         }
 461 
 462         // These are hard-coded, non-rebindable hotkeys
 463         if (id == "quit") {
 464                 names.push_back("escape");
 465         }
 466         else if (id == "quit-to-desktop") {
 467 #ifdef __APPLE__
 468                 names.push_back("cmd+q");
 469 #else
 470                 names.push_back("alt+F4");
 471 #endif
 472         }
 473 
 474         return boost::algorithm::join(names, ", ");
 475 }

 477 bool is_hotkeyable_event(const SDL_Event &event) {
 478 
 479         if (event.type == SDL_JOYBUTTONUP ||
 480                         event.type == SDL_JOYHATMOTION ||
 481                         event.type == SDL_MOUSEBUTTONUP) {
 482                 return true;
 483         }
 484 
 485         unsigned mods = sdl_get_mods();
 486 
 487         if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI) {
 488                 return event.type == SDL_KEYUP;
 489         } else {
 490                 return event.type == SDL_TEXTINPUT ||
 491                                 (event.type  == SDL_KEYUP && CKey::is_uncomposable(event.key));
 492         }
 493 }
