  43 static unsigned int sdl_get_mods()
  44 {
  45         unsigned int mods;
  46         mods = SDL_GetModState();
  47 
  48         mods &= ~KMOD_NUM;
  49         mods &= ~KMOD_CAPS;
  50         mods &= ~KMOD_MODE;
  51 
  52         // save the matching for checking right vs left keys
  53         if (mods & KMOD_SHIFT)
  54                 mods |= KMOD_SHIFT;
  55 
  56         if (mods & KMOD_CTRL)
  57                 mods |= KMOD_CTRL;
  58 
  59         if (mods & KMOD_ALT)
  60                 mods |= KMOD_ALT;
  61 
  62         if (mods & KMOD_GUI)
  63         mods |= KMOD_GUI;
  64 
  65         return mods;
  66 }

 103 bool hotkey_base::bindings_equal(hotkey_ptr other)
 104 {
 105         if (other == hotkey_ptr()) {
 106                 return false;
 107         }
 108 
 109         hk_scopes scopematch = hotkey::get_hotkey_command(get_command()).scope
 110                         & hotkey::get_hotkey_command(other->get_command()).scope;
 111 
 112         if (scopematch.none()) {
 113                 return false;
 114         }
 115 
 116         return mod_ == other->mod_ && bindings_equal_helper(other);
 117 }

 142 hotkey_ptr create_hotkey(const std::string &id, const SDL_Event &event)
 143 {
 144         hotkey_ptr base = hotkey_ptr(new hotkey_void);
 145         unsigned mods = sdl_get_mods();
 146 
 147         switch (event.type) {
 148         case SDL_KEYDOWN:
 149         case SDL_KEYUP: {
 150                 if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI || CKey::is_uncomposable(event.key)) {
 151                         hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 152                         base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 153                         SDL_Keycode code;
 154                         code = event.key.keysym.sym;
 155                         keyboard->set_keycode(code);
 156                         keyboard->set_text(SDL_GetKeyName(event.key.keysym.sym));
 157                 }
 158         }
 159                 break;
 160         case SDL_TEXTINPUT: {
 161                 hotkey_keyboard_ptr keyboard(new hotkey_keyboard());
 162                 base = std::dynamic_pointer_cast<hotkey_base>(keyboard);
 163                 std::string text =  std::string(event.text.text);
 164                 keyboard->set_text(text);
 165                 if(text == ":" || text == "`") {
 166                         mods = mods & ~KMOD_SHIFT;
 167                 }
 168         }
 169                 break;
 170         case SDL_MOUSEBUTTONDOWN:
 171         case SDL_MOUSEBUTTONUP: {
 172                 hotkey_mouse_ptr mouse(new hotkey_mouse());
 173                 base = std::dynamic_pointer_cast<hotkey_base>(mouse);
 174                 mouse->set_button(event.button.button);
 175                 break;
 176         }
 177         default:
 178                 ERR_G<< "Trying to bind an unknown event type:" << event.type << "\n";
 179                 break;
 180         }
 181 
 182         base->set_mods(mods);
 183         base->set_command(id);
 184         base->unset_default();
 185 
 186         return base;
 187 }

 253 bool hotkey_mouse::matches_helper(const SDL_Event &event) const
 254 {
 255         if (event.type != SDL_MOUSEBUTTONUP && event.type != SDL_MOUSEBUTTONDOWN) {
 256                 return false;
 257         }
 258 
 259         unsigned int mods = sdl_get_mods();
 260         if ((mods != mod_)) {
 261                 return false;
 262         }
 263 
 264         if (event.button.button != button_) {
 265                 return false;
 266         }
 267 
 268         return true;
 269 }

 289 bool hotkey_keyboard::matches_helper(const SDL_Event &event) const
 290 {
 291         unsigned int mods = sdl_get_mods();
 292 
 293         if ((event.type == SDL_KEYDOWN || event.type == SDL_KEYUP) &&
 294                                 (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI ||
 295                           CKey::is_uncomposable(event.key))) {
 296                 return event.key.keysym.sym == keycode_ && mods == mod_;
 297         }
 298 
 299         if (event.type == SDL_TEXTINPUT) {
 300                 std::string text = std::string(event.text.text);
 301                 boost::algorithm::to_lower(text);
 302                 if(text == ":" || text == "`") {
 303                         mods = mods & ~KMOD_SHIFT;
 304                 }
 305                 return text_ == text && utf8::size(utf8::string(event.text.text)) == 1 && mods == mod_;
 306         }
 307 
 308         return false;
 309 }

 358 void add_hotkey(const hotkey_ptr item)
 359 {
 360 
 361         if (item == hotkey_ptr()) {
 362                 return;
 363         }
 364 
 365         scope_changer scope_ch;
 366         set_active_scopes(hotkey::get_hotkey_command(item->get_command()).scope);
 367 
 368         if (!hotkeys_.empty()) {
 369                 hotkeys_.erase(
 370                         std::remove_if(hotkeys_.begin(), hotkeys_.end(), [item](const hotkey::hotkey_ptr& hk) { return hk->bindings_equal(item); }),
 371                         hotkeys_.end());
 372         }
 373 
 374         hotkeys_.push_back(item);
 375 
 376 }

 406 void load_hotkeys(const config& cfg, bool set_as_default)
 407 {
 408         for (const config &hk : cfg.child_range("hotkey")) {
 409 
 410                 hotkey_ptr item = load_from_config(hk);
 411                 if (!set_as_default) {
 412                         item->unset_default();
 413                 }
 414 
 415                 if (!item->null()) {
 416                         add_hotkey(item);
 417                 }
 418         }
 419 
 420         if (set_as_default) {
 421                 default_hotkey_cfg_ = cfg;
 422         }
 423 }

 453 std::string get_names(const std::string& id)
 454 {
 455         // Names are used in places like the hot-key preferences menu
 456         std::vector<std::string> names;
 457         for (const hotkey::hotkey_ptr item : hotkeys_) {
 458                 if (item->get_command() == id && !item->null() && !item->is_disabled()) {
 459                         names.push_back(item->get_name());
 460                 }
 461         }
 462 
 463         // These are hard-coded, non-rebindable hotkeys
 464         if (id == "quit") {
 465                 names.push_back("escape");
 466         }
 467         else if (id == "quit-to-desktop") {
 468 #ifdef __APPLE__
 469                 names.push_back("cmd+q");
 470 #else
 471                 names.push_back("alt+F4");
 472 #endif
 473         }
 474 
 475         return boost::algorithm::join(names, ", ");
 476 }

 478 bool is_hotkeyable_event(const SDL_Event &event) {
 479 
 480         if (event.type == SDL_JOYBUTTONUP ||
 481                         event.type == SDL_JOYHATMOTION ||
 482                         event.type == SDL_MOUSEBUTTONUP) {
 483                 return true;
 484         }
 485 
 486         unsigned mods = sdl_get_mods();
 487 
 488         if (mods & KMOD_CTRL || mods & KMOD_ALT || mods & KMOD_GUI) {
 489                 return event.type == SDL_KEYUP;
 490         } else {
 491                 return event.type == SDL_TEXTINPUT ||
 492                                 (event.type  == SDL_KEYUP && CKey::is_uncomposable(event.key));
 493         }
 494 }
