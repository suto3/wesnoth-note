  59 static void parse_about_tag(const config& cfg, std::stringstream& ss)
  60 {
  61         if(!cfg.has_child("entry")) {
  62                 return;
  63         }
  64 
  65         ss << "\n" << "<span size='x-large'>" << cfg["title"] << "</span>" << "\n";
  66 
  67         for(const auto& entry : cfg.child_range("entry")) {
  68                 ss << entry["name"] << "\n";
  69         }
  70 }

  72 void end_credits::pre_show(window& window)
  73 {
  74         window.set_callback_next_draw([this]()
  75         {
  76                 timer_id_ = add_timer(10, std::bind(&end_credits::timer_callback, this), true);
  77                 // Delay a little before beginning the scrolling
  78                 last_scroll_ = SDL_GetTicks() + 3000;
  79         });
  80 
  81         connect_signal_pre_key_press(window, std::bind(&end_credits::key_press_callback, this, _5));
  82 
  83         std::stringstream ss;
  84         std::stringstream focus_ss;
  85 
  86         const config& credits_config = about::get_about_config();
  87 
  88         for(const auto& group : credits_config.child_range("credits_group")) {
  89                 std::stringstream& group_stream = (group["id"] == focus_on_) ? focus_ss : ss;
  90 
  91                 group_stream << "\n";
  92                 if(group.has_attribute("title")) {
  93                         group_stream << "<span size='xx-large'>" << group["title"] << "</span>" << "\n";
  94                 }
  95 
  96                 if(group["sort"].to_bool(false)) {
  97                         auto sections = group.child_range("about");
  98                         std::vector<config> sorted(sections.begin(), sections.end());
  99                         std::sort(sorted.begin(), sorted.end(), [](const config& entry1, const config& entry2) {
 100                                 return translation::compare(entry1["title"].str(), entry2["title"].str()) < 0;
 101                         });
 102                         for(const auto& about : sorted) {
 103                                 parse_about_tag(about, group_stream);
 104                         }
 105                 } else {
 106                         for(const auto& about : group.child_range("about")) {
 107                                 parse_about_tag(about, group_stream);
 108                         }
 109                 }
 110         }
 111 
 112         // If a section is focused, move it to the top
 113         if(!focus_ss.str().empty()) {
 114                 focus_ss << ss.rdbuf();
 115         }
 116 
 117         // Get the appropriate background images
 118         backgrounds_ = about::get_background_images(focus_on_);
 119 
 120         if(backgrounds_.empty()) {
 121                 backgrounds_.push_back(game_config::images::game_title_background);
 122         }
 123 
 124         // TODO: implement showing all available images as the credits scroll
 125         window.get_canvas(0).set_variable("background_image", wfl::variant(backgrounds_[0]));
 126 
 127         text_widget_ = find_widget<scroll_label>(&window, "text", false, true);
 128 
 129         text_widget_->set_use_markup(true);
 130         text_widget_->set_link_aware(false);
 131         text_widget_->set_label((focus_ss.str().empty() ? ss : focus_ss).str());
 132 
 133         // HACK: always hide the scrollbar, even if it's needed.
 134         // This should probably be implemented as a scrollbar mode.
 135         // Also, for some reason hiding the whole grid doesn't work, and the elements need to be hidden manually
 136         if(grid* v_grid = dynamic_cast<grid*>(text_widget_->find("_vertical_scrollbar_grid", false))) {
 137                 find_widget<scrollbar_base>(v_grid, "_vertical_scrollbar", false).set_visible(widget::visibility::hidden);
 138 
 139                 // TODO: enable again if e24336afeb7 is reverted.
 140                 //find_widget<repeating_button>(v_grid, "_half_page_up", false).set_visible(widget::visibility::hidden);
 141                 //find_widget<repeating_button>(v_grid, "_half_page_down", false).set_visible(widget::visibility::hidden);
 142         }
 143 }

 145 void end_credits::timer_callback()
 146 {
 147         uint32_t now = SDL_GetTicks();
 148         if(last_scroll_ > now) {
 149                 return;
 150         }
 151 
 152         uint32_t missed_time = now - last_scroll_;
 153 
 154         unsigned int cur_pos = text_widget_->get_vertical_scrollbar_item_position();
 155 
 156         // Calculate how far the text should have scrolled by now
 157         // The division by 1000 is to convert milliseconds to seconds.
 158         unsigned int needed_dist = missed_time * scroll_speed_ / 1000;
 159 
 160         text_widget_->set_vertical_scrollbar_item_position(cur_pos + needed_dist);
 161 
 162         last_scroll_ = now;
 163 
 164         if(text_widget_->vertical_scrollbar_at_end()) {
 165                 remove_timer(timer_id_);
 166         }
 167 }

 169 void end_credits::key_press_callback(const SDL_Keycode key)
 170 {
 171         if(key == SDLK_UP && scroll_speed_ < 400) {
 172                 scroll_speed_ <<= 1;
 173         }
 174 
 175         if(key == SDLK_DOWN && scroll_speed_ > 50) {
 176                 scroll_speed_ >>= 1;
 177         }
 178 }
