  58 static void parse_about_tags(const config& cfg, std::stringstream& ss)
  59 {
  60         for(const auto& about : cfg.child_range("about")) {
  61                 if(!about.has_child("entry")) {
  62                         continue;
  63                 }
  64 
  65                 ss << "\n" << "<span size='x-large'>" << about["title"] << "</span>" << "\n";
  66 
  67                 for(const auto& entry : about.child_range("entry")) {
  68                         ss << entry["name"] << "\n";
  69                 }
  70         }
  71 }

  73 void end_credits::pre_show(window& window)
  74 {
  75         timer_id_ = add_timer(10, std::bind(&end_credits::timer_callback, this), true);
  76 
  77         // Delay a little before beginning the scrolling
  78         last_scroll_ = SDL_GetTicks() + 3000;
  79 
  80         connect_signal_pre_key_press(window, std::bind(&end_credits::key_press_callback, this, _5));
  81 
  82         std::stringstream ss;
  83         std::stringstream focus_ss;
  84 
  85         const config& credits_config = about::get_about_config();
  86 
  87         // First, parse all the toplevel [about] tags
  88         parse_about_tags(credits_config, ss);
  89 
  90         // Next, parse all the grouped [about] tags (usually by campaign)
  91         for(const auto& group : credits_config.child_range("credits_group")) {
  92                 std::stringstream& group_stream = (group["id"] == focus_on_) ? focus_ss : ss;
  93 
  94                 group_stream << "\n" << "<span size='xx-large'>" << group["title"] << "</span>" << "\n";
  95 
  96                 parse_about_tags(group, group_stream);
  97         }
  98 
  99         // If a section is focused, move it to the top
 100         if(!focus_ss.str().empty()) {
 101                 focus_ss << ss.rdbuf();
 102         }
 103 
 104         // Get the appropriate background images
 105         backgrounds_ = about::get_background_images(focus_on_);
 106 
 107         if(backgrounds_.empty()) {
 108                 backgrounds_.push_back(game_config::images::game_title_background);
 109         }
 110 
 111         // TODO: implement showing all available images as the credits scroll
 112         window.get_canvas(0).set_variable("background_image", wfl::variant(backgrounds_[0]));
 113 
 114         text_widget_ = find_widget<scroll_label>(&window, "text", false, true);
 115 
 116         text_widget_->set_use_markup(true);
 117         text_widget_->set_label((focus_ss.str().empty() ? ss : focus_ss).str());
 118 
 119         // HACK: always hide the scrollbar, even if it's needed.
 120         // This should probably be implemented as a scrollbar mode.
 121         // Also, for some reason hiding the whole grid doesn't work, and the elements need to be hidden manually
 122         if(grid* v_grid = dynamic_cast<grid*>(text_widget_->find("_vertical_scrollbar_grid", false))) {
 123                 find_widget<scrollbar_base>(v_grid, "_vertical_scrollbar", false).set_visible(widget::visibility::hidden);
 124 
 125                 // TODO: enable again if e24336afeb7 is reverted.
 126                 //find_widget<repeating_button>(v_grid, "_half_page_up", false).set_visible(widget::visibility::hidden);
 127                 //find_widget<repeating_button>(v_grid, "_half_page_down", false).set_visible(widget::visibility::hidden);
 128         }
 129 }

 131 void end_credits::timer_callback()
 132 {
 133         uint32_t now = SDL_GetTicks();
 134         if(last_scroll_ > now) {
 135                 return;
 136         }
 137 
 138         uint32_t missed_time = now - last_scroll_;
 139 
 140         unsigned int cur_pos = text_widget_->get_vertical_scrollbar_item_position();
 141 
 142         // Calculate how far the text should have scrolled by now
 143         // The division by 1000 is to convert milliseconds to seconds.
 144         unsigned int needed_dist = missed_time * scroll_speed_ / 1000;
 145 
 146         text_widget_->set_vertical_scrollbar_item_position(cur_pos + needed_dist);
 147 
 148         last_scroll_ = now;
 149 
 150         if(text_widget_->vertical_scrollbar_at_end()) {
 151                 remove_timer(timer_id_);
 152         }
 153 }

 155 void end_credits::key_press_callback(const SDL_Keycode key)
 156 {
 157         if(key == SDLK_UP && scroll_speed_ < 400) {
 158                 scroll_speed_ <<= 1;
 159         }
 160 
 161         if(key == SDLK_DOWN && scroll_speed_ > 50) {
 162                 scroll_speed_ >>= 1;
 163         }
 164 }
