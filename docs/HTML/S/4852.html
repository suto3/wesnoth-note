 207     # When an enemy leader of a new side arrives on the map, it could find the player's
 208     # units sitting on its castle tiles so it can't recruit, or waiting to assassinate it.
 209     # This would make it too easy for the player. In this case, the AI will get a castle-full
 210     # of units without having to recruit them. Many of them will be more powerful than what
 211     # it would have been able to recruit in the first place, making this strategy backfire
 212     # on the player.
 213 #define ADD_EXTRA_STARTING_UNITS SIDE
 214     # close_units will be units from side 1 that can reach the castle this turn (before
 215     # the leader can recruit), prevent the leader from getting to its keep, or attack
 216     # the leader.
 217     [clear_variable]
 218         name=close_units
 219     [/clear_variable]

 221     [store_locations]
 222         [filter]
 223             side={SIDE}
 224             canrecruit=yes
 225         [/filter]
 226         variable=leader_location
 227     [/store_locations]
 228     [store_locations]
 229         # If the leader managed to make it to its keep, these will be castle tiles:
 230         x=$leader_location.x
 231         y=$leader_location.y
 232         radius=1
 233         variable=adjacent_to_leader
 234     [/store_locations]

 242         [/filter]
 243         variable=side1_units
 244     [/store_unit]

 246     # For each unit on side-1, store its reachable locations. Then, check each hex
 247     # adjacent to the enemy leader to see if one matches one of those locations.
 248     [for]
 249         array=side1_units
 250         variable=u
 251         [do]
 252             [store_reachable_locations]
 253                 [filter]
 254                     id=$side1_units[$u].id
 255                 [/filter]
 256                 moves=max
 257                 variable=reachable_locations
 258             [/store_reachable_locations]
 259             [for]
 260                 array=adjacent_to_leader
 261                 variable=a
 262                 [do]
 263                     [if]
 264                         [have_location]
 265                             # If an enemy unit can reach this hex...
 266                             x=$adjacent_to_leader[$a].x
 267                             y=$adjacent_to_leader[$a].y
 268                             find_in=reachable_locations
 269                         [/have_location]
 270                         [then]
 271                             # ...then add it to the close_units list:
 272                             [set_variables]
 273                                 name=close_units
 274                                 mode=append
 275                                 to_variable=side1_units[$u]
 276                             [/set_variables]
 277                             [break]
 278                                 # So we don't store the unit again.
 279                             [/break]
 280                         [/then]
 281                     [/if]
 282                 [/do]
 283             [/for]
 284         [/do]
 285     [/for]

 393             # For each side 1 unit in range, add a unit from the list:
 394             [if]

 398                 [/variable]

 401                 [/then]
 402             [/if]

 404             # If the close unit is an ancient lich waiting to assassinate the leader, it
 405             # is going to find it protected by an equally-powerful unit. This doesn't
 406             # completely prevent an assassination, but it makes the loss of the lich
 407             # almost certain if the attempt fails (which is the most likely outcome).
 408             [if]
 409                 [have_unit]
 410                     id=$close_units[$i].id
 411                     type=Ancient Lich, Ancient Lich SotA
 412                 [/have_unit]
 413                 [then]
 414                     # Using "5" as the index chooses a level-4 unit:
 415                     {EXTRA_STARTING_UNIT {SIDE} 5}
 416                 [/then]
 417             [/if]
 418         [/do]

 421     # If the AI got just a couple units, they would go off to die one at a time. We
 422     # finish filling up the keep so that doesn't happen. Also, this will make leader
 423     # assassination by backstabbing shadows or nightgaunts very difficult. We will
 424     # do nothing if side 1 wasn't nearby, and no extra units were added. In that case,
 425     # the leader will just recruit as normal.
 426     [store_unit]
 427         [filter]
 428             side={SIDE}
 429         [/filter]
 430         variable=extra_starting_units
 431     [/store_unit]
 432     [if]
 433         [variable]
 434             name=extra_starting_units.length
 435             greater_than=1  # *One* unit is the leader.
 436         [/variable]
 437         [then]
 438             [repeat]
 439                 times="$(7 - $extra_starting_units.length)"
 440                 [do]
 441                     {EXTRA_STARTING_UNIT {SIDE} 6}
 442                 [/do]
 443             [/repeat]
 444         [/then]
 445         [else]
 446             [set_variable]
 447                 name=extra_units_message
 448                 value="empty"
 449             [/set_variable]
 450         [/else]
 451     [/if]

 454     # an extra, powerful unit to kick it off.
 455     [store_starting_location]
 456         side={SIDE}
 457         variable=keep_location
 458     [/store_starting_location]
 459     [if]
 460         [have_unit]
 461             side=1
 462             x=$keep_location.x
 463             y=$keep_location.y
 464         [/have_unit]
 465         [then]
 466             # Using "5" chooses a level-4 unit:
 467             {EXTRA_STARTING_UNIT {SIDE} 5}
 468         [/then]
 469     [/if]
