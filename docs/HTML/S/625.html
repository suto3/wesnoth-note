 165 void label::signal_handler_left_button_click(const event::ui_event /* event */, bool & handled)
 166 {
 167         DBG_GUI_E << "label click" << std::endl;
 168 
 169         if (!get_link_aware()) {
 170                 return; // without marking event as "handled".
 171         }
 172 
 173         if (!desktop::open_object_is_supported()) {
 174                 show_message(get_window()->video(), "", _("Opening links is not supported, contact your packager"), dialogs::message::auto_close);
 175                 handled = true;
 176                 return;
 177         }
 178 
 179 
 180         point mouse = get_mouse_position();
 181 
 182         mouse.x -= get_x();
 183         mouse.y -= get_y();
 184 
 185         std::string link = get_label_link(mouse);
 186 
 187         if (link.length() == 0) {
 188                 return ; // without marking event as "handled"
 189         }
 190 
 191         DBG_GUI_E << "Clicked Link:\"" << link << "\"\n";
 192 
 193         const int res = show_message(get_window()->video(), _("Confirm"), _("Do you want to open this link?") + std::string("\n\n") + link, dialogs::message::yes_no_buttons);
 194         if(res == gui2::window::OK) {
 195                 desktop::open_object(link);
 196         }
 197 
 198         handled = true;
 199 }

 201 void label::signal_handler_right_button_click(const event::ui_event /* event */, bool & handled)
 202 {
 203         DBG_GUI_E << "label right click" << std::endl;
 204 
 205         if (!get_link_aware()) {
 206                 return ; // without marking event as "handled".
 207         }
 208 
 209         point mouse = get_mouse_position();
 210 
 211         mouse.x -= get_x();
 212         mouse.y -= get_y();
 213 
 214         std::string link = get_label_link(mouse);
 215 
 216         if (link.length() == 0) {
 217                 return ; // without marking event as "handled"
 218         }
 219 
 220         DBG_GUI_E << "Right Clicked Link:\"" << link << "\"\n";
 221 
 222         desktop::clipboard::copy_to_clipboard(link, false);
 223 
 224         show_message(get_window()->video(), "", _("Copied link!"), dialogs::message::auto_close);
 225 
 226         handled = true;
 227 }

 248  * The reason is that labels are often used as visual indication of the state

 250  *
 251  * Note: The above is outdated, if "link_aware" is enabled then there is interaction.

 255  * * state_enabled, the label is enabled.
 256  * * state_disabled, the label is disabled.
 257  * @begin{parent}{name="gui/"}

 259  * @begin{tag}{name="resolution"}{min=0}{max=-1}{super="generic/widget_definition/resolution"}
 260  * @begin{table}{config}
 261  *     link_aware & f_bool & false & Whether the label is link aware. This means
 262  *                                     it is rendered with links highlighted,
 263  *                                     and responds to click events on those
 264  *                                     links. $
 265  *     link_color & string & #ffff00 & The color to render links with. This
 266  *                                     string will be used verbatim in pango
 267  *                                     markup for each link. $
 268  * @end{table}
 269  * @begin{tag}{name="state_enabled"}{min=0}{max=1}{super="generic/state"}
 270  * @end{tag}{name="state_enabled"}
 271  * @begin{tag}{name="state_disabled"}{min=0}{max=1}{super="generic/state"}
