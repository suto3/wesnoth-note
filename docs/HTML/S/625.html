  52 point spacer::calculate_best_size() const
  53 {
  54         const wfl::map_formula_callable& size = get_screen_size_variables();
  55 
  56         unsigned width = width_(size);
  57         unsigned height = height_(size);
  58 
  59         point best_size;
  60 
  61         if(width || height) {
  62                 best_size = point(width, height);
  63         }
  64 
  65         if(best_size != point()) {
  66                 return best_size;
  67         }
  68 
  69         return styled_widget::calculate_best_size();
  70 }
