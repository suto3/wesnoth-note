  36 {
  37         if(show && get_selected_item_count() == 0) {

  78                         const unsigned item_count = get_item_count();
  79                         const unsigned visible_index = get_ordered_index(index);

  81                                 // Is the last item deselected?

  83                                         // Select the second last.
  84                                         do_select_item(get_item_at_ordered(visible_index - 1));

  86                                         // Select the next item.

 145 void horizontal_list::place(const point& origin, const point& size)
 146 {
 147         /*
 148          * - Set every item to its best size.
 149          * - The origin gets increased with the width of the last item.
 150          * - No item should be higher as the size.
 151          * - In the end the origin should be the sum or the origin and the wanted
 152          *   width.
 153          */
 154 
 155         point current_origin = origin;
 156         for(size_t i = 0; i < get_item_count(); ++i) {
 157 
 158                 if(!get_item_shown(get_item_at_ordered(i))) {
 159 
 160                         continue;
 161                 }
 162 
 163                 grid& grid = item_ordered(i);
 164                 point best_size = grid.get_best_size();
 165                 assert(best_size.y <= size.y);
 166                 // FIXME should we look at grow factors???
 167                 best_size.y = size.y;
 168 
 169                 grid.place(current_origin, best_size);
 170 
 171                 current_origin.x += best_size.x;
 172         }
 173 
 174         if(current_origin.x != origin.x + size.x) {
 175                 ERR_GUI_L << "Failed to fit horizontal list to requested rect; expected right edge was " << origin.x + size.x;
 176                 ERR_GUI_L << ", actual right edge was " << current_origin.x;
 177                 ERR_GUI_L << " (left edge is " << origin.x << ")\n";
 178         }
 179 }

 568 point table::calculate_best_size() const
 569 {
 570         // The best size is the one that minimizes aspect ratio of the enclosing rect
 571         // We first calculate the best size of each item,
 572         // then find the number of rows that minimizes the aspect ratio
 573         // We try a number of columns from 1 up to sqrt(visible_items) + 2
 574         size_t n_items = get_item_count();
 575         size_t max_cols = sqrt(n_items) + 2;
 576         std::vector<point> item_sizes;
 577         for(size_t i = 0; i < n_items; i++) {
 578                 if(get_item_shown(i)) {
 579                         item_sizes.push_back(item(i).get_best_size());
 580                 }
 581         }
 582         if(item_sizes.empty()) {
 583                 return point();
 584         }
 585         std::vector<point> best_sizes(1);
 586         best_sizes[0] = std::accumulate(item_sizes.begin(), item_sizes.end(), point(), [](point a, point b) {
 587                 return point(std::max(a.x, b.x), a.y + b.y);
 588         });
 589         int max_xtra = std::min_element(item_sizes.begin(), item_sizes.end(), [](point a, point b) {
 590                 return a.x < b.x;
 591         })->x / 2;
 592         for(size_t cells_in_1st_row = 2; cells_in_1st_row <= max_cols; cells_in_1st_row++) {
 593                 int row_min_width = std::accumulate(item_sizes.begin(), item_sizes.begin() + cells_in_1st_row, 0, [](int a, point b) {
 594                         return a + b.x;
 595                 });
 596                 int row_max_width = row_min_width + max_xtra;
 597                 int row = 0;
 598                 point row_size, total_size;
 599                 for(size_t n = 0; n < item_sizes.size(); n++) {
 600                         if(row_size.x + item_sizes[n].x > row_max_width) {
 601                                 // Start new row
 602                                 row++;
 603                                 total_size.y += row_size.y;
 604                                 if(total_size.x < row_size.x) {
 605                                         total_size.x = row_size.x;
 606                                 }
 607                                 row_size = point();
 608                         }
 609                         row_size.x += item_sizes[n].x;
 610                         if(row_size.y < item_sizes[n].y) {
 611                                 row_size.y = item_sizes[n].y;
 612                         }
 613                 }
 614                 total_size.y += row_size.y;
 615                 if(total_size.x < row_size.x) {
 616                         total_size.x = row_size.x;
 617                 }
 618                 best_sizes.push_back(total_size);
 619         }
 620 
 621         return *std::min_element(best_sizes.begin(), best_sizes.end(), [](point p1, point p2) {
 622                 return std::max<double>(p1.x, p1.y) / std::min<double>(p1.x, p1.y) < std::max<double>(p2.x, p2.y) / std::min<double>(p2.x, p2.y);
 623         });
 624 }

 626 void table::place(const point& origin, const point& size)
 627 {
 628         /*
 629          * - Set every item to its best size.
 630          * - The origin gets increased with the height of the last item.
 631          * - No item should be wider as the size.
 632          * - In the end the origin should be the sum of the origin and the wanted
 633          *   height.
 634          */
 635 
 636         // TODO: Make sure all cells in a row are the same height
 637         point current_origin = origin;
 638         int row_height = 0;
 639         for(size_t i = 0; i < get_item_count(); ++i) {
 640 
 641                 if(!get_item_shown(get_item_at_ordered(i))) {
 642 
 643                         continue;
 644                 }
 645 
 646                 grid& grid = item_ordered(i);
 647                 point best_size = grid.get_best_size();
 648                 // FIXME should we look at grow factors???
 649 
 650                 if(current_origin.x + best_size.x > origin.x + size.x) {
 651                         current_origin.x = origin.x;
 652                         current_origin.y += row_height;
 653                         row_height = 0;
 654                 }
 655 
 656                 grid.place(current_origin, best_size);
 657 
 658                 current_origin.x += best_size.x;
 659                 if(best_size.y > row_height) {
 660                         row_height = best_size.y;
 661                 }
 662         }
 663 
 664         // TODO: If size is wider than best_size, the matrix will take too much vertical space.
 665         // This block is supposed to correct for that, but doesn't work properly.
 666         // To be more specific, it requires invalidating the layout to take effect.
 667         if(current_origin.y + row_height != origin.y + size.y) {
 668                 point better_size = size;
 669                 better_size.y -= current_origin.y + row_height - origin.y;
 670                 set_layout_size(better_size);
 671         }
 672 }

 674 void table::set_origin(const point& origin)
 675 {
 676         point current_origin = origin;
 677         size_t row_height = 0;
 678         for(size_t i = 0; i < get_item_count(); ++i) {
 679 
 680                 if(!get_item_shown(get_item_at_ordered(i))) {
 681 
 682                         continue;
 683                 }
 684 
 685                 grid& grid = item_ordered(i);
 686                 if(current_origin.x + grid.get_width() > origin.x + get_width()) {
 687                         current_origin.x = origin.x;
 688                         current_origin.y += row_height;
 689                         row_height = 0;
 690                 }
 691

 929 point independent::calculate_best_size() const
 930 {
 931         /*
 932          * The best size is the combination of the greatest width and greatest
 933          * height.
 934          */
 935         point result(0, 0);
 936         for(size_t i = 0; i < get_item_count(); ++i) {
 937 
 938                 const grid& grid = item(i);
 939 
 940                 const point best_size = grid.get_best_size();
 941 
 942                 if(best_size.x > result.x) {
 943                         result.x = best_size.x;
 944                 }
 945 
 946                 if(best_size.y > result.y) {
 947                         result.y = best_size.y;
 948                 }
 949         }
 950 
 951         return result;

1064                           const std::map<std::string /* widget id */, string_map>& data,

1066 {
1067         for(unsigned row = 0; row < g->get_rows(); ++row) {
1068                 for(unsigned col = 0; col < g->get_cols(); ++col) {
1069                         widget* widget = g->get_widget(row, col);
1070                         assert(widget);
1071 
1072                         grid* child_grid = dynamic_cast<grid*>(widget);
1073                         toggle_button* btn = dynamic_cast<toggle_button*>(widget);
1074                         toggle_panel* panel = dynamic_cast<toggle_panel*>(widget);
1075 
1076                         if(btn) {
1077                                 btn->set_callback_state_change(callback);
1078                                 std::map<std::string, string_map>::const_iterator itor
1079                                                 = data.find(btn->id());
1080 
1081                                 if(itor == data.end()) {
1082                                         itor = data.find("");
1083                                 }
1084                                 if(itor != data.end()) {
1085                                         btn->set_members(itor->second);
1086                                 }
1087                         } else if(panel) {
1088                                 panel->set_callback_state_change(callback);
1089                                 panel->set_child_members(data);
1090                         } else if(child_grid) {
1091                                 init(child_grid, data, callback);
1092                         } else {
1093                                 FAIL("Only toggle buttons and panels are allowed as the cells of a list definition.");
1094                         }
1095                 }
1096         }
