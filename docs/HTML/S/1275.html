  16  * @file
  17  * Execute another executable:
  18  * - wait for the process to quit, then pass through the error code
  19  * - terminate it after a timeout of "t" microseconds, then return 2
  20  */

  22 #if _WIN32_WINNT < 0x0602 //Windows version before Win8
  23         #define WIN32_LEAN_AND_MEAN
  24         #include <windows.h>

  28 #include <stdlib.h>

  31 int main(int argc, char* argv[]) {
  32 
  33         if (argc != 3) {
  34                 std::cout << "WindowsTimeout: Error USAGE1: " << argv[0] << " \"command [--options]\" t";
  35                 return -1;
  36         }
  37 
  38         PROCESS_INFORMATION pi;
  39         STARTUPINFO info = { sizeof(info) };
  40         HANDLE hStdErr = GetStdHandle(STD_ERROR_HANDLE);
  41         info.hStdError = hStdErr;
  42 
  43         if (!CreateProcess(NULL, argv[1], NULL, NULL, TRUE, 0, NULL, NULL, &info, &pi))
  44                 return -1;
  45 
  46         DWORD dwExitCode;
  47         if (::WaitForSingleObject(pi.hProcess, atoi(argv[2])) == WAIT_OBJECT_0) {
  48                 GetExitCodeProcess(pi.hProcess, &dwExitCode);
  49         }
  50         else {
  51                 TerminateProcess(pi.hProcess, 2);
  52                 dwExitCode = 2;
  53         }
  54         CloseHandle(pi.hProcess);
  55         CloseHandle(pi.hThread);
  56         CloseHandle(hStdErr);
  57         return dwExitCode;
  58 }
