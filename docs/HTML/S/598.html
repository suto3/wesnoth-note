 166 void scrollbar_container::request_reduce_height(const unsigned maximum_height)
 167 {
 168         DBG_GUI_L << LOG_HEADER << " requested height " << maximum_height << ".\n";
 169         /*
 170          * First ask the content to reduce it's height. This seems to work for now,
 171          * but maybe some sizing hints will be required later.
 172          */
 173         /** @todo Evaluate whether sizing hints are required. */
 174         assert(content_grid_);
 175         const unsigned offset
 176                         = horizontal_scrollbar_grid_
 177                                           && horizontal_scrollbar_grid_->get_visible()
 178                                                  != widget::visibility::invisible
 179                                           ? horizontal_scrollbar_grid_->get_best_size().y
 180                                           : 0;
 181 
 182         content_grid_->request_reduce_height(maximum_height - offset);
 183 
 184         // Did we manage to achieve the wanted size?
 185         point size = get_best_size();
 186         if(static_cast<unsigned>(size.y) <= maximum_height) {
 187                 DBG_GUI_L << LOG_HEADER << " child honored request, height " << size.y
 188                                   << ".\n";
 189                 return;
 190         }
 191 
 192         if(vertical_scrollbar_mode_ == ALWAYS_INVISIBLE) {
 193                 DBG_GUI_L << LOG_HEADER << " request failed due to scrollbar mode.\n";
 194                 return;
 195         }
 196 
 197         assert(vertical_scrollbar_grid_);
 198         const bool resized = vertical_scrollbar_grid_->get_visible()
 199                                                  == widget::visibility::invisible;
 200 
 201         // Always set the bar visible, is a nop is already visible.
 202         vertical_scrollbar_grid_->set_visible(widget::visibility::visible);
 203 
 204         const point scrollbar_size = vertical_scrollbar_grid_->get_best_size();
 205 
 206         // If showing the scrollbar increased the height, hide and abort.
 207         if(resized && scrollbar_size.y > size.y) {
 208                 vertical_scrollbar_grid_->set_visible(widget::visibility::invisible);
 209                 DBG_GUI_L << LOG_HEADER << " request failed, showing the scrollbar"
 210                                   << " increased the height to " << scrollbar_size.y << ".\n";
 211                 return;
 212         }
 213 
 214         if(maximum_height > static_cast<unsigned>(scrollbar_size.y)) {
 215                 size.y = maximum_height;
 216         } else {
 217                 size.y = scrollbar_size.y;
 218         }
 219 
 220         // FIXME adjust for the step size of the scrollbar
 221 
 222         set_layout_size(size);
 223         DBG_GUI_L << LOG_HEADER << " resize resulted in " << size.y << ".\n";
 224 
 225         if(resized) {
 226                 DBG_GUI_L << LOG_HEADER
 227                                   << " resize modified the width, throw notification.\n";
 228 
 229                 throw layout_exception_width_modified();
 230         }
 231 }

 233 void scrollbar_container::request_reduce_width(const unsigned maximum_width)
 234 {
 235         DBG_GUI_L << LOG_HEADER << " requested width " << maximum_width << ".\n";
 236 
 237         // First ask our content, it might be able to wrap which looks better as
 238         // a scrollbar.
 239         assert(content_grid_);
 240         const unsigned offset
 241                         = vertical_scrollbar_grid_
 242                                           && vertical_scrollbar_grid_->get_visible()
 243                                                  != widget::visibility::invisible
 244                                           ? vertical_scrollbar_grid_->get_best_size().x
 245                                           : 0;
 246 
 247         content_grid_->request_reduce_width(maximum_width - offset);
 248 
 249         // Did we manage to achieve the wanted size?
 250         point size = get_best_size();
 251         if(static_cast<unsigned>(size.x) <= maximum_width) {
 252                 DBG_GUI_L << LOG_HEADER << " child honored request, width " << size.x
 253                                   << ".\n";
 254                 return;
 255         }
 256 
 257         if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE) {
 258                 DBG_GUI_L << LOG_HEADER << " request failed due to scrollbar mode.\n";
 259                 return;
 260         }
 261 
 262         // Always set the bar visible, is a nop when it's already visible.
 263         assert(horizontal_scrollbar_grid_);
 264         horizontal_scrollbar_grid_->set_visible(widget::visibility::visible);
 265         size = get_best_size();
 266 
 267         point scrollbar_size = horizontal_scrollbar_grid_->get_best_size();
 268 
 269         /*
 270          * If the vertical bar is not invisible it's size needs to be added to the
 271          * minimum size.
 272          */
 273         if(vertical_scrollbar_grid_->get_visible()
 274            != widget::visibility::invisible) {
 275 
 276                 scrollbar_size.x += vertical_scrollbar_grid_->get_best_size().x;
 277         }
 278 
 279         // If showing the scrollbar increased the width, hide and abort.
 280         if(horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN && scrollbar_size.x
 281                                                                                                                            > size.x) {
 282 
 283                 horizontal_scrollbar_grid_->set_visible(widget::visibility::invisible);
 284                 DBG_GUI_L << LOG_HEADER << " request failed, showing the scrollbar"
 285                                   << " increased the width to " << scrollbar_size.x << ".\n";
 286                 return;
 287         }
 288 
 289         if(maximum_width > static_cast<unsigned>(scrollbar_size.x)) {
 290                 size.x = maximum_width;
 291         } else {
 292                 size.x = scrollbar_size.x;
 293         }
 294 
 295         // FIXME adjust for the step size of the scrollbar
 296 
 297         set_layout_size(size);
 298         DBG_GUI_L << LOG_HEADER << " resize resulted in " << size.x << ".\n";
 299 }

 306 point scrollbar_container::calculate_best_size() const
 307 {
 308         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 309 
 310         /***** get vertical scrollbar size *****/
 311         const point vertical_scrollbar
 312                         = vertical_scrollbar_grid_->get_visible()
 313                                           == widget::visibility::invisible
 314                                           ? point()
 315                                           : vertical_scrollbar_grid_->get_best_size();
 316 
 317         /***** get horizontal scrollbar size *****/
 318         const point horizontal_scrollbar
 319                         = horizontal_scrollbar_grid_->get_visible()
 320                                           == widget::visibility::invisible
 321                                           ? point()
 322                                           : horizontal_scrollbar_grid_->get_best_size();
 323 
 324         /***** get content size *****/
 325         assert(content_grid_);
 326         const point content = content_grid_->get_best_size();
 327 
 328         point result(
 329                         vertical_scrollbar.x + std::max(horizontal_scrollbar.x, content.x),
 330                         horizontal_scrollbar.y + std::max(vertical_scrollbar.y, content.y));
 331 
 332         //
 333         // Workaround for bug #24780. This should probably be moved somewhere more specific to
 334         // the listbox, but for now it suffices.
 335         //
 336         if(const grid* header = find_widget<const grid>(&get_grid(), "_header_grid", false, false)) {
 337                 result.y += header->get_best_size().y;
 338         }
 339 
 340         if(const grid* footer = find_widget<const grid>(&get_grid(), "_footer_grid", false, false)) {
 341                 result.y += footer->get_best_size().y;
 342         }
 343 
 344         DBG_GUI_L << LOG_HEADER << " vertical_scrollbar " << vertical_scrollbar
 345                           << " horizontal_scrollbar " << horizontal_scrollbar << " content "
 346                           << content << " result " << result << ".\n";
 347 
 348         return result;
 349 }

 351 static void
 352 set_scrollbar_mode(grid* scrollbar_grid,
 353                                    scrollbar_base* scrollbar,
 354                                    scrollbar_container::scrollbar_mode& scrollbar_mode,
 355                                    const unsigned items,
 356                                    const unsigned visible_items,
 357                                    grid* content_grid)
 358 {
 359         assert(scrollbar_grid && scrollbar);
 360 
 361         if(scrollbar_mode == scrollbar_container::ALWAYS_INVISIBLE) {
 362                 scrollbar_grid->set_visible(widget::visibility::invisible);
 363                 return;
 364         }
 365 
 366         scrollbar->set_item_count(items);
 367         scrollbar->set_item_position(0);
 368         scrollbar->set_visible_items(visible_items);
 369 
 370         if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE) {
 371 
 372                 const bool scrollbar_needed = items > visible_items;
 373 
 374                 scrollbar_grid->set_visible(scrollbar_needed
 375                                                                                         ? widget::visibility::visible
 376                                                                                         : widget::visibility::hidden);
 377         } else if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE_FIRST_RUN) {
 378                 if(items <= visible_items && content_grid != nullptr &&
 379                         scrollbar_grid->get_visible() == widget::visibility::visible) {
 380                         scrollbar_grid->set_visible(widget::visibility::invisible);
 381                         // Give newly freed space to the items.
 382                         content_grid->layout_initialize(false);
 383                 }
 384         }
 385 }
 386 static bool is_inserted_before(unsigned insertion_pos, unsigned old_item_count, unsigned old_position, unsigned visible_items)
 387 {
 388         if(old_position == 0)
 389                 return false;
 390         else if(old_position + visible_items >= old_item_count)
 391                 return true;
 392         else if(insertion_pos <= old_position)
 393                 return true;
 394         else
 395                 return false;
 396 }

 398 static void
 399 adjust_scrollbar_mode(grid* scrollbar_grid,
 400                                    scrollbar_base* scrollbar,
 401                                    scrollbar_container::scrollbar_mode& scrollbar_mode,
 402                                    const unsigned items_before,
 403                                    const unsigned items_after,
 404                                    const int insertion_pos,
 405                                    const unsigned visible_items)
 406 {
 407         assert(scrollbar_grid && scrollbar);
 408         if(items_before != scrollbar->get_item_count()) {
 409                 return set_scrollbar_mode(scrollbar_grid, scrollbar, scrollbar_mode, items_after, visible_items, nullptr);
 410         }
 411         //TODO: does this also work well in case the items were removed?
 412         const unsigned previous_item_position = scrollbar->get_item_position();
 413         //Casts insertion_pos to an unsigned so negative values are interpreted as 'at end'
 414         const bool inserted_before_visible_area = is_inserted_before(static_cast<unsigned>(insertion_pos), items_before, previous_item_position, visible_items);
 415 
 416         if(scrollbar_mode == scrollbar_container::ALWAYS_INVISIBLE) {
 417                 scrollbar_grid->set_visible(widget::visibility::invisible);
 418                 return;
 419         }
 420 
 421         scrollbar->set_item_count(items_after);
 422         scrollbar->set_item_position(inserted_before_visible_area ? previous_item_position + items_after - items_before : previous_item_position);
 423         //scrollbar->set_item_position(0);
 424         scrollbar->set_visible_items(visible_items);
 425 
 426         if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE) {
 427 
 428                 const bool scrollbar_needed = items_after > visible_items;
 429 
 430                 scrollbar_grid->set_visible(scrollbar_needed
 431                                                                                         ? widget::visibility::visible
 432                                                                                         : widget::visibility::hidden);
 433         }
 434 }

 436 void scrollbar_container::place(const point& origin, const point& size)
 437 {
 438         // Inherited.
 439         container_base::place(origin, size);
 440 
 441         // Set content size
 442         assert(content_ && content_grid_);
 443 
 444         const point content_origin = content_->get_origin();
 445 
 446         const point best_size = content_grid_->get_best_size();
 447         const point content_size(content_->get_width(), content_->get_height());
 448 
 449         const point content_grid_size(std::max(best_size.x, content_size.x),
 450                                                                    std::max(best_size.y, content_size.y));
 451 
 452         set_content_size(content_origin, content_grid_size);
 453 
 454         // Set vertical scrollbar
 455         set_scrollbar_mode(vertical_scrollbar_grid_,
 456                                            vertical_scrollbar_,
 457                                            vertical_scrollbar_mode_,
 458                                            content_grid_->get_height(),
 459                                            content_->get_height(),
 460                                            content_grid_);
 461 
 462         // Set horizontal scrollbar
 463         set_scrollbar_mode(horizontal_scrollbar_grid_,
 464                                            horizontal_scrollbar_,
 465                                            horizontal_scrollbar_mode_,
 466                                            content_grid_->get_width(),
 467                                            content_->get_width(),
 468                                            content_grid_);
 469 
 470         // Update the buttons.
 471         set_scrollbar_button_status();
 472 
 473         // Now set the visible part of the content.
 474         content_visible_area_ = content_->get_rectangle();
 475         content_grid_->set_visible_rectangle(content_visible_area_);
 476 }

 551 bool scrollbar_container::content_resize_request(const bool force_sizing)
 552 {
 553         /**
 554          * @todo Try to handle AUTO_VISIBLE_FIRST_RUN here as well.
 555          *
 556          * Handling it here makes the code a bit more complex but allows to not
 557          * reserve space for scrollbars, which will look nicer in the MP lobby.
 558          * But the extra complexity is no 1.8 material.
 559          */
 560 
 561         assert(content_ && content_grid_);
 562 
 563         point best_size = content_grid_->recalculate_best_size();
 564         point size = content_->get_size();
 565 
 566         DBG_GUI_L << LOG_HEADER << " wanted size " << best_size
 567                           << " available size " << size << ".\n";
 568 
 569         if(size == point()) {
 570                 DBG_GUI_L << LOG_HEADER << " initial setup not done, bailing out.\n";
 571                 return false;
 572         }
 573 
 574         if(best_size.x <= size.x && best_size.y <= size.y) {
 575                 const point content_size = content_grid_->get_size();
 576                 if(content_size.x > size.x || content_size.y > size.y) {
 577                         DBG_GUI_L << LOG_HEADER << " will fit, only needs a resize.\n";
 578                         goto resize;
 579                 }
 580                 if(force_sizing) {
 581                         DBG_GUI_L << LOG_HEADER << " fits, but resize forced.\n";
 582                         goto resize;
 583                 }
 584                 DBG_GUI_L << LOG_HEADER << " fits, nothing to do.\n";
 585                 return true;
 586         }
 587 
 588         if(best_size.x > size.x) {
 589                 DBG_GUI_L << LOG_HEADER << " content too wide.\n";
 590                 if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE
 591                    || (horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN
 592                            && horizontal_scrollbar_grid_->get_visible()
 593                                   == widget::visibility::invisible)) {
 594 
 595                         DBG_GUI_L << LOG_HEADER
 596                                           << " can't use horizontal scrollbar, request placement.\n";
 597                         event::message message;
 598                         fire(event::REQUEST_PLACEMENT, *this, message);
 599                         return false;
 600                 }
 601         }
 602 
 603         if(best_size.y > size.y) {
 604                 DBG_GUI_L << LOG_HEADER << " content too high.\n";
 605                 if(vertical_scrollbar_mode_ == ALWAYS_INVISIBLE
 606                    || (vertical_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN
 607                            && vertical_scrollbar_grid_->get_visible()
 608                                   == widget::visibility::invisible)) {
 609 
 610                         DBG_GUI_L << LOG_HEADER
 611                                           << " can't use vertical scrollbar, request placement.\n";
 612                         event::message message;
 613                         fire(event::REQUEST_PLACEMENT, *this, message);
 614                         return false;
 615                 }
 616         }
 617 
 618 resize:
 619         DBG_GUI_L << LOG_HEADER << " handle resizing.\n";
 620         place(get_origin(), get_size());
 621         return true;
 622 }

 624 bool scrollbar_container::content_resize_request(const int width_modification,
 625                                                                                                   const int height_modification,
 626                                                                                                   const int width_modification_pos,
 627                                                                                                   const int height_modification_pos)
 628 {
 629         DBG_GUI_L << LOG_HEADER << " wanted width modification "
 630                           << width_modification << " wanted height modification "
 631                           << height_modification << ".\n";
 632 
 633         if(get_size() == point()) {
 634                 DBG_GUI_L << LOG_HEADER << " initial setup not done, bailing out.\n";
 635                 return false;
 636         }
 637 
 638         window* window = get_window();
 639         assert(window);
 640         if(window->get_need_layout()) {
 641                 DBG_GUI_L << LOG_HEADER
 642                                   << " window already needs a layout phase, bailing out.\n";
 643                 return false;
 644         }
 645 
 646         assert(content_ && content_grid_);
 647 
 648         const bool result = content_resize_width(width_modification, width_modification_pos)
 649                                                 && content_resize_height(height_modification, height_modification_pos);
 650         scrollbar_moved();
 651         if(result) {
 652                 /*
 653                  * The subroutines set the new size of the scrollbar but don't
 654                  * update the button status.
 655                  */
 656                 set_scrollbar_button_status();
 657         }
 658 
 659         DBG_GUI_L << LOG_HEADER << " result " << result << ".\n";
 660         return result;
 661 }

 663 bool scrollbar_container::content_resize_width(const int width_modification, const int width_modification_pos)
 664 {
 665         if(width_modification == 0) {
 666                 return true;
 667         }
 668 
 669         const int new_width = content_grid_->get_width() + width_modification;
 670         DBG_GUI_L << LOG_HEADER << " current width " << content_grid_->get_width()
 671                           << " wanted width " << new_width;
 672 
 673         if(new_width < 0) {
 674                 return false;
 675         }
 676 
 677         if(static_cast<unsigned>(new_width) <= content_->get_width()) {
 678                 DBG_GUI_L << " width fits in container, test height.\n";
 679                 adjust_scrollbar_mode(horizontal_scrollbar_grid_,
 680                                                    horizontal_scrollbar_,
 681                                                    horizontal_scrollbar_mode_,
 682                                                    content_grid_->get_width(),
 683                                                    content_grid_->get_width() + width_modification,
 684                                                    width_modification_pos,
 685                                                    content_->get_width());
 686                 return true;
 687         }
 688 
 689         assert(horizontal_scrollbar_ && horizontal_scrollbar_grid_);
 690         if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE
 691            || (horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN
 692                    && horizontal_scrollbar_grid_->get_visible()
 693                           == widget::visibility::invisible)) {
 694 
 695                 DBG_GUI_L << " can't use horizontal scrollbar, ask window.\n";
 696                 window* window = get_window();
 697                 assert(window);
 698                 window->invalidate_layout();
 699                 return false;
 700         }
 701 
 702         DBG_GUI_L << " use the horizontal scrollbar, test height.\n";
 703         adjust_scrollbar_mode(horizontal_scrollbar_grid_,
 704                                            horizontal_scrollbar_,
 705                                            horizontal_scrollbar_mode_,
 706                                            content_grid_->get_width(),
 707                                            content_grid_->get_width() + width_modification,
 708                                            width_modification_pos,
 709                                            content_->get_width());
 710 
 711         return true;
 712 }

 766 void scrollbar_container::finalize_setup()
 767 {
 768         /***** Setup vertical scrollbar *****/
 769 
 770         vertical_scrollbar_grid_
 771                         = find_widget<grid>(this, "_vertical_scrollbar_grid", false, true);
 772 
 773         vertical_scrollbar_ = find_widget<scrollbar_base>(
 774                         vertical_scrollbar_grid_, "_vertical_scrollbar", false, true);
 775 
 776         connect_signal_notify_modified(
 777                         *vertical_scrollbar_,
 778                         std::bind(&scrollbar_container::vertical_scrollbar_moved, this));
 779 
 780         /***** Setup horizontal scrollbar *****/
 781         horizontal_scrollbar_grid_ = find_widget<grid>(
 782                         this, "_horizontal_scrollbar_grid", false, true);
 783 
 784         horizontal_scrollbar_ = find_widget<scrollbar_base>(
 785                         horizontal_scrollbar_grid_, "_horizontal_scrollbar", false, true);
 786 
 787         connect_signal_notify_modified(
 788                         *horizontal_scrollbar_,
 789                         std::bind(&scrollbar_container::horizontal_scrollbar_moved,
 790                                                 this));
 791 
 792         /***** Setup the scrollbar buttons *****/
 793         for(const auto & item : scroll_lookup())
 794         {
 795 
 796                 // Vertical.
 797                 clickable_item* button = find_widget<clickable_item>(
 798                                 vertical_scrollbar_grid_, item.first, false, false);
 799 
 800                 if(button) {
 801                         button->connect_click_handler(std::bind(
 802                                         &scrollbar_container::scroll_vertical_scrollbar,
 803                                         this,
 804                                         item.second));
 805                 }
 806 
 807                 // Horizontal.
 808                 button = find_widget<clickable_item>(
 809                                 horizontal_scrollbar_grid_, item.first, false, false);
 810 
 811                 if(button) {
 812                         button->connect_click_handler(std::bind(
 813                                         &scrollbar_container::scroll_horizontal_scrollbar,
 814                                         this,
 815                                         item.second));
 816                 }
 817         }
 818 
 819         /***** Setup the content *****/
 820         content_ = new spacer();
 821         content_->set_definition("default");
 822 
 823         content_grid_ = dynamic_cast<grid*>(
 824                         get_grid().swap_child("_content_grid", content_, true).release());
 825         assert(content_grid_);
 826 
 827         content_grid_->set_parent(this);
 828 
 829         /***** Let our subclasses initialize themselves. *****/
 830         finalize_subclass();
 831 }

 888 void scrollbar_container::show_content_rect(const SDL_Rect& rect)
 889 {
 890         assert(content_);
 891         assert(horizontal_scrollbar_ && vertical_scrollbar_);
 892 
 893         // Set the bottom right location first if it doesn't fit the top left
 894         // will look good. First calculate the left and top position depending on
 895         // the current position.
 896 
 897         const int left_position = horizontal_scrollbar_->get_item_position()
 898                                                           + (rect.x - content_->get_x());
 899         const int top_position = vertical_scrollbar_->get_item_position()
 900                                                          + (rect.y - content_->get_y());
 901 
 902         // bottom.
 903         const int wanted_bottom = rect.y + rect.h;
 904         const int current_bottom = content_->get_y() + content_->get_height();
 905         int distance = wanted_bottom - current_bottom;
 906         if(distance > 0) {
 907                 vertical_scrollbar_->set_item_position(
 908                                 vertical_scrollbar_->get_item_position() + distance);
 909         }
 910 
 911         // right.
 912         const int wanted_right = rect.x + rect.w;
 913         const int current_right = content_->get_x() + content_->get_width();
 914         distance = wanted_right - current_right;
 915         if(distance > 0) {
 916                 horizontal_scrollbar_->set_item_position(
 917                                 horizontal_scrollbar_->get_item_position() + distance);
 918         }
 919 
 920         // top.
 921         if(top_position
 922            < static_cast<int>(vertical_scrollbar_->get_item_position())) {
 923 
 924                 vertical_scrollbar_->set_item_position(top_position);
 925         }
 926 
 927         // left.
 928         if(left_position
 929            < static_cast<int>(horizontal_scrollbar_->get_item_position())) {
 930 
 931                 horizontal_scrollbar_->set_item_position(left_position);
 932         }
 933 
 934         // Update.
 935         scrollbar_moved();
 936 }

1142 void scrollbar_container::scrollbar_moved()
1143 {
1144         // Init.
1145         assert(content_ && content_grid_);
1146         assert(vertical_scrollbar_ && horizontal_scrollbar_);
1147 
1148         /*** Update the content location. ***/
1149         const int x_offset = horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE
1150                                                                  ? 0
1151                                                                  : horizontal_scrollbar_->get_item_position()
1152                                                                    * horizontal_scrollbar_->get_step_size();
1153 
1154         const int y_offset = vertical_scrollbar_mode_ == ALWAYS_INVISIBLE
1155                                                                  ? 0
1156                                                                  : vertical_scrollbar_->get_item_position()
1157                                                                    * vertical_scrollbar_->get_step_size();
1158 
1159         const point content_origin {content_->get_x() - x_offset, content_->get_y() - y_offset};
1160 
1161         content_grid_->set_origin(content_origin);
1162         content_grid_->set_visible_rectangle(content_visible_area_);
1163         content_grid_->set_is_dirty(true);
1164 
1165         // Update scrollbar.
1166         set_scrollbar_button_status();
1167 }
