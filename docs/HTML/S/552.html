  23 void swap_grid(grid* g, grid* content_grid, widget* widget, const std::string& id)
  24 {
  25         assert(content_grid);
  26         assert(widget);
  27 
  28         // Make sure the new child has same id.
  29         widget->set_id(id);
  30 
  31         // Get the container containing the wanted widget.
  32         grid* parent_grid = nullptr;
  33         if(g) {
  34                 parent_grid = find_widget<grid>(g, id, false, false);
  35         }
  36 
  37         if(!parent_grid) {
  38                 parent_grid = find_widget<grid>(content_grid, id, true, false);
  39         }
  40 
  41         parent_grid = dynamic_cast<grid*>(parent_grid->parent());
  42         assert(parent_grid);
  43 
  44         // Replace the child.
  45         auto old = parent_grid->swap_child(id, widget, false);
  46         assert(old);
  47 }
  48 }
