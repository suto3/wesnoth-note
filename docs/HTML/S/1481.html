  24 /**
  25  * Toggles the Wesnoth nature on the selected project
  26  */
  27 public class ToggleWesnothNatureAction extends ObjectActionDelegate
  28 {
  29     @Override
  30     @SuppressWarnings( "rawtypes" )
  31     public void run( IAction action )
  32     {
  33         if( structuredSelection_ == null ) {
  34             return;
  35         }
  36         for( Iterator it = structuredSelection_.iterator( ); it.hasNext( ); ) {
  37             Object element = it.next( );
  38             IProject project = null;
  39             if( element instanceof IProject ) {
  40                 project = ( IProject ) element;
  41             }
  42             else if( element instanceof IAdaptable ) {
  43                 project = ( IProject ) ( ( IAdaptable ) element )
  44                     .getAdapter( IProject.class );
  45             }
  46             if( project != null ) {
  47                 toggleNature( project );
  48             }
  49         }
  50     }
  51 
  52     /**
  53      * Toggles sample nature on a project
  54      * 
  55      * @param project
  56      *        to have sample nature added or removed
  57      */
  58     public void toggleNature( IProject project )
  59     {
  60         try {
  61             IProjectDescription description = project.getDescription( );
  62             String[] natures = description.getNatureIds( );
  63 
  64             for( int i = 0; i < natures.length; ++i ) {
  65                 if( WesnothProjectNature.ID_NATURE.equals( natures[i] ) ) {
  66                     // Remove the nature
  67                     String[] newNatures = new String[natures.length - 1];
  68                     System.arraycopy( natures, 0, newNatures, 0, i );
  69                     System.arraycopy( natures, i + 1, newNatures, i,
  70                         natures.length - i - 1 );
  71                     description.setNatureIds( newNatures );
  72                     project.setDescription( description,
  73                         new NullProgressMonitor( ) );
  74                     project.refreshLocal( IResource.DEPTH_INFINITE,
  75                         new NullProgressMonitor( ) );
  76                     return;
  77                 }
  78             }
  79 
  80             // Add the natures
  81             String[] newNatures = new String[natures.length + 1];
  82             System.arraycopy( natures, 0, newNatures, 0, natures.length );
  83             newNatures[natures.length] = WesnothProjectNature.ID_NATURE;
  84             description.setNatureIds( newNatures );
  85             project.setDescription( description, new NullProgressMonitor( ) );
  86             project.refreshLocal( IResource.DEPTH_INFINITE,
  87                 new NullProgressMonitor( ) );
  88         } catch( CoreException e ) {
  89             Logger.getInstance( ).logException( e );
  90         }
  91     }
  92 
  93 }
