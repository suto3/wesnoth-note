  50 static std::string do_interpolation(const std::string &str, const variable_set& set)
  51 {
  52         std::string res = str;
  53         // This needs to be able to store negative numbers to check for the while's condition
  54         // (which is only false when the previous '$' was at index 0)
  55         int rfind_dollars_sign_from = res.size();
  56         while(rfind_dollars_sign_from >= 0) {
  57                 // Going in a backwards order allows nested variable-retrieval, e.g. in arrays.
  58                 // For example, "I am $creatures[$i].user_description!"
  59                 const std::string::size_type var_begin_loc = res.rfind('$', rfind_dollars_sign_from);
  60 
  61                 // If there are no '$' left then we're done.
  62                 if(var_begin_loc == std::string::npos) {
  63                         break;
  64                 }
  65 
  66                 // For the next iteration of the loop, search for more '$'
  67                 // (not from the same place because sometimes the '$' is not replaced)
  68                 rfind_dollars_sign_from = int(var_begin_loc) - 1;
  69 
  70 
  71                 const std::string::iterator var_begin = res.begin() + var_begin_loc;
  72 
  73                 // The '$' is not part of the variable name.
  74                 const std::string::iterator var_name_begin = var_begin + 1;
  75                 std::string::iterator var_end = var_name_begin;
  76 
  77                 if(var_name_begin == res.end()) {
  78                         // Any '$' at the end of a string is just a '$'
  79                         continue;
  80                 } else if(*var_name_begin == '(') {
  81                         // The $( ... ) syntax invokes a formula
  82                         int paren_nesting_level = 0;
  83                         bool in_string = false,
  84                                 in_comment = false;
  85                         do {
  86                                 switch(*var_end) {
  87                                 case '(':
  88                                         if(!in_string && !in_comment) {
  89                                                 ++paren_nesting_level;
  90                                         }
  91                                         break;
  92                                 case ')':
  93                                         if(!in_string && !in_comment) {
  94                                                 --paren_nesting_level;
  95                                         }
  96                                         break;
  97                                 case '#':
  98                                         if(!in_string) {
  99                                                 in_comment = !in_comment;
 100                                         }
 101                                         break;
 102                                 case '\'':
 103                                         if(!in_comment) {
 104                                                 in_string = !in_string;
 105                                         }
 106                                         break;
 107                                 // TODO: support escape sequences when/if they are allowed in FormulaAI strings
 108                                 }
 109                         } while(++var_end != res.end() && paren_nesting_level > 0);
 110                         if(paren_nesting_level > 0) {
 111                                 ERR_NG << "Formula in WML string cannot be evaluated due to "
 112                                         << "a missing closing parenthesis:\n\t--> \""
 113                                         << std::string(var_begin, var_end) << "\"\n";
 114                                 res.replace(var_begin, var_end, "");
 115                                 continue;
 116                         }
 117                         try {
 118                                 const wfl::formula form(std::string(var_begin+2, var_end-1));
 119                                 res.replace(var_begin, var_end, form.evaluate().string_cast());
 120                         } catch(wfl::formula_error& e) {
 121                                 ERR_NG << "Formula in WML string cannot be evaluated due to "
 122                                         << e.type << "\n\t--> \""
 123                                         << e.formula << "\"\n";
 124                                 res.replace(var_begin, var_end, "");
 125                         }
 126                         continue;
 127                 }
 128 
 129                 // Find the maximum extent of the variable name (it may be shortened later).
 130                 for(int bracket_nesting_level = 0; var_end != res.end(); ++var_end) {
 131                         const char c = *var_end;
 132                         if(c == '[') {
 133                                 ++bracket_nesting_level;
 134                         }
 135                         else if(c == ']') {
 136                                 if(--bracket_nesting_level < 0) {
 137                                         break;
 138                                 }
 139                         }
 140                         // isascii() breaks on mingw with -std=c++0x
 141                         else if (!(((c) & ~0x7f) == 0)/*isascii(c)*/ || (!isalnum(c) && c != '.' && c != '_')) {
 142                                 break;
 143                         }
 144                 }
 145 
 146                 // Two dots in a row cannot be part of a valid variable name.
 147                 // That matters for random=, e.g. $x..$y
 148                 var_end = std::adjacent_find(var_name_begin, var_end, two_dots);
 149                 /// the default value is specified after ''?'
 150                 const std::string::iterator default_start = var_end < res.end() && *var_end == '?' ? var_end + 1 : res.end();
 151 
 152                 // If the last character is '.', then it can't be a sub-variable.
 153                 // It's probably meant to be a period instead. Don't include it.
 154                 // Would need to do it repetitively if there are multiple '.'s at the end,
 155                 // but don't actually need to do so because the previous check for adjacent '.'s would catch that.
 156                 // For example, "My score is $score." or "My score is $score..."
 157                 if(*(var_end-1) == '.'
 158                 // However, "$array[$i]" by itself does not name a variable,
 159                 // so if "$array[$i]." is encountered, then best to include the '.',
 160                 // so that it more closely follows the syntax of a variable (if only to get rid of all of it).
 161                 // (If it's the script writer's error, they'll have to fix it in either case.)
 162                 // For example in "$array[$i].$field_name", if field_name does not exist as a variable,
 163                 // then the result of the expansion should be "", not "." (which it would be if this exception did not exist).
 164                 && *(var_end-2) != ']') {
 165                         --var_end;
 166                 }
 167 
 168                 const std::string var_name(var_name_begin, var_end);
 169                 if(default_start == res.end()) {
 170                         if(var_end != res.end() && *var_end == '|') {
 171                                 // It's been used to end this variable name; now it has no more effect.
 172                                 // This can allow use of things like "$$composite_var_name|.x"
 173                                 // (Yes, that's a WML 'pointer' of sorts. They are sometimes useful.)
 174                                 // If there should still be a '|' there afterwards to affect other variable names (unlikely),
 175                                 // just put another '|' there, one matching each '$', e.g. "$$var_containing_var_name||blah"
 176                                 ++var_end;
 177                         }
 178 
 179 
 180                         if (var_name == "") {
 181                                 // Allow for a way to have $s in a string.
 182                                 // $| will be replaced by $.
 183                                 res.replace(var_begin, var_end, "$");
 184                         }
 185                         else {
 186                                 // The variable is replaced with its value.
 187                                 res.replace(var_begin, var_end,
 188                                         set.get_variable_const(var_name));
 189                         }
 190                 }
 191                 else {
 192                         var_end = default_start;
 193                         while(var_end != res.end() && *var_end != '|') {
 194                                 ++var_end;
 195                         }
 196                         const std::string::iterator default_end = var_end;
 197                         const config::attribute_value& val = set.get_variable_const(var_name);
 198                         if(var_end == res.end()) {
 199                                 res.replace(var_begin, default_start - 1, val);
 200                         }
 201                         else if(!val.empty()) {
 202                                 res.replace(var_begin, var_end + 1, val);
 203                         }
 204                         else {
 205                                 res.replace(var_begin, var_end + 1, std::string(default_start, default_end));
 206                         }
 207                 }
 208         }
 209 
 210         return res;
 211 }

 232 t_string interpolate_variables_into_tstring(const t_string &tstr, const variable_set& variables)
 233 {
 234         if(!tstr.str().empty()) {
 235                 std::string interp = utils::interpolate_variables_into_string(tstr.str(), variables);
 236                 if(tstr.str() != interp) {
 237                         return t_string(interp);
 238                 }
 239         }
 240         return tstr;
 241 }
