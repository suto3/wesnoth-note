  55 bool synced_context::run(const std::string& commandname, const config& data, bool use_undo, bool show, synced_command::error_handler_function error_handler)
  56 {
  57         DBG_REPLAY << "run_in_synced_context:" << commandname << "\n";
  58 
  59         assert(use_undo || (!resources::undo_stack->can_redo() && !resources::undo_stack->can_undo()));
  60         /*
  61                 use this after resources::recorder->add_synced_command
  62                 because set_scontext_synced sets the checkup to the last added command
  63         */
  64         set_scontext_synced sync;
  65         synced_command::map::iterator it = synced_command::registry().find(commandname);
  66         if(it == synced_command::registry().end())
  67         {
  68                 error_handler("commandname [" +commandname +"] not found", true);
  69         }
  70         else
  71         {
  72                 bool success = it->second(data, use_undo, show, error_handler);
  73                 if(!success)
  74                 {
  75                         return false;
  76                 }
  77         }
  78 
  79         // this might also be a good point to call resources::controller->check_victory();
  80         // because before for example if someone kills all units during a moveto event they don't loose.
  81         resources::controller->check_victory();
  82         sync.do_final_checkup();
  83         DBG_REPLAY << "run_in_synced_context end\n";
  84         return true;
  85 }

  87 bool synced_context::run_and_store(const std::string& commandname, const config& data, bool use_undo, bool show, synced_command::error_handler_function error_handler)
  88 {
  89         if(resources::controller->is_replay())
  90         {
  91                 ERR_REPLAY << "ignored attempt to invoke a synced command during replay\n";
  92                 return false;
  93         }
  94 
  95         assert(resources::recorder->at_end());
  96         resources::recorder->add_synced_command(commandname, data);
  97         bool success = run(commandname, data, use_undo, show, error_handler);
  98         if(!success)
  99         {
 100                 resources::recorder->undo();
 101         }
 102         return success;
 103 }

 115 bool synced_context::run_in_synced_context_if_not_already(const std::string& commandname,const config& data, bool use_undo, bool show, synced_command::error_handler_function error_handler)
 116 {
 117         switch(synced_context::get_synced_state())
 118         {
 119         case(synced_context::UNSYNCED):
 120         {
 121                 return run_and_throw(commandname, data, use_undo, show, error_handler);
 122         }
 123         case(synced_context::LOCAL_CHOICE):
 124                 ERR_REPLAY << "trying to execute action while being in a local_choice" << std::endl;
 125                 //we reject it because such actions usually change the gamestate badly which is not intented during a local_choice.
 126                 //Also we cannot invoke synced commands here, becasue multiple clients might run local choices
 127                 //simultaniously so it could result in invoking different synced commands simultaniously.
 128                 return false;
 129         case(synced_context::SYNCED):
 130         {
 131                 synced_command::map::iterator it = synced_command::registry().find(commandname);
 132                 if(it == synced_command::registry().end())
 133                 {
 134                         error_handler("commandname [" +commandname +"] not found", true);
 135                         return false;
 136                 }
 137                 else
 138                 {
 139                         return it->second(data, /*use_undo*/ false, show, error_handler);
 140                 }
 141         }
 142         default:
 143                 assert(false && "found unknown synced_context::synced_state");
 144                 return false;
 145         }
 146 }

 224 bool synced_context::can_undo()
 225 {

 280 config synced_context::ask_server_choice(const server_choice& sch)
 281 {
 282         if (!is_synced()) {
 283                 ERR_REPLAY << "Trying to ask the server for a '" << sch.name() << "' choice in a unsynced context, doing the choice locally. This can casue OOS.\n";
 284                 return sch.local_choice();
 285         }
 286         set_is_simultaneously();
 287         resources::controller->increase_server_request_number();
 288         const bool is_mp_game = resources::controller->is_networked_mp();
 289         bool did_require = false;
 290 
 291         DBG_REPLAY << "ask_server for random_seed\n";
 292         /*
 293                 as soon as random or similar is involved, undoing is impossible.
 294         */
 295         resources::undo_stack->clear();
 296         /*
 297                 there might be speak or similar commands in the replay before the user input.
 298         */
 299         while(true) {
 300 
 301                 do_replay_handle();
 302                 bool is_replay_end = resources::recorder->at_end();
 303 
 304                 if (is_replay_end && !is_mp_game)
 305                 {
 306                         /* The decision is ours, and it will be inserted
 307                         into the replay. */
 308                         DBG_REPLAY << "MP synchronization: local server choice\n";
 309                         leave_synced_context sync;
 310                         config cfg = sch.local_choice();
 311                         //-1 for "server" todo: change that.
 312                         resources::recorder->user_input(sch.name(), cfg, -1);
 313                         return cfg;
 314 
 315                 }
 316                 else if(is_replay_end && is_mp_game)
 317                 {
 318                         DBG_REPLAY << "MP synchronization: remote server choice\n";
 319 
 320                         //here we can get into the situation that the decision has already been made but not received yet.
 321                         synced_context::pull_remote_user_input();
 322                         //FIXME: we shoudl call play_controller::play_silce or the application will freeze while waiting for a remote choice.
 323                         resources::controller->play_slice();
 324                         /*
 325                                 we don't want to send multiple "require_random" to the server.
 326                         */
 327                         if(!did_require)
 328                         {
 329                                 sch.send_request();
 330                                 did_require = true;
 331                         }
 332 
 333                         SDL_Delay(10);
 334                         continue;
 335 
 336                 }
 337                 else if (!is_replay_end)
 338                 {
 339                         /* The decision has already been made, and must
 340                         be extracted from the replay. */
 341                         DBG_REPLAY << "MP synchronization: replay server choice\n";
 342                         do_replay_handle();
 343                         const config *action = resources::recorder->get_next_action();
 344                         if (!action)
 345                         {
 346                                 replay::process_error("[" + std::string(sch.name()) + "] expected but none found\n");
 347                                 resources::recorder->revert_action();
 348                                 return sch.local_choice();
 349                         }
 350                         if (!action->has_child(sch.name()))
 351                         {
 352                                 replay::process_error("[" + std::string(sch.name()) + "] expected but none found, found instead:\n " + action->debug() + "\n");
 353 
 354                                 resources::recorder->revert_action();
 355                                 return sch.local_choice();
 356                         }
 357                         if((*action)["from_side"].str() != "server" || (*action)["side_invalid"].to_bool(false) )
 358                         {
 359                                 //we can proceed without getting OOS in this case, but allowing this would allow a "player chan choose their attack results in mp" cheat
 360                                 replay::process_error("wrong from_side or side_invalid this could mean someone wants to cheat\n");
 361                         }
 362                         return action->child(sch.name());
 363                 }
 364         }
 365 }

 436 void set_scontext_synced::do_final_checkup(bool dont_throw)
 437 {
 438         assert(!did_final_checkup_);
 439         std::stringstream msg;
 440         config co;
 441         config cn {
 442                 "random_calls", new_rng_->get_random_calls(),
 443                 "next_unit_id", resources::gameboard->unit_id_manager().get_save_id() + 1,
 444         };
 445         if(checkup_instance->local_checkup(cn, co))
 446         {
 447                 return;
 448         }
 449         if(co["random_calls"].empty())
 450         {
 451                 msg << "cannot find random_calls check in replay" << std::endl;
 452         }
 453         else if(co["random_calls"] != cn["random_calls"])
 454         {
 455                 msg << "We called random " << new_rng_->get_random_calls() << " times, but the original game called random " << co["random_calls"].to_int() << " times." << std::endl;
 456         }
 457         //Ignore empty next_unit_id to prevent false positives with older saves.
 458         if(!co["next_unit_id"].empty() && co["next_unit_id"] != cn["next_unit_id"])
 459         {
 460                 msg << "Our next unit id is " << cn["next_unit_id"].to_int() << " but during the original the next unit id was " << co["next_unit_id"].to_int() << std::endl;
 461         }
 462         if(!msg.str().empty())
 463         {
 464                 msg << co.debug() << std::endl;
 465                 if(dont_throw)
 466                 {
 467                         ERR_REPLAY << msg.str() << std::flush;
 468                 }
 469                 else
 470                 {
 471                         replay::process_error(msg.str());
 472                 }
 473         }
 474         did_final_checkup_ = true;
 475 }
