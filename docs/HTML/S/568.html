  65 /** Key type for the cache. */
  66 struct key_type
  67 {
  68         key_type(const int w, const int h, const std::string& map_data)
  69                 : w(w), h(h), map_data(map_data)
  70         {
  71         }
  72 
  73         /** Width of the image. */
  74         const int w;
  75 
  76         /** Height of the image. */
  77         const int h;
  78 
  79         /** The data used to generate the image. */
  80         const std::string map_data;
  81 };
  82 
  83 static bool operator<(const key_type& lhs, const key_type& rhs)
  84 {
  85         return lhs.w < rhs.w
  86                    || (lhs.w == rhs.w
  87                            && (lhs.h < rhs.h
  88                                    || (lhs.h == rhs.h && lhs.map_data < rhs.map_data)));
  89 }

  91 /** Value type for the cache. */
  92 struct value_type
  93 {
  94         value_type(const surface& surf) : surf(surf), age(1)
  95         {
  96         }
  97 
  98         /** The cached image. */
  99         const surface surf;
 100 
 101         /**
 102          * The age of the image.
 103          *
 104          * Every time an image is used its age is increased by one. Once the cache
 105          * is full 25% of the cache is emptied. This is done by halving the age of
 106          * the items in the cache and then erase the 25% with the lowest age. If
 107          * items have the same age their order is unspecified.
 108          */
 109         unsigned age;
 110 };
 111 
 112 /**
 113  * Maximum number of items in the cache (multiple of 4).
 114  *
 115  * No testing on the optimal number is done, just seems a nice number.
 116  */
 117 static const size_t cache_max_size = 100;
 118 
 119 /**
 120  * The terrain used to create the cache.
 121  *
 122  * If another terrain config is used the cache needs to be cleared, this
 123  * normally doesn't happen a lot so the clearing of the cache is rather
 124  * unusual.
 125  */
 126 static const ::config* terrain = nullptr;

 128 /** The cache. */
 129 typedef std::map<key_type, value_type> tcache;
 130 static tcache cache;
 131 
 132 static bool compare(const std::pair<unsigned, tcache::iterator>& lhs,
 133                                         const std::pair<unsigned, tcache::iterator>& rhs)
 134 {
 135         return lhs.first < rhs.first;
 136 }

 138 static void shrink_cache()
 139 {
 140 #ifdef DEBUG_MINIMAP_CACHE
 141         std::cerr << "\nShrink cache from " << cache.size();
 142 #else
 143         DBG_GUI_D << "Shrinking the minimap cache.\n";
 144 #endif
 145 
 146         std::vector<std::pair<unsigned, tcache::iterator> > items;
 147         for(tcache::iterator itor = cache.begin(); itor != cache.end(); ++itor) {
 148 
 149                 itor->second.age /= 2;
 150                 items.emplace_back(itor->second.age, itor);
 151         }
 152 
 153         std::partial_sort(items.begin(),
 154                                           items.begin() + cache_max_size / 4,
 155                                           items.end(),
 156                                           compare);
 157 
 158         for(std::vector<std::pair<unsigned, tcache::iterator> >::iterator vitor
 159                 = items.begin();
 160                 vitor < items.begin() + cache_max_size / 4;
 161                 ++vitor) {
 162 
 163                 cache.erase(vitor->second);
 164         }
 165 
 166 #ifdef DEBUG_MINIMAP_CACHE
 167         std::cerr << " to " << cache.size() << ".\n";
 168 #endif
 169 }

 176 const surface minimap::get_image(const int w, const int h) const
 177 {
 178         if(!terrain_) {
 179                 return nullptr;
 180         }
 181 
 182         if(terrain_ != terrain) {
 183 #ifdef DEBUG_MINIMAP_CACHE
 184                 std::cerr << "\nFlush cache.\n";
 185 #else
 186                 DBG_GUI_D << "Flushing the minimap cache.\n";
 187 #endif
 188                 terrain = terrain_;
 189                 cache.clear();
 190         }
 191 
 192         const key_type key(w, h, map_data_);
 193         tcache::iterator itor = cache.find(key);
 194 
 195         if(itor != cache.end()) {
 196 #ifdef DEBUG_MINIMAP_CACHE
 197                 std::cerr << '+';
 198 #endif
 199                 itor->second.age++;
 200                 return itor->second.surf;
 201         }
 202 
 203         if(cache.size() >= cache_max_size) {
 204                 shrink_cache();
 205         }
 206 
 207         try
 208         {
 209                 const gamemap map(std::make_shared<terrain_type_data>(*terrain_), map_data_);
 210                 const surface surf = image::getMinimap(w, h, map, nullptr);
 211                 cache.emplace(key, value_type(surf));
 212 #ifdef DEBUG_MINIMAP_CACHE
 213                 std::cerr << '-';
 214 #endif
 215                 return surf;
 216         }
 217         catch(incorrect_map_format_error& e)
 218         {
 219                 ERR_CF << "Error while loading the map: " << e.message << '\n';
 220 #ifdef DEBUG_MINIMAP_CACHE
 221                 std::cerr << 'X';
 222 #endif
 223         }
 224         return nullptr;
 225 }

 227 void minimap::impl_draw_background(surface& frame_buffer,
 228                                                                         int x_offset,
 229                                                                         int y_offset)
 230 {
 231         if(!terrain_)
 232                 return;
 233         assert(terrain_);
 234 
 235         DBG_GUI_D << LOG_HEADER << " size "
 236                           << calculate_blitting_rectangle(x_offset, y_offset) << ".\n";
 237 
 238         if(map_data_.empty()) {
 239                 return;
 240         }
 241 
 242         SDL_Rect rect = calculate_blitting_rectangle(x_offset, y_offset);
 243         assert(rect.w > 0 && rect.h > 0);
 244 
 245         const ::surface surf = get_image(rect.w, rect.h);
 246         if(surf) {
 247                 sdl_blit(surf, nullptr, frame_buffer, &rect);
 248         }
 249 }
