  72 /** Key type for the cache. */
  73 struct key_type
  74 {
  75         key_type(const int w, const int h, const std::string& map_data)
  76                 : w(w), h(h), map_data(map_data)
  77         {
  78         }
  79 
  80         /** Width of the image. */
  81         const int w;
  82 
  83         /** Height of the image. */
  84         const int h;
  85 
  86         /** The data used to generate the image. */
  87         const std::string map_data;
  88 };
  89 
  90 static bool operator<(const key_type& lhs, const key_type& rhs)
  91 {
  92         return lhs.w < rhs.w
  93                    || (lhs.w == rhs.w
  94                            && (lhs.h < rhs.h
  95                                    || (lhs.h == rhs.h && lhs.map_data < rhs.map_data)));
  96 }

  98 /** Value type for the cache. */
  99 struct value_type
 100 {
 101         value_type(const surface& surf) : surf(surf), age(1)
 102         {
 103         }
 104 
 105         /** The cached image. */
 106         const surface surf;
 107 
 108         /**
 109          * The age of the image.
 110          *
 111          * Every time an image is used its age is increased by one. Once the cache
 112          * is full 25% of the cache is emptied. This is done by halving the age of
 113          * the items in the cache and then erase the 25% with the lowest age. If
 114          * items have the same age their order is unspecified.
 115          */
 116         unsigned age;
 117 };
 118 
 119 /**
 120  * Maximum number of items in the cache (multiple of 4).
 121  *
 122  * No testing on the optimal number is done, just seems a nice number.
 123  */
 124 static const size_t cache_max_size = 100;
 125 
 126 /**
 127  * The terrain used to create the cache.
 128  *
 129  * If another terrain config is used the cache needs to be cleared, this
 130  * normally doesn't happen a lot so the clearing of the cache is rather
 131  * unusual.
 132  */
 133 static const ::config* terrain = nullptr;

 135 /** The cache. */
 136 typedef std::map<key_type, value_type> tcache;
 137 static tcache cache;
 138 
 139 static bool compare(const std::pair<unsigned, tcache::iterator>& lhs,
 140                                         const std::pair<unsigned, tcache::iterator>& rhs)
 141 {
 142         return lhs.first < rhs.first;
 143 }
 144 
 145 static void shrink_cache()
 146 {
 147 #ifdef DEBUG_MINIMAP_CACHE
 148         std::cerr << "\nShrink cache from " << cache.size();
 149 #else
 150         DBG_GUI_D << "Shrinking the minimap cache.\n";
 151 #endif
 152 
 153         std::vector<std::pair<unsigned, tcache::iterator> > items;
 154         for(tcache::iterator itor = cache.begin(); itor != cache.end(); ++itor) {
 155 
 156                 itor->second.age /= 2;
 157                 items.emplace_back(itor->second.age, itor);
 158         }
 159 
 160         std::partial_sort(items.begin(),
 161                                           items.begin() + cache_max_size / 4,
 162                                           items.end(),
 163                                           compare);
 164 
 165         for(std::vector<std::pair<unsigned, tcache::iterator> >::iterator vitor
 166                 = items.begin();
 167                 vitor < items.begin() + cache_max_size / 4;
 168                 ++vitor) {
 169 
 170                 cache.erase(vitor->second);
 171         }
 172 
 173 #ifdef DEBUG_MINIMAP_CACHE
 174         std::cerr << " to " << cache.size() << ".\n";
 175 #endif
 176 }

 183 const surface minimap::get_image(const int w, const int h) const
 184 {
 185         if(!terrain_) {
 186                 return nullptr;
 187         }
 188 
 189         if(terrain_ != terrain) {
 190 #ifdef DEBUG_MINIMAP_CACHE
 191                 std::cerr << "\nFlush cache.\n";
 192 #else
 193                 DBG_GUI_D << "Flushing the minimap cache.\n";
 194 #endif
 195                 terrain = terrain_;
 196                 cache.clear();
 197         }
 198 
 199         const key_type key(w, h, map_data_);
 200         tcache::iterator itor = cache.find(key);
 201 
 202         if(itor != cache.end()) {
 203 #ifdef DEBUG_MINIMAP_CACHE
 204                 std::cerr << '+';
 205 #endif
 206                 itor->second.age++;
 207                 return itor->second.surf;
 208         }
 209 
 210         if(cache.size() >= cache_max_size) {
 211                 shrink_cache();
 212         }
 213 
 214         try
 215         {
 216                 const gamemap map(std::make_shared<terrain_type_data>(*terrain_), map_data_);
 217                 const surface surf = image::getMinimap(w, h, map, nullptr, nullptr, true);
 218                 cache.emplace(key, value_type(surf));
 219 #ifdef DEBUG_MINIMAP_CACHE
 220                 std::cerr << '-';
 221 #endif
 222                 return surf;
 223         }
 224         catch(incorrect_map_format_error& e)
 225         {
 226                 ERR_CF << "Error while loading the map: " << e.message << '\n';
 227 #ifdef DEBUG_MINIMAP_CACHE
 228                 std::cerr << 'X';
 229 #endif
 230         }
 231         return nullptr;
 232 }

 234 void minimap::impl_draw_background(surface& frame_buffer,
 235                                                                         int x_offset,
 236                                                                         int y_offset)
 237 {
 238         if(!terrain_)
 239                 return;
 240         assert(terrain_);
 241 
 242         DBG_GUI_D << LOG_HEADER << " size "
 243                           << calculate_blitting_rectangle(x_offset, y_offset) << ".\n";
 244 
 245         if(map_data_.empty()) {
 246                 return;
 247         }
 248 
 249         SDL_Rect rect = calculate_blitting_rectangle(x_offset, y_offset);
 250         assert(rect.w > 0 && rect.h > 0);
 251 
 252         const ::surface surf = get_image(rect.w, rect.h);
 253         if(surf) {
 254                 sdl_blit(surf, nullptr, frame_buffer, &rect);
 255         }
 256 }
