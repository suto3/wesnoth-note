  31 bool display_context::would_be_discovered(const map_location & loc, int side_num, bool see_all)
  32 {
  33         map_location adjs[6];
  34         get_adjacent_tiles(loc,adjs);
  35 
  36         for (const map_location &u_loc : adjs)
  37         {
  38                 unit_map::const_iterator u_it = units().find(u_loc);
  39                 if (!u_it.valid()) {
  40                         continue;
  41                 }
  42                 const unit & u = *u_it;
  43                 if (get_team(side_num).is_enemy(u.side()) && !u.incapacitated()) {
  44                         // Enemy spotted in adjacent tiles, check if we can see him.
  45                         // Watch out to call invisible with see_all=true to avoid infinite recursive calls!
  46                         if(see_all) {
  47                                 return true;
  48                         } else if (!get_team(side_num).fogged(u_loc)
  49                         && !u.invisible(u_loc, *this, true)) {
  50                                 return true;
  51                         }
  52                 }
  53         }
  54         return false;
  55 }

  57 const unit * display_context::get_visible_unit(const map_location & loc, const team &current_team, bool see_all) const
  58 {
  59         if (!map().on_board(loc)) return nullptr;
  60         const unit_map::const_iterator u = units().find(loc);
  61         if (!u.valid() || !u->is_visible_to_team(current_team, *this, see_all)) {
  62                 return nullptr;
  63         }
  64         return &*u;
  65 }

  68  * Will return true iff the unit @a u has any possible moves
  69  * it can do (including attacking etc).

  72 bool display_context::unit_can_move(const unit &u) const
  73 {
  74         if(!u.attacks_left() && u.movement_left()==0)
  75                 return false;
  76 
  77         // Units with goto commands that have already done their gotos this turn
  78         // (i.e. don't have full movement left) should have red globes.
  79         if(u.has_moved() && u.has_goto()) {
  80                 return false;
  81         }
  82 
  83         const team &current_team = get_team(u.side());
  84 
  85         map_location locs[6];
  86         get_adjacent_tiles(u.get_location(), locs);
  87         for(int n = 0; n != 6; ++n) {
  88                 if (map().on_board(locs[n])) {
  89                         const unit_map::const_iterator i = units().find(locs[n]);
  90                         if (i.valid() && !i->incapacitated() &&
  91                             current_team.is_enemy(i->side())) {
  92                                 return true;
  93                         }
  94 
  95                         if (u.movement_cost(map()[locs[n]]) <= u.movement_left()) {
  96                                 return true;
  97                         }
  98                 }
  99         }
 100 
 101         return false;
 102 }

 105 /**
 106  * Given the location of a village, will return the 0-based index
 107  * of the team that currently owns it, and -1 if it is unowned.
 108  */
 109 int display_context::village_owner(const map_location& loc) const
 110 {
 111         const std::vector<team> & t = teams();
 112         for(size_t i = 0; i != t.size(); ++i) {
 113                 if(t[i].owns_village(loc))
 114                         return i;
 115         }
 116         return -1;
 117 }

 119 /**
 120  * Determine if we are an observer, by checking if every team is not locally controlled

 122 bool display_context::is_observer() const
 123 {
 124         for (const team &t : teams()) {

 161 team_data display_context::calculate_team_data(const team& tm) const
 162 {
 163         team_data res;
 164         res.units = side_units(tm.side());
 165         res.upkeep = side_upkeep(tm.side());
 166         res.villages = tm.villages().size();
 167         res.expenses = std::max<int>(0,res.upkeep - tm.support());
 168         res.net_income = tm.total_income() - res.expenses;
 169         res.gold = tm.gold();
 170         res.teamname = tm.user_team_name();
 171         return res;
 172 }
