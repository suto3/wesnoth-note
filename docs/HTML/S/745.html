 289 bool game_launcher::init_joystick()
 290 {
 291         if (!preferences::joystick_support_enabled())
 292                 return false;
 293 
 294         if(SDL_WasInit(SDL_INIT_JOYSTICK) == 0)
 295                 if(SDL_InitSubSystem(SDL_INIT_JOYSTICK) == -1)
 296                         return false;
 297 
 298         int joysticks = SDL_NumJoysticks();
 299         if (joysticks == 0) return false;
 300 
 301         SDL_JoystickEventState(SDL_ENABLE);
 302 
 303         bool joystick_found = false;
 304         for (int i = 0; i<joysticks; i++)  {
 305 
 306                 if (SDL_JoystickOpen(i))
 307                         joystick_found = true;
 308         }
 309         return joystick_found;
 310 }

 312 bool game_launcher::init_language()
 313 {
 314         if(!::load_language_list())
 315                 return false;
 316 
 317         language_def locale;
 318         if(cmdline_opts_.language) {
 319                 std::vector<language_def> langs = get_languages();
 320                 for(const language_def & def : langs) {
 321                         if(def.localename == *cmdline_opts_.language) {
 322                                 locale = def;
 323                                 break;
 324                         }
 325                 }
 326                 if(locale.localename.empty()) {
 327                         std::cerr << "Language symbol '" << *cmdline_opts_.language << "' not found.\n";
 328                         return false;
 329                 }
 330         } else {
 331                 locale = get_locale();
 332         }
 333         ::set_language(locale);
 334 
 335         return true;
 336 }

 367 bool game_launcher::init_lua_script()
 368 {
 369         bool error = false;
 370 
 371         std::cerr << "Checking lua scripts... ";
 372 
 373         if (cmdline_opts_.script_unsafe_mode) {
 374                 plugins_manager::get()->get_kernel_base()->load_package(); //load the "package" package, so that scripts can get what packages they want
 375         }
 376 
 377         // get the application lua kernel, load and execute script file, if script file is present
 378         if (cmdline_opts_.script_file)
 379         {
 380                 filesystem::scoped_istream sf = filesystem::istream_file(*cmdline_opts_.script_file);
 381 
 382                 if (!sf->fail()) {
 383                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 384                         jump_to_multiplayer_ = false;
 385                         jump_to_editor_ = false;
 386                         jump_to_campaign_.jump_ = false;
 387 
 388                         std::string full_script((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 389 
 390                         std::cerr << "\nRunning lua script: " << *cmdline_opts_.script_file << std::endl;
 391 
 392                         plugins_manager::get()->get_kernel_base()->run(full_script.c_str());
 393                 } else {
 394                         std::cerr << "Encountered failure when opening script '" << *cmdline_opts_.script_file << "'\n";
 395                         error = true;
 396                 }
 397         }
 398 
 399         if (cmdline_opts_.plugin_file)
 400         {
 401                 std::string filename = *cmdline_opts_.plugin_file;
 402 
 403                 std::cerr << "Loading a plugin file'" << filename << "'...\n";
 404 
 405                 filesystem::scoped_istream sf = filesystem::istream_file(filename);
 406 
 407                 try {
 408                         if (sf->fail()) {
 409                                 throw std::runtime_error("failed to open plugin file");
 410                         }
 411 
 412                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 413                         jump_to_multiplayer_ = false;
 414                         jump_to_editor_ = false;
 415                         jump_to_campaign_.jump_ = false;
 416 
 417                         std::string full_plugin((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 418 
 419                         plugins_manager & pm = *plugins_manager::get();
 420 
 421                         size_t i = pm.add_plugin(filename, full_plugin);
 422 
 423                         for (size_t j = 0 ; j < pm.size(); ++j) {
 424                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 425                         }
 426 
 427                         std::cerr << "Starting a plugin...\n";
 428                         pm.start_plugin(i);
 429 
 430                         for (size_t j = 0 ; j < pm.size(); ++j) {
 431                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 432                         }
 433 
 434                         plugins_context pc("init");
 435 
 436                         for (size_t repeat = 0; repeat < 5; ++repeat) {
 437                                 std::cerr << "Playing a slice...\n";
 438                                 pc.play_slice();
 439 
 440                                 for (size_t j = 0 ; j < pm.size(); ++j) {
 441                                         std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 442                                 }
 443                         }
 444 
 445                         return true;
 446                 } catch (std::exception & e) {
 447                         gui2::show_error_message(video(), std::string("When loading a plugin, error:\n") + e.what());
 448                         error = true;
 449                 }
 450         }
 451 
 452         if (!error) {
 453                 std::cerr << "ok\n";
 454         }
 455 
 456         return !error;
 457 }

 502 // Same as play_test except that we return the results of play_game.
 503 int game_launcher::unit_test()
 504 {
 505         static bool first_time_unit = true;
 506 
 507         if(!cmdline_opts_.unit_test) {
 508                 return 0;
 509         }
 510         if(!first_time_unit)
 511                 return 0;
 512 
 513         first_time_unit = false;
 514 
 515         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::TEST;
 516         state_.classification().campaign_define = "TEST";
 517         state_.set_carryover_sides_start(
 518                 config {"next_scenario", test_scenario_}
 519         );
 520 
 521 
 522         game_config_manager::get()->
 523                 load_game_config_for_game(state_.classification());
 524 
 525         try {
 526                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 527                 LEVEL_RESULT res = ccontroller.play_game();
 528                 if (!(res == LEVEL_RESULT::VICTORY) || lg::broke_strict()) {
 529                         return 1;
 530                 }
 531         } catch(wml_exception& e) {
 532                 std::cerr << "Caught WML Exception:" << e.dev_message << std::endl;
 533                 return 1;
 534         }
 535 
 536         savegame::clean_saves(state_.classification().label);
 537 
 538         if (cmdline_opts_.noreplaycheck)
 539                 return 0; //we passed, huzzah!
 540 
 541         savegame::replay_savegame save(state_, compression::NONE);
 542         save.save_game_automatic(video(), false, "unit_test_replay"); //false means don't check for overwrite
 543 
 544         load_data_.reset(new savegame::load_game_metadata{ "unit_test_replay" , "", true, true, false });
 545 
 546         if (!load_game()) {
 547                 std::cerr << "Failed to load the replay!" << std::endl;
 548                 return 3; //failed to load replay
 549         }
 550 
 551         try {
 552                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 553                 LEVEL_RESULT res = ccontroller.play_replay();
 554                 if (!(res == LEVEL_RESULT::VICTORY)) {
 555                         std::cerr << "Observed failure on replay" << std::endl;
 556                         return 4;
 557                 }
 558         } catch(wml_exception& e) {
 559                 std::cerr << "WML Exception while playing replay: " << e.dev_message << std::endl;
 560                 return 4; //failed with an error during the replay
 561         }
 562 
 563         return 0; //we passed, huzzah!
 564 }

 581 bool game_launcher::play_render_image_mode()
 582 {
 583         if(!cmdline_opts_.render_image) {
 584                 return true;
 585         }
 586 
 587         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 588         DBG_GENERAL << "Current campaign type: " << state_.classification().campaign_type << std::endl;
 589 
 590         try {
 591                 game_config_manager::get()->
 592                         load_game_config_for_game(state_.classification());
 593         } catch(config::error& e) {
 594                 std::cerr << "Error loading game config: " << e.what() << std::endl;
 595                 return false;
 596         }
 597 
 598         // A default output filename
 599         std::string outfile = "wesnoth_image.bmp";
 600 
 601         // If a output path was given as an argument, use that instead
 602         if (cmdline_opts_.render_image_dst) {
 603                 outfile = *cmdline_opts_.render_image_dst;
 604         }
 605 
 606         if (!image::save_image(*cmdline_opts_.render_image, outfile)) {
 607                 exit(1);
 608         }
 609 
 610         return false;
 611 }

 618 bool game_launcher::load_game()
 619 {
 620         assert(game_config_manager::get());
 621 
 622         DBG_GENERAL << "Current campaign type: " << state_.classification().campaign_type << std::endl;
 623 
 624         savegame::loadgame load(video(), game_config_manager::get()->game_config(), state_);
 625         if (load_data_) {
 626                 std::unique_ptr<savegame::load_game_metadata> load_data = std::move(load_data_);
 627                 load.data() = std::move(*load_data);
 628         }
 629 
 630         try {
 631                 if(!load.load_game()) {
 632                         return false;
 633                 }
 634 
 635                 try {
 636                         game_config_manager::get()->
 637                                 load_game_config_for_game(state_.classification());
 638                 } catch(config::error&) {
 639                         return false;
 640                 }
 641 
 642                 load.set_gamestate();
 643 
 644         } catch(config::error& e) {
 645                 if(e.message.empty()) {
 646                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 647                 }
 648                 else {
 649                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 650                 }
 651                 return false;
 652         } catch(wml_exception& e) {
 653                 e.show(video());
 654                 return false;
 655         } catch(filesystem::io_exception& e) {
 656                 if(e.message.empty()) {
 657                         gui2::show_error_message(video(), _("File I/O Error while reading the game"));
 658                 } else {
 659                         gui2::show_error_message(video(), _("File I/O Error while reading the game: '") + e.message + '\'');
 660                 }
 661                 return false;
 662         } catch(game::error& e) {
 663                 if(e.message.empty()) {
 664                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 665                 }
 666                 else {
 667                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 668                 }
 669                 return false;
 670         }
 671 
 672         play_replay_ = load.data().show_replay;
 673         LOG_CONFIG << "is middle game savefile: " << (state_.is_mid_game_save() ? "yes" : "no") << "\n";
 674         LOG_CONFIG << "show replay: " << (play_replay_ ? "yes" : "no") << "\n";
 675         // in case load.data().show_replay && !state_.is_mid_game_save()
 676         // there won't be any turns to replay, but the
 677         // user gets to watch the intro sequence again ...
 678 
 679         if(state_.is_mid_game_save() && load.data().show_replay)
 680         {
 681                 statistics::clear_current_scenario();
 682         }
 683 
 684         if(state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 685                 state_.unify_controllers();
 686         }
 687 
 688         if (load.data().cancel_orders) {
 689                 state_.cancel_orders();
 690         }
 691 
 692         return true;
 693 }

 772 void game_launcher::start_wesnothd()
 773 {
 774         const std::string wesnothd_program =
 775                 preferences::get_mp_server_program_name().empty() ?
 776                 filesystem::get_program_invocation("wesnothd") : preferences::get_mp_server_program_name();
 777 
 778         std::string config = filesystem::get_user_config_dir() + "/lan_server.cfg";
 779         if (!filesystem::file_exists(config)) {
 780                 // copy file if it isn't created yet
 781                 filesystem::write_file(config, filesystem::read_file(filesystem::get_wml_location("lan_server.cfg")));
 782         }
 783 
 784 #ifndef _WIN32
 785         std::string command = "\"" + wesnothd_program +"\" -c \"" + config + "\" -d -t 2 -T 5";
 786 #else
 787         // start wesnoth as background job
 788         std::string command = "cmd /C start \"wesnoth server\" /B \"" + wesnothd_program + "\" -c \"" + config + "\" -t 2 -T 5";
 789         // Make sure wesnothd's console output is visible on the console window by
 790         // disabling SDL's stdio redirection code for this and future child
 791         // processes. No need to bother cleaning this up because it's only
 792         // meaningful to SDL applications during pre-main initialization.
 793         SetEnvironmentVariableA("SDL_STDIO_REDIRECT", "0");
 794 #endif
 795         LOG_GENERAL << "Starting wesnothd: "<< command << "\n";
 796         if (std::system(command.c_str()) == 0) {
 797                 // Give server a moment to start up
 798                 SDL_Delay(50);
 799                 return;
 800         }
 801         preferences::set_mp_server_program_name("");
 802 
 803         // Couldn't start server so throw error
 804         WRN_GENERAL << "Failed to run server start script" << std::endl;
 805         throw game::mp_server_error("Starting MP server failed!");
 806 }

 808 bool game_launcher::play_multiplayer(mp_selection res)
 809 {
 810         state_ = saved_game();
 811         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 812 
 813         try {
 814                 if (res == MP_HOST)
 815                 {
 816                         try {
 817                                 start_wesnothd();
 818                         } catch(game::mp_server_error&)
 819                         {
 820                                 preferences::show_wesnothd_server_search(video());
 821 
 822                                 try {
 823                                         start_wesnothd();
 824                                 } catch(game::mp_server_error&)
 825                                 {
 826                                         return false;
 827                                 }
 828                         }
 829 
 830 
 831                 }
 832                 //create_engine already calls game_config_manager::get()->load_config but maybe its better to have MULTIPLAYER defined while we are in the lobby.
 833                 game_config_manager::get()->load_game_config_for_create(true);
 834 
 835                 events::discard_input(); // prevent the "keylogger" effect
 836                 cursor::set(cursor::NORMAL);
 837 
 838                 if(res == MP_LOCAL) {
 839                         mp::start_local_game(video(),
 840                             game_config_manager::get()->game_config(), state_);
 841                 } else {
 842                         mp::start_client(video(), game_config_manager::get()->game_config(),
 843                                 state_, multiplayer_server_);
 844                         multiplayer_server_.clear();
 845                 }
 846 
 847         } catch(game::mp_server_error& e) {
 848                 gui2::show_error_message(video(), _("Error while starting server: ") + e.message);
 849         } catch(game::load_game_failed& e) {
 850                 gui2::show_error_message(video(), _("The game could not be loaded: ") + e.message);
 851         } catch(game::game_error& e) {
 852                 gui2::show_error_message(video(), _("Error while playing the game: ") + e.message);
 853         } catch (mapgen_exception& e) {
 854                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 855         } catch(wesnothd_error& e) {
 856                 if(e.message != "") {
 857                         ERR_NET << "caught network error: " << e.message << std::endl;
 858                         gui2::show_transient_message(video()
 859                                         , ""
 860                                         , translation::gettext(e.message.c_str()));
 861                 } else {
 862                         ERR_NET << "caught network error" << std::endl;
 863                 }
 864         } catch(config::error& e) {
 865                 if(e.message != "") {
 866                         ERR_CONFIG << "caught config::error: " << e.message << std::endl;
 867                         gui2::show_transient_message(video(), "", e.message);
 868                 } else {
 869                         ERR_CONFIG << "caught config::error" << std::endl;
 870                 }
 871         } catch(incorrect_map_format_error& e) {
 872                 gui2::show_error_message(video(), _("The game map could not be loaded: ") + e.message);
 873         } catch (savegame::load_game_exception & e) {
 874                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 875                 //this will make it so next time through the title screen loop, this game is loaded
 876         } catch(wml_exception& e) {
 877                 e.show(video());
 878         } catch (game::error & e) {
 879                 std::cerr << "caught game::error...\n";
 880                 gui2::show_error_message(video(), _("Error: ") + e.message);
 881         }
 882 
 883         return false;
 884 }

 928 void game_launcher::launch_game(RELOAD_GAME_DATA reload)
 929 {
 930         assert(!load_data_);
 931         if(play_replay_)
 932         {
 933                 play_replay();
 934                 return;
 935         }
 936 
 937         gui2::dialogs::loading_screen::display(video(), [this, reload]() {
 938 
 939                 gui2::dialogs::loading_screen::progress("load data");
 940                 if(reload == RELOAD_DATA) {
 941                         try {
 942                                 game_config_manager::get()->
 943                                         load_game_config_for_game(state_.classification());
 944                         } catch(config::error&) {
 945                                 return;
 946                         }
 947                 }
 948         });
 949 
 950         try {
 951                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types());
 952                 LEVEL_RESULT result = ccontroller.play_game();
 953                 // don't show The End for multiplayer scenario
 954                 // change this if MP campaigns are implemented
 955                 if(result == LEVEL_RESULT::VICTORY && !state_.classification().is_normal_mp_game()) {
 956                         preferences::add_completed_campaign(state_.classification().campaign, state_.classification().difficulty);
 957 
 958                         gui2::dialogs::outro::display(state_.classification().end_text, state_.classification().end_text_duration, video());
 959 
 960                         if(state_.classification().end_credits) {
 961                                 gui2::dialogs::end_credits::display(video(), state_.classification().campaign);
 962                         }
 963                 }
 964         } catch (savegame::load_game_exception &e) {
 965                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 966                 //this will make it so next time through the title screen loop, this game is loaded
 967         } catch(wml_exception& e) {
 968                 e.show(video());
 969         } catch(mapgen_exception& e) {
 970                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 971         }
 972 }

 988 editor::EXIT_STATUS game_launcher::start_editor(const std::string& filename)
 989 {
 990         while(true){
 991                 game_config_manager::get()->load_game_config_for_editor();
 992 
 993                 ::init_textdomains(game_config_manager::get()->game_config());
 994 
 995                 editor::EXIT_STATUS res = editor::start(
 996                     game_config_manager::get()->game_config(), video(), filename);
 997 
 998                 if(res != editor::EXIT_RELOAD_DATA)
 999                         return res;
1000 
1001                 game_config_manager::get()->reload_changed_game_config();
1002                 image::flush_cache();
1003         }
1004         return editor::EXIT_ERROR; // not supposed to happen
1005 }
