  57 void trigger_full_redraw() {
  58         SDL_Event event;
  59         event.type = SDL_WINDOWEVENT;
  60         event.window.event = SDL_WINDOWEVENT_RESIZED;
  61         event.window.data1 = (*frameBuffer).h;
  62         event.window.data2 = (*frameBuffer).w;
  63 
  64         for(const auto& layer : draw_layers) {
  65                 layer->handle_window_event(event);
  66         }
  67 
  68         SDL_Event drawEvent;
  69         SDL_UserEvent data;
  70 
  71         data.type = DRAW_ALL_EVENT;
  72         data.code = 0;
  73         data.data1 = nullptr;
  74         data.data2 = nullptr;
  75 
  76         drawEvent.type = DRAW_ALL_EVENT;
  77         drawEvent.user = data;
  78         SDL_FlushEvent(DRAW_ALL_EVENT);
  79         SDL_PushEvent(&drawEvent);
  80 }

 206 void CVideo::init_window()
 207 {
 208         // Position
 209         const int x = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 210         const int y = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 211 
 212         // Dimensions
 213         const int w = preferences::resolution().first;
 214         const int h = preferences::resolution().second;
 215 
 216         // Video flags
 217         int video_flags = 0;
 218 
 219         // Add any more default flags here
 220         video_flags |= SDL_WINDOW_RESIZABLE;
 221 
 222         if(preferences::fullscreen()) {
 223                 video_flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
 224         } else if(preferences::maximized()) {
 225                 video_flags |= SDL_WINDOW_MAXIMIZED;
 226         }
 227 
 228         // Initialize window
 229         window.reset(new sdl::window("", x, y, w, h, video_flags, SDL_RENDERER_SOFTWARE));
 230 
 231         std::cerr << "Setting mode to " << w << "x" << h << std::endl;
 232 
 233         window->set_minimum_size(
 234                 preferences::min_window_width,
 235                 preferences::min_window_height
 236         );
 237 
 238         SDL_DisplayMode currentDisplayMode;
 239         SDL_GetCurrentDisplayMode(window->get_display_index(), &currentDisplayMode);
 240         refresh_rate_ = currentDisplayMode.refresh_rate != 0 ? currentDisplayMode.refresh_rate : 60;
 241 
 242         event_handler_.join_global();
 243 
 244         update_framebuffer();
 245         if(frameBuffer) {
 246                 image::set_pixel_format(frameBuffer->format);
 247         }
 248 }

 250 void CVideo::setMode(int x, int y, const MODE_EVENT mode)
 251 {
 252         assert(window);
 253         if(fake_screen_) return;
 254 
 255         switch(mode) {
 256                 case TO_FULLSCREEN:
 257                         window->full_screen();
 258                         break;
 259 
 260                 case TO_WINDOWED:
 261                         window->to_window();
 262                         window->restore();
 263                         break;
 264 
 265                 case TO_MAXIMIZED_WINDOW:
 266                         window->to_window();
 267                         window->maximize();
 268                         break;
 269 
 270                 case TO_RES:
 271                         window->restore();
 272                         window->set_size(x, y);
 273                         window->center();
 274                         break;
 275         }
 276 
 277         update_framebuffer();
 278         if(frameBuffer) {
 279                 image::set_pixel_format(frameBuffer->format);
 280         }
 281 }

 351 std::pair<float, float> CVideo::get_dpi_scale_factor() const
 352 {
 353         std::pair<float, float> result {1.0f, 1.0f};
 354 
 355         if(!window) {
 356                 return result;
 357         }
 358 
 359 /* This check is here because Travis CI builds with SDL 2.0.2.
 360  * The code isn't intended to be skipped: end users, distribution packagers etc.
 361  * should build the game with SDL 2.0.4 or above.
 362  * - Jyrki, 2017-06-03 */
 363 #if SDL_VERSION_ATLEAST(2, 0, 4)
 364         float hdpi, vdpi;
 365         SDL_GetDisplayDPI(window->get_display_index(), nullptr, &hdpi, &vdpi);
 366 
 367         result.first = hdpi / MAGIC_DPI_SCALE_NUMBER;
 368         result.second = vdpi / MAGIC_DPI_SCALE_NUMBER;
 369 #endif
 370 
 371         return result;

 374 std::vector<std::pair<int, int>> CVideo::get_available_resolutions(const bool include_current)
 375 {
 376         std::vector<std::pair<int, int>> result;
 377 
 378         if(!window) {
 379                 return result;
 380         }
 381 
 382         const int display_index = window->get_display_index();
 383 
 384         const int modes = SDL_GetNumDisplayModes(display_index);
 385         if(modes <= 0) {
 386                 std::cerr << "No modes supported\n";
 387                 return result;
 388         }
 389 
 390         const std::pair<int,int> min_res = std::make_pair(preferences::min_window_width, preferences::min_window_height);
 391         const std::pair<int,int> current_res = current_resolution();
 392 
 393         float scale_h, scale_v;
 394         std::tie(scale_h, scale_v) = get_dpi_scale_factor();
 395 
 396         SDL_DisplayMode mode;
 397         for(int i = 0; i < modes; ++i) {
 398                 if(SDL_GetDisplayMode(display_index, i, &mode) == 0) {
 399                         // Exclude any results outside the range of the current DPI.
 400                         if(mode.w > current_res.first * scale_h && mode.h > current_res.second * scale_v) {
 401                                 continue;
 402                         }
 403 
 404                         if(mode.w >= min_res.first && mode.h >= min_res.second) {
 405                                 result.emplace_back(mode.w, mode.h);
 406                         }
 407                 }
 408         }
 409 
 410         if(std::find(result.begin(), result.end(), min_res) == result.end()) {
 411                 result.push_back(min_res);
 412         }
 413 
 414         if(include_current) {
 415                 result.push_back(current_res);
 416         }
 417 
 418         std::sort(result.begin(), result.end());
 419         result.erase(std::unique(result.begin(), result.end()), result.end());
 420 
 421         return result;
 422 }

 441 int CVideo::set_help_string(const std::string& str)
 442 {
 443         font::remove_floating_label(help_string_);
 444 
 445         const color_t color { 0, 0, 0, 0xbb };
 446 
 447         int size = font::SIZE_LARGE;
 448 
 449         while(size > 0) {
 450                 if(font::line_width(str, size) > getx()) {
 451                         size--;
 452                 } else {
 453                         break;
 454                 }
 455         }
 456 
 457         const int border = 5;
 458 
 459         font::floating_label flabel(str);
 460         flabel.set_font_size(size);
 461         flabel.set_position(getx()/2, gety());
 462         flabel.set_bg_color(color);
 463         flabel.set_border_size(border);
 464 
 465         help_string_ = font::add_floating_label(flabel);
 466 
 467         const SDL_Rect& rect = font::get_floating_label_rect(help_string_);
 468         font::move_floating_label(help_string_,0.0,-double(rect.h));
 469         return help_string_;
 470 }

 485 void CVideo::set_fullscreen(bool ison)
 486 {
 487 
 488         if(window && isFullScreen() != ison) {
 489 
 490                 const std::pair<int,int>& res = preferences::resolution();
 491 
 492                 MODE_EVENT mode;
 493 
 494                 if(ison) {
 495                         mode = TO_FULLSCREEN;
 496                 } else {
 497                         mode = preferences::maximized() ? TO_MAXIMIZED_WINDOW : TO_WINDOWED;
 498                 }
 499 
 500                 setMode(res.first, res.second, mode);
 501 
 502                 if(display::get_singleton()) {
 503                         display::get_singleton()->redraw_everything();
 504                 }
 505         }
 506 
 507         // Change the config value.
 508         preferences::_set_fullscreen(ison);
 509 }

 516 void CVideo::set_resolution(const unsigned width, const unsigned height)
 517 {

 519                 static_cast<unsigned int>(current_resolution().second) == height) {
 520 
 521                 return;
 522         }
 523 
 524         setMode(width, height, TO_RES);
 525 
 526         if(display::get_singleton()) {
 527                 display::get_singleton()->redraw_everything();
 528         }
 529 
 530         // Change the config values.
 531         preferences::_set_resolution(std::make_pair(width, height));
 532         preferences::_set_maximized(false);
