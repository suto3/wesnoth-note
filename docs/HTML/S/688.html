  57 void trigger_full_redraw() {
  58         SDL_Event event;
  59         event.type = SDL_WINDOWEVENT;
  60         event.window.event = SDL_WINDOWEVENT_RESIZED;
  61         event.window.data1 = (*frameBuffer).h;
  62         event.window.data2 = (*frameBuffer).w;
  63 
  64         for(const auto& layer : draw_layers) {
  65                 layer->handle_window_event(event);
  66         }
  67 
  68         SDL_Event drawEvent;
  69         SDL_UserEvent data;
  70 
  71         data.type = DRAW_ALL_EVENT;
  72         data.code = 0;
  73         data.data1 = nullptr;
  74         data.data2 = nullptr;
  75 
  76         drawEvent.type = DRAW_ALL_EVENT;
  77         drawEvent.user = data;
  78         SDL_FlushEvent(DRAW_ALL_EVENT);
  79         SDL_PushEvent(&drawEvent);
  80 }

 215 void CVideo::init_window()
 216 {
 217         // Position
 218         const int x = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 219         const int y = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 220 
 221         // Dimensions
 222         const int w = preferences::resolution().first;
 223         const int h = preferences::resolution().second;
 224 
 225         // Video flags
 226         int video_flags = 0;
 227 
 228         // Add any more default flags here
 229         video_flags |= SDL_WINDOW_RESIZABLE;
 230 
 231         if(preferences::fullscreen()) {
 232                 video_flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
 233         } else if(preferences::maximized()) {
 234                 video_flags |= SDL_WINDOW_MAXIMIZED;
 235         }
 236 
 237         // Initialize window
 238         window.reset(new sdl::window("", x, y, w, h, video_flags, SDL_RENDERER_SOFTWARE));
 239 
 240         std::cerr << "Setting mode to " << w << "x" << h << std::endl;
 241 
 242         window->set_minimum_size(
 243                 preferences::min_window_width,
 244                 preferences::min_window_height
 245         );
 246 
 247         SDL_DisplayMode currentDisplayMode;
 248         SDL_GetCurrentDisplayMode(window->get_display_index(), &currentDisplayMode);
 249         refresh_rate_ = currentDisplayMode.refresh_rate != 0 ? currentDisplayMode.refresh_rate : 60;
 250 
 251         event_handler_.join_global();
 252 
 253         update_framebuffer();
 254         if(frameBuffer) {
 255                 image::set_pixel_format(frameBuffer->format);
 256         }
 257 }

 259 void CVideo::setMode(int x, int y, const MODE_EVENT mode)
 260 {
 261         assert(window);
 262         if(fake_screen_) return;
 263 
 264         switch(mode) {
 265                 case TO_FULLSCREEN:
 266                         window->full_screen();
 267                         break;
 268 
 269                 case TO_WINDOWED:
 270                         window->to_window();
 271                         window->restore();
 272                         break;
 273 
 274                 case TO_MAXIMIZED_WINDOW:
 275                         window->to_window();
 276                         window->maximize();
 277                         break;
 278 
 279                 case TO_RES:
 280                         window->restore();
 281                         window->set_size(x, y);
 282                         window->center();
 283                         break;
 284         }
 285 
 286         update_framebuffer();
 287         if(frameBuffer) {
 288                 image::set_pixel_format(frameBuffer->format);
 289         }
 290 }

 368 std::pair<float, float> CVideo::get_dpi_scale_factor() const
 369 {

 371

 391 std::vector<std::pair<int, int>> CVideo::get_available_resolutions(const bool include_current)
 392 {
 393         std::vector<std::pair<int, int>> result;
 394 
 395         if(!window) {
 396                 return result;
 397         }
 398 
 399         const int display_index = window->get_display_index();
 400 
 401         const int modes = SDL_GetNumDisplayModes(display_index);
 402         if(modes <= 0) {
 403                 std::cerr << "No modes supported\n";
 404                 return result;
 405         }
 406 
 407         const std::pair<int,int> min_res = std::make_pair(preferences::min_window_width, preferences::min_window_height);
 408         const std::pair<int,int> current_res = current_resolution();
 409 
 410 #if 0
 411         // DPI scale factor.
 412         float scale_h, scale_v;
 413         std::tie(scale_h, scale_v) = get_dpi_scale_factor();
 414 #endif
 415 
 416         // The maximum size to which this window can be set. For some reason this won't
 417         // pop up as a display mode of its own.
 418         SDL_Rect bounds;
 419         SDL_GetDisplayBounds(display_index, &bounds);
 420 
 421         SDL_DisplayMode mode;
 422 
 423         for(int i = 0; i < modes; ++i) {
 424                 if(SDL_GetDisplayMode(display_index, i, &mode) == 0) {
 425                         // Exclude any results outside the range of the current DPI.
 426                         if(mode.w > bounds.w && mode.h > bounds.h) {
 427                                 continue;
 428                         }
 429 
 430                         if(mode.w >= min_res.first && mode.h >= min_res.second) {
 431                                 result.emplace_back(mode.w, mode.h);
 432                         }
 433                 }
 434         }
 435 
 436         if(std::find(result.begin(), result.end(), min_res) == result.end()) {
 437                 result.push_back(min_res);
 438         }
 439 
 440         if(include_current) {
 441                 result.push_back(current_res);
 442         }
 443 
 444         std::sort(result.begin(), result.end());
 445         result.erase(std::unique(result.begin(), result.end()), result.end());
 446 
 447         return result;
 448 }

 466 int CVideo::set_help_string(const std::string& str)
 467 {
 468         font::remove_floating_label(help_string_);
 469 
 470         const color_t color { 0, 0, 0, 0xbb };
 471 
 472         int size = font::SIZE_LARGE;
 473 
 474         while(size > 0) {
 475                 if(font::line_width(str, size) > getx()) {
 476                         size--;
 477                 } else {
 478                         break;
 479                 }
 480         }
 481 
 482         const int border = 5;
 483 
 484         font::floating_label flabel(str);
 485         flabel.set_font_size(size);
 486         flabel.set_position(getx()/2, gety());
 487         flabel.set_bg_color(color);
 488         flabel.set_border_size(border);
 489 
 490         help_string_ = font::add_floating_label(flabel);
 491 
 492         const SDL_Rect& rect = font::get_floating_label_rect(help_string_);
 493         font::move_floating_label(help_string_,0.0,-double(rect.h));
 494         return help_string_;
 495 }

 510 void CVideo::set_fullscreen(bool ison)
 511 {
 512 
 513         if(window && isFullScreen() != ison) {
 514 
 515                 const std::pair<int,int>& res = preferences::resolution();
 516 
 517                 MODE_EVENT mode;
 518 
 519                 if(ison) {
 520                         mode = TO_FULLSCREEN;
 521                 } else {
 522                         mode = preferences::maximized() ? TO_MAXIMIZED_WINDOW : TO_WINDOWED;
 523                 }
 524 
 525                 setMode(res.first, res.second, mode);
 526 
 527                 if(display::get_singleton()) {
 528                         display::get_singleton()->redraw_everything();
 529                 }
 530         }
 531 
 532         // Change the config value.
 533         preferences::_set_fullscreen(ison);
 534 }
