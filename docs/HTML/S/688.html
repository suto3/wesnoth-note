  57 void trigger_full_redraw() {
  58         SDL_Event event;
  59         event.type = SDL_WINDOWEVENT;
  60         event.window.event = SDL_WINDOWEVENT_RESIZED;
  61         event.window.data1 = (*frameBuffer).h;
  62         event.window.data2 = (*frameBuffer).w;
  63 
  64         for(const auto& layer : draw_layers) {
  65                 layer->handle_window_event(event);
  66         }
  67 
  68         SDL_Event drawEvent;
  69         SDL_UserEvent data;
  70 
  71         data.type = DRAW_ALL_EVENT;
  72         data.code = 0;
  73         data.data1 = nullptr;
  74         data.data2 = nullptr;
  75 
  76         drawEvent.type = DRAW_ALL_EVENT;
  77         drawEvent.user = data;
  78         SDL_FlushEvent(DRAW_ALL_EVENT);
  79         SDL_PushEvent(&drawEvent);
  80 }

 206 void CVideo::init_window()
 207 {
 208         // Position
 209         const int x = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 210         const int y = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 211 
 212         // Dimensions
 213         const int w = preferences::resolution().first;
 214         const int h = preferences::resolution().second;
 215 
 216         // Video flags
 217         int video_flags = 0;
 218 
 219         // Add any more default flags here
 220         video_flags |= SDL_WINDOW_RESIZABLE;
 221 
 222         if(preferences::fullscreen()) {
 223                 video_flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
 224         } else if(preferences::maximized()) {
 225                 video_flags |= SDL_WINDOW_MAXIMIZED;
 226         }
 227 
 228         // Initialize window
 229         window.reset(new sdl::window("", x, y, w, h, video_flags, SDL_RENDERER_SOFTWARE));
 230 
 231         std::cerr << "Setting mode to " << w << "x" << h << std::endl;
 232 
 233         window->set_minimum_size(
 234                 preferences::min_window_width,
 235                 preferences::min_window_height
 236         );
 237 
 238         event_handler_.join_global();
 239 
 240         update_framebuffer();
 241         if(frameBuffer) {
 242                 image::set_pixel_format(frameBuffer->format);
 243         }
 244 }

 246 void CVideo::setMode(int x, int y, const MODE_EVENT mode)
 247 {
 248         assert(window);
 249         if(fake_screen_) return;
 250 
 251         switch(mode) {
 252                 case TO_FULLSCREEN:
 253                         window->full_screen();
 254                         break;
 255 
 256                 case TO_WINDOWED:
 257                         window->to_window();
 258                         window->restore();
 259                         break;
 260 
 261                 case TO_MAXIMIZED_WINDOW:
 262                         window->to_window();
 263                         window->maximize();
 264                         break;
 265 
 266                 case TO_RES:
 267                         window->restore();
 268                         window->set_size(x, y);
 269                         window->center();
 270                         break;
 271         }
 272 
 273         update_framebuffer();
 274         if(frameBuffer) {
 275                 image::set_pixel_format(frameBuffer->format);
 276         }
 277 }

 347 std::pair<float, float> CVideo::get_dpi_scale_factor() const
 348 {
 349         std::pair<float, float> result {1.0f, 1.0f};
 350 
 351         if(!window) {
 352                 return result;
 353         }
 354 
 355 /* This check is here because Travis CI builds with SDL 2.0.2.
 356  * The code isn't intended to be skipped: end users, distribution packagers etc.
 357  * should build the game with SDL 2.0.4 or above.
 358  * - Jyrki, 2017-06-03 */
 359 #if SDL_VERSION_ATLEAST(2, 0, 4)
 360         float hdpi, vdpi;
 361         SDL_GetDisplayDPI(window->get_display_index(), nullptr, &hdpi, &vdpi);
 362 
 363         result.first = hdpi / MAGIC_DPI_SCALE_NUMBER;
 364         result.second = vdpi / MAGIC_DPI_SCALE_NUMBER;
 365 #endif
 366 
 367         return result;

 370 std::vector<std::pair<int, int>> CVideo::get_available_resolutions(const bool include_current)
 371 {
 372         std::vector<std::pair<int, int>> result;
 373 
 374         if(!window) {
 375                 return result;
 376         }
 377 
 378         const int display_index = window->get_display_index();
 379 
 380         const int modes = SDL_GetNumDisplayModes(display_index);
 381         if(modes <= 0) {
 382                 std::cerr << "No modes supported\n";
 383                 return result;
 384         }
 385 
 386         const std::pair<int,int> min_res = std::make_pair(preferences::min_window_width, preferences::min_window_height);
 387         const std::pair<int,int> current_res = current_resolution();
 388 
 389         float scale_h, scale_v;
 390         std::tie(scale_h, scale_v) = get_dpi_scale_factor();
 391 
 392         SDL_DisplayMode mode;
 393         for(int i = 0; i < modes; ++i) {
 394                 if(SDL_GetDisplayMode(display_index, i, &mode) == 0) {
 395                         // Exclude any results outside the range of the current DPI.
 396                         if(mode.w > current_res.first * scale_h && mode.h > current_res.second * scale_v) {
 397                                 continue;
 398                         }
 399 
 400                         if(mode.w >= min_res.first && mode.h >= min_res.second) {
 401                                 result.emplace_back(mode.w, mode.h);
 402                         }
 403                 }
 404         }
 405 
 406         if(std::find(result.begin(), result.end(), min_res) == result.end()) {
 407                 result.push_back(min_res);
 408         }
 409 
 410         if(include_current) {
 411                 result.push_back(current_res);
 412         }
 413 
 414         std::sort(result.begin(), result.end());
 415         result.erase(std::unique(result.begin(), result.end()), result.end());
 416 
 417         return result;
 418 }

 437 int CVideo::set_help_string(const std::string& str)
 438 {
 439         font::remove_floating_label(help_string_);
 440 
 441         const color_t color { 0, 0, 0, 0xbb };
 442 
 443         int size = font::SIZE_LARGE;
 444 
 445         while(size > 0) {
 446                 if(font::line_width(str, size) > getx()) {
 447                         size--;
 448                 } else {
 449                         break;
 450                 }
 451         }
 452 
 453         const int border = 5;
 454 
 455         font::floating_label flabel(str);
 456         flabel.set_font_size(size);
 457         flabel.set_position(getx()/2, gety());
 458         flabel.set_bg_color(color);
 459         flabel.set_border_size(border);
 460 
 461         help_string_ = font::add_floating_label(flabel);
 462 
 463         const SDL_Rect& rect = font::get_floating_label_rect(help_string_);
 464         font::move_floating_label(help_string_,0.0,-double(rect.h));
 465         return help_string_;
 466 }

 481 void CVideo::set_fullscreen(bool ison)
 482 {
 483 
 484         if(window && isFullScreen() != ison) {
 485 
 486                 const std::pair<int,int>& res = preferences::resolution();
 487 
 488                 MODE_EVENT mode;
 489 
 490                 if(ison) {
 491                         mode = TO_FULLSCREEN;
 492                 } else {
 493                         mode = preferences::maximized() ? TO_MAXIMIZED_WINDOW : TO_WINDOWED;
 494                 }
 495 
 496                 setMode(res.first, res.second, mode);
 497 
 498                 if(display::get_singleton()) {
 499                         display::get_singleton()->redraw_everything();
 500                 }
 501         }
 502 
 503         // Change the config value.
 504         preferences::_set_fullscreen(ison);
 505 }

 512 void CVideo::set_resolution(const unsigned width, const unsigned height)
 513 {
 514         if(static_cast<unsigned int>(current_resolution().first)  == width &&
 515                 static_cast<unsigned int>(current_resolution().second) == height) {
 516 
 517                 return;
 518         }
 519 
 520         setMode(width, height, TO_RES);
 521 
 522         if(display::get_singleton()) {
 523                 display::get_singleton()->redraw_everything();
 524         }
 525 
 526         // Change the config values.
 527         preferences::_set_resolution(std::make_pair(width, height));
 528         preferences::_set_maximized(false);
 529 }
