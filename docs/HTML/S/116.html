  38 bool isnewline(const char c)
  39 {
  40         return c == '\r' || c == '\n';
  41 }

  43 // Make sure that we can use Mac, DOS, or Unix style text files on any system
  44 // and they will work, by making sure the definition of whitespace is consistent
  45 bool portable_isspace(const char c)
  46 {
  47         // returns true only on ASCII spaces
  48         if (static_cast<unsigned char>(c) >= 128)
  49                 return false;
  50         return isnewline(c) || isspace(c);
  51 }
  52 
  53 // Make sure we regard '\r' and '\n' as a space, since Mac, Unix, and DOS
  54 // all consider these differently.
  55 bool notspace(const char c)
  56 {
  57         return !portable_isspace(c);
  58 }

  60 /**
  61  * Splits a (comma-)separated string into a vector of pieces.
  62  * @param[in]  val    A (comma-)separated string.
  63  * @param[in]  c      The separator character (usually a comma).
  64  * @param[in]  flags  Flags controlling how the split is done.
  65  *                    This is a bit field with two settings (both on by default):
  66  *                    REMOVE_EMPTY causes empty pieces to be skipped/removed.
  67  *                    STRIP_SPACES causes the leading and trailing spaces of each piece to be ignored/stripped.
  68  *
  69  *                    Basic method taken from http://stackoverflow.com/a/236803
  70  */
  71 std::vector<std::string> split(const std::string& val, const char c, const int flags)
  72 {
  73         std::vector<std::string> res;
  74 
  75         std::stringstream ss;
  76         ss.str(val);
  77 
  78         std::string item;
  79         while(std::getline(ss, item, c)) {
  80                 if(flags & STRIP_SPACES) {
  81                         boost::trim(item);
  82                 }
  83 
  84                 if(!(flags & REMOVE_EMPTY) || !item.empty()) {
  85                         res.push_back(std::move(item));
  86                 }
  87         }
  88 
  89         return res;
  90 }

  92 std::vector<std::string> square_parenthetical_split(const std::string& val,
  93                 const char separator, const std::string& left,
  94                 const std::string& right,const int flags)
  95 {
  96         std::vector< std::string > res;
  97         std::vector<char> part;
  98         bool in_parenthesis = false;
  99         std::vector<std::string::const_iterator> square_left;
 100         std::vector<std::string::const_iterator> square_right;
 101         std::vector< std::string > square_expansion;
 102 
 103         std::string lp=left;
 104         std::string rp=right;
 105 
 106         std::string::const_iterator i1 = val.begin();
 107         std::string::const_iterator i2;
 108         std::string::const_iterator j1;
 109         if (flags & STRIP_SPACES) {
 110                 while (i1 != val.end() && portable_isspace(*i1))
 111                         ++i1;
 112         }
 113         i2=i1;
 114         j1=i1;
 115 
 116         if (i1 == val.end()) return res;
 117 
 118         if (!separator) {
 119                 ERR_GENERAL << "Separator must be specified for square bracket split funtion." << std::endl;
 120                 return res;
 121         }
 122 
 123         if(left.size()!=right.size()){
 124                 ERR_GENERAL << "Left and Right Parenthesis lists not same length" << std::endl;
 125                 return res;
 126         }
 127 
 128         while (true) {
 129                 if(i2 == val.end() || (!in_parenthesis && *i2 == separator)) {
 130                         //push back square contents
 131                         size_t size_square_exp = 0;
 132                         for (size_t i=0; i < square_left.size(); i++) {
 133                                 std::string tmp_val(square_left[i]+1,square_right[i]);
 134                                 std::vector< std::string > tmp = split(tmp_val);
 135                                 for(const std::string& piece : tmp) {
 136                                         size_t found_tilde = piece.find_first_of('~');
 137                                         if (found_tilde == std::string::npos) {
 138                                                 size_t found_asterisk = piece.find_first_of('*');
 139                                                 if (found_asterisk == std::string::npos) {
 140                                                         std::string tmp2(piece);
 141                                                         boost::trim(tmp2);
 142                                                         square_expansion.push_back(tmp2);
 143                                                 }
 144                                                 else { //'*' multiple expansion
 145                                                         std::string s_begin = piece.substr(0,found_asterisk);
 146                                                         boost::trim(s_begin);
 147                                                         std::string s_end = piece.substr(found_asterisk+1);
 148                                                         boost::trim(s_end);
 149                                                         for (int ast=std::stoi(s_end); ast>0; --ast)
 150                                                                 square_expansion.push_back(s_begin);
 151                                                 }
 152                                         }
 153                                         else { //expand number range
 154                                                 std::string s_begin = piece.substr(0,found_tilde);
 155                                                 boost::trim(s_begin);
 156                                                 int begin = std::stoi(s_begin);
 157                                                 size_t padding = 0, padding_end = 0;
 158                                                 while (padding<s_begin.size() && s_begin[padding]=='0') {
 159                                                         padding++;
 160                                                 }
 161                                                 std::string s_end = piece.substr(found_tilde+1);
 162                                                 boost::trim(s_end);
 163                                                 int end = std::stoi(s_end);
 164                                                 while (padding_end<s_end.size() && s_end[padding_end]=='0') {
 165                                                         padding_end++;
 166                                                 }
 167                                                 if (padding*padding_end > 0 && s_begin.size() != s_end.size()) {
 168                                                         ERR_GENERAL << "Square bracket padding sizes not matching: "
 169                                                                                 << s_begin << " and " << s_end <<".\n";
 170                                                 }
 171                                                 if (padding_end > padding) padding = padding_end;
 172 
 173                                                 int increment = (end >= begin ? 1 : -1);
 174                                                 end+=increment; //include end in expansion
 175                                                 for (int k=begin; k!=end; k+=increment) {
 176                                                         std::string pb = std::to_string(k);
 177                                                         for (size_t p=pb.size(); p<=padding; p++)
 178                                                                 pb = std::string("0") + pb;
 179                                                         square_expansion.push_back(pb);
 180                                                 }
 181                                         }
 182                                 }
 183                                 if (i*square_expansion.size() != (i+1)*size_square_exp ) {
 184                                         std::string tmp2(i1, i2);
 185                                         ERR_GENERAL << "Square bracket lengths do not match up: " << tmp2 << std::endl;
 186                                         return res;
 187                                 }
 188                                 size_square_exp = square_expansion.size();
 189                         }
 190 
 191                         //combine square contents and rest of string for comma zone block
 192                         size_t j = 0;
 193                         size_t j_max = 0;
 194                         if (square_left.size() != 0)
 195                                 j_max = square_expansion.size() / square_left.size();
 196                         do {
 197                                 j1 = i1;
 198                                 std::string new_val;
 199                                 for (size_t i=0; i < square_left.size(); i++) {
 200                                         std::string tmp_val(j1, square_left[i]);
 201                                         new_val.append(tmp_val);
 202                                         size_t k = j+i*j_max;
 203                                         if (k < square_expansion.size())
 204                                                 new_val.append(square_expansion[k]);
 205                                         j1 = square_right[i]+1;
 206                                 }
 207                                 std::string tmp_val(j1, i2);
 208                                 new_val.append(tmp_val);
 209                                 if (flags & STRIP_SPACES)
 210                                         boost::trim_right(new_val);
 211                                 if (!(flags & REMOVE_EMPTY) || !new_val.empty())
 212                                         res.push_back(new_val);
 213                                 j++;
 214                         } while (j<j_max);
 215 
 216                         if (i2 == val.end()) //escape loop
 217                                 break;
 218                         ++i2;
 219                         if (flags & STRIP_SPACES) { //strip leading spaces
 220                                 while (i2 != val.end() && portable_isspace(*i2))
 221                                         ++i2;
 222                         }
 223                         i1=i2;
 224                         square_left.clear();
 225                         square_right.clear();
 226                         square_expansion.clear();
 227                         continue;
 228                 }
 229                 if(!part.empty() && *i2 == part.back()) {
 230                         part.pop_back();
 231                         if (*i2 == ']') square_right.push_back(i2);
 232                         if (part.empty())
 233                                 in_parenthesis = false;
 234                         ++i2;
 235                         continue;
 236                 }
 237                 bool found=false;
 238                 for(size_t i=0; i < lp.size(); i++) {
 239                         if (*i2 == lp[i]){
 240                                 if (*i2 == '[')
 241                                         square_left.push_back(i2);
 242                                 ++i2;
 243                                 part.push_back(rp[i]);
 244                                 found=true;
 245                                 break;
 246                         }
 247                 }
 248                 if(!found){
 249                         ++i2;
 250                 } else
 251                         in_parenthesis = true;
 252         }
 253 
 254         if(!part.empty()){
 255                         ERR_GENERAL << "Mismatched parenthesis:\n"<<val<< std::endl;;
 256         }
 257 
 258         return res;
 259 }

 261 std::map<std::string, std::string> map_split(
 262                   const std::string& val
 263                 , char major
 264                 , char minor
 265                 , int flags
 266                 , const std::string& default_value)
 267 {
 268         //first split by major so that we get a vector with the key-value pairs
 269         std::vector< std::string > v = split(val, major, flags);
 270 
 271         //now split by minor to extract keys and values
 272         std::map< std::string, std::string > res;
 273 
 274         for( std::vector< std::string >::iterator i = v.begin(); i != v.end(); ++i) {
 275                 size_t pos = i->find_first_of(minor);
 276                 std::string key, value;
 277 
 278                 if(pos == std::string::npos) {
 279                         key = (*i);
 280                         value = default_value;
 281                 } else {
 282                         key = i->substr(0, pos);
 283                         value = i->substr(pos + 1);
 284                 }
 285 
 286                 res[key] = value;
 287         }
 288 
 289         return res;
 290 }

 292 std::vector<std::string> parenthetical_split(const std::string& val,
 293                 const char separator, const std::string& left,
 294                 const std::string& right,const int flags)
 295 {
 296         std::vector< std::string > res;
 297         std::vector<char> part;
 298         bool in_parenthesis = false;
 299 
 300         std::string lp=left;
 301         std::string rp=right;
 302 
 303         std::string::const_iterator i1 = val.begin();
 304         std::string::const_iterator i2;
 305         if (flags & STRIP_SPACES) {
 306                 while (i1 != val.end() && portable_isspace(*i1))
 307                         ++i1;
 308         }
 309         i2=i1;
 310 
 311         if(left.size()!=right.size()){
 312                 ERR_GENERAL << "Left and Right Parenthesis lists not same length" << std::endl;
 313                 return res;
 314         }
 315 
 316         while (i2 != val.end()) {
 317                 if(!in_parenthesis && separator && *i2 == separator){
 318                         std::string new_val(i1, i2);
 319                         if (flags & STRIP_SPACES)
 320                                 boost::trim_right(new_val);
 321                         if (!(flags & REMOVE_EMPTY) || !new_val.empty())
 322                                 res.push_back(new_val);
 323                         ++i2;
 324                         if (flags & STRIP_SPACES) {
 325                                 while (i2 != val.end() && portable_isspace(*i2))
 326                                         ++i2;
 327                         }
 328                         i1=i2;
 329                         continue;
 330                 }
 331                 if(!part.empty() && *i2 == part.back()){
 332                         part.pop_back();
 333                         if(!separator && part.empty()){
 334                                 std::string new_val(i1, i2);
 335                                 if (flags & STRIP_SPACES)
 336                                         boost::trim(new_val);
 337                                 res.push_back(new_val);
 338                                 ++i2;
 339                                 i1=i2;
 340                         }else{
 341                                 if (part.empty())
 342                                         in_parenthesis = false;
 343                                 ++i2;
 344                         }
 345                         continue;
 346                 }
 347                 bool found=false;
 348                 for(size_t i=0; i < lp.size(); i++){
 349                         if (*i2 == lp[i]){
 350                                 if (!separator && part.empty()){
 351                                         std::string new_val(i1, i2);
 352                                         if (flags & STRIP_SPACES)
 353                                                 boost::trim(new_val);
 354                                         res.push_back(new_val);
 355                                         ++i2;
 356                                         i1=i2;
 357                                 }else{
 358                                         ++i2;
 359                                 }
 360                                 part.push_back(rp[i]);
 361                                 found=true;
 362                                 break;
 363                         }
 364                 }
 365                 if(!found){
 366                         ++i2;
 367                 } else
 368                         in_parenthesis = true;
 369         }
 370 
 371         std::string new_val(i1, i2);
 372         if (flags & STRIP_SPACES)
 373                 boost::trim(new_val);
 374         if (!(flags & REMOVE_EMPTY) || !new_val.empty())
 375                 res.push_back(std::move(new_val));
 376 
 377         if(!part.empty()){
 378                         ERR_GENERAL << "Mismatched parenthesis:\n"<<val<< std::endl;;
 379         }
 380 
 381         return res;
 382 }

 384 // Modify a number by string representing integer difference, or optionally %
 385 int apply_modifier( const int number, const std::string &amount, const int minimum ) {
 386         // wassert( amount.empty() == false );
 387         int value = 0;
 388         try {
 389                 value = std::stoi(amount);
 390         } catch(std::invalid_argument) {}
 391         if(amount[amount.size()-1] == '%') {
 392                 value = div100rounded(number * value);
 393         }
 394         value += number;
 395         if (( minimum > 0 ) && ( value < minimum ))
 396             value = minimum;
 397         return value;
 398 }

 400 std::string escape(const std::string &str, const char *special_chars)
 401 {
 402         std::string::size_type pos = str.find_first_of(special_chars);
 403         if (pos == std::string::npos) {
 404                 // Fast path, possibly involving only reference counting.
 405                 return str;
 406         }
 407         std::string res = str;
 408         do {
 409                 res.insert(pos, 1, '\\');
 410                 pos = res.find_first_of(special_chars, pos + 2);
 411         } while (pos != std::string::npos);
 412         return res;
 413 }

 415 std::string unescape(const std::string &str)
 416 {
 417         std::string::size_type pos = str.find('\\');
 418         if (pos == std::string::npos) {
 419                 // Fast path, possibly involving only reference counting.
 420                 return str;
 421         }
 422         std::string res = str;
 423         do {
 424                 res.erase(pos, 1);
 425                 pos = res.find('\\', pos + 1);
 426         } while (pos != std::string::npos);
 427         return str;
 428 }

 430 std::string urlencode(const std::string &str)
 431 {
 432         static const std::string nonresv_str =
 433                 "-."
 434                 "0123456789"
 435                 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
 436                 "_"
 437                 "abcdefghijklmnopqrstuvwxyz"
 438                 "~";
 439         static const std::set<char> nonresv(nonresv_str.begin(), nonresv_str.end());
 440 
 441         std::ostringstream res;
 442         res << std::hex;
 443         res.fill('0');
 444 
 445         for(char c : str) {
 446                 if(nonresv.count(c) != 0) {
 447                         res << c;
 448                         continue;
 449                 }
 450 
 451                 res << '%';
 452                 res.width(2);
 453                 res << int(c);
 454         }
 455 
 456         return res.str();
 457 }

 495 static void si_string_impl_stream_write(std::stringstream &ss, double input) {
 496         std::streamsize oldprec = ss.precision();
 497 #ifdef _MSC_VER
 498         // Visual C++ makes 'precision' set the number of decimal places.
 499         // Other platforms make it set the number of significant figures
 500         ss.precision(1);
 501         ss << std::fixed
 502            << input;
 503 #else
 504         // Workaround to display 1023 KiB instead of 1.02e3 KiB
 505         if (input >= 1000)
 506                 ss.precision(4);
 507         else
 508                 ss.precision(3);
 509         ss << input;
 510 #endif
 511         ss.precision(oldprec);

 514 std::string si_string(double input, bool base2, const std::string& unit) {
 515         const double multiplier = base2 ? 1024 : 1000;
 516 
 517         typedef std::array<std::string, 9> strings9;
 518 
 519         strings9 prefixes;
 520         strings9::const_iterator prefix;
 521         if (input == 0.0) {
 522                 strings9 tmp { { "","","","","","","","","" } };
 523                 prefixes = tmp;
 524                 prefix = prefixes.begin();
 525         } else if (input < 1.0) {
 526                 strings9 tmp { {
 527                         "",
 528                         _("prefix_milli^m"),
 529                         _("prefix_micro^µ"),
 530                         _("prefix_nano^n"),
 531                         _("prefix_pico^p"),
 532                         _("prefix_femto^f"),
 533                         _("prefix_atto^a"),
 534                         _("prefix_zepto^z"),
 535                         _("prefix_yocto^y")
 536                 } };
 537                 prefixes = tmp;
 538                 prefix = prefixes.begin();
 539                 while (input < 1.0  && *prefix != prefixes.back()) {
 540                         input *= multiplier;
 541                         ++prefix;
 542                 }
 543         } else {
 544                 strings9 tmp { {
 545                         "",
 546                         (base2 ?
 547                                 _("prefix_kibi^K") :
 548                                 _("prefix_kilo^k")
 549                         ),
 550                         _("prefix_mega^M"),
 551                         _("prefix_giga^G"),
 552                         _("prefix_tera^T"),
 553                         _("prefix_peta^P"),
 554                         _("prefix_exa^E"),
 555                         _("prefix_zetta^Z"),
 556                         _("prefix_yotta^Y")
 557                 } };
 558                 prefixes = tmp;
 559                 prefix = prefixes.begin();
 560                 while (input > multiplier && *prefix != prefixes.back()) {
 561                         input /= multiplier;
 562                         ++prefix;
 563                 }
 564         }
 565 
 566         std::stringstream ss;
 567         si_string_impl_stream_write(ss, input);
 568         ss << ' '
 569            << *prefix
 570            << (base2 && (*prefix != "") ? _("infix_binary^i") : "")
 571            << unit;
 572         return ss.str();
 573 }

 583 bool isvalid_username(const std::string& username) {
 584         const size_t alnum = std::count_if(username.begin(), username.end(), isalnum);
 585         const size_t valid_char =
 586                         std::count_if(username.begin(), username.end(), is_username_char);
 587         if ((alnum + valid_char != username.size())
 588                         || valid_char == username.size() || username.empty() )
 589         {
 590                 return false;
 591         }
 592         return true;
 593 }

 595 bool isvalid_wildcard(const std::string& username) {
 596     const size_t alnum = std::count_if(username.begin(), username.end(), isalnum);
 597         const size_t valid_char =
 598                         std::count_if(username.begin(), username.end(), is_username_char);
 599     const size_t wild_char =
 600             std::count_if(username.begin(), username.end(), is_wildcard_char);
 601         if ((alnum + valid_char + wild_char != username.size())
 602                         || valid_char == username.size() || username.empty() )
 603         {
 604                 return false;
 605         }
 606         return true;
 607 }

 610 bool word_completion(std::string& text, std::vector<std::string>& wordlist) {
 611         std::vector<std::string> matches;
 612         const size_t last_space = text.rfind(" ");
 613         // If last character is a space return.
 614         if (last_space == text.size() -1) {
 615                 wordlist = matches;
 616                 return false;
 617         }
 618 
 619         bool text_start;
 620         std::string semiword;
 621         if (last_space == std::string::npos) {
 622                 text_start = true;
 623                 semiword = text;
 624         } else {
 625                 text_start = false;
 626                 semiword.assign(text, last_space + 1, text.size());
 627         }
 628 
 629         std::string best_match = semiword;
 630         for (std::vector<std::string>::const_iterator word = wordlist.begin();
 631                         word != wordlist.end(); ++word)
 632         {
 633                 if (word->size() < semiword.size()
 634                 || !std::equal(semiword.begin(), semiword.end(), word->begin(),
 635                                 chars_equal_insensitive))
 636                 {
 637                         continue;
 638                 }
 639                 if (matches.empty()) {
 640                         best_match = *word;
 641                 } else {
 642                         int j = 0;
 643                         while (toupper(best_match[j]) == toupper((*word)[j])) j++;
 644                         if (best_match.begin() + j < best_match.end()) {
 645                                 best_match.erase(best_match.begin() + j, best_match.end());
 646                         }
 647                 }
 648                 matches.push_back(*word);
 649         }
 650         if(!matches.empty()) {
 651                 text.replace(last_space + 1, best_match.size(), best_match);
 652         }
 653         wordlist = matches;
 654         return text_start;
 655 }

 661 bool word_match(const std::string& message, const std::string& word) {
 662         size_t first = message.find(word);
 663         if (first == std::string::npos) return false;
 664         if (first == 0 || is_word_boundary(message[first - 1])) {
 665                 size_t next = first + word.size();
 666                 if (next == message.size() || is_word_boundary(message[next])) {
 667                         return true;
 668                 }
 669         }
 670         return false;
 671 }

 673 bool wildcard_string_match(const std::string& str, const std::string& match) {
 674         const bool wild_matching = (!match.empty() && match[0] == '*');
 675         const std::string::size_type solid_begin = match.find_first_not_of('*');
 676         const bool have_solids = (solid_begin != std::string::npos);
 677         // Check the simple case first
 678         if(str.empty() || !have_solids) {
 679                 return wild_matching || str == match;
 680         }
 681         const std::string::size_type solid_end = match.find_first_of('*', solid_begin);
 682         const std::string::size_type solid_len = (solid_end == std::string::npos)
 683                 ? match.length() - solid_begin : solid_end - solid_begin;
 684         std::string::size_type current = 0;
 685         bool matches;
 686         do {
 687                 matches = true;
 688                 // Now try to place the str into the solid space
 689                 const std::string::size_type test_len = str.length() - current;
 690                 for(std::string::size_type i=0; i < solid_len && matches; ++i) {
 691                         char solid_c = match[solid_begin + i];
 692                         if(i > test_len || !(solid_c == '?' || solid_c == str[current+i])) {
 693                                 matches = false;
 694                         }
 695                 }
 696                 if(matches) {
 697                         // The solid space matched, now consume it and attempt to find more
 698                         const std::string consumed_match = (solid_begin+solid_len < match.length())
 699                                 ? match.substr(solid_end) : "";
 700                         const std::string consumed_str = (solid_len < test_len)
 701                                 ? str.substr(current+solid_len) : "";
 702                         matches = wildcard_string_match(consumed_str, consumed_match);
 703                 }
 704         } while(wild_matching && !matches && ++current < str.length());
 705         return matches;
 706 }

 708 std::string indent(const std::string& string, size_t indent_size)
 709 {
 710         if(indent_size == 0) {
 711                 return string;
 712         }
 713 
 714         const std::string indent(indent_size, ' ');
 715 
 716         if(string.empty()) {
 717                 return indent;
 718         }
 719 
 720         const std::vector<std::string>& lines = split(string, '\x0A', 0);
 721         std::string res;
 722 
 723         for(size_t lno = 0; lno < lines.size();) {
 724                 const std::string& line = lines[lno];
 725 
 726                 // Lines containing only a carriage return count as empty.
 727                 if(!line.empty() && line != "\x0D") {
 728                         res += indent;
 729                 }
 730 
 731                 res += line;
 732 
 733                 if(++lno < lines.size()) {
 734                         res += '\x0A';
 735                 }
 736         }
 737 
 738         return res;
 739 }

 741 std::vector<std::string> quoted_split(const std::string& val, char c, int flags, char quote)
 742 {
 743         std::vector<std::string> res;
 744 
 745         std::string::const_iterator i1 = val.begin();
 746         std::string::const_iterator i2 = val.begin();
 747 
 748         while (i2 != val.end()) {
 749                 if (*i2 == quote) {
 750                         // Ignore quoted character
 751                         ++i2;
 752                         if (i2 != val.end()) ++i2;
 753                 } else if (*i2 == c) {
 754                         std::string new_val(i1, i2);
 755                         if (flags & STRIP_SPACES)
 756                                 boost::trim(new_val);
 757                         if (!(flags & REMOVE_EMPTY) || !new_val.empty())
 758                                 res.push_back(std::move(new_val));
 759                         ++i2;
 760                         if (flags & STRIP_SPACES) {
 761                                 while(i2 != val.end() && *i2 == ' ')
 762                                         ++i2;
 763                         }
 764 
 765                         i1 = i2;
 766                 } else {
 767                         ++i2;
 768                 }
 769         }
 770 
 771         std::string new_val(i1, i2);
 772         if (flags & STRIP_SPACES)
 773                 boost::trim(new_val);
 774         if (!(flags & REMOVE_EMPTY) || !new_val.empty())
 775                 res.push_back(new_val);
 776 
 777         return res;
 778 }

 780 std::pair<int, int> parse_range(const std::string& str)
 781 {
 782         const std::string::const_iterator dash = std::find(str.begin(), str.end(), '-');
 783         const std::string a(str.begin(), dash);
 784         const std::string b = dash != str.end() ? std::string(dash + 1, str.end()) : a;
 785         std::pair<int,int> res {0,0};
 786         try {
 787                 res = std::make_pair(std::stoi(a), std::stoi(b));
 788                 if (res.second < res.first) {
 789                         res.second = res.first;
 790                 }
 791         } catch(std::invalid_argument) {}
 792 
 793         return res;
 794 }

 796 std::vector<std::pair<int, int>> parse_ranges(const std::string& str)
 797 {
 798         std::vector< std::pair< int, int > > to_return;
 799         std::vector<std::string> strs = utils::split(str);
 800         std::vector<std::string>::const_iterator i, i_end=strs.end();
 801         for(i = strs.begin(); i != i_end; ++i) {
 802                 to_return.push_back(parse_range(*i));
 803         }
 804         return to_return;
 805 }
