 101 void mouse_handler::mouse_motion(int x, int y, const bool browse, bool update, map_location new_hex)
 102 {
 103         // we ignore the position coming from event handler
 104         // because it's always a little obsolete and we don't need
 105         // to highlight all the hexes where the mouse passed.
 106         // Also, sometimes it seems to have one *very* obsolete
 107         // and isolated mouse motion event when using drag&drop
 108         SDL_GetMouseState(&x,&y);  // <-- modify x and y
 109 
 110         if (mouse_handler_base::mouse_motion_default(x, y, update)) return;
 111 
 112         game_board & board = pc_.gamestate().board_;
 113 
 114         if (new_hex == map_location::null_location())
 115                 new_hex = gui().hex_clicked_on(x,y);
 116 
 117         if(new_hex != last_hex_) {
 118                 if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 119                         lk->mouse_over_hex_callback(new_hex);
 120                 }
 121                 update = true;
 122                 if ( pc_.get_map_const().on_board(last_hex_) ) {
 123                         // we store the previous hexes used to propose attack direction
 124                         previous_hex_ = last_hex_;
 125                         // the hex of the selected unit is also "free"
 126                         { // start planned unit map scope
 127                                 wb::future_map_if_active raii;
 128                                 if (last_hex_ == selected_hex_ || !find_unit(last_hex_)) {
 129                                         previous_free_hex_ = last_hex_;
 130                                 }
 131                         } // end planned unit map scope
 132                 }
 133                 last_hex_ = new_hex;
 134         }
 135 
 136         if (reachmap_invalid_) update = true;
 137 
 138         if (!update) return;
 139 
 140         if (reachmap_invalid_) {
 141                 reachmap_invalid_ = false;
 142                 if (!current_paths_.destinations.empty() && !show_partial_move_) {
 143                         bool selected_hex_has_unit;
 144                         { // start planned unit map scope
 145                                 wb::future_map_if_active planned_unit_map;
 146                                 selected_hex_has_unit = find_unit(selected_hex_).valid();
 147                         } // end planned unit map scope
 148                         if(selected_hex_.valid() && selected_hex_has_unit ) {
 149                                 // reselect the unit without firing events (updates current_paths_)
 150                                 select_hex(selected_hex_, true);
 151                         }
 152                         // we do never deselect here, mainly because of canceled attack-move
 153                 }
 154         }
 155 
 156         // reset current_route_ and current_paths if not valid anymore
 157         // we do it before cursor selection, because it uses current_paths_
 158         if( !pc_.get_map_const().on_board(new_hex) ) {
 159                 current_route_.steps.clear();
 160                 gui().set_route(nullptr);
 161                 pc_.get_whiteboard()->erase_temp_move();
 162         }
 163 
 164         if(unselected_paths_) {
 165                 unselected_paths_ = false;
 166                 current_paths_ = pathfind::paths();
 167                 gui().unhighlight_reach();
 168         } else if(over_route_) {
 169                 over_route_ = false;
 170                 current_route_.steps.clear();
 171                 gui().set_route(nullptr);
 172                 pc_.get_whiteboard()->erase_temp_move();
 173         }
 174 
 175         gui().highlight_hex(new_hex);
 176         pc_.get_whiteboard()->on_mouseover_change(new_hex);
 177 
 178         unit_map::iterator selected_unit;
 179         unit_map::iterator mouseover_unit;
 180         map_location attack_from;
 181 
 182         { // start planned unit map scope
 183                 wb::future_map_if_active planned_unit_map;
 184                 selected_unit = find_unit(selected_hex_);
 185                 mouseover_unit = find_unit(new_hex);
 186 
 187                 // we search if there is an attack possibility and where
 188                 attack_from = current_unit_attacks_from(new_hex);
 189 
 190                 //see if we should show the normal cursor, the movement cursor, or
 191                 //the attack cursor
 192                 //If the cursor is on WAIT, we don't change it and let the setter
 193                 //of this state end it
 194                 if (cursor::get() != cursor::WAIT) {
 195                         if (selected_unit &&
 196                                         selected_unit->side() == side_num_ &&
 197                                         !selected_unit->incapacitated() && !browse)
 198                         {
 199                                 if (attack_from.valid()) {
 200                                         cursor::set(dragging_started_ ? cursor::ATTACK_DRAG : cursor::ATTACK);
 201                                 }
 202                                 else if (!mouseover_unit &&
 203                                                 current_paths_.destinations.contains(new_hex))
 204                                 {
 205                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 206                                 } else {
 207                                         // selected unit can't attack or move there
 208                                         cursor::set(cursor::NORMAL);
 209                                 }
 210                         } else {
 211                                 // no selected unit or we can't move it
 212 
 213                                 if ( selected_hex_.valid() && mouseover_unit
 214                                                 && mouseover_unit->side() == side_num_ ) {
 215                                         // empty hex field selected and unit on our site under the cursor
 216                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 217                                 } else {
 218                                         cursor::set(cursor::NORMAL);
 219                                 }
 220                         }
 221                 }
 222         } // end planned unit map scope
 223 
 224         // show (or cancel) the attack direction indicator
 225         if (attack_from.valid() && (!browse || pc_.get_whiteboard()->is_active())) {
 226                 gui().set_attack_indicator(attack_from, new_hex);
 227         } else {
 228                 gui().clear_attack_indicator();
 229         }
 230 
 231         unit_ptr un; //will later point to unit at mouseover_hex_
 232 
 233         // the destination is the pointed hex or the adjacent hex
 234         // used to attack it
 235         map_location dest;
 236         unit_map::const_iterator dest_un;
 237         { // start planned unit map scope
 238                 wb::future_map_if_active raii;
 239                 if (attack_from.valid()) {
 240                         dest = attack_from;
 241                         dest_un = find_unit(dest);
 242                 }       else {
 243                         dest = new_hex;
 244                         dest_un = find_unit(new_hex);
 245                 }
 246 
 247                 if(dest == selected_hex_ || dest_un) {
 248                         current_route_.steps.clear();
 249                         gui().set_route(nullptr);
 250                         pc_.get_whiteboard()->erase_temp_move();
 251                 }
 252                 else if (!current_paths_.destinations.empty() &&
 253                                 board.map().on_board(selected_hex_) && board.map().on_board(new_hex))
 254                 {
 255                         if (selected_unit && !selected_unit->incapacitated()) {
 256                                 // Show the route from selected unit to mouseover hex
 257                                 current_route_ = get_route(&*selected_unit, dest, viewing_team());
 258 
 259                                 pc_.get_whiteboard()->create_temp_move();
 260 
 261                                 if(!browse) {
 262                                         gui().set_route(&current_route_);
 263                                 }
 264                         }
 265                 }
 266 
 267                 if(board.map().on_board(selected_hex_)
 268                                 && !selected_unit
 269                                 && mouseover_unit.valid()
 270                                 && mouseover_unit) {
 271                         // Show the route from selected hex to mouseover unit
 272                         current_route_ = get_route(&*mouseover_unit, selected_hex_, viewing_team());
 273 
 274                         pc_.get_whiteboard()->create_temp_move();
 275 
 276                         if(!browse) {
 277                                 gui().set_route(&current_route_);
 278                         }
 279                 } else if (!selected_unit) {
 280                         current_route_.steps.clear();
 281                         gui().set_route(nullptr);
 282                         pc_.get_whiteboard()->erase_temp_move();
 283                 }
 284 
 285                 unit_map::iterator iter = mouseover_unit;
 286                 if (iter)
 287                         un = iter.get_shared_ptr();
 288                 else
 289                         un.reset();
 290         } //end planned unit map scope
 291 
 292         if ( (!selected_hex_.valid()) && un && current_paths_.destinations.empty() &&
 293                         !gui().fogged(un->get_location()))
 294         {
 295                 if (un->side() == side_num_) {
 296                         //unit is on our team, show path if the unit has one
 297                         const map_location go_to = un->get_goto();
 298                         if(board.map().on_board(go_to)) {
 299                                 pathfind::marked_route route;
 300                                 { // start planned unit map scope
 301                                         wb::future_map_if_active raii;
 302                                         route = get_route(un.get(), go_to, current_team());
 303                                 } // end planned unit map scope
 304                                 gui().set_route(&route);
 305                         }
 306                         over_route_ = true;
 307 
 308                         wb::future_map_if_active raii;
 309                         current_paths_ = pathfind::paths(*un, false, true,
 310                                         viewing_team(), path_turns_);
 311                 } else {
 312                         //unit under cursor is not on our team
 313                         //Note: planned unit map must be activated after this is done,
 314                         //since the future state includes changes to units' movement.
 315                         unit_movement_resetter move_reset(*un);
 316 
 317                         wb::future_map_if_active raii;
 318                         current_paths_ = pathfind::paths(*un, false, true,
 319                                         viewing_team(), path_turns_);
 320                 }
 321 
 322                 unselected_paths_ = true;
 323                 gui().highlight_reach(current_paths_);
 324         }
 325 
 326 }

 352 map_location mouse_handler::current_unit_attacks_from(const map_location& loc) const
 353 {
 354         if(loc == selected_hex_)
 355                 return map_location();
 356 
 357         bool wb_active = pc_.get_whiteboard()->is_active();
 358 
 359         {
 360                 // Check the unit SOURCE of the attack
 361 
 362                 // Check that there's a selected unit
 363                 const unit_map::const_iterator source_unit = find_unit(selected_hex_);
 364                 bool source_eligible = source_unit.valid();
 365                 if (!source_eligible) return map_location();
 366 
 367                 // The selected unit must at least belong to the player currently controlling this client.
 368                 source_eligible &= source_unit->side() == gui_->viewing_side();
 369                 if (!source_eligible) return map_location();
 370 
 371                 // In addition:
 372                 // - If whiteboard is enabled, we allow planning attacks outside of player's turn
 373                 // - If whiteboard is disabled, it must be the turn of the player controlling this client
 374                 if(!wb_active) {
 375                         source_eligible &= gui_->viewing_side() == pc_.current_side();
 376                         if (!source_eligible) return map_location();
 377                 }
 378 
 379                 // Unit must have attacks left
 380                 source_eligible &= source_unit->attacks_left() != 0;
 381                 if (!source_eligible) return map_location();
 382 
 383 
 384                 // Check the unit TARGET of the attack
 385 
 386                 team const& viewer = viewing_team();
 387 
 388                 // Check that there's a unit at the target location
 389                 const unit_map::const_iterator target_unit = find_unit(loc);
 390                 bool target_eligible = target_unit.valid();
 391                 if (!target_eligible) return map_location();
 392 
 393                 // The player controlling this client must be an enemy of the target unit's side
 394                 target_eligible &= viewer.is_enemy(target_unit->side());
 395                 if (!target_eligible) return map_location();
 396 
 397                 // Sanity check: source and target of the attack shouldn't be on the same team
 398                 assert(source_unit->side() != target_unit->side());
 399 
 400                 target_eligible &= !target_unit->incapacitated();
 401                 if (!target_eligible) return map_location();
 402         }
 403 
 404         const map_location::DIRECTION preferred = loc.get_relative_dir(previous_hex_);
 405         const map_location::DIRECTION second_preferred = loc.get_relative_dir(previous_free_hex_);
 406 
 407         int best_rating = 100;//smaller is better
 408         map_location res;
 409         map_location adj[6];
 410         get_adjacent_tiles(loc,adj);
 411 
 412         for(size_t n = 0; n != 6; ++n) {
 413                 if(pc_.gamestate().board_.map().on_board(adj[n]) == false) {
 414                         continue;
 415                 }
 416 
 417                 if(adj[n] != selected_hex_ && find_unit(adj[n])) {
 418                         continue;
 419                 }
 420 
 421                 if (current_paths_.destinations.contains(adj[n]))
 422                 {
 423                         static const size_t NDIRECTIONS = map_location::NDIRECTIONS;
 424                         unsigned int difference = std::abs(int(preferred - n));
 425                         if(difference > NDIRECTIONS/2) {
 426                                 difference = NDIRECTIONS - difference;
 427                         }
 428                         unsigned int second_difference = std::abs(int(second_preferred - n));
 429                         if(second_difference > NDIRECTIONS/2) {
 430                                 second_difference = NDIRECTIONS - second_difference;
 431                         }
 432                         const int rating = difference * 2 + (second_difference > difference);
 433                         if(rating < best_rating || res.valid() == false) {
 434                                 best_rating = rating;
 435                                 res = adj[n];
 436                         }
 437                 }
 438         }
 439 
 440         return res;
 441 }

 443 pathfind::marked_route mouse_handler::get_route(const unit* un, map_location go_to, team &team) const
 444 {
 445         game_board & board = pc_.gamestate().board_;
 446 
 447         // The pathfinder will check unit visibility (fogged/stealthy).
 448         const pathfind::shortest_path_calculator calc(*un, team, board.teams(), board.map());
 449 
 450         pathfind::teleport_map allowed_teleports = pathfind::get_teleport_locations(*un, viewing_team());
 451 
 452         pathfind::plain_route route;
 453 
 454         route = pathfind::a_star_search(un->get_location(), go_to, 10000.0, calc, board.map().w(), board.map().h(), &allowed_teleports);
 455 
 456         return mark_route(route);
 457 }

 470 void mouse_handler::select_or_action(bool browse)
 471 {
 472         if (!pc_.get_map_const().on_board(last_hex_)){
 473                 tooltips::click(drag_from_x_, drag_from_y_);
 474                 return;
 475         }
 476 
 477         // Load whiteboard partial moves
 478         wb::future_map_if_active planned_unit_map;
 479 
 480         unit_map::iterator clicked_u = find_unit(last_hex_);
 481         unit_map::iterator selected_u = find_unit(selected_hex_);
 482         if ( clicked_u &&
 483                  (!selected_u ||
 484                   selected_u->side() != side_num_ ||
 485                   (clicked_u->side() == side_num_ && clicked_u->id() != selected_u->id())) )
 486         {
 487                 select_hex(last_hex_, false);
 488         }
 489         else
 490         {
 491                 move_action(browse);
 492         }
 493 }

 495 void mouse_handler::move_action(bool browse)
 496 {
 497         // Lock whiteboard activation state to avoid problems due to
 498         // its changing while an animation takes place.
 499         wb::whiteboard_lock wb_lock = pc_.get_whiteboard()->get_activation_state_lock();
 500 
 501         //we use the last registered highlighted hex
 502         //since it's what update our global state
 503         map_location hex = last_hex_;
 504 
 505         // TODO
 506         //      // Clicks on border hexes mean to deselect.
 507         //      // (Check this before doing processing that might not be needed.)
 508         //      if ( !pc_.get_map_const().on_board(hex) ) {
 509         //              deselect_hex();
 510         //              return false;
 511         //      }
 512         if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 513                 lk->select_hex_callback(last_hex_);
 514         }
 515         unit_map::iterator u;
 516         unit_map::iterator clicked_u;
 517         map_location src;
 518         pathfind::paths orig_paths;
 519         map_location attack_from;
 520         { // start planned unit map scope
 521                 wb::future_map_if_active planned_unit_map;
 522                 u = find_unit(selected_hex_);
 523 
 524                 //if the unit is selected and then itself clicked on,
 525                 //any goto command is canceled
 526                 if (u && !browse && selected_hex_ == hex && u->side() == side_num_) {
 527                         u->set_goto(map_location());
 528                 }
 529 
 530                 clicked_u = find_unit(hex);
 531 
 532                 src = selected_hex_;
 533                 orig_paths = current_paths_;
 534                 attack_from = current_unit_attacks_from(hex);
 535         } // end planned unit map scope
 536 
 537         //see if we're trying to do a attack or move-and-attack
 538         if((!browse || pc_.get_whiteboard()->is_active()) && attack_from.valid()) {
 539 
 540                 // Ignore this command if commands are disabled.
 541                 if ( commands_disabled )
 542                         return;
 543 
 544                 if (((u.valid() && u->side() == side_num_) || pc_.get_whiteboard()->is_active()) && clicked_u.valid() ) {
 545                         if (attack_from == selected_hex_) { //no move needed
 546                                 int choice = -1;
 547                                 { wb::future_map_if_active planned_unit_map; //start planned unit map scope
 548                                         choice = show_attack_dialog(attack_from, clicked_u->get_location());
 549                                 } // end planned unit map scope
 550                                 if (choice >=0 ) {
 551                                         if (pc_.get_whiteboard()->is_active()) {
 552                                                 save_whiteboard_attack(attack_from, clicked_u->get_location(), choice);
 553                                         } else {
 554                                                 // clear current unit selection so that any other unit selected
 555                                                 // triggers a new selection
 556                                                 selected_hex_ = map_location();
 557 
 558                                                 attack_enemy(u->get_location(), clicked_u->get_location(), choice);
 559                                         }
 560                                 }
 561                                 return;
 562                         }
 563                         else {
 564 
 565                                 int choice = -1; //for the attack dialog
 566 
 567                                 { wb::future_map_if_active planned_unit_map; //start planned unit map scope
 568                                         // we will now temporary move next to the enemy
 569                                         pathfind::paths::dest_vect::const_iterator itor =
 570                                                         current_paths_.destinations.find(attack_from);
 571                                         if(itor == current_paths_.destinations.end()) {
 572                                                 // can't reach the attacking location
 573                                                 // not supposed to happen, so abort
 574                                                 return;
 575                                         }
 576 
 577                                         // block where we temporary move the unit
 578                                         {
 579                                                 temporary_unit_mover temp_mover(pc_.gamestate().board_.units_, src, attack_from,
 580                                                                                 itor->move_left);
 581                                                 choice = show_attack_dialog(attack_from, clicked_u->get_location());
 582                                         }
 583 
 584                                         if (choice < 0) {
 585                                                 // user hit cancel, don't start move+attack
 586                                                 return;
 587                                         }
 588                                 } // end planned unit map scope
 589 
 590                                 if (pc_.get_whiteboard()->is_active()) {
 591                                         save_whiteboard_attack(attack_from, hex, choice);
 592                                 }
 593                                 else {
 594                                         bool not_interrupted = move_unit_along_current_route();
 595                                         bool alt_unit_selected = (selected_hex_ != src);
 596                                         src = selected_hex_;
 597                                         // clear current unit selection so that any other unit selected
 598                                         // triggers a new selection
 599                                         selected_hex_ = map_location();
 600 
 601                                         if (not_interrupted)
 602                                                 attack_enemy(attack_from, hex, choice); // Fight !!
 603 
 604                                         //TODO: Maybe store the attack choice so "press t to continue"
 605                                         //      can also continue the attack?
 606 
 607                                         if (alt_unit_selected && !selected_hex_.valid()) {
 608                                                 //reselect other unit if selected during movement animation
 609                                                 select_hex(src, browse);
 610                                         }
 611                                 }
 612 
 613                                 return;
 614                         }
 615                 }
 616         }
 617         //otherwise we're trying to move to a hex
 618         else if (
 619                         // the old use case: move selected unit to mouse hex field
 620                         ( (!browse || pc_.get_whiteboard()->is_active()) &&
 621                                         selected_hex_.valid() && selected_hex_ != hex &&
 622                                         u.valid() &&
 623                                         (u->side() == side_num_ || pc_.get_whiteboard()->is_active()) &&
 624                                         !clicked_u &&
 625                                         !current_route_.steps.empty() &&
 626                                         current_route_.steps.front() == selected_hex_
 627                         )
 628                         || // the new use case: move mouse unit to selected hex field
 629                         ( (!browse || pc_.get_whiteboard()->is_active()) &&
 630                                         selected_hex_.valid() && selected_hex_ != hex &&
 631                                         clicked_u &&
 632                                         !current_route_.steps.empty() &&
 633                                         current_route_.steps.back() == selected_hex_
 634                                         && !u
 635                                         && clicked_u->side() == side_num_
 636                         )
 637         ) {
 638 
 639                 // Ignore this command if commands are disabled.
 640                 if ( commands_disabled )
 641                         return;
 642 
 643                 // If the whiteboard is active, it intercepts any unit movement
 644                 if (pc_.get_whiteboard()->is_active()) {
 645                                 // deselect the current hex, and create planned move for whiteboard
 646                                 selected_hex_ = map_location();
 647                                 gui().select_hex(map_location());
 648                                 gui().clear_attack_indicator();
 649                                 gui().set_route(nullptr);
 650                                 show_partial_move_ = false;
 651                                 gui().unhighlight_reach();
 652                                 current_paths_ = pathfind::paths();
 653                                 current_route_.steps.clear();
 654 
 655                                 pc_.get_whiteboard()->save_temp_move();
 656 
 657                 // Otherwise proceed to normal unit movement
 658                 } else {
 659                         //Don't move if the unit already has actions
 660                         //from the whiteboard.
 661                         if (pc_.get_whiteboard()->unit_has_actions(
 662                                         u ? &*u : &*clicked_u )) {
 663                                 return;
 664                         }
 665 
 666                         move_unit_along_current_route();
 667                         // during the move, we may have selected another unit
 668                         // (but without triggering a select event (command was disabled)
 669                         // in that case reselect it now to fire the event (+ anim & sound)
 670                         if (selected_hex_ != src) {
 671                                 select_hex(selected_hex_, browse);
 672                         }
 673                 }
 674                 return;
 675         }
 676 }

 678 void mouse_handler::select_hex(const map_location& hex, const bool browse, const bool highlight, const bool fire_event) {

 695

 717                                         // ensure unit map is back to normal while event is fired
 718                                         wb::real_map srum;
 719                                         pc_.pump().fire("select", hex);
 720                                         //end forced real unit map
 721                                 }
 722                         }
 723                 }
 724                 return;
 725         }
 726 
 727         if (selected_hex_.valid() && !unit) {
 728                 // compute unit in range of the empty selected_hex field
 729 
 730                 gui_->unhighlight_reach();
 731 
 732                 pathfind::paths reaching_unit_locations;
 733 
 734                 pathfind::paths clicked_location;
 735                 clicked_location.destinations.insert(hex);
 736 
 737                 for(unit_map::iterator u = pc_.gamestate().board_.units_.begin(); u != pc_.gamestate().board_.units_.end(); ++u) {
 738                         bool invisible = u->invisible(u->get_location(), gui_->get_disp_context());
 739 
 740                         if (!gui_->fogged(u->get_location()) && !u->incapacitated() && !invisible)
 741                         {
 742

 744                                                 viewing_team(), path_turns_, false, false);

 746                                 if (path.destinations.find(hex) != path.destinations.end()) {
 747                                         reaching_unit_locations.destinations.insert(u->get_location());
 748                                         gui_->highlight_another_reach(clicked_location);
 749                                 }
 750                         }
 751                 }
 752                 gui_->highlight_another_reach(reaching_unit_locations);
 753         } else {
 754                 if (!pc_.gamestate().board_.units_.find(last_hex_)) {
 755                         gui_->unhighlight_reach();
 756                 }
 757                 current_paths_ = pathfind::paths();
 758                 current_route_.steps.clear();
 759 
 760                 pc_.get_whiteboard()->on_deselect_hex();
 761         }

 768 /**
 769  * Moves a unit along the currently cached route.
 770  *
 771  * @returns  true if the end of the route was reached and no information was
 772  *           uncovered that would warrant interrupting a chain of actions;
 773  *           false otherwise.
 774  */
 775 bool mouse_handler::move_unit_along_current_route()
 776 {
 777         // Copy the current route to ensure it remains valid throughout the animation.
 778         const std::vector<map_location> steps = current_route_.steps;
 779 
 780         // do not show footsteps during movement
 781         gui().set_route(nullptr);
 782         gui().unhighlight_reach();
 783 
 784         // do not keep the hex highlighted that we started from
 785         selected_hex_ = map_location();
 786         gui().select_hex(map_location());
 787 
 788         bool interrupted = false;
 789         if ( steps.size() > 1 )
 790         {
 791                 size_t num_moves = move_unit_along_route(steps, interrupted);
 792 
 793                 interrupted =  interrupted || num_moves + 1 < steps.size();
 794                 next_unit_ = steps[num_moves];
 795         }
 796 
 797         // invalid after the move
 798         current_paths_ = pathfind::paths();
 799         current_route_.steps.clear();
 800 
 801         return !interrupted;
 802 }

 804 /**
 805  * Moves a unit across the board for a player.
 806  * This is specifically for movement at the time it is initiated by a player,
 807  * whether via a mouse click or executing whiteboard actions. Continued moves
 808  * (including goto execution) can bypass this and call actions::move_unit() directly.
 809  * This function call may include time for an animation, so make sure the
 810  * provided route will remain unchanged (the caller should probably make a local
 811  * copy).
 812  *
 813  * @param[in]   steps           The route to be traveled. The unit to be moved is at the beginning of this route.
 814  * @param[out]  interrupted     This is set to true if information was uncovered that warrants interrupting a chain of actions (and set to false otherwise).
 815  *
 816  * @returns The number of hexes entered. This can safely be used as an index
 817  *          into steps to get the location where movement ended, provided
 818  *          steps is not empty (the return value is guaranteed to be less
 819  *          than steps.size() ).
 820  */
 821 size_t mouse_handler::move_unit_along_route(const std::vector<map_location> & steps,
 822                                             bool & interrupted)
 823 {
 824         if(steps.empty()) {
 825                 interrupted = false;
 826                 return 0;
 827         }
 828 
 829         // Default return value.
 830         interrupted = true;
 831 
 832         //If this is a leader on a keep, ask permission to the whiteboard to move it
 833         //since otherwise it may cause planned recruits to be erased.
 834         if ( pc_.get_map_const().is_keep(steps.front()) )
 835         {
 836                 unit_map::const_iterator const u = pc_.gamestate().board_.units().find(steps.front());
 837 
 838                 if ( u &&  u->can_recruit()  &&
 839                      u->side() == gui().viewing_side()        &&
 840                      !pc_.get_whiteboard()->allow_leader_to_move(*u) )
 841                 {
 842                         gui2::show_transient_message(gui_->video(), "",
 843                                 _("You cannot move your leader away from the keep with some planned recruits or recalls left."));
 844                         return 0;
 845                 }
 846         }
 847 
 848         LOG_NG << "move unit along route  from " << steps.front() << " to " << steps.back() << "\n";
 849         size_t moves = actions::move_unit_and_record(steps, &pc_.get_undo_stack(), false, true, &interrupted);
 850 
 851         cursor::set(cursor::NORMAL);
 852         gui().invalidate_game_status();
 853 
 854         if(moves == 0)
 855                 return 0;
 856 
 857         if ( interrupted  &&  moves + 1 < steps.size() ) {
 858                 // reselect the unit (for "press t to continue")
 859                 select_hex(steps[moves], false, false, false);
 860                 // the new discovery is more important than the new movement range
 861                 show_partial_move_ = true;
 862         }
 863 
 864         return moves;
 865 }

 898 int mouse_handler::fill_weapon_choices(std::vector<battle_context>& bc_vector, unit_map::iterator attacker, unit_map::iterator defender)
 899 {
 900         int best = 0;
 901         for (unsigned int i = 0; i < attacker->attacks().size(); i++) {
 902                 // skip weapons with attack_weight=0
 903                 if (attacker->attacks()[i].attack_weight() > 0) {
 904                         battle_context bc(pc_.gamestate().board_.units_, attacker->get_location(), defender->get_location(), i);
 905                         if (!bc_vector.empty() && bc.better_attack(bc_vector[best], 0.5)) {
 906                                 // as some weapons can be hidden, i is not a valid index into the resulting vector
 907                                 best = bc_vector.size();
 908                         }
 909                         bc_vector.push_back(bc);
 910                 }
 911         }
 912         return best;
 913 }

 941                           attacker

1035 /**
1036  * Causes attackable hexes to be highlighted.
1037  *
1038  * This checks the hexes that the provided unit can attack. If there is a valid
1039  * target there, that location is inserted into current_paths_.destinations.
1040  */
1041 void mouse_handler::show_attack_options(const unit_map::const_iterator &u)
1042 {
1043         // Cannot attack if no attacks are left.
1044         if (u->attacks_left() == 0)
1045               return;
1046 
1047         // Get the teams involved.
1048         const team & cur_team = current_team();
1049         const team & u_team = pc_.gamestate().board_.teams_[u->side()-1];
1050 
1051         // Check each adjacent hex.
1052         map_location adj[6];
1053         get_adjacent_tiles(u->get_location(), adj);
1054         for(const map_location &loc : adj)
1055         {
1056                 // No attack option shown if no visible unit present.
1057                 // (Visible to current team, not necessarily the unit's team.)
1058                 if (!pc_.get_map_const().on_board(loc)) continue;
1059                 unit_map::const_iterator i = pc_.gamestate().board_.units().find(loc);
1060                 if ( !i ||  !i->is_visible_to_team(cur_team, gui_->get_disp_context(), false) )
1061                         continue;
1062                 const unit &target = *i;
1063                 // Can only attack non-petrified enemies.
1064                 if ( u_team.is_enemy(target.side())  &&  !target.incapacitated() )
1065                         current_paths_.destinations.insert(loc);
1066         }
1067 }

1069 bool mouse_handler::unit_in_cycle(unit_map::const_iterator it)
1070 {
1071         game_board & board = pc_.gamestate().board_;
1072 
1073         if (!it) {
1074                 return false;
1075         }
1076 
1077         if (it->side() != side_num_ || it->user_end_turn()
1078             || gui().fogged(it->get_location()) || !board.unit_can_move(*it))
1079                 return false;
1080 
1081         if (current_team().is_enemy(int(gui().viewing_team()+1)) &&
1082             it->invisible(it->get_location(), gui().get_disp_context()))
1083                 return false;
1084 
1085         if (it->get_hidden())
1086                 return false;
1087 
1088         return true;

1092 void mouse_handler::cycle_units(const bool browse, const bool reverse)
1093 {
1094         game_board & board = pc_.gamestate().board_;
1095 
1096         if (board.units().begin() == board.units().end()) {
1097                 return;
1098         }
1099 
1100         unit_map::const_iterator it = find_unit(next_unit_);
1101         if (!it)
1102                 it = board.units().begin();
1103         const unit_map::const_iterator itx = it;
1104 
1105         do {
1106                 if (reverse) {
1107                         if (it == board.units().begin())
1108                                 it = board.units().end();
1109                         --it;
1110                 } else {
1111                         if (it == board.units().end())
1112                                 it = board.units().begin();
1113                         else
1114                                 ++it;
1115                 }
1116         } while (it != itx && !unit_in_cycle(it));
1117 
1118         if (unit_in_cycle(it)) {
1119                 gui().scroll_to_tile(it->get_location(), game_display::WARP);
1120                 select_hex(it->get_location(), browse);
1121 //              mouse_update(browse);
1122         }
1123 }
