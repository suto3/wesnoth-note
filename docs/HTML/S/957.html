 308 void team::build(const config &cfg, const gamemap& map, int gold)
 309 {
 310         gold_ = gold;
 311         info_.read(cfg);
 312 
 313         fog_.set_enabled(cfg["fog"].to_bool());
 314         fog_.read(cfg["fog_data"]);
 315         shroud_.set_enabled(cfg["shroud"].to_bool());
 316         shroud_.read(cfg["shroud_data"]);
 317         auto_shroud_updates_ = cfg["auto_shroud"].to_bool(auto_shroud_updates_);
 318 
 319         LOG_NG << "team::team(...): team_name: " << info_.team_name
 320                << ", shroud: " << uses_shroud() << ", fog: " << uses_fog() << ".\n";
 321 
 322         // Load the WML-cleared fog.
 323         const config &fog_override = cfg.child("fog_override");
 324         if ( fog_override ) {
 325                 const std::vector<map_location> fog_vector =
 326                         map.parse_location_range(fog_override["x"], fog_override["y"], true);
 327                 fog_clearer_.insert(fog_vector.begin(), fog_vector.end());
 328         }
 329 
 330         // To ensure some minimum starting gold,
 331         // gold is the maximum of 'gold' and what is given in the config file
 332         gold_ = std::max(gold, info_.gold);
 333         if (gold_ != info_.gold)
 334                 info_.start_gold = gold;
 335         // Old code was doing:
 336         // info_.start_gold = std::to_string(gold) + " (" + info_.start_gold + ")";
 337         // Was it correct?
 338 
 339         // Load in the villages the side controls at the start
 340         for (const config &v : cfg.child_range("village"))
 341         {
 342                 map_location loc(v);
 343                 if (map.is_village(loc)) {
 344                         villages_.insert(loc);
 345                 } else {
 346                         WRN_NG << "[side] " << current_player() << " [village] points to a non-village location " << loc << std::endl;
 347                 }
 348         }
 349 
 350         countdown_time_ = cfg["countdown_time"];
 351         action_bonus_count_ = cfg["action_bonus_count"];
 352 
 353         planned_actions_.reset(new wb::side_actions());
 354         planned_actions_->set_team_index(info_.side - 1);
 355 }

 379 bool team::get_village(const map_location& loc, const int owner_side, game_data * gamedata)
 380 {
 381         villages_.insert(loc);
 382         bool gamestate_changed = false;
 383         if(gamedata) {
 384                 config::attribute_value& var = gamedata->get_variable("owner_side");
 385                 const config::attribute_value old_value = var;
 386                 var = owner_side;
 387 
 388                 // During team building, game_events pump is not guaranteed to exist yet. (At current revision.) We skip capture events in this case.
 389                 if (resources::game_events) {
 390                         gamestate_changed = resources::game_events->pump().fire("capture",loc);
 391                 }
 392 
 393                 if(old_value.blank())
 394                         gamedata->clear_variable("owner_side");
 395                 else
 396                         var = old_value;
 397         }
 398         return gamestate_changed;
 399 }

 422 int team::minimum_recruit_price() const
 423 {
 424         if(info_.minimum_recruit_price){
 425                 return info_.minimum_recruit_price;
 426         }else{
 427                 int min = 20;
 428                 for(std::string recruit : info_.can_recruit) {
 429                         const unit_type *ut = unit_types.find(recruit);
 430                         if(!ut)
 431                                 continue;
 432                         else{
 433                                 if(ut->cost() < min)
 434                                         min = ut->cost();
 435                         }
 436 
 437                 }
 438                 info_.minimum_recruit_price = min;
 439         }
 440         return info_.minimum_recruit_price;

 454 bool team::calculate_is_enemy(size_t index) const
 455 {
 456         // We're not enemies of ourselves
 457         if(&resources::gameboard->teams()[index] == this) {
 458                 return false;
 459         }
 460 
 461         // We are friends with anyone who we share a teamname with
 462         std::vector<std::string> our_teams = utils::split(info_.team_name),
 463                                                    their_teams = utils::split(resources::gameboard->teams()[index].info_.team_name);
 464 
 465         LOG_NGE << "team " << info_.side << " calculates if it has enemy in team "<<index+1 << "; our team_name ["<<info_.team_name<<"], their team_name is ["<<resources::gameboard->teams()[index].info_.team_name<<"]"<< std::endl;
 466         for(std::vector<std::string>::const_iterator t = our_teams.begin(); t != our_teams.end(); ++t) {
 467                 if(std::find(their_teams.begin(), their_teams.end(), *t) != their_teams.end())
 468                 {
 469                         LOG_NGE << "team " << info_.side << " found same team name [" << *t << "] in team "<< index+1 << std::endl;
 470                         return false;
 471                 } else {
 472                         LOG_NGE << "team " << info_.side << " not found same team name [" << *t << "] in team "<< index+1 << std::endl;
 473                 }
 474         }
 475         LOG_NGE << "team " << info_.side << " has enemy in team " << index+1 << std::endl;
 476         return true;
 477 }

 512 void team::change_controller_by_wml(const std::string& new_controller_string)
 513 {
 514         CONTROLLER new_controller;
 515         if(!new_controller.parse(new_controller_string)) {
 516                 WRN_NG << "ignored attempt to change controller to " << new_controller_string << std::endl;
 517                 return;
 518         }
 519         if(new_controller == CONTROLLER::EMPTY && resources::controller->current_side() == this->side()) {
 520                 WRN_NG << "ignored attempt to change the currently playing side's controller to 'null'" << std::endl;
 521                 return;
 522         }
 523         config choice = synced_context::ask_server_choice(controller_server_choice(new_controller, *this));
 524         if(!new_controller.parse(choice["controller"])) {
 525                 //TODO: this should be more than a ERR_NG message.
 526                 // GL-2016SEP02 Oh? So why was ERR_NG defined as warning level? Making the call fit the definition.
 527                 WRN_NG << "recieved an invalid controller string from the server" << choice["controller"] << std::endl;
 528         }
 529         if(!resources::controller->is_replay()) {
 530                 set_local(choice["is_local"].to_bool());
 531         }
 532         change_controller(new_controller);
 533 }

 576 bool team::fogged(const map_location& loc) const
 577 {
 578         if(shrouded(loc)) return true;
 579 
 580         // Check for an override of fog.

 582                 return false;

 585                 return fog_.value(loc.wml_x(),loc.wml_y());
 586 
 587         return fog_.shared_value(ally_fog(resources::gameboard->teams()),loc.wml_x(),loc.wml_y());
 588 }

 592         if(ally_shroud_.empty()) {

 596                         }

 598         }

 616 bool team::knows_about_team(size_t index) const
 617 {
 618         const team& t = resources::gameboard->teams()[index];
 619 
 620         // We know about our own team
 621         if(this == &t) return true;
 622 
 623         // If we aren't using shroud or fog, then we know about everyone
 624         if(!uses_shroud() && !uses_fog()) return true;
 625 
 626         // We don't know about enemies
 627         if(is_enemy(index+1)) return false;
 628 
 629         // We know our human allies.
 630         if (t.is_human()) return true;
 631 
 632         // We know about allies we're sharing maps with
 633         if(share_maps() && t.uses_shroud()) return true;
 634 
 635         // We know about allies we're sharing view with
 636         if(share_view() && (t.uses_fog() || t.uses_shroud())) return true;
 637 
 638         return false;
 639 }

 641 /**
 642  * Removes the record of hexes that were cleared of fog via WML.
 643  * @param[in] hexes     The hexes to no longer keep clear.
 644  */
 645 void team::remove_fog_override(const std::set<map_location> &hexes)
 646 {
 647         // Take a set difference.
 648         std::vector<map_location> result(fog_clearer_.size());
 649         std::vector<map_location>::iterator result_end =
 650                 std::set_difference(fog_clearer_.begin(), fog_clearer_.end(),
 651                                     hexes.begin(), hexes.end(), result.begin());
 652 
 653         // Put the result into fog_clearer_.
 654         fog_clearer_.clear();
 655         fog_clearer_.insert(result.begin(), result_end);
 656 }

 669 bool team::shroud_map::clear(int x, int y)
 670 {
 671         if(enabled_ == false || x < 0 || y < 0)
 672                 return false;
 673 
 674         if(x >= static_cast<int>(data_.size()))
 675                 data_.resize(x+1);
 676 
 677         if(y >= static_cast<int>(data_[x].size()))
 678                 data_[x].resize(y+1);
 679 
 680         if(data_[x][y] == false) {
 681                 data_[x][y] = true;
 682                 return true;
 683         } else {
 684                 return false;
 685         }
 686 }

 688 void team::shroud_map::place(int x, int y)
 689 {
 690         if(enabled_ == false || x < 0 || y < 0)
 691                 return;
 692 
 693         if (x >= static_cast<int>(data_.size())) {
 694                 DBG_NG << "Couldn't place shroud on invalid x coordinate: ("
 695                         << x << ", " << y << ") - max x: " << data_.size() - 1 << "\n";
 696         } else if (y >= static_cast<int>(data_[x].size())) {
 697                 DBG_NG << "Couldn't place shroud on invalid y coordinate: ("
 698                         << x << ", " << y << ") - max y: " << data_[x].size() - 1 << "\n";
 699         } else {
 700                 data_[x][y] = false;
 701         }
 702 }

 704 void team::shroud_map::reset()
 705 {
 706         if(enabled_ == false)
 707                 return;
 708 
 709         for(std::vector<std::vector<bool> >::iterator i = data_.begin(); i != data_.end(); ++i) {
 710                 std::fill(i->begin(),i->end(),false);
 711         }
 712 }

 714 bool team::shroud_map::value(int x, int y) const
 715 {
 716         if ( !enabled_ )
 717                 return false;
 718 
 719         // Locations for which we have no data are assumed to still be covered.
 720         if ( x < 0  ||  x >= static_cast<int>(data_.size()) )
 721                 return true;
 722         if ( y < 0  ||  y >= static_cast<int>(data_[x].size()) )
 723                 return true;
 724 
 725         // data_ stores whether or not a location has been cleared, while
 726         // we want to return whether or not a location is covered.
 727         return !data_[x][y];
 728 }
 729 
 730 bool team::shroud_map::shared_value(const std::vector<const shroud_map*>& maps, int x, int y) const
 731 {
 732         if ( !enabled_ )
 733                 return false;
 734 
 735         // A quick abort:
 736         if ( x < 0  ||  y < 0 )
 737                 return true;
 738 
 739         // A tile is uncovered if it is uncovered on any shared map.
 740         for (const shroud_map * const shared_map : maps) {
 741                 if ( shared_map->enabled_  &&  !shared_map->value(x,y) )
 742                         return false;
 743         }
 744         return true;
 745 }

 747 std::string team::shroud_map::write() const
 748 {
 749         std::stringstream shroud_str;
 750         for(std::vector<std::vector<bool> >::const_iterator sh = data_.begin(); sh != data_.end(); ++sh) {
 751                 shroud_str << '|';
 752 
 753                 for(std::vector<bool>::const_iterator i = sh->begin(); i != sh->end(); ++i) {
 754                         shroud_str << (*i ? '1' : '0');
 755                 }
 756 
 757                 shroud_str << '\n';
 758         }
 759 
 760         return shroud_str.str();
 761 }
 762 
 763 void team::shroud_map::read(const std::string& str)
 764 {
 765         data_.clear();
 766         for(std::string::const_iterator sh = str.begin(); sh != str.end(); ++sh) {
 767                 if(*sh == '|')
 768                         data_.resize(data_.size()+1);
 769 
 770                 if(data_.empty() == false) {
 771                         if(*sh == '1')
 772                                 data_.back().push_back(true);
 773                         else if(*sh == '0')
 774                                 data_.back().push_back(false);
 775                 }
 776         }
 777 }

 779 void team::shroud_map::merge(const std::string& str)
 780 {
 781         int x=0, y=0;
 782         for(std::string::const_iterator sh = str.begin(); sh != str.end(); ++sh) {
 783                 if(*sh == '|' && sh != str.begin()) {
 784                         y=0;
 785                         x++;
 786                 } else if(*sh == '1') {
 787                         clear(x,y);
 788                         y++;
 789                 } else if(*sh == '0') {
 790                         y++;
 791                 }
 792         }
 793 }

 795 bool team::shroud_map::copy_from(const std::vector<const shroud_map*>& maps)
 796 {
 797         if(enabled_ == false)
 798                 return false;
 799 
 800         bool cleared = false;
 801         for(std::vector<const shroud_map*>::const_iterator i = maps.begin(); i != maps.end(); ++i) {
 802                 if((*i)->enabled_ == false)
 803                         continue;
 804 
 805                 const std::vector<std::vector<bool> >& v = (*i)->data_;
 806                 for(size_t x = 0; x != v.size(); ++x) {
 807                         for(size_t y = 0; y != v[x].size(); ++y) {
 808                                 if(v[x][y]) {
 809                                         cleared |= clear(x,y);
 810                                 }
 811                         }
 812                 }
 813         }
 814         return cleared;
 815 }
