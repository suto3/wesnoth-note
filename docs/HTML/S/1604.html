  18 /**
  19  * We use an auxiliary class just to do a 'better' equals check
  20  */
  21 public class LinkedFileEditorInput extends FileEditorInput
  22 {
  23     /**
  24      * Creates a new {@link LinkedFileEditorInput}
  25      * 
  26      * @param file
  27      *        The file attached to this editor input
  28      */
  29     public LinkedFileEditorInput( IFile file )
  30     {
  31         super( file );
  32     }
  33 
  34     @Override
  35     public boolean equals( Object obj )
  36     {
  37         if( this == obj ) {
  38             return true;
  39         }
  40         String targetUri = ""; //$NON-NLS-1$
  41         if( ! ( obj instanceof IFileEditorInput ) ) {
  42             if( obj instanceof FileStoreEditorInput ) {
  43                 FileStoreEditorInput other = ( FileStoreEditorInput ) obj;
  44                 targetUri = other.getURI( ).toString( );
  45             }
  46         }
  47         else {
  48             IFileEditorInput other = ( IFileEditorInput ) obj;
  49             targetUri = other.getFile( ).getLocationURI( ).toString( );
  50         }
  51         if( targetUri.isEmpty( ) ) {
  52             return false;
  53         }
  54         return getFile( ).getLocationURI( ).toString( )
  55             .toLowerCase( Locale.ENGLISH )
  56             .equals( targetUri.toLowerCase( Locale.ENGLISH ) );
  57     }
  58 }
