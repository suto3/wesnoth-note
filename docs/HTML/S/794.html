  73 static std::string get_system_username()
  74 {
  75         std::string res;
  76 #ifdef _WIN32
  77         wchar_t buffer[300];
  78         DWORD size = 300;
  79         if(GetUserNameW(buffer, &size)) {
  80                 //size includes a terminating null character.
  81                 assert(size > 0);
  82                 res = unicode_cast<utf8::string>(boost::iterator_range<wchar_t*>(buffer, buffer + size - 1));
  83         }
  84 #else
  85         if(char* const login = getenv("USER")) {
  86                 res = login;
  87         }
  88 #endif
  89         return res;
  90 }

  92 static void clear_credentials()
  93 {
  94         // Zero them before clearing.
  95         // Probably overly paranoid, but doesn't hurt?
  96         for(auto& cred : credentials) {
  97                 std::fill(cred.username.begin(), cred.username.end(), '\0');
  98                 std::fill(cred.server.begin(), cred.server.end(), '\0');
  99         }
 100         credentials.clear();

 107         std::string login()
 108         {
 109                 std::string name = preferences::get("login", EMPTY_LOGIN);
 110                 if(name == EMPTY_LOGIN) {
 111                         name = get_system_username();
 112                 } else if(name.size() > 2 && name[0] == '@' && name[name.size() - 1] == '@') {
 113                         name = name.substr(1, name.size() - 2);
 114                 } else {
 115                         ERR_CFG << "malformed user credentials (did you manually edit the preferences file?)" << std::endl;
 116                 }
 117                 if(name.empty()) {
 118                         return "player";
 119                 }
 120                 return name;
 121         }

 144         std::string password(const std::string& server, const std::string& login)
 145         {
 146                 if(!remember_password()) {
 147                         if(!credentials.empty() && credentials[0].username == login && credentials[0].server == server) {
 148                                 auto temp = decrypt(credentials[0].key, build_key(server, login));
 149                                 return std::string(temp.begin(), temp.end());
 150                         } else {
 151                                 return "";
 152                         }
 153                 }
 154                 auto cred = std::find_if(credentials.begin(), credentials.end(), [&](const login_info& cred) {
 155                         return cred.server == server && cred.username == login;
 156                 });
 157                 if(cred == credentials.end()) {
 158                         return "";
 159                 }
 160                 auto temp = decrypt(cred->key, build_key(server, login));
 161                 return std::string(temp.begin(), temp.end());
 162         }

 164         void set_password(const std::string& server, const std::string& login, const std::string& key)
 165         {
 166                 secure_buffer temp(key.begin(), key.end());
 167                 if(!remember_password()) {
 168                         clear_credentials();
 169                         credentials.emplace_back(login, server, encrypt(temp, build_key(server, login)));
 170                         return;
 171                 }
 172                 auto cred = std::find_if(credentials.begin(), credentials.end(), [&](const login_info& cred) {
 173                         return cred.server == server && cred.username == login;
 174                 });
 175                 if(cred == credentials.end()) {
 176                         // This is equivalent to emplace_back, but also returns the iterator to the new element
 177                         cred = credentials.emplace(credentials.end(), login, server);
 178                 }
 179                 cred->key = encrypt(temp, build_key(server, login));
 180         }

 182         void load_credentials()
 183         {
 184                 if(!remember_password()) {
 185                         return;
 186                 }
 187                 clear_credentials();
 188                 std::string cred_file = filesystem::get_credentials_file();
 189                 if(!filesystem::file_exists(cred_file)) {
 190                         return;
 191                 }
 192                 filesystem::scoped_istream stream = filesystem::istream_file(cred_file, false);
 193                 // Credentials file is a binary blob, so use streambuf iterator
 194                 secure_buffer data((std::istreambuf_iterator<char>(*stream)), (std::istreambuf_iterator<char>()));
 195                 data = decrypt(data, build_key("global", get_system_username()));
 196                 if(data.empty() || data[0] != CREDENTIAL_SEPARATOR) {
 197                         ERR_CFG << "Invalid data in credentials file\n";
 198                         return;
 199                 }
 200                 for(const std::string& elem : utils::split(std::string(data.begin(), data.end()), CREDENTIAL_SEPARATOR, utils::REMOVE_EMPTY)) {
 201                         size_t at = elem.find_last_of('@');
 202                         size_t eq = elem.find_first_of('=', at + 1);
 203                         if(at != std::string::npos && eq != std::string::npos) {
 204                                 secure_buffer key(elem.begin() + eq + 1, elem.end());
 205                                 credentials.emplace_back(elem.substr(0, at), elem.substr(at + 1, eq - at - 1), unescape(key));
 206                         }
 207                 }
 208         }

 210         void save_credentials()
 211         {
 212                 if(!remember_password()) {
 213                         filesystem::delete_file(filesystem::get_credentials_file());
 214                         return;
 215                 }
 216                 secure_buffer credentials_data(1, CREDENTIAL_SEPARATOR);
 217                 size_t offset = 1;
 218                 for(const auto& cred : credentials) {
 219                         credentials_data.resize(credentials_data.size() + cred.size(), CREDENTIAL_SEPARATOR);
 220                         std::copy(cred.username.begin(), cred.username.end(), credentials_data.begin() + offset);
 221                         offset += cred.username.size();
 222                         credentials_data[offset++] = '@';
 223                         std::copy(cred.server.begin(), cred.server.end(), credentials_data.begin() + offset);
 224                         offset += cred.server.size();
 225                         credentials_data[offset++] = '=';
 226                         secure_buffer key_escaped = escape(cred.key);
 227                         // Escaping may increase the length, so resize again if so
 228                         credentials_data.resize(credentials_data.size() + key_escaped.size() - cred.key.size());
 229                         std::copy(key_escaped.begin(), key_escaped.end(), credentials_data.begin() + offset);
 230                         offset += key_escaped.size() + 1;
 231                 }
 232                 try {
 233                         filesystem::scoped_ostream credentials_file = filesystem::ostream_file(filesystem::get_credentials_file());
 234                         secure_buffer encrypted = encrypt(credentials_data, build_key("global", get_system_username()));
 235                         credentials_file->write(reinterpret_cast<const char*>(encrypted.data()), encrypted.size());
 236                 } catch(filesystem::io_exception&) {
 237                         ERR_CFG << "error writing to credentials file '" << filesystem::get_credentials_file() << "'" << std::endl;
 238                 }
 239         }

 242 // TODO: Key-stretching (bcrypt was recommended)
 243 secure_buffer build_key(const std::string& server, const std::string& login)
 244 {
 245         std::string sysname = get_system_username();
 246         secure_buffer result(std::max<size_t>(server.size() + login.size() + sysname.size(), 32));
 247         unsigned char i = 0;
 248         std::generate(result.begin(), result.end(), [&i]() {return 'x' ^ i++;});
 249         std::copy(login.begin(), login.end(), result.begin());
 250         std::copy(sysname.begin(), sysname.end(), result.begin() + login.size());
 251         std::copy(server.begin(), server.end(), result.begin() + login.size() + sysname.size());
 252         return result;
 253 }

 255 static secure_buffer rc4_crypt(const secure_buffer& text, const secure_buffer& key)
 256 {
 257         RC4_KEY cipher_key;
 258         RC4_set_key(&cipher_key, key.size(), key.data());
 259         const size_t block_size = key.size();
 260         const size_t blocks = text.size() / block_size;
 261         const size_t extra = text.size() % block_size;
 262         secure_buffer result(text.size(), '\0');
 263         for(size_t i = 0; i < blocks * block_size; i += block_size) {
 264                 RC4(&cipher_key, block_size, text.data() + i, result.data() + i);
 265         }
 266         if(extra) {
 267                 size_t i = blocks * block_size;
 268                 RC4(&cipher_key, extra, text.data() + i, result.data() + i);
 269         }
 270         return result;
 271 }

 283 secure_buffer unescape(const secure_buffer& text)
 284 {
 285         secure_buffer unescaped;
 286         unescaped.reserve(text.size());
 287         bool escaping = false;
 288         for(char c : text) {
 289                 if(escaping) {
 290                         if(c == '\xa') {
 291                                 unescaped.push_back('\xc');
 292                         } else if(c == '.') {
 293                                 unescaped.push_back('@');
 294                         } else {
 295                                 unescaped.push_back(c);
 296                         }
 297                         escaping = false;
 298                 } else if(c == '\x1') {
 299                         escaping = true;
 300                 } else {
 301                         unescaped.push_back(c);
 302                 }
 303         }
 304         assert(!escaping);
 305         return unescaped;
 306 }

 308 secure_buffer escape(const secure_buffer& text)
 309 {
 310         secure_buffer escaped;
 311         escaped.reserve(text.size());
 312         for(char c : text) {
 313                 if(c == '\x1') {
 314                         escaped.push_back('\x1');
 315                         escaped.push_back('\x1');
 316                 } else if(c == '\xc') {
 317                         escaped.push_back('\x1');
 318                         escaped.push_back('\xa');
 319                 } else if(c == '@') {
 320                         escaped.push_back('\x1');
 321                         escaped.push_back('.');
 322                 } else {
 323                         escaped.push_back(c);
 324                 }
 325         }
 326         return escaped;
 327 }
