 145 # Functions to extract data from different repo types.
 146 # For git repos
 147 # shellcheck disable=SC2039,SC2164,SC2155
 148 gitRepo() {
 149         ${LOCAL} oldPath="${PWD}"
 150 
 151         cd "$(git rev-parse --show-toplevel)"
 152 
 153         VCS_TYPE="git"
 154 
 155         VCS_BASENAME="$(basename "${PWD}")"
 156 
 157         ${LOCAL} currentRev="$(git rev-parse HEAD)"
 158 
 159         VCS_UUID="$(git rev-list --max-parents=0 --date-order --reverse "${currentRev}" 2>/dev/null | sed -n 1p)"
 160         if [ -z "${VCS_UUID}" ]; then
 161                 VCS_UUID="$(git rev-list --topo-order "${currentRev}" | tail -n 1)"
 162         fi
 163 
 164         # Is the working copy clean?
 165         test -z "$(git status --untracked-files=normal --porcelain)"
 166         VCS_WC_MODIFIED="${?}"
 167 
 168         # Enumeration of changesets
 169         VCS_NUM="$(git rev-list --count "${currentRev}" 2>/dev/null)"
 170         if [ -z "${VCS_NUM}" ]; then
 171                 if [ ! "${WARNHIDE}" = "1" ]; then
 172                         echo "warning: Counting the number of revisions may be slower due to an outdated git version less than 1.7.2.3. If something breaks, please update it." 1>&2
 173                 fi
 174                 VCS_NUM="$(git rev-list HEAD | wc -l)"
 175         fi
 176 
 177         # This may be a git-svn remote.  If so, report the Subversion revision.
 178         if [ -z "$(git config svn-remote.svn.url 2>/dev/null)" ]; then
 179                 # The full revision hash
 180                 VCS_FULL_HASH="$(git rev-parse "${currentRev}")"
 181 
 182                 # The short hash
 183                 VCS_SHORT_HASH="$(git rev-parse --short "${currentRev}")"
 184         else
 185                 # The git-svn revision number
 186                 VCS_FULL_HASH="$(git svn find-rev "${currentRev}")"
 187                 VCS_SHORT_HASH="${VCS_FULL_HASH}"
 188         fi
 189 
 190         # Current branch
 191         VCS_BRANCH="$(git rev-parse --symbolic-full-name --verify "$(git name-rev --name-only --no-undefined "${currentRev}" 2>/dev/null)" 2>/dev/null | sed -e 's:refs/heads/::' | sed -e 's:refs/::')"
 192 
 193         # Cache the description
 194         ${LOCAL} DESCRIPTION="$(git describe --long --tags "${currentRev}" 2>/dev/null)"
 195 
 196         # Current or last tag ancestor (empty if no tags)
 197         VCS_TAG="$(echo "${DESCRIPTION}" | sed -e "s:-g${VCS_SHORT_HASH}\$::" -e 's:-[0-9]*$::')"
 198 
 199         # Distance to last tag or an alias of VCS_NUM if there is no tag
 200         if [ ! -z "${DESCRIPTION}" ]; then
 201                 VCS_TICK="$(echo "${DESCRIPTION}" | sed -e "s:${VCS_TAG}-::" -e "s:-g${VCS_SHORT_HASH}::")"
 202         else
 203                 VCS_TICK="${VCS_NUM}"
 204         fi
 205 
 206         # Date of the current commit
 207         VCS_DATE="$(TZ=UTC git show -s --date=iso-strict-local --pretty=format:%cd "${currentRev}" 2>/dev/null | sed -e 's|+00:00|Z|')"
 208         if [ -z "${VCS_DATE}" ]; then
 209                 if [ ! "${WARNHIDE}" = "1" ]; then
 210                         echo "warning: Action stamps require git version 2.7+." 1>&2
 211                 fi
 212                 VCS_DATE="$(git log -1 --pretty=format:%ci "${currentRev}" | sed -e 's: :T:' -e 's: ::' -e 's|+00:00|Z|')"
 213                 ${LOCAL} ASdis="1"
 214         fi
 215 
 216         # Action Stamp
 217         if [ -z "${ASdis}" ]; then
 218                 VCS_ACTION_STAMP="${VCS_DATE}!$(git show -s --pretty=format:%cE "${currentRev}")"
 219         else
 220                 VCS_ACTION_STAMP=""
 221         fi
 222 
 223         cd "${oldPath}"
 224 }

 226 # For hg repos
 227 # shellcheck disable=SC2039,SC2164
 228 hgRepo() {
 229         ${LOCAL} oldPath="${PWD}"
 230 
 231         cd "$(hg root)"
 232 
 233         VCS_TYPE="hg"
 234 
 235         VCS_BASENAME="$(basename "${PWD}")"
 236 
 237         VCS_UUID="$(hg log -r "0" -l 1 --template '{node}\n')"
 238 
 239         # Is the working copy clean?
 240         test -z "$(hg status -duram)"
 241         VCS_WC_MODIFIED="${?}"
 242 
 243         # Enumeration of changesets
 244         VCS_NUM="$(hg id -n | tr -d '+')"
 245 
 246         # The full revision hash
 247         VCS_FULL_HASH="$(hg log -r "${VCS_NUM}" -l 1 --template '{node}\n')"
 248 
 249         # The short hash
 250         VCS_SHORT_HASH="$(hg id -i | tr -d '+')"
 251 
 252         # Current bookmark (bookmarks are roughly equivalent to git's branches)
 253         # or branch if no bookmark
 254         VCS_BRANCH="$(hg id -B | cut -d ' ' -f 1)"
 255         # Fall back to the branch if there are no bookmarks
 256         if [ -z "${VCS_BRANCH}" ]; then
 257                 VCS_BRANCH="$(hg id -b)"
 258         fi
 259 
 260         # Current or last tag ancestor (excluding auto tags, empty if no tags)
 261         VCS_TAG="$(hg log -r "${VCS_NUM}" -l 1 --template '{latesttag}\n' 2>/dev/null | sed -e 's:qtip::' -e 's:tip::' -e 's:qbase::' -e 's:qparent::' -e "s:$(hg --config 'extensions.color=' --config 'extensions.mq=' --color never qtop 2>/dev/null)::" | cut -d ' ' -f 1)"
 262 
 263         # Distance to last tag or an alias of VCS_NUM if there is no tag
 264         if [ ! -z "${VCS_TAG}" ]; then
 265                 VCS_TICK="$(hg log -r "${VCS_NUM}" -l 1 --template '{latesttagdistance}\n' 2>/dev/null)"
 266         else
 267                 VCS_TICK="${VCS_NUM}"
 268         fi
 269 
 270         # Date of the current commit
 271         VCS_DATE="$(hg log -r "${VCS_NUM}" -l 1 --template '{date|isodatesec}\n' 2>/dev/null | sed -e 's: :T:' -e 's: ::' -e 's|+00:00|Z|')"
 272 
 273         # Action Stamp
 274         VCS_ACTION_STAMP="$(TZ=UTC hg log -r "${VCS_NUM}" -l 1 --template '{date|localdate|rfc3339date}\n' 2>/dev/null | sed -e 's|+00:00|Z|')!$(hg log -r "${VCS_NUM}" -l 1 --template '{author|email}\n' 2>/dev/null)"
 275 
 276         cd "${oldPath}"
 277 }

 281 bzrRepo() {
 282         ${LOCAL} oldPath="${PWD}"
 283 
 284         cd "$(bzr root)"
 285 
 286         VCS_TYPE="bzr"
 287 
 288         VCS_BASENAME="$(basename "${PWD}")"
 289 
 290         # Currently unimplemented because more investigation is needed.
 291         VCS_UUID=""
 292 
 293         # Is the working copy clean?
 294         bzr version-info --custom --template='{clean}\n' | grep -q '1'
 295         VCS_WC_MODIFIED="${?}"
 296 
 297         # Enumeration of changesets
 298         VCS_NUM="$(bzr revno)"
 299 
 300         # The full revision hash
 301         VCS_FULL_HASH="$(bzr version-info --custom --template='{revision_id}\n')"
 302 
 303         # The short hash
 304         VCS_SHORT_HASH="${VCS_NUM}"
 305 
 306         # Nick of the current branch
 307         VCS_BRANCH="$(bzr nick)"
 308 
 309         # Current or last tag ancestor (excluding auto tags, empty if no tags)
 310         VCS_TAG="$(bzr tags --sort=time | sed '/?$/d' | tail -n1 | cut -d ' ' -f1)"
 311 
 312         # Distance to last tag or an alias of VCS_NUM if there is no tag
 313         if [ ! -z "${VCS_TAG}" ]; then
 314                 VCS_TICK="$(bzr log --line -r "tag:${VCS_TAG}.." | tail -n +2 | wc -l | sed -e 's:^ *::')"
 315         else
 316                 VCS_TICK="${VCS_NUM}"
 317         fi
 318 
 319         # Date of the current commit
 320         VCS_DATE="$(bzr version-info --custom --template='{date}\n' | sed -e 's: :T:' -e 's: ::')"
 321 
 322         # Action Stamp
 323         # Currently unimplemented because more investigation is needed.
 324         VCS_ACTION_STAMP=""
 325 
 326         cd "${oldPath}"
 327 }

 329 # For svn repos
 330 # shellcheck disable=SC2039,SC2164,SC2155
 331 svnRepo() {
 332         ${LOCAL} oldPath="${PWD}"
 333 
 334         VCS_TYPE="svn"
 335 
 336         case "${PWD}" in
 337         /*trunk*|/*branches*|/*tags*)
 338                 ${LOCAL} fn="${PWD}"
 339                 while [ "$(basename "${fn}")" != 'trunk' ] && [ "$(basename "${fn}")" != 'branches' ] && [ "$(basename "${fn}")" != 'tags' ] && [ "$(basename "${fn}")" != '/' ]; do
 340                         ${LOCAL} fn="$(dirname "${fn}")"
 341                 done
 342                 ${LOCAL} fn="$(dirname "${fn}")"
 343                 if [ "${fn}" = '/' ]; then
 344                         VCS_BASENAME="$(basename "${PWD}")"
 345                 else
 346                         VCS_BASENAME="$(basename "${fn}")"
 347                 fi
 348                 ;;
 349         *) VCS_BASENAME="$(basename "${PWD}")" ;;
 350         esac
 351 
 352         VCS_UUID="$(svn info --xml | sed -n -e 's:<uuid>::' -e 's:</uuid>::p')"
 353 
 354         # Cache svnversion output
 355         ${LOCAL} SVNVERSION="$(svnversion)"
 356 
 357         # Is the working copy clean?
 358         echo "${SVNVERSION}" | grep -q "M"
 359         case "${?}" in
 360                 0)
 361                         VCS_WC_MODIFIED="1"
 362                 ;;
 363                 1)
 364                         if [ ! -z "${UNTRACKEDFILES}" ]; then
 365                         # `svnversion` does not detect untracked files and `svn status` is really slow, so only run it if we really have to.
 366                                 if [ -z "$(svn status)" ]; then
 367                                         VCS_WC_MODIFIED="0"
 368                                 else
 369                                         VCS_WC_MODIFIED="1"
 370                                 fi
 371                         else
 372                                 VCS_WC_MODIFIED="0"
 373                         fi
 374                 ;;
 375         esac
 376 
 377         # Enumeration of changesets
 378         VCS_NUM="$(echo "${SVNVERSION}" | cut -d : -f 1 | sed -e 's:M::' -e 's:S::' -e 's:P::')"
 379 
 380         # The full revision hash
 381         VCS_FULL_HASH="${SVNVERSION}"
 382 
 383         # The short hash
 384         VCS_SHORT_HASH="${VCS_NUM}"
 385 
 386         # Current branch
 387         case "${PWD}" in
 388         /*trunk*|/*branches*|/*tags*)
 389                 ${LOCAL} lastbase=""
 390                 ${LOCAL} fn="${PWD}"
 391                 while :
 392                 do
 393                         base="$(basename "${fn}")"
 394                         if [ "${base}" = 'trunk' ]; then
 395                                 VCS_BRANCH='trunk'
 396                                 break
 397                         elif [ "${base}" = 'branches' ] || [ "${base}" = 'tags' ]; then
 398                                 VCS_BRANCH="${lastbase}"
 399                                 break
 400                         elif [ "${base}" = '/' ]; then
 401                                 VCS_BRANCH=""
 402                                 break
 403                         fi
 404                         ${LOCAL} lastbase="${base}"
 405                         ${LOCAL} fn="$(dirname "${fn}")"
 406                 done
 407                 ;;
 408         *) VCS_BRANCH="" ;;
 409         esac
 410 
 411         # Current or last tag ancestor (empty if no tags). But "current
 412         # tag" can't be extracted reliably because Subversion doesn't
 413         # have tags the way other VCSes do.
 414         VCS_TAG=""
 415         VCS_TICK=""
 416 
 417         # Date of the current commit
 418         VCS_DATE="$(svn info --xml | sed -n -e 's:<date>::' -e 's:</date>::p')"
 419 
 420         # Action Stamp
 421         VCS_ACTION_STAMP="${VCS_DATE}!$(svn log --xml -l 1 -r "${VCS_SHORT_HASH}" | sed -n -e 's:<author>::' -e 's:</author>::p')"
 422 
 423         cd "${oldPath}"
 424 }

1125 # Helper functions
1126 # Count path segments
1127 # shellcheck disable=SC2039
1128 pathSegment() {
1129         ${LOCAL} pathz="${1}"
1130         ${LOCAL} depth="0"
1131 
1132         if [ ! -z "${pathz}" ]; then
1133                 # Continue until we are at / or there are no path separators left.
1134                 while [ ! "${pathz}" = "/" ] && [ ! "${pathz}" = "$(echo "${pathz}" | sed -e 's:/::')" ]; do
1135                         pathz="$(dirname "${pathz}")"
1136                         depth="$((depth+1))"
1137                 done
1138         fi
1139         echo "${depth}"
1140 }

1142 # Largest of four numbers
1143 # shellcheck disable=SC2039
1144 multiCompare() {
1145         ${LOCAL} larger="${1}"
1146         ${LOCAL} numA="${2}"
1147         ${LOCAL} numB="${3}"
1148         ${LOCAL} numC="${4}"
1149 
1150         [ "${numA}" -gt "${larger}" ] && larger="${numA}"
1151         [ "${numB}" -gt "${larger}" ] && larger="${numB}"
1152         [ "${numC}" -gt "${larger}" ] && larger="${numC}"
1153         echo "${larger}"
1154 }

1156 # Test for repositories
1157 # shellcheck disable=SC2155,SC2039
1158 repoTest() {
1159         REPONUM="0"
1160         if command -v git > /dev/null 2>&1; then
1161                 ${LOCAL} gitPath="$(git rev-parse --show-toplevel 2>/dev/null)"
1162                 ${LOCAL} gitDepth="$(pathSegment "${gitPath}")"
1163                 if [ ! -z "${gitPath}" ]; then
1164                         REPONUM="$((REPONUM+1))"
1165                 fi
1166         else
1167                 ${LOCAL} gitDepth="0"
1168         fi
1169         if command -v hg > /dev/null 2>&1; then
1170                 ${LOCAL} hgPath="$(hg root 2>/dev/null)"
1171                 ${LOCAL} hgDepth="$(pathSegment "${hgPath}")"
1172                 if [ ! -z "${hgPath}" ]; then
1173                         REPONUM="$((REPONUM+1))"
1174                 fi
1175         else
1176                 ${LOCAL} hgDepth="0"
1177         fi
1178         if command -v bzr > /dev/null 2>&1; then
1179                 ${LOCAL} bzrPath="$(bzr root 2>/dev/null)"
1180                 ${LOCAL} bzrDepth="$(pathSegment "${bzrPath}")"
1181                 if [ ! -z "${bzrPath}" ]; then
1182                         REPONUM="$((REPONUM+1))"
1183                 fi
1184         else
1185                 ${LOCAL} bzrDepth="0"
1186         fi
1187         if command -v svn > /dev/null 2>&1; then
1188                 ${LOCAL} stringz="<wcroot-abspath>"
1189                 ${LOCAL} stringx="</wcroot-abspath>"
1190                 ${LOCAL} svnPath="$(svn info --xml 2>/dev/null | sed -n -e "s:${stringz}::" -e "s:${stringx}::p")"
1191                 # An old enough svn will not be able give us a path; default
1192                 # to 1 for that case.
