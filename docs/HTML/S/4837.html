 377     [event]
 378         # This event fires when the king sees a unit approaching the castle, and he
 379         # is alerted that he needs more units. We want the player to be able to see
 380         # the king while he talks about this, so that's what we're actually going to
 381         # check: That a player's unit can see the side-2 leader. We can't use
 382         # "sighted" for this because that only works when a unit is revealed for the
 383         # first time, and the king may have already been revealed by a shadow or other
 384         # hidden unit. (But, of course, the king wouldn't have been alerted in that
 385         # case.) We fire a custom event to have the king open another chest of gold.
 386         name=moveto
 387         first_time_only=no
 388         id=king_alerted_1
 389 
 390         [filter]
 391             side=1
 392         [/filter]
 393         [store_reachable_locations]
 394             [filter]
 395                 id=$unit.id
 396                 # These three units are hidden, so they don't count:
 397                 [not]
 398                     type=Shadow, Nightgaunt
 399                 [/not]
 400                 [and]
 401                     [not]
 402                         type=Fugitive
 403                         [filter_location]
 404                             terrain=*^V*  # Hidden only in villages
 405                         [/filter_location]
 406                     [/not]
 407                 [/and]
 408             [/filter]
 409             range=vision
 410             [filter_location]
 411                 x,y=20,26
 412             [/filter_location]
 413             variable=can_see_the_king
 414         [/store_reachable_locations]
 415 
 416         [if]
 417             [variable]
 418                 name=can_see_the_king.length
 419                 greater_than=0
 420             [/variable]
 421             [then]
 422                 [store_unit]
 423                     [filter]
 424                         id=$unit.id
 425                     [/filter]
 426                     variable=alerting_unit
 427                 [/store_unit]
 428                 [fire_event]
 429                     name=get_more_gold
 430                 [/fire_event]
 431             [/then]
 432             [else]
 433                 # If a unit can see the king, this else won't be reached, so undo
 434                 # won't work for that unit. To solve that, this whole event will
 435                 # be removed by the get_more_gold event, and undo will only be
 436                 # blocked on the one move that causes get_more_gold to fire.
 437                 [allow_undo][/allow_undo]
 438             [/else]
 439         [/if]
 440     [/event]
