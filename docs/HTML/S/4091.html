  94     def read_encoded(self, filename):
  95         """
  96         Helper for gracefully handling non-utf8 files and fixing up non-unix
  97         line endings.
  98         """
  99         try:

 101         except IOError:
 102             sys.stderr.write("Cannot open file %s!\n" % filename)
 103             return ""
 104         try:
 105             u = text.decode("utf8")
 106         except UnicodeDecodeError:
 107             u = text.decode("latin1")

 109         text = text.replace("\r\n", "\n").replace("\t", " ").replace("\r", "\n")
 110         if text == "" or text[-1] != "\n":

 221     def read_until(self, sep):
 222         """Read until a character inside the string sep is found."""
 223         mob = re.compile(".*?[" + sep + "]", re.S).match(self.text, self.textpos)
 224         if mob:
 225             found = mob.group(0)
 226             self.line += found.count("\n")
 227             self.textpos = mob.end(0)
 228             if self.textpos == len(self.text) and not self.stay_in_file:
 229                 if len(self.texts): self.pop_text()
 230             return found
 231         else:
 232             found = self.text[self.textpos:]
 233             self.line += found.count("\n")
 234             self.textpos = len(self.text)
 235             if len(self.texts) and not self.stay_in_file:
 236                 self.pop_text()
 237                 found += self.read_until(sep)
 238             return found

 257     def read_lines_until(self, string):
 258         """
 259         Read lines until one contains the given string, but throw away any
 260         comments.
 261         """
 262         text = ""
 263         in_string = False
 264         while 1:
 265             if self.at_end():
 266                 return None
 267             line = self.read_until("\n")
 268 
 269             line_start = 0
 270             if in_string:
 271                 string_end = line.find('"')
 272                 if string_end < 0:
 273                     text += line
 274                     continue
 275                 in_string = False
 276                 line_start = string_end + 1
 277 
 278             elif line.lstrip().startswith("#"):
 279                 possible_comment = line.lstrip()
 280                 for com in self.preprocessor_commands:
 281                     if possible_comment.startswith(com):
 282                         break
 283                 else:
 284                     continue
 285 
 286 
 287             quotation = line.find('"', line_start)
 288             while quotation >= 0:
 289                 in_string = True
 290                 string_end = line.find('"', quotation + 1)
 291                 if string_end < 0: break
 292                 line_start = string_end + 1
 293                 in_string = False
 294                 quotation = line.find('"', line_start)
 295 
 296             if not in_string:
 297                 end = line.find(string, line_start)
 298                 if end >= 0:
 299                     text += line[:end]
 300                     break
 301             text += line
 302         return text

 334     def parse_macro(self):
 335         """No recursive macro processing is done here. If a macro is passed as
 336         parameter to a macro, then whoever parses the macro replacement will do
 337         the macro recursion.
 338         Actually - I'm too tired right now to think this through. Maybe it
 339         should be done the other way around, starting expansion with the
 340         innermost macro?
 341         """
 342         params = []
 343         balance = 1
 344         macro = ""
 345         while balance > 0:
 346             macro += self.read_until("{}")
 347             if macro[-1] == "{":
 348                 balance += 1
 349             elif macro[-1] == "}":
 350                 balance -= 1
 351             else:
 352                 raise Error(self, "Unclosed macro")
 353                 return
 354 
 355         preserve = macro
 356         macro = macro[:-1] # Get rid of final }
 357 
 358         if self.macro_callback:
 359             if not self.macro_callback(macro): return None
 360 
 361         # If the macro starts with ~, assume a file in userdata.
 362         if macro[0] == "~":
 363             if self.user_dir:
 364                 dirpath = self.user_dir + "/" + macro[1:]
 365             else:
 366                 dirpath = None
 367         # If the macro starts with @, look first in data then in userdata.
 368         elif macro[0] == "@":
 369             if self.data_dir:
 370                 dirpath = self.data_dir + "/" + macro[1:]
 371                 if not os.path.exists(dirpath): dirpath = None
 372             else:
 373                 dirpath = None
 374             if not dirpath and self.user_dir:
 375                 dirpath = self.user_dir + "/" + macro[1:]
 376         # If the macro starts with ., look relative to the currently parsed
 377         # file.
 378         elif macro[0] == ".":
 379             dirpath = self.current_path + macro[1:]
 380         # Otherwise, try to interpret the macro as a filename in the data dir.
 381         elif self.data_dir != None:
 382             dirpath = self.data_dir + "/" + macro
 383         else:
 384             dirpath = None
 385 
 386         if dirpath != None and os.path.exists(dirpath):
 387             dirpath = os.path.normpath(dirpath)
 388             if self.only_expand_pathes:
 389                 if not [x for x in self.only_expand_pathes if os.path.commonprefix([dirpath, x]) == x]:
 390                     return None
 391             # If it is a directory, parse all cfg files within.
 392             if os.path.isdir(dirpath):
 393                 # Note: glob.glob will try to return unicode filenames
 394                 # if you pass it an unicode string - but to deal with
 395                 # non-unicode filenames as are allowed in linux we
 396                 # convert to a byte-string.
 397                 # Execute all WML files in the directory.
 398                 files = glob.glob(str(dirpath + "/*.cfg"))
 399                 # And also execute directories with a _main.cfg.
 400                 files += glob.glob(str(dirpath + "/*/_main.cfg"))
 401                 files.sort()
 402                 mc = dirpath + "/_main.cfg"
 403                 fc = dirpath + "/_final.cfg"
 404 
 405                 if mc in files:
 406                     # If there's a file called _main.cfg, only parse that.
 407                     files = [mc]
 408                 elif fc in files:
 409                     # If there's a file called _final.cfg, parse it only after
 410                     # all others.
 411                     files.remove(fc)
 412                     files.append(fc)
 413             else:
 414                 files = [dirpath]
 415             files.reverse()
 416             for path in files:
 417                 self.push_text(path, self.read_encoded(path), cd = os.path.dirname(path))
 418             return None
 419 
 420         # It's OK for user directories not to exist.
 421         # Nothing prefixed with ~ can be a macro.
 422         if macro.startswith("~") and not self.verbose:
 423             return None
 424 
 425         # No file was found, try to do macro expansion. First, push the
 426         # macro call again. E.g. {blah 1 2 3}.
 427         self.push_text("macro", preserve)
 428 
 429         # Find all parameters.
 430         while 1:
 431             read = self.read_until('"}{ (\n')
 432             if not read:
 433                 sys.stderr.write("? %s\n" % macro)
 434                 sys.stderr.write(" (%s)\n" % params)
 435                 raise Error(self, "Unexpected end of file")
 436                 break
 437             sep = read[-1]
 438             read = read[:-1]
 439 
 440             if sep == "}":
 441                 if read: params += [read]
 442                 break
 443 
 444             elif sep == "{":
 445                 balance = 1
 446                 param = sep + read
 447                 while balance:
 448                     c = self.read_next()
 449                     if c == "{":
 450                         balance += 1
 451                     elif c == "}":
 452                         balance -= 1
 453                     param += c
 454                 params += [param]
 455 
 456             elif sep == '"':
 457                 # Cannot parse strings here.. it must be passed to the macro as
 458                 # is, so the real string parser can handle it. Else there will
 459                 # be subtle bugs, e.g. when a MACRO evaluates to a + at the end
 460                 # of line.
 461                 read += '"' + self.read_until('"')
 462                 params += [read]
 463 
 464             elif sep == "(":
 465                 in_string = False
 466                 balance = 1
 467                 param = read
 468                 while balance:
 469                     c = self.read_next()
 470                     # Ignore () within strings
 471                     if c == '"':
 472                         in_string = not in_string
 473 
 474                     if not in_string:
 475                         if c == "(":
 476                             balance += 1
 477                         elif c == ")":
 478                             balance -= 1
 479                     param += c
 480                 params += [param[:-1]]
 481 
 482             else:
 483                 if read:
 484                     params += [read]
 485                 self.read_while(" \n")
 486 
 487         if self.no_macros:
 488             return wmldata.DataMacro("macro", " ".join(params))
 489 
 490         name = params[0]
 491         if name in self.macros:
 492             macro = self.macros[name]
 493             text = macro.text
 494             for i, j in enumerate(macro.params):
 495                 if 1 + i >= len(params):
 496                     raise Error(self, "Not enough parameters for macro %s. " % name +
 497                         "%d given but %d needed %s." % (len(params) - 1,
 498                             len(macro.params), macro.params))
 499                 rep = params[1 + i]
 500                 # Handle gettext replacement here, since inside the macro
 501                 # the textdomain will be wrong.
 502                 if self.gettext and rep and rep[0] == "_":
 503                     q = rep.find('"')
 504                     qe = rep.find('"', q + 1)
 505                     rep = self.gettext(self.textdomain, rep[q + 1:qe])
 506                     rep = '"' + rep + '"'
 507                 if self.verbose:
 508                     #s = "Replacing {%s} with %s\n" % (j, rep)
 509                     ##sys.stderr.write(s.encode("utf8"))
 510                     pass
 511                 text = text.replace("{%s}" % j, rep)
 512

 514                 self.push_text(name, text, initial_textdomain = macro.textdomain)
 515             else:
 516                 pass # empty macro, nothing to do
 517         else:
 518             if self.macro_not_found_callback:
 519                 keep_macro = self.macro_not_found_callback(self, name, params)
 520                 if keep_macro: return keep_macro
 521             if self.verbose:
 522                 sys.stderr.write("No macro %s.\n" % name.encode("utf8"))
 523             if self.verbose:
 524                 sys.stderr.write(" (%s:%d)\n" % (self.filename, self.line))
 525             return name
 526         return None

 528     def parse_string(self):
 529         text = ""
 530         match_read_end = '"'
 531         if not self.no_macros_in_string:
 532             match_read_end += '{'
 533         while not self.at_end():
 534             text += self.read_until(match_read_end)
 535             if text[-1] == '"':
 536                 if self.peek_next() == '"':
 537                     self.read_next()
 538                 else:
 539                     return text[:-1]
 540             elif text[-1] == '{':
 541                 text = text[:-1]
 542                 not_found = self.parse_macro()
 543                 if not isinstance(not_found, wmldata.Data):
 544                     if not_found:
 545                         text += not_found
 546             else:
 547                 break
 548         raise Error(self, "Unclosed string")

 634                     else:

 638                         value += c

 642             raise Error(self, "= expected for \"%s\"" % variable)

 657     def parse_top(self, data, state = None):
 658         while 1:
 659             self.skip_whitespace_and_newlines()
 660             if self.at_end():
 661                 if state:
 662                     raise Error(self, "Tag stack non-empty (%s) at end of data" % state)
 663                 break
 664             c = self.read_next()
 665             if c == "#": # comment or preprocessor
 666                 if self.check_for("define "):
 667                     self.read_until(" ")
 668                     params = []
 669                     while 1:
 670                         name = self.read_until(" \n")
 671                         sep = name[-1]
 672                         name = name[:-1]
 673                         if name: params += [name]
 674                         if sep == "\n": break
 675                         self.read_while(" ")

 678                     if text == None:
 679                         raise Error(self, "#define without #enddef")
