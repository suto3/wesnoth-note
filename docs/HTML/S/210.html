  88 template<typename T>
  89 auto find_attack(T* u, const std::string& id) -> attack_ptr_in<T>
  90 {
  91         auto attacks = u->attacks();
  92         for(auto at = attacks.begin(); at != attacks.end(); ++at) {
  93                 if(at->id() == id) {
  94                         return *at.base();
  95                 }
  96         }
  97         return nullptr;
  98 }

 100 template<typename T>
 101 auto find_attack(T* u, size_t i) -> attack_ptr_in<T>
 102 {
 103         auto attacks = u->attacks();
 104         if(i < static_cast<size_t>(attacks.size())) {
 105                 auto iter = attacks.begin();
 106                 iter += i;
 107                 return *iter.base();
 108         }
 109         return nullptr;
 110 }

 112 /**
 113  * Gets the attacks of a unit or unit type (__index metamethod).
 114  * - Arg 1: table containing the userdata containing the unit or unit type.
 115  * - Arg 2: index (int) or id (string) identifying a particular attack.
 116  * - Ret 1: the unit's attacks.
 117  */
 118 static int impl_unit_attacks_get(lua_State *L)
 119 {
 120         if(!lua_istable(L, 1)) {
 121                 return luaW_type_error(L, 1, "unit attacks");
 122         }
 123         lua_rawgeti(L, 1, 0);
 124         lua_unit* lu = luaW_tounit_ref(L, -1);
 125         const unit_type* ut = luaW_tounittype(L, -1);
 126         if(lu && lu->get()) {
 127                 unit* u = lu->get();
 128                 attack_ptr atk = lua_isnumber(L, 2) ? find_attack(u, luaL_checkinteger(L, 2) - 1) : find_attack(u, luaL_checkstring(L, 2));
 129                 luaW_pushweapon(L, atk);
 130         } else if(ut) {
 131                 const_attack_ptr atk = lua_isnumber(L, 2) ? find_attack(ut, luaL_checkinteger(L, 2) - 1) : find_attack(ut, luaL_checkstring(L, 2));
 132                 luaW_pushweapon(L, atk);
 133         } else {
 134                 return luaL_argerror(L, 1, "unit not found");
 135         }
 136         return 1;
 137 }

 139 static attack_itors::iterator get_attack_iter(unit& u, attack_ptr atk)
 140 {
 141         // This is slightly inefficient since it walks the attack list a second time...
 142         return std::find_if(u.attacks().begin(), u.attacks().end(), [&atk](const attack_type& atk2) {
 143                 return &atk2 == atk.get();
 144         });
 145 }

 147 static int impl_unit_attacks_set(lua_State* L)
 148 {
 149         if(!lua_istable(L, 1)) {
 150                 return luaW_type_error(L, 1, "unit attacks");
 151         }
 152         lua_rawgeti(L, 1, 0);
 153         const unit_type* ut = luaW_tounittype(L, -1);
 154         if(ut) {
 155                 return luaL_argerror(L, 1, "unit type attack table is immutable");
 156         }
 157 
 158         unit& u = luaW_checkunit(L, -1);
 159         attack_ptr atk = lua_isnumber(L, 2) ? find_attack(&u, luaL_checkinteger(L, 2) - 1) : find_attack(&u, luaL_checkstring(L, 2));
 160         if(lua_isnumber(L, 2) && lua_tonumber(L, 2) - 1 > u.attacks().size()) {
 161                 return luaL_argerror(L, 2, "attack can only be added at the end of the list");
 162         }
 163 
 164         if(lua_isnil(L, 3)) {
 165                 // Delete the attack
 166                 u.remove_attack(atk);
 167                 return 0;
 168         }
 169 
 170         auto iter = get_attack_iter(u, atk), end = u.attacks().end();
 171         if(const_attack_ptr atk2 = luaW_toweapon(L, 3)) {
 172                 if(iter == end) {
 173                         atk = u.add_attack(end, *atk2);
 174                 } else {
 175                         iter.base()->reset(new attack_type(*atk2));
 176                         atk = *iter.base();
 177                 }
 178         } else {
 179                 config cfg = luaW_checkconfig(L, 3);
 180                 if(iter == end) {
 181                         atk = u.add_attack(end, cfg);
 182                 } else {
 183                         iter.base()->reset(new attack_type(cfg));
 184                         atk = *iter.base();
 185                 }
 186         }
 187         if(!lua_isnumber(L, 2)) {
 188                 atk->set_id(lua_tostring(L, 2));
 189         }
 190         return 0;
 191 }

 193 /**
 194  * Counts the attacks of a unit (__len metamethod).
 195  * - Arg 1: table containing the userdata containing the unit id.
 196  * - Ret 1: size of unit attacks vector.
 197  */
 198 static int impl_unit_attacks_len(lua_State *L)
 199 {
 200         if(!lua_istable(L, 1)) {
 201                 return luaW_type_error(L, 1, "unit attacks");
 202         }
 203         lua_rawgeti(L, 1, 0);
 204         const unit* u = luaW_tounit(L, -1);
 205         const unit_type* ut = luaW_tounittype(L, -1);
 206         if(!u && !ut) {
 207                 return luaL_argerror(L, 1, "unknown unit");
 208         }
 209         lua_pushinteger(L, (u ? u->attacks() : ut->attacks()).size());
 210         return 1;
 211 }

 213 static int impl_unit_attacks_next(lua_State *L)
 214 {
 215         lua_len(L, 1);
 216         int n = luaL_checknumber(L, 2) + 1;
 217         int max_n = luaL_checknumber(L, -1);
 218         if(n > max_n) {
 219                 return 0;
 220         }
 221         lua_pushnumber(L, n);
 222         lua_pushvalue(L, -1);
 223         lua_gettable(L, 1);
 224         return 2;

 235 /**
 236  * Gets a property of a units attack (__index metamethod).
 237  * - Arg 1: table containing the userdata containing the unit id. and a string identifying the attack.
 238  * - Arg 2: string
 239  * - Ret 1:

 241 static int impl_unit_attack_get(lua_State *L)
 242 {
 243         attack_ref& atk_ref = luaW_checkweapon_ref(L, 1);
 244         const attack_type& attack = *atk_ref.cattack;
 245         char const *m = luaL_checkstring(L, 2);
 246         return_bool_attrib("read_only", atk_ref.attack == nullptr);
 247         return_string_attrib("description", attack.name());
 248         return_string_attrib("name", attack.id());
 249         return_string_attrib("type", attack.type());
 250         return_string_attrib("icon", attack.icon());
 251         return_string_attrib("range", attack.range());
 252         return_int_attrib("damage", attack.damage());
 253         return_int_attrib("number", attack.num_attacks());
 254         return_int_attrib("attack_weight", attack.attack_weight());
 255         return_int_attrib("defense_weight", attack.defense_weight());
 256         return_int_attrib("accuracy", attack.accuracy());
 257         return_int_attrib("movement_used", attack.movement_used());
 258         return_int_attrib("parry", attack.parry());
 259         return_cfgref_attrib("specials", attack.specials());
 260         return_cfgref_attrib("__cfg", attack.to_config());
 261         if(luaW_getmetafield(L, 1, m)) {
 262                 return 1;
 263         }
 264         std::string err_msg = "unknown property of attack: ";
 265         err_msg += m;
 266         return luaL_argerror(L, 2, err_msg.c_str());
 267 }

 280         modify_string_attrib("name", attack.set_id(value));
 281         modify_string_attrib("type", attack.set_type(value));

 283         modify_string_attrib("range", attack.set_range(value));
 284         modify_int_attrib("damage", attack.set_damage(value));
 285         modify_int_attrib("number", attack.set_num_attacks(value));
 286         modify_int_attrib("attack_weight", attack.set_attack_weight(value));
 287         modify_int_attrib("defense_weight", attack.set_defense_weight(value));
 288         modify_int_attrib("accuracy", attack.set_accuracy(value));
 289         modify_int_attrib("movement_used", attack.set_movement_used(value));
 290         modify_int_attrib("parry", attack.set_parry(value));
 291 
 292         if(strcmp(m, "specials") == 0) {
 293                 attack.set_specials(luaW_checkconfig(L, 3));
 294                 return 0;
 295         }
 296 
 297         std::string err_msg = "unknown modifiable property of attack: ";
 298         err_msg += m;
 299         return luaL_argerror(L, 2, err_msg.c_str());

 337                 // Create the unit attacks metatable.

 353                 luaL_newmetatable(L, uattackKey);
 354                 lua_pushcfunction(L, impl_unit_attack_get);

 359                 lua_setfield(L, -2, "__eq");
 360                 lua_pushcfunction(L, impl_unit_attack_collect);
 361                 lua_setfield(L, -2, "__gc");
 362                 lua_pushstring(L, uattackKey);
 363                 lua_setfield(L, -2, "__metatable");
 364                 lua_pushcfunction(L, impl_unit_attack_match);
 365                 lua_setfield(L, -2, "matches");

 367                 // Add create_attack
 368                 luaW_getglobal(L, "wesnoth");
 369                 lua_pushcfunction(L, intf_create_attack);
 370                 lua_setfield(L, -2, "create_weapon");
 371                 lua_pop(L, 1);
 372 
 373                 return cmd_out.str();
