  68 static bool check_side_number(const team &t, const std::string &str)
  69 {
  70                 std::vector<std::pair<int,int> > ranges = utils::parse_ranges(str);
  71                 int side_number = t.side();
  72 
  73                 std::vector<std::pair<int,int> >::const_iterator range, range_end = ranges.end();
  74                 for (range = ranges.begin(); range != range_end; ++range) {
  75                         if(side_number >= range->first && side_number <= range->second) {
  76                                 return true;
  77                         }
  78                 }
  79                 return false;

  82 bool side_filter::match_internal(const team &t) const
  83 {
  84         assert(fc_);
  85 
  86         if (cfg_.has_attribute("side_in")) {
  87                 if (!check_side_number(t,cfg_["side_in"])) {
  88                         return false;
  89                 }
  90         }
  91         if (cfg_.has_attribute("side")) {
  92                 if (!check_side_number(t,cfg_["side"])) {
  93                         return false;
  94                 }
  95         }
  96         if (!side_string_.empty()) {
  97                 if (!check_side_number(t,side_string_)) {
  98                         return false;
  99                 }
 100         }
 101 
 102         config::attribute_value cfg_team_name = cfg_["team_name"];
 103         if (!cfg_team_name.blank()) {
 104                 const std::string& that_team_name = cfg_team_name;
 105                 const std::string& this_team_name = t.team_name();
 106 
 107                 if(std::find(this_team_name.begin(), this_team_name.end(), ',') == this_team_name.end()) {
 108                         if(this_team_name != that_team_name) return false;
 109                 }
 110                 else {
 111                         const std::vector<std::string>& these_team_names = utils::split(this_team_name);
 112                         bool search_futile = true;
 113                         for(const std::string& this_single_team_name : these_team_names) {
 114                                 if(this_single_team_name == that_team_name) {
 115                                         search_futile = false;
 116                                         break;
 117                                 }
 118                         }
 119                         if(search_futile) return false;
 120                 }
 121         }
 122 
 123         //Allow filtering on units
 124         if(cfg_.has_child("has_unit")) {
 125                 const vconfig & ufilt_cfg = cfg_.child("has_unit");
 126                 if (!ufilter_)
 127                         ufilter_.reset(new unit_filter(ufilt_cfg, fc_, flat_));
 128                 bool found = false;
 129                 for(const unit &u : fc_->get_disp_context().units()) {
 130                         if (u.side() != t.side()) {
 131                                 continue;
 132                         }
 133                         if (ufilter_->matches(u)) {
 134                                 found = true;
 135                                 break;
 136                         }
 137                 }
 138                 if(!found && ufilt_cfg["search_recall_list"].to_bool(false)) {
 139                         for(const unit_const_ptr & u : t.recall_list()) {
 140                                 scoped_recall_unit this_unit("this_unit", t.save_id(),t.recall_list().find_index(u->id()));
 141                                 if(ufilter_->matches(*u)) {
 142                                         found = true;
 143                                         break;
 144                                 }
 145                         }
 146                 }
 147                 if (!found) {
 148                         return false;
 149                 }
 150         }
 151 
 152         const vconfig& enemy_of = cfg_.child("enemy_of");
 153         if(!enemy_of.null()) {
 154                 if (!enemy_filter_)
 155                         enemy_filter_.reset(new side_filter(enemy_of, fc_));
 156                 const std::vector<int>& teams = enemy_filter_->get_teams();
 157                 if(teams.empty()) return false;
 158                 for(const int side : teams) {
 159                         if(!(fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 160                                 return false;
 161                 }
 162         }
 163 
 164         const vconfig& allied_with = cfg_.child("allied_with");
 165         if(!allied_with.null()) {
 166                 if (!allied_filter_)
 167                         allied_filter_.reset(new side_filter(allied_with, fc_));
 168                 const std::vector<int>& teams = allied_filter_->get_teams();
 169                 if(teams.empty()) return false;
 170                 for(const int side : teams) {
 171                         if((fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 172                                 return false;
 173                 }
 174         }
 175 
 176         const vconfig& has_enemy = cfg_.child("has_enemy");
 177         if(!has_enemy.null()) {
 178                 if (!has_enemy_filter_)
 179                         has_enemy_filter_.reset(new side_filter(has_enemy, fc_));
 180                 const std::vector<int>& teams = has_enemy_filter_->get_teams();
 181                 bool found = false;
 182                 for(const int side : teams) {
 183                         if((fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 184                         {
 185                                 found = true;
 186                                 break;
 187                         }
 188                 }
 189                 if (!found) return false;
 190         }
 191 
 192         const vconfig& has_ally = cfg_.child("has_ally");
 193         if(!has_ally.null()) {
 194                 if (!has_ally_filter_)
 195                         has_ally_filter_.reset(new side_filter(has_ally, fc_));
 196                 const std::vector<int>& teams = has_ally_filter_->get_teams();
 197                 bool found = false;
 198                 for(const int side : teams) {
 199                         if(!(fc_->get_disp_context().teams())[side - 1].is_enemy(t.side()))
 200                         {
 201                                 found = true;
 202                                 break;
 203                         }
 204                 }
 205                 if (!found) return false;
 206         }
 207 
 208 
 209         const config::attribute_value cfg_controller = cfg_["controller"];
 210         if (!cfg_controller.blank())
 211         {
 212                 if (resources::controller->is_networked_mp() && synced_context::is_synced()) {
 213                         ERR_NG << "ignoring controller= in SSF due to danger of OOS errors" << std::endl;
 214                 }
 215                 else {
 216                         bool found = false;
 217                         for(const std::string& controller : utils::split(cfg_controller))
 218                         {
 219                                 if(t.controller().to_string() == controller) {
 220                                         found = true;
 221                                 }
 222                         }
 223                         if(!found) {
 224                                 return false;
 225                         }
 226                 }
 227         }
 228 
 229         if (cfg_.has_attribute("formula")) {
 230                 try {
 231                         const wfl::team_callable callable(t);
 232                         const wfl::formula form(cfg_["formula"], new wfl::gamestate_function_symbol_table);
 233                         if(!form.evaluate(callable).as_bool()) {
 234                                 return false;
 235                         }
 236                         return true;
 237                 } catch(wfl::formula_error& e) {
 238                         lg::wml_error() << "Formula error in side filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 239                         // Formulae with syntax errors match nothing
 240                         return false;
 241                 }
 242         }
 243 
 244         return true;
 245 }

 253 bool side_filter::match(const team& t) const
 254 {
 255         bool matches = match_internal(t);
 256 
 257         //handle [and], [or], and [not] with in-order precedence
 258         vconfig::all_children_iterator cond = cfg_.ordered_begin();
 259         vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 260         while (cond != cond_end) {
 261                 const std::string& cond_name = cond.get_key();
 262                 const vconfig& cond_cfg = cond.get_child();
 263 
 264                 //handle [and]
 265                 if(cond_name == "and")
 266                 {
 267                         matches = matches && side_filter(cond_cfg, fc_, flat_).match(t);
 268                 }
 269                 //handle [or]
 270                 else if(cond_name == "or")
 271                 {
 272                         matches = matches || side_filter(cond_cfg, fc_, flat_).match(t);
 273                 }
 274                 //handle [not]
 275                 else if(cond_name == "not")
 276                 {
 277                         matches = matches && !side_filter(cond_cfg, fc_, flat_).match(t);
 278                 }
 279                         ++cond;
 280         }
 281         return matches;
 282 }
