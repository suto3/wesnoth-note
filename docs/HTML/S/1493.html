 147     /**
 148      * Loads this class from the {@code .wesnoth} file and
 149      * loads the other cached files
 150      */
 151     public void loadCache( )
 152     {
 153         ResourceUtils.createWesnothFile( wesnothFile_.getAbsolutePath( ), false );
 154 
 155         FileInputStream inputStream = null;
 156         ObjectInputStream deserializer = null;
 157         try {
 158             inputStream = new FileInputStream( wesnothFile_ );
 159             deserializer = new ObjectInputStream(
 160                 inputStream );
 161             ProjectCache cache = ( ProjectCache ) deserializer.readObject( );
 162 
 163             properties_ = cache.properties_;
 164             configFiles_ = cache.configFiles_;
 165             dependTree_ = cache.dependTree_;
 166             variables_ = cache.variables_;
 167 
 168             dependTree_.deserialize( project_ );
 169 
 170             deserializer.close( );
 171             inputStream.close( );
 172         }
 173         // invalid file contents. just save this instance to it.
 174         catch( EOFException e ) {
 175             saveCache( );
 176         } catch( StreamCorruptedException e ) {
 177             saveCache( );
 178         } catch( ClassCastException ex ) {
 179             saveCache( );
 180         } catch( ClassNotFoundException e ) {
 181             saveCache( );
 182         } catch( IOException e ) {
 183             Logger.getInstance( ).logException( e );
 184         } finally {
 185             try {
 186                 if( deserializer != null ) {
 187                     deserializer.close( );
 188                 }
 189 
 190                 if( inputStream != null ) {
 191                     inputStream.close( );
 192                 }
 193             } catch( IOException e ) {
 194                 Logger.getInstance( ).logException( e );
 195             }
 196         }
 197 
 198         readDefines( true );
 199     }

 215     /**
 216      * Saves th defines in the defines file for the project
 217      * 
 218      * @return
 219      */
 220     private boolean saveDefines( )
 221     {
 222         try {
 223             if( ! definesFile_.exists( ) ) {
 224                 ResourceUtils.createNewFile( definesFile_.getAbsolutePath( ) );
 225             }
 226 
 227             FileWriter writer = new FileWriter( definesFile_ );
 228 
 229             try {
 230                 for( Define define: defines_.values( ) ) {
 231                     writer.write( define.toString( ) );
 232                     writer.write( "\n" );
 233                 }
 234             } finally {
 235                 writer.close( );
 236             }
 237         } catch( Exception e ) {
 238             Logger.getInstance( ).logException( e );
 239         }
 240         return false;
 241     }

 243     /**
 244      * Serializes the current ProjectCache instance to the file
 245      * 
 246      * @return
 247      */
 248     private boolean serializeCacheToFile( )
 249     {
 250         ResourceUtils.createWesnothFile( wesnothFile_.getAbsolutePath( ), false );
 251 
 252         try {
 253             FileOutputStream outputStream = new FileOutputStream( wesnothFile_ );
 254             ObjectOutputStream serializer = new ObjectOutputStream( outputStream );
 255             try {
 256                 serializer.writeObject( this );
 257             } finally {
 258                 serializer.close( );
 259                 outputStream.close( );
 260             }
 261 
 262             return true;
 263         } catch( Exception e ) {
 264             Logger.getInstance( ).logException( e );
 265             return false;
 266         }
 267     }

 269     /**
 270      * Reads the defines files for this project
 271      * 
 272      * @param force
 273      *        Read the defines even if the defines file's contents
 274      *        haven't changed since last time read.
 275      */
 276     public void readDefines( boolean force )
 277     {
 278         if( force == false && definesFile_.lastModified( ) <= definesTimestamp_ ) {
 279             return;
 280         }
 281 
 282         if( definesFile_.exists( ) == false ) {
 283             return;
 284         }
 285 
 286         defines_.putAll( Define.readDefines( definesFile_.getAbsolutePath( ) ) );
 287         definesTimestamp_ = definesFile_.lastModified( );
 288     }
