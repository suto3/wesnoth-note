 132 def parse_frequency(string):
 133     if string == "required":
 134         return REQUIRED
 135     elif string == "optional":
 136         return OPTIONAL
 137     elif string == "repeated":
 138         return REPEATED
 139     elif string == "forbidden":
 140         return FORBIDDEN
