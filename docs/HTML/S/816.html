  42 /**
  43 * Change the log level of a log domain.
  44 *
  45 * @param data string of the form: "@<level@> @<domain@>"
  46 */
  47 void chat_handler::change_logging(const std::string& data) {
  48         const std::string::const_iterator j =
  49                 std::find(data.begin(), data.end(), ' ');
  50         if (j == data.end()) return;
  51         const std::string level(data.begin(), j);
  52         const std::string domain(j + 1, data.end());
  53         int severity;
  54         if (level == "error") severity = lg::err().get_severity();
  55         else if (level == "warning") severity = lg::warn().get_severity();
  56         else if (level == "info") severity = lg::info().get_severity();
  57         else if (level == "debug") severity = lg::debug().get_severity();
  58         else {
  59                 utils::string_map symbols;
  60                 symbols["level"] = level;
  61                 const std::string& msg =
  62                         vgettext("Unknown debug level: '$level'.", symbols);
  63                 ERR_NG << msg << std::endl;
  64                 add_chat_message(time(nullptr), _("error"), 0, msg);
  65                 return;
  66         }
  67         if (!lg::set_log_domain_severity(domain, severity)) {
  68                 utils::string_map symbols;
  69                 symbols["domain"] = domain;
  70                 const std::string& msg =
  71                         vgettext("Unknown debug domain: '$domain'.", symbols);
  72                 ERR_NG << msg << std::endl;
  73                 add_chat_message(time(nullptr), _("error"), 0, msg);
  74                 return;
  75         }
  76         else {
  77                 utils::string_map symbols;
  78                 symbols["level"] = level;
  79                 symbols["domain"] = domain;
  80                 const std::string& msg =
  81                         vgettext("Switched domain: '$domain' to level: '$level'.", symbols);
  82                 LOG_NG << msg << "\n";
  83                 add_chat_message(time(nullptr), "log", 0, msg);
  84         }
  85 }

  87 void chat_handler::send_command(const std::string& cmd, const std::string& args /* = "" */) {
  88         config data;
  89         if (cmd == "muteall") {
  90                 data.add_child(cmd);
  91         }
  92         else if (cmd == "query") {
  93                 data.add_child(cmd)["type"] = args;
  94         }
  95         else if (cmd == "ban" || cmd == "unban" || cmd == "kick"
  96                 || cmd == "mute" || cmd == "unmute") {
  97                 data.add_child(cmd)["username"] = args;
  98         }
  99         else if (cmd == "ping") {
 100                 data[cmd] = std::to_string(time(nullptr));
 101         }
 102         else if (cmd == "green") {
 103                 data.add_child("query")["type"] = "lobbymsg @" + args;
 104         }
 105         else if (cmd == "red") {
 106                 data.add_child("query")["type"] = "lobbymsg #" + args;
 107         }
 108         else if (cmd == "yellow") {
 109                 data.add_child("query")["type"] = "lobbymsg <255,255,0>" + args;
 110         }
 111         else if (cmd == "report") {
 112                 data.add_child("query")["type"] = "report " + args;
 113         }
 114         else if (cmd == "join") {
 115                 data.add_child("room_join")["room"] = args;
 116         }
 117         else if (cmd == "part") {
 118                 data.add_child("room_part")["room"] = args;
 119         }
 120         send_to_server(data);
 121 }

 123 void chat_handler::do_speak(const std::string& message, bool allies_only)
 124 {
 125         if (message == "" || message == "/") {
 126                 return;
 127         }
 128         bool is_command = (message[0] == '/');
 129         bool quoted_command = (is_command && message[1] == ' ');
 130 
 131         if (!is_command) {
 132                 send_chat_message(message, allies_only);
 133                 return;
 134         }
 135         else if (quoted_command) {
 136                 send_chat_message(std::string(message.begin() + 2, message.end()), allies_only);
 137                 return;
 138         }
 139         std::string cmd(message.begin() + 1, message.end());
 140         chat_command_handler cch(*this, allies_only);
 141         cch.dispatch(cmd);
 142 }

 148 void chat_handler::send_whisper(const std::string& receiver, const std::string& message)
 149 {
 150         config cwhisper, data;
