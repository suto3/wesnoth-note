  87 static std::string format_level_string(const int level)
  88 {
  89         std::string lvl = std::to_string(level);
  90 
  91         if(level < 1) {
  92                 return "<span color='#969696'>" + lvl + "</span>";
  93         } else if(level == 1) {
  94                 return lvl;
  95         } else if(level == 2) {
  96                 return "<b>" + lvl + "</b>";
  97         } else if(level > 2 ) {
  98                 return"<b><span color='#ffffff'>" + lvl + "</span></b>";
  99         }
 100 
 101         return lvl;
 102 }

 104 static std::string format_cost_string(int unit_recall_cost, const int team_recall_cost)
 105 {
 106         std::stringstream str;
 107 
 108         if(unit_recall_cost < 0) {
 109                 unit_recall_cost = team_recall_cost;
 110         }
 111 
 112         if(unit_recall_cost > team_recall_cost) {
 113                 str << "<span color='#ff0000'>" << unit_recall_cost << "</span>";
 114         } else if(unit_recall_cost == team_recall_cost) {
 115                 str << unit_recall_cost;
 116         } else if(unit_recall_cost < team_recall_cost) {
 117                 str << "<span color='#00ff00'>" << unit_recall_cost << "</span>";
 118         }
 119 
 120         return str.str();
 121 }

 123 static std::string get_title_suffix(int side_num)
 124 {
 125         if(!resources::gameboard) {
 126                 return "";
 127         }
 128 
 129         unit_map& units = resources::gameboard->units();
 130 
 131         int controlled_recruiters = 0;
 132         for(const auto& team : resources::gameboard->teams()) {
 133                 if(team.is_local_human() && !team.recruits().empty() && units.find_leader(team.side()) !=units.end()) {
 134                         ++controlled_recruiters;
 135                 }
 136         }
 137 
 138         std::stringstream msg;
 139         if(controlled_recruiters >= 2) {
 140                 unit_map::const_iterator leader = resources::gameboard->units().find_leader(side_num);
 141                 if(leader != resources::gameboard->units().end() && !leader->name().empty()) {
 142                         msg << " (" << leader->name(); msg << ")";
 143                 }
 144         }
 145 
 146         return msg.str();
 147 }

 149 void unit_recall::pre_show(window& window)
 150 {
 151         label& title = find_widget<label>(&window, "title", true);
 152         title.set_label(title.get_label() + get_title_suffix(team_.side()));
 153 
 154         text_box* filter
 155                         = find_widget<text_box>(&window, "filter_box", false, true);
 156 
 157         filter->set_text_changed_callback(
 158                         std::bind(&unit_recall::filter_text_changed, this, _1, _2));
 159 
 160         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 161 
 162 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 163         connect_signal_notify_modified(*list,
 164                         std::bind(&unit_recall::list_item_clicked,
 165                                 *this, std::ref(window)));
 166 #else
 167         list.set_callback_value_change(
 168                         dialog_callback<unit_recall, &unit_recall::list_item_clicked>);
 169 #endif
 170 
 171         list.clear();
 172 
 173         window.keyboard_capture(filter);
 174         window.add_to_keyboard_chain(&list);
 175 
 176         connect_signal_mouse_left_click(
 177                 find_widget<button>(&window, "dismiss", false),
 178                 std::bind(&unit_recall::dismiss_unit, this, std::ref(window)));
 179 
 180         connect_signal_mouse_left_click(
 181                 find_widget<button>(&window, "show_help", false),
 182                 std::bind(&unit_recall::show_help, this, std::ref(window)));
 183 
 184         for(const unit_const_ptr& unit : recall_list_) {
 185                 std::map<std::string, string_map> row_data;
 186                 string_map column;
 187 
 188                 std::string mods = unit->image_mods();
 189 
 190                 if(unit->can_recruit()) {
 191                         mods += "~BLIT(" + unit::leader_crown() + ")";
 192                 }
 193 
 194                 for(const std::string& overlay : unit->overlays()) {
 195                         mods += "~BLIT(" + overlay + ")";
 196                 }
 197 
 198                 column["use_markup"] = "true";
 199 
 200                 column["label"] = unit->absolute_image() + mods;
 201                 row_data.emplace("unit_image", column);
 202 
 203                 column["label"] = unit->type_name();
 204                 row_data.emplace("unit_type", column);
 205 
 206                 column["label"] = format_cost_string(unit->recall_cost(), team_.recall_cost());
 207                 row_data.emplace("unit_recall_cost", column);
 208 
 209                 const std::string& name = !unit->name().empty() ? unit->name().str() : font::unicode_en_dash;
 210                 column["label"] = name;
 211                 row_data.emplace("unit_name", column);
 212 
 213                 column["label"] = format_level_string(unit->level());
 214                 row_data.emplace("unit_level", column);
 215 
 216                 std::stringstream exp_str;
 217                 exp_str << font::span_color(unit->xp_color()) << unit->experience() << "/"
 218                         << (unit->can_advance() ? std::to_string(unit->max_experience()) : font::unicode_en_dash) << "</span>";
 219 
 220                 column["label"] = exp_str.str();
 221                 row_data.emplace("unit_experience", column);
 222 
 223                 // Since the table widgets use heavy formatting, we save a bare copy
 224                 // of certain options to filter on.
 225                 std::string filter_text = unit->type_name() + " " + name + " " + std::to_string(unit->level());
 226 
 227                 std::string traits;
 228                 for(const std::string& trait : unit->trait_names()) {
 229                         traits += (traits.empty() ? "" : "\n") + trait;
 230                         filter_text += " " + trait;
 231                 }
 232 
 233                 column["label"] = !traits.empty() ? traits : font::unicode_en_dash;
 234                 row_data.emplace("unit_traits", column);
 235 
 236                 list.add_row(row_data);
 237                 filter_options_.push_back(filter_text);
 238         }
 239 
 240         list.register_sorting_option(0, [this](const int i) { return recall_list_[i]->type_name().str(); });
 241         list.register_sorting_option(1, [this](const int i) { return recall_list_[i]->name().str(); });
 242         list.register_sorting_option(2, [this](const int i) { return recall_list_[i]->level(); });
 243         list.register_sorting_option(3, [this](const int i) { return recall_list_[i]->experience(); });
 244         list.register_sorting_option(4, [this](const int i) {
 245                 return !recall_list_[i]->trait_names().empty() ? recall_list_[i]->trait_names().front().str() : "";
 246         });
 247 
 248         list.set_active_sorting_option(sort_last.first >= 0 ? sort_last : sort_default);
 249 
 250         list_item_clicked(window);
 251 }

 253 void unit_recall::dismiss_unit(window& window)
 254 {
 255         LOG_DP << "Recall list units:\n"; dump_recall_list_to_console(recall_list_);
 256 
 257         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 258         const int index = list.get_selected_row();
 259 
 260         const unit& u = *recall_list_[index].get();
 261 
 262         // If the unit is of level > 1, or is close to advancing, we warn the player about it
 263         std::stringstream message;
 264         if(u.loyal()) {
 265                 message << _("This unit is loyal and requires no upkeep.") << " " << (u.gender() == unit_race::MALE
 266                          ? _("Do you really want to dismiss him?")
 267                          : _("Do you really want to dismiss her?"));
 268 
 269         } else if(u.level() > 1) {
 270                 message << _("This unit is an experienced one, having advanced levels.") << " " << (u.gender() == unit_race::MALE
 271                          ? _("Do you really want to dismiss him?")
 272                          : _("Do you really want to dismiss her?"));
 273 
 274         } else if(u.experience() > u.max_experience()/2) {
 275                 message << _("This unit is close to advancing a level.") << " " << (u.gender() == unit_race::MALE
 276                          ? _("Do you really want to dismiss him?")
 277                          : _("Do you really want to dismiss her?"));
 278         }
 279 
 280         if(!message.str().empty()) {
 281                 const int res = gui2::show_message(window.video(), _("Dismiss Unit"), message.str(), message::yes_no_buttons);
 282 
 283                 if(res != gui2::window::OK) {
 284                         return;
 285                 }
 286         }
 287 
 288         recall_list_.erase(recall_list_.begin() + index);
 289 
 290         // Remove the entry from the dialog list
 291         list.remove_row(index);
 292         list_item_clicked(window);
 293 
 294         // Remove the entry from the filter list
 295         filter_options_.erase(filter_options_.begin() + index);
 296         assert(filter_options_.size() == list.get_item_count());
 297 
 298         LOG_DP << "Dismissing a unit, side = " << u.side() << ", id = '" << u.id() << "'\n";
 299         LOG_DP << "That side's recall list:\n";
 300         dump_recall_list_to_console(team_.recall_list());
 301 
 302         // Find the unit in the recall list.
 303         unit_ptr dismissed_unit = team_.recall_list().find_if_matches_id(u.id());
 304         assert(dismissed_unit);
 305 
 306         // Record the dismissal, then delete the unit.
 307         synced_context::run_and_throw("disband", replay_helper::get_disband(dismissed_unit->id()));
 308 
 309         // Close the dialog if all units are dismissed
 310         if(list.get_item_count() == 0) {
 311                 window.set_retval(window::CANCEL);
 312         }
 313 }

 320 void unit_recall::list_item_clicked(window& window)
 321 {
 322         const int selected_row
 323                 = find_widget<listbox>(&window, "recall_list", false).get_selected_row();
 324

 326                 return;
 327         }
 328 
 329         find_widget<unit_preview_pane>(&window, "unit_details", false)

 343 void unit_recall::filter_text_changed(text_box_base* textbox, const std::string& text)
 344 {
 345         window& window = *textbox->get_window();
 346 
 347         listbox& list = find_widget<listbox>(&window, "recall_list", false);
 348 
 349         const std::vector<std::string> words = utils::split(text, ' ');
 350 
 351         if(words == last_words_)
 352                 return;
 353         last_words_ = words;
 354 
 355         boost::dynamic_bitset<> show_items;
 356         show_items.resize(list.get_item_count(), true);
 357 
 358         if(!text.empty()) {
 359                 for(unsigned int i = 0; i < list.get_item_count(); i++) {
 360                         bool found = false;
 361 
 362                         for(const auto & word : words) {
 363                                 found = std::search(filter_options_[i].begin(),
 364                                                         filter_options_[i].end(),
 365                                                         word.begin(),
 366                                                         word.end(),
 367                                                         chars_equal_insensitive)
 368                                                 != filter_options_[i].end();
 369 
 370                                 if(!found) {
 371                                         // one word doesn't match, we don't reach words.end()
 372                                         break;
 373                                 }
 374                         }
 375 
 376                         show_items[i] = found;
 377                 }
 378         }
 379 
 380         list.set_row_shown(show_items);
 381 }
