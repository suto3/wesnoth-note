 260 void menu_handler::recruit(int side_num, const map_location& last_hex)
 261 {
 262         std::vector<const unit_type*> sample_units;
 263 
 264         std::set<std::string> recruits = actions::get_recruits(side_num, last_hex);
 265 
 266         for(const auto& recruit : recruits) {
 267                 const unit_type* type = unit_types.find(recruit);
 268                 if(!type) {
 269                         ERR_NG << "could not find unit '" << recruit << "'" << std::endl;
 270                         return;
 271                 }
 272 
 273                 sample_units.push_back(type);
 274         }
 275 
 276         if(sample_units.empty()) {
 277                 gui2::show_transient_message(gui_->video(), "", _("You have no units available to recruit."));
 278                 return;
 279         }
 280 
 281         gui2::dialogs::unit_recruit dlg(sample_units, board().get_team(side_num));
 282 
 283         dlg.show(gui_->video());
 284 
 285         if(dlg.get_retval() == gui2::window::OK) {
 286                 do_recruit(sample_units[dlg.get_selected_index()]->id(), side_num, last_hex);
 287         }

 298 bool menu_handler::do_recruit(const std::string& name, int side_num, const map_location& last_hex)
 299 {
 300         team& current_team = board().get_team(side_num);
 301 
 302         // search for the unit to be recruited in recruits
 303         if(!utils::contains(actions::get_recruits(side_num, last_hex), name)) {
 304                 return false;
 305         }
 306 
 307         const unit_type* u_type = unit_types.find(name);
 308         assert(u_type);
 309 
 310         if(u_type->cost() > current_team.gold() - (pc_.get_whiteboard()
 311                         ? pc_.get_whiteboard()->get_spent_gold_for(side_num)
 312                         : 0))
 313         {
 314                 gui2::show_transient_message(gui_->video(), "", _("You do not have enough gold to recruit that unit"));
 315                 return false;
 316         }
 317 
 318         current_team.last_recruit(name);
 319         const events::command_disabler disable_commands;
 320 
 321         map_location loc = last_hex;
 322         map_location recruited_from = map_location::null_location();
 323 
 324         std::string msg;
 325         {
 326                 wb::future_map_if_active future; /* start planned unit map scope if in planning mode */
 327                 msg = actions::find_recruit_location(side_num, loc, recruited_from, name);
 328         } // end planned unit map scope
 329 
 330         if(!msg.empty()) {
 331                 gui2::show_transient_message(gui_->video(), "", msg);
 332                 return false;
 333         }
 334 
 335         if(!pc_.get_whiteboard() || !pc_.get_whiteboard()->save_recruit(name, side_num, loc)) {
 336                 // MP_COUNTDOWN grant time bonus for recruiting
 337                 current_team.set_action_bonus_count(1 + current_team.action_bonus_count());
 338 
 339                 // Do the recruiting.
 340 
 341                 synced_context::run_and_throw("recruit", replay_helper::get_recruit(u_type->id(), loc, recruited_from));
 342                 return true;
 343         }
 344 
 345         return false;
 346 }

 348 void menu_handler::recall(int side_num, const map_location& last_hex)
 349 {
 350         if(pc_.get_disallow_recall()) {
 351                 gui2::show_transient_message(
 352                                 gui_->video(), "", _("You are separated from your soldiers and may not recall them"));
 353                 return;
 354         }
 355 
 356         team& current_team = board().get_team(side_num);
 357 
 358         std::vector<unit_const_ptr> recall_list_team;
 359         {
 360                 wb::future_map future; // ensures recall list has planned recalls removed
 361                 recall_list_team = actions::get_recalls(side_num, last_hex);
 362         }
 363 
 364         DBG_WB << "menu_handler::recall: Contents of wb-modified recall list:\n";
 365         for(const unit_const_ptr& unit : recall_list_team) {
 366                 DBG_WB << unit->name() << " [" << unit->id() << "]\n";
 367         }
 368 
 369         if(current_team.recall_list().empty()) {
 370                 gui2::show_transient_message(gui_->video(), "",
 371                         _("There are no troops available to recall\n(You must have veteran survivors from a previous scenario)"));
 372                 return;
 373         }
 374         if(recall_list_team.empty()) {
 375                 gui2::show_transient_message(gui_->video(), "", _("You currently can't recall at the highlighted location"));
 376                 return;
 377         }
 378 
 379         gui2::dialogs::unit_recall dlg(recall_list_team, current_team);
 380 
 381         dlg.show(gui_->video());
 382 
 383         if(dlg.get_retval() != gui2::window::OK) {
 384                 return;
 385         }
 386 
 387         int res = dlg.get_selected_index();
 388         int unit_cost = current_team.recall_cost();
 389 
 390         // we need to check if unit has a specific recall cost
 391         // if it does we use it elsewise we use the team.recall_cost()
 392         // the magic number -1 is what it gets set to if the unit doesn't
 393         // have a special recall_cost of its own.
 394         if(recall_list_team[res]->recall_cost() > -1) {
 395                 unit_cost = recall_list_team[res]->recall_cost();
 396         }
 397 
 398         int wb_gold = pc_.get_whiteboard() ? pc_.get_whiteboard()->get_spent_gold_for(side_num) : 0;
 399         if(current_team.gold() - wb_gold < unit_cost) {
 400                 utils::string_map i18n_symbols;
 401                 i18n_symbols["cost"] = std::to_string(unit_cost);
 402                 std::string msg = VNGETTEXT("You must have at least 1 gold piece to recall a unit",
 403                                 "You must have at least $cost gold pieces to recall this unit", unit_cost, i18n_symbols);
 404                 gui2::show_transient_message(gui_->video(), "", msg);
 405                 return;
 406         }
 407 
 408         LOG_NG << "recall index: " << res << "\n";
 409         const events::command_disabler disable_commands;
 410 
 411         map_location recall_location = last_hex;
 412         map_location recall_from = map_location::null_location();
 413         std::string err;
 414         {
 415                 wb::future_map_if_active
 416                                 future; // future unit map removes invisible units from map, don't do this outside of planning mode
 417                 err = actions::find_recall_location(side_num, recall_location, recall_from, *recall_list_team[res].get());
 418         } // end planned unit map scope
 419 
 420         if(!err.empty()) {
 421                 gui2::show_transient_message(gui_->video(), "", err);
 422                 return;
 423         }
 424 
 425         if(!pc_.get_whiteboard()
 426                         || !pc_.get_whiteboard()->save_recall(*recall_list_team[res].get(), side_num, recall_location)) {
 427                 bool success = synced_context::run_and_throw("recall",
 428                                 replay_helper::get_recall(recall_list_team[res]->id(), recall_location, recall_from), true, true,
 429                                 synced_context::ignore_error_function);
 430 
 431                 if(!success) {
 432                         ERR_NG << "menu_handler::recall(): Unit does not exist in the recall list." << std::endl;
 433                 }
 434         }
 435 }

 495         for(auto& unit : units) {

 518                 }
 519         }

 525 bool menu_handler::end_turn(int side_num)
 526 {
 527         if(!gamedata().allow_end_turn()) {
 528                 gui2::show_transient_message((*gui_).video(), "", _("You cannot end your turn yet!"));
 529                 return false;
 530         }
 531 
 532         size_t team_num = static_cast<size_t>(side_num - 1);
 533         if(team_num < teams().size() && teams()[team_num].no_turn_confirmation()) {
 534                 // Skip the confirmations that follow.
 535         }
 536         // Ask for confirmation if the player hasn't made any moves.
 537         else if(preferences::confirm_no_moves() && !pc_.get_undo_stack().player_acted()
 538                         && (!pc_.get_whiteboard() || !pc_.get_whiteboard()->current_side_has_actions())
 539                         && units_alive(side_num, units())) {
 540                 const int res = gui2::show_message((*gui_).video(), "",
 541                                 _("You have not started your turn yet. Do you really want to end your turn?"),
 542                                 gui2::dialogs::message::yes_no_buttons);
 543                 if(res == gui2::window::CANCEL) {
 544                         return false;
 545                 }
 546         }
 547         // Ask for confirmation if units still have some movement left.
 548         else if(preferences::yellow_confirm() && partmoved_units(side_num, units(), board(), pc_.get_whiteboard())) {
 549                 const int res = gui2::show_message((*gui_).video(), "",
 550                                 _("Some units have movement left. Do you really want to end your turn?"),
 551                                 gui2::dialogs::message::yes_no_buttons);
 552                 if(res == gui2::window::CANCEL) {
 553                         return false;
 554                 }
 555         }
 556         // Ask for confirmation if units still have all movement left.
 557         else if(preferences::green_confirm() && unmoved_units(side_num, units(), board(), pc_.get_whiteboard())) {
 558                 const int res = gui2::show_message((*gui_).video(), "",

 841 void menu_handler::execute_gotos(mouse_handler& mousehandler, int side)
 842 {
 843         // we will loop on all gotos and try to fully move a maximum of them,
 844         // but we want to avoid multiple blocking of the same unit,
 845         // so, if possible, it's better to first wait that the blocker move
 846 
 847         bool wait_blocker_move = true;
 848         std::set<map_location> fully_moved;
 849 
 850         bool change = false;
 851         bool blocked_unit = false;
 852         do {
 853                 change = false;
 854                 blocked_unit = false;
 855                 for(auto& unit : units()) {
 856                         if(unit.side() != side || unit.movement_left() == 0) {
 857                                 continue;
 858                         }
 859 
 860                         const map_location& current_loc = unit.get_location();
 861                         const map_location& goto_loc = unit.get_goto();
 862 
 863                         if(goto_loc == current_loc) {
 864                                 unit.set_goto(map_location());
 865                                 continue;
 866                         }
 867 
 868                         if(!map().on_board(goto_loc)) {
 869                                 continue;
 870                         }
 871 
 872                         // avoid pathfinding calls for finished units
 873                         if(fully_moved.count(current_loc)) {
 874                                 continue;
 875                         }
 876 
 877                         pathfind::marked_route route = mousehandler.get_route(&unit, goto_loc, board().get_team(side));
 878 
 879                         if(route.steps.size() <= 1) { // invalid path
 880                                 fully_moved.insert(current_loc);
 881                                 continue;
 882                         }
 883 
 884                         // look where we will stop this turn (turn_1 waypoint or goto)
 885                         map_location next_stop = goto_loc;
 886                         pathfind::marked_route::mark_map::const_iterator w = route.marks.begin();
 887                         for(; w != route.marks.end(); ++w) {
 888                                 if(w->second.turns == 1) {
 889                                         next_stop = w->first;
 890                                         break;
 891                                 }
 892                         }
 893 
 894                         if(next_stop == current_loc) {
 895                                 fully_moved.insert(current_loc);
 896                                 continue;
 897                         }
 898 
 899                         // we delay each blocked move because some other change
 900                         // may open a another not blocked path
 901                         if(units().count(next_stop)) {
 902                                 blocked_unit = true;
 903                                 if(wait_blocker_move)
 904                                         continue;
 905                         }
 906 
 907                         gui_->set_route(&route);
 908 
 909                         {
 910                                 LOG_NG << "execute goto from " << route.steps.front() << " to " << route.steps.back() << "\n";
 911                                 int moves = actions::move_unit_and_record(route.steps, &pc_.get_undo_stack());
 912                                 change = moves > 0;
 913                         }
 914 
 915                         if(change) {
 916                                 // something changed, resume waiting blocker (maybe one can move now)
 917                                 wait_blocker_move = true;
 918                         }
 919                 }
 920 
 921                 if(!change && wait_blocker_move) {
 922                         // no change when waiting, stop waiting and retry
 923                         wait_blocker_move = false;
 924                         change = true;
 925                 }
 926         } while(change && blocked_unit);
 927 
 928         // erase the footsteps after movement
 929         gui_->set_route(nullptr);
 930         gui_->invalidate_game_status();
 931 }

1287 void menu_handler::do_search(const std::string& new_search)
1288 {
1289         if(new_search.empty() == false && new_search != last_search_)
1290                 last_search_ = new_search;
1291 
1292         if(last_search_.empty())
1293                 return;
1294 
1295         bool found = false;
1296         map_location loc = last_search_hit_;
1297         // If this is a location search, just center on that location.
1298         std::vector<std::string> args = utils::split(last_search_, ',');
1299         if(args.size() == 2) {
1300                 int x, y;
1301                 x = lexical_cast_default<int>(args[0], 0) - 1;
1302                 y = lexical_cast_default<int>(args[1], 0) - 1;
1303                 if(x >= 0 && x < map().w() && y >= 0 && y < map().h()) {
1304                         loc = map_location(x, y);
1305                         found = true;
1306                 }
1307         }
1308         // Start scanning the game map
1309         if(loc.valid() == false) {
1310                 loc = map_location(map().w() - 1, map().h() - 1);
1311         }
1312 
1313         map_location start = loc;
1314         while(!found) {
1315                 // Move to the next location
1316                 loc.x = (loc.x + 1) % map().w();
1317                 if(loc.x == 0)
1318                         loc.y = (loc.y + 1) % map().h();
1319 
1320                 // Search label
1321                 if(!gui_->shrouded(loc)) {
1322                         const terrain_label* label = gui_->labels().get_label(loc);
1323                         if(label) {
1324                                 std::string label_text = label->text().str();
1325                                 if(std::search(label_text.begin(), label_text.end(), last_search_.begin(), last_search_.end(),
1326                                                    chars_equal_insensitive)
1327                                                 != label_text.end()) {
1328                                         found = true;
1329                                 }
1330                         }
1331                 }
1332                 // Search unit name
1333                 if(!gui_->fogged(loc)) {
1334                         unit_map::const_iterator ui = units().find(loc);
1335                         if(ui != units().end()) {
1336                                 const std::string name = ui->name();
1337                                 if(std::search(
1338                                                    name.begin(), name.end(), last_search_.begin(), last_search_.end(), chars_equal_insensitive)
1339                                                 != name.end()) {
1340                                         if(!teams()[gui_->viewing_team()].is_enemy(ui->side())
1341                                                         || !ui->invisible(ui->get_location(), gui_->get_disp_context())) {
1342                                                 found = true;
1343                                         }
1344                                 }
1345                         }
1346                 }
1347 
1348                 if(loc == start)
1349                         break;
1350         }
1351 
1352         if(found) {
1353                 last_search_hit_ = loc;
1354                 gui_->scroll_to_tile(loc, game_display::ONSCREEN, false);
1355                 gui_->highlight_hex(loc);
1356         } else {
1357                 last_search_hit_ = map_location();
1358                 // Not found, inform the player
1359                 utils::string_map symbols;
1360                 symbols["search"] = last_search_;
1361                 const std::string msg = vgettext("Could not find label or unit "
1362                                                                                  "containing the string ‘$search’.",
1363                                 symbols);
1364                 gui2::show_message(gui_->video(), "", msg, gui2::dialogs::message::auto_close);
1365         }
1366 }

1736 void console_handler::do_set_alias()
1737 {
1738         const std::string data = get_data();
1739         const std::string::const_iterator j = std::find(data.begin(), data.end(), '=');
1740         const std::string alias(data.begin(), j);
1741         if(j != data.end()) {
1742                 const std::string command(j + 1, data.end());
1743                 if(!command.empty()) {
1744                         register_alias(command, alias);
1745                 } else {

1747                         // equal to itself here. Later preferences will filter empty alias.

1799 void console_handler::do_unit()
1800 {
1801         // prevent SIGSEGV due to attempt to set HP during a fight
1802         if(events::commands_disabled > 0) {
1803                 return;
1804         }
1805 
1806         unit_map::iterator i = menu_handler_.current_unit();
1807         if(i == menu_handler_.units().end()) {
1808                 return;
1809         }
1810 
1811         const map_location loc = i->get_location();
1812         const std::string data = get_data(1);
1813         std::vector<std::string> parameters = utils::split(data, '=', utils::STRIP_SPACES);
1814         if(parameters.size() < 2) {
1815                 return;
1816         }
1817 
1818         if(parameters[0] == "alignment") {
1819                 unit_type::ALIGNMENT alignment;
1820                 if(!alignment.parse(parameters[1])) {
1821                         utils::string_map symbols;
1822                         symbols["alignment"] = get_arg(1);
1823                         command_failed(VGETTEXT(
1824                                         "Invalid alignment: '$alignment', needs to be one of lawful, neutral, chaotic, or liminal.",
1825                                         symbols));
1826                         return;
1827                 }
1828         }
1829 
1830         synced_context::run_and_throw("debug_unit",
1831                 config {
1832                         "x", loc.wml_x(),
1833                         "y", loc.wml_y(),
1834                         "name", parameters[0],
1835                         "value", parameters[1],
1836                 }
1837         );
1838 }
