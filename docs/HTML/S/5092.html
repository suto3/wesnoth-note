   6 #define ABILITY_INITIATIVE
   7     [dummy]
   8         id=initiative
   9         name= _ "initiative"
  10         description= _ "All adjacent friendly units will strike first in melee combat, even when defending."
  11     [/dummy]
  12 #enddef
  13 
  14 #define SPECIAL_NOTES_INITIATIVE
  15 _"This unit's grasp of melee tactics allows adjacent allies to strike the first blow even when defending."#enddef

  17 #define INITIATIVE_OBJECT FILTER
  18     [object]
  19         silent=yes
  20 
  21         [filter]
  22             {FILTER}
  23         [/filter]
  24 
  25         [effect]
  26             apply_to=attack
  27             range=melee
  28 
  29             [set_specials]
  30                 mode=append
  31 
  32                 [firststrike]
  33                     id=initiative
  34                     name=""
  35                     description=""
  36                     [filter_self]
  37                         [filter_adjacent]
  38                             id="Li'sar"
  39                         [/filter_adjacent]
  40                     [/filter_self]
  41                 [/firststrike]
  42             [/set_specials]
  43         [/effect]
  44     [/object]
  45 #enddef
  46 
  47 #define LACKS_INITIATIVE_FILTER
  48     side=1,2
  49 
  50     [filter_side]
  51         [allied_with]
  52             [has_unit]
  53                 id="Li'sar"
  54             [/has_unit]
  55         [/allied_with]
  56     [/filter_side]
  57 
  58     [filter_wml]
  59         [attack]
  60             range=melee
  61         [/attack]
  62     [/filter_wml]
  63 
  64     [not]
  65         [filter_wml]
  66             [attack]
  67                 range=melee
  68                 [specials]
  69                     [firststrike]
  70                     [/firststrike]
  71                 [/specials]
  72             [/attack]
  73         [/filter_wml]
  74     [/not]
  75 #enddef
