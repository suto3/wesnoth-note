   3 # Finds the nearest hex to (X,Y) within MAX_DISTANCE that matches the FILTER, and stores the location in VARIABLE with x, y, and terrain. If there is more than one location found, VARIABLE will be an array.
   4 #define NEAREST_HEX X Y MAX_DISTANCE_NUMBER FILTER VAR_NAME
   5     [set_variable]
   6         name=distance
   7         value=0
   8     [/set_variable]
   9 
  10     [set_variable]
  11         name=location_found
  12         value=no
  13     [/set_variable]
  14 
  15     # Store any matching locations within a radius of "distance". If nothing got stored, add one to "distance" and try again. Do this until a location is found, or "distance" goes over the MAX_DISTANCE_NUMBER.
  16     [while]
  17         [variable]
  18             name=location_found
  19             equals=no
  20         [/variable]
  21         [and]
  22             [variable]
  23                 name=distance
  24                 less_than_equal_to={MAX_DISTANCE_NUMBER}
  25             [/variable]
  26         [/and]
  27 
  28         [do]
  29             [store_locations]
  30                 {FILTER}
  31                 [and]
  32                     x={X}
  33                     y={Y}
  34                     radius=$distance
  35                 [/and]
  36                 variable={VAR_NAME}
  37             [/store_locations]
  38 
  39             [if]
  40                 [variable]
  41                     name={VAR_NAME}.length
  42                     greater_than=0
  43                 [/variable]
  44                 [then]
  45                     [set_variable]
  46                         name=location_found
  47                         value=yes
  48                     [/set_variable]
  49                 [/then]
  50             [/if]
  51 
  52             [set_variable]
  53                 name=distance
  54                 add=1
  55             [/set_variable]
  56         [/do]
  57     [/while]
  58 
  59     [clear_variable]
  60         name=distance
  61     [/clear_variable]
  62 
  63     [clear_variable]
  64         name=location_found
  65     [/clear_variable]
  66 #enddef
