  97 void playmp_controller::play_linger_turn()
  98 {
  99         if (is_host()) {
 100                 end_turn_enable(true);
 101         }
 102 
 103         while(end_turn_ == END_TURN_NONE) {
 104                 config cfg;
 105                 if(network_reader_.read(cfg)) {
 106                         if(turn_data_.process_network_data(cfg) == turn_info::PROCESS_END_LINGER)
 107                         {
 108                                 end_turn();

 110                 }
 111                 play_slice();
 112                 gui_->draw();

 116 void playmp_controller::play_human_turn()
 117 {
 118         LOG_NG << "playmp::play_human_turn...\n";
 119         assert(!linger_);
 120         remove_blindfold();
 121         const std::unique_ptr<countdown_clock> timer(saved_game_.mp_settings().mp_countdown
 122         ? new countdown_clock(current_team())
 123         : nullptr);
 124         show_turn_dialog();
 125         if(undo_stack().can_undo()) {
 126                 // If we reload a networked mp game we cannot undo moves made before the save
 127                 // Becasue other players already received them
 128                 if(!current_team().auto_shroud_updates()) {
 129                         synced_context::run_and_store("update_shroud", replay_helper::get_update_shroud());
 130                 }
 131                 undo_stack().clear();
 132         }
 133         if (!preferences::disable_auto_moves()) {
 134                 execute_gotos();
 135         }
 136 
 137         end_turn_enable(true);
 138         while(!should_return_to_play_side()) {
 139                 try {
 140                         process_network_data();
 141                         if (player_type_changed_)
 142                         {
 143                                 // Clean undo stack if turn has to be restarted (losing control)
 144                                 if ( undo_stack().can_undo() )
 145                                 {
 146                                         font::floating_label flabel(_("Undoing moves not yet transmitted to the server."));
 147 
 148                                         color_t color {255,255,255,SDL_ALPHA_OPAQUE};
 149                                         flabel.set_color(color);
 150                                         SDL_Rect rect = gui_->map_area();
 151                                         flabel.set_position(rect.w/2, rect.h/2);
 152                                         flabel.set_lifetime(150);
 153                                         flabel.set_clip_rect(rect);
 154 
 155                                         font::add_floating_label(flabel);
 156                                 }
 157 
 158                                 while( undo_stack().can_undo() )
 159                                         undo_stack().undo();
 160 
 161                         }
 162                         check_objectives();
 163                         play_slice_catch();
 164                         if(timer)
 165                         {
 166                                 bool time_left = timer->update();
 167                                 if(!time_left)
 168                                 {
 169                                         end_turn_ = END_TURN_REQUIRED;
 170                                 }
 171                         }
 172                 }
 173                 catch(...)
 174                 {
 175                         turn_data_.send_data();
 176                         throw;
 177                 }
 178                 turn_data_.send_data();
 179 
 180                 gui_->draw();
 181         }
 182 }

 229 void playmp_controller::linger()
 230 {
 231         LOG_NG << "beginning end-of-scenario linger\n";
 232         linger_ = true;
 233         // If we need to set the status depending on the completion state
 234         // we're needed here.
 235         gui_->set_game_mode(game_display::LINGER);
 236         // End all unit moves
 237         gamestate().board_.set_all_units_user_end_turn();
 238 
 239         set_end_scenario_button();
 240         assert(is_regular_game_end());
 241         if ( get_end_level_data_const().transient.reveal_map ) {
 242                 // Change the view of all players and observers
 243                 // to see the whole map regardless of shroud and fog.
 244                 update_gui_to_player(gui_->viewing_team(), true);
 245         }
 246         bool quit;
 247         do {
 248                 quit = true;
 249                 try {
 250                         // reimplement parts of play_side()
 251                         turn_data_.send_data();
 252                         end_turn_ = END_TURN_NONE;
 253                         play_linger_turn();
 254                         after_human_turn();
 255                         LOG_NG << "finished human turn" << std::endl;
 256                 } catch (savegame::load_game_exception&) {
 257                         LOG_NG << "caught load-game-exception" << std::endl;
 258                         // this should not happen, the option to load a game is disabled
 259                         throw;
 260                 } catch (ingame_wesnothd_error&) {
 261                         LOG_NG << "caught network-error-exception" << std::endl;
 262                         quit = false;
 263                 }
 264         } while (!quit);
 265 
 266         reset_end_scenario_button();
 267 
 268         LOG_NG << "ending end-of-scenario linger\n";
 269 }

 271 void playmp_controller::wait_for_upload()
 272 {
 273         // If the host is here we'll never leave since we wait for the host to
 274         // upload the next scenario.
 275         assert(!is_host());
 276 
 277         config cfg;
 278         network_reader_.set_source(playturn_network_adapter::get_source_from_config(cfg));
 279         while(true) {
 280                 try {
 281                         const bool  res = gui2::dialogs::network_transmission::wesnothd_receive_dialog(
 282                                 gui_->video(), "next scenario", cfg, mp_info_->connection);
 283 
 284                         if(res) {
 285                                 if (turn_data_.process_network_data_from_reader() == turn_info::PROCESS_END_LINGER) {
 286                                         break;
 287                                 }
 288                         }
 289                         else
 290                         {
 291                                 throw_quit_game_exception();
 292                         }
 293 
 294                 } catch(const quit_game_exception&) {
 295                         network_reader_.set_source([this](config& cfg) { return recieve_from_wesnothd(cfg);});
 296                         turn_data_.send_data();
 297                         throw;
 298                 }
 299         }
 300         network_reader_.set_source([this](config& cfg) { return recieve_from_wesnothd(cfg);});
 301 }

 303 void playmp_controller::after_human_turn(){
 304         if(saved_game_.mp_settings().mp_countdown)
 305         {
 306                 //time_left + turn_bonus + (action_bouns * number of actions done)
 307                 const int new_time_in_secs = (current_team().countdown_time() / 1000)
 308                         + saved_game_.mp_settings().mp_countdown_turn_bonus
 309                         + saved_game_.mp_settings().mp_countdown_action_bonus * current_team().action_bonus_count();
 310                 const int new_time = 1000 * std::min<int>(new_time_in_secs, saved_game_.mp_settings().mp_countdown_reservoir_time);
 311 
 312                 current_team().set_action_bonus_count(0);
 313                 current_team().set_countdown_time(new_time);
 314                 resources::recorder->add_countdown_update(new_time, current_side());
 315         }
 316         LOG_NG << "playmp::after_human_turn...\n";
 317 
 318         // Normal post-processing for human turns (clear undos, end the turn, etc.)
 319         playsingle_controller::after_human_turn();
 320         //send one more time to make sure network is up-to-date.
 321         turn_data_.send_data();
 322 
 323 }

 325 void playmp_controller::play_network_turn(){
 326         LOG_NG << "is networked...\n";
 327 
 328         end_turn_enable(false);
 329         turn_data_.send_data();
 330 
 331         while(end_turn_ != END_TURN_SYNCED && !is_regular_game_end() && !player_type_changed_)
 332         {
 333                 if (!network_processing_stopped_) {
 334                         process_network_data();
 335                         if (!mp_info_ || mp_info_->current_turn == turn()) {
 336                                 skip_replay_ = false;
 337                         }
 338                 }
 339 
 340                 play_slice_catch();
 341                 if (!network_processing_stopped_){
 342                         turn_data_.send_data();
 343                 }
 344 
 345                 gui_->draw();
 346         }
 347 
 348         LOG_NG << "finished networked...\n";
 349 }

 352 void playmp_controller::process_oos(const std::string& err_msg) const {
 353         // Notify the server of the oos error.
 354         config cfg;
 355         config& info = cfg.add_child("info");
 356         info["type"] = "termination";
 357         info["condition"] = "out of sync";
 358         send_to_wesnothd(cfg);
 359 
 360         std::stringstream temp_buf;
 361         std::vector<std::string> err_lines = utils::split(err_msg,'\n');
 362         temp_buf << _("The game is out of sync, and cannot continue. There are a number of reasons this could happen: this can occur if you or another player have modified their game settings. This may mean one of the players is attempting to cheat. It could also be due to a bug in the game, but this is less likely.\n\nDo you want to save an error log of your game?");
 363         if(!err_msg.empty()) {
 364                 temp_buf << " \n \n"; //and now the "Details:"
 365                 for(std::vector<std::string>::iterator i=err_lines.begin(); i!=err_lines.end(); ++i)
 366                 {
 367                         temp_buf << *i << '\n';
 368                 }
 369                 temp_buf << " \n";
 370         }
 371         scoped_savegame_snapshot snapshot(*this);
 372         savegame::oos_savegame save(saved_game_, *gui_, ignore_replay_errors_);
 373         save.save_game_interactive(gui_->video(), temp_buf.str(), savegame::savegame::YES_NO);
 374 }

 376 void playmp_controller::handle_generic_event(const std::string& name){
 377         turn_data_.send_data();
 378 
 379         if (name == "ai_user_interact")
 380         {
 381                 playsingle_controller::handle_generic_event(name);
 382                 turn_data_.send_data();
 383         }
 384         else if (name == "ai_gamestate_changed")
 385         {
 386                 turn_data_.send_data();
 387         }
 388         else if (name == "host_transfer"){
 389                 assert(mp_info_);
 390                 mp_info_->is_host = true;
 391                 if (linger_){
 392                         end_turn_enable(true);
 393                         gui_->invalidate_theme();
 394                 }
 395         }

 409 void playmp_controller::maybe_linger()
 410 {
 411         // mouse_handler expects at least one team for linger mode to work.
 412         assert(is_regular_game_end());
 413         if (!get_end_level_data_const().transient.linger_mode || gamestate().board_.teams().empty() || gui_->video().faked()) {
 414                 const bool has_next_scenario = !gamestate().gamedata_.next_scenario().empty() && gamestate().gamedata_.next_scenario() != "null";
 415                 if(!is_host() && has_next_scenario) {
 416                         // If we continue without lingering we need to
 417                         // make sure the host uploads the next scenario
 418                         // before we attempt to download it.
 419                         wait_for_upload();
 420                 }
 421         } else {
 422                 linger();
 423         }
 424 }

 426 void playmp_controller::pull_remote_choice()
 427 {
 428         // when using a remote user choice undoing must be impossible because that network traffic cannot be undone
 429         // Also turn_data_.sync_network() (which calls turn_data_.send_data()) won't work if the undo stack isn't empty because undoable moves won't be sended
 430         // Also undo_stack()clear() must be called synced so we cannot do that here.
 431         assert(!current_team().is_local() || !undo_stack().can_undo());
 432         turn_info::PROCESS_DATA_RESULT res = turn_data_.sync_network();
 433         assert(res != turn_info::PROCESS_END_LINGER);
 434         assert(res != turn_info::PROCESS_END_TURN);
 435         if(res == turn_info::PROCESS_RESTART_TURN)
 436         {
 437                 player_type_changed_ = true;
 438         }
 439 }

 450 void playmp_controller::process_network_data()
 451 {
 452         if(end_turn_ == END_TURN_SYNCED || is_regular_game_end() || player_type_changed_) {
 453                 return;
 454         }
 455         turn_info::PROCESS_DATA_RESULT res = turn_info::PROCESS_CONTINUE;
 456         config cfg;
 457         if(!resources::recorder->at_end()) {
 458                 res = turn_info::replay_to_process_data_result(do_replay());
 459         }
 460         else if(network_reader_.read(cfg)) {
 461                 res = turn_data_.process_network_data(cfg);
 462         }
 463 
 464         if (res == turn_info::PROCESS_RESTART_TURN) {
 465                 player_type_changed_ = true;
 466         }
 467         else if (res == turn_info::PROCESS_END_TURN) {
 468                 end_turn_ = END_TURN_SYNCED;
 469         }
 470         else if (res == turn_info::PROCESS_END_LEVEL) {
 471         }
 472         else if (res == turn_info::PROCESS_END_LINGER) {
 473                 replay::process_error("Received unexpected next_scenario during the game");
 474         }
 475 }
