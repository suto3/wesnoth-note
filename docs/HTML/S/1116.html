 178 void configure_engine::write_parameters()
 179 {
 180         config& scenario = this->state_.get_starting_pos();
 181         const mp_game_settings& params = this->state_.mp_settings();
 182 
 183         if (params.random_start_time) {
 184                 if (!tod_manager::is_start_ToD(scenario["random_start_time"])) {
 185                         scenario["random_start_time"] = true;
 186                 }
 187         }
 188         else {
 189                 scenario["random_start_time"] = false;
 190         }
 191         scenario["experience_modifier"] = params.xp_modifier;
 192         scenario["turns"] = params.num_turns;
 193 
 194         for (config& side : scenario.child_range("side"))
 195         {
 196                 side["fog"] = params.fog_game;
 197                 side["shroud"] = params.shroud_game;
 198                 if (!params.use_map_settings) {
 199                         side["village_gold"] = params.village_gold;
 200                         side["village_support"] = params.village_support;
 201                 }
 202         }
 203 }
