  66          * This ensures that if an event handler adds more handlers, the new handlers
  67          * won't be called for the event that caused them to be added.

  72 bool context::remove_handler(sdl_handler* ptr)
  73 {
  74         static int depth = 0;
  75         ++depth;
  76 
  77         // The handler is most likely on the back of the events list,
  78         // so look there first, otherwise do a complete search.
  79         if(!handlers.empty() && handlers.back() == ptr) {
  80                 if(focused_handler != handlers.end() && *focused_handler == ptr) {
  81                         focused_handler = handlers.end();
  82                 }
  83 
  84                 handlers.pop_back();
  85         } else {
  86                 const handler_list::iterator i = std::find(handlers.begin(), handlers.end(), ptr);
  87 
  88                 if(i == handlers.end()) {
  89                         --depth;
  90 
  91                         // The handler may be in the staging area. Search it from there.
  92                         auto j = std::find(staging_handlers.begin(), staging_handlers.end(), ptr);
  93                         if(j != staging_handlers.end()) {
  94                                 staging_handlers.erase(j);
  95                                 return true;
  96                         } else {
  97                                 return false;
  98                         }
  99                 }
 100 
 101                 if(i == focused_handler) {
 102                         focused_handler != handlers.begin() ? --focused_handler : ++focused_handler;
 103                 }
 104 
 105                 handlers.erase(i);
 106         }
 107 
 108         --depth;
 109 
 110         if(depth == 0) {
 111                 cycle_focus();
 112         } else {
 113                 focused_handler = handlers.end();
 114         }
 115 
 116         return true;
 117 }

 119 void context::cycle_focus()
 120 {
 121         if(handlers.begin() == handlers.end()) {
 122                 return;
 123         }
 124 
 125         handler_list::iterator current = focused_handler;
 126         handler_list::iterator last = focused_handler;
 127 
 128         if(last != handlers.begin()) {
 129                 --last;
 130         }
 131 
 132         if(current == handlers.end()) {
 133                 current = handlers.begin();
 134         } else {
 135                 ++current;
 136         }
 137 
 138         while(current != last) {
 139                 if(current != handlers.end() && (*current)->requires_event_focus()) {
 140                         focused_handler = current;
 141                         break;
 142                 }
 143 
 144                 if(current == handlers.end()) {
 145                         current = handlers.begin();
 146                 } else {
 147                         ++current;
 148                 }
 149         }
 150 }

 240 void sdl_handler::join(context& c)
 241 {
 242         if(has_joined_global_) {
 243                 leave_global();
 244         }
 245 
 246         if(has_joined_) {
 247                 leave(); // should not be in multiple event contexts
 248         }
 249 
 250         // join self
 251         c.add_handler(this);
 252         has_joined_ = true;
 253 
 254         // instruct members to join
 255         for(auto member : handler_members()) {
 256                 member->join(c);
 257         }

 260 void sdl_handler::join_same(sdl_handler* parent)
 261 {
 262         if(has_joined_) {
 263                 leave(); // should not be in multiple event contexts
 264         }

 267                 handler_list& handlers = context.handlers;

 277 void sdl_handler::leave()
 278 {
 279         sdl_handler_vector members = handler_members();
 280 
 281         if(members.empty()) {
 282                 assert(event_contexts.empty() == false);
 283         }
 284 
 285         for(auto member : members) {
 286                 member->leave();
 287         }
 288 
 289         for(auto& context : boost::adaptors::reverse(event_contexts)) {
 290                 if(context.remove_handler(this)) {
 291                         break;
 292                 }
 293         }
 294 
 295         has_joined_ = false;
 296 }

 336 bool has_focus(const sdl_handler* hand, const SDL_Event* event)
 337 {
 338         if(event_contexts.empty()) {
 339                 return true;
 340         }
 341 
 342         if(hand->requires_event_focus(event) == false) {
 343                 return true;
 344         }
 345 
 346         const handler_list::iterator foc = event_contexts.back().focused_handler;
 347         auto& handlers = event_contexts.back().handlers;
 348 
 349         // If no-one has focus at the moment, this handler obviously wants
 350         // focus, so give it to it.
 351         if(foc == handlers.end()) {
 352                 focus_handler(hand);
 353                 return true;
 354         }
 355 
 356         sdl_handler* const foc_hand = *foc;
 357         if(foc_hand == hand) {
 358                 return true;
 359         } else if(!foc_hand->requires_event_focus(event)) {
 360                 // If the currently focused handler doesn't need focus for this event
 361                 // allow the most recent interested handler to take care of it
 362                 for(auto i = handlers.rbegin(); i != handlers.rend(); ++i) {
 363                         sdl_handler* const thief_hand = *i;
 364 
 365                         if(thief_hand != foc_hand && thief_hand->requires_event_focus(event)) {
 366                                 // Steal focus
 367                                 focus_handler(thief_hand);
 368 
 369                                 // Position the previously focused handler to allow stealing back
 370                                 handlers.splice(handlers.end(), handlers, foc);
 371 
 372                                 return thief_hand == hand;
 373                         }
 374                 }
 375         }
 376 
 377         return false;
 378 }

 384 static bool remove_on_resize(const SDL_Event& a)
 385 {
 386         if(a.type == DRAW_EVENT || a.type == DRAW_ALL_EVENT) {
 387                 return true;
 388         }
 389 
 390         if(a.type == SHOW_HELPTIP_EVENT) {
 391                 return true;
 392         }
 393 
 394         if((a.type == SDL_WINDOWEVENT) && (
 395                 a.window.event == SDL_WINDOWEVENT_RESIZED ||
 396                 a.window.event == SDL_WINDOWEVENT_SIZE_CHANGED ||
 397                 a.window.event == SDL_WINDOWEVENT_EXPOSED)
 398         ) {
 399                 return true;
 400         }
 401 
 402         return false;
 403 }

 408 void pump()
 409 {
 410         if(boost::this_thread::get_id() != main_thread) {
 411                 // Can only call this on the main thread!
 412                 return;
 413         }
 414 
 415         peek_for_resize();
 416         pump_info info;
 417 
 418         // Used to keep track of double click events
 419         static int last_mouse_down = -1;
 420         static int last_click_x = -1, last_click_y = -1;
 421 
 422         SDL_Event temp_event;
 423         int poll_count = 0;
 424         int begin_ignoring = 0;
 425 
 426         std::vector<SDL_Event> events;
 427         while(SDL_PollEvent(&temp_event)) {
 428                 if(temp_event.type == INVOKE_FUNCTION_EVENT) {
 429                         static_cast<invoked_function_data*>(temp_event.user.data1)->call();
 430                         continue;
 431                 }
 432 
 433                 ++poll_count;
 434                 peek_for_resize();
 435 
 436                 if(!begin_ignoring && temp_event.type == SDL_WINDOWEVENT && (
 437                         temp_event.window.event == SDL_WINDOWEVENT_ENTER ||
 438                         temp_event.window.event == SDL_WINDOWEVENT_FOCUS_GAINED)
 439                 ) {
 440                         begin_ignoring = poll_count;
 441                 } else if(begin_ignoring > 0 && is_input(temp_event)) {
 442                         // ignore user input events that occurred after the window was activated
 443                         continue;
 444                 }
 445 
 446                 events.push_back(temp_event);
 447         }
 448 
 449         std::vector<SDL_Event>::iterator ev_it = events.begin();
 450         for(int i = 1; i < begin_ignoring; ++i) {
 451                 if(is_input(*ev_it)) {
 452                         // ignore user input events that occurred before the window was activated
 453                         ev_it = events.erase(ev_it);
 454                 } else {
 455                         ++ev_it;
 456                 }
 457         }
 458 
 459         std::vector<SDL_Event>::iterator ev_end = events.end();
 460         bool resize_found = false;
 461         for(ev_it = events.begin(); ev_it != ev_end; ++ev_it) {
 462                 SDL_Event& event = *ev_it;
 463                 if(event.type == SDL_WINDOWEVENT && event.window.event == SDL_WINDOWEVENT_RESIZED) {
 464                         resize_found = true;
 465                         last_resize_event = event;
 466                         last_resize_event_used = false;
 467                 }
 468         }
 469 
 470         // remove all inputs, draw events and only keep the last of the resize events
 471         // This will turn horrible after ~38 days when the Uint32 wraps.
 472         if(resize_found || SDL_GetTicks() <= last_resize_event.window.timestamp + resize_timeout) {
 473                 events.erase(std::remove_if(events.begin(), events.end(), remove_on_resize), events.end());
 474         } else if(SDL_GetTicks() > last_resize_event.window.timestamp + resize_timeout && !last_resize_event_used) {
 475                 events.insert(events.begin(), last_resize_event);
 476                 last_resize_event_used = true;
 477         }
 478 
 479         // Move all draw events to the end of the queue
 480         auto first_draw_event = std::stable_partition(events.begin(), events.end(),
 481                 [](const SDL_Event& e) { return e.type != DRAW_EVENT; }
 482         );
 483 
 484         if(first_draw_event != events.end()) {
 485                 // Remove all draw events except one
 486                 events.erase(first_draw_event + 1, events.end());
 487         }
 488 
 489         ev_end = events.end();
 490 
 491         for(ev_it = events.begin(); ev_it != ev_end; ++ev_it) {
 492                 for(context& c : event_contexts) {
 493                         c.add_staging_handlers();
 494                 }
 495 
 496                 SDL_Event& event = *ev_it;
 497                 switch(event.type) {
 498                 case SDL_WINDOWEVENT:
 499                         switch(event.window.event) {
 500                         case SDL_WINDOWEVENT_ENTER:
 501                         case SDL_WINDOWEVENT_FOCUS_GAINED:
 502                                 cursor::set_focus(1);
 503                                 break;
 504 
 505                         case SDL_WINDOWEVENT_LEAVE:
 506                         case SDL_WINDOWEVENT_FOCUS_LOST:
 507                                 cursor::set_focus(1);
 508                                 break;
 509 
 510                         case SDL_WINDOWEVENT_RESIZED:
 511                                 info.resize_dimensions.first = event.window.data1;
 512                                 info.resize_dimensions.second = event.window.data2;
 513                                 break;
 514                         }
 515 
 516                         // make sure this runs in it's own scope.
 517                         {
 518                                 flip_locker flip_lock(CVideo::get_singleton());
 519                                 for(auto& context : event_contexts) {
 520                                         for(auto handler : context.handlers) {
 521                                                 handler->handle_window_event(event);
 522                                         }
 523                                 }
 524 
 525                                 for(auto global_handler : event_contexts.front().handlers) {
 526                                         global_handler->handle_window_event(event);
 527                                 }
 528                         }
 529 
 530                         // This event was just distributed, don't re-distribute.
 531                         continue;
 532 
 533                 case SDL_MOUSEMOTION: {
 534                         // Always make sure a cursor is displayed if the mouse moves or if the user clicks
 535                         cursor::set_focus(true);
 536                         raise_help_string_event(event.motion.x, event.motion.y);
 537                         break;
 538                 }
 539 
 540                 case SDL_MOUSEBUTTONDOWN: {
 541                         // Always make sure a cursor is displayed if the mouse moves or if the user clicks
 542                         cursor::set_focus(true);
 543                         if(event.button.button == SDL_BUTTON_LEFT) {
 544                                 static const int DoubleClickTime = 500;
 545                                 static const int DoubleClickMaxMove = 3;
 546 
 547                                 if(last_mouse_down >= 0 && info.ticks() - last_mouse_down < DoubleClickTime
 548                                                 && std::abs(event.button.x - last_click_x) < DoubleClickMaxMove
 549                                                 && std::abs(event.button.y - last_click_y) < DoubleClickMaxMove
 550                                 ) {
 551                                         SDL_UserEvent user_event;
 552                                         user_event.type = DOUBLE_CLICK_EVENT;
 553                                         user_event.code = 0;
 554                                         user_event.data1 = reinterpret_cast<void*>(event.button.x);
 555                                         user_event.data2 = reinterpret_cast<void*>(event.button.y);
 556                                         ::SDL_PushEvent(reinterpret_cast<SDL_Event*>(&user_event));
 557                                 }
 558 
 559                                 last_mouse_down = info.ticks();
 560                                 last_click_x = event.button.x;
 561                                 last_click_y = event.button.y;
 562                         }
 563                         break;
 564                 }
 565 
 566                 case DRAW_ALL_EVENT: {
 567                         flip_locker flip_lock(CVideo::get_singleton());
 568 
 569                         /* Iterate backwards as the most recent things will be at the top */
 570                         // FIXME? ^ that isn't happening here.
 571                         for(auto& context : event_contexts) {
 572                                 for(auto handler : context.handlers) {
 573                                         handler->handle_event(event);
 574                                 }
 575                         }
 576 
 577                         continue; // do not do further handling here
 578                 }
 579 
 580 #ifndef __APPLE__
 581                 case SDL_KEYDOWN: {
 582                         if(event.key.keysym.sym == SDLK_F4 &&
 583                                 (event.key.keysym.mod == KMOD_RALT || event.key.keysym.mod == KMOD_LALT)
 584                         ) {
 585                                 quit_confirmation::quit_to_desktop();
 586                                 continue; // this event is already handled
 587                         }
 588                         break;
 589                 }
 590 #endif
 591 
 592 #if defined(_X11) && !defined(__APPLE__)
 593                 case SDL_SYSWMEVENT: {
 594                         // clipboard support for X11
 595                         desktop::clipboard::handle_system_event(event);
 596                         break;
 597                 }
 598 #endif
 599 
 600 #if defined _WIN32
 601                 case SDL_SYSWMEVENT: {
 602                         windows_tray_notification::handle_system_event(event);
 603                         break;
 604                 }
 605 #endif
 606 
 607                 case SDL_QUIT: {
 608                         quit_confirmation::quit_to_desktop();
 609                         continue; // this event is already handled.
 610                 }
 611                 }
 612 
 613                 for(auto global_handler : event_contexts.front().handlers) {
 614                         global_handler->handle_event(event);
 615                 }
 616 
 617                 if(event_contexts.empty() == false) {
 618                         for(auto handler : event_contexts.back().handlers) {
 619                                 handler->handle_event(event);
 620                         }
 621                 }
 622         }
 623 
 624         // Inform the pump monitors that an events::pump() has occurred
 625         for(auto monitor : pump_monitors) {
 626                 monitor->process(info);
 627         }
 628 }

 735 void peek_for_resize()
 736 {
 737         SDL_Event events[100];
 738         int num = SDL_PeepEvents(events, 100, SDL_PEEKEVENT, SDL_WINDOWEVENT, SDL_WINDOWEVENT);
 739         for(int i = 0; i < num; ++i) {
 740                 if(events[i].type == SDL_WINDOWEVENT && events[i].window.event == SDL_WINDOWEVENT_RESIZED) {
 741                         CVideo::get_singleton().update_framebuffer();
 742                 }
 743         }
 744 }

 746 void call_in_main_thread(const std::function<void(void)>& f)
 747 {
 748         if(boost::this_thread::get_id() == main_thread) {
 749                 // nothing special to do if called from the main thread.
 750                 f();
 751                 return;
 752         }
 753 
 754         invoked_function_data fdata{false, f};
 755 
 756         SDL_Event sdl_event;
 757         SDL_UserEvent sdl_userevent;
 758 
 759         sdl_userevent.type = INVOKE_FUNCTION_EVENT;
 760         sdl_userevent.code = 0;
 761         sdl_userevent.data1 = &fdata;
 762         sdl_userevent.data2 = nullptr;
 763 
 764         sdl_event.type = INVOKE_FUNCTION_EVENT;
 765         sdl_event.user = sdl_userevent;
 766 
 767         SDL_PushEvent(&sdl_event);
 768 
 769         while(!fdata.finished) {
 770                 SDL_Delay(10);
 771         }
 772 }
