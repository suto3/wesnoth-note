 105 std::unique_ptr<widget> grid::swap_child(const std::string& id,
 106                                                    widget* w,
 107                                                    const bool recurse,
 108                                                    widget* new_parent)
 109 {
 110         assert(w);
 111 
 112         for(auto & child : children_)
 113         {
 114                 if(child.id() != id) {
 115 
 116                         if(recurse) {
 117                                 // decent in the nested grids.
 118                                 grid* g = dynamic_cast<grid*>(child.get_widget());
 119                                 if(g) {
 120 
 121                                         std::unique_ptr<widget> old = g->swap_child(id, w, true);
 122                                         if(old) {
 123                                                 return old;
 124                                         }
 125                                 }
 126                         }
 127 
 128                         continue;
 129                 }
 130 
 131                 // When find the widget there should be a widget.
 132                 widget* old = child.get_widget();
 133                 assert(old);
 134                 old->set_parent(new_parent);
 135 
 136                 w->set_parent(this);
 137                 w->set_visible(old->get_visible());
 138                 child.set_widget(w);
 139 
 140                 return std::unique_ptr<widget>(old);
 141         }
 142 
 143         return nullptr;
 144 }

 243 void grid::request_reduce_width(const unsigned maximum_width)
 244 {
 245         point size = get_best_size();
 246         if(size.x <= static_cast<int>(maximum_width)) {
 247                 /** @todo this point shouldn't be reached, find out why it does. */
 248                 return;
 249         }
 250 
 251         const unsigned too_wide = size.x - maximum_width;
 252         unsigned reduced = 0;
 253         for(size_t col = 0; col < cols_; ++col) {
 254                 if(too_wide - reduced >= col_width_[col]) {
 255                         DBG_GUI_L << LOG_HEADER << " column " << col
 256                                           << " is too small to be reduced.\n";
 257                         continue;
 258                 }
 259 
 260                 const unsigned wanted_width = col_width_[col] - (too_wide - reduced);
 261                 const unsigned width
 262                                 = grid_implementation::column_request_reduce_width(
 263                                                 *this, col, wanted_width);
 264 
 265                 if(width < col_width_[col]) {
 266                         unsigned reduction = col_width_[col] - width;
 267 
 268                         DBG_GUI_L << LOG_HEADER << " reduced " << reduction
 269                                           << " pixels for column " << col << ".\n";
 270 
 271                         size.x -= reduction;
 272                         reduced += reduction;
 273                 }
 274 
 275                 if(size.x <= static_cast<int>(maximum_width)) {
 276                         break;
 277                 }
 278         }
 279 
 280         set_layout_size(calculate_best_size());
 281 }

 321 void grid::request_reduce_height(const unsigned maximum_height)
 322 {
 323         point size = get_best_size();
 324         if(size.y <= static_cast<int>(maximum_height)) {
 325                 /** @todo this point shouldn't be reached, find out why it does. */
 326                 return;
 327         }
 328 
 329         const unsigned too_high = size.y - maximum_height;
 330         unsigned reduced = 0;
 331         for(size_t row = 0; row < rows_; ++row) {
 332                 unsigned wanted_height = row_height_[row] - (too_high - reduced);
 333                 /**
 334                  * @todo Improve this code.
 335                  *
 336                  * Now we try every item to be reduced, maybe items need a flag whether
 337                  * or not to try to reduce and also evaluate whether the force
 338                  * reduction is still needed.
 339                  */
 340                 if(too_high - reduced >= row_height_[row]) {
 341                         DBG_GUI_L << LOG_HEADER << " row " << row << " height "
 342                                           << row_height_[row] << " want to reduce " << too_high
 343                                           << " is too small to be reduced fully try 1 pixel.\n";
 344 
 345                         wanted_height = 1;
 346                 }
 347 
 348                 /* Reducing the height of a widget causes the widget to save its new size
 349                 in widget::layout_size_. After that, get_best_size() will return that
 350                 size and not the originally calculated optimal size.
 351                 Thus, it's perfectly correct that grid::calculate_best_size() that we
 352                 call later calls get_best_size() for child widgets as if size reduction
 353                 had never happened. */
 354                 const unsigned height = grid_implementation::row_request_reduce_height(
 355                                 *this, row, wanted_height);
 356 
 357                 if(height < row_height_[row]) {
 358                         unsigned reduction = row_height_[row] - height;
 359 
 360                         DBG_GUI_L << LOG_HEADER << " row " << row << " height "
 361                                           << row_height_[row] << " want to reduce " << too_high
 362                                           << " reduced " << reduction << " pixels.\n";
 363 
 364                         size.y -= reduction;
 365                         reduced += reduction;
 366                 }
 367 
 368                 if(size.y <= static_cast<int>(maximum_height)) {
 369                         break;
 370                 }
 371         }
 372 
 373         size = calculate_best_size();
 374 
 375         DBG_GUI_L << LOG_HEADER << " Requested maximum " << maximum_height
 376                           << " resulting height " << size.y << ".\n";
 377 
 378         set_layout_size(size);
 379 }

 386 void grid::request_placement(dispatcher&, const event::ui_event, bool& handled, bool&)
 387 {
 388         if (get_window()->invalidate_layout_blocked()) {
 389                 handled = true;
 390                 return;
 391         }
 392 
 393         point size = get_size();
 394         point best_size = calculate_best_size();
 395         if(size.x >= best_size.x && size.y >= best_size.y) {
 396                 place(get_origin(), size);
 397                 handled = true;
 398                 return;
 399         }
 400 
 401         recalculate_best_size();
 402 
 403         if(size.y >= best_size.y) {
 404                 // We have enough space in the Y direction, but not in the X direction.
 405                 // Try wrapping the content.
 406                 request_reduce_width(size.x);
 407                 best_size = get_best_size();
 408 
 409                 if(size.x >= best_size.x && size.y >= best_size.y) {
 410                         // Wrapping succeeded, we still fit vertically.
 411                         place(get_origin(), size);
 412                         handled = true;
 413                         return;
 414                 } else {
 415                         // Wrapping failed, we no longer fit.
 416                         // Reset the sizes of child widgets.
 417                         layout_initialize(true);
 418                 }
 419         }
 420 
 421         /*
 422         Not enough space.
 423         Let the event flow higher up.
 424         This is a pre-event handler, so the event flows upwards. */
 425 }

 434 point grid::calculate_best_size() const
 435 {
 436         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 437 
 438         // Reset the cached values.
 439         row_height_.clear();
 440         row_height_.resize(rows_, 0);
 441         col_width_.clear();
 442         col_width_.resize(cols_, 0);
 443 
 444         // First get the sizes for all items.
 445         for(unsigned row = 0; row < rows_; ++row) {
 446                 for(unsigned col = 0; col < cols_; ++col) {
 447 
 448                         const point size = get_child(row, col).get_best_size();
 449 
 450                         if(size.x > static_cast<int>(col_width_[col])) {
 451                                 col_width_[col] = size.x;
 452                         }
 453 
 454                         if(size.y > static_cast<int>(row_height_[row])) {
 455                                 row_height_[row] = size.y;
 456                         }
 457                 }
 458         }
 459 
 460         for(unsigned row = 0; row < rows_; ++row) {
 461                 DBG_GUI_L << LOG_HEADER << " the row_height_ for row " << row
 462                                   << " will be " << row_height_[row] << ".\n";
 463         }
 464 
 465         for(unsigned col = 0; col < cols_; ++col) {
 466                 DBG_GUI_L << LOG_HEADER << " the col_width_ for column " << col
 467                                   << " will be " << col_width_[col] << ".\n";
 468         }
 469 
 470         const point result(
 471                         std::accumulate(col_width_.begin(), col_width_.end(), 0),
 472                         std::accumulate(row_height_.begin(), row_height_.end(), 0));
 473 
 474         DBG_GUI_L << LOG_HEADER << " returning " << result << ".\n";
 475         return result;
 476 }

 491 void grid::place(const point& origin, const point& size)
 492 {
 493         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 494 
 495         /***** INIT *****/
 496 
 497         widget::place(origin, size);
 498 
 499         if(!rows_ || !cols_) {
 500                 return;
 501         }
 502 
 503         // call the calculate so the size cache gets updated.
 504         const point best_size = calculate_best_size();
 505 
 506         assert(row_height_.size() == rows_);
 507         assert(col_width_.size() == cols_);
 508         assert(row_grow_factor_.size() == rows_);
 509         assert(col_grow_factor_.size() == cols_);
 510 
 511         DBG_GUI_L << LOG_HEADER << " best size " << best_size << " available size "
 512                           << size << ".\n";
 513 
 514         /***** BEST_SIZE *****/
 515 
 516         if(best_size == size) {
 517                 layout(origin);
 518                 return;
 519         }
 520 
 521         if(best_size.x > size.x || best_size.y > size.y) {
 522                 // The assertion below fails quite often so try to give as much information as possible.
 523                 std::stringstream out;
 524                 out << " Failed to place a grid, we have " << size << " space but we need " << best_size << " space.";
 525                 out << " This happened at a grid with the id '" << id() << "'";
 526                 widget* pw = parent();
 527                 while(pw != nullptr) {
 528                         out << " in a '" << typeid(*pw).name() << "' with the id '" << pw->id() << "'";
 529                         pw = pw->parent();
 530                 }
 531                 ERR_GUI_L << LOG_HEADER << out.str() << ".\n";
 532 
 533                 return;
 534         }
 535 
 536         /***** GROW *****/
 537 
 538         // expand it.
 539         if(size.x > best_size.x) {
 540                 const unsigned w = size.x - best_size.x;
 541                 unsigned w_size = std::accumulate(
 542                         col_grow_factor_.begin(), col_grow_factor_.end(), 0);
 543 
 544                 DBG_GUI_L << LOG_HEADER << " extra width " << w
 545                         << " will be divided amount " << w_size << " units in "
 546                         << cols_ << " columns.\n";
 547 
 548                 if(w_size == 0) {
 549                         // If all sizes are 0 reset them to 1
 550                         for(auto & val : col_grow_factor_)
 551                         {
 552                                 val = 1;
 553                         }
 554                         w_size = cols_;
 555                 }
 556                 // We might have a bit 'extra' if the division doesn't fix exactly
 557                 // but we ignore that part for now.
 558                 const unsigned w_normal = w / w_size;
 559                 for(unsigned i = 0; i < cols_; ++i) {
 560                         col_width_[i] += w_normal * col_grow_factor_[i];
 561                         DBG_GUI_L << LOG_HEADER << " column " << i
 562                                 << " with grow factor " << col_grow_factor_[i]
 563                         << " set width to " << col_width_[i] << ".\n";
 564                 }
 565         }
 566 
 567         if(size.y > best_size.y) {
 568                 const unsigned h = size.y - best_size.y;
 569                 unsigned h_size = std::accumulate(
 570                         row_grow_factor_.begin(), row_grow_factor_.end(), 0);
 571                 DBG_GUI_L << LOG_HEADER << " extra height " << h
 572                         << " will be divided amount " << h_size << " units in "
 573                         << rows_ << " rows.\n";
 574 
 575                 if(h_size == 0) {
 576                         // If all sizes are 0 reset them to 1
 577                         for(auto & val : row_grow_factor_)
 578                         {
 579                                 val = 1;
 580                         }
 581                         h_size = rows_;
 582                 }
 583                 // We might have a bit 'extra' if the division doesn't fix exactly
 584                 // but we ignore that part for now.
 585                 const unsigned h_normal = h / h_size;
 586                 for(unsigned i = 0; i < rows_; ++i) {
 587                         row_height_[i] += h_normal * row_grow_factor_[i];
 588                         DBG_GUI_L << LOG_HEADER << " row " << i << " with grow factor "
 589                                 << row_grow_factor_[i] << " set height to "
 590                                 << row_height_[i] << ".\n";
 591                 }
 592         }
 593 
 594         layout(origin);
 595         return;
 596 }

 697         }
 698

 777 void grid::child::place(point origin, point size)
 778 {
 779         assert(get_widget());
 780         if(get_widget()->get_visible() == widget::visibility::invisible) {
 781                 return;
 782         }
 783 
 784         if(border_size_) {
 785                 if(flags_ & BORDER_TOP) {
 786                         origin.y += border_size_;
 787                         size.y -= border_size_;
 788                 }
 789                 if(flags_ & BORDER_BOTTOM) {
 790                         size.y -= border_size_;
 791                 }
 792 
 793                 if(flags_ & BORDER_LEFT) {
 794                         origin.x += border_size_;
 795                         size.x -= border_size_;
 796                 }
 797                 if(flags_ & BORDER_RIGHT) {
 798                         size.x -= border_size_;
 799                 }
 800         }
 801 
 802         // If size smaller or equal to best size set that size.
 803         // No need to check > min size since this is what we got.
 804         const point best_size = get_widget()->get_best_size();
 805         if(size <= best_size) {
 806                 DBG_GUI_L << LOG_CHILD_HEADER
 807                                   << " in best size range setting widget to " << origin << " x "
 808                                   << size << ".\n";
 809 
 810                 get_widget()->place(origin, size);
 811                 return;
 812         }
 813 
 814         const styled_widget* control = dynamic_cast<const styled_widget*>(get_widget());
 815         const point maximum_size = control ? control->get_config_maximum_size()
 816                                                                                 : point();
 817 
 818         if((flags_ & (HORIZONTAL_MASK | VERTICAL_MASK))
 819            == (HORIZONTAL_GROW_SEND_TO_CLIENT | VERTICAL_GROW_SEND_TO_CLIENT)) {
 820 
 821                 if(maximum_size == point() || size <= maximum_size) {
 822 
 823                         DBG_GUI_L << LOG_CHILD_HEADER
 824                                           << " in maximum size range setting widget to " << origin
 825                                           << " x " << size << ".\n";
 826 
 827                         get_widget()->place(origin, size);
 828                         return;
 829                 }
 830         }
 831 
 832         point widget_size = point(std::min(size.x, best_size.x), std::min(size.y, best_size.y));
 833         point widget_orig = origin;
 834 
 835         const unsigned v_flag = flags_ & VERTICAL_MASK;
 836 
 837         if(v_flag == VERTICAL_GROW_SEND_TO_CLIENT) {
 838                 if(maximum_size.y) {
 839                         widget_size.y = std::min(size.y, maximum_size.y);
 840                 } else {
 841                         widget_size.y = size.y;
 842                 }
 843                 DBG_GUI_L << LOG_CHILD_HEADER << " vertical growing from "
 844                                   << best_size.y << " to " << widget_size.y << ".\n";
 845 
 846         } else if(v_flag == VERTICAL_ALIGN_TOP) {
 847                 // Do nothing.
 848 
 849                 DBG_GUI_L << LOG_CHILD_HEADER << " vertically aligned at the top.\n";
 850 
 851         } else if(v_flag == VERTICAL_ALIGN_CENTER) {
 852 
 853                 widget_orig.y += (size.y - widget_size.y) / 2;
 854                 DBG_GUI_L << LOG_CHILD_HEADER << " vertically centered.\n";
 855 
 856         } else if(v_flag == VERTICAL_ALIGN_BOTTOM) {
 857 
 858                 widget_orig.y += (size.y - widget_size.y);
 859                 DBG_GUI_L << LOG_CHILD_HEADER << " vertically aligned at the bottom.\n";
 860 
 861         } else {
 862                 ERR_GUI_L << LOG_CHILD_HEADER << " Invalid vertical alignment '"
 863                                   << v_flag << "' specified.\n";
 864                 assert(false);
 865         }
 866 
 867         const unsigned h_flag = flags_ & HORIZONTAL_MASK;
 868 
 869         if(h_flag == HORIZONTAL_GROW_SEND_TO_CLIENT) {
 870                 if(maximum_size.x) {
 871                         widget_size.x = std::min(size.x, maximum_size.x);
 872                 } else {
 873                         widget_size.x = size.x;
 874                 }
 875                 DBG_GUI_L << LOG_CHILD_HEADER << " horizontal growing from "
 876                                   << best_size.x << " to " << widget_size.x << ".\n";
 877 
 878         } else if(h_flag == HORIZONTAL_ALIGN_LEFT) {
 879                 // Do nothing.
 880                 DBG_GUI_L << LOG_CHILD_HEADER << " horizontally aligned at the left.\n";
 881 
 882         } else if(h_flag == HORIZONTAL_ALIGN_CENTER) {
 883 
 884                 widget_orig.x += (size.x - widget_size.x) / 2;
 885                 DBG_GUI_L << LOG_CHILD_HEADER << " horizontally centered.\n";
 886 
 887         } else if(h_flag == HORIZONTAL_ALIGN_RIGHT) {
 888 
 889                 widget_orig.x += (size.x - widget_size.x);
 890                 DBG_GUI_L << LOG_CHILD_HEADER
 891                                   << " horizontally aligned at the right.\n";
 892 
 893         } else {
 894                 ERR_GUI_L << LOG_CHILD_HEADER << " No horizontal alignment '" << h_flag
 895                                   << "' specified.\n";
 896                 assert(false);
 897         }
 898 
 899         DBG_GUI_L << LOG_CHILD_HEADER << " resize widget to " << widget_orig
 900                           << " x " << widget_size << ".\n";
 901 
 902         get_widget()->place(widget_orig, widget_size);
 903 }

 955 void grid::set_child_alignment(widget* widget, unsigned set_flag, unsigned mode_mask)
 956 {
 957         grid::child* cell = get_child(widget);
 958         if(!cell) {
 959                 return;
 960         }
 961 
 962         unsigned flags = cell->get_flags();
 963 
 964         if((flags & mode_mask) == HORIZONTAL_GROW_SEND_TO_CLIENT) {
 965                 ERR_GUI_G << "Cannot set horizontal alignment (grid cell specifies dynamic growth)" << std::endl;
 966                 return;
 967         }
 968 
 969         if((flags & mode_mask) == VERTICAL_GROW_SEND_TO_CLIENT) {
 970                 ERR_GUI_G << "Cannot set vertical alignment (grid cell specifies dynamic growth)" << std::endl;
 971                 return;
 972         }
 973 
 974         flags &= ~mode_mask;
 975         flags |= set_flag;
 976 
 977         cell->set_flags(flags);
 978 
 979         event::message message;
 980         fire(event::REQUEST_PLACEMENT, *this, message);
 981 }

 983 void grid::layout(const point& origin)
 984 {
 985         point orig = origin;
 986         for(unsigned row = 0; row < rows_; ++row) {
 987                 for(unsigned col = 0; col < cols_; ++col) {
 988 
 989                         const point size(col_width_[col], row_height_[row]);
 990                         DBG_GUI_L << LOG_HEADER << " set widget at " << row << ',' << col
 991                                           << " at origin " << orig << " with size " << size
 992                                           << ".\n";
 993 
 994                         if(get_child(row, col).get_widget()) {
 995                                 get_child(row, col).place(orig, size);
 996                         }
 997 
 998                         orig.x += col_width_[col];
 999                 }
1000                 orig.y += row_height_[row];
1001                 orig.x = origin.x;
1002         }
1003 }

1005 void grid::impl_draw_children(surface& frame_buffer, int x_offset, int y_offset)
1006 {
1007         /*
1008          * The call to SDL_PumpEvents seems a bit like black magic.
1009          * With the call the resizing doesn't seem to lose resize events.
1010          * But when added the queue still only contains one resize event and the
1011          * internal SDL queue doesn't seem to overflow (rarely more than 50 pending
1012          * events).
1013          * Without the call when resizing larger a black area of remains, this is
1014          * the area not used for resizing the screen, this call `fixes' that.
1015          */
1016 
1017         assert(get_visible() == widget::visibility::visible);
1018         set_is_dirty(false);
1019 
1020         for(auto & child : children_)
1021         {
1022 
1023                 widget* widget = child.get_widget();
1024                 assert(widget);
1025 
1026                 if(widget->get_visible() != widget::visibility::visible) {
1027                         continue;
1028                 }
1029 
1030                 if(widget->get_drawing_action() == widget::redraw_action::none) {
1031                         continue;
1032                 }
1033 
1034                 widget->draw_background(frame_buffer, x_offset, y_offset);
1035                 widget->draw_children(frame_buffer, x_offset, y_offset);
1036                 widget->draw_foreground(frame_buffer, x_offset, y_offset);
1037                 widget->set_is_dirty(false);
1038         }
1039 }

1132 /*WIKI
1133  * @page = GUILayout
1134  *
1135  * {{Autogenerated}}
1136  *
1137  * = Abstract =
1138  *
1139  * In the widget library the placement and sizes of elements is determined by
1140  * a grid. Therefore most widgets have no fixed size.
1141  *
1142  *
1143  * = Theory =
1144  *
1145  * We have two examples for the addon dialog, the first example the lower
1146  * buttons are in one grid, that means if the remove button gets wider
1147  * (due to translations) the connect button (4.1 - 2.2) will be aligned
1148  * to the left of the remove button. In the second example the connect
1149  * button will be partial underneath the remove button.
1150  *
1151  * A grid exists of x rows and y columns for all rows the number of columns
1152  * needs to be the same, there is no column (nor row) span. If spanning is
1153  * required place a nested grid to do so. In the examples every row has 1 column
1154  * but rows 3, 4 (and in the second 5) have a nested grid to add more elements
1155  * per row.
1156  *
1157  * In the grid every cell needs to have a widget, if no widget is wanted place
1158  * the special widget ''spacer''. This is a non-visible item which normally
1159  * shouldn't have a size. It is possible to give a spacer a size as well but
1160  * that is discussed elsewhere.
1161  *
1162  * Every row and column has a ''grow_factor'', since all columns in a grid are
1163  * aligned only the columns in the first row need to define their grow factor.
1164  * The grow factor is used to determine with the extra size available in a
1165  * dialog. The algorithm determines the extra size work like this:
1166  *
1167  * * determine the extra size
1168  * * determine the sum of the grow factors
1169  * * if this sum is 0 set the grow factor for every item to 1 and sum to sum of items.
1170  * * divide the extra size with the sum of grow factors
1171  * * for every item multiply the grow factor with the division value
1172  *
1173  * eg
1174  *  extra size 100
1175  *  grow factors 1, 1, 2, 1
1176  *  sum 5
1177  *  division 100 / 5 = 20
1178  *  extra sizes 20, 20, 40, 20
1179  *
1180  * Since we force the factors to 1 if all zero it's not possible to have non
1181  * growing cells. This can be solved by adding an extra cell with a spacer and a
1182  * grow factor of 1. This is used for the buttons in the examples.
1183  *
1184  * Every cell has a ''border_size'' and ''border'' the ''border_size'' is the
1185  * number of pixels in the cell which aren't available for the widget. This is
1186  * used to make sure the items in different cells aren't put side to side. With
1187  * ''border'' it can be determined which sides get the border. So a border is
1188  * either 0 or ''border_size''.

1191  * determines where in the cell the widget is placed.
