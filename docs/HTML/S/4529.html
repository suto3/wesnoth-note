   8 # The maximum width of the text
   9 #define MAX_TEXT_WIDTH
  10 675
  11 #enddef
  12 
  13 #define ___GUI_IMAGE_WIDTH
  14 (
  15         max(250, min(500, best_size))
  16 where
  17         best_size = (
  18                 if((best_width * aspect_ratio) > max_height,
  19                         floor(max_height / aspect_ratio),
  20                         best_width
  21                 )
  22         where
  23                 best_width = (
  24                         if(gamemap_width > text_width_saturation,
  25                                 gamemap_width - {MAX_TEXT_WIDTH},
  26                                 gamemap_width / size_factor
  27                         )
  28                 where
  29                         # gamemap_width width needed to reach the MAX_TEXT_WIDTH #
  30                         text_width_saturation = 3 * height_offset / 2 where height_offset = 25,
  31                         size_factor = 3
  32                 ),
  33                 aspect_ratio = (
  34                         if(image_original_width > 0,
  35                                 as_decimal(image_original_height) / image_original_width,
  36                                 0
  37                         )
  38                 ),
  39                 max_height = (gamemap_height - 4)
  40         )
  41 )#enddef

  64 # This is the macro for the image width which is can be used as string
  65 # parameter.

  69 # the value the image needs to be scaled by, this version can be used as string
  70 # parameter.
  71 #define _GUI_IMAGE_SCALE_FACTOR
  72 (
  73         {__GUI_IMAGE_WIDTH} / 500
  74 )#enddef

  77 # The formula to set the widthe of the real image width depending on the
  78 # image real size
  79 #define __GUI_IMAGE_DISPLAYED_WIDTH
  80 (
  81         if((image_original_width < 300) and (image_original_height < 300),
  82                 image_original_width,
  83                 ((image_original_width * calculated_width) / 500)
  84         )
  85 where calculated_width = {__GUI_IMAGE_WIDTH}
  86 )#enddef
  87 
  88 # The formula to set the heighte of the real image height depending on the
  89 # image real size
  90 #define __GUI_IMAGE_DISPLAYED_HEIGHT
  91 (
  92         if((image_original_width < 300) and (image_original_height < 300),
  93                 image_original_height,
  94                 ((image_original_height * calculated_width) / 500)
  95         )
  96 where calculated_width = {__GUI_IMAGE_WIDTH}
  97 )#enddef
  98 
  99 # The X location of the image to display, only tested on the left side.
 100 #define __GUI_IMAGE_DISPLAYED_X
 101 (
 102         if(image_original_width > 100,
 103                 0,
 104                 (calculated_width - image_width) / 2
 105         )
 106 where calculated_width = {__GUI_IMAGE_WIDTH}
 107 )#enddef
 108 
 109 # The Y location of the image to display, only tested on the left side.
 110 # Note since we don't know the exact height of the text we can't centre
 111 # the image. Instead we use the centre of the reserved image space.
 112 #
 113 # TODO the Y can't be determined properly since we don't know the height
 114 # of the panel. Disabled the centring for now.
 115 #
 116 #define __GUI_IMAGE_DISPLAYED_Y
 117 (
 118         if(image_original_height > 0,
 119                 (height - image_height),
 120                 height - ((calculated_width - image_height) / 2)
 121         )
 122 where calculated_width = {__GUI_IMAGE_WIDTH}
 123 )#enddef
