   8 ;; Update WML information using WML built-in to Wesnoth.
   9 
  10 ;; The following should be added to your .emacs so that `wesnoth-update' can
  11 ;; correctly generate WML data:
  12 ;;   (setq wesnoth-root-directory "/path/to/wesnoth/"
  13 ;;         wesnoth-update-output-directory "/path/to/wesnoth-mode/"
  14 ;;         wesnoth-addition-file
  15 ;;         "/path/to/wesnoth-mode/wesnoth-wml-additions.cfg")
  16 ;; Specifying the appropriate path in each case.
  17 
  18 ;; Although WML data is provided along with wesnoth-mode, you can generate
  19 ;; update-to-date, version-specific WML reference data for `wesnoth-mode'
  20 ;; using `wesnoth-update'.  This requires Wesnoth to be install and its
  21 ;; pathname set for this to behave correctly.  for example:
  22 ;; (setq wesnoth-root-directory "/usr/share/wesnoth/")
  23 
  24 ;; Then set the output directory for `wesnoth-update's results:
  25 ;; (setq wesnoth-update-output-directory "/path/to/wesnoth-mode/")
  26 ;; This is recommended to be in the same directory as `wesnoth-mode' and
  27 ;; must be in `load-path'.
  28 
  29 ;; Once set, `wesnoth-update' will produce 'wesnoth-wml-data.el' in
  30 ;; `wesnoth-update-output-directory' and the information will automatically
  31 ;; be available in the future sessions.
  32 
  33 ;; Although much data is retreived, it is unlikely to be completely
  34 ;; comprehensive.  wesnoth-mode can be taught about additional tags,
  35 ;; attributes and macros using the current project, or a single file, using
  36 ;; `wesnoth-update-wml-additions'.
  37 
  38 ;; To teach wesnoth-mode about elements it may have missed, you can extend the
  39 ;; sample additions included with wesnoth-mode; namely
  40 ;; wesnoth-wml-additions.cfg (although any source of WML can be used).  To
  41 ;; enable this, do the following:
  42 ;; Set `wesnoth-addition-file' appropriately, for example:
  43 ;; (setq wesnoth-addition-file "/path/to/wesnoth-wml-additions.cfg")
  44 
  45 ;; Once set correctly, running M-x wesnoth-update will update the WML data
  46 ;; available to `wesnoth-mode'.

 114 (defun wesnoth-create-wml-hash-table (tag-data &optional force)
 115   "Handle generation of `wesnoth-tag-hash-table'.
 116 TAG-DATA is the data to add to the hash-table.  If FORCE is
 117 non-nil, update the hash-table regardless of whether it replacing
 118 any existing data."
 119   (when (or (= (hash-table-count wesnoth-tag-hash-table) 0)
 120             force)
 121     (clrhash wesnoth-tag-hash-table)
 122     (dolist (tag tag-data)
 123       (puthash (car tag) (cdr tag) wesnoth-tag-hash-table))))

 125 (defun wesnoth-file-cfg-p (file)
 126   "Return non-nil if FILE has a '.cfg' extension."
 127   (and (not (file-directory-p file)) (string-match "\\.cfg$" file)))

 129 (defun wesnoth-fetch-all-dirs (dir)
 130   "Retrieve a list of subdirectories to scan.
 131 DIR is the directory to check."
 132   (let ((dirs-to-scan (wesnoth-files-in-dir dir)))
 133     (while dirs-to-scan
 134         (setq dirs-to-scan (append (wesnoth-files-in-dir (pop dirs-to-scan))
 135                                    dirs-to-scan)))))

 137 (defun wesnoth-files-in-dir (dir)
 138   "Add cfgs to `wesnoth-files-in-dir'.
 139 Returns a list of sub-directories in DIR."
 140   (let ((cfgs (wesnoth-cfg-files-in-dir dir)))
 141     (when cfgs
 142       (setq wesnoth-found-cfgs (append cfgs wesnoth-found-cfgs))))
 143   (let ((dirs '()))
 144     (dolist (file (directory-files dir t))
 145       (when (and (not (string-match "^\\..*" (file-name-nondirectory file)))
 146                  (not (file-symlink-p file)))
 147         (cond ((file-directory-p file)
 148                (add-to-list 'dirs file))
 149               ((wesnoth-file-cfg-p file)
 150                (add-to-list 'wesnoth-found-cfgs file)))))
 151     dirs))

 153 (defun wesnoth-cfg-files-in-dir (dir)
 154   "Return all cfg files in DIR."
 155   (let ((result '()))
 156     (dolist (file (directory-files dir t))
 157       (and (wesnoth-file-cfg-p file)
 158            (add-to-list 'result file)))
 159     result))

 161 (defun wesnoth-determine-details (dir-or-file function)
 162   "Process .cfg files in DIR-OR-FILE using FUNCTION.
 163 DIR-OR-FILE can be a file, a directory, or a list of files."
 164   (cond ((listp dir-or-file)
 165          (dolist (file dir-or-file)
 166            (wesnoth-handle-file function file)))
 167         ((and (file-exists-p dir-or-file)
 168               (not (file-directory-p dir-or-file)))
 169          (wesnoth-handle-file function dir-or-file))
 170         (t
 171          (wesnoth-fetch-all-dirs dir-or-file)
 172          (while wesnoth-found-cfgs
 173            (message "Updating WML semantics: %s..."
 174                     (substring (car wesnoth-found-cfgs)
 175                                (length (expand-file-name
 176                                         wesnoth-root-directory))))
 177            (unless (string-match "^\\..+" (file-name-nondirectory
 178                                            (car wesnoth-found-cfgs)))
 179              (wesnoth-handle-file function (car wesnoth-found-cfgs))
 180              (setq wesnoth-found-cfgs (cdr wesnoth-found-cfgs)))))))

 182 (defun wesnoth-handle-file (function file)

 186       (insert-file-contents file)
 187       (funcall function))))

 189 (defun wesnoth-extract-tag-information ()
 190   "Retrieve relevant tag and attribute information."
 191   (let ((unmatched-tag-list '()))
 192     (goto-char (point-min))
 193     (wesnoth-determine-macro-information)
 194     (while (search-forward-regexp
 195             "^[\t ]*\\(\\[[+/]?\\(\\(\\w\\|_\\)+\\)\\]\\|\\(\\w\\|_\\)+=\\)"
 196             (point-max) t)
 197       (beginning-of-line)
 198       (cond
 199        ((and (save-excursion
 200                (search-backward-regexp
 201                 "^[\t ]*\\(\\[[^/]]?\\|#define \\|#enddef \\)"
 202                 (point-min) t))
 203              (string-match "#define " (match-string 1))
 204              (looking-at "^[\t ]*\\[\\+?\\(\\(\\w\\|_\\)+\\)\\]"))
 205         (wesnoth-append-tag-information (match-string-no-properties 1)
 206                                         nil nil)
 207         (setq unmatched-tag-list (cons (match-string-no-properties 1)
 208                                        unmatched-tag-list)))
 209        ((looking-at "^[\t ]*\\[\\+?\\(\\(\\w\\|_\\)+\\)\\]")
 210         (wesnoth-append-tag-information (car unmatched-tag-list)
 211                                         (match-string-no-properties 1)
 212                                         nil)
 213         (wesnoth-append-tag-information (match-string-no-properties 1)
 214                                         nil nil)
 215         (setq unmatched-tag-list (cons (match-string-no-properties 1)
 216                                        unmatched-tag-list)))
 217        ((looking-at "^[\t ]*\\(\\(\\w\\|_\\)+\\)=")
 218         (wesnoth-append-tag-information (car unmatched-tag-list)
 219                                         nil (match-string-no-properties 1)))
 220        ((looking-at "^[\t ]*\\[/\\(\\(\\w\\|_\\)+\\)\\]\\|")
 221         (when (string= (match-string-no-properties 1)
 222                        (car unmatched-tag-list))
 223           (setq unmatched-tag-list (cdr unmatched-tag-list)))))
 224       (end-of-line))))

 226 (defun wesnoth-append-tag-information (tag subtag attribute)
 227   "Add the information regarding TAG to the list.
 228 SUBTAG and ATTRIBUTE are a children of TAG to be added."
 229   (let ((match (assoc tag wesnoth-tmp-tag-data)))
 230     (if (not match)
 231         (add-to-list 'wesnoth-tmp-tag-data
 232                      (list tag (and subtag (list subtag))
 233                            (and attribute (list attribute))))
 234       (if subtag
 235           (let ((tmp (nth 1 match)))
 236             (when (not (member subtag tmp))
 237               (add-to-list 'tmp subtag)
 238               (setq match (list tag tmp (car (last match))))))
 239         (when attribute (let ((tmp (nth 2 match)))
 240                           (when (not (member attribute tmp))
 241                             (add-to-list 'tmp attribute)
 242                             (setq match (list tag (nth 1 match) tmp))))))
 243       (setq wesnoth-tmp-tag-data
 244             (remove (assoc tag wesnoth-tmp-tag-data)
 245                     wesnoth-tmp-tag-data))
 246       (add-to-list 'wesnoth-tmp-tag-data match))))

 248 (defun wesnoth-determine-macro-information ()
 249   "Process the buffer, retrieving macro definition information."
 250   (save-excursion
 251     (goto-char (point-min))
 252     (while (search-forward-regexp
 253             "#define \\(\\(?:\\w\\|_\\)+\\)\\(\\([\t ]+\\(\\w\\|_\\)+\\)*\\)"
 254             (point-max) t)
 255       (beginning-of-line)
 256       (add-to-list 'wesnoth-tmp-macro-data
 257                    (list (match-string-no-properties 1)
 258                          (and (match-string 2)
 259                               (split-string
 260                                (match-string-no-properties 2)))))
 261       (end-of-line))
 262     wesnoth-tmp-macro-data))

 295 (defun wesnoth-update ()
 296   "Update WML information.
 297 Path to WML information included in wesnoth is set by
 298 `wesnoth-root-directory.'."
 299   (interactive)
 300   (setq wesnoth-tag-data nil
 301         wesnoth-macro-data nil
 302         wesnoth-found-cfgs nil
 303         wesnoth-tmp-macro-data nil
 304         wesnoth-tmp-tag-data nil)
 305   (unless (and (stringp wesnoth-root-directory)
 306                (file-exists-p wesnoth-root-directory))
 307     ;; Update failed; restore data.
 308     (load "wesnoth-wml-data")
 309     (error "%s: directory does not exist"
 310            wesnoth-root-directory))
 311   (message "Updating WML semantics: Searching for .cfg's...")
 312   (wesnoth-determine-details wesnoth-root-directory
 313                              'wesnoth-extract-tag-information)
 314   (setq wesnoth-tag-data wesnoth-tmp-tag-data
 315         wesnoth-tmp-tag-data nil
 316         wesnoth-macro-data wesnoth-tmp-macro-data
 317         wesnoth-tmp-macro-data nil)
 318   (with-temp-buffer
 319     (insert "(setq wesnoth-tag-data '"
 320             (pp wesnoth-tag-data) ")\n\n"
 321             "(setq wesnoth-macro-data '"
 322             (pp wesnoth-macro-data) ")\n\n"
 323             "(provide 'wesnoth-wml-data)\n")
 324     (write-file (expand-file-name "wesnoth-wml-data.el"
 325                                   (wesnoth-output-path)))
 326     (load "wesnoth-wml-data"))
 327   (message "Updating WML semantics: done"))

 329 (defun wesnoth-merge-macro-data (&rest macro-data)
 330   "Merge WML macro information and return the result.
 331 MACRO-DATA is the macro-data to merge."
 332   (let ((set-data '())
 333         (macro-base-data (car macro-data)))
 334     (while (setq macro-data (cdr macro-data))
 335       (setq set-data (car macro-data))
 336       (while set-data
 337         (setq macro-base-data
 338               (append (list (car set-data))
 339                       (remove (assoc (car (car set-data)) macro-base-data)
 340                               macro-base-data))
 341               set-data (cdr set-data))))
 342     macro-base-data))
 343 
 344 (defun wesnoth-merge-tag-data (&rest tag-data)
 345   "Merge WML tag information and return the result.
 346 TAG-DATA is the tag-data to merge."
 347   (setq wesnoth-tmp-tag-data (car tag-data))
 348   (let ((set-data '()))
 349     (while (setq tag-data (cdr tag-data))
 350       (setq set-data (car tag-data))
 351       (while set-data
 352         (let ((subtags (nth 1 (car set-data))))
 353           (while subtags
 354             (wesnoth-append-tag-information (caar set-data) (car subtags)
 355                                             nil)
 356             (setq subtags (cdr subtags))))
 357         (let ((attributes (nth 2 (car set-data))))
 358           (while attributes
 359             (wesnoth-append-tag-information (caar set-data) nil
 360                                             (car attributes))
 361             (setq attributes (cdr attributes))))
 362         (setq set-data (cdr set-data))))
 363     (wesnoth-read-tmp-tag-data)))

 365 (defun wesnoth-update-project-information (&optional clear)
 366   "Update WML macro information for the current project.
 367 If CLEAR is non-nil, reset `wesnoth-local-macro-data'."
 368   (interactive "P")
 369   (setq wesnoth-tmp-macro-data nil)
 370   (if clear
 371       (setq wesnoth-local-macro-data nil)
 372     (setq wesnoth-local-macro-data
 373           (wesnoth-merge-macro-data wesnoth-local-macro-data

 377 (defun wesnoth-refresh-wml-data ()
 378   "Return merged WML tag data and WML data from the addition file."
 379   (save-match-data
