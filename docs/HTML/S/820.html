 103         template <typename T>
 104         unsigned int rng::get_random_element(T first, T last)
 105         {
 106                 double target = get_random_double();
 107                 double sum = 0.0;
 108                 T it = first;
 109                 sum += *it;
 110                 while (sum <= target)
 111                 {
 112                         ++it;
 113                         if (it != last)
 114                         {
 115                                 sum += *it;
 116                         }
 117                         else
 118                         {
 119                                 break;
 120                         }
 121                 }
 122                 return std::distance(first, it);
 123         }
