  57 point slider::calculate_best_size() const
  58 {
  59         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
  60 
  61         // Inherited.
  62         point result = styled_widget::calculate_best_size();
  63         if(best_slider_length_ != 0) {
  64 
  65                 // Override length.
  66                 std::shared_ptr<const slider_definition::resolution> conf
  67                                 = std::static_pointer_cast<const slider_definition::
  68                                                                                                           resolution>(config());
  69 
  70                 assert(conf);
  71 
  72                 result.x = conf->left_offset + best_slider_length_ + conf->right_offset;
  73         }
  74 
  75         DBG_GUI_L << LOG_HEADER << " best_slider_length " << best_slider_length_
  76                           << " result " << result << ".\n";
  77         return result;
  78 }

  80 void slider::set_value(const int value)
  81 {
  82         if(value == get_value()) {
  83                 return;
  84         }
  85 
  86         if(value < minimum_value_) {
  87                 set_value(minimum_value_);
  88         } else if(value > get_maximum_value()) {
  89                 set_value(get_maximum_value());
  90         } else {
  91                 set_item_position(value - minimum_value_);
  92         }
  93 
  94         fire(event::NOTIFY_MODIFIED, *this, nullptr);
  95 }

  97 void slider::set_minimum_value(const int minimum_value)
  98 {
  99         if(minimum_value == minimum_value_) {
 100                 return;
 101         }
 102 
 103         /** @todo maybe make it a VALIDATE. */
 104         assert(minimum_value <= get_maximum_value());
 105 
 106         const int value = get_value();
 107         const int maximum_value = get_maximum_value();
 108         minimum_value_ = minimum_value;
 109 
 110         // The number of items needs to include the begin and end so distance step
 111         // size.
 112         set_item_count(maximum_value - minimum_value_ + get_step_size());
 113 
 114         if(value < minimum_value_) {
 115                 set_item_position(0);
 116         }
 117 }

 119 void slider::set_maximum_value(const int maximum_value)
 120 {
 121         if(maximum_value == get_maximum_value()) {
 122                 return;
 123         }
 124 
 125         /** @todo maybe make it a VALIDATE. */
 126         assert(minimum_value_ <= maximum_value);
 127 
 128         const int value = get_value();
 129 
 130         // The number of items needs to include the begin and end so distance + step
 131         // size.
 132         set_item_count(maximum_value - minimum_value_ + get_step_size());
 133 
 134         if(value > maximum_value) {
 135                 set_item_position(get_maximum_value());
 136         }

 139 t_string slider::get_value_label() const
 140 {
 141         if(value_labels_) {
 142                 return value_labels_(get_item_position(), get_item_count());
 143         } else if(!minimum_value_label_.empty() && get_value()
 144                                                                                            == get_minimum_value()) {
 145                 return minimum_value_label_;
 146         } else if(!maximum_value_label_.empty() && get_value()
 147                                                                                            == get_maximum_value()) {
 148                 return maximum_value_label_;
 149         }
 150 
 151         return t_string(formatter() << get_value());
 152 }

 195 bool slider::on_positioner(const point& coordinate) const
 196 {
 197         // Note we assume the positioner is over the entire height of the widget.
 198         return coordinate.x >= static_cast<int>(get_positioner_offset())
 199                    && coordinate.x < static_cast<int>(get_positioner_offset()
 200                                                                                           + get_positioner_length())
 201                    && coordinate.y > 0 && coordinate.y < static_cast<int>(get_height());
 202 }

 204 int slider::on_bar(const point& coordinate) const
 205 {
 206         const unsigned x = static_cast<size_t>(coordinate.x);
 207         const unsigned y = static_cast<size_t>(coordinate.y);
 208 
 209         // Not on the widget, leave.
 210         if(x > get_width() || y > get_height()) {
 211                 return 0;
 212         }
 213 
 214         // we also assume the bar is over the entire height of the widget.
 215         if(x < get_positioner_offset()) {
 216                 return -1;
 217         } else if(x > get_positioner_offset() + get_positioner_length()) {
 218                 return 1;
 219         }
 220 
 221         return 0;
 222 }

 477 builder_slider::builder_slider(const config& cfg)
 478         : implementation::builder_styled_widget(cfg)
 479         , best_slider_length_(cfg["best_slider_length"])
 480         , minimum_value_(cfg["minimum_value"])
 481         , maximum_value_(cfg["maximum_value"])
 482         , step_size_(cfg["step_size"])
 483         , value_(cfg["value"])
 484         , minimum_value_label_(cfg["minimum_value_label"].t_str())
 485         , maximum_value_label_(cfg["maximum_value_label"].t_str())
 486         , value_labels_()
 487 {
 488         const config& labels = cfg.child("value_labels");
 489         if(!labels) {
 490                 return;
 491         }
 492 
 493         for(const auto & label : labels.child_range("value"))
 494         {
 495                 value_labels_.push_back(label["label"]);
 496         }
 497 }

 499 widget* builder_slider::build() const
 500 {
 501         slider* widget = new slider();
 502 
 503         init_control(widget);
 504 
 505         widget->set_best_slider_length(best_slider_length_);
 506         widget->set_maximum_value(maximum_value_);
 507         widget->set_minimum_value(minimum_value_);
 508         widget->set_step_size(step_size_);
 509         widget->set_value(value_);
 510 
 511         if(!value_labels_.empty()) {
 512                 VALIDATE(value_labels_.size() == widget->get_item_count(),
 513                                  _("The number of value_labels and values don't match."));
 514 
 515                 widget->set_value_labels(value_labels_);
 516 
 517         } else {
 518                 widget->set_minimum_value_label(minimum_value_label_);
 519                 widget->set_maximum_value_label(maximum_value_label_);
 520         }
 521 
 522         DBG_GUI_G << "Window builder: placed slider '" << id
 523                           << "' with definition '" << definition << "'.\n";
 524 
 525         return widget;
 526 }
