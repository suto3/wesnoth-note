   1 # This test checks that sighted events are being triggered
   2 # at the appropriate time. An elvish archer moves towards
   3 # and discovers an orcish grunt.
   4 # We pass if the elvish archer is the discoverer, and the
   5 # variable "sighted_should_trigger" is set.
   6 # Companion tests check what happens if the orc is discovered
   7 # on the elf's turn, and also that we fail if we dont set the
   8 # flag.
   9 
  10 #define TEST_SIGHTED_EVENTS_SCEN ID BOB_CONDITION SET_FLAG
  11     [test]
  12         name = "Unit Test {ID}"
  13         map_data = "{test/maps/sighted_events.map}"
  14         turns = 4
  15         id = {ID}
  16         random_start_time = no
  17         is_unit_test = yes
  18 
  19         {DAWN}
  20 
  21         [side]
  22             side=1
  23             controller=human
  24             name = "Alice"
  25             type = Elvish Archer
  26             id=alice
  27             fog=yes
  28         [/side]
  29         [side]
  30             side=2
  31             controller=human
  32             name = "Bob"
  33             type = Orcish Grunt
  34             id=bob
  35             fog=yes
  36         [/side]
  37 
  38         [event]
  39             name = prestart
  40             {VARIABLE sighted_should_trigger no}
  41         [/event]
  42 
  43         [event]
  44             name = sighted
  45             {ASSERT ({VARIABLE_CONDITIONAL unit.id equals bob})}
  46             {ASSERT ({VARIABLE_CONDITIONAL second_unit.id equals alice})}
  47             {RETURN ({VARIABLE_CONDITIONAL sighted_should_trigger boolean_equals yes})}
  48         [/event]
  49 
  50         #Alice moves towards Bob.
  51         [event]
  52             name = side 1 turn 1 refresh
  53             [do_command]
  54                 [move]
  55                     x=1,2,3,4,5,6,7
  56                     y=3,3,3,3,3,3,3
  57                 [/move]
  58             [/do_command]
  59         [/event]
  60 
  61         #Bob moves either down, then up (unseen), or left, then right (seen)
  62         [event]
  63             name = side 2 turn 1 refresh
  64             [if]
  65                 {BOB_CONDITION}
  66                 [then]
  67                     [do_command]
  68                         [move]
  69                             x=16,16,16
  70                             y=3,4,5
  71                         [/move]
  72                     [/do_command]
  73                     [do_command]
  74                         [move]
  75                             x=16,16,16
  76                             y=5,4,3
  77                         [/move]
  78                     [/do_command]
  79                 [/then]
  80                 [else]
  81                     [if]
  82                         {SET_FLAG}
  83                         [then]
  84                             {VARIABLE sighted_should_trigger yes}
  85                         [/then]
  86                     [/if]
  87                     [do_command]
  88                         [move]
  89                             x=16,15,14
  90                             y=3,3,3
  91                         [/move]
  92                     [/do_command]
  93                     [do_command]
  94                         [move]
  95                             x=14,15,16
  96                             y=3,3,3
  97                         [/move]
  98                     [/do_command]
  99                 [/else]
 100             [/if]
 101         [/event]
 102 
 103         #Alice discovers Bob.
 104         [event]
 105             name = side 1 turn 2 refresh
 106             [do_command]
 107                 [move]
 108                     x=7,8
 109                     y=3,3
 110                 [/move]
 111             [/do_command]
 112             #With the next stop Bob should be visible.
 113             [if]
 114                 {SET_FLAG}
 115                 [then]
 116                     {VARIABLE sighted_should_trigger yes}
 117                 [/then]
 118             [/if]
 119             [do_command]
 120                 [move]
 121                     x=8,9
 122                     y=3,3
 123                 [/move]
 124             [/do_command]
 125         [/event]
 126 
 127         #If we got this far without triggering sighted, we fail the test.
 128         [event]
 129             name = side 2 turn 2
 130             {FAIL}
 131         [/event]
 132 
 133         #This makes the sides pass their turns, when the other events have taken place.
 134         [event]
 135             name = side turn
 136             first_time_only = no
 137             [end_turn]
 138             [/end_turn]
 139         [/event]
 140     [/test]
 141 #enddef
