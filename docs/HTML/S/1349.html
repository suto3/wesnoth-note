 101     if (no_change and image_got_smaller):
 102         os.rename(new_img, source_img) # move new image to old image // os.rename(src, dest)

 107 def run_imagemagick(image, tmpimage):
 108     shutil.copy(image, tmpimage)
 109     debugprint("imagemagick ")
 110     cmd = [ EXEC_IMAGEMAGICK,
 111             "-strip",
 112             "-define",
 113             "png:color-type=6",
 114             image,
 115             tmpimage
 116     ]
 117     subprocess.call(cmd)

 119 def run_optipng(image, tmpimage):
 120     debugprint("optipng...")
 121     shutil.copy(image, tmpimage)
 122     cmd = [ EXEC_OPTIPNG,
 123             "-q",
 124             "-o5",
 125             "-nb",
 126             "-nc",
 127             "-np",
 128             tmpimage
 129     ]
 130     subprocess.call(cmd)

 158 def optimize_image(image):
 159     size_initial = os.path.getsize(image)
 160     with open(image, 'rb') as f:
 161         initial_file_content = f.read()
 162 
 163

 165     it=0
 166     size_after = 0
 167     size_before = os.path.getsize(image)
 168     while ((size_before > size_after) or (not it)):
 169         it+=1
 170         debugprint(("iteration " + str(it)))
 171         size_before = os.path.getsize(image)
 172         tmpimage  = image + ".tmp"
 173 
 174         run_imagemagick(image, tmpimage)
 175         verify_images(image, tmpimage, "imagemagick")
 176 
 177         run_optipng(image, tmpimage)
 178         verify_images(image, tmpimage, "optipng")
 179 
 180         run_advdef(image, tmpimage)
 181         verify_images(image, tmpimage, "advdef")
 182 
 183         size_after = os.path.getsize(image)
 184         size_delta = size_after - size_initial
 185         perc_delta = (size_delta/size_initial) *100
 186 
 187     if (DEBUG and (size_initial < size_after)):
 188         debugprint("WARNING: " + str(image) + "got bigger !")
 189     if os.path.isfile(tmpimage): # clean up
 190         os.remove(tmpimage)
 191 
 192     if (os.path.getsize(image) > size_initial) or (perc_delta*-1 < THRESHOLD) : # got bigger, or exceeds threshold
 193         with open(image, 'wb') as f: # write back original file
 194             f.write(initial_file_content)
 195     else:
 196         print("optimized  {image}  from {size_initial} to {size_after}, {size_delta}b, {perc_delta}%".format(image=image, size_initial=size_initial, size_after=size_after, size_delta=size_delta, perc_delta=str(perc_delta)[0:6]))
