 109 #define ON_SIGHTING ID SIGHTING_SIDE SIGHTED_FILTER ACTION_WML
 110     # NOTE: As of version 1.11, this macro is unnecessary, since sighted
 111     # events now work as intended.
 112     #
 113     # This is intended to be used when a regular sighted event can't be relied
 114     # on (that'd be most of the time). This event triggers when a unit matching
 115     # SIGHTED_FILTER is first seen by a unit of side SIGHTING_SIDE. This works
 116     # whether the sighting happens due to moving, recruiting, recalling,
 117     # attacking (in case DSU was on before the attack) or advancing. If none of
 118     # those actions manage to "catch" the moment of sighting, then it will occur
 119     # when the current side ends their turn.
 120     #
 121     # The 'unit' and 'second_unit' message speaker shortcuts as well as the
 122     # auto-stored $unit and $second_unit should work normally: the sighting unit
 123     # is the primary unit and the sighted unit is the secondary unit.
 124     #
 125     # You need to provide a unique ID if you use several of these events in the
 126     # same scenario. Otherwise it can be left empty.
 127     #
 128     # Only works if SIGHTING_SIDE has fog and/or shroud.
 129     #
 130     # Example:
 131     #! {ON_SIGHTING () 1 side=2,3 (
 132     #!     [message]
 133     #!         speaker=unit
 134     #!         message= _ "I can see a side $second_unit.side $second_unit.language_name|!"
 135     #!     [/message]
 136     #!     [message]
 137     #!         speaker=second_unit
 138     #!         message= _ "Oh my! I was seen by a side $unit.side $unit.language_name|!"
 139     #!     [/message]
 140     #! )}
 141     #!
 142     [event]
 143         name=SIGHTED_{ID}_trigger_action_WML
 144 
 145         {ACTION_WML}
 146     [/event]
 147 
 148     [event]
 149         name=moveto,recruit,recall,attack,post advance,side turn
 150         first_time_only=no
 151 
 152         [filter]
 153             side={SIGHTING_SIDE}
 154 
 155             [or]
 156                 {SIGHTED_FILTER}
 157             [/or]
 158         [/filter]
 159 
 160         [if]
 161             [variable]
 162                 name=SIGHTED_{ID}_finished
 163                 not_equals=yes
 164             [/variable]
 165 
 166             [then]
 167                 [store_side]
 168                     side={SIGHTING_SIDE}
 169                     variable=stored_sighting_side
 170                 [/store_side]
 171 
 172                 [if]
 173                     [have_unit]
 174                         {SIGHTED_FILTER}
 175 
 176                         [filter_vision]
 177                             side={SIGHTING_SIDE}
 178                         [/filter_vision]
 179                     [/have_unit]
 180 
 181                     [and]
 182                         [variable]
 183                             name=stored_sighting_side.fog
 184                             boolean_equals=yes
 185                         [/variable]
 186 
 187                         [or]
 188                             [variable]
 189                                 name=stored_sighting_side.shroud
 190                                 boolean_equals=yes
 191                             [/variable]
 192                         [/or]
 193                     [/and]
 194 
 195                     [then]
 196                         [fire_event]
 197                             name=SIGHTED_{ID}_trigger_action_WML
 198 
 199                             [primary_unit]
 200                                 side={SIGHTING_SIDE}
 201 
 202                                 [filter_location]
 203                                     [filter]
 204                                         {SIGHTED_FILTER}
 205 
 206                                         [not]
 207                                             side={SIGHTING_SIDE}
 208                                         [/not]
 209                                     [/filter]
 210 
 211                                     radius="$($this_unit.max_moves + 1)"
 212                                 [/filter_location]
 213                             [/primary_unit]
 214 
 215                             [secondary_unit]
 216                                 {SIGHTED_FILTER}
 217 
 218                                 [filter_vision]
 219                                     side={SIGHTING_SIDE}
 220                                 [/filter_vision]
 221                             [/secondary_unit]
 222                         [/fire_event]
 223 
 224                         [set_variable]
 225                             name=SIGHTED_{ID}_finished
 226                             value=yes
 227                         [/set_variable]
 228                     [/then]
 229 
 230                     [else]
 231                         [allow_undo]
 232                         [/allow_undo]
 233                     [/else]
 234                 [/if]
 235 
 236                 [clear_variable]
 237                     name=stored_sighting_side
 238                 [/clear_variable]
 239             [/then]
 240 
 241             [else]
 242                 [allow_undo]
 243                 [/allow_undo]
 244             [/else]
 245         [/if]
 246     [/event]
 247 
 248     [event]
 249         name=victory
 250 
 251         [clear_variable]
 252             name=SIGHTED_{ID}_finished
 253         [/clear_variable]
 254     [/event]
 255 #enddef

 309 #define FOREACH ARRAY_VALUE VAR

 311 {VARIABLE {VAR} 0}
 312 [while]
 313     [variable]
 314     name={VAR}
 315     less_than=${ARRAY_VALUE}.length
 316     [/variable]
 317     [do]
 318 #enddef

 320 #define NEXT VAR
 321 # Macro to end a WML clause that iterates over an array.
 322     [set_variable]
 323     name={VAR}
 324     add=1
 325     [/set_variable]
 326     [/do]
 327 [/while]
 328 {CLEAR_VARIABLE {VAR}}
 329 #enddef
