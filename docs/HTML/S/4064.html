   1 #!/usr/bin/env python3
   2 # encoding: utf-8
   3 
   4 """
   5 A script to create the "Terrain Table" on the TerrainCodeTableWML wiki page.
   6 Add the output to the wiki whenever a new terrain is added to mainline.
   7 The terrain_url below should point to the latest stable release.
   8 """
   9 
  10 import os
  11 import sys
  12 import re
  13 import argparse
  14 
  15 # Where to get terrain images
  16 terrain_url = "https://raw.github.com/wesnoth/wesnoth/1.14/data/core/images/terrain/%s.png"
  17 
  18 
  19 def parse_terrain(data):
  20     """
  21     Parses the terrains. Input looks like this:
  22 
  23     [terrain_type]
  24     symbol_image=water/ocean-grey-tile
  25     id=deep_water_gray
  26     editor_name= _ "Gray Deep Water"
  27     string=Wog
  28     aliasof=Wo
  29     submerge=0.5
  30     editor_group=water
  31     [/terrain_type]
  32 
  33     Output is a text in wiki format.
  34     """
  35 
  36     # Remove all comments.
  37     data = "\n".join([i for i in data.split("\n") if not re.match("^\s*#", i)])
  38     terrains = re.compile(r"\[terrain_type\](.*?)\[\/terrain_type\]", re.DOTALL).findall(data)
  39 
  40     data = """{{AutogeneratedWML}}{| border="1"
  41 !terrain
  42 !name
  43 !string
  44 !alias of
  45 !editor group
  46 """
  47 
  48     for i in terrains:
  49         # Strip unneeded things.
  50         i = i[5:]
  51         i = i.split("\n    ")
  52         # Don't parse special files that are hacks. They shouldn't be used
  53         # directly. (They're only there to make aliasing work.)
  54         if i[0].startswith(" "):
  55             continue
  56         # This avoids problems due to additional = in strings. Exact string
  57         # removal does not matter as long as we do not print help_topic_text
  58         # in the wiki page.
  59         removeus = ("<italic>text='", "'</italic>", "<ref>dst='", "text='", "'</ref>")
  60         for text in removeus:
  61             i = [a.replace(text, "") for a in i]
  62         # Create a dictionary of key and values
  63         content = {k: v for (k, v) in [p.strip().split("=", 1) for p in i]}
  64         # Hidden things shouldn't be displayed
  65         if 'hidden' in content:
  66             continue
  67         data += """|-
  68 | %s
  69 | %s
  70 | <code>%s</code>
  71 | <code>%s</code>
  72 | %s
  73 """ % (
  74 terrain_url % (content['editor_image'] if 'editor_image' in content else content['symbol_image']),
  75 content['editor_name'][4:-1] if 'editor_name' in content else content['name'][4:-1],
  76 content['string'].replace("# wmllint: ignore", "").replace("|", "&#124;"),
  77 content['aliasof'].replace("|", "&#124;") if 'aliasof' in content else "",
  78 content['editor_group'])
  79 
  80     data += "|}"
  81     return data
  82 
  83 if __name__ == "__main__":
  84     parser = argparse.ArgumentParser(description='terrain2wiki is a tool to\
  85 convert the terrain codes located in terrain.cfg to wiki formatted text.')
  86     parser.add_argument('-f', '--file', default='data/core/terrain.cfg',
  87 dest='path', help="The location of the terrain.cfg file.")
  88     parser.add_argument('-o', '--output', default='/tmp/TerrainCodeTableWML',
  89 dest='output_path', help="The location of the output file.")
  90     args = parser.parse_args()
  91 
  92     path = args.path
  93     output_path = args.output_path
  94 
  95     if not os.path.exists(path) or not path.endswith('.cfg'):
  96         print("Invalid path: '%s' does not exist or is not a .cfg file." % path)
  97         sys.exit(1)
  98 
  99     with open(path, "r", encoding="utf8") as input_file:
 100         data = input_file.read()
 101     data = parse_terrain(data)
 102 
 103     with open(output_path, "w", encoding="utf8") as output:
 104         output.write(data)
