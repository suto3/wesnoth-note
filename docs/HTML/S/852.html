  45 /**
  46  * Function that will find a location on the board that is as near
  47  * to @a loc as possible, but which is unoccupied by any units.
  48  * If no valid location can be found, it will return a null location.
  49  * If @a pass_check is provided, the found location must have a terrain
  50  * that this unit can enter.
  51  * If @a shroud_check is provided, only locations not covered by this
  52  * team's shroud will be considered.
  53  */
  54 map_location find_vacant_tile(const map_location& loc, VACANT_TILE_TYPE vacancy,
  55                               const unit* pass_check, const team* shroud_check, const game_board* board)
  56 {
  57         if (!board) {
  58                 board = resources::gameboard;
  59                 assert(board);
  60         }
  61         const gamemap & map = board->map();
  62         const unit_map & units = board->units();
  63 
  64         if (!map.on_board(loc)) return map_location();
  65 
  66         const bool do_shroud = shroud_check  &&  shroud_check->uses_shroud();
  67         std::set<map_location> pending_tiles_to_check, tiles_checked;
  68         pending_tiles_to_check.insert(loc);
  69         // Iterate out 50 hexes from loc
  70         for (int distance = 0; distance < 50; ++distance) {
  71                 if (pending_tiles_to_check.empty())
  72                         return map_location();
  73                 //Copy over the hexes to check and clear the old set
  74                 std::set<map_location> tiles_checking;
  75                 tiles_checking.swap(pending_tiles_to_check);
  76                 //Iterate over all the hexes we need to check
  77                 for (const map_location &l : tiles_checking)
  78                 {
  79                         // Skip shrouded locations.
  80                         if ( do_shroud  &&  shroud_check->shrouded(l) )
  81                                 continue;
  82                         //If this area is not a castle but should, skip it.
  83                         if ( vacancy == VACANT_CASTLE  &&  !map.is_castle(l) ) continue;
  84                         const bool pass_check_and_unreachable = pass_check
  85                                 && pass_check->movement_cost(map[l]) == movetype::UNREACHABLE;
  86                         //If the unit can't reach the tile and we have searched
  87                         //an area of at least radius 10 (arbitrary), skip the tile.
  88                         //Neccessary for cases such as an unreachable
  89                         //starting hex surrounded by 6 other unreachable hexes, in which case
  90                         //the algorithm would not even search distance==1
  91                         //even if there's a reachable hex for distance==2.
  92                         if (pass_check_and_unreachable && distance > 10) continue;
  93                         //If the hex is empty and we do either no pass check or the hex is reachable, return it.
  94                         if (units.find(l) == units.end() && !pass_check_and_unreachable) return l;
  95                         map_location adjs[6];
  96                         get_adjacent_tiles(l,adjs);
  97                         for (const map_location &l2 : adjs)
  98                         {
  99                                 if (!map.on_board(l2)) continue;
 100                                 // Add the tile to be checked if it hasn't already been and
 101                                 // isn't being checked.
 102                                 if (tiles_checked.find(l2) == tiles_checked.end() &&
 103                                     tiles_checking.find(l2) == tiles_checking.end())
 104                                 {
 105                                         pending_tiles_to_check.insert(l2);
 106                                 }
 107                         }
 108                 }
 109                 tiles_checked.swap(tiles_checking);
 110         }
 111         return map_location();
 112 }

 126 /**
 127  * Determines if a given location is in an enemy zone of control.
 128  *
 129  * @param current_team  The moving team (only ZoC of enemies of this team are considered).
 130  * @param loc           The location to check.
 131  * @param viewing_team  Only units visible to this team are considered.
 132  * @param see_all       If true, all units are considered (and viewing_team is ignored).
 133  *
 134  * @return true iff a visible enemy exerts zone of control over loc.
 135  */
 136 bool enemy_zoc(team const &current_team, map_location const &loc,
 137                team const &viewing_team, bool see_all)
 138 {
 139         // Check the adjacent tiles.
 140         map_location locs[6];
 141         get_adjacent_tiles(loc,locs);
 142         for (int i = 0; i != 6; ++i)
 143         {
 144                 const unit *u = resources::gameboard->get_visible_unit(locs[i], viewing_team, see_all);
 145                 if ( u  &&  current_team.is_enemy(u->side())  &&  u->emits_zoc() )
 146                         return true;
 147         }
 148 
 149         // No adjacent tiles had an enemy exerting ZoC over loc.
 150         return false;
 151 }

 155         /**
 156          * Nodes used by find_routes().
 157          * These store the information necessary for extending the path
 158          * and for tracing the route back to the source.
 159          */
 160         struct findroute_node {
 161                 int moves_left, turns_left;
 162                 map_location prev;
 163                 // search_num is used to detect which nodes have been collected
 164                 // in the current search. (More than just a boolean value so
 165                 // that nodes can be stored between searches.)
 166                 unsigned search_num;
 167 
 168                 // Constructors.
 169                 findroute_node(int moves, int turns, const map_location &prev_loc, unsigned search_count)
 170                         : moves_left(moves)
 171                         , turns_left(turns)
 172                         , prev(prev_loc)
 173                         , search_num(search_count)
 174                 { }
 175                 findroute_node()
 176                         : moves_left(0)
 177                         , turns_left(0)
 178                         , prev()
 179                         , search_num(0)
 180                 { }
 181 
 182                 // Compare these nodes based on movement consumed.
 183                 bool operator<(const findroute_node& o) const {
 184                         return turns_left > o.turns_left ||
 185                                   (turns_left == o.turns_left && moves_left > o.moves_left);
 186                 }
 187         };
 188 
 189         /**
 190          * Converts map locations to and from integer indices.
 191          */
 192         struct findroute_indexer {
 193                 int w, h; // Width and height of the map.
 194 
 195                 // Constructor:
 196                 findroute_indexer(int a, int b) : w(a), h(b) { }
 197                 // Convert to an index: (returns -1 on out of bounds)
 198                 int operator()(int x, int y) const {
 199                         if ( x < 0  || w <= x  ||  y < 0 || h <= y )
 200                                 return -1;
 201                         else
 202                                 return x + y*w;
 203                 }
 204                 int operator()(const map_location& loc) const {
 205                         return (*this)(loc.x, loc.y);
 206                 }
 207                 // Convert from an index:
 208                 map_location operator()(int index) const {
 209                         return map_location(index%w, index/w);
 210                 }
 211         };

 213         /**
 214          * A function object for comparing indices.
 215          */
 216         struct findroute_comp {
 217                 const std::vector<findroute_node>& nodes;
 218 
 219                 // Constructor:
 220                 findroute_comp(const std::vector<findroute_node>& n) : nodes(n) { }
 221                 // Binary predicate evaluating the order of its arguments:
 222                 bool operator()(int l, int r) const {
 223                         return nodes[r] < nodes[l];
 224                 }

 228 /**
 229  * Creates a list of routes that a unit can traverse from the provided location.
 230  * (This is called when creating pathfind::paths and descendant classes.)
 231  *
 232  * @param[in]  origin        The location at which to begin the routes.
 233  * @param[in]  costs         The costs to use for route finding.
 234  * @param[in]  slowed        Whether or not to use the slowed costs.
 235  * @param[in]  moves_left    The number of movement points left for the current turn.
 236  * @param[in]  max_moves     The number of movement points in each future turn.
 237  * @param[in]  turns_left    The number of future turns of movement to calculate.
 238  * @param[out] destinations  The traversable routes.
 239  * @param[out] edges         The hexes (possibly off-map) adjacent to those in
 240  *                           destinations. (It is permissible for this to contain
 241  *                           some hexes that are also in destinations.)
 242  *
 243  * @param[in]  teleporter    If not nullptr, teleportation will be considered, using
 244  *                           this unit's abilities.
 245  * @param[in]  current_team  If not nullptr, enemies of this team can obstruct routes
 246  *                           both by occupying hexes and by exerting zones of control.
 247  *                           In addition, the presence of units can affect
 248  *                           teleportation options.
 249  * @param[in]  skirmisher    If not nullptr, use this to determine where ZoC can and
 250  *                           cannot be ignored (due to this unit having or not
 251  *                           having the skirmisher ability).
 252  *                           If nullptr, then ignore all zones of control.
 253  *                           (No effect if current_team is nullptr).
 254  * @param[in]  viewing_team  If not nullptr, use this team's vision when detecting
 255  *                           enemy units and teleport destinations.
 256  *                           If nullptr, then "see all".
 257  *                           (No effect if teleporter and current_team are both nullptr.)
 258  * @param[in]  jamming_map   The relevant "jamming" of the costs being used
 259  *                           (currently only used with vision costs).
 260  * @param[out] full_cost_map If not nullptr, build a cost_map instead of destinations.
 261  *                           Destinations is ignored.
 262  *                           full_cost_map is a vector of pairs. The first entry is the
 263  *                           cost itself, the second how many units already visited this hex
 264  * @param[in]  check_vision  If true, use vision check for teleports, that is, ignore
 265  *                           units potentially blocking the teleport exit
 266  */
 267 static void find_routes(
 268                 const map_location & origin, const movetype::terrain_costs & costs,
 269                 bool slowed, int moves_left, int max_moves, int turns_left,
 270                 paths::dest_vect & destinations, std::set<map_location> * edges,
 271                 const unit * teleporter, const team * current_team,
 272                 const unit * skirmisher, const team * viewing_team,
 273                 const std::map<map_location, int> * jamming_map=nullptr,
 274                 std::vector<std::pair<int, int> > * full_cost_map=nullptr, bool check_vision=false)
 275 {
 276         const gamemap& map = resources::gameboard->map();
 277 
 278         const bool see_all =  viewing_team == nullptr;
 279         // When see_all is true, the viewing team never matters, but we still
 280         // need to supply one to some functions.
 281         if ( viewing_team == nullptr )
 282                 viewing_team = &resources::gameboard->teams().front();
 283 
 284         // Build a teleport map, if needed.
 285         const teleport_map teleports = teleporter ?
 286                         get_teleport_locations(*teleporter, *viewing_team, see_all, current_team == nullptr, check_vision) :
 287                         teleport_map();
 288 
 289         // Since this is called so often, keep memory reserved for the node list.
 290         static std::vector<findroute_node> nodes;
 291         static unsigned search_counter = 0;
 292         // Incrementing search_counter means we ignore results from earlier searches.
 293         ++search_counter;
 294         // Whenever the counter cycles, trash the contents of nodes and restart at 1.
 295         if ( search_counter == 0 ) {
 296                 nodes.resize(0);
 297                 search_counter = 1;
 298         }
 299         // Initialize the nodes for this search.
 300         nodes.resize(map.w() * map.h());
 301         findroute_comp node_comp(nodes);
 302         findroute_indexer index(map.w(), map.h());
 303 
 304         // Check if full_cost_map has the correct size.
 305         // If not, ignore it. If yes, initialize the start position.
 306         if ( full_cost_map ) {
 307                 if ( full_cost_map->size() != static_cast<unsigned>(map.w() * map.h()) )
 308                         full_cost_map = nullptr;
 309                 else {
 310                         if ( (*full_cost_map)[index(origin)].second == 0 )
 311                                 (*full_cost_map)[index(origin)].first = 0;
 312                         (*full_cost_map)[index(origin)].second += 1;
 313                 }
 314         }
 315 
 316         // Used to optimize the final collection of routes.
 317         int xmin = origin.x, xmax = origin.x, ymin = origin.y, ymax = origin.y;
 318         int nb_dest = 1;
 319 
 320         // Record the starting location.
 321         assert(index(origin) >= 0);
 322         nodes[index(origin)] = findroute_node(moves_left, turns_left,
 323                                               map_location::null_location(),
 324                                               search_counter);
 325         // Begin the search at the starting location.
 326         std::vector<int> hexes_to_process(1, index(origin));  // Will be maintained as a heap.
 327 
 328         while ( !hexes_to_process.empty() ) {
 329                 // Process the hex closest to the origin.
 330                 const int cur_index = hexes_to_process.front();
 331                 const map_location cur_hex = index(cur_index);
 332                 const findroute_node& current = nodes[cur_index];
 333                 // Remove from the heap.
 334                 std::pop_heap(hexes_to_process.begin(), hexes_to_process.end(), node_comp);
 335                 hexes_to_process.pop_back();
 336 
 337                 // Get the locations adjacent to current.
 338                 std::vector<map_location> adj_locs(6);
 339                 get_adjacent_tiles(cur_hex, &adj_locs[0]);
 340                 if ( teleporter ) {
 341                         std::set<map_location> allowed_teleports;
 342                         teleports.get_adjacents(allowed_teleports, cur_hex);
 343                         adj_locs.insert(adj_locs.end(), allowed_teleports.begin(), allowed_teleports.end());
 344                 }
 345                 for ( int i = adj_locs.size()-1; i >= 0; --i ) {
 346                         // Get the node associated with this location.
 347                         const map_location & next_hex = adj_locs[i];
 348                         const int next_index = index(next_hex);
 349                         if ( next_index < 0 ) {
 350                                 // Off the map.
 351                                 if ( edges != nullptr )
 352                                         edges->insert(next_hex);
 353                                 continue;
 354                         }
 355                         findroute_node & next = nodes[next_index];
 356 
 357                         // Skip nodes we have already collected.
 358                         // (Since no previously checked routes were longer than
 359                         // the current one, the current route cannot be shorter.)
 360                         // (Significant difference from classic Dijkstra: we have
 361                         // vertex weights, not edge weights.)
 362                         if ( next.search_num == search_counter )
 363                                 continue;
 364 
 365                         // If we go to next, it will be from current.
 366                         next.prev = cur_hex;
 367 
 368                         // Calculate the cost of entering next_hex.
 369                         int cost = costs.cost(map[next_hex], slowed);
 370                         if ( jamming_map ) {
 371                                 const std::map<map_location, int>::const_iterator jam_it =
 372                                         jamming_map->find(next_hex);
 373                                 if ( jam_it != jamming_map->end() )
 374                                         cost += jam_it->second;
 375                         }
 376 
 377                         // Calculate movement remaining after entering next_hex.
 378                         next.moves_left = current.moves_left - cost;
 379                         next.turns_left = current.turns_left;
 380                         if ( next.moves_left < 0 ) {
 381                                 // Have to delay until the next turn.
 382                                 next.turns_left--;
 383                                 next.moves_left = max_moves - cost;
 384                         }
 385                         if ( next.moves_left < 0 || next.turns_left < 0 ) {
 386                                 // Either can never enter this hex or out of turns.
 387                                 if ( edges != nullptr )
 388                                         edges->insert(next_hex);
 389                                 continue;
 390                         }
 391 
 392                         if ( current_team ) {
 393                                 // Account for enemy units.
 394                                 const unit *v = resources::gameboard->get_visible_unit(next_hex, *viewing_team, see_all);
 395                                 if ( v && current_team->is_enemy(v->side()) ) {
 396                                         // Cannot enter enemy hexes.
 397                                         if ( edges != nullptr )
 398                                                 edges->insert(next_hex);
 399                                         continue;
 400                                 }
 401 
 402                                 if ( skirmisher  &&  next.moves_left > 0  &&
 403                                      enemy_zoc(*current_team, next_hex, *viewing_team, see_all)  &&
 404                                      !skirmisher->get_ability_bool("skirmisher", next_hex, *resources::gameboard) ) {
 405                                         next.moves_left = 0;
 406                                 }
 407                         }
 408 
 409                         // Update full_cost_map
 410                         if ( full_cost_map ) {
 411                                 if ( (*full_cost_map)[next_index].second == 0 )
 412                                         (*full_cost_map)[next_index].first = 0;
 413                                 int summed_cost = (turns_left - next.turns_left + 1) * max_moves - next.moves_left;
 414                                 (*full_cost_map)[next_index].first += summed_cost;
 415                                 (*full_cost_map)[next_index].second += 1;
 416                         }
 417 
 418                         // Mark next as being collected.
 419                         next.search_num = search_counter;
 420 
 421                         // Add this node to the heap.
 422                         hexes_to_process.push_back(next_index);
 423                         std::push_heap(hexes_to_process.begin(), hexes_to_process.end(), node_comp);
 424 
 425                         // Bookkeeping (for later).
 426                         ++nb_dest;
 427                         if ( next_hex.x < xmin )
 428                                 xmin = next_hex.x;
 429                         else if ( xmax < next_hex.x )
 430                                 xmax = next_hex.x;
 431                         if ( next_hex.y < ymin )
 432                                 ymin = next_hex.y;
 433                         else if ( ymax < next_hex.y )
 434                                 ymax = next_hex.y;
 435                 }//for (i)
 436         }//while (hexes_to_process)
 437 
 438         // Currently the only caller who uses full_cost_map doesn't need the
 439         // destinations. We can skip this part.
 440         if ( full_cost_map ) {
 441                 return;
 442         }
 443 
 444         // Build the routes for every map_location that we reached.
 445         // The ordering must be compatible with map_location::operator<.
 446         destinations.reserve(nb_dest);
 447         for (int x = xmin; x <= xmax; ++x) {
 448                 for (int y = ymin; y <= ymax; ++y)
 449                 {
 450                         const findroute_node &n = nodes[index(x,y)];
 451                         if ( n.search_num == search_counter ) {
 452                                 paths::step s =
 453                                         { map_location(x,y), n.prev, n.moves_left + n.turns_left*max_moves };
 454                                 destinations.push_back(s);
 455                         }
 456                 }
 457         }
 458 }

 460 static bool step_compare(const paths::step& a, const map_location& b) {
 461         return a.curr < b;
 462 }

 464 paths::dest_vect::const_iterator paths::dest_vect::find(const map_location &loc) const
 465 {
 466         const_iterator i = std::lower_bound(begin(), end(), loc, step_compare);
 467         if (i != end() && i->curr != loc) return end();
 468         return i;
 469 }
 470 
 471 void paths::dest_vect::insert(const map_location &loc)
 472 {
 473         iterator i = std::lower_bound(begin(), end(), loc, step_compare);
 474         if (i != end() && i->curr == loc) return;
 475         paths::step s { loc, map_location(), 0 };
 476         std::vector<step>::insert(i, s);
 477 }

 479 /**
 480  * Returns the path going from the source point (included) to the
 481  * destination point @a j (excluded).
 482  */
 483 std::vector<map_location> paths::dest_vect::get_path(const const_iterator &j) const
 484 {
 485         std::vector<map_location> path;
 486         if (!j->prev.valid()) {
 487                 path.push_back(j->curr);
 488         } else {
 489                 const_iterator i = j;
 490                 do {
 491                         i = find(i->prev);
 492                         assert(i != end());
 493                         path.push_back(i->curr);
 494                 } while (i->prev.valid());
 495         }
 496         std::reverse(path.begin(), path.end());
 497         return path;
 498 }

 501 {
 502         return find(loc) != end();

 505 /**
 506  * Construct a list of paths for the specified unit.
 507  *
 508  * This function is used for several purposes, including showing a unit's
 509  * potential moves and generating currently possible paths.
 510  * @param u                The unit whose moves and movement type will be used.
 511  * @param force_ignore_zoc Set to true to completely ignore zones of control.
 512  * @param allow_teleport   Set to true to consider teleportation abilities.
 513  * @param viewing_team     Usually the current team, except for "show enemy moves", etc.
 514  * @param additional_turns The number of turns to account for, in addition to the current.
 515  * @param see_all          Set to true to remove unit visibility from consideration.
 516  * @param ignore_units     Set to true if units should never obstruct paths (implies ignoring ZoC as well).
 517  */
 518 paths::paths(const unit& u, bool force_ignore_zoc,
 519                 bool allow_teleport, const team &viewing_team,
 520                 int additional_turns, bool see_all, bool ignore_units)
 521         : destinations()
 522 {
 523         std::vector<team> const &teams = resources::gameboard->teams();
 524         if (u.side() < 1 || u.side() > int(teams.size())) {
 525                 return;
 526         }
 527 
 528         find_routes(u.get_location(), u.movement_type().get_movement(),
 529                     u.get_state(unit::STATE_SLOWED), u.movement_left(),
 530                     u.total_movement(), additional_turns, destinations, nullptr,
 531                     allow_teleport ? &u : nullptr,
 532                     ignore_units ? nullptr : &teams[u.side()-1],
 533                     force_ignore_zoc ? nullptr : &u,
 534                     see_all ? nullptr : &viewing_team);
 535 }

 544 /**
 545  * Constructs a list of vision paths for a unit.
 546  *
 547  * This is used to construct a list of hexes that the indicated unit can see.
 548  * It differs from pathfinding in that it will only ever go out one turn,
 549  * and that it will also collect a set of border hexes (the "one hex beyond"
 550  * movement to which vision extends).
 551  * @param viewer     The unit doing the viewing.
 552  * @param loc        The location from which the viewing occurs
 553  *                   (does not have to be the unit's location).
 554  */
 555 vision_path::vision_path(const unit& viewer, map_location const &loc,
 556                          const std::map<map_location, int>& jamming_map)
 557         : paths(), edges()
 558 {
 559         const int sight_range = viewer.vision();
 560 
 561         // The three nullptr parameters indicate (in order):
 562         // ignore units, ignore ZoC (no effect), and don't build a cost_map.
 563         team const& viewing_team = resources::gameboard->teams()[resources::screen->viewing_team()];
 564         find_routes(loc, viewer.movement_type().get_vision(),
 565                     viewer.get_state(unit::STATE_SLOWED), sight_range, sight_range,
 566                     0, destinations, &edges, &viewer, nullptr, nullptr, &viewing_team, &jamming_map, nullptr, true);
 567 }

 572  * This constructor is provided so that only the relevant portion of a unit's

 625 marked_route mark_route(const plain_route &rt)
 626 {
 627         marked_route res;
 628 
 629         if (rt.steps.empty()) return marked_route();
 630         res.route = rt;
 631 
 632         unit_map::const_iterator it = resources::gameboard->units().find(rt.steps.front());
 633         if (it == resources::gameboard->units().end()) return marked_route();
 634         unit const& u = *it;
 635 
 636         int turns = 0;
 637         int movement = u.movement_left();
 638         const team& unit_team = resources::gameboard->get_team(u.side());
 639         bool zoc = false;
 640 
 641         std::vector<map_location>::const_iterator i = rt.steps.begin();
 642 
 643         for (; i !=rt.steps.end(); ++i) {
 644                 bool last_step = (i+1 == rt.steps.end());
 645 
 646                 // move_cost of the next step is irrelevant for the last step
 647                 assert(last_step || resources::gameboard->map().on_board(*(i+1)));
 648                 const int move_cost = last_step ? 0 : u.movement_cost((resources::gameboard->map())[*(i+1)]);
 649 
 650                 team const& viewing_team = resources::gameboard->teams()[resources::screen->viewing_team()];
 651 
 652                 if (last_step || zoc || move_cost > movement) {
 653                         // check if we stop an a village and so maybe capture it
 654                         // if it's an enemy unit and a fogged village, we assume a capture
 655                         // (if he already owns it, we can't know that)
 656                         // if it's not an enemy, we can always know if he owns the village
 657                         bool capture = resources::gameboard->map().is_village(*i) && ( !unit_team.owns_village(*i)
 658                                  || (viewing_team.is_enemy(u.side()) && viewing_team.fogged(*i)) );
 659 
 660                         ++turns;
 661 
 662                         bool invisible = u.invisible(*i, *resources::gameboard, false);
 663 
 664                         res.marks[*i] = marked_route::mark(turns, zoc, capture, invisible);
 665 
 666                         if (last_step) break; // finished and we used dummy move_cost
 667 
 668                         movement = u.total_movement();
 669                         if(move_cost > movement) {
 670                                 return res; //we can't reach destination
 671                         }
 672                 }
 673 
 674                 zoc = enemy_zoc(unit_team, *(i + 1), viewing_team)
 675                                         && !u.get_ability_bool("skirmisher", *(i+1), *resources::gameboard);
 676 
 677                 if (zoc) {
 678                         movement = 0;
 679                 } else {
 680                         movement -= move_cost;
 681                 }
 682         }
 683 
 684         return res;
 685 }

 697 double shortest_path_calculator::cost(const map_location& loc, const double so_far) const
 698 {
 699         assert(map_.on_board(loc));
 700 
 701         // loc is shrouded, consider it impassable
 702         // NOTE: This is why AI must avoid to use shroud
 703         if (!see_all_ && viewing_team_.shrouded(loc))
 704                 return getNoPathValue();
 705 
 706         const t_translation::terrain_code terrain = map_[loc];
 707         const int terrain_cost = unit_.movement_cost(terrain);
 708         // Pathfinding heuristic: the cost must be at least 1
 709         VALIDATE(terrain_cost >= 1, _("Terrain with a movement cost less than 1 encountered."));
 710 
 711         // Compute how many movement points are left in the game turn
 712         // needed to reach the previous hex.
 713         // total_movement_ is not zero, thanks to the pathfinding heuristic
 714         int remaining_movement = movement_left_ - static_cast<int>(so_far);
 715         if (remaining_movement < 0) {
 716                 remaining_movement = total_movement_ - (-remaining_movement) % total_movement_;
 717         }
 718 
 719         if (terrain_cost >= movetype::UNREACHABLE || (total_movement_ < terrain_cost && remaining_movement < terrain_cost)) {
 720                 return getNoPathValue();
 721         }
 722 
 723         int other_unit_subcost = 0;
 724         if (!ignore_unit_) {
 725                 const unit *other_unit =
 726                         resources::gameboard->get_visible_unit(loc, viewing_team_, see_all_);
 727 
 728                 // We can't traverse visible enemy and we also prefer empty hexes
 729                 // (less blocking in multi-turn moves and better when exploring fog,
 730                 // because we can't stop on a friend)
 731 
 732                 if (other_unit)
 733                 {
 734                         if (teams_[unit_.side() - 1].is_enemy(other_unit->side()))
 735                                 return getNoPathValue();
 736                         else
 737                                 // This value will be used with the defense_subcost (see below)
 738                                 // The 1 here means: consider occupied hex as a -1% defense
 739                                 // (less important than 10% defense because friends may move)
 740                                 other_unit_subcost = 1;
 741                 }
 742         }
 743 
 744         // this will sum all different costs of this move
 745         int move_cost = 0;
 746 
 747         // Suppose that we have only 2 remaining MP and want to move onto a hex
 748         // costing 3 MP. We don't have enough MP now, so we must end our turn here,
 749         // thus spend our remaining MP by waiting (next turn, with full MP, we will
 750         // be able to move on that hex)
 751         if (remaining_movement < terrain_cost) {
 752                 move_cost += remaining_movement;
 753                 remaining_movement = total_movement_; // we consider having full MP now
 754         }
 755 
 756         // check ZoC
 757         if (!ignore_unit_ && remaining_movement != terrain_cost
 758             && enemy_zoc(teams_[unit_.side()-1], loc, viewing_team_, see_all_)
 759                         && !unit_.get_ability_bool("skirmisher", loc, *resources::gameboard)) {
 760                 // entering ZoC cost all remaining MP
 761                 move_cost += remaining_movement;
 762         } else {
 763                 // empty hex, pay only the terrain cost
 764                 move_cost += terrain_cost;
 765         }
 766 
 767         // We will add a tiny cost based on terrain defense, so the pathfinding
 768         // will prefer good terrains between 2 with the same MP cost
 769         // Keep in mind that defense_modifier is inverted (= 100 - defense%)
 770         const int defense_subcost = ignore_defense_ ? 0 : unit_.defense_modifier(terrain);
 771 
 772         // We divide subcosts by 100 * 100, because defense is 100-based and
 773         // we don't want any impact on move cost for less then 100-steps path
 774         // (even ~200 since mean defense is around ~50%)
 775         return move_cost + (defense_subcost + other_unit_subcost) / 10000.0;
 776 }

 783 // This is an simplified version of shortest_path_calculator (see above for explanation)
 784 double move_type_path_calculator::cost(const map_location& loc, const double so_far) const
 785 {
 786         assert(map_.on_board(loc));
 787         if (viewing_team_.shrouded(loc))
 788                 return getNoPathValue();
 789 
 790         const t_translation::terrain_code terrain = map_[loc];
 791         const int terrain_cost = movement_type_.movement_cost(terrain);
 792 
 793         if (total_movement_ < terrain_cost)
 794                 return getNoPathValue();
 795 
 796         int remaining_movement = movement_left_ - static_cast<int>(so_far);
 797         if (remaining_movement < 0)
 798                 remaining_movement = total_movement_ - (-remaining_movement) % total_movement_;
 799 
 800         int move_cost = 0;
 801 
 802         if (remaining_movement < terrain_cost) {
 803                 move_cost += remaining_movement;
 804         }
 805 
 806         move_cost += terrain_cost;
 807 
 808         return move_cost;
 809 }

 869 /**
 870  * Adds a units cost map to cost_map (increments the elements in cost_map)
 871  * @param u a real existing unit on the map
 872  */
 873 void full_cost_map::add_unit(const unit& u, bool use_max_moves)
 874 {
 875         std::vector<team> const &teams = resources::gameboard->teams();
 876         if (u.side() < 1 || u.side() > int(teams.size())) {
 877                 return;
 878         }
 879 
 880         // We don't need the destinations, but find_routes() wants to have this parameter
 881         paths::dest_vect dummy = paths::dest_vect();
 882 
 883                 find_routes(u.get_location(), u.movement_type().get_movement(),
 884                             u.get_state(unit::STATE_SLOWED),
 885                             (use_max_moves) ? u.total_movement() : u.movement_left(),
 886                             u.total_movement(), 99, dummy, nullptr,
 887                             allow_teleport_ ? &u : nullptr,
 888                             ignore_units_ ? nullptr : &teams[u.side()-1],
 889                             force_ignore_zoc_ ? nullptr : &u,
 890                             see_all_ ? nullptr : &viewing_team_,
 891                             nullptr, &cost_map);
 892 }

 911 /**
 912  * Accessor for the cost/reach-amount pairs.
 913  * Read comment in pathfind.hpp to cost_map.
 914  *
 915  * @return the entry of the cost_map at (x, y)
 916  *         or (-1, 0) if value is not set or (x, y) is invalid.
 917  */
 918 std::pair<int, int> full_cost_map::get_pair_at(int x, int y) const
 919 {
 920         const gamemap& map = resources::gameboard->map();
 921         assert(cost_map.size() == static_cast<unsigned>(map.w() * map.h()));
 922 
 923         if (x < 0 || x >= map.w() || y < 0 || y >= map.h()) {
 924                 return std::make_pair(-1, 0);  // invalid
 925         }
 926 
 927         return cost_map[x + (y * map.w())];
 928 }
