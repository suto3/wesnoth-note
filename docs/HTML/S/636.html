 178 widget* builder_drawing::build() const
 179 {
 180         drawing* widget = new drawing();
 181 
 182         init_control(widget);
 183 
 184         const wfl::map_formula_callable& size = get_screen_size_variables();
 185 
 186         const unsigned w = width(size);
 187         const unsigned h = height(size);
 188 
 189         if(w || h) {
 190                 widget->set_best_size(point(w, h));
 191         }
 192 
 193         widget->set_drawing_data(draw);
 194 
 195         DBG_GUI_G << "Window builder: placed drawing '" << id
 196                           << "' with definition '" << definition << "'.\n";
 197 
 198         return widget;
 199 }
