   7 #define LOAD_SUBMAP X_SPAN Y_SPAN OFFSET_X OFFSET_Y MAP
   8     [store_unit]
   9         [filter]
  10             x=1-999
  11             y=1-999
  12         [/filter]
  13         variable=MOVE_UNIT_store
  14         kill=yes
  15     [/store_unit]
  16     [store_villages]
  17         variable=villages
  18     [/store_villages]
  19 
  20     [foreach]
  21         array=MOVE_UNIT_store
  22         [do]
  23             {VARIABLE_OP this_item.x add {OFFSET_X}}
  24             {VARIABLE_OP this_item.y add {OFFSET_Y}}
  25             [if]
  26                 [variable]
  27                     name=this_item.goto_x
  28                     numerical_not_equals=0
  29                 [/variable]
  30                 [then]
  31                     {VARIABLE_OP this_item.goto_x add {OFFSET_X}}
  32                 [/then]
  33             [/if]
  34             [if]
  35                 [variable]
  36                     name=this_item.goto_y
  37                     numerical_not_equals=0
  38                 [/variable]
  39                 [then]
  40                     {VARIABLE_OP this_item.goto_y add {OFFSET_Y}}
  41                 [/then]
  42             [/if]
  43         [/do]
  44     [/foreach]
  45 
  46     [foreach]
  47         array=villages
  48         [do]
  49             #Make the villages on the old map neutral
  50             [capture_village]
  51                 x=$this_item.x
  52                 y=$this_item.y
  53             [/capture_village]
  54             {VARIABLE_OP this_item.x add {OFFSET_X}}
  55             {VARIABLE_OP this_item.y add {OFFSET_Y}}
  56         [/do]
  57     [/foreach]
  58 
  59     [replace_map]
  60         x={X_SPAN}
  61         y={Y_SPAN}
  62         {LOW_MAP {MAP}}
  63     [/replace_map]
  64     [shift_labels]
  65         x={OFFSET_X}
  66         y={OFFSET_Y}
  67     [/shift_labels]
  68 
  69     [foreach]
  70         array=MOVE_UNIT_store
  71         [do]
  72             [unstore_unit]
  73                 variable=this_item
  74                 find_vacant=no
  75             [/unstore_unit]
  76         [/do]
  77     [/foreach]
  78     {CLEAR_VARIABLE MOVE_UNIT_store}
  79 
  80     [foreach]
  81         array=villages
  82         [do]
  83             [capture_village]
  84                 x=$this_item.x
  85                 y=$this_item.y
  86                 side=$this_item.owner_side
  87             [/capture_village]
  88         [/do]
  89     [/foreach]
  90     {CLEAR_VARIABLE villages}
  91 #enddef
