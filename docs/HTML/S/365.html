  88  * Fetch the selected game's config from the server and prompts an initial faction selection.
  89  */
  90 bool mp_join_game::fetch_game_config(CVideo& video)
  91 {
  92         // Ask for the next scenario data, if applicable
  93         if(!first_scenario_) {
  94                 wesnothd_connection_.send_data(config("load_next_scenario"));
  95         }
  96 
  97         bool has_scenario_and_controllers = false;
  98         while(!has_scenario_and_controllers) {
  99                 config revc;
 100                 const bool data_res = gui2::dialogs::network_transmission::wesnothd_receive_dialog(
 101                         video, "download level data", revc, wesnothd_connection_);
 102 
 103                 if(!data_res) {
 104                         return false;
 105                 }
 106 
 107                 mp::check_response(data_res, revc);
 108 
 109                 if(revc.child("leave_game")) {
 110                         return false;
 111                 } else if(config& next_scenario = revc.child("next_scenario")) {
 112                         level_.swap(next_scenario);
 113                 } else if(revc.has_attribute("version")) {
 114                         level_.swap(revc);
 115 
 116                         has_scenario_and_controllers = true;
 117                 } else if(config& controllers = revc.child("controllers")) {
 118                         int index = 0;
 119                         for(const config& controller : controllers.child_range("controller")) {
 120                                 if(config& side = get_scenario().child("side", index)) {
 121                                         side["is_local"] = controller["is_local"];
 122                                 }
 123                                 ++index;
 124                         }
 125 
 126                         has_scenario_and_controllers = true;
 127                 }
 128         }
 129 
 130         if(level_["started"].to_bool()) {
 131                 mp::level_to_gamestate(level_, state_);
 132                 return true;
 133         }
 134 
 135         if(first_scenario_) {
 136                 state_ = saved_game();
 137                 state_.classification() = game_classification(level_);
 138 
 139                 // Make sure that we have the same config as host, if possible.
 140                 game_config_manager::get()->load_game_config_for_game(state_.classification());
 141         }
 142 
 143         game_config::add_color_info(get_scenario());
 144 
 145         // If we're just an observer, we don't need to find an appropriate side and set faction selection
 146         if(observe_game_) {
 147                 return true;
 148         }
 149 
 150         // Search for an appropriate vacant slot. If a description is set (i.e. we're loading from a saved game),
 151         // then prefer to get the side with the same description as our login. Otherwise just choose the first
 152         // available side.
 153         const config* side_choice = nullptr;
 154 
 155         int side_num = 0, nb_sides = 0;
 156         for(const config& side : get_scenario().child_range("side")) {
 157                 if(side["controller"] == "reserved" && side["current_player"] == preferences::login()) {
 158                         side_choice = &side;
 159                         side_num = nb_sides;
 160                         break;
 161                 }
 162 
 163                 if(side["controller"] == "human" && side["player_id"].empty()) {
 164                         if(!side_choice) { // Found the first empty side
 165                                 side_choice = &side;
 166                                 side_num = nb_sides;
 167                         }
 168 
 169                         if(side["current_player"] == preferences::login()) {
 170                                 side_choice = &side;
 171                                 side_num = nb_sides;
 172                                 break;  // Found the preferred one
 173                         }
 174                 }
 175 
 176                 if(side["player_id"] == preferences::login()) {
 177                         // We already own a side in this game
 178                         return true;
 179                 }
 180 
 181                 ++nb_sides;
 182         }
 183 
 184         if(!side_choice) {
 185                 return false;
 186         }
 187 
 188         // If the client is allowed to choose their team, do that here instead of having it set by the server
 189         if((*side_choice)["allow_changes"].to_bool(true)) {
 190                 events::event_context context;
 191 
 192                 const config& era = level_.child("era");
 193                 // TODO: Check whether we have the era. If we don't, inform the player
 194                 if(!era) {
 195                         throw config::error(_("No era information found."));
 196                 }
 197 
 198                 config::const_child_itors possible_sides = era.child_range("multiplayer_side");
 199                 if(possible_sides.empty()) {
 200                         return false;
 201                 }
 202 
 203                 const std::string color = (*side_choice)["color"].str();
 204 
 205                 std::vector<const config*> era_factions;
 206                 for(const config& side : possible_sides) {
 207                         era_factions.push_back(&side);
 208                 }
 209 
 210                 const bool is_mp = state_.classification().is_normal_mp_game();
 211                 const bool lock_settings = get_scenario()["force_lock_settings"].to_bool(!is_mp);
 212                 const bool use_map_settings = level_.child("multiplayer")["mp_use_map_settings"].to_bool();
 213                 const bool saved_game = level_.child("multiplayer")["savegame"].to_bool();
 214 
 215                 ng::flg_manager flg(era_factions, *side_choice, lock_settings, use_map_settings, saved_game);
 216 
 217                 gui2::dialogs::faction_select dlg(flg, color, side_num);
 218                 dlg.show(video);
 219 
 220                 if(dlg.get_retval() != gui2::window::OK) {
 221                         return false;
 222                 }
 223 
 224                 config faction;
 225                 config& change = faction.add_child("change_faction");
 226                 change["change_faction"] = true;
 227                 change["name"] = preferences::login();
 228                 change["faction"] = flg.current_faction()["id"];
 229                 change["leader"] = flg.current_leader();
 230                 change["gender"] = flg.current_gender();
 231 
 232                 wesnothd_connection_.send_data(faction);
 233         }
 234 
 235         return true;
 236 }

 238 static std::string generate_user_description(const config& side)
 239 {
 240         // Allow the host to override, since only the host knows the ai_algorithm.
 241         if(const config::attribute_value* desc = side.get("user_description")) {
 242                 return desc->str();
 243         }
 244 
 245         const std::string controller_type = side["controller"].str();
 246         const std::string reservation = side["reserved_for"].str();
 247         const std::string owner = side["player_id"].str();
 248 
 249         if(controller_type == "ai") {
 250                 return _("Computer Player");
 251         } else if(controller_type == "null") {
 252                 return _("Empty slot");
 253         } else if(controller_type == "reserved") {
 254                 return vgettext("Reserved for $playername", {{"playername", reservation}});
 255         } else if(owner.empty()) {
 256                 return _("Vacant slot");
 257         } else if(controller_type == "human" || controller_type == "network") {
 258                 return owner;
 259         } else {
 260                 return _("empty");
 261         }
 262 }

 264 void mp_join_game::pre_show(window& window)
 265 {
 266         window.set_enter_disabled(true);
 267 
 268         //
 269         // Set title
 270         //
 271         label& title = find_widget<label>(&window, "title", false);
 272         title.set_label((formatter() << title.get_label() << " " << font::unicode_em_dash << " " << get_scenario()["name"].t_str()).str());
 273 
 274         //
 275         // Set up sides list
 276         //
 277         generate_side_list(window);
 278 
 279         //
 280         // Initialize chatbox and game rooms
 281         //
 282         chatbox& chat = find_widget<chatbox>(&window, "chat", false);
 283 
 284         chat.set_lobby_info(lobby_info_);
 285         chat.set_wesnothd_connection(wesnothd_connection_);
 286 
 287         chat.room_window_open("this game", true, false); // TODO: better title?
 288         chat.active_window_changed();
 289 
 290         //
 291         // Set up player list
 292         //
 293         update_player_list(window);
 294 
 295         //
 296         // Set up the network handling
 297         //
 298         update_timer_ = add_timer(game_config::lobby_network_timer, std::bind(&mp_join_game::network_handler, this, std::ref(window)), true);
 299 
 300         //
 301         // Set up the Lua plugin context
 302         //
 303         plugins_context_.reset(new plugins_context("Multiplayer Join"));
 304 
 305         plugins_context_->set_callback("launch", [&window](const config&) { window.set_retval(window::OK); }, false);
 306         plugins_context_->set_callback("quit",   [&window](const config&) { window.set_retval(window::CANCEL); }, false);
 307         plugins_context_->set_callback("chat",   [&chat](const config& cfg) { chat.send_chat_message(cfg["message"], false); }, true);
 308 }

 310 void mp_join_game::generate_side_list(window& window)
 311 {
 312         if(stop_updates_) {
 313                 return;
 314         }
 315 
 316         tree_view& tree = find_widget<tree_view>(&window, "side_list", false);
 317 
 318         window.keyboard_capture(&tree);
 319 
 320         tree.clear();
 321         team_tree_map_.clear();
 322         const std::map<std::string, string_map> empty_map;
 323 
 324         for(const auto& side : get_scenario().child_range("side")) {
 325                 if(!side["allow_player"].to_bool(true)) {
 326                         continue;
 327                 }
 328 
 329                 // Check to see whether we've added a toplevel tree node for this team. If not, add one
 330                 if(team_tree_map_.find(side["team_name"].str()) == team_tree_map_.end()) {
 331                         std::map<std::string, string_map> data;
 332                         string_map item;
 333 
 334                         item["label"] = (formatter() << _("Team:") << " " << side["user_team_name"]).str();
 335                         data.emplace("tree_view_node_label", item);
 336 
 337                         tree_view_node& team_node = tree.add_node("team_header", data);
 338                         team_node.add_sibling("side_spacer", empty_map);
 339 
 340                         team_tree_map_[side["team_name"].str()] = &team_node;
 341                 }
 342 
 343                 std::map<std::string, string_map> data;
 344                 string_map item;
 345 
 346                 const std::string color = !side["color"].empty() ? side["color"] : side["side"].str();
 347 
 348                 item["label"] = (formatter() << "<span color='" << font::get_pango_color_from_id(color) << "'>" << side["side"] << "</span>").str();
 349                 data.emplace("side_number", item);
 350 
 351                 std::string leader_image = ng::random_enemy_picture;
 352                 std::string leader_type = side["type"];
 353                 std::string leader_name;
 354 
 355                 const std::string leader_gender = side["gender"];
 356 
 357                 // If there is a unit which can recruit, use it as a leader.
 358                 // Necessary to display leader information when loading saves.
 359                 for(const config& side_unit : side.child_range("unit")) {
 360                         if(side_unit["canrecruit"].to_bool()) {
 361                                 leader_type = side_unit["type"].str();
 362                                 break;
 363                         }
 364                 }
 365 
 366                 if(const unit_type* ut = unit_types.find(leader_type)) {
 367                         const unit_type& type = ut->get_gender_unit_type(leader_gender);
 368 
 369                         leader_image = formatter() << type.image() << "~RC(" << type.flag_rgb() << ">" << color << ")";
 370                         leader_name = type.type_name();
 371                 }
 372 
 373                 item["label"] = leader_image;
 374                 data.emplace("leader_image", item);
 375 
 376                 std::string description = generate_user_description(side);
 377                 if(!leader_name.empty()) {
 378                         description += formatter() << " (<i>" << leader_name << "</i>)";
 379                 }
 380 
 381                 item["label"] = description;
 382                 data.emplace("leader_type", item);
 383 
 384                 item["label"] = (formatter() << "<span color='#a69275'>" << side["faction_name"] << "</span>").str();
 385                 data.emplace("leader_faction", item);
 386 
 387                 std::string gender_icon = "icons/icon-random.png";
 388                 if(side["gender"] != "null") {
 389                         gender_icon = formatter() << "icons/icon-" << leader_gender << ".png";
 390                 }
 391 
 392                 item["label"] = gender_icon;
 393                 item["tooltip"] = side["gender"];
 394                 data.emplace("leader_gender", item);
 395 
 396                 item.clear();
 397 
 398                 // Don't show gold for saved games
 399                 // TODO: gold icon
 400                 if(side["allow_changes"].to_bool()) {
 401                         item["label"] = side["gold"].str() + " " + _("Gold");
 402                         data.emplace("side_gold", item);
 403                 }
 404 
 405                 const int income_amt = side["income"];
 406                 if(income_amt != 0) {
 407                         const std::string income_string = formatter() << (income_amt > 0 ? "+" : "") << income_amt << " " << _("Income");
 408 
 409                         item["label"] = income_string;
 410                         data.emplace("side_income", item);
 411                 }
 412 
 413                 tree_view_node& node = team_tree_map_[side["team_name"].str()]->add_child("side_panel", data);
 414 
 415                 grid& row_grid = node.get_grid();
 416 
 417                 if(income_amt == 0) {
 418                         find_widget<image>(&row_grid, "income_icon", false).set_visible(widget::visibility::invisible);
 419                         find_widget<label>(&row_grid, "side_income", false).set_visible(widget::visibility::invisible);
 420                 }
 421         }
 422 }

 441 void mp_join_game::network_handler(window& window)
 442 {
 443         // If the game has already started, close the dialog immediately.
 444         if (level_["started"].to_bool())
 445         {
 446                 window.set_retval(window::OK);
 447                 return;
 448         }
 449 
 450         config data;
 451         if(!wesnothd_connection_.receive_data(data)) {
 452                 return;
 453         }
 454 
 455         // Update chat
 456         find_widget<chatbox>(&window, "chat", false).process_network_data(data);
 457 
 458         if(!data["message"].empty()) {
 459                 gui2::show_transient_message(window.video(), _("Response") , data["message"]);
 460         }
 461 
 462         if(data["failed"].to_bool()) {
 463                 window.set_retval(window::CANCEL);
 464         } else if(data.child("start_game")) {
 465                 level_["started"] = true;
 466                 window.set_retval(window::OK);
 467         } else if(data.child("leave_game")) {
 468                 window.set_retval(window::CANCEL);
 469         }
 470 
 471         if(data.child("stop_updates")) {
 472                 stop_updates_ = true;
 473         } else if(const config& c = data.child("scenario_diff")) {
 474                 // TODO: We should catch config::error and then leave the game.
 475                 level_.apply_diff(c);
 476 
 477                 generate_side_list(window);
 478         } else if(const config& change = data.child("change_controller")) {
 479                 if(config& side_to_change = get_scenario().find_child("side", "side", change["side"])) {
 480                         side_to_change.merge_with(change);
 481                 }
 482         } else if(data.has_child("scenario") || data.has_child("snapshot") || data.child("next_scenario")) {
 483                 level_ = first_scenario_ ? data : data.child("next_scenario");
 484 
 485                 generate_side_list(window);
 486         }
 487 
 488         if (data.has_child("turn")) {
 489                 ERR_MP << "received replay data\n" << data << "\n in mp join\n";
 490         }
 491 
 492         // Update player list
 493         // TODO: optimally, it wouldn't regenerate the entire list every single refresh cycle
 494         update_player_list(window);
 495 }

 497 config& mp_join_game::get_scenario()

 500                 return scenario;

 508 void mp_join_game::post_show(window& window)
 509 {
 510         if(update_timer_ != 0) {
 511                 remove_timer(update_timer_);
 512                 update_timer_ = 0;
 513         }
 514 
 515         if(window.get_retval() == window::OK) {
 516                 if(const config& stats = level_.child("statistics")) {
 517                         statistics::fresh_stats();
 518                         statistics::read_stats(stats);
 519                 }
 520 
 521                 mp::level_to_gamestate(level_, state_);
 522 
 523                 mp_ui_alerts::game_has_begun();
 524         } else {
 525                 wesnothd_connection_.send_data(config("leave_game"));
 526         }
 527 }
