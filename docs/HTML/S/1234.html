  64 unit_const_ptr find_backup_leader(const unit & leader)
  65 {
  66         assert(leader.can_recruit());
  67         assert(resources::gameboard->map().is_keep(leader.get_location()));
  68         for (unit_map::const_iterator unit = resources::gameboard->units().begin(); unit != resources::gameboard->units().end(); unit++)
  69         {
  70                 if (unit->can_recruit() && unit->id() != leader.id())
  71                 {
  72                         if ( dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*unit, leader.get_location()) )
  73                                 return unit.get_shared_ptr();
  74                 }
  75         }
  76         return unit_const_ptr();
  77 }

  79 unit* find_recruiter(size_t team_index, map_location const& hex)
  80 {
  81         if ( !resources::gameboard->map().is_castle(hex) )
  82                 return nullptr;
  83 
  84         for(unit& u : resources::gameboard->units())
  85                 if(u.can_recruit()
  86                                 && u.side() == static_cast<int>(team_index+1)
  87                                 && dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(u, hex))
  88                         return &u;
  89         return nullptr;
  90 }

 113 int path_cost(std::vector<map_location> const& path, unit const& u)
 114 {
 115         if(path.size() < 2)
 116                 return 0;
 117 
 118         team const& u_team = resources::gameboard->get_team(u.side());
 119         map_location const& dest = path.back();
 120         if ( (resources::gameboard->map().is_village(dest) && !u_team.owns_village(dest))
 121              || pathfind::enemy_zoc(u_team, dest, u_team) )
 122                 return u.total_movement();
 123 
 124         int result = 0;
 125         gamemap const& map = resources::gameboard->map();
 126         for(map_location const& loc : std::make_pair(path.begin()+1,path.end())) {
 127                 result += u.movement_cost(map[loc]);
 128         }
 129         return result;
 130 }

 169 void for_each_action(std::function<void(action*)> function, team_filter team_filter)
 170 {
 171         bool end = false;
 172         for(size_t turn=0; !end; ++turn) {
 173                 end = true;
 174                 for(team &side : resources::gameboard->teams()) {
 175                         side_actions &actions = *side.get_side_actions();
 176                         if(turn < actions.num_turns() && team_filter(side)) {
 177                                 for(auto iter = actions.turn_begin(turn); iter != actions.turn_end(turn); ++iter) {
 178                                         function(iter->get());
 179                                 }
 180                                 end = false;
 181                         }
 182                 }
 183         }
 184 }

 186 action_ptr find_action_at(map_location hex, team_filter team_filter)
 187 {
 188         action_ptr result;
 189         size_t result_turn = std::numeric_limits<size_t>::max();
 190 
 191         for(team &side : resources::gameboard->teams()) {
 192                 side_actions &actions = *side.get_side_actions();
 193                 if(team_filter(side)) {
 194                         side_actions::iterator chall = actions.find_first_action_at(hex);
 195                         if(chall == actions.end()) {
 196                                 continue;
 197                         }
 198 
 199                         size_t chall_turn = actions.get_turn(chall);
 200                         if(chall_turn < result_turn) {
 201                                 result = *chall;
 202                                 result_turn = chall_turn;
 203                         }
 204                 }
 205         }
 206 
 207         return result;
 208 }
