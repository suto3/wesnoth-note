  38 bool modal_dialog::show(CVideo& video, const unsigned auto_close_time)
  39 {
  40         if(video.faked() && !show_even_without_video_) {
  41                 if(!allow_plugin_skip_) {
  42                         return false;
  43                 }
  44 
  45                 plugins_manager* pm = plugins_manager::get();
  46                 if (pm && pm->any_running())
  47                 {
  48                         plugins_context pc("Dialog");
  49                         pc.set_callback("skip_dialog", [this](config) { retval_ = window::OK; }, false);
  50                         pc.set_callback("quit", [](config) {}, false);
  51                         pc.play_slice();
  52                 }
  53 
  54                 return false;
  55         }
  56 
  57         std::unique_ptr<window> window(build_window(video));
  58         assert(window.get());
  59 
  60         post_build(*window);
  61 
  62         window->set_owner(this);
  63 
  64         init_fields(*window);
  65 
  66         pre_show(*window);
  67 
  68         open_window_stack.push_back(window.get());
  69 
  70         retval_ = window->show(restore_, auto_close_time);
  71 
  72         open_window_stack.pop_back();
  73 
  74         /*
  75          * It can happen that when two clicks follow each other fast that the event
  76          * handling code in events.cpp generates a DOUBLE_CLICK_EVENT. For some
  77          * reason it can happen that this event gets pushed in the queue when the
  78          * window is shown, but processed after the window is closed. This causes
  79          * the next window to get this pending event.
  80          *
  81          * This caused a bug where double clicking in the campaign selection dialog
  82          * directly selected a difficulty level and started the campaign. In order
  83          * to avoid that problem, filter all pending DOUBLE_CLICK_EVENT events after
  84          * the window is closed.
  85          */
  86         SDL_FlushEvent(DOUBLE_CLICK_EVENT);
  87 
  88         finalize_fields(*window, (retval_ == window::OK || always_save_fields_));
  89 
  90         post_show(*window);
  91 
  92         // post_show may have updated the windoe retval. Update it here.
  93         retval_ = window->get_retval();
  94 
  95         return retval_ == window::OK;
  96 }

 218 void modal_dialog::init_fields(window& window)
 219 {
 220         for(auto field : fields_)
 221         {
 222                 field->attach_to_window(window);
 223                 field->widget_init(window);
 224         }
 225 
 226         if(!focus_.empty()) {
 227                 if(widget* widget = window.find(focus_, false)) {
 228                         window.keyboard_capture(widget);
 229                 }
 230         }
 231 }
 232 
 233 void modal_dialog::finalize_fields(window& window, const bool save_fields)
 234 {
 235         for(auto field : fields_)
 236         {
 237                 if(save_fields) {
 238                         field->widget_finalize(window);
 239                 }
 240                 field->detach_from_window();
 241         }
 242 }
