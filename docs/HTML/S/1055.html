  92 void replay_controller::add_replay_theme()

  94         const config &theme_cfg = controller_.get_theme(game_config_manager::get()->game_config(), controller_.theme());

  96         {

 199 void replay_controller::replay_ui_playback_should_stop()
 200 {
 201         if(!replay_ui_has_all_buttons())
 202                 return;
 203 
 204         if(!resources::recorder->at_end()) {
 205                 play_button()->enable(true);
 206                 reset_button()->enable(true);
 207                 play_turn_button()->enable(true);
 208                 play_side_button()->enable(true);
 209                 play_move_button()->enable(true);
 210 
 211                 play_button()->release();
 212                 play_turn_button()->release();
 213                 play_side_button()->release();
 214                 play_move_button()->release();
 215         } else {
 216                 reset_button()->enable(true);
 217                 stop_button()->enable(false);
 218         }
 219 
 220         if(stop_condition_->should_stop()) {
 221                 //user interrupted
 222                 stop_button()->release();
 223         }
 224 }

 292 REPLAY_RETURN replay_controller::play_side_impl()
 293 {
 294         while(!return_to_play_side_ && !static_cast<playsingle_controller&>(controller_).get_player_type_changed())
 295         {
 296                 if(!stop_condition_->should_stop())
 297                 {
 298                         if(resources::recorder->at_end()) {
 299                                 //Gather more replay data
 300                                 on_end_replay_();
 301                         }
 302                         else {
 303                                 REPLAY_RETURN res = do_replay(true);
 304                                 if(res == REPLAY_FOUND_END_MOVE) {
 305                                         stop_condition_->move_done();
 306                                 }
 307                                 if(res == REPLAY_FOUND_END_TURN) {
 308                                         return res;
 309                                 }
 310                                 if(res == REPLAY_RETURN_AT_END) {
 311                                         stop_replay();
 312                                 }
 313                                 if(res == REPLAY_FOUND_INIT_TURN)
 314                                 {
 315                                         stop_condition_->new_side_turn(controller_.current_side(), controller_.gamestate().tod_manager_.turn());
 316                                 }
 317                         }
 318                         controller_.play_slice(false);
 319                 }
 320                 else
 321                 {
 322                         controller_.play_slice(true);
 323                         replay_ui_playback_should_stop();
 324                 }
 325         }
 326         return REPLAY_FOUND_END_MOVE;
 327 }
 328 bool replay_controller::can_execute_command(const hotkey::hotkey_command& cmd, int) const
 329 {
 330         hotkey::HOTKEY_COMMAND command = cmd.id;
 331 
 332         switch(command) {
 333         case hotkey::HOTKEY_REPLAY_SKIP_ANIMATION:
 334                 return true;
 335         case hotkey::HOTKEY_REPLAY_SHOW_EVERYTHING:
 336         case hotkey::HOTKEY_REPLAY_SHOW_EACH:
 337         case hotkey::HOTKEY_REPLAY_SHOW_TEAM1:
 338                 return is_controlling_view();
 339         //commands we only can do before the end of the replay
 340         case hotkey::HOTKEY_REPLAY_STOP:
 341                 return !recorder_at_end();
 342         case hotkey::HOTKEY_REPLAY_PLAY:
 343         case hotkey::HOTKEY_REPLAY_NEXT_TURN:
 344         case hotkey::HOTKEY_REPLAY_NEXT_SIDE:
 345         case hotkey::HOTKEY_REPLAY_NEXT_MOVE:
 346                 //we have one events_disabler when starting the replay_controller and a second when entering the synced context.
 347                 return should_stop() && (events::commands_disabled <= 1 ) && !recorder_at_end();
 348         case hotkey::HOTKEY_REPLAY_RESET:
 349                 return allow_reset_replay() && events::commands_disabled <= 1;
 350         default:
 351                 assert(false);
 352                 return false;
 353         }
 354 }
