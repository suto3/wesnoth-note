  18 function ca_hang_out:evaluation(cfg, data)
  19     -- Return 0 if the mobilize condition has previously been met
  20     if MAISD.get_mai_self_data(data, cfg.ai_id, "mobilize_units") then
  21         return 0
  22     end
  23 
  24     -- Otherwise check if any of the mobilize conditions are now met
  25     local mobilize_condition = H.get_child(cfg, "mobilize_condition")
  26     if (mobilize_condition and wesnoth.eval_conditional(mobilize_condition))
  27         or (cfg.mobilize_on_gold_less_than and (wesnoth.sides[wesnoth.current.side].gold < cfg.mobilize_on_gold_less_than))
  28     then
  29         MAISD.insert_mai_self_data(data, cfg.ai_id, { mobilize_units = true })
  30 
  31         -- Need to unmark all units also (all units, with and without moves)
  32         local units = wesnoth.get_units { side = wesnoth.current.side, { "and", H.get_child(cfg, "filter") } }
  33         for _,unit in ipairs(units) do
  34             MAIUV.delete_mai_unit_variables(unit, cfg.ai_id)
  35         end
  36 
  37         return 0
  38     end
  39 
  40     if get_hang_out_units(cfg)[1] then return cfg.ca_score end
  41     return 0
  42 end

  44 function ca_hang_out:execution(cfg)
  45     local units = get_hang_out_units(cfg)
  46 
  47     -- Get the locations close to which the units should hang out
  48     -- cfg.filter_location defaults to the location of the side leader(s)
  49     local filter_location = H.get_child(cfg, "filter_location") or {
  50         { "filter", { side = wesnoth.current.side, canrecruit = "yes" } }
  51     }
  52 
  53     local width, height = wesnoth.get_map_size()
  54     local locs = wesnoth.get_locations {
  55         x = '1-' .. width,
  56         y = '1-' .. height,
  57         { "and", filter_location }
  58     }
  59 
  60     -- Get map of locations to be avoided.
  61     -- Use [micro_ai][avoid] tag with priority over [ai][avoid].
  62     -- If neither is given, default to all castle terrain.
  63     local avoid_tag = H.get_child(cfg, "avoid")
  64     local avoid_map
  65     if avoid_tag then
  66         avoid_map = LS.of_pairs(wesnoth.get_locations(avoid_tag))
  67     else
  68         local ai_tag = H.get_child(wesnoth.sides[wesnoth.current.side].__cfg, 'ai')
  69         for aspect in H.child_range(ai_tag, 'aspect') do
  70             if (aspect.id == 'avoid') then
  71                 local facet = H.get_child(aspect, 'facet')
  72                 if facet or aspect.name ~= "composite_aspect" then
  73                     -- If there's a [facet] child, it's set as a composite aspect,
  74                     -- with at least one facet.
  75                     -- But it might not be a composite aspect; it could be
  76                     -- a Lua aspect or a standard aspect.
  77                     avoid_map = LS.of_pairs(ai.aspects.avoid)
  78                     break
  79                 end
  80             end
  81         end
  82     end
  83     if avoid_map == nil then
  84         avoid_map = LS.of_pairs(wesnoth.get_locations { terrain = 'C*,C*^*,*^C*' })
  85     end
  86 
  87     local max_rating, best_hex, best_unit = -9e99
  88     for _,unit in ipairs(units) do
  89         -- Only consider units that have not been marked yet
  90         if (not MAIUV.get_mai_unit_variables(unit, cfg.ai_id, "moved")) then
  91             local max_rating_unit, best_hex_unit = -9e99
  92 
  93             -- Check out all unoccupied hexes the unit can reach
  94             local reach_map = AH.get_reachable_unocc(unit)
  95             reach_map:iter( function(x, y, v)
  96                 if (not avoid_map:get(x, y)) then
  97                     for _,loc in ipairs(locs) do
  98                         -- Main rating is the distance from any of the goal hexes
  99                         local rating = -M.distance_between(x, y, loc[1], loc[2])
 100 
 101                         -- Fastest unit moves first
 102                         rating = rating + unit.max_moves / 100.
 103 
 104                         -- Minor penalty for distance from current position of unit
 105                         -- so that there's not too much shuffling around
 106                         local rating = rating - M.distance_between(x, y, unit.x, unit.y) / 1000.
 107 
 108                         if (rating > max_rating_unit) then
 109                             max_rating_unit = rating
 110                             best_hex_unit = { x, y }
 111                         end
 112                     end
 113                 end
 114             end)
 115 
 116             -- Only consider a unit if the best hex found for it is not its current location
 117             if best_hex_unit and ((best_hex_unit[1] ~= unit.x) or (best_hex_unit[2] ~= unit.y)) then
 118                 if (max_rating_unit > max_rating) then
 119                     max_rating = max_rating_unit
 120                     best_hex, best_unit = best_hex_unit, unit
 121                 end
 122             end
 123         end
 124     end
 125 
 126     -- If no valid locations/units were found or all units are in their
 127     -- respective best locations already, we take moves away from all units
 128     if (not best_unit) then
 129         for _,unit in ipairs(units) do
 130             AH.checked_stopunit_moves(ai, unit)
 131 
 132             -- Also remove the markers
 133             if unit and unit.valid then MAIUV.delete_mai_unit_variables(unit, cfg.ai_id) end
 134         end
 135     else
 136         -- Otherwise move unit (partial move only) and mark as having been used
 137         AH.checked_move(ai, best_unit, best_hex[1], best_hex[2])
 138 
 139         if best_unit and best_unit.valid then
 140             MAIUV.set_mai_unit_variables(best_unit, cfg.ai_id, { moved = true })
 141         end
 142     end
 143 end
 144 
 145 return ca_hang_out
