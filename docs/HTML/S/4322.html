  21 function ca_patrol:execution(cfg)

  23     local patrol_vars = MAIUV.get_mai_unit_variables(patrol, cfg.ai_id)
  24 
  25     -- Set up waypoints, taking into account whether 'reverse' is set
  26     -- This works even the first time, when patrol_vars.patrol_reverse is not set yet
  27     cfg.waypoint_x = AH.split(cfg.waypoint_x, ",")
  28     cfg.waypoint_y = AH.split(cfg.waypoint_y, ",")
  29     local n_wp = #cfg.waypoint_x
  30     local waypoints = {}
  31     for i = 1,n_wp do
  32         if patrol_vars.patrol_reverse then
  33             waypoints[i] = { tonumber(cfg.waypoint_x[n_wp-i+1]), tonumber(cfg.waypoint_y[n_wp-i+1]) }
  34         else
  35             waypoints[i] = { tonumber(cfg.waypoint_x[i]), tonumber(cfg.waypoint_y[i]) }
  36         end
  37     end
  38 
  39     -- If not set, set next location (first move)
  40     -- This needs to be in WML format, so that it persists over save/load cycles
  41     if (not patrol_vars.patrol_x) then
  42         patrol_vars.patrol_x = waypoints[1][1]
  43         patrol_vars.patrol_y = waypoints[1][2]
  44         patrol_vars.patrol_reverse = false
  45         MAIUV.set_mai_unit_variables(patrol, cfg.ai_id, patrol_vars)
  46     end
  47 
  48     while patrol.moves > 0 do
  49         -- Check whether one of the enemies to be attacked is next to the patroller
  50         -- If so, don't move, but attack that enemy
  51         local adjacent_enemy = AH.get_attackable_enemies {
  52             id = cfg.attack,
  53             { "filter_adjacent", { id = patrol.id } }
  54         }[1]
  55         if adjacent_enemy then break end
  56 
  57         -- Also check whether we're next to any unit (enemy or ally) which is on the next waypoint
  58         local unit_on_wp = AH.get_visible_units(wesnoth.current.side, {
  59             x = patrol_vars.patrol_x,
  60             y = patrol_vars.patrol_y,
  61             { "filter_adjacent", { id = patrol.id } }
  62         })[1]
  63 
  64         for i,wp in ipairs(waypoints) do

  66             if ((patrol.x == wp[1]) and (patrol.y == wp[2]))
  67                 or (unit_on_wp and ((unit_on_wp.x == wp[1]) and (unit_on_wp.y == wp[2])))
  68             then

  70                     -- Move him to the first one (or reverse route), if he's on the last waypoint
  71                     -- Unless cfg.one_time_only is set
  72                     if cfg.one_time_only then

  74                         patrol_vars.patrol_y = waypoints[n_wp][2]
  75                         MAIUV.set_mai_unit_variables(patrol, cfg.ai_id, patrol_vars)
  76                     else

  78                         if cfg.out_and_back then

  80                             patrol_vars.patrol_y = waypoints[n_wp-1][2]
  81                             -- We also need to reverse the waypoints right here, as this might not be the end of the move
  82                             patrol_vars.patrol_reverse = not patrol_vars.patrol_reverse
  83                             MAIUV.set_mai_unit_variables(patrol, cfg.ai_id, patrol_vars)
  84 
  85                             local tmp_wp = {}
  86                             for j,wp2 in ipairs(waypoints) do tmp_wp[n_wp-j+1] = wp2 end
  87                             waypoints = tmp_wp
  88                         else
  89                             patrol_vars.patrol_x = waypoints[1][1]
  90                             patrol_vars.patrol_y = waypoints[1][2]
  91                             MAIUV.set_mai_unit_variables(patrol, cfg.ai_id, patrol_vars)
  92                         end
  93                     end
  94                 else
  95                     -- ... else move him on toward the next waypoint
  96                     patrol_vars.patrol_x = waypoints[i+1][1]
  97                     patrol_vars.patrol_y = waypoints[i+1][2]
  98                     MAIUV.set_mai_unit_variables(patrol, cfg.ai_id, patrol_vars)
  99                 end

 101         end
 102 
 103         -- If we're on the last waypoint on one_time_only is set, stop here
 104         if cfg.one_time_only and
 105             (patrol.x == waypoints[n_wp][1]) and (patrol.y == waypoints[n_wp][2])
 106         then
 107             AH.checked_stopunit_moves(ai, patrol)
 108         else  -- Otherwise move toward next WP
 109             local x, y = wesnoth.find_vacant_tile(patrol_vars.patrol_x, patrol_vars.patrol_y, patrol)
 110             local nh = AH.next_hop(patrol, x, y)
 111             if nh and ((nh[1] ~= patrol.x) or (nh[2] ~= patrol.y)) then
 112                 AH.checked_move(ai, patrol, nh[1], nh[2])
 113             else
 114                 AH.checked_stopunit_moves(ai, patrol)
 115             end
 116         end
 117         if (not patrol) or (not patrol.valid) then return end
 118     end
 119 
 120     -- Attack unit on the last waypoint under all circumstances if cfg.one_time_only is set
 121     local adjacent_enemy
 122     if cfg.one_time_only then
 123         adjacent_enemy = AH.get_attackable_enemies {
 124             x = waypoints[n_wp][1],
 125             y = waypoints[n_wp][2],
 126             { "filter_adjacent", { id = patrol.id } }
 127         }[1]
 128     end
 129 
 130     -- Otherwise attack adjacent enemy (if specified)
 131     if (not adjacent_enemy) then
 132         adjacent_enemy = AH.get_attackable_enemies {
 133             id = cfg.attack,
 134             { "filter_adjacent", { id = patrol.id } }
 135         }[1]
 136     end
 137 
 138     if adjacent_enemy then AH.checked_attack(ai, patrol, adjacent_enemy) end
 139     if (not patrol) or (not patrol.valid) then return end
