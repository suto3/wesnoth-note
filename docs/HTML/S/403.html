  64 void wml_message_base::pre_show(window& window)
  65 {
  66         set_restore(true);
  67 
  68         window.get_canvas(1).set_variable("portrait_image", wfl::variant(portrait_));
  69         window.get_canvas(1).set_variable("portrait_mirror", wfl::variant(mirror_));
  70 
  71         // Set the markup
  72         label& title = find_widget<label>(&window, "title", false);
  73         title.set_label(title_);
  74         title.set_use_markup(true);
  75         title.set_can_wrap(true);
  76 
  77         styled_widget& message = find_widget<styled_widget>(&window, "message", false);
  78         message.set_label(message_);
  79         message.set_use_markup(true);
  80         // The message label might not always be a scroll_label but the capturing
  81         // shouldn't hurt.
  82         window.keyboard_capture(&message);
  83 
  84         // Find the input box related fields.
  85         label& caption = find_widget<label>(&window, "input_caption", false);
  86         text_box& input = find_widget<text_box>(&window, "input", true);
  87 
  88         if(has_input_) {
  89                 caption.set_label(input_caption_);
  90                 caption.set_use_markup(true);
  91                 input.set_value(*input_text_);
  92                 input.set_maximum_length(input_maximum_length_);
  93                 window.keyboard_capture(&input);
  94                 window.set_click_dismiss(false);
  95                 window.set_escape_disabled(true);
  96         } else {
  97                 caption.set_visible(widget::visibility::invisible);
  98                 input.set_visible(widget::visibility::invisible);
  99         }
 100 
 101         // Find the option list related fields.
 102         listbox& options = find_widget<listbox>(&window, "input_list", true);
 103 
 104         if(!option_list_.empty()) {
 105                 std::map<std::string, string_map> data;
 106                 for(const wml_message_option& item : option_list_) {
 107                         // Add the data.
 108                         data["icon"]["label"] = item.image();
 109                         data["label"]["label"] = item.label();
 110                         data["label"]["use_markup"] = "true";
 111                         data["description"]["label"] = item.description();
 112                         data["description"]["use_markup"] = "true";
 113                         options.add_row(data);
 114                 }
 115 
 116                 // Avoid negative and 0 since item 0 is already selected.
 117                 if(*chosen_option_ > 0 && static_cast<size_t>(*chosen_option_)
 118                                                                   < option_list_.size()) {
 119 
 120                         options.select_row(*chosen_option_);
 121                 }
 122 
 123                 if(!has_input_) {
 124                         window.keyboard_capture(&options);
 125                         window.set_click_dismiss(false);
 126                         window.set_escape_disabled(true);
 127                 } else {
 128                         window.add_to_keyboard_chain(&options);
 129                         // click_dismiss has been disabled due to the input.
 130                 }
 131         } else {
 132                 options.set_visible(widget::visibility::invisible);
 133         }
 134         window.set_click_dismiss(!has_input_ && option_list_.empty());
 135 }

 137 void wml_message_base::post_show(window& window)
 138 {
 139         if(has_input_) {
 140                 *input_text_
 141                                 = find_widget<text_box>(&window, "input", true).get_value();
 142         }
 143 
 144         if(!option_list_.empty()) {
 145                 *chosen_option_ = find_widget<listbox>(&window, "input_list", true)
 146                                                                   .get_selected_row();
 147         }
 148 }

 163 int show_wml_message(CVideo& video,
 164                                          const std::string& title,
 165                                          const std::string& message,
 166                                          const wml_message_portrait* left,
 167                                          const wml_message_portrait* right,
 168                                          const wml_message_options& options,
 169                                          const wml_message_input& input)
 170 {
 171         std::shared_ptr<wml_message_base> dlg;
 172         if(left && !right) {
 173                 dlg.reset(new wml_message_left(title, message, left->portrait, left->mirror));
 174         } else if(!left && right) {
 175                 dlg.reset(new wml_message_right(title, message, right->portrait, right->mirror));
 176         } else {
 177                 dlg.reset(new wml_message_double(title, message, left->portrait, left->mirror, right->portrait, right->mirror));
 178         }
 179         assert(dlg.get());
 180 
 181         if(input.text_input_was_specified) {
 182                 dlg->set_input(input.caption, &input.text, input.maximum_length);
 183         }
 184 
 185         if(!options.option_list.empty()) {
 186                 dlg->set_option_list(options.option_list, &options.chosen_option);
 187         }
 188 
 189         dlg->show(video);
 190         return dlg->get_retval();
 191 }
