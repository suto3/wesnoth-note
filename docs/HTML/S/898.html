  37 /// Calculates the number of blows resulting from swarm.
  38 inline unsigned swarm_blows(unsigned min_blows, unsigned max_blows, unsigned hp,
  39                             unsigned max_hp)
  40 {
  41         return hp >= max_hp          ? max_blows :
  42                max_blows < min_blows ? min_blows - (min_blows-max_blows) * hp / max_hp :
  43                                        min_blows + (max_blows-min_blows) * hp / max_hp;
  44 }
