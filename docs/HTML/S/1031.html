  56 /**
  57  * Detects whether we are running on Wine or not.
  58  *
  59  * This is for informational purposes only and all Windows code should assume
  60  * we are running on the real thing instead.
  61  */
  62 bool on_wine()
  63 {
  64         HMODULE ntdll = GetModuleHandle(L"ntdll.dll");
  65         if(!ntdll) {
  66                 return false;
  67         }
  68 
  69         return GetProcAddress(ntdll, "wine_get_version") != nullptr;
  70 }

  74 /**
  75  * Release policy for POSIX pipe streams opened with popen(3).
  76  */
  77 struct posix_pipe_release_policy
  78 {
  79         void operator()(std::FILE* f) const { if(f != nullptr) { pclose(f); } }
  80 };
  81 
  82 /**
  83  * Scoped POSIX pipe stream.
  84  *
  85  * The stream object type is the same as a regular file stream, but the release
  86  * policy is different, as required by popen(3).
  87  */
  88 typedef std::unique_ptr<std::FILE, posix_pipe_release_policy> scoped_posix_pipe;
  89 
  90 /**
  91  * Read a single line from the specified pipe.
  92  *
  93  * @returns An empty string if the pipe is invalid or nothing could be read.
  94  */
  95 std::string read_pipe_line(scoped_posix_pipe& p)
  96 {
  97         if(!p.get()) {
  98                 return "";
  99         }
 100 
 101         std::string ver;
 102         int c;
 103 
 104         ver.reserve(64);
 105 
 106         // We only want the first line.
 107         while((c = std::fgetc(p.get())) && c != EOF && c != '\n' && c != '\r') {
 108                 ver.push_back(static_cast<char>(c));
 109         }
 110 
 111         return ver;
 112 }
 113 #endif

 117 std::string os_version()
 118 {
 119 #if defined(_X11) || defined(__APPLE__)
 120 
 121 #ifdef __APPLE__
 122 
 123         //
 124         // Standard Mac OSX version
 125         //
 126 
 127         static const std::string version_plist = "/System/Library/CoreServices/SystemVersion.plist";
 128         static const std::string defaults_bin = "/usr/bin/defaults";
 129 
 130         if(filesystem::file_exists(defaults_bin) && filesystem::file_exists(version_plist)) {
 131                 static const std::string cmdline
 132                                 = defaults_bin + " read " + version_plist + " ProductUserVisibleVersion";
 133 
 134                 scoped_posix_pipe p(popen(cmdline.c_str(), "r"));
 135                 const std::string& ver = read_pipe_line(p);
 136 
 137                 if(!ver.empty()) {
 138                         return "Apple OS X " + ver;
 139                 }
 140         }
 141 
 142 #else
 143 
 144         //
 145         // Linux Standard Base version.
 146         //
 147 
 148         static const std::string lsb_release_bin = "/usr/bin/lsb_release";
 149 
 150         if(filesystem::file_exists(lsb_release_bin)) {
 151                 static const std::string cmdline = lsb_release_bin + " -s -d";
 152 
 153                 scoped_posix_pipe p(popen(cmdline.c_str(), "r"));
 154                 std::string ver = read_pipe_line(p);
 155 
 156                 if(ver.length() >= 2 && ver[0] == '"' && ver[ver.length() - 1] == '"') {
 157                         ver.erase(ver.length() - 1, 1);
 158                         ver.erase(0, 1);
 159                 }
 160 
 161                 // Check this again in case we got "" above for some weird reason.
 162                 if(!ver.empty()) {
 163                         return ver;
 164                 }
 165         }
 166 #endif
 167 
 168         //
 169         // POSIX uname version.
 170         //
 171 
 172         utsname u;
 173 
 174         if(uname(&u) != 0) {
 175                 ERR_DU << "os_version: uname error (" << strerror(errno) << ")\n";
 176         }
 177 
 178         return formatter() << u.sysname << ' '
 179                                                 << u.release << ' '
 180                                                 << u.version << ' '
 181                                                 << u.machine;
 182 
 183 #elif defined(_WIN32)
 184 
 185         //
 186         // Windows version.
 187         //
 188 
 189         static const std::string base
 190                         = !on_wine() ? "Microsoft Windows" : "Wine/Microsoft Windows";
 191 
 192         OSVERSIONINFOEX v { sizeof(OSVERSIONINFOEX) };
 193 
 194         if(!GetVersionEx(reinterpret_cast<OSVERSIONINFO*>(&v))) {
 195                 ERR_DU << "os_version: GetVersionEx error ("
 196                            << GetLastError() << ")\n";
 197                 return base;
 198         }
 199 
 200         const DWORD vnum = v.dwMajorVersion * 100 + v.dwMinorVersion;
 201         std::string version;
 202 
 203         switch(vnum)
 204         {
 205                 case 500:
 206                         version = "2000";
 207                         break;
 208                 case 501:
 209                         version = "XP";
 210                         break;
 211                 case 502:
 212                         // This will misidentify XP x64 but who really cares?
 213                         version = "Server 2003";
 214                         break;
 215                 case 600:
 216                         if(v.wProductType == VER_NT_WORKSTATION) {
 217                                 version = "Vista";
 218                         } else {
 219                                 version = "Server 2008";
 220                         }
 221                         break;
 222                 case 601:
 223                         if(v.wProductType == VER_NT_WORKSTATION) {
 224                                 version = "7";
 225                         } else {
 226                                 version = "Server 2008 R2";
 227                         }
 228                         break;
 229                 case 602:
 230                         if(v.wProductType == VER_NT_WORKSTATION) {
 231                                 version = "8";
 232                         } else {
 233                                 version = "Server 2012";
 234                         }
 235                         break;
 236                 case 603:
 237                         if(v.wProductType == VER_NT_WORKSTATION) {
 238                                 version = "8.1";
 239                         } else {
 240                                 version = "Server 2012 R2";
 241                         }
 242                         break;
 243                 case 1000:
 244                         if(v.wProductType == VER_NT_WORKSTATION) {
 245                                 version = "10";
 246                                 break;
 247                         } // else fallback to default
 248                 default:
 249                         if(v.wProductType != VER_NT_WORKSTATION) {
 250                                 version = "Server";
 251                         }
 252         }
 253 
 254         if(v.szCSDVersion && *v.szCSDVersion) {
 255                 version += " ";
 256                 version += unicode_cast<std::string>(std::wstring(v.szCSDVersion));
 257         }
 258 
 259         version += " (";
 260         // Add internal version numbers.
 261         version += formatter()
 262                         << v.dwMajorVersion << '.'
 263                         << v.dwMinorVersion << '.'
 264                         << v.dwBuildNumber;
 265         version += ")";
 266 
 267         return base + " " + version;
 268 
 269 #else
 270 
 271         //
 272         // "I don't know where I am" version.
 273         //
 274 
 275         ERR_DU << "os_version(): unsupported platform\n";
 276         return _("operating_system^<unknown>");
 277 
 278 #endif
 279 }
