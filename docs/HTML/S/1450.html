  90     /**
  91      * Checks if the tag name exists in the schema.
  92      * 
  93      * @param tag
  94      *        The {@link WMLTag} to check
  95      */
  96     @Check( CheckType.NORMAL )
  97     public void checkNormalTagName( WMLTag tag )
  98     {
  99         if( ! isValidationEnabled( ) ) {
 100             return;
 101         }
 102 
 103         ICompositeNode node = NodeModelUtils.getNode( tag );
 104         if( node != null ) {
 105             ILeafNode parentNode = NodeModelUtils.findLeafNodeAtOffset(
 106                 node.getParent( ), node.getParent( ).getOffset( ) + 2 );
 107 
 108             boolean found = false;
 109             String searchName = parentNode.getText( );
 110             if( node.getParent( ) == null ) // root node
 111             {
 112                 searchName = "root"; //$NON-NLS-1$
 113             }
 114 
 115             WMLTag schemaTag = getSchema( tag ).getTags( ).get( searchName );
 116 
 117             if( schemaTag != null ) {
 118                 for( WMLExpression expression: schemaTag.getExpressions( ) ) {
 119                     if( expression.getName( ).equals( tag.getName( ) ) ) {
 120                         found = true;
 121                         break;
 122                     }
 123                 }
 124                 if( found == false ) {
 125                     warning( Messages.WMLJavaValidator_1,
 126                         WmlPackage.Literals.WML_EXPRESSION__NAME );
 127                 }
 128             }
 129         }
 130     }

 132     private void checkTagsCardinalities( SchemaParser schema,
 133         Iterable< WMLTag > tags )
 134     {
 135         Map< String, Integer > ocurrences = new HashMap< String, Integer >( );
 136 
 137         for( WMLTag tag: tags ) {
 138             Integer currentValue = ocurrences.get( tag.getName( ) );
 139             if( currentValue == null ) {
 140                 currentValue = 0;
 141             }
 142 
 143             ocurrences.put( tag.getName( ), currentValue + 1 );
 144         }
 145 
 146         for( Entry< String, Integer > entry: ocurrences.entrySet( ) ) {
 147             WMLTag schemaTag = schema.getTags( ).get( entry.getKey( ) );
 148 
 149             if( schemaTag == null ) {
 150                 continue;
 151             }
 152 
 153             if( schemaTag.getAllowedCount( ) < entry.getValue( ) ) {
 154                 warning( "Tag " + entry.getKey( ) + " cannot appear more"
 155                     + "than " + schemaTag.getAllowedCount( ) + " times. ",
 156                     WmlPackage.Literals.WML_EXPRESSION__NAME );
 157             }
 158         }
 159     }

 161     private void checkKeysCardinalities( WMLTag parentTag,
 162         Iterable< WMLKey > keys )
 163     {
 164         Map< String, Integer > ocurrences = new HashMap< String, Integer >( );
 165 
 166         for( WMLKey key: keys ) {
 167             Integer currentValue = ocurrences.get( key.getName( ) );
 168             if( currentValue == null ) {
 169                 currentValue = 0;
 170             }
 171 
 172             ocurrences.put( key.getName( ), currentValue + 1 );
 173         }
 174 
 175         for( Entry< String, Integer > entry: ocurrences.entrySet( ) ) {
 176             WMLKey schemaKey = WMLUtils
 177                 .getKeyByName( parentTag, entry.getKey( ) );
 178 
 179             if( schemaKey == null ) {
 180                 continue;
 181             }
 182 
 183             if( schemaKey.getAllowedCount( ) < entry.getValue( ) ) {
 184                 warning(
 185                     "Key " + entry.getKey( ) + ", in tag "
 186                         + parentTag.getName( )
 187                         + "cannot appear more than "
 188                         + schemaKey.getAllowedCount( ) + " times. ",
 189                     WmlPackage.Literals.WML_EXPRESSION__NAME );
 190             }
 191         }
 192     }
