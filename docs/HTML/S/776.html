 124 {
 125         const attack_type_callable* att_callable = dynamic_cast<const attack_type_callable*>(callable);
 126         if(att_callable == nullptr) {
 127                 return formula_callable::do_compare(callable);
 128         }
 129 
 130         if(att_->damage() != att_callable->att_->damage()) {
 131                 return att_->damage() - att_callable->att_->damage();
 132         }
 133 
 134         if(att_->num_attacks() != att_callable->att_->num_attacks()) {
 135                 return att_->num_attacks() - att_callable->att_->num_attacks();

 137

 401 struct fai_variant_visitor : public boost::static_visitor<variant>
 402 {
 403         variant operator()(bool b) const               { return variant(b ? 1 : 0); }
 404         variant operator()(int i) const                { return variant(i); }
 405         variant operator()(unsigned long long i) const { return variant(i); }
 406         variant operator()(double i) const             { return variant(i * 1000, variant::DECIMAL_VARIANT); }
 407         // TODO: Should comma-separated lists of stuff be returned as a list?
 408         // The challenge is to distinguish them from ordinary strings that happen to contain a comma
 409         // (or should we assume that such strings will be translatable?).
 410         variant operator()(const std::string& s) const { return variant(s); }
 411         variant operator()(const t_string& s) const    { return variant(s.str()); }
 412         variant operator()(boost::blank) const         { return variant(); }
 413 };

 415 variant config_callable::get_value(const std::string& key) const
 416 {
 417         if(cfg_.has_attribute(key)) {
 418                 return cfg_[key].apply_visitor(fai_variant_visitor());
 419         } else if(cfg_.has_child(key)) {
 420                 std::vector<variant> result;
 421                 for(const auto& child : cfg_.child_range(key)) {
 422                         result.emplace_back(std::make_shared<config_callable>(child));
 423                 }
 424 
 425                 return variant(result);
 426         } else if(key == "__all_children") {
 427                 std::vector<variant> result;
 428                 for(const auto& child : cfg_.all_children_range()) {
 429                         const variant cfg_child(std::make_shared<config_callable>(child.cfg));
 430                         const variant kv(std::make_shared<key_value_pair>(variant(child.key), cfg_child));
 431                         result.push_back(kv);
 432                 }
 433 
 434                 return variant(result);
 435         } else if(key == "__children") {
 436                 std::map<std::string, std::vector<variant> > build;
 437                 for(const auto& child : cfg_.all_children_range()) {
 438                         const variant cfg_child(std::make_shared<config_callable>(child.cfg));
 439                         build[child.key].push_back(cfg_child);
 440                 }
 441 
 442                 std::map<variant,variant> result;
 443                 for(auto& p : build) {
 444                         result[variant(p.first)] = variant(p.second);
 445                 }
 446 
 447                 return variant(result);
 448         } else if(key == "__attributes") {
 449                 std::map<variant,variant> result;
 450                 for(const auto& val : cfg_.attribute_range()) {
 451                         result[variant(val.first)] = val.second.apply_visitor(fai_variant_visitor());
 452                 }
 453 
 454                 return variant(result);
 455         }
 456 
 457         return variant();
 458 }

 725 variant safe_call_callable::execute_self(variant ctxt)
 726 {
 727         variant res;
 728         if(auto action = main_.try_convert<action_callable>()) {
 729                 res = action->execute_self(ctxt);
 730         }
 731 
 732         if(res.try_convert<safe_call_result>()) {
 733                 /* If we have safe_call formula and either an error occurred, or the current action
 734                  * was not recognized, then evaluate backup formula from safe_call and execute it
 735                  * during the next loop
 736                  */
 737 
 738                 map_formula_callable callable(ctxt.as_callable());
 739                 callable.add("error", res);
 740 
 741                 /* Store the result in safe_call_callable in case we would like to display it to the user,
 742                  * for example if this formula was executed from the commandline.
 743                  */
 744                 backup_ = get_backup()->evaluate(callable);
 745                 ctxt.execute_variant(backup_);
 746         }
 747         return variant(true);
 748 }
