  60 static void verify(const unit_map& units, const config& cfg) {
  61         std::stringstream errbuf;
  62         LOG_REPLAY << "verifying unit structure...\n";
  63 
  64         const size_t nunits = cfg["num_units"].to_size_t();
  65         if(nunits != units.size()) {
  66                 errbuf << "SYNC VERIFICATION FAILED: number of units from data source differ: "
  67                            << nunits << " according to data source. " << units.size() << " locally\n";
  68 
  69                 std::set<map_location> locs;
  70                 for (const config &u : cfg.child_range("unit"))
  71                 {
  72                         const map_location loc(u);
  73                         locs.insert(loc);
  74 
  75                         if(units.count(loc) == 0) {
  76                                 errbuf << "data source says there is a unit at "
  77                                            << loc << " but none found locally\n";
  78                         }
  79                 }
  80 
  81                 for(unit_map::const_iterator j = units.begin(); j != units.end(); ++j) {
  82                         if (locs.count(j->get_location()) == 0) {
  83                                 errbuf << "local unit at " << j->get_location()
  84                                            << " but none in data source\n";
  85                         }
  86                 }
  87                 replay::process_error(errbuf.str());
  88                 errbuf.clear();
  89         }
  90 
  91         for (const config &un : cfg.child_range("unit"))
  92         {
  93                 const map_location loc(un);
  94                 const unit_map::const_iterator u = units.find(loc);
  95                 if(u == units.end()) {
  96                         errbuf << "SYNC VERIFICATION FAILED: data source says there is a '"
  97                                    << un["type"] << "' (side " << un["side"] << ") at "
  98                                    << loc << " but there is no local record of it\n";
  99                         replay::process_error(errbuf.str());
 100                         errbuf.clear();
 101                 }
 102 
 103                 config u_cfg;
 104                 u->write(u_cfg);
 105 
 106                 bool is_ok = true;
 107                 static const std::string fields[] {"type","hitpoints","experience","side",""};
 108                 for(const std::string* str = fields; str->empty() == false; ++str) {
 109                         if (u_cfg[*str] != un[*str]) {
 110                                 errbuf << "ERROR IN FIELD '" << *str << "' for unit at "
 111                                            << loc << " data source: '" << un[*str]
 112                                            << "' local: '" << u_cfg[*str] << "'\n";
 113                                 is_ok = false;
 114                         }
 115                 }
 116 
 117                 if(!is_ok) {
 118                         errbuf << "(SYNC VERIFICATION FAILED)\n";
 119                         replay::process_error(errbuf.str());
 120                         errbuf.clear();
 121                 }
 122         }
 123 
 124         LOG_REPLAY << "verification passed\n";
 125 }

 127 static time_t get_time(const config &speak)
 128 {
 129         time_t time;
 130         if (!speak["time"].empty())
 131         {
 132                 std::stringstream ss(speak["time"].str());
 133                 ss >> time;
 134         }
 135         else
 136         {
 137                 //fallback in case sender uses wesnoth that doesn't send timestamps
 138                 time = ::time(nullptr);
 139         }
 140         return time;
 141 }

 388 config replay::get_data_range(int cmd_start, int cmd_end, DATA_TYPE data_type) const
 389 {
 390         config res;
 391

 393         {
 394                 config &c = command(cmd);
 395                 //prevent creating 'blank' attribute values during checks
 396                 const config &cc = c;
 397                 if ((data_type == ALL_DATA || !cc["undo"].to_bool(true)) && !cc["sent"].to_bool(false))
 398                 {
 399                         res.add_child("command", c);
 400                         if (data_type == NON_UNDO_DATA) c["sent"] = true;

 428 config& replay::get_last_real_command()
 429 {
 430         for (int cmd_num = base_->get_pos() - 1; cmd_num >= 0; --cmd_num)
 431         {
 432                 config &c = command(cmd_num);
 433                 const config &cc = c;

 444 /// fixes a rename command when undoing a earlier command.
 445 /// @return: true if the command should be removed.
 446 static bool fix_rename_command(const config& c, config& async_child)
 447 {
 448         if (const config &child = c.child("move"))
 449         {
 450                 // A unit's move is being undone.
 451                 // Repair unsynced cmds whose locations depend on that unit's location.
 452                 std::vector<map_location> steps;
 453 
 454                 try {
 455                         read_locations(child,steps);
 456                 } catch (bad_lexical_cast &) {
 457                         WRN_REPLAY << "Warning: Path data contained something which could not be parsed to a sequence of locations:" << "\n config = " << child.debug() << std::endl;
 458                 }
 459 
 460                 if (steps.empty()) {
 461                         ERR_REPLAY << "trying to undo a move using an empty path";
 462                 }
 463                 else {
 464                         const map_location &src = steps.front();
 465                         const map_location &dst = steps.back();
 466                         map_location aloc(async_child);
 467                         if (dst == aloc) src.write(async_child);
 468                 }
 469         }
 470         else
 471         {
 472                 const config *chld = &c.child("recruit");
 473                 if (!*chld) chld = &c.child("recall");
 474                 if (*chld) {
 475                         // A unit is being un-recruited or un-recalled.
 476                         // Remove unsynced commands that would act on that unit.
 477                         map_location src(*chld);
 478                         map_location aloc(async_child);
 479                         if (src == aloc) {
 480                                 return true;
 481                         }
 482                 }
 483         }
 484         return false;
 485 }

 487 void replay::undo_cut(config& dst)
 488 {
 489         assert(dst.empty());
 490         //assert that we are not undoing a command which we didn't execute yet.
 491         assert(at_end());
 492 
 493         //calculate the index of the last synced user action (which we want to undo).
 494         int cmd_index = ncommands() - 1;
 495         for (; cmd_index >= 0; --cmd_index)
 496         {
 497                 //"undo"=no means speak/label/remove_label, especialy attack, recruits etc. have "undo"=yes
 498                 //"async"=yes means rename_unit
 499                 //"dependent"=true means user input
 500                 const config &c = command(cmd_index);
 501 
 502                 if(c["undo"].to_bool(true) && !c["async"].to_bool(false) && !c["dependent"].to_bool(false))
 503                 {
 504                         if(c["sent"].to_bool(false))
 505                         {
 506                                 ERR_REPLAY << "trying to undo a command that was already sent.\n";
 507                                 return;
 508                         }
 509                         else
 510                         {
 511                                 break;
 512                         }
 513                 }
 514         }
 515 
 516         if (cmd_index < 0)
 517         {
 518                 ERR_REPLAY << "trying to undo a command but no command was found.\n";
 519                 return;
 520         }
 521         //Fix the [command]s after the undone action. This includes dependent commands for that user actions and async user action.
 522         for(int i = ncommands() - 1; i >= cmd_index; --i)
 523         {
 524                 config &c = command(i);
 525                 const config &cc = c;
 526                 if(!cc["undo"].to_bool(true))
 527                 {
 528                         //Leave these commands on the replay.
 529                 }
 530                 else if(cc["async"].to_bool(false))
 531                 {
 532                         if(config& rename = c.child("rename"))
 533                         {
 534                                 if(fix_rename_command(command(cmd_index), rename))
 535                                 {
 536                                         //remove the command from the replay if fix_rename_command requested it.
 537                                         remove_command(i);
 538                                 }
 539                         }
 540                 }
 541                 else if(cc["dependent"].to_bool(false) || i == cmd_index)
 542                 {
 543                         //we loop backwars so we must insert new insert at beginning to preserve order.
 544                         dst.add_child_at("command", config(), 0).swap(c);
 545                         remove_command(i);
 546                 }
 547                 else
 548                 {
 549                         ERR_REPLAY << "Coudn't handle command:\n" << cc << "\nwhen undoing.\n";
 550                 }
 551         }
 552         set_to_end();
 553 }

 664 REPLAY_RETURN do_replay(bool one_move)
 665 {
 666         log_scope("do replay");
 667 
 668         if (!resources::controller->is_skipping_replay()) {
 669                 resources::screen->recalculate_minimap();
 670         }
 671 
 672         update_locker lock_update(resources::screen->video(), resources::controller->is_skipping_replay());
 673         return do_replay_handle(one_move);
 674 }
 675 /**
 676         @returns:
 677                 if we expect a user choice and found something that prevents us from moving on we return REPLAY_FOUND_DEPENDENT (even if it is not a dependent command)
 678                 else if we found an [end_turn] we return REPLAY_FOUND_END_TURN
 679                 else if we found a player action and one_move=true we return REPLAY_FOUND_END_MOVE
 680                 else (<=> we reached the end of the replay) we return REPLAY_RETURN_AT_END
 681 */
 682 REPLAY_RETURN do_replay_handle(bool one_move)
 683 {
 684 
 685         //team &current_team = resources::gameboard->get_team(side_num);
 686 
 687         const int side_num = resources::controller->current_side();
 688         while(true)
 689         {
 690                 const config *cfg = resources::recorder->get_next_action();
 691                 const bool is_synced = synced_context::is_synced();
 692 
 693                 DBG_REPLAY << "in do replay with is_synced=" << is_synced << "\n";
 694 
 695                 if (cfg != nullptr)
 696                 {
 697                         DBG_REPLAY << "Replay data:\n" << *cfg << "\n";
 698                 }
 699                 else
 700                 {
 701                         DBG_REPLAY << "Replay data at end\n";
 702                         return REPLAY_RETURN_AT_END;
 703                 }
 704 
 705 
 706                 const config::const_all_children_itors ch_itors = cfg->all_children_range();
 707                 //if there is an empty command tag or a start tag
 708                 if (ch_itors.empty() || cfg->has_child("start"))
 709                 {
 710                         //this shouldn't happen anymore because replaycontroller now moves over the [start] with get_next_action
 711                         //also we removed the the "add empty replay entry at scenario reload" behavior.
 712                         ERR_REPLAY << "found "<<  cfg->debug() <<" in replay" << std::endl;
 713                         //do nothing
 714                 }
 715                 else if (const config &speak = cfg->child("speak"))
 716                 {
 717                         const std::string &team_name = speak["team_name"];
 718                         const std::string &speaker_name = speak["id"];
 719                         const std::string &message = speak["message"];
 720                         //if (!preferences::parse_should_show_lobby_join(speaker_name, message)) return;
 721                         bool is_whisper = (speaker_name.find("whisper: ") == 0);
 722                         if(resources::recorder->add_chat_message_location()) {
 723                                 DBG_REPLAY << "tried to add a chat message twice.\n";
 724                                 if (!resources::controller->is_skipping_replay() || is_whisper) {
 725                                         int side = speak["side"];
 726                                         resources::screen->get_chat_manager().add_chat_message(get_time(speak), speaker_name, side, message,
 727                                                 (team_name.empty() ? events::chat_handler::MESSAGE_PUBLIC
 728                                                 : events::chat_handler::MESSAGE_PRIVATE),
 729                                                 preferences::message_bell());
 730                                 }
 731                         }
 732                 }
 733                 else if (const config &label_config = cfg->child("label"))
 734                 {
 735                         terrain_label label(resources::screen->labels(), label_config);
 736 
 737                         resources::screen->labels().set_label(label.location(),
 738                                                 label.text(),
 739                                                 label.creator(),
 740                                                 label.team_name(),
 741                                                 label.color());
 742                 }
 743                 else if (const config &clear_labels = cfg->child("clear_labels"))
 744                 {
 745                         resources::screen->labels().clear(std::string(clear_labels["team_name"]), clear_labels["force"].to_bool());
 746                 }
 747                 else if (const config &rename = cfg->child("rename"))
 748                 {
 749                         const map_location loc(rename);
 750                         const std::string &name = rename["name"];
 751 
 752                         unit_map::iterator u = resources::gameboard->units().find(loc);
 753                         if (u.valid() && !u->unrenamable()) {
 754                                 u->rename(name);
 755                         } else {
 756                                 // Users can rename units while it's being killed or at another machine.
 757                                 // This since the player can rename units when it's not his/her turn.
 758                                 // There's not a simple way to prevent that so in that case ignore the
 759                                 // rename instead of throwing an OOS.
 760                                 // The same way it is possible that an unrenamable unit moves to a
 761                                 // hex where previously a renamable unit was.
 762                                 WRN_REPLAY << "attempt to rename unit at location: "
 763                                    << loc << (u.valid() ? ", which is unrenamable" : ", where none exists (anymore)") << "\n";
 764                         }
 765                 }
 766 
 767                 else if (cfg->child("init_side"))
 768                 {
 769 
 770                         if(is_synced)
 771                         {
 772                                 replay::process_error("found side initialization in replay expecting a user choice\n" );
 773                                 resources::recorder->revert_action();
 774                                 return REPLAY_FOUND_DEPENDENT;
 775                         }
 776                         else
 777                         {
 778                                 resources::controller->do_init_side();
 779                                 if (one_move) {
 780                                         return REPLAY_FOUND_INIT_TURN;
 781                                 }
 782                         }
 783                 }
 784 
 785                 //if there is an end turn directive
 786                 else if (cfg->child("end_turn"))
 787                 {
 788                         if(is_synced)
 789                         {
 790                                 replay::process_error("found turn end in replay while expecting a user choice\n" );
 791                                 resources::recorder->revert_action();
 792                                 return REPLAY_FOUND_DEPENDENT;
 793                         }
 794                         else
 795                         {
 796                                 if (const config &cfg_verify = cfg->child("verify")) {
 797                                         verify(resources::gameboard->units(), cfg_verify);
 798                                 }
 799 
 800                                 return REPLAY_FOUND_END_TURN;
 801                         }
 802                 }
 803                 else if (const config &countdown_update = cfg->child("countdown_update"))
 804                 {
 805                         int val = countdown_update["value"];
 806                         int tval = countdown_update["team"];
 807                         if (tval <= 0  || tval > int(resources::gameboard->teams().size())) {
 808                                 std::stringstream errbuf;
 809                                 errbuf << "Illegal countdown update \n"
 810                                         << "Received update for :" << tval << " Current user :"
 811                                         << side_num << "\n" << " Updated value :" << val;
 812 
 813                                 replay::process_error(errbuf.str());
 814                         } else {
 815                                 resources::gameboard->get_team(tval).set_countdown_time(val);
 816                         }
 817                 }
 818                 else if ((*cfg)["dependent"].to_bool(false))
 819                 {
 820                         if(!is_synced)
 821                         {
 822                                 replay::process_error("found dependent command in replay while is_synced=false\n" );
 823                                 //ignore this command
 824                                 continue;
 825                         }
 826                         //this means user choice.
 827                         // it never makes sense to try to execute a user choice.
 828                         // but we are called from
 829                         // the only other option for "dependent" command is checksum wich is already checked.
 830                         assert(cfg->all_children_count() == 1);
 831                         std::string child_name = cfg->all_children_range().front().key;
 832                         DBG_REPLAY << "got an dependent action name = " << child_name <<"\n";
 833                         resources::recorder->revert_action();
 834                         return REPLAY_FOUND_DEPENDENT;
 835                 }
 836                 else
 837                 {
 838                         //we checked for empty commands at the beginning.
 839                         const std::string & commandname = cfg->ordered_begin()->key;
 840                         config data = cfg->ordered_begin()->cfg;
 841 
 842                         if(is_synced)
 843                         {
 844                                 replay::process_error("found [" + commandname + "] command in replay expecting a user choice\n" );
 845                                 resources::recorder->revert_action();
 846                                 return REPLAY_FOUND_DEPENDENT;
 847                         }
 848                         else
 849                         {
 850                                 LOG_REPLAY << "found commandname " << commandname << "in replay";
 851 
 852                                 if((*cfg)["from_side"].to_int(0) != resources::controller->current_side()) {
 853                                         ERR_REPLAY << "recieved a synced [command] from side " << (*cfg)["from_side"].to_int(0) << ". Expacted was a [command] from side " << resources::controller->current_side() << "\n";
 854                                 }
 855                                 else if((*cfg)["side_invalid"].to_bool(false)) {
 856                                         ERR_REPLAY << "recieved a synced [command] from side " << (*cfg)["from_side"].to_int(0) << ". Sent from wrong client.\n";
 857                                 }
 858                                 /*
 859                                         we need to use the undo stack during replays in order to make delayed shroud updated work.
 860                                 */
 861                                 synced_context::run(commandname, data, true, !resources::controller->is_skipping_replay(), show_oos_error_error_function);
 862                                 if(resources::controller->is_regular_game_end()) {
 863                                         return REPLAY_FOUND_END_LEVEL;
 864                                 }
 865                                 if (one_move) {
 866                                         return REPLAY_FOUND_END_MOVE;
 867                                 }
 868                         }
 869                 }
 870 
 871                 if (const config &child = cfg->child("verify")) {
 872                         verify(resources::gameboard->units(), child);
 873                 }
 874         }
 875 }
