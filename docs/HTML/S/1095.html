 767 namespace detail {
 768         template<typename... T>
 769         struct config_construct_unpacker;
 770 
 771         template<>
 772         struct config_construct_unpacker<>
 773         {
 774                 void visit(config&) {}
 775         };
 776 
 777         template<typename K, typename V, typename... Rest>
 778         struct config_construct_unpacker<K, V, Rest...>
 779         {
 780                 template<typename K2 = K, typename V2 = V>
 781                 void visit(config& cfg, K2&& key, V2&& val, Rest... fwd)
 782                 {
 783                         cfg.insert(std::forward<K>(key), std::forward<V>(val));
 784                         config_construct_unpacker<Rest...> unpack;
 785                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 786                 }
 787         };
 788 
 789         template<typename T, typename... Rest>
 790         struct config_construct_unpacker<T, config, Rest...>
 791         {
 792                 template<typename T2 = T, typename C = config>
 793                 void visit(config& cfg, T2&& tag, C&& child, Rest... fwd)
 794                 {
 795                         cfg.add_child(std::forward<T>(tag), std::forward<config>(child));
 796                         config_construct_unpacker<Rest...> unpack;
 797                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 798                 }
 799         };
 800 
 801         template<typename T, typename... Rest>
 802         struct config_construct_unpacker<T, config&, Rest...>
 803         {
 804                 template<typename T2 = T>
 805                 void visit(config& cfg, T2&& tag, config& child, Rest... fwd)
 806                 {
 807                         cfg.add_child(std::forward<T>(tag), std::forward<config>(child));
 808                         config_construct_unpacker<Rest...> unpack;
 809                         unpack.visit(cfg, std::forward<Rest>(fwd)...);
 810                 }
 811         };
 812 }
 813 
 814 template<typename... T>
 815 inline config::config(config_key_type first, T&&... args)
 816 {
 817         detail::config_construct_unpacker<config_key_type, T...> unpack;
 818         unpack.visit(*this, first, std::forward<T>(args)...);
 819 }
