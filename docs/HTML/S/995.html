 164 void unit_type::build_full(const movement_type_map &mv_types,
 165         const race_map &races, const config::const_child_itors &traits)
 166 {
 167         // Don't build twice.
 168         if ( FULL <= build_status_ )
 169                 return;
 170         // Make sure we are built to the preceding build level.
 171         build_help_index(mv_types, races, traits);
 172 
 173         for (int i = 0; i < 2; ++i) {
 174                 if (gender_types_[i]) {
 175                         gender_types_[i]->build_full(mv_types, races, traits);
 176                 }
 177         }
 178 
 179         if ( race_ != &unit_race::null_race )
 180         {
 181                 if (undead_variation_.empty()) {
 182                         undead_variation_ = race_->undead_variation();
 183                 }
 184         }
 185 
 186         zoc_ = cfg_["zoc"].to_bool(level_ > 0);
 187 
 188         game_config::add_color_info(cfg_);
 189 
 190         hp_bar_scaling_ = cfg_["hp_bar_scaling"].to_double(game_config::hp_bar_scaling);
 191         xp_bar_scaling_ = cfg_["xp_bar_scaling"].to_double(game_config::xp_bar_scaling);
 192 
 193         // Propagate the build to the variations.
 194         for (variations_map::value_type & variation : variations_) {
 195                 variation.second.build_full(mv_types, races, traits);
 196         }
 197 
 198         // Deprecation messages, only seen when unit is parsed for the first time.
 199 
 200         build_status_ = FULL;
 201 }

 204  * Partially load data into an empty unit_type (build to HELP_INDEXED).

 206 void unit_type::build_help_index(const movement_type_map &mv_types,
 207         const race_map &races, const config::const_child_itors &traits)
 208 {
 209         // Don't build twice.
 210         if ( HELP_INDEXED <= build_status_ )
 211                 return;
 212         // Make sure we are built to the preceding build level.
 213         build_created(mv_types, races, traits);
 214 
 215         type_name_ = cfg_["name"];
 216         description_ = cfg_["description"];
 217         hitpoints_ = cfg_["hitpoints"].to_int(1);
 218         level_ = cfg_["level"];
 219         recall_cost_ = cfg_["recall_cost"].to_int(-1);
 220         movement_ = cfg_["movement"].to_int(1);
 221         vision_ = cfg_["vision"].to_int(-1);
 222         jamming_ = cfg_["jamming"].to_int(0);
 223         max_attacks_ = cfg_["attacks"].to_int(1);
 224         cost_ = cfg_["cost"].to_int(1);
 225         usage_ = cfg_["usage"].str();
 226         undead_variation_ = cfg_["undead_variation"].str();
 227         image_ = cfg_["image"].str();
 228         icon_ = cfg_["image_icon"].str();
 229         small_profile_ = cfg_["small_profile"].str();
 230         profile_ = cfg_["profile"].str();
 231         adjust_profile(profile_);
 232 
 233         alignment_ = unit_type::ALIGNMENT::NEUTRAL;
 234         alignment_.parse(cfg_["alignment"].str());
 235 
 236         for (int i = 0; i < 2; ++i) {
 237                 if (gender_types_[i])
 238                         gender_types_[i]->build_help_index(mv_types, races, traits);
 239         }
 240 
 241         const race_map::const_iterator race_it = races.find(cfg_["race"]);
 242         if(race_it != races.end()) {
 243                 race_ = &race_it->second;
 244         } else {
 245                 race_ = &unit_race::null_race;
 246         }
 247 
 248         // if num_traits is not defined, we use the num_traits from race
 249         num_traits_ = cfg_["num_traits"].to_int(race_->num_traits());
 250 
 251         const std::vector<std::string> genders = utils::split(cfg_["gender"]);
 252         for(std::vector<std::string>::const_iterator g = genders.begin(); g != genders.end(); ++g) {
 253                 genders_.push_back(string_gender(*g));
 254         }
 255         // For simplicity in other parts of the code, we must have at least one gender.
 256         if(genders_.empty()) {
 257                 genders_.push_back(unit_race::MALE);
 258         }
 259 
 260         if (const config &abil_cfg = cfg_.child("abilities"))
 261         {
 262                 for (const config::any_child &ab : abil_cfg.all_children_range()) {
 263                         const config::attribute_value &name = ab.cfg["name"];
 264                         if (!name.empty()) {
 265                                 abilities_.push_back(name.t_str());
 266                                 ability_tooltips_.push_back( ab.cfg["description"].t_str() );
 267                         }
 268                 }
 269         }
 270 
 271         for (const config &adv : cfg_.child_range("advancement"))
 272         {
 273                 for (const config &effect : adv.child_range("effect"))
 274                 {
 275                         const config &abil_cfg = effect.child("abilities");
 276                         if (!abil_cfg || effect["apply_to"] != "new_ability") {
 277                                 continue;
 278                         }
 279                         for (const config::any_child &ab : abil_cfg.all_children_range()) {
 280                                 const config::attribute_value &name = ab.cfg["name"];
 281                                 if (!name.empty()) {
 282                                         adv_abilities_.push_back(name.t_str());
 283                                         adv_ability_tooltips_.push_back( ab.cfg["description"].t_str() );
 284                                 }
 285                         }
 286                 }
 287         }
 288 
 289         // Set the movement type.
 290         const std::string move_type = cfg_["movement_type"];
 291         const movement_type_map::const_iterator find_it = mv_types.find(move_type);
 292         if ( find_it != mv_types.end() ) {
 293                 DBG_UT << "inheriting from movement_type '" << move_type << "'\n";
 294                 movement_type_ = find_it->second;
 295         }
 296         else if ( !move_type.empty() ) {
 297                 DBG_UT << "movement_type '" << move_type << "' not found\n";
 298         }
 299         // Override parts of the movement type with what is in our config.
 300         movement_type_.merge(cfg_);
 301 
 302         for (const config &t : traits)
 303         {
 304                 possible_traits_.add_child("trait", t);
 305         }
 306         if ( race_ != &unit_race::null_race )
 307         {
 308                 if (!race_->uses_global_traits()) {
 309                         possible_traits_.clear();
 310                 }
 311                 if ( cfg_["ignore_race_traits"].to_bool() ) {
 312                         possible_traits_.clear();
 313                 } else {
 314                         for (const config &t : race_->additional_traits())
 315                         {
 316                                 if (alignment_ != unit_type::ALIGNMENT::NEUTRAL || t["id"] != "fearless")
 317                                         possible_traits_.add_child("trait", t);
 318                         }
 319                 }
 320                 if (undead_variation_.empty()) {
 321                         undead_variation_ = race_->undead_variation();
 322                 }
 323         }
 324         // Insert any traits that are just for this unit type
 325         for (const config &trait : cfg_.child_range("trait"))
 326         {
 327                 possible_traits_.add_child("trait", trait);
 328         }
 329 
 330         for (const config &var_cfg : cfg_.child_range("variation"))
 331         {
 332                 const std::string& var_id = var_cfg["variation_id"].empty() ?
 333                                 var_cfg["variation_name"] : var_cfg["variation_id"];
 334 
 335                 variations_map::iterator ut;
 336                 bool success;
 337                 std::tie(ut, success) = variations_.emplace(var_id, unit_type(var_cfg, id_));
 338                 if(success) {
 339                         ut->second.debug_id_ = debug_id_ + " [" + var_id + "]";
 340                         ut->second.base_id_ = base_id_;  // In case this is not id_.
 341                         ut->second.build_help_index(mv_types, races, traits);
 342                 } else {
 343                         ERR_CF << "Skipping duplicate unit variation ID: " << var_id << "\n";
 344                 }
 345         }
 346 
 347         hide_help_= cfg_["hide_help"].to_bool();
 348 
 349         build_status_ = HELP_INDEXED;
 350 }

 352 /**
 353  * Load the most needed data into an empty unit_type (build to CREATE).
 354  * This creates the gender-specific types (if needed) and also defines how much
 355  * experience is needed to advance as well as what this advances to.
 356  */
 357 void unit_type::build_created(const movement_type_map &mv_types,
 358         const race_map &races, const config::const_child_itors &traits)
 359 {
 360         // Don't build twice.
 361         if ( CREATED <= build_status_ )
 362                 return;
 363         // There is no preceding build level (other than being constructed).
 364 
 365         // These should still be nullptr from the constructor.
 366         assert(!gender_types_[0]);
 367         assert(!gender_types_[1]);
 368 
 369         if ( const config &male_cfg = cfg_.child("male") ) {
 370                 gender_types_[0].reset(new unit_type(male_cfg, id_));
 371                 gender_types_[0]->debug_id_ = debug_id_ + " (male)";
 372         }
 373 
 374         if ( const config &female_cfg = cfg_.child("female") ) {
 375                 gender_types_[1].reset(new unit_type(female_cfg, id_));
 376                 gender_types_[1]->debug_id_ = debug_id_ + " (female)";
 377         }
 378 
 379         for (unsigned i = 0; i < gender_types_.size(); ++i) {
 380                 if (gender_types_[i]) {
 381                         gender_types_[i]->build_created(mv_types, races, traits);
 382                 }
 383         }
 384 
 385     const std::string& advances_to_val = cfg_["advances_to"];
 386     if(advances_to_val != "null" && advances_to_val != "")
 387         advances_to_ = utils::split(advances_to_val);
 388     DBG_UT << "unit_type '" << log_id() << "' advances to : " << advances_to_val << "\n";
 389 
 390         experience_needed_ = cfg_["experience"].to_int(500);
 391 
 392         build_status_ = CREATED;
 393 }

 395 /**
 396  * Performs a build of this to the indicated stage.
 397  */
 398 void unit_type::build(BUILD_STATUS status, const movement_type_map &movement_types,
 399                       const race_map &races, const config::const_child_itors &traits)
 400 {
 401         DBG_UT << "Building unit type " << log_id() << ", level " << status << '\n';
 402 
 403         switch (status) {
 404         case NOT_BUILT:
 405                 // Already done in the constructor.
 406                 return;
 407 
 408         case CREATED:
 409                 // Build the basic data.
 410                 build_created(movement_types, races, traits);
 411                 return;
 412 
 413         case VARIATIONS: // Implemented as part of HELP_INDEXED
 414         case HELP_INDEXED:
 415                 // Build the data needed to feed the help index.
 416                 build_help_index(movement_types, races, traits);
 417                 return;
 418 
 419         case FULL:
 420                 build_full(movement_types, races, traits);
 421                 return;
 422 
 423         default:
 424                 ERR_UT << "Build of unit_type to unrecognized status (" << status << ") requested." << std::endl;
 425                 // Build as much as possible.
 426                 build_full(movement_types, races, traits);
 427                 return;
 428         }
 429 }

 560 void unit_type::add_advancement(const unit_type &to_unit,int xp)
 561 {
 562         const std::string &to_id = to_unit.id_;
 563 
 564         // Add extra advancement path to this unit type
 565         LOG_CONFIG << "adding advancement from " << log_id() << " to " << to_unit.log_id() << "\n";
 566         if(std::find(advances_to_.begin(), advances_to_.end(), to_id) == advances_to_.end()) {
 567                 advances_to_.push_back(to_id);
 568         } else {
 569                 LOG_CONFIG << "advancement from " << log_id() << " to "
 570                            << to_unit.log_id() << " already known, ignoring.\n";
 571                 return;
 572         }
 573 
 574         if(xp > 0) {
 575                 //xp is 0 in case experience= wasn't given.
 576                 if(!in_advancefrom_) {
 577                         //This function is called for and only for an [advancefrom] tag in a unit_type referencing this unit_type.
 578                         in_advancefrom_ = true;
 579                         experience_needed_ = xp;
 580                         DBG_UT << "Changing experience_needed from " << experience_needed_
 581                                << " to " << xp << " due to (first) [advancefrom] of "
 582                                << to_unit.log_id() << "\n";
 583                 }
 584                 else if(experience_needed_ > xp) {
 585                         experience_needed_ = xp;
 586                         DBG_UT << "Lowering experience_needed from " << experience_needed_
 587                                << " to " << xp << " due to (multiple, lower) [advancefrom] of "
 588                                << to_unit.log_id() << "\n";
 589                 }
 590                 else
 591                         DBG_UT << "Ignoring experience_needed change from " << experience_needed_
 592                                << " to " << xp << " due to (multiple, higher) [advancefrom] of "
 593                                << to_unit.log_id() << "\n";
 594         }
 595 
 596         // Add advancements to gendered subtypes, if supported by to_unit
 597         for(int gender=0; gender<=1; ++gender) {
 598                 if(!gender_types_[gender]) {
 599                         continue;
 600                 }
 601                 if(!to_unit.gender_types_[gender]) {
 602                         WRN_CF << to_unit.log_id() << " does not support gender " << gender << std::endl;
 603                         continue;
 604                 }
 605                 LOG_CONFIG << "gendered advancement " << gender << ": ";
 606                 gender_types_[gender]->add_advancement(*(to_unit.gender_types_[gender]), xp);
 607         }
 608 
 609         if ( cfg_.has_child("variation") ) {
 610                 // Make sure the variations are created.
 611                 unit_types.build_unit_type(*this, VARIATIONS);
 612 
 613                 // Add advancements to variation subtypes.
 614                 // Since these are still a rare and special-purpose feature,
 615                 // we assume that the unit designer knows what they're doing,
 616                 // and don't block advancements that would remove a variation.
 617                 for(variations_map::iterator v=variations_.begin();
 618                         v!=variations_.end(); ++v) {
 619                         LOG_CONFIG << "variation advancement: ";
 620                         v->second.add_advancement(to_unit,xp);
 621                 }
 622         }
 623 }

 625 static void advancement_tree_internal(const std::string& id, std::set<std::string>& tree)
 626 {
 627         const unit_type *ut = unit_types.find(id);
 628         if (!ut)
 629                 return;
 630 
 631         for (const std::string& adv : ut->advances_to()) {
 632                 if (tree.insert(adv).second) {
 633                         // insertion succeed, expand the new type
 634                         advancement_tree_internal(adv, tree);
 635                 }
 636         }
 637 }

 642         advancement_tree_internal(id_, tree);
 643         return tree;
 644 }

 663 // This function is only meant to return the likely state a given status
 664 // for a new recruit of this type. It should not be used to check if
 665 // a particular unit has it, use get_state(status_name) for that.
 666 bool unit_type::musthave_status(const std::string& status_name) const
 667 {
 668         // Statuses default to absent.
 669         bool current_status = false;
 670 
 671         // Look at all of the "musthave" traits to see if the
 672         // status gets changed. In the unlikely event it gets changed
 673         // multiple times, we want to try to do it in the same order
 674         // that unit::apply_modifications does things.
 675         for (const config &mod : possible_traits())
 676         {
 677                 if (mod["availability"] != "musthave")
 678                         continue;
 679 
 680                 for (const config &effect : mod.child_range("effect"))
 681                 {
 682                         // See if the effect only applies to
 683                         // certain unit types But don't worry
 684                         // about gender checks, since we don't
 685                         // know what the gender of the
 686                         // hypothetical recruit is.
 687                         const std::string &ut = effect["unit_type"];
 688                         if (!ut.empty()) {
 689                                 const std::vector<std::string> &types = utils::split(ut);
 690                                 if(std::find(types.begin(), types.end(), id()) == types.end())
 691                                         continue;
 692                         }
 693 
 694                         // We're only interested in status changes.
 695                         if (effect["apply_to"] != "status") {
 696                                 continue;
 697                         }
 698                         if (effect["add"] == status_name) {
 699                                 current_status = true;
 700                         }
 701                         if (effect["remove"] == status_name) {
 702                                 current_status = false;
 703                         }
 704                 }
 705         }
 706 
 707         return current_status;
 708 }

 775 int unit_type::resistance_against(const std::string& damage_name, bool attacker) const
 776 {
 777         int resistance = movement_type_.resistance_against(damage_name);
 778         unit_ability_list resistance_abilities;
 779         if (const config &abilities = cfg_.child("abilities")) {
 780                 for (const config& cfg : abilities.child_range("resistance")) {
 781                         if (!cfg["affect_self"].to_bool(true)) {
 782                                 continue;
 783                         }
 784                         if (!resistance_filter_matches(cfg, attacker, damage_name, 100 - resistance)) {
 785                                 continue;
 786                         }
 787                         resistance_abilities.push_back(unit_ability(&cfg, map_location::null_location()));
 788                 }
 789         }
 790         if (!resistance_abilities.empty()) {
 791                 unit_abilities::effect resist_effect(resistance_abilities, 100 - resistance, false);
 792                 resistance = 100 - std::min<int>(resist_effect.get_composite_value(),
 793                                 resistance_abilities.highest("max_value").first);
 794         }
 795         return resistance;
 796 }

 798 bool unit_type::resistance_filter_matches(const config& cfg, bool attacker, const std::string& damage_name, int res) const
 799 {
 800         if(!(cfg["active_on"].empty() || (attacker && cfg["active_on"]=="offense") || (!attacker && cfg["active_on"]=="defense"))) {
 801                 return false;
 802         }
 803         const std::string& apply_to = cfg["apply_to"];
 804         if(!apply_to.empty()) {
 805                 if(damage_name != apply_to) {
 806                         if ( apply_to.find(',') != std::string::npos  &&
 807                              apply_to.find(damage_name) != std::string::npos ) {
 808                                 const std::vector<std::string>& vals = utils::split(apply_to);
 809                                 if(std::find(vals.begin(),vals.end(),damage_name) == vals.end()) {
 810                                         return false;
 811                                 }
 812                         } else {
 813                                 return false;
 814                         }
 815                 }
 816         }
 817         if (!unit_abilities::filter_base_matches(cfg, res)) return false;
 818         return true;
 819 }

 861 namespace { // Helpers for set_config()
 862         /**
 863          * Spits out an error message and throws a config::error.
 864          * Called when apply_base_unit() detects a cycle.
 865          * (This exists merely to take the error message out of that function.)
 866          */
 867         void throw_base_unit_recursion_error(
 868                 const std::vector<std::string> & base_tree, const std::string & base_id)
 869         {
 870                 std::stringstream ss;
 871                 ss << "[base_unit] recursion loop in [unit_type] ";
 872                 for (const std::string &step : base_tree)
 873                         ss << step << "->";
 874                 ss << base_id;
 875                 ERR_CF << ss.str() << '\n';
 876                 throw config::error(ss.str());
 877         }
 878 
 879         /**
 880          * Locates the config for the unit type with id= @a key within @a all_types.
 881          * Throws a config::error if the unit type cannot be found.
 882          */
 883         config & find_unit_type_config(const std::string & key, config & all_types)
 884         {
 885                 config & cfg = all_types.find_child("unit_type", "id", key);
 886                 if ( cfg )
 887                         return cfg;
 888 
 889                 // Bad WML!
 890                 ERR_CF << "unit type not found: " << key << std::endl;
 891                 ERR_CF << all_types << std::endl;
 892                 throw config::error("unit type not found: " + key);
 893         }
 894 
 895         /**
 896          * Modifies the provided config by merging all base units into it.
 897          * The @a base_tree parameter is used solely for detecting and reporting
 898          * cycles of base units; it is no longer needed to prevent infinite loops.
 899          */
 900         void apply_base_unit(config & ut_cfg, config & all_types,
 901                              std::vector<std::string> &base_tree)
 902         {
 903                 // Get a list of base units to apply.
 904                 std::vector<std::string> base_ids;
 905                 for (config & base : ut_cfg.child_range("base_unit"))
 906                         base_ids.push_back(base["id"]);
 907 
 908                 if ( base_ids.empty() )
 909                         // Nothing to do.
 910                         return;
 911 
 912                 // Store the base ids for the help system.
 913                 ut_cfg["base_ids"] = utils::join(base_ids);
 914 
 915                 // Clear the base units (otherwise they could interfere with the merge).
 916                 // This has the side-effect of breaking cycles, hence base_tree is
 917                 // merely for error detection, not error recovery.
 918                 ut_cfg.clear_children("base_unit");
 919 
 920                 // Merge the base units, in order.
 921                 for (const std::string & base_id : base_ids) {
 922                         // Detect recursion so the WML author is made aware of an error.
 923                         if ( std::find(base_tree.begin(), base_tree.end(), base_id) != base_tree.end() )
 924                                 throw_base_unit_recursion_error(base_tree, base_id);
 925 
 926                         // Find the base unit.
 927                         config & base_cfg = find_unit_type_config(base_id, all_types);
 928                         // Make sure the base unit has had its base units accounted for.
 929                         base_tree.push_back(base_id);
 930                         apply_base_unit(base_cfg, all_types, base_tree);
 931                         base_tree.pop_back();
 932 
 933                         // Merge the base unit "under" our config.
 934                         ut_cfg.inherit_from(base_cfg);
 935                 }
 936         }
 937 
 938         /**
 939          * Handles inheritance for configs of [male], [female], and [variation].
 940          * Also removes gendered children, as those serve no purpose.
 941          * @a default_inherit is the default value for inherit=.
 942          */
 943         void fill_unit_sub_type(config & var_cfg, const config & parent,
 944                                 bool default_inherit)
 945         {
 946                 if ( var_cfg["inherit"].to_bool(default_inherit) ) {
 947                         var_cfg.inherit_from(parent);
 948                 }
 949                 var_cfg.clear_children("male");
 950                 var_cfg.clear_children("female");
 951         }
 952 
 953         /**
 954          * Processes [variation] tags of @a ut_cfg, handling inheritance and
 955          * child clearing.
 956          */
 957         void handle_variations(config & ut_cfg)
 958         {
 959                 // Most unit types do not have variations.
 960                 if ( !ut_cfg.has_child("variation") )
 961                         return;
 962 
 963                 // Pull the variations out of the base unit type.
 964                 config variations;
 965                 variations.splice_children(ut_cfg, "variation");
 966 
 967                 // Handle each variation's inheritance.
 968                 for (config &var_cfg : variations.child_range("variation"))
 969                         fill_unit_sub_type(var_cfg, ut_cfg, false);
 970 
 971                 // Restore the variations.
 972                 ut_cfg.splice_children(variations, "variation");
 973         }

 975         const boost::regex fai_identifier("[a-zA-Z_]+");
 976 
 977         template<typename MoveT>
 978         void patch_movetype(MoveT& mt, const std::string& new_key, const std::string& formula_str, int default_val, bool replace) {
 979                 config temp_cfg, original_cfg;
 980                 mt.write(original_cfg);
 981                 if (!replace && !original_cfg[new_key].blank()) {
 982                         // Don't replace if the key already exists in the config (even if empty).
 983                         return;
 984                 }
 985                 gui2::typed_formula<int> formula(formula_str);
 986                 wfl::map_formula_callable original;
 987                 boost::sregex_iterator m(formula_str.begin(), formula_str.end(), fai_identifier);
 988                 for (const boost::sregex_iterator::value_type& p : std::make_pair(m, boost::sregex_iterator())) {
 989                         const std::string var_name = p.str();
 990                         wfl::variant val(original_cfg[var_name].to_int(default_val));
 991                         original.add(var_name, val);
 992                 }
 993                 temp_cfg[new_key] = formula(original);
 994                 mt.merge(temp_cfg, true);
 995         }

 998 /**
 999  * Resets all data based on the provided config.
1000  * This includes some processing of the config, such as expanding base units.
1001  * A pointer to the config is stored, so the config must be persistent.
1002  */
1003 void unit_type_data::set_config(config &cfg)
1004 {
1005         DBG_UT << "unit_type_data::set_config, name: " << cfg["name"] << "\n";
1006 
1007         clear();
1008         unit_cfg_ = &cfg;
1009 
1010         for (const config &mt : cfg.child_range("movetype"))
1011         {
1012                 movement_types_.emplace(mt["name"].str(), movetype(mt));
1013                 gui2::dialogs::loading_screen::progress();
1014         }
1015 
1016         for (const config &r : cfg.child_range("race"))
1017         {
1018                 const unit_race race(r);
1019                 races_.emplace(race.id(), race);
1020                 gui2::dialogs::loading_screen::progress();
1021         }
1022 
1023         // Movetype resistance patching
1024         for (const config &r : cfg.child_range("resistance_defaults"))
1025         {
1026                 const std::string& dmg_type = r["id"];
1027                 config temp_cfg;
1028                 for (const config::attribute &attr : r.attribute_range()) {
1029                         const std::string &mt = attr.first;
1030                         if (mt == "id" || mt == "default" || movement_types_.find(mt) == movement_types_.end()) {
1031                                 continue;
1032                         }
1033                         patch_movetype(movement_types_[mt].get_resistances(), dmg_type, attr.second, 100, true);
1034                 }
1035                 if (r.has_attribute("default")) {
1036                         for (movement_type_map::value_type &mt : movement_types_) {
1037                                 // Don't apply a default if a value is explicitly specified.
1038                                 if (r.has_attribute(mt.first)) {
1039                                         continue;
1040                                 }
1041                                 patch_movetype(mt.second.get_resistances(), dmg_type, r["default"], 100, false);
1042                         }
1043                 }
1044         }
1045 
1046         // Movetype move/defend patching
1047         for (const config &terrain : cfg.child_range("terrain_defaults"))
1048         {
1049                 const std::string& ter_type = terrain["id"];
1050                 config temp_cfg;
1051                 static const std::string terrain_info_tags[] {"movement", "vision", "jamming", "defense"};
1052                 for (const std::string &tag : terrain_info_tags) {
1053                         if (!terrain.has_child(tag)) {
1054                                 continue;
1055                         }
1056                         const config& info = terrain.child(tag);
1057                         for (const config::attribute &attr : info.attribute_range()) {
1058                                 const std::string &mt = attr.first;
1059                                 if (mt == "default" || movement_types_.find(mt) == movement_types_.end()) {
1060                                         continue;
1061                                 }
1062                                 if (tag == "defense") {
1063                                         patch_movetype(movement_types_[mt].get_defense(), ter_type, attr.second, 100, true);
1064                                 } else if (tag == "vision") {
1065                                         patch_movetype(movement_types_[mt].get_vision(), ter_type, attr.second, 99, true);
1066                                 } else if (tag == "movement") {
1067                                         patch_movetype(movement_types_[mt].get_movement(), ter_type, attr.second, 99, true);
1068                                 } else if (tag == "jamming") {
1069                                         patch_movetype(movement_types_[mt].get_jamming(), ter_type, attr.second, 99, true);
1070                                 }
1071                         }
1072                         if (info.has_attribute("default")) {
1073                                 for (movement_type_map::value_type &mt : movement_types_) {
1074                                         // Don't apply a default if a value is explicitly specified.
1075                                         if (info.has_attribute(mt.first)) {
1076                                                 continue;
1077                                         }
1078                                         if (tag == "defense") {
1079                                                 patch_movetype(mt.second.get_defense(), ter_type, info["default"], 100, false);
1080                                         } else if (tag == "vision") {
1081                                                 patch_movetype(mt.second.get_vision(), ter_type, info["default"], 99, false);
1082                                         } else if (tag == "movement") {
1083                                                 patch_movetype(mt.second.get_movement(), ter_type, info["default"], 99, false);
1084                                         } else if (tag == "jamming") {
1085                                                 patch_movetype(mt.second.get_jamming(), ter_type, info["default"], 99, false);
1086                                         }
1087                                 }
1088                         }
1089                 }
1090         }
1091 
1092         // Apply base units.
1093         for (config &ut : cfg.child_range("unit_type"))
1094         {
1095                 if ( ut.has_child("base_unit") ) {
1096                         // Derived units must specify a new id.
1097                         // (An error message will be emitted later if id is empty.)
1098                         const std::string id = ut["id"];
1099                         if ( !id.empty() ) {
1100                                 std::vector<std::string> base_tree(1, id);
1101                                 apply_base_unit(ut, cfg, base_tree);
1102                                 gui2::dialogs::loading_screen::progress();
1103                         }
1104                 }
1105         }
1106 
1107         // Handle inheritance and recording of unit types.
1108         for (config &ut : cfg.child_range("unit_type"))
1109         {
1110                 std::string id = ut["id"];
1111                 // Every type is required to have an id.
1112                 if ( id.empty() ) {
1113                         ERR_CF << "[unit_type] with empty id=, ignoring:\n" << ut.debug();
1114                         continue;
1115                 }
1116 
1117                 // Complete the gender-specific children of the config.
1118                 if ( config &male_cfg = ut.child("male") ) {
1119                         fill_unit_sub_type(male_cfg, ut, true);
1120                         handle_variations(male_cfg);
1121                 }
1122                 if ( config &female_cfg = ut.child("female") ) {
1123                         fill_unit_sub_type(female_cfg, ut, true);
1124                         handle_variations(female_cfg);
1125                 }
1126 
1127                 // Complete the variation-defining children of the config.
1128                 handle_variations(ut);
1129 
1130                 // Record this unit type.
1131                 if(types_.emplace(id, unit_type(ut)).second) {
1132                         LOG_CONFIG << "added " << id << " to unit_type list (unit_type_data.unit_types)\n";
1133                 } else {
1134                         ERR_CF << "Multiple [unit_type]s with id=" << id << " encountered." << std::endl;
1135                 }
1136 
1137                 gui2::dialogs::loading_screen::progress();
1138         }
1139 
1140         // Build all unit types. (This was not done within the loop for performance.)
1141         build_all(unit_type::CREATED);
1142 
1143         // Suppress some unit types (presumably used as base units) from the help.
1144         if (const config &hide_help = cfg.child("hide_help")) {
1145                 hide_help_all_ = hide_help["all"].to_bool();
1146                 read_hide_help(hide_help);
1147         }
1148 }

1194 void unit_type_data::build_all(unit_type::BUILD_STATUS status)
1195 {
1196         // Nothing to do if already built to the requested level.
1197         if ( status <= build_status_ )
1198                 return;
1199         assert(unit_cfg_ != nullptr);
1200 
1201         for (unit_type_map::iterator u = types_.begin(), u_end = types_.end(); u != u_end; ++u) {
1202                 build_unit_type(u->second, status);
1203                 gui2::dialogs::loading_screen::progress();
1204         }
1205         // Handle [advancefrom] (once) after building to (at least) the CREATED level.
1206         // (Currently, this could be simply a test for build_status_ == NOT_BUILT,
1207         // but to guard against future changes, use a more thorough test.)
1208         if ( build_status_ < unit_type::CREATED  &&  unit_type::CREATED <= status )
1209                 for (unit_type_map::iterator u = types_.begin(), u_end = types_.end(); u != u_end; ++u) {
1210                         add_advancement(u->second);
1211                 }
1212 
1213         build_status_ = status;
1214 }

1216 void unit_type_data::read_hide_help(const config& cfg)
1217 {
1218         if (!cfg)
1219                 return;
1220 
1221         hide_help_race_.emplace_back();
1222         hide_help_type_.emplace_back();
1223 
1224         std::vector<std::string> races = utils::split(cfg["race"]);
1225         hide_help_race_.back().insert(races.begin(), races.end());
1226 
1227         std::vector<std::string> types = utils::split(cfg["type"]);
1228         hide_help_type_.back().insert(types.begin(), types.end());
1229 
1230         std::vector<std::string> trees = utils::split(cfg["type_adv_tree"]);
1231         hide_help_type_.back().insert(trees.begin(), trees.end());
1232         for (const std::string& t_id : trees) {
1233                 unit_type_map::iterator ut = types_.find(t_id);
1234                 if (ut != types_.end()) {
1235                         std::set<std::string> adv_tree = ut->second.advancement_tree();
1236                         hide_help_type_.back().insert(adv_tree.begin(), adv_tree.end());
1237                 }
1238         }
1239 
1240         // we call recursively all the imbricated [not] tags
1241         read_hide_help(cfg.child("not"));
1242 }

1244 bool unit_type_data::hide_help(const std::string& type, const std::string& race) const
1245 {
1246         bool res = hide_help_all_;
1247         int lvl = hide_help_all_ ? 1 : 0; // first level is covered by 'all=yes'
1248 
1249         // supposed to be equal but let's be cautious
1250         int lvl_nb = std::min(hide_help_race_.size(), hide_help_type_.size());
1251 
1252         for (; lvl < lvl_nb; ++lvl) {
1253                 if (hide_help_race_[lvl].count(race) || hide_help_type_[lvl].count(type))
1254                         res = !res; // each level is a [not]
1255         }
1256         return res;
1257 }

1259 void unit_type_data::add_advancement(unit_type& to_unit) const
1260 {
1261     const config& cfg = to_unit.get_cfg();
1262 
1263     for (const config &af : cfg.child_range("advancefrom"))
1264     {
1265         const std::string &from = af["unit"];
1266         int xp = af["experience"];
1267 
1268         unit_type_data::unit_type_map::iterator from_unit = types_.find(from);
1269 
1270                 if (from_unit == types_.end()) {
1271                         std::ostringstream msg;
1272                         msg << "unit type '" << from << "' not found when resolving [advancefrom] tag for '"
1273                                 << to_unit.log_id() << "'";
1274                         throw config::error(msg.str());
1275                 }
1276 
1277         // Fix up advance_from references
1278         from_unit->second.add_advancement(to_unit, xp);
1279 
1280         DBG_UT << "Added advancement ([advancefrom]) from " << from << " to " << to_unit.log_id() << "\n";
1281     }
1282 }

1290 void unit_type::check_id(std::string& id)
1291 {
1292         assert(!id.empty());
1293         //we don't allow leading whitepaces
1294         if (id[0] == ' ') {
1295                 throw error("Found unit type id with a leading whitespace \"" + id + "\"");
1296         }
1297         bool gave_wanrning = false;
1298         for (size_t pos = 0; pos < id.size(); ++pos) {
1299                 const char c = id[pos];
1300                 const bool valid = c == '_' || c == ' ' || (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9');
1301                 if (!valid) {
1302                         if (!gave_wanrning) {
1303                                 ERR_UT << "Found unit type id with invalid chracters: \"" << id << "\"\n";
1304                                 gave_wanrning = true;
1305                         }
1306                         id[pos] = '_';
1307                 }
1308         }
1309 
1310 }

1315 void adjust_profile(std::string& profile)
1316 {
1317         // Create a temp copy
1318         std::string temp = profile;
1319 
1320         static const std::string path_adjust = "/transparent";
1321         const std::string::size_type offset = profile.find_last_of('/', profile.find('~'));
1322 
1323         // If the path already refers to /transparent...
1324         if(profile.find(path_adjust) != std::string::npos && offset != std::string::npos) {
1325                 if(!image::locator(profile).file_exists()) {
1326                         profile.replace(profile.find(path_adjust), path_adjust.length(), "");
1327                 }
1328 
1329                 return;
1330         }
1331 
1332         // else, check for the file with /transparent appended...
1333         offset != std::string::npos ?
1334                 temp.insert(offset, path_adjust) : temp = path_adjust + temp;
1335 
1336         // and use that path if it exists
1337         if(image::locator(temp).file_exists()) {
1338                 profile = temp;
1339         }
1340 }
