 118 bool affects_side(const config& cfg, const std::vector<team>& teams, size_t side, size_t other_side)
 119 {
 120         if (side == other_side)
 121                 return cfg["affect_allies"].to_bool(true);
 122         if (teams[side - 1].is_enemy(other_side))
 123                 return cfg["affect_enemies"].to_bool();
 124         else
 125                 return cfg["affect_allies"].to_bool();

 131 bool unit::get_ability_bool(const std::string& tag_name, const map_location& loc, const display_context& dc) const
 132 {
 133         for (const config &i : this->abilities_.child_range(tag_name)) {
 134                 if (ability_active(tag_name, i, loc) &&
 135                         ability_affects_self(tag_name, i, loc))
 136                 {
 137                         return true;
 138                 }
 139         }
 140 
 141         assert(resources::units);
 142         const unit_map& units = *resources::units;
 143 
 144         map_location adjacent[6];
 145         get_adjacent_tiles(loc,adjacent);
 146         for(int i = 0; i != 6; ++i) {
 147                 const unit_map::const_iterator it = units.find(adjacent[i]);
 148                 if (it == units.end() || it->incapacitated())
 149                         continue;
 150                 // Abilities may be tested at locations other than the unit's current
 151                 // location. This is intentional to allow for less messing with the unit
 152                 // map during calculations, particularly with regards to movement.
 153                 // Thus, we need to make sure the adjacent unit (*it) is not actually
 154                 // ourself.
 155                 if ( &*it == this )
 156                         continue;
 157                 for (const config &j : it->abilities_.child_range(tag_name)) {
 158                         if (affects_side(j, dc.teams(), side(), it->side()) &&
 159                             it->ability_active(tag_name, j, adjacent[i]) &&
 160                             ability_affects_adjacent(tag_name,  j, i, loc, *it))
 161                         {
 162                                 return true;
 163                         }
 164                 }
 165         }
 166 
 167 
 168         return false;
 169 }

 275                 }
 276                 else
 277                 {
 278                         // See if an inactive name was specified.
 279                         config::attribute_value const &inactive_value =
 280                                 gender_value(ab.cfg, gender_, "name_inactive",

 297 bool unit::ability_active(const std::string& ability,const config& cfg,const map_location& loc) const
 298 {
 299         bool illuminates = ability == "illuminates";
 300 
 301         if (const config &afilter = cfg.child("filter"))
 302                 if ( !unit_filter(vconfig(afilter), resources::filter_con, illuminates).matches(*this, loc) )
 303                         return false;
 304 
 305         map_location adjacent[6];
 306         get_adjacent_tiles(loc,adjacent);
 307 
 308         assert(resources::units);
 309         const unit_map& units = *resources::units;
 310 
 311         for (const config &i : cfg.child_range("filter_adjacent"))
 312         {
 313                 size_t count = 0;
 314                 const unit_filter ufilt(vconfig(i), resources::filter_con, illuminates);
 315                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 316                 for (const map_location::DIRECTION index : dirs)
 317                 {
 318                         if (index == map_location::NDIRECTIONS)
 319                                 continue;
 320                         unit_map::const_iterator unit = units.find(adjacent[index]);
 321                         if (unit == units.end())
 322                                 return false;
 323                         if (!ufilt(*unit, *this))
 324                                 return false;
 325                         if (i.has_attribute("is_enemy")) {
 326                                 const display_context& dc = resources::filter_con->get_disp_context();
 327                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(side_)) {
 328                                         continue;
 329                                 }
 330                         }
 331                         count++;
 332                 }
 333                 if (i["count"].empty() && count != dirs.size()) {
 334                         return false;
 335                 }
 336                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 337                         return false;
 338                 }
 339         }
 340 
 341         for (const config &i : cfg.child_range("filter_adjacent_location"))
 342         {
 343                 size_t count = 0;
 344                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 345                 adj_filter.flatten(illuminates);
 346 
 347                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 348                 for (const map_location::DIRECTION index : dirs)
 349                 {
 350                         if (index == map_location::NDIRECTIONS) {
 351                                 continue;
 352                         }
 353                         if(!adj_filter.match(adjacent[index])) {
 354                                 return false;
 355                         }
 356                         count++;
 357                 }
 358                 if (i["count"].empty() && count != dirs.size()) {
 359                         return false;
 360                 }
 361                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 362                         return false;
 363                 }
 364         }
 365         return true;
 366 }

 368 bool unit::ability_affects_adjacent(const std::string& ability, const config& cfg,int dir,const map_location& loc,const unit& from) const
 369 {
 370         bool illuminates = ability == "illuminates";
 371 
 372         assert(dir >=0 && dir <= 5);
 373         map_location::DIRECTION direction = static_cast<map_location::DIRECTION>(dir);
 374 
 375         for (const config &i : cfg.child_range("affect_adjacent"))
 376         {
 377                 if (i.has_attribute("adjacent")) { //key adjacent defined
 378                         std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 379                         if (std::find(dirs.begin(), dirs.end(), direction) == dirs.end()) {
 380                                 continue;
 381                         }
 382                 }
 383                 const config &filter = i.child("filter");
 384                 if (!filter || //filter tag given
 385                         unit_filter(vconfig(filter), resources::filter_con, illuminates).matches(*this, loc, from) ) {
 386                         return true;
 387                 }
 388         }
 389         return false;
 390 }

 406 std::pair<int,map_location> unit_ability_list::highest(const std::string& key, int def) const
 407 {
 408         if ( cfgs_.empty() ) {
 409                 return std::make_pair(def, map_location());
 410         }
 411         // The returned location is the best non-cumulative one, if any,
 412         // the best absolute cumulative one otherwise.
 413         map_location best_loc;
 414         bool only_cumulative = true;
 415         int abs_max = 0;
 416         int flat = 0;
 417         int stack = 0;
 418         for (unit_ability const &p : cfgs_)
 419         {
 420                 int value = (*p.first)[key].to_int(def);
 421                 if ((*p.first)["cumulative"].to_bool()) {
 422                         stack += value;
 423                         if (value < 0) value = -value;
 424                         if (only_cumulative && value >= abs_max) {
 425                                 abs_max = value;
 426                                 best_loc = p.second;
 427                         }
 428                 } else if (only_cumulative || value > flat) {
 429                         only_cumulative = false;
 430                         flat = value;
 431                         best_loc = p.second;
 432                 }
 433         }
 434         return std::make_pair(flat + stack, best_loc);
 435 }

 498         /**
 499          * Gets the children of @parent (which should be the specials for an
 500          * attack_type) and places the ones whose tag or id= matches @a id into
 501          * @a result.
 502          * If @a just_peeking is set to true, then @a result is not touched;
 503          * instead the return value is used to indicate if any matching children
 504          * were found.
 505          *
 506          * @returns  true if @a just_peeking is true and a match was found;
 507          *           false otherwise.
 508          */
 509         bool get_special_children(std::vector<const config*>& result, const config& parent,
 510                                    const std::string& id, bool just_peeking=false) {
 511                 for (const config::any_child &sp : parent.all_children_range())
 512                 {
 513                         if (sp.key == id || sp.cfg["id"] == id) {
 514                                 if(just_peeking) {
 515                                         return true; // peek succeeded; done
 516                                 } else {
 517                                         result.push_back(&sp.cfg);
 518                                 }
 519                         }
 520                 }
 521                 return false;
 522         }
 523 }

 525 /**
 526  * Returns whether or not @a *this has a special with a tag or id equal to
 527  * @a special. If @a simple_check is set to true, then the check is merely
 528  * for being present. Otherwise (the default), the check is for a special
 529  * active in the current context (see set_specials_context), including
 530  * specials obtained from the opponent's attack.
 531  */
 532 bool attack_type::get_special_bool(const std::string& special, bool simple_check) const
 533 {
 534         {
 535                 std::vector<const config*> list;
 536                 if ( get_special_children(list, specials_, special, simple_check) ) {
 537                         return true;
 538                 }
 539                 // If we make it to here, then either list.empty() or !simple_check.
 540                 // So if the list is not empty, then this is not a simple check and
 541                 // we need to check each special in the list to see if any are active.
 542                 for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 543                         if ( special_active(**i, AFFECT_SELF) ) {
 544                                 return true;
 545                         }
 546                 }
 547         }
 548         // Skip checking the opponent's attack?
 549         if ( simple_check || !other_attack_ ) {
 550                 return false;
 551         }
 552 
 553         std::vector<const config*> list;
 554         get_special_children(list, other_attack_->specials_, special);
 555         for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 556                 if ( other_attack_->special_active(**i, AFFECT_OTHER) ) {
 557                         return true;
 558                 }
 559         }
 560         return false;
 561 }

 630         //log_scope("weapon_specials");
 631         std::string res;
 632         for (const config::any_child &sp : specials_.all_children_range())
 633         {
 634                 if ( only_active  &&  !special_active(sp.cfg, AFFECT_EITHER, is_backstab) )
 635                         continue;
 636 
 637                 const std::string& name = sp.cfg["name"].str();
 638                 if (!name.empty()) {
 639                         if (!res.empty()) res += ", ";
 640                         res += name;
 641                 }
 642         }
 643 
 644         return res;
 645 }

 689 /**
 690  * Calculates the number of attacks this weapon has, considering specials.
 691  * This returns two numbers because of the swarm special. The actual number of
 692  * attacks depends on the unit's health and should be:
 693  *   min_attacks + (max_attacks - min_attacks) * (current hp) / (max hp)
 694  * c.f. swarm_blows()
 695  */
 696 void attack_type::modified_attacks(bool is_backstab, unsigned & min_attacks,
 697                                    unsigned & max_attacks) const
 698 {
 699         // Apply [attacks].
 700         unit_abilities::effect attacks_effect(get_specials("attacks"),
 701                                               num_attacks(), is_backstab);
 702         int attacks_value = attacks_effect.get_composite_value();
 703         if ( attacks_value < 0 ) {
 704                 attacks_value = num_attacks();
 705                 ERR_NG << "negative number of strikes after applying weapon specials" << std::endl;
 706         }
 707 
 708         // Apply [swarm].
 709         unit_ability_list swarm_specials = get_specials("swarm");
 710         if ( !swarm_specials.empty() ) {
 711                 min_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_min").first);
 712                 max_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_max", attacks_value).first);
 713         } else {
 714                 min_attacks = max_attacks = attacks_value;
 715         }
 716 }

 732          * Returns whether or not the given special affects the opponent of the unit
 733          * with the special.
 734          * @param[in]  special      a weapon special WML structure
 735          * @param[in]  is_attacker  whether or not the unit with the special is the attacker
 736          */
 737         bool special_affects_opponent(const config& special, bool is_attacker)
 738         {
 739                 //log_scope("special_affects_opponent");
 740                 const std::string& apply_to = special["apply_to"];
 741                 if ( apply_to.empty() )
 742                         return false;
 743                 if ( apply_to == "both" )
 744                         return true;
 745                 if ( apply_to == "opponent" )
 746                         return true;
 747                 if ( is_attacker  &&  apply_to == "defender" )
 748                         return true;
 749                 if ( !is_attacker &&  apply_to == "attacker" )
 750                         return true;
 751                 return false;
 752         }

 763                 if ( apply_to.empty() )
 764                         return true;

 766                         return true;

 768                         return true;

 776         /**
 777          * Determines if a unit/weapon combination matches the specified child
 778          * (normally a [filter_*] child) of the provided filter.
 779          * @param[in]  un_it      The unit to filter.
 780          * @param[in]  loc        The presumed location of @a un_it.
 781          * @param[in]  weapon     The attack_type to filter.
 782          * @param[in]  filter     The filter containing the child filter to use.
 783          * @param[in]  child_tag  The tag of the child filter to use.
 784          */
 785         static bool special_unit_matches(const unit_map::const_iterator & un_it,
 786                                                                          const unit_map::const_iterator & u2,
 787                                              const map_location & loc,
 788                                              const_attack_ptr weapon,
 789                                              const config & filter,
 790                                                                          const bool for_listing,
 791                                              const std::string & child_tag)
 792         {
 793                 if (for_listing && !loc.valid())
 794                         // The special's context was set to ignore this unit, so assume we pass.
 795                         // (This is used by reports.cpp to show active specials when the
 796                         // opponent is not known. From a player's perspective, the special
 797                         // is active, in that it can be used, even though the player might
 798                         // need to select an appropriate opponent.)
 799                         return true;
 800 
 801                 const config & filter_child = filter.child(child_tag);
 802                 if ( !filter_child )
 803                         // The special does not filter on this unit, so we pass.
 804                         return true;
 805 
 806                 // If the primary unit doesn't exist, there's nothing to match
 807                 if (!un_it.valid()) {
 808                         return false;
 809                 }
 810 
 811                 unit_filter ufilt(vconfig(filter_child), resources::filter_con);
 812 
 813                 // If the other unit doesn't exist, try matching without it
 814                 if (!u2.valid()) {
 815                         return ufilt.matches(*un_it, loc);
 816                 }
 817 
 818                 // Check for a unit match.
 819                 if (!ufilt.matches(*un_it, loc, *u2)) {
 820                         return false;
 821                 }
 822 
 823                 // Check for a weapon match.
 824                 if ( const config & filter_weapon = filter_child.child("filter_weapon") ) {
 825                         if ( !weapon || !weapon->matches_filter(filter_weapon) )
 826                                 return false;
 827                 }
 828 
 829                 // Passed.
 830                 return true;
 831         }

 835 /**
 836  * Returns whether or not the given special is active for the specified unit,
 837  * based on the current context (see set_specials_context).
 838  * @param[in] special           a weapon special WML structure
 839  * @param[in] whom              specifies which combatant we care about
 840  * @param[in] include_backstab  false if backstab specials should not be active
 841  *                              (usually true since backstab is usually accounted
 842  *                              for elsewhere)
 843  */
 844 bool attack_type::special_active(const config& special, AFFECTS whom,
 845                                  bool include_backstab) const
 846 {
 847         //log_scope("special_active");
 848 
 849         // Backstab check
 850         if ( !include_backstab )
 851                 if ( special["backstab"].to_bool() )
 852                         return false;
 853 
 854         // Does this affect the specified unit?
 855         if ( whom == AFFECT_SELF ) {
 856                 if ( !special_affects_self(special, is_attacker_) )
 857                         return false;
 858         }
 859         if ( whom == AFFECT_OTHER ) {
 860                 if ( !special_affects_opponent(special, is_attacker_) )
 861                         return false;
 862         }
 863 
 864         // Is this active on attack/defense?
 865         const std::string & active_on = special["active_on"];
 866         if ( !active_on.empty() ) {
 867                 if ( is_attacker_  &&  active_on != "offense" )
 868                         return false;
 869                 if ( !is_attacker_  &&  active_on != "defense" )
 870                         return false;
 871         }
 872 
 873         // Get the units involved.
 874         assert(resources::units);
 875         const unit_map& units = *resources::units;
 876 
 877         unit_map::const_iterator self  = units.find(self_loc_);
 878         unit_map::const_iterator other = units.find(other_loc_);
 879 
 880         // Make sure they're facing each other.
 881         temporary_facing self_facing(self, self_loc_.get_relative_dir(other_loc_));
 882         temporary_facing other_facing(other, other_loc_.get_relative_dir(self_loc_));
 883 
 884         // Translate our context into terms of "attacker" and "defender".
 885         unit_map::const_iterator & att = is_attacker_ ? self : other;
 886         unit_map::const_iterator & def = is_attacker_ ? other : self;
 887         const map_location & att_loc   = is_attacker_ ? self_loc_ : other_loc_;
 888         const map_location & def_loc   = is_attacker_ ? other_loc_ : self_loc_;
 889         const_attack_ptr att_weapon = is_attacker_ ? shared_from_this() : other_attack_;
 890         const_attack_ptr def_weapon = is_attacker_ ? other_attack_ : shared_from_this();
 891 
 892         // Filter the units involved.
 893         if (!special_unit_matches(self, other, self_loc_, shared_from_this(), special, is_for_listing_, "filter_self"))
 894                 return false;
 895         if (!special_unit_matches(other, self, other_loc_, other_attack_, special, is_for_listing_, "filter_opponent"))
 896                 return false;
 897         if (!special_unit_matches(att, def, att_loc, att_weapon, special, is_for_listing_, "filter_attacker"))
 898                 return false;
 899         if (!special_unit_matches(def, att, def_loc, def_weapon, special, is_for_listing_, "filter_defender"))
 900                 return false;
 901 
 902         map_location adjacent[6];
 903         get_adjacent_tiles(self_loc_, adjacent);
 904 
 905         // Filter the adjacent units.
 906         for (const config &i : special.child_range("filter_adjacent"))
 907         {
 908                 size_t count = 0;
 909                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 910                 unit_filter filter(vconfig(i), resources::filter_con);
 911                 for (const map_location::DIRECTION index : dirs)
 912                 {
 913                         if (index == map_location::NDIRECTIONS)
 914                                 continue;
 915                         unit_map::const_iterator unit = units.find(adjacent[index]);
 916                         if (unit == units.end() || !filter.matches(*unit, adjacent[index], *self))
 917                                 return false;
 918                         if (i.has_attribute("is_enemy")) {
 919                                 const display_context& dc = resources::filter_con->get_disp_context();
 920                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(self->side())) {
 921                                         continue;
 922                                 }
 923                         }
 924                         count++;
 925                 }
 926                 if (i["count"].empty() && count != dirs.size()) {
 927                         return false;
 928                 }
 929                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 930                         return false;
 931                 }
 932         }
 933 
 934         // Filter the adjacent locations.
 935         for (const config &i : special.child_range("filter_adjacent_location"))
 936         {
 937                 size_t count = 0;
 938                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 939                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 940                 for (const map_location::DIRECTION index : dirs)
 941                 {
 942                         if (index == map_location::NDIRECTIONS)
 943                                 continue;
 944                         if(!adj_filter.match(adjacent[index])) {
 945                                 return false;
 946                         }
 947                         count++;
 948                 }
 949                 if (i["count"].empty() && count != dirs.size()) {
 950                         return false;
 951                 }
 952                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 953                         return false;
 954                 }
 955         }
 956 
 957         return true;
 958 }

 973 bool filter_base_matches(const config& cfg, int def)
 974 {
 975         if (const config &apply_filter = cfg.child("filter_base_value")) {
 976                 config::attribute_value cond_eq = apply_filter["equals"];
 977                 config::attribute_value cond_ne = apply_filter["not_equals"];
 978                 config::attribute_value cond_lt = apply_filter["less_than"];
 979                 config::attribute_value cond_gt = apply_filter["greater_than"];
 980                 config::attribute_value cond_ge = apply_filter["greater_than_equal_to"];
 981                 config::attribute_value cond_le = apply_filter["less_than_equal_to"];
 982                 return  (cond_eq.empty() || def == cond_eq.to_int()) &&
 983                         (cond_ne.empty() || def != cond_ne.to_int()) &&
 984                         (cond_lt.empty() || def <  cond_lt.to_int()) &&
 985                         (cond_gt.empty() || def >  cond_gt.to_int()) &&
 986                         (cond_ge.empty() || def >= cond_ge.to_int()) &&
 987                         (cond_le.empty() || def <= cond_le.to_int());
 988         }
 989         return true;
 990 }
 991 
 992 effect::effect(const unit_ability_list& list, int def, bool backstab) :
 993         effect_list_(),
 994         composite_value_(0)
 995 {
 996 
 997         int value_set = def;
 998         bool value_is_set = false;
 999         std::map<std::string,individual_effect> values_add;
1000         std::map<std::string,individual_effect> values_mul;
1001         std::map<std::string,individual_effect> values_div;
1002 
1003         individual_effect set_effect;
1004 
1005         for (const unit_ability & ability : list) {
1006                 const config& cfg = *ability.first;
1007                 const std::string& effect_id = cfg[cfg["id"].empty() ? "name" : "id"];
1008 
1009                 if (!cfg["backstab"].blank()) {
1010                         lg::wml_error() << "The backstab= key in weapon specials is deprecated; use [filter_adjacent] instead\n";
1011                 }
1012 
1013                 if (!backstab && cfg["backstab"].to_bool())
1014                         continue;
1015                 if (!filter_base_matches(cfg, def))
1016                         continue;
1017 
1018                 if (const config::attribute_value *v = cfg.get("value")) {
1019                         int value = *v;
1020                         bool cumulative = cfg["cumulative"].to_bool();
1021                         if (!value_is_set && !cumulative) {
1022                                 value_set = value;
1023                                 set_effect.set(SET, value, ability.first, ability.second);
1024                         } else {
1025                                 if (cumulative) value_set = std::max<int>(value_set, def);
1026                                 if (value > value_set) {
1027                                         value_set = value;
1028                                         set_effect.set(SET, value, ability.first, ability.second);
1029                                 }
1030                         }
1031                         value_is_set = true;
1032                 }
1033 
1034                 if (const config::attribute_value *v = cfg.get("add")) {
1035                         int add = *v;
1036                         std::map<std::string,individual_effect>::iterator add_effect = values_add.find(effect_id);
1037                         if(add_effect == values_add.end() || add > add_effect->second.value) {
1038                                 values_add[effect_id].set(ADD, add, ability.first, ability.second);
1039                         }
1040                 }
1041                 if (const config::attribute_value *v = cfg.get("sub")) {
1042                         int sub = - *v;
1043                         std::map<std::string,individual_effect>::iterator sub_effect = values_add.find(effect_id);
1044                         if(sub_effect == values_add.end() || sub > sub_effect->second.value) {
1045                                 values_add[effect_id].set(ADD, sub, ability.first, ability.second);
1046                         }
1047                 }
1048                 if (const config::attribute_value *v = cfg.get("multiply")) {
1049                         int multiply = int(v->to_double() * 100);
1050                         std::map<std::string,individual_effect>::iterator mul_effect = values_mul.find(effect_id);
1051                         if(mul_effect == values_mul.end() || multiply > mul_effect->second.value) {
1052                                 values_mul[effect_id].set(MUL, multiply, ability.first, ability.second);
1053                         }
1054                 }
1055                 if (const config::attribute_value *v = cfg.get("divide")) {
1056                         if (*v == 0) {
1057                                 ERR_NG << "division by zero with divide= in ability/weapon special " << effect_id << std::endl;
1058                         }
1059                         else {
1060                                 int divide = int(v->to_double() * 100);
1061                                 std::map<std::string,individual_effect>::iterator div_effect = values_div.find(effect_id);
1062                                 if(div_effect == values_div.end() || divide > div_effect->second.value) {
1063                                         values_div[effect_id].set(DIV, divide, ability.first, ability.second);
1064                                 }
1065                         }
1066                 }
1067         }
1068 
1069         if(value_is_set && set_effect.type != NOT_USED) {
1070                 effect_list_.push_back(set_effect);
1071         }
1072 
1073         /* Do multiplication with floating point values rather than integers
1074          * We want two places of precision for each multiplier
1075          * Using integers multiplied by 100 to keep precision causes overflow
1076          *   after 3-4 abilities for 32-bit values and ~8 for 64-bit
1077          * Avoiding the overflow by dividing after each step introduces rounding errors
1078          *   that may vary depending on the order effects are applied
1079          * As the final values are likely <1000 (always true for mainline), loss of less significant digits is not an issue
1080          */
1081         double multiplier = 1.0;
1082         double divisor = 1.0;
1083         std::map<std::string,individual_effect>::const_iterator e, e_end;
1084         for (e = values_mul.begin(), e_end = values_mul.end(); e != e_end; ++e) {
1085                 multiplier *= e->second.value/100.0;
1086                 effect_list_.push_back(e->second);
1087         }
1088         for (e = values_div.begin(), e_end = values_div.end(); e != e_end; ++e) {
1089                 divisor *= e->second.value/100.0;
1090                 effect_list_.push_back(e->second);
1091         }
1092         int addition = 0;
1093         for (e = values_add.begin(), e_end = values_add.end(); e != e_end; ++e) {
1094                 addition += e->second.value;
1095                 effect_list_.push_back(e->second);
1096         }
1097 
1098         composite_value_ = int((value_set + addition) * multiplier / divisor);
1099 }
