  41 # The amount of extra recruit cost is kept in $recruit_cost_extra; when this
  42 # macro is called for the first time, it'll disallow recruitment of
  43 # "Desert Fighter" and then allow recruitment of "Desert FighterX". The second
  44 # call then disallows recruitment of "Desert FighterY" and allows recruitment of
  45 # "Desert Fighter(X+Y)", and so on. After $recruit_cost_extra has reached 10,
  46 # nothing will happen anymore.
  47 
  48 #define INCREASE_RECRUIT_COSTS X
  49     {VARIABLE num $recruit_cost_extra}
  50     {VARIABLE_OP recruit_cost_extra add {X}}
  51     [if]
  52         [variable]
  53             name=recruit_cost_extra
  54             greater_than=10
  55         [/variable]
  56         [then]
  57             {VARIABLE recruit_cost_extra 10}
  58         [/then]
  59     [/if]
  60 
  61 #ifdef UTBSNEW
  62     #{DEBUG_MSG "disallowing: Quenoth Fighter$num,Tauroch Rider$num,Quenoth Scout$num,Quenoth Mystic$num"}
  63     [disallow_recruit]
  64         type=Quenoth Fighter$num,Tauroch Rider$num,Quenoth Scout$num,Quenoth Mystic$num
  65         side=1
  66     [/disallow_recruit]
  67 
  68     {VARIABLE num $recruit_cost_extra}

  70     [allow_recruit]
  71         type=Quenoth Fighter$num,Tauroch Rider$num,Quenoth Scout$num,Quenoth Mystic$num

  74 #else

  78         side=1

 107 # To give the human units in scenarios 8 and 9 a more unique look, we shift
 108 # the blue steel colors to more of a bronze hue. Because the blue colors don't
 109 # adhere to an exact consistent palette, we can't use ~RC() for this but instead
 110 # we need to do it with a rather hacky formula.
 111 
 112 #define ARMOR_SHIFT NEWCHAN ORIGCHAN
 113 if(red <= green and green <= blue + 3, ({NEWCHAN} / 255.0)^2 * 255.0, {ORIGCHAN})#enddef
 114 
 115 #define ARMOR_SHIFT_IPF
 116 CHAN({ARMOR_SHIFT blue red}, {ARMOR_SHIFT green green}, {ARMOR_SHIFT red blue})#enddef
 117

 128             [/not]

 136                     [/modifications]
 137                 [/filter_wml]
 138             [/not]
 139         [/filter]
 140

 142             armor_color_shift=yes
 143             silent=yes
 144

 151 #enddef
