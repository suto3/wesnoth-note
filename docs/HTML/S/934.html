 121 /**
 122  * Ensures that *this manages its own memory, making it safe for *this to
 123  * outlive the config it was ultimately constructed from.
 124  * It is perfectly safe to call this for a vconfig that already manages its memory.
 125  * This does not work on a null() vconfig.
 126  */
 127 void vconfig::make_safe() const
 128 {
 129         // Nothing to do if we already manage our own memory.
 130         if ( memory_managed() )
 131                 return;
 132 
 133         // Make a copy of our config.
 134         cache_.reset(new config(*cfg_));
 135         // Use our copy instead of the original.
 136         cfg_ = cache_.get();
 137 }

 139 config vconfig::get_parsed_config() const
 140 {
 141         // Keeps track of insert_tag variables.
 142         static std::set<std::string> vconfig_recursion;
 143 
 144         config res;
 145 
 146         for (const config::attribute &i : cfg_->attribute_range()) {
 147                 res[i.first] = expand(i.first);
 148         }
 149 
 150         for (const config::any_child &child : cfg_->all_children_range())
 151         {
 152                 if (child.key == "insert_tag") {
 153                         vconfig insert_cfg(child.cfg);
 154                         std::string name = insert_cfg["name"];
 155                         std::string vname = insert_cfg["variable"];
 156                         if(!vconfig_recursion.insert(vname).second) {
 157                                 throw recursion_error("vconfig::get_parsed_config() infinite recursion detected, aborting");
 158                         }
 159                         try
 160                         {
 161                                 config::const_child_itors range = as_nonempty_range(vname);
 162                                 for (const config& ch : range)
 163                                 {
 164                                         res.add_child(name, vconfig(ch).get_parsed_config());
 165                                 }
 166                         }
 167                         catch(const invalid_variablename_exception&)
 168                         {
 169                                 res.add_child(name);
 170                         }
 171                         catch(recursion_error &err) {
 172                                 vconfig_recursion.erase(vname);
 173                                 WRN_NG << err.message << std::endl;
 174                                 if(vconfig_recursion.empty()) {
 175                                         res.add_child("insert_tag", insert_cfg.get_config());
 176                                 } else {
 177                                         // throw to the top [insert_tag] which started the recursion
 178                                         throw;
 179                                 }
 180                         }
 181                         vconfig_recursion.erase(vname);
 182                 } else {
 183                         res.add_child(child.key, vconfig(child.cfg).get_parsed_config());
 184                 }
 185         }
 186         return res;
 187 }

 189 vconfig::child_list vconfig::get_children(const std::string& key) const
 190 {
 191         vconfig::child_list res;
 192 
 193         for (const config::any_child &child : cfg_->all_children_range())
 194         {
 195                 if (child.key == key) {
 196                         res.push_back(vconfig(child.cfg, cache_));
 197                 } else if (child.key == "insert_tag") {
 198                         vconfig insert_cfg(child.cfg);
 199                         if(insert_cfg["name"] == key)
 200                         {
 201                                 try
 202                                 {
 203                                         config::const_child_itors range = as_nonempty_range(insert_cfg["variable"]);
 204                                         for (const config& ch : range)
 205                                         {
 206                                                 res.push_back(vconfig(ch, true));
 207                                         }
 208                                 }
 209                                 catch(const invalid_variablename_exception&)
 210                                 {
 211                                         res.push_back(empty_vconfig());
 212                                 }
 213                         }
 214                 }
 215         }
 216         return res;
 217 }

 219 size_t vconfig::count_children(const std::string& key) const
 220 {
 221         size_t n = 0;
 222 
 223         for (const config::any_child &child : cfg_->all_children_range())
 224         {
 225                 if (child.key == key) {
 226                         n++;
 227                 } else if (child.key == "insert_tag") {
 228                         vconfig insert_cfg(child.cfg);
 229                         if(insert_cfg["name"] == key)
 230                         {
 231                                 try
 232                                 {
 233                                         config::const_child_itors range = as_nonempty_range(insert_cfg["variable"]);
 234                                         n += range.size();
 235                                 }
 236                                 catch(const invalid_variablename_exception&)
 237                                 {
 238                                         n++;
 239                                 }
 240                         }
 241                 }
 242         }

 246 /**
 247  * Returns a child of *this whose key is @a key.
 248  * If no such child exists, returns an unconstructed vconfig (use null() to test
 249  * for this).
 250  */
 251 vconfig vconfig::child(const std::string& key) const
 252 {
 253         if (const config &natural = cfg_->child(key)) {
 254                 return vconfig(natural, cache_);
 255         }
 256         for (const config &ins : cfg_->child_range("insert_tag"))
 257         {
 258                 vconfig insert_cfg(ins);
 259                 if(insert_cfg["name"] == key)
 260                 {
 261                         try
 262                         {
 263                                 config::const_child_itors range = as_nonempty_range(insert_cfg["variable"]);
 264                                 return vconfig(range.front(), true);
 265                         }
 266                         catch(const invalid_variablename_exception&)
 267                         {
 268                                 return empty_vconfig();
 269                         }
 270                 }
 271         }
 272         return unconstructed_vconfig();
 273 }

 275 /**
 276  * Returns whether or not *this has a child whose key is @a key.
 277  */
 278 bool vconfig::has_child(const std::string& key) const
 279 {
 280         if (cfg_->child(key)) {
 281                 return true;
 282         }
 283         for (const config &ins : cfg_->child_range("insert_tag"))
 284         {
 285                 vconfig insert_cfg(ins);
 286                 if(insert_cfg["name"] == key) {
 287                         return true;
 288                 }
 289         }
 290         return false;
 291 }

 463 config &scoped_wml_variable::store(const config &var_value)
 464 {
 465         try
 466         {
 467                 for (const config &i : resources::gamedata->get_variables().child_range(var_name_)) {
 468                         previous_val_.add_child(var_name_, i);
 469                 }
 470                 resources::gamedata->clear_variable_cfg(var_name_);
 471                 config &res = resources::gamedata->add_variable_cfg(var_name_, var_value);
 472                 LOG_NG << "scoped_wml_variable: var_name \"" << var_name_ << "\" has been auto-stored.\n";
 473                 activated_ = true;
 474                 return res;
 475         }
 476         catch(const invalid_variablename_exception&)
 477         {
 478                 assert(false && "invalid variable name of autostored varaible");
 479                 throw "assertion ignored";
 480         }
 481 
 482 }

 484 scoped_wml_variable::~scoped_wml_variable()
 485 {
 486         if(!resources::gamedata) {
 487                 return;
 488         }
 489 
 490         if(activated_) {
 491                 resources::gamedata->clear_variable_cfg(var_name_);
 492                 for(const config &i : previous_val_.child_range(var_name_))
 493                 {
 494                         try
 495                         {
 496                                 resources::gamedata->add_variable_cfg(var_name_, i);
 497                         }
 498                         catch(const invalid_variablename_exception&)
 499                         {
 500                         }
 501                 }
 502                 LOG_NG << "scoped_wml_variable: var_name \"" << var_name_ << "\" has been reverted.\n";
 503         }
 504 
 505         assert(resources::gamedata->scoped_variables.back() == this);
 506         resources::gamedata->scoped_variables.pop_back();
 507 }

 530 void scoped_recall_unit::activate()
 531 {
 532         assert(resources::gameboard);
 533 
 534         const std::vector<team>& teams = resources::gameboard->teams();
 535 
 536         std::vector<team>::const_iterator team_it = std::find_if(teams.begin(), teams.end(), [&](const team& t) { return t.save_id() == player_; });
 537 
 538         if(team_it != teams.end()) {
 539                 if(team_it->recall_list().size() > recall_index_) {
 540                         config &tmp_cfg = store();
 541                         team_it->recall_list()[recall_index_]->write(tmp_cfg);
 542                         tmp_cfg["x"] = "recall";
 543                         tmp_cfg["y"] = "recall";
 544                         LOG_NG << "auto-storing $" << name() << " for player: " << player_
 545                                 << " at recall index: " << recall_index_ << '\n';
 546                 } else {
 547                         ERR_NG << "failed to auto-store $" << name() << " for player: " << player_
 548                                 << " at recall index: " << recall_index_ << '\n';
 549                 }
 550         } else {
 551                 ERR_NG << "failed to auto-store $" << name() << " for player: " << player_ << '\n';
