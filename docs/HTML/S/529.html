 117         /**
 118          * A helper class to find out whether dispatcher has an handler for a
 119          * certain event.
 120          */
 121         class has_handler
 122         {
 123         public:
 124                 /**
 125                  * Constructor.
 126                  *
 127                  * @param event_type      The type of event to look for.
 128                  * @param dispatcher      The dispatcher whose signal queue is used.
 129                  */
 130                 has_handler(const dispatcher::event_queue_type event_type, dispatcher& dispatcher)
 131                         : event_type_(event_type), dispatcher_(dispatcher)
 132                 {
 133                 }
 134 
 135                 /**
 136                  * Tests whether a handler for an event is available.
 137                  *
 138                  * It tests for both the event and the event_type send in the
 139                  * constructor.
 140                  *
 141                  * @tparam T              A key from an event set used to instantiate
 142                  *                        the proper @p event_signal function.
 143                  * @param event           The event to get the signal for.
 144                  *
 145                  * @returns               Whether or not the handler is found.
 146                  */
 147                 // not called operator() to work around a problem in MSVC
 148                 // (known to affect all versions up to 2015)
 149                 template<typename T>
 150                 bool oper(ui_event event)
 151                 {
 152                         if((event_type_ & dispatcher::pre)
 153                            && !event_signal<T>(dispatcher_, event).pre_child.empty()) {
 154                                 return true;
 155                         }
 156 
 157                         if((event_type_ & dispatcher::child)
 158                            && !event_signal<T>(dispatcher_, event).child.empty()) {
 159                                 return true;
 160                         }
 161 
 162                         if((event_type_ & dispatcher::post)
 163                            && !event_signal<T>(dispatcher_, event).post_child.empty()) {
 164                                 return true;
 165                         }
 166 
 167                         return false;
 168                 }
 169 
 170         private:
 171                 dispatcher::event_queue_type event_type_;
 172                 dispatcher& dispatcher_;
 173         };
 174 };

 180 /** Specialized class when itor == end */
 181 template<bool done = true>
 182 struct find
 183 {
 184         template<typename itor, typename end, typename E, typename F>
 185         static bool execute(itor*, end*, E, F)
 186         {
 187                 return false;
 188         }
 189 };
 190 
 191 /** Specialized class when itor != end */
 192 template<>
 193 struct find<false>
 194 {
 195         template<typename itor, typename end, typename E, typename F>
 196         static bool execute(itor*, end*, E event, F functor)
 197         {
 198                 typedef typename boost::mpl::deref<itor>::type item;
 199                 typedef typename boost::mpl::apply1<boost::mpl::identity<>, item>::type arg;
 200 
 201                 boost::value_initialized<arg> x;
 202 
 203                 if(boost::get(x) == event) {
 204                         return functor.template oper<item>(event);
 205                 } else {
 206                         typedef typename boost::mpl::next<itor>::type itor_t;
 207                         return find<std::is_same<itor_t, end>::value>::execute(
 208                                         static_cast<itor_t*>(nullptr),
 209                                         static_cast<end*>(nullptr),
 210                                         event,
 211                                         functor);
 212                 }
 213         }
 214 };

 218 /**
 219  * Tests whether an event handler is available.
 220  *
 221  * The code is based on boost::mpl_for_each, which doesn't allow to call a
 222  * template function with the dereferred iterator as template parameter.
 223  *
 224  * The function first tries to match whether the value in the sequence matches
 225  * event, once that matched it will execute the functor with the key found as
 226  * template parameter and the event as parameter.
 227  *
 228  * @tparam sequence               The sequence to test upon.
 229  * @tparam E                      The value type of the item in the sequence
 230  * @tparam F                      Type of the functor.
 231  *
 232  * @param event                   The event to look for.
 233  * @param functor                 The predicate which should is executed if the
 234  *                                event is matched.
 235  *
 236  * @returns                       Whether or not the function found a result.
 237  */
 238 template<typename sequence, typename E, typename F>
 239 inline bool find(E event, F functor)
 240 {
 241         typedef typename boost::mpl::begin<sequence>::type begin;
 242         typedef typename boost::mpl::end<sequence>::type end;
 243 
 244         return implementation::find<std::is_same<begin, end>::value>::execute(
 245                 static_cast<begin*>(nullptr), static_cast<end*>(nullptr), event, functor);
 246 }

 259  *  | | --------------- | |

 268  * Note that the firing routine fires the events from:

 271  * - post child for chain.begin() -> chain.end()

 274 /**
 275  * Build the event chain.
 276  *
 277  * The event chain is a chain of events starting from the first parent of the
 278  * widget until (and including) the wanted parent. For all these widgets it
 279  * will be tested whether they have either a pre or post handler for the event.
 280  * This ways there will be list of widgets to try to send the events to.
 281  * If there's no line from widget to parent the result is undefined.
 282  * (If widget == dispatcher the result will always be empty.)
 283  *
 284  * @pre                           dispatcher != nullptr
 285  * @pre                           widget != nullptr
 286  *
 287  * @param event                   The event to test.
 288  * @param dispatcher              The final widget to test, this is also the
 289  *                                dispatcher the sends the event.
 290  * @param widget                  The widget should parent(s) to check.
 291  *
 292  * @returns                       The list of widgets with a handler.
 293  *                                The order will be (assuming all have a
 294  *                                handler):
 295  *                                * container 2
 296  *                                * container 1
 297  *                                * dispatcher
 298  */
 299 template<typename T>
 300 inline std::vector<std::pair<widget*, ui_event>>
 301 build_event_chain(const ui_event event, widget* dispatcher, widget* w)
 302 {
 303         assert(dispatcher);
 304         assert(w);
 305 
 306         std::vector<std::pair<widget*, ui_event>> result;
 307 
 308         while(w != dispatcher) {
 309                 w = w->parent();
 310                 assert(w);
 311 
 312                 if(w->has_event(event, dispatcher::event_queue_type(dispatcher::pre | dispatcher::post))) {
 313                         result.emplace_back(w, event);
 314                 }
 315         }
 316 
 317         return result;
 318 }

 320 /**
 321  * Build the event chain for signal_notification_function.
 322  *
 323  * The notification is only send to the receiver it returns an empty chain.
 324  * Since the pre and post queues are unused, it validates whether they are
 325  * empty (using asserts).
 326  *
 327  * @returns                       An empty vector.
 328  */
 329 template<>
 330 inline std::vector<std::pair<widget*, ui_event>>
 331 build_event_chain<signal_notification_function>(const ui_event event, widget* dispatcher, widget* w)
 332 {
 333         assert(dispatcher);
 334         assert(w);
 335 
 336         assert(!w->has_event(event, dispatcher::event_queue_type(dispatcher::pre | dispatcher::post)));
 337 
 338         return std::vector<std::pair<widget*, ui_event>>();
 339 }

 345 /**
 346  * Build the event chain for signal_message_function.
 347  *
 348  * This function expects that the widget sending it is also the receiver. This
 349  * assumption might change, but is valid for now. The function doesn't build an
 350  * event chain from @p dispatcher to @p widget but from @p widget to its
 351  * toplevel item (the first one without a parent) which we call @p window.
 352  *
 353  * @pre                           dispatcher == widget
 354  *
 355  * @returns                       The list of widgets with a handler.
 356  *                                The order will be (assuming all have a
 357  *                                handler):
 358  *                                * window
 359  *                                * container 1
 360  *                                * container 2
 361  */
 362 template<>
 363 inline std::vector<std::pair<widget*, ui_event>>
 364 build_event_chain<signal_message_function>(const ui_event event, widget* dispatcher, widget* w)
 365 {
 366         assert(dispatcher);
 367         assert(w);
 368         assert(w == dispatcher);
 369 
 370         std::vector<std::pair<widget*, ui_event>> result;
 371 
 372         /* We only should add the parents of the widget to the chain. */
 373         while((w = w->parent())) {
 374                 assert(w);
 375 
 376                 if(w->has_event(event, dispatcher::event_queue_type(dispatcher::pre | dispatcher::post))) {
 377                         result.emplace(result.begin(), w, event);
 378                 }
 379         }
 380 
 381         return result;
 382 }

 387 /**
 388  * Helper function for fire_event.
 389  *
 390  * This is called with the same parameters as fire_event except for the
 391  * event_chain, which contains the widgets with the events to call for them.
 392  */
 393 template<typename T, typename... F>
 394 inline bool fire_event(const ui_event event,
 395                                            std::vector<std::pair<widget*, ui_event>>& event_chain,
 396                                            widget* dispatcher,
 397                                            widget* w,
 398                                            F&&... params)
 399 {
 400         bool handled = false;
 401         bool halt = false;
 402 
 403         /***** ***** ***** Pre ***** ***** *****/
 404         for(auto& ritor_widget : boost::adaptors::reverse(event_chain)) {
 405                 dispatcher::signal_type<T>& signal =
 406                         dispatcher_implementation::event_signal<T>(*ritor_widget.first, ritor_widget.second);
 407 
 408                 for(auto& pre_func : signal.pre_child) {
 409                         pre_func(*dispatcher, ritor_widget.second, handled, halt, std::forward<F>(params)...);
 410 
 411                         if(halt) {
 412                                 assert(handled);
 413                                 break;
 414                         }
 415                 }
 416 
 417                 if(handled) {
 418                         return true;
 419                 }
 420         }
 421 
 422         /***** ***** ***** Child ***** ***** *****/
 423         if(w->has_event(event, dispatcher::child)) {
 424                 dispatcher::signal_type<T>& signal =
 425                         dispatcher_implementation::event_signal<T>(*w, event);
 426 
 427                 for(auto& func : signal.child) {
 428                         func(*dispatcher, event, handled, halt, std::forward<F>(params)...);
 429 
 430                         if(halt) {
 431                                 assert(handled);
 432                                 break;
 433                         }
 434                 }
 435 
 436                 if(handled) {
 437                         return true;
 438                 }
 439         }
 440 
 441         /***** ***** ***** Post ***** ***** *****/
 442         for(auto& itor_widget : event_chain) {
 443                 dispatcher::signal_type<T>& signal =
 444                         dispatcher_implementation::event_signal<T>(*itor_widget.first, itor_widget.second);
 445 
 446                 for(auto& post_func : signal.post_child) {
 447                         post_func(*dispatcher, itor_widget.second, handled, halt, std::forward<F>(params)...);
 448 
 449                         if(halt) {
 450                                 assert(handled);
 451                                 break;
 452                         }
 453                 }
 454 
 455                 if(handled) {
 456                         return true;
 457                 }
 458         }
 459 
 460         /**** ***** ***** Unhandled ***** ***** *****/
 461         assert(handled == false);
 462         return false;
 463 }

 467 /**
 468  * Fires an event.
 469  *
 470  * A helper to allow the common event firing code to be shared between the
 471  * different signal function types.
 472  *
 473  * @pre                           d != nullptr
 474  * @pre                           w != nullptr
 475  *
 476  * @tparam T                      The signal type of the event to handle.
 477  * @tparam F                      The paramater pack type.
 478  *
 479  *
 480  * @param event                   The event to fire.
 481  * @param d                       The dispatcher that handles the event.
 482  * @param w                       The widget that should receive the event.
 483  * @param params                  Zero or more additional arguments to pass
 484  *                                to the signal function when it's executed.
 485  *
 486  * @returns                       Whether or not the event was handled.
 487  */
 488 template<typename T, typename... F>
 489 inline bool
 490 fire_event(const ui_event event, dispatcher* d, widget* w, F&&... params)
 491 {
 492         assert(d);
 493         assert(w);
 494 
 495         widget* dispatcher_w = dynamic_cast<widget*>(d);
 496 
 497         std::vector<std::pair<widget*, ui_event>> event_chain =
 498                 implementation::build_event_chain<T>(event, dispatcher_w, w);
 499 
 500         return implementation::fire_event<T>(event, event_chain, dispatcher_w, w, std::forward<F>(params)...);
 501 }

 503 template<ui_event click,
 504                   ui_event double_click,
 505                   bool(event_executor::*wants_double_click)() const,
 506                   typename T,
 507                   typename... F>
 508 inline bool
 509 fire_event_double_click(dispatcher* dsp, widget* wgt, F&&... params)
 510 {
 511         assert(dsp);
 512         assert(wgt);
 513 
 514         std::vector<std::pair<widget*, ui_event>> event_chain;
 515         widget* w = wgt;
 516         widget* d = dynamic_cast<widget*>(dsp);
 517 
 518         while(w != d) {
 519                 w = w->parent();
 520                 assert(w);
 521 
 522                 if((w->*wants_double_click)()) {
 523                         if(w->has_event(double_click, dispatcher::event_queue_type(dispatcher::pre | dispatcher::post))) {
 524                                 event_chain.emplace_back(w, double_click);
 525                         }
 526                 } else {
 527                         if(w->has_event(click, dispatcher::event_queue_type(dispatcher::pre | dispatcher::post))) {
 528                                 event_chain.emplace_back(w, click);
 529                         }
 530                 }
 531         }
 532 
 533         if((wgt->*wants_double_click)()) {
 534                 return implementation::fire_event<T>(
 535                         double_click, event_chain, d, wgt, std::forward<F>(params)...);
 536         } else {
 537                 return implementation::fire_event<T>(
 538                         click, event_chain, d, wgt, std::forward<F>(params)...);
 539         }
 540 }
