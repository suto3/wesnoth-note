 167 void halo_impl::effect::set_location(int x, int y)
 168 {
 169         int new_x = x - disp->get_location_x(map_location::ZERO());
 170         int new_y = y - disp->get_location_y(map_location::ZERO());
 171         if (new_x != x_ || new_y != y_) {
 172                 unrender();
 173                 x_ = new_x;
 174                 y_ = new_y;
 175                 buffer_.assign(nullptr);
 176                 overlayed_hexes_.clear();
 177         }
 178 }

 180 bool halo_impl::effect::render()
 181 {
 182         if(disp == nullptr) {
 183                 return false;
 184         }
 185 
 186         if(loc_.x != -1 && loc_.y != -1) {
 187                 if(disp->shrouded(loc_)) {
 188                         return false;
 189                 } else {
 190                         // The location of a halo is an x,y value and not a map location.
 191                         // This means when a map is zoomed, the halo's won't move,
 192                         // This glitch is most visible on [item] haloes.
 193                         // This workaround always recalculates the location of the halo
 194                         // (item haloes have a location parameter to hide them under the shroud)
 195                         // and reapplies that location.
 196                         // It might be optimized by storing and comparing the zoom value.
 197                         set_location(
 198                                 disp->get_location_x(loc_) + disp->hex_size() / 2,
 199                                 disp->get_location_y(loc_) + disp->hex_size() / 2);
 200                 }
 201         }
 202 
 203         images_.update_last_draw_time();
 204         surf_.assign(image::get_image(current_image(),image::SCALED_TO_ZOOM));
 205         if(surf_ == nullptr) {
 206                 return false;
 207         }
 208         if(orientation_ == HREVERSE || orientation_ == HVREVERSE) {
 209                 surf_.assign(image::reverse_image(surf_));
 210         }
 211         if(orientation_ == VREVERSE || orientation_ == HVREVERSE) {
 212                 surf_.assign(flop_surface(surf_));
 213         }
 214 
 215         const int screenx = disp->get_location_x(map_location::ZERO());
 216         const int screeny = disp->get_location_y(map_location::ZERO());
 217 
 218         const int xpos = x_ + screenx - surf_->w/2;
 219         const int ypos = y_ + screeny - surf_->h/2;
 220 
 221         SDL_Rect rect {xpos, ypos, surf_->w, surf_->h};
 222         rect_ = rect;
 223         SDL_Rect clip_rect = disp->map_outside_area();
 224 
 225         // If rendered the first time, need to determine the area affected.
 226         // If a halo changes size, it is not updated.
 227         if(location_not_known()) {
 228                 display::rect_of_hexes hexes = disp->hexes_under_rect(rect);
 229                 display::rect_of_hexes::iterator i = hexes.begin(), end = hexes.end();
 230                 for (;i != end; ++i) {
 231                         overlayed_hexes_.push_back(*i);
 232                 }
 233         }
 234 
 235         if(sdl::rects_overlap(rect,clip_rect) == false) {
 236                 buffer_.assign(nullptr);
 237                 return false;
 238         }
 239 
 240         surface& screen = disp->get_screen_surface();
 241 
 242         const clip_rect_setter clip_setter(screen, &clip_rect);
 243         if(buffer_ == nullptr || buffer_->w != rect.w || buffer_->h != rect.h) {
 244                 SDL_Rect rect2 = rect_;
 245                 buffer_.assign(get_surface_portion(screen,rect2));
 246         } else {
 247                 SDL_Rect rect2 = rect_;
 248                 sdl_copy_portion(screen,&rect2,buffer_,nullptr);
 249         }
 250 
 251         sdl_blit(surf_,nullptr,screen,&rect);
 252 
 253         return true;
 254 }

 256 void halo_impl::effect::unrender()
 257 {
 258         if (!surf_ || !buffer_) {
 259                 return;
 260         }
 261 
 262         // Shrouded haloes are never rendered unless shroud has been re-placed; in
 263         // that case, unrendering causes the hidden terrain (and previous halo
 264         // frame, when dealing with animated halos) to glitch through shroud. We
 265         // don't need to unrender them because shroud paints over the underlying
 266         // area anyway.
 267         if (loc_.x != -1 && loc_.y != -1 && disp->shrouded(loc_)) {
 268                 return;
 269         }
 270 
 271         surface& screen = disp->get_screen_surface();
 272 
 273         SDL_Rect clip_rect = disp->map_outside_area();
 274         const clip_rect_setter clip_setter(screen, &clip_rect);
 275 
 276         // Due to scrolling, the location of the rendered halo
 277         // might have changed; recalculate
 278         const int screenx = disp->get_location_x(map_location::ZERO());
 279         const int screeny = disp->get_location_y(map_location::ZERO());
 280 
 281         const int xpos = x_ + screenx - surf_->w/2;
 282         const int ypos = y_ + screeny - surf_->h/2;
 283 
 284         SDL_Rect rect {xpos, ypos, surf_->w, surf_->h};
 285         sdl_blit(buffer_,nullptr,screen,&rect);
 286 }

 288 bool halo_impl::effect::on_location(const std::set<map_location>& locations) const
 289 {
 290         for(std::vector<map_location>::const_iterator itor = overlayed_hexes_.begin();
 291                         itor != overlayed_hexes_.end(); ++itor) {
 292                 if(locations.find(*itor) != locations.end()) {
 293                         return true;
 294                 }
 295         }
 296         return false;
 297 }

 315 int halo_impl::add(int x, int y, const std::string& image, const map_location& loc,
 316                 ORIENTATION orientation, bool infinite)
 317 {
 318         const int id = halo_id++;
 319         animated<image::locator>::anim_description image_vector;
 320         std::vector<std::string> items = utils::square_parenthetical_split(image, ',');
 321 
 322         for(const std::string& item : items) {
 323                 const std::vector<std::string>& sub_items = utils::split(item, ':');
 324                 std::string str = item;
 325                 int time = 100;
 326 
 327                 if(sub_items.size() > 1) {
 328                         str = sub_items.front();
 329                         try {
 330                                 time = std::stoi(sub_items.back());
 331                         } catch(std::invalid_argument) {
 332                                 ERR_DP << "Invalid time value found when constructing halo: " << sub_items.back() << "\n";
 333                         }
 334                 }
 335                 image_vector.push_back(animated<image::locator>::frame_description(time,image::locator(str)));
 336 
 337         }
 338         haloes.emplace(id, effect(disp, x, y, image_vector, loc, orientation, infinite));
 339         new_haloes.insert(id);
 340         if(haloes.find(id)->second.does_change() || !infinite) {
 341                 changing_haloes.insert(id);
 342         }
 343         return id;
 344 }

 354 void halo_impl::remove(int handle)
 355 {
 356         // Silently ignore invalid haloes.
 357         // This happens when Wesnoth is being terminated as well.
 358         if(handle == NO_HALO || haloes.find(handle) == haloes.end())  {
 359                 return;
 360         }
 361 
 362         deleted_haloes.insert(handle);
 363 }

 365 void halo_impl::unrender(std::set<map_location> invalidated_locations)
 366 {
 367         if(preferences::show_haloes() == false || haloes.empty()) {
 368                 return;
 369         }
 370         //assert(invalidated_haloes.empty());
 371 
 372         // Remove expired haloes
 373         std::map<int, effect>::iterator itor = haloes.begin();
 374         for(; itor != haloes.end(); ++itor ) {
 375                 if(itor->second.expired()) {
 376                         deleted_haloes.insert(itor->first);
 377                 }
 378         }
 379 
 380         // Add the haloes marked for deletion to the invalidation set
 381         std::set<int>::const_iterator set_itor = deleted_haloes.begin();
 382         for(;set_itor != deleted_haloes.end(); ++set_itor) {
 383                 invalidated_haloes.insert(*set_itor);
 384                 haloes.find(*set_itor)->second.add_overlay_location(invalidated_locations);
 385         }
 386 
 387         // Test the multi-frame haloes whether they need an update
 388         for(set_itor = changing_haloes.begin();
 389                         set_itor != changing_haloes.end(); ++set_itor) {
 390                 if(haloes.find(*set_itor)->second.need_update()) {
 391                         invalidated_haloes.insert(*set_itor);
 392                         haloes.find(*set_itor)->second.add_overlay_location(invalidated_locations);
 393                 }
 394         }
 395 
 396         // Find all halo's in a the invalidated area
 397         size_t halo_count;
 398 
 399         // Repeat until set of haloes in the invalidated area didn't change
 400         // (including none found) or all existing haloes are found.
 401         do {
 402                 halo_count = invalidated_haloes.size();
 403                 for(itor = haloes.begin(); itor != haloes.end(); ++itor) {
 404                         // Test all haloes not yet in the set
 405                         // which match one of the locations
 406                         if(invalidated_haloes.find(itor->first) == invalidated_haloes.end() &&
 407                                         (itor->second.location_not_known() ||
 408                                         itor->second.on_location(invalidated_locations))) {
 409 
 410                                 // If found, add all locations which the halo invalidates,
 411                                 // and add it to the set
 412                                 itor->second.add_overlay_location(invalidated_locations);
 413                                 invalidated_haloes.insert(itor->first);
 414                         }
 415                 }
 416         } while (halo_count != invalidated_haloes.size() && halo_count != haloes.size());
 417 
 418         if(halo_count == 0) {
 419                 return;
 420         }
 421 
 422         // Render the haloes:
 423         // iterate through all the haloes and invalidate if in set
 424         for(std::map<int, effect>::reverse_iterator ritor = haloes.rbegin(); ritor != haloes.rend(); ++ritor) {
 425                 if(invalidated_haloes.find(ritor->first) != invalidated_haloes.end()) {
 426                         ritor->second.unrender();
 427                 }
 428         }
 429 
 430         // Really delete the haloes marked for deletion
 431         for(set_itor = deleted_haloes.begin(); set_itor != deleted_haloes.end(); ++set_itor) {
 432                 // It can happen a deleted halo hasn't been rendered yet, invalidate them as well
 433                 new_haloes.erase(*set_itor);
 434 
 435                 changing_haloes.erase(*set_itor);
 436                 invalidated_haloes.erase(*set_itor);
 437                 haloes.erase(*set_itor);
 438         }
 439 
 440         deleted_haloes.clear();
 441 }

 443 void halo_impl::render()
 444 {
 445         if(preferences::show_haloes() == false || haloes.empty() ||
 446                         (new_haloes.empty() && invalidated_haloes.empty())) {
 447                 return;
 448         }
 449 
 450         // Keep track of not rendered new images they have to be kept scheduled
 451         // for rendering otherwise the invalidation area is never properly set
 452         std::set<int> unrendered_new_haloes;
 453 
 454         // Render the haloes:
 455         // iterate through all the haloes and draw if in either set
 456         for(std::map<int, effect>::iterator itor = haloes.begin();
 457                         itor != haloes.end(); ++itor) {
 458 
 459                 if(new_haloes.find(itor->first) != new_haloes.end() &&
 460                                 ! itor->second.render()) {
 461 
 462                         unrendered_new_haloes.insert(itor->first);
 463                 } else if(invalidated_haloes.find(itor->first) != invalidated_haloes.end()) {
 464                         itor->second.render();
 465                 }
 466         }
 467 
 468         invalidated_haloes.clear();
 469         new_haloes = unrendered_new_haloes;
 470 }

 532 halo_record::~halo_record()
 533 {
 534         if (!valid()) return;
 535 
 536         std::shared_ptr<halo_impl> man = my_manager_.lock();
 537 
 538         if (man) {
 539                 try {
 540                         man->remove(id_);
 541                 } catch (std::exception & e) {
 542                         std::cerr << "Caught an exception in halo::halo_record destructor: \n" << e.what() << std::endl;
 543                 } catch (...) {}
 544         }
