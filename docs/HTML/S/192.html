  38 class lua_callable : public formula_callable {
  39         lua_State* mState;
  40         int table_i;
  41 public:
  42         lua_callable(lua_State* L, int i) : mState(L), table_i(lua_absindex(L,i)) {}
  43         variant get_value(const std::string& key) const {
  44                 if(key == "__list") {
  45                         std::vector<variant> values;
  46                         size_t n = lua_rawlen(mState, table_i);
  47                         if(n == 0) {
  48                                 return variant();
  49                         }
  50                         for(size_t i = 1; i <= n; i++) {
  51                                 lua_pushinteger(mState, i);
  52                                 lua_gettable(mState, table_i);
  53                                 values.push_back(luaW_tofaivariant(mState, -1));
  54                         }
  55                         return variant(values);
  56                 } else if(key == "__map") {
  57                         std::map<variant,variant> values;
  58                         for(lua_pushnil(mState); lua_next(mState, table_i); lua_pop(mState, 1)) {
  59                                 values[luaW_tofaivariant(mState, -2)] = luaW_tofaivariant(mState, -1);
  60                         }
  61                         return variant(values);
  62                 }
  63                 lua_pushlstring(mState, key.c_str(), key.size());
  64                 lua_gettable(mState, table_i);
  65                 variant result = luaW_tofaivariant(mState, -1);
  66                 lua_pop(mState, 1);
  67                 return result;
  68         }
  69         void get_inputs(formula_input_vector& inputs) const {
  70                 add_input(inputs, "__list");
  71                 add_input(inputs, "__map");
  72                 for(lua_pushnil(mState); lua_next(mState, table_i); lua_pop(mState,1)) {
  73                         if(lua_isstring(mState, -2) && !lua_isnumber(mState, -2)) {
  74                                 std::string key = lua_tostring(mState, -2);
  75                                 if(key.find_first_not_of(formula::id_chars) != std::string::npos) {
  76                                         add_input(inputs, key);
  77                                 }
  78                         }
  79                 }
  80         }
  81         int do_compare(const formula_callable* other) const {
  82                 const lua_callable* lua = dynamic_cast<const lua_callable*>(other);
  83                 if(lua == nullptr) {
  84                         return formula_callable::do_compare(other);
  85                 }
  86                 if(mState == lua->mState) { // Which should always be the case, but let's be safe here
  87                         if(lua_compare(mState, table_i, lua->table_i, LUA_OPEQ)) {
  88                                 return 0;
  89                         }
  90                         int top = lua_gettop(mState);
  91                         if(lua_getmetatable(mState, table_i)) {
  92                                 lua_getfield(mState, -1, "__lt");
  93                                 if(!lua_isnoneornil(mState, -1)) {
  94                                         if(lua_getmetatable(mState, lua->table_i)) {
  95                                                 lua_getfield(mState, -1, "__lt");
  96                                                 if(!lua_isnoneornil(mState, -1)) {
  97                                                         lua_settop(mState, top);
  98                                                         return lua_compare(mState, table_i, lua->table_i, LUA_OPLT) ? -1 : 1;
  99                                                 }
 100                                                 if(lua_compare(mState, -4, -2, LUA_OPEQ)) {
 101                                                         lua_settop(mState, top);
 102                                                         return 0;
 103                                                 }
 104                                                 const void* lhs = lua_topointer(mState, -4);
 105                                                 const void* rhs = lua_topointer(mState, -2);
 106                                                 lua_settop(mState, top);
 107                                                 return lhs < rhs ? -1 : (lhs > rhs ? 1 : 0);
 108                                         }
 109                                 }
 110                         }
 111                         lua_settop(mState, top);
 112                         return lua_topointer(mState, -2) < lua_topointer(mState, -1) ? -1 : 1;
 113                 }
 114                 return mState < lua->mState ? -1 : 1;
 115         }

 118 void luaW_pushfaivariant(lua_State* L, variant val) {
 119         if(val.is_int()) {
 120                 lua_pushinteger(L, val.as_int());
 121         } else if(val.is_decimal()) {
 122                 lua_pushnumber(L, val.as_decimal() / 1000.0);
 123         } else if(val.is_string()) {
 124                 const std::string result_string = val.as_string();
 125                 lua_pushlstring(L, result_string.c_str(), result_string.size());
 126         } else if(val.is_list()) {
 127                 lua_newtable(L);
 128                 for(const variant& v : val.as_list()) {
 129                         lua_pushinteger(L, lua_rawlen(L, -1) + 1);
 130                         luaW_pushfaivariant(L, v);
 131                         lua_settable(L, -3);
 132                 }
 133         } else if(val.is_map()) {
 134                 typedef std::map<variant,variant>::value_type kv_type;
 135                 lua_newtable(L);
 136                 for(const kv_type& v : val.as_map()) {
 137                         luaW_pushfaivariant(L, v.first);
 138                         luaW_pushfaivariant(L, v.second);
 139                         lua_settable(L, -3);
 140                 }
 141         } else if(val.is_callable()) {
 142                 // First try a few special cases
 143                 if(auto u_ref = val.try_convert<unit_callable>()) {
 144                         const unit& u = u_ref->get_unit();
 145                         unit_map::iterator un_it = resources::gameboard->units().find(u.get_location());
 146                         if(&*un_it == &u) {
 147                                 luaW_pushunit(L, u.underlying_id());
 148                         } else {
 149                                 luaW_pushunit(L, u.side(), u.underlying_id());
 150                         }
 151                 } else if(auto loc_ref = val.try_convert<location_callable>()) {
 152                         luaW_pushlocation(L, loc_ref->loc());
 153                 } else {
 154                         // If those fail, convert generically to a map
 155                         auto obj = val.as_callable();
 156                         formula_input_vector inputs;
 157                         obj->get_inputs(inputs);
 158                         lua_newtable(L);
 159                         for(const formula_input& attr : inputs) {
 160                                 if(attr.access == FORMULA_WRITE_ONLY) {
 161                                         continue;
 162                                 }
 163                                 lua_pushstring(L, attr.name.c_str());
 164                                 luaW_pushfaivariant(L, obj->query_value(attr.name));
 165                                 lua_settable(L, -3);
 166                         }
 167                 }
 168         } else if(val.is_null()) {
 169                 lua_pushnil(L);
 170         }
 171 }

 173 variant luaW_tofaivariant(lua_State* L, int i) {
 174         switch(lua_type(L, i)) {
 175                 case LUA_TBOOLEAN:
 176                         return variant(lua_tointeger(L, i));
 177                 case LUA_TNUMBER:
 178                         return variant(lua_tonumber(L, i), variant::DECIMAL_VARIANT);
 179                 case LUA_TSTRING:
 180                         return variant(lua_tostring(L, i));
 181                 case LUA_TTABLE:
 182                         return variant(std::make_shared<lua_callable>(L, i));
 183                 case LUA_TUSERDATA:
 184                         static t_string tstr;
 185                         static vconfig vcfg = vconfig::unconstructed_vconfig();
 186                         static map_location loc;
 187                         if(luaW_totstring(L, i, tstr)) {
 188                                 return variant(tstr.str());
 189                         } else if(luaW_tovconfig(L, i, vcfg)) {
 190                                 return variant(std::make_shared<config_callable>(vcfg.get_parsed_config()));
 191                         } else if(unit* u = luaW_tounit(L, i)) {
 192                                 return variant(std::make_shared<unit_callable>(*u));
 193                         } else if(luaW_tolocation(L, i, loc)) {
 194                                 return variant(std::make_shared<location_callable>(loc));
 195                         }
 196                         break;
 197         }
 198         return variant();
 199 }

 201 /**
 202  * Evaluates a formula in the formula engine.
 203  * - Arg 1: Formula string.
 204  * - Arg 2: optional context; can be a unit or a Lua table.
 205  * - Ret 1: Result of the formula.
 206  */
 207 int lua_formula_bridge::intf_eval_formula(lua_State *L)
 208 {
 209         bool need_delete = false;
 210         fwrapper* form;
 211         if(void* ud = luaL_testudata(L, 1, formulaKey)) {
 212                 form = static_cast<fwrapper*>(ud);
 213         } else {
 214                 need_delete = true;
 215                 form = new fwrapper(luaL_checkstring(L, 1));
 216         }
 217         std::shared_ptr<formula_callable> context, fallback;
 218         if(unit* u = luaW_tounit(L, 2)) {
 219                 context.reset(new unit_callable(*u));
 220         } else if(lua_istable(L, 2)) {
 221                 context.reset(new lua_callable(L, 2));
 222         } else {
 223                 context.reset(new map_formula_callable);
 224         }
 225         variant result = form->evaluate(*context);
 226         luaW_pushfaivariant(L, result);
 227         if(need_delete) {
 228                 delete form;
 229         }
 230         return 1;
 231 }

 234 {
 235         if(!lua_isstring(L, 1)) {
 236                 luaW_type_error(L, 1, "string");
 237         }
