  46 /**
  47  * Shows a transient message to the user.
  48  *
  49  * This shows a dialog with a short message which can be dismissed with a
  50  * single click.
  51  *
  52  * @note The message _should_ be small enough to fit on the window, the text
  53  * can contain newlines and will wrap when needed.
  54  *
  55  * @param video               The video which contains the surface to draw
  56  *                            upon.
  57  * @param title               The title of the dialog.
  58  * @param message             The message to show in the dialog.
  59  * @param image               An image to show in the dialog.
  60  * @param message_use_markup  Use markup for the message?
  61  * @param title_use_markup    Use markup for the title?
  62  * @param restore_background  Restore the background to the state it was before
  63  *                                                        the message appeared
  64  */
  65 void show_transient_message(CVideo& video,
  66                                                         const std::string& title,
  67                                                         const std::string& message,
  68                                                         const std::string& image = std::string(),
  69                                                         const bool message_use_markup = false,
  70                                                         const bool title_use_markup = false,
  71                                                         const bool restore_background = false);
