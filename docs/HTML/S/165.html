  46 void server_base::start_server()
  47 {
  48         boost::asio::ip::tcp::endpoint endpoint(boost::asio::ip::tcp::v4(), port_);
  49         acceptor_.open(endpoint.protocol());
  50         acceptor_.set_option(boost::asio::ip::tcp::acceptor::reuse_address(true));
  51         acceptor_.set_option(boost::asio::ip::tcp::acceptor::keep_alive(keep_alive_));
  52         acceptor_.bind(endpoint);
  53         acceptor_.listen();
  54         serve();
  55 
  56         handshake_response_.connection_num = 42;
  57 
  58 #ifndef _WIN32
  59         sighup_.async_wait(
  60                 [=](const boost::system::error_code& error, int sig)
  61                         { this->handle_sighup(error, sig); });
  62 #endif
  63         sigs_.async_wait(std::bind(&server_base::handle_termination, this, _1, _2));
  64 }

 100 void server_base::serverside_handshake(socket_ptr socket)
 101 {
 102         boost::shared_array<char> handshake(new char[4]);
 103         async_read(
 104                                 *socket, boost::asio::buffer(handshake.get(), 4),
 105                                 std::bind(&server_base::handle_handshake, this, _1, socket, handshake)
 106                                 );
 107 }

 109 void server_base::handle_handshake(const boost::system::error_code& error, socket_ptr socket, boost::shared_array<char> handshake)
 110 {
 111         if(check_error(error, socket))
 112                 return;
 113 
 114         if(strcmp(handshake.get(), "\0\0\0\0") != 0) {
 115                 ERR_SERVER << client_address(socket) << "\tincorrect handshake\n";
 116                 return;
 117         }
 118         async_write(
 119                                 *socket, boost::asio::buffer(handshake_response_.buf, 4),
 120                                 [=](const boost::system::error_code& error, size_t)
 121                                         { if(!check_error(error, socket)) this->handle_new_client(socket); }
 122         );
 123 }

 135 void server_base::handle_termination(const boost::system::error_code& error, int signal_number)
 136 {
 137         assert(!error);
 138 
 139         std::string signame;
 140         if(signal_number == SIGINT) signame = "SIGINT";
 141         else if(signal_number == SIGTERM) signame = "SIGTERM";
 142         else signame = lexical_cast<std::string>(signal_number);
 143         LOG_SERVER << signame << " caught, exiting without cleanup immediately.\n";
 144         exit(128 + signal_number);
 145 }

 156 std::string client_address(socket_ptr socket)
 157 {
 158         boost::system::error_code error;
 159         std::string result = socket->remote_endpoint(error).address().to_string();
 160         if(error)
 161                 return "<unknown address>";
 162         else
 163                 return result;
 164 }

 166 bool check_error(const boost::system::error_code& error, socket_ptr socket)
 167 {
 168         if(error) {
 169                 if(error == boost::asio::error::eof)
 170                         LOG_SERVER << client_address(socket) << "\tconnection closed\n";
 171                 else
 172                         ERR_SERVER << client_address(socket) << "\t" << error.message() << "\n";
 173                 return true;
 174         }
 175         return false;
 176 }
