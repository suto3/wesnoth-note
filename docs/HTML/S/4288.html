   9 function ca_wolves_multipacks_wander:evaluation(cfg)
  10     -- When there's nothing to attack, the wolves wander and regroup into their packs
  11 
  12     local wolves = AH.get_units_with_moves {
  13         side = wesnoth.current.side,
  14         type = cfg.type or "Wolf"
  15     }
  16 
  17     if wolves[1] then return cfg.ca_score end
  18     return 0
  19 end

  21 function ca_wolves_multipacks_wander:execution(cfg)
  22     local packs = WMPF.assign_packs(cfg)
  23 
  24     for pack_number,pack in pairs(packs) do
  25         -- If any of the wolves has a goal set, this is used for the entire pack
  26         local wolves, goal = {}, {}
  27         for _,loc in ipairs(pack) do
  28             local wolf = wesnoth.get_unit(loc.x, loc.y)
  29             table.insert(wolves, wolf)
  30 
  31             -- If any of the wolves in the pack has a goal set, we use that one
  32             local wolf_goal = MAIUV.get_mai_unit_variables(wolf, cfg.ai_id)
  33             if wolf_goal.goal_x then
  34                 goal = { wolf_goal.goal_x, wolf_goal.goal_y }
  35             end
  36         end
  37 
  38         -- If the position of any of the wolves is at the goal, delete it
  39         for _,wolf in ipairs(wolves) do
  40             if (wolf.x == goal[1]) and (wolf.y == goal[2]) then goal = {} end
  41         end
  42 
  43         -- Pack gets a new goal if none exist or on any move with 10% random chance
  44         local rand = math.random(10)
  45         if (not goal[1]) or (rand == 1) then
  46             local width, height = wesnoth.get_map_size()
  47             local locs = wesnoth.get_locations { x = '1-'..width, y = '1-'..height }
  48 
  49             -- Need to find reachable terrain for this to be a viable goal
  50             -- We only check whether the first wolf can get there
  51             local unreachable = true
  52             while unreachable do
  53                 local rand = math.random(#locs)
  54                 local next_hop = AH.next_hop(wolves[1], locs[rand][1], locs[rand][2])
  55                 if next_hop then
  56                     goal = { locs[rand][1], locs[rand][2] }
  57                     unreachable = nil
  58                 end
  59             end
  60         end
  61 
  62         -- This goal is saved with every wolf of the pack
  63         for _,wolf in ipairs(wolves) do
  64             MAIUV.insert_mai_unit_variables(wolf, cfg.ai_id, { goal_x = goal[1], goal_y = goal[2] })
  65         end
  66 
  67         -- The pack wanders with only 2 considerations
  68         -- 1. Keeping the pack together (most important)
  69         --   Going through all combinations of all hexes for all wolves is too expensive
  70         --   -> find hexes that can be reached by all wolves
  71         -- 2. Getting closest to the goal
  72 
  73         -- Number of wolves that can reach each hex,
  74         local reach_map = LS.create()
  75         for _,wolf in ipairs(wolves) do
  76             local reach = wesnoth.find_reach(wolf)
  77             for _,loc in ipairs(reach) do
  78                 reach_map:insert(loc[1], loc[2], (reach_map:get(loc[1], loc[2]) or 0) + 100)
  79             end
  80         end
  81 
  82         -- Keep only those hexes that can be reached by all wolves in the pack
  83         -- and add distance from goal for those
  84         local max_rating, goto_hex = -9e99
  85         reach_map:iter( function(x, y, v)
  86             local rating = reach_map:get(x, y)
  87             if (rating == #pack * 100) then
  88                 rating = rating - M.distance_between(x, y, goal[1], goal[2])
  89                 reach_map:insert(x,y, rating)
  90                 if rating > max_rating then
  91                     max_rating, goto_hex = rating, { x, y }
  92                 end
  93             else
  94                 reach_map:remove(x, y)
  95             end
  96         end)
  97 
  98         -- Sort wolves by MP, the one with fewest moves goes first
  99         table.sort(wolves, function(a, b) return a.moves < b.moves end)
 100 
 101         -- If there's no hex that all units can reach, use the 'center of gravity' between them
 102         -- Then we move the first wolf (fewest MP) toward that hex, and the position of that wolf
 103         -- becomes the goto coordinates for the others
 104         if (not goto_hex) then
 105             local cg = { 0, 0 }  -- Center of gravity hex
 106             for _,wolf in ipairs(wolves) do
 107                 cg = { cg[1] + wolf.x, cg[2] + wolf.y }
 108             end
 109             cg[1] = math.floor(cg[1] / #pack)
 110             cg[2] = math.floor(cg[2] / #pack)
 111 
 112             -- Find closest move for Wolf #1 to that position, which then becomes the goto hex
 113             goto_hex = AH.find_best_move(wolves[1], function(x, y)
 114                 return -M.distance_between(x, y, cg[1], cg[2])
 115             end)
 116             -- We could move this wolf right here, but for convenience all the actual moves
 117             -- are done together below. This should be a small extra calculation cost
 118         end
 119 
 120         -- Now all wolves in the pack are moved toward goto_hex, starting with the one with fewest MP
 121         -- Distance to goal hex is taken into account as secondary criterion
 122         for _,wolf in ipairs(wolves) do
 123             local best_hex = AH.find_best_move(wolf, function(x, y)
 124                 local rating = -M.distance_between(x, y, goto_hex[1], goto_hex[2])
 125                 rating = rating -M.distance_between(x, y, goal[1], goal[2]) / 100.
 126                 return rating
 127             end)
 128 
 129             if cfg.show_pack_number then
 130                 WMPF.clear_label(wolf.x, wolf.y)
 131             end
 132 
 133             local move_result = AH.movefull_stopunit(ai, wolf, best_hex)
 134 
 135             if cfg.show_pack_number and wolf and wolf.valid then
 136                 WMPF.put_label(wolf.x, wolf.y, pack_number)
 137             end
 138 
 139             -- If the wolf was ambushed, return and reconsider
 140             if (AH.is_incomplete_move(move_result)) then return end
 141         end
 142     end
 143 end
 144 
 145 return ca_wolves_multipacks_wander
