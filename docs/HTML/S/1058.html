 107         fake_unit_ptr create_fake_unit(const vconfig& cfg)
 108         {
 109                 std::string type = cfg["type"];
 110                 std::string variation = cfg["variation"];
 111                 std::string img_mods = cfg["image_mods"];
 112 
 113                 size_t side_num = cfg["side"].to_int(1);
 114                 if (!resources::gameboard->has_team(side_num)) {
 115                         side_num = 1;
 116                 }
 117 
 118                 unit_race::GENDER gender = string_gender(cfg["gender"]);
 119                 const unit_type *ut = unit_types.find(type);
 120                 if (!ut) return fake_unit_ptr();
 121                 fake_unit_ptr fake = fake_unit_ptr(unit_ptr(new unit(*ut, side_num, false, gender)));
 122 
 123                 if(!variation.empty()) {
 124                         config mod;
 125                         config &effect = mod.add_child("effect");
 126                         effect["apply_to"] = "variation";
 127                         effect["name"] = variation;
 128                         fake->add_modification("variation",mod);
 129                 }
 130 
 131                 if(!img_mods.empty()) {
 132                         config mod;
 133                         config &effect = mod.add_child("effect");
 134                         effect["apply_to"] = "image_mod";
 135                         effect["add"] = img_mods;
 136                         fake->add_modification("image_mod",mod);
 137                 }
 138 
 139                 return fake;
 140         }

 142         std::vector<map_location> fake_unit_path(const unit& fake_unit, const std::vector<std::string>& xvals, const std::vector<std::string>& yvals)
 143         {
 144                 const gamemap *game_map = & resources::gameboard->map();
 145                 std::vector<map_location> path;
 146                 map_location src;
 147                 map_location dst;
 148                 for(size_t i = 0; i != std::min(xvals.size(),yvals.size()); ++i) {
 149                         if(i==0){
 150                                 try {
 151                                         src.set_wml_x(std::stoi(xvals[i]));
 152                                         src.set_wml_y(std::stoi(yvals[i]));
 153                                 } catch(std::invalid_argument) {
 154                                         ERR_CF << "Invalid move_unit_fake source: " << xvals[i] << ", " << yvals[i] << '\n';
 155                                         continue;
 156                                 }
 157                                 if (!game_map->on_board(src)) {
 158                                         ERR_CF << "Invalid move_unit_fake source: " << src << '\n';
 159                                         break;
 160                                 }
 161                                 path.push_back(src);
 162                                 continue;
 163                         }
 164                         pathfind::shortest_path_calculator calc(fake_unit,
 165                                         resources::gameboard->get_team(fake_unit.side()),
 166                                         resources::gameboard->teams(),
 167                                         *game_map);
 168 
 169                         try {
 170                                 dst.set_wml_x(std::stoi(xvals[i]));
 171                                 dst.set_wml_y(std::stoi(yvals[i]));
 172                         } catch(std::invalid_argument) {
 173                                 ERR_CF << "Invalid move_unit_fake destination: " << xvals[i] << ", " << yvals[i] << '\n';
 174                         }
 175                         if (!game_map->on_board(dst)) {
 176                                 ERR_CF << "Invalid move_unit_fake destination: " << dst << '\n';
 177                                 break;
 178                         }
 179 
 180                         pathfind::plain_route route = pathfind::a_star_search(src, dst, 10000, calc,
 181                                 game_map->w(), game_map->h());
 182 
 183                         if (route.steps.empty()) {
 184                                 WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring complexities" << std::endl;
 185                                 pathfind::emergency_path_calculator emergency_calc(fake_unit, *game_map);
 186 
 187                                 route = pathfind::a_star_search(src, dst, 10000, emergency_calc,
 188                                                 game_map->w(), game_map->h());
 189                                 if(route.steps.empty()) {
 190                                         // This would occur when trying to do a MUF of a unit
 191                                         // over locations which are unreachable to it (infinite movement
 192                                         // costs). This really cannot fail.
 193                                         WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring terrain" << std::endl;
 194                                         pathfind::dummy_path_calculator dummy_calc(fake_unit, *game_map);
 195                                         route = a_star_search(src, dst, 10000, dummy_calc, game_map->w(), game_map->h());
 196                                         assert(!route.steps.empty());
 197                                 }
 198                         }
 199                         // we add this section to the end of the complete path
 200                         // skipping section's head because already included
 201                         // by the previous iteration
 202                         path.insert(path.end(),
 203                                         route.steps.begin()+1, route.steps.end());
 204 
 205                         src = dst;
 206                 }
 207                 return path;
 208         }

 271 // This tag exposes part of the code path used to handle [command]'s in replays
 272 // This allows to perform scripting in WML that will use the same code path as player actions, for example.
 273 WML_HANDLER_FUNCTION(do_command,, cfg)
 274 {
 275         // Doing this in a whiteboard applied context will cause bugs
 276         // Note that even though game_events::wml_event_pump() will always apply the real unit map
 277         // It is still possible get a wml commands to run in a whiteboard applied context
 278         // With the theme_items lua callbacks
 279         if(resources::whiteboard->has_planned_unit_map())
 280         {
 281                 ERR_NG << "[do_command] called while whiteboard is applied, ignoring" << std::endl;
 282                 return;
 283         }
 284 
 285         static const std::set<std::string> allowed_tags {"attack", "move", "recruit", "recall", "disband", "fire_event", "lua_ai"};
 286 
 287         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 288         if(is_too_early)
 289         {
 290                 ERR_NG << "[do_command] called too early, only allowed at START or later" << std::endl;
 291                 return;
 292         }
 293         if(!resources::controller->current_team().is_local() && synced_context::get_synced_state() == synced_context::UNSYNCED)
 294         {
 295                 ERR_NG << "[do_command] can only be used from clients that control the currently playing side" << std::endl;
 296                 return;
 297         }
 298         for(vconfig::all_children_iterator i = cfg.ordered_begin(); i != cfg.ordered_end(); ++i)
 299         {
 300                 if(allowed_tags.find( i.get_key()) == allowed_tags.end()) {
 301                         ERR_NG << "unsupported tag [" << i.get_key() << "] in [do_command]" << std::endl;
 302                         std::stringstream o;
 303                         std::copy(allowed_tags.begin(), allowed_tags.end(), std::ostream_iterator<std::string>(o, " "));
 304                         ERR_NG << "allowed tags: " << o.str() << std::endl;
 305                         continue;
 306                 }
 307                 // TODO: afaik run_in_synced_context_if_not_already thows exceptions when the executed action end the scenario or the turn.
 308                 //       This could cause problems, specially when its unclear whether that excetion is caught by lua or not...
 309 
 310                 //Note that this fires related events and everthing else that also happen normally.
 311                 //have to watch out with the undo stack, therefore forbid [auto_shroud] and [update_shroud] here...
 312                 synced_context::run_in_synced_context_if_not_already(
 313                         /*commandname*/ i.get_key(),
 314                         /*data*/ i.get_child().get_parsed_config(),
 315                         /*use_undo*/ true,
 316                         /*show*/ true,
 317                         /*error_handler*/ &on_replay_error
 318                 );
 319         }
 320 }

 329 WML_HANDLER_FUNCTION(modify_turns,, cfg)
 330 {
 331         config::attribute_value value = cfg["value"];
 332         std::string add = cfg["add"];
 333         config::attribute_value current = cfg["current"];
 334         tod_manager& tod_man = *resources::tod_manager;
 335         if(!add.empty()) {
 336                 tod_man.modify_turns_by_wml(add);
 337         } else if(!value.empty()) {
 338                 tod_man.set_number_of_turns_by_wml(value.to_int(-1));
 339         }
 340         // change current turn only after applying mods
 341         if(!current.empty()) {
 342                 const unsigned int current_turn_number = tod_man.turn();
 343                 int new_turn_number = current.to_int(current_turn_number);
 344                 const unsigned int new_turn_number_u = static_cast<unsigned int>(new_turn_number);
 345                 if(new_turn_number_u < 1 || (new_turn_number > tod_man.number_of_turns() && tod_man.number_of_turns() != -1)) {
 346                         ERR_NG << "attempted to change current turn number to one out of range (" << new_turn_number << ")" << std::endl;
 347                 } else if(new_turn_number_u != current_turn_number) {
 348                         tod_man.set_turn_by_wml(new_turn_number_u, resources::gamedata);
 349                         resources::screen->new_turn();
 350                 }
 351         }
 352 }

 357 {

 360                 return;

 365         const std::string y = cfg["y"];

 368         const std::vector<std::string> yvals = utils::split(y);

 373                 unit_display::move_unit(path, dummy_unit.get_unit_ptr(), true, map_location::NDIRECTIONS, force_scroll);
 374         }

 377 WML_HANDLER_FUNCTION(move_units_fake,, cfg)
 378 {
 379         LOG_NG << "Processing [move_units_fake]\n";
 380 
 381         const vconfig::child_list unit_cfgs = cfg.get_children("fake_unit");
 382         size_t num_units = unit_cfgs.size();
 383         std::vector<fake_unit_ptr > units;
 384         units.reserve(num_units);
 385         std::vector<std::vector<map_location> > paths;
 386         paths.reserve(num_units);
 387 
 388         LOG_NG << "Moving " << num_units << " units\n";
 389 
 390         size_t longest_path = 0;
 391 
 392         for (const vconfig& config : unit_cfgs) {
 393                 const std::vector<std::string> xvals = utils::split(config["x"]);
 394                 const std::vector<std::string> yvals = utils::split(config["y"]);
 395                 int skip_steps = config["skip_steps"];
 396                 fake_unit_ptr u = create_fake_unit(config);
 397                 units.push_back(u);
 398                 paths.push_back(fake_unit_path(*u, xvals, yvals));
 399                 if(skip_steps > 0)
 400                         paths.back().insert(paths.back().begin(), skip_steps, paths.back().front());
 401                 longest_path = std::max(longest_path, paths.back().size());
 402                 DBG_NG << "Path " << paths.size() - 1 << " has length " << paths.back().size() << '\n';
 403 
 404                 u->set_location(paths.back().front());
 405                 u.place_on_fake_unit_manager(resources::fake_units);
 406         }
 407 
 408         LOG_NG << "Units placed, longest path is " << longest_path << " long\n";
 409 
 410         std::vector<map_location> path_step(2);
 411         path_step.resize(2);
 412         for(size_t step = 1; step < longest_path; ++step) {
 413                 DBG_NG << "Doing step " << step << "...\n";
 414                 for(size_t un = 0; un < num_units; ++un) {
 415                         if(step >= paths[un].size() || paths[un][step - 1] == paths[un][step])
 416                                 continue;
 417                         DBG_NG << "Moving unit " << un << ", doing step " << step << '\n';
 418                         path_step[0] = paths[un][step - 1];
 419                         path_step[1] = paths[un][step];
 420                         unit_display::move_unit(path_step, units[un].get_unit_ptr());
 421                         units[un]->set_location(path_step[1]);
 422                         units[un]->anim_comp().set_standing();
 423                 }
 424         }
 425 
 426         LOG_NG << "Units moved\n";
 427 }

 479                                                 DBG_NG << "...matched the leader filter and is able to recall the unit.\n";
 480                                                 if(!resources::gameboard->map().on_board(loc))
 481                                                         loc = leader->get_location();
 482                                                 if(pass_check || (resources::gameboard->units().count(loc) > 0))
 483                                                         loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);

 485                                                         DBG_NG << "...valid location for the recall found. Recalling.\n";

 488                                                                                cfg["show"].to_bool(true), cfg["fire_event"].to_bool(false),
 489                                                                                true, true);

 492                                         }
 493                                 }
 494                                 if (resources::gameboard->map().on_board(cfg_loc)) {
 495                                         map_location loc = cfg_loc;
 496                                         if(pass_check || (resources::gameboard->units().count(loc) > 0))
 497                                                 loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 498                                         // Check if we still have a valid location

 500                                                 DBG_NG << "No usable leader found, but found usable location. Recalling.\n";
 501                                                 avail.erase(u); // Erase before recruiting, since recruiting can fire more events

 527                         static const std::string s_all = "(" + s_prefix + s_terrain + s_sep + ")+";
 528                         static const boost::regex r_all(s_all);
 529 
 530                         const std::string& mapfile = filesystem::get_wml_location(filename_);
 531                         std::string res = "";
 532                         if(filesystem::file_exists(mapfile)) {
 533                                 res = filesystem::read_file(mapfile);
 534                         }
 535                         config retv;
 536                         if(boost::regex_match(res, r_all))
 537                         {
 538                                 retv["map_data"] = res;
 539                         }
 540                         return retv;
 541                 }
 542                 virtual config random_choice(int /*side*/) const
 543                 {
 544                         return config();
 545                 }
 546                 virtual std::string description() const
 547                 {
 548                         return "Map Data";
 549                 }

 551         };

 554 /// Experimental map replace
 555 /// @todo Finish experimenting.
 556 WML_HANDLER_FUNCTION(replace_map,, cfg)
 557 {
 558         /*
 559          * When a hex changes from a village terrain to a non-village terrain, and
 560          * a team owned that village it loses that village. When a hex changes from
 561          * a non-village terrain to a village terrain and there is a unit on that
 562          * hex it does not automatically capture the village. The reason for not
 563          * capturing villages it that there are too many choices to make; should a
 564          * unit loose its movement points, should capture events be fired. It is
 565          * easier to do this as wanted by the author in WML.
 566          */
 567 
 568         const gamemap * game_map = & resources::gameboard->map();
 569         gamemap map(*game_map);
 570 
 571         try {
 572                 if(!cfg["map_file"].empty()) {
 573                         config file_cfg = mp_sync::get_user_choice("map_data", map_choice(cfg["map_file"].str()));
 574                         map.read(file_cfg["map_data"].str(), false);
 575                 } else {
 576                         map.read(cfg["map"], false);
 577                 }
 578         } catch(incorrect_map_format_error&) {
 579                 const std::string log_map_name = cfg["map"].empty() ? cfg["file"] : std::string("from inline data");
 580                 lg::wml_error() << "replace_map: Unable to load map " << log_map_name << std::endl;
 581                 return;
 582         } catch(wml_exception& e) {
 583                 e.show(resources::screen->video());
 584                 return;
 585         }
 586 
 587         if (map.total_width() > game_map->total_width()
 588         || map.total_height() > game_map->total_height()) {
 589                 if (!cfg["expand"].to_bool()) {
 590                         lg::wml_error() << "replace_map: Map dimension(s) increase but expand is not set" << std::endl;
 591                         return;
 592                 }
 593         }
 594 
 595         if (map.total_width() < game_map->total_width()
 596         || map.total_height() < game_map->total_height()) {
 597                 if (!cfg["shrink"].to_bool()) {
 598                         lg::wml_error() << "replace_map: Map dimension(s) decrease but shrink is not set" << std::endl;
 599                         return;
 600                 }
 601         }
 602 
 603         boost::optional<std::string> errmsg = resources::gameboard->replace_map(map);
 604 
 605         if (errmsg) {
 606                 lg::wml_error() << *errmsg << std::endl;
 607         }
 608 
 609         resources::screen->reload_map();
 610         resources::screen->needs_rebuild(true);
 611         ai::manager::raise_map_changed();
 612 }

 622 WML_HANDLER_FUNCTION(set_variables,, cfg)
 623 {
 624         const t_string& name = cfg["name"];
 625         variable_access_create dest = resources::gamedata->get_variable_access_write(name);
 626         if(name.empty()) {
 627                 ERR_NG << "trying to set a variable with an empty name:\n" << cfg.get_config().debug();
 628                 return;
 629         }
 630 
 631         std::vector<config> data;
 632         if(cfg.has_attribute("to_variable"))
 633         {
 634                 try
 635                 {
 636                         variable_access_const tovar = resources::gamedata->get_variable_access_read(cfg["to_variable"]);
 637                         for (const config& c : tovar.as_array())
 638                         {
 639                                 data.push_back(c);
 640                         }
 641                 }
 642                 catch(const invalid_variablename_exception&)
 643                 {
 644                         ERR_NG << "Cannot do [set_variables] with invalid to_variable variable: " << cfg["to_variable"] << " with " << cfg.get_config().debug() << std::endl;
 645                 }
 646         } else {
 647                 typedef std::pair<std::string, vconfig> vchild;
 648                 for (const vchild& p : cfg.all_ordered()) {
 649                         if(p.first == "value") {
 650                                 data.push_back(p.second.get_parsed_config());
 651                         } else if(p.first == "literal") {
 652                                 data.push_back(p.second.get_config());
 653                         } else if(p.first == "split") {
 654                                 const vconfig & split_element = p.second;
 655 
 656                                 std::string split_string=split_element["list"];
 657                                 std::string separator_string=split_element["separator"];
 658                                 std::string key_name=split_element["key"];
 659                                 if(key_name.empty())
 660                                 {
 661                                         key_name="value";
 662                                 }
 663 
 664                                 bool remove_empty = split_element["remove_empty"].to_bool();
 665 
 666                                 char* separator = separator_string.empty() ? nullptr : &separator_string[0];
 667 
 668                                 std::vector<std::string> split_vector;
 669 
 670                                 //if no separator is specified, explode the string
 671                                 if(separator == nullptr)
 672                                 {
 673                                         for(std::string::iterator i=split_string.begin(); i!=split_string.end(); ++i)
 674                                         {
 675                                                 split_vector.push_back(std::string(1, *i));
 676                                         }
 677                                 }
 678                                 else {
 679                                         split_vector=utils::split(split_string, *separator, remove_empty ? utils::REMOVE_EMPTY | utils::STRIP_SPACES : utils::STRIP_SPACES);
 680                                 }
 681 
 682                                 for(std::vector<std::string>::iterator i=split_vector.begin(); i!=split_vector.end(); ++i)
 683                                 {
 684                                         data.emplace_back(config_of(key_name, *i));
 685                                 }
 686                         }
 687                 }
 688         }
 689         try
 690         {
 691                 const std::string& mode = cfg["mode"];
 692                 if(mode == "merge")
 693                 {
 694                         if(dest.explicit_index() && data.size() > 1)
 695                         {
 696                                 //merge children into one
 697                                 config merged_children;
 698                                 for (const config &ch : data) {
 699                                         merged_children.append(ch);
 700                                 }
 701                                 data = {merged_children};
 702                         }
 703                         dest.merge_array(data);
 704                 }
 705                 else if(mode == "insert")
 706                 {
 707                         dest.insert_array(data);
 708                 }
 709                 else if(mode == "append")
 710                 {
 711                         dest.append_array(data);
 712                 }
 713                 else /*default if(mode == "replace")*/
 714                 {
 715                         dest.replace_array(data);
 716                 }
 717         }
 718         catch(const invalid_variablename_exception&)
 719         {
 720                 ERR_NG << "Cannot do [set_variables] with invalid destination variable: " << name << " with " << cfg.get_config().debug() << std::endl;
 721         }
 722 }

 724 /// Store the relative direction from one hex to another in a WML variable.
 725 /// This is mainly useful as a diagnostic tool, but could be useful
 726 /// for some kind of scenario.
 727 WML_HANDLER_FUNCTION(store_relative_direction,, cfg)
 728 {
 729         if (!cfg.child("source")) {
 730                 WRN_NG << "No source in [store_relative_direction]" << std::endl;
 731                 return;
 732         }
 733         if (!cfg.child("destination")) {
 734                 WRN_NG << "No destination in [store_relative_direction]" << std::endl;
 735                 return;
 736         }
 737         if (!cfg.has_attribute("variable")) {
 738                 WRN_NG << "No variable in [store_relative_direction]" << std::endl;
 739                 return;
 740         }
 741 
 742         const map_location src = cfg_to_loc(cfg.child("source"));
 743         const map_location dst = cfg_to_loc(cfg.child("destination"));
 744 
 745         std::string variable = cfg["variable"];
 746         map_location::RELATIVE_DIR_MODE mode = static_cast<map_location::RELATIVE_DIR_MODE> (cfg["mode"].to_int(0));
 747         try
 748         {
 749                 variable_access_create store = resources::gamedata->get_variable_access_write(variable);
 750 
 751                 store.as_scalar() = map_location::write_direction(src.get_relative_dir(dst,mode));
 752         }
 753         catch(const invalid_variablename_exception&)
 754         {
 755                 ERR_NG << "Cannot do [store_relative_direction] with invalid destination variable: " << variable << " with " << cfg.get_config().debug() << std::endl;
 756         }

 857 WML_HANDLER_FUNCTION(tunnel,, cfg)
 858 {
 859         const bool remove = cfg["remove"].to_bool(false);
 860         const bool delay = cfg["delayed_variable_substitution"].to_bool(true);

 862                 const std::vector<std::string> ids = utils::split(cfg["id"]);
 863                 for (const std::string &id : ids) {
 864                         resources::tunnels->remove(id);
 865                 }

 867                 cfg.get_children("target").empty() ||
 868                 cfg.get_children("filter").empty()) {
 869                 ERR_WML << "[tunnel] is missing a mandatory tag:\n"
 870                          << cfg.get_config().debug();
 871         } else {
 872                 pathfind::teleport_group tunnel(delay ? cfg : vconfig(cfg.get_parsed_config()), false);
 873                 resources::tunnels->add(tunnel);
 874 
 875                 if(cfg["bidirectional"].to_bool(true)) {
 876                         tunnel = pathfind::teleport_group(delay ? cfg : vconfig(cfg.get_parsed_config()), true);
 877                         resources::tunnels->add(tunnel);
 878                 }

 882 /// If we should spawn a new unit on the map somewhere
 883 WML_HANDLER_FUNCTION(unit,, cfg)
 884 {
 885         config parsed_cfg = cfg.get_parsed_config();
 886 
 887         config::attribute_value to_variable = cfg["to_variable"];
 888         if (!to_variable.blank())
 889         {
 890                 parsed_cfg.remove_attribute("to_variable");
 891                 unit new_unit(parsed_cfg, true, &cfg);
 892                 try
 893                 {
 894                         config &var = resources::gamedata->get_variable_cfg(to_variable);
 895                         var.clear();
 896                         new_unit.write(var);
 897                         if (const config::attribute_value *v = parsed_cfg.get("x")) var["x"] = *v;
 898                         if (const config::attribute_value *v = parsed_cfg.get("y")) var["y"] = *v;
 899                 }
 900                 catch(const invalid_variablename_exception&)
 901                 {
 902                         ERR_NG << "Cannot do [unit] with invalid to_variable:  " << to_variable << " with " << cfg.get_config().debug() << std::endl;
 903                 }
 904                 return;
 905 
 906         }
 907 
 908         int side = parsed_cfg["side"].to_int(1);
 909 
 910 
 911         if ((side<1)||(side > static_cast<int>(resources::gameboard->teams().size()))) {
 912                 ERR_NG << "wrong side in [unit] tag - no such side: "<<side<<" ( number of teams :"<<resources::gameboard->teams().size()<<")"<<std::endl;
 913                 DBG_NG << parsed_cfg.debug();
 914                 return;
 915         }
 916         team &tm = resources::gameboard->get_team(side);
 917 
 918         unit_creator uc(tm,resources::gameboard->map().starting_position(side));
 919 
 920         uc
 921                 .allow_add_to_recall(true)
 922                 .allow_discover(true)
 923                 .allow_get_village(true)
 924                 .allow_invalidate(true)
 925                 .allow_rename_side(true)
 926                 .allow_show(true);
 927 
 928         uc.add_unit(parsed_cfg, &cfg);
 929 
 930 }
