 106         fake_unit_ptr create_fake_unit(const vconfig& cfg)
 107         {
 108                 std::string type = cfg["type"];
 109                 std::string variation = cfg["variation"];
 110                 std::string img_mods = cfg["image_mods"];
 111 
 112                 size_t side_num = cfg["side"].to_int(1);
 113                 if (!resources::gameboard->has_team(side_num)) {
 114                         side_num = 1;
 115                 }
 116 
 117                 unit_race::GENDER gender = string_gender(cfg["gender"]);
 118                 const unit_type *ut = unit_types.find(type);
 119                 if (!ut) return fake_unit_ptr();
 120                 fake_unit_ptr fake = fake_unit_ptr(unit_ptr(new unit(*ut, side_num, false, gender)));
 121 
 122                 if(!variation.empty()) {
 123                         config mod;
 124                         config &effect = mod.add_child("effect");
 125                         effect["apply_to"] = "variation";
 126                         effect["name"] = variation;
 127                         fake->add_modification("variation",mod);
 128                 }
 129 
 130                 if(!img_mods.empty()) {
 131                         config mod;
 132                         config &effect = mod.add_child("effect");
 133                         effect["apply_to"] = "image_mod";
 134                         effect["add"] = img_mods;
 135                         fake->add_modification("image_mod",mod);
 136                 }
 137 
 138                 return fake;
 139         }

 141         std::vector<map_location> fake_unit_path(const unit& fake_unit, const std::vector<std::string>& xvals, const std::vector<std::string>& yvals)
 142         {
 143                 const gamemap *game_map = & resources::gameboard->map();
 144                 std::vector<map_location> path;
 145                 map_location src;
 146                 map_location dst;
 147                 for(size_t i = 0; i != std::min(xvals.size(),yvals.size()); ++i) {
 148                         if(i==0){
 149                                 try {
 150                                         src.set_wml_x(std::stoi(xvals[i]));
 151                                         src.set_wml_y(std::stoi(yvals[i]));
 152                                 } catch(std::invalid_argument) {
 153                                         ERR_CF << "Invalid move_unit_fake source: " << xvals[i] << ", " << yvals[i] << '\n';
 154                                         continue;
 155                                 }
 156                                 if (!game_map->on_board(src)) {
 157                                         ERR_CF << "Invalid move_unit_fake source: " << src << '\n';
 158                                         break;
 159                                 }
 160                                 path.push_back(src);
 161                                 continue;
 162                         }
 163                         pathfind::shortest_path_calculator calc(fake_unit,
 164                                         resources::gameboard->get_team(fake_unit.side()),
 165                                         resources::gameboard->teams(),
 166                                         *game_map);
 167 
 168                         try {
 169                                 dst.set_wml_x(std::stoi(xvals[i]));
 170                                 dst.set_wml_y(std::stoi(yvals[i]));
 171                         } catch(std::invalid_argument) {
 172                                 ERR_CF << "Invalid move_unit_fake destination: " << xvals[i] << ", " << yvals[i] << '\n';
 173                         }
 174                         if (!game_map->on_board(dst)) {
 175                                 ERR_CF << "Invalid move_unit_fake destination: " << dst << '\n';
 176                                 break;
 177                         }
 178 
 179                         pathfind::plain_route route = pathfind::a_star_search(src, dst, 10000, calc,
 180                                 game_map->w(), game_map->h());
 181 
 182                         if (route.steps.empty()) {
 183                                 WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring complexities" << std::endl;
 184                                 pathfind::emergency_path_calculator emergency_calc(fake_unit, *game_map);
 185 
 186                                 route = pathfind::a_star_search(src, dst, 10000, emergency_calc,
 187                                                 game_map->w(), game_map->h());
 188                                 if(route.steps.empty()) {
 189                                         // This would occur when trying to do a MUF of a unit
 190                                         // over locations which are unreachable to it (infinite movement
 191                                         // costs). This really cannot fail.
 192                                         WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring terrain" << std::endl;
 193                                         pathfind::dummy_path_calculator dummy_calc(fake_unit, *game_map);
 194                                         route = a_star_search(src, dst, 10000, dummy_calc, game_map->w(), game_map->h());
 195                                         assert(!route.steps.empty());
 196                                 }
 197                         }
 198                         // we add this section to the end of the complete path
 199                         // skipping section's head because already included
 200                         // by the previous iteration
 201                         path.insert(path.end(),
 202                                         route.steps.begin()+1, route.steps.end());
 203 
 204                         src = dst;
 205                 }
 206                 return path;
 207         }

 270 // This tag exposes part of the code path used to handle [command]'s in replays
 271 // This allows to perform scripting in WML that will use the same code path as player actions, for example.
 272 WML_HANDLER_FUNCTION(do_command,, cfg)
 273 {
 274         // Doing this in a whiteboard applied context will cause bugs
 275         // Note that even though game_events::wml_event_pump() will always apply the real unit map
 276         // It is still possible get a wml commands to run in a whiteboard applied context
 277         // With the theme_items lua callbacks
 278         if(resources::whiteboard->has_planned_unit_map())
 279         {
 280                 ERR_NG << "[do_command] called while whiteboard is applied, ignoring" << std::endl;
 281                 return;
 282         }
 283 
 284         static const std::set<std::string> allowed_tags {"attack", "move", "recruit", "recall", "disband", "fire_event", "lua_ai"};
 285 
 286         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 287         if(is_too_early)
 288         {
 289                 ERR_NG << "[do_command] called too early, only allowed at START or later" << std::endl;
 290                 return;
 291         }
 292         if(!resources::controller->current_team().is_local() && synced_context::get_synced_state() == synced_context::UNSYNCED)
 293         {
 294                 ERR_NG << "[do_command] can only be used from clients that control the currently playing side" << std::endl;
 295                 return;
 296         }
 297         for(vconfig::all_children_iterator i = cfg.ordered_begin(); i != cfg.ordered_end(); ++i)
 298         {
 299                 if(allowed_tags.find( i.get_key()) == allowed_tags.end()) {
 300                         ERR_NG << "unsupported tag [" << i.get_key() << "] in [do_command]" << std::endl;
 301                         std::stringstream o;
 302                         std::copy(allowed_tags.begin(), allowed_tags.end(), std::ostream_iterator<std::string>(o, " "));
 303                         ERR_NG << "allowed tags: " << o.str() << std::endl;
 304                         continue;
 305                 }
 306                 // TODO: afaik run_in_synced_context_if_not_already thows exceptions when the executed action end the scenario or the turn.
 307                 //       This could cause problems, specially when its unclear whether that excetion is caught by lua or not...
 308 
 309                 //Note that this fires related events and everthing else that also happen normally.
 310                 //have to watch out with the undo stack, therefore forbid [auto_shroud] and [update_shroud] here...
 311                 synced_context::run_in_synced_context_if_not_already(
 312                         /*commandname*/ i.get_key(),
 313                         /*data*/ i.get_child().get_parsed_config(),
 314                         /*use_undo*/ true,
 315                         /*show*/ true,
 316                         /*error_handler*/ &on_replay_error
 317                 );
 318         }
 319 }

 328 WML_HANDLER_FUNCTION(modify_turns,, cfg)
 329 {
 330         config::attribute_value value = cfg["value"];
 331         std::string add = cfg["add"];
 332         config::attribute_value current = cfg["current"];
 333         tod_manager& tod_man = *resources::tod_manager;
 334         if(!add.empty()) {
 335                 tod_man.modify_turns_by_wml(add);
 336         } else if(!value.empty()) {
 337                 tod_man.set_number_of_turns_by_wml(value.to_int(-1));
 338         }
 339         // change current turn only after applying mods
 340         if(!current.empty()) {
 341                 const unsigned int current_turn_number = tod_man.turn();
 342                 int new_turn_number = current.to_int(current_turn_number);
 343                 const unsigned int new_turn_number_u = static_cast<unsigned int>(new_turn_number);
 344                 if(new_turn_number_u < 1 || (new_turn_number > tod_man.number_of_turns() && tod_man.number_of_turns() != -1)) {
 345                         ERR_NG << "attempted to change current turn number to one out of range (" << new_turn_number << ")" << std::endl;
 346                 } else if(new_turn_number_u != current_turn_number) {
 347                         tod_man.set_turn_by_wml(new_turn_number_u, resources::gamedata);
 348                         resources::screen->new_turn();
 349                 }
 350         }
 351 }

 355 WML_HANDLER_FUNCTION(move_unit_fake,, cfg)
 356 {
 357         fake_unit_ptr dummy_unit(create_fake_unit(cfg));
 358         if(!dummy_unit.get())
 359                 return;
 360 
 361         const bool force_scroll = cfg["force_scroll"].to_bool(true);
 362 
 363         const std::string x = cfg["x"];
 364         const std::string y = cfg["y"];
 365 
 366         const std::vector<std::string> xvals = utils::split(x);
 367         const std::vector<std::string> yvals = utils::split(y);
 368 
 369         const std::vector<map_location>& path = fake_unit_path(*dummy_unit, xvals, yvals);
 370         if (!path.empty()) {
 371                 // Always scroll.
 372                 unit_display::move_unit(path, dummy_unit.get_unit_ptr(), true, map_location::NDIRECTIONS, force_scroll);
 373         }

 376 WML_HANDLER_FUNCTION(move_units_fake,, cfg)
 377 {
 378         LOG_NG << "Processing [move_units_fake]\n";
 379 
 380         const vconfig::child_list unit_cfgs = cfg.get_children("fake_unit");
 381         size_t num_units = unit_cfgs.size();
 382         std::vector<fake_unit_ptr > units;
 383         units.reserve(num_units);
 384         std::vector<std::vector<map_location> > paths;
 385         paths.reserve(num_units);
 386 
 387         LOG_NG << "Moving " << num_units << " units\n";
 388 
 389         size_t longest_path = 0;
 390 
 391         for (const vconfig& config : unit_cfgs) {
 392                 const std::vector<std::string> xvals = utils::split(config["x"]);
 393                 const std::vector<std::string> yvals = utils::split(config["y"]);
 394                 int skip_steps = config["skip_steps"];
 395                 fake_unit_ptr u = create_fake_unit(config);
 396                 units.push_back(u);
 397                 paths.push_back(fake_unit_path(*u, xvals, yvals));
 398                 if(skip_steps > 0)
 399                         paths.back().insert(paths.back().begin(), skip_steps, paths.back().front());
 400                 longest_path = std::max(longest_path, paths.back().size());
 401                 DBG_NG << "Path " << paths.size() - 1 << " has length " << paths.back().size() << '\n';
 402 
 403                 u->set_location(paths.back().front());
 404                 u.place_on_fake_unit_manager(resources::fake_units);
 405         }
 406 
 407         LOG_NG << "Units placed, longest path is " << longest_path << " long\n";
 408 
 409         std::vector<map_location> path_step(2);
 410         path_step.resize(2);
 411         for(size_t step = 1; step < longest_path; ++step) {
 412                 DBG_NG << "Doing step " << step << "...\n";
 413                 for(size_t un = 0; un < num_units; ++un) {
 414                         if(step >= paths[un].size() || paths[un][step - 1] == paths[un][step])
 415                                 continue;
 416                         DBG_NG << "Moving unit " << un << ", doing step " << step << '\n';
 417                         path_step[0] = paths[un][step - 1];
 418                         path_step[1] = paths[un][step];
 419                         unit_display::move_unit(path_step, units[un].get_unit_ptr());
 420                         units[un]->set_location(path_step[1]);
 421                         units[un]->anim_comp().set_standing();
 422                 }
 423         }
 424 
 425         LOG_NG << "Units moved\n";
 426 }

 428 /// If we should recall units that match a certain description.
 429 // If you change attributes specific to [recall] (that is, not a Standard Unit Filter)
 430 // be sure to update data/lua/wml_tag, auto_recall feature for [role] to reflect your changes.
 431 WML_HANDLER_FUNCTION(recall,, cfg)
 432 {
 433         LOG_NG << "recalling unit...\n";
 434         config temp_config(cfg.get_config());
 435         // Prevent the recall unit filter from using the location as a criterion
 436 
 437         /**
 438          * @todo FIXME: we should design the WML to avoid these types of
 439          * collisions; filters should be named consistently and always have a
 440          * distinct scope.
 441          */
 442         temp_config["x"] = "recall";
 443         temp_config["y"] = "recall";
 444         vconfig unit_filter_cfg(temp_config);
 445         const vconfig & leader_filter = cfg.child("secondary_unit");
 446 
 447         for(int index = 0; index < int(resources::gameboard->teams().size()); ++index) {
 448                 LOG_NG << "for side " << index + 1 << "...\n";
 449                 const std::string player_id = resources::gameboard->teams()[index].save_id();
 450 
 451                 if(resources::gameboard->teams()[index].recall_list().size() < 1) {
 452                         DBG_NG << "recall list is empty when trying to recall!\n"
 453                                    << "player_id: " << player_id << " side: " << index+1 << "\n";
 454                         continue;
 455                 }
 456 
 457                 recall_list_manager & avail = resources::gameboard->teams()[index].recall_list();
 458                 std::vector<unit_map::unit_iterator> leaders = resources::gameboard->units().find_leaders(index + 1);
 459 
 460                 const unit_filter ufilt(unit_filter_cfg, resources::filter_con);
 461                 const unit_filter lfilt(leader_filter, resources::filter_con); // Note that if leader_filter is null, this correctly gives a null filter that matches all units.
 462                 for(std::vector<unit_ptr>::iterator u = avail.begin(); u != avail.end(); ++u) {
 463                         DBG_NG << "checking unit against filter...\n";
 464                         scoped_recall_unit auto_store("this_unit", player_id, u - avail.begin());
 465                         if (ufilt(*(*u), map_location())) {
 466                                 DBG_NG << (*u)->id() << " matched the filter...\n";
 467                                 const unit_ptr to_recruit = *u;
 468                                 const unit* pass_check = to_recruit.get();
 469                                 if(!cfg["check_passability"].to_bool(true)) pass_check = nullptr;
 470                                 const map_location cfg_loc = cfg_to_loc(cfg);
 471 
 472                                 /// @todo fendrin: comment this monster
 473                                 for (unit_map::const_unit_iterator leader : leaders) {
 474                                         DBG_NG << "...considering " + leader->id() + " as the recalling leader...\n";
 475                                         map_location loc = cfg_loc;
 476                                         if ( lfilt(*leader)  &&
 477                                              unit_filter(vconfig(leader->recall_filter()), resources::filter_con).matches( *(*u),map_location() ) ) {
 478                                                 DBG_NG << "...matched the leader filter and is able to recall the unit.\n";
 479                                                 if(!resources::gameboard->map().on_board(loc))
 480                                                         loc = leader->get_location();
 481                                                 if(pass_check || (resources::gameboard->units().count(loc) > 0))
 482                                                         loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 483                                                 if(resources::gameboard->map().on_board(loc)) {
 484                                                         DBG_NG << "...valid location for the recall found. Recalling.\n";
 485                                                         avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 486                                                         actions::place_recruit(to_recruit, loc, leader->get_location(), 0, true,
 487                                                                                cfg["show"].to_bool(true), cfg["fire_event"].to_bool(false),
 488                                                                                true, true);
 489                                                         return;
 490                                                 }
 491                                         }
 492                                 }
 493                                 if (resources::gameboard->map().on_board(cfg_loc)) {
 494                                         map_location loc = cfg_loc;
 495                                         if(pass_check || (resources::gameboard->units().count(loc) > 0))
 496                                                 loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 497                                         // Check if we still have a valid location
 498                                         if (resources::gameboard->map().on_board(loc)) {
 499                                                 DBG_NG << "No usable leader found, but found usable location. Recalling.\n";
 500                                                 avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 501                                                 map_location null_location = map_location::null_location();
 502                                                 actions::place_recruit(to_recruit, loc, null_location, 0, true, cfg["show"].to_bool(true),
 503                                                                        cfg["fire_event"].to_bool(false), true, true);
 504                                                 return;
 505                                         }
 506                                 }
 507                         }
 508                 }
 509         }
 510         LOG_WML << "A [recall] tag with the following content failed:\n" << cfg.get_config().debug();
 511 }

 525                         static const std::string s_prefix = R"""((\d+ )?)""";
 526                         static const std::string s_all = "(" + s_prefix + s_terrain + s_sep + ")+";
 527                         static const boost::regex r_all(s_all);
 528 
 529                         const std::string& mapfile = filesystem::get_wml_location(filename_);
 530                         std::string res = "";
 531                         if(filesystem::file_exists(mapfile)) {
 532                                 res = filesystem::read_file(mapfile);
 533                         }
 534                         config retv;
 535                         if(boost::regex_match(res, r_all))
 536                         {
 537                                 retv["map_data"] = res;
 538                         }
 539                         return retv;
 540                 }
 541                 virtual config random_choice(int /*side*/) const
 542                 {
 543                         return config();
 544                 }
 545                 virtual std::string description() const
 546                 {
 547                         return "Map Data";
 548                 }

 553 /// Experimental map replace
 554 /// @todo Finish experimenting.
 555 WML_HANDLER_FUNCTION(replace_map,, cfg)
 556 {
 557         /*
 558          * When a hex changes from a village terrain to a non-village terrain, and
 559          * a team owned that village it loses that village. When a hex changes from
 560          * a non-village terrain to a village terrain and there is a unit on that
 561          * hex it does not automatically capture the village. The reason for not
 562          * capturing villages it that there are too many choices to make; should a
 563          * unit loose its movement points, should capture events be fired. It is
 564          * easier to do this as wanted by the author in WML.
 565          */
 566 
 567         const gamemap * game_map = & resources::gameboard->map();
 568         gamemap map(*game_map);
 569 
 570         try {
 571                 if(!cfg["map_file"].empty()) {
 572                         config file_cfg = mp_sync::get_user_choice("map_data", map_choice(cfg["map_file"].str()));
 573                         map.read(file_cfg["map_data"].str(), false);
 574                 } else {
 575                         map.read(cfg["map"], false);
 576                 }
 577         } catch(incorrect_map_format_error&) {
 578                 const std::string log_map_name = cfg["map"].empty() ? cfg["file"] : std::string("from inline data");
 579                 lg::wml_error() << "replace_map: Unable to load map " << log_map_name << std::endl;
 580                 return;
 581         } catch(wml_exception& e) {
 582                 e.show(resources::screen->video());
 583                 return;
 584         }
 585 
 586         if (map.total_width() > game_map->total_width()
 587         || map.total_height() > game_map->total_height()) {
 588                 if (!cfg["expand"].to_bool()) {
 589                         lg::wml_error() << "replace_map: Map dimension(s) increase but expand is not set" << std::endl;
 590                         return;
 591                 }
 592         }
 593 
 594         if (map.total_width() < game_map->total_width()
 595         || map.total_height() < game_map->total_height()) {
 596                 if (!cfg["shrink"].to_bool()) {
 597                         lg::wml_error() << "replace_map: Map dimension(s) decrease but shrink is not set" << std::endl;
 598                         return;
 599                 }
 600         }
 601 
 602         boost::optional<std::string> errmsg = resources::gameboard->replace_map(map);
 603 
 604         if (errmsg) {
 605                 lg::wml_error() << *errmsg << std::endl;
 606         }
 607 
 608         resources::screen->reload_map();
 609         resources::screen->needs_rebuild(true);
 610         ai::manager::raise_map_changed();
 611 }

 621 WML_HANDLER_FUNCTION(set_variables,, cfg)
 622 {
 623         const t_string& name = cfg["name"];
 624         variable_access_create dest = resources::gamedata->get_variable_access_write(name);
 625         if(name.empty()) {
 626                 ERR_NG << "trying to set a variable with an empty name:\n" << cfg.get_config().debug();
 627                 return;
 628         }
 629 
 630         std::vector<config> data;
 631         if(cfg.has_attribute("to_variable"))
 632         {
 633                 try
 634                 {
 635                         variable_access_const tovar = resources::gamedata->get_variable_access_read(cfg["to_variable"]);
 636                         for (const config& c : tovar.as_array())
 637                         {
 638                                 data.push_back(c);
 639                         }
 640                 }
 641                 catch(const invalid_variablename_exception&)
 642                 {
 643                         ERR_NG << "Cannot do [set_variables] with invalid to_variable variable: " << cfg["to_variable"] << " with " << cfg.get_config().debug() << std::endl;
 644                 }
 645         } else {
 646                 typedef std::pair<std::string, vconfig> vchild;
 647                 for (const vchild& p : cfg.all_ordered()) {
 648                         if(p.first == "value") {
 649                                 data.push_back(p.second.get_parsed_config());
 650                         } else if(p.first == "literal") {
 651                                 data.push_back(p.second.get_config());
 652                         } else if(p.first == "split") {
 653                                 const vconfig & split_element = p.second;
 654 
 655                                 std::string split_string=split_element["list"];
 656                                 std::string separator_string=split_element["separator"];
 657                                 std::string key_name=split_element["key"];
 658                                 if(key_name.empty())
 659                                 {
 660                                         key_name="value";
 661                                 }
 662 
 663                                 bool remove_empty = split_element["remove_empty"].to_bool();
 664 
 665                                 char* separator = separator_string.empty() ? nullptr : &separator_string[0];
 666 
 667                                 std::vector<std::string> split_vector;
 668 
 669                                 //if no separator is specified, explode the string
 670                                 if(separator == nullptr)
 671                                 {
 672                                         for(std::string::iterator i=split_string.begin(); i!=split_string.end(); ++i)
 673                                         {
 674                                                 split_vector.push_back(std::string(1, *i));
 675                                         }
 676                                 }
 677                                 else {
 678                                         split_vector=utils::split(split_string, *separator, remove_empty ? utils::REMOVE_EMPTY | utils::STRIP_SPACES : utils::STRIP_SPACES);
 679                                 }
 680 
 681                                 for(std::vector<std::string>::iterator i=split_vector.begin(); i!=split_vector.end(); ++i)
 682                                 {
 683                                         data.emplace_back(config {key_name, *i});
 684                                 }
 685                         }
 686                 }
 687         }
 688         try
 689         {
 690                 const std::string& mode = cfg["mode"];
 691                 if(mode == "merge")
 692                 {
 693                         if(dest.explicit_index() && data.size() > 1)
 694                         {
 695                                 //merge children into one
 696                                 config merged_children;
 697                                 for (const config &ch : data) {
 698                                         merged_children.append(ch);
 699                                 }
 700                                 data = {merged_children};
 701                         }
 702                         dest.merge_array(data);
 703                 }
 704                 else if(mode == "insert")
 705                 {
 706                         dest.insert_array(data);
 707                 }
 708                 else if(mode == "append")
 709                 {
 710                         dest.append_array(data);
 711                 }
 712                 else /*default if(mode == "replace")*/
 713                 {
 714                         dest.replace_array(data);
 715                 }
 716         }
 717         catch(const invalid_variablename_exception&)
 718         {
 719                 ERR_NG << "Cannot do [set_variables] with invalid destination variable: " << name << " with " << cfg.get_config().debug() << std::endl;
 720         }
 721 }

 723 /// Store the relative direction from one hex to another in a WML variable.
 724 /// This is mainly useful as a diagnostic tool, but could be useful
 725 /// for some kind of scenario.
 726 WML_HANDLER_FUNCTION(store_relative_direction,, cfg)
 727 {
 728         if (!cfg.child("source")) {
 729                 WRN_NG << "No source in [store_relative_direction]" << std::endl;
 730                 return;
 731         }
 732         if (!cfg.child("destination")) {
 733                 WRN_NG << "No destination in [store_relative_direction]" << std::endl;
 734                 return;
 735         }
 736         if (!cfg.has_attribute("variable")) {
 737                 WRN_NG << "No variable in [store_relative_direction]" << std::endl;
 738                 return;
 739         }
 740 
 741         const map_location src = cfg_to_loc(cfg.child("source"));
 742         const map_location dst = cfg_to_loc(cfg.child("destination"));
 743 
 744         std::string variable = cfg["variable"];
 745         map_location::RELATIVE_DIR_MODE mode = static_cast<map_location::RELATIVE_DIR_MODE> (cfg["mode"].to_int(0));
 746         try
 747         {
 748                 variable_access_create store = resources::gamedata->get_variable_access_write(variable);
 749 
 750                 store.as_scalar() = map_location::write_direction(src.get_relative_dir(dst,mode));
 751         }
 752         catch(const invalid_variablename_exception&)
 753         {
 754                 ERR_NG << "Cannot do [store_relative_direction] with invalid destination variable: " << variable << " with " << cfg.get_config().debug() << std::endl;
 755         }

 857 {

 870         } else {
 871                 pathfind::teleport_group tunnel(delay ? cfg : vconfig(cfg.get_parsed_config()), false);
 872                 resources::tunnels->add(tunnel);

 876                         resources::tunnels->add(tunnel);

 881 /// If we should spawn a new unit on the map somewhere
 882 WML_HANDLER_FUNCTION(unit,, cfg)
 883 {
 884         config parsed_cfg = cfg.get_parsed_config();
 885 
 886         config::attribute_value to_variable = cfg["to_variable"];
 887         if (!to_variable.blank())
 888         {
 889                 parsed_cfg.remove_attribute("to_variable");
 890                 unit new_unit(parsed_cfg, true, &cfg);
 891                 try
 892                 {
 893                         config &var = resources::gamedata->get_variable_cfg(to_variable);
 894                         var.clear();
 895                         new_unit.write(var);
 896                         if (const config::attribute_value *v = parsed_cfg.get("x")) var["x"] = *v;
 897                         if (const config::attribute_value *v = parsed_cfg.get("y")) var["y"] = *v;
 898                 }
 899                 catch(const invalid_variablename_exception&)
 900                 {
 901                         ERR_NG << "Cannot do [unit] with invalid to_variable:  " << to_variable << " with " << cfg.get_config().debug() << std::endl;
 902                 }
 903                 return;
 904 
 905         }
 906 
 907         int side = parsed_cfg["side"].to_int(1);
 908 
 909 
 910         if ((side<1)||(side > static_cast<int>(resources::gameboard->teams().size()))) {
 911                 ERR_NG << "wrong side in [unit] tag - no such side: "<<side<<" ( number of teams :"<<resources::gameboard->teams().size()<<")"<<std::endl;
 912                 DBG_NG << parsed_cfg.debug();
 913                 return;
 914         }
 915         team &tm = resources::gameboard->get_team(side);
 916 
 917         unit_creator uc(tm,resources::gameboard->map().starting_position(side));
 918 
 919         uc
 920                 .allow_add_to_recall(true)
 921                 .allow_discover(true)
 922                 .allow_get_village(true)
 923                 .allow_invalidate(true)
 924                 .allow_rename_side(true)
 925                 .allow_show(true);
 926 
 927         uc.add_unit(parsed_cfg, &cfg);
 928 
 929 }
