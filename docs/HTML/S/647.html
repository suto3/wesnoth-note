  36 /**

  39  * @note In this context a widget is a @ref gui2::styled_widget and not a @ref

  56         }

  62         /**
  63          * Attaches the field to a window.
  64          *
  65          * When attached the widget which we're a wrapper around is stored linked
  66          * in here.
  67          *
  68          * @warning After attaching the window must remain a valid. Before the
  69          * window is destroyed the @ref detach_from_window function must be called.
  70          *
  71          * @todo Most functions that have a window parameter only use it to get the
  72          * widget. Evaluate and remove the window parameter where applicable.
  73          *
  74          * @pre widget_ == nullptr
  75          *
  76          * @param window               The window to be attached to.
  77          */
  78         void attach_to_window(window& window)
  79         {
  80                 assert(!widget_);
  81                 widget_ = find_widget<styled_widget>(&window, id(), false, mandatory_);
  82         }

  84         /**
  85          * Initializes the widget.
  86          *
  87          * This routine is called before the dialog is shown and the pre_show() is
  88          * called. So the user can override the values set. This routine does the
  89          * following:
  90          * - If no widget available exit gives feedback it the widget must exist.
  91          * - If a getter is defined we use to set value_ and the widget.
  92          * - If no setter is defined we use the widget value to set value_.
  93          *
  94          * The function calls two functions
  95          *  - init_generic which is to be used in the template subclass.
  96          *  - init_specialized which is to be used in subclasses of the template
  97          *     class. This way they can override this function without to use their
  98          *     signature to inherit.
  99          *
 100          * @param window              The window containing the widget.
 101          */
 102         void widget_init(window& window)
 103         {
 104                 init_generic(window);
 105                 init_specialized(window);
 106         }
 107 
 108         /**
 109          * Finalizes the widget.
 110          *
 111          * This routine is called after the dialog is closed with OK. It's called
 112          * before post_show(). This routine does the following:
 113          * - if no active widget available exit.
 114          * - if a setter is defined the widget value is saved in the setter.
 115          * - The widget value is saved in value_.
 116          *
 117          * Like widget_init it calls two functions with the same purpose.
 118          *
 119          * @param window              The window containing the widget.
 120          */
 121         void widget_finalize(window& window)
 122         {
 123                 finalize_generic(window);
 124                 finalize_specialized(window);
 125         }

 141          * It can be a window must be recreated, in that case the state needs to be

 159         /**
 160          * Enables a widget.
 161          *
 162          * @param window              The window containing the widget.
 163          * @param enable              If true enables the widget, disables
 164          *                            otherwise.
 165          * @param sync                If the state is changed do we need to
 166          *                            synchronize. Upon disabling, write the value
 167          *                            of the widget in the variable value_. Upon
 168          *                            enabling write the value of value_ in the
 169          *                            widget.
 170          */
 171         void widget_set_enabled(window& window, const bool enable, const bool sync)
 172         {
 173                 styled_widget* widget = dynamic_cast<styled_widget*>(window.find(id(), false));
 174 
 175                 if(!widget) {
 176                         return;
 177                 }
 178 
 179                 const bool widget_state = widget->get_active();
 180                 if(widget_state == enable) {
 181                         return;
 182                 }
 183 
 184                 if(sync) {
 185                         if(enable) {
 186                                 widget_restore(window);
 187                         } else {
 188                                 widget_save(window);
 189                         }
 190                 }
 191 
 192                 widget->set_active(enable);
 193         }

 245 /**
 246  * Template class to implement the generic field implementation.
 247  *
 248  * @tparam T                      The type of the item to show in the widget.
 249  * @tparam W                      The type of widget to show, this is not a
 250  *                                widget class but a behavior class.
 251  * @tparam CT                     The type tp be used in the
 252  *                                callback_save_value callback. Normally this
 253  *                                is const T but for example with strings it
 254  *                                can be const T&. Note the const needs to be
 255  *                                in the template otherwise compilation on
 256  *                                GCC-4.3 fails (not sure whether compiler bug
 257  *                                or not).
 258  */
 259 template <class T, class W, class CT>
 260 class field : public field_base
 261 {
 262 public:
 263         /**
 264          * Constructor.
 265          *
 266          * @param id                  The id of the widget to connect to the window.
 267          *                            A widget can only be connected once.
 268          * @param mandatory           Is the widget mandatory?
 269          * @param callback_load_value A callback function which is called when the
 270          *                            window is shown. This callback returns the
 271          *                            initial value of the field.
 272          * @param callback_save_value A callback function which is called when the
 273          *                            window closed with the OK button. The
 274          *                            callback is executed with the new value of
 275          *                            the field. It's meant to set the value of
 276          *                            some variable in the engine after the window
 277          *                            is closed with OK.
 278          */
 279         field(const std::string& id,
 280                    const bool mandatory,
 281                    const std::function<T()>& callback_load_value,
 282                    const std::function<void(CT)>& callback_save_value)
 283                 : field_base(id, mandatory)
 284                 , value_(T())
 285                 , link_(value_)
 286                 , callback_load_value_(callback_load_value)
 287                 , callback_save_value_(callback_save_value)
 288         {
 289                 static_assert((!std::is_same<styled_widget, W>::value), "Second template argument cannot be styled_widget");
 290         }
 291 
 292         /**
 293          * Constructor.
 294          *
 295          * @param id                  The id of the widget to connect to the window.
 296          *                            A widget can only be connected once.
 297          * @param mandatory           Is the widget mandatory?
 298          * @param linked_variable     The variable which is linked to the field.
 299          *                            * Upon loading its value is used as initial
 300          *                              value of the widget.
 301          *                            * Upon closing:
 302          *                              * with OK its value is set to the value of
 303          *                                the widget.
 304          *                              * else, its value is undefined.
 305          */
 306         field(const std::string& id, const bool mandatory, T& linked_variable)
 307                 : field_base(id, mandatory)
 308                 , value_(T())
 309                 , link_(linked_variable)
 310                 , callback_load_value_(std::function<T()>())
 311                 , callback_save_value_(std::function<void(CT)>())
 312         {
 313                 static_assert((!std::is_same<styled_widget, W>::value), "Second template argument cannot be styled_widget");
 314         }
 315 
 316         /**
 317          * Constructor.
 318          *
 319          * This version is used for read only variables.
 320          *
 321          * @note The difference between this constructor and the one above is the
 322          * sending of the third parameter as const ref instead of a non-const ref.
 323          * So it feels a bit tricky. Since this constructor is only used for a
 324          * the @ref styled_widget class and the other constructors not the issue is
 325          * solved by using static asserts to test whether the proper constructor
 326          * is used.
 327          *
 328          * @param mandatory            Is the widget mandatory?
 329          * @param id                  The id of the widget to connect to the window.
 330          *                            A widget can only be connected once.
 331          * @param value               The value of the widget.
 332          */
 333         field(const std::string& id, const bool mandatory, const T& value)
 334                 : field_base(id, mandatory)
 335                 , value_(value)
 336                 , link_(value_)
 337                 , callback_load_value_(std::function<T()>())
 338                 , callback_save_value_(std::function<void(CT)>())
 339         {
 340                 static_assert((std::is_same<styled_widget, W>::value), "Second template argument must be styled_widget");

 382                 save(window, false);
 383         }

 385         /**
 386          * Gets the value of the field.
 387          *
 388          * This function gets the value of the widget and stores that in the
 389          * internal cache, then that value is returned.
 390          *
 391          * @deprecated Use references to a variable instead.
 392          *
 393          * @param window              The window containing the widget.
 394          *
 395          * @returns                   The current value of the widget.

 397         T get_widget_value(window& window)

 408         T value_;

 413          * When set determines the initial value and the final value is stored here

 416         T& link_;

 418         /**

 423         std::function<T()> callback_load_value_;

 425         /** Inherited from field_base. */
 426         void init_generic(window& window)
 427         {
 428                 validate_widget(window);
 429 
 430                 if(callback_load_value_) {
 431                         value_ = callback_load_value_();
 432                 } else {
 433                         value_ = link_;
 434                 }
 435 
 436                 restore(window);
 437         }
 438 
 439         /** Inherited from field_base. */
 440         void finalize_generic(window& window)
 441         {
 442                 save(window, true);
 443 
 444                 if(callback_save_value_) {
 445                         callback_save_value_(value_);
 446                 } else {
 447                         link_ = value_;
 448                 }
 449         }
 450 
 451         /**
 452          * The callback function to save the value.
 453          *
 454          * Once the dialog has been successful this function is used to store the
 455          * result of this widget.
 456          */
 457         std::function<void(CT)> callback_save_value_;
 458 
 459         /**
 460          * Test whether the widget exists if the widget is mandatory.
 461          *
 462          * @param window              The window containing the widget.
 463          */
 464         void validate_widget(window& window)
 465         {
 466                 if(!is_mandatory()) {
 467                         return;
 468                 }
 469                 find_widget<const W>(&window, id(), false);
 470         }

 472         /**
 473          * Stores the value in the widget in the interval value_.
 474          *
 475          * @param window              The window containing the widget.
 476          * @param must_be_active      If true only active widgets will store their
 477          *value.
 478          */
 479         void save(window& window, const bool must_be_active);
 480 
 481         /**
 482          * Stores the internal value_ in the widget.
 483          *
 484          * @param window              The window containing the widget.
 485          */
 486         void restore(window& window);
