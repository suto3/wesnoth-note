 139 bool get_village(const map_location& loc, int side, bool *action_timebonus, bool fire_event)
 140 {
 141         std::vector<team> &teams = resources::gameboard->teams();
 142         team *t = unsigned(side - 1) < teams.size() ? &teams[side - 1] : nullptr;
 143         if (t && t->owns_village(loc)) {
 144                 return false;
 145         }
 146 
 147         bool not_defeated = t && !resources::gameboard->team_is_defeated(*t);
 148 
 149         bool grants_timebonus = false;
 150 
 151         int old_owner_side = 0;
 152         // We strip the village off all other sides, unless it is held by an ally
 153         // and our side is already defeated (and thus we can't occupy it)
 154         for(std::vector<team>::iterator i = teams.begin(); i != teams.end(); ++i) {
 155                 int i_side = i - teams.begin() + 1;
 156                 if (!t || not_defeated || t->is_enemy(i_side)) {
 157                         if(i->owns_village(loc)) {
 158                                 old_owner_side = i_side;
 159                                 i->lose_village(loc);
 160                         }
 161                         if (side != i_side && action_timebonus) {
 162                                 grants_timebonus = true;
 163                         }
 164                 }
 165         }
 166 
 167         if (!t) return false;
 168 
 169         if(grants_timebonus) {
 170                 t->set_action_bonus_count(1 + t->action_bonus_count());
 171                 *action_timebonus = true;
 172         }
 173 
 174         if(not_defeated) {
 175                 if (resources::screen != nullptr) {
 176                         resources::screen->invalidate(loc);
 177                 }
 178                 return t->get_village(loc, old_owner_side, fire_event ? resources::gamedata : nullptr);
 179         }
 180 
 181         return false;
 182 }

 416         /**
 417          * Checks for ambushers around @a hex, setting flags as appropriate.
 418          */
 419         inline void unit_mover::check_for_ambushers(const map_location & hex)
 420         {
 421                 const unit_map &units = resources::gameboard->units();
 422 
 423                 // Need to check each adjacent hex for hidden enemies.
 424                 map_location adjacent[6];
 425                 get_adjacent_tiles(hex, adjacent);
 426                 for ( int i = 0; i != 6; ++i )
 427                 {
 428                         const unit_map::const_iterator neighbor_it = units.find(adjacent[i]);
 429 
 430                         if ( neighbor_it != units.end()  &&
 431                              current_team_->is_enemy(neighbor_it->side())  &&
 432                              neighbor_it->invisible(adjacent[i], *resources::gameboard) )
 433                         {
 434                                 // Ambushed!
 435                                 ambushed_ = true;
 436                                 ambush_stop_ = hex;
 437                                 ambushers_.push_back(adjacent[i]);
 438                         }
 439                 }
 440         }

 443         /**
 444          * Makes sure the path is not obstructed by a unit.
 445          * @param  hex       The hex to check.
 446          * @param  prev_hex  The previous hex in the route (used to detect a teleport).
 447          * @return true if @a hex is obstructed.
 448          */
 449         inline bool unit_mover::check_for_obstructing_unit(const map_location & hex,
 450                                                            const map_location & prev_hex)
 451         {
 452                 const unit_map::const_iterator blocking_unit = resources::gameboard->units().find(hex);
 453 
 454                 // If no unit, then the path is not obstructed.
 455                 if (blocking_unit == resources::gameboard->units().end()) {
 456                         return false;
 457                 }
 458 
 459                 // Check for units blocking a teleport exit. This can now only happen
 460                 // if these units are not visible to the current side, as otherwise no
 461                 // valid path is found.
 462                 if ( !tiles_adjacent(hex, prev_hex) ) {
 463                         if ( current_team_->is_enemy(blocking_unit->side()) ) {
 464                                 // Enemy units always block the tunnel.
 465                                 teleport_failed_ = true;
 466                                 return true;
 467                         } else {
 468                                 // By contrast, allied units (of a side which does not share vision) only
 469                                 // block the tunnel if pass_allied_units=true. Whether the teleport is possible
 470                                 // is checked by getting the teleport map with the see_all flag set to true.
 471                                 const pathfind::teleport_map teleports = pathfind::get_teleport_locations(*move_it_, *current_team_, true, false, false);
 472                                 std::set<map_location> allowed_teleports;
 473                                 teleports.get_adjacents(allowed_teleports, prev_hex);
 474 
 475                                 bool found_valid_teleport = false;
 476                                 std::set<map_location>::const_iterator it = allowed_teleports.begin();
 477                                 while( it!=allowed_teleports.end() && !found_valid_teleport ) {
 478                                         if ( *it == hex ) {
 479                                                 found_valid_teleport = true;
 480                                         }
 481                                         ++it;
 482                                 }
 483 
 484                                 if (!found_valid_teleport) {
 485                                         teleport_failed_ = true;
 486                                         return true;
 487                                 }
 488                         }
 489                 }
 490 
 491                 if ( current_team_->is_enemy(blocking_unit->side()) ) {
 492                         // Trying to go through an enemy.
 493                         blocked_loc_ = hex;
 494                         return true;
 495                 }
 496 
 497                 // If we get here, the unit does not interfere with movement.
 498                 return false;
 499         }

 502         /**
 503          * Moves the unit the next step.
 504          * @a step_to is the hex being moved to.
 505          * @a step_from is the hex before that in the route.
 506          * (The unit is actually at *move_loc_.)
 507          * @a animator is the unit_display::unit_mover being used.
 508          * @return whether or not we started a new animation.
 509          */
 510         inline bool unit_mover::do_move(const route_iterator & step_from,
 511                                         const route_iterator & step_to,
 512                                         unit_display::unit_mover & animator)
 513         {
 514                 game_display &disp = *resources::screen;
 515 
 516                 // Adjust the movement even if we cannot move yet.
 517                 // We will eventually be able to move if nothing unexpected
 518                 // happens, and if something does happen, this movement is the
 519                 // cost to discover it.
 520                 move_it_->set_movement(moves_left_.front(), true);
 521                 moves_left_.pop_front();
 522 
 523                 // Invalidate before moving so we invalidate neighbor hexes if needed.
 524                 move_it_->anim_comp().invalidate(disp);
 525 
 526                 // Attempt actually moving.
 527                 // (Fails if *step_to is occupied).
 528                 std::pair<unit_map::iterator, bool> move_result =
 529                         resources::gameboard->units().move(*move_loc_, *step_to);
 530                 if ( move_result.second )
 531                 {
 532                         // Update the moving unit.
 533                         move_it_ = move_result.first;
 534                         move_it_->set_facing(step_from->get_relative_dir(*step_to));
 535                         // Disable bars. The expectation here is that the animation
 536                         // unit_mover::finish() will clean after us at a later point. Ugly,
 537                         // but it works.
 538                         move_it_->anim_comp().set_standing(false);
 539                         disp.invalidate_unit_after_move(*move_loc_, *step_to);
 540                         disp.invalidate(*step_to);
 541                         move_loc_ = step_to;
 542 
 543                         // Show this move.
 544                         const size_t current_tracking = resources::game_events->pump().wml_tracking();
 545                         animator.proceed_to(move_it_.get_shared_ptr(), step_to - begin_,
 546                                             current_tracking != do_move_track_, false);
 547                         do_move_track_ = current_tracking;
 548                         disp.redraw_minimap();
 549                 }
 550 
 551                 return move_result.second;
 552         }

 587          * (E.g. it would not kill movement by making an unintended village capture.)

 590         {
 591                 // We cannot reasonably stop if move_it_ could not be moved to this

 593                 if (*move_loc_ != hex) {

 595                 }

 598                 return !resources::gameboard->map().is_village(hex) ||
 599                        current_team_->owns_village(hex);
 600         }

 630         /**
 631          * Makes sure the units in ambushers_ still exist.
 632          */
 633         inline void unit_mover::validate_ambushers()
 634         {
 635                 const unit_map &units = resources::gameboard->units();
 636 
 637                 // Loop through the previously-detected ambushers.
 638                 size_t i = 0;
 639                 while ( i != ambushers_.size() ) {
 640                         if (units.count(ambushers_[i]) == 0) {
 641                                 // Ambusher is gone.
 642                                 ambushers_.erase(ambushers_.begin() + i);

 646                                 ++i;

 654         /**
 655          * Checks the expected route for hidden units.
 656          * This basically handles all the checks for surprises that can be done
 657          * without visibly notifying a player. Thus this can be called at the
 658          * start of movement and immediately after events, rather than tie up
 659          * CPU time in the middle of animating a move.
 660          *
 661          * @param[in]  start           The beginning of the path to check.
 662          * @param[in]  stop            The end of the path to check.
 663          */
 664         void unit_mover::cache_hidden_units(const route_iterator & start,
 665                                             const route_iterator & stop)
 666         {
 667                 // Clear the old cache.
 668                 obstructed_ = full_end_;
 669                 blocked_loc_ = map_location::null_location();
 670                 teleport_failed_ = false;
 671                 // The ambush cache needs special treatment since we cannot re-detect
 672                 // an ambush if we are already at the ambushed location.
 673                 ambushed_ =  ambushed_  &&  ambush_stop_ == *start;
 674                 if ( ambushed_ ) {
 675                         validate_ambushers();
 676                         ambushed_ = !ambushers_.empty();
 677                 }
 678                 if ( !ambushed_ ) {
 679                         ambush_stop_ = map_location::null_location();
 680                         ambushers_.clear();
 681                 }
 682 
 683                 // Update the shroud clearer.
 684                 clearer_.cache_units(current_uses_fog_ ? current_team_ : nullptr);
 685 
 686 
 687                 // Abort for null routes.
 688                 if ( start == stop ) {
 689                         ambush_limit_ = start;
 690                         return;
 691                 }
 692 
 693                 // This loop will end with ambush_limit_ pointing one element beyond
 694                 // where the unit would be forced to stop by a hidden unit.
 695                 for ( ambush_limit_ = start+1; ambush_limit_ != stop; ++ambush_limit_ ) {
 696                         // Check if we need to stop in the previous hex.
 697                         if ( ambushed_ ) {
 698                                         break;
 699                         }
 700                         // Check for being unable to enter this hex.
 701                         if ( check_for_obstructing_unit(*ambush_limit_, *(ambush_limit_-1)) ) {
 702                                 // No replay check here? Makes some sense, I guess.
 703                                 obstructed_ = ambush_limit_++; // The limit needs to be after obstructed_ in order for the latter to do anything.
 704                                 break;
 705                         }
 706 
 707                         // We can enter this hex.
 708                         // See if we are stopped in this hex.
 709                         check_for_ambushers(*ambush_limit_);
 710                 }
 711         }

 714         /**
 715          * Fires the enter_hex or exit_hex event and updates our data as needed.
 716          *
 717          * @param[in]  event_name  The name of the event ("enter_hex" or "exit_hex").
 718          * @param[in]  current     The currently occupied hex.
 719          * @param[in]  other       The secondary hex to provide to the event.
 720          *
 721          * @return true if this event should interrupt movement.
 722          * (This is also stored in event_mutated_.)
 723          */
 724         bool unit_mover::fire_hex_event(const std::string & event_name,
 725                                             const route_iterator & current,
 726                                         const route_iterator & other)
 727         {
 728                 const size_t track = resources::game_events->pump().wml_tracking();
 729                 bool valid = true;
 730 
 731                 const game_events::entity_location mover(*move_it_, *current);
 732                 const bool event = resources::game_events->pump().fire(event_name, mover, *other);
 733 
 734                 if (track != resources::game_events->pump().wml_tracking()) {
 735                         // Some WML fired, so update our status.
 736                         valid = post_wml(current);
 737                 }
 738                 if (event || !valid) {
 739                         event_mutated_ = true;
 740                 }
 741                 return event || !valid;
 742         }

 745         /**
 746          * Checks how far it appears we can move this turn.
 747          *
 748          * @param[in] start         The beginning of the plotted path.
 749          * @param[in] stop          The end of the plotted path.
 750          *
 751          * @return An end iterator for the path that can be traversed this turn.
 752          */
 753         unit_mover::route_iterator unit_mover::plot_turn(const route_iterator & start,
 754                                                          const route_iterator & stop)
 755         {
 756                 const gamemap &map = resources::gameboard->map();
 757 
 758                 // Handle null routes.
 759                 if (start == stop) {
 760                         return start;
 761                 }
 762 
 763                 int remaining_moves = move_it_->movement_left();
 764                 zoc_stop_ = map_location::null_location();
 765                 moves_left_.clear();
 766 
 767                 if ( start != begin_ ) {
 768                         // Check for being unable to leave the current hex.
 769                         if ( !move_it_->get_ability_bool("skirmisher", *start, *resources::gameboard) &&
 770                               pathfind::enemy_zoc(*current_team_, *start, *current_team_) )
 771                                 zoc_stop_ = *start;
 772                 }
 773 
 774                 // This loop will end with end pointing one element beyond where the
 775                 // unit thinks it will stop (the usual notion of "end" for iterators).
 776                 route_iterator end = start + 1;
 777                 for ( ; end != stop; ++end )
 778                 {
 779                         // Break out of the loop if we cannot leave the previous hex.
 780                         if ( zoc_stop_ != map_location::null_location() ) {
 781                                 break;
 782                         }
 783                         remaining_moves -= move_it_->movement_cost(map[*end]);
 784                         if ( remaining_moves < 0 ) {
 785                                         break;
 786                         }
 787 
 788                         // We can enter this hex. Record the cost.
 789                         moves_left_.push_back(remaining_moves);
 790 
 791                         // Check for being unable to leave this hex.
 792                         if (!move_it_->get_ability_bool("skirmisher", *end, *resources::gameboard) &&
 793                                 pathfind::enemy_zoc(*current_team_, *end, *current_team_))
 794                         {
 795                                 zoc_stop_ = *end;
 796                         }
 797                 }
 798 
 799                 route_iterator min_end =  start == begin_ ? start : start + 1;
 800                 while (end != min_end  &&  resources::gameboard->has_visible_unit(*(end - 1), *current_team_)) {
 801                         // Backtrack.
 802                         --end;
 803                 }
 804 
 805                 return end;
 806         }

 809         /**
 810          * Updates our stored info after a WML event might have changed something.
 811          *
 812          * @param step      Indicates the position in the path where we might need to start recalculating movement.
 813          *                  Set this to full_end_ (or do not supply it) to skip such recalculations (because movement has finished).
 814          *
 815          * @returns false if continuing is impossible (i.e. we lost the moving unit).
 816          */
 817         bool unit_mover::post_wml(const route_iterator & step)
 818         {
 819                 // Re-find the moving unit.
 820                 move_it_ = resources::gameboard->units().find(*move_loc_);
 821                 const bool found = move_it_ != resources::gameboard->units().end();
 822 
 823                 // Update the current unit data.
 824                 current_side_ = found ? move_it_->side() : orig_side_;
 825                 current_team_ = &resources::gameboard->get_team(current_side_);
 826                 current_uses_fog_ = current_team_->fog_or_shroud()  &&
 827                                     ( current_side_ != orig_side_  ||
 828                                       current_team_->auto_shroud_updates() );
 829 
 830                 // Update the path.
 831                 if ( found  &&  step != full_end_ ) {
 832                         const route_iterator new_limit = plot_turn(step, expected_end_);
 833                         cache_hidden_units(step, new_limit);
 834                         // Just in case: length 0 paths become length 1 paths.
 835                         if (ambush_limit_ == step) {
 836                                 ++ambush_limit_;
 837                         }
 838                 }
 839 
 840                 return found;
 841         }

 845          * Fires the sighted events that were raised earlier.
 846          *
 847          * @param[in]  from  Points to the hex the sighting unit currently occupies.
 848          *
 849          * @return sets event_mutated_ to true if this event should interrupt movement.
 850          */
 851         void unit_mover::pump_sighted(const route_iterator & from)
 852         {
 853                 const size_t track = resources::game_events->pump().wml_tracking();
 854                 bool valid = true;
 855

 859                         // Some WML fired, so update our status.
 860                         valid = post_wml(from);
 861                 }
 862                 if (non_undoable_events_happened || !valid) {
 863                         event_mutated_ = true;
 864                 }
 865         }

 889          * May fire "sighted" events.

 891         void unit_mover::reveal_ambusher(const map_location & hex, bool update_alert)

 935          * This is based solely on data known to the player, and will not plot a move

 950         /**
 951          * Attempts to move the unit along the expected path.
 952          * (This will do nothing unless check_expected_movement() was called first.)
 953          *
 954          * @param[in]  show            Set to false to suppress animations.
 955          */
 956         void unit_mover::try_actual_movement(bool show)
 957         {
 958                 static const std::string enter_hex_str("enter hex");
 959                 static const std::string exit_hex_str("exit hex");
 960 
 961 
 962                 bool obstructed_stop = false;
 963 
 964 
 965                 // Check for hidden units along the expected path before we start
 966                 // animating and firing events.
 967                 cache_hidden_units(begin_, expected_end_);
 968 
 969                 if ( begin_ != ambush_limit_ ) {
 970                         // Cache the moving unit's visibility.
 971                         std::vector<int> not_seeing = get_sides_not_seeing(*move_it_);
 972 
 973                         // Prepare to animate.
 974                         unit_display::unit_mover animator(route_, show);
 975                         animator.start(move_it_.get_shared_ptr());
 976 
 977                         // Traverse the route to the hex where we need to stop.
 978                         // Each iteration performs the move from real_end_-1 to real_end_.
 979                         for ( real_end_ = begin_+1; real_end_ != ambush_limit_; ++real_end_ ) {
 980                                 const route_iterator step_from = real_end_ - 1;
 981 
 982                                 // See if we can leave *step_from.
 983                                 // Already accounted for: ambusher
 984                                 if ( event_mutated_ )
 985                                 {
 986                                         break;
 987                                 }
 988                                 if ( sighted_ && is_reasonable_stop(*step_from) )
 989                                 {
 990                                         sighted_stop_ = true;
 991                                         break;
 992                                 }
 993                                 // Already accounted for: ZoC
 994                                 // Already accounted for: movement cost
 995                                 if ( fire_hex_event(exit_hex_str, step_from, real_end_) ) {
 996                                         break;
 997                                 }
 998                                 if ( real_end_ == obstructed_ ) {
 999                                         // We did not check for being a replay when checking for an
1000                                         // obstructed hex, so we do not check can_break here.
1001                                         obstructed_stop = true;
1002                                         break;
1003                                 }
1004 
1005                                 // We can leave *step_from. Make the move to *real_end_.
1006                                 bool new_animation = do_move(step_from, real_end_, animator);
1007                                 // Update the fog.
1008                                 if ( current_uses_fog_ )
1009                                         handle_fog(*real_end_, new_animation);
1010                                 animator.wait_for_anims();
1011 
1012                                 // Fire the events for this step.
1013                                 // (These return values are not checked since real_end_ still
1014                                 // needs to be incremented. The event_mutated_ check will break
1015                                 // us out of the loop if needed.)
1016                                 fire_hex_event(enter_hex_str, real_end_, step_from);
1017                                 // Sighted events only fire if we could stop due to sighting.
1018                                 if (is_reasonable_stop(*real_end_)) {
1019                                         pump_sighted(real_end_);
1020                                 }
1021                         }//for
1022                         // Make sure any remaining sighted events get fired.
1023                         pump_sighted(real_end_-1);
1024 
1025                         if ( move_it_.valid() ) {
1026                                 // Finish animating.
1027                                 animator.finish(move_it_.get_shared_ptr());
1028                                 // Check for the moving unit being seen.
1029                                 event_mutated_ |= actor_sighted(*move_it_, &not_seeing);
1030                         }
1031                 }//if
1032 
1033                 // Some flags were set to indicate why we might stop.
1034                 // Update those to reflect whether or not we got to them.
1035                 ambushed_ = ambushed_ && real_end_ == ambush_limit_;
1036                 if (!obstructed_stop) {
1037                         blocked_loc_ = map_location::null_location();
1038                 }
1039                 teleport_failed_ = teleport_failed_ && obstructed_stop;
1040                 // event_mutated_ does not get unset, regardless of other reasons
1041                 // for stopping, but we do save its current value.
1042                 event_mutated_mid_move_ = event_mutated_;
1043         }

1046         /**
1047          * Does some bookkeeping and event firing, for use after movement.
1048          * This includes village capturing and the undo stack.
1049          */
1050         void unit_mover::post_move(undo_list *undo_stack)
1051         {
1052                 const map_location & final_loc = final_hex();
1053 
1054                 int orig_village_owner = 0;
1055                 bool action_time_bonus = false;
1056 
1057                 // Reveal ambushers?
1058                 if (ambushed_ || blocked()) {
1059                         reveal_ambushers();
1060                 }
1061                 else if (teleport_failed_ && spectator_) {
1062                         spectator_->set_failed_teleport(resources::gameboard->units().find(*obstructed_));
1063                 }
1064                 unit::clear_status_caches();
1065 
1066                 if ( move_it_.valid() ) {
1067                         // Update the moving unit.
1068                         move_it_->set_interrupted_move(
1069                                 sighted_stop_ && !resources::whiteboard->is_executing_actions() ?
1070                                         *(full_end_-1) :
1071                                         map_location::null_location());
1072                         if (ambushed_ || final_loc == zoc_stop_) {
1073                                 move_it_->set_movement(0, true);
1074                         }
1075 
1076                         // Village capturing.
1077                         if ( resources::gameboard->map().is_village(final_loc) ) {
1078                                 // Is this a capture?
1079                                 orig_village_owner = resources::gameboard->village_owner(final_loc) + 1;
1080                                 if ( orig_village_owner != current_side_) {
1081                                         // Captured. Zap movement and take over the village.
1082                                         move_it_->set_movement(0, true);
1083                                         event_mutated_ |= get_village(final_loc, current_side_, &action_time_bonus);
1084                                         post_wml();
1085                                 }
1086                         }
1087                 }
1088 
1089                 // Finally, the moveto event.
1090                 event_mutated_ |= resources::game_events->pump().fire("moveto", final_loc, *begin_);
1091                 post_wml();
1092 
1093                 // Record keeping.
1094                 if (spectator_) {
1095                         spectator_->set_unit(move_it_);
1096                 }
1097                 if ( undo_stack ) {
1098                         const bool mover_valid = move_it_.valid();
1099 
1100                         if ( mover_valid ) {
1101                                 // MP_COUNTDOWN: added param
1102                                 undo_stack->add_move(
1103                                         move_it_.get_shared_ptr(), begin_, real_end_, orig_moves_,
1104                                         action_time_bonus, orig_village_owner, orig_dir_);
1105                         }
1106 
1107                         if ( !mover_valid  ||  undo_blocked()  ||
1108                                 (resources::whiteboard->is_active() && resources::whiteboard->should_clear_undo()) || !synced_context::can_undo())
1109                         {
1110                                 undo_stack->clear();
1111                         }
1112                 }
1113 
1114                 // Update the screen.
1115                 resources::screen->redraw_minimap();
1116                 resources::screen->draw();
1117         }

1202 static size_t move_unit_internal(undo_list* undo_stack,
1203                  bool show_move,
1204                  bool* interrupted,
1205                                  unit_mover& mover)
1206 {

1208         // Default return value.
1209         if (interrupted) {
1210                 *interrupted = false;
1211         }
1212 
1213         // Attempt moving.
1214         mover.try_actual_movement(show_move);
1215 
1216         config co;
1217         config cn {
1218                 "stopped_early", mover.stopped_early(),
1219                 "final_hex_x", mover.final_hex().wml_x(),
1220                 "final_hex_y", mover.final_hex().wml_y(),
1221         };
1222         bool matches_replay = checkup_instance->local_checkup(cn,co);
1223         if(!matches_replay)
1224         {
1225                 replay::process_error("calculated movement destination (x="+ cn["final_hex_x"].str() +  " y=" + cn["final_hex_y"].str() +
1226                         ") didn't match the original destination(x="+ co["final_hex_x"].str() +  " y=" + co["final_hex_y"].str() + ")\n");
1227 
1228                 //TODO: move the unit by force to the desired destination with something like mover.reset_final_hex(co["x"], co["y"]);
1229         }
1230 
1231         // Bookkeeping, etc.
1232         // also fires the moveto event
1233         mover.post_move(undo_stack);
1234         if (show_move) {
1235                 mover.feedback();
1236         }
1237 
1238         // Set return value.
1239         if (interrupted) {
1240                 *interrupted = mover.interrupted();
1241         }
1242 
1243         return mover.steps_travelled();
1244 }

1246 /**
1247  * Moves a unit across the board.
1248  *
1249  * This function handles actual movement, checking terrain costs as well as
1250  * things that might interrupt movement (e.g. ambushes). If the full path
1251  * cannot be reached this turn, the remainder is stored as the unit's "goto"
1252  * instruction. (The unit itself is whatever unit is at the beginning of the
1253  * supplied path.)
1254  *
1255  * @param[in]  steps                The route to be traveled. The unit to be moved is at the beginning of this route.
1256  * @param      undo_stack           If supplied, then either this movement will be added to the stack or the stack will be cleared.
1257  * @param[in]  continued_move       If set to true, this is a continuation of an earlier move (movement is not interrupted should units be spotted).
1258  * @param[in]  show_move            Controls whether or not the movement is animated for the player.
1259  * @param[out] interrupted          If supplied, then this is set to true if information was uncovered that warrants interrupting a chain of actions (and set to false otherwise).
1260  * @param[out] move_spectator       If supplied, this will be given the information uncovered by the move (and the unit's "goto" instruction will be preserved).
1261  *
1262  * @returns The number of hexes entered. This can safely be used as an index
1263  *          into @a steps to get the location where movement ended, provided
1264  *          @a steps is not empty (the return value is guaranteed to be less
1265  *          than steps.size() ).
1266  */
1267 size_t move_unit_and_record(const std::vector<map_location> &steps,
1268                  undo_list* undo_stack,
1269                  bool continued_move, bool show_move,
1270                  bool* interrupted,
1271                  move_unit_spectator* move_spectator)
1272 {
1273 
1274         // Avoid some silliness.
1275         if ( steps.size() < 2  ||  (steps.size() == 2 && steps.front() == steps.back()) ) {
1276                 DBG_NG << "Ignoring a unit trying to jump on its hex at " <<
1277                           ( steps.empty() ? map_location::null_location() : steps.front() ) << ".\n";
1278                 return 0;
1279         }
1280         //if we have no fog activated then we always skip sighted
1281         if(resources::gameboard->units().find(steps.front()) != resources::gameboard->units().end())
1282         {
1283                 const team &current_team = resources::gameboard->teams()[
1284                         resources::gameboard->units().find(steps.front())->side() - 1];
1285                 continued_move |= !current_team.fog_or_shroud();
1286         }
1287         const bool skip_ally_sighted = !preferences::interrupt_when_ally_sighted();
1288 
1289         // Evaluate this move.
1290         unit_mover mover(steps, move_spectator, continued_move, skip_ally_sighted);
1291         if ( !mover.check_expected_movement() )
1292                 return 0;
1293         if(synced_context::get_synced_state() != synced_context::SYNCED)
1294         {
1295                 /*
1296                         enter the synced mode and do the actual movement.
1297                 */
1298                 resources::recorder->add_synced_command("move",replay_helper::get_movement(steps, continued_move, skip_ally_sighted));
1299                 set_scontext_synced sync;
1300                 size_t r =  move_unit_internal(undo_stack, show_move, interrupted, mover);
1301                 resources::controller->check_victory();
1302                 resources::controller->maybe_throw_return_to_play_side();
1303                 sync.do_final_checkup();
1304                 return r;
1305         }
1306         else
1307         {
1308                 //we are already in synced mode and don't need to reenter it again.
1309                 return  move_unit_internal(undo_stack, show_move, interrupted, mover);
1310         }
1311 }

1313 size_t move_unit_from_replay(const std::vector<map_location> &steps,
1314                  undo_list* undo_stack,
1315                  bool continued_move,bool skip_ally_sighted, bool show_move)
1316 {
1317         // Evaluate this move.
1318         unit_mover mover(steps, nullptr, continued_move,skip_ally_sighted);
1319         if ( !mover.check_expected_movement() )
1320         {
1321                 replay::process_error("found corrupt movement in replay.");
1322                 return 0;
1323         }
1324 
1325         return move_unit_internal(undo_stack, show_move, nullptr, mover);
1326 }
