  60 const std::set<std::string> get_recruits(int side, const map_location &recruit_loc)
  61 {
  62         const team & current_team = resources::gameboard->get_team(side);
  63 
  64         LOG_NG << "getting recruit list for side " << side << " at location " << recruit_loc << "\n";
  65 
  66         std::set<std::string> local_result;
  67         std::set<std::string> global_result;
  68         unit_map::const_iterator u = resources::gameboard->units().begin(),
  69                         u_end = resources::gameboard->units().end();
  70 
  71         bool leader_in_place = false;
  72         bool allow_local = resources::gameboard->map().is_castle(recruit_loc);
  73 
  74 
  75         // Check for a leader at recruit_loc (means we are recruiting from there,
  76         // rather than to there).
  77         unit_map::const_iterator find_it = resources::gameboard->units().find(recruit_loc);
  78         if ( find_it != u_end ) {
  79                 if ( find_it->can_recruit()  &&  find_it->side() == side  &&
  80                      resources::gameboard->map().is_keep(recruit_loc) )
  81                 {
  82                         // We have been requested to get the recruit list for this
  83                         // particular leader.
  84                         leader_in_place = true;
  85                         local_result.insert(find_it->recruits().begin(),
  86                                             find_it->recruits().end());
  87                 }
  88                 else if ( find_it->is_visible_to_team(current_team, *resources::gameboard, false) )
  89                 {
  90                         // This hex is visibly occupied, so we cannot recruit here.
  91                         allow_local = false;
  92                 }
  93         }
  94 
  95         if ( !leader_in_place ) {
  96                 // Check all leaders for their ability to recruit here.
  97                 for( ; u != u_end; ++u ) {
  98                         // Only consider leaders on this side.
  99                         if ( !(u->can_recruit() && u->side() == side) )
 100                                 continue;
 101 
 102                         // Check if the leader is on a connected keep.
 103                         if ( allow_local && dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*u, recruit_loc) ) {
 104                                 leader_in_place= true;
 105                                 local_result.insert(u->recruits().begin(), u->recruits().end());
 106                         }
 107                         else if ( !leader_in_place )
 108                                 global_result.insert(u->recruits().begin(), u->recruits().end());
 109                 }
 110         }
 111 
 112         // Determine which result set to use.
 113         std::set<std::string> & result = leader_in_place ? local_result : global_result;
 114 
 115         // Add the team-wide recruit list.
 116         const std::set<std::string>& recruit_list = current_team.recruits();
 117         result.insert(recruit_list.begin(), recruit_list.end());
 118 
 119         return result;
 120 }

 123 namespace { // Helpers for get_recalls()
 124         /**
 125          * Adds to @a result those units that @a leader (assumed a leader) can recall.
 126          * If @a already_added is supplied, it contains the underlying IDs of units
 127          * that can be skipped (because they are already in @a result), and the
 128          * underlying ID of units added to @a result will be added to @a already_added.
 129          */
 130         void add_leader_filtered_recalls(const unit_const_ptr leader,
 131                                          std::vector< unit_const_ptr > & result,
 132                                          std::set<size_t> * already_added = nullptr)
 133         {
 134                 const team& leader_team = resources::gameboard->get_team(leader->side());
 135                 const std::string& save_id = leader_team.save_id();
 136 
 137                 const unit_filter ufilt(vconfig(leader->recall_filter()), resources::filter_con);
 138 
 139                 for (const unit_const_ptr & recall_unit_ptr : leader_team.recall_list())
 140                 {
 141                         const unit & recall_unit = *recall_unit_ptr;
 142                         // Do not add a unit twice.
 143                         size_t underlying_id = recall_unit.underlying_id();
 144                         if ( !already_added  ||  already_added->count(underlying_id) == 0 )
 145                         {
 146                                 // Only units that match the leader's recall filter are valid.
 147                                 scoped_recall_unit this_unit("this_unit", save_id, leader_team.recall_list().find_index(recall_unit.id()));
 148 
 149                                 if ( ufilt(recall_unit, map_location::null_location()) )
 150                                 {
 151                                         result.push_back(recall_unit_ptr);
 152                                         if ( already_added != nullptr )
 153                                                 already_added->insert(underlying_id);
 154                                 }
 155                         }
 156                 }
 157         }

 160 std::vector<unit_const_ptr > get_recalls(int side, const map_location &recall_loc)
 161 {
 162         LOG_NG << "getting recall list for side " << side << " at location " << recall_loc << "\n";
 163 
 164         std::vector<unit_const_ptr > result;
 165 
 166         /*
 167          * We have three use cases:
 168          * 1. An empty castle tile is highlighted; we return only the units recallable there.
 169          * 2. A leader on a keep is highlighted; we return only the units recallable by that leader.
 170          * 3. Otherwise, we return all units in the recall list that can be recalled by any leader on the map.
 171          */
 172 
 173         bool leader_in_place = false;
 174         bool allow_local = resources::gameboard->map().is_castle(recall_loc);
 175 
 176 
 177         // Check for a leader at recall_loc (means we are recalling from there,
 178         // rather than to there).
 179         const unit_map::const_iterator find_it = resources::gameboard->units().find(recall_loc);
 180         if ( find_it != resources::gameboard->units().end() ) {
 181                 if ( find_it->can_recruit()  &&  find_it->side() == side  &&
 182                      resources::gameboard->map().is_keep(recall_loc) )
 183                 {
 184                         // We have been requested to get the recalls for this
 185                         // particular leader.
 186                         add_leader_filtered_recalls(find_it.get_shared_ptr(), result);
 187                         return result;
 188                 }
 189                 else if ( find_it->is_visible_to_team(resources::gameboard->get_team(side), *resources::gameboard, false) )
 190                 {
 191                         // This hex is visibly occupied, so we cannot recall here.
 192                         allow_local = false;
 193                 }
 194         }
 195 
 196         if ( allow_local )
 197         {
 198                 unit_map::const_iterator u = resources::gameboard->units().begin(),
 199                                 u_end = resources::gameboard->units().end();
 200                 std::set<size_t> valid_local_recalls;
 201 
 202                 for(; u != u_end; ++u) {
 203                         //We only consider leaders on our side.
 204                         if (!(u->can_recruit() && u->side() == side))
 205                                 continue;
 206 
 207                         // Check if the leader is on a connected keep.
 208                         if ( !dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*u, recall_loc) )
 209                                 continue;
 210                         leader_in_place= true;
 211 
 212                         add_leader_filtered_recalls(u.get_shared_ptr(), result, &valid_local_recalls);
 213                 }
 214         }
 215 
 216         if ( !leader_in_place )
 217         {
 218                 std::set<size_t> valid_local_recalls;
 219 
 220                 for(auto u = resources::gameboard->units().begin(); u != resources::gameboard->units().end(); ++u) {
 221                         //We only consider leaders on our side.
 222                         if(!u->can_recruit() || u->side() != side) {
 223                                 continue;
 224                         }
 225 
 226                         add_leader_filtered_recalls(u.get_shared_ptr(), result, &valid_local_recalls);
 227                 }
 228         }
 229 
 230         return result;
 231 }

 233 namespace { // Helpers for check_recall_location()
 234         /**
 235          * Checks if @a recaller can recall @a recall_unit at @a preferred.
 236          * If recalling can occur but not at the preferred location, then a
 237          * permissible location is stored in @a alternative.
 238          * @returns the reason why recalling is not allowed (or RECRUIT_OK).
 239          */
 240         RECRUIT_CHECK check_unit_recall_location(
 241                 const unit & recaller, const unit & recall_unit,
 242                 const map_location & preferred, map_location & alternative)
 243         {
 244                 // Make sure the unit can actually recall.
 245                 if ( !recaller.can_recruit() )
 246                         return RECRUIT_NO_LEADER;
 247 
 248                 // Make sure the recalling unit can recall this specific unit.
 249                 team& recall_team = (*resources::gameboard).get_team(recaller.side());
 250                 scoped_recall_unit this_unit("this_unit", recall_team.save_id(),
 251                                                 recall_team.recall_list().find_index(recall_unit.id()));
 252 
 253                 const unit_filter ufilt(vconfig(recaller.recall_filter()), resources::filter_con);
 254                 if ( !ufilt(recall_unit, map_location::null_location()) )
 255                         return RECRUIT_NO_ABLE_LEADER;
 256 
 257                 // Make sure the unit is on a keep.
 258                 if ( !resources::gameboard->map().is_keep(recaller.get_location()) )
 259                         return RECRUIT_NO_KEEP_LEADER;
 260 
 261                 // Make sure there is a permissible location to which to recruit.
 262                 map_location permissible = pathfind::find_vacant_castle(recaller);
 263                 if ( !permissible.valid() )
 264                         return RECRUIT_NO_VACANCY;
 265 
 266                 // See if the preferred location cannot be used.
 267                 if ( !dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(recaller, preferred) ) {
 268                         alternative = permissible;
 269                         return RECRUIT_ALTERNATE_LOCATION;
 270                 }
 271 
 272                 // All tests passed.
 273                 return RECRUIT_OK;
 274         }

 277 /// Checks if there is a location on which to recall @a unit_recall.
 278 RECRUIT_CHECK check_recall_location(const int side, map_location& recall_location,
 279                                     map_location& recall_from,
 280                                     const unit &unit_recall)
 281 {
 282         const unit_map & units = resources::gameboard->units();
 283         const unit_map::const_iterator u_end = units.end();
 284 
 285         map_location check_location = recall_location;
 286         map_location alternative;       // Set by check_unit_recall_location().
 287 
 288         // If the specified location is occupied, proceed as if no location was specified.
 289         if ( resources::gameboard->units().count(recall_location) != 0 )
 290                 check_location = map_location::null_location();
 291 
 292         // If the check location is not valid, we will never get an "OK" result.
 293         RECRUIT_CHECK const goal_result = check_location.valid() ? RECRUIT_OK :
 294                                                                    RECRUIT_ALTERNATE_LOCATION;
 295         RECRUIT_CHECK best_result = RECRUIT_NO_LEADER;
 296 
 297         // Test the specified recaller (if there is one).
 298         unit_map::const_iterator u = units.find(recall_from);
 299         if ( u != u_end  &&  u->side() == side ) {
 300                 best_result =
 301                         check_unit_recall_location(*u, unit_recall, check_location, alternative);
 302         }
 303 
 304         // Loop through all units on the specified side.
 305         for ( u = units.begin(); best_result < goal_result  &&  u != u_end; ++u ) {
 306                 if ( u->side() != side )
 307                         continue;
 308 
 309                 // Check this unit's viability as a recaller.
 310                 RECRUIT_CHECK current_result =
 311                         check_unit_recall_location(*u, unit_recall, check_location, alternative);
 312 
 313                 // If this is not an improvement, proceed to the next unit.
 314                 if ( current_result <= best_result )
 315                         continue;
 316                 best_result = current_result;
 317 
 318                 // If we have a viable recaller, record its location.
 319                 if ( current_result >= RECRUIT_ALTERNATE_LOCATION )
 320                         recall_from = u->get_location();
 321         }
 322 
 323         if ( best_result == RECRUIT_ALTERNATE_LOCATION )
 324                 // Report the alternate location to the caller.
 325                 recall_location = alternative;
 326 
 327         return best_result;
 328 }

 330 std::string find_recall_location(const int side, map_location& recall_location, map_location& recall_from, const unit &unit_recall)
 331 {
 332         LOG_NG << "finding recall location for side " << side << " and unit " << unit_recall.id() << "\n";
 333 
 334         // This function basically translates check_recall_location() to a
 335         // human-readable string.
 336         switch ( check_recall_location(side, recall_location, recall_from, unit_recall) )
 337         {
 338         case RECRUIT_NO_LEADER:
 339                 LOG_NG << "No leaders on side " << side << " when recalling " << unit_recall.id() << ".\n";
 340                 return _("You do not have a leader to recall with.");
 341 
 342         case RECRUIT_NO_ABLE_LEADER:
 343                 LOG_NG << "No leader is able to recall " << unit_recall.id() << " on side " << side << ".\n";
 344                 return _("None of your leaders are able to recall that unit.");
 345 
 346         case RECRUIT_NO_KEEP_LEADER:
 347                 LOG_NG << "No leader able to recall " << unit_recall.id() << " is on a keep.\n";
 348                 return _("You must have a leader on a keep who is able to recall that unit.");
 349 
 350         case RECRUIT_NO_VACANCY:
 351                 LOG_NG << "No vacant castle tiles around a keep are available for recalling " << unit_recall.id() << "; requested location is " << recall_location << ".\n";
 352                 return _("There are no vacant castle tiles in which to recall the unit.");
 353 
 354         case RECRUIT_ALTERNATE_LOCATION:
 355         case RECRUIT_OK:
 356                 return std::string();
 357         }
 358 
 359         // We should never get down to here. But just in case someone decides to
 360         // mess with the enum without updating this function:
 361         ERR_NG << "Unrecognized enum in find_recall_location()" << std::endl;
 362         return _("An unrecognized error has occurred.");
 363 }

 534         /**
 535          * Locates a leader on side @a side who can recruit at @a recruit_location.
 536          * A leader at @a recruited_from is chosen in preference to others.
 537          */
 538         const map_location & find_recruit_leader(int side,
 539                 const map_location &recruit_location, const map_location &recruited_from)
 540         {
 541                 const unit_map & units = resources::gameboard->units();
 542 
 543                 // See if the preferred location is an option.
 544                 unit_map::const_iterator leader = units.find(recruited_from);
 545                 if ( leader != units.end()  &&  leader->can_recruit()  &&
 546                          leader->side() == side && dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*leader, recruit_location) )
 547                         return leader->get_location();
 548 
 549                 // Check all units.
 550                 for ( leader = units.begin(); leader != units.end(); ++leader )
 551                         if ( leader->can_recruit()  &&  leader->side() == side  &&
 552                                  dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*leader, recruit_location) )
 553                                 return leader->get_location();
 554 
 555                 // No usable leader found.
 556                 return map_location::null_location();
 557         }

 576         }

 578         void set_recruit_facing(unit_map::iterator &new_unit_itor, const unit &new_unit,
 579                 const map_location &recruit_loc, const map_location &leader_loc)
 580         {
 581                 // Find closest enemy and turn towards it (level 2s count more than level 1s, etc.)
 582                 const gamemap *map = & resources::gameboard->map();
 583                 const unit_map & units = resources::gameboard->units();
 584                 unit_map::const_iterator unit_itor;
 585                 map_location min_loc;
 586                 int min_dist = INT_MAX;
 587 
 588                 for ( unit_itor = units.begin(); unit_itor != units.end(); ++unit_itor ) {
 589                         if (resources::gameboard->get_team(unit_itor->side()).is_enemy(new_unit.side()) &&
 590                                 unit_itor->is_visible_to_team(resources::gameboard->get_team(new_unit.side()), *resources::gameboard, false)) {
 591                                 int dist = distance_between(unit_itor->get_location(),recruit_loc) - unit_itor->level();
 592                                 if (dist < min_dist) {
 593                                         min_dist = dist;
 594                                         min_loc = unit_itor->get_location();
 595                                 }
 596                         }
 597                 }
 598                 if (min_dist < INT_MAX) {
 599                         // Face towards closest enemy
 600                         new_unit_itor->set_facing(recruit_loc.get_relative_dir(min_loc));
 601                 } else if (leader_loc != map_location::null_location()) {
 602                         // Face away from leader
 603                         new_unit_itor->set_facing(map_location::get_opposite_dir(recruit_loc.get_relative_dir(leader_loc)));
 604                 } else {
 605                         // Face towards center of map
 606                         const map_location center(map->w()/2, map->h()/2);
 607                         new_unit_itor->set_facing(recruit_loc.get_relative_dir(center));
 608                 }
 609         }
 610 }// anonymous namespace
 611 //Used by recalls and recruits
 612 place_recruit_result place_recruit(unit_ptr u, const map_location &recruit_location, const map_location& recruited_from,
 613     int cost, bool is_recall, bool show, bool fire_event, bool full_movement,
 614     bool wml_triggered)
 615 {
 616         place_recruit_result res(false, 0, false);
 617         LOG_NG << "placing new unit on location " << recruit_location << "\n";
 618         if (full_movement) {
 619                 u->set_movement(u->total_movement(), true);
 620         } else {
 621                 u->set_movement(0, true);
 622                 u->set_attacks(0);
 623         }
 624         u->heal_fully();
 625         u->set_hidden(true);
 626 
 627         // Get the leader location before adding the unit to the board.
 628         const map_location leader_loc = !show ? map_location::null_location() :
 629                         find_recruit_leader(u->side(), recruit_location, recruited_from);
 630         u->set_location(recruit_location);
 631         // Add the unit to the board.
 632         std::pair<unit_map::iterator, bool> add_result = resources::gameboard->units().insert(u);
 633         assert(add_result.second);
 634         unit_map::iterator & new_unit_itor = add_result.first;
 635         map_location current_loc = recruit_location;
 636 
 637         set_recruit_facing(new_unit_itor, *u, recruit_location, leader_loc);
 638 
 639         // Do some bookkeeping.
 640         recruit_checksums(*u, wml_triggered);
 641         resources::whiteboard->on_gamestate_change();
 642 
 643         resources::game_events->pump().fire("unit_placed", current_loc);
 644 
 645         if ( fire_event ) {
 646                 const std::string event_name = is_recall ? "prerecall" : "prerecruit";
 647                 LOG_NG << "firing " << event_name << " event\n";
 648                 {
 649                         std::get<0>(res) |= resources::game_events->pump().fire(event_name, current_loc, recruited_from);
 650                 }
 651                 if ( !validate_recruit_iterator(new_unit_itor, current_loc) )
 652                         return std::make_tuple(true, 0, false);
 653                 new_unit_itor->set_hidden(true);
 654         }
 655         preferences::encountered_units().insert(new_unit_itor->type_id());
 656         resources::gameboard->get_team(u->side()).spend_gold(cost);
 657 
 658         if ( show ) {
 659                 unit_display::unit_recruited(current_loc, leader_loc);
 660         }
 661         // Make sure the unit appears (if either !show or the animation is suppressed).
 662         new_unit_itor->set_hidden(false);
 663         if ( resources::screen != nullptr ) {
 664                 resources::screen->invalidate(current_loc);
 665                 resources::screen->redraw_minimap();
 666         }
 667 
 668         // Village capturing.
 669         if ( resources::gameboard->map().is_village(current_loc) ) {
 670                 std::get<1>(res) = resources::gameboard->village_owner(current_loc) + 1;
 671                 std::get<0>(res) |= actions::get_village(current_loc, new_unit_itor->side(), &std::get<2>(res));
 672                 if ( !validate_recruit_iterator(new_unit_itor, current_loc) )
 673                         return std::make_tuple(true, 0, false);
 674         }
 675 
 676         // Fog clearing.
 677         actions::shroud_clearer clearer;
 678         if ( !wml_triggered ) // To preserve current WML behavior.
 679                 std::get<0>(res) |= clearer.clear_unit(current_loc, *new_unit_itor);
 680 
 681         if ( fire_event ) {
 682                 const std::string event_name = is_recall ? "recall" : "recruit";
 683                 LOG_NG << "firing " << event_name << " event\n";
 684                 {
 685                         std::get<0>(res) |= resources::game_events->pump().fire(event_name, current_loc, recruited_from);
 686                 }
 687         }
 688 
 689         // "sighted" event(s).
 690         std::get<0>(res) |= clearer.fire_events();
 691         if ( new_unit_itor.valid() )
 692                 std::get<0>(res) |= actions::actor_sighted(*new_unit_itor);
 693 
 694         return res;
 695 }

 698 /**
 699  * Recruits a unit of the given type for the given side.
 700  */
 701 void recruit_unit(const unit_type & u_type, int side_num, const map_location & loc,
 702                   const map_location & from, bool show, bool use_undo)
 703 {
 704         const unit_ptr new_unit = unit_ptr( new unit(u_type, side_num, true));
 705 
 706 
 707         // Place the recruit.
 708         place_recruit_result res = place_recruit(new_unit, loc, from, u_type.cost(), false, show);
 709         statistics::recruit_unit(*new_unit);
 710 
 711         // To speed things a bit, don't bother with the undo stack during
 712         // an AI turn. The AI will not undo nor delay shroud updates.
 713         // (Undo stack processing is also suppressed when redoing a recruit.)
 714         if ( use_undo ) {
 715                 resources::undo_stack->add_recruit(new_unit, loc, from, std::get<1>(res), std::get<2>(res));
 716                 // Check for information uncovered or randomness used.
 717 
 718                 if ( std::get<0>(res) || !synced_context::can_undo()) {
 719                         resources::undo_stack->clear();
 720                 }
 721         }
 722 
 723         // Update the screen.
 724         if ( resources::screen != nullptr )
 725                 resources::screen->invalidate_game_status();
 726                 // Other updates were done by place_recruit().
 727 }

 730 /**
 731  * Recalls the unit with the indicated ID for the provided team.
 732  */
 733 bool recall_unit(const std::string & id, team & current_team,
 734                  const map_location & loc, const map_location & from,
 735                  bool show, bool use_undo)
 736 {
 737         unit_ptr recall = current_team.recall_list().extract_if_matches_id(id);
 738 
 739         if ( !recall )
 740                 return false;
 741 
 742 
 743         // ** IMPORTANT: id might become invalid at this point!
 744         // (Use recall.id() instead, if needed.)
 745 
 746         // Place the recall.
 747         // We also check to see if a custom unit level recall has been set if not,
 748         // we use the team's recall cost otherwise the unit's.
 749         place_recruit_result res;
 750         if (recall->recall_cost() < 0) {
 751                 res = place_recruit(recall, loc, from, current_team.recall_cost(),
 752                                      true, show);
 753         }
 754         else {
 755                 res = place_recruit(recall, loc, from, recall->recall_cost(),
 756                                      true, show);
 757         }
 758         statistics::recall_unit(*recall);
 759 
 760         // To speed things a bit, don't bother with the undo stack during
 761         // an AI turn. The AI will not undo nor delay shroud updates.
 762         // (Undo stack processing is also suppressed when redoing a recall.)
 763         if ( use_undo ) {
 764                 resources::undo_stack->add_recall(recall, loc, from, std::get<1>(res), std::get<2>(res));
 765                 if ( std::get<0>(res) || !synced_context::can_undo()) {
 766                         resources::undo_stack->clear();
 767                 }
 768         }
 769 
 770         // Update the screen.
 771         if ( resources::screen != nullptr )
 772                 resources::screen->invalidate_game_status();
 773                 // Other updates were done by place_recruit().
 774 
 775         return true;
 776 }
