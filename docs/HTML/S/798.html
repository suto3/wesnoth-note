  85 base_manager::base_manager()
  86 {
  87         event_handler_.join_global();
  88 
  89         try{
  90 #ifdef DEFAULT_PREFS_PATH
  91                 filesystem::scoped_istream stream = filesystem::istream_file(filesystem::get_default_prefs_file(),false);
  92                 read(prefs, *stream);
  93 
  94                 config user_prefs;
  95                 stream = filesystem::istream_file(filesystem::get_prefs_file());
  96                 read(user_prefs, *stream);
  97 
  98                 prefs.merge_with(user_prefs);
  99 #else
 100                 filesystem::scoped_istream stream = filesystem::istream_file(filesystem::get_prefs_file(),false);
 101                 read(prefs, *stream);
 102 #endif
 103         } catch(const config::error& e) {
 104                 ERR_CFG << "Error loading preference, message: "
 105                                 << e.what()
 106                                 << std::endl;
 107         }
 108         preferences::load_credentials();
 109 }

 125 /*
 126  * Hook for setting window state variables on window resize and maximize
 127  * events. Since there is no fullscreen window event, that setter is called
 128  * from the CVideo function instead.
 129  */
 130 void prefs_event_handler::handle_window_event(const SDL_Event& event)
 131 {
 132 
 133         // Saftey check to make sure this is a window event
 134         if (event.type != SDL_WINDOWEVENT) return;
 135 
 136         switch(event.window.event) {
 137         case SDL_WINDOWEVENT_RESIZED:
 138                 _set_resolution(std::make_pair(event.window.data1,event.window.data2));
 139 
 140                 break;
 141 
 142         case SDL_WINDOWEVENT_MAXIMIZED:
 143                 _set_maximized(true);
 144 
 145                 break;
 146 
 147         case SDL_WINDOWEVENT_RESTORED:
 148                 _set_maximized(fullscreen() || false);
 149 
 150                 break;
 151         }
 152 }

 154 void write_preferences()
 155 {
 156 #ifndef _WIN32
 157     bool prefs_file_existed = access(filesystem::get_prefs_file().c_str(), F_OK) == 0;
 158 #endif
 159 
 160         try {
 161                 filesystem::scoped_ostream prefs_file = filesystem::ostream_file(filesystem::get_prefs_file());
 162                 write(*prefs_file, prefs);
 163         } catch(filesystem::io_exception&) {
 164                 ERR_FS << "error writing to preferences file '" << filesystem::get_prefs_file() << "'" << std::endl;
 165         }
 166 
 167         preferences::save_credentials();
 168 
 169 #ifndef _WIN32
 170     if(!prefs_file_existed) {
 171 
 172         if(chmod(filesystem::get_prefs_file().c_str(), 0600) == -1) {
 173                         ERR_FS << "error setting permissions of preferences file '" << filesystem::get_prefs_file() << "'" << std::endl;
 174         }
 175 
 176     }
 177 #endif
 178 }

 293         if (color.substr(0,4) == "orb_") {

 295                         return color.substr(5);
 296                 } else {
 297                         return color.substr(4);

 299         }

 373 std::pair<int,int> resolution()
 374 {
 375         const std::string& x = prefs["xresolution"], y = prefs["yresolution"];
 376 
 377         if (!x.empty() && !y.empty()) {
 378                 try {
 379                         return std::make_pair(
 380                                 std::max(std::stoi(x), min_window_width),
 381                                 std::max(std::stoi(y), min_window_height));
 382                 } catch(std::invalid_argument) {}
 383         }
 384 
 385         return std::pair<int,int>(def_window_width, def_window_height);
 386 }

 515 {
 516         // Sounds don't sound good on Windows unless the buffer size is 4k,
 517         // but this seems to cause crashes on other systems...
 518         #ifdef _WIN32

 523

 527 void save_sound_buffer_size(const size_t size)
 528 {
 529         #ifdef _WIN32
 530                 const char* buf_size = "4096";
 531         #else
 532                 const char* buf_size = "1024";
 533         #endif
 534 
 535         const std::string new_size = lexical_cast_default<std::string>(size, buf_size);
 536         if (get("sound_buffer_size") == new_size)
 537                 return;
 538 
 539         preferences::set("sound_buffer_size", new_size);
 540 
 541         sound::reset_sound();
 542 }

 609 bool set_turn_bell(bool ison)
 610 {
 611         if(!turn_bell() && ison) {
 612                 preferences::set("turn_bell", true);
 613                 if(!music_on() && !sound_on() && !UI_sound_on()) {
 614                         if(!sound::init_sound()) {
 615                                 preferences::set("turn_bell", false);
 616                                 return false;
 617                         }
 618                 }
 619         } else if(turn_bell() && !ison) {
 620                 preferences::set("turn_bell", false);
 621                 sound::stop_bell();
 622                 if(!music_on() && !sound_on() && !UI_sound_on())
 623                         sound::close_sound();
 624         }
 625         return true;
 626 }
