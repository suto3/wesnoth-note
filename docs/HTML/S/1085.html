  74 void t_string_base::walker::update()
  75 {
  76         unsigned int id;
  77 
  78         if(begin_ == string_.size())
  79                 return;
  80 
  81         switch(string_[begin_]) {
  82         case TRANSLATABLE_PART: {
  83                 // Format: [TRANSLATABLE_PART]textdomain[TEXTDOMAIN_SEPARATOR]msgid[...]
  84                 std::string::size_type textdomain_end =
  85                         string_.find(TEXTDOMAIN_SEPARATOR, begin_ + 1);
  86 
  87                 if(textdomain_end == std::string::npos || textdomain_end >= string_.size() - 1) {
  88                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
  89                         begin_ = string_.size();
  90                         return;
  91                 }
  92 
  93                 end_ = string_.find_first_of(mark, textdomain_end + 1);
  94                 if(end_ == std::string::npos)
  95                         end_ = string_.size();
  96 
  97                 textdomain_ = std::string(string_, begin_+1, textdomain_end - begin_ - 1);
  98                 translatable_ = true;
  99                 begin_ = textdomain_end + 1;
 100 
 101                 break;
 102         }
 103         case ID_TRANSLATABLE_PART:
 104                 // Format: [ID_TRANSLATABLE_PART][2-byte textdomain ID]msgid[...]
 105                 if(begin_ + 3 >= string_.size()) {
 106                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 107                         begin_ = string_.size();
 108                         return;
 109                 }
 110                 end_ = string_.find_first_of(mark, begin_ + 3);
 111                 if(end_ == std::string::npos)
 112                         end_ = string_.size();
 113 
 114                 id = string_[begin_ + 1] + string_[begin_ + 2] * 256;
 115                 if(id >= id_to_textdomain.size()) {
 116                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 117                         begin_ = string_.size();
 118                         return;
 119                 }
 120                 textdomain_ = id_to_textdomain[id];
 121                 begin_ += 3;
 122                 translatable_ = true;
 123 
 124                 break;
 125 
 126         case UNTRANSLATABLE_PART:
 127                 end_ = string_.find_first_of(mark, begin_ + 1);
 128                 if(end_ == std::string::npos)
 129                         end_ = string_.size();
 130 
 131                 if(end_ <= begin_ + 1) {
 132                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 133                         begin_ = string_.size();
 134                         return;
 135                 }
 136 
 137                 translatable_ = false;
 138                 textdomain_ = "";
 139                 begin_ += 1;
 140                 break;
 141 
 142         case PLURAL_PART:
 143                 begin_ = string_.find_first_of(mark, end_ + 5);
 144                 if(begin_ == std::string::npos)
 145                         begin_ = string_.size();
 146                 if(string_[begin_] == PLURAL_PART) {
 147                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 148                         begin_ = string_.size();
 149                         return;
 150                 }
 151                 update();
 152                 break;
 153 
 154         default:
 155                 end_ = string_.size();
 156                 translatable_ = false;
 157                 textdomain_ = "";
 158                 break;
 159         }
 160 
 161         if(translatable_ && string_[end_] == PLURAL_PART) {
 162                 // Format: [PLURAL_PART][4-byte count]msgid_plural[...]
 163                 if(end_ + 5 >= string_.size()) {
 164                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 165                         begin_ = string_.size();
 166                         return;
 167                 }
 168                 std::string::size_type real_end = string_.find_first_of(mark, end_ + 6);
 169                 if(real_end < string_.size() && string_[real_end] == PLURAL_PART) {
 170                         ERR_CF << "Error: invalid string: " << string_ << std::endl;
 171                         begin_ = string_.size();
 172                         return;
 173                 }
 174                 countable_ = true;
 175                 union {
 176                         int32_t count;
 177                         char data[4];
 178                 } cvt;
 179                 std::copy_n(string_.data() + end_ + 1, 4, cvt.data);
 180                 count_ = cvt.count;
 181         } else {
 182                 countable_ = false;
 183                 count_ = 0;
 184         }
 185 }

 240 t_string_base::t_string_base(const std::string& string, const std::string& textdomain) :
 241         value_(1, ID_TRANSLATABLE_PART),
 242         translated_value_(),
 243         translation_timestamp_(0),
 244         translatable_(true),
 245         last_untranslatable_(false)
 246 {
 247         if (string.empty()) {
 248                 value_.clear();
 249                 translatable_ = false;
 250                 return;
 251         }
 252 
 253         std::map<std::string, unsigned int>::const_iterator idi = textdomain_to_id.find(textdomain);
 254         unsigned int id;
 255 
 256         if(idi == textdomain_to_id.end()) {
 257                 id = id_to_textdomain.size();
 258                 textdomain_to_id[textdomain] = id;
 259                 id_to_textdomain.push_back(textdomain);
 260         } else {
 261                 id = idi->second;
 262         }
 263 
 264         value_ += char(id & 0xff);
 265         value_ += char(id >> 8);
 266         value_ += string;
 267 }

 283         unsigned int id;

 286                 id = id_to_textdomain.size();
 287                 textdomain_to_id[textdomain] = id;
 288                 id_to_textdomain.push_back(textdomain);
 289         } else {
 290                 id = idi->second;
 291         }
 292 
 293         value_ += char(id & 0xff);
 294         value_ += char(id >> 8);
 295         value_ += sing;
 296         value_ += PLURAL_PART;
 297

 305         }

 318 t_string_base t_string_base::from_serialized(const std::string& string)
 319 {
 320         t_string_base orig(string);
 321 
 322         if(!string.empty() && (string[0] == TRANSLATABLE_PART || string[0] == UNTRANSLATABLE_PART)) {
 323                 orig.translatable_ = true;
 324         } else {
 325                 orig.translatable_ = false;
 326         }
 327 
 328         t_string_base res;
 329 
 330         for(walker w(orig); !w.eos(); w.next()) {
 331                 std::string substr(w.begin(), w.end());
 332 
 333                 if(w.translatable()) {
 334                         res += t_string_base(substr, w.textdomain());
 335                 } else {
 336                         res += substr;
 337                 }
 338         }
 339 
 340         return res;
 341 }

 352 std::string t_string_base::to_serialized() const
 353 {
 354         t_string_base res;
 355 
 356         for(walker w(*this); !w.eos(); w.next()) {
 357                 t_string_base chunk;
 358 
 359                 std::string substr(w.begin(), w.end());
 360                 if(w.translatable()) {
 361                         chunk.translatable_ = true;
 362                         chunk.last_untranslatable_ = false;
 363                         chunk.value_ = TRANSLATABLE_PART + w.textdomain() +
 364                                 TEXTDOMAIN_SEPARATOR + substr;
 365                 } else {
 366                         chunk.translatable_ = false;
 367                         chunk.value_ = substr;
 368                 }
 369 
 370                 res += chunk;
 371         }
 372 
 373         return res.value();
 374 }

 430 t_string_base& t_string_base::operator+=(const t_string_base& string)
 431 {
 432         if (string.value_.empty())
 433                 return *this;
 434         if (value_.empty()) {
 435                 *this = string;
 436                 return *this;
 437         }
 438 
 439         if(translatable_ || string.translatable_) {
 440                 if(!translatable_) {
 441                         value_ = UNTRANSLATABLE_PART + value_;
 442                         translatable_ = true;
 443                         last_untranslatable_ = true;
 444                 } else
 445                         translated_value_ = "";
 446                 if(string.translatable_) {
 447                         if (last_untranslatable_ && string.value_[0] == UNTRANSLATABLE_PART)
 448                                 value_.append(string.value_.begin() + 1, string.value_.end());
 449                         else
 450                                 value_ += string.value_;
 451                         last_untranslatable_ = string.last_untranslatable_;
 452                 } else {
 453                         if (!last_untranslatable_) {
 454                                 value_ += UNTRANSLATABLE_PART;
 455                                 last_untranslatable_ = true;
 456                         }
 457                         value_ += string.value_;
 458                 }
 459         } else {
 460                 value_ += string.value_;
 461         }
 462 
 463         return *this;
 464 }

 532 const std::string& t_string_base::str() const
 533 {
 534         if(!translatable_)
 535                 return value_;
 536 
 537         if (translatable_ && !translated_value_.empty() && translation_timestamp_ == language_counter)
 538                 return translated_value_;
 539 
 540         translated_value_.clear();
 541 
 542         for(walker w(*this); !w.eos(); w.next()) {
 543                 std::string part(w.begin(), w.end());
 544 
 545                 if(w.translatable()) {
 546                         if(w.countable()) {
 547                                 std::string plural(w.plural_begin(), w.plural_end());
 548                                 translated_value_ += translation::dsngettext(w.textdomain().c_str(), part.c_str(), plural.c_str(), w.count());
 549                         } else {
 550                                 translated_value_ += translation::dsgettext(w.textdomain().c_str(), part.c_str());
 551                         }
 552                 } else {
 553                         translated_value_ += part;
 554                 }
 555         }
 556 
 557         translation_timestamp_ = language_counter;
 558         return translated_value_;
 559 }
