  14 /**
  15         Some information about savefiles:
  16         A saveile can contain:
  17          * General information (toplevel atributes, [multiplayer])
  18             This is present in all savefiles
  19          * [statistics]
  20             This is present in all savefiles but it's not handled by playcampaign/play_controller/saved_game.
  21                 It's handled by savegame.cpp
  22          * [snapshot]
  23             If a savegame was saved during a scenario this contains a snapshot of the game at the point when it was saved.
  24          * [carryover_sides_start]
  25             At start-of-scenrio saves this contains data from the previous scenario that was preserved
  26          * [carryover_sides]
  27             In savefile made during the game, this tag contains data from [carryover_sides_start] that was not used in the current scenario but should be saved for a next scenario
  28          * [replay_start]
  29             A snapshot made very early to replay the game from.
  30          * [replay]
  31             A record of game actions that was made between the creation of [replay_start] and [snapshot].
  32 
  33 
  34 
  35         The following types of savegames are known:
  36          * Start of scenario savefiles
  37             These files only contain general information, statistics, and [carryover_sides_start]
  38                 When these saves are loaded, the scenario data is loaded form the game config using the next_scenario attribute from [carryover_sides_start]
  39          * Expanded Start of scenario savefiles
  40             Similar to normal Start-of-scenario savefiles, but the also contain a [scenario] that contains the scenario data.
  41                 This type is only used internally and usually doesn't get written to the disk.
  42          * Ingame savefile
  43             These files contain general information, statistics, [snapshot], [replay], [replay_start], [snapshot], [carryover_sides]
  44                 These files don't contain a [carryover_sides_start] because both starting points ([replay_start] and [snapshot])
  45                 were made after [carryover_sides_start] was merged into the scenario.
  46          * Replay savefiles
  47             Like a Ingame save made during linger mode, but without the [snapshot]
  48 */

 167 void saved_game::set_defaults()
 168 {
 169         const bool is_loaded_game = this->starting_pos_type_ != STARTINGPOS_SCENARIO;
 170         const bool is_multiplayer_tag = classification().get_tagname() == "multiplayer";
 171         static const std::vector<std::string> team_defaults {
 172                 "carryover_percentage",
 173                 "carryover_add",
 174         };
 175         for(config& side : starting_pos_.child_range("side"))
 176         {
 177                 // Set save_id default value directly after loading to its default to prevent different default behaviour in mp_connect code and sp code.
 178                 if(side["save_id"].empty())
 179                 {
 180                         side["save_id"] = side["id"];
 181                 }
 182                 if(!is_multiplayer_tag && side["side_name"].blank())
 183                 {
 184                         side["side_name"] = side["name"];
 185                 }
 186                 if(!is_loaded_game && !side["current_player"].empty())
 187                 {
 188                         ERR_NG << "Removed invalid 'current_player' attribute from [side] while loading a scenario. Consider using 'side_name' instead\n";
 189                         side["current_player"] = config::attribute_value();
 190                 }
 191                 // Set some team specific values to their defaults specified in scenario
 192                 for(const std::string& att_name : team_defaults)
 193                 {
 194                         const config::attribute_value* scenario_value = starting_pos_.get(att_name);
 195                         config::attribute_value& team_value = side[att_name];
 196                         if(scenario_value && team_value.empty())
 197                         {
 198                                 team_value = *scenario_value;
 199                         }
 200                 }
 201         }
 202 }

 204 void saved_game::expand_scenario()
 205 {
 206         if(this->starting_pos_type_ == STARTINGPOS_NONE && !has_carryover_expanded_)
 207         {
 208                 game_config_manager::get()->load_game_config_for_game(this->classification());
 209                 const config& game_config = game_config_manager::get()->game_config();
 210                 const config& scenario = game_config.find_child(classification().get_tagname(), "id", carryover_["next_scenario"]);
 211                 if(scenario)
 212                 {
 213                         this->starting_pos_type_ = STARTINGPOS_SCENARIO;
 214                         this->starting_pos_ = scenario;
 215                         // A hash has to be generated using an unmodified scenario data.
 216                         mp_settings_.hash = scenario.hash();
 217 
 218                         // Add addon_id information if it exists.
 219                         if (!scenario["addon_id"].empty() && scenario["require_scenario"].to_bool(false)) {
 220                                 mp_settings_.update_addon_requirements(config_of("id",scenario["addon_id"])("version", scenario["addon_version"])("min_version", scenario["addon_min_version"]));
 221                         }
 222 
 223                         update_label();
 224                         set_defaults();
 225                 }
 226                 else
 227                 {
 228                         this->starting_pos_type_ = STARTINGPOS_INVALID;
 229                         this->starting_pos_ = config();
 230                 }
 231         }
 232 }

 263 void saved_game::load_mod(const std::string& type, const std::string& id)
 264 {
 265         if(const config& cfg = game_config_manager::get()->
 266                 game_config().find_child(type, "id", id))
 267         {
 268                 // Note the addon_id if this mod is required to play the game in mp
 269                 std::string require_attr = "require_" + type;
 270                 bool require_default = (type == "era"); // By default, eras have "require_era = true", and mods have "require_modification = false"
 271                 if (!cfg["addon_id"].empty() && cfg[require_attr].to_bool(require_default)) {
 272                         mp_settings_.update_addon_requirements(config_of("id",cfg["addon_id"])("version", cfg["addon_version"])("min_version", cfg["addon_min_version"]));
 273                 }
 274 
 275                 // Copy events
 276                 for(const config& modevent : cfg.child_range("event"))
 277                 {
 278                         if(modevent["enable_if"].empty() || variable_to_bool(carryover_.child_or_empty("variables"), modevent["enable_if"]))
 279                         {
 280                                 this->starting_pos_.add_child("event", modevent);
 281                         }
 282                 }
 283                 // Copy lua
 284                 for(const config& modlua : cfg.child_range("lua"))
 285                 {
 286                         this->starting_pos_.add_child("lua", modlua);
 287                 }
 288                 // Copy load_resource
 289                 for(const config& load_resource : cfg.child_range("load_resource"))
 290                 {
 291                         this->starting_pos_.add_child("load_resource", load_resource);
 292                 }
 293         }
 294         else
 295         {
 296                 //TODO: A user message instead?
 297                 ERR_NG << "Couldn't find [" << type<< "] with id=" << id <<std::endl;
 298         }
 299 }

 301 // Gets the ids of the mp_era and modifications which were set to be active, then fetches these configs from the game_config and copies their [event] and [lua] to the starting_pos_.
 302 // At this time, also collect the addon_id attributes which appeared in them and put this list in the addon_ids attribute of the mp_settings.
 303 void saved_game::expand_mp_events()
 304 {
 305         expand_scenario();
 306         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !this->starting_pos_["has_mod_events"].to_bool(false))
 307         {
 308                 std::vector<modevents_entry> mods;
 309                 std::set<std::string> loaded_resources;
 310                 boost::copy( mp_settings_.active_mods
 311                         | boost::adaptors::transformed(modevents_entry_for("modification"))
 312                         , std::back_inserter(mods) );
 313                 if(mp_settings_.mp_era != "") //We don't want the error message below if there is no era (= if this is a sp game)
 314                 { mods.emplace_back("era", mp_settings_.mp_era); }
 315                 if(classification_.campaign != "")
 316                 { mods.emplace_back("campaign", classification_.campaign); }
 317 
 318                 // In the first iteration mod contains no [resource]s in all other iterations, mods contains only [resource]s
 319                 do {
 320                         for(modevents_entry& mod : mods)
 321                         {
 322                                 load_mod(mod.type, mod.id);
 323                         }
 324                         mods.clear();
 325                         for(const config& cfg : starting_pos_.child_range("load_resource"))
 326                         {
 327                                 if(loaded_resources.find(cfg["id"].str()) == loaded_resources.end()) {
 328                                         mods.emplace_back("resource", cfg["id"].str());
 329                                         loaded_resources.insert(cfg["id"].str());
 330                                 }
 331                         }
 332                         starting_pos_.clear_children("load_resource");
 333                 } while(!mods.empty());
 334                 this->starting_pos_["has_mod_events"] = true;
 335         }
 336 }

 338 void saved_game::expand_mp_options()
 339 {
 340         if(starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_)
 341         {
 342                 std::vector<modevents_entry> mods;
 343 
 344                 boost::copy( mp_settings_.active_mods
 345                         | boost::adaptors::transformed(modevents_entry_for("modification"))
 346                         , std::back_inserter(mods) );
 347                 mods.emplace_back("era", mp_settings_.mp_era);
 348                 mods.emplace_back("multiplayer", get_scenario_id());
 349                 mods.emplace_back("campaign", classification().campaign);
 350 
 351                 config& variables = carryover_.child_or_add("variables");
 352                 for(modevents_entry& mod : mods)
 353                 {
 354                         if(const config& cfg = this->mp_settings().options.find_child(mod.type, "id", mod.id))
 355                         {
 356                                 for(const config& option : cfg.child_range("option"))
 357                                 {
 358                                         try
 359                                         {
 360                                                 variable_access_create(option["id"], variables).as_scalar() = option["value"];
 361                                         }
 362                                         catch(const invalid_variablename_exception&)
 363                                         {
 364                                                 ERR_NG << "variable " << option["id"] << "cannot be set to " << option["value"] << std::endl;
 365                                         }
 366                                 }
 367                         }
 368                         else
 369                         {
 370                                 LOG_NG << "Couldn't find [" << mod.type<< "] with id=" << mod.id << " for [option]s" << std::endl;
 371                         }
 372                 }
 373         }
 374 }

 376 void saved_game::expand_random_scenario()
 377 {
 378         expand_scenario();
 379         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO)
 380         {
 381                 // If the entire scenario should be randomly generated
 382                 if(!starting_pos_["scenario_generation"].empty())
 383                 {
 384                         LOG_NG << "randomly generating scenario...\n";
 385                         const cursor::setter cursor_setter(cursor::WAIT);
 386 
 387                         config scenario_new = random_generate_scenario(starting_pos_["scenario_generation"],
 388                                 starting_pos_.child("generator"));
 389                         //Preserve "story" form the scenario toplevel.
 390                         for(config& story : starting_pos_.child_range("story"))
 391                         {
 392                                 scenario_new.add_child("story", story);
 393                         }
 394                         scenario_new["id"] = starting_pos_["id"];
 395                         starting_pos_ = scenario_new;
 396                         update_label();
 397                         set_defaults();
 398                 }
 399 
 400                 // If no map_data is provided, try to load the specified file directly
 401                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_file"].empty()) {
 402                         starting_pos_["map_data"] = filesystem::read_map(starting_pos_["map_file"]);
 403                 }
 404 
 405                 // If the map should be randomly generated
 406                 // We dont want that we accidentally to this twice so we check for starting_pos_["map_data"].empty()
 407                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_generation"].empty()) {
 408                         LOG_NG << "randomly generating map...\n";
 409                         const cursor::setter cursor_setter(cursor::WAIT);
 410 
 411                         starting_pos_["map_data"] = random_generate_map(
 412                                 starting_pos_["map_generation"], starting_pos_.child("generator"));
 413                 }
 414         }
 415 }

 417 void saved_game::expand_carryover()
 418 {
 419         expand_scenario();
 420         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_)
 421         {
 422                 carryover_info sides(carryover_);
 423 
 424                 sides.transfer_to(get_starting_pos());
 425                 for(config& side_cfg : get_starting_pos().child_range("side"))
 426                 {
 427                         sides.transfer_all_to(side_cfg);
 428                 }
 429 
 430                 carryover_ = sides.to_config();
 431                 has_carryover_expanded_ = true;
 432         }
 433 }

 467 const config& saved_game::get_replay_starting_pos()
 468 {
 469         if(!replay_start_.empty())
 470         {
 471                 return replay_start_;
 472         }
 473         if(!has_carryover_expanded_)
 474         {
 475                 //Try to load the scenario form game config or from [scenario] if there is no [replay_start]
 476                 expand_scenario();
 477                 expand_carryover();
 478         }
 479         if(starting_pos_type_ == STARTINGPOS_SCENARIO)
 480         {
 481                 return starting_pos_;
 482         }
 483         return this->replay_start_.child("some_non_existet_invalid");

 486 void saved_game::convert_to_start_save()
 487 {
 488         assert(starting_pos_type_ == STARTINGPOS_SNAPSHOT);
 489         carryover_info sides(starting_pos_, true);
 490         sides.merge_old_carryover(carryover_info(carryover_));
 491         sides.rng().rotate_random();
 492         carryover_ = sides.to_config();
 493         has_carryover_expanded_ = false;
 494         replay_data_ = replay_recorder_base();
 495         replay_start_ = config();
 496         remove_snapshot();
 497 }

 522 std::string saved_game::get_scenario_id()
 523 {
 524         std::string scenario_id;

 526                 || this->starting_pos_type_ == STARTINGPOS_SCENARIO)

 530         else if(!has_carryover_expanded_)

 556 void saved_game::cancel_orders()
 557 {
 558         for(config &side : this->starting_pos_.child_range("side"))
 559         {
 560                 // for humans "goto_x/y" is used for multi-turn-moves
 561                 // for the ai "goto_x/y" is a way for wml to order the ai to move a unit to a certain place.
 562                 // we want to cancel human order but not to break wml.
 563                 if (side["controller"] != "human" && side["controller"] != "network") continue;
 564                 for(config &unit : side.child_range("unit"))
 565                 {
 566                         unit["goto_x"] = -999;
 567                         unit["goto_y"] = -999;
 568                 }
 569         }
 570 }

 601 void saved_game::set_data(config& cfg)
 602 {
 603         log_scope("read_game");
 604 
 605         if(config & caryover_sides = cfg.child("carryover_sides"))
 606         {
 607                 carryover_.swap(caryover_sides);
 608                 has_carryover_expanded_ = true;
 609         }
 610         else if(config & caryover_sides_start = cfg.child("carryover_sides_start"))
 611         {
 612                 carryover_.swap(caryover_sides_start);
 613                 has_carryover_expanded_ = false;
 614         }
 615         else
 616         {

 648                 this->starting_pos_type_ = STARTINGPOS_NONE;
 649                 this->starting_pos_ = config();

 652         LOG_NG << "scenario: '" << carryover_["next_scenario"].str() << "'\n";
 653 
 654         if (const config &stats = cfg.child("statistics")) {
 655                 statistics::fresh_stats();

 658 
 659         classification_ = game_classification(cfg);
