  85 cave_map_generator::cave_map_generator_job::cave_map_generator_job(const cave_map_generator& pparams, boost::optional<uint32_t> randomseed)
  86         : params(pparams)
  87         , flipx_(false)
  88         , flipy_(false)
  89         , map_(t_translation::ter_map(params.width_ + 2 * gamemap::default_border, params.height_ + 2 * gamemap::default_border/*, params.wall_*/))
  90         , starting_positions_()
  91         , chamber_ids_()
  92         , chambers_()
  93         , passages_()
  94         , res_(params.cfg_.child_or_empty("settings"))
  95         , rng_() //initialises with rand()
  96 {
  97         res_.add_child("event", config_of
  98                 ("name", "start")
  99                 ("message", config_of
 100                         ("message", "scenario_generation=cave is deprecated and will be removed soon.")
 101                 )
 102         );
 103         uint32_t seed = randomseed.get_ptr() ? *randomseed.get_ptr() : seed_rng::next_seed();
 104         rng_.seed(seed);
 105         LOG_NG << "creating random cave with seed: " << seed << '\n';
 106         flipx_ = int(rng_() % 100) < params.flipx_chance_;
 107         flipy_ = int(rng_() % 100) < params.flipy_chance_;
 108 
 109         LOG_NG << "creating scenario....\n";
 110         generate_chambers();
 111 
 112         LOG_NG << "placing chambers...\n";
 113         for(std::vector<chamber>::const_iterator c = chambers_.begin(); c != chambers_.end(); ++c) {
 114                 place_chamber(*c);
 115         }
 116 
 117         LOG_NG << "placing passages...\n";
 118 
 119         for(std::vector<passage>::const_iterator p = passages_.begin(); p != passages_.end(); ++p) {
 120                 place_passage(*p);
 121         }
 122         LOG_NG << "outputting map....\n";
 123 
 124         res_["map_data"] = t_translation::write_game_map(map_, starting_positions_);
 125 }

 127 void cave_map_generator::cave_map_generator_job::build_chamber(map_location loc, std::set<map_location>& locs, size_t size, size_t jagged)
 128 {
 129         if(size == 0 || locs.count(loc) != 0 || !params.on_board(loc))
 130                 return;
 131 
 132         locs.insert(loc);
 133 
 134         map_location adj[6];
 135         get_adjacent_tiles(loc,adj);
 136         for(size_t n = 0; n != 6; ++n) {
 137                 if(int(rng_() % 100) < (100l - static_cast<long>(jagged))) {
 138                         build_chamber(adj[n],locs,size-1,jagged);
 139                 }
 140         }
 141 }

 143 void cave_map_generator::cave_map_generator_job::generate_chambers()
 144 {
 145         for (const config &ch : params.cfg_.child_range("chamber"))
 146         {
 147                 // If there is only a chance of the chamber appearing, deal with that here.
 148                 if (ch.has_attribute("chance") && int(rng_() % 100) < ch["chance"].to_int()) {
 149                         continue;
 150                 }
 151 
 152                 const std::string &xpos = ch["x"];
 153                 const std::string &ypos = ch["y"];
 154 
 155                 size_t min_xpos = 0, min_ypos = 0, max_xpos = params.width_, max_ypos = params.height_;
 156 
 157                 if (!xpos.empty()) {
 158                         const std::vector<std::string>& items = utils::split(xpos, '-');
 159                         if(items.empty() == false) {
 160                                 try {
 161                                         min_xpos = std::stoi(items.front()) - 1;
 162                                         max_xpos = std::stoi(items.back());
 163                                 } catch(std::invalid_argument) {
 164                                         lg::wml_error() << "Invalid min/max coordinates in cave_map_generator: " << items.front() << ", " << items.back() << "\n";
 165                                         continue;
 166                                 }
 167                         }
 168                 }
 169 
 170                 if (!ypos.empty()) {
 171                         const std::vector<std::string>& items = utils::split(ypos, '-');
 172                         if(items.empty() == false) {
 173                                 try {
 174                                         min_ypos = std::stoi(items.front()) - 1;
 175                                         max_ypos = std::stoi(items.back());
 176                                 } catch(std::invalid_argument) {
 177                                         lg::wml_error() << "Invalid min/max coordinates in cave_map_generator: " << items.front() << ", " << items.back() << "\n";
 178                                 }
 179                         }
 180                 }
 181                 const size_t x = translate_x(min_xpos + (rng_()%(max_xpos-min_xpos)));
 182                 const size_t y = translate_y(min_ypos + (rng_()%(max_ypos-min_ypos)));
 183 
 184                 int chamber_size = ch["size"].to_int(3);
 185                 int jagged_edges = ch["jagged"];
 186 
 187                 chamber new_chamber;
 188                 new_chamber.center = map_location(x,y);
 189                 build_chamber(new_chamber.center,new_chamber.locs,chamber_size,jagged_edges);
 190 
 191                 const config &items = ch.child("items");
 192                 new_chamber.items = items ? &items : nullptr;
 193 
 194                 const std::string &id = ch["id"];
 195                 if (!id.empty()) {
 196                         chamber_ids_[id] = chambers_.size();
 197                 }
 198 
 199                 chambers_.push_back(new_chamber);
 200 
 201                 for(const config &p : ch.child_range("passage"))
 202                 {
 203                         const std::string &dst = p["destination"];
 204 
 205                         // Find the destination of this passage
 206                         const std::map<std::string,size_t>::const_iterator itor = chamber_ids_.find(dst);
 207                         if(itor == chamber_ids_.end())
 208                                 continue;
 209 
 210                         assert(itor->second < chambers_.size());
 211 
 212                         passages_.emplace_back(new_chamber.center, chambers_[itor->second].center, p);
 213                 }
 214         }
 215 }

 217 void cave_map_generator::cave_map_generator_job::place_chamber(const chamber& c)
 218 {
 219         for(std::set<map_location>::const_iterator i = c.locs.begin(); i != c.locs.end(); ++i) {
 220                 set_terrain(*i,params.clear_);
 221         }
 222 
 223         if (c.items == nullptr || c.locs.empty()) return;
 224 
 225         size_t index = 0;
 226         for (const config::any_child &it : c.items->all_children_range())
 227         {
 228                 config cfg = it.cfg;
 229                 config &filter = cfg.child("filter");
 230                 config* object_filter = nullptr;
 231                 if (config &object = cfg.child("object")) {
 232                         if (config &of = object.child("filter"))
 233                                 object_filter = &of;
 234                 }
 235 
 236                 if (!it.cfg["same_location_as_previous"].to_bool()) {
 237                         index = rng_()%c.locs.size();
 238                 }
 239                 std::string loc_var = it.cfg["store_location_as"];
 240 
 241                 std::set<map_location>::const_iterator loc = c.locs.begin();
 242                 std::advance(loc,index);
 243 
 244                 cfg["x"] = loc->x + 1;
 245                 cfg["y"] = loc->y + 1;
 246 
 247                 if (filter) {
 248                         filter["x"] = loc->x + 1;
 249                         filter["y"] = loc->y + 1;
 250                 }
 251 
 252                 if (object_filter) {
 253                         (*object_filter)["x"] = loc->x + 1;
 254                         (*object_filter)["y"] = loc->y + 1;
 255                 }
 256 
 257                 // If this is a side, place a castle for the side
 258                 if (it.key == "side" && !it.cfg["no_castle"].to_bool()) {
 259                         place_castle(it.cfg["side"].to_int(-1), *loc);
 260                 }
 261 
 262                 res_.add_child(it.key, cfg);
 263 
 264                 if(!loc_var.empty()) {
 265                         config &temp = res_.add_child("event");
 266                         temp["name"] = "prestart";
 267                         config &xcfg = temp.add_child("set_variable");
 268                         xcfg["name"] = loc_var + "_x";
 269                         xcfg["value"] = loc->x + 1;
 270                         config &ycfg = temp.add_child("set_variable");
 271                         ycfg["name"] = loc_var + "_y";
 272                         ycfg["value"] = loc->y + 1;
 273                 }
 274         }
 275 }

 277 struct passage_path_calculator : pathfind::cost_calculator
 278 {
 279         passage_path_calculator(const t_translation::ter_map& mapdata,
 280                                 const t_translation::terrain_code & wall,
 281                                 double laziness, size_t windiness,
 282                                                         std::mt19937& rng) :
 283                 map_(mapdata), wall_(wall), laziness_(laziness), windiness_(windiness), rng_(rng)
 284         {}
 285 
 286         virtual double cost(const map_location& loc, const double so_far) const;
 287 private:
 288         const t_translation::ter_map& map_;
 289         t_translation::terrain_code wall_;
 290         double laziness_;
 291         size_t windiness_;
 292         std::mt19937& rng_;
 293 };

 295 double passage_path_calculator::cost(const map_location& loc, const double) const
 296 {
 297         double res = 1.0;
 298         if (map_.get(loc.x + gamemap::default_border, loc.y + gamemap::default_border) == wall_) {
 299                 res = laziness_;
 300         }
 301 
 302         if(windiness_ > 1) {
 303                 res *= double(rng_()%windiness_);
 304         }
 305 
 306         return res;
 307 }

 309 void cave_map_generator::cave_map_generator_job::place_passage(const passage& p)
 310 {
 311         const std::string& chance = p.cfg["chance"];
 312         if(chance != "" && int(rng_()%100) < std::stoi(chance)) {
 313                 return;
 314         }
 315 
 316 
 317         int windiness = p.cfg["windiness"];
 318         double laziness = std::max<double>(1.0, p.cfg["laziness"].to_double());
 319 
 320         passage_path_calculator calc(map_, params.wall_, laziness, windiness, rng_);
 321 
 322         pathfind::plain_route rt = a_star_search(p.src, p.dst, 10000.0, calc, params.width_, params.height_);
 323 
 324         int width = std::max<int>(1, p.cfg["width"].to_int());
 325         int jagged = p.cfg["jagged"];
 326 
 327         for(std::vector<map_location>::const_iterator i = rt.steps.begin(); i != rt.steps.end(); ++i) {
 328                 std::set<map_location> locs;
 329                 build_chamber(*i,locs,width,jagged);
 330                 for(std::set<map_location>::const_iterator j = locs.begin(); j != locs.end(); ++j) {
 331                         set_terrain(*j, params.clear_);
 332                 }
 333         }
 334 }

 336 void cave_map_generator::cave_map_generator_job::set_terrain(map_location loc, const t_translation::terrain_code & t)
 337 {
 338         if (params.on_board(loc)) {
 339                 t_translation::terrain_code& c = map_.get(loc.x + gamemap::default_border, loc.y + gamemap::default_border);
 340 
 341                 if(c == params.clear_ || c == params.wall_ || c == params.village_) {
 342                         // Change this terrain.
 343                         if ( t == params.clear_  &&  int(rng_() % 1000) < params.village_density_ )
 344                                 // Override with a village.
 345                                 c = params.village_;
 346                         else
 347                                 c = t;
 348                 }
 349         }
 350 }

 352 void cave_map_generator::cave_map_generator_job::place_castle(int starting_position, const map_location &loc)
 353 {
 354         if (starting_position != -1) {
 355                 set_terrain(loc, params.keep_);
 356 
 357                 t_translation::coordinate coord(
 358                                   loc.x + gamemap::default_border
 359                                 , loc.y + gamemap::default_border);
 360                 starting_positions_.insert(t_translation::starting_positions::value_type(std::to_string(starting_position), coord));
 361         }
 362 
 363         map_location adj[6];
 364         get_adjacent_tiles(loc,adj);
 365         for(size_t n = 0; n != 6; ++n) {
 366                 set_terrain(adj[n], params.castle_);
 367         }
 368 }
