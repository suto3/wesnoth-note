 145 #define NR_WHITE_MAGE_RESPAWN_AT HUB_CANDIDATE
 146     # Find a respawn point, starting within 1 hex of the candidate,
 147     # and increading in range if no suitable hex is found.
 148     # Tallin will be the hub candidate if the specified unit is still jailed
 149     [find_respawn_point]
 150         respawn_near={HUB_CANDIDATE}
 151         variable=respawn_point
 152     [/find_respawn_point]
 153 
 154     {FLASH_WHITE (
 155         [teleport]
 156             [filter]
 157                 id=$unit.id
 158             [/filter]
 159             x=$respawn_point.x
 160             y=$respawn_point.y
 161             animate=no
 162         [/teleport]
 163     )}
 164 
 165     [redraw][/redraw]
 166 
 167     [scroll_to]
 168         x=$respawn_point.x
 169         y=$respawn_point.y
 170     [/scroll_to]
 171 
 172     [delay]
 173         time=500
 174     [/delay]
 175 
 176     [heal_unit]
 177         [filter]
 178             id=$unit.id
 179         [/filter]
 180         [filter_second]
 181             id={HUB_CANDIDATE}
 182         [/filter_second]
 183 
 184         amount=full
 185         animate=yes
 186     [/heal_unit]
 187 
 188     {CLEAR_VARIABLE respawn_point}
 189 #enddef
