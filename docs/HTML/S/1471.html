  17 /**
  18  * A viewer sorter that sorts the items in the Project Explorer in the following
  19  * order:
  20  * 1) Wesnoth Core Library
  21  * 2) Containers
  22  * 3) Files
  23  */
  24 public class WesnothProjectsExplorerViewerSorter extends ViewerSorter
  25 {
  26     /**
  27      * Creates a new {@link WesnothProjectsExplorerViewerSorter}
  28      */
  29     public WesnothProjectsExplorerViewerSorter( )
  30     {
  31     }
  32 
  33     /**
  34      * Creates a new {@link WesnothProjectsExplorerViewerSorter}
  35      * 
  36      * @param collator
  37      *        A {@link Collator} instance
  38      */
  39     public WesnothProjectsExplorerViewerSorter( Collator collator )
  40     {
  41         super( collator );
  42     }
  43 
  44     @Override
  45     public int compare( Viewer viewer, Object e1, Object e2 )
  46     {
  47         // The core library container should be first everytime
  48         int result = super.compare( viewer, e1, e2 );
  49 
  50         if( result != 0
  51             && e1 instanceof IContainer
  52             && ( ( IContainer ) e1 ).getName( ).equals(
  53                 WesnothProjectsExplorer.CORE_LIBRARY_NAME ) ) {
  54             return - 1;
  55         }
  56 
  57         if( result != 0
  58             && e2 instanceof IContainer
  59             && ( ( IContainer ) e2 ).getName( ).equals(
  60                 WesnothProjectsExplorer.CORE_LIBRARY_NAME ) ) {
  61             return 1;
  62         }
  63 
  64         if( e1 instanceof IContainer && ! ( e2 instanceof IContainer ) ) {
  65             return - 1;
  66         }
  67 
  68         if( e2 instanceof IContainer && ! ( e1 instanceof IContainer ) ) {
  69             return 1;
  70         }
  71 
  72         return result;
  73     }
  74 }
