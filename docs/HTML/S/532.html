 270         /**
 271          * The position where to add a new callback in the signal handler.
 272          *
 273          * The signal handler has three callback queues:
 274          * * pre_child These callbacks are called before a container widget sends it
 275          *   to the child items. Widgets without children should also use this
 276          *   queue.
 277          * * child The callbacks for the proper child widget(s) are called.
 278          * * post_child The callbacks for the parent container to be called after
 279          *   the child.
 280          *
 281          * For every queue it's possible to add a new event in the front or in the
 282          * back.
 283          *
 284          * Whether all three queues are executed depend on the whether the
 285          * callbacks modify the handled and halt flag.
 286          * * When the halt flag is set execution of the current queue stops, when
 287          *   doing so the handled flag must be set as well.
 288          * * When the handled flag is set the events in that queue are executed and
 289          *   no more queues afterwards.
 290          *
 291          * Here are some use case examples.
 292          * A button that plays a sound and executes an optional user callback:
 293          * * The buttons internal click handler is invoked and sets the handled
 294          *   flag
 295          * * The callback installed by the user is in the same queue and gets
 296          *   executed afterwards.
 297          *
 298          * A toggle button may or may not be toggled:
 299          * * The user inserts a callback, that validates whether the action is
 300          *   allowed, if not allowed it sets the halt flag (and handled), else
 301          *   leaves the flags untouched.
 302          * * The normal buttons toggle function then might get invoked and if so
 303          *   sets the handled flag.
 304          * * Optionally there is another user callback invoked at this point.
 305          */
 306         enum queue_position {
 307                 front_pre_child,
 308                 back_pre_child,
 309                 front_child,
 310                 back_child,
 311                 front_post_child,
 312                 back_post_child
 313         };

 572         /**
 573          * The behavior of the mouse events.
 574          *
 575          * Normally for mouse events there's first checked whether a dispatcher has
 576          * captured the mouse if so it gets the event.
 577          * If not the dispatcher is searched from the back to the front in the
 578          * layers and its behavior is checked.
 579          * * none The event is never send to the layer and goes on the the next
 580          *   layer. This is used for tooltips who might cover a button but a click
 581          *   on the tooltips should still click the button.
 582          * * all The event is always send to this layer and stops the search for a
 583          *   next layer.
 584          * * hit If the mouse is inside the dispatcher area the event is send and
 585          *   no longer searched further. If not inside tests the last layer.
 586          *
 587          * If after these tests no dispatcher is found the event is ignored.
 588          */
 589         enum mouse_behavior {
 590                 all,
 591                 hit,
 592                 none
 593         };

 642         /** Helper struct to generate the various event queues. */
 643         template <class T>
 644         struct signal_queue
 645         {
 646                 signal_queue() : queue()
 647                 {
 648                 }
 649 
 650                 std::map<ui_event, signal_type<T> > queue;
 651 
 652                 void connect_signal(const ui_event event,
 653                                                         const queue_position position,
 654                                                         const T& signal)
 655                 {
 656                         switch(position) {
 657                                 case front_pre_child:
 658                                         queue[event].pre_child.push_front(signal);
 659                                         break;
 660                                 case back_pre_child:
 661                                         queue[event].pre_child.push_back(signal);
 662                                         break;
 663 
 664                                 case front_child:
 665                                         queue[event].child.push_front(signal);
 666                                         break;
 667                                 case back_child:
 668                                         queue[event].child.push_back(signal);
 669                                         break;
 670 
 671                                 case front_post_child:
 672                                         queue[event].post_child.push_front(signal);
 673                                         break;
 674                                 case back_post_child:
 675                                         queue[event].post_child.push_back(signal);
 676                                         break;
 677                         }
 678                 }

 680                 void disconnect_signal(const ui_event event,
 681                                                            const queue_position position,
 682                                                            const T& signal)
 683                 {
 684                         signal_type<T>& signal_queue = queue[event];
 685 
 686                         /* The function doesn't differentiate between front and back position so fall
 687                          * down from front to back.
 688                          *
 689                          * NOTE: This used to only remove the first signal of matching target type.
 690                          *       That behavior could be restored in the future if needed.
 691                          * - vultraz 5/2/2017
 692                          */
 693                         switch(position) {
 694                                 case front_pre_child:
 695                                 case back_pre_child: {
 696                                         signal_queue.pre_child.remove_if(
 697                                                 [&signal](T& element) { return signal.target_type() == element.target_type(); }
 698                                         );
 699                                 } break;
 700 
 701                                 case front_child:
 702                                 case back_child: {
 703                                         signal_queue.child.remove_if(
 704                                                 [&signal](T& element) { return signal.target_type() == element.target_type(); }
 705                                         );
 706                                 } break;
 707 
 708                                 case front_post_child:
 709                                 case back_post_child: {
 710                                         signal_queue.post_child.remove_if(
 711                                                 [&signal](T& element) { return signal.target_type() == element.target_type(); }
 712                                         );
 713                                 } break;
 714                         }
 715                 }
 716         };
