  34 static bool contains_ignore_case(const std::string& str1, const std::string& str2)
  35 {
  36         if(str2.size() > str1.size()) {
  37                 return false;
  38         }
  39 
  40         for(size_t i = 0; i < str1.size() - str2.size() + 1; i++) {
  41                 bool ok = true;
  42                 for(size_t j = 0; j < str2.size(); j++) {
  43                         if(std::tolower(str1[i + j]) != std::tolower(str2[j])) {
  44                                 ok = false;
  45                                 break;
  46                         }
  47                 }
  48 
  49                 if(ok) {
  50                         return true;
  51                 }
  52         }
  53 
  54         return false;
  55 }

 424                 std::vector<level_ptr> games;
 425                 std::vector<size_t> games_filtered;
 426 
 427                 void apply_filter(const int player_count, const std::string& name_filter)
 428                 {
 429                         games_filtered.clear();
 430 
 431                         for(size_t i = 0; i < games.size(); i++) {
 432                                 const bool num_players_match = player_count == 1 || games[i]->player_count_filter(player_count);
 433 
 434                                 if(contains_ignore_case(games[i]->name(), name_filter) && num_players_match) {
 435                                         games_filtered.push_back(i);
 436                                 }
 437                         }
 438                 }
 439 
 440                 void reset_filter()
 441                 {
 442                         games_filtered.resize(games.size());
 443                         std::iota(games_filtered.begin(), games_filtered.end(), 0);
 444                 }
 445         };
