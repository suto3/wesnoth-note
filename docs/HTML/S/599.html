 143 void scrollbar_container::request_reduce_height(const unsigned maximum_height)
 144 {
 145         DBG_GUI_L << LOG_HEADER << " requested height " << maximum_height << ".\n";
 146 
 147         /*
 148          * First ask the content to reduce it's height. This seems to work for now,
 149          * but maybe some sizing hints will be required later.
 150          */
 151         /** @todo Evaluate whether sizing hints are required. */
 152         assert(content_grid_);
 153         const unsigned offset =
 154                 horizontal_scrollbar_grid_ && horizontal_scrollbar_grid_->get_visible() != widget::visibility::invisible
 155                         ? horizontal_scrollbar_grid_->get_best_size().y
 156                         : 0;
 157 
 158         content_grid_->request_reduce_height(maximum_height - offset);
 159 
 160         // Did we manage to achieve the wanted size?
 161         point size = get_best_size();
 162         if(static_cast<unsigned>(size.y) <= maximum_height) {
 163                 DBG_GUI_L << LOG_HEADER << " child honored request, height " << size.y << ".\n";
 164                 return;
 165         }
 166 
 167         if(vertical_scrollbar_mode_ == ALWAYS_INVISIBLE) {
 168                 DBG_GUI_L << LOG_HEADER << " request failed due to scrollbar mode.\n";
 169                 return;
 170         }
 171 
 172         assert(vertical_scrollbar_grid_);
 173         const bool resized = vertical_scrollbar_grid_->get_visible() == widget::visibility::invisible;
 174 
 175         // Always set the bar visible, is a nop is already visible.
 176         vertical_scrollbar_grid_->set_visible(widget::visibility::visible);
 177 
 178         const point scrollbar_size = vertical_scrollbar_grid_->get_best_size();
 179 
 180         // If showing the scrollbar increased the height, hide and abort.
 181         if(resized && scrollbar_size.y > size.y) {
 182                 vertical_scrollbar_grid_->set_visible(widget::visibility::invisible);
 183                 DBG_GUI_L << LOG_HEADER << " request failed, showing the scrollbar"
 184                                   << " increased the height to " << scrollbar_size.y << ".\n";
 185                 return;
 186         }
 187 
 188         if(maximum_height > static_cast<unsigned>(scrollbar_size.y)) {
 189                 size.y = maximum_height;
 190         } else {
 191                 size.y = scrollbar_size.y;
 192         }
 193 
 194         // FIXME adjust for the step size of the scrollbar
 195 
 196         set_layout_size(size);
 197         DBG_GUI_L << LOG_HEADER << " resize resulted in " << size.y << ".\n";
 198 
 199         if(resized) {
 200                 DBG_GUI_L << LOG_HEADER << " resize modified the width, throw notification.\n";
 201 
 202                 throw layout_exception_width_modified();
 203         }
 204 }

 206 void scrollbar_container::request_reduce_width(const unsigned maximum_width)
 207 {
 208         DBG_GUI_L << LOG_HEADER << " requested width " << maximum_width << ".\n";
 209 
 210         // First ask our content, it might be able to wrap which looks better as
 211         // a scrollbar.
 212         assert(content_grid_);
 213         const unsigned offset =
 214                 vertical_scrollbar_grid_ && vertical_scrollbar_grid_->get_visible() != widget::visibility::invisible
 215                         ? vertical_scrollbar_grid_->get_best_size().x
 216                         : 0;
 217 
 218         content_grid_->request_reduce_width(maximum_width - offset);
 219 
 220         // Did we manage to achieve the wanted size?
 221         point size = get_best_size();
 222         if(static_cast<unsigned>(size.x) <= maximum_width) {
 223                 DBG_GUI_L << LOG_HEADER << " child honored request, width " << size.x << ".\n";
 224                 return;
 225         }
 226 
 227         if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE) {
 228                 DBG_GUI_L << LOG_HEADER << " request failed due to scrollbar mode.\n";
 229                 return;
 230         }
 231 
 232         // Always set the bar visible, is a nop when it's already visible.
 233         assert(horizontal_scrollbar_grid_);
 234         horizontal_scrollbar_grid_->set_visible(widget::visibility::visible);
 235         size = get_best_size();
 236 
 237         point scrollbar_size = horizontal_scrollbar_grid_->get_best_size();
 238 
 239         /*
 240          * If the vertical bar is not invisible it's size needs to be added to the
 241          * minimum size.
 242          */
 243         if(vertical_scrollbar_grid_->get_visible() != widget::visibility::invisible) {
 244                 scrollbar_size.x += vertical_scrollbar_grid_->get_best_size().x;
 245         }
 246 
 247         // If showing the scrollbar increased the width, hide and abort.
 248         if(horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN && scrollbar_size.x > size.x) {
 249                 horizontal_scrollbar_grid_->set_visible(widget::visibility::invisible);
 250                 DBG_GUI_L << LOG_HEADER << " request failed, showing the scrollbar"
 251                                   << " increased the width to " << scrollbar_size.x << ".\n";
 252                 return;
 253         }
 254 
 255         if(maximum_width > static_cast<unsigned>(scrollbar_size.x)) {
 256                 size.x = maximum_width;
 257         } else {
 258                 size.x = scrollbar_size.x;
 259         }
 260 
 261         // FIXME adjust for the step size of the scrollbar
 262 
 263         set_layout_size(size);
 264         DBG_GUI_L << LOG_HEADER << " resize resulted in " << size.x << ".\n";
 265 }

 272 point scrollbar_container::calculate_best_size() const
 273 {
 274         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 275 
 276         /***** get vertical scrollbar size *****/
 277         const point vertical_scrollbar = vertical_scrollbar_grid_->get_visible() == widget::visibility::invisible
 278                 ? point()
 279                 : vertical_scrollbar_grid_->get_best_size();
 280 
 281         /***** get horizontal scrollbar size *****/
 282         const point horizontal_scrollbar = horizontal_scrollbar_grid_->get_visible() == widget::visibility::invisible
 283                 ? point()
 284                 : horizontal_scrollbar_grid_->get_best_size();
 285 
 286         /***** get content size *****/
 287         assert(content_grid_);
 288         const point content = content_grid_->get_best_size();
 289 
 290         point result(
 291                 vertical_scrollbar.x + std::max(horizontal_scrollbar.x, content.x),
 292                 horizontal_scrollbar.y + std::max(vertical_scrollbar.y, content.y));
 293 
 294         //
 295         // Workaround for bug #24780. This should probably be moved somewhere more specific to
 296         // the listbox, but for now it suffices.
 297         //
 298         if(const grid* header = find_widget<const grid>(&get_grid(), "_header_grid", false, false)) {
 299                 result.y += header->get_best_size().y;
 300         }
 301 
 302         if(const grid* footer = find_widget<const grid>(&get_grid(), "_footer_grid", false, false)) {
 303                 result.y += footer->get_best_size().y;
 304         }
 305 
 306         DBG_GUI_L << LOG_HEADER << " vertical_scrollbar " << vertical_scrollbar << " horizontal_scrollbar "
 307                           << horizontal_scrollbar << " content " << content << " result " << result << ".\n";
 308 
 309         return result;
 310 }

 312 static void set_scrollbar_mode(grid* scrollbar_grid,
 313                 scrollbar_base* scrollbar,
 314                 scrollbar_container::scrollbar_mode& scrollbar_mode,
 315                 const unsigned items,
 316                 const unsigned visible_items,
 317                 grid* content_grid)
 318 {
 319         assert(scrollbar_grid && scrollbar);
 320 
 321         if(scrollbar_mode == scrollbar_container::ALWAYS_INVISIBLE) {
 322                 scrollbar_grid->set_visible(widget::visibility::invisible);
 323                 return;
 324         }
 325 
 326         scrollbar->set_item_count(items);
 327         scrollbar->set_item_position(0);
 328         scrollbar->set_visible_items(visible_items);
 329 
 330         if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE) {
 331                 const bool scrollbar_needed = items > visible_items;
 332                 scrollbar_grid->set_visible(scrollbar_needed ? widget::visibility::visible : widget::visibility::hidden);
 333         } else if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE_FIRST_RUN) {
 334                 if(items <= visible_items && content_grid != nullptr
 335                                 && scrollbar_grid->get_visible() == widget::visibility::visible
 336                 ) {
 337                         scrollbar_grid->set_visible(widget::visibility::invisible);
 338                         // Give newly freed space to the items.
 339                         content_grid->layout_initialize(false);
 340                 }
 341         }
 342 }
 343 static bool is_inserted_before(
 344                 unsigned insertion_pos, unsigned old_item_count, unsigned old_position, unsigned visible_items)
 345 {
 346         if(old_position == 0) {
 347                 return false;
 348         } else if(old_position + visible_items >= old_item_count) {
 349                 return true;
 350         } else if(insertion_pos <= old_position) {
 351                 return true;
 352         }
 353 
 354         return false;
 355 }

 357 static void adjust_scrollbar_mode(grid* scrollbar_grid,
 358                 scrollbar_base* scrollbar,
 359                 scrollbar_container::scrollbar_mode& scrollbar_mode,
 360                 const unsigned items_before,
 361                 const unsigned items_after,
 362                 const int insertion_pos,
 363                 const unsigned visible_items)
 364 {
 365         assert(scrollbar_grid && scrollbar);
 366         if(items_before != scrollbar->get_item_count()) {
 367                 return set_scrollbar_mode(scrollbar_grid, scrollbar, scrollbar_mode, items_after, visible_items, nullptr);
 368         }
 369 
 370         // TODO: does this also work well in case the items were removed?
 371         const unsigned previous_item_position = scrollbar->get_item_position();
 372 
 373         // Casts insertion_pos to an unsigned so negative values are interpreted as 'at end'
 374         const bool inserted_before_visible_area = is_inserted_before(
 375                         static_cast<unsigned>(insertion_pos), items_before, previous_item_position, visible_items);
 376 
 377         if(scrollbar_mode == scrollbar_container::ALWAYS_INVISIBLE) {
 378                 scrollbar_grid->set_visible(widget::visibility::invisible);
 379                 return;
 380         }
 381 
 382         scrollbar->set_item_count(items_after);
 383         scrollbar->set_item_position(inserted_before_visible_area
 384                 ? previous_item_position + items_after - items_before
 385                 : previous_item_position);
 386 
 387         // scrollbar->set_item_position(0);
 388         scrollbar->set_visible_items(visible_items);
 389 
 390         if(scrollbar_mode == scrollbar_container::AUTO_VISIBLE) {
 391                 const bool scrollbar_needed = items_after > visible_items;
 392                 scrollbar_grid->set_visible(scrollbar_needed ? widget::visibility::visible : widget::visibility::hidden);
 393         }
 394 }

 396 void scrollbar_container::place(const point& origin, const point& size)
 397 {
 398         // Inherited.
 399         container_base::place(origin, size);
 400 
 401         // Set content size
 402         assert(content_ && content_grid_);
 403 
 404         const point content_origin = content_->get_origin();
 405 
 406         const point best_size = content_grid_->get_best_size();
 407         const point content_size(content_->get_width(), content_->get_height());
 408 
 409         const point content_grid_size(std::max(best_size.x, content_size.x), std::max(best_size.y, content_size.y));
 410 
 411         set_content_size(content_origin, content_grid_size);
 412 
 413         // Set vertical scrollbar
 414         set_scrollbar_mode(
 415                 vertical_scrollbar_grid_,
 416                 vertical_scrollbar_,
 417                 vertical_scrollbar_mode_,
 418                 content_grid_->get_height(),
 419                 content_->get_height(),
 420                 content_grid_.get()
 421         );
 422 
 423         // Set horizontal scrollbar
 424         set_scrollbar_mode(
 425                 horizontal_scrollbar_grid_,
 426                 horizontal_scrollbar_,
 427                 horizontal_scrollbar_mode_,
 428                 content_grid_->get_width(),
 429                 content_->get_width(),
 430                 content_grid_.get()
 431         );
 432 
 433         // Update the buttons.
 434         set_scrollbar_button_status();
 435 
 436         // Now set the visible part of the content.
 437         content_visible_area_ = content_->get_rectangle();
 438         content_grid_->set_visible_rectangle(content_visible_area_);
 439 }

 514 bool scrollbar_container::content_resize_request(const bool force_sizing)
 515 {
 516         /**
 517          * @todo Try to handle AUTO_VISIBLE_FIRST_RUN here as well.
 518          *
 519          * Handling it here makes the code a bit more complex but allows to not
 520          * reserve space for scrollbars, which will look nicer in the MP lobby.
 521          * But the extra complexity is no 1.8 material.
 522          */
 523 
 524         assert(content_ && content_grid_);
 525 
 526         point best_size = content_grid_->recalculate_best_size();
 527         point size = content_->get_size();
 528 
 529         DBG_GUI_L << LOG_HEADER << " wanted size " << best_size << " available size " << size << ".\n";
 530 
 531         if(size == point()) {
 532                 DBG_GUI_L << LOG_HEADER << " initial setup not done, bailing out.\n";
 533                 return false;
 534         }
 535 
 536         if(best_size.x <= size.x && best_size.y <= size.y) {
 537                 const point content_size = content_grid_->get_size();
 538 
 539                 if(content_size.x > size.x || content_size.y > size.y) {
 540                         DBG_GUI_L << LOG_HEADER << " will fit, only needs a resize.\n";
 541                         goto resize;
 542                 }
 543 
 544                 if(force_sizing) {
 545                         DBG_GUI_L << LOG_HEADER << " fits, but resize forced.\n";
 546                         goto resize;
 547                 }
 548 
 549                 DBG_GUI_L << LOG_HEADER << " fits, nothing to do.\n";
 550                 return true;
 551         }
 552 
 553         if(best_size.x > size.x) {
 554                 DBG_GUI_L << LOG_HEADER << " content too wide.\n";
 555 
 556                 if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE ||
 557                         (
 558                                 horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN &&
 559                                 horizontal_scrollbar_grid_->get_visible() == widget::visibility::invisible
 560                         )
 561                 ) {
 562                         DBG_GUI_L << LOG_HEADER << " can't use horizontal scrollbar, request placement.\n";
 563 
 564                         event::message message;
 565                         fire(event::REQUEST_PLACEMENT, *this, message);
 566                         return false;
 567                 }
 568         }
 569 
 570         if(best_size.y > size.y) {
 571                 DBG_GUI_L << LOG_HEADER << " content too high.\n";
 572 
 573                 if(vertical_scrollbar_mode_ == ALWAYS_INVISIBLE ||
 574                         (
 575                                 vertical_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN &&
 576                                 vertical_scrollbar_grid_->get_visible() == widget::visibility::invisible
 577                         )
 578                 ) {
 579                         DBG_GUI_L << LOG_HEADER << " can't use vertical scrollbar, request placement.\n";
 580 
 581                         event::message message;
 582                         fire(event::REQUEST_PLACEMENT, *this, message);
 583                         return false;
 584                 }
 585         }
 586 
 587 resize:
 588         DBG_GUI_L << LOG_HEADER << " handle resizing.\n";
 589 
 590         place(get_origin(), get_size());
 591         return true;
 592 }

 594 bool scrollbar_container::content_resize_request(const int width_modification,
 595                 const int height_modification,
 596                 const int width_modification_pos,
 597                 const int height_modification_pos)
 598 {
 599         DBG_GUI_L << LOG_HEADER << " wanted width modification " << width_modification << " wanted height modification "
 600                           << height_modification << ".\n";
 601 
 602         if(get_size() == point()) {
 603                 DBG_GUI_L << LOG_HEADER << " initial setup not done, bailing out.\n";
 604                 return false;
 605         }
 606 
 607         window* window = get_window();
 608         assert(window);
 609 
 610         if(window->get_need_layout()) {
 611                 DBG_GUI_L << LOG_HEADER << " window already needs a layout phase, bailing out.\n";
 612                 return false;
 613         }
 614 
 615         assert(content_ && content_grid_);
 616 
 617         const bool result =
 618                 content_resize_width(width_modification, width_modification_pos) &&
 619                 content_resize_height(height_modification, height_modification_pos);
 620 
 621         scrollbar_moved();
 622 
 623         /*
 624          * The subroutines set the new size of the scrollbar but don't
 625          * update the button status.
 626          */
 627         if(result) {
 628                 set_scrollbar_button_status();
 629         }
 630 
 631         DBG_GUI_L << LOG_HEADER << " result " << result << ".\n";
 632         return result;
 633 }

 635 bool scrollbar_container::content_resize_width(const int width_modification, const int width_modification_pos)
 636 {
 637         if(width_modification == 0) {
 638                 return true;
 639         }
 640 
 641         const int new_width = content_grid_->get_width() + width_modification;
 642         DBG_GUI_L << LOG_HEADER << " current width " << content_grid_->get_width() << " wanted width " << new_width;
 643 
 644         if(new_width < 0) {
 645                 return false;
 646         }
 647 
 648         if(static_cast<unsigned>(new_width) <= content_->get_width()) {
 649                 DBG_GUI_L << " width fits in container, test height.\n";
 650 
 651                 adjust_scrollbar_mode(horizontal_scrollbar_grid_, horizontal_scrollbar_, horizontal_scrollbar_mode_,
 652                                 content_grid_->get_width(), content_grid_->get_width() + width_modification, width_modification_pos,
 653                                 content_->get_width());
 654                 return true;
 655         }
 656 
 657         assert(horizontal_scrollbar_ && horizontal_scrollbar_grid_);
 658         if(horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE ||
 659                 (
 660                         horizontal_scrollbar_mode_ == AUTO_VISIBLE_FIRST_RUN &&
 661                         horizontal_scrollbar_grid_->get_visible() == widget::visibility::invisible
 662                 )
 663         ) {
 664                 DBG_GUI_L << " can't use horizontal scrollbar, ask window.\n";
 665 
 666                 window* window = get_window();
 667                 assert(window);
 668 
 669                 window->invalidate_layout();
 670                 return false;
 671         }
 672 
 673         DBG_GUI_L << " use the horizontal scrollbar, test height.\n";
 674         adjust_scrollbar_mode(horizontal_scrollbar_grid_, horizontal_scrollbar_, horizontal_scrollbar_mode_,
 675                         content_grid_->get_width(), content_grid_->get_width() + width_modification, width_modification_pos,
 676                         content_->get_width());
 677 
 678         return true;
 679 }

 727 void scrollbar_container::finalize_setup()
 728 {
 729         /***** Setup vertical scrollbar *****/
 730         vertical_scrollbar_grid_ = find_widget<grid>(this, "_vertical_scrollbar_grid", false, true);
 731 
 732         vertical_scrollbar_ =
 733                 find_widget<scrollbar_base>(vertical_scrollbar_grid_, "_vertical_scrollbar", false, true);
 734 
 735         connect_signal_notify_modified(*vertical_scrollbar_,
 736                 std::bind(&scrollbar_container::vertical_scrollbar_moved, this));
 737 
 738         /***** Setup horizontal scrollbar *****/
 739         horizontal_scrollbar_grid_ = find_widget<grid>(this, "_horizontal_scrollbar_grid", false, true);
 740 
 741         horizontal_scrollbar_ =
 742                 find_widget<scrollbar_base>(horizontal_scrollbar_grid_, "_horizontal_scrollbar", false, true);
 743 
 744         connect_signal_notify_modified(*horizontal_scrollbar_,
 745                 std::bind(&scrollbar_container::horizontal_scrollbar_moved, this));
 746 
 747         /***** Setup the scrollbar buttons *****/
 748         for(const auto& item : scroll_lookup()) {
 749                 // Vertical.
 750                 clickable_item* button = find_widget<clickable_item>(vertical_scrollbar_grid_, item.first, false, false);
 751 
 752                 if(button) {
 753                         button->connect_click_handler(
 754                                 std::bind(&scrollbar_container::scroll_vertical_scrollbar, this, item.second));
 755                 }
 756 
 757                 // Horizontal.
 758                 button = find_widget<clickable_item>(horizontal_scrollbar_grid_, item.first, false, false);
 759 
 760                 if(button) {
 761                         button->connect_click_handler(
 762                                 std::bind(&scrollbar_container::scroll_horizontal_scrollbar, this, item.second));
 763                 }
 764         }
 765 
 766         /***** Setup the content *****/
 767         content_ = build_single_widget_instance<spacer>("spacer");
 768 
 769         // TODO: possibly move this unique_ptr casting functionality to a helper function.
 770         content_grid_.reset(dynamic_cast<grid*>(get_grid().swap_child("_content_grid", content_, true).release()));
 771         assert(content_grid_);
 772 
 773         content_grid_->set_parent(this);
 774 
 775         /***** Let our subclasses initialize themselves. *****/
 776         finalize_subclass();
 777 }

 827 void scrollbar_container::show_content_rect(const SDL_Rect& rect)
 828 {
 829         assert(content_);
 830         assert(horizontal_scrollbar_ && vertical_scrollbar_);
 831 
 832         // Set the bottom right location first if it doesn't fit the top left
 833         // will look good. First calculate the left and top position depending on
 834         // the current position.
 835 
 836         const int left_position = horizontal_scrollbar_->get_item_position() + (rect.x - content_->get_x());
 837         const int top_position = vertical_scrollbar_->get_item_position() + (rect.y - content_->get_y());
 838 
 839         // bottom.
 840         const int wanted_bottom = rect.y + rect.h;
 841         const int current_bottom = content_->get_y() + content_->get_height();
 842 
 843         int distance = wanted_bottom - current_bottom;
 844         if(distance > 0) {
 845                 vertical_scrollbar_->set_item_position(vertical_scrollbar_->get_item_position() + distance);
 846         }
 847 
 848         // right.
 849         const int wanted_right = rect.x + rect.w;
 850         const int current_right = content_->get_x() + content_->get_width();
 851 
 852         distance = wanted_right - current_right;
 853         if(distance > 0) {
 854                 horizontal_scrollbar_->set_item_position(horizontal_scrollbar_->get_item_position() + distance);
 855         }
 856 
 857         // top.
 858         if(top_position < static_cast<int>(vertical_scrollbar_->get_item_position())) {
 859                 vertical_scrollbar_->set_item_position(top_position);
 860         }
 861 
 862         // left.
 863         if(left_position < static_cast<int>(horizontal_scrollbar_->get_item_position())) {
 864                 horizontal_scrollbar_->set_item_position(left_position);
 865         }
 866 
 867         // Update.
 868         scrollbar_moved();
 869 }

1055 void scrollbar_container::scrollbar_moved()
1056 {
1057         // Init.
1058         assert(content_ && content_grid_);
1059         assert(vertical_scrollbar_ && horizontal_scrollbar_);
1060 
1061         /*** Update the content location. ***/
1062         const int x_offset = horizontal_scrollbar_mode_ == ALWAYS_INVISIBLE
1063                 ? 0
1064                 : horizontal_scrollbar_->get_item_position() * horizontal_scrollbar_->get_step_size();
1065 
1066         const int y_offset = vertical_scrollbar_mode_ == ALWAYS_INVISIBLE
1067                 ? 0
1068                 : vertical_scrollbar_->get_item_position() * vertical_scrollbar_->get_step_size();
1069 
1070         const point content_origin {content_->get_x() - x_offset, content_->get_y() - y_offset};
1071 
1072         content_grid_->set_origin(content_origin);
1073         content_grid_->set_visible_rectangle(content_visible_area_);
1074         content_grid_->set_is_dirty(true);
1075 
1076         // Update scrollbar.
1077         set_scrollbar_button_status();
1078 }
