  36         }

  38         /** Is the styled_widget selected? */

  45         virtual unsigned num_states() const = 0;
  46 
  47         bool get_value_bool() const
  48         {
  49                 assert(num_states() == 2);
  50                 return get_value() != 0;
  51         }
  52 
  53         void set_value_bool(const bool value)
  54         {
  55                 assert(num_states() == 2);
  56                 return set_value(value);
  57         }
