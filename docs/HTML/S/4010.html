   7 #define HIGHLIGHT_IMAGE X Y IMAGE BACKGROUND_VALUE
   8     # Scrolls to the given location, blinks the given image in and out a few
   9     # times, and then leaves the image in place. Use this to pinpoint an
  10     # important location to the player. If there's an existing [item] on the
  11     # location, specify it as the BACKGROUND_VALUE or else it'll get lost
  12     # during the blinking.
  13     [scroll_to]
  14         x,y={X},{Y}
  15         check_fogged=no
  16     [/scroll_to]
  17 
  18     [repeat]
  19         times=3
  20         [do]
  21             [item]
  22                 x,y={X},{Y}
  23                 halo={IMAGE}
  24             [/item]
  25 
  26             [redraw][/redraw]
  27 
  28             [delay]
  29                 time=300
  30             [/delay]
  31 
  32             [remove_item]
  33                 x,y={X},{Y}
  34             [/remove_item]
  35 
  36             [item]
  37                 x,y={X},{Y}
  38                 image={BACKGROUND_VALUE}
  39             [/item]
  40 
  41             [redraw][/redraw]
  42 
  43             [delay]
  44                 time=300
  45             [/delay]
  46         [/do]
  47     [/repeat]
  48 
  49     [item]
  50         x,y={X},{Y}
  51         image={IMAGE}
  52     [/item]
  53 
  54     [redraw][/redraw]
  55 #enddef

  57 #define CLEAR_FOG SIDE X Y RADIUS
  58     # Clears fog in a specific area for the given side.
  59     # This will persist until {UNCLEAR_FOG} (below) is used.
  60     [lift_fog]
  61         [filter_side]
  62             side={SIDE}
  63         [/filter_side]
  64         x,y={X},{Y}
  65         radius={RADIUS}
  66         multiturn=yes
  67     [/lift_fog]
  68 #enddef
  69 
  70 #define UNCLEAR_FOG
  71     # Restores fog that was temporarily cleared (check CLEAR_FOG above). Only
  72     # one UNCLEAR_FOG is required no matter how many times CLEAR_FOG was used.
  73     [reset_fog]
  74     [/reset_fog]
  75 #enddef

 101 #define FAKE_UNIT_MOVE FROM_X TO_X FROM_Y TO_Y SIDE TYPE MODIFIER_WML
 102     # Moves a fake unit using TYPE's animations, from (FROM_X, FROM_Y)
 103     # to (TO_X, TO_Y).  You can also specify modifiers that change the
 104     # baseframe displayed by unit's gender or variation, or none at
 105     # all.
 106     #
 107     # Example:
 108     #! {FAKE_UNIT_MOVE 1 2 1 2 1 (Dark Adept) (
 109     #!     gender=female
 110     #!     variation=fancy
 111     #! )}
 112     [move_unit_fake]
 113         type={TYPE}
 114         side={SIDE}
 115         x={FROM_X},{TO_X}
 116         y={FROM_Y},{TO_Y}
 117         {MODIFIER_WML}
 118     [/move_unit_fake]
 119 #enddef

 190 #define COLOR_ADJUST RED GREEN BLUE
 191     # Adjusts the color of the screen by a tint or red, green and blue

 201 # Flashes the screen with a given color performing ACTION_WML
 202 # These macros come in WHITE, RED, GREEN and BLUE and can
 203 # easily be expanded for ORANGE, PURPLE etc.
 204 #
 205 # Example: flash the screen to scare the player:
 206 #! {FLASH_GREEN ()}
 207 #! {FLASH_RED ()}
 208 #! {FLASH_BLUE ()}
 209 #! {FLASH_WHITE ()}
 210 
 211 #define FLASH_WHITE ACTION_WML
 212     # Flash the screen momentarily white.
 213     {COLOR_ADJUST 67 67 67}
 214     {COLOR_ADJUST 100 100 100}
 215     {ACTION_WML}
 216     {COLOR_ADJUST 33 33 33}
 217     {COLOR_ADJUST 0 0 0}
 218 #enddef
 219 
 220 #define FLASH_RED ACTION_WML
 221     # Flash the screen momentarily red.
 222     {COLOR_ADJUST 67 0 0}
 223     {COLOR_ADJUST 100 0 0}
 224     {ACTION_WML}
 225     {COLOR_ADJUST 33 0 0}
 226     {COLOR_ADJUST 0 0 0}
 227 #enddef
 228 
 229 #define FLASH_GREEN ACTION_WML
 230     # Flash the screen momentarily green.
 231     {COLOR_ADJUST 0 67 0}
 232     {COLOR_ADJUST 0 100 0}
 233     {ACTION_WML}
 234     {COLOR_ADJUST 0 33 0}
 235     {COLOR_ADJUST 0 0 0}
 236 #enddef
 237 
 238 #define FLASH_BLUE ACTION_WML
 239     # Flash the screen momentarily blue.

 241     {COLOR_ADJUST 0 0 100}
 242     {ACTION_WML}
 243     {COLOR_ADJUST 0 0 33}
 244     {COLOR_ADJUST 0 0 0}
