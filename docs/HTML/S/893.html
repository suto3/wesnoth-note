  58         int advance_unit_dialog(const map_location &loc)
  59         {
  60                 const unit& u = *resources::gameboard->units().find(loc);
  61                 std::vector<unit_const_ptr> previews;
  62 
  63                 for (const std::string& advance : u.advances_to()) {
  64                         preferences::encountered_units().insert(advance);
  65                         previews.push_back(get_advanced_unit(u, advance));
  66                 }
  67 
  68                 size_t num_real_advances = previews.size();
  69                 bool always_display = false;
  70 
  71                 for (const config& advance : u.get_modification_advances()) {
  72                         if (advance["always_display"]) {
  73                                 always_display = true;
  74                         }
  75                         previews.push_back(get_amla_unit(u, advance));
  76                 }
  77 
  78                 if (previews.size() > 1 || always_display) {
  79                         gui2::dialogs::unit_advance dlg(previews, num_real_advances);
  80 
  81                         dlg.show(CVideo::get_singleton());
  82 
  83                         if (dlg.get_retval() == gui2::window::OK) {
  84                                 return dlg.get_selected_index();
  85                         }
  86 
  87                         // This should be unreachable, since canceling is disabled for the dialog
  88                         assert(false && "Unit advance dialog was cancelled, which should be impossible.");
  89                 }
  90 
  91                 return 0;
  92         }

  94         bool animate_unit_advancement(const map_location &loc, size_t choice, const bool &fire_event, const bool animate)
  95         {
  96                 const events::command_disabler cmd_disabler;
  97 
  98                 unit_map::iterator u = resources::gameboard->units().find(loc);
  99                 if (u == resources::gameboard->units().end()) {
 100                         LOG_DP << "animate_unit_advancement suppressed: invalid unit\n";
 101                         return false;
 102                 }
 103                 else if (!u->advances()) {
 104                         LOG_DP << "animate_unit_advancement suppressed: unit does not advance\n";
 105                         return false;
 106                 }
 107 
 108                 const std::vector<std::string>& options = u->advances_to();
 109                 std::vector<config> mod_options = u->get_modification_advances();
 110 
 111                 if (choice >= options.size() + mod_options.size()) {
 112                         LOG_DP << "animate_unit_advancement suppressed: invalid option\n";
 113                         return false;
 114                 }
 115 
 116                 // When the unit advances, it fades to white, and then switches
 117                 // to the new unit, then fades back to the normal color
 118 
 119                 if (animate && !resources::screen->video().update_locked()) {
 120                         unit_animator animator;
 121                         bool with_bars = true;
 122                         animator.add_animation(&*u, "levelout", u->get_location(), map_location(), 0, with_bars);
 123                         animator.start_animations();
 124                         animator.wait_for_end();
 125                 }
 126 
 127                 if (choice < options.size()) {
 128                         // chosen_unit is not a reference, since the unit may disappear at any moment.
 129                         std::string chosen_unit = options[choice];
 130                         ::advance_unit(loc, chosen_unit, fire_event);
 131                 }
 132                 else {
 133                         const config &mod_option = mod_options[choice - options.size()];
 134                         ::advance_unit(loc, &mod_option, fire_event);
 135                 }
 136 
 137                 u = resources::gameboard->units().find(loc);
 138                 resources::screen->invalidate_unit();
 139 
 140                 if (animate && u != resources::gameboard->units().end() && !resources::screen->video().update_locked()) {
 141                         unit_animator animator;
 142                         animator.add_animation(&*u, "levelin", u->get_location(), map_location(), 0, true);
 143                         animator.start_animations();
 144                         animator.wait_for_end();
 145                         animator.set_all_standing();
 146                         resources::screen->invalidate(loc);
 147                         resources::screen->draw();
 148                         events::pump();
 149                 }
 150 
 151                 resources::screen->invalidate_all();
 152                 resources::screen->draw();
 153 
 154                 return true;
 155         }

 157         class unit_advancement_choice : public mp_sync::user_choice
 158         {
 159         public:
 160                 unit_advancement_choice(const map_location& loc, int total_opt, int side_num, const ai::unit_advancements_aspect* ai_advancement, bool force_dialog)
 161                         : loc_ (loc), nb_options_(total_opt), side_num_(side_num), ai_advancement_(ai_advancement), force_dialog_(force_dialog)
 162                 {
 163                 }
 164 
 165                 virtual ~unit_advancement_choice()
 166                 {
 167                 }
 168 
 169                 virtual config query_user(int /*side*/) const
 170                 {
 171                         //the 'side' parameter might differ from side_num_-
 172                         int res = 0;
 173                         team t = resources::gameboard->get_team(side_num_);
 174                         //i wonder how this got included here ?
 175                         bool is_mp = resources::controller->is_networked_mp();
 176                         bool is_current_side = resources::controller->current_side() == side_num_;
 177                         //note, that the advancements for networked sides are also determined on the current playing side.
 178 
 179                         //to make mp games equal we only allow selecting advancements to the current side.
 180                         //otherwise we'd give an unfair advantage to the side that hosts ai sides if units advance during ai turns.
 181                         if(!CVideo::get_singleton().non_interactive() && (force_dialog_ || (t.is_local_human() && !t.is_idle() && (is_current_side || !is_mp))))
 182                         {
 183                                 res = advance_unit_dialog(loc_);
 184                         }
 185                         else if(t.is_local_ai() || t.is_network_ai() || t.is_empty())
 186                         {
 187                                 res = rand() % nb_options_;
 188 
 189                                 //if ai_advancement_ is the default advancement the following code will
 190                                 //have no effect because get_advancements returns an empty list.
 191                                 if(ai_advancement_ != nullptr)
 192                                 {
 193                                         unit_map::iterator u = resources::gameboard->units().find(loc_);
 194                                         const std::vector<std::string>& options = u->advances_to();
 195                                         const std::vector<std::string>& allowed = ai_advancement_->get_advancements(u);
 196 
 197                                         for(std::vector<std::string>::const_iterator a = options.begin(); a != options.end(); ++a) {
 198                                                 if (std::find(allowed.begin(), allowed.end(), *a) != allowed.end()){
 199                                                         res = a - options.begin();
 200                                                         break;
 201                                                 }
 202                                         }
 203                                 }
 204 
 205                         }
 206                         else
 207                         {
 208                                 //we are in the situation, that the unit is owned by a human, but he's not allowed to do this decision.
 209                                 //because it's a mp game and it's not his turn.
 210                                 //note that it doesn't matter whether we call randomness::generator->next_random() or rand().
 211                                 res = randomness::generator->get_random_int(0, nb_options_-1);
 212                         }
 213                         LOG_NG << "unit at position " << loc_ << "choose advancement number " << res << "\n";
 214                         config retv;
 215                         retv["value"] = res;
 216                         return retv;
 217 
 218                 }
 219                 virtual config random_choice(int /*side*/) const
 220                 {
 221                         config retv;
 222                         retv["value"] = 0;
 223                         return retv;
 224                 }
 225                 virtual std::string description() const
 226                 {
 227                         return "an advancement choice";
 228                 }

 232                 int side_num_;
 233                 const ai::unit_advancements_aspect* ai_advancement_;
 234                 bool force_dialog_;
 235         };

 238 /*
 239 advances the unit and stores data in the replay (or reads data from replay).
 240 */
 241 void advance_unit_at(const advance_unit_params& params)
 242 {
 243         //i just don't want infinite loops...
 244         // the 20 is picked rather randomly.
 245         for(int advacment_number = 0; advacment_number < 20; advacment_number++)
 246         {
 247                 unit_map::iterator u = resources::gameboard->units().find(params.loc_);
 248                 //this implies u.valid()
 249                 if(!unit_helper::will_certainly_advance(u)) {
 250                         return;
 251                 }
 252 
 253                 if(params.fire_events_)
 254                 {
 255                         LOG_NG << "Firing pre advance event at " << params.loc_ <<".\n";
 256                         resources::game_events->pump().fire("pre_advance", params.loc_);
 257                         //TODO: maybe use id instead of location here ?.
 258                         u = resources::gameboard->units().find(params.loc_);
 259                         if(!unit_helper::will_certainly_advance(u))
 260                         {
 261                                 LOG_NG << "pre advance event aborted advancing.\n";
 262                                 return;
 263                         }
 264                 }
 265                 //we don't want to let side 1 decide it during start/prestart.
 266                 int side_for = resources::gamedata->phase() == game_data::PLAY ? 0: u->side();
 267                 config selected = mp_sync::get_user_choice("choose",
 268                         unit_advancement_choice(params.loc_, unit_helper::number_of_possible_advances(*u), u->side(), params.ai_advancements_, params.force_dialog_), side_for);
 269                 //calls actions::advance_unit.
 270                 bool result = animate_unit_advancement(params.loc_, selected["value"], params.fire_events_, params.animate_);
 271 
 272                 DBG_NG << "animate_unit_advancement result = " << result << std::endl;
 273                 u = resources::gameboard->units().find(params.loc_);
 274                 // level 10 unit gives 80 XP and the highest mainline is level 5
 275                 if (u.valid() && u->experience() > 80)
 276                 {
 277                         WRN_NG << "Unit has too many (" << u->experience() << ") XP left; cascade leveling goes on still." << std::endl;
 278                 }
 279         }
 280         ERR_NG << "unit at " << params.loc_ << "tried to advance more than 20 times. Advancing was aborted" << std::endl;
 281 }

 283 unit_ptr get_advanced_unit(const unit &u, const std::string& advance_to)
 284 {
 285         const unit_type *new_type = unit_types.find(advance_to);
 286         if (!new_type) {
 287                 throw game::game_error("Could not find the unit being advanced"
 288                         " to: " + advance_to);
 289         }
 290         unit_ptr new_unit(new unit(u));
 291         new_unit->set_experience(new_unit->experience() - new_unit->max_experience());
 292         new_unit->advance_to(*new_type);
 293         new_unit->heal_fully();
 294         new_unit->set_state(unit::STATE_POISONED, false);
 295         new_unit->set_state(unit::STATE_SLOWED, false);
 296         new_unit->set_state(unit::STATE_PETRIFIED, false);
 297         new_unit->set_user_end_turn(false);
 298         new_unit->set_hidden(false);
 299         return new_unit;
 300 }

 315 void advance_unit(map_location loc, const advancement_option &advance_to, bool fire_event)
 316 {
 317         unit_map::unit_iterator u = resources::gameboard->units().find(loc);
 318         if(!u.valid()) {
 319                 return;
 320         }
 321         // original_type is not a reference, since the unit may disappear at any moment.
 322         std::string original_type = u->type_id();
 323 
 324         // "advance" event.
 325         if(fire_event)
 326         {
 327                 LOG_NG << "Firing advance event at " << loc <<".\n";
 328                 resources::game_events->pump().fire("advance",loc);
 329 
 330                 if (!u.valid() || u->experience() < u->max_experience() ||
 331                         u->type_id() != original_type)
 332                 {
 333                         LOG_NG << "WML has invalidated the advancing unit. Aborting.\n";
 334                         return;
 335                 }
 336                 // In case WML moved the unit:
 337                 loc = u->get_location();
 338         }
 339 
 340         // This is not normally necessary, but if a unit loses power when leveling
 341         // (e.g. loses "jamming" or ambush), it could be discovered as a result of
 342         // the advancement.
 343         std::vector<int> not_seeing = actions::get_sides_not_seeing(*u);
 344 
 345         // Create the advanced unit.
 346         bool use_amla = boost::get<std::string>(&advance_to) == nullptr;
 347         unit_ptr new_unit = use_amla ? get_amla_unit(*u, *boost::get<const config*>(advance_to)) :
 348                                    get_advanced_unit(*u, boost::get<std::string>(advance_to));
 349         new_unit->set_location(loc);
 350         if ( !use_amla )
 351         {
 352                 statistics::advance_unit(*new_unit);
 353                 preferences::encountered_units().insert(new_unit->type_id());
 354                 LOG_CF << "Added '" << new_unit->type_id() << "' to the encountered units.\n";
 355         }
 356         resources::gameboard->units().erase(loc);
 357         u = resources::gameboard->units().insert(new_unit).first;
 358 
 359         // Update fog/shroud.
 360         actions::shroud_clearer clearer;
 361         clearer.clear_unit(loc, *new_unit);
 362 
 363         // "post_advance" event.
 364         if(fire_event)
 365         {
 366                 LOG_NG << "Firing post_advance event at " << loc << ".\n";
 367                 resources::game_events->pump().fire("post_advance",loc);
 368         }
 369 
 370         // "sighted" event(s).
 371         clearer.fire_events();
 372         if ( u.valid() )
 373                 actions::actor_sighted(*u, &not_seeing);
 374 
 375         resources::whiteboard->on_gamestate_change();
 376 }
