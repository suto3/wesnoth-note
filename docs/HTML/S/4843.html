   5 # least 50 gold because that's probably necessary to win Entering_the_Northlands on hard.

   7 # come all the way back to this scenario for more gold would be annoying--hence the tip.
   8 # You get the ability to recruit chocobones when a unit inspects the bones of a
   9 # flightless bird half-buried in the sand. There is one long event for this. Almost the
  10 # same event appears in Mountain_Pass if the player doesn't find the bones here.

 109     # This long event creates a chocobone when a unit moves to where bird bones are
 110     # buried in the sand.
 111     # There is a short conversation, but there are many possibilities for which units
 112     # are involved, so there are a lot of ifs. There is also a lot of animation around
 113     # the appearance of the chocobone involving a skeleton and a flightless-bird unit.
 114     [event]
 115         name=moveto
 116         [filter]
 117             x,y=32,12
 118             side=1
 119         [/filter]
 120 
 121         [if]
 122             [variable]
 123                 name=unit.race
 124                 equals=undead
 125             [/variable]
 126             [then]
 127                 # Bones examined by a ghost, skeleton, ghoul, or corpse.
 128                 [message]
 129                     speaker=$unit.id
 130                     message= _ "Bones. Claws and wings."
 131                 [/message]
 132             [/then]
 133             [else]
 134                 # Bones examined by a human.
 135                 [message]
 136                     speaker=$unit.id
 137                     message= _ "There are some large bones here. I can see claws and wings."
 138                 [/message]
 139             [/else]
 140         [/if]

 353         # We want to be able to add Ras-Tabahn's gold to the total when the two
 354         # characters are reunited, but only Ardonna's gold will be carried over
 355         # from the scenario right before that. We are storing R-T's gold now,
 356         # but we want to store what it would be after the bonus. We need to
 357         # calculate the bonus and store it before ending the scenario.
 358         [store_turns]
 359             variable=turns
 360         [/store_turns]
 361         [store_villages]
 362             variable=villages
 363         [/store_villages]
 364         [store_gold]
 365             side=1
 366             variable=gold
 367         [/store_gold]

 369         # This is the formula used by the game:
 370         [set_variable]
 371             name=rastabahn_gold
 372             value="$((($turns - $turn_number) * ($villages.length + 2) + ($gold)) * 0.4)"
 373         [/set_variable]
 374         [set_variable]
 375             name=rastabahn_gold
 376             round=0
 377         [/set_variable]

 410         # During the next message about gold, the carryover report will not have been
 411         # shown yet, so the player won't know how much gold R-T is talking about. To
 412         # fix that, we temporarily set the amount of gold shown at the top of the
 413         # screen to be equal to the carryover amount. Then we have to set it back to
 414         # make the carryover report correct.
 415         [gold]
 416             side=1
 417             amount="$(-1 * $gold + $rastabahn_gold)"
 418         [/gold]
 419         [message]
 420             speaker=Ras-Tabahn
 421             message=$gold_message
 422         [/message]
 423         [gold]
 424             side=1
 425             amount="$(-1 * $rastabahn_gold + $gold)"
 426         [/gold]
