  72     # This menu item macro is used for "recruiting" walking corpses, skeletons, and skeleton
  73     # archers from graves. UNIT is the unit type, UNIT_STRING is similar but has underscores
  74     # and is used for the id and icon filename. GOLD is the cost (a number).
  75 #define SET_MENU_ITEM_FOR UNIT UNIT_STRING GOLD
  76     # We need a variable to use in the switch statement:
  77     [set_variable]
  78         name=unit_string
  79         value={UNIT_STRING}
  80     [/set_variable]
  81     # The following sentences could be built from the macro arguments, but that might
  82     # make them impossible to translate properly. So, we use a switch statement instead.
  83     [switch]
  84         variable=unit_string
  85         [case]
  86             value=walking_corpse
  87             [set_variable]
  88                 name=description_text
  89                 value= _ "Raise Walking Corpse (8 Gold)"
  90             [/set_variable]
  91         [/case]
  92         [case]
  93             value=skeleton
  94             [set_variable]
  95                 name=description_text
  96                 value= _ "Raise Skeleton (15 Gold)"
  97             [/set_variable]
  98         [/case]
  99         [case]
 100             value=skeleton_archer
 101             [set_variable]
 102                 name=description_text
 103                 value= _ "Raise Skeleton Archer (14 Gold)"
 104             [/set_variable]
 105         [/case]
 106     [/switch]
 107     [clear_variable]
 108         name=unit_string
 109     [/clear_variable]
