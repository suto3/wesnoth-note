 134     # The next two events take four HP from Ras-Tabahn. If his HP is very low, he needs
 135     # to receive any healing (from villages or resting) *first*, or he will die. On the
 136     # other hand, if he has a lot of HP, and he is healed first, he will not receive
 137     # the full benefit. That is why we need two events. We will also be slowing him at
 138     # this time.
 139     [event]
 140         name=side 1 turn
 141         first_time_only=no

 143             [filter]
 144                 id=Ras-Tabahn
 145             [/filter]
 146             variable=rastabahn_stored
 147         [/store_unit]

 149             name=rastabahn_harmed
 150             value=no
 151         [/set_variable]

 156             [/variable]

 158                 [harm_unit]
 159                     [filter]
 160                         id=Ras-Tabahn
 161                     [/filter]

 164                 [/harm_unit]

 166                     name=rastabahn_harmed
 167                     value=yes
