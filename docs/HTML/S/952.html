 101 std::map<int,config> mp_sync::get_user_choice_multiple_sides(const std::string &name, const mp_sync::user_choice &uch,
 102         std::set<int> sides)
 103 {
 104         //pass sides by copy because we need a copy.
 105         const bool is_synced = synced_context::is_synced();
 106         const int max_side  = static_cast<int>(resources::gameboard->teams().size());
 107         //we currently don't check for too early because luas sync choice doesn't necessarily show screen dialogs.
 108         //It (currently) in the responsibility of the user of sync choice to not use dialogs during prestart events..
 109         if(!is_synced)
 110         {
 111                 //we got called from inside luas wesnoth.synchronize_choice or from a select event.
 112                 replay::process_error("MP synchronization only works in a synced context (for example Select or preload events are no synced context).\n");
 113                 return std::map<int,config>();
 114         }
 115 
 116         /*
 117                 for empty sides we want to use random choice instead.
 118         */
 119         std::set<int> empty_sides;
 120         for(int side : sides)
 121         {
 122                 assert(1 <= side && side <= max_side);
 123                 if( resources::gameboard->get_team(side).is_empty())
 124                 {
 125                         empty_sides.insert(side);
 126                 }
 127         }
 128 
 129         for(int side : empty_sides)
 130         {
 131                 sides.erase(side);
 132         }
 133 
 134         std::map<int,config> retv =  user_choice_manager::get_user_choice_internal(name, uch, sides);
 135 
 136         for(int side : empty_sides)
 137         {
 138                 retv[side] = uch.random_choice(side);
 139         }
 140         return retv;
 141 
 142 }

 147 config mp_sync::get_user_choice(const std::string &name, const mp_sync::user_choice &uch,
 148         int side)
 149 {
 150         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 151         const bool is_synced = synced_context::is_synced();
 152         const bool is_mp_game = resources::controller->is_networked_mp();//Only used in debugging output below
 153         const int max_side  = static_cast<int>(resources::gameboard->teams().size());
 154         bool is_side_null_controlled;
 155 
 156         if(!is_synced)
 157         {
 158                 //we got called from inside luas wesnoth.synchronize_choice or from a select event (or maybe a preload event?).
 159                 //This doesn't cause problems and someone could use it for example to use a [message][option] inside a wesnoth.synchronize_choice which could be useful,
 160                 //so just give a warning.
 161                 LOG_REPLAY << "MP synchronization called during an unsynced context.\n";
 162                 return uch.query_user(side);
 163         }
 164         if(is_too_early && uch.is_visible())
 165         {
 166                 //We are in a prestart event or even earlier.
 167                 //Although we are able to sync them, we cannot use query_user,
 168                 //because we cannot (or shouldn't) put things on the screen inside a prestart event, this is true for SP and MP games.
 169                 //Quotation form event wiki: "For things displayed on-screen such as character dialog, use start instead"
 170                 return uch.random_choice(side);
 171         }
 172         //in start events it's unclear to decide on which side the function should be executed (default= side1 still).
 173         //But for advancements we can just decide on the side that owns the unit and that's in the responsibility of advance_unit_at.
 174         //For [message][option] and luas sync_choice the scenario designer is responsible for that.
 175         //For [get_global_variable] side is never null.
 176 
 177         /*
 178                 side = 0 should default to the currently active side per definition.
 179         */
 180         if ( side < 1  ||  max_side < side )
 181         {
 182                 if(side != 0)
 183                 {
 184                         ERR_REPLAY << "Invalid parameter for side in get_user_choice." << std::endl;
 185                 }
 186                 side = resources::controller->current_side();
 187                 LOG_REPLAY << " side changed to " << side << "\n";
 188         }
 189         is_side_null_controlled = resources::gameboard->get_team(side).is_empty();
 190 
 191         LOG_REPLAY << "get_user_choice_called with"
 192                         << " name=" << name
 193                         << " is_synced=" << is_synced
 194                         << " is_mp_game=" << is_mp_game
 195                         << " is_side_null_controlled=" << is_side_null_controlled << "\n";
 196 
 197         if (is_side_null_controlled)
 198         {
 199                 DBG_REPLAY << "MP synchronization: side 1 being null-controlled in get_user_choice.\n";
 200                 //most likely we are in a start event with an empty side 1
 201                 //but calling [set_global_variable] to an empty side might also cause this.
 202                 //i think in that case we should better use uch.random_choice(),
 203                 //which could return something like config {"invalid", true};
 204                 side = 1;
 205                 while ( side <= max_side  &&  resources::gameboard->get_team(side).is_empty() )
 206                         side++;
 207                 assert(side <= max_side);
 208         }
 209 
 210 
 211         assert(1 <= side && side <= max_side);
 212 
 213         std::set<int> sides;
 214         sides.insert(side);
 215         std::map<int, config> retv = user_choice_manager::get_user_choice_internal(name, uch, sides);
 216         if(retv.find(side) == retv.end())
 217         {
 218                 //An error occured, get_user_choice_internal should have given an oos error message
 219                 return config();
 220         }
 221         return retv[side];
 222 }

 224 user_choice_manager::user_choice_manager(const std::string &name, const mp_sync::user_choice &uch, const std::set<int>& sides)
 225         : required_(sides)
 226         , res_()
 227         , local_choice_(0)
 228         , wait_message_()
 229         , oos_(false)
 230         , uch_(uch)
 231         , tagname_(name)
 232         , current_side_(resources::controller->current_side())
 233         , changed_event_("user_choice_update")
 234 {
 235         update_local_choice();
 236         const int max_side  = static_cast<int>(resources::gameboard->teams().size());
 237 
 238         for(int side : required_)
 239         {
 240                 assert(1 <= side && side <= max_side);
 241                 const team& t = resources::gameboard->get_team(side);
 242                 assert(!t.is_empty());
 243                 if(side != current_side_)
 244                 {
 245                         synced_context::set_is_simultaneously();
 246                 }
 247         }
 248 
 249         do_replay_handle();
 250         search_in_replay();
 251 
 252 }

 254 void user_choice_manager::search_in_replay()
 255 {
 256         while(!finished() && !oos_)
 257         {
 258                 do_replay_handle();
 259                 if(resources::recorder->at_end()) {
 260                         return;
 261                 }
 262 
 263                 DBG_REPLAY << "MP synchronization: extracting choice from replay with has_local_side=" << has_local_choice() << "\n";
 264 
 265                 const config *action = resources::recorder->get_next_action();
 266                 assert(action); //action cannot be null because resources::recorder->at_end() returned false.
 267                 if( !action->has_child(tagname_) || !(*action)["dependent"].to_bool())
 268                 {
 269                         replay::process_error("[" + tagname_ + "] expected but none found\n. found instead:\n" + action->debug());
 270                         //We save this action for later
 271                         resources::recorder->revert_action();
 272                         // execute this local choice locally
 273                         oos_ = true;
 274                         changed_event_.notify_observers();
 275                         return;
 276                 }
 277                 int from_side = (*action)["from_side"].to_int(0);
 278                 if((*action)["side_invalid"].to_bool(false) == true)
 279                 {
 280                         //since this 'cheat' can have a quite heavy effect especialy in umc content we give an oos error .
 281                         replay::process_error("MP synchronization: side_invalid in replay data, this could mean someone wants to cheat.\n");
 282                 }
 283                 if(required_.find(from_side) == required_.end())
 284                 {
 285                         replay::process_error("MP synchronization: we got an answer from side " + std::to_string(from_side) + "for [" + tagname_ + "] which is not was we expected\n");
 286                 }
 287                 if(res_.find(from_side) != res_.end())
 288                 {
 289                         replay::process_error("MP synchronization: we got already our answer from side " + std::to_string(from_side) + "for [" + tagname_ + "] now we have it twice.\n");
 290                 }
 291                 res_[from_side] = action->child(tagname_);
 292                 changed_event_.notify_observers();
 293         }

 295 void user_choice_manager::pull()
 296 {
 297         // there might be speak or similar commands in the replay before the user input.
 298         do_replay_handle();
 299         synced_context::pull_remote_user_input();
 300         do_replay_handle();
 301         update_local_choice();
 302         search_in_replay();

 305 void user_choice_manager::update_local_choice()
 306 {
 307         int local_choice_prev = local_choice_;
 308         //equals to any side in sides that is local, 0 if no such side exists.
 309         local_choice_ = 0;
 310         //if for any side from which we need an answer
 311         std::string sides_str;
 312         for(int side : required_)
 313         {
 314                 //and we havent already received our answer from that side
 315                 if(res_.find(side) == res_.end())
 316                 {
 317                         sides_str += " ";
 318                         sides_str += std::to_string(side);
 319                         //and it is local
 320                         if(resources::gameboard->get_team(side).is_local() && !resources::gameboard->get_team(side).is_idle())
 321                         {
 322                                 //then we have to make a local choice.
 323                                 local_choice_ = side;
 324                                 break;
 325                         }
 326                 }
 327         }
 328         wait_message_ = vgettext("waiting for $desc from side(s) $sides", {std::make_pair("desc", uch_.description()), std::make_pair("sides", sides_str)});
 329         if(local_choice_prev != local_choice_) {
 330                 changed_event_.notify_observers();
 331         }
 332 }

 334 void user_choice_manager::ask_local_choice()
 335 {
 336         assert(local_choice_ != 0);
 337 
 338         leave_synced_context sync;
 339         /* At least one of the decisions is ours, and it will be inserted
 340         into the replay. */
 341         DBG_REPLAY << "MP synchronization: local choice\n";
 342         config cfg = uch_.query_user(local_choice_);
 343         if(res_.find(local_choice_) != res_.end()) {
 344                 // It might be possible that we this choice was already made by another client while we were in uch_.query_user
 345                 // becase our side might be ressigned while we made our choice.
 346                 WRN_REPLAY << "Discarding a local choice becasue we found it already on the replay";
 347                 return;
 348         }
 349         resources::recorder->user_input(tagname_, cfg, local_choice_);
 350         res_[local_choice_] = cfg;
 351 
 352         //send data to others.
 353         //but if there wasn't any data sended during this turn, we don't want to bein wth that now.
 354         //TODO: we should send user choices during nonundoable actions immideatley.
 355         if(synced_context::is_simultaneously() || current_side_ != local_choice_)
 356         {
 357                 synced_context::send_user_choice();
 358         }
 359         update_local_choice();
 360 }

 362 void user_choice_manager::fix_oos()
 363 {
 364         assert(oos_);
 365         ERR_REPLAY << "A sync error appeared while waiting for a synced user choice of type '" << uch_.description() << "' ([" + tagname_ + "]), doing the choice locally\n";
 366         for(int side : required_)
 367         {
 368                 if(res_.find(side) == res_.end())
 369                 {
 370                         ERR_REPLAY << "Doing a local choice for side " << side << "\n";
 371                         res_[side] = uch_.query_user(side);
 372                 }
 373         }
 374         oos_ = false;
 375 }

 377 static void wait_ingame(user_choice_manager& man)
 378 {
 379         user_choice_notifer_ingame notifer;
 380         while(!man.finished() && man.waiting())
 381         {
 382                 if(resources::gamedata->phase() == game_data::PLAY || resources::gamedata->phase() == game_data::START)
 383                 {
 384                         //during the prestart/preload event the screen is locked and we shouldn't call user_interact.
 385                         //because that might result in crashs if someone clicks anywhere during screenlock.
 386 
 387                         // calls man.pull via events.cpp -> pump_monitor::process
 388                         resources::controller->play_slice();
 389                 }
 390 
 391                 notifer.update(man.wait_message());
 392         }
 393 }

 401 std::map<int, config> user_choice_manager::get_user_choice_internal(const std::string &name, const mp_sync::user_choice &uch, const std::set<int>& sides)
 402 {
 403         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 404         user_choice_manager man(name, uch, sides);
 405         while(!man.finished())
 406         {
 407                 if(man.waiting())
 408                 {
 409                         if(is_too_early) {
 410                                 wait_prestart(man);
 411                         }
 412                         else {
 413                                 wait_ingame(man);
 414                         }
 415                 }
 416                 else if(man.has_local_choice())
 417                 {
 418                         man.ask_local_choice();
 419                 }
 420                 else
 421                 {
 422                         man.fix_oos();
 423                 }
 424         }
 425         return man.res_;
 426 }
