 102 void game_board::check_victory(bool & continue_level, bool & found_player, bool & found_network_player, bool & cleared_villages, std::set<unsigned> & not_defeated, bool remove_from_carryover_on_defeat)
 103 {
 104         continue_level = true;
 105         found_player = false;
 106         found_network_player = false;
 107         cleared_villages = false;
 108 
 109         not_defeated = std::set<unsigned>();
 110 
 111         for (const unit & i : units())
 112         {
 113                 DBG_EE << "Found a unit: " << i.id() << " on side " << i.side() << std::endl;
 114                 const team& tm = get_team(i.side());
 115                 DBG_EE << "That team's defeat condition is: " << tm.defeat_condition() << std::endl;
 116                 if (i.can_recruit() && tm.defeat_condition() == team::DEFEAT_CONDITION::NO_LEADER) {
 117                         not_defeated.insert(i.side());
 118                 } else if (tm.defeat_condition() == team::DEFEAT_CONDITION::NO_UNITS) {
 119                         not_defeated.insert(i.side());
 120                 }
 121         }
 122 
 123         for (team& tm : teams_)
 124         {
 125                 if(tm.defeat_condition() == team::DEFEAT_CONDITION::NEVER)
 126                 {
 127                         not_defeated.insert(tm.side());
 128                 }
 129                 // Clear villages for teams that have no leader and
 130                 // mark side as lost if it should be removed from carryover.
 131                 if (not_defeated.find(tm.side()) == not_defeated.end())
 132                 {
 133                         tm.clear_villages();
 134                         // invalidate_all() is overkill and expensive but this code is
 135                         // run rarely so do it the expensive way.
 136                         cleared_villages = true;
 137 
 138                         if (remove_from_carryover_on_defeat)
 139                         {
 140                                 tm.set_lost(true);
 141                         }
 142                 }
 143                 else if(remove_from_carryover_on_defeat)
 144                 {
 145                         tm.set_lost(false);
 146                 }
 147         }
 148 
 149         for (std::set<unsigned>::iterator n = not_defeated.begin(); n != not_defeated.end(); ++n) {
 150                 size_t side = *n - 1;
 151 
 152                 DBG_EE << "Side " << (side+1) << " is a not-defeated team" << std::endl;
 153 
 154                 std::set<unsigned>::iterator m(n);
 155                 for (++m; m != not_defeated.end(); ++m) {
 156                         if (teams()[side].is_enemy(*m)) {
 157                                 return;
 158                         }
 159                         DBG_EE << "Side " << (side+1) << " and " << *m << " are not enemies." << std::endl;
 160                 }
 161 
 162                 if (teams()[side].is_local_human()) {
 163                         found_player = true;
 164                 }
 165 
 166                 if (teams()[side].is_network_human()) {
 167                         found_network_player = true;
 168                 }
 169         }
 170 
 171         continue_level = false;
 172 }

 174 unit_map::iterator game_board::find_visible_unit(const map_location &loc,
 175         const team& current_team, bool see_all)
 176 {
 177         if (!map_->on_board(loc)) return units_.end();
 178         unit_map::iterator u = units_.find(loc);
 179         if (!u.valid() || !u->is_visible_to_team(current_team, *this, see_all))
 180                 return units_.end();
 181         return u;
 182 }

 184 bool game_board::has_visible_unit(const map_location & loc, const team& current_team, bool see_all) const
 185 {
 186         if (!map_->on_board(loc)) return false;
 187         unit_map::const_iterator u = units_.find(loc);
 188         if (!u.valid() || !u->is_visible_to_team(current_team, *this, see_all))
 189                 return false;

 201 void game_board::side_drop_to(int side_num, team::CONTROLLER ctrl, team::PROXY_CONTROLLER proxy) {
 202         team &tm = teams_[side_num-1];

 210         unit_map::iterator leader = units_.find_leader(side_num);

 231 bool game_board::team_is_defeated(const team& t) const
 232 {
 233         switch(t.defeat_condition().v)
 234         {
 235         case team::DEFEAT_CONDITION::ALWAYS:
 236                 return true;
 237         case team::DEFEAT_CONDITION::NO_LEADER:
 238                 return !units_.find_leader(t.side()).valid();
 239         case team::DEFEAT_CONDITION::NO_UNITS:
 240                 for (const unit& u : units_)
 241                 {
 242                         if(u.side() == t.side())
 243                                 return false;
 244                 }
 245                 return true;
 246         case team::DEFEAT_CONDITION::NEVER:
 247         default:
 248                 return false;
 249         }
 250 }

 259 boost::optional<std::string> game_board::replace_map(const gamemap & newmap) {
 260         boost::optional<std::string> ret = boost::optional<std::string> ();
 261 
 262         /* Remember the locations where a village is owned by a side. */
 263         std::map<map_location, int> villages;
 264         for(const auto& village : map_->villages()) {
 265                 const int owner = village_owner(village);
 266                 if(owner != -1) {
 267                         villages[village] = owner;
 268                 }
 269         }
 270 
 271         for (unit_map::iterator itor = units_.begin(); itor != units_.end(); ) {
 272                 if (!newmap.on_board(itor->get_location())) {
 273                         if (!try_add_unit_to_recall_list(itor->get_location(), itor.get_shared_ptr())) {
 274                                 *ret = std::string("replace_map: Cannot add a unit that would become off-map to the recall list\n");
 275                         }
 276                         units_.erase(itor++);
 277                 } else {
 278                         ++itor;
 279                 }
 280         }
 281 
 282         /* Disown villages that are no longer villages. */
 283         for(const auto& village : villages) {
 284                 if(!newmap.is_village(village.first)) {
 285                         teams_[village.second].lose_village(village.first);
 286                 }
 287         }
 288 
 289         *map_ = newmap;
 290         return ret;
 291 }

 299 bool game_board::change_terrain(const map_location &loc, const std::string &t_str,
 300                     const std::string & mode_str, bool replace_if_failed)
 301 {
 302         //Code internalized from the implementation in lua.cpp
 303         t_translation::terrain_code terrain = t_translation::read_terrain_code(t_str);
 304         if (terrain == t_translation::NONE_TERRAIN) return false;
 305 
 306         terrain_type_data::merge_mode mode = terrain_type_data::BOTH;
 307 
 308         if (mode_str == "base") mode = terrain_type_data::BASE;
 309         else if (mode_str == "overlay") mode = terrain_type_data::OVERLAY;
 310 
 311         /*
 312          * When a hex changes from a village terrain to a non-village terrain, and
 313          * a team owned that village it loses that village. When a hex changes from
 314          * a non-village terrain to a village terrain and there is a unit on that
 315          * hex it does not automatically capture the village. The reason for not
 316          * capturing villages it that there are too many choices to make; should a
 317          * unit loose its movement points, should capture events be fired. It is
 318          * easier to do this as wanted by the author in WML.
 319          */
 320 
 321         t_translation::terrain_code
 322                 old_t = map_->get_terrain(loc),
 323                 new_t = map_->tdata()->merge_terrains(old_t, terrain, mode, replace_if_failed);
 324         if (new_t == t_translation::NONE_TERRAIN) return false;
 325         preferences::encountered_terrains().insert(new_t);
 326 
 327         if (map_->tdata()->is_village(old_t) && !map_->tdata()->is_village(new_t)) {
 328                 int owner = village_owner(loc);
 329                 if (owner != -1)
 330                         teams_[owner].lose_village(loc);
 331         }
 332 
 333         map_->set_terrain(loc, new_t);
 334 
 335         for(const t_translation::terrain_code &ut : map_->underlying_union_terrain(loc)) {
 336                 preferences::encountered_terrains().insert(ut);
 337         }
 338         return true;
 339 }

 341 void game_board::write_config(config & cfg) const
 342 {
 343         cfg["next_underlying_unit_id"] = unit_id_manager_.get_save_id();
 344         for(std::vector<team>::const_iterator t = teams_.begin(); t != teams_.end(); ++t) {
 345                 int side_num = t - teams_.begin() + 1;
 346 
 347                 config& side = cfg.add_child("side");
 348                 t->write(side);
 349                 side["no_leader"] = true;
 350                 side["side"] = std::to_string(side_num);
 351 
 352                 //current units
 353                 for(const unit & i : units_) {
 354                         if(i.side() == side_num) {
 355                                 config& u = side.add_child("unit");
 356                                 i.get_location().write(u);
 357                                 i.write(u);
 358                         }
 359                 }
 360                 //recall list
 361                 for(const unit_const_ptr & j : t->recall_list()) {
 362                         config& u = side.add_child("unit");
 363                         j->write(u);
 364                 }
 365         }
 366 
 367         //write the map
 368         cfg["map_data"] = map_->write();
 369 }

 414 /**
 415  * Constructor
 416  * This version will change the unit's current movement to @a new_moves while
 417  * the unit is moved (and restored to its previous value upon this object's
 418  * destruction).
 419  */
 420 temporary_unit_mover::temporary_unit_mover(unit_map& m, const map_location& src,
 421                                            const map_location& dst, int new_moves)
 422         : m_(m), src_(src), dst_(dst), old_moves_(-1),
 423           temp_(src == dst ? unit_ptr() : m_.extract(dst))

 429         {
 430                 old_moves_ = move_result.first->movement_left(true);

 474
