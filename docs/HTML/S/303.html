 620         /**
 621          *  Rotates a template rule to a given angle.
 622          *
 623          *  Template rules are defined like normal rules, except that:
 624          *  * Flags and image filenames may contain template strings of the form
 625          *@verbatim
 626          *  <code>@Rn</code>, n being a number from 0 to 5.
 627          *@endverbatim
 628          *  * The rule contains the rotations=r0,r1,r2,r3,r4,r5, with r0 to r5
 629          *    being strings describing the 6 different positions, typically,
 630          *    n, ne, se, s, sw, and nw (but maybe anything else.)
 631          *
 632          *  A template rule will generate 6 rules, which are similar
 633          *  to the template, except that:
 634          *
 635          *  * The map of constraints ( [tile]s ) of this rule will be
 636          *    rotated by an angle, of 0 to 5 pi / 6
 637          *
 638          *  * On the rule which is rotated to 0rad, the template strings
 639          *@verbatim
 640          *    @R0, @R1, @R2, @R3, @R4, @R5,
 641          *@endverbatim
 642          *    will be replaced by the corresponding r0, r1, r2, r3, r4, r5
 643          *    variables given in the rotations= element.
 644          *
 645          *  * On the rule which is rotated to pi/3 rad, the template strings
 646          *@verbatim
 647          *    @R0, @R1, @R2 etc.
 648          *@endverbatim
 649          *    will be replaced by the corresponding
 650          *    <strong>r1, r2, r3, r4, r5, r0</strong> (note the shift in indices).
 651          *
 652          *  * On the rule rotated 2pi/3, those will be replaced by
 653          *    r2, r3, r4, r5, r0, r1 and so on.
 654          *
 655          */
 656         void rotate_rule(building_rule& rule, int angle, const std::vector<std::string>& angle_name);
