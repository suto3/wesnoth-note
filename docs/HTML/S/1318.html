  53     def write(self, filebuf, fuzzy):
  54         if self.wmlinfos is not None:
  55             for i in self.wmlinfos:
  56                 print('#.', i, file=filebuf)
  57         if self.addedinfos is not None:
  58             for i in self.addedinfos:
  59                 print('#.', i, file=filebuf)
  60         if self.finfos is not None:
  61             for i in self.finfos:
  62                 print('#:', i, file=filebuf)
  63         if fuzzy:
  64             print('#, fuzzy', file=filebuf)
  65         self.sentence = '"' + self.sentence + '"'
  66         if self.ismultiline:
  67             lf = r'\\n"' + '\n"'
  68             self.sentence = re.sub(r'(\n\r|\r\n|[\n\r])', lf, self.sentence)
  69             self.sentence = '""\n' + self.sentence
  70         print('msgid', self.sentence, file=filebuf)
  71         if self.plural is None:
  72             print('msgstr ""', file=filebuf)
  73         else:
  74             # self.plural must be an instance of PoCommentedStringPL
  75             print('msgid_plural', self.plural.value, file=filebuf)
  76             print('msgstr[0] ""', file=filebuf)
  77             print('msgstr[1] ""', file=filebuf)

 129     def add_sentence(self, sentence, *, ismultiline, lineno, 
 130                      lineno_sub=0, plural=None, override=None, addition=None):
 131         if self.sentences is None:
 132             self.sentences = []
 133         # 'plural' parameter accepted by WmlNode.add_sentence can be:
 134         #     1) A string
 135         #     2) A tuple of two values (string, bool)
 136         #     3) An actual PoCommentedStringPL object
 137         # This is why we need to add an extra variable here (plural_value)
 138         # wich will store an PoCommentedStringPL object (or will be None)
 139         plural_value = None 
 140         if plural is not None:
 141             if isinstance(plural, str):
 142                 plural_value = PoCommentedStringPL(plural)
 143             elif isinstance(plural, tuple) or isinstance(plural, list):
 144                 if len(plural) == 2:
 145                     if(isinstance(plural[0], str) and 
 146                           isinstance(plural[1], bool)):
 147                         plural_value = PoCommentedStringPL(plural[0],
 148                                                            plural[1])
 149                     elif(isinstance(plural[0], bool) and 
 150                           isinstance(plural[1], str)):
 151                         plural_value = PoCommentedStringPL(plural[1],
 152                                                            plural[0])
 153             elif isinstance(plural, PoCommentedStringPL):
 154                 plural_value = plural
 155                 plural = None
 156         self.sentences.append( WmlNodeSentence(sentence, 
 157                                           ismultiline=ismultiline, 
 158                                           lineno=lineno, 
 159                                           lineno_sub=lineno_sub,
 160                                           plural=plural_value,
 161                                           override=override, 
 162                                           addition=addition) )

 164     def assemble_wmlinfo(self):
 165         if self.tagname == "":
 166             value = []
 167             return value
 168         else:
 169             mystr = self.tagname
 170             intstr = 0
 171             if self.wmlinfos is not None:
 172                 if len(self.wmlinfos) > 0:
 173                     mystr += ": "
 174                     for k in self.wmlinfos:
 175                         intstr += 1
 176                         if intstr > 1:
 177                             mystr += ", "
 178                         mystr += k
 179             return mystr
