 103 BOOST_AUTO_TEST_CASE( test_wildcard_string_match )
 104 {
 105         const std::string str = "foo bar baz";
 106 
 107         BOOST_CHECK(utils::wildcard_string_match(str, "*bar*"));
 108 
 109         BOOST_CHECK(!utils::wildcard_string_match(str, "*BAR*"));
 110         BOOST_CHECK(!utils::wildcard_string_match(str, "bar"));
 111 
 112         BOOST_CHECK(utils::wildcard_string_match(str, "*ba? b*"));
 113         BOOST_CHECK(utils::wildcard_string_match(str, "*?a?*"));
 114 
 115         BOOST_CHECK(!utils::wildcard_string_match(str, "foo? "));
 116         BOOST_CHECK(!utils::wildcard_string_match(str, "?foo"));
 117 
 118         std::string superfluous_mask;
 119 
 120         superfluous_mask = std::string(str.length(), '?');
 121         BOOST_CHECK(utils::wildcard_string_match(str, superfluous_mask));
 122         BOOST_CHECK(utils::wildcard_string_match(str, superfluous_mask + '?'));
 123 
 124         superfluous_mask = std::string(str.length(), '*');
 125         BOOST_CHECK(utils::wildcard_string_match(str, superfluous_mask));
 126         BOOST_CHECK(utils::wildcard_string_match(str, superfluous_mask + '*'));
 127 
 128         BOOST_CHECK(utils::wildcard_string_match("", ""));
 129         BOOST_CHECK(!utils::wildcard_string_match(str, ""));
 130 
 131         BOOST_CHECK(utils::wildcard_string_match("", "*"));
 132         BOOST_CHECK(utils::wildcard_string_match("", "***?**"));
 133         BOOST_CHECK(!utils::wildcard_string_match("", "?"));
 134         BOOST_CHECK(!utils::wildcard_string_match("", "???"));
 135 }
