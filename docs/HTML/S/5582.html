   2 # Note: As an Orcish Grunt without traits, Bob starts with 38 hitpoints.
   3 # Thus, the first test sets his hitpoints to 19, and the second test
   4 # sets his level to 2.
   5 
   6 {GENERIC_UNIT_TEST filter_this_unit_wml (
   7         [event]
   8                 name=prestart
   9                 [modify_unit]
  10                         [filter]
  11                                 id=bob
  12                         [/filter]
  13                         hitpoints="$($this_unit.max_hitpoints / 2)"
  14                 [/modify_unit]
  15                 {RETURN (
  16                         [have_unit]
  17                                 id=bob
  18                                 [filter_wml]
  19                                         hitpoints="$($this_unit.max_hitpoints / 2)"
  20                                 [/filter_wml]
  21                         [/have_unit]
  22                 )}
  23         [/event]
  24 )}
