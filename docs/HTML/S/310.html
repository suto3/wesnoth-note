 100 bool terrain_filter::match_internal(const map_location& loc, const bool ignore_xy) const
 101 {
 102         if (!this->fc_->get_disp_context().map().on_board_with_border(loc)) {
 103                 return false;
 104         }
 105 
 106         std::string lua_function = cfg_["lua_function"];
 107         if (!lua_function.empty() && fc_->get_lua_kernel()) {
 108                 if (!fc_->get_lua_kernel()->run_filter(lua_function.c_str(), loc)) {
 109                         return false;
 110                 }
 111         }
 112 
 113         //Filter Areas
 114         if (cfg_.has_attribute("area") &&
 115                 fc_->get_tod_man().get_area_by_id(cfg_["area"]).count(loc) == 0)
 116                 return false;
 117 
 118         if(cfg_.has_attribute("terrain")) {
 119                 if(cache_.parsed_terrain == nullptr) {
 120                         cache_.parsed_terrain = new t_translation::ter_match(cfg_["terrain"]);
 121                 }
 122                 if(!cache_.parsed_terrain->is_empty) {
 123                         const t_translation::terrain_code letter = fc_->get_disp_context().map().get_terrain_info(loc).number();
 124                         if(!t_translation::terrain_matches(letter, *cache_.parsed_terrain)) {
 125                                 return false;
 126                         }
 127                 }
 128         }
 129 
 130         //Allow filtering on location ranges
 131         if (!ignore_xy) {
 132                 if (!loc.matches_range(cfg_["x"], cfg_["y"])) {
 133                         return false;
 134                 }
 135                 //allow filtering by searching a stored variable of locations
 136                 if (cfg_.has_attribute("find_in")) {
 137                         if (const game_data * gd = fc_->get_game_data()) {
 138                                 try
 139                                 {
 140                                         variable_access_const vi = gd->get_variable_access_read(cfg_["find_in"]);
 141 
 142                                         bool found = false;
 143                                         for (const config &cfg : vi.as_array()) {
 144                                                 if (map_location(cfg, nullptr) == loc) {
 145                                                         found = true;
 146                                                         break;
 147                                                 }
 148                                         }
 149                                         if (!found) return false;
 150                                 }
 151                                 catch (const invalid_variablename_exception&)
 152                                 {
 153                                         return false;
 154                                 }
 155                         }
 156                 }
 157                 if (cfg_.has_attribute("location_id")) {
 158                         if (loc != fc_->get_disp_context().map().special_location(cfg_["location_id"])) {
 159                                 return false;
 160                         }
 161                 }
 162         }
 163         //Allow filtering on unit
 164         if(cfg_.has_child("filter")) {
 165                 const unit_map::const_iterator u = fc_->get_disp_context().units().find(loc);
 166                 if (!u.valid())
 167                         return false;
 168                 if (!cache_.ufilter_)
 169                         cache_.ufilter_.reset(new unit_filter(cfg_.child("filter"), fc_, flat_));
 170                 if (!cache_.ufilter_->matches(*u, loc))
 171                         return false;
 172         }
 173 
 174         // Allow filtering on visibility to a side
 175         if (cfg_.has_child("filter_vision")) {
 176                 const vconfig::child_list& vis_filt = cfg_.get_children("filter_vision");
 177                 vconfig::child_list::const_iterator i, i_end = vis_filt.end();
 178                 for (i = vis_filt.begin(); i != i_end; ++i) {
 179                         bool visible = (*i)["visible"].to_bool(true);
 180                         bool respect_fog = (*i)["respect_fog"].to_bool(true);
 181 
 182                         side_filter ssf(*i, fc_);
 183                         std::vector<int> sides = ssf.get_teams();
 184 
 185                         bool found = false;
 186                         for (const int side : sides) {
 187                                 const team &viewing_team = fc_->get_disp_context().get_team(side);
 188                                 bool viewer_sees = respect_fog ? !viewing_team.fogged(loc) : !viewing_team.shrouded(loc);
 189                                 if (visible == viewer_sees) {
 190                                         found = true;
 191                                         break;
 192                                 }
 193                         }
 194                         if (!found) {return false;}
 195                 }
 196         }
 197 
 198         //Allow filtering on adjacent locations
 199         if(cfg_.has_child("filter_adjacent_location")) {
 200                 map_location adjacent[6];
 201                 get_adjacent_tiles(loc, adjacent);
 202                 const vconfig::child_list& adj_cfgs = cfg_.get_children("filter_adjacent_location");
 203                 vconfig::child_list::const_iterator i, i_end, i_begin = adj_cfgs.begin();
 204                 for (i = i_begin, i_end = adj_cfgs.end(); i != i_end; ++i) {
 205                         int match_count = 0;
 206                         vconfig::child_list::difference_type index = i - i_begin;
 207                         std::vector<map_location::DIRECTION> dirs = (*i).has_attribute("adjacent")
 208                                 ? map_location::parse_directions((*i)["adjacent"]) : map_location::default_dirs();
 209                         std::vector<map_location::DIRECTION>::const_iterator j, j_end = dirs.end();
 210                         for (j = dirs.begin(); j != j_end; ++j) {
 211                                 map_location &adj = adjacent[*j];
 212                                 if (fc_->get_disp_context().map().on_board(adj)) {
 213                                         if(cache_.adjacent_matches == nullptr) {
 214                                                 while(index >= std::distance(cache_.adjacent_match_cache.begin(), cache_.adjacent_match_cache.end())) {
 215                                                         const vconfig& adj_cfg = adj_cfgs[cache_.adjacent_match_cache.size()];
 216                                                         std::pair<terrain_filter, std::map<map_location,bool> > amc_pair(
 217                                                                 terrain_filter(adj_cfg, *this),
 218                                                                 std::map<map_location,bool>());
 219                                                         cache_.adjacent_match_cache.push_back(amc_pair);
 220                                                 }
 221                                                 terrain_filter &amc_filter = cache_.adjacent_match_cache[index].first;
 222                                                 std::map<map_location,bool> &amc = cache_.adjacent_match_cache[index].second;
 223                                                 std::map<map_location,bool>::iterator lookup = amc.find(adj);
 224                                                 if(lookup == amc.end()) {
 225                                                         if(amc_filter(adj)) {
 226                                                                 amc[adj] = true;
 227                                                                 ++match_count;
 228                                                         } else {
 229                                                                 amc[adj] = false;
 230                                                         }
 231                                                 } else if(lookup->second) {
 232                                                         ++match_count;
 233                                                 }
 234                                         } else {
 235                                                 assert(index < std::distance(cache_.adjacent_matches->begin(), cache_.adjacent_matches->end()));
 236                                                 std::set<map_location> &amc = (*cache_.adjacent_matches)[index];
 237                                                 if(amc.find(adj) != amc.end()) {
 238                                                         ++match_count;
 239                                                 }
 240                                         }
 241                                 }
 242                         }
 243                         static std::vector<std::pair<int,int> > default_counts = utils::parse_ranges("1-6");
 244                         std::vector<std::pair<int,int> > counts = (*i).has_attribute("count")
 245                                 ? utils::parse_ranges((*i)["count"]) : default_counts;
 246                         if(!in_ranges(match_count, counts)) {
 247                                 return false;
 248                         }
 249                 }
 250         }
 251 
 252         const t_string& t_tod_type = cfg_["time_of_day"];
 253         const t_string& t_tod_id = cfg_["time_of_day_id"];
 254         const std::string& tod_type = t_tod_type;
 255         const std::string& tod_id = t_tod_id;
 256         if(!tod_type.empty() || !tod_id.empty()) {
 257                 // creating a time_of_day is expensive, only do it if we will use it
 258                 time_of_day tod;
 259 
 260                 if(flat_) {
 261                         tod = fc_->get_tod_man().get_time_of_day(loc);
 262                 } else {
 263                         tod = fc_->get_tod_man().get_illuminated_time_of_day(fc_->get_disp_context().units(), fc_->get_disp_context().map(),loc);
 264                 }
 265 
 266                 if(!tod_type.empty()) {
 267                         const std::vector<std::string>& vals = utils::split(tod_type);
 268                         if(tod.lawful_bonus<0) {
 269                                 if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::CHAOTIC)) == vals.end()) {
 270                                         return false;
 271                                 }
 272                         } else if(tod.lawful_bonus>0) {
 273                                 if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::LAWFUL)) == vals.end()) {
 274                                         return false;
 275                                 }
 276                         } else if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::NEUTRAL)) == vals.end()) {
 277                                 return false;
 278                         }
 279                 }
 280 
 281                 if(!tod_id.empty()) {
 282                         if(tod_id != tod.id) {
 283                                 if(std::find(tod_id.begin(),tod_id.end(),',') != tod_id.end() &&
 284                                         std::search(tod_id.begin(),tod_id.end(),
 285                                         tod.id.begin(),tod.id.end()) != tod_id.end()) {
 286                                         const std::vector<std::string>& vals = utils::split(tod_id);
 287                                         if(std::find(vals.begin(),vals.end(),tod.id) == vals.end()) {
 288                                                 return false;
 289                                         }
 290                                 } else {
 291                                         return false;
 292                                 }
 293                         }
 294                 }
 295         }
 296 
 297         //allow filtering on owner (for villages)
 298         const config::attribute_value &owner_side = cfg_["owner_side"];
 299         const vconfig& filter_owner = cfg_.child("filter_owner");
 300         if(!filter_owner.null()) {
 301                 if(!owner_side.empty()) {
 302                         WRN_NG << "duplicate side information in a SLF, ignoring inline owner_side=" << std::endl;
 303                 }
 304                 if(!fc_->get_disp_context().map().is_village(loc))
 305                         return false;
 306                 side_filter ssf(filter_owner, fc_);
 307                 const std::vector<int>& sides = ssf.get_teams();
 308                 bool found = false;
 309                 if(sides.empty() && fc_->get_disp_context().village_owner(loc) == -1)
 310                         found = true;
 311                 for(const int side : sides) {
 312                         if(fc_->get_disp_context().get_team(side).owns_village(loc)) {
 313                                 found = true;
 314                                 break;
 315                         }
 316                 }
 317                 if(!found)
 318                         return false;
 319         }
 320         else if(!owner_side.empty()) {
 321                 const int side_index = owner_side.to_int(0) - 1;
 322                 if(fc_->get_disp_context().village_owner(loc) != side_index) {
 323                         return false;
 324                 }
 325         }
 326 
 327         if(cfg_.has_attribute("formula")) {
 328                 try {
 329                         const gamemap& map = fc_->get_disp_context().map();
 330                         t_translation::terrain_code t = map.get_terrain(loc);
 331                         const terrain_type& ter = map.tdata()->get_terrain_info(t);
 332                         const wfl::terrain_callable callable(ter,loc);
 333                         const wfl::formula form(cfg_["formula"]);
 334                         if(!form.evaluate(callable).as_bool()) {
 335                                 return false;
 336                         }
 337                         return true;
 338                 } catch(wfl::formula_error& e) {
 339                         lg::wml_error() << "Formula error in location filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 340                         // Formulae with syntax errors match nothing
 341                         return false;
 342                 }
 343         }
 344 
 345         return true;
 346 }

 348 bool terrain_filter::match(const map_location& loc) const
 349 {
 350         if(cfg_["x"] == "recall" && cfg_["y"] == "recall") {
 351                 return !fc_->get_disp_context().map().on_board(loc);
 352         }
 353         std::set<map_location> hexes;
 354         std::vector<map_location> loc_vec(1, loc);
 355 
 356         //handle radius
 357         size_t radius = cfg_["radius"].to_size_t(0);
 358         if(radius > max_loop_) {
 359                 ERR_NG << "terrain_filter: radius greater than " << max_loop_
 360                 << ", restricting\n";
 361                 radius = max_loop_;
 362         }
 363         if ( radius == 0 )
 364                 hexes.insert(loc_vec.begin(), loc_vec.end());
 365         else if ( cfg_.has_child("filter_radius") ) {
 366                 terrain_filter r_filter(cfg_.child("filter_radius"), *this);
 367                 get_tiles_radius(fc_->get_disp_context().map(), loc_vec, radius, hexes, false, r_filter);
 368         } else {
 369                 get_tiles_radius(fc_->get_disp_context().map(), loc_vec, radius, hexes);
 370         }
 371 
 372         size_t loop_count = 0;
 373         std::set<map_location>::const_iterator i;
 374         for(i = hexes.begin(); i != hexes.end(); ++i) {
 375                 bool matches = match_internal(*i, false);
 376 
 377                 //handle [and], [or], and [not] with in-order precedence
 378                 vconfig::all_children_iterator cond = cfg_.ordered_begin();
 379                 vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 380                 while(cond != cond_end)
 381                 {
 382                         const std::string& cond_name = cond.get_key();
 383                         const vconfig& cond_cfg = cond.get_child();
 384 
 385                         //handle [and]
 386                         if(cond_name == "and")
 387                         {
 388                                 matches = matches && terrain_filter(cond_cfg, *this)(*i);
 389                         }
 390                         //handle [or]
 391                         else if(cond_name == "or")
 392                         {
 393                                 matches = matches || terrain_filter(cond_cfg, *this)(*i);
 394                         }
 395                         //handle [not]
 396                         else if(cond_name == "not")
 397                         {
 398                                 matches = matches && !terrain_filter(cond_cfg, *this)(*i);
 399                         }
 400                         ++cond;
 401                 }
 402                 if(matches) {
 403                         return true;
 404                 }
 405                 if(++loop_count > max_loop_) {
 406                         std::set<map_location>::const_iterator temp = i;
 407                         if(++temp != hexes.end()) {
 408                                 ERR_NG << "terrain_filter: loop count greater than " << max_loop_
 409                                 << ", aborting\n";
 410                                 break;
 411                         }
 412                 }
 413         }
 414         return false;
 415 }

 419 public:
 420         using location_set = std::set<map_location>;
 421         struct no_start_set_yet {};

 427         template<typename T, typename F1, typename F2, typename F3>
 428         static void filter_final(T&& src, location_set& dest, const terrain_filter&, const F1& f1, const F2& f2, const F3& f3)
 429         {
 430                 for (const map_location &loc : src) {
 431                         if (f1(loc) && f2(loc) && f3(loc)) {
 432                                 dest.insert(loc);
 433                         }
 434                 }
 435         }
 436 
 437         template<typename T, typename F1, typename F2>
 438         static void filter_special_loc(T&& src, location_set& dest, const terrain_filter& filter, const F1& f1, const F2& f2)
 439         {
 440                 if (filter.cfg_.has_attribute("location_id")) {
 441                         map_location loc2 = filter.fc_->get_disp_context().map().special_location(filter.cfg_["location_id"]);
 442                         filter_final(src, dest, filter, f1, f2, [loc2](const map_location& loc) { return loc == loc2; });
 443                 }
 444                 else {
 445                         filter_final(src, dest, filter, f1, f2, no_filter());
 446                 }
 447         }
 448 
 449         template<typename T, typename F1>
 450         static void filter_area(T&& src, location_set& dest, const terrain_filter& filter, const F1& f1)
 451         {
 452                 if (filter.cfg_.has_attribute("area")) {
 453                         const std::set<map_location>& area = filter.fc_->get_tod_man().get_area_by_id(filter.cfg_["area"]);
 454                         filter_special_loc(src, dest, filter, f1, [&area](const map_location& loc) { return area.find(loc) != area.end(); });
 455                 }
 456                 else {
 457                         filter_special_loc(src, dest, filter, f1, no_filter());
 458                 }
 459         }
 460 
 461         template<typename T>
 462         static void filter_xy(T&& src, location_set& dest, const terrain_filter& filter, bool with_border)
 463         {
 464                 if (filter.cfg_.has_attribute("x") || filter.cfg_.has_attribute("y")) {
 465                         std::vector<map_location> xy_vector = filter.fc_->get_disp_context().map().parse_location_range(filter.cfg_["x"], filter.cfg_["y"], with_border);
 466                         filter_area(src, dest, filter, [&xy_vector](const map_location& loc) { return std::find(xy_vector.begin(), xy_vector.end(), loc) != xy_vector.end(); });
 467                 }
 468                 else {
 469                         filter_area(src, dest, filter, no_filter());
 470                 }
 471         }
 472 };

 479 void terrain_filter::get_locations(std::set<map_location>& locs, bool with_border) const
 480 {
 481         std::set<map_location> match_set;
 482 
 483         // See if the caller provided an override to with_border
 484         with_border = cfg_["include_borders"].to_bool(with_border);
 485 
 486         if (cfg_.has_attribute("find_in")) {
 487 
 488                 if (const game_data * gd = fc_->get_game_data()) {
 489                         try
 490                         {
 491                                 auto ar = gd->get_variable_access_read(cfg_["find_in"]).as_array();
 492                                 terrain_filterimpl::filter_xy(ar | boost::adaptors::transformed(cfg_to_loc()), match_set, *this, with_border);
 493                         }
 494                         catch (const invalid_variablename_exception&)
 495                         {
 496                                 //Do nothing
 497                         }
 498                 }
 499         }
 500         else if (cfg_.has_attribute("x") || cfg_.has_attribute("y")) {
 501                 std::vector<map_location> xy_vector = fc_->get_disp_context().map().parse_location_range(cfg_["x"], cfg_["y"], with_border);
 502                 terrain_filterimpl::filter_area(xy_vector, match_set, *this, terrain_filterimpl::no_filter());
 503         }
 504         else if (cfg_.has_attribute("area")) {
 505                 const std::set<map_location>& area = fc_->get_tod_man().get_area_by_id(cfg_["area"]);
 506                 terrain_filterimpl::filter_special_loc(area, match_set, *this, terrain_filterimpl::no_filter(), terrain_filterimpl::no_filter());
 507         }
 508         else if (cfg_.has_attribute("location_id")) {
 509                 map_location loc2 = fc_->get_disp_context().map().special_location(cfg_["location_id"]);
 510                 if (loc2.valid()) {
 511                         match_set.insert(loc2);
 512                 }
 513         }
 514         else {
 515                 //consider all locations on the map
 516                 int bs = fc_->get_disp_context().map().border_size();
 517                 int w = with_border ? fc_->get_disp_context().map().w() + bs : fc_->get_disp_context().map().w();
 518                 int h = with_border ? fc_->get_disp_context().map().h() + bs : fc_->get_disp_context().map().h();
 519                 for (int x = with_border ? 0 - bs : 0; x < w; ++x) {
 520                         for (int y = with_border ? 0 - bs : 0; y < h; ++y) {
 521                                 match_set.insert(map_location(x, y));
 522                         }
 523                 }
 524         }
 525 
 526         //handle location filter
 527         if(cfg_.has_child("filter_adjacent_location")) {
 528                 if(cache_.adjacent_matches == nullptr) {
 529                         cache_.adjacent_matches = new std::vector<std::set<map_location> >();
 530                 }
 531                 const vconfig::child_list& adj_cfgs = cfg_.get_children("filter_adjacent_location");
 532                 for (unsigned i = 0; i < adj_cfgs.size(); ++i) {
 533                         std::set<map_location> adj_set;
 534                         /* GCC-3.3 doesn't like operator[] so use at(), which has the same result */
 535                         terrain_filter(adj_cfgs.at(i), *this).get_locations(adj_set, with_border);
 536                         cache_.adjacent_matches->push_back(adj_set);
 537                         if(i >= max_loop_ && i+1 < adj_cfgs.size()) {
 538                                 ERR_NG << "terrain_filter: loop count greater than " << max_loop_
 539                                 << ", aborting\n";
 540                                 break;
 541                         }
 542                 }
 543         }
 544         std::set<map_location>::iterator loc_itor = match_set.begin();
 545         while(loc_itor != match_set.end()) {
 546                 if(match_internal(*loc_itor, true)) {
 547                         ++loc_itor;
 548                 } else {
 549                         match_set.erase(loc_itor++);
 550                 }
 551         }
 552 
 553         //handle [and], [or], and [not] with in-order precedence
 554         vconfig::all_children_iterator cond = cfg_.ordered_begin();
 555         vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 556         int ors_left = std::count_if(cond, cond_end, cfg_isor());
 557         while(cond != cond_end)
 558         {
 559                 //if there are no locations or [or] conditions left, go ahead and return empty
 560                 if(match_set.empty() && ors_left <= 0) {
 561                         return;
 562                 }
 563 
 564                 const std::string& cond_name = cond.get_key();
 565                 const vconfig& cond_cfg = cond.get_child();
 566 
 567                 //handle [and]
 568                 if(cond_name == "and") {
 569                         std::set<map_location> intersect_hexes;
 570                         terrain_filter(cond_cfg, *this).get_locations(intersect_hexes, with_border);
 571                         std::set<map_location>::iterator intersect_itor = match_set.begin();
 572                         while(intersect_itor != match_set.end()) {
 573                                 if(intersect_hexes.find(*intersect_itor) == intersect_hexes.end()) {
 574                                         match_set.erase(*intersect_itor++);
 575                                 } else {
 576                                         ++intersect_itor;
 577                                 }
 578                         }
 579                 }
 580                 //handle [or]
 581                 else if(cond_name == "or") {
 582                         std::set<map_location> union_hexes;
 583                         terrain_filter(cond_cfg, *this).get_locations(union_hexes, with_border);
 584                         //match_set.insert(union_hexes.begin(), union_hexes.end()); //doesn't compile on MSVC
 585                         std::set<map_location>::iterator insert_itor = union_hexes.begin();
 586                         while(insert_itor != union_hexes.end()) {
 587                                 match_set.insert(*insert_itor++);
 588                         }
 589                         --ors_left;
 590                 }
 591                 //handle [not]
 592                 else if(cond_name == "not") {
 593                         std::set<map_location> removal_hexes;
 594                         terrain_filter(cond_cfg, *this).get_locations(removal_hexes, with_border);
 595                         std::set<map_location>::iterator erase_itor = removal_hexes.begin();
 596                         while(erase_itor != removal_hexes.end()) {
 597                                 match_set.erase(*erase_itor++);
 598                         }
 599                 }
 600                 ++cond;
 601         }
 602         if(match_set.empty()) {
 603                 return;
 604         }
 605 
 606         //handle radius
 607         size_t radius = cfg_["radius"].to_size_t(0);
 608         if(radius > max_loop_) {
 609                 ERR_NG << "terrain_filter: radius greater than " << max_loop_
 610                 << ", restricting\n";
 611                 radius = max_loop_;
 612         }
 613         if(radius > 0) {
 614                 std::vector<map_location> xy_vector (match_set.begin(), match_set.end());
 615                 if(cfg_.has_child("filter_radius")) {
 616                         terrain_filter r_filter(cfg_.child("filter_radius"), *this);
 617                         get_tiles_radius(fc_->get_disp_context().map(), xy_vector, radius, locs, with_border, r_filter);
 618                 } else {
 619                         get_tiles_radius(fc_->get_disp_context().map(), xy_vector, radius, locs, with_border);
 620                 }
 621         } else {
 622                 locs.insert(match_set.begin(), match_set.end());
 623         }
 624 }

 627 {
 628         return cfg_.get_config();
 629 }
