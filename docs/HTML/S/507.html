  74 window* build(CVideo& video, const builder_window::window_resolution* definition)
  75 {
  76         // We set the values from the definition since we can only determine the
  77         // best size (if needed) after all widgets have been placed.
  78         window* win = new window(video,
  79                                                                   definition->x,
  80                                                                   definition->y,
  81                                                                   definition->width,
  82                                                                   definition->height,
  83                                                                   definition->reevaluate_best_size,
  84                                                                   definition->functions,
  85                                                                   definition->automatic_placement,
  86                                                                   definition->horizontal_placement,
  87                                                                   definition->vertical_placement,
  88                                                                   definition->maximum_width,
  89                                                                   definition->maximum_height,
  90                                                                   definition->definition,
  91                                                                   definition->tooltip,
  92                                                                   definition->helptip);
  93         assert(win);
  94 
  95         for(const auto & lg : definition->linked_groups)
  96         {
  97 
  98                 if(win->has_linked_size_group(lg.id)) {
  99                         t_string msg = vgettext("Linked '$id' group has multiple definitions.", {{"id", lg.id}});
 100 
 101                         FAIL(msg);
 102                 }
 103 
 104                 win->init_linked_size_group(lg.id, lg.fixed_width, lg.fixed_height);
 105         }
 106 
 107         win->set_click_dismiss(definition->click_dismiss);
 108 
 109         std::shared_ptr<const window_definition::resolution>
 110         conf = std::static_pointer_cast<const window_definition::resolution>(
 111                         win->config());
 112         assert(conf);
 113 
 114         if(conf->grid) {
 115                 win->init_grid(conf->grid);
 116                 win->finalize(definition->grid);
 117         } else {
 118                 win->init_grid(definition->grid);
 119         }
 120 
 121         win->add_to_keyboard_chain(win);
 122 
 123         return win;
 124 }

 150 builder_widget_ptr create_builder_widget(const config& cfg)
 151 {
 152         config::const_all_children_itors children = cfg.all_children_range();
 153         VALIDATE(children.size() == 1, "Grid cell does not have exactly 1 child.");
 154 
 155         for(const auto & item : builder_widget_lookup())
 156         {
 157                 if(item.first == "window" || item.first == "tooltip") {
 158                         continue;
 159                 }
 160                 if(const config& c = cfg.child(item.first)) {
 161                         return item.second(c);
 162                 }
 163         }
 164 
 165         if(const config& c = cfg.child("grid")) {
 166                 return std::make_shared<builder_grid>(c);
 167         }
 168 
 169         if(const config& instance = cfg.child("instance")) {
 170                 return std::make_shared<implementation::builder_instance>(instance);
 171         }
 172 
 173         if(const config& pane = cfg.child("pane")) {
 174                 return std::make_shared<implementation::builder_pane>(pane);
 175         }
 176 
 177         if(const config& viewport = cfg.child("viewport")) {
 178                 return std::make_shared<implementation::builder_viewport>(viewport);
 179         }

 254 const std::string& builder_window::read(const config& cfg)
 255 {
 256         id_ = cfg["id"].str();
 257         description_ = cfg["description"].str();
 258 
 259         VALIDATE(!id_.empty(), missing_mandatory_wml_key("window", "id"));
 260         VALIDATE(!description_.empty(),
 261                          missing_mandatory_wml_key("window", "description"));
 262 
 263         DBG_GUI_P << "Window builder: reading data for window " << id_ << ".\n";
 264 
 265         config::const_child_itors cfgs = cfg.child_range("resolution");
 266         VALIDATE(!cfgs.empty(), _("No resolution defined."));
 267         for(const auto & i : cfgs)
 268         {
 269                 resolutions.emplace_back(i);
 270         }
 271 
 272         return id_;
 273 }

 379 builder_window::window_resolution::window_resolution(const config& cfg)
 380         : window_width(cfg["window_width"])
 381         , window_height(cfg["window_height"])
 382         , automatic_placement(cfg["automatic_placement"].to_bool(true))
 383         , x(cfg["x"])
 384         , y(cfg["y"])
 385         , width(cfg["width"])
 386         , height(cfg["height"])
 387         , reevaluate_best_size(cfg["reevaluate_best_size"])
 388         , functions()
 389         , vertical_placement(implementation::get_v_align(cfg["vertical_placement"]))
 390         , horizontal_placement(
 391                           implementation::get_h_align(cfg["horizontal_placement"]))
 392         , maximum_width(cfg["maximum_width"])
 393         , maximum_height(cfg["maximum_height"])
 394         , click_dismiss(cfg["click_dismiss"].to_bool())
 395         , definition(cfg["definition"])
 396         , linked_groups()
 397         , tooltip(cfg.child_or_empty("tooltip"), "tooltip")
 398         , helptip(cfg.child_or_empty("helptip"), "helptip")
 399         , grid(0)
 400 {
 401         if(!cfg["functions"].empty()) {
 402                 wfl::formula(cfg["functions"], &functions).evaluate();
 403         }
 404 
 405         const config& c = cfg.child("grid");
 406 
 407         VALIDATE(c, _("No grid defined."));
 408 
 409         grid = std::make_shared<builder_grid>(builder_grid(c));
 410 
 411         if(!automatic_placement) {
 412                 VALIDATE(width.has_formula() || width(),
 413                                  missing_mandatory_wml_key("resolution", "width"));
 414                 VALIDATE(height.has_formula() || height(),
 415                                  missing_mandatory_wml_key("resolution", "height"));
 416         }
 417 
 418         DBG_GUI_P << "Window builder: parsing resolution " << window_width << ','
 419                           << window_height << '\n';
 420 
 421         if(definition.empty()) {
 422                 definition = "default";
 423         }
 424 
 425         linked_groups = parse_linked_group_definitions(cfg);
 426 }

 442  *                                      used that often, but is allowed. $
 443  *     linked_group & string & 0 &       $
 444  * @end{table}
 445  *
 446  * Every grid cell has some cell configuration values and one widget in the grid
 447  * cell. Here we describe the what is available more information about the usage
 448  * can be found here [[GUILayout]].
 449  *
 450  * == Row values ==
 451  * @begin{tag}{name="row"}{min="0"}{max="-1"}
 452  * For every row the following variables are available:
 453  *
 454  * @begin{table}{config}
 455  *     grow_factor & unsigned & 0 &      The grow factor for a row. $
 456  * @end{table}
 457  *
 458  * == Cell values ==
 459  * @begin{tag}{name="column"}{min="0"}{max="-1"}
 460  * @allow{link}{name="gui/window/resolution/grid"}
 461  * For every column the following variables are available:
 462  * @begin{table}{config}
 463  *     grow_factor & unsigned & 0 &    The grow factor for a column, this
 464  *                                     value is only read for the first row. $
 465  *
 466  *     border_size & unsigned & 0 &    The border size for this grid cell. $
 467  *     border & border & "" &          Where to place the border in this grid
 468  *                                     cell. $
 469  *
 470  *     vertical_alignment & v_align & "" &
 471  *                                     The vertical alignment of the widget in
 472  *                                     the grid cell. (This value is ignored if
 473  *                                     vertical_grow is true.) $
 474  *     horizontal_alignment & h_align & "" &
 475  *                                     The horizontal alignment of the widget in
 476  *                                     the grid cell.(This value is ignored if
 477  *                                     horizontal_grow is true.) $
 478  *
 479  *     vertical_grow & bool & false &    Does the widget grow in vertical
 480  *                                     direction when the grid cell grows in the
 481  *                                     vertical direction. This is used if the
 482  *                                     grid cell is wider as the best width for
 483  *                                     the widget. $
 484  *     horizontal_grow & bool & false &  Does the widget grow in horizontal
 485  *                                     direction when the grid cell grows in the
 486  *                                     horizontal direction. This is used if the
 487  *                                     grid cell is higher as the best width for
 488  *                                     the widget. $
 489  * @end{table}
 490  * @end{tag}{name="column"}
 491  * @end{tag}{name="row"}
 492  * @end{tag}{name="grid"}
 493  * @end{parent}{name="gui/window/resolution/"}
 494  *
 495  */
 496 builder_grid::builder_grid(const config& cfg)
 497         : builder_widget(cfg)
 498         , rows(0)
 499         , cols(0)
 500         , row_grow_factor()
 501         , col_grow_factor()
 502         , flags()
 503         , border_size()
 504         , widgets()
 505 {
 506         log_scope2(log_gui_parse, "Window builder: parsing a grid");
 507 
 508         for(const auto & row : cfg.child_range("row"))
 509         {
 510                 unsigned col = 0;
 511 
 512                 row_grow_factor.push_back(row["grow_factor"]);
 513 
 514                 for(const auto & c : row.child_range("column"))
 515                 {
 516                         flags.push_back(implementation::read_flags(c));
 517                         border_size.push_back(c["border_size"]);
 518                         if(rows == 0) {
 519                                 col_grow_factor.push_back(c["grow_factor"]);
 520                         }
 521 
 522                         widgets.push_back(create_builder_widget(c));
 523 
 524                         ++col;
 525                 }
 526 
 527                 ++rows;
 528                 if(rows == 1) {
 529                         cols = col;
 530                 } else {
 531                         VALIDATE(col, _("A row must have a column."));
 532                         VALIDATE(col == cols, _("Number of columns differ."));
 533                 }
 534         }
 535 
 536         DBG_GUI_P << "Window builder: grid has " << rows << " rows and " << cols
 537                           << " columns.\n";
 538 }

 552 grid* builder_grid::build(grid* grid) const
 553 {
 554         grid->set_id(id);
 555         grid->set_linked_group(linked_group);
 556         grid->set_rows_cols(rows, cols);
 557 
 558         log_scope2(log_gui_general, "Window builder: building grid");
 559 
 560         DBG_GUI_G << "Window builder: grid '" << id << "' has " << rows
 561                           << " rows and " << cols << " columns.\n";
 562 
 563         for(unsigned x = 0; x < rows; ++x) {
 564                 grid->set_row_grow_factor(x, row_grow_factor[x]);
 565                 for(unsigned y = 0; y < cols; ++y) {
 566 
 567                         if(x == 0) {
 568                                 grid->set_column_grow_factor(y, col_grow_factor[y]);
 569                         }
 570 
 571                         DBG_GUI_G << "Window builder: adding child at " << x << ',' << y
 572                                           << ".\n";
 573 
 574                         widget* widget = widgets[x * cols + y]->build();
 575                         grid->set_child(widget,
 576                                                         x,
 577                                                         y,
 578                                                         flags[x * cols + y],
 579                                                         border_size[x * cols + y]);
 580                 }
 581         }
 582 
 583         return grid;
 584 }
