  38 /**
  39  * Helper to implement private functions without modifying the header.
  40  *
  41  * The class is a helper to avoid recompilation and only has static
  42  * functions. It also facilitates to create duplicates of functions for a const
  43  * and a non-const member function.
  44  */
  45 struct pane_implementation
  46 {
  47         /**
  48          * Implementation for the wrappers for
  49          * [const] widget* pane::find_at(const point&, const bool) [const].
  50          *
  51          * @tparam W                  A pointer to the pane.
  52          */
  53         template <class W>
  54         static utils::const_clone_ptr<widget, W>
  55         find_at(W pane, point coordinate, const bool must_be_active)
  56         {
  57 
  58                 /*
  59                  * First test whether the mouse is at the pane.
  60                  */
  61                 if(pane->widget::find_at(coordinate, must_be_active) != pane) {
  62                         return nullptr;
  63                 }
  64 
  65                 for(auto item : pane->items_)
  66                 {
  67 
  68                         if(item.item_grid->get_visible() == widget::visibility::invisible) {
  69                                 continue;
  70                         }
  71 
  72                         /*
  73                          * If the adjusted coordinate is in the item's grid let the grid
  74                          * resolve the coordinate.
  75                          */
  76                         const SDL_Rect rect = item.item_grid->get_rectangle();
  77                         if(coordinate.x >= rect.x && coordinate.y >= rect.y
  78                            && coordinate.x < rect.x + rect.w
  79                            && coordinate.y < rect.y + rect.h) {
  80 
  81                                 return item.item_grid->find_at(coordinate, must_be_active);
  82                         }
  83                 }
  84 
  85                 return nullptr;
  86         }
  87 
  88         /**
  89          * Implementation for the wrappers for
  90          * [const] grid* pane::grid(const unsigned id) [const].
  91          *
  92          * @tparam W                  A pointer to the pane.
  93          */
  94         template <class W>
  95         static utils::const_clone_ptr<grid, W>
  96         get_grid(W pane, const unsigned id)
  97         {
  98                 for(auto item : pane->items_)
  99                 {
 100 
 101                         if(item.id == id) {
 102                                 return item.item_grid;
 103                         }
 104                 }
 105 
 106                 return nullptr;
 107         }
 108 };

 141 unsigned pane::create_item(const std::map<std::string, string_map>& item_data,
 142                                                         const std::map<std::string, std::string>& tags)
 143 {
 144         item item = { item_id_generator_++, tags, item_builder_->build() };
 145 
 146         item.item_grid->set_parent(this);
 147 
 148         for(const auto & data : item_data)
 149         {
 150                 styled_widget* control
 151                                 = find_widget<styled_widget>(item.item_grid, data.first, false, false);
 152 
 153                 if(control) {
 154                         control->set_members(data.second);
 155                 }
 156         }
 157 
 158         items_.push_back(item);
 159 
 160         event::message message;
 161         fire(event::REQUEST_PLACEMENT, *this, message);
 162 
 163         return item.id;
 164 }

 277 void pane::place_children()
 278 {
 279         prepare_placement();
 280         unsigned index = 0;
 281         for(auto & item : items_)
 282         {
 283                 if(item.item_grid->get_visible() == widget::visibility::invisible) {
 284                         continue;
 285                 }
 286 
 287                 const point origin = placer_->get_origin(index);
 288                 item.item_grid->place(origin, item.item_grid->get_best_size());
 289                 ++index;
 290         }
 291 }

 309 void pane::place_or_set_origin_children()
 310 {
 311         prepare_placement();
 312         unsigned index = 0;
 313         for(auto & item : items_)
 314         {
 315                 if(item.item_grid->get_visible() == widget::visibility::invisible) {
 316                         continue;
 317                 }
 318 
 319                 const point origin = placer_->get_origin(index);
 320                 if(item.item_grid->get_size() != item.item_grid->get_best_size()) {
 321                         item.item_grid->place(origin, item.item_grid->get_best_size());
 322                 } else {
 323                         item.item_grid->set_origin(origin);
 324                 }
 325                 ++index;
 326         }
 327 }

 329 void pane::prepare_placement() const
 330 {
 331         assert(placer_.get());
 332         placer_->initialize();
 333 
 334         for(const auto & item : items_)
 335         {
 336                 if(item.item_grid->get_visible() == widget::visibility::invisible) {
 337                         continue;
 338                 }
 339 
 340                 placer_->add_item(item.item_grid->get_best_size());
 341         }
 342 }

 344 void pane::signal_handler_request_placement(dispatcher& dispatcher,
 345                                                                                          const event::ui_event event,
 346                                                                                          bool& handled)
 347 {
 348         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 349 
 350         widget* wgt = dynamic_cast<widget*>(&dispatcher);
 351         if(wgt) {
 352                 for(auto & item : items_)
 353                 {
 354                         if(item.item_grid->has_widget(*wgt)) {
 355                                 if(item.item_grid->get_visible() != widget::visibility::invisible) {
 356 
 357                                         /*
 358                                          * This time we call init layout but also the linked widget
 359                                          * update this makes things work properly for the
 360                                          * addon_list. This code can use some more tuning,
 361                                          * polishing and testing.
 362                                          */
 363                                         item.item_grid->layout_initialize(false);
 364                                         get_window()->layout_linked_widgets();
 365 
 366                                         /*
 367                                          * By not calling init layout it uses its previous size
 368                                          * what seems to work properly when showing and hiding
 369                                          * items. Might fail with new items (haven't tested yet).
 370                                          */
 371                                         item.item_grid->place(point(), item.item_grid->get_best_size());
 372                                 }
 373                                 place_or_set_origin_children();
 374                                 DBG_GUI_E << LOG_HEADER << ' ' << event << " handled.\n";
 375                                 handled = true;
 376                                 return;
 377                         }
 378                 }
 379         }
 380 
 381         DBG_GUI_E << LOG_HEADER << ' ' << event << " failed to handle.\n";
 382         assert(false);
 383         handled = false;
 384 }
