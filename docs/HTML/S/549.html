 101 void styled_widget::set_members(const string_map& data)
 102 {
 103         /** @todo document this feature on the wiki. */
 104         /** @todo do we need to add the debug colors here as well? */
 105         string_map::const_iterator itor = data.find("id");
 106         if(itor != data.end()) {
 107                 set_id(itor->second);
 108         }
 109 
 110         itor = data.find("linked_group");
 111         if(itor != data.end()) {
 112                 set_linked_group(itor->second);
 113         }
 114 
 115         itor = data.find("label");
 116         if(itor != data.end()) {
 117                 set_label(itor->second);
 118         }
 119 
 120         itor = data.find("tooltip");
 121         if(itor != data.end()) {
 122                 set_tooltip(itor->second);

 206 void styled_widget::request_reduce_width(const unsigned maximum_width)
 207 {
 208         assert(config_);
 209 
 210         if(!label_.empty() && can_wrap()) {
 211 
 212                 point size = get_best_text_size(
 213                                 point(), point(maximum_width - config_->text_extra_width, 0));
 214 
 215                 size.x += config_->text_extra_width;
 216                 size.y += config_->text_extra_height;
 217 
 218                 set_layout_size(size);
 219 
 220                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 221                                   << "' maximum_width " << maximum_width << " result " << size
 222                                   << ".\n";
 223 
 224         } else if(label_.empty() || text_can_shrink()) {
 225                 point size = get_best_size();
 226                 point min_size = get_config_minimum_size();
 227                 size.x = std::min(size.x, std::max<int>(maximum_width, min_size.x));
 228                 set_layout_size(size);
 229 
 230                 DBG_GUI_L << LOG_HEADER << " styled_widget " << id()
 231                           << " maximum_width " << maximum_width << " result " << size
 232                           << ".\n";
 233         } else {
 234                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 235                                   << "' failed; either no label or wrapping not allowed.\n";
 236         }
 237 }

 255 point styled_widget::calculate_best_size() const
 256 {
 257         assert(config_);
 258         if(label_.empty()) {
 259                 DBG_GUI_L << LOG_HEADER << " empty label return default.\n";
 260                 return get_config_default_size();
 261         }
 262 
 263         const point minimum = get_config_default_size();
 264         const point maximum = get_config_maximum_size();
 265 
 266         /**
 267          * @todo The value send should subtract the border size
 268          * and read it after calculation to get the proper result.
 269          */
 270         point result = get_best_text_size(minimum, maximum);
 271         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 272                           << "' result " << result << ".\n";
 273         return result;
 274 }

 276 void styled_widget::place(const point& origin, const point& size)
 277 {
 278         // resize canvasses
 279         for(auto & canvas : canvas_)
 280         {
 281                 canvas.set_width(size.x);
 282                 canvas.set_height(size.y);
 283         }
 284 
 285         // Note we assume that the best size has been queried but otherwise it
 286         // should return false.
 287         if(renderer_.is_truncated() && use_tooltip_on_label_overflow_
 288            && tooltip_.empty()) {
 289 
 290                 set_tooltip(label_);
 291         }
 292 
 293         // inherited
 294         widget::place(origin, size);
 295 
 296         // update the state of the canvas after the sizes have been set.
 297         update_canvas();
 298 }

 401 void styled_widget::update_canvas()
 402 {
 403         const int max_width = get_text_maximum_width();
 404         const int max_height = get_text_maximum_height();
 405 
 406         // set label in canvases
 407         for(auto & canvas : canvas_)
 408         {
 409                 canvas.set_variable("text", wfl::variant(label_));
 410                 canvas.set_variable("text_markup", wfl::variant(use_markup_));
 411                 canvas.set_variable("text_link_aware", wfl::variant(get_link_aware()));
 412                 // Possible TODO: Consider making a formula_callable for colours
 413                 color_t link_color = get_link_color();
 414                 std::vector<wfl::variant> link_color_as_list{wfl::variant(link_color.r), wfl::variant(link_color.g), wfl::variant(link_color.b), wfl::variant(link_color.a)};
 415                 canvas.set_variable("text_link_color", wfl::variant(link_color_as_list));
 416                 canvas.set_variable("text_alignment",
 417                                                         wfl::variant(encode_text_alignment(text_alignment_)));
 418                 canvas.set_variable("text_maximum_width", wfl::variant(max_width));
 419                 canvas.set_variable("text_maximum_height", wfl::variant(max_height));
 420                 canvas.set_variable("text_wrap_mode", wfl::variant(get_text_ellipse_mode()));
 421                 canvas.set_variable("text_characters_per_line",
 422                                                         wfl::variant(get_characters_per_line()));
 423         }
 424 }

 461 void styled_widget::definition_load_configuration(const std::string& control_type)
 462 {
 463         assert(!config());
 464 
 465         set_config(get_control(control_type, definition_));
 466         if(get_canvases().size() != config()->state.size())
 467         {
 468                 // TODO: Some widgets (toggle panel, toggle button) have a variable canvas count which is determined by its definition.
 469                 // I think we should remove the canvas_count from tcontrols constructor and always read it from the definition.
 470                 DBG_GUI_L << "Corrected canvas count to " << config()->state.size() << std::endl;
 471                 get_canvases() = std::vector<canvas>(config()->state.size());
 472         }
 473         for(size_t i = 0; i < get_canvases().size(); ++i) {
 474                 get_canvas(i) = config()->state[i].canvas_;
 475         }
 476 
 477         update_canvas();
 478 }

 480 point styled_widget::get_best_text_size(point minimum_size,
 481                                                                         point maximum_size) const
 482 {
 483         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 484 
 485         assert(!label_.empty());
 486 
 487         const point border(config_->text_extra_width, config_->text_extra_height);
 488         point size = minimum_size - border;
 489 
 490         renderer_.set_link_aware(get_link_aware())
 491                         .set_link_color(get_link_color());
 492 
 493         renderer_.set_text(label_, use_markup_);
 494 
 495         renderer_.set_family_class(config_->text_font_family);
 496         renderer_.set_font_size(config_->text_font_size);
 497         renderer_.set_font_style(config_->text_font_style);
 498         renderer_.set_alignment(text_alignment_);
 499 
 500         // Try with the minimum wanted size.
 501         const int maximum_width = text_maximum_width_ != 0 ? text_maximum_width_
 502                                                                                                            : maximum_size.x;
 503 
 504         renderer_.set_maximum_width(maximum_width);
 505 
 506         renderer_.set_ellipse_mode(get_text_ellipse_mode());
 507 
 508         renderer_.set_characters_per_line(get_characters_per_line());
 509         if(get_characters_per_line() != 0 && !can_wrap()) {
 510                 WRN_GUI_L
 511                 << LOG_HEADER << " Limited the number of characters per line, "
 512                 << "but wrapping is not set, output may not be as expected.\n";
 513         }
 514 
 515         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 516                           << "' status: "
 517                           << " minimum_size " << minimum_size << " maximum_size "
 518                           << maximum_size << " text_maximum_width_ " << text_maximum_width_
 519                           << " can_wrap " << can_wrap() << " characters_per_line "
 520                           << get_characters_per_line() << " truncated "
 521                           << renderer_.is_truncated() << " renderer size "
 522                           << renderer_.get_size() << ".\n";
 523 
 524         // If doesn't fit try the maximum.
 525         if(renderer_.is_truncated() && !can_wrap()) {
 526                 // FIXME if maximum size is defined we should look at that
 527                 // but also we don't adjust for the extra text space yet!!!
 528                 maximum_size = point(config_->max_width, config_->max_height);
 529                 renderer_.set_maximum_width(maximum_size.x ? maximum_size.x - border.x
 530                                                                                                    : -1);
 531         }
 532 
 533         size = renderer_.get_size() + border;
 534 
 535         if(size.x < minimum_size.x) {
 536                 size.x = minimum_size.x;
 537         }
 538 
 539         if(size.y < minimum_size.y) {
 540                 size.y = minimum_size.y;
 541         }
 542 
 543         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 544                           << "' result " << size << ".\n";
 545         return size;
 546 }

 548 void styled_widget::signal_handler_show_tooltip(const event::ui_event event,
 549                                                                                    bool& handled,
 550                                                                                    const point& location)
 551 {
 552         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 553 
 554         if(!tooltip_.empty()) {
 555                 std::string tip = tooltip_;
 556                 if(!help_message_.empty()) {
 557                         utils::string_map symbols;
 558                         symbols["hotkey"] = hotkey::get_names(
 559                                         hotkey::hotkey_command::get_command_by_command(
 560                                                         hotkey::GLOBAL__HELPTIP).command);
 561 
 562                         tip = tooltip_ + utils::interpolate_variables_into_string(
 563                                                                          settings::has_helptip_message, &symbols);
 564                 }
 565 
 566                 event::message_show_tooltip message(tip, location, get_rectangle());
 567                 handled = fire(event::MESSAGE_SHOW_TOOLTIP, *this, message);
 568         }
 569 }

 640  *     label & t_string & "" &          Most widgets have some text associated

 643  *                                     for other purposes, this is documented

 651  *                                     be empty when 'help' is set. $
 652  *
 653  *
 654  *     help & t_string & "" &           If you hover over a widget and press F10
 655  *                                     (or the key the user defined for the help
 656  *                                     tip) a help message can show up. This
 657  *                                     help message might be the same as the
 658  *                                     tooltip but in general (if used) this
 659  *                                     message should show more help. This
 660  *                                     defines the text of that message. $
 661  *
 662  *    use_tooltip_on_label_overflow & bool & true &
 663  *                                     If the text on the label is truncated and
 664  *                                     the tooltip is empty the label can be
 665  *                                     used for the tooltip. If this variable is
 666  *                                     set to true this will happen. $
 667  *
 668  *   debug_border_mode & unsigned & 0 &
 669  *                                     The mode for showing the debug border.
 670  *                                     This border shows the area reserved for
 671  *                                     a widget. This function is only meant
 672  *                                     for debugging and might not be
 673  *                                     available in all Wesnoth binaries.
 674  *                                     Available modes:
 675  *                                     @* 0 no border.
 676  *                                     @* 1 1 pixel border.
 677  *                                     @* 2 floodfill the widget area. $
 678  *
 679  *   debug_border_color & color & "" & The color of the debug border. $
 680  * @end{table}
 681  * @end{tag}{name="widget_instance"}
 682  * @end{parent}{name="generic/"}
 683  */

 693         , help(cfg["help"].t_str())
 694         , use_tooltip_on_label_overflow(true)
 695         , use_markup(cfg["use_markup"].to_bool(false))

 705                                                  << "' helptip '" << help << "'.");

 708         DBG_GUI_P << "Window builder: found styled_widget with id '" << id
 709                           << "' and definition '" << definition << "'.\n";
