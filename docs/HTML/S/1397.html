 184     private void updatePageIsComplete( )
 185     {
 186         setPageComplete( false );
 187         if( radioNewFile.getSelection( ) ) {
 188             IResource container = ResourcesPlugin.getWorkspace( ).getRoot( )
 189                 .findMember( new Path( getDirectoryName( ) ) );
 190             String fileName = getFileName( );
 191 
 192             if( getDirectoryName( ).isEmpty( ) ) {
 193                 setErrorMessage( Messages.WizardLauncherPage0_10 );
 194                 return;
 195             }
 196 
 197             if( container == null || ! container.exists( )
 198                 || ! ( container instanceof IContainer ) ) {
 199                 setErrorMessage( Messages.WizardLauncherPage0_11 );
 200                 return;
 201             }
 202 
 203             if( fileName.isEmpty( ) ) {
 204                 setErrorMessage( Messages.WizardLauncherPage0_12 );
 205                 return;
 206             }
 207 
 208             if( fileName.replace( '\\', '/' ).indexOf( '/', 1 ) > 0 ) {
 209                 setErrorMessage( Messages.WizardLauncherPage0_13 );
 210                 return;
 211             }
 212 
 213             int dotLoc = fileName.lastIndexOf( '.' );
 214             if( dotLoc == - 1
 215                 || fileName.substring( dotLoc + 1 )
 216                     .equalsIgnoreCase( "cfg" ) == false ) //$NON-NLS-1$
 217             {
 218                 setErrorMessage( Messages.WizardLauncherPage0_15 );
 219                 return;
 220             }
 221         }
 222         else {
 223             // current file checking
 224             if( EditorUtils.getEditedFile( ) != null ) {
 225                 lblCurrentFileOpened.setText( Messages.WizardLauncherPage0_16
 226                     + EditorUtils.getEditedFile( ).getEditorInput( )
 227                         .getName( ) + Messages.WizardLauncherPage0_17 );
 228             }
 229             else {
 230                 lblCurrentFileOpened.setText( Messages.WizardLauncherPage0_18 );
 231                 setErrorMessage( Messages.WizardLauncherPage0_19 );
 232                 return;
 233             }
 234         }
 235         setPageComplete( true );
 236         setErrorMessage( null );
 237     }
