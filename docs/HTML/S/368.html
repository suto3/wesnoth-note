  60 synched_choice_wait::synched_choice_wait(user_choice_manager& mgr)
  61         : mgr_(mgr)
  62         , message_()
  63         , window_(nullptr)
  64 {
  65         mgr_.changed_event_.attach_handler(this);
  66 }
  67 
  68 synched_choice_wait::~synched_choice_wait()
  69 {
  70         mgr_.changed_event_.detach_handler(this);
  71 }
  72 
  73 void synched_choice_wait::pre_show(window& window)
  74 {
  75         window_ = &window;
  76         message_ = find_widget<label>(&window, "lblMessage", false, true);
  77 
  78 
  79         button& quit_button = find_widget<button>(
  80                                 &window, "btn_quit_game", false);
  81 
  82         connect_signal_mouse_left_click(quit_button,
  83                 std::bind(&quit_confirmation::quit_to_title));
  84 
  85         message_->set_label(mgr_.wait_message());
  86         if(mgr_.finished() || !mgr_.waiting()) {
  87                 window_->close();
  88         }
  89 }
  90 
  91 void synched_choice_wait::handle_generic_event(const std::string& event_name)
  92 {
  93         assert(event_name == "user_choice_update");
  94         assert(message_);
  95         message_->set_label(mgr_.wait_message());
  96         if(mgr_.finished() || !mgr_.waiting()) {
  97                 window_->close();
  98         }
  99 }
