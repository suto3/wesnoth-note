 261         # Here we set up an array that holds the possible ambusher unit types.
 262         # We do this instead of just using a random type when creating the
 263         # ambusher so that the even distribution of different types is
 264         # guaranteed. For now, Orcish Archer is listed twice, so they appear
 265         # more often than the other types.
 266 
 267         {VARIABLE random_ambusher_type_table[0].type "Orcish Grunt"}
 268         {VARIABLE random_ambusher_type_table[1].type "Orcish Archer"}
 269         {VARIABLE random_ambusher_type_table[2].type "Orcish Archer"}
 270         {VARIABLE random_ambusher_type_table[3].type "Orcish Assassin"}
 271         {VARIABLE random_ambusher_type_table[4].type "Goblin Spearman"}
 272 
 273         {VARIABLE random_ambusher_type_i 0}
 274 
 275         # With this macro we place a set number of ambushers (from the above list) to a
 276         # given rectangular area. We do this in blocks so that we can guarantee that
 277         # every major area has at least some ambushers: if we randomized the ambush
 278         # locations across all possible locations at once, the RNG could theoretically
 279         # clump all the ambushers into a single corner of the map.
 280 
 281 #define ORC_AMBUSH_AREA SIDE X_SPAN Y_SPAN NUMBER
 282     # Any hill that's completely surrounded by other hills, mountains or
 283     # forest is eligible as an ambush location.
 284     [random_placement]
 285         variable=ambush_location
 286         num_items={NUMBER}
 287         [filter_location]
 288             x={X_SPAN}
 289             y={Y_SPAN}
 290             terrain=Hh
 291 
 292             [not]
 293                 [filter_adjacent_location]
 294                     [not]
 295                         terrain=Hh*,Mm*,*^F*
 296                     [/not]
 297                 [/filter_adjacent_location]
 298             [/not]
 299         [/filter_location]
 300         [command]
 301             [if]
 302                 [variable]
 303                     name=random_ambusher_type_i
 304                     less_than=4
 305                 [/variable]
 306 
 307                 [then]
 308                     {VARIABLE_OP random_ambusher_type_i add 1}
 309                 [/then]
 310 
 311                 [else]
 312                     {VARIABLE random_ambusher_type_i 0}
 313                 [/else]
 314             [/if]
 315 
 316             # And here we place an ambusher of a type specified by the
 317             # random_ambusher_type_table and random_ambusher_type_i
 318             # variables, thus always having the same ratio of different
 319             # types.
 320 
 321             [if]
 322                 [not]
 323                     [have_unit]
 324                         x,y=$ambush_location.x,$ambush_location.y
 325                     [/have_unit]
 326                 [/not]
 327                 [then]
 328                     [unit]
 329                         x,y=$ambush_location.x,$ambush_location.y
 330                         type=$random_ambusher_type_table[$random_ambusher_type_i].type
 331                         side={SIDE}
 332                         facing=sw
 333                         role=ambusher
 334                         random_traits=yes
 335 
 336                         [modifications]
 337                             {TRAIT_LOYAL}
 338                         [/modifications]
 339                         {IS_LOYAL}
 340                     [/unit]
 341 
 342                     [object]
 343                         silent=yes
 344 
 345                         [filter]
 346                             x,y=$ambush_location.x,$ambush_location.y
 347                         [/filter]
 348 
 349                         [effect]
 350                             apply_to=new_ability
 351 
 352                             [abilities]
 353                                 [hides]
 354                                     id=orc_ambush
 355                                     affect_self=yes
 356 
 357                                     [filter_self]
 358                                         role=ambusher
 359 
 360                                         [filter_location]
 361                                             terrain=Hh,Mm
 362                                         [/filter_location]
 363                                     [/filter_self]
 364                                 [/hides]
 365                             [/abilities]
 366                         [/effect]
 367                     [/object]
 368                 [/then]
 369             [/if]
 370         [/command]
 371     [/random_placement]
 372     {CLEAR_VARIABLE ambush_location}
 373 
 374 #enddef

 376         # Only the number of ambushers differs by difficulty here.
 377 
 378         # Adjust for leaders killed in S01 ('The Elves Besieged')
 379         # Reduce 25% for each. Note: implicit floor() due to loop
 380         # GL-2016Aug
 381         #
 382         #             3  2  1  0 # killed
 383         #
 384         #      5 -->  1  2  3  5
 385         #      6 -->  1  3  4  6
 386         #      7 -->  1  3  5  7
 387         #
 388         #      9 -->  2  4  6  9
 389         #     10 -->  2  5  7 10
 390         #     11 -->  2  5  8 11
 391         #
 392         # Net result
 393         #
 394         #   Easy -->  6 12 18 29
 395         # Normal -->  6 17 23 34
 396         #   Hard -->  6 17 28 39
 397 
 398         {VARIABLE small_number "$(((4 - $besieged_enemies_killed?0|) * {ON_DIFFICULTY 5 6 7}) / 4)"}
 399         {VARIABLE large_number "$(((4 - $besieged_enemies_killed?0|) * {ON_DIFFICULTY 9 10 11}) / 4)"}
 400 
 401         {ORC_AMBUSH_AREA 2 5-35 1-7 $small_number}
 402         {ORC_AMBUSH_AREA 2 9-24 8-15 $small_number}
 403         {ORC_AMBUSH_AREA 3 1-17 17-24 $large_number}
 404         {ORC_AMBUSH_AREA 3 18-25 16-23 $small_number}
 405         {ORC_AMBUSH_AREA 3 26-40 16-23 $small_number}
 406 
 407         {CLEAR_VARIABLE random_ambusher_type_table}
 408         {CLEAR_VARIABLE random_ambusher_type_i,small_number,large_number}
 409     [/event]
