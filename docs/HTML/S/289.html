  82 void holder::init( side_number side )
  83 {
  84         if (side_context_ == nullptr) {
  85                 side_context_ = new side_context_impl(side,cfg_);
  86         } else {
  87                 side_context_->set_side(side);
  88         }
  89         if (readonly_context_ == nullptr){
  90                 readonly_context_ = new readonly_context_impl(*side_context_,cfg_);
  91                 readonly_context_->on_readonly_context_create();
  92         }
  93         if (readwrite_context_ == nullptr){
  94                 readwrite_context_ = new readwrite_context_impl(*readonly_context_,cfg_);
  95         }
  96         if (default_ai_context_ == nullptr){
  97                 default_ai_context_ = new default_ai_context_impl(*readwrite_context_,cfg_);
  98         }
  99         if (!this->ai_){
 100                 ai_ = std::shared_ptr<ai_composite>(new ai_composite(*default_ai_context_,cfg_));
 101         }
 102 
 103         if (this->ai_) {
 104                 ai_->on_create();
 105                 for (config &mod_ai : cfg_.child_range("modify_ai")) {
 106                         if (!mod_ai.has_attribute("side")) {
 107                                 mod_ai["side"] = side;
 108                         }
 109                         modify_ai(mod_ai);
 110                 }
 111                 cfg_.clear_children("modify_ai");
 112 
 113                 std::vector<engine_ptr> engines = ai_->get_engines();
 114                 for (std::vector<engine_ptr>::iterator it = engines.begin(); it != engines.end(); ++it)
 115                 {
 116                         (*it)->set_ai_context(&(ai_->get_ai_context()));
 117                 }
 118 
 119         } else {
 120                 ERR_AI_MANAGER << describe_ai()<<"AI lazy initialization error!" << std::endl;
 121         }
 122 
 123 }

 150 void holder::modify_ai(const config &cfg)
 151 {
 152         if (!this->ai_) {
 153                 // if not initialized, initialize now.
 154                 get_ai_ref();
 155         }
 156         const std::string &act = cfg["action"];
 157         LOG_AI_MOD << "side "<< side_ << "        "<<act<<"_ai_component \""<<cfg["path"]<<"\""<<std::endl;
 158         DBG_AI_MOD << std::endl << cfg << std::endl;
 159         DBG_AI_MOD << "side "<< side_ << " before "<<act<<"_ai_component"<<std::endl << to_config() << std::endl;
 160         bool res = false;
 161         if (act == "add") {
 162                 res = component_manager::add_component(&*this->ai_,cfg["path"],cfg);
 163         } else if (act == "change") {
 164                 res = component_manager::change_component(&*this->ai_,cfg["path"],cfg);
 165         } else if (act == "delete") {
 166                 res = component_manager::delete_component(&*this->ai_,cfg["path"]);
 167         } else {
 168                 ERR_AI_MOD << "modify_ai tag has invalid 'action' attribute " << act << std::endl;
 169         }
 170         DBG_AI_MOD << "side "<< side_ << "  after [modify_ai]"<<act<<std::endl << to_config() << std::endl;
 171         if (!res) {
 172                 LOG_AI_MOD << act << "_ai_component failed"<< std::endl;
 173         } else {
 174                 LOG_AI_MOD << act << "_ai_component success"<< std::endl;
 175         }
 176 
 177 }

 179 void holder::append_ai(const config& cfg)
 180 {
 181         if(!this->ai_) {
 182                 get_ai_ref();
 183         }
 184         for(const config& aspect : cfg.child_range("aspect")) {
 185                 const std::string& id = aspect["id"];
 186                 for(const config& facet : aspect.child_range("facet")) {
 187                         ai_->add_facet(id, facet);
 188                 }
 189         }
 190         for(const config& goal : cfg.child_range("goal")) {
 191                 ai_->add_goal(goal);
 192         }
 193         for(const config& stage : cfg.child_range("stage")) {
 194                 if(stage["name"] != "empty") {
 195                         ai_->add_stage(stage);
 196                 }
 197         }
 198 }

 467 const std::string manager::evaluate_command( side_number side, const std::string& str )
 468 {
 469         //insert new command into history
 470         history_.emplace_back(history_item_counter_++,str);
 471 
 472         //prune history - erase 1/2 of it if it grows too large
 473         if (history_.size()>MAX_HISTORY_SIZE){
 474                 history_.erase(history_.begin(),history_.begin()+MAX_HISTORY_SIZE/2);
 475                 LOG_AI_MANAGER << "AI MANAGER: pruned history" << std::endl;
 476         }
 477 
 478         if (!should_intercept(str)){
 479                 ai_composite& ai = get_active_ai_for_side(side);
 480                 raise_gamestate_changed();
 481                 return ai.evaluate(str);
 482         }
 483 
 484         return internal_evaluate_command(side,str);
 485 }

 488 bool manager::should_intercept( const std::string& str )
 489 {
 490         if (str.length()<1) {
 491                 return false;
 492         }
 493         if (str.at(0)=='!'){
 494                 return true;
 495         }
 496         if (str.at(0)=='?'){
 497                 return true;
 498         }
 499         return false;
 500 
 501 }

 503 std::deque< command_history_item > manager::history_;
 504 long manager::history_item_counter_ = 1;
 505 
 506 //this is stub code to allow testing of basic 'history', 'repeat-last-command', 'add/remove/replace ai' capabilities.
 507 //yes, it doesn't look nice. but it is usable.
 508 //to be refactored at earliest opportunity
 509 ///@todo 1.9 extract to separate class which will use fai or lua parser
 510 const std::string manager::internal_evaluate_command( side_number side, const std::string& str ){
 511         const int MAX_HISTORY_VISIBLE = 30;
 512 
 513         //repeat last command
 514         if (str=="!") {
 515                         //this command should not be recorded in history
 516                         if (!history_.empty()){
 517                                 history_.pop_back();
 518                                 history_item_counter_--;
 519                         }
 520 
 521                         if (history_.empty()){
 522                                 return "AI MANAGER: empty history";
 523                         }
 524                         return evaluate_command(side, history_.back().get_command());//no infinite loop since '!' commands are not present in history
 525         };
 526         //show last command
 527         if (str=="?") {
 528                 //this command should not be recorded in history
 529                 if (!history_.empty()){
 530                         history_.pop_back();
 531                         history_item_counter_--;
 532                 }
 533 
 534                 if (history_.empty()){
 535                         return "AI MANAGER: History is empty";
 536                 }
 537 
 538                 int n = std::min<int>( MAX_HISTORY_VISIBLE, history_.size() );
 539                 std::stringstream strstream;
 540                 strstream << "AI MANAGER: History - last "<< n <<" commands:\n";
 541                 std::deque< command_history_item >::reverse_iterator j = history_.rbegin();
 542 
 543                 for (int cmd_id=n; cmd_id>0; --cmd_id){
 544                         strstream << j->get_number() << "    :" << j->get_command() << '\n';
 545                         ++j;//this is *reverse* iterator
 546                 }
 547 
 548                 return strstream.str();
 549         };
 550 
 551 
 552         std::vector< std::string > cmd = utils::parenthetical_split(str, ' ',"'","'");
 553 
 554         if (cmd.size()==3){
 555                 //!add_ai side file
 556                 if (cmd.at(0)=="!add_ai"){
 557                         side = std::stoi(cmd.at(1));
 558                         std::string file = cmd.at(2);
 559                         if (add_ai_for_side_from_file(side,file,false)){
 560                                 return std::string("AI MANAGER: added [")+manager::get_active_ai_identifier_for_side(side)+std::string("] AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 561                         } else {
 562                                 return std::string("AI MANAGER: failed attempt to add AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 563                         }
 564                 }
 565                 //!replace_ai side file
 566                 if (cmd.at(0)=="!replace_ai"){
 567                         side = std::stoi(cmd.at(1));
 568                         std::string file = cmd.at(2);
 569                         if (add_ai_for_side_from_file(side,file,true)){
 570                                         return std::string("AI MANAGER: added [")+manager::get_active_ai_identifier_for_side(side)+std::string("] AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 571                         } else {
 572                                         return std::string("AI MANAGER: failed attempt to add AI for side ")+std::to_string(side)+std::string(" from file ")+file;
 573                         }
 574                 }
 575 
 576         } else if (cmd.size()==2){
 577                 //!remove_ai side
 578                 if (cmd.at(0)=="!remove_ai"){
 579                         side = std::stoi(cmd.at(1));
 580                         remove_ai_for_side(side);
 581                         return std::string("AI MANAGER: made an attempt to remove AI for side ")+std::to_string(side);
 582                 }
 583                 if (cmd.at(0)=="!"){
 584                         //this command should not be recorded in history
 585                         if (!history_.empty()){
 586                                 history_.pop_back();
 587                                 history_item_counter_--;
 588                         }
 589 
 590                         int command = std::stoi(cmd.at(1));
 591                         std::deque< command_history_item >::reverse_iterator j = history_.rbegin();
 592                         //yes, the iterator could be precisely positioned (since command numbers go 1,2,3,4,..). will do it later.
 593                         while ( (j!=history_.rend()) && (j->get_number()!=command) ){
 594                                 ++j;// this is *reverse* iterator
 595                         }
 596                         if (j!=history_.rend()){
 597                                 return evaluate_command(side,j->get_command());//no infinite loop since '!' commands are not present in history
 598                         }
 599                         return "AI MANAGER: no command with requested number found";
 600                 }
 601         } else if (cmd.size()==1){
 602                 if (cmd.at(0)=="!help") {
 603                         return
 604                                 "known commands:\n"
 605                                 "!    - repeat last command (? and ! do not count)\n"
 606                                 "! NUMBER    - repeat numbered command\n"
 607                                 "?    - show a history list\n"
 608                                 "!add_ai TEAM FILE    - add a AI to side (0 - command AI, N - AI for side #N) from file\n"
 609                                 "!remove_ai TEAM    - remove AI from side (0 - command AI, N - AI for side #N)\n"
 610                                 "!replace_ai TEAM FILE    - replace AI of side (0 - command AI, N - AI for side #N) from file\n"
 611                                 "!help    - show this help message";
 612                 }
 613         }
 614 
 615 
 616         return "AI MANAGER: nothing to do";
 617 }
