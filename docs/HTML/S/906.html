 139 game_events::pump_result_t get_village(const map_location& loc, int side, bool *action_timebonus, bool fire_event)
 140 {
 141         std::vector<team> &teams = resources::gameboard->teams();
 142         team *t = unsigned(side - 1) < teams.size() ? &teams[side - 1] : nullptr;
 143         if (t && t->owns_village(loc)) {
 144                 return game_events::pump_result_t();
 145         }
 146 
 147         bool not_defeated = t && !resources::gameboard->team_is_defeated(*t);
 148 
 149         bool grants_timebonus = false;
 150 
 151         int old_owner_side = 0;
 152         // We strip the village off all other sides, unless it is held by an ally
 153         // and our side is already defeated (and thus we can't occupy it)
 154         for(std::vector<team>::iterator i = teams.begin(); i != teams.end(); ++i) {
 155                 int i_side = i - teams.begin() + 1;
 156                 if (!t || not_defeated || t->is_enemy(i_side)) {
 157                         if(i->owns_village(loc)) {
 158                                 old_owner_side = i_side;
 159                                 i->lose_village(loc);
 160                         }
 161                         if (side != i_side && action_timebonus) {
 162                                 grants_timebonus = true;
 163                         }
 164                 }
 165         }
 166 
 167         if (!t) {
 168                 return game_events::pump_result_t();
 169         }
 170         
 171         if(grants_timebonus) {
 172                 t->set_action_bonus_count(1 + t->action_bonus_count());
 173                 *action_timebonus = true;
 174         }
 175 
 176         if(not_defeated) {
 177                 if (resources::screen != nullptr) {
 178                         resources::screen->invalidate(loc);
 179                 }
 180                 return t->get_village(loc, old_owner_side, fire_event ? resources::gamedata : nullptr);
 181         }
 182 
 183         return game_events::pump_result_t();
 184 }

 415         /**
 416          * Checks for ambushers around @a hex, setting flags as appropriate.
 417          */
 418         inline void unit_mover::check_for_ambushers(const map_location & hex)
 419         {
 420                 const unit_map &units = resources::gameboard->units();
 421 
 422                 // Need to check each adjacent hex for hidden enemies.
 423                 map_location adjacent[6];
 424                 get_adjacent_tiles(hex, adjacent);
 425                 for ( int i = 0; i != 6; ++i )
 426                 {
 427                         const unit_map::const_iterator neighbor_it = units.find(adjacent[i]);
 428 
 429                         if ( neighbor_it != units.end()  &&
 430                              current_team_->is_enemy(neighbor_it->side())  &&
 431                              neighbor_it->invisible(adjacent[i], *resources::gameboard) )
 432                         {
 433                                 // Ambushed!
 434                                 ambushed_ = true;
 435                                 ambush_stop_ = hex;
 436                                 ambushers_.push_back(adjacent[i]);
 437                         }
 438                 }
 439         }

 442         /**
 443          * Makes sure the path is not obstructed by a unit.
 444          * @param  hex       The hex to check.
 445          * @param  prev_hex  The previous hex in the route (used to detect a teleport).
 446          * @return true if @a hex is obstructed.
 447          */
 448         inline bool unit_mover::check_for_obstructing_unit(const map_location & hex,
 449                                                            const map_location & prev_hex)
 450         {
 451                 const unit_map::const_iterator blocking_unit = resources::gameboard->units().find(hex);
 452 
 453                 // If no unit, then the path is not obstructed.
 454                 if (blocking_unit == resources::gameboard->units().end()) {
 455                         return false;
 456                 }
 457 
 458                 // Check for units blocking a teleport exit. This can now only happen
 459                 // if these units are not visible to the current side, as otherwise no
 460                 // valid path is found.
 461                 if ( !tiles_adjacent(hex, prev_hex) ) {
 462                         if ( current_team_->is_enemy(blocking_unit->side()) ) {
 463                                 // Enemy units always block the tunnel.
 464                                 teleport_failed_ = true;
 465                                 return true;
 466                         } else {
 467                                 // By contrast, allied units (of a side which does not share vision) only
 468                                 // block the tunnel if pass_allied_units=true. Whether the teleport is possible
 469                                 // is checked by getting the teleport map with the see_all flag set to true.
 470                                 const pathfind::teleport_map teleports = pathfind::get_teleport_locations(*move_it_, *current_team_, true, false, false);
 471                                 std::set<map_location> allowed_teleports;
 472                                 teleports.get_adjacents(allowed_teleports, prev_hex);
 473 
 474                                 bool found_valid_teleport = false;
 475                                 std::set<map_location>::const_iterator it = allowed_teleports.begin();
 476                                 while( it!=allowed_teleports.end() && !found_valid_teleport ) {
 477                                         if ( *it == hex ) {
 478                                                 found_valid_teleport = true;
 479                                         }
 480                                         ++it;
 481                                 }
 482 
 483                                 if (!found_valid_teleport) {
 484                                         teleport_failed_ = true;
 485                                         return true;
 486                                 }
 487                         }
 488                 }
 489 
 490                 if ( current_team_->is_enemy(blocking_unit->side()) ) {
 491                         // Trying to go through an enemy.
 492                         blocked_loc_ = hex;
 493                         return true;
 494                 }
 495 
 496                 // If we get here, the unit does not interfere with movement.
 497                 return false;
 498         }

 501         /**
 502          * Moves the unit the next step.
 503          * @a step_to is the hex being moved to.
 504          * @a step_from is the hex before that in the route.
 505          * (The unit is actually at *move_loc_.)
 506          * @a animator is the unit_display::unit_mover being used.
 507          * @return whether or not we started a new animation.
 508          */
 509         inline bool unit_mover::do_move(const route_iterator & step_from,
 510                                         const route_iterator & step_to,
 511                                         unit_display::unit_mover & animator)
 512         {
 513                 game_display &disp = *resources::screen;
 514 
 515                 // Adjust the movement even if we cannot move yet.
 516                 // We will eventually be able to move if nothing unexpected
 517                 // happens, and if something does happen, this movement is the
 518                 // cost to discover it.
 519                 move_it_->set_movement(moves_left_.front(), true);
 520                 moves_left_.pop_front();
 521 
 522                 // Invalidate before moving so we invalidate neighbor hexes if needed.
 523                 move_it_->anim_comp().invalidate(disp);
 524 
 525                 // Attempt actually moving.
 526                 // (Fails if *step_to is occupied).
 527                 std::pair<unit_map::iterator, bool> move_result =
 528                         resources::gameboard->units().move(*move_loc_, *step_to);
 529                 if ( move_result.second )
 530                 {
 531                         // Update the moving unit.
 532                         move_it_ = move_result.first;
 533                         move_it_->set_facing(step_from->get_relative_dir(*step_to));
 534                         // Disable bars. The expectation here is that the animation
 535                         // unit_mover::finish() will clean after us at a later point. Ugly,
 536                         // but it works.
 537                         move_it_->anim_comp().set_standing(false);
 538                         disp.invalidate_unit_after_move(*move_loc_, *step_to);
 539                         disp.invalidate(*step_to);
 540                         move_loc_ = step_to;
 541 
 542                         // Show this move.
 543                         animator.proceed_to(move_it_.get_shared_ptr(), step_to - begin_,
 544                                             move_it_->appearance_changed(), false);
 545                         move_it_->set_appearance_changed(false);
 546                         disp.redraw_minimap();
 547                 }
 548 
 549                 return move_result.second;
 550         }

 553         /**
 554          * Clears fog/shroud and raises events for units being sighted.
 555          * Only call this if the current team uses fog or shroud.
 556          * @a hex is both the center of fog clearing and the filtered location of
 557          * the moving unit when the sighted events will be fired.
 558          */
 559         inline void unit_mover::handle_fog(const map_location & hex,
 560                                            bool new_animation)
 561         {
 562                 // Clear the fog.
 563                 if ( clearer_.clear_unit(hex, *move_it_, *current_team_, nullptr,
 564                                          &enemy_count_, &friend_count_, spectator_,
 565                                          !new_animation) )
 566                 {
 567                         clearer_.invalidate_after_clear();
 568                         fog_changed_ = true;
 569                 }
 570 
 571                 // Check for sighted units?
 572                 if ( !skip_sighting_ ) {
 573                         sighted_ = enemy_count_ != 0 ;
 574                 }
 575                 if( !skip_sighting_ && !skip_ally_sighting_ )
 576                 {
 577                         sighted_ |= (friend_count_ != 0);
 578                 }
 579         }

 587         inline bool unit_mover::is_reasonable_stop(const map_location & hex) const
 588         {
 589                 // We cannot reasonably stop if move_it_ could not be moved to this
 590                 // hex (the hex was occupied by someone else).
 591                 if (*move_loc_ != hex) {
 592                         return false;
 593                 }
 594 
 595                 // We can reasonably stop if the hex is not an unowned village.
 596                 return !resources::gameboard->map().is_village(hex) ||
 597                        current_team_->owns_village(hex);
 598         }

 628         /**
 629          * Makes sure the units in ambushers_ still exist.
 630          */
 631         inline void unit_mover::validate_ambushers()
 632         {
 633                 const unit_map &units = resources::gameboard->units();
 634 
 635                 // Loop through the previously-detected ambushers.
 636                 size_t i = 0;
 637                 while ( i != ambushers_.size() ) {
 638                         if (units.count(ambushers_[i]) == 0) {
 639                                 // Ambusher is gone.
 640                                 ambushers_.erase(ambushers_.begin() + i);
 641                         }
 642                         else {
 643                                 // Proceed to the next ambusher.
 644                                 ++i;
 645                         }
 646                 }
 647         }

 652         /**
 653          * Checks the expected route for hidden units.
 654          * This basically handles all the checks for surprises that can be done
 655          * without visibly notifying a player. Thus this can be called at the
 656          * start of movement and immediately after events, rather than tie up
 657          * CPU time in the middle of animating a move.
 658          *
 659          * @param[in]  start           The beginning of the path to check.
 660          * @param[in]  stop            The end of the path to check.
 661          */
 662         void unit_mover::cache_hidden_units(const route_iterator & start,
 663                                             const route_iterator & stop)
 664         {
 665                 // Clear the old cache.
 666                 obstructed_ = full_end_;
 667                 blocked_loc_ = map_location::null_location();
 668                 teleport_failed_ = false;
 669                 // The ambush cache needs special treatment since we cannot re-detect
 670                 // an ambush if we are already at the ambushed location.
 671                 ambushed_ =  ambushed_  &&  ambush_stop_ == *start;
 672                 if ( ambushed_ ) {
 673                         validate_ambushers();
 674                         ambushed_ = !ambushers_.empty();
 675                 }
 676                 if ( !ambushed_ ) {
 677                         ambush_stop_ = map_location::null_location();
 678                         ambushers_.clear();
 679                 }
 680 
 681                 // Update the shroud clearer.
 682                 clearer_.cache_units(current_uses_fog_ ? current_team_ : nullptr);
 683 
 684 
 685                 // Abort for null routes.
 686                 if ( start == stop ) {
 687                         ambush_limit_ = start;
 688                         return;
 689                 }
 690 
 691                 // This loop will end with ambush_limit_ pointing one element beyond
 692                 // where the unit would be forced to stop by a hidden unit.
 693                 for ( ambush_limit_ = start+1; ambush_limit_ != stop; ++ambush_limit_ ) {
 694                         // Check if we need to stop in the previous hex.
 695                         if ( ambushed_ ) {
 696                                         break;
 697                         }
 698                         // Check for being unable to enter this hex.
 699                         if ( check_for_obstructing_unit(*ambush_limit_, *(ambush_limit_-1)) ) {
 700                                 // No replay check here? Makes some sense, I guess.
 701                                 obstructed_ = ambush_limit_++; // The limit needs to be after obstructed_ in order for the latter to do anything.
 702                                 break;
 703                         }
 704 
 705                         // We can enter this hex.
 706                         // See if we are stopped in this hex.
 707                         check_for_ambushers(*ambush_limit_);
 708                 }
 709         }

 731         /**
 732          * Checks how far it appears we can move this turn.
 733          *
 734          * @param[in] start         The beginning of the plotted path.
 735          * @param[in] stop          The end of the plotted path.
 736          *
 737          * @return An end iterator for the path that can be traversed this turn.
 738          */
 739         unit_mover::route_iterator unit_mover::plot_turn(const route_iterator & start,
 740                                                          const route_iterator & stop)
 741         {
 742                 const gamemap &map = resources::gameboard->map();
 743 
 744                 // Handle null routes.
 745                 if (start == stop) {
 746                         return start;
 747                 }
 748 
 749                 int remaining_moves = move_it_->movement_left();
 750                 zoc_stop_ = map_location::null_location();
 751                 moves_left_.clear();
 752 
 753                 if ( start != begin_ ) {
 754                         // Check for being unable to leave the current hex.
 755                         if ( !move_it_->get_ability_bool("skirmisher", *start, *resources::gameboard) &&
 756                               pathfind::enemy_zoc(*current_team_, *start, *current_team_) )
 757                                 zoc_stop_ = *start;
 758                 }
 759 
 760                 // This loop will end with end pointing one element beyond where the
 761                 // unit thinks it will stop (the usual notion of "end" for iterators).
 762                 route_iterator end = start + 1;
 763                 for ( ; end != stop; ++end )
 764                 {
 765                         // Break out of the loop if we cannot leave the previous hex.
 766                         if ( zoc_stop_ != map_location::null_location() ) {
 767                                 break;
 768                         }
 769                         remaining_moves -= move_it_->movement_cost(map[*end]);
 770                         if ( remaining_moves < 0 ) {
 771                                         break;
 772                         }
 773 
 774                         // We can enter this hex. Record the cost.
 775                         moves_left_.push_back(remaining_moves);
 776 
 777                         // Check for being unable to leave this hex.
 778                         if (!move_it_->get_ability_bool("skirmisher", *end, *resources::gameboard) &&
 779                                 pathfind::enemy_zoc(*current_team_, *end, *current_team_))
 780                         {
 781                                 zoc_stop_ = *end;
 782                         }
 783                 }
 784 
 785                 route_iterator min_end =  start == begin_ ? start : start + 1;
 786                 while (end != min_end  &&  resources::gameboard->has_visible_unit(*(end - 1), *current_team_)) {
 787                         // Backtrack.
 788                         --end;
 789                 }
 790 
 791                 return end;
 792         }

 795         /**
 796          * Updates our stored info after a WML event might have changed something.
 797          *
 798          * @param step      Indicates the position in the path where we might need to start recalculating movement.
 799          *                  Set this to full_end_ (or do not supply it) to skip such recalculations (because movement has finished).
 800          *
 801          * @returns false if continuing is impossible (i.e. we lost the moving unit).
 802          */
 803         void unit_mover::post_wml(game_events::pump_result_t pump_res, const route_iterator & step)
 804         {
 805                 wml_move_aborted_ |= std::get<1>(pump_res);
 806                 wml_undo_disabled_ |= std::get<0>(pump_res);
 807 
 808                 // Re-find the moving unit.
 809                 move_it_ = resources::gameboard->units().find(*move_loc_);
 810                 const bool found = move_it_ != resources::gameboard->units().end();
 811 
 812                 // Update the current unit data.
 813                 current_side_ = found ? move_it_->side() : orig_side_;
 814                 current_team_ = &resources::gameboard->get_team(current_side_);
 815                 current_uses_fog_ = current_team_->fog_or_shroud()  &&
 816                                     ( current_side_ != orig_side_  ||
 817                                       current_team_->auto_shroud_updates() );
 818 
 819                 // Update the path.
 820                 if ( found  &&  step != full_end_ ) {
 821                         const route_iterator new_limit = plot_turn(step, expected_end_);
 822                         cache_hidden_units(step, new_limit);
 823                         // Just in case: length 0 paths become length 1 paths.
 824                         if (ambush_limit_ == step) {
 825                                 ++ambush_limit_;
 826                         }
 827                 }
 828 
 829                 wml_removed_unit_ |= !found;
 830         }

 835          *
 836          * @param[in]  from  Points to the hex the sighting unit currently occupies.
 837          *
 838          * @return sets event_mutated_ || wml_move_aborted_ to true if this event should interrupt movement.

 840         void unit_mover::pump_sighted(const route_iterator & from)

 845 
 846                 if (track != resources::game_events->pump().wml_tracking()) {
 847                         // Some WML fired, so update our status.
 848                         post_wml(pump_res, from);
 849                 }               
 850         }

 874          * May fire "sighted" events.

 879                 unit_map &units = resources::gameboard->units();

 881 
 882                 // Find the unit at the indicated location.
 883                 unit_map::iterator ambusher = units.find(hex);
 884                 if ( ambusher != units.end() ) {
 885                         // Prepare for sighted events.
 886                         std::vector<int> sight_cache(get_sides_not_seeing(*ambusher));

 888                         // we had to backtrack).
 889                         ambusher->set_state(unit::STATE_UNCOVERED, true);

 891                         // Record this in the move spectator.
 892                         if (spectator_) {
 893                                 spectator_->set_ambusher(ambusher);

 912                         bool wml_move_aborted = false;
 913                 
 914                         std::tie(wml_undo_blocked, wml_move_aborted) = actor_sighted(*ambusher, &sight_cache);

 916                         wml_move_aborted_ |= wml_move_aborted;
 917                         wml_undo_disabled_ |= wml_undo_blocked;
 918                 }
 919         }

 926          * This is based solely on data known to the player, and will not plot a move
 927          * that ends on another (known) unit.
 928          * (For example, this prevents a player from plotting a multi-turn move that
 929          * has this turn's movement ending on a (slower) unit, and thereby clearing

 931          *
 932          * @returns false if the expectation is to not move at all.

 937                 return expected_end_ != begin_;
 938         }

 941         /**
 942          * Attempts to move the unit along the expected path.
 943          * (This will do nothing unless check_expected_movement() was called first.)
 944          *
 945          * @param[in]  show            Set to false to suppress animations.
 946          */
 947         void unit_mover::try_actual_movement(bool show)
 948         {
 949                 static const std::string enter_hex_str("enter hex");
 950                 static const std::string exit_hex_str("exit hex");
 951 
 952 
 953                 bool obstructed_stop = false;
 954 
 955 
 956                 // Check for hidden units along the expected path before we start
 957                 // animating and firing events.
 958                 cache_hidden_units(begin_, expected_end_);
 959 
 960                 if ( begin_ != ambush_limit_ ) {
 961                         // Cache the moving unit's visibility.
 962                         std::vector<int> not_seeing = get_sides_not_seeing(*move_it_);
 963 
 964                         // Prepare to animate.
 965                         unit_display::unit_mover animator(route_, show);
 966                         animator.start(move_it_.get_shared_ptr());
 967 
 968                         // Traverse the route to the hex where we need to stop.
 969                         // Each iteration performs the move from real_end_-1 to real_end_.
 970                         for ( real_end_ = begin_+1; real_end_ != ambush_limit_; ++real_end_ ) {
 971                                 const route_iterator step_from = real_end_ - 1;
 972 
 973                                 // See if we can leave *step_from.
 974                                 // Already accounted for: ambusher
 975                                 if ( wml_removed_unit_ || wml_move_aborted_) {
 976                                         break;
 977                                 }
 978                                 if ( sighted_ && is_reasonable_stop(*step_from) ) {
 979                                         sighted_stop_ = true;
 980                                         break;
 981                                 }
 982                                 // Already accounted for: ZoC
 983                                 // Already accounted for: movement cost
 984                                 fire_hex_event(exit_hex_str, step_from, real_end_);
 985                                 if (wml_removed_unit_ || wml_move_aborted_) {
 986                                         break;
 987                                 }
 988                                 if ( real_end_ == obstructed_ ) {
 989                                         // We did not check for being a replay when checking for an
 990                                         // obstructed hex, so we do not check can_break here.
 991                                         obstructed_stop = true;
 992                                         break;
 993                                 }
 994 
 995                                 // We can leave *step_from. Make the move to *real_end_.
 996                                 bool new_animation = do_move(step_from, real_end_, animator);
 997                                 // Update the fog.
 998                                 if ( current_uses_fog_ )
 999                                         handle_fog(*real_end_, new_animation);
1000                                 animator.wait_for_anims();
1001 
1002                                 // Fire the events for this step.
1003                                 // (These return values are not checked since real_end_ still
1004                                 // needs to be incremented. The wml_move_aborted_ check will break
1005                                 // us out of the loop if needed.)
1006                                 fire_hex_event(enter_hex_str, real_end_, step_from);
1007                                 // Sighted events only fire if we could stop due to sighting.
1008                                 if (is_reasonable_stop(*real_end_)) {
1009                                         pump_sighted(real_end_);
1010                                 }
1011                         }//for
1012                         // Make sure any remaining sighted events get fired.
1013                         pump_sighted(real_end_-1);
1014 
1015                         if ( move_it_.valid() ) {
1016                                 // Finish animating.
1017                                 animator.finish(move_it_.get_shared_ptr());
1018                                 // Check for the moving unit being seen.
1019 
1020                                 bool wml_undo_blocked = false;
1021                                 bool wml_move_aborted = false;
1022 
1023                                 std::tie(wml_undo_blocked, wml_move_aborted) = actor_sighted(*move_it_, &not_seeing);
1024                                 // TODO: should we call post_wml ?
1025                                 wml_move_aborted_ |= wml_move_aborted;
1026                                 wml_undo_disabled_ |= wml_undo_blocked;
1027                         }
1028                 }//if
1029 
1030                 // Some flags were set to indicate why we might stop.
1031                 // Update those to reflect whether or not we got to them.
1032                 ambushed_ = ambushed_ && real_end_ == ambush_limit_;
1033                 if (!obstructed_stop) {
1034                         blocked_loc_ = map_location::null_location();
1035                 }
1036                 teleport_failed_ = teleport_failed_ && obstructed_stop;
1037                 // event_mutated_ does not get unset, regardless of other reasons
1038                 // for stopping, but we do save its current value.
1039                 event_mutated_mid_move_ = wml_removed_unit_ || wml_move_aborted_;
1040         }

1043         /**
1044          * Does some bookkeeping and event firing, for use after movement.
1045          * This includes village capturing and the undo stack.
1046          */
1047         void unit_mover::post_move(undo_list *undo_stack)
1048         {
1049                 const map_location & final_loc = final_hex();
1050 
1051                 int orig_village_owner = 0;
1052                 bool action_time_bonus = false;
1053 
1054                 // Reveal ambushers?
1055                 if (ambushed_ || blocked()) {
1056                         reveal_ambushers();
1057                 }
1058                 else if (teleport_failed_ && spectator_) {
1059                         spectator_->set_failed_teleport(resources::gameboard->units().find(*obstructed_));
1060                 }
1061                 unit::clear_status_caches();
1062 
1063                 if ( move_it_.valid() ) {
1064                         // Update the moving unit.
1065                         move_it_->set_interrupted_move(
1066                                 sighted_stop_ && !resources::whiteboard->is_executing_actions() ?
1067                                         *(full_end_-1) :
1068                                         map_location::null_location());
1069                         if (ambushed_ || final_loc == zoc_stop_) {
1070                                 move_it_->set_movement(0, true);
1071                         }
1072 
1073                         // Village capturing.
1074                         if ( resources::gameboard->map().is_village(final_loc) ) {
1075                                 // Is this a capture?
1076                                 orig_village_owner = resources::gameboard->village_owner(final_loc) + 1;
1077                                 if ( orig_village_owner != current_side_) {
1078                                         // Captured. Zap movement and take over the village.
1079                                         move_it_->set_movement(0, true);
1080                                         post_wml(get_village(final_loc, current_side_, &action_time_bonus));
1081                                 }
1082                         }
1083                 }
1084 
1085                 // Finally, the moveto event.
1086                 post_wml(resources::game_events->pump().fire("moveto", final_loc, *begin_));
1087 
1088                 // Record keeping.
1089                 if (spectator_) {
1090                         spectator_->set_unit(move_it_);
1091                 }
1092                 if ( undo_stack ) {
1093                         const bool mover_valid = move_it_.valid();
1094 
1095                         if ( mover_valid ) {
1096                                 // MP_COUNTDOWN: added param
1097                                 undo_stack->add_move(
1098                                         move_it_.get_shared_ptr(), begin_, real_end_, orig_moves_,
1099                                         action_time_bonus, orig_village_owner, orig_dir_);
1100                         }
1101 
1102                         if ( !mover_valid  ||  undo_blocked()  ||
1103                                 (resources::whiteboard->is_active() && resources::whiteboard->should_clear_undo()) || !synced_context::can_undo())
1104                         {
1105                                 undo_stack->clear();
1106                         }
1107                 }
1108 
1109                 // Update the screen.
1110                 resources::screen->redraw_minimap();
1111                 resources::screen->draw();
1112         }

1116          * Shows the various on-screen messages, for use after movement.
1117          */
1118         void unit_mover::feedback() const
1119         {
1120                 // Alias some resources.
1121                 game_display &disp = *resources::screen;
1122 
1123                 bool redraw = false;
1124 
1125                 // Multiple messages may be displayed simultaneously
1126                 // this variable is used to keep them from overlapping
1127                 std::string message_prefix = "";
1128 
1129                 // Ambush feedback?
1130                 if ( ambushed_  &&  show_ambush_alert_ ) {
1131                         disp.announce(message_prefix + ambush_string_, font::BAD_COLOR);

1141                         redraw = true;
1142                 }
1143 
1144                 // Sighted units feedback?
1145                 if ( playing_team_is_viewing_  &&  (enemy_count_ != 0 || friend_count_ != 0) ) {

1162                                 msg_color = font::BAD_COLOR;
1163                         } else if ( friend_count_ != 0 ) {
1164                                 // Only friends sighted -- good message.
1165                                 message = VNGETTEXT("Friendly unit sighted", "$friends friendly units sighted", friend_count_, symbols);
1166                                 msg_color = font::GOOD_COLOR;
1167                         }
1168 
1169                         disp.announce(message_prefix + message, msg_color);
1170                         message_prefix += " \n";
1171                         redraw = true;
1172                 }
1173 
1174                 // Suggest "continue move"?
1175                 if ( playing_team_is_viewing_ && sighted_stop_ && !resources::whiteboard->is_executing_actions() ) {
1176                         // See if the "Continue Move" action has an associated hotkey
1177                         std::string name = hotkey::get_names(hotkey::hotkey_command::get_command_by_command(hotkey::HOTKEY_CONTINUE_MOVE).command);
1178                         if ( !name.empty() ) {
1179                                 utils::string_map symbols;
1180                                 symbols["hotkey"] = name;
1181                                 std::string message = vgettext("(press $hotkey to keep moving)", symbols);
1182                                 disp.announce(message_prefix + message, font::NORMAL_COLOR);
1183                                 message_prefix += " \n";
1184                                 redraw = true;
1185                         }
1186                 }
1187

1201 {
1202         const events::command_disabler disable_commands;
1203         // Default return value.
1204         if (interrupted) {
1205                 *interrupted = false;
1206         }
1207 
1208         // Attempt moving.

1210

1212         config cn {
1213                 "stopped_early", mover.stopped_early(),
1214                 "final_hex_x", mover.final_hex().wml_x(),
1215                 "final_hex_y", mover.final_hex().wml_y(),
1216         };
1217         bool matches_replay = checkup_instance->local_checkup(cn,co);
1218         if(!matches_replay)
1219         {
1220                 replay::process_error("calculated movement destination (x="+ cn["final_hex_x"].str() +  " y=" + cn["final_hex_y"].str() +
1221                         ") didn't match the original destination(x="+ co["final_hex_x"].str() +  " y=" + co["final_hex_y"].str() + ")\n");
1222 
1223                 //TODO: move the unit by force to the desired destination with something like mover.reset_final_hex(co["x"], co["y"]);
1224         }
1225 
1226         // Bookkeeping, etc.
1227         // also fires the moveto event
1228         mover.post_move(undo_stack);
1229         if (show_move) {
1230                 mover.feedback();
1231         }
1232 
1233         // Set return value.
1234         if (interrupted) {
1235                 *interrupted = mover.interrupted();
1236         }
1237

1241 /**
1242  * Moves a unit across the board.
1243  *
1244  * This function handles actual movement, checking terrain costs as well as
1245  * things that might interrupt movement (e.g. ambushes). If the full path
1246  * cannot be reached this turn, the remainder is stored as the unit's "goto"
1247  * instruction. (The unit itself is whatever unit is at the beginning of the
1248  * supplied path.)
1249  *
1250  * @param[in]  steps                The route to be traveled. The unit to be moved is at the beginning of this route.
1251  * @param      undo_stack           If supplied, then either this movement will be added to the stack or the stack will be cleared.
1252  * @param[in]  continued_move       If set to true, this is a continuation of an earlier move (movement is not interrupted should units be spotted).
1253  * @param[in]  show_move            Controls whether or not the movement is animated for the player.
1254  * @param[out] interrupted          If supplied, then this is set to true if information was uncovered that warrants interrupting a chain of actions (and set to false otherwise).
1255  * @param[out] move_spectator       If supplied, this will be given the information uncovered by the move (and the unit's "goto" instruction will be preserved).
1256  *
1257  * @returns The number of hexes entered. This can safely be used as an index
1258  *          into @a steps to get the location where movement ended, provided
1259  *          @a steps is not empty (the return value is guaranteed to be less
1260  *          than steps.size() ).
1261  */
1262 size_t move_unit_and_record(const std::vector<map_location> &steps,
1263                  undo_list* undo_stack,
1264                  bool continued_move, bool show_move,
1265                  bool* interrupted,
1266                  move_unit_spectator* move_spectator)
1267 {
1268 
1269         // Avoid some silliness.
1270         if ( steps.size() < 2  ||  (steps.size() == 2 && steps.front() == steps.back()) ) {
1271                 DBG_NG << "Ignoring a unit trying to jump on its hex at " <<
1272                           ( steps.empty() ? map_location::null_location() : steps.front() ) << ".\n";
1273                 return 0;
1274         }
1275         //if we have no fog activated then we always skip sighted
1276         if(resources::gameboard->units().find(steps.front()) != resources::gameboard->units().end())
1277         {
1278                 const team &current_team = resources::gameboard->teams()[
1279                         resources::gameboard->units().find(steps.front())->side() - 1];
1280                 continued_move |= !current_team.fog_or_shroud();
1281         }
1282         const bool skip_ally_sighted = !preferences::interrupt_when_ally_sighted();
1283 
1284         // Evaluate this move.
1285         unit_mover mover(steps, move_spectator, continued_move, skip_ally_sighted);
1286         if ( !mover.check_expected_movement() )
1287                 return 0;
1288         if(synced_context::get_synced_state() != synced_context::SYNCED)
1289         {
1290                 /*
1291                         enter the synced mode and do the actual movement.
1292                 */
1293                 resources::recorder->add_synced_command("move",replay_helper::get_movement(steps, continued_move, skip_ally_sighted));
1294                 set_scontext_synced sync;
1295                 size_t r =  move_unit_internal(undo_stack, show_move, interrupted, mover);
1296                 resources::controller->check_victory();
1297                 resources::controller->maybe_throw_return_to_play_side();
1298                 sync.do_final_checkup();
1299                 return r;
1300         }
1301         else
1302         {
1303                 //we are already in synced mode and don't need to reenter it again.
1304                 return  move_unit_internal(undo_stack, show_move, interrupted, mover);
1305         }
1306 }

1308 size_t move_unit_from_replay(const std::vector<map_location> &steps,
1309                  undo_list* undo_stack,
1310                  bool continued_move,bool skip_ally_sighted, bool show_move)
1311 {
1312         // Evaluate this move.
1313         unit_mover mover(steps, nullptr, continued_move,skip_ally_sighted);
1314         if ( !mover.check_expected_movement() )
1315         {
1316                 replay::process_error("found corrupt movement in replay.");
1317                 return 0;
1318         }
1319 
1320         return move_unit_internal(undo_stack, show_move, nullptr, mover);
