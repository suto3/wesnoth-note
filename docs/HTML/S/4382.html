  46         local get_hp_efficiency = function (table, recruit_id)
  47             -- raw durability is a function of hp and the regenerates ability
  48             -- efficiency decreases faster than cost increases to avoid recruiting many expensive units
  49             -- there is a requirement for bodies in order to block movement
  50 
  51             -- There is currently an assumption that opponents will average about 15 damage per strike
  52             -- and that two units will attack per turn until the unit dies to estimate the number of hp
  53             -- gained from regeneration
  54             local effective_hp = wesnoth.unit_types[recruit_id].max_hitpoints
  55 
  56             local unit = wesnoth.create_unit {
  57                 type = recruit_id,
  58                 random_traits = false,
  59                 name = "X",
  60                 random_gender = false
  61             }
  62             -- Find the best regeneration ability and use it to estimate hp regained by regeneration
  63             local abilities = H.get_child(unit.__cfg, "abilities")
  64             local regen_amount = 0
  65             if abilities then
  66                 for regen in H.child_range(abilities, "regenerate") do
  67                     if regen.value > regen_amount then
  68                         regen_amount = regen.value
  69                     end
  70                 end
  71                 effective_hp = effective_hp + (regen_amount * effective_hp/30)
  72             end
  73             local hp_score = math.max(math.log(effective_hp/20),0.01)
  74             local efficiency = hp_score/(wesnoth.unit_types[recruit_id].cost^2)
  75             local no_village_efficiency = hp_score/(no_village_cost(recruit_id)^2)
  76 
  77             table[recruit_id] = {efficiency, no_village_efficiency}
  78             return {efficiency, no_village_efficiency}
  79         end

 105         function analyze_enemy_unit(enemy_type, ally_type)
 106             local function get_best_attack(attacker, defender, defender_defense, attacker_defense, can_poison)
 107                 -- Try to find the average damage for each possible attack and return the one that deals the most damage.
 108                 -- Would be preferable to call simulate combat, but that requires the defender to be on the map according
 109                 -- to documentation and we are looking for hypothetical situations so would have to search for available
 110                 -- locations for the defender that would have the desired defense. We would also need to remove nearby units
 111                 -- in order to ensure that adjacent units are not modifying the result. In addition, the time of day is
 112                 -- assumed to be neutral here, which is not assured in the simulation.
 113                 -- Ideally, this function would be a clone of simulate combat, but run for each time of day in the scenario and on arbitrary terrain.
 114                 -- In several cases this function only approximates the correct value (eg Thunderguard vs Goblin Spearman has damage capped by target health)
 115                 -- In some cases (like poison), this approximation is preferred to the actual value.
 116                 local best_damage = 0
 117                 local best_attack = nil
 118                 local best_poison_damage = 0
 119                 -- Steadfast is currently disabled because it biases the AI too much in favour of Guardsmen
 120                 -- Basically it sees the defender stats for damage and wrongfully concludes that the unit is amazing
 121                 -- This may be rectifiable by looking at retaliation damage as well.
 122                 local steadfast = false
 123 
 124                 for attack in H.child_range(wesnoth.unit_types[attacker.type].__cfg, "attack") do
 125                     local defense = defender_defense
 126                     local poison = false
 127                     local damage_multiplier = 1
 128                     local damage_bonus = 0
 129                     local weapon_damage = attack.damage
 130 
 131                     for special in H.child_range(attack, 'specials') do
 132                         local mod
 133                         if H.get_child(special, 'poison') and can_poison then
 134                             poison = true
 135                         end
 136 
 137                         -- Handle marksman and magical
 138                         mod = H.get_child(special, 'chance_to_hit')
 139                         if mod then
 140                             if mod.value then
 141                                 if mod.cumulative then
 142                                     if mod.value > defense then
 143                                         defense = mod.value
 144                                     end
 145                                 else
 146                                     defense = mod.value
 147                                 end
 148                             elseif mod.add then
 149                                 defense = defense + mod.add
 150                             elseif mod.sub then
 151                                 defense = defense - mod.sub
 152                             elseif mod.multiply then
 153                                 defense = defense * mod.multiply
 154                             elseif mod.divide then
 155                                 defense = defense / mod.divide
 156                             end
 157                         end
 158 
 159                         -- Handle most damage specials (assumes all are cumulative)
 160                         mod = H.get_child(special, 'damage')
 161                         if mod and mod.active_on ~= "defense" then
 162                             local special_multiplier = 1
 163                             local special_bonus = 0
 164 
 165                             if mod.multiply then
 166                                 special_multiplier = special_multiplier*mod.multiply
 167                             end
 168                             if mod.divide then
 169                                 special_multiplier = special_multiplier/mod.divide
 170                             end
 171                             if mod.add then
 172                                 special_bonus = special_bonus+mod.add
 173                             end
 174                             if mod.subtract then
 175                                 special_bonus = special_bonus-mod.subtract
 176                             end
 177 
 178                             if mod.backstab then
 179                                 -- Assume backstab happens on only 1/2 of attacks
 180                                 -- TODO: find out what actual probability of getting to backstab is
 181                                 damage_multiplier = damage_multiplier*(special_multiplier*0.5 + 0.5)
 182                                 damage_bonus = damage_bonus+(special_bonus*0.5)
 183                                 if mod.value ~= nil then
 184                                     weapon_damage = (weapon_damage+mod.value)/2
 185                                 end
 186                             else
 187                                 damage_multiplier = damage_multiplier*special_multiplier
 188                                 damage_bonus = damage_bonus+special_bonus
 189                                 if mod.value ~= nil then
 190                                     weapon_damage = mod.value
 191                                 end
 192                             end
 193                         end
 194                     end
 195 
 196                     -- Handle drain for defender
 197                     local drain_recovery = 0
 198                     for defender_attack in H.child_range(defender.__cfg, 'attack') do
 199                         if (defender_attack.range == attack.range) then
 200                             for special in H.child_range(defender_attack, 'specials') do
 201                                 if H.get_child(special, 'drains') and drainable(attacker) then
 202                                     -- TODO: calculate chance to hit
 203                                     -- currently assumes 50% chance to hit using supplied constant
 204                                     local attacker_resistance = wesnoth.unit_resistance(attacker, defender_attack.type)
 205                                     drain_recovery = (defender_attack.damage*defender_attack.number*attacker_resistance*attacker_defense/2)/10000
 206                                 end
 207                             end
 208                         end
 209                     end
 210 
 211                     defense = defense/100.0
 212                     local resistance = wesnoth.unit_resistance(defender, attack.type)
 213                     if steadfast and (resistance < 100) then
 214                         resistance = 100 - ((100 - resistance) * 2)
 215                         if (resistance < 50) then
 216                             resistance = 50
 217                         end
 218                     end
 219                     local base_damage = (weapon_damage+damage_bonus)*resistance*damage_multiplier
 220                     if (resistance > 100) then
 221                         base_damage = base_damage-1
 222                     end
 223                     base_damage = math.floor(base_damage/100 + 0.5)
 224                     if (base_damage < 1) and (attack.damage > 0) then
 225                         -- Damage is always at least 1
 226                         base_damage = 1
 227                     end
 228                     local attack_damage = base_damage*attack.number*defense-drain_recovery
 229 
 230                     local poison_damage = 0
 231                     if poison then
 232                         -- Add poison damage * probability of poisoning
 233                         poison_damage = 8*(1-((1-defense)^attack.number))
 234                     end
 235 
 236                     if (not best_attack) or (attack_damage+poison_damage > best_damage+best_poison_damage) then
 237                         best_damage = attack_damage
 238                         best_poison_damage = poison_damage
 239                         best_attack = attack
 240                     end
 241                 end
 242 
 243                 return best_attack, best_damage, best_poison_damage
 244             end
 245 
 246             -- Use cached information when possible: this is expensive
 247             local analysis = {}
 248             if not recruit_data.analyses then
 249                 recruit_data.analyses = {}
 250             else
 251                 if recruit_data.analyses[enemy_type] then
 252                     analysis = recruit_data.analyses[enemy_type] or {}
 253                 end
 254             end
 255             if analysis[ally_type] then
 256                 return analysis[ally_type]
 257             end
 258 
 259             local unit = wesnoth.create_unit {
 260                 type = enemy_type,
 261                 random_traits = false,
 262                 name = "X",
 263                 random_gender = false
 264             }
 265             local can_poison = poisonable(unit) and (not wesnoth.unit_ability(unit, 'regenerate'))
 266             local flat_defense = wesnoth.unit_defense(unit, "Gt")
 267             local best_defense = get_best_defense(unit)
 268 
 269             local recruit = wesnoth.create_unit {
 270                 type = ally_type,
 271                 random_traits = false,
 272                 name = "X",
 273                 random_gender = false
 274             }
 275             local recruit_flat_defense = wesnoth.unit_defense(recruit, "Gt")
 276             local recruit_best_defense = get_best_defense(recruit)
 277 
 278             local can_poison_retaliation = poisonable(recruit) and (not wesnoth.unit_ability(recruit, 'regenerate'))
 279             best_flat_attack, best_flat_damage, flat_poison = get_best_attack(recruit, unit, flat_defense, recruit_best_defense, can_poison)
 280             best_high_defense_attack, best_high_defense_damage, high_defense_poison = get_best_attack(recruit, unit, best_defense, recruit_flat_defense, can_poison)
 281             best_retaliation, best_retaliation_damage, retaliation_poison = get_best_attack(unit, recruit, recruit_flat_defense, best_defense, can_poison_retaliation)
 282 
 283             local result = {
 284                 offense = { attack = best_flat_attack, damage = best_flat_damage, poison_damage = flat_poison },
 285                 defense = { attack = best_high_defense_attack, damage = best_high_defense_damage, poison_damage = high_defense_poison },
 286                 retaliation = { attack = best_retaliation, damage = best_retaliation_damage, poison_damage = retaliation_poison }
 287             }
 288             analysis[ally_type] = result
 289 
 290             -- Cache result before returning
 291             recruit_data.analyses[enemy_type] = analysis
 292             return analysis[ally_type]
 293         end

 306         function get_hp_ratio_with_gold()
 307             function sum_gold_for_sides(side_filter)
 308                 -- sum positive amounts of gold for a set of sides
 309                 -- positive only because it is used to estimate the number of enemy units that could appear
 310                 -- and negative numbers shouldn't subtract from the number of units on the map
 311                 local gold = 0
 312                 local sides = wesnoth.get_sides(side_filter)
 313                 for i,s in ipairs(sides) do
 314                     if s.gold > 0 then
 315                         gold = gold + s.gold
 316                     end
 317                 end
 318 
 319                 return gold
 320             end
 321 
 322             -- Hitpoint ratio of own units / enemy units
 323             -- Also convert available gold to a hp estimate
 324             my_units = AH.get_live_units {
 325                 { "filter_side", {{"allied_with", {side = wesnoth.current.side} }} }
 326             }
 327             enemies = AH.get_live_units {
 328                 { "filter_side", {{"enemy_of", {side = wesnoth.current.side} }} }
 329             }
 330 
 331             local my_hp, enemy_hp = 0, 0
 332             for i,u in ipairs(my_units) do my_hp = my_hp + u.hitpoints end
 333             for i,u in ipairs(enemies) do enemy_hp = enemy_hp + u.hitpoints end
 334 
 335             my_hp = my_hp + sum_gold_for_sides({{"allied_with", {side = wesnoth.current.side} }})*2.3
 336             enemy_hp = enemy_hp+sum_gold_for_sides({{"enemy_of", {side = wesnoth.current.side} }})*2.3
 337             hp_ratio = my_hp/(enemy_hp + 1e-6)
 338 
 339             return hp_ratio
 340         end

 342         function do_recruit_eval(data)
 343             -- Check if leader is on keep
 344             local leader = wesnoth.get_units { side = wesnoth.current.side, canrecruit = 'yes' }[1]
 345 
 346             if (not leader) or (not wesnoth.get_terrain_info(wesnoth.get_terrain(leader.x, leader.y)).keep) then
 347                 return 0
 348             end
 349 
 350             -- Check if there is enough gold to recruit a unit
 351             local cheapest_unit_cost = AH.get_cheapest_recruit_cost()
 352             if cheapest_unit_cost > wesnoth.sides[wesnoth.current.side].gold then
 353                 return 0
 354             end
 355 
 356             -- Check for space to recruit a unit
 357             get_current_castle(leader, data)
 358             local no_space = true
 359             for i,c in ipairs(data.castle.locs) do
 360                 local unit = wesnoth.get_unit(c[1], c[2])
 361                 if (not AH.is_visible_unit(wesnoth.current.side, unit)) then
 362                     no_space = false
 363                     break
 364                 end
 365             end
 366             if no_space then
 367                 return 0
 368             end
 369 
 370             -- Check for minimal recruit option
 371             if wesnoth.current.turn == 1 and params.min_turn_1_recruit and params.min_turn_1_recruit() then
 372                 if not get_village_target(leader, data)[1] then
 373                     return 0
 374                 end
 375             end
 376 
 377             if data.recruit == nil then
 378                 data.recruit = init_data(leader)
 379             end
 380             data.recruit.cheapest_unit_cost = cheapest_unit_cost
 381 
 382             local score = 180000 -- default score if one not provided. Same as RCA AI
 383             if params.score_function then
 384                 score = params.score_function()
 385             end
 386             return score
 387         end

 389         function init_data(leader)
 390             local data = {}
 391 
 392             -- Count enemies of each type
 393             local enemies = AH.get_live_units {
 394                 { "filter_side", {{"enemy_of", {side = wesnoth.current.side} }}}
 395             }
 396             local enemy_counts = {}
 397             local enemy_types = {}
 398             local possible_enemy_recruit_count = 0
 399 
 400             local function add_unit_type(unit_type)
 401                 if enemy_counts[unit_type] == nil then
 402                     table.insert(enemy_types, unit_type)
 403                     enemy_counts[unit_type] = 1
 404                 else
 405                     enemy_counts[unit_type] = enemy_counts[unit_type] + 1
 406                 end
 407             end
 408 
 409             -- Collect all enemies on map
 410             for i, unit in ipairs(enemies) do
 411                 add_unit_type(unit.type)
 412             end
 413             -- Collect all possible enemy recruits and count them as virtual enemies
 414             local enemy_sides = wesnoth.get_sides({
 415                 { "enemy_of", {side = wesnoth.current.side} },
 416                 { "has_unit", { canrecruit = true }} })
 417             for i, side in ipairs(enemy_sides) do
 418                 possible_enemy_recruit_count = possible_enemy_recruit_count + #(wesnoth.sides[side.side].recruit)
 419                 for j, unit_type in ipairs(wesnoth.sides[side.side].recruit) do
 420                     add_unit_type(unit_type)
 421                 end
 422             end
 423             data.enemy_counts = enemy_counts
 424             data.enemy_types = enemy_types
 425             data.num_enemies = #enemies
 426             data.possible_enemy_recruit_count = possible_enemy_recruit_count
 427 
 428             return data
 429         end

 445         function ai_cas:recruit_rushers_exec()
 446             if AH.show_messages() then W.message { speaker = 'narrator', message = 'Recruiting' } end
 447 
 448             local enemy_counts = recruit_data.recruit.enemy_counts
 449             local enemy_types = recruit_data.recruit.enemy_types
 450             local num_enemies =  recruit_data.recruit.num_enemies
 451             local hp_ratio = get_hp_ratio_with_gold()
 452 
 453             -- Determine effectiveness of recruitable units against each enemy unit type
 454             local recruit_effectiveness = {}
 455             local recruit_vulnerability = {}
 456             local attack_type_count = {} -- The number of units who will likely use a given attack type
 457             local attack_range_count = {} -- The number of units who will likely use a given attack range
 458             local unit_attack_type_count = {} -- The attack types a unit will use
 459             local unit_attack_range_count = {} -- The ranges a unit will use
 460             local enemy_type_count = 0
 461             local poisoner_count = 0.1 -- Number of units with a poison attack (set to slightly > 0 because we divide by it later)
 462             local poisonable_count = 0 -- Number of units that the opponents control that are hurt by poison
 463             local recruit_count = {}
 464             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 465                 recruit_count[recruit_id] = #(AH.get_live_units { side = wesnoth.current.side, type = recruit_id, canrecruit = 'no' })
 466             end
 467 
 468             for i, unit_type in ipairs(enemy_types) do
 469                 enemy_type_count = enemy_type_count + 1
 470                 local poison_vulnerable = false
 471                 for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 472                     local analysis = analyze_enemy_unit(unit_type, recruit_id)
 473 
 474                     if recruit_effectiveness[recruit_id] == nil then
 475                         recruit_effectiveness[recruit_id] = {damage = 0, poison_damage = 0}
 476                         recruit_vulnerability[recruit_id] = 0
 477                     end
 478 
 479                     recruit_effectiveness[recruit_id].damage = recruit_effectiveness[recruit_id].damage + analysis.defense.damage * enemy_counts[unit_type]^2
 480                     if analysis.defense.poison_damage and analysis.defense.poison_damage > 0 then
 481                         poison_vulnerable = true
 482                         recruit_effectiveness[recruit_id].poison_damage = recruit_effectiveness[recruit_id].poison_damage +
 483                             analysis.defense.poison_damage * enemy_counts[unit_type]^2
 484                     end
 485                     recruit_vulnerability[recruit_id] = recruit_vulnerability[recruit_id] + (analysis.retaliation.damage * enemy_counts[unit_type])^3
 486 
 487                     local attack_type = analysis.defense.attack.type
 488                     if attack_type_count[attack_type] == nil then
 489                         attack_type_count[attack_type] = 0
 490                     end
 491                     attack_type_count[attack_type] = attack_type_count[attack_type] + recruit_count[recruit_id]
 492 
 493                     local attack_range = analysis.defense.attack.range
 494                     if attack_range_count[attack_range] == nil then
 495                         attack_range_count[attack_range] = 0
 496                     end
 497                     attack_range_count[attack_range] = attack_range_count[attack_range] + recruit_count[recruit_id]
 498 
 499                     if unit_attack_type_count[recruit_id] == nil then
 500                         unit_attack_type_count[recruit_id] = {}
 501                     end
 502                     unit_attack_type_count[recruit_id][attack_type] = true
 503 
 504                     if unit_attack_range_count[recruit_id] == nil then
 505                         unit_attack_range_count[recruit_id] = {}
 506                     end
 507                     unit_attack_range_count[recruit_id][attack_range] = true
 508                 end
 509                 if poison_vulnerable then
 510                     poisonable_count = poisonable_count + enemy_counts[unit_type]
 511                 end
 512             end
 513             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 514                 -- Count the number of units with the poison ability
 515                 -- This could be wrong if all the units on the enemy side are immune to poison, but since poison has no effect then anyway it doesn't matter
 516                 if recruit_effectiveness[recruit_id].poison_damage > 0 then
 517                     poisoner_count = poisoner_count + recruit_count[recruit_id]
 518                 end
 519             end
 520             -- Subtract the number of possible recruits for the enemy from the list of poisonable units
 521             -- This works perfectly unless some of the enemy recruits cannot be poisoned.
 522             -- However, there is no problem with this since poison is generally less useful in such situations and subtracting them too discourages such recruiting
 523             local poison_modifier = math.max(0, math.min(((poisonable_count-recruit_data.recruit.possible_enemy_recruit_count) / (poisoner_count*5)), 1))^2
 524             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 525                 -- Ensure effectiveness and vulnerability are positive.
 526                 -- Negative values imply that drain is involved and the amount drained is very high
 527                 if recruit_effectiveness[recruit_id].damage <= 0 then
 528                     recruit_effectiveness[recruit_id].damage = 0.01
 529                 else
 530                     recruit_effectiveness[recruit_id].damage = (recruit_effectiveness[recruit_id].damage / (num_enemies)^2)^0.5
 531                 end
 532                 recruit_effectiveness[recruit_id].poison_damage = (recruit_effectiveness[recruit_id].poison_damage / (num_enemies)^2)^0.5 * poison_modifier
 533                 if recruit_vulnerability[recruit_id] <= 0 then
 534                     recruit_vulnerability[recruit_id] = 0.01
 535                 else
 536                     recruit_vulnerability[recruit_id] = (recruit_vulnerability[recruit_id] / ((num_enemies)^2))^0.5
 537                 end
 538             end
 539             -- Correct count of units for each range
 540             local most_common_range = nil
 541             local most_common_range_count = 0
 542             for range, count in pairs(attack_range_count) do
 543                 attack_range_count[range] = count/enemy_type_count
 544                 if attack_range_count[range] > most_common_range_count then
 545                     most_common_range = range
 546                     most_common_range_count = attack_range_count[range]
 547                 end
 548             end
 549             -- Correct count of units for each attack type
 550             for attack_type, count in pairs(attack_type_count) do
 551                 attack_type_count[attack_type] = count/enemy_type_count
 552             end
 553 
 554             local recruit_type = nil
 555             local leader = wesnoth.get_units { side = wesnoth.current.side, canrecruit = 'yes' }[1]
 556             repeat
 557                 recruit_data.recruit.best_hex, recruit_data.recruit.target_hex = ai_cas:find_best_recruit_hex(leader, recruit_data)
 558                 recruit_type = ai_cas:find_best_recruit(attack_type_count, unit_attack_type_count, recruit_effectiveness, recruit_vulnerability, attack_range_count, unit_attack_range_count, most_common_range_count)
 559             until recruit_type ~= nil
 560 
 561             if wesnoth.unit_types[recruit_type].cost <= wesnoth.sides[wesnoth.current.side].gold then
 562                 AH.checked_recruit(ai, recruit_type, recruit_data.recruit.best_hex[1], recruit_data.recruit.best_hex[2])
 563 
 564                 -- If the recruited unit cannot reach the target hex, return it to the pool of targets
 565                 if recruit_data.recruit.target_hex ~= nil and recruit_data.recruit.target_hex[1] ~= nil then
 566                     local unit = wesnoth.get_unit(recruit_data.recruit.best_hex[1], recruit_data.recruit.best_hex[2])
 567                     local path, cost = wesnoth.find_path(unit, recruit_data.recruit.target_hex[1], recruit_data.recruit.target_hex[2], {viewing_side=0, max_cost=unit.max_moves+1})
 568                     if cost > unit.max_moves then
 569                         -- The last village added to the list should be the one we tried to aim for, check anyway
 570                         local last = #recruit_data.castle.assigned_villages_x
 571                         if (recruit_data.castle.assigned_villages_x[last] == recruit_data.recruit.target_hex[1]) and (recruit_data.castle.assigned_villages_y[last] == recruit_data.recruit.target_hex[2]) then
 572                             table.remove(recruit_data.castle.assigned_villages_x)
 573                             table.remove(recruit_data.castle.assigned_villages_y)
 574                         end
 575                     end
 576                 end
 577 
 578                 return true
 579             else
 580                 return false
 581             end
 582         end

 604         function ai_cas:find_best_recruit_hex(leader, data)
 605             -- Find the best recruit hex
 606             -- First choice: a hex that can reach an unowned village
 607             -- Second choice: a hex close to the enemy
 608             get_current_castle(leader, data)
 609 
 610             local best_hex, village = get_village_target(leader, data)
 611             if village[1] then
 612                 table.insert(data.castle.assigned_villages_x, village[1])
 613                 table.insert(data.castle.assigned_villages_y, village[2])
 614             else
 615                 -- no available village, look for hex closest to enemy leader
 616                 -- and also the closest enemy
 617                 local max_rating = -1
 618 
 619                 local enemy_leaders = AH.get_attackable_enemies { canrecruit = 'yes' }
 620                 local closest_enemy_distance, closest_enemy_location = AH.get_closest_enemy()
 621 
 622                 for i,c in ipairs(data.castle.locs) do
 623                     local rating = 0
 624                     local unit = wesnoth.get_unit(c[1], c[2])
 625                     if (not AH.is_visible_unit(wesnoth.current.side, unit)) then
 626                         for j,e in ipairs(enemy_leaders) do
 627                             rating = rating + 1 / M.distance_between(c[1], c[2], e.x, e.y) ^ 2.
 628                         end
 629                         if closest_enemy_location then
 630                             rating = rating + 1 / M.distance_between(c[1], c[2], closest_enemy_location.x, closest_enemy_location.y) ^ 2.
 631                         end
 632                         if (rating > max_rating) then
 633                             max_rating, best_hex = rating, { c[1], c[2] }
 634                         end
 635                     end
 636                 end
 637             end
 638 
 639             if AH.print_eval() then
 640                 if village[1] then
 641                     print("Recruit at: " .. best_hex[1] .. "," .. best_hex[2] .. " -> " .. village[1] .. "," .. village[2])
 642                 else
 643                     print("Recruit at: " .. best_hex[1] .. "," .. best_hex[2])
 644                 end
 645             end
 646             return best_hex, village
 647         end

 649         function ai_cas:find_best_recruit(attack_type_count, unit_attack_type_count, recruit_effectiveness, recruit_vulnerability, attack_range_count, unit_attack_range_count, most_common_range_count)
 650             -- Find best recruit based on damage done to enemies present, speed, and hp/gold ratio
 651             local recruit_scores = {}
 652             local best_scores = {offense = 0, defense = 0, move = 0}
 653             local best_hex = recruit_data.recruit.best_hex
 654             local target_hex = recruit_data.recruit.target_hex
 655             local distance_to_enemy, enemy_location
 656             if target_hex[1] then
 657                 distance_to_enemy, enemy_location = AH.get_closest_enemy(target_hex, wesnoth.current.side, { viewing_side = 0 })
 658             else
 659                 distance_to_enemy, enemy_location = AH.get_closest_enemy(best_hex, wesnoth.current.side, { viewing_side = 0 })
 660             end
 661 
 662             local gold_limit = 9e99
 663             if recruit_data.castle.loose_gold_limit >= recruit_data.recruit.cheapest_unit_cost then
 664                 gold_limit = recruit_data.castle.loose_gold_limit
 665             end
 666             --print (recruit_data.castle.loose_gold_limit .. " " .. recruit_data.recruit.cheapest_unit_cost .. " " .. gold_limit)
 667 
 668             local recruitable_units = {}
 669 
 670             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 671                 -- Count number of units with the same attack type. Used to avoid recruiting too many of the same unit
 672                 local attack_types = 0
 673                 local recruit_count = 0
 674                 for attack_type, count in pairs(unit_attack_type_count[recruit_id]) do
 675                     attack_types = attack_types + 1
 676                     recruit_count = recruit_count + attack_type_count[attack_type]
 677                 end
 678                 recruit_count = recruit_count / attack_types
 679                 local recruit_modifier = 1+recruit_count/50
 680                 local efficiency_index = 1
 681                 local unit_cost = wesnoth.unit_types[recruit_id].cost
 682 
 683                 -- Use time to enemy to encourage recruiting fast units when the opponent is far away (game is beginning or we're winning)
 684                 -- Base distance on
 685                 local recruit_unit = wesnoth.create_unit {
 686                     type = recruit_id,
 687                     x = best_hex[1],
 688                     y = best_hex[2],
 689                     random_traits = false,
 690                     name = "X",
 691                     random_gender = false
 692                 }
 693                 if target_hex[1] then
 694                     local path, cost = wesnoth.find_path(recruit_unit, target_hex[1], target_hex[2], {viewing_side=0, max_cost=wesnoth.unit_types[recruit_id].max_moves+1})
 695                     if cost > wesnoth.unit_types[recruit_id].max_moves then
 696                         -- Unit cost is effectively higher if cannot reach the village
 697                         efficiency_index = 2
 698                         unit_cost = no_village_cost(recruit_id)
 699                     end
 700 
 701                     -- Later calculations are based on where the unit will be after initial move
 702                     recruit_unit.x = target_hex[1]
 703                     recruit_unit.y = target_hex[2]
 704                 end
 705 
 706                 local path, cost = wesnoth.find_path(recruit_unit, enemy_location.x, enemy_location.y, {ignore_units = true})
 707                 local time_to_enemy = cost / wesnoth.unit_types[recruit_id].max_moves
 708                 local move_score = 1 / (time_to_enemy * unit_cost^0.5)
 709 
 710                 local eta = math.ceil(time_to_enemy)
 711                 if target_hex[1] then
 712                     -- expect a 1 turn delay to reach village
 713                     eta = eta + 1
 714                 end
 715                 -- divide the lawful bonus by eta before running it through the function because the function converts from 0 centered to 1 centered
 716 
 717                 local lawful_bonus = 0
 718                 local eta_turn = wesnoth.current.turn + eta
 719                 if eta_turn <= wesnoth.game_config.last_turn then
 720                     lawful_bonus = wesnoth.get_time_of_day(wesnoth.current.turn + eta).lawful_bonus / eta^2
 721                 end
 722                 local damage_bonus = AH.get_unit_time_of_day_bonus(recruit_unit.__cfg.alignment, lawful_bonus)
 723                 -- Estimate effectiveness on offense and defense
 724                 local offense_score =
 725                     (recruit_effectiveness[recruit_id].damage*damage_bonus+recruit_effectiveness[recruit_id].poison_damage)
 726                     /(wesnoth.unit_types[recruit_id].cost^0.3*recruit_modifier^4)
 727                 local defense_score = efficiency[recruit_id][efficiency_index]/recruit_vulnerability[recruit_id]
 728 
 729                 local unit_score = {offense = offense_score, defense = defense_score, move = move_score}
 730                 recruit_scores[recruit_id] = unit_score
 731                 for key, score in pairs(unit_score) do
 732                     if score > best_scores[key] then
 733                         best_scores[key] = score
 734                     end
 735                 end
 736 
 737                 if can_slow(recruit_unit) then
 738                     unit_score["slows"] = true
 739                 end
 740                 if wesnoth.match_unit(recruit_unit, { ability = "healing" }) then
 741                     unit_score["heals"] = true
 742                 end
 743                 if wesnoth.match_unit(recruit_unit, { ability = "skirmisher" }) then
 744                     unit_score["skirmisher"] = true
 745                 end
 746                 recruitable_units[recruit_id] = recruit_unit
 747             end
 748             local healer_count, healable_count = get_unit_counts_for_healing()
 749             local best_score = 0
 750             local recruit_type = nil
 751             local offense_weight = 2.5
 752             local defense_weight = 1/hp_ratio^0.5
 753             local move_weight = math.max((distance_to_enemy/20)^2, 0.25)
 754             local randomness = params.randomness or 0.1
 755             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 756                 local scores = recruit_scores[recruit_id]
 757                 local offense_score = (scores["offense"]/best_scores["offense"])^0.5
 758                 local defense_score = (scores["defense"]/best_scores["defense"])^0.5
 759                 local move_score = (scores["move"]/best_scores["move"])^0.5
 760 
 761                 local bonus = math.random()*randomness
 762                 if scores["slows"] then
 763                     bonus = bonus + 0.4
 764                 end
 765                 if scores["heals"] then
 766                     bonus = bonus + (healable_count/(healer_count+1))/20
 767                 end
 768                 if scores["skirmisher"] then
 769                     bonus = bonus + 0.1
 770                 end
 771                 for attack_range, count in pairs(unit_attack_range_count[recruit_id]) do
 772                     bonus = bonus + 0.02 * most_common_range_count / (attack_range_count[attack_range]+1)
 773                 end
 774                 bonus = bonus + 0.03 * wesnoth.races[wesnoth.unit_types[recruit_id].__cfg.race].num_traits^2
 775                 if target_hex[1] then
 776                     recruitable_units[recruit_id].x = best_hex[1]
 777                     recruitable_units[recruit_id].y = best_hex[2]
 778                     local path, cost = wesnoth.find_path(recruitable_units[recruit_id], target_hex[1], target_hex[2], {viewing_side=0, max_cost=wesnoth.unit_types[recruit_id].max_moves+1})
 779                     if cost > wesnoth.unit_types[recruit_id].max_moves then
 780                         -- penalty if the unit can't reach the target village
 781                         bonus = bonus - 0.2
 782                     end
 783                 end
 784 
 785                 local score = offense_score*offense_weight + defense_score*defense_weight + move_score*move_weight + bonus
 786 
 787                 if AH.print_eval() then
 788                     print(recruit_id .. " score: " .. offense_score*offense_weight .. " + " .. defense_score*defense_weight .. " + " .. move_score*move_weight  .. " + " .. bonus  .. " = " .. score)
 789                 end
 790                 if score > best_score and wesnoth.unit_types[recruit_id].cost <= gold_limit then
 791                     best_score = score
 792                     recruit_type = recruit_id
 793                 end
 794             end
 795 
 796             return recruit_type
 797         end

 812         function get_village_target(leader, data)
 813             -- Only consider villages reachable by our fastest unit
 814             local fastest_unit_speed = 0
 815             for i, recruit_id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do
 816                 if wesnoth.unit_types[recruit_id].max_moves > fastest_unit_speed then
 817                     fastest_unit_speed = wesnoth.unit_types[recruit_id].max_moves
 818                 end
 819             end
 820 
 821             local locsx, locsy = AH.split_location_list_to_strings(data.castle.locs)
 822 
 823             -- get a list of all unowned villages within fastest_unit_speed
 824             -- TODO get list of villages not owned by allies instead
 825             -- this may have false positives (villages that can't be reached due to difficult/impassible terrain)
 826             local exclude_x, exclude_y = "0", "0"
 827             if data.castle.assigned_villages_x ~= nil and data.castle.assigned_villages_x[1] then
 828                 exclude_x = table.concat(data.castle.assigned_villages_x, ",")
 829                 exclude_y = table.concat(data.castle.assigned_villages_y, ",")
 830             end
 831             local villages = wesnoth.get_villages {
 832                 owner_side = 0,
 833                 { "and", {
 834                     radius = fastest_unit_speed,
 835                     x = locsx, y = locsy
 836                 }},
 837                 { "not", {
 838                     x = exclude_x,
 839                     y = exclude_y
 840                 }}
 841             }
 842 
 843             local hex, target, shortest_distance = {}, {}, AH.no_path
 844 
 845             if not data.castle.assigned_villages_x then
 846                 data.castle.assigned_villages_x = {}
 847                 data.castle.assigned_villages_y = {}
 848 
 849                 if not params.leader_takes_village or params.leader_takes_village() then
 850                     -- skip one village for the leader
 851                     for i,v in ipairs(villages) do
 852                         local path, cost = wesnoth.find_path(leader, v[1], v[2], {max_cost = leader.max_moves+1})
 853                         if cost <= leader.max_moves then
 854                             table.insert(data.castle.assigned_villages_x, v[1])
 855                             table.insert(data.castle.assigned_villages_y, v[2])
 856                             table.remove(villages, i)
 857                             break
 858                         end
 859                     end
 860                 end
 861             end
 862 
 863             local village_count = #villages
 864             local test_units = get_test_units()
 865             local num_recruits = #test_units
 866             local total_village_distance = {}
 867             for j,c in ipairs(data.castle.locs) do
 868                 c_index = c[1] + c[2]*1000
 869                 total_village_distance[c_index] = 0
 870                 for i,v in ipairs(villages) do
 871                     total_village_distance[c_index] = total_village_distance[c_index] + M.distance_between(c[1], c[2], v[1], v[2])
 872                 end
 873             end
 874 
 875             local width,height,border = wesnoth.get_map_size()
 876             for i,v in ipairs(villages) do
 877                 local close_castle_hexes = wesnoth.get_locations {
 878                     x = locsx, y = locsy,
 879                     { "and", {
 880                       x = v[1], y = v[2],
 881                       radius = fastest_unit_speed
 882                     }},
 883                     { "not", { { "filter", {} } } }
 884                 }
 885                 for u,unit in ipairs(test_units) do
 886                     test_units[u].x = v[1]
 887                     test_units[u].y = v[2]
 888                 end
 889 
 890                 local viable_village = false
 891                 local village_best_hex, village_shortest_distance = {}, AH.no_path
 892                 for j,c in ipairs(close_castle_hexes) do
 893                     if c[1] > 0 and c[2] > 0 and c[1] <= width and c[2] <= height then
 894                         local distance = 0
 895                         for x,unit in ipairs(test_units) do
 896                             local path, unit_distance = wesnoth.find_path(unit, c[1], c[2], {viewing_side=0, max_cost=fastest_unit_speed+1})
 897                             distance = distance + unit_distance
 898 
 899                             -- Village is only viable if at least one unit can reach it
 900                             if unit_distance <= unit.max_moves then
 901                                 viable_village = true
 902                             end
 903                         end
 904                         distance = distance / num_recruits
 905 
 906                         if distance < village_shortest_distance
 907                         or (distance == village_shortest_distance and distance < AH.no_path
 908                             and total_village_distance[c[1] + c[2]*1000] > total_village_distance[village_best_hex[1]+village_best_hex[2]*1000])
 909                         then
 910                             village_best_hex = c
 911                             village_shortest_distance = distance
 912                         end
 913                     end
 914                 end
 915                 if village_shortest_distance < shortest_distance then
 916                     hex = village_best_hex
 917                     target = v
 918                     shortest_distance = village_shortest_distance
 919                 end
 920 
 921                 if not viable_village then
 922                     -- this village could not be reached by any unit
 923                     -- eliminate it from consideration
 924                     table.insert(data.castle.assigned_villages_x, v[1])
 925                     table.insert(data.castle.assigned_villages_y, v[2])
 926                     village_count = village_count - 1
 927                 end
 928             end
 929 
 930             data.castle.loose_gold_limit = math.floor(wesnoth.sides[wesnoth.current.side].gold/village_count + 0.5)
 931 
 932             return hex, target
 933         end

 935         function get_test_units()
 936             local test_units, num_recruits = {}, 0
 937             local movetypes = {}
 938             for x,id in ipairs(wesnoth.sides[wesnoth.current.side].recruit) do

 946                         movetypes[movetype] = wesnoth.unit_types[id].max_moves
 947                     end
 948                     num_recruits = num_recruits + 1
 949                     test_units[num_recruits] =  wesnoth.create_unit({
 950                         type = id,
 951                         side = wesnoth.current.side,
 952                         random_traits = false,
 953                         name = "X",
 954                         random_gender = false
 955                     })
 956                 end
 957             end
 958 
 959             return test_units
 960         end
