  34 inline void sdl_copy_portion(const surface& screen, SDL_Rect* screen_rect, surface& dst, SDL_Rect* dst_rect){
  35         SDL_SetSurfaceBlendMode(screen, SDL_BLENDMODE_NONE);
  36         SDL_SetSurfaceBlendMode(dst, SDL_BLENDMODE_NONE);
  37         SDL_BlitSurface(screen, screen_rect, dst, dst_rect);
  38         SDL_SetSurfaceBlendMode(screen, SDL_BLENDMODE_BLEND);

 283 /**
 284  * Replacement for sdl_blit.
 285  *
 286  * sdl_blit has problems with blitting partly transparent surfaces so
 287  * this is a replacement. It ignores the SDL_SRCALPHA and SDL_SRCCOLORKEY
 288  * flags. src and dst will have the SDL_RLEACCEL flag removed.
 289  * The return value of SDL_BlistSurface is normally ignored so no return value.
 290  * The rectangles are const and will not be modified.
 291  *
 292  * @pre @p src contains a valid canvas.
 293  * @pre @p dst contains a valid neutral canvas.
 294  * @pre The caller must make sure the @p src fits on the @p dst.
 295  *
 296  * @param src          The surface to blit.
 297  * @param srcrect      The region of the surface to blit
 298  * @param dst          The surface to blit on.
 299  * @param dstrect      The offset to blit the surface on, only x and y are used.
