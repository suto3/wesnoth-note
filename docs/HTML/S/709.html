  46 BOOST_AUTO_TEST_CASE( test_insertion )
  47 {
  48         side_actions_container sac;
  49         std::shared_ptr<dummy_action> dact;
  50 
  51         // Basic insertions
  52         std::shared_ptr<dummy_action> act1(new dummy_action(0, false, 1));
  53         std::shared_ptr<dummy_action> act2(new dummy_action(0, false, 2));
  54         std::shared_ptr<dummy_action> act3(new dummy_action(0, false, 3));
  55 
  56         sac.queue(0, act2);
  57         sac.queue(0, act3);
  58         sac.insert(sac.begin(), act1);
  59 
  60         BOOST_REQUIRE(sac.num_turns() == 1);
  61 
  62         int tmp=0;
  63         for(action_ptr act : sac) {
  64                 ++tmp;
  65                 BOOST_REQUIRE(dact = std::dynamic_pointer_cast<dummy_action>(act));
  66                 BOOST_REQUIRE(dact->id_ == tmp);
  67         }
  68 
  69         // Multi-turn insertions
  70         std::shared_ptr<dummy_action> act4(new dummy_action(0, false, 4));
  71         std::shared_ptr<dummy_action> act5(new dummy_action(0, false, 5));
  72         std::shared_ptr<dummy_action> act6(new dummy_action(0, false, 6));
  73         std::shared_ptr<dummy_action> act7(new dummy_action(0, false, 7));
  74         std::shared_ptr<dummy_action> act8(new dummy_action(0, false, 8));
  75         sac.queue(1, act5);
  76         sac.queue(2, act8);
  77         sac.queue(1, act7);
  78         sac.queue(0, act4);
  79         sac.insert(sac.turn_begin(1)+1, act6);
  80 
  81         BOOST_REQUIRE(sac.num_turns() == 3);
  82 
  83         tmp=0;
  84         for(action_ptr act : sac) {
  85                 ++tmp;
  86                 BOOST_REQUIRE(dact = std::dynamic_pointer_cast<dummy_action>(act));
  87                 BOOST_REQUIRE(dact->id_ == tmp);
  88         }
  89 
  90         BOOST_REQUIRE(dact = std::dynamic_pointer_cast<dummy_action>(*sac.turn_begin(1)));
  91         BOOST_REQUIRE(dact->id_ == 5);
  92 
  93         BOOST_REQUIRE(dact = std::dynamic_pointer_cast<dummy_action>(*(1+sac.turn_begin(1))));
  94         BOOST_REQUIRE(dact->id_ == 6);
  95 
  96         BOOST_REQUIRE(sac.turn_size(1) == 3);
  97         BOOST_REQUIRE(3+sac.turn_begin(1) == sac.turn_end(1));
  98 }

 100 BOOST_AUTO_TEST_CASE( test_removal )
 101 {
 102         side_actions_container sac;
 103         std::shared_ptr<dummy_action> dact;
 104 
 105         std::shared_ptr<dummy_action> act1(new dummy_action(0, false, 1));
 106         std::shared_ptr<dummy_action> act2(new dummy_action(0, false, 2));
 107         std::shared_ptr<dummy_action> act3(new dummy_action(0, false, 3));
 108         std::shared_ptr<dummy_action> act4(new dummy_action(0, false, 4));
 109         std::shared_ptr<dummy_action> act5(new dummy_action(0, false, 5));
 110         std::shared_ptr<dummy_action> act6(new dummy_action(0, false, 6));
 111 
 112         sac.queue(0, act1);
 113         side_actions::iterator ite2 = sac.queue(0, act2);
 114         sac.queue(0, act3);
 115         side_actions::iterator ite4 = sac.queue(1, act4);
 116         sac.queue(1, act5);
 117         side_actions::iterator ite6 = sac.queue(2, act6);
 118 
 119         sac.erase(ite2);
 120         sac.erase(ite4);
 121         sac.erase(ite6);
 122 
 123         BOOST_REQUIRE(sac.num_turns() == 2);
 124         side_actions::iterator it = sac.begin();
 125         for(int i=1; i<6; i+=2, ++it){
 126                 BOOST_REQUIRE(dact = std::dynamic_pointer_cast<dummy_action>(*it));
 127                 BOOST_REQUIRE(dact->id_ == i);
 128         }
