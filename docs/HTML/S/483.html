  92 void unit_create::pre_show(window& window)
  93 {
  94         toggle_button& male_toggle
  95                         = find_widget<toggle_button>(&window, "male_toggle", false);
  96         toggle_button& female_toggle
  97                         = find_widget<toggle_button>(&window, "female_toggle", false);
  98 
  99         gender_toggle.add_member(&male_toggle, unit_race::MALE);
 100         gender_toggle.add_member(&female_toggle, unit_race::FEMALE);
 101 
 102         gender_toggle.set_member_states(last_gender);
 103 
 104         gender_toggle.set_callback_on_value_change(
 105                         dialog_callback<unit_create, &unit_create::gender_toggle_callback>);
 106 
 107         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 108 
 109         text_box* filter
 110                         = find_widget<text_box>(&window, "filter_box", false, true);
 111 
 112         filter->set_text_changed_callback(
 113                         std::bind(&unit_create::filter_text_changed, this, _1, _2));
 114 
 115         window.keyboard_capture(filter);
 116         window.add_to_keyboard_chain(&list);
 117 
 118 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 119         connect_signal_notify_modified(*list,
 120                                                                    std::bind(&unit_create::list_item_clicked,
 121                                                                                            *this,
 122                                                                                            std::ref(window)));
 123 #else
 124         list.set_callback_value_change(
 125                         dialog_callback<unit_create, &unit_create::list_item_clicked>);
 126 #endif
 127 
 128         list.clear();
 129 
 130         for(const auto & i : unit_types.types())
 131         {
 132                 if(i.second.do_not_list())
 133                         continue;
 134 
 135                 // Make sure this unit type is built with the data we need.
 136                 unit_types.build_unit_type(i.second, unit_type::FULL);
 137 
 138                 units_.push_back(&i.second);
 139 
 140                 std::map<std::string, string_map> row_data;
 141                 string_map column;
 142 
 143                 column["label"] = units_.back()->race()->plural_name();
 144                 row_data.emplace("race", column);
 145 
 146                 column["label"] = units_.back()->type_name();
 147                 row_data.emplace("unit_type", column);
 148 
 149                 list.add_row(row_data);
 150 
 151                 // Select the previous choice, if any.
 152                 if(choice_.empty() != true && choice_ == i.first) {
 153                         list.select_row(list.get_item_count() - 1);
 154                 }
 155         }
 156 
 157         if(units_.empty()) {
 158                 ERR_GUI_G << "no unit types found for unit create dialog; not good"
 159                                   << std::endl;
 160         }
 161 
 162         list.register_sorting_option(0, [this](const int i) { return (*units_[i]).race()->plural_name().str(); });
 163         list.register_sorting_option(1, [this](const int i) { return (*units_[i]).type_name().str(); });
 164 
 165         list_item_clicked(window);
 166 }

 168 void unit_create::post_show(window& window)
 169 {
 170         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 171 
 172         choice_ = "";
 173 
 174         if(get_retval() != window::OK) {
 175                 return;
 176         }
 177 
 178         const int selected_row = list.get_selected_row();
 179         if(selected_row < 0) {
 180                 return;
 181         } else if(static_cast<size_t>(selected_row) >= units_.size()) {
 182                 // FIXME: maybe assert?
 183                 ERR_GUI_G << "unit create dialog has more list items than known unit "
 184                                          "types; not good\n";
 185                 return;
 186         }
 187 
 188         last_chosen_type_id = choice_ = units_[selected_row]->id();
 189         last_gender = gender_;
 190 }

 192 void unit_create::list_item_clicked(window& window)
 193 {
 194         const int selected_row
 195                 = find_widget<listbox>(&window, "unit_type_list", false).get_selected_row();
 196 
 197         if(selected_row == -1) {
 198                 return;
 199         }
 200 
 201         find_widget<unit_preview_pane>(&window, "unit_details", false)
 202                 .set_displayed_type(*units_[selected_row]);
 203 
 204         gender_toggle.set_members_enabled([&](const unit_race::GENDER& gender)->bool {
 205                 return units_[selected_row]->has_gender_variation(gender);
 206         });
 207 }

 209 void unit_create::filter_text_changed(text_box_base* textbox, const std::string& text)
 210 {
 211         window& window = *textbox->get_window();
 212 
 213         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 214 
 215         const std::vector<std::string> words = utils::split(text, ' ');
 216 
 217         if(words == last_words_)
 218                 return;
 219         last_words_ = words;
 220 
 221         boost::dynamic_bitset<> show_items;
 222         show_items.resize(list.get_item_count(), true);
 223 
 224         if(!text.empty()) {
 225                 for(unsigned int i = 0; i < list.get_item_count(); i++) {
 226                         grid* row = list.get_row_grid(i);
 227 
 228                         grid::iterator it = row->begin();
 229                         label& type_label
 230                                         = find_widget<label>(*it, "unit_type", false);
 231 
 232                         bool found = false;
 233                         for(const auto & word : words)
 234                         {
 235                                 found = std::search(type_label.get_label().str().begin(),
 236                                                                         type_label.get_label().str().end(),
 237                                                                         word.begin(),
 238                                                                         word.end(),
 239                                                                         chars_equal_insensitive)
 240                                                 != type_label.get_label().str().end();
 241 
 242                                 if(!found) {
 243                                         // one word doesn't match, we don't reach words.end()
 244                                         break;
 245                                 }
 246                         }
 247 
 248                         show_items[i] = found;
 249                 }
 250         }
 251 
 252         list.set_row_shown(show_items);
 253 }
