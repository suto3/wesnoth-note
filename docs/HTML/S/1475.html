 202     /**
 203      * Downloads the currently selected addon
 204      */
 205     protected void downloadAddon( )
 206     {
 207         if( tableAddons_.getSelectionIndex( ) == - 1 ) {
 208             GUIUtils.showErrorMessageBox( "No addon selected" );
 209             return;
 210         }
 211 
 212         TableItem[] selection = tableAddons_.getSelection( );
 213         final String addonName = selection[0].getText( 1 );
 214 
 215         WorkspaceJob downloadJob = new WorkspaceJob( "Download" ) {
 216 
 217             @Override
 218             public IStatus runInWorkspace( final IProgressMonitor monitor )
 219                 throws CoreException
 220             {
 221                 monitor.beginTask( "Downloading addon " + addonName, 100 );
 222 
 223                 String installName = Preferences.getDefaultInstallName( );
 224 
 225                 RunnableWithResult< String > runnable = new RunnableWithResult< String >( ) {
 226                     @Override
 227                     public void run( )
 228                     {
 229                         // ask the user to select the install for the project
 230                         SelectWesnothInstallDialog dialog = new SelectWesnothInstallDialog(
 231                             null );
 232                         if( dialog.open( ) == SWT.OK ) {
 233                             setResult( dialog.getSelectedInstallName( ) );
 234                         }
 235                     }
 236                 };
 237 
 238                 Display.getDefault( ).syncExec( runnable );
 239                 if( ! StringUtils.isNullOrEmpty( runnable.getResult( ) ) ) {
 240                     installName = runnable.getResult( );
 241                 }
 242 
 243                 final Paths paths = Preferences.getPaths( installName );
 244 
 245                 OutputStream console = GUIUtils.createConsole(
 246                     "Wesnoth Addon Manager", null, false )
 247                     .newOutputStream( );
 248 
 249                 ExternalToolInvoker tool = WMLTools.runWesnothAddonManager(
 250                     installName,
 251                     null,
 252                     currentPort_,
 253                     Arrays.asList( "-d", addonName, "-c",
 254                         paths.getAddonsDir( ) ),
 255                     new OutputStream[] { console },
 256                     new OutputStream[] { console } );
 257 
 258                 tool.waitForTool( );
 259 
 260                 monitor.worked( 50 );
 261 
 262                 // ask user if he wants to create a project
 263                 if( GUIUtils
 264                     .showMessageBox(
 265                         "Do you want to create a new project for the downloaded addon?",
 266                         SWT.YES | SWT.NO ) == SWT.YES ) {
 267 
 268                     ProjectUtils.createWesnothProject( addonName,
 269                         paths.getAddonsDir( ) + addonName, installName,
 270                         monitor );
 271                 }
 272                 monitor.done( );
 273 
 274                 return Status.OK_STATUS;
 275             }
 276         };
 277         downloadJob.schedule( );
 278     }

 280     /**
 281      * Refreshes the list of addons
 282      */
 283     protected void refreshAddons( )
 284     {
 285         if( cmbAddonServer_ == null || tableAddons_ == null ) {
 286             return;
 287         }
 288 
 289         if( loading_ ) {
 290             GUIUtils
 291                 .showInfoMessageBox( "Please wait for the previous query to finish." );
 292             return;
 293         }
 294 
 295         if( cmbAddonServer_.getSelectionIndex( ) == - 1 ) {
 296             currentPort_ = cmbAddonServer_.getText( );
 297         }
 298         else {
 299             currentPort_ = ports_.get( cmbAddonServer_.getSelectionIndex( ) );
 300         }
 301 
 302         if( StringUtils.isNullOrEmpty( currentPort_ ) ) {
 303             return;
 304         }
 305 
 306         int portNumber;
 307         try {
 308             portNumber = Integer.parseInt( currentPort_ );
 309         } catch( NumberFormatException e ) {
 310             GUIUtils.showInfoMessageBox( "Invalid port: " + currentPort_ );
 311             return;
 312         }
 313 
 314         currentPort_ = String.valueOf( portNumber );
 315 
 316         loading_ = true;
 317         tableAddons_.setItemCount( 0 );
 318         tableAddons_.clearAll( );
 319 
 320         WorkspaceJob loadAddons = new WorkspaceJob( "Retrieving list..." ) {
 321 
 322             @Override
 323             public IStatus runInWorkspace( IProgressMonitor monitor )
 324                 throws CoreException
 325             {
 326                 monitor.beginTask( "Retrieving list...", 100 );
 327                 monitor.worked( 10 );
 328 
 329                 String installName = Preferences.getDefaultInstallName( );
 330 
 331                 OutputStream stderr = GUIUtils.createConsole(
 332                     "Wesnoth Addon Manager", null, false )
 333                     .newOutputStream( );
 334 
 335                 ExternalToolInvoker tool = WMLTools.runWesnothAddonManager(
 336                     installName, null, currentPort_,
 337                     Arrays.asList( "-w", "-l" ), // list addons in raw
 338                                                  // mode
 339                     null, new OutputStream[] { stderr } );
 340                 tool.waitForTool( );
 341 
 342                 /**
 343                  * parse the contents columns = [[" 0 - type", "1 - name",
 344                  * "2 - title", "3 - author", "4 - version", "uploads",
 345                  * "5 - downloads", "size", "timestamp", "translate"]]
 346                  */
 347                 final String[] lines = StringUtils.getLines( tool
 348                     .getOutputContent( ) );
 349                 final List< String[] > addons = new ArrayList< String[] >( );
 350 
 351                 String[] tmpColumns = null;
 352                 int index = - 1;
 353 
 354                 for( String line: lines ) {
 355                     index = - 1;
 356 
 357                     if( line.startsWith( "\\ campaign" ) ) {
 358                         if( tmpColumns != null ) {
 359                             addons.add( tmpColumns );
 360                         }
 361                         tmpColumns = new String[6];
 362                     }
 363                     else if( line.startsWith( "  \\ type" ) ) {
 364                         index = 0;
 365                     }
 366                     else if( line.startsWith( "  \\ name" ) ) {
 367                         index = 1;
 368                     }
 369                     else if( line.startsWith( "  \\ title" ) ) {
 370                         index = 2;
 371                     }
 372                     else if( line.startsWith( "  \\ author" ) ) {
 373                         index = 3;
 374                     }
 375                     else if( line.startsWith( "  \\ version" ) ) {
 376                         index = 4;
 377                     }
 378                     else if( line.startsWith( "  \\ downloads" ) ) {
 379                         index = 5;
 380                     }
 381 
 382                     // got something interesting? parse it
 383                     if( tmpColumns != null && index != - 1 ) {
 384                         int firstIndex = line.indexOf( '\'' ) + 1;
 385                         int lastIndex = line.lastIndexOf( '\'' );
 386 
 387                         if( lastIndex < firstIndex ) {
 388                             lastIndex = line.length( );
 389                         }
 390 
 391                         tmpColumns[index] = line.substring(
 392                             firstIndex,
 393                             lastIndex ).trim( );
 394                     }
 395                 }
 396 
 397                 // need GUI Thread access
 398                 Display.getDefault( ).syncExec( new Runnable( ) {
 399                     @Override
 400                     public void run( )
 401                     {
 402                         // skipp 1st line since it's just the header
 403                         for( String[] addon: addons ) {
 404 
 405                             TableItem tableItem = new TableItem(
 406                                 tableAddons_, SWT.NONE );
 407                             tableItem.setText( new String[] { addon[0],
 408                                 addon[1], addon[2], addon[3], addon[4],
 409                                 addon[5] } );
 410 
 411                         }
 412 
 413                         tableAddons_.forceFocus( );
 414                     }
 415                 } );
 416                 loading_ = false;
 417 
 418                 monitor.worked( 100 );
 419                 monitor.done( );
 420                 return Status.OK_STATUS;
 421             }
 422         };
 423         loadAddons.schedule( );
 424     }
