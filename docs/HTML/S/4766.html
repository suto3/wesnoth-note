  32 function wesnoth.wml_actions.replace_map(cfg)
  33         if not cfg.x and not cfg.y then
  34                 return replace_map(cfg)
  35         end
  36         local x1,x2 = string.match(cfg.x, "(%d+)-(%d+)")
  37         local y1,y2 = string.match(cfg.y, "(%d+)-(%d+)")
  38         local map = cfg.map_data
  39         x1 = tonumber(x1)
  40         y1 = tonumber(y1)
  41         x2 = x2 + 2
  42         y2 = y2 + 2
  43         local t = {}
  44         local y = 1
  45         for row in string.gmatch(map, "[^\n]+") do
  46                 if y >= y1 and y <= y2 then
  47                         local r = {}
  48                         local x = 1
  49                         for tile in string.gmatch(row, "[^,]+") do
  50                                 if x >= x1 and x <= x2 then r[x - x1 + 1] = tile end
  51                                 x = x + 1
  52                         end
  53                         t[y - y1 + 1] = table.concat(r, ',')
  54                 end
  55                 y = y + 1
  56         end
  57         local new_map = table.concat(t, '\n')
  58         replace_map { map = new_map, expand = true, shrink = true }
  59 end
