  87 void textbox::append_text(const std::string& text, bool auto_scroll, const color_t& color)
  88 {
  89         if(text_image_.get() == nullptr) {
  90                 set_text(text, color);
  91                 return;
  92         }
  93 
  94         //disallow adding multi-line text to a single-line text box
  95         if(wrap_ == false && std::find_if(text.begin(),text.end(),utils::isnewline) != text.end()) {
  96                 return;
  97         }
  98         const bool is_at_bottom = get_position() == get_max_position();
  99         const ucs4::string& wtext = unicode_cast<ucs4::string>(text);
 100 
 101         surface new_text = add_text_line(wtext, color);
 102         surface new_surface = create_compatible_surface(text_image_,std::max<size_t>(text_image_->w,new_text->w),text_image_->h+new_text->h);
 103 
 104         adjust_surface_alpha(new_text, SDL_ALPHA_TRANSPARENT);
 105         adjust_surface_alpha(text_image_, SDL_ALPHA_TRANSPARENT);
 106         SDL_SetSurfaceBlendMode(text_image_, SDL_BLENDMODE_NONE);
 107         sdl_blit(text_image_,nullptr,new_surface,nullptr);
 108         SDL_SetSurfaceBlendMode(text_image_, SDL_BLENDMODE_BLEND);
 109 
 110         SDL_Rect target {
 111                           0
 112                         , text_image_->h
 113                         , new_text->w
 114                         , new_text->h
 115         };
 116         SDL_SetSurfaceBlendMode(new_text, SDL_BLENDMODE_NONE);
 117         sdl_blit(new_text,nullptr,new_surface,&target);
 118         text_image_.assign(new_surface);
 119 
 120         text_.insert(text_.end(), wtext.begin(), wtext.end());
 121 
 122         set_dirty(true);
 123         update_text_cache(false);
 124         if(auto_scroll && is_at_bottom) scroll_to_bottom();
 125         handle_text_changed(text_);
 126 }

 186 void textbox::draw_contents()
 187 {
 188         SDL_Rect const &loc = inner_location();
 189 
 190         surface& surf = video().getSurface();
 191         sdl::draw_solid_tinted_rectangle(loc.x,loc.y,loc.w,loc.h,0,0,0,
 192                                     focus(nullptr) ? alpha_focus_ : alpha_, surf);
 193 
 194         SDL_Rect src;
 195 
 196         if(text_image_ == nullptr) {
 197                 update_text_cache(true);
 198         }
 199 
 200         if(text_image_ != nullptr) {
 201                 src.y = yscroll_;
 202                 src.w = std::min<size_t>(loc.w,text_image_->w);
 203                 src.h = std::min<size_t>(loc.h,text_image_->h);
 204                 src.x = text_pos_;
 205                 SDL_Rect dest = screen_area();
 206                 dest.x = loc.x;
 207                 dest.y = loc.y;
 208 
 209                 // Fills the selected area
 210                 if(enabled() && is_selection()) {
 211                         const int start = std::min<int>(selstart_,selend_);
 212                         const int end = std::max<int>(selstart_,selend_);
 213                         int startx = char_x_[start];
 214                         int starty = char_y_[start];
 215                         const int endx = char_x_[end];
 216                         const int endy = char_y_[end];
 217 
 218                         while(starty <= endy) {
 219                                 const size_t right = starty == endy ? endx : text_image_->w;
 220                                 if(right <= size_t(startx)) {
 221                                         break;
 222                                 }
 223 
 224                                 SDL_Rect rect = sdl::create_rect(loc.x + startx
 225                                                 , loc.y + starty - src.y
 226                                                 , right - startx
 227                                                 , line_height_);
 228 
 229                                 const clip_rect_setter clipper(surf, &loc);
 230 
 231                                 Uint32 color = SDL_MapRGB(surf->format, 0, 0, 160);
 232                                 sdl::fill_rect_alpha(rect, color, 140, surf);
 233 
 234                                 starty += int(line_height_);
 235                                 startx = 0;
 236                         }
 237                 }
 238 
 239                 if(enabled()) {
 240                         sdl_blit(text_image_, &src, surf, &dest);
 241                 } else {
 242                         // HACK: using 30% opacity allows white text to look as though it is grayed out,
 243                         // while not changing any applicable non-grayscale AA. Actual colored text will
 244                         // not look as good, but this is not currently a concern since GUI1 textboxes
 245                         // are not used much nowadays, and they will eventually all go away.
 246                         adjust_surface_alpha(text_image_, ftofxp(0.3));
 247                         sdl_blit(text_image_, &src, surf, &dest);
 248                 }
 249         }
 250 
 251         draw_cursor((cursor_pos_ == 0 ? 0 : cursor_pos_ - 1), video());
 252 }

 294 surface textbox::add_text_line(const ucs4::string& text, const color_t& color)
 295 {
 296         line_height_ = font::get_max_height(font_size_);
 297 
 298         if(char_y_.empty()) {
 299                 char_y_.push_back(0);
 300         } else {
 301                 char_y_.push_back(char_y_.back() + line_height_);
 302         }
 303 
 304         char_x_.push_back(0);
 305 
 306         // Re-calculate the position of each glyph. We approximate this by asking the
 307         // width of each substring, but this is a flawed assumption which won't work with
 308         // some more complex scripts (that is, RTL languages). This part of the work should
 309         // actually be done by the font-rendering system.
 310         std::string visible_string;
 311         ucs4::string wrapped_text;
 312 
 313         ucs4::string::const_iterator backup_itor = text.end();
 314 
 315         ucs4::string::const_iterator itor = text.begin();
 316         while(itor != text.end()) {
 317                 //If this is a space, save copies of the current state so we can roll back
 318                 if(char(*itor) == ' ') {
 319                         backup_itor = itor;
 320                 }
 321                 visible_string.append(unicode_cast<utf8::string>(*itor));
 322 
 323                 if(char(*itor) == '\n') {
 324                         backup_itor = text.end();
 325                         visible_string = "";
 326                 }
 327 
 328                 int w = font::line_width(visible_string, font_size_);
 329 
 330                 if(wrap_ && w >= inner_location().w) {
 331                         if(backup_itor != text.end()) {
 332                                 int backup = itor - backup_itor;
 333                                 itor = backup_itor + 1;
 334                                 if(backup > 0) {
 335                                         char_x_.erase(char_x_.end()-backup, char_x_.end());
 336                                         char_y_.erase(char_y_.end()-backup, char_y_.end());
 337                                         wrapped_text.erase(wrapped_text.end()-backup, wrapped_text.end());
 338                                 }
 339                         } else {
 340                                 if (visible_string == std::string("").append(unicode_cast<utf8::string>(*itor))) {
 341                                         break;  //breaks infinite loop where when running with a fake display, we word wrap a single character infinitely.
 342                                 }
 343                         }
 344                         backup_itor = text.end();
 345                         wrapped_text.push_back(ucs4::char_t('\n'));
 346                         char_x_.push_back(0);
 347                         char_y_.push_back(char_y_.back() + line_height_);
 348                         visible_string = "";
 349                 } else {
 350                         wrapped_text.push_back(*itor);
 351                         char_x_.push_back(w);
 352                         char_y_.push_back(char_y_.back() + (char(*itor) == '\n' ? line_height_ : 0));
 353                         ++itor;
 354                 }
 355         }
 356 
 357         const std::string s = unicode_cast<utf8::string>(wrapped_text);
 358         const surface res(font::get_rendered_text(s, font_size_, color));
 359 
 360         return res;
 361 }

 364 void textbox::update_text_cache(bool changed, const color_t& color)
 365 {
 366         if(changed) {
 367                 char_x_.clear();
 368                 char_y_.clear();
 369 
 370                 text_image_.assign(add_text_line(text_, color));
 371         }
 372 
 373         int cursor_x = char_x_[cursor_];
 374 
 375         if(cursor_x - text_pos_ > location().w) {
 376                 text_pos_ = cursor_x - location().w;
 377         } else if(cursor_x - text_pos_ < 0) {
 378                 text_pos_ = cursor_x;
 379         }
 380         cursor_pos_ = cursor_x - text_pos_;
 381 
 382         if (!text_image_.null()) {
 383                 set_full_size(text_image_->h);
 384                 set_shown_size(location().h);
 385         }
 386 }

 393 void textbox::erase_selection()
 394 {
 395         if(!is_selection())
 396                 return;
 397 
 398         ucs4::string::iterator itor = text_.begin() + std::min(selstart_, selend_);
 399         text_.erase(itor, itor + std::abs(selend_ - selstart_));
 400         cursor_ = std::min(selstart_, selend_);
 401         selstart_ = selend_ = -1;

 414 bool textbox::requires_event_focus(const SDL_Event* event) const
 415 {
 416         if(!focus_ || hidden() || !enabled()) {
 417                 return false;
 418         }
 419         if(event == nullptr) {
 420                 //when event is not specified, signal that focus may be desired later
 421                 return true;
 422         }
 423 
 424         if(event->type == SDL_KEYDOWN) {
 425                 SDL_Keycode key = event->key.keysym.sym;
 426                 switch(key) {
 427                 case SDLK_UP:
 428                 case SDLK_DOWN:
 429                 case SDLK_PAGEUP:
 430                 case SDLK_PAGEDOWN:
 431                         //in the future we may need to check for input history or multi-line support
 432                         //for now, just return false since these events are not handled.
 433                         return false;
 434                 default:
 435                         return true;
 436                 }
 437         }
 438         //mouse events are processed regardless of focus
 439         return false;
 440 }

 449 {
 450         bool changed = false;

 452         ucs4::string s = unicode_cast<ucs4::string>(str);
 453 
 454         DBG_G << "Char: " << str << "\n";
 455 
 456         if (editable_) {
 457                 changed = true;
 458                 if (is_selection())
 459                         erase_selection();
 460 
 461                 if (text_.size() + 1 <= max_size_) {

 464                         cursor_ += s.size();

 472 bool textbox::handle_key_down(const SDL_Event &event)
 473 {
 474         bool changed = false;
 475 
 476         const SDL_Keysym& key = reinterpret_cast<const SDL_KeyboardEvent&>(event).keysym;
 477         const SDL_Keymod modifiers = SDL_GetModState();
 478 
 479         const int c = key.sym;
 480         const int old_cursor = cursor_;
 481 
 482         listening_ = true;
 483 
 484         if(editable_) {
 485                 if(c == SDLK_LEFT && cursor_ > 0)
 486                         --cursor_;
 487 
 488                 if(c == SDLK_RIGHT && cursor_ < static_cast<int>(text_.size()))
 489                         ++cursor_;
 490 
 491                 // ctrl-a, ctrl-e and ctrl-u are readline style shortcuts, even on Macs
 492                 if(c == SDLK_END || (c == SDLK_e && (modifiers & KMOD_CTRL)))
 493                         cursor_ = text_.size();
 494 
 495                 if(c == SDLK_HOME || (c == SDLK_a && (modifiers & KMOD_CTRL)))
 496                         cursor_ = 0;
 497 
 498                 if((old_cursor != cursor_) && (modifiers & KMOD_SHIFT)) {
 499                         if(selstart_ == -1)
 500                                 selstart_ = old_cursor;
 501                         selend_ = cursor_;
 502                 }
 503         } else if(c == SDLK_LEFT || c == SDLK_RIGHT || c == SDLK_END || c == SDLK_HOME) {
 504                 pass_event_to_target(event);
 505         }
 506 
 507         if(editable_) {
 508                 if(c == SDLK_BACKSPACE) {
 509                         changed = true;
 510                         if(is_selection()) {
 511                                 erase_selection();
 512                         } else if(cursor_ > 0) {
 513                                 --cursor_;
 514                                 text_.erase(text_.begin()+cursor_);
 515                         }
 516                 }
 517 
 518                 if(c == SDLK_u && (modifiers & KMOD_CTRL)) { // clear line
 519                         changed = true;
 520                         cursor_ = 0;
 521                         text_.resize(0);
 522                 }
 523 
 524                 if(c == SDLK_DELETE && !text_.empty()) {
 525                         changed = true;
 526                         if(is_selection()) {
 527                                 erase_selection();
 528                         } else {
 529                                 if(cursor_ < static_cast<int>(text_.size())) {
 530                                         text_.erase(text_.begin()+cursor_);
 531                                 }
 532                         }
 533                 }
 534         } else if(c == SDLK_BACKSPACE || c == SDLK_DELETE || (c == SDLK_u && (modifiers & KMOD_CTRL))) {
 535                 pass_event_to_target(event);
 536         }
 537 
 538 
 539         //movement characters may have a "Unicode" field on some platforms, so ignore it.
 540         if(!(c == SDLK_UP || c == SDLK_DOWN || c == SDLK_LEFT || c == SDLK_RIGHT ||
 541                         c == SDLK_DELETE || c == SDLK_BACKSPACE || c == SDLK_END || c == SDLK_HOME ||
 542                         c == SDLK_PAGEUP || c == SDLK_PAGEDOWN)) {
 543                 if((event.key.keysym.mod & copypaste_modifier)
 544                                 //on windows SDL fires for AltGr lctrl+ralt (needed to access @ etc on certain keyboards)
 545 #ifdef _WIN32
 546                                 && !(event.key.keysym.mod & KMOD_ALT)
 547 #endif
 548                 ) {
 549                         switch(c) {
 550                         case SDLK_v: // paste
 551                         {
 552                                 if(!editable()) {
 553                                         pass_event_to_target(event);
 554                                         break;
 555                                 }
 556 
 557                                 changed = true;
 558                                 if(is_selection())
 559                                         erase_selection();
 560 
 561                                 std::string str = desktop::clipboard::copy_from_clipboard(false);
 562 
 563                                 //cut off anything after the first newline
 564                                 str.erase(std::find_if(str.begin(),str.end(),utils::isnewline),str.end());
 565 
 566                                 ucs4::string s = unicode_cast<ucs4::string>(str);
 567 
 568                                 if(text_.size() < max_size_) {
 569                                         if(s.size() + text_.size() > max_size_) {
 570                                                 s.resize(max_size_ - text_.size());
 571                                         }
 572                                         text_.insert(text_.begin()+cursor_, s.begin(), s.end());
 573                                         cursor_ += s.size();
 574                                 }
 575 
 576                         }
 577 
 578                         break;
 579 
 580                         case SDLK_c: // copy
 581                         {
 582                                 if(is_selection())
 583                                 {
 584                                         const size_t beg = std::min<size_t>(size_t(selstart_),size_t(selend_));
 585                                         const size_t end = std::max<size_t>(size_t(selstart_),size_t(selend_));
 586 
 587                                         ucs4::string ws(text_.begin() + beg, text_.begin() + end);
 588                                         std::string s = unicode_cast<utf8::string>(ws);
 589                                         desktop::clipboard::copy_to_clipboard(s, false);
 590                                 }
 591                         }
 592                         break;
 593                         }
 594                 }
 595                 else {
 596                         pass_event_to_target(event);
 597                 }
 598         }
 599 
 600         return changed;
 601 }

 603 void textbox::handle_event(const SDL_Event& event, bool was_forwarded)
 604 {
 605         if(!enabled())
 606                 return;
 607 
 608         scrollarea::handle_event(event);
 609         if(hidden())
 610                 return;
 611 
 612         bool changed = false;
 613 
 614         const int old_selstart = selstart_;
 615         const int old_selend = selend_;
 616 
 617         //Sanity check: verify that selection start and end are within text
 618         //boundaries
 619         if(is_selection() && !(size_t(selstart_) <= text_.size() && size_t(selend_) <= text_.size())) {
 620                 WRN_DP << "out-of-boundary selection" << std::endl;
 621                 selstart_ = selend_ = -1;
 622         }
 623 
 624         int mousex, mousey;
 625         const Uint8 mousebuttons = SDL_GetMouseState(&mousex,&mousey);
 626         if(!(mousebuttons & SDL_BUTTON(1))) {
 627                 grabmouse_ = false;
 628         }
 629 
 630         SDL_Rect const &loc = inner_location();
 631         bool clicked_inside = !mouse_locked() && (event.type == SDL_MOUSEBUTTONDOWN
 632                                            && (mousebuttons & SDL_BUTTON(1))
 633                                            && sdl::point_in_rect(mousex, mousey, loc));
 634         if(clicked_inside) {
 635                 set_focus(true);
 636         }
 637         if ((grabmouse_ && (!mouse_locked() && event.type == SDL_MOUSEMOTION)) || clicked_inside) {
 638                 const int x = mousex - loc.x + text_pos_;
 639                 const int y = mousey - loc.y;
 640                 int pos = 0;
 641                 int distance = x;
 642 
 643                 for(unsigned int i = 1; i < char_x_.size(); ++i) {
 644                         if(static_cast<int>(yscroll_) + y < char_y_[i]) {
 645                                 break;
 646                         }
 647 
 648                         // Check individually each distance (if, one day, we support
 649                         // RTL languages, char_x_[c] may not be monotonous.)
 650                         if(std::abs(x - char_x_[i]) < distance && yscroll_ + y < char_y_[i] + line_height_) {
 651                                 pos = i;
 652                                 distance = std::abs(x - char_x_[i]);
 653                         }
 654                 }
 655 
 656                 cursor_ = pos;
 657 
 658                 if(grabmouse_)
 659                         selend_ = cursor_;
 660 
 661                 update_text_cache(false);
 662 
 663                 if(!grabmouse_ && (mousebuttons & SDL_BUTTON(1))) {
 664                         grabmouse_ = true;
 665                         selstart_ = selend_ = cursor_;
 666                 } else if (! (mousebuttons & SDL_BUTTON(1))) {
 667                         grabmouse_ = false;
 668                 }
 669 
 670                 set_dirty();
 671         }
 672 
 673         //if we don't have the focus, then see if we gain the focus,
 674         //otherwise return
 675         if(!was_forwarded && focus(&event) == false) {
 676                 if (!mouse_locked() && event.type == SDL_MOUSEMOTION && sdl::point_in_rect(mousex, mousey, loc))
 677                         events::focus_handler(this);
 678 
 679                 return;
 680         }
 681 
 682         const int old_cursor = cursor_;
 683 
 684         if (event.type == SDL_TEXTINPUT && listening_) {
 685                 changed = handle_text_input(event);
 686         } else
 687                 if (event.type == SDL_KEYDOWN) {
 688                         changed = handle_key_down(event);
 689         }
 690         else {
 691                 if(event.type != SDL_KEYDOWN || (!was_forwarded && focus(&event) != true)) {
 692                         draw();
 693                         return;
 694                 }
 695         }
 696 
 697 
 698         if(is_selection() && (selend_ != cursor_))
 699                 selstart_ = selend_ = -1;
 700 
 701         //since there has been cursor activity, make the cursor appear for
 702         //at least the next 500ms.
 703         show_cursor_ = true;
 704         show_cursor_at_ = SDL_GetTicks();
 705 
 706         if(changed || old_cursor != cursor_ || old_selstart != selstart_ || old_selend != selend_) {
 707                 text_image_ = nullptr;
 708                 handle_text_changed(text_);
 709         }
 710 
 711         set_dirty(true);
 712 }

 715 {
 716         if(edit_target_ && edit_target_->editable()) {
 717                 edit_target_->handle_event(event, true);
 718         }
 719 }
