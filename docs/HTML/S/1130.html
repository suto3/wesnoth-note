 102 void scenario::set_sides()
 103 {
 104         if(map_.get() != nullptr) {
 105                 // If there are fewer sides in the configuration than there are
 106                 // starting positions, then generate the additional sides
 107                 const int map_positions = map_->num_valid_starting_positions();
 108 
 109                 for(int pos = data_.child_count("side"); pos < map_positions; ++pos) {
 110                         config& side = data_.add_child("side");
 111                         side["side"] = pos + 1;
 112                         side["team_name"] = "Team " + std::to_string(pos + 1);
 113                         side["canrecruit"] = true;
 114                         side["controller"] = "human";
 115                 }
 116 
 117                 num_players_ = 0;
 118                 for(const config& scenario : data_.child_range("side")) {
 119                         if(scenario["allow_player"].to_bool(true)) {
 120                                 ++num_players_;
 121                         }
 122                 }
 123         }
 124 }

 309 void create_engine::init_generated_level_data()
 310 {
 311         DBG_MP << "initializing generated level data\n";
 312 
 313         //DBG_MP << "current data:\n";
 314         //DBG_MP << current_level().data().debug();
 315 
 316         random_map * cur_lev = dynamic_cast<random_map *> (&current_level());
 317 
 318         if(!cur_lev) {
 319                 WRN_MP << "Tried to initialized generated level data on a level that wasn't a random map\n";
 320                 return;
 321         }
 322 
 323         try {
 324                 if(!cur_lev->generate_whole_scenario())
 325                 {
 326                         DBG_MP << "** replacing map ** \n";
 327 
 328                         config data = cur_lev->data();
 329 
 330                         data["map_data"] = generator_->create_map();
 331 
 332                         cur_lev->set_data(data);
 333 
 334                 } else { //scenario generation
 335 
 336                         DBG_MP << "** replacing scenario ** \n";
 337 
 338                         config data = generator_->create_scenario();
 339 
 340                         // Set the scenario to have placing of sides
 341                         // based on the terrain they prefer
 342                         if(!data.has_attribute("modify_placing")) {
 343                                 data["modify_placing"] = "true";
 344                         }
 345 
 346                         const std::string& description = cur_lev->data()["description"];
 347                         data["description"] = description;
 348 
 349                         cur_lev->set_data(data);
 350                 }
 351         } catch (mapgen_exception & e) {
 352                 config data = cur_lev->data();
 353 
 354                 data["error_message"] = e.what();
 355 
 356                 cur_lev->set_data(data);
 357         }
 358 
 359         //DBG_MP << "final data:\n";
 360         //DBG_MP << current_level().data().debug();
 361 }

 418 std::string create_engine::select_campaign_difficulty(int set_value)
 419 {
 420         // Verify the existence of difficulties
 421         std::vector<std::string> difficulties;
 422 
 423         for(const config& d : current_level().data().child_range("difficulty")) {
 424                 difficulties.push_back(d["define"]);
 425         }
 426 
 427         if(difficulties.empty()) {
 428                 difficulties = utils::split(current_level().data()["difficulties"]);
 429         }
 430 
 431         // No difficulties found. Exit
 432         if(difficulties.empty()) {
 433                 return "";
 434         }
 435 
 436         // One difficulty found. Use it
 437         if(difficulties.size() == 1) {
 438                 return difficulties[0];
 439         }
 440 
 441         // A specific difficulty value was passed
 442         // Use a minimalistic interface to get the specified define
 443         if(set_value != -1) {
 444                 if(set_value > static_cast<int>(difficulties.size())) {
 445                         std::cerr << "incorrect difficulty number: [" <<
 446                                 set_value << "]. maximum is [" << difficulties.size() << "].\n";
 447                         return "FAIL";
 448                 } else if(set_value < 1) {
 449                         std::cerr << "incorrect difficulty number: [" <<
 450                                 set_value << "]. minimum is [1].\n";
 451                         return "FAIL";
 452                 } else {
 453                         return difficulties[set_value - 1];
 454                 }
 455         }
 456 
 457         // If not, let the user pick one from the prompt
 458         // We don't pass the difficulties vector here because additional data is required
 459         // to constrict the dialog
 460         gui2::dialogs::campaign_difficulty dlg(current_level().data());
 461         dlg.show(video_);
 462 
 463         selected_campaign_difficulty_ = dlg.selected_difficulty();
 464 
 465         return selected_campaign_difficulty_;
 466 }

 532 void create_engine::set_current_level(const size_t index)
 533 {
 534         try {
 535                 current_level_index_ = type_map_.at(current_level_type_.v).games_filtered.at(index);
 536         } catch (std::out_of_range&) {
 537                 current_level_index_ = 0u;
 538         }
 539 
 540         if(current_level_type_ == level::TYPE::RANDOM_MAP) {
 541                 random_map* current_random_map = dynamic_cast<random_map*>(&current_level());
 542 
 543                 // If dynamic cast has failed then we somehow have gotten all the pointers mixed together.
 544                 assert(current_random_map);
 545 
 546                 generator_.reset(current_random_map->create_map_generator());
 547         } else {
 548                 generator_.reset(nullptr);
 549         }
 550 
 551         if(current_level_type_ != level::TYPE::CAMPAIGN &&
 552                 current_level_type_ != level::TYPE::SP_CAMPAIGN) {
 553 
 554                 dependency_manager_->try_scenario(current_level().id());
 555         }

 675 void create_engine::init_all_levels()
 676 {
 677         if(const config& generic_multiplayer = game_config_manager::get()->game_config().child("generic_multiplayer")) {
 678                 config gen_mp_data = generic_multiplayer;
 679 
 680                 // User maps.
 681                 int dep_index_offset = 0;
 682                 for(size_t i = 0; i < user_map_names_.size(); i++)
 683                 {
 684                         config user_map_data = gen_mp_data;
 685                         user_map_data["map_data"] = filesystem::read_map(user_map_names_[i]);
 686 
 687                         // Check if a file is actually a map.
 688                         // Note that invalid maps should be displayed in order to
 689                         // show error messages in the GUI.
 690                         bool add_map = true;
 691                         std::unique_ptr<gamemap> map;
 692                         try {
 693                                 map.reset(new gamemap(game_config_manager::get()->terrain_types(), user_map_data["map_data"]));
 694                         } catch (incorrect_map_format_error& e) {
 695                                 user_map_data["description"] = _("Map could not be loaded: ") + e.message;
 696 
 697                                 ERR_CF << "map could not be loaded: " << e.message << '\n';
 698                         } catch (wml_exception&) {
 699                                 add_map = false;
 700                                 dep_index_offset++;
 701                         }
 702 
 703                         if(add_map) {
 704                                 user_map_ptr new_user_map(new user_map(user_map_data, user_map_names_[i], map.get()));
 705 
 706                                 type_map_[level::TYPE::USER_MAP].games.push_back(new_user_map);
 707 
 708                                 // Since user maps are treated as scenarios, some dependency info is required
 709                                 config depinfo;
 710                                 depinfo["id"] = user_map_names_[i];
 711                                 depinfo["name"] = user_map_names_[i];
 712                                 dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset);
 713                         }
 714                 }
 715 
 716                 // User made scenarios.
 717                 dep_index_offset = 0;
 718                 for(size_t i = 0; i < user_scenario_names_.size(); i++)
 719                 {
 720                         config data;
 721                         try {
 722                                 read(data, *preprocess_file(filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]));
 723                         } catch (config::error & e) {
 724                                 ERR_CF << "Caught a config error while parsing user made (editor) scenarios:\n" << e.message << std::endl;
 725                                 ERR_CF << "Skipping file: " << (filesystem::get_user_data_dir() + "/editor/scenarios/" + user_scenario_names_[i]) << std::endl;
 726                                 continue;
 727                         }
 728 
 729                         scenario_ptr new_scenario(new scenario(data));
 730                         if(new_scenario->id().empty()) continue;
 731 
 732                         type_map_[level::TYPE::USER_SCENARIO].games.push_back(new_scenario);
 733 
 734                         // Since user scenarios are treated as scenarios, some dependency info is required
 735                         config depinfo;
 736                         depinfo["id"] = data["id"];
 737                         depinfo["name"] = data["name"];
 738                         dependency_manager_->insert_element(depcheck::SCENARIO, depinfo, i - dep_index_offset++);
 739                 }
 740         }
 741 
 742         // Stand-alone scenarios.
 743         for(const config& data : game_config_manager::get()->game_config().child_range("multiplayer"))
 744         {
 745                 if(!data["allow_new_game"].to_bool(true))
 746                         continue;
 747 
 748                 if(!data["campaign_id"].empty())
 749                         continue;
 750 
 751                 if(data.has_attribute("map_generation") || data.has_attribute("scenario_generation")) {
 752                         random_map_ptr new_random_map(new random_map(data));
 753 
 754                         type_map_[level::TYPE::RANDOM_MAP].games.push_back(new_random_map);
 755                 } else {
 756                         scenario_ptr new_scenario(new scenario(data));
 757 
 758                         type_map_[level::TYPE::SCENARIO].games.push_back(new_scenario);
 759                 }
 760         }
 761 
 762         // Campaigns.
 763         for(const config& data : game_config_manager::get()->game_config().child_range("campaign"))
 764         {
 765                 const std::string& type = data["type"];
 766                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 767 
 768                 if(type == "mp" || (type == "hybrid" && mp)) {
 769                         campaign_ptr new_campaign(new campaign(data));
 770 
 771                         type_map_[level::TYPE::CAMPAIGN].games.push_back(new_campaign);
 772                 }
 773 
 774                 if(type == "sp" || type.empty() || (type == "hybrid" && !mp)) {
 775                         campaign_ptr new_sp_campaign(new campaign(data));
 776                         new_sp_campaign->mark_if_completed();
 777 
 778                         type_map_[level::TYPE::SP_CAMPAIGN].games.push_back(new_sp_campaign);
 779                 }
 780         }
 781 
 782         auto& sp_campaigns = type_map_[level::TYPE::SP_CAMPAIGN].games;
 783 
 784         // Sort sp campaigns by rank.
 785         std::stable_sort(sp_campaigns.begin(), sp_campaigns.end(),
 786         [](const create_engine::level_ptr& a, const create_engine::level_ptr& b) {
 787                         return a->data()["rank"].to_int(1000) < b->data()["rank"].to_int(1000);
 788                 }
 789     );
 790 }

 792 void create_engine::init_extras(const MP_EXTRA extra_type)
 793 {
 794         std::vector<extras_metadata_ptr>& extras = get_extras_by_type(extra_type);
 795         const std::string extra_name = (extra_type == ERA) ? "era" : "modification";
 796 
 797         ng::depcheck::component_availabilty default_availabilty = (extra_type == ERA)
 798                 ? ng::depcheck::component_availabilty::MP
 799                 : ng::depcheck::component_availabilty::HYBRID;
 800 
 801         for(const config& extra : game_config_manager::get()->game_config().child_range(extra_name))
 802         {
 803                 ng::depcheck::component_availabilty type = extra["type"].to_enum(default_availabilty);
 804                 bool mp = state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 805 
 806                 if((type != ng::depcheck::component_availabilty::MP || mp) && (type != ng::depcheck::component_availabilty::SP || !mp) )
 807                 {
 808                         extras_metadata_ptr new_extras_metadata(new extras_metadata());
 809                         new_extras_metadata->id = extra["id"].str();
 810                         new_extras_metadata->name = extra["name"].str();
 811                         new_extras_metadata->description = extra["description"].str();
 812                         new_extras_metadata->cfg = &extra;
 813 
 814                         extras.push_back(new_extras_metadata);
 815                 }
 816         }
 817 }
