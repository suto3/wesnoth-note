 108         fake_unit_ptr create_fake_unit(const vconfig& cfg)
 109         {
 110                 std::string type = cfg["type"];
 111                 std::string variation = cfg["variation"];
 112                 std::string img_mods = cfg["image_mods"];
 113 
 114                 size_t side_num = cfg["side"].to_int(1);
 115                 if (!resources::gameboard->has_team(side_num)) {
 116                         side_num = 1;
 117                 }
 118 
 119                 unit_race::GENDER gender = string_gender(cfg["gender"]);
 120                 const unit_type *ut = unit_types.find(type);
 121                 if (!ut) return fake_unit_ptr();
 122                 fake_unit_ptr fake = fake_unit_ptr(unit_ptr(new unit(*ut, side_num, false, gender)));
 123 
 124                 if(!variation.empty()) {
 125                         config mod;
 126                         config &effect = mod.add_child("effect");
 127                         effect["apply_to"] = "variation";
 128                         effect["name"] = variation;
 129                         fake->add_modification("variation",mod);
 130                 }
 131 
 132                 if(!img_mods.empty()) {
 133                         config mod;
 134                         config &effect = mod.add_child("effect");
 135                         effect["apply_to"] = "image_mod";
 136                         effect["add"] = img_mods;
 137                         fake->add_modification("image_mod",mod);
 138                 }
 139 
 140                 return fake;
 141         }

 143         std::vector<map_location> fake_unit_path(const unit& fake_unit, const std::vector<std::string>& xvals, const std::vector<std::string>& yvals)
 144         {
 145                 const gamemap *game_map = & resources::gameboard->map();
 146                 std::vector<map_location> path;
 147                 map_location src;
 148                 map_location dst;
 149                 for(size_t i = 0; i != std::min(xvals.size(),yvals.size()); ++i) {
 150                         if(i==0){
 151                                 try {
 152                                         src.set_wml_x(std::stoi(xvals[i]));
 153                                         src.set_wml_y(std::stoi(yvals[i]));
 154                                 } catch(std::invalid_argument) {
 155                                         ERR_CF << "Invalid move_unit_fake source: " << xvals[i] << ", " << yvals[i] << '\n';
 156                                         continue;
 157                                 }
 158                                 if (!game_map->on_board(src)) {
 159                                         ERR_CF << "Invalid move_unit_fake source: " << src << '\n';
 160                                         break;
 161                                 }
 162                                 path.push_back(src);
 163                                 continue;
 164                         }
 165                         pathfind::shortest_path_calculator calc(fake_unit,
 166                                         resources::gameboard->get_team(fake_unit.side()),
 167                                         resources::gameboard->teams(),
 168                                         *game_map);
 169 
 170                         try {
 171                                 dst.set_wml_x(std::stoi(xvals[i]));
 172                                 dst.set_wml_y(std::stoi(yvals[i]));
 173                         } catch(std::invalid_argument) {
 174                                 ERR_CF << "Invalid move_unit_fake destination: " << xvals[i] << ", " << yvals[i] << '\n';
 175                         }
 176                         if (!game_map->on_board(dst)) {
 177                                 ERR_CF << "Invalid move_unit_fake destination: " << dst << '\n';
 178                                 break;
 179                         }
 180 
 181                         pathfind::plain_route route = pathfind::a_star_search(src, dst, 10000, calc,
 182                                 game_map->w(), game_map->h());
 183 
 184                         if (route.steps.empty()) {
 185                                 WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring complexities" << std::endl;
 186                                 pathfind::emergency_path_calculator emergency_calc(fake_unit, *game_map);
 187 
 188                                 route = pathfind::a_star_search(src, dst, 10000, emergency_calc,
 189                                                 game_map->w(), game_map->h());
 190                                 if(route.steps.empty()) {
 191                                         // This would occur when trying to do a MUF of a unit
 192                                         // over locations which are unreachable to it (infinite movement
 193                                         // costs). This really cannot fail.
 194                                         WRN_NG << "Could not find move_unit_fake route from " << src << " to " << dst << ": ignoring terrain" << std::endl;
 195                                         pathfind::dummy_path_calculator dummy_calc(fake_unit, *game_map);
 196                                         route = a_star_search(src, dst, 10000, dummy_calc, game_map->w(), game_map->h());
 197                                         assert(!route.steps.empty());
 198                                 }
 199                         }
 200                         // we add this section to the end of the complete path
 201                         // skipping section's head because already included
 202                         // by the previous iteration
 203                         path.insert(path.end(),
 204                                         route.steps.begin()+1, route.steps.end());
 205 
 206                         src = dst;
 207                 }
 208                 return path;
 209         }

 272 // This tag exposes part of the code path used to handle [command]'s in replays
 273 // This allows to perform scripting in WML that will use the same code path as player actions, for example.
 274 WML_HANDLER_FUNCTION(do_command,, cfg)
 275 {
 276         // Doing this in a whiteboard applied context will cause bugs
 277         // Note that even though game_events::wml_event_pump() will always apply the real unit map
 278         // It is still possible get a wml commands to run in a whiteboard applied context
 279         // With the theme_items lua callbacks
 280         if(resources::whiteboard->has_planned_unit_map())
 281         {
 282                 ERR_NG << "[do_command] called while whiteboard is applied, ignoring" << std::endl;
 283                 return;
 284         }
 285 
 286         static const std::set<std::string> allowed_tags {"attack", "move", "recruit", "recall", "disband", "fire_event", "lua_ai"};
 287 
 288         const bool is_too_early = resources::gamedata->phase() != game_data::START && resources::gamedata->phase() != game_data::PLAY;
 289         if(is_too_early)
 290         {
 291                 ERR_NG << "[do_command] called too early, only allowed at START or later" << std::endl;
 292                 return;
 293         }
 294         if(!resources::controller->current_team().is_local() && synced_context::get_synced_state() == synced_context::UNSYNCED)
 295         {
 296                 ERR_NG << "[do_command] can only be used from clients that control the currently playing side" << std::endl;
 297                 return;
 298         }
 299         for(vconfig::all_children_iterator i = cfg.ordered_begin(); i != cfg.ordered_end(); ++i)
 300         {
 301                 if(allowed_tags.find( i.get_key()) == allowed_tags.end()) {
 302                         ERR_NG << "unsupported tag [" << i.get_key() << "] in [do_command]" << std::endl;
 303                         std::stringstream o;
 304                         std::copy(allowed_tags.begin(), allowed_tags.end(), std::ostream_iterator<std::string>(o, " "));
 305                         ERR_NG << "allowed tags: " << o.str() << std::endl;
 306                         continue;
 307                 }
 308                 // TODO: afaik run_in_synced_context_if_not_already thows exceptions when the executed action end the scenario or the turn.
 309                 //       This could cause problems, specially when its unclear whether that excetion is caught by lua or not...
 310 
 311                 //Note that this fires related events and everthing else that also happen normally.
 312                 //have to watch out with the undo stack, therefore forbid [auto_shroud] and [update_shroud] here...
 313                 synced_context::run_in_synced_context_if_not_already(
 314                         /*commandname*/ i.get_key(),
 315                         /*data*/ i.get_child().get_parsed_config(),
 316                         /*use_undo*/ true,
 317                         /*show*/ true,
 318                         /*error_handler*/ &on_replay_error
 319                 );
 320         }
 321 }

 330 WML_HANDLER_FUNCTION(modify_turns,, cfg)
 331 {
 332         config::attribute_value value = cfg["value"];
 333         std::string add = cfg["add"];
 334         config::attribute_value current = cfg["current"];
 335         tod_manager& tod_man = *resources::tod_manager;
 336         if(!add.empty()) {
 337                 tod_man.modify_turns_by_wml(add);
 338         } else if(!value.empty()) {
 339                 tod_man.set_number_of_turns_by_wml(value.to_int(-1));
 340         }
 341         // change current turn only after applying mods
 342         if(!current.empty()) {
 343                 const unsigned int current_turn_number = tod_man.turn();
 344                 int new_turn_number = current.to_int(current_turn_number);
 345                 const unsigned int new_turn_number_u = static_cast<unsigned int>(new_turn_number);
 346                 if(new_turn_number_u < 1 || (new_turn_number > tod_man.number_of_turns() && tod_man.number_of_turns() != -1)) {
 347                         ERR_NG << "attempted to change current turn number to one out of range (" << new_turn_number << ")" << std::endl;
 348                 } else if(new_turn_number_u != current_turn_number) {
 349                         tod_man.set_turn_by_wml(new_turn_number_u, resources::gamedata);
 350                         resources::screen->new_turn();
 351                 }
 352         }
 353 }

 378 WML_HANDLER_FUNCTION(move_units_fake,, cfg)
 379 {
 380         LOG_NG << "Processing [move_units_fake]\n";
 381 
 382         const vconfig::child_list unit_cfgs = cfg.get_children("fake_unit");
 383         size_t num_units = unit_cfgs.size();
 384         std::vector<fake_unit_ptr > units;
 385         units.reserve(num_units);
 386         std::vector<std::vector<map_location> > paths;
 387         paths.reserve(num_units);
 388 
 389         LOG_NG << "Moving " << num_units << " units\n";
 390 
 391         size_t longest_path = 0;
 392 
 393         for (const vconfig& config : unit_cfgs) {
 394                 const std::vector<std::string> xvals = utils::split(config["x"]);
 395                 const std::vector<std::string> yvals = utils::split(config["y"]);
 396                 int skip_steps = config["skip_steps"];
 397                 fake_unit_ptr u = create_fake_unit(config);
 398                 units.push_back(u);
 399                 paths.push_back(fake_unit_path(*u, xvals, yvals));
 400                 if(skip_steps > 0)
 401                         paths.back().insert(paths.back().begin(), skip_steps, paths.back().front());
 402                 longest_path = std::max(longest_path, paths.back().size());
 403                 DBG_NG << "Path " << paths.size() - 1 << " has length " << paths.back().size() << '\n';
 404 
 405                 u->set_location(paths.back().front());
 406                 u.place_on_fake_unit_manager(resources::fake_units);
 407         }
 408 
 409         LOG_NG << "Units placed, longest path is " << longest_path << " long\n";
 410 
 411         std::vector<map_location> path_step(2);
 412         path_step.resize(2);
 413         for(size_t step = 1; step < longest_path; ++step) {
 414                 DBG_NG << "Doing step " << step << "...\n";
 415                 for(size_t un = 0; un < num_units; ++un) {
 416                         if(step >= paths[un].size() || paths[un][step - 1] == paths[un][step])
 417                                 continue;
 418                         DBG_NG << "Moving unit " << un << ", doing step " << step << '\n';
 419                         path_step[0] = paths[un][step - 1];
 420                         path_step[1] = paths[un][step];
 421                         unit_display::move_unit(path_step, units[un].get_unit_ptr());
 422                         units[un]->set_location(path_step[1]);
 423                         units[un]->anim_comp().set_standing();
 424                 }
 425         }
 426 
 427         LOG_NG << "Units moved\n";
 428 }

 437         // Prevent the recall unit filter from using the location as a criterion
 438 
 439         /**
 440          * @todo FIXME: we should design the WML to avoid these types of
 441          * collisions; filters should be named consistently and always have a
 442          * distinct scope.
 443          */

 445         temp_config["y"] = "recall";

 447         const vconfig & leader_filter = cfg.child("secondary_unit");

 449         for(int index = 0; index < int(resources::gameboard->teams().size()); ++index) {
 450                 LOG_NG << "for side " << index + 1 << "...\n";

 457                 }
 458

 460                 std::vector<unit_map::unit_iterator> leaders = resources::gameboard->units().find_leaders(index + 1);

 464                 for(std::vector<unit_ptr>::iterator u = avail.begin(); u != avail.end(); ++u) {

 472                                 const map_location cfg_loc = cfg_to_loc(cfg);
 473

 476                                         DBG_NG << "...considering " + leader->id() + " as the recalling leader...\n";
 477                                         map_location loc = cfg_loc;

 479                                              unit_filter(vconfig(leader->recall_filter()), resources::filter_con).matches( *(*u),map_location() ) ) {
 480                                                 DBG_NG << "...matched the leader filter and is able to recall the unit.\n";
 481                                                 if(!resources::gameboard->map().on_board(loc))
 482                                                         loc = leader->get_location();
 483                                                 if(pass_check || (resources::gameboard->units().count(loc) > 0))
 484                                                         loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 485                                                 if(resources::gameboard->map().on_board(loc)) {
 486                                                         DBG_NG << "...valid location for the recall found. Recalling.\n";
 487                                                         avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 488                                                         actions::place_recruit(to_recruit, loc, leader->get_location(), 0, true,
 489                                                                                cfg["show"].to_bool(true), cfg["fire_event"].to_bool(false),
 490                                                                                true, true);
 491                                                         return;
 492                                                 }
 493                                         }
 494                                 }
 495                                 if (resources::gameboard->map().on_board(cfg_loc)) {
 496                                         map_location loc = cfg_loc;
 497                                         if(pass_check || (resources::gameboard->units().count(loc) > 0))
 498                                                 loc = pathfind::find_vacant_tile(loc, pathfind::VACANT_ANY, pass_check);
 499                                         // Check if we still have a valid location
 500                                         if (resources::gameboard->map().on_board(loc)) {
 501                                                 DBG_NG << "No usable leader found, but found usable location. Recalling.\n";
 502                                                 avail.erase(u); // Erase before recruiting, since recruiting can fire more events
 503                                                 map_location null_location = map_location::null_location();
 504                                                 actions::place_recruit(to_recruit, loc, null_location, 0, true, cfg["show"].to_bool(true),
 505                                                                        cfg["fire_event"].to_bool(false), true, true);
 506                                                 return;
 507                                         }
 508                                 }
 509                         }
 510                 }
 511         }
 512         LOG_WML << "A [recall] tag with the following content failed:\n" << cfg.get_config().debug();

 528                         static const std::string s_all = "(" + s_prefix + s_terrain + s_sep + ")+";
 529                         static const boost::regex r_all(s_all);
 530 
 531                         const std::string& mapfile = filesystem::get_wml_location(filename_);
 532                         std::string res = "";
 533                         if(filesystem::file_exists(mapfile)) {
 534                                 res = filesystem::read_file(mapfile);
 535                         }
 536                         config retv;
 537                         if(boost::regex_match(res, r_all))
 538                         {
 539                                 retv["map_data"] = res;
 540                         }
 541                         return retv;
 542                 }
 543                 virtual config random_choice(int /*side*/) const
 544                 {
 545                         return config();
 546                 }
 547                 virtual std::string description() const
 548                 {
 549                         return "Map Data";
 550                 }
 551 
 552         };

 555 /// Experimental map replace
 556 /// @todo Finish experimenting.
 557 WML_HANDLER_FUNCTION(replace_map,, cfg)
 558 {
 559         /*
 560          * When a hex changes from a village terrain to a non-village terrain, and
 561          * a team owned that village it loses that village. When a hex changes from
 562          * a non-village terrain to a village terrain and there is a unit on that
 563          * hex it does not automatically capture the village. The reason for not
 564          * capturing villages it that there are too many choices to make; should a
 565          * unit loose its movement points, should capture events be fired. It is
 566          * easier to do this as wanted by the author in WML.
 567          */
 568 
 569         const gamemap * game_map = & resources::gameboard->map();
 570         gamemap map(*game_map);
 571 
 572         try {
 573                 if(!cfg["map_file"].empty()) {
 574                         config file_cfg = mp_sync::get_user_choice("map_data", map_choice(cfg["map_file"].str()));
 575                         map.read(file_cfg["map_data"].str(), false);
 576                 } else {
 577                         map.read(cfg["map"], false);
 578                 }
 579         } catch(incorrect_map_format_error&) {
 580                 const std::string log_map_name = cfg["map"].empty() ? cfg["file"] : std::string("from inline data");
 581                 lg::wml_error() << "replace_map: Unable to load map " << log_map_name << std::endl;
 582                 return;
 583         } catch(wml_exception& e) {
 584                 e.show(resources::screen->video());
 585                 return;
 586         }
 587 
 588         if (map.total_width() > game_map->total_width()
 589         || map.total_height() > game_map->total_height()) {
 590                 if (!cfg["expand"].to_bool()) {
 591                         lg::wml_error() << "replace_map: Map dimension(s) increase but expand is not set" << std::endl;
 592                         return;
 593                 }
 594         }
 595 
 596         if (map.total_width() < game_map->total_width()
 597         || map.total_height() < game_map->total_height()) {
 598                 if (!cfg["shrink"].to_bool()) {
 599                         lg::wml_error() << "replace_map: Map dimension(s) decrease but shrink is not set" << std::endl;
 600                         return;
 601                 }
 602         }
 603 
 604         boost::optional<std::string> errmsg = resources::gameboard->replace_map(map);
 605 
 606         if (errmsg) {
 607                 lg::wml_error() << *errmsg << std::endl;
 608         }
 609 
 610         resources::screen->reload_map();
 611         resources::screen->needs_rebuild(true);
 612         ai::manager::raise_map_changed();
 613 }

 623 WML_HANDLER_FUNCTION(set_variables,, cfg)
 624 {
 625         const t_string& name = cfg["name"];
 626         variable_access_create dest = resources::gamedata->get_variable_access_write(name);
 627         if(name.empty()) {
 628                 ERR_NG << "trying to set a variable with an empty name:\n" << cfg.get_config().debug();
 629                 return;
 630         }
 631 
 632         std::vector<config> data;
 633         if(cfg.has_attribute("to_variable"))
 634         {
 635                 try
 636                 {
 637                         variable_access_const tovar = resources::gamedata->get_variable_access_read(cfg["to_variable"]);
 638                         for (const config& c : tovar.as_array())
 639                         {
 640                                 data.push_back(c);
 641                         }
 642                 }
 643                 catch(const invalid_variablename_exception&)
 644                 {
 645                         ERR_NG << "Cannot do [set_variables] with invalid to_variable variable: " << cfg["to_variable"] << " with " << cfg.get_config().debug() << std::endl;
 646                 }
 647         } else {
 648                 typedef std::pair<std::string, vconfig> vchild;
 649                 for (const vchild& p : cfg.all_ordered()) {
 650                         if(p.first == "value") {
 651                                 data.push_back(p.second.get_parsed_config());
 652                         } else if(p.first == "literal") {
 653                                 data.push_back(p.second.get_config());
 654                         } else if(p.first == "split") {
 655                                 const vconfig & split_element = p.second;
 656 
 657                                 std::string split_string=split_element["list"];
 658                                 std::string separator_string=split_element["separator"];
 659                                 std::string key_name=split_element["key"];
 660                                 if(key_name.empty())
 661                                 {
 662                                         key_name="value";
 663                                 }
 664 
 665                                 bool remove_empty = split_element["remove_empty"].to_bool();
 666 
 667                                 char* separator = separator_string.empty() ? nullptr : &separator_string[0];
 668 
 669                                 std::vector<std::string> split_vector;
 670 
 671                                 //if no separator is specified, explode the string
 672                                 if(separator == nullptr)
 673                                 {
 674                                         for(std::string::iterator i=split_string.begin(); i!=split_string.end(); ++i)
 675                                         {
 676                                                 split_vector.push_back(std::string(1, *i));
 677                                         }
 678                                 }
 679                                 else {
 680                                         split_vector=utils::split(split_string, *separator, remove_empty ? utils::REMOVE_EMPTY | utils::STRIP_SPACES : utils::STRIP_SPACES);
 681                                 }
 682 
 683                                 for(std::vector<std::string>::iterator i=split_vector.begin(); i!=split_vector.end(); ++i)
 684                                 {
 685                                         data.emplace_back(config_of(key_name, *i));
 686                                 }
 687                         }
 688                 }
 689         }
 690         try
 691         {
 692                 const std::string& mode = cfg["mode"];
 693                 if(mode == "merge")
 694                 {
 695                         if(dest.explicit_index() && data.size() > 1)
 696                         {
 697                                 //merge children into one
 698                                 config merged_children;
 699                                 for (const config &ch : data) {
 700                                         merged_children.append(ch);
 701                                 }
 702                                 data = {merged_children};
 703                         }
 704                         dest.merge_array(data);
 705                 }
 706                 else if(mode == "insert")
 707                 {
 708                         dest.insert_array(data);
 709                 }
 710                 else if(mode == "append")
 711                 {
 712                         dest.append_array(data);
 713                 }
 714                 else /*default if(mode == "replace")*/
 715                 {
 716                         dest.replace_array(data);
 717                 }
 718         }
 719         catch(const invalid_variablename_exception&)
 720         {
 721                 ERR_NG << "Cannot do [set_variables] with invalid destination variable: " << name << " with " << cfg.get_config().debug() << std::endl;
 722         }
 723 }

 725 /// Store the relative direction from one hex to another in a WML variable.
 726 /// This is mainly useful as a diagnostic tool, but could be useful
 727 /// for some kind of scenario.
 728 WML_HANDLER_FUNCTION(store_relative_direction,, cfg)
 729 {
 730         if (!cfg.child("source")) {
 731                 WRN_NG << "No source in [store_relative_direction]" << std::endl;
 732                 return;
 733         }
 734         if (!cfg.child("destination")) {
 735                 WRN_NG << "No destination in [store_relative_direction]" << std::endl;
 736                 return;
 737         }
 738         if (!cfg.has_attribute("variable")) {
 739                 WRN_NG << "No variable in [store_relative_direction]" << std::endl;
 740                 return;
 741         }
 742 
 743         const map_location src = cfg_to_loc(cfg.child("source"));
 744         const map_location dst = cfg_to_loc(cfg.child("destination"));
 745 
 746         std::string variable = cfg["variable"];
 747         map_location::RELATIVE_DIR_MODE mode = static_cast<map_location::RELATIVE_DIR_MODE> (cfg["mode"].to_int(0));
 748         try
 749         {
 750                 variable_access_create store = resources::gamedata->get_variable_access_write(variable);
 751 
 752                 store.as_scalar() = map_location::write_direction(src.get_relative_dir(dst,mode));
 753         }
 754         catch(const invalid_variablename_exception&)
 755         {
 756                 ERR_NG << "Cannot do [store_relative_direction] with invalid destination variable: " << variable << " with " << cfg.get_config().debug() << std::endl;
 757         }

 858 WML_HANDLER_FUNCTION(tunnel,, cfg)
 859 {
 860         const bool remove = cfg["remove"].to_bool(false);
 861         const bool delay = cfg["delayed_variable_substitution"].to_bool(true);

 863                 const std::vector<std::string> ids = utils::split(cfg["id"]);
 864                 for (const std::string &id : ids) {
 865                         resources::tunnels->remove(id);
 866                 }

 868                 cfg.get_children("target").empty() ||
 869                 cfg.get_children("filter").empty()) {
 870                 ERR_WML << "[tunnel] is missing a mandatory tag:\n"
 871                          << cfg.get_config().debug();
 872         } else {
 873                 pathfind::teleport_group tunnel(delay ? cfg : vconfig(cfg.get_parsed_config()), false);
 874                 resources::tunnels->add(tunnel);
 875 
 876                 if(cfg["bidirectional"].to_bool(true)) {
 877                         tunnel = pathfind::teleport_group(delay ? cfg : vconfig(cfg.get_parsed_config()), true);
 878                         resources::tunnels->add(tunnel);
 879                 }

 883 /// If we should spawn a new unit on the map somewhere
 884 WML_HANDLER_FUNCTION(unit,, cfg)
 885 {
 886         config parsed_cfg = cfg.get_parsed_config();
 887 
 888         config::attribute_value to_variable = cfg["to_variable"];
 889         if (!to_variable.blank())
 890         {
 891                 parsed_cfg.remove_attribute("to_variable");
 892                 unit new_unit(parsed_cfg, true, &cfg);
 893                 try
 894                 {
 895                         config &var = resources::gamedata->get_variable_cfg(to_variable);
 896                         var.clear();
 897                         new_unit.write(var);
 898                         if (const config::attribute_value *v = parsed_cfg.get("x")) var["x"] = *v;
 899                         if (const config::attribute_value *v = parsed_cfg.get("y")) var["y"] = *v;
 900                 }
 901                 catch(const invalid_variablename_exception&)
 902                 {
 903                         ERR_NG << "Cannot do [unit] with invalid to_variable:  " << to_variable << " with " << cfg.get_config().debug() << std::endl;
 904                 }
 905                 return;
 906 
 907         }
 908 
 909         int side = parsed_cfg["side"].to_int(1);
 910 
 911 
 912         if ((side<1)||(side > static_cast<int>(resources::gameboard->teams().size()))) {
 913                 ERR_NG << "wrong side in [unit] tag - no such side: "<<side<<" ( number of teams :"<<resources::gameboard->teams().size()<<")"<<std::endl;
 914                 DBG_NG << parsed_cfg.debug();
 915                 return;
 916         }
 917         team &tm = resources::gameboard->get_team(side);
 918 
 919         unit_creator uc(tm,resources::gameboard->map().starting_position(side));
 920 
 921         uc
 922                 .allow_add_to_recall(true)
 923                 .allow_discover(true)
 924                 .allow_get_village(true)
 925                 .allow_invalidate(true)
 926                 .allow_rename_side(true)
 927                 .allow_show(true);
 928 
 929         uc.add_unit(parsed_cfg, &cfg);
 930 
 931 }
