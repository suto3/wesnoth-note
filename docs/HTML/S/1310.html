  14 sub filelabel {
  15   my $self = shift;
  16   my ($file,$level) = @_;
  17   $level = $graphincludes::params::minshow unless defined $level;
  18 
  19   ## 0: file
  20 
  21   $file =~ s/^$self->{PFXSTRIP}// if defined $self->{PFXSTRIP};
  22 
  23   return $file if $level == 0;
  24 
  25 
  26   ## 1: "module", aka. "compilation unit"
  27 
  28   $file =~ s/\.[^.]*$//;
  29   $file='ai' if $file =~ m/^ai_(move|attack)$/;
  30 
  31   return $file if $level == 1;
  32 
  33 
  34   ## 2: small groups
  35 
  36   if ($file =~ m!^(variable|server/variable)\.! ) {
  37     $file='variable';
  38   } elsif ($file =~ m!^(ai|multiplayer|replay).*!) {
  39     $file=$1;
  40   } elsif ($file =~ m!^(mapgen|mapgen_dialog|cavegen|map_create)\..*!) {
  41     $file='mapcreator';
  42   } elsif ($file =~ m!^(serialization|widgets)/.*!) {
  43     $file=$1;
  44   }
  45 
  46   return $file if $level == 2;
  47 
  48 
  49   ## 3: subsystems
  50 
  51   # core: low-level stuff, non-graphical plumbing
  52   if ($file =~ m!^(array|astarnode|cavegen|config|filesystem|game_config|game_errors|gettext|global|language|log|map|mapgen|pathfind|pathutils|preferences|race|random|serialization|scoped_resource|terrain|thread|tstring|unit|unit_types|util|variable|wassert|wml_separators|(.*/xcoll))$!) {
  53     $file='core';
  54   }
  55 
  56   # graphics: low-level graphical stuff, not part of the gameplay
  57   elsif ($file =~ m!^(animated|clipboard|cursor|font|halo|image|sdl_utils|tooltips|video)$!) {
  58     $file='graphics';
  59   }
  60 
  61   # uicore: building blocks for the GUI, not gameplay-related
  62   elsif ($file =~ m!^(builder|display|events|key|theme|widgets)$!) {
  63     $file='uicore';
  64   }
  65 
  66   # ui: game-related GUI elements (eg. usable in editor)
  67   elsif ($file =~ m!^(about|hotkeys|leader_list|mapgen_dialog|marked-up_text|menu_events|preferences_display|replay|show_dialog)$!) {
  68     $file='ui';
  69   }
  70 
  71   # gameclient: modules specific to the gameclient (ie. not for server
  72   # or editor)
  73   elsif ($file =~ m!^(ai|game|help|intro|mouse_events|multiplayer|playcampaign|playlevel|playturn|titlescreen)$!) {
  74     $file='gameclient';
  75   }
  76 
  77   # those subsystems living in their own directory
  78   elsif ($file =~ m!^(campaign_server|editor|server|tools)/.*!) {
  79     $file=$1;
  80   }
  81 
  82 
  83   # poor homeless modules
  84   return $file;
  85 }
