  77     /**
  78      * Runs the current tool
  79      */
  80     public void runTool( )
  81     {
  82         try {
  83             process_ = processBuilder_.start( );
  84             Reader stdoutReader = null;
  85             Reader stderrReader = null;
  86 
  87             if( arguments_.get( 0 ).toLowerCase( Locale.ENGLISH )
  88                 .contains( "wesnoth.exe" ) && //$NON-NLS-1$
  89                 Constants.IS_WINDOWS_MACHINE ) {
  90                 String wesnothParent = new File( arguments_.get( 0 ) )
  91                     .getParent( ) + "/"; //$NON-NLS-1$
  92                 if( new File( wesnothParent + "stdout.txt" ).exists( ) ) {
  93                     stdoutReader = new FileReader( wesnothParent + "stdout.txt" ); //$NON-NLS-1$
  94                 }
  95                 if( new File( wesnothParent + "stderr.txt" ).exists( ) ) {
  96                     stderrReader = new FileReader( wesnothParent + "stderr.txt" ); //$NON-NLS-1$
  97                 }
  98             }
  99             else {
 100                 stdoutReader = new InputStreamReader( process_.getInputStream( ) );
 101                 stderrReader = new InputStreamReader( process_.getErrorStream( ) );
 102             }
 103 
 104             if( stdoutReader != null ) {
 105                 bufferedReaderOutput_ = new BufferedReader( stdoutReader );
 106             }
 107 
 108             if( stderrReader != null ) {
 109                 bufferedReaderError_ = new BufferedReader( stderrReader );
 110             }
 111         } catch( IOException e ) {
 112             Logger.getInstance( ).logToolException( e );
 113         }
 114     }

 188     /**
 189      * Starts a new thread monitoring stderr.
 190      * All "Error" output will be available to be read from
 191      * <code>getErrorContent()</code>
 192      * 
 193      * @param extraStreams
 194      *        The extra streams array where stderr will be written or null
 195      *        if none
 196      */
 197     public void startErrorMonitor( final OutputStream[] extraStreams )
 198     {
 199         monitorErrorThread_ = new Thread( new Runnable( ) {
 200             @Override
 201             public void run( )
 202             {
 203                 try {
 204                     String line = ""; //$NON-NLS-1$
 205                     while( ( line = readErrorLine( ) ) != null ) {
 206                         if( extraStreams != null && extraStreams.length > 0 ) {
 207                             for( OutputStream stream: extraStreams ) {
 208                                 stream.write( ( line + "\n" ).getBytes( ) ); //$NON-NLS-1$
 209                             }
 210                         }
 211                         errorContent_.append( line + "\n" ); //$NON-NLS-1$
 212                     }
 213                 } catch( IOException e ) {
 214                     e.printStackTrace( );
 215                 }
 216             }
 217         } );
 218         monitorErrorThread_.start( );

 356     /**
 357      * Kills the current opened tool. No effect is tool is already killed
 358      * 
 359      * @param waitForKilling
 360      *        true to wait until the process is killed, so when the call
 361      *        returns the process will be already finished (it has
 362      *        "exitValue")
 363      */
 364     public void kill( boolean waitForKilling )
 365     {
 366         if( process_ != null ) {
 367             process_.destroy( );
 368             if( waitForKilling ) {
 369                 try {
 370                     process_.waitFor( );
 371                 } catch( InterruptedException e ) {
 372                 }
 373             }
 374             monitorErrorThread_.interrupt( );
 375             monitorOutputThread_.interrupt( );
 376         }
 377     }

 379     /**
 380      * Launches the specified tool, with the specified argument list.
 381      * The caller returns immediatelly
 382      * 
 383      * @param fileName
 384      *        the full path to the executable to be launched
 385      * @param args
 386      *        the arguments list
 387      * @param stdout
 388      *        An array of outputstreams where the stdout from the tool will
 389      *        be written
 390      * @param stderr
 391      *        An array of outputstreams where the stderr from the tool will
 392      *        be written
 393      * @return An {@link ExternalToolInvoker} instance that launched the
 394      *         specified file.
 395      */
 396     public static ExternalToolInvoker launchTool( final String fileName,
 397         final List< String > args, final OutputStream[] stdout,
 398         final OutputStream[] stderr )
 399     {
 400         final ExternalToolInvoker toolInvoker = new ExternalToolInvoker(
 401             fileName, args );
 402         toolInvoker.runTool( );
 403         Thread outputStreamThread = new Thread( new Runnable( ) {
 404             @Override
 405             public void run( )
 406             {
 407                 try {
 408                     String line = ""; //$NON-NLS-1$
 409                     while( ( line = toolInvoker.readOutputLine( ) ) != null ) {
 410                         for( OutputStream out: stdout ) {
 411                             out.write( ( line + "\n" ).getBytes( ) ); //$NON-NLS-1$
 412                         }
 413                     }
 414                 } catch( IOException e ) {
 415                     Logger.getInstance( ).logToolException( e );
 416                 }
 417             }
 418         } );
 419         Thread errorStreamThread = new Thread( new Runnable( ) {
 420             @Override
 421             public void run( )
 422             {
 423                 try {
 424                     String line = ""; //$NON-NLS-1$
 425                     while( ( line = toolInvoker.readErrorLine( ) ) != null ) {
 426                         for( OutputStream err: stderr ) {
 427                             err.write( ( line + "\n" ).getBytes( ) ); //$NON-NLS-1$
 428                         }
 429                     }
 430                 } catch( IOException e ) {
 431                     Logger.getInstance( ).logToolException( e );
 432                 }
 433             }
 434         } );
 435         outputStreamThread.start( );
 436         errorStreamThread.start( );
 437         return toolInvoker;
 438     }
