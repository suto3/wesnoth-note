  61 unsigned toggle_button::num_states() const
  62 {
  63         std::div_t res = std::div(this->config()->state.size(), COUNT);
  64         assert(res.rem == 0);
  65         assert(res.quot > 0);
  66         return res.quot;

  99 void toggle_button::update_canvas()

 114 void toggle_button::set_value(const unsigned selected)
 115 {
 116         if(selected == get_value()) {
 117                 return;
 118         }
 119         state_num_ = selected % num_states();
 120         set_is_dirty(true);
 121 
 122         // Check for get_window() is here to prevent the callback from
 123         // being called when the initial value is set.
 124         if(callback_state_change_ && get_window() != nullptr) {
 125                 callback_state_change_(*this);
 126         }
 127 }

 169 void toggle_button::signal_handler_left_button_click(const event::ui_event event,
 170                                                                                                           bool& handled)
 171 {
 172         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 173 
 174         sound::play_UI_sound(settings::sound_toggle_button_click);
 175 
 176         set_value(get_value() + 1);
 177 
 178         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 179 
 180         handled = true;
 181 }
 182 
 183 void toggle_button::signal_handler_left_button_double_click(
 184                 const event::ui_event event, bool& handled)
 185 {
 186         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 187 
 188         if(retval_ == 0) {
 189                 return;
 190         }
 191 
 192         window* window = get_window();
 193         assert(window);
 194 
 195         window->set_retval(retval_);
 196 
 197         handled = true;
 198 }

 214  * == Toggle button ==
 215  *
 216  * The definition of a toggle button.
 217  *
 218  * The following states exist:
 219  * * state_enabled, the button is enabled and not selected.
 220  * * state_disabled, the button is disabled and not selected.
 221  * * state_focused, the mouse is over the button and not selected.
 222  *
 223  * * state_enabled_selected, the button is enabled and selected.
 224  * * state_disabled_selected, the button is disabled and selected.
 225  * * state_focused_selected, the mouse is over the button and selected.
 226  * @begin{parent}{name="gui/"}
 227  * @begin{tag}{name="oggle_button_definition"}{min=0}{max=-1}{super="generic/widget_definition"}
 228  * @begin{tag}{name="resolution"}{min=0}{max=-1}{super="generic/widget_definition/resolution"}
 229  * @begin{tag}{name="state_enabled"}{min=0}{max=1}{super="generic/state"}
 230  * @end{tag}{name="state_enabled"}
 231  * @begin{tag}{name="state_disabled"}{min=0}{max=1}{super="generic/state"}
 232  * @end{tag}{name="state_disabled"}
 233  * @begin{tag}{name="state_focused"}{min=0}{max=1}{super="generic/state"}
 234  * @end{tag}{name="state_focused"}
 235  * @begin{tag}{name="state_enabled_selected"}{min=0}{max=1}{super="generic/state"}
 236  * @end{tag}{name="state_enabled_selected"}
 237  * @begin{tag}{name="state_disabled_selected"}{min=0}{max=1}{super="generic/state"}
 238  * @end{tag}{name="state_disabled_selected"}
 239  * @begin{tag}{name="state_focused_selected"}{min=0}{max=1}{super="generic/state"}

 245 toggle_button_definition::resolution::resolution(const config& cfg)
 246         : resolution_definition(cfg)
 247 {
 248         // Note the order should be the same as the enum state_t in
 249         // toggle_button.hpp.
 250         for(const auto& c : cfg.child_range("state"))
 251         {
 252                 state.emplace_back(c.child("enabled"));
 253                 state.emplace_back(c.child("disabled"));
 254                 state.emplace_back(c.child("focused"));
 255         }
 256 }
