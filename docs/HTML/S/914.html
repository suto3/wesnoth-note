  70 /* Secure password storage functions */
  71 bool authenticate(config& campaign, const config::attribute_value& passphrase)
  72 {
  73         return utils::md5(passphrase, campaign["passsalt"]).hex_digest() == campaign["passhash"];
  74 }
  75 
  76 std::string generate_salt(size_t len)
  77 {
  78         static const std::string itoa64 = "./0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz";
  79         boost::mt19937 mt(time(0));
  80         std::string salt = std::string(len, '0');
  81         boost::uniform_int<> from_str(0, itoa64.length() - 1);
  82         boost::variate_generator< boost::mt19937, boost::uniform_int<> > get_char(mt, from_str);
  83 
  84         for(size_t i = 0; i < len; i++) {
  85                 salt[i] = itoa64[get_char()];
  86         }
  87 
  88         return salt;
  89 }
  90 
  91 void set_passphrase(config& campaign, std::string passphrase)
  92 {
  93         std::string salt = generate_salt(16);
  94         campaign["passsalt"] = salt;
  95         campaign["passhash"] = utils::md5(passphrase, salt).hex_digest();
  96 }

 155 void server::load_config()
 156 {
 157         LOG_CS << "Reading configuration from " << cfg_file_ << "...\n";
 158 
 159         filesystem::scoped_istream in = filesystem::istream_file(cfg_file_);
 160         read(cfg_, *in);
 161 
 162         read_only_ = cfg_["read_only"].to_bool(false);
 163 
 164         if(read_only_) {
 165                 LOG_CS << "READ-ONLY MODE ACTIVE\n";
 166         }
 167 
 168         // Seems like compression level above 6 is a waste of CPU cycles.
 169         compress_level_ = cfg_["compress_level"].to_int(6);
 170 
 171         const config& svinfo_cfg = server_info();
 172         if(svinfo_cfg) {
 173                 feedback_url_format_ = svinfo_cfg["feedback_url_format"].str();
 174         }
 175 
 176         blacklist_file_ = cfg_["blacklist_file"].str();
 177         load_blacklist();
 178 
 179         // Load any configured hooks.
 180         hooks_.emplace(std::string("hook_post_upload"), cfg_["hook_post_upload"]);
 181         hooks_.emplace(std::string("hook_post_erase"), cfg_["hook_post_erase"]);
 182 
 183 #ifndef _WIN32
 184         // Open the control socket if enabled.
 185         if(!cfg_["control_socket"].empty()) {
 186                 const std::string& path = cfg_["control_socket"].str();
 187 
 188                 if(path != fifo_path_) {
 189                         const int res = mkfifo(path.c_str(),0660);
 190                         if(res != 0 && errno != EEXIST) {
 191                                 ERR_CS << "could not make fifo at '" << path << "' (" << strerror(errno) << ")\n";
 192                         } else {
 193                                 input_.close();
 194                                 int fifo = open(path.c_str(), O_RDWR|O_NONBLOCK);
 195                                 input_.assign(fifo);
 196                                 LOG_CS << "opened fifo at '" << path << "'. Server commands may be written to this file.\n";
 197                                 read_from_fifo();
 198                                 fifo_path_ = path;
 199                         }
 200                 }
 201         }
 202 #endif
 203 
 204         // Ensure the campaigns list WML exists even if empty, other functions
 205         // depend on its existence.
 206         cfg_.child_or_add("campaigns");
 207 
 208         // Certain config values are saved to WML again so that a given server
 209         // instance's parameters remain constant even if the code defaults change
 210         // at some later point.
 211         cfg_["compress_level"] = compress_level_;
 212 
 213         // But not the listening port number.
 214         port_ = cfg_["port"].to_int(default_campaignd_port);
 215 }

 224 void server::handle_request(socket_ptr socket, std::shared_ptr<simple_wml::document> doc)
 225 {
 226         config data;
 227         read(data, doc->output());
 228 
 229         config::all_children_iterator i = data.ordered_begin();
 230 
 231         if(i != data.ordered_end()) {
 232                 // We only handle the first child.
 233                 const config::any_child& c = *i;
 234 
 235                 request_handlers_table::const_iterator j
 236                                 = handlers_.find(c.key);
 237 
 238                 if(j != handlers_.end()) {
 239                         // Call the handler.
 240                         j->second(this, request(c.key, c.cfg, socket));
 241                 } else {
 242                         send_error("Unrecognized [" + c.key + "] request.",socket);
 243                 }
 244         }
 245 }

 249 void server::handle_read_from_fifo(const boost::system::error_code& error, std::size_t)
 250 {
 251         if(error) {
 252                 if(error == boost::asio::error::operation_aborted)
 253                         // This means fifo was closed by load_config() to open another fifo
 254                         return;
 255                 ERR_CS << "Error reading from fifo: " << error.message() << '\n';
 256                 return;
 257         }
 258 
 259         std::istream is(&admin_cmd_);
 260         std::string cmd;
 261         std::getline(is, cmd);
 262 
 263         const control_line ctl = cmd;
 264 
 265         if(ctl == "shut_down") {
 266                 LOG_CS << "Shut down requested by admin, shutting down...\n";
 267                 throw server_shutdown("Shut down via fifo command");
 268         } else if(ctl == "readonly") {
 269                 if(ctl.args_count()) {
 270                         cfg_["read_only"] = read_only_ = utils::string_bool(ctl[1], true);
 271                 }
 272 
 273                 LOG_CS << "Read only mode: " << (read_only_ ? "enabled" : "disabled") << '\n';
 274         } else if(ctl == "flush") {
 275                 LOG_CS << "Flushing config to disk...\n";
 276                 write_config();
 277         } else if(ctl == "reload") {
 278                 if(ctl.args_count()) {
 279                         if(ctl[1] == "blacklist") {
 280                                 LOG_CS << "Reloading blacklist...\n";
 281                                 load_blacklist();
 282                         } else {
 283                                 ERR_CS << "Unrecognized admin reload argument: " << ctl[1] << '\n';
 284                         }
 285                 } else {
 286                         LOG_CS << "Reloading all configuration...\n";
 287                         load_config();
 288                         LOG_CS << "Reloaded configuration\n";
 289                 }
 290         } else if(ctl == "setpass") {
 291                 if(ctl.args_count() != 2) {
 292                         ERR_CS << "Incorrect number of arguments for 'setpass'\n";
 293                 } else {
 294                         const std::string& addon_id = ctl[1];
 295                         const std::string& newpass = ctl[2];
 296                         config& campaign = get_campaign(addon_id);
 297 
 298                         if(!campaign) {
 299                                 ERR_CS << "Add-on '" << addon_id << "' not found, cannot set passphrase\n";
 300                         } else if(newpass.empty()) {
 301                                 // Shouldn't happen!
 302                                 ERR_CS << "Add-on passphrases may not be empty!\n";
 303                         } else {
 304                                 set_passphrase(campaign, newpass);
 305                                 write_config();
 306                                 LOG_CS << "New passphrase set for '" << addon_id << "'\n";
 307                         }
 308                 }
 309         } else {
 310                 ERR_CS << "Unrecognized admin command: " << ctl.full() << '\n';
 311         }
 312 
 313         read_from_fifo();
 314 }

 345 void server::load_blacklist()
 346 {
 347         // We *always* want to clear the blacklist first, especially if we are
 348         // reloading the configuration and the blacklist is no longer enabled.
 349         blacklist_.clear();
 350 
 351         if(blacklist_file_.empty()) {
 352                 return;
 353         }
 354 
 355         try {
 356                 filesystem::scoped_istream in = filesystem::istream_file(blacklist_file_);
 357                 config blcfg;
 358 
 359                 read(blcfg, *in);
 360 
 361                 blacklist_.read(blcfg);
 362                 LOG_CS << "using blacklist from " << blacklist_file_ << '\n';
 363         } catch(const config::error&) {
 364                 ERR_CS << "failed to read blacklist from " << blacklist_file_ << ", blacklist disabled\n";
 365         }
 366 }

 376 void server::fire(const std::string& hook, const std::string& addon)
 377 {
 378         const std::map<std::string, std::string>::const_iterator itor = hooks_.find(hook);
 379         if(itor == hooks_.end()) {
 380                 return;
 381         }
 382 
 383         const std::string& script = itor->second;
 384         if(script.empty()) {
 385                 return;
 386         }
 387 
 388 #if defined(_WIN32)
 389         (void)addon;
 390         ERR_CS << "Tried to execute a script on an unsupported platform\n";
 391         return;
 392 #else
 393         pid_t childpid;
 394 
 395         if((childpid = fork()) == -1) {
 396                 ERR_CS << "fork failed while updating campaign " << addon << '\n';
 397                 return;
 398         }
 399 
 400         if(childpid == 0) {
 401                 // We are the child process. Execute the script. We run as a
 402                 // separate thread sharing stdout/stderr, which will make the
 403                 // log look ugly.
 404                 execlp(script.c_str(), script.c_str(), addon.c_str(), static_cast<char *>(nullptr));
 405 
 406                 // exec() and family never return; if they do, we have a problem
 407                 std::cerr << "ERROR: exec failed with errno " << errno << " for addon " << addon
 408                                   << '\n';
 409                 exit(errno);
 410 
 411         } else {
 412                 return;
 413         }
 414 #endif
 415 }

 446 void server::handle_request_campaign_list(const server::request& req)
 447 {
 448         LOG_CS << "sending campaign list to " << req.addr << " using gzip\n";
 449 
 450         time_t epoch = time(nullptr);
 451         config campaign_list;
 452 
 453         campaign_list["timestamp"] = epoch;
 454         if(req.cfg["times_relative_to"] != "now") {
 455                 epoch = 0;
 456         }
 457 
 458         bool before_flag = false;
 459         time_t before = epoch;
 460         try {
 461                 before = before + lexical_cast<time_t>(req.cfg["before"]);
 462                 before_flag = true;
 463         } catch(bad_lexical_cast) {}
 464 
 465         bool after_flag = false;
 466         time_t after = epoch;
 467         try {
 468                 after = after + lexical_cast<time_t>(req.cfg["after"]);
 469                 after_flag = true;
 470         } catch(bad_lexical_cast) {}
 471 
 472         const std::string& name = req.cfg["name"];
 473         const std::string& lang = req.cfg["language"];
 474 
 475         for(const config& i : campaigns().child_range("campaign"))
 476         {
 477                 if(!name.empty() && name != i["name"]) {
 478                         continue;
 479                 }
 480 
 481                 const std::string& tm = i["timestamp"];
 482 
 483                 if(before_flag && (tm.empty() || lexical_cast_default<time_t>(tm, 0) >= before)) {
 484                         continue;
 485                 }
 486                 if(after_flag && (tm.empty() || lexical_cast_default<time_t>(tm, 0) <= after)) {
 487                         continue;
 488                 }
 489 
 490                 if(!lang.empty()) {
 491                         bool found = false;
 492 
 493                         for(const config& j : i.child_range("translation"))
 494                         {
 495                                 if(j["language"] == lang) {
 496                                         found = true;
 497                                         break;
 498                                 }
 499                         }
 500 
 501                         if(!found) {
 502                                 continue;
 503                         }
 504                 }
 505 
 506                 campaign_list.add_child("campaign", i);
 507         }
 508 
 509         for(config& j : campaign_list.child_range("campaign"))
 510         {
 511                 j["passphrase"] = "";
 512                 j["passhash"] = "";
 513                 j["passsalt"] = "";
 514                 j["upload_ip"] = "";
 515                 j["email"] = "";
 516                 j["feedback_url"] = "";
 517 
 518                 // Build a feedback_url string attribute from the
 519                 // internal [feedback] data.
 520                 const config& url_params = j.child_or_empty("feedback");
 521                 if(!url_params.empty() && !feedback_url_format_.empty()) {
 522                         j["feedback_url"] = format_addon_feedback_url(feedback_url_format_, url_params);
 523                 }
 524 
 525                 // Clients don't need to see the original data, so discard it.
 526                 j.clear_children("feedback");
 527         }
 528 
 529         config response;
 530         response.add_child("campaigns", campaign_list);
 531 
 532         std::ostringstream ostr;
 533         write(ostr, response);
 534         std::string wml = ostr.str();
 535         simple_wml::document doc(wml.c_str(), simple_wml::INIT_STATIC);
 536         doc.compress();
 537 
 538         async_send_doc(req.sock, doc, std::bind(&server::handle_new_client, this, _1));
 539 }

 541 void server::handle_request_campaign(const server::request& req)
 542 {
 543         LOG_CS << "sending campaign '" << req.cfg["name"] << "' to " << req.addr << " using gzip\n";
 544 
 545         config& campaign = get_campaign(req.cfg["name"]);
 546 
 547         if(!campaign) {
 548                 send_error("Add-on '" + req.cfg["name"].str() + "' not found.", req.sock);
 549         } else {
 550                 const int size = filesystem::file_size(campaign["filename"]);
 551 
 552                 if(size < 0) {
 553                         std::cerr << " size: <unknown> KiB\n";
 554                         ERR_CS << "File size unknown, aborting send.\n";
 555                         send_error("Add-on '" + req.cfg["name"].str() + "' could not be read by the server.", req.sock);
 556                         return;
 557                 }
 558 
 559                 std::cerr << " size: " << size/1024 << "KiB\n";
 560                 async_send_file(req.sock, campaign["filename"],
 561                                 std::bind(&server::handle_new_client, this, _1), null_handler);
 562                 // Clients doing upgrades or some other specific thing shouldn't bump
 563                 // the downloads count. Default to true for compatibility with old
 564                 // clients that won't tell us what they are trying to do.
 565                 if(req.cfg["increase_downloads"].to_bool(true)) {
 566                         const int downloads = campaign["downloads"].to_int() + 1;
 567                         campaign["downloads"] = downloads;
 568                 }
 569         }
 570 }

 587 void server::handle_upload(const server::request& req)
 588 {
 589         const config& upload = req.cfg;
 590 
 591         LOG_CS << "uploading campaign '" << upload["name"] << "' from " << req.addr << ".\n";
 592         config data = upload.child("data");
 593 
 594         const std::string& name = upload["name"];
 595         config *campaign = nullptr;
 596 
 597         bool passed_name_utf8_check = false;
 598 
 599         try {
 600                 const std::string& lc_name = utf8::lowercase(name);
 601                 passed_name_utf8_check = true;
 602 
 603                 for(config& c : campaigns().child_range("campaign"))
 604                 {
 605                         if(utf8::lowercase(c["name"]) == lc_name) {
 606                                 campaign = &c;
 607                                 break;
 608                         }
 609                 }
 610         } catch(const utf8::invalid_utf8_exception&) {
 611                 if(!passed_name_utf8_check) {
 612                         LOG_CS << "Upload aborted - invalid_utf8_exception caught on handle_upload() check 1, "
 613                                    << "the add-on pbl info contains invalid UTF-8\n";
 614                         send_error("Add-on rejected: The add-on name contains an invalid UTF-8 sequence.", req.sock);
 615                 } else {
 616                         LOG_CS << "Upload aborted - invalid_utf8_exception caught on handle_upload() check 2, "
 617                                    << "the internal add-ons list contains invalid UTF-8\n";
 618                         send_error("Server error: The server add-ons list is damaged.", req.sock);
 619                 }
 620 
 621                 return;
 622         }
 623 
 624         if(read_only_) {
 625                 LOG_CS << "Upload aborted - uploads not permitted in read-only mode.\n";
 626                 send_error("Add-on rejected: The server is currently in read-only mode.", req.sock);
 627         } else if(!data) {
 628                 LOG_CS << "Upload aborted - no add-on data.\n";
 629                 send_error("Add-on rejected: No add-on data was supplied.", req.sock);
 630         } else if(!addon_name_legal(upload["name"])) {
 631                 LOG_CS << "Upload aborted - invalid add-on name.\n";
 632                 send_error("Add-on rejected: The name of the add-on is invalid.", req.sock);
 633         } else if(is_text_markup_char(upload["name"].str()[0])) {
 634                 LOG_CS << "Upload aborted - add-on name starts with an illegal formatting character.\n";
 635                 send_error("Add-on rejected: The name of the add-on starts with an illegal formatting character.", req.sock);
 636         } else if(upload["title"].empty()) {
 637                 LOG_CS << "Upload aborted - no add-on title specified.\n";
 638                 send_error("Add-on rejected: You did not specify the title of the add-on in the pbl file!", req.sock);
 639         } else if(is_text_markup_char(upload["title"].str()[0])) {
 640                 LOG_CS << "Upload aborted - add-on title starts with an illegal formatting character.\n";
 641                 send_error("Add-on rejected: The title of the add-on starts with an illegal formatting character.", req.sock);
 642         } else if(get_addon_type(upload["type"]) == ADDON_UNKNOWN) {
 643                 LOG_CS << "Upload aborted - unknown add-on type specified.\n";
 644                 send_error("Add-on rejected: You did not specify a known type for the add-on in the pbl file! (See PblWML: wiki.wesnoth.org/PblWML)", req.sock);
 645         } else if(upload["author"].empty()) {
 646                 LOG_CS << "Upload aborted - no add-on author specified.\n";
 647                 send_error("Add-on rejected: You did not specify the author(s) of the add-on in the pbl file!", req.sock);
 648         } else if(upload["version"].empty()) {
 649                 LOG_CS << "Upload aborted - no add-on version specified.\n";
 650                 send_error("Add-on rejected: You did not specify the version of the add-on in the pbl file!", req.sock);
 651         } else if(upload["description"].empty()) {
 652                 LOG_CS << "Upload aborted - no add-on description specified.\n";
 653                 send_error("Add-on rejected: You did not specify a description of the add-on in the pbl file!", req.sock);
 654         } else if(upload["email"].empty()) {
 655                 LOG_CS << "Upload aborted - no add-on email specified.\n";
 656                 send_error("Add-on rejected: You did not specify your email address in the pbl file!", req.sock);
 657         } else if(!check_names_legal(data)) {
 658                 LOG_CS << "Upload aborted - invalid file names in add-on data.\n";
 659                 send_error("Add-on rejected: The add-on contains an illegal file or directory name."
 660                                    " File or directory names may not contain whitespace or any of the following characters: '/ \\ : ~'",
 661                                    req.sock);
 662         } else if(campaign && !authenticate(*campaign, upload["passphrase"])) {
 663                 LOG_CS << "Upload aborted - incorrect passphrase.\n";
 664                 send_error("Add-on rejected: The add-on already exists, and your passphrase was incorrect.", req.sock);
 665         } else {
 666                 const time_t upload_ts = time(nullptr);
 667 
 668                 LOG_CS << "Upload is owner upload.\n";
 669 
 670                 try {
 671                         if(blacklist_.is_blacklisted(name,
 672                                                                                  upload["title"].str(),
 673                                                                                  upload["description"].str(),
 674                                                                                  upload["author"].str(),
 675                                                                                  req.addr,
 676                                                                                  upload["email"].str()))
 677                         {
 678                                 LOG_CS << "Upload denied - blacklisted add-on information.\n";
 679                                 send_error("Add-on upload denied. Please contact the server administration for assistance.", req.sock);
 680                                 return;
 681                         }
 682                 } catch(const utf8::invalid_utf8_exception&) {
 683                         LOG_CS << "Upload aborted - the add-on pbl info contains invalid UTF-8 and cannot be "
 684                                    << "checked against the blacklist\n";
 685                         send_error("Add-on rejected: The add-on publish information contains an invalid UTF-8 sequence.", req.sock);
 686                         return;
 687                 }
 688 
 689                 const bool existing_upload = campaign != nullptr;
 690 
 691                 std::string message = "Add-on accepted.";
 692 
 693                 if(campaign == nullptr) {
 694                         campaign = &campaigns().add_child("campaign");
 695                         (*campaign)["original_timestamp"] = upload_ts;
 696                 }
 697 
 698                 (*campaign)["title"] = upload["title"];
 699                 (*campaign)["name"] = upload["name"];
 700                 (*campaign)["filename"] = "data/" + upload["name"].str();
 701                 (*campaign)["author"] = upload["author"];
 702                 (*campaign)["description"] = upload["description"];
 703                 (*campaign)["version"] = upload["version"];
 704                 (*campaign)["icon"] = upload["icon"];
 705                 (*campaign)["translate"] = upload["translate"];
 706                 (*campaign)["dependencies"] = upload["dependencies"];
 707                 (*campaign)["upload_ip"] = req.addr;
 708                 (*campaign)["type"] = upload["type"];
 709                 (*campaign)["email"] = upload["email"];
 710 
 711                 if(!existing_upload) {
 712                         set_passphrase(*campaign, upload["passphrase"]);
 713                 }
 714 
 715                 if((*campaign)["downloads"].empty()) {
 716                         (*campaign)["downloads"] = 0;
 717                 }
 718                 (*campaign)["timestamp"] = upload_ts;
 719 
 720                 int uploads = (*campaign)["uploads"].to_int() + 1;
 721                 (*campaign)["uploads"] = uploads;
 722 
 723                 (*campaign).clear_children("feedback");
 724                 if(const config& url_params = upload.child("feedback")) {
 725                         (*campaign).add_child("feedback", url_params);
 726                 }
 727 
 728                 const std::string& filename = (*campaign)["filename"].str();
 729                 data["title"] = (*campaign)["title"];
 730                 data["name"] = "";
 731                 data["campaign_name"] = (*campaign)["name"];
 732                 data["author"] = (*campaign)["author"];
 733                 data["description"] = (*campaign)["description"];
 734                 data["version"] = (*campaign)["version"];
 735                 data["timestamp"] = (*campaign)["timestamp"];
 736                 data["original_timestamp"] = (*campaign)["original_timestamp"];
 737                 data["icon"] = (*campaign)["icon"];
 738                 data["type"] = (*campaign)["type"];
 739                 (*campaign).clear_children("translation");
 740                 find_translations(data, *campaign);
 741 
 742                 add_license(data);
 743 
 744                 {
 745                         filesystem::scoped_ostream campaign_file = filesystem::ostream_file(filename);
 746                         config_writer writer(*campaign_file, true, compress_level_);
 747                         writer.write(data);
 748                 }
 749 
 750                 (*campaign)["size"] = filesystem::file_size(filename);
 751 
 752                 write_config();
 753 
 754                 send_message(message, req.sock);
 755 
 756                 fire("hook_post_upload", upload["name"]);
 757         }
 758 }

 760 void server::handle_delete(const server::request& req)
 761 {
 762         const config& erase = req.cfg;
 763 
 764         if(read_only_) {
 765                 LOG_CS << "in read-only mode, request to delete '" << erase["name"] << "' from " << req.addr << " denied\n";
 766                 send_error("Cannot delete add-on: The server is currently in read-only mode.", req.sock);
 767                 return;
 768         }
 769 
 770         LOG_CS << "deleting campaign '" << erase["name"] << "' requested from " << req.addr << "\n";
 771 
 772         config& campaign = get_campaign(erase["name"]);
 773 
 774         if(!campaign) {
 775                 send_error("The add-on does not exist.", req.sock);
 776                 return;
 777         }
 778 
 779         if(!authenticate(campaign, erase["passphrase"])
 780                 && (campaigns()["master_password"].empty()
 781                 || campaigns()["master_password"] != erase["passphrase"]))
 782         {
 783                 send_error("The passphrase is incorrect.", req.sock);
 784                 return;
 785         }
 786 
 787         // Erase the campaign.
 788         filesystem::write_file(campaign["filename"], std::string());
 789         if(remove(campaign["filename"].str().c_str()) != 0) {
 790                 ERR_CS << "failed to delete archive for campaign '" << erase["name"]
 791                            << "' (" << campaign["filename"] << "): " << strerror(errno)
 792                            << '\n';
 793         }
 794 
 795         config::child_itors itors = campaigns().child_range("campaign");
 796         for(size_t index = 0; !itors.empty(); ++index, itors.pop_front())
 797         {
 798                 if(&campaign == &itors.front()) {
 799                         campaigns().remove_child("campaign", index);
 800                         break;
 801                 }
 802         }
 803 
 804         write_config();
 805 
 806         send_message("Add-on deleted.", req.sock);
 807 
 808         fire("hook_post_erase", erase["name"]);
 809 
 810 }

 812 void server::handle_change_passphrase(const server::request& req)
 813 {
 814         const config& cpass = req.cfg;
 815 
 816         if(read_only_) {
 817                 LOG_CS << "in read-only mode, request to change passphrase denied\n";
 818                 send_error("Cannot change passphrase: The server is currently in read-only mode.", req.sock);
 819                 return;
 820         }
 821 
 822         config& campaign = get_campaign(cpass["name"]);
 823 
 824         if(!campaign) {
 825                 send_error("No add-on with that name exists.", req.sock);
 826         } else if(!authenticate(campaign, cpass["passphrase"])) {
 827                 send_error("Your old passphrase was incorrect.", req.sock);
 828         } else if(cpass["new_passphrase"].empty()) {
 829                 send_error("No new passphrase was supplied.", req.sock);
 830         } else {
 831                 set_passphrase(campaign, cpass["new_passphrase"]);
 832                 write_config();
 833                 send_message("Passphrase changed.", req.sock);
 834         }
 835 }
