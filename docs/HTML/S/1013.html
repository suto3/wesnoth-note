 124 /// The basic unit filter gives a generic implementation of the match fcn
 125 class basic_unit_filter_impl : public unit_filter_abstract_impl {
 126 public:
 127         basic_unit_filter_impl(const vconfig & vcfg, const filter_context & fc, bool flat_tod)
 128                 : fc_(fc)
 129                 , vcfg(vcfg)
 130                 , use_flat_tod_(flat_tod)
 131                 , cond_children_()
 132                 , cond_child_types_()
 133         {
 134                 // Handle [and], [or], and [not] with in-order precedence
 135                 vconfig::all_children_iterator cond = vcfg.ordered_begin();
 136                 vconfig::all_children_iterator cond_end = vcfg.ordered_end();
 137                 while(cond != cond_end)
 138                 {
 139                         const std::string& cond_name = cond.get_key();
 140                         conditional::TYPE type;
 141                         if(type.parse(cond_name)) {
 142                                 const vconfig& cond_filter = cond.get_child();
 143 
 144                                 cond_children_.emplace_back(cond_filter, &fc_, use_flat_tod_);
 145                                 cond_child_types_.push_back(type);
 146                         }
 147                         else {
 148                                 static const int NUM_VALID_TAGS = 5;
 149                                 static const std::string valid_tags[NUM_VALID_TAGS] {
 150                                         "filter_vision",
 151                                         "filter_adjacent",
 152                                         "filter_location",
 153                                         "filter_side",
 154                                         "filter_wml",
 155                                 };
 156                                 static const std::string* const valid_tags_end = valid_tags + NUM_VALID_TAGS;
 157 
 158                                 if (std::find(valid_tags, valid_tags_end, cond_name) == valid_tags_end){
 159                                         std::stringstream errmsg;
 160                                         errmsg << "encountered a child [" << cond_name << "] of a standard unit filter, it is being ignored";
 161                                         DBG_CF << errmsg.str() << std::endl; //FAIL( errmsg.str() );
 162                                 }
 163 
 164                         }
 165                         ++cond;
 166                 }
 167                 this->vcfg.make_safe();
 168         }
 169 
 170         virtual bool matches(const unit & u, const map_location & loc, const unit * u2) const;
 171         virtual std::vector<const unit *> all_matches_on_map(unsigned max_matches, const map_location* loc = nullptr, const unit* u2 = nullptr) const;
 172         virtual unit_const_ptr first_match_on_map() const;
 173         config to_config() const {
 174                 return vcfg.get_config();
 175         }
 176 
 177         virtual ~basic_unit_filter_impl() {}
 178 private:
 179         const filter_context & fc_;
 180         const vconfig vcfg;
 181         bool use_flat_tod_;
 182 
 183         std::vector<unit_filter> cond_children_;
 184         std::vector<conditional::TYPE> cond_child_types_;
 185 
 186         bool internal_matches_filter(const unit & u, const map_location & loc, const unit* u2) const;
 187 };

 193 static std::shared_ptr<unit_filter_abstract_impl> construct(const vconfig & vcfg, const filter_context & fc, bool flat_tod)
 194 {
 195         if (vcfg.empty()) {
 196                 return std::make_shared<null_unit_filter_impl> (fc);
 197         }
 198         if (vcfg.get_config().attribute_count() == 1 && vcfg.get_config().all_children_count() == 0 && vcfg.has_attribute("limit")) {
 199                 return std::make_shared<null_unit_filter_impl> (fc);
 200         }
 201         return std::make_shared<basic_unit_filter_impl>(vcfg, fc, flat_tod);
 202         //TODO: Add more efficient implementations for special cases
 203 }

 224 bool basic_unit_filter_impl::matches(const unit & u, const map_location& loc, const unit * u2) const
 225 {
 226         bool matches = true;
 227 
 228         if(loc.valid()) {
 229                 scoped_xy_unit auto_store("this_unit", loc, fc_.get_disp_context().units());
 230                 if (u2) {
 231                         const map_location& loc2 = u2->get_location();
 232                         scoped_xy_unit u2_auto_store("other_unit", loc2, fc_.get_disp_context().units());
 233                         matches = internal_matches_filter(u, loc, u2);
 234                 } else {
 235                         matches = internal_matches_filter(u, loc, u2);
 236                 }
 237         } else {
 238                 // If loc is invalid, then this is a recall list unit (already been scoped)
 239                 matches = internal_matches_filter(u, loc, nullptr);
 240         }
 241 
 242         // Handle [and], [or], and [not] with in-order precedence
 243         for (size_t i = 0; i < cond_children_.size(); i++) {
 244                 switch (cond_child_types_[i].v) {
 245                         case conditional::TYPE::AND:
 246                                 if(u2) {
 247                                         matches = matches && cond_children_[i].matches(u,loc,*u2);
 248                                 } else {
 249                                         matches = matches && cond_children_[i].matches(u,loc);
 250                                 }
 251                                 break;
 252                         case conditional::TYPE::OR:
 253                                 if(u2) {
 254                                         matches = matches || cond_children_[i].matches(u,loc,*u2);
 255                                 } else {
 256                                         matches = matches || cond_children_[i].matches(u,loc);
 257                                 }
 258                                 break;
 259                         case conditional::TYPE::NOT:
 260                                 if(u2) {
 261                                         matches = matches && !cond_children_[i].matches(u,loc,*u2);
 262                                 } else {
 263                                         matches = matches && !cond_children_[i].matches(u,loc);
 264                                 }
 265                 }
 266         }
 267         return matches;
 268 }

 270 bool basic_unit_filter_impl::internal_matches_filter(const unit & u, const map_location& loc, const unit* u2) const
 271 {
 272         if (!vcfg["name"].blank() && vcfg["name"].t_str() != u.name()) {
 273                 return false;
 274         }
 275 
 276         if (!vcfg["id"].blank()) {
 277                 std::vector<std::string> id_list = utils::split(vcfg["id"]);
 278                 if (std::find(id_list.begin(), id_list.end(), u.id()) == id_list.end()) {
 279                         return false;
 280                 }
 281         }
 282 
 283         // Allow 'speaker' as an alternative to id, since people use it so often
 284         if (!vcfg["speaker"].blank() && vcfg["speaker"].str() != u.id()) {
 285                 return false;
 286         }
 287 
 288         if (vcfg.has_child("filter_location")) {
 289                 if (vcfg.count_children("filter_location") > 1) {
 290                         FAIL("Encountered multiple [filter_location] children of a standard unit filter. "
 291                                  "This is not currently supported and in all versions of wesnoth would have "
 292                                  "resulted in the later children being ignored. You must use [and] or similar "
 293                                  "to achieve the desired result.");
 294                 }
 295                 terrain_filter filt(vcfg.child("filter_location"), &fc_, use_flat_tod_);
 296                 if (!filt.match(loc)) {
 297                         return false;
 298                 }
 299         }
 300 
 301         if(vcfg.has_child("filter_side")) {
 302                 if (vcfg.count_children("filter_side") > 1) {
 303                         FAIL("Encountered multiple [filter_side] children of a standard unit filter. "
 304                                  "This is not currently supported and in all versions of wesnoth would have "
 305                                  "resulted in the later children being ignored. You must use [and] or similar "
 306                                  "to achieve the desired result.");
 307                 }
 308                 side_filter filt(vcfg.child("filter_side"), &fc_);
 309                 if(!filt.match(u.side()))
 310                         return false;
 311         }
 312 
 313         // Also allow filtering on location ranges outside of the location filter
 314         if (!vcfg["x"].blank() || !vcfg["y"].blank()){
 315                 if(vcfg["x"] == "recall" && vcfg["y"] == "recall") {
 316                         //locations on the map are considered to not be on a recall list
 317                         if (fc_.get_disp_context().map().on_board(loc))
 318                         {
 319                                 return false;
 320                         }
 321                 } else if(vcfg["x"].empty() && vcfg["y"].empty()) {
 322                         return false;
 323                 } else if(!loc.matches_range(vcfg["x"], vcfg["y"])) {
 324                         return false;
 325                 }
 326         }
 327 
 328         // The type could be a comma separated list of types
 329         if (!vcfg["type"].empty()) {
 330                 std::vector<std::string> types = utils::split(vcfg["type"]);
 331                 if (std::find(types.begin(), types.end(), u.type_id()) == types.end()) {
 332                         return false;
 333                 }
 334         }
 335 
 336         // Shorthand for all advancements of a given type
 337         if (!vcfg["type_adv_tree"].empty()) {
 338                 std::set<std::string> types;
 339                 for(const std::string type : utils::split(vcfg["type_adv_tree"])) {
 340                         if(types.count(type)) {
 341                                 continue;
 342                         }
 343                         if(const unit_type* ut = unit_types.find(type)) {
 344                                 const auto& tree = ut->advancement_tree();
 345                                 types.insert(tree.begin(), tree.end());
 346                                 types.insert(type);
 347                         }
 348                 }
 349                 if(types.find(u.type_id()) == types.end()) {
 350                         return false;
 351                 }
 352         }
 353 
 354         // The variation_type could be a comma separated list of types
 355         if (!vcfg["variation"].empty())
 356         {
 357                 std::vector<std::string> types = utils::split(vcfg["variation"]);
 358                 if (std::find(types.begin(), types.end(), u.variation()) == types.end()) {
 359                         return false;
 360                 }
 361         }
 362 
 363         // The has_variation_type could be a comma separated list of types
 364         if (!vcfg["has_variation"].empty())
 365         {
 366                 bool match = false;
 367                 // If this unit is a variation itself then search in the base unit's variations.
 368                 const unit_type* const type = u.variation().empty() ? &u.type() : unit_types.find(u.type().base_id());
 369                 assert(type);
 370 
 371                 for (const std::string& variation_id : utils::split(vcfg["has_variation"])) {
 372                         if (type->has_variation(variation_id)) {
 373                                 match = true;
 374                                 break;
 375                         }
 376                 }
 377                 if (!match) return false;
 378         }
 379 
 380         if (!vcfg["ability"].empty())
 381         {
 382                 bool match = false;
 383 
 384                 for (const std::string& ability_id : utils::split(vcfg["ability"])) {
 385                         if (u.has_ability_by_id(ability_id)) {
 386                                 match = true;
 387                                 break;
 388                         }
 389                 }
 390                 if (!match) return false;
 391         }
 392 
 393         if (!vcfg["ability_type"].empty())
 394         {
 395                 bool match = false;
 396 
 397                 for (const std::string& ability : utils::split(vcfg["ability_type"])) {
 398                         if (u.has_ability_type(ability)) {
 399                                 match = true;
 400                                 break;
 401                         }
 402                 }
 403                 if (!match) return false;
 404         }
 405 
 406         if(!vcfg["ability_type_active"].empty()) {
 407                 bool match = false;
 408 
 409                 for(const std::string& ability : utils::split(vcfg["ability_type_active"])) {
 410                         if(!u.get_abilities(ability, loc).empty()) {
 411                                 match = true;
 412                                 break;
 413                         }
 414                 }
 415                 if(!match) {
 416                         return false;
 417                 }
 418         }
 419 
 420         if(!vcfg["trait"].empty()) {
 421                 std::vector<std::string> check_traits = utils::split(vcfg["trait"]), have_traits = u.get_traits_list(), isect;
 422                 std::sort(check_traits.begin(), check_traits.end());
 423                 std::sort(have_traits.begin(), have_traits.end());
 424                 std::set_intersection(check_traits.begin(), check_traits.end(), have_traits.begin(), have_traits.end(), std::back_inserter(isect));
 425                 if(isect.empty()) {
 426                         return false;
 427                 }
 428         }
 429 
 430         if (!vcfg["race"].empty()) {
 431                 std::vector<std::string> races = utils::split(vcfg["race"]);
 432                 if (std::find(races.begin(), races.end(), u.race()->id()) == races.end()) {
 433                         return false;
 434                 }
 435         }
 436 
 437         if (!vcfg["gender"].blank() && string_gender(vcfg["gender"]) != u.gender()) {
 438                 return false;
 439         }
 440 
 441         if (!vcfg["side"].empty() && vcfg["side"].to_int(-999) != u.side()) {
 442                 std::vector<std::string> sides = utils::split(vcfg["side"]);
 443                 const std::string u_side = std::to_string(u.side());
 444                 if (std::find(sides.begin(), sides.end(), u_side) == sides.end()) {
 445                         return false;
 446                 }
 447         }
 448 
 449         // handle statuses list
 450         if (!vcfg["status"].empty()) {
 451                 bool status_found = false;
 452 
 453                 for (const std::string status : utils::split(vcfg["status"])) {
 454                         if(u.get_state(status)) {
 455                                 status_found = true;
 456                                 break;
 457                         }
 458                 }
 459 
 460                 if(!status_found) {
 461                         return false;
 462                 }
 463         }
 464 
 465         if (vcfg.has_child("has_attack")) {
 466                 const vconfig& weap_filter = vcfg.child("has_attack");
 467                 bool has_weapon = false;
 468                 for(const attack_type& a : u.attacks()) {
 469                         if(a.matches_filter(weap_filter.get_parsed_config())) {
 470                                 has_weapon = true;
 471                                 break;
 472                         }
 473                 }
 474                 if(!has_weapon) {
 475                         return false;
 476                 }
 477         } else if (!vcfg["has_weapon"].blank()) {
 478                 std::string weapon = vcfg["has_weapon"];
 479                 bool has_weapon = false;
 480                 for(const attack_type& a : u.attacks()) {
 481                         if(a.id() == weapon) {
 482                                 has_weapon = true;
 483                                 break;
 484                         }
 485                 }
 486                 if(!has_weapon) {
 487                         return false;
 488                 }
 489         }
 490 
 491         if (!vcfg["role"].blank() && vcfg["role"].str() != u.get_role()) {
 492                 return false;
 493         }
 494 
 495         if (!vcfg["ai_special"].blank() && ((vcfg["ai_special"].str() == "guardian") != u.get_state(unit::STATE_GUARDIAN))) {
 496                 return false;
 497         }
 498 
 499         if (!vcfg["canrecruit"].blank() && vcfg["canrecruit"].to_bool() != u.can_recruit()) {
 500                 return false;
 501         }
 502 
 503         if (!vcfg["recall_cost"].blank()) {
 504                 bool match_found = false;
 505                 for(auto cost : utils::parse_ranges(vcfg["recall_cost"])) {
 506                         if(cost.first <= u.recall_cost() && u.recall_cost() <= cost.second) {
 507                                 match_found = true;
 508                                 break;
 509                         }
 510                 }
 511                 if(!match_found) {
 512                         return false;
 513                 }
 514         }
 515 
 516         if(!vcfg["level"].blank()) {
 517                 bool match_found = false;
 518                 for(auto lvl : utils::parse_ranges(vcfg["level"])) {
 519                         if(lvl.first <= u.level() && u.level() <= lvl.second) {
 520                                 match_found = true;
 521                                 break;
 522                         }
 523                 }
 524                 if(!match_found) {
 525                         return false;
 526                 }
 527         }
 528 
 529         if(!vcfg["defense"].blank()) {
 530                 bool match_found = false;
 531                 int actual_defense = u.defense_modifier(fc_.get_disp_context().map().get_terrain(loc));
 532                 for(auto def : utils::parse_ranges(vcfg["defense"])) {
 533                         if(def.first <= actual_defense && actual_defense <= def.second) {
 534                                 match_found = true;
 535                                 break;
 536                         }
 537                 }
 538                 if(!match_found) {
 539                         return false;
 540                 }
 541         }
 542 
 543         if(!vcfg["movement_cost"].blank()) {
 544                 bool match_found = false;
 545                 int actual_cost = u.movement_cost(fc_.get_disp_context().map().get_terrain(loc));
 546                 for(auto cost : utils::parse_ranges(vcfg["movement_cost"])) {
 547                         if(cost.first <= actual_cost && actual_cost <= cost.second) {
 548                                 match_found = true;
 549                                 break;
 550                         }
 551                 }
 552                 if(!match_found) {
 553                         return false;
 554                 }
 555         }
 556 
 557         if(!vcfg["vision_cost"].blank()) {
 558                 bool match_found = false;
 559                 int actual_cost = u.vision_cost(fc_.get_disp_context().map().get_terrain(loc));
 560                 for(auto cost : utils::parse_ranges(vcfg["vision_cost"])) {
 561                         if(cost.first <= actual_cost && actual_cost <= cost.second) {
 562                                 match_found = true;
 563                                 break;
 564                         }
 565                 }
 566                 if(!match_found) {
 567                         return false;
 568                 }
 569         }
 570 
 571         if(!vcfg["jamming_cost"].blank()) {
 572                 bool match_found = false;
 573                 int actual_cost = u.jamming_cost(fc_.get_disp_context().map().get_terrain(loc));
 574                 for(auto cost : utils::parse_ranges(vcfg["jamming_cost"])) {
 575                         if(cost.first <= actual_cost && actual_cost <= cost.second) {
 576                                 match_found = true;
 577                                 break;
 578                         }
 579                 }
 580                 if(!match_found) {
 581                         return false;
 582                 }
 583         }
 584 
 585         // Now start with the new WML based comparison.
 586         // If a key is in the unit and in the filter, they should match
 587         // filter only => not for us
 588         // unit only => not filtered
 589         config unit_cfg; // No point in serializing the unit once for each [filter_wml]!
 590         for (const vconfig& wmlcfg : vcfg.get_children("filter_wml")) {
 591                         config fwml = wmlcfg.get_parsed_config();
 592                         /* Check if the filter only cares about variables.
 593                            If so, no need to serialize the whole unit. */
 594                         config::all_children_itors ci = fwml.all_children_range();
 595                         if (fwml.all_children_count() == 1 &&
 596                                 fwml.attribute_count() == 1 &&
 597                             ci.front().key == "variables") {
 598                                 if (!u.variables().matches(ci.front().cfg))
 599                                         return false;
 600                         } else {
 601                                 if (unit_cfg.empty())
 602                                         u.write(unit_cfg);
 603                                 if (!unit_cfg.matches(fwml))
 604                                         return false;
 605                         }
 606         }
 607 
 608         for (const vconfig& vision : vcfg.get_children("filter_vision")) {
 609                 std::set<int> viewers;
 610 
 611                 // Use standard side filter
 612                 side_filter ssf(vision, &fc_);
 613                 std::vector<int> sides = ssf.get_teams();
 614                 viewers.insert(sides.begin(), sides.end());
 615 
 616                 bool found = false;
 617                 for (const int viewer : viewers) {
 618                         bool fogged = fc_.get_disp_context().get_team(viewer).fogged(loc);
 619                         bool hiding = u.invisible(loc, fc_.get_disp_context())
 620                                 && fc_.get_disp_context().get_team(viewer).is_enemy(u.side());
 621                         bool unit_hidden = fogged || hiding;
 622                         if (vision["visible"].to_bool(true) != unit_hidden) {
 623                                 found = true;
 624                                 break;
 625                         }
 626                 }
 627                 if (!found) {return false;}
 628         }
 629 
 630         if (vcfg.has_child("filter_adjacent")) {
 631                 const unit_map& units = fc_.get_disp_context().units();
 632                 map_location adjacent[6];
 633                 get_adjacent_tiles(loc, adjacent);
 634 
 635                 for (const vconfig& adj_cfg : vcfg.get_children("filter_adjacent")) {
 636                         int match_count=0;
 637                         unit_filter filt(adj_cfg, &fc_, use_flat_tod_);
 638 
 639                         config::attribute_value i_adjacent = adj_cfg["adjacent"];
 640                         std::vector<map_location::DIRECTION> dirs;
 641                         if (i_adjacent.blank()) {
 642                                 dirs = map_location::default_dirs();
 643                         } else {
 644                                 dirs = map_location::parse_directions(i_adjacent);
 645                         }
 646 
 647                         std::vector<map_location::DIRECTION>::const_iterator j, j_end = dirs.end();
 648                         for (j = dirs.begin(); j != j_end; ++j) {
 649                                 unit_map::const_iterator unit_itor = units.find(adjacent[*j]);
 650                                 if (unit_itor == units.end() || !filt(*unit_itor, u)) {
 651                                         continue;
 652                                 }
 653                                 boost::optional<bool> is_enemy;
 654                                 if (!adj_cfg["is_enemy"].blank()) {
 655                                         is_enemy = adj_cfg["is_enemy"].to_bool();
 656                                 }
 657                                 if (!is_enemy || *is_enemy ==
 658                                     fc_.get_disp_context().get_team(u.side()).is_enemy(unit_itor->side())) {
 659                                         ++match_count;
 660                                 }
 661                         }
 662 
 663                         static std::vector<std::pair<int,int> > default_counts = utils::parse_ranges("1-6");
 664                         config::attribute_value i_count = adj_cfg["count"];
 665                         if(!in_ranges(match_count, !i_count.blank() ? utils::parse_ranges(i_count) : default_counts)) {
 666                                 return false;
 667                         }
 668                 }
 669         }
 670 
 671         if (!vcfg["find_in"].blank()) {
 672                 // Allow filtering by searching a stored variable of units
 673                 if (const game_data * gd = fc_.get_game_data()) {
 674                         try
 675                         {
 676                                 variable_access_const vi = gd->get_variable_access_read(vcfg["find_in"]);
 677                                 bool found_id = false;
 678                                 for (const config& c : vi.as_array())
 679                                 {
 680                                         if(c["id"] == u.id())
 681                                                 found_id = true;
 682                                 }
 683                                 if(!found_id)
 684                                 {
 685                                         return false;
 686                                 }
 687                         }
 688                         catch(const invalid_variablename_exception&)
 689                         {
 690                                 return false;
 691                         }
 692                 }
 693         }
 694         if (!vcfg["formula"].blank()) {
 695                 try {
 696                         const wfl::unit_callable main(loc,u);
 697                         wfl::map_formula_callable callable(main.fake_ptr());
 698                         if (u2) {
 699                                 std::shared_ptr<wfl::unit_callable> secondary(new wfl::unit_callable(*u2));
 700                                 callable.add("other", wfl::variant(secondary));
 701                                 // It's not destroyed upon scope exit because the variant holds a reference
 702                         }
 703                         const wfl::formula form(vcfg["formula"], new wfl::gamestate_function_symbol_table);
 704                         if(!form.evaluate(callable).as_bool()) {
 705                                 return false;
 706                         }
 707                         return true;
 708                 } catch(wfl::formula_error& e) {
 709                         lg::wml_error() << "Formula error in unit filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 710                         // Formulae with syntax errors match nothing
 711                         return false;
 712                 }
 713         }
 714 
 715         if (!vcfg["lua_function"].blank()) {
 716                 if (game_lua_kernel * lk = fc_.get_lua_kernel()) {
 717                         bool b = lk->run_filter(vcfg["lua_function"].str().c_str(), u);
 718                         if (!b) return false;
 719                 }
 720         }
 721 
 722         return true;
 723 }

 725 std::vector<const unit *> basic_unit_filter_impl::all_matches_on_map(unsigned max_matches, const map_location* loc, const unit* u2) const {
 726         std::vector<const unit *> ret;
 727         for (const unit & u : fc_.get_disp_context().units()) {
 728                 if (matches(u, loc ? *loc : u.get_location(), u2)) {
 729                         if(max_matches == 0) {
 730                                 return ret;
 731                         }
 732                         --max_matches;
 733                         ret.push_back(&u);
 734                 }
 735         }
 736         return ret;
 737 }

 739 unit_const_ptr basic_unit_filter_impl::first_match_on_map() const {
 740         const unit_map & units = fc_.get_disp_context().units();

 742                 if (matches(*u,u->get_location(),nullptr)) {
 743                         return u.get_shared_ptr();
 744                 }
 745         }
 746         return unit_const_ptr();
