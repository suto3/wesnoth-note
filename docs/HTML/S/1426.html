 365     protected void removeInstall( WesnothInstall install )
 366     {
 367         if( install != null ) {
 368             installs_.remove( install.getName( ) );
 369             installsTableViewer_.refresh( );
 370 
 371             // unset all settings.
 372             IPreferenceStore prefs = Preferences.getPreferences( );
 373             String installPrefix = Preferences.getInstallPrefix( install
 374                 .getName( ) );
 375             prefs.setToDefault( installPrefix + Preferences.WESNOTH_EXEC_PATH );
 376             prefs.setToDefault( installPrefix + Preferences.WESNOTH_USER_DIR );
 377             prefs.setToDefault( installPrefix
 378                 + Preferences.WESNOTH_WMLTOOLS_DIR );
 379             prefs
 380                 .setToDefault( installPrefix + Preferences.WESNOTH_WORKING_DIR );
 381 
 382             // unset the default install if this was that
 383             // and select another one (the first) - if any - as default
 384             if( install.getName( )
 385                 .equals( Preferences.getDefaultInstallName( ) ) ) {
 386                 Preferences.setDefaultInstallName( "" ); //$NON-NLS-1$
 387 
 388                 if( ! installs_.isEmpty( ) ) {
 389                     // get the first item from the iterator
 390                     Iterator< WesnothInstall > itor = installs_.values( )
 391                         .iterator( );
 392                     setInstallAsDefault( itor.next( ) );
 393                 }
 394             }
 395 
 396             // clear the current info
 397             newInstall( );
 398         }

 463     /**
 464      * Tries the list of available paths for current os
 465      */
 466     private void guessDefaultPaths( )
 467     {
 468         testAndSetPaths( wesnothExecutablePaths_, wesnothExecutableField_ );
 469         testAndSetPaths( wesnothDataDirPaths_, wesnothWorkingDirField_ );
 470         testAndSetPaths( wesnothUserDirPaths_, wesnothUserDirField_ );
 471 
 472         // guess the working dir based on the executable's path
 473         Text textControl = wesnothWorkingDirField_
 474             .getTextControl( getFieldEditorParent( ) );
 475 
 476         String workingDirValue = wesnothWorkingDirField_.getStringValue( );
 477         String wesnothExecValue = wesnothExecutableField_.getStringValue( );
 478 
 479         IPath guessedWorkingDir = new Path( wesnothExecValue );
 480         // remove the filename
 481         guessedWorkingDir = guessedWorkingDir.removeLastSegments( 1 );
 482         // The working dir should contain the data directory
 483         guessedWorkingDir = guessedWorkingDir.append( "/data/" );
 484 
 485         if( workingDirValue.isEmpty( )
 486             && ! wesnothExecValue.isEmpty( )
 487             && new File( guessedWorkingDir.toOSString( ) ).exists( ) ) {
 488 
 489             workingDirValue = guessedWorkingDir.removeLastSegments( 1 )
 490                 .toOSString( );
 491             // don't retain the /data/ directory
 492             textControl.setText( workingDirValue );
 493         }
 494 
 495         // guess the wmltools path
 496         if( wmlToolsField_.getStringValue( ).isEmpty( )
 497             && ! workingDirValue.isEmpty( ) ) {
 498             String path = workingDirValue + "/data/tools"; //$NON-NLS-1$
 499             if( testWMLToolsPath( path ) ) {
 500                 wmlToolsField_.setStringValue( path );
 501             }
 502         }
 503 
 504         String userDirValue = wesnothUserDirField_.getStringValue( );
 505         // guess the userdata path
 506         if( userDirValue.isEmpty( ) && ! workingDirValue.isEmpty( ) ) {
 507             String path = workingDirValue + "/userdata"; //$NON-NLS-1$
 508             testAndSetPaths( new String[] { path }, wesnothUserDirField_ );
 509         }
 510     }

 570     /**
 571      * Saves the current install
 572      * 
 573      * @return true if the save was successfully, false otherwise
 574      */
 575     private boolean saveInstall( )
 576     {
 577         String installName = txtInstallName_.getText( );
 578 
 579         // if it's editable, it means we are creating a new install
 580         if( txtInstallName_.getEditable( ) ) {
 581             boolean isFieldsEmpty = isFieldsEmpty( );
 582 
 583             if( installName.isEmpty( ) == true ) {
 584                 // if we haven't completed anything,
 585                 // we can skip the saving without alerting the user.
 586                 if( ! isFieldsEmpty ) {
 587                     GUIUtils
 588                         .showErrorMessageBox( Messages.WesnothInstallsPage_19 );
 589                 }
 590 
 591                 // we consider successfully save if the fields are all
 592                 // empty
 593                 return isFieldsEmpty;
 594             }
 595 
 596             if( cmbVersion_.getText( ).isEmpty( ) == true ) {
 597                 GUIUtils.showErrorMessageBox( Messages.WesnothInstallsPage_20 );
 598                 return false;
 599             }
 600 
 601             // update the fields preferences names
 602             setFieldsPreferenceName(
 603                 Preferences.getInstallPrefix( installName ), false );
 604 
 605             WesnothInstall newInstall = new WesnothInstall( installName,
 606                 cmbVersion_.getText( ) );
 607 
 608             installs_.put( installName, newInstall );
 609 
 610             // if there is not any install set as default, set this one
 611             if( Preferences.getDefaultInstallName( ).isEmpty( ) ) {
 612                 setInstallAsDefault( newInstall );
 613             }
 614 
 615             // we are creating a new install. Clear the editable
 616             // flag after we save it, to prevent renaming.
 617             txtInstallName_.setEditable( false );
 618         }
 619         else if( getSelectedInstall( ) != null ) { // just saving
 620             // the fields are automatically saved by Eclipse.
 621             // we just need to save the new version.
 622             getSelectedInstall( ).setVersion( cmbVersion_.getText( ) );
 623         }
 624 
 625         installsTableViewer_.refresh( );
 626 
 627         return true;
 628     }

 638         }
 639         if( ! wesnothUserDirField_.isValid( ) ) {

 642         if( ! wesnothWorkingDirField_.isValid( ) ) {

 644         }
 645         if( ! wmlToolsField_.isValid( ) ) {
 646             wmlToolsField_.setStringValue( "" ); //$NON-NLS-1$

 648 
 649         if( saveInstall( ) == false ) {

 654         return true;
