 139 void formula_debugger::check_breakpoints()
 140 {
 141         for(std::list<breakpoint_ptr>::iterator b = breakpoints_.begin(); b != breakpoints_.end(); ++b) {
 142                 if ((*b)->is_break_now()){
 143                         current_breakpoint_ = (*b);
 144                         show_gui();
 145                         current_breakpoint_ = breakpoint_ptr();
 146                         if ((*b)->is_one_time_only()) {
 147                                 b = breakpoints_.erase(b);
 148                         }
 149                         break;
 150                 }
 151         }
 152 }

 154 void formula_debugger::show_gui()
 155 {
 156         if (resources::screen == nullptr) {
 157                 WRN_FDB << "do not showing debug window due to nullptr gui" << std::endl;
 158                 return;
 159         }

 161                 gui2::dialogs::formula_debugger debug_dialog(*this);
 162                 debug_dialog.show(resources::screen->video());
 163         } else {

 165         }

 168 void formula_debugger::call_stack_push(const std::string &str)
 169 {
 170         call_stack_.emplace_back(arg_number_extra_debug_info,counter_++,call_stack_.size(),f_name_extra_debug_info,str,variant(),false);
 171         arg_number_extra_debug_info = -1;
 172         f_name_extra_debug_info = "";
 173         execution_trace_.push_back(call_stack_.back());
 174 }

 194 variant formula_debugger::evaluate_arg_callback(const formula_expression &expression, const formula_callable &variables)
 195 {
 196         call_stack_push(expression.str());
 197         check_breakpoints();
 198         msg(" evaluating expression: ",call_stack_.back());
 199         variant v = expression.execute(variables,this);
 200         call_stack_set_value(v);
 201         call_stack_set_evaluated(true);
 202         msg(" evaluated expression: ",call_stack_.back()," to ",v.to_debug_string(true).c_str());
 203         check_breakpoints();
 204         call_stack_pop();
 205         return v;
 206 }

 262 class end_breakpoint : public base_breakpoint {
 263 public:
 264         end_breakpoint(formula_debugger &fdb)
 265                 : base_breakpoint(fdb,"End", true)
 266         {
 267         }
 268 
 269         virtual ~end_breakpoint()
 270         {
 271         }
 272 
 273         virtual bool is_break_now() const
 274         {
 275                 const std::list<debug_info> &call_stack = fdb_.get_call_stack();
 276                 if ((call_stack.size() == 1) && (call_stack.front().evaluated()) ) {
 277                         return true;
 278                 }
 279                 return false;
 280         }
