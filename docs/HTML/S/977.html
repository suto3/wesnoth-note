  17 /**

  19  * Contains a wrapper class for the @ref SDL_Window class.
  20  */
  21 
  22 #include <SDL_video.h>
  23 
  24 #include <string>
  25 
  26 class surface;
  27 struct SDL_Renderer;
  28 
  29 namespace sdl
  30 {
  31 
  32 /**
  33  * The wrapper class for the @ref SDL_Window class.
  34  *
  35  * At the moment of writing it is not certain yet how many windows will be
  36  * created. At least one as main window, but maybe the GUI dialogues will have
  37  * their own window. Once that is known it might be a good idea to evaluate
  38  * whether the class should become a singleton or not.
  39  *
  40  * The class also wraps several functions operating on @ref SDL_Window objects.
  41  * For functions not wrapped the class offers an implicit conversion operator
  42  * to a pointer to the @ref SDL_Window object it owns.
  43  */
  44 class window
  45 {
  46 public:
  47         window(const window&) = delete;
  48         window& operator=(const window&) = delete;
  49 
  50         /***** ***** ***** Constructor and destructor. ***** ***** *****/
  51 
  52         /**
  53          * Constructor.
  54          *
  55          * The function calls @ref SDL_CreateWindow and @ref SDL_CreateRenderer.
  56          *
  57          * @param title               Used as title for @ref SDL_CreateWindow.
  58          * @param x                   Used as x for @ref SDL_CreateWindow.
  59          * @param y                   Used as y for @ref SDL_CreateWindow.
  60          * @param w                   Used as w for @ref SDL_CreateWindow.
  61          * @param h                   Used as x for @ref SDL_CreateWindow.
  62          * @param window_flags        Used as flags for @ref SDL_CreateWindow.
  63          * @param render_flags        Used as flags for @ref SDL_CreateRenderer.
  64          */
  65         window(const std::string& title,
  66                         const int x,
  67                         const int y,
  68                         const int w,
  69                         const int h,
  70                         const Uint32 window_flags,
  71                         const Uint32 render_flags);
  72 
  73         ~window();
  74 
  75 
  76         /***** ***** ***** Operations. ***** ***** *****/
  77 
  78         /**
  79          * Wrapper for @ref SDL_SetWindowSize.
  80          *
  81          * @param w                   Used as w for @ref SDL_SetWindowSize.
  82          * @param h                   Used as x for @ref SDL_SetWindowSize.
  83          */
  84         void set_size(const int w, const int h);
  85 
  86         /**
  87          * Dummy function for centering the window.
  88          */
  89         void center();
  90 
  91         /**
  92          * Dummy function for maximizing the window.
  93          */
  94         void maximize();
  95 
  96         /**
  97          * Dummy function for restoring the window.
  98          */
  99         void restore();
 100 
 101         /**
 102          * Dummy function for returning the window to windowed mode.
 103          */
 104         void to_window();
 105 
 106         /**
 107          * Dummy function for setting the window to fullscreen mode.
 108          */
 109         void full_screen();
 110 
 111         /**
 112          * Clears the contents of the window with a given color.
 113          *
 114          * @param r                   Red value of the color.
 115          * @param g                   Green value of the color.
 116          * @param b                   Blue value of the color.
 117          */
 118         void fill(Uint8 r, Uint8 g, Uint8 b, Uint8 a = 0);
 119 
 120         /** Renders the contents of the window. */
 121         void render();
 122 
 123         /**
 124          * Sets the title of the window.
 125          *
 126          * This is a wrapper for @ref SDL_SetWindowTitle.
 127          *
 128          * @param title               The new title for the window.
 129          */
 130         void set_title(const std::string& title);
 131 
 132         /**
 133          * Sets the icon of the window.
 134          *
 135          * This is a wrapper for @ref SDL_SetWindowIcon.
 136          *
 137          * @note The @p icon is a @ref SDL_Surface and not a @ref SDL_Texture, this
 138          * is part of the SDL 2 API.
 139          *
 140          * @param icon                The new icon for the window.
 141          */
 142         void set_icon(const surface& icon);
 143 
 144         int get_flags();
 145 
 146         /**
 147          * Set mimimum size of the window.
 148          *
 149          * This is a wrapper for @ref SDL_SetWindowMinimumWindowSize.
 150          */
 151         void set_minimum_size(int min_w, int min_h);
 152 
 153         int get_display_index();
 154 
 155         /***** ***** ***** Conversion operators. ***** ***** *****/
 156 
 157         /**
 158          * Conversion operator to a SDL_Window*.
 159          */
 160         operator SDL_Window*();
 161 
 162 
 163 private:
 164         /**
 165          * Conversion operator to a SDL_Renderer*.
 166          *
 167          * @todo Evaluate whether the function should become public or not.
 168          */
 169         operator SDL_Renderer*();
 170 
 171         /***** ***** ***** Members. ***** ***** *****/
 172 
 173         /** The @ref SDL_Window we own. */
 174         SDL_Window* window_;
 175 
 176         /** The preferred pixel format for the renderer. */
 177         Uint32 pixel_format_;
 178 };
 179 
 180 } // namespace sdl
