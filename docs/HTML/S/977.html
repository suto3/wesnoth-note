  21 bool point_in_rect(int x, int y, const SDL_Rect& rect)
  22 {
  23         return x >= rect.x && y >= rect.y && x < rect.x + rect.w && y < rect.y + rect.h;
  24 }

  31 bool rects_overlap(const SDL_Rect& rect1, const SDL_Rect& rect2)
  32 {
  33         return (rect1.x < rect2.x+rect2.w && rect2.x < rect1.x+rect1.w &&
  34                         rect1.y < rect2.y+rect2.h && rect2.y < rect1.y+rect1.h);

  37 SDL_Rect intersect_rects(SDL_Rect const &rect1, SDL_Rect const &rect2)
  38 {
  39         SDL_Rect res;
  40         res.x = std::max<int>(rect1.x, rect2.x);
  41         res.y = std::max<int>(rect1.y, rect2.y);
  42         int w = std::min<int>(rect1.x + rect1.w, rect2.x + rect2.w) - res.x;
  43         int h = std::min<int>(rect1.y + rect1.h, rect2.y + rect2.h) - res.y;
  44         if (w <= 0 || h <= 0) return empty_rect;
  45         res.w = w;
  46         res.h = h;
  47         return res;
  48 }

  50 SDL_Rect union_rects(SDL_Rect const &rect1, SDL_Rect const &rect2)
  51 {
  52         if (rect1.w == 0 || rect1.h == 0) return rect2;
  53         if (rect2.w == 0 || rect2.h == 0) return rect1;
  54         SDL_Rect res;
  55         res.x = std::min<int>(rect1.x, rect2.x);
  56         res.y = std::min<int>(rect1.y, rect2.y);
  57         res.w = std::max<int>(rect1.x + rect1.w, rect2.x + rect2.w) - res.x;
  58         res.h = std::max<int>(rect1.y + rect1.h, rect2.y + rect2.h) - res.y;
  59         return res;
  60 }
