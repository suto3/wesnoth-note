  74 /**
  75  * Compares 2 version strings - which is newer.
  76  * - Args 1,3: version strings
  77  * - Arg 2: comparison operator (string)
  78  * - Ret 1: comparison result
  79  */
  80 static int intf_compare_versions(lua_State* L)
  81 {
  82         char const *v1 = luaL_checkstring(L, 1);
  83 
  84         const VERSION_COMP_OP vop = parse_version_op(luaL_checkstring(L, 2));
  85         if(vop == OP_INVALID) return luaL_argerror(L, 2, "unknown version comparison operator - allowed are ==, !=, <, <=, > and >=");
  86 
  87         char const *v2 = luaL_checkstring(L, 3);
  88 
  89         const bool result = do_version_check(version_info(v1), vop, version_info(v2));
  90         lua_pushboolean(L, result);
  91 
  92         return 1;
  93 }

 160 static int intf_name_generator(lua_State *L)
 161 {
 162         std::string type = luaL_checkstring(L, 1);
 163         name_generator* gen = nullptr;
 164         try {
 165                 if(type == "markov" || type == "markov_chain") {
 166                         std::vector<std::string> input;
 167                         if(lua_istable(L, 2)) {
 168                                 input = lua_check<std::vector<std::string>>(L, 2);
 169                         } else {
 170                                 input = utils::parenthetical_split(luaL_checkstring(L, 2), ',');
 171                         }
 172                         int chain_sz = luaL_optinteger(L, 3, 2);
 173                         int max_len = luaL_optinteger(L, 4, 12);
 174                         gen = new(L) markov_generator(input, chain_sz, max_len);
 175                         // Ensure the pointer didn't change when cast
 176                         assert(static_cast<void*>(gen) == dynamic_cast<markov_generator*>(gen));
 177                 } else if(type == "context_free" || type == "cfg" || type == "CFG") {
 178                         if(lua_istable(L, 2)) {
 179                                 std::map<std::string, std::vector<std::string>> data;
 180                                 for(lua_pushnil(L); lua_next(L, 2); lua_pop(L, 1)) {
 181                                         if(!lua_isstring(L, -2)) {
 182                                                 lua_pushstring(L, "CFG generator: invalid nonterminal name (must be a string)");
 183                                                 return lua_error(L);
 184                                         }
 185                                         if(lua_isstring(L, -1)) {
 186                                                 data[lua_tostring(L,-2)] = utils::split(lua_tostring(L,-1),'|');
 187                                         } else if(lua_istable(L, -1)) {
 188                                                 data[lua_tostring(L,-2)] = lua_check<std::vector<std::string>>(L, -1);
 189                                         } else {
 190                                                 lua_pushstring(L, "CFG generator: invalid noterminal value (must be a string or list of strings)");
 191                                                 return lua_error(L);
 192                                         }
 193                                 }
 194                                 if(!data.empty()) {
 195                                         gen = new(L) context_free_grammar_generator(data);
 196                                 }
 197                         } else {
 198                                 gen = new(L) context_free_grammar_generator(luaL_checkstring(L, 2));
 199                         }
 200                         if(gen) {
 201                                 assert(static_cast<void*>(gen) == dynamic_cast<context_free_grammar_generator*>(gen));
 202                         }
 203                 } else {
 204                         return luaL_argerror(L, 1, "should be either 'markov_chain' or 'context_free'");
 205                 }
 206         }
 207         catch (const name_generator_invalid_exception& ex) {
 208                 lua_pushstring(L, ex.what());
 209                 return lua_error(L);
 210         }
 211 
 212         // We set the metatable now, even if the generator is invalid, so that it
 213         // will be properly collected if it was invalid.
 214         luaL_getmetatable(L, Gen);
 215         lua_setmetatable(L, -2);
 216 
 217         return 1;
 218 }

 220 /**
 221 * Returns a random numer, same interface as math.random.
 222 */
 223 static int intf_random(lua_State *L)
 224 {
 225         if (lua_isnoneornil(L, 1)) {
 226                 double r = double(randomness::generator->next_random());
 227                 double r_max = double(std::numeric_limits<uint32_t>::max());
 228                 lua_push(L, r / (r_max + 1));
 229                 return 1;
 230         }
 231         else {
 232                 int32_t min;
 233                 int32_t max;
 234                 if (lua_isnumber(L, 2)) {
 235                         min = lua_check<int32_t>(L, 1);
 236                         max = lua_check<int32_t>(L, 2);
 237                 }
 238                 else {
 239                         min = 1;
 240                         max = lua_check<int32_t>(L, 1);
 241                 }
 242                 if (min > max) {
 243                         return luaL_argerror(L, 1, "min > max");
 244                 }
 245                 lua_push(L, randomness::generator->get_random_int(min, max));
 246                 return 1;
 247         }
 248 }

 258 /**
 259 * Logs a message
 260 * Arg 1: (optional) Logger
 261 * Arg 2: Message
 262 */
 263 int intf_log(lua_State *L) {
 264         const std::string& logger = lua_isstring(L, 2) ? luaL_checkstring(L, 1) : "";
 265         std::string msg = lua_isstring(L, 2) ? luaL_checkstring(L, 2) : luaL_checkstring(L, 1);
 266         if(msg.empty() || msg[msg.size() - 1] != '\n') {
 267                 msg += '\n';
 268         }
 269 
 270         if(logger == "err" || logger == "error") {
 271                 LOG_STREAM(err, log_user) << msg;
 272         } else if(logger == "warn" || logger == "wrn" || logger == "warning") {
 273                 LOG_STREAM(warn, log_user) << msg;
 274         } else if((logger == "debug" || logger == "dbg")) {
 275                 LOG_STREAM(debug, log_user) << msg;
 276         } else {
 277                 LOG_STREAM(info, log_user) << msg;
 278         }
 279         return 0;
 280 }

 542                 { "__call", &impl_name_generator_call},
 543                 { "__gc", &impl_name_generator_collect},
 544                 { nullptr, nullptr}
 545         };
 546         luaL_setfuncs(L, generator, 0);
 547 
 548         // Create formula bridge metatables
 549         cmd_log_ << lua_formula_bridge::register_metatables(L);
 550 
 551         // Loading ilua:
 552         cmd_log_ << "Loading ilua...\n";
 553 
 554         lua_settop(L, 0);
 555         luaW_getglobal(L, "wesnoth", "require");
 556         lua_pushstring(L, "lua/ilua.lua");
 557         if(protected_call(1, 1)) {
 558                 //run "ilua.set_strict()"
 559                 lua_pushstring(L, "set_strict");
 560                 lua_gettable(L, -2);
 561                 if (!this->protected_call(0,0, std::bind(&lua_kernel_base::log_error, this, _1, _2))) {
 562                         cmd_log_ << "Failed to activate strict mode.\n";
 563                 } else {
 564                         cmd_log_ << "Activated strict mode.\n";
 565                 }
 566 
 567                 lua_setglobal(L, "ilua"); //save ilua table as a global
 568         } else {
 569                 cmd_log_ << "Error: failed to load ilua.\n";
 570         }
 571         lua_settop(L, 0);
 572 }

 609 bool lua_kernel_base::protected_call(lua_State * L, int nArgs, int nRets, error_handler e_h)
 610 {
 611         int errcode = luaW_pcall_internal(L, nArgs, nRets);
 612 
 613         if (errcode != LUA_OK) {
 614                 char const * msg = lua_tostring(L, -1);
 615 
 616                 std::string context = "When executing, ";
 617                 if (errcode == LUA_ERRRUN) {
 618                         context += "Lua runtime error: ";
 619                 } else if (errcode == LUA_ERRERR) {
 620                         context += "Lua error in attached debugger: ";
 621                 } else if (errcode == LUA_ERRMEM) {
 622                         context += "Lua out of memory error: ";
 623                 } else if (errcode == LUA_ERRGCMM) {
 624                         context += "Lua error in garbage collection metamethod: ";
 625                 } else {
 626                         context += "unknown lua error: ";
 627                 }
 628                 if(lua_isstring(L, -1)) {
 629                         context +=  msg ? msg : "null string";
 630                 } else {
 631                         context += lua_typename(L, lua_type(L, -1));
 632                 }
 633 
 634                 lua_pop(L, 1);
 635 
 636                 e_h(context.c_str(), "Lua Error");
 637 
 638                 return false;
 639         }
 640 
 641         return true;
 642 }

 644 bool lua_kernel_base::load_string(char const * prog, error_handler e_h)
 645 {
 646         int errcode = luaL_loadstring(mState, prog);
 647         if (errcode != LUA_OK) {
 648                 char const * msg = lua_tostring(mState, -1);
 649                 std::string message = msg ? msg : "null string";
 650 
 651                 std::string context = "When parsing a string to lua, ";
 652 
 653                 if (errcode == LUA_ERRSYNTAX) {
 654                         context += " a syntax error";
 655                 } else if(errcode == LUA_ERRMEM){
 656                         context += " a memory error";
 657                 } else if(errcode == LUA_ERRGCMM) {
 658                         context += " an error in garbage collection metamethod";
 659                 } else {
 660                         context += " an unknown error";
 661                 }
 662 
 663                 lua_pop(mState, 1);
 664 
 665                 e_h(message.c_str(), context.c_str());
 666 
 667                 return false;
 668         }
 669         return true;
 670 }

 703 // Tests if a program resolves to an expression, and pretty prints it if it is, otherwise it runs it normally. Throws exceptions.
 704 void lua_kernel_base::interactive_run(char const * prog) {
 705         std::string experiment = "ilua._pretty_print(";
 706         experiment += prog;
 707         experiment += ")";
 708 
 709         error_handler eh = std::bind(&lua_kernel_base::throw_exception, this, _1, _2 );
 710 
 711         try {
 712                 // Try to load the experiment without syntax errors
 713                 this->load_string(experiment.c_str(), eh);
 714         } catch (game::lua_error &) {
 715                 this->throwing_run(prog, 0);    // Since it failed, fall back to the usual throwing_run, on the original input.
 716                 return;
 717         }
 718         // experiment succeeded, now run but log normally.
 719         cmd_log_ << "$ " << prog << "\n";
 720         this->protected_call(0, 0, eh);
 721 }
 722 /**
 723  * Loads and executes a Lua file.
 724  * - Arg 1: string containing the file name.
 725  * - Ret *: values returned by executing the file body.
 726  */
 727 int lua_kernel_base::intf_dofile(lua_State* L)
 728 {
 729         luaL_checkstring(L, 1);
 730         lua_rotate(L, 1, -1);
 731         if (lua_fileops::load_file(L) != 1) return 0;
 732         //^ should end with the file contents loaded on the stack. actually it will call lua_error otherwise, the return 0 is redundant.
 733         lua_rotate(L, 1, 1);
 734         // Using a non-protected call here appears to fix an issue in plugins.
 735         // The protected call isn't technically necessary anyway, because this function is called from Lua code,
 736         // which should already be in a protected environment.
 737         lua_call(L, lua_gettop(L) - 1, LUA_MULTRET);
 738         return lua_gettop(L);
 739 }

 741 /**
 742  * Loads and executes a Lua file, if there is no corresponding entry in wesnoth.package.
 743  * Stores the result of the script in wesnoth.package and returns it.
 744  * - Arg 1: string containing the file name.
 745  * - Ret 1: value returned by the script.
 746  */
 747 int lua_kernel_base::intf_require(lua_State* L)
 748 {
 749         const char * m = luaL_checkstring(L, 1);
 750         if(!m) {
 751                 return luaL_argerror(L, 1, "found a null string argument to wesnoth require");
 752         }
 753 
 754         // Check if there is already an entry.
 755 
 756         lua_getglobal(L, "wesnoth");
 757         lua_pushstring(L, "package");
 758         lua_rawget(L, -2);
 759         lua_pushvalue(L, 1);
 760         lua_rawget(L, -2);
 761         if(!lua_isnil(L, -1) && !game_config::debug_lua) {
 762                 return 1;
 763         }
 764         lua_pop(L, 1);
 765         lua_pushvalue(L, 1);
 766         // stack is now [packagename] [wesnoth] [package] [packagename]
 767 
 768         if(lua_fileops::load_file(L) != 1) {
 769                 // should end with the file contents loaded on the stack. actually it will call lua_error otherwise, the return 0 is redundant.
 770                 // stack is now [packagename] [wesnoth] [package] [chunk]
 771                 return 0;
 772         }
 773         DBG_LUA << "require: loaded a file, now calling it\n";
 774 
 775         if (!this->protected_call(L, 0, 1, std::bind(&lua_kernel_base::log_error, this, _1, _2))) {
 776                 // historically if wesnoth.require fails it just yields nil and some logging messages, not a lua error
 777                 return 0;
 778     }
 779         // stack is now [packagename] [wesnoth] [package] [results]
 780 
 781         lua_pushvalue(L, 1);
 782         lua_pushvalue(L, -2);
 783         // stack is now [packagename] [wesnoth] [package] [results] [packagename] [results]
 784         // Add the return value to the table.
 785 
 786         lua_settable(L, -4);
 787         // stack is now [packagename] [wesnoth] [package] [results]
 788         return 1;
 789 }

 844 /**
 845  * Gets all the global variable names in the Lua environment. This is useful for tab completion.
 846  */
 847 std::vector<std::string> lua_kernel_base::get_global_var_names()
 848 {
 849         std::vector<std::string> ret;
 850 
 851         lua_State *L = mState;
 852 
 853         int idx = lua_gettop(L);
 854         lua_getglobal(L, "_G");
 855         lua_pushnil(L);
 856 
 857         while (lua_next(L, idx+1) != 0) {
 858                 if (lua_isstring(L, -2)) {
 859                         ret.push_back(lua_tostring(L,-2));
 860                 }
 861                 lua_pop(L,1);
 862         }
 863         lua_settop(L, idx);
 864         return ret;
 865 }

 867 /**
 868  * Gets all attribute names of an extended variable name. This is useful for tab completion.
 869  */
 870 std::vector<std::string> lua_kernel_base::get_attribute_names(const std::string & input)
 871 {
 872         std::vector<std::string> ret;
 873         std::string base_path = input;
 874         size_t last_dot = base_path.find_last_of('.');
 875         std::string partial_name = base_path.substr(last_dot + 1);
 876         base_path.erase(last_dot);
 877         std::string load = "return " + base_path;
 878 
 879         lua_State* L = mState;
 880         int save_stack = lua_gettop(L);
 881         int result = luaL_loadstring(L, load.c_str());
 882         if(result != LUA_OK) {
 883                 // This isn't at error level because it's a really low priority error; it just means the user tried to tab-complete something that doesn't exist.
 884                 LOG_LUA << "Error when attempting tab completion:\n";
 885                 LOG_LUA << luaL_checkstring(L, -1) << '\n';
 886                 // Just return an empty list; no matches were found
 887                 lua_settop(L, save_stack);
 888                 return ret;
 889         }
 890 
 891         luaW_pcall(L, 0, 1);
 892         if(lua_istable(L, -1) || lua_isuserdata(L, -1)) {
 893                 int top = lua_gettop(L);
 894                 int obj = lua_absindex(L, -1);
 895                 if(luaL_getmetafield(L, obj, "__tab_enum") == LUA_TFUNCTION) {
 896                         lua_pushvalue(L, obj);
 897                         lua_pushlstring(L, partial_name.c_str(), partial_name.size());
 898                         luaW_pcall(L, 2, 1);
 899                         ret = lua_check<std::vector<std::string>>(L, -1);
 900                 } else {
 901                         lua_settop(L, top);
 902                         // Metafunction not found, so use lua_next to enumerate the table
 903                         for(lua_pushnil(L); lua_next(L, obj); lua_pop(L, 1)) {
 904                                 if(lua_isstring(L, -2)) {
 905                                         std::string attr = lua_tostring(L, -2);
 906                                         if(attr.empty()) {
 907                                                 continue;
 908                                         }
 909                                         if(!isalpha(attr[0]) && attr[0] != '_') {
 910                                                 continue;
 911                                         }
 912                                         if(std::any_of(attr.begin(), attr.end(), [](char c){
 913                                                 return !isalpha(c) && !isdigit(c) && c != '_';
 914                                         })) {
 915                                                 continue;
 916                                         }
 917                                         if(attr.substr(0, partial_name.size()) == partial_name) {
 918                                                 ret.push_back(base_path + "." + attr);
 919                                         }
 920                                 }
 921                         }
 922                 }
 923         }
 924         lua_settop(L, save_stack);
 925         return ret;
 926 }
