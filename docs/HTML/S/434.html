  59 static void parse_about_tag(const config& cfg, std::stringstream& ss)
  60 {
  61         if(!cfg.has_child("entry")) {
  62                 return;
  63         }
  64 
  65         ss << "\n" << "<span size='x-large'>" << cfg["title"] << "</span>" << "\n";
  66 
  67         for(const auto& entry : cfg.child_range("entry")) {
  68                 ss << entry["name"] << "\n";
  69         }
  70 }

  72 void end_credits::pre_show(window& window)
  73 {
  74         timer_id_ = add_timer(10, std::bind(&end_credits::timer_callback, this), true);
  75 
  76         // Delay a little before beginning the scrolling
  77         last_scroll_ = SDL_GetTicks() + 3000;
  78 
  79         connect_signal_pre_key_press(window, std::bind(&end_credits::key_press_callback, this, _5));
  80 
  81         std::stringstream ss;
  82         std::stringstream focus_ss;
  83 
  84         const config& credits_config = about::get_about_config();
  85 
  86         for(const auto& group : credits_config.child_range("credits_group")) {
  87                 std::stringstream& group_stream = (group["id"] == focus_on_) ? focus_ss : ss;
  88 
  89                 group_stream << "\n";
  90                 if(group.has_attribute("title")) {
  91                         group_stream << "<span size='xx-large'>" << group["title"] << "</span>" << "\n";
  92                 }
  93 
  94                 if(group["sort"].to_bool(false)) {
  95                         auto sections = group.child_range("about");
  96                         std::vector<config> sorted(sections.begin(), sections.end());
  97                         std::sort(sorted.begin(), sorted.end(), [](const config& entry1, const config& entry2) {
  98                                 return translation::compare(entry1["title"].str(), entry2["title"].str()) < 0;
  99                         });
 100                         for(const auto& about : sorted) {
 101                                 parse_about_tag(about, group_stream);
 102                         }
 103                 } else {
 104                         for(const auto& about : group.child_range("about")) {
 105                                 parse_about_tag(about, group_stream);
 106                         }
 107                 }
 108         }
 109 
 110         // If a section is focused, move it to the top
 111         if(!focus_ss.str().empty()) {
 112                 focus_ss << ss.rdbuf();
 113         }
 114 
 115         // Get the appropriate background images
 116         backgrounds_ = about::get_background_images(focus_on_);
 117 
 118         if(backgrounds_.empty()) {
 119                 backgrounds_.push_back(game_config::images::game_title_background);
 120         }
 121 
 122         // TODO: implement showing all available images as the credits scroll
 123         window.get_canvas(0).set_variable("background_image", wfl::variant(backgrounds_[0]));
 124 
 125         text_widget_ = find_widget<scroll_label>(&window, "text", false, true);
 126 
 127         text_widget_->set_use_markup(true);
 128         text_widget_->set_label((focus_ss.str().empty() ? ss : focus_ss).str());
 129 
 130         // HACK: always hide the scrollbar, even if it's needed.
 131         // This should probably be implemented as a scrollbar mode.
 132         // Also, for some reason hiding the whole grid doesn't work, and the elements need to be hidden manually
 133         if(grid* v_grid = dynamic_cast<grid*>(text_widget_->find("_vertical_scrollbar_grid", false))) {
 134                 find_widget<scrollbar_base>(v_grid, "_vertical_scrollbar", false).set_visible(widget::visibility::hidden);
 135 
 136                 // TODO: enable again if e24336afeb7 is reverted.
 137                 //find_widget<repeating_button>(v_grid, "_half_page_up", false).set_visible(widget::visibility::hidden);
 138                 //find_widget<repeating_button>(v_grid, "_half_page_down", false).set_visible(widget::visibility::hidden);
 139         }
 140 }

 142 void end_credits::timer_callback()
 143 {
 144         uint32_t now = SDL_GetTicks();
 145         if(last_scroll_ > now) {
 146                 return;
 147         }
 148 
 149         uint32_t missed_time = now - last_scroll_;
 150 
 151         unsigned int cur_pos = text_widget_->get_vertical_scrollbar_item_position();
 152 
 153         // Calculate how far the text should have scrolled by now
 154         // The division by 1000 is to convert milliseconds to seconds.
 155         unsigned int needed_dist = missed_time * scroll_speed_ / 1000;
 156 
 157         text_widget_->set_vertical_scrollbar_item_position(cur_pos + needed_dist);
 158 
 159         last_scroll_ = now;
 160 
 161         if(text_widget_->vertical_scrollbar_at_end()) {
 162                 remove_timer(timer_id_);
 163         }
 164 }

 166 void end_credits::key_press_callback(const SDL_Keycode key)
 167 {
 168         if(key == SDLK_UP && scroll_speed_ < 400) {
 169                 scroll_speed_ <<= 1;
 170         }
 171 
 172         if(key == SDLK_DOWN && scroll_speed_ > 50) {
 173                 scroll_speed_ >>= 1;
 174         }
 175 }
