  91 bool load_language_list()
  92 {
  93         config cfg;
  94         try {
  95                 filesystem::scoped_istream stream = preprocess_file(filesystem::get_wml_location("hardwired/language.cfg"));
  96                 read(cfg, *stream);
  97         } catch(config::error &) {
  98                 return false;
  99         }
 100 
 101         known_languages.clear();
 102         known_languages.emplace_back("", t_string(N_("System default language"), "wesnoth"), "ltr", "", "A");
 103 
 104         for (const config &lang : cfg.child_range("locale"))
 105         {
 106                 known_languages.emplace_back(
 107                         lang["locale"], lang["name"], lang["dir"],
 108                         lang["alternates"], lang["sort_name"]);
 109         }
 110 
 111         return true;
 112 }

 122 static void wesnoth_setlocale(int category, const std::string& slocale,
 123         std::vector<std::string> const *alternates)
 124 {
 125         std::string locale = slocale;
 126         // FIXME: ideally we should check LANGUAGE and on first invocation
 127         // use that value, so someone with es would get the game in Spanish
 128         // instead of en_US the first time round
 129         // LANGUAGE overrides other settings, so for now just get rid of it
 130         // FIXME: add configure check for unsetenv
 131 
 132         //category is never LC_MESSAGES since that case was moved to gettext.cpp to remove the dependency to libintl.h in this file
 133         //that's why code like if (category == LC_MESSAGES) is outcommented here.
 134 #ifndef _WIN32
 135         unsetenv ("LANGUAGE"); // void so no return value to check
 136 #endif
 137 #ifdef __APPLE__
 138         //if (category == LC_MESSAGES && setenv("LANG", locale.c_str(), 1) == -1) {
 139         //      ERR_G << "setenv LANG failed: " << strerror(errno);
 140         //}
 141 #endif
 142 
 143 #ifdef _WIN32
 144         std::string win_locale(locale, 0, 2);
 145         #include "language_win32.ii"
 146         //if(category == LC_MESSAGES) {
 147         //      SetEnvironmentVariableA("LANG", win_locale.c_str());
 148         //      std::string env = "LANGUAGE=" + locale;
 149         //      _putenv(env.c_str());
 150         //      return;
 151         //}
 152         locale = win_locale;
 153 #endif
 154 
 155         char *res = nullptr;
 156         std::vector<std::string>::const_iterator i;
 157         if (alternates) i = alternates->begin();
 158 
 159         for (;;)
 160         {
 161                 std::string lang = locale, extra;
 162                 std::string::size_type pos = locale.find('@');
 163                 if (pos != std::string::npos) {
 164                         lang.erase(pos);
 165                         extra = locale.substr(pos);
 166                 }
 167 
 168                 /*
 169                  * The "" is the last item to work-around a problem in glibc picking
 170                  * the non utf8 locale instead an utf8 version if available.
 171                  */
 172                 char const *encoding[] { ".utf-8", ".UTF-8", "" };
 173                 for (int j = 0; j != 3; ++j)
 174                 {
 175                         locale = lang + encoding[j] + extra;
 176                         res = std::setlocale(category, locale.c_str());
 177                         if (res) {
 178                                 LOG_G << "Set locale to '" << locale << "' result: '" << res << "'.\n";
 179                                 goto done;
 180                         }
 181                 }
 182 
 183                 if (!alternates || i == alternates->end()) break;
 184                 locale = *i;
 185                 ++i;
 186         }
 187 
 188         WRN_G << "setlocale() failed for '" << slocale << "'." << std::endl;
 189 
 190         if (category == LC_TIME) {
 191                 time_locale_correct() = false;
 192         }
 193 
 194 #ifndef _WIN32
 195                 //if(category == LC_MESSAGES) {
 196                 //      WRN_G << "Setting LANGUAGE to '" << slocale << "'." << std::endl;
 197                 //      setenv("LANGUAGE", slocale.c_str(), 1);
 198                 //      std::setlocale(LC_MESSAGES, "");
 199                 //}
 200 #endif
 201 
 202         done:
 203         DBG_G << "Numeric locale: " << std::setlocale(LC_NUMERIC, nullptr) << '\n';
 204         DBG_G << "Full locale: " << std::setlocale(LC_ALL, nullptr) << '\n';
 205 }

 207 void set_language(const language_def& locale)
 208 {
 209         strings_.clear();
 210 
 211         std::string locale_lc;
 212         locale_lc.resize(locale.localename.size());
 213         std::transform(locale.localename.begin(),locale.localename.end(),locale_lc.begin(),tolower);
 214 
 215         current_language = locale;
 216         time_locale_correct() = true;
 217 
 218         wesnoth_setlocale(LC_COLLATE, locale.localename, &locale.alternates);
 219         wesnoth_setlocale(LC_TIME, locale.localename, &locale.alternates);
 220         translation::set_language(locale.localename, &locale.alternates);
 221         load_strings(false);
 222 }

 224 bool load_strings(bool complain)
 225 {
 226         DBG_G << "Loading strings\n";
 227         config cfg;
 228 
 229         LOG_G << "There are " << languages_.size() << " [language] blocks\n";
 230         if (complain && languages_.empty()) {
 231                 std::cerr << "No [language] block found\n";
 232                 return false;
 233         }
 234         for (const config &lang : languages_) {
 235                 DBG_G << "[language]\n";
 236                 for (const config::attribute &j : lang.attribute_range()) {
 237                         DBG_G << j.first << "=\"" << j.second << "\"\n";
 238                         strings_[j.first] = j.second;
 239                 }
 240                 DBG_G << "[/language]\n";
 241         }
 242         DBG_G << "done\n";
 243 
 244         return true;
 245 }

 289 void init_textdomains(const config& cfg)
 290 {
 291         for (const config &t : cfg.child_range("textdomain"))
 292         {
 293                 const std::string &name = t["name"];
 294                 const std::string &path = t["path"];
 295 
 296                 if(path.empty()) {
 297                         t_string::add_textdomain(name, filesystem::get_intl_dir());
 298                 } else {
 299                         std::string location = filesystem::get_binary_dir_location("", path);
 300 
 301                         if (location.empty()) {
 302                                 //if location is empty, this causes a crash on Windows, so we
 303                                 //disallow adding empty domains
 304                                 WRN_G << "no location found for '" << path << "', skipping textdomain" << std::endl;
 305                         } else {
 306                                 t_string::add_textdomain(name, location);
 307                         }
 308                 }
 309         }
 310 }
