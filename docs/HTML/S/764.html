  46 void display_chat_manager::add_chat_message(const time_t& time, const std::string& speaker,
  47                 int side, const std::string& message, events::chat_handler::MESSAGE_TYPE type,
  48                 bool bell)
  49 {
  50         const bool whisper = speaker.find("whisper: ") == 0;
  51         std::string sender = speaker;
  52         if (whisper) {
  53                 sender.assign(speaker, 9, speaker.size());
  54                 add_whisperer( sender );
  55         }
  56         //remove disconnected user from whisperer
  57         std::string::size_type pos = message.find(" has disconnected");
  58         if (pos != std::string::npos){
  59                 for(std::set<std::string>::const_iterator w = whisperers().begin(); w != whisperers().end(); ++w){
  60                         if (*w == message.substr(0,pos)) remove_whisperer(*w);
  61                 }
  62         }
  63 
  64         if (!preferences::parse_should_show_lobby_join(sender, message)) return;
  65         if (preferences::is_ignored(sender)) return;
  66 
  67         preferences::parse_admin_authentication(sender, message);
  68 
  69         bool is_observer = false;
  70         { //TODO: Clean this block up somehow
  71 
  72                 const game_board * board = dynamic_cast<const game_board*>(&my_disp_.get_disp_context());
  73 
  74                 if (board) {
  75                         is_observer = board->is_observer();
  76                 }
  77         }
  78 
  79         if (bell) {
  80                 if ((type == events::chat_handler::MESSAGE_PRIVATE && (!is_observer || whisper))
  81                         || utils::word_match(message, preferences::login())) {
  82                         mp_ui_alerts::private_message(false, sender, message);
  83                 } else if (preferences::is_friend(sender)) {
  84                         mp_ui_alerts::friend_message(false, sender, message);
  85                 } else if (sender == "server") {
  86                         mp_ui_alerts::server_message(false, sender, message);
  87                 } else {
  88                         mp_ui_alerts::public_message(false, sender, message);
  89                 }
  90         }
  91 
  92         bool action = false;
  93 
  94         std::string msg;
  95 
  96         if (message.compare(0,4,"/me ") == 0) {
  97                 msg.assign(message, 4, message.size());
  98                 action = true;
  99         } else {
 100                 msg = message;
 101         }
 102 
 103         try {
 104                 // We've had a joker who send an invalid utf-8 message to crash clients
 105                 // so now catch the exception and ignore the message.
 106                 msg = my_disp_.video().faked() ? "" : font::word_wrap_text(msg,font::SIZE_NORMAL,my_disp_.map_outside_area().w*3/4);
 107         } catch (utf8::invalid_utf8_exception&) {
 108                 ERR_NG << "Invalid utf-8 found, chat message is ignored." << std::endl;
 109                 return;
 110         }
 111 
 112         int ypos = chat_message_x;
 113         for(std::vector<chat_message>::const_iterator m = chat_messages_.begin(); m != chat_messages_.end(); ++m) {
 114                 ypos += std::max(font::get_floating_label_rect(m->handle).h,
 115                         font::get_floating_label_rect(m->speaker_handle).h);
 116         }
 117         color_t speaker_color {255,255,255,SDL_ALPHA_OPAQUE};
 118         if(side >= 1) {
 119                 speaker_color = team::get_side_color_range(side).mid();
 120         }
 121 
 122         color_t message_color = chat_message_color;
 123         std::stringstream str;
 124         std::stringstream message_str;
 125 
 126         if(type ==  events::chat_handler::MESSAGE_PUBLIC) {
 127                 if(action) {
 128                         str << "<" << speaker << " " << msg << ">";
 129                         message_color = speaker_color;
 130                         message_str << " ";
 131                 } else {
 132                         if (!speaker.empty())
 133                                 str << "<" << speaker << ">";
 134                         message_str << msg;
 135                 }
 136         } else {
 137                 if(action) {
 138                         str << "*" << speaker << " " << msg << "*";
 139                         message_color = speaker_color;
 140                         message_str << " ";
 141                 } else {
 142                         if (!speaker.empty())
 143                                 str << "*" << speaker << "*";
 144                         message_str << msg;
 145                 }
 146         }
 147 
 148         // Prepend message with timestamp.
 149         std::stringstream message_complete;
 150         message_complete << preferences::get_chat_timestamp(time) << str.str();
 151 
 152         const SDL_Rect rect = my_disp_.map_outside_area();
 153 
 154         font::floating_label spk_flabel(message_complete.str());
 155         spk_flabel.set_font_size(font::SIZE_NORMAL);
 156         spk_flabel.set_color(speaker_color);
 157         spk_flabel.set_position(rect.x + chat_message_x, rect.y + ypos);
 158         spk_flabel.set_clip_rect(rect);
 159         spk_flabel.set_alignment(font::LEFT_ALIGN);
 160         spk_flabel.set_bg_color(chat_message_bg);
 161         spk_flabel.set_border_size(chat_message_border);
 162         spk_flabel.use_markup(false);
 163 
 164         int speaker_handle = font::add_floating_label(spk_flabel);
 165 
 166         font::floating_label msg_flabel(message_str.str());
 167         msg_flabel.set_font_size(font::SIZE_NORMAL);
 168         msg_flabel.set_color(message_color);
 169         msg_flabel.set_position(rect.x + chat_message_x + font::get_floating_label_rect(speaker_handle).w,
 170         rect.y + ypos);
 171         msg_flabel.set_clip_rect(rect);
 172         msg_flabel.set_alignment(font::LEFT_ALIGN);
 173         msg_flabel.set_bg_color(chat_message_bg);
 174         msg_flabel.set_border_size(chat_message_border);
 175         msg_flabel.use_markup(false);
 176 
 177         int message_handle = font::add_floating_label(msg_flabel);
 178 
 179         chat_messages_.emplace_back(speaker_handle,message_handle);
 180 
 181         prune_chat_messages();
 182 }

 184 void display_chat_manager::prune_chat_messages(bool remove_all)
 185 {
 186         const unsigned message_aging = preferences::chat_message_aging();
 187         const unsigned message_ttl = remove_all ? 0 : message_aging * 60 * 1000;
 188         const unsigned max_chat_messages = preferences::chat_lines();
 189         int movement = 0;
 190 
 191         if(message_aging != 0 || remove_all || chat_messages_.size() > max_chat_messages) {
 192                 while (!chat_messages_.empty() &&
 193                        (chat_messages_.front().created_at + message_ttl < SDL_GetTicks() ||
 194                         chat_messages_.size() > max_chat_messages))
 195                 {
 196                         const chat_message &old = chat_messages_.front();
 197                         movement += font::get_floating_label_rect(old.handle).h;
 198                         font::remove_floating_label(old.speaker_handle);
 199                         font::remove_floating_label(old.handle);
 200                         chat_messages_.erase(chat_messages_.begin());
 201                 }
 202         }
 203 
 204         for(const chat_message &cm : chat_messages_) {
 205                 font::move_floating_label(cm.speaker_handle, 0, - movement);
 206                 font::move_floating_label(cm.handle, 0, - movement);
 207         }
 208 }
