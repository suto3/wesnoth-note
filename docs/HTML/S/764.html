  61 turn_info::PROCESS_DATA_RESULT turn_info::sync_network()
  62 {
  63         //there should be nothing left on the replay and we should get turn_info::PROCESS_CONTINUE back.
  64         turn_info::PROCESS_DATA_RESULT retv = replay_to_process_data_result(do_replay());
  65         if(resources::controller->is_networked_mp()) {
  66 
  67                 //receive data first, and then send data. When we sent the end of
  68                 //the AI's turn, we don't want there to be any chance where we
  69                 //could get data back pertaining to the next turn.
  70                 config cfg;
  71                 while( (retv == turn_info::PROCESS_CONTINUE) &&  network_reader_.read(cfg)) {
  72                         retv = process_network_data(cfg);
  73                         cfg.clear();
  74                 }
  75                 send_data();
  76         }
  77         return retv;
  78 }

 108 turn_info::PROCESS_DATA_RESULT turn_info::process_network_data_from_reader()
 109 {
 110         config cfg;
 111         while(this->network_reader_.read(cfg))
 112         {
 113                 PROCESS_DATA_RESULT res = process_network_data(cfg);
 114                 if(res != PROCESS_CONTINUE)
 115                 {
 116                         return res;
 117                 }
 118                 cfg.clear();
 119         }
 120         return PROCESS_CONTINUE;
 121 }

 123 turn_info::PROCESS_DATA_RESULT turn_info::process_network_data(const config& cfg)
 124 {
 125         // the simple wesnothserver implementation in wesnoth was removed years ago.
 126         assert(cfg.all_children_count() == 1);
 127         assert(cfg.attribute_range().empty());
 128         if(!resources::recorder->at_end())
 129         {
 130                 ERR_NW << "processing network data while still having data on the replay." << std::endl;
 131         }
 132 
 133         if (const config &message = cfg.child("message"))
 134         {
 135                 resources::screen->get_chat_manager().add_chat_message(time(nullptr), message["sender"], message["side"],
 136                                 message["message"], events::chat_handler::MESSAGE_PUBLIC,
 137                                 preferences::message_bell());
 138         }
 139         else if (const config &whisper = cfg.child("whisper") /*&& is_observer()*/)
 140         {
 141                 resources::screen->get_chat_manager().add_chat_message(time(nullptr), "whisper: " + whisper["sender"].str(), 0,
 142                                 whisper["message"], events::chat_handler::MESSAGE_PRIVATE,
 143                                 preferences::message_bell());
 144         }
 145         else if (const config &observer = cfg.child("observer") )
 146         {
 147                 resources::screen->get_chat_manager().add_observer(observer["name"]);
 148         }
 149         else if (const config &observer_quit = cfg.child("observer_quit"))
 150         {
 151                 resources::screen->get_chat_manager().remove_observer(observer_quit["name"]);
 152         }
 153         else if (cfg.child("leave_game")) {
 154                 throw ingame_wesnothd_error("");
 155         }
 156         else if (const config &turn = cfg.child("turn"))
 157         {
 158                 return handle_turn(turn);
 159         }
 160         else if (cfg.has_child("whiteboard"))
 161         {
 162                 resources::whiteboard->process_network_data(cfg);
 163         }
 164         else if (const config &change = cfg.child("change_controller"))
 165         {
 166                 if(change.empty()) {
 167                         ERR_NW << "Bad [change_controller] signal from server, [change_controller] tag was empty." << std::endl;
 168                         return PROCESS_CONTINUE;
 169                 }
 170 
 171                 const int side = change["side"].to_int();
 172                 const bool is_local = change["is_local"].to_bool();
 173                 const std::string player = change["player"];
 174                 const size_t index = side - 1;
 175                 if(index >= resources::gameboard->teams().size()) {
 176                         ERR_NW << "Bad [change_controller] signal from server, side out of bounds: " << change.debug() << std::endl;
 177                         return PROCESS_CONTINUE;
 178                 }
 179 
 180                 const team & tm = resources::gameboard->teams().at(index);
 181                 const bool was_local = tm.is_local();
 182 
 183                 resources::gameboard->side_change_controller(side, is_local, player);
 184 
 185                 if (!was_local && tm.is_local()) {
 186                         resources::controller->on_not_observer();
 187                 }
 188 
 189                 if (resources::gameboard->is_observer() || (resources::gameboard->teams())[resources::screen->playing_team()].is_local_human()) {
 190                         resources::screen->set_team(resources::screen->playing_team());
 191                         resources::screen->redraw_everything();
 192                         resources::screen->recalculate_minimap();
 193                 } else if (tm.is_local_human()) {
 194                         resources::screen->set_team(side - 1);
 195                         resources::screen->redraw_everything();
 196                         resources::screen->recalculate_minimap();
 197                 }
 198 
 199                 resources::whiteboard->on_change_controller(side,tm);
 200 
 201                 resources::screen->labels().recalculate_labels();
 202 
 203                 const bool restart = resources::screen->playing_side() == side && (was_local || tm.is_local());
 204                 return restart ? PROCESS_RESTART_TURN : PROCESS_CONTINUE;
 205         }
 206 
 207         else if (const config &side_drop_c = cfg.child("side_drop"))
 208         {
 209                 const int  side_drop = side_drop_c["side_num"].to_int(0);
 210                 size_t index = side_drop -1;
 211 
 212                 bool restart = side_drop == resources::screen->playing_side();
 213 
 214                 if (index >= resources::gameboard->teams().size()) {
 215                         ERR_NW << "unknown side " << side_drop << " is dropping game" << std::endl;
 216                         throw ingame_wesnothd_error("");
 217                 }
 218 
 219                 team::CONTROLLER ctrl;
 220                 if(!ctrl.parse(side_drop_c["controller"])) {
 221                         ERR_NW << "unknown controller type issued from server on side drop: " << side_drop_c["controller"] << std::endl;
 222                         throw ingame_wesnothd_error("");
 223                 }
 224 
 225                 if (ctrl == team::CONTROLLER::AI) {
 226                         resources::gameboard->side_drop_to(side_drop, ctrl);
 227                         return restart ? PROCESS_RESTART_TURN:PROCESS_CONTINUE;
 228                 }
 229                 //null controlled side cannot be dropped becasue they aren't controlled by anyone.
 230                 else if (ctrl != team::CONTROLLER::HUMAN) {
 231                         ERR_NW << "unknown controller type issued from server on side drop: " << ctrl.to_cstring() << std::endl;
 232                         throw ingame_wesnothd_error("");
 233                 }
 234 
 235                 int action = 0;
 236                 int first_observer_option_idx = 0;
 237                 int control_change_options = 0;
 238                 bool has_next_scenario = !resources::gamedata->next_scenario().empty() && resources::gamedata->next_scenario() != "null";
 239 
 240                 std::vector<std::string> observers;
 241                 std::vector<const team *> allies;
 242                 std::vector<std::string> options;
 243 
 244                 const team &tm = resources::gameboard->teams()[index];
 245 
 246                 for (const team &t : resources::gameboard->teams()) {
 247                         if (!t.is_enemy(side_drop) && !t.is_local_human() && !t.is_local_ai() && !t.is_network_ai() && !t.is_empty()
 248                                 && t.current_player() != tm.current_player()) {
 249                                 allies.push_back(&t);
 250                         }
 251                 }
 252 
 253                 // We want to give host chance to decide what to do for side
 254                 if (!resources::controller->is_linger_mode() || has_next_scenario) {
 255                         utils::string_map t_vars;
 256 
 257                         //get all allies in as options to transfer control
 258                         for (const team *t : allies) {
 259                                 //if this is an ally of the dropping side and it is not us (choose local player
 260                                 //if you want that) and not ai or empty and if it is not the dropping side itself,
 261                                 //get this team in as well
 262                                 t_vars["player"] = t->current_player();
 263                                 options.emplace_back(vgettext("Give control to their ally $player", t_vars));
 264                                 control_change_options++;
 265                         }
 266 
 267                         first_observer_option_idx = options.size();
 268 
 269                         //get all observers in as options to transfer control
 270                         for (const std::string &screen_observers : resources::screen->observers()) {
 271                                 t_vars["player"] = screen_observers;
 272                                 options.emplace_back(vgettext("Give control to observer $player", t_vars));
 273                                 observers.push_back(screen_observers);
 274                                 control_change_options++;
 275                         }
 276 
 277                         options.emplace_back(_("Replace with AI"));
 278                         options.emplace_back(_("Replace with local player"));
 279                         options.emplace_back(_("Set side to idle"));
 280                         options.emplace_back(_("Save and abort game"));
 281 
 282                         t_vars["player"] = tm.current_player();
 283                         const std::string gettext_message =  vgettext("$player has left the game. What do you want to do?", t_vars);
 284                         gui2::dialogs::simple_item_selector dlg("", gettext_message, options);
 285                         dlg.set_single_button(true);
 286                         dlg.show(resources::screen->video());
 287                         action = dlg.selected_index();
 288 
 289                         // If esc was pressed, default to setting side to idle
 290                         if (action == -1) {
 291                                 action = control_change_options + 2;
 292                         }
 293                 } else {
 294                         // Always set leaving side to idle if in linger mode and there is no next scenario
 295                         action = 2;
 296                 }
 297 
 298                 if (action < control_change_options) {
 299                         // Grant control to selected ally
 300 
 301                         {
 302                                 // Server thinks this side is ours now so in case of error transferring side we have to make local state to same as what server thinks it is.
 303                                 resources::gameboard->side_drop_to(side_drop, team::CONTROLLER::HUMAN, team::PROXY_CONTROLLER::PROXY_IDLE);
 304                         }
 305 
 306                         if (action < first_observer_option_idx) {
 307                                 change_side_controller(side_drop, allies[action]->current_player());
 308                         } else {
 309                                 change_side_controller(side_drop, observers[action - first_observer_option_idx]);
 310                         }
 311 
 312                         return restart ? PROCESS_RESTART_TURN : PROCESS_CONTINUE;
 313                 } else {
 314                         action -= control_change_options;
 315 
 316                         //make the player an AI, and redo this turn, in case
 317                         //it was the current player's team who has just changed into
 318                         //an AI.
 319                         switch(action) {
 320                                 case 0:
 321                                         resources::controller->on_not_observer();
 322                                         resources::gameboard->side_drop_to(side_drop, team::CONTROLLER::HUMAN, team::PROXY_CONTROLLER::PROXY_AI);
 323 
 324                                         return restart?PROCESS_RESTART_TURN:PROCESS_CONTINUE;
 325 
 326                                 case 1:
 327                                         resources::controller->on_not_observer();
 328                                         resources::gameboard->side_drop_to(side_drop, team::CONTROLLER::HUMAN, team::PROXY_CONTROLLER::PROXY_HUMAN);
 329 
 330                                         return restart?PROCESS_RESTART_TURN:PROCESS_CONTINUE;
 331                                 case 2:
 332                                         resources::gameboard->side_drop_to(side_drop, team::CONTROLLER::HUMAN, team::PROXY_CONTROLLER::PROXY_IDLE);
 333 
 334                                         return restart?PROCESS_RESTART_TURN:PROCESS_CONTINUE;
 335 
 336                                 case 3:
 337                                         //The user pressed "end game". Don't throw a network error here or he will get
 338                                         //thrown back to the title screen.
 339                                         do_save();
 340                                         throw_quit_game_exception();
 341                                 default:
 342                                         break;
 343                         }
 344                 }
 345         }
 346 
 347         // The host has ended linger mode in a campaign -> enable the "End scenario" button
 348         // and tell we did get the notification.
 349         else if (cfg.child("notify_next_scenario")) {
 350                 std::shared_ptr<gui::button> btn_end = resources::screen->find_action_button("button-endturn");
 351                 if(btn_end) {
 352                         btn_end->enable(true);
 353                 }
 354                 return PROCESS_END_LINGER;
 355         }
 356 
 357         //If this client becomes the new host, notify the play_controller object about it
 358         else if (cfg.child("host_transfer")){
 359                 host_transfer_.notify_observers();
 360         }
 361         else
 362         {
 363                 ERR_NW << "found unknown command:\n" << cfg.debug() << std::endl;
 364         }
 365 
 366         return PROCESS_CONTINUE;
 367 }

 379 turn_info::PROCESS_DATA_RESULT turn_info::replay_to_process_data_result(REPLAY_RETURN replayreturn)
 380 {
 381         switch(replayreturn)
 382         {
 383         case REPLAY_RETURN_AT_END:
 384                 return PROCESS_CONTINUE;
 385         case REPLAY_FOUND_DEPENDENT:
 386                 return PROCESS_FOUND_DEPENDENT;
 387         case REPLAY_FOUND_END_TURN:
 388                 return PROCESS_END_TURN;
 389         case REPLAY_FOUND_END_LEVEL:
 390                 return PROCESS_END_LEVEL;
 391         default:
 392                 assert(false);
 393                 throw "found invalid REPLAY_RETURN";
 394         }
 395 }
