 117 /**

 123 static int impl_side_set(lua_State *L)
 124 {
 125         // Hidden metamethod, so arg1 has to be a pointer to a team.

 129         // Find the corresponding attribute.
 130         modify_int_attrib("gold", t.set_gold(value));
 131         modify_tstring_attrib("objectives", t.set_objectives(value, true));
 132         //maybe add a setter for save_id too?
 133         modify_int_attrib("village_gold", t.set_village_gold(value));

 139         modify_bool_attrib("scroll_to_leader", t.set_scroll_to_leader(value));
 140         modify_tstring_attrib("user_team_name", t.change_team(t.team_name(), value));
 141         modify_string_attrib("team_name", t.change_team(value, t.user_team_name()));

 146         modify_bool_attrib("carryover_add", t.set_carryover_add(value));
 147         modify_bool_attrib("lost", t.set_lost(value));
 148         modify_bool_attrib("persistent", t.set_persistent(value));
 149         modify_bool_attrib("suppress_end_turn_confirmation", t.set_no_turn_confirmation(value));
 150         modify_bool_attrib("shroud", t.set_shroud(value));

 152         modify_string_attrib("flag_icon", t.set_flag_icon(value));
 153         modify_string_attrib("share_vision", {
 154                 team::SHARE_VISION v;
 155                 if(v.parse(value)) {
 156                         t.set_share_vision(v);
 157                 } else {
 158                         return luaL_argerror(L, 3, "Invalid share_vision value (should be 'all', 'none', or 'shroud')");
 159                 }
 160         });
 161 
 162         if (strcmp(m, "carryover_bonus") == 0) {
 163                 t.set_carryover_bonus(luaL_checknumber(L, 3));
 164                 return 0;
 165         }
 166 
 167         if (strcmp(m, "recruit") == 0) {
 168                 t.set_recruits(std::set<std::string>());
 169                 if (!lua_istable(L, 3)) return 0;
 170                 for (int i = 1;; ++i) {
 171                         lua_rawgeti(L, 3, i);
 172                         if (lua_isnil(L, -1)) break;
 173                         t.add_recruit(lua_tostring(L, -1));
 174                         lua_pop(L, 1);
 175                 }
 176                 return 0;
 177         }
 178 
 179         std::string err_msg = "unknown modifiable property of side: ";
 180         err_msg += m;
 181         return luaL_argerror(L, 2, err_msg.c_str());
 182 }

 184 static int impl_side_equal(lua_State *L)
 185 {
 186         // Hidden metamethod, so arg1 has to be a pointer to a team.
 187         team &t1 = luaW_checkteam(L, 1);
 188         if(team* t2 = luaW_toteam(L, 2)) {
 189                 lua_pushboolean(L, t1.side() == t2->side());
 190         } else {
 191                 lua_pushboolean(L, false);
 192         }
 193         return 1;
 194 }

 198         std::string register_metatable(lua_State * L)
 199         {
 200                 luaL_newmetatable(L, Team);
 201 
 202                 static luaL_Reg const callbacks[] {
 203                         { "__index",        &impl_side_get},
 204                         { "__newindex",     &impl_side_set},
 205                         { "__eq",               &impl_side_equal},
 206                         { nullptr, nullptr }
 207                 };
 208                 luaL_setfuncs(L, callbacks, 0);
 209 
 210                 lua_pushstring(L, Team);
 211                 lua_setfield(L, -2, "__metatable");
 212                 // Side methods
 213                 luaW_getglobal(L, "wesnoth", "match_side");
 214                 lua_setfield(L, -2, "matches");
 215 
 216                 return "Adding getside metatable...\n";
 217         }
 218 }

 220 void luaW_pushteam(lua_State *L, team & tm)
 221 {
 222         team** t = static_cast<team**>(lua_newuserdata(L, sizeof(team*)));
 223         *t = &tm;
 224         luaL_setmetatable(L, Team);
 225 }

 227 team& luaW_checkteam(lua_State* L, int idx)
 228 {
 229         return **static_cast<team **>(luaL_checkudata(L, idx, Team));
 230 }
 231 
 232 team* luaW_toteam(lua_State* L, int idx)
 233 {
 234         if(void* p = luaL_testudata(L, idx, Team)) {
 235                 return *static_cast<team **>(p);
 236         }
 237         return nullptr;
 238 }
