  31 template<size_t len>
  32 std::string encode_hash(const std::array<uint8_t, len>& input) {
  33         std::string encoded_hash;
  34 
  35         unsigned int i = 0;
  36         do {
  37                 unsigned value = input[i++];
  38                 encoded_hash.append(itoa64.substr(value & 0x3f,1));
  39                 if(i < len)
  40                         value |= static_cast<int>(input[i]) << 8;
  41                 encoded_hash.append(itoa64.substr((value >> 6) & 0x3f,1));
  42                 if(i++ >= len)
  43                         break;
  44                 if(i < len)
  45                         value |= static_cast<int>(input[i]) << 16;
  46                 encoded_hash.append(itoa64.substr((value >> 12) & 0x3f,1));
  47                 if(i++ >= len)
  48                         break;
  49                 encoded_hash.append(itoa64.substr((value >> 18) & 0x3f,1));
  50         } while (i < len);
  51 
  52         return encoded_hash;
  53 }

  74 bool md5::is_valid_hash(const std::string& hash) {
  75         if(hash.size() != 34) return false;
  76         if(hash.substr(0,3) != hash_prefix) return false;
  77 
  78         const int iteration_count = get_iteration_count(hash);
  79         if(iteration_count < 7 || iteration_count > 30) return false;
  80 
  81         return true;
  82 }

  84 md5::md5(const std::string& password, const std::string& salt, int iteration_count)
  85 {
  86         iteration_count = 1 << iteration_count;
  87 
  88         hash = md5(salt + password).raw_digest();
  89         do {
  90                 hash = md5(std::string(hash.begin(), hash.end()).append(password)).raw_digest();
  91         } while(--iteration_count);
  92 }
