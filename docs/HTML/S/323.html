  86 /**
  87  * Variant for storing WML attributes.
  88  * The most efficient type is used when assigning a value. For instance,
  89  * strings "yes", "no", "true", "false" will be detected and stored as boolean.
  90  * @note The blank variant is only used when querying missing attributes.
  91  *       It is not stored in config objects.
  92  */
  93 class config_attribute_value
  94 {
  95         /// A wrapper for bool to get the correct streaming ("true"/"false").
  96         /// Most visitors can simply treat this as bool.
  97 public:
  98         class true_false
  99         {
 100                 bool value_;
 101         public:
 102                 explicit true_false(bool value = false) : value_(value) {}
 103                 operator bool() const { return value_; }
 104 
 105                 const std::string & str() const
 106                 {
 107                         return value_ ? config_attribute_value::s_true : config_attribute_value::s_false;
 108                 }
 109         };
 110         friend std::ostream& operator<<(std::ostream &os, const true_false &v) { return os << v.str(); }
 111 
 112         /// A wrapper for bool to get the correct streaming ("yes"/"no").
 113         /// Most visitors can simply treat this as bool.
 114         class yes_no
 115         {
 116                 bool value_;
 117         public:
 118                 explicit yes_no(bool value = false) : value_(value) {}
 119                 operator bool() const { return value_; }
 120 
 121                 const std::string & str() const
 122                 {
 123                         return value_ ? config_attribute_value::s_yes : config_attribute_value::s_no;
 124                 }
 125         };
 126         friend std::ostream& operator<<(std::ostream &os, const yes_no &v) { return os << v.str(); }
 127 private:

 133         // Data will be stored in a variant, allowing for the possibility of
 134         // boolean, numeric, and translatable data in addition to basic string
 135         // data. For most purposes, int is the preferred type for numeric data
 136         // as it is fast (often natural word size). While it is desirable to
 137         // use few types (to keep the overhead low), we do have use cases for
 138         // fractions (double) and huge numbers (up to the larger of LLONG_MAX
 139         // and SIZE_MAX).
 140         typedef boost::variant<boost::blank,
 141                 true_false, yes_no,
 142                 int, unsigned long long, double,
 143                 std::string, t_string
 144         > value_type;
 145         /// The stored value will always use the first type from the variant
 146         /// definition that can represent it and that can be streamed to the
 147         /// correct string representation (if applicable).
 148         /// This is enforced upon assignment.
 149         value_type value_;
