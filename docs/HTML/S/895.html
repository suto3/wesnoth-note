  65 battle_context_unit_stats::battle_context_unit_stats(const unit &u,
  66                 const map_location& u_loc, int u_attack_num, bool attacking,
  67                 const unit &opp, const map_location& opp_loc,
  68                 const_attack_ptr opp_weapon, const unit_map& units) :
  69         weapon(nullptr),
  70         attack_num(u_attack_num),
  71         is_attacker(attacking),
  72         is_poisoned(u.get_state(unit::STATE_POISONED)),
  73         is_slowed(u.get_state(unit::STATE_SLOWED)),
  74         slows(false),
  75         drains(false),
  76         petrifies(false),
  77         plagues(false),
  78         poisons(false),
  79         backstab_pos(false),
  80         swarm(false),
  81         firststrike(false),
  82         disable(false),
  83         experience(u.experience()),
  84         max_experience(u.max_experience()),
  85         level(u.level()),
  86         rounds(1),
  87         hp(0),
  88         max_hp(u.max_hitpoints()),
  89         chance_to_hit(0),
  90         damage(0),
  91         slow_damage(0),
  92         drain_percent(0),
  93         drain_constant(0),
  94         num_blows(0),
  95         swarm_min(0),
  96         swarm_max(0),
  97         plague_type()
  98 {
  99         // Get the current state of the unit.
 100         if (attack_num >= 0) {
 101                 weapon = u.attacks()[attack_num].shared_from_this();
 102         }
 103         if(u.hitpoints() < 0) {
 104                 LOG_CF << "Unit with " << u.hitpoints() << " hitpoints found, set to 0 for damage calculations\n";
 105                 hp = 0;
 106         } else if(u.hitpoints() > u.max_hitpoints()) {
 107                 // If a unit has more hp than its maximum, the engine will fail
 108                 // with an assertion failure due to accessing the prob_matrix
 109                 // out of bounds.
 110                 hp = u.max_hitpoints();
 111         } else {
 112                 hp = u.hitpoints();
 113         }
 114 
 115         // Get the weapon characteristics, if any.
 116         if (weapon) {
 117                 weapon->set_specials_context(u_loc, opp_loc, attacking, opp_weapon);
 118                 if (opp_weapon)
 119                         opp_weapon->set_specials_context(opp_loc, u_loc, !attacking, weapon);
 120                 slows = weapon->get_special_bool("slow");
 121                 drains = !opp.get_state("undrainable") && weapon->get_special_bool("drains");
 122                 petrifies = weapon->get_special_bool("petrifies");
 123                 poisons = !opp.get_state("unpoisonable") && weapon->get_special_bool("poison") && !opp.get_state(unit::STATE_POISONED);
 124                 backstab_pos = is_attacker && backstab_check(u_loc, opp_loc, units, resources::gameboard->teams());
 125                 rounds = weapon->get_specials("berserk").highest("value", 1).first;
 126                 firststrike = weapon->get_special_bool("firststrike");
 127                 {
 128                         const int distance = distance_between(u_loc, opp_loc);
 129                         const bool out_of_range = distance > weapon->max_range() || distance < weapon->min_range();
 130                         disable = weapon->get_special_bool("disable") || out_of_range;
 131                 }
 132 
 133                 // Handle plague.
 134                 unit_ability_list plague_specials = weapon->get_specials("plague");
 135                 plagues = !opp.get_state("unplagueable") && !plague_specials.empty() &&
 136                         strcmp(opp.undead_variation().c_str(), "null") && !resources::gameboard->map().is_village(opp_loc);
 137 
 138                 if (plagues) {
 139                         plague_type = (*plague_specials.front().first)["type"].str();
 140                         if (plague_type.empty())
 141                                 plague_type = u.type().base_id();
 142                 }
 143 
 144                 // Compute chance to hit.
 145                 chance_to_hit = opp.defense_modifier(
 146                         resources::gameboard->map().get_terrain(opp_loc)) + weapon->accuracy() -
 147                         (opp_weapon ? opp_weapon->parry() : 0);
 148                 if(chance_to_hit > 100) {
 149                         chance_to_hit = 100;
 150                 }
 151 
 152                 unit_ability_list cth_specials = weapon->get_specials("chance_to_hit");
 153                 unit_abilities::effect cth_effects(cth_specials, chance_to_hit, backstab_pos);
 154                 chance_to_hit = cth_effects.get_composite_value();
 155 
 156                 if (opp.get_state("invulnerable"))
 157                 {
 158                         chance_to_hit = 0;
 159                 }
 160 
 161                 // Compute base damage done with the weapon.
 162                 int base_damage = weapon->modified_damage(backstab_pos);
 163 
 164                 // Get the damage multiplier applied to the base damage of the weapon.
 165                 int damage_multiplier = 100;
 166                 // Time of day bonus.
 167                 damage_multiplier += combat_modifier(resources::gameboard->units(), resources::gameboard->map(), u_loc, u.alignment(), u.is_fearless());
 168                 // Leadership bonus.
 169                 int leader_bonus = under_leadership(units, u_loc).first;
 170                 if (leader_bonus != 0)
 171                         damage_multiplier += leader_bonus;
 172                 // Resistance modifier.
 173                 damage_multiplier *= opp.damage_from(*weapon, !attacking, opp_loc);
 174 
 175                 // Compute both the normal and slowed damage.
 176                 damage = round_damage(base_damage, damage_multiplier, 10000);
 177                 slow_damage = round_damage(base_damage, damage_multiplier, 20000);
 178                 if (is_slowed)
 179                         damage = slow_damage;
 180 
 181                 // Compute drain amounts only if draining is possible.
 182                 if(drains) {
 183                         unit_ability_list drain_specials = weapon->get_specials("drains");
 184 
 185                         // Compute the drain percent (with 50% as the base for backward compatibility)
 186                         unit_abilities::effect drain_percent_effects(drain_specials, 50, backstab_pos);
 187                         drain_percent = drain_percent_effects.get_composite_value();
 188                 }
 189 
 190                 // Add heal_on_hit (the drain constant)
 191                 unit_ability_list heal_on_hit_specials = weapon->get_specials("heal_on_hit");
 192                 unit_abilities::effect heal_on_hit_effects(heal_on_hit_specials, 0, backstab_pos);
 193                 drain_constant += heal_on_hit_effects.get_composite_value();
 194 
 195                 drains = drain_constant || drain_percent;
 196 
 197                 // Compute the number of blows and handle swarm.
 198                 weapon->modified_attacks(backstab_pos, swarm_min, swarm_max);
 199                 swarm = swarm_min != swarm_max;
 200                 num_blows = calc_blows(hp);
 201         }
 202 }

 322 battle_context::battle_context(const unit_map& units,
 323                 const map_location& attacker_loc, const map_location& defender_loc,
 324                 int attacker_weapon, int defender_weapon, double aggression,
 325                 const combatant *prev_def, const unit* attacker_ptr) :
 326         attacker_stats_(nullptr), defender_stats_(nullptr), attacker_combatant_(nullptr),
 327         defender_combatant_(nullptr)
 328 {
 329         const unit &attacker = attacker_ptr ? *attacker_ptr : *units.find(attacker_loc);
 330         const unit &defender = *units.find(defender_loc);
 331         const double harm_weight = 1.0 - aggression;
 332 
 333         if (attacker_weapon == -1 && attacker.attacks().size() == 1 && attacker.attacks()[0].attack_weight() > 0 && !attacker.attacks()[0].get_special_bool("disable", true))
 334                 attacker_weapon = 0;
 335 
 336         if (attacker_weapon == -1) {
 337                 attacker_weapon = choose_attacker_weapon(attacker, defender, units,
 338                         attacker_loc, defender_loc,
 339                                 harm_weight, &defender_weapon, prev_def);
 340         } else if (defender_weapon == -1) {
 341                 defender_weapon = choose_defender_weapon(attacker, defender, attacker_weapon,
 342                         units, attacker_loc, defender_loc, prev_def);
 343         }
 344 
 345         // If those didn't have to generate statistics, do so now.
 346         if (!attacker_stats_) {
 347                 const_attack_ptr adef = nullptr;
 348                 const_attack_ptr ddef = nullptr;
 349                 if (attacker_weapon >= 0) {
 350                         VALIDATE(attacker_weapon < static_cast<int>(attacker.attacks().size()),
 351                                         _("An invalid attacker weapon got selected."));
 352                         adef = attacker.attacks()[attacker_weapon].shared_from_this();
 353                 }
 354                 if (defender_weapon >= 0) {
 355                         VALIDATE(defender_weapon < static_cast<int>(defender.attacks().size()),
 356                                         _("An invalid defender weapon got selected."));
 357                         ddef = defender.attacks()[defender_weapon].shared_from_this();
 358                 }
 359                 assert(!defender_stats_ && !attacker_combatant_ && !defender_combatant_);
 360                 attacker_stats_ = new battle_context_unit_stats(attacker, attacker_loc, attacker_weapon,
 361                                 true, defender, defender_loc, ddef, units);
 362                 defender_stats_ = new battle_context_unit_stats(defender, defender_loc, defender_weapon, false,
 363                                 attacker, attacker_loc, adef, units);
 364         }
 365 
 366         // There have been various bugs where only one of these was set
 367         assert(attacker_stats_);
 368         assert(defender_stats_);
 369 }

 443 // Does combat A give us a better result than combat B?
 444 bool battle_context::better_combat(const combatant &us_a, const combatant &them_a,
 445                 const combatant &us_b, const combatant &them_b, double harm_weight)
 446 {
 447         double a, b;
 448 
 449         // Compare: P(we kill them) - P(they kill us).
 450         a = them_a.hp_dist[0] - us_a.hp_dist[0] * harm_weight;
 451         b = them_b.hp_dist[0] - us_b.hp_dist[0] * harm_weight;
 452         if (a - b < -0.01)
 453                 return false;
 454         if (a - b > 0.01)
 455                 return true;
 456 
 457         // Add poison to calculations
 458         double poison_a_us = (us_a.poisoned) * game_config::poison_amount;
 459         double poison_a_them = (them_a.poisoned) * game_config::poison_amount;
 460         double poison_b_us = (us_b.poisoned) * game_config::poison_amount;
 461         double poison_b_them = (them_b.poisoned) * game_config::poison_amount;
 462         // Compare: damage to them - damage to us (average_hp replaces -damage)
 463         a = (us_a.average_hp()-poison_a_us)*harm_weight - (them_a.average_hp()-poison_a_them);
 464         b = (us_b.average_hp()-poison_b_us)*harm_weight - (them_b.average_hp()-poison_b_them);
 465         if (a - b < -0.01)
 466                 return false;
 467         if (a - b > 0.01)
 468                 return true;
 469 
 470         // All else equal: go for most damage.
 471         return them_a.average_hp() < them_b.average_hp();
 472 }

 474 int battle_context::choose_attacker_weapon(const unit &attacker,
 475                 const unit &defender, const unit_map& units,
 476                 const map_location& attacker_loc, const map_location& defender_loc,
 477                 double harm_weight, int *defender_weapon, const combatant *prev_def)
 478 {
 479         std::vector<unsigned int> choices;
 480 
 481         // What options does attacker have?
 482         unsigned int i;
 483         for (i = 0; i < attacker.attacks().size(); ++i) {
 484                 const attack_type &att = attacker.attacks()[i];
 485                 if (att.attack_weight() > 0) {
 486                         choices.push_back(i);
 487                 }
 488         }
 489         if (choices.empty())
 490                 return -1;
 491         if (choices.size() == 1) {
 492                 *defender_weapon = choose_defender_weapon(attacker, defender, choices[0], units,
 493                         attacker_loc, defender_loc, prev_def);
 494                 const_attack_ptr def_weapon = *defender_weapon >= 0 ? defender.attacks()[*defender_weapon].shared_from_this() : nullptr;
 495                 attacker_stats_ = new battle_context_unit_stats(attacker, attacker_loc, choices[0],
 496                                                 true, defender, defender_loc, def_weapon, units);
 497                 if (attacker_stats_->disable) {
 498                         delete attacker_stats_;
 499                         attacker_stats_ = nullptr;
 500                         return -1;
 501                 }
 502                 const attack_type &att = attacker.attacks()[choices[0]];
 503                 defender_stats_ = new battle_context_unit_stats(defender, defender_loc, *defender_weapon, false,
 504                         attacker, attacker_loc, att.shared_from_this(), units);
 505                 return choices[0];
 506         }
 507 
 508         // Multiple options: simulate them, save best.
 509         battle_context_unit_stats *best_att_stats = nullptr, *best_def_stats = nullptr;
 510         combatant *best_att_comb = nullptr, *best_def_comb = nullptr;
 511 
 512         for (i = 0; i < choices.size(); ++i) {
 513                 const attack_type &att = attacker.attacks()[choices[i]];
 514                 int def_weapon = choose_defender_weapon(attacker, defender, choices[i], units,
 515                         attacker_loc, defender_loc, prev_def);
 516                 // If that didn't simulate, do so now.
 517                 if (!attacker_combatant_) {
 518                         const_attack_ptr def = nullptr;
 519                         if (def_weapon >= 0) {
 520                                 def = defender.attacks()[def_weapon].shared_from_this();
 521                         }
 522                         attacker_stats_ = new battle_context_unit_stats(attacker, attacker_loc, choices[i],
 523                                 true, defender, defender_loc, def, units);
 524                         if (attacker_stats_->disable) {
 525                                 delete attacker_stats_;
 526                                 attacker_stats_ = nullptr;
 527                                 continue;
 528                         }
 529                         defender_stats_ = new battle_context_unit_stats(defender, defender_loc, def_weapon, false,
 530                                 attacker, attacker_loc, att.shared_from_this(), units);
 531                         attacker_combatant_ = new combatant(*attacker_stats_);
 532                         defender_combatant_ = new combatant(*defender_stats_, prev_def);
 533                         attacker_combatant_->fight(*defender_combatant_);
 534                 } else {
 535                         if (attacker_stats_->disable) {
 536                                 delete attacker_stats_;
 537                                 attacker_stats_ = nullptr;
 538                                 continue;
 539                         }
 540                 }
 541                 if (!best_att_comb || better_combat(*attacker_combatant_, *defender_combatant_,
 542                                         *best_att_comb, *best_def_comb, harm_weight)) {
 543                         delete best_att_comb;
 544                         delete best_def_comb;
 545                         delete best_att_stats;
 546                         delete best_def_stats;
 547                         best_att_comb = attacker_combatant_;
 548                         best_def_comb = defender_combatant_;
 549                         best_att_stats = attacker_stats_;
 550                         best_def_stats = defender_stats_;
 551                 } else {
 552                         delete attacker_combatant_;
 553                         delete defender_combatant_;
 554                         delete attacker_stats_;
 555                         delete defender_stats_;
 556                 }
 557                 attacker_combatant_ = nullptr;
 558                 defender_combatant_ = nullptr;
 559                 attacker_stats_ = nullptr;
 560                 defender_stats_ = nullptr;
 561         }
 562 
 563         attacker_combatant_ = best_att_comb;
 564         defender_combatant_ = best_def_comb;
 565         attacker_stats_ = best_att_stats;
 566         defender_stats_ = best_def_stats;
 567 
 568         // These currently mean the same thing, but assumptions like that have been broken before
 569         if (!defender_stats_ || !attacker_stats_) {
 570                 return -1;
 571         }
 572         *defender_weapon = defender_stats_->attack_num;
 573         return attacker_stats_->attack_num;
 574 }

 576 /** @todo FIXME: Hand previous defender unit in here.
 577  */
 578 int battle_context::choose_defender_weapon(const unit &attacker,
 579                 const unit &defender, unsigned attacker_weapon, const unit_map& units,
 580                 const map_location& attacker_loc, const map_location& defender_loc,
 581                 const combatant *prev_def)
 582 {
 583         VALIDATE(attacker_weapon < attacker.attacks().size(),
 584                         _("An invalid attacker weapon got selected."));
 585         const attack_type &att = attacker.attacks()[attacker_weapon];
 586         std::vector<unsigned int> choices;
 587 
 588         // What options does defender have?
 589         unsigned int i;
 590         for (i = 0; i < defender.attacks().size(); ++i) {
 591                 const attack_type &def = defender.attacks()[i];
 592                 if (def.range() == att.range() && def.defense_weight() > 0) {
 593                         choices.push_back(i);
 594                 }
 595         }
 596         if (choices.empty())
 597                 return -1;
 598         if (choices.size() == 1) {
 599                 const battle_context_unit_stats def_stats(defender, defender_loc,
 600                                 choices[0], false, attacker, attacker_loc, att.shared_from_this(), units);
 601                 return (def_stats.disable) ? -1 : choices[0];
 602         }
 603 
 604         // Multiple options:
 605         // First pass : get the best weight and the minimum simple rating for this weight.
 606         // simple rating = number of blows * damage per blows (resistance taken in account) * cth * weight
 607         // Eligible attacks for defense should have a simple rating greater or equal to this weight.
 608 
 609         int min_rating = 0;
 610         {
 611                 double max_weight = 0.0;
 612 
 613                 for (i = 0; i < choices.size(); ++i) {
 614                         const attack_type &def = defender.attacks()[choices[i]];
 615                         if (def.defense_weight() >= max_weight) {
 616                                 const battle_context_unit_stats def_stats(defender, defender_loc,
 617                                                 choices[i], false, attacker, attacker_loc, att.shared_from_this(), units);
 618                                 if (def_stats.disable) continue;
 619                                 max_weight = def.defense_weight();
 620                                 int rating = static_cast<int>(def_stats.num_blows * def_stats.damage *
 621                                                 def_stats.chance_to_hit * def.defense_weight());
 622                                 if (def.defense_weight() > max_weight || rating < min_rating ) {
 623                                         min_rating = rating;
 624                                 }
 625                         }
 626                 }
 627         }
 628 
 629         // Multiple options: simulate them, save best.
 630         for (i = 0; i < choices.size(); ++i) {
 631                 const attack_type &def = defender.attacks()[choices[i]];
 632                 battle_context_unit_stats *att_stats = new battle_context_unit_stats(attacker, attacker_loc, attacker_weapon,
 633                                 true, defender, defender_loc, def.shared_from_this(), units);
 634                 battle_context_unit_stats *def_stats = new battle_context_unit_stats(defender, defender_loc, choices[i], false,
 635                                 attacker, attacker_loc, att.shared_from_this(), units);
 636                 if (def_stats->disable) {
 637                         delete att_stats;
 638                         delete def_stats;
 639                         continue;
 640                 }
 641                 combatant *att_comb = new combatant(*att_stats);
 642                 combatant *def_comb = new combatant(*def_stats, prev_def);
 643                 att_comb->fight(*def_comb);
 644 
 645                 int simple_rating = static_cast<int>(def_stats->num_blows *
 646                                 def_stats->damage * def_stats->chance_to_hit * def.defense_weight());
 647 
 648                 if (simple_rating >= min_rating &&
 649                                 ( !attacker_combatant_ || better_combat(*def_comb, *att_comb, *defender_combatant_, *attacker_combatant_, 1.0) )
 650                    ) {
 651                         delete attacker_combatant_;
 652                         delete defender_combatant_;
 653                         delete attacker_stats_;
 654                         delete defender_stats_;
 655                         attacker_combatant_ = att_comb;
 656                         defender_combatant_ = def_comb;
 657                         attacker_stats_ = att_stats;
 658                         defender_stats_ = def_stats;
 659                 } else {
 660                         delete att_comb;
 661                         delete def_comb;
 662                         delete att_stats;
 663                         delete def_stats;
 664                 }
 665         }
 666 
 667         return defender_stats_ ? defender_stats_->attack_num : -1;
 668 }

 897         bool attack::perform_hit(bool attacker_turn, statistics::attack_context &stats)
 898         {
 899                 unit_info
 900                         &attacker = *(attacker_turn ? &a_ : &d_),
 901                         &defender = *(attacker_turn ? &d_ : &a_);
 902                 const battle_context_unit_stats
 903                         *&attacker_stats = *(attacker_turn ? &a_stats_ : &d_stats_),
 904                         *&defender_stats = *(attacker_turn ? &d_stats_ : &a_stats_);
 905                 int &abs_n = *(attacker_turn ? &abs_n_attack_ : &abs_n_defend_);
 906                 bool &update_fog = *(attacker_turn ? &update_def_fog_ : &update_att_fog_);
 907 
 908                 int ran_num = randomness::generator->get_random_int(0,99);
 909                 bool hits = (ran_num < attacker.cth_);
 910 
 911                 int damage = 0;
 912                 if (hits) {
 913                         damage = attacker.damage_;
 914                         resources::gamedata->get_variable("damage_inflicted") = damage;
 915                 }
 916 
 917                 // Make sure that if we're serializing a game here,
 918                 // we got the same results as the game did originally.
 919                 const config local_results {"chance", attacker.cth_, "hits", hits, "damage", damage};
 920                 config replay_results;
 921                 bool equals_replay = checkup_instance->local_checkup(local_results, replay_results);
 922                 if (!equals_replay)
 923                 {
 924                         check_replay_attack_result(hits, ran_num, damage, replay_results, attacker);
 925                 }
 926 
 927                 // can do no more damage than the defender has hitpoints
 928                 int damage_done = std::min<int>(defender.get_unit().hitpoints(), attacker.damage_);
 929                 // expected damage = damage potential * chance to hit (as a percentage)
 930                 double expected_damage = damage_done*attacker.cth_*0.01;
 931                 if (attacker_turn) {
 932                         stats.attack_expected_damage(expected_damage, 0);
 933                 } else {
 934                         stats.attack_expected_damage(0, expected_damage);
 935                 }
 936 
 937                 int drains_damage = 0;
 938                 if (hits && attacker_stats->drains) {
 939                         drains_damage = damage_done * attacker_stats->drain_percent / 100 + attacker_stats->drain_constant;
 940                         // don't drain so much that the attacker gets more than his maximum hitpoints
 941                         drains_damage = std::min<int>(drains_damage, attacker.get_unit().max_hitpoints() - attacker.get_unit().hitpoints());
 942                         // if drain is negative, don't allow drain to kill the attacker
 943                         drains_damage = std::max<int>(drains_damage, 1 - attacker.get_unit().hitpoints());
 944                 }
 945 
 946                 if (update_display_)
 947                 {
 948                         std::ostringstream float_text;
 949                         std::vector<std::string> extra_hit_sounds;
 950                         if (hits)
 951                         {
 952                                 const unit &defender_unit = defender.get_unit();
 953                                 if (attacker_stats->poisons && !defender_unit.get_state(unit::STATE_POISONED)) {
 954                                         float_text << (defender_unit.gender() == unit_race::FEMALE ?
 955                                                 _("female^poisoned") : _("poisoned")) << '\n';
 956 
 957                                         extra_hit_sounds.push_back(game_config::sounds::status::poisoned);
 958                                 }
 959 
 960                                 if (attacker_stats->slows && !defender_unit.get_state(unit::STATE_SLOWED)) {
 961                                         float_text << (defender_unit.gender() == unit_race::FEMALE ?
 962                                                 _("female^slowed") : _("slowed")) << '\n';
 963 
 964                                         extra_hit_sounds.push_back(game_config::sounds::status::slowed);
 965                                 }
 966 
 967                                 if (attacker_stats->petrifies) {
 968                                         float_text << (defender_unit.gender() == unit_race::FEMALE ?
 969                                                 _("female^petrified") : _("petrified")) << '\n';
 970 
 971                                         extra_hit_sounds.push_back(game_config::sounds::status::petrified);
 972                                 }
 973                         }
 974 
 975                         unit_display::unit_attack(game_display::get_singleton(), *resources::gameboard,
 976                                 attacker.loc_, defender.loc_, damage,
 977                                 *attacker_stats->weapon, defender_stats->weapon,
 978                                 abs_n, float_text.str(), drains_damage, "", &extra_hit_sounds);
 979                 }
 980 
 981                 bool dies = defender.get_unit().take_hit(damage);
 982                 LOG_NG << "defender took " << damage << (dies ? " and died\n" : "\n");
 983                 if (attacker_turn) {
 984                         stats.attack_result(hits
 985                                 ? (dies ? statistics::attack_context::KILLS : statistics::attack_context::HITS)
 986                                 : statistics::attack_context::MISSES, damage_done, drains_damage);
 987                 } else {
 988                         stats.defend_result(hits
 989                                 ? (dies ? statistics::attack_context::KILLS : statistics::attack_context::HITS)
 990                                 : statistics::attack_context::MISSES, damage_done, drains_damage);
 991                 }
 992 
 993 
 994                 replay_results.clear();
 995                 // there was also a attribute cfg["unit_hit"] which was never used so i deleted.
 996                 equals_replay = checkup_instance->local_checkup(config {"dies", dies}, replay_results);
 997                 if (!equals_replay)
 998                 {
 999                         bool results_dies = replay_results["dies"].to_bool();
1000 
1001                                 errbuf_ << "SYNC: In attack " << a_.dump() << " vs " << d_.dump()
1002                                         << ": the data source says the "
1003                                         << (attacker_turn ? "defender" : "attacker") << ' '
1004                                         << (results_dies ? "perished" : "survived")
1005                                         << " while in-game calculations show it "
1006                                         << (dies ? "perished" : "survived")
1007                                         << " (over-riding game calculations with data source results)\n";
1008                                 dies = results_dies;
1009                                 // Set hitpoints to 0 so later checks don't invalidate the death.
1010                                 if (results_dies) defender.get_unit().set_hitpoints(0);
1011                                 OOS_error_ = true;
1012                 }
1013 
1014                 if (hits)
1015                 {
1016                         try {
1017                                 fire_event(attacker_turn ? "attacker_hits" : "defender_hits");
1018                         } catch (attack_end_exception) {
1019                                 refresh_bc();
1020                                 return false;
1021                         }
1022                 }
1023                 else
1024                 {
1025                         try {
1026                                 fire_event(attacker_turn ? "attacker_misses" : "defender_misses");
1027                         } catch (attack_end_exception) {
1028                                 refresh_bc();
1029                                 return false;
1030                         }
1031                 }
1032                 refresh_bc();
1033 
1034                 bool attacker_dies = false;
1035                 if (drains_damage > 0) {
1036                         attacker.get_unit().heal(drains_damage);
1037                 } else if(drains_damage < 0) {
1038                         attacker_dies = attacker.get_unit().take_hit(-drains_damage);
1039                 }
1040 
1041                 if (dies) {
1042                         unit_killed(attacker, defender, attacker_stats, defender_stats, false);
1043                         update_fog = true;
1044                 }
1045                 if (attacker_dies) {
1046                         unit_killed(defender, attacker, defender_stats, attacker_stats, true);
1047                         *(attacker_turn ? &update_att_fog_ : &update_def_fog_) = true;
1048                 }
1049 
1050                 if(dies) {
1051                         update_minimap_ = true;
1052                         return false;
1053                 }
1054 
1055                 if (hits)
1056                 {
1057                         unit &defender_unit = defender.get_unit();
1058                         if (attacker_stats->poisons && !defender_unit.get_state(unit::STATE_POISONED)) {
1059                                 defender_unit.set_state(unit::STATE_POISONED, true);
1060                                 LOG_NG << "defender poisoned\n";
1061                         }
1062 
1063                         if (attacker_stats->slows && !defender_unit.get_state(unit::STATE_SLOWED)) {
1064                                 defender_unit.set_state(unit::STATE_SLOWED, true);
1065                                 update_fog = true;
1066                                 defender.damage_ = defender_stats->slow_damage;
1067                                 LOG_NG << "defender slowed\n";
1068                         }
1069 
1070                         // If the defender is petrified, the fight stops immediately
1071                         if (attacker_stats->petrifies) {
1072                                 defender_unit.set_state(unit::STATE_PETRIFIED, true);
1073                                 update_fog = true;
1074                                 attacker.n_attacks_ = 0;
1075                                 defender.n_attacks_ = -1; // Petrified.
1076                                 resources::game_events->pump().fire("petrified", defender.loc_, attacker.loc_);
1077                                 refresh_bc();
1078                         }
1079                 }
1080 
1081                 // Delay until here so that poison and slow go through
1082                 if(attacker_dies) {
1083                         update_minimap_ = true;
1084                         return false;
1085                 }
1086 
1087                 --attacker.n_attacks_;
1088                 return true;
1089         }

1462 bool backstab_check(const map_location& attacker_loc,
1463                     const map_location& defender_loc,
1464                     const unit_map& units, const std::vector<team>& teams)
1465 {
1466         const unit_map::const_iterator defender = units.find(defender_loc);
1467         if(defender == units.end()) return false; // No defender
1468 
1469         map_location adj[6];
1470         get_adjacent_tiles(defender_loc, adj);
1471         int i;
1472         for(i = 0; i != 6; ++i) {
1473                 if(adj[i] == attacker_loc)
1474                         break;
1475         }
1476         if(i >= 6) return false;  // Attack not from adjacent location
1477 
1478         const unit_map::const_iterator opp =
1479                 units.find(adj[(i+3)%6]);
1480         if(opp == units.end()) return false; // No opposite unit
1481         if (opp->incapacitated()) return false;
1482         if (size_t(defender->side() - 1) >= teams.size() || size_t(opp->side() - 1) >= teams.size())
1483                 return true; // If sides aren't valid teams, then they are enemies
1484         if (teams[defender->side() - 1].is_enemy(opp->side()))
1485                 return true; // Defender and opposite are enemies
1486         return false; // Defender and opposite are friends
1487 }
