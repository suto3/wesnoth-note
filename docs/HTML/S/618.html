  35 tree_view_node::tree_view_node(
  36                 const std::string& id,
  37                 tree_view_node* parent_node,
  38                 tree_view& parent_tree_view,
  39                 const std::map<std::string /* widget id */, string_map>& data)
  40         : widget()
  41         , parent_node_(parent_node)
  42         , tree_view_(&parent_tree_view)
  43         , grid_()
  44         , children_()
  45         , toggle_(nullptr)
  46         , label_(nullptr)
  47         , unfolded_(false)
  48         , callback_state_change_()
  49         , callback_state_to_folded_()
  50         , callback_state_to_unfolded_()
  51 {
  52         grid_.set_parent(this);
  53         set_parent(&parent_tree_view);
  54 
  55         if(id == "root") {
  56                 unfolded_ = true;
  57                 return;
  58         }
  59 
  60         for(const auto& node_definition : get_tree_view().get_node_definitions()) {
  61                 if(node_definition.id != id) {
  62                         continue;
  63                 }
  64 
  65                 node_definition.builder->build(&grid_);
  66                 init_grid(&grid_, data);
  67 
  68                 if(parent_node_ && parent_node_->toggle_) {
  69                         dynamic_cast<widget&>(*parent_node_->toggle_).set_visible(widget::visibility::visible);
  70                 }
  71 
  72                 if(node_definition.unfolded) {
  73                         unfolded_ = true;
  74                 }
  75 
  76                 widget* toggle_widget = grid_.find("tree_view_node_toggle", false);
  77                 toggle_ = dynamic_cast<selectable_item*>(toggle_widget);
  78 
  79                 if(toggle_) {
  80                         toggle_widget->set_visible(widget::visibility::hidden);
  81 
  82                         toggle_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  83                                         &tree_view_node::signal_handler_left_button_click, this, _2));
  84                         toggle_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  85                                         &tree_view_node::signal_handler_left_button_click, this, _2),
  86                                         event::dispatcher::back_post_child);
  87 
  88                         if(unfolded_) {
  89                                 toggle_->set_value(1);
  90                         }
  91                 }
  92 
  93                 widget* label_widget = grid_.find("tree_view_node_label", false);
  94                 label_ = dynamic_cast<selectable_item*>(label_widget);
  95 
  96                 if(label_) {
  97                         label_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
  98                                         &tree_view_node::signal_handler_label_left_button_click, this, _2, _3, _4),
  99                                         event::dispatcher::front_child);
 100                         label_widget->connect_signal<event::LEFT_BUTTON_CLICK>(std::bind(
 101                                         &tree_view_node::signal_handler_label_left_button_click, this, _2, _3, _4),
 102                                         event::dispatcher::front_pre_child);
 103 
 104                         if(!get_tree_view().selected_item_) {
 105                                 get_tree_view().selected_item_ = this;
 106                                 label_->set_value(true);
 107                         }
 108                 }
 109 
 110                 return;
 111         }
 112 
 113         // Only reached if no matching node definiton was found in the above loop
 114         FAIL_WITH_DEV_MESSAGE(_("Unknown builder id for tree view node."), id);
 115 }

 132 tree_view_node& tree_view_node::add_child(
 133                 const std::string& id,
 134                 const std::map<std::string /* widget id */, string_map>& data,
 135                 const int index)
 136 {
 137         boost::ptr_vector<tree_view_node>::iterator itor = children_.end();
 138 
 139         if(static_cast<size_t>(index) < children_.size()) {
 140                 itor = children_.begin() + index;
 141         }
 142 
 143         itor = children_.insert(itor, new tree_view_node(id, this, get_tree_view(), data));
 144 
 145         if(is_folded() /*|| is_root_node()*/) {
 146                 return *itor;
 147         }
 148 
 149         if(get_tree_view().get_size() == point()) {
 150                 return *itor;
 151         }
 152 
 153         assert(get_tree_view().content_grid());
 154         const point current_size = get_tree_view().content_grid()->get_size();
 155 
 156         // Calculate width modification.
 157         // This increases tree width if the width of the new node is greater than the current width.
 158         point best_size = itor->get_best_size();
 159         best_size.x += get_indentation_level() * get_tree_view().indentation_step_size_;
 160 
 161         const int width_modification = best_size.x > current_size.x
 162                 ? best_size.x - current_size.x
 163                 : 0;
 164 
 165         // Calculate height modification.
 166         // For this, we only increase height if the best size of the tree (that is, the size with the new node)
 167         // is larger than its current size. This prevents the scrollbar being reserved even when there's obviously
 168         // enough visual space.
 169 
 170         // Throw away cached best size to force a recomputation.
 171         get_tree_view().layout_initialize(false);
 172 
 173         const point tree_best_size = get_tree_view().get_best_size();
 174 
 175         const int height_modification = tree_best_size.y > current_size.y && get_tree_view().layout_size() == point()
 176                 ? tree_best_size.y - current_size.y
 177                 : 0;
 178 
 179         assert(height_modification >= 0);
 180 
 181         // Request new size.
 182         get_tree_view().resize_content(width_modification, height_modification, -1, itor->calculate_ypos());
 183 
 184         return *itor;
 185 }

 249 void tree_view_node::fold_internal()
 250 {
 251         const point current_size(get_current_size().x, get_unfolded_size().y);
 252         const point new_size = get_folded_size();
 253 
 254         const int width_modification = std::max(0, new_size.x - current_size.x);
 255         const int height_modification = new_size.y - current_size.y;
 256         assert(height_modification <= 0);
 257 
 258         get_tree_view().resize_content(width_modification, height_modification, -1, calculate_ypos());
 259         unfolded_ = false;
 260 
 261         if(callback_state_to_folded_) {
 262                 callback_state_to_folded_(*this);
 263         }
 264 }

 283 void tree_view_node::clear()
 284 {
 285         /** @todo Also try to find the optimal width. */
 286         int height_reduction = 0;
 287 
 288         if(!is_folded()) {
 289                 for(const auto & node : children_)
 290                 {
 291                         height_reduction += node.get_current_size().y;
 292                 }
 293         }
 294 
 295         children_.clear();
 296 
 297         if(height_reduction == 0) {
 298                 return;
 299         }
 300 
 301         get_tree_view().resize_content(0, -height_reduction,  -1, calculate_ypos());
 302 }

 304 struct tree_view_node_implementation

 306 private:
 307         template <class W, class It>
 308         static W* find_at_aux(It begin,
 309                                                   It end,
 310                                                   const point& coordinate,
 311                                                   const bool must_be_active)
 312         {
 313                 for(It it = begin; it != end; ++it) {
 314                         if(W* widget = it->find_at(coordinate, must_be_active)) {
 315                                 return widget;
 316                         }
 317                 }
 318                 return nullptr;
 319         }
 320 
 321 public:
 322         template <class W>
 323         static W* find_at(utils::const_clone_ref<tree_view_node, W> tree_view_node,
 324                                           const point& coordinate,
 325                                           const bool must_be_active)
 326         {
 327                 if(W* widget = tree_view_node.grid_.find_at(coordinate, must_be_active)) {
 328                         return widget;
 329                 }
 330 
 331                 if(tree_view_node.is_folded()) {
 332                         return nullptr;
 333                 }
 334 
 335                 return find_at_aux<W>(tree_view_node.children_.begin(),
 336                                                           tree_view_node.children_.end(),
 337                                                           coordinate,
 338                                                           must_be_active);
 339         }
 340 };

 421 point tree_view_node::get_current_size(bool assume_visible) const
 422 {
 423         if(!assume_visible && parent_node_ && parent_node_->is_folded()) {
 424                 return point();
 425         }
 426 
 427         point size = get_folded_size();
 428         if(is_folded()) {
 429                 return size;
 430         }
 431 
 432         for(const auto& node : children_) {
 433                 if(node.grid_.get_visible() == widget::visibility::invisible) {
 434                         continue;
 435                 }
 436 
 437                 point node_size = node.get_current_size();
 438 
 439                 size.y += node_size.y;
 440                 size.x = std::max(size.x, node_size.x);
 441         }
 442 
 443         return size;
 444 }

 476 point tree_view_node::calculate_best_size(const int indentation_level,
 477                                                                                         const unsigned indentation_step_size) const
 478 {
 479         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 480 
 481         point best_size = grid_.get_best_size();
 482         if(indentation_level > 0) {
 483                 best_size.x += indentation_level * indentation_step_size;
 484         }
 485 
 486         DBG_GUI_L << LOG_HEADER << " own grid best size " << best_size << ".\n";
 487 
 488         for(const auto& node : children_) {
 489                 if(node.grid_.get_visible() == widget::visibility::invisible) {
 490                         continue;
 491                 }
 492 
 493                 const point node_size = node.calculate_best_size(indentation_level + 1, indentation_step_size);
 494 
 495                 if(!is_folded()) {
 496                         best_size.y += node_size.y;
 497                 }
 498 
 499                 best_size.x = std::max(best_size.x, node_size.x);
 500         }
 501 
 502         DBG_GUI_L << LOG_HEADER << " result " << best_size << ".\n";
 503         return best_size;
 504 }

 523 unsigned tree_view_node::place(const unsigned indentation_step_size,
 524                                                                 point origin,
 525                                                                 unsigned width)
 526 {
 527         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 528         DBG_GUI_L << LOG_HEADER << " origin " << origin << ".\n";
 529 
 530         const unsigned offset = origin.y;
 531         point best_size = grid_.get_best_size();
 532         best_size.x = width;
 533 
 534         grid_.place(origin, best_size);
 535 
 536         if(!is_root_node()) {
 537                 origin.x += indentation_step_size;
 538                 assert(width >= indentation_step_size);
 539                 width -= indentation_step_size;
 540         }
 541 
 542         origin.y += best_size.y;
 543 
 544         if(is_folded()) {
 545                 DBG_GUI_L << LOG_HEADER << " folded node done.\n";
 546                 return origin.y - offset;
 547         }
 548 
 549         DBG_GUI_L << LOG_HEADER << " set children.\n";
 550         for(auto & node : children_) {
 551                 origin.y += node.place(indentation_step_size, origin, width);
 552         }
 553 
 554         // Inherited.
 555         widget::set_size(point(width, origin.y - offset));
 556 
 557         DBG_GUI_L << LOG_HEADER << " result " << (origin.y - offset) << ".\n";
 558         return origin.y - offset;
 559 }

 617 void tree_view_node::signal_handler_label_left_button_click(
 618                 const event::ui_event event, bool& handled, bool& halt)
 619 {
 620         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 621 
 622         assert(label_);
 623 
 624         // Normally, this is only an event hook and not full handling; however, if
 625         // the currently selected item was selected, we halt the event to prevent
 626         // deselection (which would leave no items selected).
 627         if(label_->get_value()) {
 628                 halt = handled = true;
 629                 return;
 630         }
 631 
 632         // Select the new item if a different one was selected
 633         if(get_tree_view().selected_item_ && get_tree_view().selected_item_->label_) {
 634                 get_tree_view().selected_item_->label_->set_value(false);
 635         }
 636 
 637         get_tree_view().selected_item_ = this;
 638 
 639         if(get_tree_view().selection_change_callback_) {
 640                 get_tree_view().selection_change_callback_(get_tree_view());
 641         }

 644 void tree_view_node::init_grid(

 646                 const std::map<std::string /* widget id */, string_map>& data)
 647 {
 648         assert(g);
 649 
 650         for(unsigned row = 0; row < g->get_rows(); ++row) {
 651                 for(unsigned col = 0; col < g->get_cols(); ++col) {
 652                         widget* wgt = g->get_widget(row, col);
 653                         assert(wgt);
 654 
 655                         // toggle_button* btn = dynamic_cast<toggle_button*>(widget);
 656 
 657                         if(toggle_panel* panel = dynamic_cast<toggle_panel*>(wgt)) {
 658                                 panel->set_child_members(data);
 659                         } else if(grid* child_grid = dynamic_cast<grid*>(wgt)) {
 660                                 init_grid(child_grid, data);
 661                         } else if(styled_widget* control = dynamic_cast<styled_widget*>(wgt)) {
 662                                 auto itor = data.find(control->id());
 663 
 664                                 if(itor == data.end()) {
 665                                         itor = data.find("");
 666                                 }
 667 
 668                                 if(itor != data.end()) {
 669                                         control->set_members(itor->second);
 670                                 }
 671                                 // control->set_members(data);
 672                         } else {
 673                                 // ERROR_LOG("Widget type '" << typeid(*widget).name() << "'.");
 674                         }
 675                 }
 676         }

 691 std::vector<int> tree_view_node::describe_path()
 692 {
 693         if(is_root_node()) {
 694                 return std::vector<int>();
 695         }
 696 
 697         std::vector<int> res = parent_node_->describe_path();
 698         for(size_t i = 0; i < parent_node_->count_children(); ++i) {
 699                 if(&parent_node_->children_[i] == this) {
 700                         res.push_back(i);
 701                         return res;
 702                 }
 703         }
 704 
 705         assert(!"tree_view_node was not found in parent nodes children");
 706         throw "assertion ignored"; //To silence 'no return value in this codepath' warning.
 707 }

 709 int tree_view_node::calculate_ypos()
 710 {
 711         if(!parent_node_) {
 712                 return 0;
 713         }
 714 
 715         int res = parent_node_->calculate_ypos();
 716         for(const auto& node : parent_node_->children_) {
 717                 if(&node == this) {
 718                         break;
 719                 }
 720 
 721                 res += node.get_current_size(true).y;
 722         }
 723 
 724         return res;

 737 tree_view_node* tree_view_node::get_node_above()
 738 {
 739         assert(!is_root_node());
 740 
 741         tree_view_node* cur = nullptr;
 742         for(size_t i = 0; i < parent_node_->count_children(); ++i) {
 743                 if(&parent_node_->children_[i] == this) {
 744                         if(i == 0) {
 745                                 return parent_node_->is_root_node() ? nullptr : parent_node_;
 746                         } else {
 747                                 cur = &parent_node_->children_[i - 1];
 748                                 break;
 749                         }
 750                 }
 751         }
 752 
 753         while(!cur->is_folded() && cur->count_children() > 0) {
 754                 cur = &cur->get_child_at(cur->count_children() - 1);
 755         }
 756 
 757         return cur;
 758 }

 760 tree_view_node* tree_view_node::get_node_below()
 761 {
 762         assert(!is_root_node());
 763         if(!is_folded() && count_children() > 0) {
 764                 return &get_child_at(0);
 765         }
 766 
 767         tree_view_node* cur = this;
 768         while(cur->parent_node_ != nullptr) {
 769                 tree_view_node& parent = *cur->parent_node_;
 770 
 771                 for(size_t i = 0; i < parent.count_children(); ++i) {
 772                         if(&parent.children_[i] == cur) {
 773                                 if(i < parent.count_children() - 1) {
 774                                         return &parent.children_[i + 1];
 775                                 } else {
 776                                         cur = &parent;
 777                                 }
 778                                 break;
 779                         }
 780                 }
 781         }
 782 
 783         return nullptr;
 784 }

 804 void tree_view_node::select_node(bool expand_parents)
 805 {
 806         if(!label_ || label_->get_value_bool()) {
 807                 return;
 808         }
 809 
 810         if(expand_parents) {
 811                 tree_view_node* root = &get_tree_view().get_root_node();
 812                 for(tree_view_node* cur = parent_node_; cur != root; cur = cur->parent_node_) {
 813                         cur->unfold();
 814                 }
 815         }
 816 
 817         if(get_tree_view().selected_item_ && get_tree_view().selected_item_->label_) {
 818                 get_tree_view().selected_item_->label_->set_value(false);
 819         }
 820         get_tree_view().selected_item_ = this;
 821 
 822         if(get_tree_view().selection_change_callback_) {
 823                 get_tree_view().selection_change_callback_(get_tree_view());
 824         }
 825         label_->set_value_bool(true);
 826 }
