 144 double recruitment::evaluate() {
 145         // Check if the recruitment list has changed.
 146         // Then cheapest_unit_costs_ is not valid anymore.
 147         if (recruit_situation_change_observer_.recruit_list_changed()) {
 148                 cheapest_unit_costs_.clear();
 149                 recruit_situation_change_observer_.set_recruit_list_changed(false);
 150         }
 151 
 152         // When evaluate() is called the first time this turn,
 153         // we'll retrieve the recruitment-instruction aspect.
 154         if (resources::tod_manager->turn() != recruitment_instructions_turn_) {
 155                 recruitment_instructions_ = get_recruitment_instructions();
 156                 integrate_recruitment_pattern_in_recruitment_instructions();
 157                 recruitment_instructions_turn_ = resources::tod_manager->turn();
 158                 LOG_AI_RECRUITMENT << "Recruitment-instructions updated:\n";
 159                 LOG_AI_RECRUITMENT << recruitment_instructions_ << "\n";
 160         }
 161 
 162         // Check if we have something to do.
 163         const config* job = get_most_important_job();
 164         if (!job) {
 165                 return BAD_SCORE;
 166         }
 167 
 168         const unit_map& units = resources::gameboard->units();
 169         const std::vector<unit_map::const_iterator> leaders = units.find_leaders(get_side());
 170 
 171         for (const unit_map::const_iterator& leader : leaders) {
 172                 if (leader == resources::gameboard->units().end()) {
 173                         return BAD_SCORE;
 174                 }
 175                 // Check Gold. But proceed if there is a unit with cost <= 0 (WML can do that)
 176                 int cheapest_unit_cost = get_cheapest_unit_cost_for_leader(leader);
 177                 if (current_team().gold() < cheapest_unit_cost && cheapest_unit_cost > 0) {
 178                         continue;
 179                 }
 180 
 181                 const map_location& loc = leader->get_location();
 182                 if (resources::gameboard->map().is_keep(loc) &&
 183                                 pathfind::find_vacant_castle(*leader) != map_location::null_location()) {
 184                         return get_score();
 185                 }
 186         }
 187 
 188         return BAD_SCORE;
 189 }

 191 void recruitment::execute() {
 192         LOG_AI_RECRUITMENT << "\n\n\n------------AI RECRUITMENT BEGIN---------------\n\n";
 193         LOG_AI_RECRUITMENT << "TURN: " << resources::tod_manager->turn() <<
 194                         " SIDE: " << current_team().side() << "\n";
 195 
 196         /*
 197          * Check which leaders can recruit and collect them in leader_data.
 198          */
 199 
 200         const unit_map& units = resources::gameboard->units();
 201         const gamemap& map = resources::gameboard->map();
 202         const std::vector<unit_map::const_iterator> leaders = units.find_leaders(get_side());
 203 
 204         // This is the central datastructure with all score_tables in it.
 205         std::vector<data> leader_data;
 206 
 207         std::set<std::string> global_recruits;
 208 
 209         for (const unit_map::const_iterator& leader : leaders) {
 210                 const map_location& keep = leader->get_location();
 211                 if (!resources::gameboard->map().is_keep(keep)) {
 212                         LOG_AI_RECRUITMENT << "Leader " << leader->name() << " is not on keep. \n";
 213                         continue;
 214                 }
 215                 if (pathfind::find_vacant_castle(*leader) == map_location::null_location()) {
 216                         LOG_AI_RECRUITMENT << "Leader " << leader->name() << " has no free hexes \n";
 217                         continue;
 218                 }
 219                 int cheapest_unit_cost = get_cheapest_unit_cost_for_leader(leader);
 220                 if (current_team().gold() < cheapest_unit_cost && cheapest_unit_cost > 0) {
 221                         LOG_AI_RECRUITMENT << "Leader " << leader->name() << " recruits are too expensive. \n";
 222                         continue;
 223                 }
 224 
 225                 // Leader can recruit.
 226 
 227                 data data(leader);
 228 
 229                 // Add team recruits.
 230                 for (const std::string& recruit : current_team().recruits()) {
 231                         if (!unit_types.find(recruit)) {
 232                                 lg::wml_error() << "Unit-type \"" << recruit << "\" doesn't exist.\n";
 233                         }
 234                         data.recruits.insert(recruit);
 235                         data.scores[recruit] = 0.0;
 236                         global_recruits.insert(recruit);
 237                 }
 238 
 239                 // Add extra recruits.
 240                 for (const std::string& recruit : leader->recruits()) {
 241                         if (!unit_types.find(recruit)) {
 242                                 lg::wml_error() << "Unit-type \"" << recruit << "\" doesn't exist.\n";
 243                         }
 244                         data.recruits.insert(recruit);
 245                         data.scores[recruit] = 0.0;
 246                         global_recruits.insert(recruit);
 247                 }
 248 
 249                 // Add recalls.
 250                 // Recalls are treated as recruits. While recruiting
 251                 // we'll check if we can do a recall instead of a recruitment.
 252                 for (const unit_const_ptr & recall : current_team().recall_list()) {
 253                         // Check if this leader is allowed to recall this unit.
 254                         const unit_filter ufilt( vconfig(leader->recall_filter()), resources::filter_con);
 255                         if (!ufilt(*recall, map_location::null_location())) {
 256                                 continue;
 257                         }
 258                         data.recruits.insert(recall->type_id());
 259                         data.scores[recall->type_id()] = 0.0;
 260                         global_recruits.insert(recall->type_id());
 261                 }
 262 
 263                 // Check if leader is in danger. (If a enemies unit can attack the leader)
 264                 data.in_danger = power_projection(leader->get_location(), get_enemy_dstsrc()) > 0;
 265 
 266                 // If yes, set ratio_score very high, so this leader will get priority while recruiting.
 267                 if (data.in_danger) {
 268                         data.ratio_score = 50;
 269                         state_ = LEADER_IN_DANGER;
 270                         LOG_AI_RECRUITMENT << "Leader " << leader->name() << " is in danger.\n";
 271                 }
 272 
 273                 leader_data.push_back(data);
 274         }
 275 
 276         if (leader_data.empty()) {
 277                 LOG_AI_RECRUITMENT << "No leader available for recruiting. \n";
 278                 return;  // This CA is going to be blacklisted for this turn.
 279         }
 280 
 281         if (global_recruits.empty()) {
 282                 LOG_AI_RECRUITMENT << "All leaders have empty recruitment lists. \n";
 283                 return;  // This CA is going to be blacklisted for this turn.
 284         }
 285 
 286         /**
 287          * Find important hexes and calculate other static things.
 288          */
 289 
 290         update_important_hexes();
 291         // Show "x" on important hexes if debug mode is activated AND
 292         // the log domain "ai/recruitment" is used.
 293         if (game_config::debug && !lg::info().dont_log(log_ai_recruitment)) {
 294                 show_important_hexes();
 295         }
 296 
 297         for (const map_location& hex : important_hexes_) {
 298                 ++important_terrain_[map[hex]];
 299         }
 300 
 301         update_average_lawful_bonus();
 302         update_own_units_count();
 303         update_scouts_wanted();
 304 
 305         /**
 306          * Fill scores.
 307          */
 308 
 309         do_combat_analysis(&leader_data);
 310 
 311         LOG_AI_RECRUITMENT << "Scores before extra treatments:\n";
 312         for (const data& data : leader_data) {
 313                 LOG_AI_RECRUITMENT << "\n" << data.to_string();
 314         }
 315 
 316         do_similarity_penalty(&leader_data);
 317         do_randomness(&leader_data);
 318         handle_recruitment_more(&leader_data);
 319 
 320         LOG_AI_RECRUITMENT << "Scores after extra treatments:\n";
 321         for (const data& data : leader_data) {
 322                 LOG_AI_RECRUITMENT << "\n" << data.to_string();
 323         }
 324 
 325         /**
 326          * Do recruitment according to [recruit]-tags and scores.
 327          * Note that the scores don't indicate the preferred mix to recruit but rather
 328          * the preferred mix of all units. So already existing units are considered.
 329          */
 330 
 331         action_result_ptr action_result;
 332         config* job = nullptr;
 333         do {  // Recruitment loop
 334                 recruit_situation_change_observer_.reset_gamestate_changed();
 335 
 336                 // Check if we may want to save gold by not recruiting.
 337                 update_state();
 338                 int save_gold_turn = get_recruitment_save_gold()["active"].to_int(2);  // From aspect.
 339                 int current_turn = resources::tod_manager->turn();
 340                 bool save_gold_active = save_gold_turn > 0 && save_gold_turn <= current_turn;
 341                 if (state_ == SAVE_GOLD && save_gold_active) {
 342                         break;
 343                 }
 344 
 345                 job = get_most_important_job();
 346                 if (!job) {
 347                         LOG_AI_RECRUITMENT << "All recruitment jobs (recruitment_instructions) done.\n";
 348                         break;
 349                 }
 350                 LOG_AI_RECRUITMENT << "Executing this job:\n" << *job << "\n";
 351 
 352                 data* best_leader_data = get_best_leader_from_ratio_scores(leader_data, job);
 353                 if (!best_leader_data) {
 354                         LOG_AI_RECRUITMENT << "Leader with job (recruitment_instruction) is not on keep.\n";
 355                         if (remove_job_if_no_blocker(job)) {
 356                                 continue;
 357                         } else {
 358                                 break;
 359                         }
 360                 }
 361                 LOG_AI_RECRUITMENT << "We want to have " << scouts_wanted_ << " more scouts.\n";
 362 
 363                 const std::string best_recruit = get_best_recruit_from_scores(*best_leader_data, job);
 364                 if (best_recruit.empty()) {
 365                         LOG_AI_RECRUITMENT << "Cannot fulfill recruitment-instruction.\n";
 366                         if (remove_job_if_no_blocker(job)) {
 367                                 continue;
 368                         } else {
 369                                 break;
 370                         }
 371                 }
 372 
 373                 LOG_AI_RECRUITMENT << "Best recruit is: " << best_recruit << "\n";
 374                 const std::string* recall_id = get_appropriate_recall(best_recruit, *best_leader_data);
 375                 if (recall_id) {
 376                         LOG_AI_RECRUITMENT << "Found appropriate recall with id: " << *recall_id << "\n";
 377                         action_result = execute_recall(*recall_id, *best_leader_data);
 378                 } else {
 379                         action_result = execute_recruit(best_recruit, *best_leader_data);
 380                 }
 381 
 382                 if (action_result->is_ok()) {
 383                         ++own_units_count_[best_recruit];
 384                         ++total_own_units_;
 385                         if (recruit_matches_type(best_recruit, "scout")) {
 386                                 --scouts_wanted_;
 387                         }
 388 
 389                         // Update the current job.
 390                         if (!job->operator[]("total").to_bool(false)) {
 391                                 job->operator[]("number") = job->operator[]("number").to_int(99999) - 1;
 392                         }
 393 
 394                         // Check if something changed in the recruitment list (WML can do that).
 395                         // If yes, just return/break. evaluate() and execute() will be called again.
 396                         if (recruit_situation_change_observer_.recruit_list_changed()) {
 397                                 break;
 398                         }
 399                         // Check if the gamestate changed more than once.
 400                         // (Recruitment will trigger one gamestate change, WML could trigger more changes.)
 401                         // If yes, just return/break. evaluate() and execute() will be called again.
 402                         if (recruit_situation_change_observer_.gamestate_changed() > 1) {
 403                                 break;
 404                         }
 405 
 406                 } else {
 407                         LOG_AI_RECRUITMENT << "Recruit result not ok.\n";
 408                         // We'll end up here if
 409                         // 1. We haven't enough gold,
 410                         // 2. There aren't any free hexes around leaders,
 411                         // 3. This leader can not recruit this type (this can happen after a recall)
 412                 }
 413         } while((action_result && action_result->is_ok()) || !action_result);
 414         // A action_result may be uninitialized if a job was removed. Continue then anyway.
 415 
 416         // Recruiting is done now.
 417         // Update state_ for next execution().
 418 
 419         if (state_ == LEADER_IN_DANGER) {
 420                 state_ = NORMAL;
 421         }
 422 
 423         int status = (action_result) ? action_result->get_status() : -1;
 424         bool no_gold = (status == recruit_result::E_NO_GOLD || status == recall_result::E_NO_GOLD);
 425         if (state_ == SPEND_ALL_GOLD && no_gold) {
 426                 state_ = SAVE_GOLD;
 427         }
 428         if (job && no_gold) {
 429                 remove_job_if_no_blocker(job);
 430         }
 431 }

 463 /**
 464  * A helper function for execute().
 465  * Checks if this unit type can be recalled.
 466  * If yes, we calculate a estimated value in gold of the recall unit.
 467  * If this value is less then the recall cost, we dismiss the unit.
 468  * The unit with the highest value will be returned.
 469  */
 470 const std::string* recruitment::get_appropriate_recall(const std::string& type,
 471                 const data& leader_data) const {
 472         const std::string* best_recall_id = nullptr;
 473         double best_recall_value = -1;
 474         for (const unit_const_ptr & recall_unit : current_team().recall_list()) {
 475                 if (type != recall_unit->type_id()) {
 476                         continue;
 477                 }
 478                 // Check if this leader is allowed to recall this unit.
 479                 const unit_filter ufilt(vconfig(leader_data.leader->recall_filter()), resources::filter_con);
 480                 if (!ufilt(*recall_unit, map_location::null_location())) {
 481                         LOG_AI_RECRUITMENT << "Refused recall because of filter: " << recall_unit->id() << "\n";
 482                         continue;
 483                 }
 484                 double average_cost_of_advanced_unit = 0;
 485                 int counter = 0;
 486                 for (const std::string& advancement : recall_unit->advances_to()) {
 487                         const unit_type* advancement_type = unit_types.find(advancement);
 488                         if (!advancement_type) {
 489                                 continue;
 490                         }
 491                         average_cost_of_advanced_unit += advancement_type->cost();
 492                         ++counter;
 493                 }
 494                 if (counter > 0) {
 495                         average_cost_of_advanced_unit /= counter;
 496                 } else {
 497                         // Unit don't have advancements. Use cost of unit itself.
 498                         average_cost_of_advanced_unit = recall_unit->cost();
 499                 }
 500                 double xp_quantity = static_cast<double>(recall_unit->experience()) /
 501                                 recall_unit->max_experience();
 502                 double recall_value = recall_unit->cost() + xp_quantity * average_cost_of_advanced_unit;
 503                 if (recall_value < current_team().recall_cost()) {
 504                         continue;  // Unit is not worth to get recalled.
 505                 }
 506                 if (recall_value > best_recall_value) {
 507                         best_recall_id = &recall_unit->id();
 508                         best_recall_value = recall_value;
 509                 }
 510         }
 511         return best_recall_id;
 512 }

 514 /**
 515  * A helper function for execute().
 516  * Decides according to the leaders ratio scores which leader should recruit.
 517  */
 518 data* recruitment::get_best_leader_from_ratio_scores(std::vector<data>& leader_data,
 519                 const config* job) const {
 520         assert(job);
 521         // Find things for normalization.
 522         int total_recruit_count = 0;
 523         double ratio_score_sum = 0.0;
 524         for (const data& data : leader_data) {
 525                 ratio_score_sum += data.ratio_score;
 526                 total_recruit_count += data.recruit_count;
 527         }
 528         assert(ratio_score_sum > 0.0);
 529 
 530         // Shuffle leader_data to break ties randomly.
 531         std::shuffle(leader_data.begin(), leader_data.end(), randomness::rng::default_instance());
 532 
 533         // Find which leader should recruit according to ratio_scores.
 534         data* best_leader_data = nullptr;
 535         double biggest_difference = -99999.;
 536         for (data& data : leader_data) {
 537                 if (!leader_matches_job(data, job)) {
 538                         continue;
 539                 }
 540                 double desired_ammount = data.ratio_score / ratio_score_sum * (total_recruit_count + 1);
 541                 double current_ammount = data.recruit_count;
 542                 double difference = desired_ammount - current_ammount;
 543                 if (difference > biggest_difference) {
 544                         biggest_difference = difference;
 545                         best_leader_data = &data;
 546                 }
 547         }
 548         return best_leader_data;
 549 }

 551 /**
 552  * A helper function for execute().
 553  * Counts own units and then decides what unit should be recruited so that the
 554  * unit distribution approaches the given scores.
 555  */
 556 const std::string recruitment::get_best_recruit_from_scores(const data& leader_data,
 557                 const config* job) {
 558         assert(job);
 559         std::string pattern_type = get_random_pattern_type_if_exists(leader_data, job);
 560         if (!pattern_type.empty()) {
 561                 LOG_AI_RECRUITMENT << "Randomly chosen pattern_type: " << pattern_type << "\n";
 562         }
 563         std::string best_recruit = "";
 564         double biggest_difference = -99999.;
 565         for (const score_map::value_type& i : leader_data.get_normalized_scores()) {
 566                 const std::string& unit = i.first;
 567                 const double score = i.second;
 568 
 569                 if (!limit_ok(unit)) {
 570                         continue;
 571                 }
 572                 if (!pattern_type.empty()) {
 573                         if (!recruit_matches_type(unit, pattern_type)) {
 574                                 continue;
 575                         }
 576                 } else {
 577                         if (!recruit_matches_job(unit, job)) {
 578                                 continue;
 579                         }
 580                 }
 581 
 582                 double desired_ammount = score * (total_own_units_ + 1);
 583                 double current_ammount = own_units_count_[unit];
 584                 double difference = desired_ammount - current_ammount;
 585                 if (scouts_wanted_ > 0 && recruit_matches_type(unit, "scout")) {
 586                         difference += 1000.;
 587                 }
 588                 if (difference > biggest_difference) {
 589                         biggest_difference = difference;
 590                         best_recruit = unit;
 591                 }
 592         }
 593         return best_recruit;
 594 }

 596 /**
 597  * For Map Analysis
 598  * Computes from our cost map and the combined cost map of all enemies the important hexes.
 599  */
 600 void recruitment::compare_cost_maps_and_update_important_hexes(
 601                 const pathfind::full_cost_map& my_cost_map,
 602                 const pathfind::full_cost_map& enemy_cost_map) {
 603 
 604         const gamemap& map = resources::gameboard->map();
 605 
 606         // First collect all hexes where the average costs are similar in important_hexes_candidates
 607         // Then chose only those hexes where the average costs are relatively low.
 608         // This is done to remove hexes to where the teams need a similar amount of moves but
 609         // which are relatively far away comparing to other important hexes.
 610         typedef std::map<map_location, double> border_cost_map;
 611         border_cost_map important_hexes_candidates;
 612         double smallest_border_movecost = 999999;
 613         double biggest_border_movecost = 0;
 614         for(int x = 0; x < map.w(); ++x) {
 615                 for (int y = 0; y < map.h(); ++y) {
 616                         double my_cost_average = my_cost_map.get_average_cost_at(x, y);
 617                         double enemy_cost_average = enemy_cost_map.get_average_cost_at(x, y);
 618                         if (my_cost_average == -1 || enemy_cost_average == -1) {
 619                                 continue;
 620                         }
 621                         // We multiply the threshold MAP_BORDER_THICKNESS by the average_local_cost
 622                         // to favor high cost hexes (a bit).
 623                         if (std::abs(my_cost_average - MAP_OFFENSIVE_SHIFT - enemy_cost_average) <
 624                                         MAP_BORDER_THICKNESS * average_local_cost_[map_location(x, y)]) {
 625                                 double border_movecost = (my_cost_average + enemy_cost_average) / 2;
 626 
 627                                 important_hexes_candidates[map_location(x, y)] = border_movecost;
 628 
 629                                 if (border_movecost < smallest_border_movecost) {
 630                                         smallest_border_movecost = border_movecost;
 631                                 }
 632                                 if (border_movecost > biggest_border_movecost) {
 633                                         biggest_border_movecost = border_movecost;
 634                                 }
 635                         }
 636                 }  // for
 637         }  // for
 638         double threshold = (biggest_border_movecost - smallest_border_movecost) *
 639                         MAP_BORDER_WIDTH + smallest_border_movecost;
 640         for (const border_cost_map::value_type& candidate : important_hexes_candidates) {
 641                 if (candidate.second < threshold) {
 642                         important_hexes_.insert(candidate.first);
 643                 }
 644         }
 645 }

 647 /**
 648  * For Map Analysis.
 649  * Calculates for a given unit the average defense on the map.
 650  * (According to important_hexes_ / important_terrain_)
 651  */
 652 double recruitment::get_average_defense(const std::string& u_type) const {
 653         const unit_type* const u_info = unit_types.find(u_type);
 654         if (!u_info) {
 655                 return 0.0;
 656         }
 657         long summed_defense = 0;
 658         int total_terrains = 0;
 659         for (const terrain_count_map::value_type& entry : important_terrain_) {
 660                 const t_translation::terrain_code& terrain = entry.first;
 661                 int count = entry.second;
 662                 int defense = 100 - u_info->movement_type().defense_modifier(terrain);
 663                 summed_defense += defense * count;
 664                 total_terrains += count;
 665         }
 666         double average_defense = (total_terrains == 0) ? 0.0 :
 667                         static_cast<double>(summed_defense) / total_terrains;
 668         return average_defense;
 669 }

 672  * For Map Analysis.
 673  * Creates cost maps for a side. Each hex is map to
 674  * a) the summed movecost and
 675  * b) how many units can reach this hex
 676  * for all units of side.
 677  */
 678 const  pathfind::full_cost_map recruitment::get_cost_map_of_side(int side) const {
 679         const unit_map& units = resources::gameboard->units();
 680         const team& team = resources::gameboard->get_team(side);
 681 
 682         pathfind::full_cost_map cost_map(true, true, team, true, true);
 683 
 684         // First add all existing units to cost_map.
 685         unsigned int unit_count = 0;
 686         for (const unit& unit : units) {
 687                 if (unit.side() != side || unit.can_recruit() ||
 688                                 unit.incapacitated() || unit.total_movement() <= 0) {
 689                         continue;
 690                 }
 691                 ++unit_count;
 692                 cost_map.add_unit(unit);
 693         }
 694 
 695         // If this side has not so many units yet, add unit_types with the leaders position as origin.
 696         if (unit_count < UNIT_THRESHOLD) {
 697                 std::vector<unit_map::const_iterator> leaders = units.find_leaders(side);
 698                 for (const unit_map::const_iterator& leader : leaders) {
 699                         // First add team-recruits (it's fine when (team-)recruits are added multiple times).
 700                         for (const std::string& recruit : team.recruits()) {
 701                                 cost_map.add_unit(leader->get_location(), unit_types.find(recruit), side);
 702                         }
 703 
 704                         // Next add extra-recruits.
 705                         for (const std::string& recruit : leader->recruits()) {
 706                                 cost_map.add_unit(leader->get_location(), unit_types.find(recruit), side);
 707                         }
 708                 }
 709         }
 710         return cost_map;

 774 /**
 775  * For Map Analysis.
 776  * Creates a std::set of hexes where a fight will occur with high probability.
 777  */
 778 void recruitment::update_important_hexes() {
 779         important_hexes_.clear();
 780         important_terrain_.clear();
 781         own_units_in_combat_counter_ = 0;
 782 
 783         update_average_local_cost();
 784         const gamemap& map = resources::gameboard->map();
 785         const unit_map& units = resources::gameboard->units();
 786 
 787         // Mark battle areas as important
 788         // This are locations where one of my units is adjacent
 789         // to a enemies unit.
 790         for (const unit& unit : units) {
 791                 if (unit.side() != get_side()) {
 792                         continue;
 793                 }
 794                 if (is_enemy_in_radius(unit.get_location(), 1)) {
 795                         // We found a enemy next to us. Mark our unit and all adjacent
 796                         // hexes as important.
 797                         std::vector<map_location> surrounding;
 798                         get_tiles_in_radius(unit.get_location(), 1, surrounding);
 799                         important_hexes_.insert(unit.get_location());
 800                         std::copy(surrounding.begin(), surrounding.end(),
 801                                         std::inserter(important_hexes_, important_hexes_.begin()));
 802                         ++own_units_in_combat_counter_;
 803                 }
 804         }
 805 
 806         // Mark area between me and enemies as important
 807         // This is done by creating a cost_map for each team.
 808         // A cost_map maps to each hex the average costs to reach this hex
 809         // for all units of the team.
 810         // The important hexes are those where my value on the cost map is
 811         // similar to a enemies one.
 812         const pathfind::full_cost_map my_cost_map = get_cost_map_of_side(get_side());
 813         for (const team& team : resources::gameboard->teams()) {
 814                 if (current_team().is_enemy(team.side())) {
 815                         const pathfind::full_cost_map enemy_cost_map = get_cost_map_of_side(team.side());
 816 
 817                         compare_cost_maps_and_update_important_hexes(my_cost_map, enemy_cost_map);
 818                 }
 819         }
 820 
 821         // Mark 'near' villages and area around them as important
 822         // To prevent a 'feedback' of important locations collect all
 823         // important villages first and add them and their surroundings
 824         // to important_hexes_ in a second step.
 825         std::vector<map_location> important_villages;
 826         for (const map_location& village : map.villages()) {
 827                 std::vector<map_location> surrounding;
 828                 get_tiles_in_radius(village, MAP_VILLAGE_NEARNESS_THRESHOLD, surrounding);
 829                 for (const map_location& hex : surrounding) {
 830                         if (important_hexes_.find(hex) != important_hexes_.end()) {
 831                                 important_villages.push_back(village);
 832                                 break;
 833                         }
 834                 }
 835         }
 836         for (const map_location& village : important_villages) {
 837                 important_hexes_.insert(village);
 838                 std::vector<map_location> surrounding;
 839                 get_tiles_in_radius(village, MAP_VILLAGE_SURROUNDING, surrounding);
 840                 for (const map_location& hex : surrounding) {
 841                         // only add hex if one of our units can reach the hex
 842                         if (map.on_board(hex) && my_cost_map.get_cost_at(hex.x, hex.y) != -1) {
 843                                 important_hexes_.insert(hex);
 844                         }
 845                 }
 846         }
 847 }

 849 /**
 850  * For Combat Analysis.
 851  * Calculates how good unit-type a is against unit type b.
 852  * If the value is bigger then 0, a is better then b.
 853  * If the value is 2.0 then unit-type a is twice as good as unit-type b.
 854  * Since this function is called very often it uses a cache.
 855  */
 856 double recruitment::compare_unit_types(const std::string& a, const std::string& b) {
 857         const unit_type* const type_a = unit_types.find(a);
 858         const unit_type* const type_b = unit_types.find(b);
 859         if (!type_a || !type_b) {
 860                 ERR_AI_RECRUITMENT << "Couldn't find unit type: " << ((type_a) ? b : a) << "." << std::endl;
 861                 return 0.0;
 862         }
 863         double defense_a = get_average_defense(a);
 864         double defense_b = get_average_defense(b);
 865 
 866         const double* cache_value = get_cached_combat_value(a, b, defense_a, defense_b);
 867         if (cache_value) {
 868                 return *cache_value;
 869         }
 870 
 871         double damage_to_a = 0.0;
 872         double damage_to_b = 0.0;
 873 
 874         // a attacks b
 875         simulate_attack(type_a, type_b, defense_a, defense_b, &damage_to_a, &damage_to_b);
 876         // b attacks a
 877         simulate_attack(type_b, type_a, defense_b, defense_a, &damage_to_b, &damage_to_a);
 878 
 879         int a_cost = (type_a->cost() > 0) ? type_a->cost() : 1;
 880         int b_cost = (type_b->cost() > 0) ? type_b->cost() : 1;
 881         int a_max_hp = (type_a->hitpoints() > 0) ? type_a->hitpoints() : 1;
 882         int b_max_hp = (type_b->hitpoints() > 0) ? type_b->hitpoints() : 1;
 883 
 884         double retval = 1.;
 885         // There are rare cases where a unit deals 0 damage (eg. Elvish Lady).
 886         // Then we just set the value to something reasonable.
 887         if (damage_to_a <= 0 && damage_to_b <= 0) {
 888                 retval = 0.;
 889         } else if (damage_to_a <= 0) {
 890                 retval = 2.;
 891         } else if (damage_to_b <= 0) {
 892                 retval = -2.;
 893         } else {
 894                 // Normal case
 895                 double value_of_a = damage_to_b / (b_max_hp * a_cost);
 896                 double value_of_b = damage_to_a / (a_max_hp * b_cost);
 897 
 898                 if (value_of_a > value_of_b) {
 899                         return value_of_a / value_of_b;
 900                 } else if (value_of_a < value_of_b) {
 901                         return -value_of_b / value_of_a;
 902                 } else {
 903                         return 0.;
 904                 }
 905         }
 906 
 907         // Insert in cache.
 908         const cached_combat_value entry(defense_a, defense_b, retval);
 909         std::set<cached_combat_value>& cache = combat_cache_[a][b];
 910         cache.insert(entry);
 911 
 912         return retval;
 913 }

 915 /**
 916  * Combat Analysis.
 917  * Main function.
 918  * Compares all enemy units with all of our possible recruits and fills
 919  * the scores.
 920  */
 921 void recruitment::do_combat_analysis(std::vector<data>* leader_data) {
 922         const unit_map& units = resources::gameboard->units();
 923 
 924         // Collect all enemy units (and their hp) we want to take into account in enemy_units.
 925         typedef std::vector<std::pair<std::string, int> > unit_hp_vector;
 926         unit_hp_vector enemy_units;
 927         for (const unit& unit : units) {
 928                 if (!current_team().is_enemy(unit.side()) || unit.incapacitated()) {
 929                         continue;
 930                 }
 931                 enemy_units.emplace_back(unit.type_id(), unit.hitpoints());
 932         }
 933         if (enemy_units.size() < UNIT_THRESHOLD) {
 934                 // Use also enemies recruitment lists and insert units into enemy_units.
 935                 for (const team& team : resources::gameboard->teams()) {
 936                         if (!current_team().is_enemy(team.side())) {
 937                                 continue;
 938                         }
 939                         std::set<std::string> possible_recruits;
 940                         // Add team recruits.
 941                         possible_recruits.insert(team.recruits().begin(), team.recruits().end());
 942                         // Add extra recruits.
 943                         const std::vector<unit_map::const_iterator> leaders = units.find_leaders(team.side());
 944                         for (unit_map::const_iterator leader : leaders) {
 945                                 possible_recruits.insert(leader->recruits().begin(), leader->recruits().end());
 946                         }
 947                         // Insert set in enemy_units.
 948                         for (const std::string& possible_recruit : possible_recruits) {
 949                                 const unit_type* recruit_type = unit_types.find(possible_recruit);
 950                                 if (recruit_type) {
 951                                         int hp = recruit_type->hitpoints();
 952                                         enemy_units.emplace_back(possible_recruit, hp);
 953                                 }
 954                         }
 955                 }
 956         }
 957 
 958         for (data& leader : *leader_data) {
 959                 if (leader.recruits.empty()) {
 960                         continue;
 961                 }
 962                 typedef std::map<std::string, double> simple_score_map;
 963                 simple_score_map temp_scores;
 964 
 965                 for (const unit_hp_vector::value_type& entry : enemy_units) {
 966                         const std::string& enemy_unit = entry.first;
 967                         int enemy_unit_hp = entry.second;
 968                         for (const std::string& recruit : leader.recruits) {
 969                                 double score = compare_unit_types(recruit, enemy_unit);
 970                                 score *= enemy_unit_hp;
 971                                 score = pow(score, COMBAT_SCORE_POWER);
 972                                 temp_scores[recruit] += score;
 973                         }
 974                 }
 975 
 976                 if (temp_scores.empty()) {
 977                         return;
 978                 }
 979                 // Find things for normalization.
 980                 double max = -99999.;
 981                 double sum = 0;
 982                 for (const simple_score_map::value_type& entry : temp_scores) {
 983                         double score = entry.second;
 984                         if (score > max) {
 985                                 max = score;
 986                         }
 987                         sum += score;
 988                 }
 989                 double average = sum / temp_scores.size();
 990 
 991                 // What we do now is a linear transformation.
 992                 // We want to map the scores in temp_scores to something between 0 and 100.
 993                 // The max score shall always be 100.
 994                 // The min score depends on the aspect "recruitment_diversity".
 995                 double new_100 = max;
 996                 double score_threshold = get_recruitment_diversity();
 997                 if (score_threshold <= 0) {
 998                         score_threshold = 0.0001;
 999                 }
1000                 double new_0 = max - (score_threshold * (max - average));
1001                 if (new_100 == new_0) {
1002                         // This can happen if max == average. (E.g. only one possible recruit)
1003                         new_0 -= 0.000001;
1004                 }
1005 
1006                 for (const simple_score_map::value_type& entry : temp_scores) {
1007                         const std::string& recruit = entry.first;
1008                         double score = entry.second;
1009 
1010                         // Here we transform.
1011                         // (If score <= new_0 then normalized_score will be 0)
1012                         // (If score = new_100 then normalized_score will be 100)
1013                         double normalized_score = 100 * ((score - new_0) / (new_100 - new_0));
1014                         if (normalized_score < 0) {
1015                                 normalized_score = 0;
1016                         }
1017                         leader.scores[recruit] += normalized_score;
1018                 }
1019         }  // for all leaders
1020 }

1022 /**
1023  * For Combat Analysis.
1024  * Returns the cached combat value for two unit types
1025  * or nullptr if there is none or terrain defenses are not within range.
1026  */
1027 const double* recruitment::get_cached_combat_value(const std::string& a, const std::string& b,
1028                 double a_defense, double b_defense) {
1029         double best_distance = 999;
1030         const double* best_value = nullptr;
1031         const std::set<cached_combat_value>& cache = combat_cache_[a][b];
1032         for (const cached_combat_value& entry : cache) {
1033                 double distance_a = std::abs(entry.a_defense - a_defense);
1034                 double distance_b = std::abs(entry.b_defense - b_defense);
1035                 if (distance_a <= COMBAT_CACHE_TOLERANCY && distance_b <= COMBAT_CACHE_TOLERANCY) {
1036                         if(distance_a + distance_b <= best_distance) {
1037                                 best_distance = distance_a + distance_b;
1038                                 best_value = &entry.value;
1039                         }
1040                 }
1041         }
1042         return best_value;
1043 }

1045 /**
1046  * For Combat Analysis.
1047  * This struct encapsulates all information for one attack simulation.
1048  * One attack simulation is defined by the unit-types, the weapons and the units defenses.
1049  */
1050 struct attack_simulation {
1051         const unit_type* attacker_type;
1052         const unit_type* defender_type;
1053         const battle_context_unit_stats attacker_stats;
1054         const battle_context_unit_stats defender_stats;
1055         combatant attacker_combatant;
1056         combatant defender_combatant;
1057 
1058         attack_simulation(const unit_type* attacker, const unit_type* defender,
1059                         double attacker_defense, double defender_defense,
1060                         const_attack_ptr att_weapon, const_attack_ptr def_weapon,
1061                         int average_lawful_bonus) :
1062                         attacker_type(attacker),
1063                         defender_type(defender),
1064                         attacker_stats(attacker, att_weapon, true, defender, def_weapon,
1065                                         round_double(defender_defense), average_lawful_bonus),
1066                         defender_stats(defender, def_weapon, false, attacker, att_weapon,
1067                                         round_double(attacker_defense), average_lawful_bonus),
1068                         attacker_combatant(attacker_stats),
1069                         defender_combatant(defender_stats)
1070         {
1071                 attacker_combatant.fight(defender_combatant);
1072         }
1073 
1074         bool better_result(const attack_simulation* other, bool for_defender) {
1075                 assert(other);
1076                 if (for_defender) {
1077                         return battle_context::better_combat(
1078                                         defender_combatant, attacker_combatant,
1079                                         other->defender_combatant, other->attacker_combatant, 0);
1080                 } else {
1081                         return battle_context::better_combat(
1082                                         attacker_combatant, defender_combatant,
1083                                         other->attacker_combatant, other->defender_combatant, 0);
1084                 }
1085         }
1086 
1087         double get_avg_hp_of_defender() const {
1088                 return get_avg_hp_of_combatant(false);
1089         }
1090 
1091         double get_avg_hp_of_attacker() const {
1092                 return get_avg_hp_of_combatant(true);
1093         }
1094         double get_avg_hp_of_combatant(bool attacker) const {
1095                 const combatant& combatant = (attacker) ? attacker_combatant : defender_combatant;
1096                 const unit_type* unit_type = (attacker) ? attacker_type : defender_type;
1097                 double avg_hp = combatant.average_hp(0);
1098 
1099                 // handle poisson
1100                 avg_hp -= combatant.poisoned * game_config::poison_amount;
1101 
1102                 avg_hp = std::max(0., avg_hp);
1103                 avg_hp = std::min(static_cast<double>(unit_type->hitpoints()), avg_hp);
1104                 return avg_hp;
1105         }
1106 };

1108 /**
1109  * For Combat Analysis.
1110  * Simulates a attack with a attacker and a defender.
1111  * The function will use battle_context::better_combat() to decide which weapon to use.
1112  */
1113 void recruitment::simulate_attack(
1114                         const unit_type* const attacker, const unit_type* const defender,
1115                         double attacker_defense, double defender_defense,
1116                         double* damage_to_attacker, double* damage_to_defender) const {
1117         if(!attacker || !defender || !damage_to_attacker || !damage_to_defender) {
1118                 ERR_AI_RECRUITMENT << "nullptr pointer in simulate_attack()" << std::endl;
1119                 return;
1120         }
1121         const_attack_itors attacker_weapons = attacker->attacks();
1122         const_attack_itors defender_weapons = defender->attacks();
1123 
1124         std::shared_ptr<attack_simulation> best_att_attack;
1125 
1126         // Let attacker choose weapon
1127         for (const attack_type& att_weapon : attacker_weapons) {
1128                 std::shared_ptr<attack_simulation> best_def_response;
1129                 // Let defender choose weapon
1130                 for (const attack_type& def_weapon : defender_weapons) {
1131                         if (att_weapon.range() != def_weapon.range()) {
1132                                 continue;
1133                         }

1135                                         attacker, defender,
1136                                         attacker_defense, defender_defense,
1137                                         att_weapon.shared_from_this(), def_weapon.shared_from_this(), average_lawful_bonus_));
1138                         if (!best_def_response || simulation->better_result(best_def_response.get(), true)) {
1139                                 best_def_response = simulation;
1140                         }
1141                 }  // for defender weapons
1142

1146                                         attacker, defender,
1147                                         attacker_defense, defender_defense,
1148                                         att_weapon.shared_from_this(), nullptr, average_lawful_bonus_));

1153         }  // for attacker weapons
1154

1156                 return;
1157         }
1158 
1159         *damage_to_defender += (defender->hitpoints() - best_att_attack->get_avg_hp_of_defender());

1453 /**
1454  * For Aspect "recruitment_save_gold".
1455  * Returns our_total_unit_costs / enemy_total_unit_costs.
1456  */
1457 double recruitment::get_unit_ratio() const {
1458         const unit_map& units = resources::gameboard->units();
1459         double own_total_value = 0.;
1460         double team_total_value = 0.;
1461         double enemy_total_value = 0.;
1462         for (const unit& unit : units) {
1463                 if (unit.incapacitated() || unit.total_movement() <= 0 || unit.can_recruit()) {
1464                         continue;
1465                 }
1466                 double value = unit.cost() * unit.hitpoints() / unit.max_hitpoints();
1467                 if (current_team().is_enemy(unit.side())) {
1468                         enemy_total_value += value;
1469                 } else {
1470                         team_total_value += value;
1471                         if (unit.side() == current_team().side()) {
1472                                 own_total_value += value;
1473                         }
1474                 }
1475         }
1476         int allies_count = 0;
1477         for (const team& team : resources::gameboard->teams()) {
1478                 if (!current_team().is_enemy(team.side())) {
1479                         ++allies_count;
1480                 }
1481         }
1482         // If only the leader is left, the values could be 0.
1483         // Catch those cases and return something reasonable.
1484         if ((own_total_value == 0. || team_total_value == 0) && enemy_total_value == 0.) {
1485                 return 0.;  // do recruit
1486         } else if (enemy_total_value == 0.) {
1487                 return 999.;  // save money
1488         }
1489 
1490         // We calculate two ratios: One for the team and one for just our self.
1491         // Then we return the minimum.
1492         // This prevents cases where side1 will recruit until the save_gold begin threshold
1493         // is reached, and side2 won't recruit anything. (assuming side1 and side2 are allied)
1494         double own_ratio = (own_total_value / enemy_total_value) * allies_count;
1495         double team_ratio = team_total_value / enemy_total_value;
1496         return std::min<double>(own_ratio, team_ratio);
1497 }

1551 /**
1552  * Will give a penalty to similar units. Similar units are units in one advancement tree.
1553  * Example (Archer can advance to Ranger):
1554  *                 before    after
1555  * Elvish Fighter:   50        50
1556  * Elvish Archer:    50        25
1557  * Elvish Ranger:    50        25
1558  */
1559 void recruitment::do_similarity_penalty(std::vector<data>* leader_data) const {
1560         if (!leader_data) {
1561                 return;
1562         }
1563         for (data& data : *leader_data) {
1564                 // First we count how many similarities each recruit have to other ones (in a map).
1565                 // Some examples:
1566                 // If unit A and unit B have nothing to do with each other, they have similarity = 0.
1567                 // If A advances to B both have similarity = 1.
1568                 // If A advances to B and B to C, A, B and C have similarity = 2.
1569                 // If A advances to B or C, A have similarity = 2. B and C have similarity = 1.
1570                 typedef std::map<std::string, int> similarity_map;
1571                 similarity_map similarities;
1572                 for (const score_map::value_type& entry : data.scores) {
1573                         const std::string& recruit = entry.first;
1574                         const unit_type* recruit_type = unit_types.find(recruit);
1575                         if (!recruit_type) {
1576                                 continue;
1577                         }
1578                         for (const std::string& advanced_type : recruit_type->advancement_tree()) {
1579                                 if (data.scores.count(advanced_type) != 0) {
1580                                         ++similarities[recruit];
1581                                         ++similarities[advanced_type];
1582                                 }
1583                         }
1584                 }
1585                 // Now we divide each score by similarity + 1.
1586                 for (score_map::value_type& entry : data.scores) {
1587                         const std::string& recruit = entry.first;
1588                         double& score = entry.second;
1589                         score /= (similarities[recruit] + 1);
1590                 }
1591         }
1592 }

1700  * scouts we want to recruit.

1702 void recruitment::update_scouts_wanted() {

1704         if (get_villages_per_scout() == 0) {
1705                 return;
1706         }
1707         int neutral_villages = 0;
1708         // We recruit the initial allocation of scouts
1709         // based on how many neutral villages there are.
1710         for (const map_location& village : resources::gameboard->map().villages()) {
1711                 if (resources::gameboard->village_owner(village) == -1) {
1712                         ++neutral_villages;
1713                 }

1716 
1717         // The villages per scout is for a two-side battle,
1718         // accounting for all neutral villages on the map.

1720         // making us get twice as many scouts.
1721         double villages_per_scout = (VILLAGE_PER_SCOUT_MULTIPLICATOR * get_villages_per_scout()) / 2;
1722 
1723         scouts_wanted_ = (villages_per_scout > 0) ? round_double(our_share / villages_per_scout) : 0;
1724 
1725         if (scouts_wanted_ == 0) {
1726                 return;
1727         }
1728 
1729         // Subtract already recruited scouts.
1730         for (const count_map::value_type& entry : own_units_count_) {
1731                 const std::string& unit_type = entry.first;
1732                 const int count = entry.second;
1733                 if (recruit_matches_type(unit_type, "scout")) {
1734                         scouts_wanted_ -= count;
1735                 }
1736         }
1737 }

1779 recruitment_aspect::recruitment_aspect(readonly_context &context, const config &cfg, const std::string &id)
1780         : standard_aspect<config>(context, cfg, id)
1781 {
1782         config parsed_cfg(cfg.has_child("value") ? cfg.child("value") : cfg);
1783         // First, transform simplified tags into [recruit] tags.
1784         for (config pattern : parsed_cfg.child_range("pattern")) {
1785                 parsed_cfg["pattern"] = true;
1786                 parsed_cfg.add_child("recruit", pattern);
1787         }
1788         for (config total : parsed_cfg.child_range("total")) {
1789                 parsed_cfg["total"] = true;
1790                 parsed_cfg.add_child("recruit", total);
1791         }
1792         parsed_cfg.clear_children("pattern", "total");
1793         // Then, if there's no [recruit], add one.
1794         if (!parsed_cfg.has_child("recruit")) {
1795                 parsed_cfg.add_child("recruit", config_of("importance", 0));
1796         }
1797         // Finally, populate our lists
1798         for (config job : parsed_cfg.child_range("recruit")) {
1799                 create_job(jobs_, job);
1800         }
1801         for (config lim : parsed_cfg.child_range("limit")) {
1802                 create_limit(limits_, lim);
1803         }
1804         std::function<void(std::vector<std::shared_ptr<recruit_job> >&, const config&)> factory_jobs =
1805                 std::bind(&recruitment_aspect::create_job,*this,_1,_2);
1806         std::function<void(std::vector<std::shared_ptr<recruit_limit> >&, const config&)> factory_limits =
1807                 std::bind(&recruitment_aspect::create_limit,*this,_1,_2);
1808         register_vector_property(property_handlers(), "recruit", jobs_, factory_jobs);
1809         register_vector_property(property_handlers(), "limit", limits_, factory_limits);
1810 }
