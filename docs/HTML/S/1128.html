  73 connect_engine::connect_engine(saved_game& state, const bool first_scenario, mp_campaign_info* campaign_info)
  74         : level_()
  75         , state_(state)
  76         , params_(state.mp_settings())
  77         , default_controller_(campaign_info ? CNTR_NETWORK : CNTR_LOCAL)
  78         , campaign_info_(campaign_info)
  79         , first_scenario_(first_scenario)
  80         , force_lock_settings_()
  81         , side_engines_()
  82         , era_factions_()
  83         , team_names_()
  84         , user_team_names_()
  85         , player_teams_()
  86 {
  87         // Initial level config from the mp_game_settings.
  88         level_ = mp::initial_level_config(state_);
  89         if(level_.empty()) {
  90                 return;
  91         }
  92 
  93         const bool is_mp = state_.classification().is_normal_mp_game();
  94         force_lock_settings_ = (!state.mp_settings().saved_game) && scenario()["force_lock_settings"].to_bool(!is_mp);
  95 
  96         // Original level sides.
  97         config::child_itors sides = current_config()->child_range("side");
  98 
  99         // AI algorithms.
 100         ai::configuration::add_era_ai_from_config(level_.child("era"));
 101         ai::configuration::add_mod_ai_from_config(level_.child_range("modification"));
 102 
 103         // Set the team name lists and modify the original level sides if necessary.
 104         std::vector<std::string> original_team_names;
 105         std::string team_prefix(_("Team") + " ");
 106 
 107         int side_count = 1;
 108         for(config& side : sides) {
 109                 const std::string side_str = std::to_string(side_count);
 110 
 111                 config::attribute_value& team_name = side["team_name"];
 112                 config::attribute_value& user_team_name = side["user_team_name"];
 113 
 114                 // Revert to default values if appropriate.
 115                 if(team_name.empty()) {
 116                         team_name = side_str;
 117                 }
 118 
 119                 if(params_.use_map_settings && user_team_name.empty()) {
 120                         user_team_name = team_name;
 121                 }
 122 
 123                 bool add_team = true;
 124                 if(params_.use_map_settings) {
 125                         // Only add a team if it is not found.
 126                         bool found = std::find(team_names_.begin(), team_names_.end(),
 127                                 team_name.str()) != team_names_.end();
 128 
 129                         if(found) {
 130                                 add_team = false;
 131                         }
 132                 } else {
 133                         // Always add a new team for every side, but leave the specified team assigned to a side if there is one.
 134                         auto name_itor = std::find(original_team_names.begin(), original_team_names.end(), team_name.str());
 135 
 136                         // Note that the prefix "Team " is untranslatable, as team_name is not meant to be translated. This is needed
 137                         // so that the attribute is not interpretted as an int when reading from config, which causes bugs later.
 138                         if(name_itor == original_team_names.end()) {
 139                                 original_team_names.push_back(team_name);
 140 
 141                                 team_name = "Team " + std::to_string(original_team_names.size());
 142                         } else {
 143                                 team_name = "Team " + std::to_string(name_itor - original_team_names.begin() + 1);
 144                         }
 145 
 146                         user_team_name = team_prefix + side_str;
 147                 }
 148 
 149                 if(add_team) {
 150                         team_names_.push_back(params_.use_map_settings ? team_name : "Team " + side_str);
 151                         user_team_names_.push_back(user_team_name.t_str().to_serialized());
 152 
 153                         if(side["allow_player"].to_bool(true) || game_config::debug) {
 154                                 player_teams_.push_back(user_team_name.str());
 155                         }
 156                 }
 157 
 158                 ++side_count;
 159         }
 160 
 161         // Selected era's factions.
 162         for(const config& era : level_.child("era").child_range("multiplayer_side")) {
 163                 era_factions_.push_back(&era);
 164         }
 165 
 166         game_config::add_color_info(scenario());
 167 
 168         // Create side engines.
 169         int index = 0;
 170         for(const config& s : sides) {
 171                 side_engine_ptr new_side_engine(new side_engine(s, *this, index));
 172                 side_engines_.push_back(new_side_engine);
 173 
 174                 index++;
 175         }
 176 
 177         if(first_scenario_) {
 178                 // Add host to the connected users list.
 179                 import_user(preferences::login(), false);
 180         } else {
 181                 // Add host but don't assign a side to him.
 182                 import_user(preferences::login(), true);
 183 
 184                 // Load reserved players information into the sides.
 185                 load_previous_sides_users();
 186         }
 187 
 188         // Only updates the sides in the level.
 189         update_level();
 190 
 191         // If we are connected, send data to the connected host.
 192         send_level_data();
 193 }

 211 void connect_engine::import_user(const config& data, const bool observer, int side_taken)
 212 {
 213         const std::string& username = data["name"];
 214         assert(!username.empty());
 215         if(campaign_info_) {
 216                 connected_users_rw().insert(username);
 217         }
 218 
 219         update_side_controller_options();
 220 
 221         if(observer) {
 222                 return;
 223         }
 224 
 225         bool side_assigned = false;
 226         if(side_taken >= 0) {
 227                 side_engines_[side_taken]->place_user(data, true);
 228                 side_assigned = true;
 229         }
 230 
 231         // Check if user has a side(s) reserved for him.
 232         for(side_engine_ptr side : side_engines_) {
 233                 if(side->reserved_for() == username && side->player_id().empty() && side->controller() != CNTR_COMPUTER) {
 234                         side->place_user(data);
 235 
 236                         side_assigned = true;
 237                 }
 238         }
 239 
 240         // If no sides were assigned for a user,
 241         // take a first available side.
 242         if(side_taken < 0 && !side_assigned) {
 243                 for(side_engine_ptr side : side_engines_) {
 244                         if(side->available_for_user(username) ||
 245                                 side->controller() == CNTR_LOCAL) {
 246                                         side->place_user(data);
 247 
 248                                         side_assigned = true;
 249                                         break;
 250                         }
 251                 }
 252         }
 253 
 254         // Check if user has taken any sides, which should get control
 255         // over any other sides.
 256         for(side_engine_ptr user_side : side_engines_) {
 257                 if(user_side->player_id() == username && !user_side->previous_save_id().empty()) {
 258                         for(side_engine_ptr side : side_engines_){
 259                                 if(side->player_id().empty() && side->previous_save_id() == user_side->previous_save_id()) {
 260                                         side->place_user(data);
 261                                 }
 262                         }
 263                 }
 264         }
 265 }

 302 bool connect_engine::can_start_game() const
 303 {
 304         if(side_engines_.size() == 0) {
 305                 return true;
 306         }
 307 
 308         // First check if all sides are ready to start the game.
 309         for(side_engine_ptr side : side_engines_) {
 310                 if(!side->ready_for_start()) {
 311                         const int side_num = side->index() + 1;
 312                         DBG_MP << "not all sides are ready, side " <<
 313                                 side_num << " not ready\n";
 314 
 315                         return false;
 316                 }
 317         }
 318 
 319         DBG_MP << "all sides are ready" << std::endl;
 320 
 321         /*
 322          * If at least one human player is slotted with a player/ai we're allowed
 323          * to start. Before used a more advanced test but it seems people are
 324          * creative in what is used in multiplayer [1] so use a simpler test now.
 325          * [1] http://bugs.debian.org/cgi-bin/bugreport.cgi?bug=568029
 326          */
 327         for(side_engine_ptr side : side_engines_) {
 328                 if(side->controller() != CNTR_EMPTY && side->allow_player()) {
 329                         return true;
 330                 }
 331         }
 332 
 333         return false;
 334 }

 388 void connect_engine::start_game()
 389 {
 390         DBG_MP << "starting a new game" << std::endl;
 391 
 392         // Resolves the "random faction", "random gender" and "random message"
 393         // Must be done before shuffle sides, or some cases will cause errors
 394         randomness::mt_rng rng; // Make an RNG for all the shuffling and random faction operations
 395         for(side_engine_ptr side : side_engines_) {
 396                 std::vector<std::string> avoid_faction_ids;
 397 
 398                 // If we aren't resolving random factions independently at random, calculate which factions should not appear for this side.
 399                 if(params_.random_faction_mode != mp_game_settings::RANDOM_FACTION_MODE::DEFAULT) {
 400                         for(side_engine_ptr side2 : side_engines_) {
 401                                 if(!side2->flg().is_random_faction()) {
 402                                         switch(params_.random_faction_mode.v) {
 403                                                 case mp_game_settings::RANDOM_FACTION_MODE::NO_MIRROR:
 404                                                         avoid_faction_ids.push_back(side2->flg().current_faction()["id"].str());
 405                                                         break;
 406                                                 case mp_game_settings::RANDOM_FACTION_MODE::NO_ALLY_MIRROR:
 407                                                         if(side2->team() == side->team()) {// TODO: When the connect engines are fixed to allow multiple teams, this should be changed to "if side1 and side2 are allied, i.e. their list of teams has nonempty intersection"
 408                                                                 avoid_faction_ids.push_back(side2->flg().current_faction()["id"].str());
 409                                                         }
 410                                                         break;
 411                                                 default:
 412                                                         break; // assert(false);
 413                                         }
 414                                 }
 415                         }
 416                 }
 417                 side->resolve_random(rng, avoid_faction_ids);
 418         }
 419 
 420         // Shuffle sides (check settings and if it is a re-loaded game).
 421         // Must be done after resolve_random() or shuffle sides, or they won't work.
 422         if(state_.mp_settings().shuffle_sides && !force_lock_settings_ && !(level_.child("snapshot") && level_.child("snapshot").child("side"))) {
 423 
 424                 // Only playable sides should be shuffled.
 425                 std::vector<int> playable_sides;
 426                 for(side_engine_ptr side : side_engines_) {
 427                         if(side->allow_player() && side->allow_shuffle()) {
 428                                 playable_sides.push_back(side->index());
 429                         }
 430                 }
 431 
 432                 // Fisher-Yates shuffle.
 433                 for(int i = playable_sides.size(); i > 1; i--) {
 434                         int j_side = playable_sides[rng.get_next_random() % i];
 435                         int i_side = playable_sides[i - 1];
 436 
 437                         if(i_side == j_side) continue; //nothing to swap
 438 
 439                         // First we swap everything about a side with another
 440                         side_engine_ptr tmp_side = side_engines_[j_side];
 441                         side_engines_[j_side] = side_engines_[i_side];
 442                         side_engines_[i_side] = tmp_side;
 443 
 444                         // Some 'child' variables such as village ownership and
 445                         // initial side units need to be swapped over as well
 446                         std::multimap<std::string, config> tmp_side_children = side_engines_[j_side]->get_side_children();
 447                         side_engines_[j_side]->set_side_children(side_engines_[i_side]->get_side_children());
 448                         side_engines_[i_side]->set_side_children(tmp_side_children);
 449 
 450                         // Then we revert the swap for fields that are unique to
 451                         // player control and the team they selected
 452                         int tmp_index = side_engines_[j_side]->index();
 453                         side_engines_[j_side]->set_index(side_engines_[i_side]->index());
 454                         side_engines_[i_side]->set_index(tmp_index);
 455 
 456                         int tmp_team = side_engines_[j_side]->team();
 457                         side_engines_[j_side]->set_team(side_engines_[i_side]->team());
 458                         side_engines_[i_side]->set_team(tmp_team);
 459                 }
 460         }
 461 
 462         // Make other clients not show the results of resolve_random().
 463         config lock("stop_updates");
 464         send_to_server(lock);
 465 
 466         update_and_send_diff(true);
 467 
 468         save_reserved_sides_information();
 469 
 470         // Build the gamestate object after updating the level.
 471         mp::level_to_gamestate(level_, state_);
 472 
 473         send_to_server(config("start_game"));
 474 }

 476 void connect_engine::start_game_commandline(const commandline_options& cmdline_opts)
 477 {
 478         DBG_MP << "starting a new game in commandline mode" << std::endl;
 479 
 480         typedef std::tuple<unsigned int, std::string> mp_option;
 481 
 482         randomness::mt_rng rng;
 483 
 484         unsigned num = 0;
 485         for(side_engine_ptr side : side_engines_) {
 486                 num++;
 487 
 488                 // Set the faction, if commandline option is given.
 489                 if(cmdline_opts.multiplayer_side) {
 490                         for(const mp_option& option : *cmdline_opts.multiplayer_side) {
 491 
 492                                 if(std::get<0>(option) == num) {
 493                                         if(std::find_if(era_factions_.begin(), era_factions_.end(), [&option](const config* faction) { return (*faction)["id"] == std::get<1>(option); }) != era_factions_.end()) {
 494                                                 DBG_MP << "\tsetting side " << std::get<0>(option) << "\tfaction: " << std::get<1>(option) << std::endl;
 495 
 496                                                 side->set_faction_commandline(std::get<1>(option));
 497                                         }
 498                                         else {
 499                                                 ERR_MP << "failed to set side " << std::get<0>(option) << " to faction " << std::get<1>(option) << std::endl;
 500                                         }
 501                                 }
 502                         }
 503                 }
 504 
 505                 // Set the controller, if commandline option is given.
 506                 if(cmdline_opts.multiplayer_controller) {
 507                         for(const mp_option& option : *cmdline_opts.multiplayer_controller) {
 508 
 509                                 if(std::get<0>(option) == num) {
 510                                         DBG_MP << "\tsetting side " << std::get<0>(option) <<
 511                                                 "\tfaction: " << std::get<1>(option) << std::endl;
 512 
 513                                         side->set_controller_commandline(std::get<1>(option));
 514                                 }
 515                         }
 516                 }
 517 
 518                 // Set AI algorithm to RCA AI for all sides,
 519                 // then override if commandline option was given.
 520                 side->set_ai_algorithm("ai_default_rca");
 521                 if(cmdline_opts.multiplayer_algorithm) {
 522                         for(const mp_option& option : *cmdline_opts.multiplayer_algorithm) {
 523 
 524                                 if(std::get<0>(option) == num) {
 525                                         DBG_MP << "\tsetting side " << std::get<0>(option) <<
 526                                                 "\tfaction: " << std::get<1>(option) << std::endl;
 527 
 528                                         side->set_ai_algorithm(std::get<1>(option));
 529                                 }
 530                         }
 531                 }
 532 
 533                 // Finally, resolve "random faction",
 534                 // "random gender" and "random message", if any remains unresolved.
 535                 side->resolve_random(rng);
 536         } // end top-level loop
 537 
 538         update_and_send_diff(true);
 539 
 540         // Update sides with commandline parameters.
 541         if(cmdline_opts.multiplayer_turns) {
 542                 DBG_MP << "\tsetting turns: " << *cmdline_opts.multiplayer_turns <<
 543                         std::endl;
 544                 scenario()["turns"] = *cmdline_opts.multiplayer_turns;
 545         }
 546 
 547         for(config &side : scenario().child_range("side")) {
 548                 if(cmdline_opts.multiplayer_ai_config) {
 549                         for(const mp_option& option : *cmdline_opts.multiplayer_ai_config) {
 550 
 551                                 if(std::get<0>(option) == side["side"].to_unsigned()) {
 552                                         DBG_MP << "\tsetting side " << side["side"] <<
 553                                                 "\tai_config: " << std::get<1>(option) << std::endl;
 554 
 555                                         side["ai_config"] = std::get<1>(option);
 556                                 }
 557                         }
 558                 }
 559 
 560                 // Having hard-coded values here is undesirable,
 561                 // but that's how it is done in the MP lobby
 562                 // part of the code also.
 563                 // Should be replaced by settings/constants in both places
 564                 if(cmdline_opts.multiplayer_ignore_map_settings) {
 565                         side["gold"] = 100;
 566                         side["income"] = 1;
 567                 }
 568 
 569                 typedef std::tuple<unsigned int, std::string, std::string> mp_parameter;
 570 
 571                 if(cmdline_opts.multiplayer_parm) {
 572                         for(const mp_parameter& parameter : *cmdline_opts.multiplayer_parm) {
 573 
 574                                 if(std::get<0>(parameter) == side["side"].to_unsigned()) {
 575                                         DBG_MP << "\tsetting side " << side["side"] << " " <<
 576                                                 std::get<1>(parameter) << ": " << std::get<2>(parameter) << std::endl;
 577 
 578                                         side[std::get<1>(parameter)] = std::get<2>(parameter);
 579                                 }
 580                         }
 581                 }
 582     }
 583 
 584         save_reserved_sides_information();
 585 
 586         // Build the gamestate object after updating the level
 587         mp::level_to_gamestate(level_, state_);
 588         send_to_server(config("start_game"));
 589 }

 598 std::pair<bool, bool> connect_engine::process_network_data(const config& data)
 599 {
 600         std::pair<bool, bool> result(std::make_pair(false, true));
 601 
 602         if(data.child("leave_game")) {
 603                 result.first = true;
 604                 return result;
 605         }
 606 
 607         // A side has been dropped.
 608         if(const config& side_drop = data.child("side_drop")) {
 609                 unsigned side_index = side_drop["side_num"].to_int() - 1;
 610 
 611                 if(side_index < side_engines_.size()) {
 612                         side_engine_ptr side_to_drop = side_engines_[side_index];
 613 
 614                         // Remove user, whose side was dropped.
 615                         connected_users_rw().erase(side_to_drop->player_id());
 616                         update_side_controller_options();
 617 
 618                         side_to_drop->reset();
 619 
 620                         update_and_send_diff();
 621 
 622                         return result;
 623                 }
 624         }
 625 
 626         // A player is connecting to the game.
 627         if(!data["side"].empty()) {
 628                 unsigned side_taken = data["side"].to_int() - 1;
 629 
 630                 // Checks if the connecting user has a valid and unique name.
 631                 const std::string name = data["name"];
 632                 if(name.empty()) {
 633                         config response;
 634                         response["failed"] = true;
 635                         send_to_server(response);
 636 
 637                         ERR_CF << "ERROR: No username provided with the side." << std::endl;
 638 
 639                         return result;
 640                 }
 641 
 642                 if(connected_users().find(name) != connected_users().end()) {
 643                         // TODO: Seems like a needless limitation
 644                         // to only allow one side per player.
 645                         if(find_user_side_index_by_id(name) != -1) {
 646                                 config response;
 647                                 response["failed"] = true;
 648                                 response["message"] = "The nickname '" + name +
 649                                         "' is already in use.";
 650                                 send_to_server(response);
 651 
 652                                 return result;
 653                         } else {
 654                                 connected_users_rw().erase(name);
 655                                 update_side_controller_options();
 656                                 config observer_quit;
 657                                 observer_quit.add_child("observer_quit")["name"] = name;
 658                                 send_to_server(observer_quit);
 659                         }
 660                 }
 661 
 662                 // Assigns this user to a side.
 663                 if(side_taken < side_engines_.size()) {
 664                         if(!side_engines_[side_taken]->available_for_user(name)) {
 665                                 // This side is already taken.
 666                                 // Try to reassing the player to a different position.
 667                                 side_taken = 0;
 668                                 for(side_engine_ptr s : side_engines_) {
 669                                         if(s->available_for_user()) {
 670                                                 break;
 671                                         }
 672 
 673                                         side_taken++;
 674                                 }
 675 
 676                                 if(side_taken >= side_engines_.size()) {
 677                                         config response;
 678                                         response["failed"] = true;
 679                                         send_to_server(response);
 680 
 681                                         config res;
 682                                         config& kick = res.add_child("kick");
 683                                         kick["username"] = data["name"];
 684                                         send_to_server(res);
 685 
 686                                         update_and_send_diff();
 687 
 688                                         ERR_CF << "ERROR: Couldn't assign a side to '" <<
 689                                                 name << "'\n";
 690 
 691                                         return result;
 692                                 }
 693                         }
 694 
 695                         LOG_CF << "client has taken a valid position\n";
 696 
 697                         import_user(data, false, side_taken);
 698                         update_and_send_diff();
 699 
 700                         // Wait for them to choose faction if allowed.
 701                         side_engines_[side_taken]->set_waiting_to_choose_status(side_engines_[side_taken]->allow_changes());
 702                         LOG_MP << "waiting to choose status = " << side_engines_[side_taken]->allow_changes() << std::endl;
 703                         result.second = false;
 704 
 705                         LOG_NW << "sent player data\n";
 706                 } else {
 707                         ERR_CF << "tried to take illegal side: " << side_taken << std::endl;
 708 
 709                         config response;
 710                         response["failed"] = true;
 711                         send_to_server(response);
 712                 }
 713         }
 714 
 715 
 716         if(const config& change_faction = data.child("change_faction")) {
 717                 int side_taken = find_user_side_index_by_id(change_faction["name"]);
 718                 if(side_taken != -1 || !first_scenario_) {
 719                         import_user(change_faction, false, side_taken);
 720                         update_and_send_diff();
 721                 }
 722         }
 723 
 724         if(const config& observer = data.child("observer")) {
 725                 import_user(observer, true);
 726                 update_and_send_diff();
 727         }
 728 
 729         if(const config& observer = data.child("observer_quit")) {
 730                 const t_string& observer_name = observer["name"];
 731                 if(!observer_name.empty()) {
 732                         if((connected_users().find(observer_name) != connected_users().end()) &&
 733                                 (find_user_side_index_by_id(observer_name) != -1)) {
 734 
 735                                 connected_users_rw().erase(observer_name);
 736                                 update_side_controller_options();
 737                                 update_and_send_diff();
 738                         }
 739                 }
 740         }
 741 
 742         return result;
 743 }

 848 side_engine::side_engine(const config& cfg, connect_engine& parent_engine, const int index)
 849         : cfg_(cfg)
 850         , parent_(parent_engine)
 851         , controller_(CNTR_NETWORK)
 852         , current_controller_index_(0)
 853         , controller_options_()
 854         , allow_player_(cfg["allow_player"].to_bool(true))
 855         , controller_lock_(cfg["controller_lock"].to_bool(parent_.force_lock_settings_) && parent_.params_.use_map_settings)
 856         , index_(index)
 857         , team_(0)
 858         , color_(std::min(index, gamemap::MAX_PLAYERS - 1))
 859         , gold_(cfg["gold"].to_int(100))
 860         , income_(cfg["income"])
 861         , reserved_for_(cfg["current_player"])
 862         , player_id_()
 863         , ai_algorithm_()
 864         , chose_random_(cfg["chose_random"].to_bool(false))
 865         , disallow_shuffle_(cfg["disallow_shuffle"].to_bool(false))
 866         , flg_(parent_.era_factions_, cfg_, parent_.force_lock_settings_, parent_.params_.use_map_settings, parent_.params_.saved_game)
 867         , allow_changes_(!parent_.params_.saved_game && !(flg_.choosable_factions().size() == 1 && flg_.choosable_leaders().size() == 1 && flg_.choosable_genders().size() == 1))
 868         , waiting_to_choose_faction_(allow_changes_)
 869         , custom_color_()
 870         , color_id_()
 871         , color_options_()
 872 {
 873         // Save default attributes that could be overwirtten by the faction, so that correct faction lists would be
 874         // initialized by flg_manager when the new side config is sent over network.
 875         cfg_.add_child("default_faction", config {
 876                 "type",    cfg_["type"],
 877                 "gender",  cfg_["gender"],
 878                 "faction", cfg_["faction"],
 879                 "recruit", cfg_["recruit"],
 880         });
 881 
 882         if(cfg_["side"].to_int(index_ + 1) != index_ + 1) {
 883                 ERR_CF << "found invalid side=" << cfg_["side"].to_int(index_ + 1) << " in definition of side number " << index_ + 1 << std::endl;
 884         }
 885 
 886         cfg_["side"] = index_ + 1;
 887 
 888         // Check if this side should give its control to some other side.
 889         const size_t side_cntr_index = cfg_["controller"].to_int(-1) - 1;
 890         if(side_cntr_index < parent_.side_engines().size()) {
 891                 // Remove this attribute to avoid locking side
 892                 // to non-existing controller type.
 893                 cfg_.remove_attribute("controller");
 894 
 895                 cfg_["previous_save_id"] = parent_.side_engines()[side_cntr_index]->previous_save_id();
 896                 ERR_MP << "contoller=<number> is deperecated\n";
 897         }
 898 
 899         if(!parent_.params_.saved_game && cfg_["save_id"].str().empty()) {
 900                 assert(cfg_["id"].empty()); // we already setted "save_id" to "id" if "id" existed.
 901                 cfg_["save_id"] = parent_.scenario()["id"].str() + "_" + std::to_string(index);
 902         }
 903 
 904         if(cfg_["controller"] != "human" && cfg_["controller"] != "ai" && cfg_["controller"] != "null") {
 905                 //an invalid contoller type was specified. Remove it to prevent asertion failures later.
 906                 cfg_.remove_attribute("controller");
 907         }
 908 
 909         update_controller_options();
 910 
 911         // Tweak the controllers.
 912         if(parent_.state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::SCENARIO && cfg_["controller"].blank()) {
 913                 cfg_["controller"] = "ai";
 914         }
 915 
 916         if(cfg_["controller"] == "null") {
 917                 set_controller(CNTR_EMPTY);
 918         } else if(cfg_["controller"] == "ai") {
 919                 set_controller(CNTR_COMPUTER);
 920         } else if(parent_.default_controller_ == CNTR_NETWORK && !reserved_for_.empty()) {
 921                 // Reserve a side for "current_player", unless the side
 922                 // is played by an AI.
 923                 set_controller(CNTR_RESERVED);
 924         } else if(allow_player_) {
 925                 set_controller(parent_.default_controller_);
 926         } else {
 927                 // AI is the default.
 928                 set_controller(CNTR_COMPUTER);
 929         }
 930 
 931         // Initialize team and color.
 932         unsigned team_name_index = 0;
 933         for(const std::string& name : parent_.team_names_) {
 934                 if(name == cfg["team_name"]) {
 935                         break;
 936                 }
 937 
 938                 team_name_index++;
 939         }
 940         if(team_name_index >= parent_.team_names_.size()) {
 941                 assert(!parent_.team_names_.empty());
 942                 team_ = 0;
 943                 WRN_MP << "In side_engine constructor: Could not find my team_name " << cfg["team_name"] << " among the mp connect engine's list of team names. I am being assigned to the first team. This may indicate a bug!" << std::endl;
 944         } else {
 945                 team_ = team_name_index;
 946         }
 947 
 948         color_options_ = game_config::default_colors;
 949         color_id_ = color_options_[color_];
 950 
 951         if(!cfg["color"].empty()) {
 952                 if(cfg["color"].to_int()) {
 953                         color_ = cfg["color"].to_int() - 1;
 954                         color_id_ = color_options_[color_];
 955                 } else {
 956                         custom_color_ = cfg["color"].str();
 957 
 958                         const auto iter = std::find(color_options_.begin(), color_options_.end(), custom_color_);
 959 
 960                         if(iter != color_options_.end()) {
 961                                 color_id_ = *iter;
 962                                 color_ = iter - color_options_.begin();
 963                         } else {
 964                                 color_options_.push_back(custom_color_);
 965 
 966                                 color_id_ = custom_color_;
 967                                 color_ = color_options_.size() - 1;
 968                         }
 969                 }
 970         }
 971 
 972         // Initialize ai algorithm.
 973         if(const config& ai = cfg.child("ai")) {
 974                 ai_algorithm_ = ai["ai_algorithm"].str();
 975         }
 976 }

 994 config side_engine::new_config() const
 995 {
 996         config res = cfg_;
 997 
 998         // In case of 'shuffle sides' the side index in cfg_ might be wrong which will confuse the team constuctor later.
 999         res["side"] = index_ + 1;
1000 
1001         // If the user is allowed to change type, faction, leader etc,  then import their new values in the config.
1002         if(!parent_.params_.saved_game) {
1003                 // Merge the faction data to res.
1004                 config faction = flg_.current_faction();
1005                 LOG_MP << "side_engine::new_config: side=" << index_ + 1 << " faction=" << faction["id"] << " recruit=" << faction["recruit"] << "\n";
1006                 res["faction_name"] = faction["name"];
1007                 res["faction"] = faction["id"];
1008                 faction.remove_attributes("id", "name", "image", "gender", "type");
1009                 res.append(faction);
1010         }
1011 
1012         res["controller"] = controller_names[controller_];
1013 
1014         // The hosts receives the serversided controller tweaks after the start event, but
1015         // for mp sync it's very important that the controller types are correct
1016         // during the start/prestart event (otherwise random unit creation during prestart fails).
1017         res["is_local"] = player_id_ == preferences::login() || controller_ == CNTR_COMPUTER || controller_ == CNTR_LOCAL;
1018 
1019         // This function (new_config) is only meant to be called by the host's machine, which is why this check
1020         // works. It essentially certifies that whatever side has the player_id that matches the host's login
1021         // will be flagged. The reason we cannot check mp_campaign_info::is_host is because that flag is *always*
1022         // true on the host's machine, meaning this flag would be set to true for every side.
1023         res["is_host"] = player_id_ == preferences::login();
1024 
1025         std::string desc = user_description();
1026         if(!desc.empty()) {
1027                 res["user_description"] = t_string(desc, "wesnoth");
1028 
1029                 desc = vgettext("$playername $side", {
1030                         {"playername", _(desc.c_str())},
1031                         {"side", res["side"].str()}
1032                 });
1033         } else if(!player_id_.empty()) {
1034                 desc = player_id_;
1035         }
1036 
1037         if(res["name"].str().empty() && !desc.empty()) {
1038                 //TODO: maybe we should add this in to the leaders config instead of the side config?
1039                 res["name"] = desc;
1040         }
1041 
1042         assert(controller_ != CNTR_LAST);
1043         if(controller_ == CNTR_COMPUTER && allow_player_) {
1044                 // Do not import default ai cfg otherwise - all is set by scenario config.
1045                 res.add_child_at("ai", config {"ai_algorithm", ai_algorithm_}, 0);
1046         }
1047 
1048         if(controller_ == CNTR_EMPTY) {
1049                 res["no_leader"] = true;
1050         }
1051 
1052         // A side's "current_player" is the player which has currently taken that side or the one for which it is reserved.
1053         // The "player_id" is the id of the client who controls that side. It's always the host for Local and AI players and
1054         // always empty for free/reserved sides or null controlled sides. You can use !res["player_id"].empty() to check
1055         // whether a side is already taken.
1056         assert(!preferences::login().empty());
1057         if(controller_ == CNTR_LOCAL) {
1058                 res["player_id"] = preferences::login();
1059                 res["current_player"] = preferences::login();
1060         } else if(controller_ == CNTR_RESERVED) {
1061                 res.remove_attribute("player_id");
1062                 res["current_player"] = reserved_for_;
1063         } else if(controller_ == CNTR_COMPUTER) {
1064                 // TODO: what is the content of player_id_ here ?
1065                 res["current_player"] = desc;
1066                 res["player_id"] = preferences::login();
1067         } else if(!player_id_.empty()) {
1068                 res["player_id"] = player_id_;
1069                 res["current_player"] = player_id_;
1070         }
1071 
1072         res["allow_changes"] = allow_changes_;
1073         res["chose_random"] = chose_random_;
1074 
1075         if(!parent_.params_.saved_game) {
1076                 // Find a config where a default leader is and set a new type and gender values for it.
1077                 config* leader = &res;
1078 
1079                 if(flg_.default_leader_cfg() != nullptr) {
1080                         for(config& side_unit : res.child_range("unit")) {
1081                                 if(*flg_.default_leader_cfg() != side_unit) {
1082                                         continue;
1083                                 }
1084 
1085                                 leader = &side_unit;
1086 
1087                                 if(flg_.current_leader() != (*leader)["type"]) {
1088                                         // If a new leader type was selected from carryover, make sure that we reset the leader.
1089                                         std::string leader_id = (*leader)["id"];
1090                                         leader->clear();
1091 
1092                                         if(!leader_id.empty()) {
1093                                                 (*leader)["id"] = leader_id;
1094                                         }
1095                                 }
1096 
1097                                 break;
1098                         }
1099                 }
1100 
1101                 // NOTE: the presence of a type= key overrides no_leader
1102                 if(controller_ != CNTR_EMPTY) {
1103                         (*leader)["type"] = flg_.current_leader();
1104                         (*leader)["gender"] = flg_.current_gender();
1105                         LOG_MP << "side_engine::new_config: side=" << index_ + 1 << " type=" << (*leader)["type"] << " gender=" << (*leader)["gender"] << "\n";
1106                 } else {
1107                         // TODO: FIX THIS SHIT! We shouldn't have a special string to denote no-leader-ness...
1108                         (*leader)["type"] = "null";
1109                         (*leader)["gender"] = "null";
1110                 }
1111 
1112                 res["team_name"] = parent_.team_names_[team_];
1113 
1114                 // TODO: Fix this mess!
1115                 //
1116                 // There is a fundamental disconnect, here. One the one hand we have the idea of
1117                 // 'teams' (which don't actually exist). A 'team' has a name (internal key:
1118                 // team_name) and a translatable display name (internal key: user_team_name). But
1119                 // what we actually have are sides. Sides relate to each other by 'team' (internal
1120                 // key: team_name) and each side has it's own name for the team (internal key:
1121                 // user_team_name).
1122                 //
1123                 // The confusion is that the keys from the side have names which one might expect
1124                 // always refer to the 'team' concept. THEY DO NOT! They are simply named in such
1125                 // a way to confuse the unwary.
1126                 //
1127                 // There is no simple, clean way to clear up the confusion. So, I'm applying the
1128                 // Principle of Least Surprise. The user can see the user_team_name, and it should
1129                 // not change. So if the side already has a user_team_name, use it.
1130                 //
1131                 // In the rare and unlikely (like, probably never happens) case that the side does
1132                 // not have a user_team_name, but an (nebulous and non-deterministic term here)
1133                 // EARLIER side has the same team_name and that side gives a user_team_name, we'll
1134                 // use it.
1135                 //
1136                 // The effect of this mess, and my lame fix for it, is probably only visible when
1137                 // randomizing the sides on a team for multi-player games. But the effect when it's
1138                 // not fixed is an obvious mistake on the player's screen when playing a campaign
1139                 // in single-player mode.
1140                 //
1141                 // At some level, all this is probably wrong, but it is the least breakage from the
1142                 // mess I found; so deal with it, or fix it.
1143                 //
1144                 // If, by now, you get the impression this is a kludged-together mess which cries
1145                 // out for an honest design and a thoughtful implementation, you're correct! But
1146                 // I'm tired, and I'm cranky from wasting a over day on this, and so I'm exercising
1147                 // my prerogative as a grey-beard and leaving this for someone else to clean up.
1148                 if(res["user_team_name"].empty() || !parent_.params_.use_map_settings) {
1149                         res["user_team_name"] = parent_.user_team_names_[team_];
1150                 }
1151 
1152                 res["allow_player"] = allow_player_;
1153                 res["color"] = color_id_;
1154                 res["gold"] = gold_;
1155                 res["income"] = income_;
1156         }
1157 
1158         if(parent_.params_.use_map_settings && !parent_.params_.saved_game) {
1159                 config trimmed = cfg_;
1160 
1161                 for(const std::string& attribute : attributes_to_trim) {
1162                         trimmed.remove_attribute(attribute);
1163                 }
1164 
1165                 if(controller_ != CNTR_COMPUTER) {
1166                         // Only override names for computer controlled players.
1167                         trimmed.remove_attribute("user_description");
1168                 }
1169 
1170                 res.merge_with(trimmed);
1171         }
1172 
1173         return res;
1174 }

1177 {
1178         if(!allow_player_) {
1179                 // Sides without players are always ready.
1180                 return true;

1185                 (controller_ == CNTR_LOCAL)) {
1186

1190         if(available_for_user()) {

1192                 return false;

1302 void side_engine::update_current_controller_index()
1303 {
1304         int i = 0;
1305         for(const controller_option& option : controller_options_) {
1306                 if(option.first == controller_) {
1307                         current_controller_index_ = i;
1308 
1309                         if(player_id_.empty() || player_id_ == option.second) {
1310                                 // Stop searching if no user is assigned to a side
1311                                 // or the selected user is found.
1312                                 break;
1313                         }
1314                 }
1315 
1316                 i++;
1317         }
1318 
1319         assert(current_controller_index_ < controller_options_.size());
1320 }

1322 bool side_engine::controller_changed(const int selection)
1323 {
1324         const ng::controller selected_cntr = controller_options_[selection].first;
1325         if(selected_cntr == CNTR_LAST) {
1326                 return false;
1327         }
1328 
1329         // Check if user was selected. If so assign a side to him/her.
1330         // If not, make sure that no user is assigned to this side.
1331         if(selected_cntr == parent_.default_controller_ && selection != 0) {
1332                 player_id_ = controller_options_[selection].second;
1333                 set_waiting_to_choose_status(false);
1334         } else {
1335                 player_id_.clear();
1336         }
1337 
1338         set_controller(selected_cntr);
1339 
1340         return true;
1341 }
