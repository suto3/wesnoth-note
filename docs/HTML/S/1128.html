  24 /**
  25  * Implements a quit confirmation dialog.
  26  *
  27  * Any object of this type will prevent the game from quitting immediately.
  28  * Instead, a confirmation dialog will pop up when attempting to close.
  29  */
  30 class quit_confirmation
  31 {
  32 public:
  33         explicit quit_confirmation(const std::function<bool()>& prompt = &quit_confirmation::default_prompt)
  34                 : prompt_(prompt) { blockers_.push_back(this); }
  35 
  36         ~quit_confirmation() { blockers_.pop_back(); }
  37 
  38         /**
  39          * Shows the quit confirmation if needed.
  40          *
  41          * @throws CVideo::quit If the user chooses to quit or no prompt was
  42          *                      displayed.
  43          */
  44         static bool quit();
  45         static void quit_to_title();
  46         static void quit_to_desktop();
  47 
  48         static bool show_prompt(const std::string& message);
  49         static bool default_prompt();
  50 
  51 private:
  52         // noncopyable
  53         quit_confirmation(const quit_confirmation&) = delete;
  54         const quit_confirmation& operator=(const quit_confirmation&) = delete;
  55         static std::vector<quit_confirmation*> blockers_;
  56         static bool open_;
  57 
  58         std::function<bool()> prompt_;
  59 };
