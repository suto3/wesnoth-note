  97 void help_text_area::set_items()
  98 {
  99         last_row_.clear();
 100         items_.clear();
 101         curr_loc_.first = 0;
 102         curr_loc_.second = 0;
 103         curr_row_height_ = min_row_height_;
 104         // Add the title item.
 105         const std::string show_title =
 106                 font::make_text_ellipsis(shown_topic_->title, title_size, inner_location().w);
 107         surface surf(font::get_rendered_text(show_title, title_size,
 108                                              font::NORMAL_COLOR, TTF_STYLE_BOLD));
 109         if (surf != nullptr) {
 110                 add_item(item(surf, 0, 0, show_title));
 111                 curr_loc_.second = title_spacing_;
 112                 contents_height_ = title_spacing_;
 113                 down_one_line();
 114         }
 115         // Parse and add the text.
 116         std::vector<std::string> const &parsed_items = shown_topic_->text.parsed_text();
 117         std::vector<std::string>::const_iterator it;
 118         for (it = parsed_items.begin(); it != parsed_items.end(); ++it) {
 119                 if (*it != "" && (*it)[0] == '[') {
 120                         // Should be parsed as WML.
 121                         try {
 122                                 config cfg;
 123                                 std::istringstream stream(*it);
 124                                 read(cfg, stream);
 125 
 126 #define TRY(name) do { \
 127                                 if (config &child = cfg.child(#name)) \
 128                                         handle_##name##_cfg(child); \
 129                                 } while (0)
 130 
 131                                 TRY(ref);
 132                                 TRY(img);
 133                                 TRY(bold);
 134                                 TRY(italic);
 135                                 TRY(header);
 136                                 TRY(jump);
 137                                 TRY(format);
 138 
 139 #undef TRY
 140 
 141                         }
 142                         catch (config::error& e) {
 143                                 std::stringstream msg;
 144                                 msg << "Error when parsing help markup as WML: '" << e.message << "'";
 145                                 throw parse_error(msg.str());
 146                         }
 147                 }
 148                 else {
 149                         add_text_item(*it);
 150                 }
 151         }
 152         down_one_line(); // End the last line.
 153         int h = height();
 154         set_position(0);
 155         set_full_size(contents_height_);
 156         set_shown_size(h);
 157 }

 159 void help_text_area::handle_ref_cfg(const config &cfg)
 160 {
 161         const std::string dst = cfg["dst"];
 162         const std::string text = cfg["text"];
 163         bool force = cfg["force"].to_bool();
 164 
 165         if (dst == "") {
 166                 std::stringstream msg;
 167                 msg << "Ref markup must have dst attribute. Please submit a bug"
 168                        " report if you have not modified the game files yourself. Erroneous config: ";
 169                 write(msg, cfg);
 170                 throw parse_error(msg.str());
 171         }
 172 
 173         if (find_topic(toplevel_, dst) == nullptr && !force) {
 174                 // detect the broken link but quietly silence the hyperlink for normal user
 175                 add_text_item(text, game_config::debug ? dst : "", true);
 176 
 177                 // FIXME: workaround: if different campaigns define different
 178                 // terrains, some terrains available in one campaign will
 179                 // appear in the list of seen terrains, and be displayed in the
 180                 // help, even if the current campaign does not handle such
 181                 // terrains. This will lead to the unit page generator creating
 182                 // invalid references.
 183                 //
 184                 // Disabling this is a kludgey workaround until the
 185                 // encountered_terrains system is fixed
 186                 //
 187                 // -- Ayin apr 8 2005
 188 #if 0
 189                 if (game_config::debug) {
 190                         std::stringstream msg;
 191                         msg << "Reference to non-existent topic '" << dst
 192                             << "'. Please submit a bug report if you have not"
 193                                "modified the game files yourself. Erroneous config: ";
 194                         write(msg, cfg);
 195                         throw parse_error(msg.str());
 196                 }
 197 #endif
 198         } else {
 199                 add_text_item(text, dst);
 200         }
 201 }

 242 void help_text_area::handle_jump_cfg(const config &cfg)
 243 {
 244         const std::string amount_str = cfg["amount"];
 245         const std::string to_str = cfg["to"];
 246         if (amount_str == "" && to_str == "") {
 247                 throw parse_error("Jump markup must have either a to or an amount attribute.");
 248         }
 249         unsigned jump_to = curr_loc_.first;
 250         if (amount_str != "") {
 251                 unsigned amount;
 252                 try {
 253                         amount = lexical_cast<unsigned, std::string>(amount_str);
 254                 }
 255                 catch (bad_lexical_cast) {
 256                         throw parse_error("Invalid amount the amount attribute in jump markup.");
 257                 }
 258                 jump_to += amount;
 259         }
 260         if (to_str != "") {
 261                 unsigned to;
 262                 try {
 263                         to = lexical_cast<unsigned, std::string>(to_str);
 264                 }
 265                 catch (bad_lexical_cast) {
 266                         throw parse_error("Invalid amount in the to attribute in jump markup.");
 267                 }
 268                 if (to < jump_to) {
 269                         down_one_line();
 270                 }
 271                 jump_to = to;
 272         }
 273         if (jump_to != 0 && static_cast<int>(jump_to) <
 274             get_max_x(curr_loc_.first, curr_row_height_)) {
 275 
 276                 curr_loc_.first = jump_to;
 277         }
 278 }

 293 void help_text_area::add_text_item(const std::string& text, const std::string& ref_dst,
 294                                                                    bool broken_link, int _font_size, bool bold, bool italic,
 295                                                                    color_t text_color
 296 )
 297 {
 298         const int font_size = _font_size < 0 ? normal_font_size : _font_size;
 299         // font::line_width(), font::get_rendered_text() are not use scaled font inside
 300         const int scaled_font_size = preferences::font_scaled(font_size);
 301         if (text.empty())
 302                 return;
 303         const int remaining_width = get_remaining_width();
 304         size_t first_word_start = text.find_first_not_of(" ");
 305         if (first_word_start == std::string::npos) {
 306                 first_word_start = 0;
 307         }
 308         if (text[first_word_start] == '\n') {
 309                 down_one_line();
 310                 std::string rest_text = text;
 311                 rest_text.erase(0, first_word_start + 1);
 312                 add_text_item(rest_text, ref_dst, broken_link, _font_size, bold, italic, text_color);
 313                 return;
 314         }
 315         const std::string first_word = get_first_word(text);
 316         int state = 0;
 317         state |= bold ? TTF_STYLE_BOLD : 0;
 318         state |= italic ? TTF_STYLE_ITALIC : 0;
 319         if (curr_loc_.first != get_min_x(curr_loc_.second, curr_row_height_)
 320                 && remaining_width < font::line_width(first_word, scaled_font_size, state)) {
 321                 // The first word does not fit, and we are not at the start of
 322                 // the line. Move down.
 323                 down_one_line();
 324                 std::string s = remove_first_space(text);
 325                 add_text_item(s, ref_dst, broken_link, _font_size, bold, italic, text_color);
 326         }
 327         else {
 328                 std::vector<std::string> parts = split_in_width(text, font_size, remaining_width);
 329                 std::string first_part = parts.front();
 330                 // Always override the color if we have a cross reference.
 331                 color_t color;
 332                 if(ref_dst.empty())
 333                         color = text_color;
 334                 else if(broken_link)
 335                         color = font::BAD_COLOR;
 336                 else
 337                         color = font::YELLOW_COLOR;
 338 
 339                 // In split_in_width(), no_break_after() and no_break_before() are used(see marked-up_text.cpp).
 340                 // Thus, even if there is enough remaining_width for the next word,
 341                 // sometimes empty string is returned from split_in_width().
 342                 if (first_part.empty()) {
 343                         down_one_line();
 344                 }
 345                 else {
 346                         surface surf(font::get_rendered_text(first_part, scaled_font_size, color, state));
 347                         if (!surf.null())
 348                                 add_item(item(surf, curr_loc_.first, curr_loc_.second, first_part, ref_dst));
 349                 }
 350                 if (parts.size() > 1) {
 351 
 352                         std::string& s = parts.back();
 353 
 354                         const std::string first_word_before = get_first_word(s);
 355                         const std::string first_word_after = get_first_word(remove_first_space(s));
 356                         if (get_remaining_width() >= font::line_width(first_word_after, scaled_font_size, state)
 357                                 && get_remaining_width()
 358                                 < font::line_width(first_word_before, scaled_font_size, state)) {
 359                                 // If the removal of the space made this word fit, we
 360                                 // must move down a line, otherwise it will be drawn
 361                                 // without a space at the end of the line.
 362                                 s = remove_first_space(s);
 363                                 down_one_line();
 364                         }
 365                         else if (!(font::line_width(first_word_before, scaled_font_size, state)
 366                                            < get_remaining_width())) {
 367                                 s = remove_first_space(s);
 368                         }
 369                         add_text_item(s, ref_dst, broken_link, _font_size, bold, italic, text_color);
 370 
 371                 }
 372         }
 373 }

 375 void help_text_area::add_img_item(const std::string& path, const std::string& alignment,
 376                                                                   const bool floating, const bool box)
 377 {
 378         surface surf(image::get_image(path));
 379         if (surf.null())
 380                 return;
 381         ALIGNMENT align = str_to_align(alignment);
 382         if (align == HERE && floating) {
 383                 WRN_DP << "Floating image with align HERE, aligning left." << std::endl;
 384                 align = LEFT;
 385         }
 386         const int width = surf->w + (box ? box_width * 2 : 0);
 387         int xpos;
 388         int ypos = curr_loc_.second;
 389         int text_width = inner_location().w;
 390         switch (align) {
 391         case HERE:
 392                 xpos = curr_loc_.first;
 393                 break;
 394         case LEFT:
 395         default:
 396                 xpos = 0;
 397                 break;
 398         case MIDDLE:
 399                 xpos = text_width / 2 - width / 2 - (box ? box_width : 0);
 400                 break;
 401         case RIGHT:
 402                 xpos = text_width - width - (box ? box_width * 2 : 0);
 403                 break;
 404         }
 405         if (curr_loc_.first != get_min_x(curr_loc_.second, curr_row_height_)
 406                 && (xpos < curr_loc_.first || xpos + width > text_width)) {
 407                 down_one_line();
 408                 add_img_item(path, alignment, floating, box);
 409         }
 410         else {
 411                 if (!floating) {
 412                         curr_loc_.first = xpos;
 413                 }
 414                 else {
 415                         ypos = get_y_for_floating_img(width, xpos, ypos);
 416                 }
 417                 add_item(item(surf, xpos, ypos, floating, box, align));
 418         }
 419 }

 421 int help_text_area::get_y_for_floating_img(const int width, const int x, const int desired_y)
 422 {
 423         int min_y = desired_y;
 424         for (std::list<item>::const_iterator it = items_.begin(); it != items_.end(); ++it) {
 425                 const item& itm = *it;
 426                 if (itm.floating) {
 427                         if ((itm.rect.x + itm.rect.w > x && itm.rect.x < x + width)
 428                                 || (itm.rect.x > x && itm.rect.x < x + width)) {
 429                                 min_y = std::max<int>(min_y, itm.rect.y + itm.rect.h);
 430                         }
 431                 }
 432         }
 433         return min_y;
 434 }

 436 int help_text_area::get_min_x(const int y, const int height)
 437 {
 438         int min_x = 0;
 439         for (std::list<item>::const_iterator it = items_.begin(); it != items_.end(); ++it) {
 440                 const item& itm = *it;
 441                 if (itm.floating) {
 442                         if (itm.rect.y < y + height && itm.rect.y + itm.rect.h > y && itm.align == LEFT) {
 443                                 min_x = std::max<int>(min_x, itm.rect.w + 5);
 444                         }
 445                 }
 446         }
 447         return min_x;
 448 }

 450 int help_text_area::get_max_x(const int y, const int height)
 451 {
 452         int text_width = inner_location().w;
 453         int max_x = text_width;
 454         for (std::list<item>::const_iterator it = items_.begin(); it != items_.end(); ++it) {
 455                 const item& itm = *it;
 456                 if (itm.floating) {
 457                         if (itm.rect.y < y + height && itm.rect.y + itm.rect.h > y) {
 458                                 if (itm.align == RIGHT) {
 459                                         max_x = std::min<int>(max_x, text_width - itm.rect.w - 5);
 460                                 } else if (itm.align == MIDDLE) {
 461                                         max_x = std::min<int>(max_x, text_width / 2 - itm.rect.w / 2 - 5);
 462                                 }
 463                         }
 464                 }
 465         }
 466         return max_x;
 467 }

 469 void help_text_area::add_item(const item &itm)
 470 {
 471         items_.push_back(itm);
 472         if (!itm.floating) {
 473                 curr_loc_.first += itm.rect.w;
 474                 curr_row_height_ = std::max<int>(itm.rect.h, curr_row_height_);
 475                 contents_height_ = std::max<int>(contents_height_, curr_loc_.second + curr_row_height_);
 476                 last_row_.push_back(&items_.back());
 477         }
 478         else {
 479                 if (itm.align == LEFT) {
 480                         curr_loc_.first = itm.rect.w + 5;
 481                 }
 482                 contents_height_ = std::max<int>(contents_height_, itm.rect.y + itm.rect.h);
 483         }
 484 }

 487 help_text_area::ALIGNMENT help_text_area::str_to_align(const std::string &cmp_str)
 488 {
 489         if (cmp_str == "left") {
 490                 return LEFT;
 491         } else if (cmp_str == "middle") {
 492                 return MIDDLE;
 493         } else if (cmp_str == "right") {
 494                 return RIGHT;
 495         } else if (cmp_str == "here" || cmp_str == "") { // Make the empty string be "here" alignment.
 496                 return HERE;
 497         }
 498         std::stringstream msg;
 499         msg << "Invalid alignment string: '" << cmp_str << "'";
 500         throw parse_error(msg.str());
 501 }

 503 void help_text_area::down_one_line()
 504 {
 505         adjust_last_row();
 506         last_row_.clear();
 507         curr_loc_.second += curr_row_height_ + (curr_row_height_ == min_row_height_ ? 0 : 2);
 508         curr_row_height_ = min_row_height_;
 509         contents_height_ = std::max<int>(curr_loc_.second + curr_row_height_, contents_height_);
 510         curr_loc_.first = get_min_x(curr_loc_.second, curr_row_height_);
 511 }
 512 
 513 void help_text_area::adjust_last_row()
 514 {
 515         for (std::list<item *>::iterator it = last_row_.begin(); it != last_row_.end(); ++it) {
 516                 item &itm = *(*it);
 517                 const int gap = curr_row_height_ - itm.rect.h;
 518                 itm.rect.y += gap / 2;
 519         }
 520 }
 521 
 522 int help_text_area::get_remaining_width()
 523 {
 524         const int total_w = get_max_x(curr_loc_.second, curr_row_height_);
 525         return total_w - curr_loc_.first;
 526 }

 528 void help_text_area::draw_contents()
 529 {
 530         SDL_Rect const &loc = inner_location();
 531         bg_restore();
 532         surface& screen = video().getSurface();
 533         clip_rect_setter clip_rect_set(screen, &loc);
 534         for(std::list<item>::const_iterator it = items_.begin(), end = items_.end(); it != end; ++it) {
 535                 SDL_Rect dst = it->rect;
 536                 dst.y -= get_position();
 537                 if (dst.y < static_cast<int>(loc.h) && dst.y + it->rect.h > 0) {
 538                         dst.x += loc.x;
 539                         dst.y += loc.y;
 540                         if (it->box) {
 541                                 for (int i = 0; i < box_width; ++i) {
 542                                         SDL_Rect draw_rect {
 543                                                 dst.x,
 544                                                 dst.y,
 545                                                 it->rect.w - i * 2,
 546                                                 it->rect.h - i * 2
 547                                         };
 548         
 549                                         sdl::draw_rectangle(draw_rect, {0, 0, 0, 0});
 550                                         ++dst.x;
 551                                         ++dst.y;
 552                                 }
 553                         }
 554                         sdl_blit(it->surf, nullptr, screen, &dst);
 555                 }
 556         }
 557 }

 571 std::string help_text_area::ref_at(const int x, const int y)
 572 {
 573         const int local_x = x - location().x;
 574         const int local_y = y - location().y;
 575         if (local_y < height() && local_y > 0) {
 576                 const int cmp_y = local_y + get_position();
 577                 const std::list<item>::const_iterator it =
 578                         std::find_if(items_.begin(), items_.end(), item_at(local_x, cmp_y));
 579                 if (it != items_.end()) {
 580                         if ((*it).ref_to != "") {
 581                                 return ((*it).ref_to);
 582                         }
 583                 }
 584         }
 585         return "";
 586 }
