  53 // get filename associated to this code
  54 static std::string get_filename(const std::string& file_code)
  55 {
  56         if(!encode_filename) {
  57                 return file_code;
  58         }
  59 
  60         std::stringstream s;
  61         s << file_code;
  62         int n = 0;
  63         s >> std::hex >> n;
  64 
  65         for(const auto& p : file_number_map) {

 379 /**
 380  * Called by an STL stream whenever it has reached the end of #out_buffer_.
 381  * Fills #buffer_ by calling the #current_ preprocessor, then copies its
 382  * content into #out_buffer_.
 383  * @return the first character of #out_buffer_ if any, EOF otherwise.
 384  */
 385 int preprocessor_streambuf::underflow()
 386 {
 387         unsigned sz = 0;
 388         if(char* gp = gptr()) {
 389                 if(gp < egptr()) {
 390                         // Sanity check: the internal buffer has not been totally consumed,
 391                         // should we force the caller to use what remains first?
 392                         return *gp;
 393                 }
 394 
 395                 // The buffer has been completely read; fill it again.
 396                 // Keep part of the previous buffer, to ensure putback capabilities.
 397                 sz = out_buffer_.size();
 398                 buffer_.str(std::string());
 399 
 400                 if(sz > 3) {
 401                         buffer_ << out_buffer_.substr(sz - 3);
 402                         sz = 3;
 403                 } else {
 404                         buffer_ << out_buffer_;
 405                 }
 406         } else {
 407                 // The internal get-data pointer is null
 408         }
 409 
 410         const int desired_fill_amount = 2000;
 411 
 412         while(current() && buffer_.rdbuf()->in_avail() < desired_fill_amount) {
 413                 // Process files and data chunks until the desired buffer size is reached
 414                 if(!current()->get_chunk()) {
 415                         // Drop the current preprocessor item from the queue.
 416                         restore_old_preprocessor();
 417                 }
 418         }
 419 
 420         // Update the internal state and data pointers
 421         out_buffer_ = buffer_.str();
 422         if(out_buffer_.empty()) {
 423                 return EOF;
 424         }
 425 
 426         char* begin = &*out_buffer_.begin();
 427         unsigned bs = out_buffer_.size();
 428 
 429         setg(begin, begin + sz, begin + bs);
 430 
 431         if(sz >= bs) {
 432                 return EOF;
 433         }
 434 
 435         return static_cast<unsigned char>(*(begin + sz));
 436 }

 438 /**
 439 * Restores the old preprocessing context.
 440 * Appends location and domain directives to the buffer, so that the parser
 441 * notices these changes.
 442 */
 443 void preprocessor_streambuf::restore_old_preprocessor()
 444 {
 445         preprocessor* current = this->current();
 446 
 447         if(!current->old_location_.empty()) {
 448                 buffer_ << OUTPUT_SEPARATOR << "line " << current->old_linenum_ << ' ' << current->old_location_ << '\n';
 449         }
 450 
 451         if(!current->old_textdomain_.empty() && textdomain_ != current->old_textdomain_) {
 452                 buffer_ << OUTPUT_SEPARATOR << "textdomain " << current->old_textdomain_ << '\n';
 453         }
 454 
 455         location_ = current->old_location_;
 456         linenum_ = current->old_linenum_;
 457         textdomain_ = current->old_textdomain_;
 458 
 459         // Drop the preprocessor from the queue.
 460         drop_preprocessor();
 461 }

 463 std::string preprocessor_streambuf::get_current_file()
 464 {
 465         unsigned nested_level = 0;
 466 
 467         preprocessor* pre = nullptr;
 468 
 469         // Iterate backwards over queue to get the last non-macro preprocessor.
 470         for(auto p = preprocessor_queue_.rbegin(); p != preprocessor_queue_.rend(); ++p) {
 471                 pre = p->get();
 472 
 473                 if(!pre || pre->parse_mode() == preprocessor::PARSES_FILE) {
 474                         break;
 475                 }
 476 
 477                 if(pre->parse_mode() == preprocessor::PARSES_MACRO) {
 478                         ++nested_level;
 479                 }
 480         }
 481 
 482         std::string res;
 483         std::vector<std::string> pos = utils::quoted_split(location_, ' ');
 484 
 485         if(pos.size() <= 2 * nested_level) {
 486                 return res;
 487         }
 488 
 489         return get_filename(pos[2 * nested_level]);
 490 }

 492 std::string lineno_string(const std::string& lineno)
 493 {
 494         std::vector<std::string> pos = utils::quoted_split(lineno, ' ');
 495         std::vector<std::string>::const_iterator i = pos.begin(), end = pos.end();
 496         std::string included_from = preprocessor_error_detail_prefix + "included from ";
 497         std::string res;
 498 
 499         while(i != end) {
 500                 const std::string& line = *(i++);
 501 
 502                 if(!res.empty()) {
 503                         res += included_from;
 504                 }
 505 
 506                 if(i != end) {
 507                         res += get_filename(*(i++));
 508                 } else {
 509                         res += "<unknown>";
 510                 }
 511 
 512                 res += ':' + line;
 513         }
 514 
 515         if(res.empty()) {
 516                 res = "???";
 517         }
 518 
 519         return res;
 520 }

 842 void preprocessor_data::push_token(token_desc::TOKEN_TYPE t)
 843 {
 844         tokens_.emplace_back(t, strings_.size(), linenum_);
 845 
 846         if(t == token_desc::MACRO_SPACE) {
 847                 // Macro expansions do not have any associated storage at start.
 848                 return;
 849         } else if(t == token_desc::STRING || t == token_desc::VERBATIM) {
 850                 /* Quoted strings are always inlined in the parent token. So
 851                  * they need neither storage nor metadata, unless the parent
 852                  * token is a macro expansion.
 853                  */
 854                 token_desc::TOKEN_TYPE& outer_type = tokens_[tokens_.size() - 2].type;
 855                 if(outer_type != token_desc::MACRO_SPACE) {
 856                         return;
 857                 }
 858 
 859                 outer_type = token_desc::MACRO_CHUNK;
 860                 tokens_.back().stack_pos = strings_.size() + 1;
 861         }
 862 
 863         std::ostringstream s;
 864         if(!skipping_ && slowpath_) {
 865                 s << OUTPUT_SEPARATOR << "line " << linenum_ << ' ' << parent_.location_ << "\n"
 866                   << OUTPUT_SEPARATOR << "textdomain " << parent_.textdomain_ << '\n';
 867         }
 868 
 869         strings_.push_back(s.str());
 870 }

 872 void preprocessor_data::pop_token()
 873 {
 874         token_desc::TOKEN_TYPE inner_type = tokens_.back().type;
 875         unsigned stack_pos = tokens_.back().stack_pos;
 876 
 877         tokens_.pop_back();
 878 
 879         token_desc::TOKEN_TYPE& outer_type = tokens_.back().type;
 880 
 881         if(inner_type == token_desc::MACRO_PARENS) {
 882                 // Parenthesized macro arguments are left on the stack.
 883                 assert(outer_type == token_desc::MACRO_SPACE);
 884                 return;
 885         }
 886 
 887         if(inner_type == token_desc::STRING || inner_type == token_desc::VERBATIM) {
 888                 // Quoted strings are always inlined.
 889                 assert(stack_pos == strings_.size());
 890                 return;
 891         }
 892 
 893         if(outer_type == token_desc::MACRO_SPACE) {
 894                 /* A macro expansion does not have any associated storage.
 895                  * Instead, storage of the inner token is not discarded
 896                  * but kept as a new macro argument. But if the inner token
 897                  * was a macro expansion, it is about to be appended, so
 898                  * prepare for it.
 899                  */
 900                 if(inner_type == token_desc::MACRO_SPACE || inner_type == token_desc::MACRO_CHUNK) {
 901                         strings_.erase(strings_.begin() + stack_pos, strings_.end());
 902                         strings_.emplace_back();
 903                 }
 904 
 905                 assert(stack_pos + 1 == strings_.size());
 906                 outer_type = token_desc::MACRO_CHUNK;
 907 
 908                 return;
 909         }
 910 
 911         strings_.erase(strings_.begin() + stack_pos, strings_.end());
 912 }

1051 bool preprocessor_data::get_chunk()
1052 {
1053         char c = static_cast<char>(in_.get());
1054         token_desc& token = tokens_.back();
1055 
1056         if(in_.eof()) {
1057                 // The end of file was reached.
1058                 // Make sure we don't have any incomplete tokens.
1059                 char const* s;
1060 
1061                 switch(token.type) {
1062                 case token_desc::START:
1063                         return false; // everything is fine
1064                 case token_desc::PROCESS_IF:
1065                 case token_desc::SKIP_IF:
1066                 case token_desc::PROCESS_ELSE:
1067                 case token_desc::SKIP_ELSE:
1068                         s = "#ifdef or #ifndef";
1069                         break;
1070                 case token_desc::STRING:
1071                         s = "Quoted string";
1072                         break;
1073                 case token_desc::VERBATIM:
1074                         s = "Verbatim string";
1075                         break;
1076                 case token_desc::MACRO_CHUNK:
1077                 case token_desc::MACRO_SPACE:
1078                         s = "Macro substitution";
1079                         break;
1080                 case token_desc::MACRO_PARENS:
1081                         s = "Macro argument";
1082                         break;
1083                 default:
1084                         s = "???";
1085                 }
1086 
1087                 parent_.error(std::string(s) + " not terminated", token.linenum);
1088         }
1089 
1090         if(c == '\n') {
1091                 ++linenum_;
1092         }
1093 
1094         if(c == OUTPUT_SEPARATOR) {
1095                 std::string buffer(1, c);
1096 
1097                 for(;;) {
1098                         char d = static_cast<char>(in_.get());
1099 
1100                         if(in_.eof() || d == '\n') {
1101                                 break;
1102                         }
1103 
1104                         buffer += d;
1105                 }
1106 
1107                 buffer += '\n';
1108                 // line_change = 1-1 = 0
1109                 put(buffer);
1110         } else if(token.type == token_desc::VERBATIM) {
1111                 put(c);
1112 
1113                 if(c == '>' && in_.peek() == '>') {
1114                         put(in_.get());
1115                         pop_token();
1116                 }
1117         } else if(c == '<' && in_.peek() == '<') {
1118                 in_.get();
1119                 push_token(token_desc::VERBATIM);
1120                 put('<');
1121                 put('<');
1122         } else if(c == '"') {
1123                 if(token.type == token_desc::STRING) {
1124                         parent_.quoted_ = false;
1125                         put(c);
1126                         pop_token();
1127                 } else if(!parent_.quoted_) {
1128                         parent_.quoted_ = true;
1129                         push_token(token_desc::STRING);
1130                         put(c);
1131                 } else {
1132                         parent_.error("Nested quoted string", linenum_);
1133                 }
1134         } else if(c == '{') {
1135                 push_token(token_desc::MACRO_SPACE);
1136                 ++slowpath_;
1137         } else if(c == ')' && token.type == token_desc::MACRO_PARENS) {
1138                 pop_token();
1139         } else if(c == '#' && !parent_.quoted_) {
1140                 std::string command = read_word();
1141                 bool comment = false;
1142 
1143                 if(command == "define") {
1144                         skip_spaces();
1145                         int linenum = linenum_;
1146                         std::vector<std::string> items = utils::split(read_line(), ' ');
1147                         std::map<std::string, std::string> optargs;
1148 
1149                         if(items.empty()) {
1150                                 parent_.error("No macro name found after #define directive", linenum);
1151                         }
1152 
1153                         std::string symbol = items.front();
1154                         items.erase(items.begin());
1155                         int found_arg = 0, found_enddef = 0;
1156                         std::string buffer;
1157                         for(;;) {
1158                                 if(in_.eof())
1159                                         break;
1160                                 char d = static_cast<char>(in_.get());
1161                                 if(d == '\n')
1162                                         ++linenum_;
1163                                 buffer += d;
1164                                 if(d == '#') {
1165                                         if(in_.peek() == 'a') {
1166                                                 found_arg = 1;
1167                                         } else {
1168                                                 found_enddef = 1;
1169                                         }
1170                                 } else {
1171                                         if(found_arg > 0 && ++found_arg == 4) {
1172                                                 if(std::equal(buffer.end() - 3, buffer.end(), "arg")) {
1173                                                         buffer.erase(buffer.end() - 4, buffer.end());
1174 
1175                                                         skip_spaces();
1176                                                         std::string argname = read_word();
1177                                                         skip_eol();
1178 
1179                                                         std::string argbuffer;
1180 
1181                                                         int found_endarg = 0;
1182                                                         for(;;) {
1183                                                                 if(in_.eof()) {
1184                                                                         break;
1185                                                                 }
1186 
1187                                                                 char e = static_cast<char>(in_.get());
1188                                                                 if(e == '\n') {
1189                                                                         ++linenum_;

1192                                                                 argbuffer += e;

1410                         while((pos = symbol.find(OUTPUT_SEPARATOR)) != std::string::npos) {
1411                                 std::string::iterator b = symbol.begin(); // invalidated at each iteration
1412                                 symbol.erase(b + pos, b + symbol.find('\n', pos + 1) + 1);
1413                         }
1414 
1415                         std::map<std::string, std::string>::const_iterator arg;
1416                         preproc_map::const_iterator macro;
1417 
1418                         // If this is a known pre-processing symbol, then we insert it,
1419                         // otherwise we assume it's a file name to load.
1420                         if(symbol == current_file_str && strings_.size() - token.stack_pos == 1) {
1421                                 pop_token();
1422                                 put(parent_.get_current_file());
1423                         } else if(symbol == current_dir_str && strings_.size() - token.stack_pos == 1) {
1424                                 pop_token();
1425                                 put(filesystem::directory_name(parent_.get_current_file()));
1426                         } else if(local_defines_ && (arg = local_defines_->find(symbol)) != local_defines_->end()) {
1427                                 if(strings_.size() - token.stack_pos != 1) {
1428                                         std::ostringstream error;
1429                                         error << "Macro argument '" << symbol << "' does not expect any arguments";
1430                                         parent_.error(error.str(), linenum_);
1431                                 }
1432 
1433                                 std::ostringstream v;
1434                                 v << arg->second << OUTPUT_SEPARATOR << "line " << linenum_ << ' ' << parent_.location_ << "\n"
1435                                   << OUTPUT_SEPARATOR << "textdomain " << parent_.textdomain_ << '\n';
1436 
1437                                 pop_token();
1438                                 put(v.str());
1439                         } else if(parent_.depth() < 100 && (macro = parent_.defines_->find(symbol)) != parent_.defines_->end()) {
1440                                 const preproc_define& val = macro->second;
1441                                 size_t nb_arg = strings_.size() - token.stack_pos - 1;
1442                                 size_t optional_arg_num = 0;
1443 
1444                                 std::map<std::string, std::string>* defines = new std::map<std::string, std::string>;
1445                                 const std::string& dir = filesystem::directory_name(val.location.substr(0, val.location.find(' ')));
1446 
1447                                 for(size_t i = 0; i < nb_arg; ++i) {
1448                                         if(i < val.arguments.size()) {
1449                                                 // Normal mandatory arguments
1450 
1451                                                 (*defines)[val.arguments[i]] = strings_[token.stack_pos + i + 1];
1452                                         } else {
1453                                                 // These should be optional argument overrides
1454 
1455                                                 std::string str = strings_[token.stack_pos + i + 1];
1456                                                 size_t equals_pos = str.find_first_of("=");
1457 
1458                                                 if(equals_pos != std::string::npos) {
1459                                                         size_t argname_pos = str.substr(0, equals_pos).find_last_of(" \n") + 1;
1460 
1461                                                         std::string argname = str.substr(argname_pos, equals_pos - argname_pos);
1462 
1463                                                         if(val.optional_arguments.find(argname) != val.optional_arguments.end()) {
1464                                                                 (*defines)[argname] = str.substr(equals_pos + 1);
1465 
1466                                                                 optional_arg_num++;
1467 
1468                                                                 DBG_PREPROC << "Found override for " << argname << " in call to macro " << symbol
1469                                                                                         << "\n";
1470                                                         } else {
1471                                                                 std::ostringstream warning;
1472                                                                 warning << "Unrecognized optional argument passed to macro '" << symbol << "': '"
1473                                                                                 << argname << "'";
1474                                                                 parent_.warning(warning.str(), linenum_);
1475 
1476                                                                 optional_arg_num++; // To prevent the argument number check from blowing up
1477                                                         }
1478                                                 }
1479                                         }
1480                                 }
1481 
1482                                 // If the macro definition has any optional arguments, insert their defaults
1483                                 if(val.optional_arguments.size() > 0) {
1484                                         for(const auto& argument : val.optional_arguments) {
1485                                                 if(defines->find(argument.first) == defines->end()) {
1486                                                         std::unique_ptr<preprocessor_streambuf> buf(new preprocessor_streambuf(parent_));
1487 
1488                                                         buf->textdomain_ = parent_.textdomain_;
1489                                                         std::istream in(buf.get());
1490 
1491                                                         std::istringstream* buffer = new std::istringstream(argument.second);
1492 
1493                                                         std::map<std::string, std::string>* temp_defines = new std::map<std::string, std::string>;
1494                                                         temp_defines->insert(defines->begin(), defines->end());
1495 
1496                                                         buf->add_preprocessor<preprocessor_data>(
1497                                                                 buffer, val.location, "", val.linenum, dir, val.textdomain, temp_defines, false);
1498 
1499                                                         std::ostringstream res;
1500                                                         res << in.rdbuf();
1501 
1502                                                         DBG_PREPROC << "Setting default for optional argument " << argument.first << " in macro "
1503                                                                                 << symbol << "\n";
1504 
1505                                                         (*defines)[argument.first] = res.str();
1506                                                 }
1507                                         }
1508                                 }
1509 
1510                                 if(nb_arg - optional_arg_num != val.arguments.size()) {
1511                                         const std::vector<std::string>& locations = utils::quoted_split(val.location, ' ');
1512                                         std::ostringstream error;
1513                                         error << "Preprocessor symbol '" << symbol << "' defined at " << get_filename(locations[0]) << ":"
1514                                                   << val.linenum << " expects " << val.arguments.size() << " arguments, but has "
1515                                                   << nb_arg - optional_arg_num << " arguments";
1516                                         parent_.error(error.str(), linenum_);
1517                                 }
1518 
1519                                 std::istringstream* buffer = new std::istringstream(val.value);
1520 
1521                                 pop_token();
1522 
1523                                 if(!slowpath_) {
1524                                         DBG_PREPROC << "substituting macro " << symbol << '\n';
1525 
1526                                         parent_.add_preprocessor<preprocessor_data>(
1527                                                 buffer, val.location, "", val.linenum, dir, val.textdomain, defines, true);
1528                                 } else {
1529                                         DBG_PREPROC << "substituting (slow) macro " << symbol << '\n';
1530 
1531                                         std::unique_ptr<preprocessor_streambuf> buf(new preprocessor_streambuf(parent_));
1532 
1533                                         // Make the nested preprocessor_data responsible for
1534                                         // restoring our current textdomain if needed.
1535                                         buf->textdomain_ = parent_.textdomain_;
1536 
1537                                         std::ostringstream res;
1538                                         {
1539                                                 std::istream in(buf.get());
1540                                                 buf->add_preprocessor<preprocessor_data>(
1541                                                         buffer, val.location, "", val.linenum, dir, val.textdomain, defines, true);
1542 
1543                                                 res << in.rdbuf();
1544                                         }
1545 
1546                                         put(res.str());
1547                                 }
1548                         } else if(parent_.depth() < 40) {
1549                                 LOG_PREPROC << "Macro definition not found for " << symbol << " , attempting to open as file.\n";
1550                                 pop_token();
1551 
1552                                 std::string nfname = filesystem::get_wml_location(symbol, directory_);
1553                                 if(!nfname.empty()) {
1554                                         if(!slowpath_)
1555                                                 // nfname.size() - symbol.size() gives you an index into nfname
1556                                                 // This does not necessarily match the symbol though, as it can start with ~ or ./
1557                                                 parent_.add_preprocessor<preprocessor_file>(nfname, nfname.size() - symbol.size());
1558                                         else {
1559                                                 std::unique_ptr<preprocessor_streambuf> buf(new preprocessor_streambuf(parent_));
1560 
1561                                                 std::ostringstream res;
1562                                                 {
1563                                                         std::istream in(buf.get());
1564                                                         buf->add_preprocessor<preprocessor_file>(nfname, nfname.size() - symbol.size());
1565 
1566                                                         res << in.rdbuf();
1567                                                 }
1568 
1569                                                 put(res.str());
1570                                         }
1571                                 } else {
1572                                         std::ostringstream error;
1573                                         error << "Macro/file '" << symbol << "' is missing";
1574                                         parent_.error(error.str(), linenum_);
1575                                 }
1576                         } else {
1577                                 parent_.error("Too many nested preprocessing inclusions", linenum_);
1578                         }
1579                 } else if(!skipping_) {
1580                         if(token.type == token_desc::MACRO_SPACE) {
1581                                 std::ostringstream s;
1582                                 s << OUTPUT_SEPARATOR << "line " << linenum_ << ' ' << parent_.location_ << "\n"
1583                                   << OUTPUT_SEPARATOR << "textdomain " << parent_.textdomain_ << '\n';
1584 
1585                                 strings_.push_back(s.str());
1586                                 token.type = token_desc::MACRO_CHUNK;
1587                         }
1588                         put(c);
1589                 }
1590         } else {
1591                 put(c);
1592         }
1593 
1594         return true;
1595 }

1603 {
1604         preprocessor_scope_helper(const std::string& fname, preproc_map* defines)
1605                 : std::basic_istream<char>(nullptr)
1606                 , buf_(nullptr)
1607                 , local_defines_(nullptr)
1608         {
1609                 //
1610                 // If no defines were provided, we create a new local preproc_map and assign
1611                 // it to defines temporarily. In this case, the map will be deleted once this
1612                 // object is destroyed and defines will still be subsequently null.
1613                 //
1614                 if(!defines) {
1615                         local_defines_.reset(new preproc_map);
1616                         defines = local_defines_.get();
1617                 }
1618 
1619                 buf_.reset(new preprocessor_streambuf(defines));
1620 
1621                 // Begin processing.
1622                 buf_->add_preprocessor<preprocessor_file>(fname);
1623 
1624                 //
1625                 // TODO: not sure if this call is needed. Previously, this call was passed a
1626                 // preprocessor_streambuf pointer and the std::basic_istream constructor was
1627                 // called with its contents. However, at that point the preprocessing should
1628                 // already have completed, meaning this call might be redundant. Not sure.
1629                 //
1630                 // - vultraz, 2017-08-31
1631                 //
1632                 init(buf_.get());
1633         }
1634 
1635         ~preprocessor_scope_helper()
1636         {
1637                 clear(std::ios_base::goodbit);
1638                 exceptions(std::ios_base::goodbit);
1639                 rdbuf(nullptr);
1640         }

1642         std::unique_ptr<preprocessor_streambuf> buf_;

1659 void preprocess_resource(const std::string& res_name,
1660                 preproc_map* defines_map,
1661                 bool write_cfg,
1662                 bool write_plain_cfg,
1663                 const std::string& parent_directory)
1664 {
1665         if(filesystem::is_directory(res_name)) {
1666                 std::vector<std::string> dirs, files;
1667 
1668                 filesystem::get_files_in_dir(res_name, &files, &dirs, filesystem::ENTIRE_FILE_PATH, filesystem::SKIP_MEDIA_DIR,
1669                                 filesystem::DO_REORDER);
1670 
1671                 // Subdirectories
1672                 for(const std::string& dir : dirs) {
1673                         LOG_PREPROC << "processing sub-dir: " << dir << '\n';
1674                         preprocess_resource(dir, defines_map, write_cfg, write_plain_cfg, parent_directory);
1675                 }
1676 
1677                 // Files in current directory
1678                 for(const std::string& file : files) {
1679                         preprocess_resource(file, defines_map, write_cfg, write_plain_cfg, parent_directory);
1680                 }
1681 
1682                 return;
1683         }
1684 
1685         // process only config files.
1686         if(!filesystem::ends_with(res_name, ".cfg")) {
1687                 return;
1688         }
1689 
1690         LOG_PREPROC << "processing resource: " << res_name << '\n';
1691 
1692         // disable filename encoding to get clear #line in cfg.plain
1693         encode_filename = false;
1694 
1695         filesystem::scoped_istream stream = preprocess_file(res_name, defines_map);
1696 
1697         std::stringstream ss;
1698 
1699         // Set the failbit so if we get any preprocessor exceptions (e.g.:preproc_config::error)
1700         // they will be propagated in the main program, instead of just setting the
1701         // failbit on the stream. This was necessary in order for the MSVC and GCC
1702         // binaries to behave the same way.
1703         ss.exceptions(std::ios_base::failbit);
1704 
1705         ss << (*stream).rdbuf();
1706 
1707         LOG_PREPROC << "processing finished\n";
1708 
1709         if(write_cfg || write_plain_cfg) {
1710                 config cfg;
1711                 std::string streamContent = ss.str();
1712 
1713                 read(cfg, streamContent);
1714 
1715                 const std::string preproc_res_name = parent_directory + "/" + filesystem::base_name(res_name);
1716 
1717                 // Write the processed cfg file
1718                 if(write_cfg) {
1719                         LOG_PREPROC << "writing cfg file: " << preproc_res_name << '\n';
1720 
1721                         filesystem::create_directory_if_missing_recursive(filesystem::directory_name(preproc_res_name));
1722                         filesystem::scoped_ostream outStream(filesystem::ostream_file(preproc_res_name));
1723 
1724                         write(*outStream, cfg);
1725                 }
1726 
1727                 // Write the plain cfg file
1728                 if(write_plain_cfg) {
1729                         LOG_PREPROC << "writing plain cfg file: " << (preproc_res_name + ".plain") << '\n';
1730 
1731                         filesystem::create_directory_if_missing_recursive(filesystem::directory_name(preproc_res_name));
1732                         filesystem::write_file(preproc_res_name + ".plain", streamContent);
1733                 }
1734         }
