  53 // get filename associated to this code
  54 static std::string get_filename(const std::string& file_code){
  55         if(!encode_filename)
  56                 return file_code;
  57 
  58         std::stringstream s;
  59         s << file_code;
  60         int n = 0;
  61         s >> std::hex >> n;
  62 
  63         for(const auto& p : file_number_map) {
  64                 if(p.second == n)
  65                         return p.first;
  66         }
  67         return "<unknown>";
  68 }

  70 // get code associated to this filename
  71 static std::string get_file_code(const std::string& filename){
  72         if(!encode_filename)
  73                 return filename;
  74 
  75         // current number of encountered filenames
  76         static int current_file_number = 0;
  77 
  78         int& fnum = file_number_map[utils::escape(filename, " \\")];
  79         if(fnum == 0)
  80                 fnum = ++current_file_number;
  81 
  82         std::ostringstream shex;
  83         shex << std::hex << fnum;
  84         return shex.str();
  85 }

  87 // decode the filenames placed in a location
  88 static std::string get_location(const std::string& loc)
  89 {
  90         std::string res;
  91         std::vector< std::string > pos = utils::quoted_split(loc, ' ');
  92         if(pos.empty())
  93                 return res;
  94         std::vector< std::string >::const_iterator i = pos.begin(), end = pos.end();
  95         while (true) {
  96                 res += get_filename(*(i++));
  97                 if(i == end) break;
  98                 res += ' ';
  99                 res += *(i++);
 100                 if(i == end) break;
 101                 res += ' ';
 102         }
 103         return res;
 104 }

 246 // decode the filenames placed in a location
 247 std::string preprocessor_streambuf::get_current_file()
 248 {
 249         unsigned nested_level = 0;
 250         preprocessor* pre = current_;
 251         while(pre && pre->is_macro()) {
 252                 if(pre->is_macro() == 1) {
 253                         ++nested_level;
 254                 }
 255                 pre = pre->get_old_preprocessor();
 256         }
 257         std::string res;
 258         std::vector< std::string > pos = utils::quoted_split(location_, ' ');
 259         if(pos.size() <= 2*nested_level) {
 260                 return res;
 261         }
 262         else {
 263                 return get_filename(pos[2*nested_level]);
 264         }
 265 }

 297 /**
 298  * Called by an STL stream whenever it has reached the end of #out_buffer_.
 299  * Fills #buffer_ by calling the #current_ preprocessor, then copies its
 300  * content into #out_buffer_.
 301  * @return the first character of #out_buffer_ if any, EOF otherwise.
 302  */
 303 int preprocessor_streambuf::underflow()
 304 {
 305         unsigned sz = 0;
 306         if (char *gp = gptr()) {
 307                 if (gp < egptr()) {
 308                         // Sanity check: the internal buffer has not been totally consumed,
 309                         // should we force the caller to use what remains first?
 310                         return *gp;
 311                 }
 312                 // The buffer has been completely read; fill it again.
 313                 // Keep part of the previous buffer, to ensure putback capabilities.
 314                 sz = out_buffer_.size();
 315                 buffer_.str(std::string());
 316                 if (sz > 3) {
 317                         buffer_ << out_buffer_.substr(sz - 3);
 318                         sz = 3;
 319                 }
 320                 else
 321                 {
 322                         buffer_ << out_buffer_;
 323                 }
 324         } else {
 325                 // The internal get-data pointer is null
 326         }
 327         const int desired_fill_amount = 2000;
 328         while (current_ && buffer_.rdbuf()->in_avail() < desired_fill_amount)
 329         {
 330                 // Process files and data chunks until the desired buffer size is reached
 331                 if (!current_->get_chunk()) {
 332                          // Delete the current preprocessor item to restore its predecessor
 333                         delete current_;
 334                 }
 335         }
 336         // Update the internal state and data pointers
 337         out_buffer_ = buffer_.str();
 338         if(out_buffer_.empty()) {
 339                 return EOF;
 340         }
 341         char *begin = &*out_buffer_.begin();
 342         unsigned bs = out_buffer_.size();
 343         setg(begin, begin + sz, begin + bs);
 344         if (sz >= bs)
 345                 return EOF;
 346         return static_cast<unsigned char>(*(begin + sz));
 347 }

 349 std::string lineno_string(const std::string &lineno)
 350 {
 351         std::vector< std::string > pos = utils::quoted_split(lineno, ' ');
 352         std::vector< std::string >::const_iterator i = pos.begin(), end = pos.end();
 353         std::string included_from =
 354                 preprocessor_error_detail_prefix + "included from ";
 355         std::string res;
 356         while (i != end) {
 357                 const std::string& line = *(i++);
 358                 if (!res.empty())
 359                         res += included_from;
 360                 if (i != end)
 361                         res += get_filename(*(i++));
 362                 else
 363                         res += "<unknown>";
 364                 res += ':' + line;
 365         }
 366         if (res.empty()) res = "???";
 367         return res;
 368 }

 447  * This is the core of the preprocessor.

 449 class preprocessor_data: preprocessor

 451         /** Description of a preprocessing chunk. */
 452         struct token_desc
 453         {
 454                 enum TOKEN_TYPE {
 455                         START,        // Toplevel
 456                         PROCESS_IF,   // Processing the "if" branch of a ifdef/ifndef (the "else" branch will be skipped)

 458                         SKIP_IF,      // Skipping the "if" branch of a ifdef/ifndef (the "else" branch, if any, will be processed)

 547 preprocessor_file::preprocessor_file(preprocessor_streambuf &t, const std::string& name, size_t symbol_index=-1) :
 548         preprocessor(t),
 549         files_(),
 550         pos_(),
 551         end_()
 552 {
 553         if (filesystem::is_directory(name)) {
 554 
 555                 filesystem::get_files_in_dir(name, &files_, nullptr, filesystem::ENTIRE_FILE_PATH, filesystem::SKIP_MEDIA_DIR, filesystem::DO_REORDER);
 556 
 557                 for(std::string fname : files_) {
 558                         size_t cpos = fname.rfind(" ");
 559                         if (cpos != std::string::npos && cpos >= symbol_index) {
 560                                 std::stringstream ss;
 561                                 ss << "Found filename containing whitespace: '" << filesystem::base_name(fname) << "' in included directory '" << name << "'.\nThe included symbol probably looks similar to '"
 562                                  << filesystem::directory_name(fname.substr(symbol_index)) << "'";
 563                                 // TODO: find a real linenumber
 564                                 target_.error(ss.str(), -1);
 565                         }
 566                 }
 567         }
 568         else {
 569                 filesystem::scoped_istream file_stream = filesystem::istream_file(name);
 570                 if (!file_stream->good()) {
 571                         ERR_PREPROC << "Could not open file " << name << std::endl;
 572                 }
 573                 else
 574                         new preprocessor_data(t, file_stream.release(), "", filesystem::get_short_wml_path(name),
 575                                 1, filesystem::directory_name(name), t.textdomain_, nullptr);
 576         }
 577         pos_ = files_.begin();
 578         end_ = files_.end();
 579 }

 581 /**
 582  * preprocessor_file::get_chunk()
 583  *
 584  * Inserts and processes the next file in the list of included files.
 585  * @return      false if there is no next file.
 586  */
 587 bool preprocessor_file::get_chunk()
 588 {
 589         while (pos_ != end_) {
 590                 const std::string &name = *(pos_++);
 591                 unsigned sz = name.size();
 592                 // Use reverse iterator to optimize testing
 593                 if (sz < 5 || !std::equal(name.rbegin(), name.rbegin() + 4, "gfc."))
 594                         continue;
 595                 new preprocessor_file(target_, name);
 596                 return true;
 597         }
 598         return false;
 599 }

 646 void preprocessor_data::push_token(token_desc::TOKEN_TYPE t)
 647 {
 648         token_desc token(t, strings_.size(), linenum_);
 649         tokens_.push_back(token);
 650         if (t == token_desc::MACRO_SPACE) {
 651                 // Macro expansions do not have any associated storage at start.
 652                 return;
 653         } else if (t == token_desc::STRING || t == token_desc::VERBATIM) {
 654                 /* Quoted strings are always inlined in the parent token. So
 655                    they need neither storage nor metadata, unless the parent
 656                    token is a macro expansion. */
 657                 token_desc::TOKEN_TYPE &outer_type = tokens_[tokens_.size() - 2].type;
 658                 if (outer_type != token_desc::MACRO_SPACE)
 659                         return;
 660                 outer_type = token_desc::MACRO_CHUNK;
 661                 tokens_.back().stack_pos = strings_.size() + 1;
 662         }
 663         std::ostringstream s;
 664         if (!skipping_ && slowpath_) {
 665                 s << "\376line " << linenum_ << ' ' << target_.location_
 666                   << "\n\376textdomain " << target_.textdomain_ << '\n';
 667         }
 668         strings_.push_back(s.str());
 669 }
 670 
 671 void preprocessor_data::pop_token()
 672 {
 673         token_desc::TOKEN_TYPE inner_type = tokens_.back().type;
 674         unsigned stack_pos = tokens_.back().stack_pos;
 675         tokens_.pop_back();
 676         token_desc::TOKEN_TYPE &outer_type = tokens_.back().type;
 677         if (inner_type == token_desc::MACRO_PARENS) {
 678                 // Parenthesized macro arguments are left on the stack.
 679                 assert(outer_type == token_desc::MACRO_SPACE);
 680                 return;
 681         }
 682         if (inner_type == token_desc::STRING || inner_type == token_desc::VERBATIM) {
 683                 // Quoted strings are always inlined.
 684                 assert(stack_pos == strings_.size());
 685                 return;
 686         }
 687         if (outer_type == token_desc::MACRO_SPACE) {
 688                 /* A macro expansion does not have any associated storage.
 689                    Instead, storage of the inner token is not discarded
 690                    but kept as a new macro argument. But if the inner token
 691                    was a macro expansion, it is about to be appended, so
 692                    prepare for it. */
 693                 if (inner_type == token_desc::MACRO_SPACE || inner_type == token_desc::MACRO_CHUNK) {
 694                         strings_.erase(strings_.begin() + stack_pos, strings_.end());
 695                         strings_.emplace_back();
 696                 }
 697                 assert(stack_pos + 1 == strings_.size());
 698                 outer_type = token_desc::MACRO_CHUNK;
 699                 return;
 700         }
 701         strings_.erase(strings_.begin() + stack_pos, strings_.end());
 702 }

 704 void preprocessor_data::skip_spaces()
 705 {
 706         for(;;) {
 707                 int c = in_.peek();
 708                 if (in_.eof() || (c != ' ' && c != '\t'))
 709                         return;
 710                 in_.get();
 711         }
 712 }
 713 
 714 void preprocessor_data::skip_eol()
 715 {
 716         for(;;) {
 717                 int c = in_.get();
 718                 if (c == '\n') {
 719                         ++linenum_;
 720                         return;
 721                 }
 722                 if (in_.eof())
 723                         return;
 724         }
 725 }

 727 std::string preprocessor_data::read_word()
 728 {
 729         std::string res;
 730         for(;;) {
 731                 int c = in_.peek();
 732                 if (c == preprocessor_streambuf::traits_type::eof() || utils::portable_isspace(c)) {
 733                         // DBG_PREPROC << "(" << res << ")\n";
 734                         return res;
 735                 }
 736                 in_.get();
 737                 res += static_cast<char>(c);
 738         }
 739 }

 741 std::string preprocessor_data::read_line()
 742 {
 743         std::string res;
 744         for(;;) {
 745                 int c = in_.get();
 746                 if (c == '\n') {
 747                         ++linenum_;
 748                         return res;
 749                 }
 750                 if (in_.eof())
 751                         return res;
 752                 if (c != '\r')
 753                         res += static_cast<char>(c);
 754         }
 755 }

 757 std::string preprocessor_data::read_rest_of_line()
 758 {
 759         std::string res;
 760         while(in_.peek() != '\n' && !in_.eof()) {
 761                 int c = in_.get();
 762                 if (c != '\r')
 763                         res += static_cast<char>(c);
 764         }
 765         return res;
 766 }

 768 void preprocessor_data::put(char c)
 769 {
 770         if (skipping_)
 771                 return;
 772         if (slowpath_) {
 773                 strings_.back() += c;
 774                 return;
 775         }
 776 
 777         int cond_linenum = c == '\n' ? linenum_ - 1 : linenum_;
 778         if (unsigned diff = cond_linenum - target_.linenum_)
 779         {
 780                 target_.linenum_ = cond_linenum;
 781                 if (diff <= target_.location_.size() + 11) {
 782                         target_.buffer_ << std::string(diff, '\n');
 783                 } else {
 784                         target_.buffer_ << "\376line " << target_.linenum_
 785                                 << ' ' << target_.location_ << '\n';
 786                 }
 787         }
 788 
 789         if (c == '\n')
 790                 ++target_.linenum_;
 791         target_.buffer_ << c;
 792 }

 812 bool preprocessor_data::get_chunk()
 813 {
 814         char c = in_.get();
 815         token_desc &token = tokens_.back();
 816         if (in_.eof()) {
 817                 // The end of file was reached.
 818                 // Make sure we don't have any incomplete tokens.
 819                 char const *s;
 820                 switch (token.type) {
 821                 case token_desc::START: return false; // everything is fine
 822                 case token_desc::PROCESS_IF:
 823                 case token_desc::SKIP_IF:
 824                 case token_desc::PROCESS_ELSE:
 825                 case token_desc::SKIP_ELSE: s = "#ifdef or #ifndef"; break;
 826                 case token_desc::STRING: s = "Quoted string"; break;
 827                 case token_desc::VERBATIM: s = "Verbatim string"; break;
 828                 case token_desc::MACRO_CHUNK:
 829                 case token_desc::MACRO_SPACE: s = "Macro substitution"; break;
 830                 case token_desc::MACRO_PARENS: s = "Macro argument"; break;
 831                 default: s = "???";
 832                 }
 833                 target_.error(std::string(s) + " not terminated", token.linenum);
 834         }
 835         if (c == '\n')
 836                 ++linenum_;
 837         if (c == '\376') {
 838                 std::string buffer(1, c);
 839                 for(;;) {
 840                         char d = in_.get();
 841                         if (in_.eof() || d == '\n')
 842                                 break;
 843                         buffer += d;
 844                 }
 845                 buffer += '\n';
 846                 // line_change = 1-1 = 0
 847                 put(buffer);
 848         } else if (token.type == token_desc::VERBATIM) {
 849                 put(c);
 850                 if (c == '>' && in_.peek() == '>') {
 851                         put(in_.get());
 852                         pop_token();
 853                 }
 854         } else if (c == '<' && in_.peek() == '<') {
 855                 in_.get();
 856                 push_token(token_desc::VERBATIM);
 857                 put('<');
 858                 put('<');
 859         } else if (c == '"') {
 860                 if (token.type == token_desc::STRING) {
 861                         target_.quoted_ = false;
 862                         put(c);
 863                         pop_token();
 864                 } else if (!target_.quoted_) {
 865                         target_.quoted_ = true;
 866                         push_token(token_desc::STRING);
 867                         put(c);
 868                 } else {
 869                         target_.error("Nested quoted string" , linenum_);
 870                 }
 871         } else if (c == '{') {
 872                 push_token(token_desc::MACRO_SPACE);
 873                 ++slowpath_;
 874         } else if (c == ')' && token.type == token_desc::MACRO_PARENS) {
 875                 pop_token();
 876         } else if (c == '#' && !target_.quoted_) {
 877                 std::string command = read_word();
 878                 bool comment = false;
 879                 if (command == "define") {
 880                         skip_spaces();
 881                         int linenum = linenum_;
 882                         std::vector< std::string > items = utils::split(read_line(), ' ');
 883                         std::map< std::string, std::string> optargs;
 884                         if (items.empty()) {
 885                                 target_.error("No macro name found after #define directive", linenum);
 886                         }
 887                         std::string symbol = items.front();
 888                         items.erase(items.begin());
 889                         int found_arg = 0, found_enddef = 0;
 890                         std::string buffer;
 891                         for(;;) {
 892                                 if (in_.eof())
 893                                         break;
 894                                 char d = in_.get();
 895                                 if (d == '\n')
 896                                         ++linenum_;
 897                                 buffer += d;
 898                                 if (d == '#') {
 899                                         if (in_.peek() == 'a') {
 900                                                 found_arg = 1;
 901                                         } else {
 902                                                 found_enddef = 1;
 903                                         }
 904                                 } else {
 905                                         if (found_arg > 0 && ++found_arg == 4) {
 906                                                 if (std::equal(buffer.end() - 3, buffer.end(), "arg")) {
 907                                                         buffer.erase(buffer.end() - 4, buffer.end());
 908 
 909                                                         skip_spaces();
 910                                                         std::string argname = read_word();
 911                                                         skip_eol();
 912 
 913                                                         std::string argbuffer;
 914 
 915                                                         int found_endarg = 0;
 916                                                         for(;;) {
 917                                                                 if (in_.eof())
 918                                                                         break;
 919                                                                 char e = in_.get();
 920                                                                 if (e == '\n')
 921                                                                         ++linenum_;
 922                                                                 argbuffer += e;
 923 
 924                                                                 if (e == '#') {
 925                                                                         found_endarg = 1;
 926                                                                 } else if (found_endarg > 0 && ++found_endarg == 7) {
 927                                                                         if (std::equal(argbuffer.end() - 6, argbuffer.end(), "endarg")) {
 928                                                                                 argbuffer.erase(argbuffer.end() - 7, argbuffer.end());
 929                                                                                 optargs[argname] = argbuffer;
 930                                                                                 skip_eol();
 931                                                                                 break;
 932                                                                         } else {
 933                                                                                 target_.error("Unterminated #arg definition", linenum_);
 934                                                                         }
 935                                                                 }
 936                                                         }
 937                                                 }
 938                                         }
 939 
 940                                         if (found_enddef > 0 && ++found_enddef == 7) {
 941                                                 if (std::equal(buffer.end() - 6, buffer.end(), "enddef"))
 942                                                         break;
 943                                                 else {
 944                                                         found_enddef = 0;
 945                                                         if (std::equal(buffer.end()-6, buffer.end(), "define")) { //TODO: Maybe add support for this? This would fill feature request #21343
 946                                                                 target_.error("Preprocessor error: #define is not allowed inside a #define/#enddef pair", linenum);
 947                                                         }
 948                                                 }
 949                                         }
 950                                 }
 951                         }
 952                         if (found_enddef != 7) {
 953                                 target_.error("Unterminated preprocessor definition", linenum_);
 954                         }
 955                         if (!skipping_) {
 956                                 preproc_map::const_iterator old_i = target_.defines_->find(symbol);
 957                                 if (old_i != target_.defines_->end()) {
 958                                         std::ostringstream new_pos, old_pos;
 959                                         const preproc_define& old_d = old_i->second;
 960 
 961                                         new_pos << linenum << ' ' << target_.location_;
 962                                         old_pos << old_d.linenum << ' ' << old_d.location;
 963 
 964                                         WRN_PREPROC << "Redefining macro " << symbol
 965                                                                 << " without explicit #undef at "
 966                                                                 << lineno_string(new_pos.str()) << '\n'
 967                                                                 << "previously defined at "
 968                                                                 << lineno_string(old_pos.str()) << '\n';
 969                                 }
 970 
 971                                 buffer.erase(buffer.end() - 7, buffer.end());
 972                                 (*target_.defines_)[symbol] = preproc_define(buffer, items, optargs, target_.textdomain_,
 973                                                                linenum, target_.location_);

 975                         }
 976                 } else if (command == "ifdef") {

 980                         DBG_PREPROC << "testing for macro " << symbol << ": "

 983                 } else if (command == "ifndef") {
 984                         skip_spaces();
 985                         const std::string& symbol = read_word();
 986                         bool found = target_.defines_->count(symbol) != 0;
 987                         DBG_PREPROC << "testing for macro " << symbol << ": "
 988                                 << (found ? "defined" : "not defined") << '\n';
 989                         conditional_skip(found);
 990                 } else if (command == "ifhave") {
 991                         skip_spaces();

 994                         DBG_PREPROC << "testing for file or directory " << symbol << ": "
 995                                 << (found ? "found" : "not found") << '\n';
 996                         conditional_skip(!found);
 997                 } else if (command == "ifnhave") {
 998                         skip_spaces();
 999                         const std::string& symbol = read_word();
1000                         bool found = !filesystem::get_wml_location(symbol, directory_).empty();
1001                         DBG_PREPROC << "testing for file or directory " << symbol << ": "
1002                                 << (found ? "found" : "not found") << '\n';
1003                         conditional_skip(found);
1004                 } else if (command == "ifver" || command == "ifnver") {
1005                         skip_spaces();
1006                         const std::string& vsymstr = read_word();
1007                         skip_spaces();
1008                         const std::string& vopstr = read_word();
1009                         skip_spaces();
1010                         const std::string& vverstr = read_word();
1011 
1012                         const VERSION_COMP_OP vop = parse_version_op(vopstr);
1013

1017                                 preproc_define const& sym = (*target_.defines_)[vsymstr];
1018

1020                                         target_.error("First argument macro in #ifver/#ifnver should not require arguments", linenum_);
1021                                 }

1028                                         << (found ? "match" : "no match") << '\n';
1029

1039                                 pop_token();

1041                                 push_token(token_desc::PROCESS_ELSE);
1042                         } else if (token.type == token_desc::PROCESS_IF) {
1043                                 pop_token();

1045                                 push_token(token_desc::SKIP_IF);
1046                         } else {
1047                                 target_.error("Unexpected #else", linenum_);
1048                         }
1049                 } else if (command == "endif") {
1050                         switch (token.type) {
1051                         case token_desc::SKIP_IF:
1052                         case token_desc::SKIP_ELSE: --skipping_;
1053                         case token_desc::PROCESS_IF:
1054                         case token_desc::PROCESS_ELSE: break;
1055                         default:
1056                                 target_.error("Unexpected #endif", linenum_);
1057                         }
1058                         pop_token();
1059                 } else if (command == "textdomain") {
1060                         skip_spaces();
1061                         const std::string& s = read_word();
1062                         if (s != target_.textdomain_) {
1063                                 put("#textdomain ");
1064                                 put(s);
1065                                 target_.textdomain_ = s;
1066                         }
1067                         comment = true;
1068                 } else if (command == "enddef") {
1069                         target_.error("Unexpected #enddef", linenum_);
1070                 } else if (command == "undef") {
1071                         skip_spaces();
1072                         const std::string& symbol = read_word();
1073                         if (!skipping_) {
1074                                 target_.defines_->erase(symbol);
1075                                 LOG_PREPROC << "undefine macro " << symbol << " (location " << get_location(target_.location_) << ")\n";
1076                         }
1077                 } else if (command == "error") {
1078                         if (!skipping_) {
1079                                 skip_spaces();
1080                                 std::ostringstream error;
1081                                 error << "#error: \"" << read_rest_of_line() << '"';
1082                                 target_.error(error.str(), linenum_);
1083                         } else
1084                                 DBG_PREPROC << "Skipped an error\n";
1085                 } else if (command == "warning") {
1086                         if (!skipping_) {
1087                                 skip_spaces();
1088                                 std::ostringstream warning;
1089                                 warning << "#warning: \"" << read_rest_of_line() << '"';
1090                                 target_.warning(warning.str(), linenum_);
1091                         } else
1092                                 DBG_PREPROC << "Skipped a warning\n";
1093                 } else
1094                         comment = token.type != token_desc::MACRO_SPACE;
1095                 skip_eol();
1096                 if (comment)
1097                         put('\n');
1098         } else if (token.type == token_desc::MACRO_SPACE || token.type == token_desc::MACRO_CHUNK) {
1099                 if (c == '(') {
1100                         // If a macro argument was started, it is implicitly ended.
1101                         token.type = token_desc::MACRO_SPACE;
1102                         push_token(token_desc::MACRO_PARENS);
1103                 } else if (utils::portable_isspace(c)) {
1104                         // If a macro argument was started, it is implicitly ended.
1105                         token.type = token_desc::MACRO_SPACE;
1106                 } else if (c == '}') {
1107                         --slowpath_;
1108                         if (skipping_) {
1109                                 pop_token();
1110                                 return true;
1111                         }
1112                         // FIXME: is this obsolete?
1113                         //if (token.type == token_desc::MACRO_SPACE) {

1118                         //              location<<linenum_<<' '<<target_.location_;
1119                         //              target_.error(error.str(), location.str());
1120                         //      }
1121                         //      strings_.pop_back();
1122                         //}
1123 
1124                         if(strings_.size() <= static_cast<size_t>(token.stack_pos)) {
1125                                 target_.error("No macro or file substitution target specified", linenum_);
1126                         }
1127 
1128                         std::string symbol = strings_[token.stack_pos];
1129                         std::string::size_type pos;
1130                         while ((pos = symbol.find('\376')) != std::string::npos) {
1131                                 std::string::iterator b = symbol.begin(); // invalidated at each iteration
1132                                 symbol.erase(b + pos, b + symbol.find('\n', pos + 1) + 1);
1133                         }
1134                         std::map<std::string, std::string>::const_iterator arg;
1135                         preproc_map::const_iterator macro;
1136                         // If this is a known pre-processing symbol, then we insert it,
1137                         // otherwise we assume it's a file name to load.

1139                                 pop_token();

1141                         }
1142                         else if(symbol == current_dir_str && strings_.size() - token.stack_pos == 1) {
1143                                 pop_token();
1144                                 put(filesystem::directory_name(target_.get_current_file()));
1145                         }

1147                             (arg = local_defines_->find(symbol)) != local_defines_->end())

1150                                 {
1151                                         std::ostringstream error;
1152                                         error << "Macro argument '" << symbol
1153                                               << "' does not expect any arguments";
1154                                         target_.error(error.str(), linenum_);
1155                                 }

1157                                 v << arg->second << "\376line " << linenum_ << ' ' << target_.location_
1158                                   << "\n\376textdomain " << target_.textdomain_ << '\n';

1161                         }

1241                                 std::istringstream *buffer = new std::istringstream(val.value);
1242                                 pop_token();
1243                                 if (!slowpath_) {
1244                                         DBG_PREPROC << "substituting macro " << symbol << '\n';
1245                                         new preprocessor_data(target_, buffer, val.location, "",
1246                                                               val.linenum, dir, val.textdomain, defines, true);
1247                                 } else {
1248                                         DBG_PREPROC << "substituting (slow) macro " << symbol << '\n';
1249                                         std::ostringstream res;
1250                                         preprocessor_streambuf *buf =
1251                                                 new preprocessor_streambuf(target_);
1252                                         // Make the nested preprocessor_data responsible for
1253                                         // restoring our current textdomain if needed.
1254                                         buf->textdomain_ = target_.textdomain_;
1255                                         {       std::istream in(buf);
1256                                                 new preprocessor_data(*buf, buffer, val.location, "",
1257                                                                       val.linenum, dir, val.textdomain, defines, true);
1258                                                 res << in.rdbuf(); }
1259                                         delete buf;
1260                                         put(res.str());
1261                                 }
1262                         } else if (target_.depth_ < 40) {
1263                                 LOG_PREPROC << "Macro definition not found for " << symbol << " , attempting to open as file.\n";
1264                                 pop_token();
1265                                 std::string nfname = filesystem::get_wml_location(symbol, directory_);
1266                                 if (!nfname.empty())
1267                                 {
1268                                         if (!slowpath_)
1269                                                 // nfname.size() - symbol.size() gives you an index into nfname
1270                                                 // This does not necessarily match the symbol though, as it can start with ~ or ./
1271                                                 new preprocessor_file(target_, nfname, nfname.size() - symbol.size());
1272                                         else {
1273                                                 std::ostringstream res;
1274                                                 preprocessor_streambuf *buf =
1275                                                         new preprocessor_streambuf(target_);
1276                                                 {       std::istream in(buf);
1277                                                         new preprocessor_file(*buf, nfname, nfname.size() - symbol.size());
1278                                                         res << in.rdbuf(); }
1279                                                 delete buf;
1280                                                 put(res.str());
1281                                         }
1282                                 }
1283                                 else
1284                                 {
1285                                         std::ostringstream error;
1286                                         error << "Macro/file '" << symbol << "' is missing";
1287                                         target_.error(error.str(), linenum_);
1288                                 }
1289                         } else {
1290                                 target_.error("Too many nested preprocessing inclusions", linenum_);
1291                         }
1292                 }
1293                 else if (!skipping_)
1294                 {
1295                         if (token.type == token_desc::MACRO_SPACE)
1296                         {
1297                                 std::ostringstream s;
1298                                 s << "\376line " << linenum_ << ' ' << target_.location_
1299                                   << "\n\376textdomain " << target_.textdomain_ << '\n';
1300                                 strings_.push_back(s.str());
1301                                 token.type = token_desc::MACRO_CHUNK;
1302                         }
1303                         put(c);
1304                 }
1305         } else
1306                 put(c);
1307         return true;
1308 }

1333 filesystem::scoped_istream preprocess_file(const std::string& fname, preproc_map *defines)
1334 {
1335         log_scope("preprocessing file " + fname + " ...");
1336         preproc_map *owned_defines = nullptr;
1337         if (!defines) {
1338                 // If no preproc_map has been given, create a new one,
1339                 // and ensure it is destroyed when the stream is
1340 // ??
1341                 // by giving it to the deleter.
1342                 owned_defines = new preproc_map;
1343                 defines = owned_defines;
1344         }
1345         preprocessor_streambuf *buf = new preprocessor_streambuf(defines);
1346 
1347         new preprocessor_file(*buf, fname);
1348         return filesystem::scoped_istream(new preprocessor_deleter(buf, owned_defines));
1349 }

1351 void preprocess_resource(const std::string& res_name, preproc_map *defines_map,
1352                          bool write_cfg, bool write_plain_cfg,const std::string& target_directory)
1353 {
1354         if (filesystem::is_directory(res_name))
1355         {
1356                 std::vector<std::string> dirs,files;
1357 
1358                 filesystem::get_files_in_dir(res_name, &files, &dirs, filesystem::ENTIRE_FILE_PATH, filesystem::SKIP_MEDIA_DIR, filesystem::DO_REORDER);
1359 
1360                 // subdirectories
1361                 for(const std::string& dir : dirs)
1362                 {
1363                         LOG_PREPROC << "processing sub-dir: " << dir << '\n';
1364                         preprocess_resource(dir, defines_map, write_cfg, write_plain_cfg, target_directory);
1365                 }
1366 
1367                 // files in current directory
1368                 for(const std::string& file : files)
1369                 {
1370                         preprocess_resource(file, defines_map, write_cfg, write_plain_cfg, target_directory);
1371                 }
1372                 return;
1373         }
1374 
1375         // process only config files.
1376         if (filesystem::ends_with(res_name, ".cfg") == false)
1377                 return;
1378 
1379         LOG_PREPROC << "processing resource: " << res_name << '\n';
1380 
1381         //disable filename encoding to get clear #line in cfg.plain
1382         encode_filename = false;
1383 
1384         filesystem::scoped_istream stream = preprocess_file(res_name, defines_map);
1385         std::stringstream ss;
1386         // Set the failbit so if we get any preprocessor exceptions (e.g.:preproc_config::error)
1387         // they will be propagated in the main program, instead of just setting the
1388         // failbit on the stream. This was necessary in order for the MSVC and GCC
1389         // binaries to behave the same way.
1390         ss.exceptions(std::ios_base::failbit);
1391 
1392         ss << (*stream).rdbuf();
1393 
1394         LOG_PREPROC << "processing finished\n";
1395 
1396         if (write_cfg == true || write_plain_cfg == true)
1397         {
1398                 config cfg;
1399                 std::string streamContent = ss.str();
1400 
1401                 read(cfg, streamContent);
1402                 const std::string preproc_res_name = target_directory + "/" + filesystem::base_name(res_name);
1403 
1404                 // write the processed cfg file
1405                 if (write_cfg == true)
1406                 {
1407                         LOG_PREPROC << "writing cfg file: " << preproc_res_name << '\n';
1408                         filesystem::create_directory_if_missing_recursive(filesystem::directory_name(preproc_res_name));
1409                         filesystem::scoped_ostream outStream(filesystem::ostream_file(preproc_res_name));
1410                         write(*outStream, cfg);
1411                 }
1412 
1413                 // write the plain cfg file
1414                 if (write_plain_cfg == true)
1415                 {
1416                         LOG_PREPROC << "writing plain cfg file: " << (preproc_res_name + ".plain") << '\n';
1417                         filesystem::create_directory_if_missing_recursive(filesystem::directory_name(preproc_res_name));
1418                         filesystem::write_file(preproc_res_name + ".plain", streamContent);
1419                 }
1420         }
1421 }
