  80 void stacked_widget::select_layer_impl(std::function<bool(unsigned int i)> display_condition)
  81 {
  82         const unsigned int num_layers = get_layer_count();
  83 
  84         // Deselect all layers except the chosen ones.
  85         for(unsigned int i = 0; i < num_layers; ++i) {
  86                 const bool selected = display_condition(i);
  87 
  88                 /* Selecting a previously selected item will deselect it, regardless of the what is passed to
  89                  * select_item. This causes issues if this function is called when all layers are visible (for
  90                  * example, initialization). For layers other than the chosen one, this is the desired behavior.
  91                  * However the chosen layer could *also* be deselected undesirably due to the conditions outlined
  92                  * above, and as this widget's generator does not stipulate a minimum selection, it's possible to
  93                  * end up with no layers visible at all.
  94                  *
  95                  * This works around that by performing no selection unless necessary to change states.
  96                  */
  97                 if(generator_->is_selected(i) != selected) {
  98                         generator_->select_item(i, selected);
  99                 }
 100         }
 101 
 102         // If we already have our chosen layers, exit.
 103         if(selected_layer_ >= 0) {
 104                 return;
 105         }
 106 
 107         // Else, re-show all layers.
 108         for(unsigned int i = 0; i < num_layers; ++i) {
 109                 /* By design, only the last selected item will receive events even if multiple items are visible
 110                  * and said item is not at the top of the stack. If this point is reached, all layers have already
 111                  * been hidden by the loop above, so the last layer selected will be the top-most one, as desired.
 112                  */
 113                 generator_->select_item(i, true);
 114         }
 115 }

 128 void stacked_widget::select_layer(const int layer)
 129 {
 130         update_selected_layer_index(layer);
 131 
 132         select_layer_impl([this](unsigned int i)
 133         {
 134                 return i == static_cast<unsigned int>(selected_layer_);
 135         });
 136 }
 137 
 138 void stacked_widget::select_layers(const boost::dynamic_bitset<>& mask)
 139 {
 140         assert(mask.size() == get_layer_count());
 141 
 142         select_layer_impl([&](unsigned int i)
 143         {
 144                 if(mask[i]) {
 145                         update_selected_layer_index(i);
 146                 }
 147 
 148                 return mask[i];
 149         });
 150 }

 177  * == Stacked widget ==
 178  *
 179  * A stacked widget holds several widgets on top of each other. This can be used
 180  * for various effects; add an optional overlay to an image, stack it with a
 181  * spacer to force a minimum size of a widget. The latter is handy to avoid
 182  * making a separate definition for a single instance with a fixed size.
 183  *
 184  * A stacked widget has no states.

 186  * @begin{tag}{name="stacked_widget_definition"}{min=0}{max=-1}{super="generic/widget_definition"}

 191  * @end{parent}{name="gui/"}

 231 builder_stacked_widget::builder_stacked_widget(const config& real_cfg)
 232         : builder_styled_widget(real_cfg), stack()
 233 {
 234         const config& cfg = real_cfg.has_child("stack") ? real_cfg.child("stack") : real_cfg;
 235         if(&cfg != &real_cfg) {
 236                 lg::wml_error() << "Stacked widgets no longer require a [stack] tag. Instead, place [layer] tags directly in the widget definition.\n";
 237         }
 238         VALIDATE(cfg.has_child("layer"), _("No stack layers defined."));
 239         for(const auto & layer : cfg.child_range("layer"))
 240         {
 241                 stack.emplace_back(std::make_shared<builder_grid>(layer));
 242         }
 243 }

 245 widget* builder_stacked_widget::build() const
 246 {
 247         stacked_widget* widget = new stacked_widget(*this);
 248 
 249         DBG_GUI_G << "Window builder: placed stacked widget '" << id
 250                           << "' with definition '" << definition << "'.\n";
 251 
 252         const auto conf = widget->cast_config_to<stacked_widget_definition>();
 253         assert(conf);
 254 
 255         widget->init_grid(conf->grid);
 256 
 257         widget->finalize(stack);
 258 
 259         return widget;
 260 }
