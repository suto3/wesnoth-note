  15 /**
  16  * @file
  17  * Editor action classes. Some important points:
  18  * - This is a polymorphic hierarchy of classes, so actions are usually passed around
  19  *   as editor_action pointers
  20  * - The pointers can, in general, be null. Always check for null before doing anything.
  21  *   The helper functions perform_ that take a pointer do that.
  22  * - The perform() functions can throw when an error occurs. Use smart pointers if you
  23  *   need to ensure the pointer is deleted.
  24  */

  51 /**
  52  * Base class for actions that:
  53  * 1) operate on an area
  54  * 2) can be used as undo for a click-drag operation
  55  * 3) can be extended so one undo action undos several actual drag actions
  56  */
  57 class editor_action_extendable : public editor_action
  58 {
  59         public:
  60                 editor_action_extendable()
  61                 {
  62                 }
  63                 /**
  64                  * The crux of the extendable contract. This member function must be
  65                  * implemented so that the undo behavior is consistent, exactly the
  66                  * same as would be with separate undo actions for every part of
  67                  * the drag.
  68                  */
  69                 virtual void extend(const editor_map& map, const std::set<map_location>& locs) = 0;
  70                 const char* get_name() const { return "extendable"; }
  71 };

  73 /**
  74  * Container action wrapping several actions into one.
  75  * The actions are performed in the order they are added,
  76  * i.e. in the usual iteration order through the container.
  77  */
  78 class editor_action_chain : public editor_action
  79 {
  80         public:
  81                 /**
  82                  * Create an empty action chain
  83                  */
  84                 editor_action_chain() :
  85                         actions_()
  86                 {
  87                 }
  88 
  89                 editor_action_chain(const editor_action_chain& other);
  90 
  91                 editor_action_chain& operator=(const editor_action_chain& other);
  92 
  93                 editor_action_chain* clone() const;
  94 
  95                 /**
  96                  * Create an action chain from a deque of action pointers.
  97                  * Note: the action chain assumes ownership of the pointers.
  98                  */
  99                 explicit editor_action_chain(std::deque<editor_action*> actions)
 100                 : actions_(actions)
 101                 {
 102                 }
 103 
 104                 /**
 105                  * Create an action chain by wrapping around a single action pointer.
 106                  * Note: the action chain assumes ownership of the pointer.
 107                  */
 108                 explicit editor_action_chain(editor_action* action)
 109                 : actions_(1, action)
 110                 {
 111                 }
 112 
 113                 /**
 114                  * The destructor deletes all the owned action pointers
 115                  */
 116                 ~editor_action_chain();
 117 
 118                 /**
 119                  * Go through the chain and add up all the action counts
 120                  */
 121                 int action_count() const;
 122 
 123                 /**
 124                  * Add an action at the end of the chain
 125                  */
 126                 void append_action(editor_action* a);
 127 
 128                 /**
 129                  * Add an action at the beginning of the chain
 130                  */
 131                 void prepend_action(editor_action* a);
 132 
 133                 /**
 134                  * @return true when there are no actions in the chain. Empty
 135                  * action chains should usually be discarded as to not keep
 136                  * "empty" actions around.
 137                  */
 138                 bool empty() const;
 139 
 140                 /**
 141                  * Remove the last added action and return it, transferring
 142                  * ownership to the caller
 143                  */
 144                 editor_action* pop_last_action();
 145 
 146                 /**
 147                  * Remove the first added action and return it, transferring
 148                  * ownership to the caller
 149                  */
 150                 editor_action* pop_first_action();
 151 
 152                 /**
 153                  * Perform all the actions in order and create a undo action chain
 154                  */
 155                 editor_action_chain* perform(map_context& m) const;
 156 
 157                 /**
 158                  * Perform all the actions in order
 159                  */
 160                 void perform_without_undo(map_context& m) const;
 161 
 162                 const char* get_name() const { return "chain"; }
 163 
 164         protected:
 165                 /**
 166                  * The action pointers owned by this action chain
 167                  */
 168                 std::deque<editor_action*> actions_;
 169 };
