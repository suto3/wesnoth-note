 149     private void createLink( IProject project, IFile linkFile, java.net.URI uri )
 150         throws CoreException
 151     {
 152         IPath path = linkFile.getFullPath( );
 153 
 154         IPath folders = path.removeLastSegments( 1 ).removeFirstSegments( 1 );
 155         IPath checkPath = Path.ROOT;
 156         int segmentCount = folders.segmentCount( );
 157         for( int i = 0; i < segmentCount; i++ ) {
 158             checkPath = checkPath.addTrailingSeparator( ).append(
 159                 folders.segment( i ) );
 160             IFolder folder = project.getFolder( checkPath );
 161             if( ! folder.exists( ) ) {
 162                 folder.create( true, true, null );
 163             }
 164         }
 165         linkFile.createLink( uri, IResource.ALLOW_MISSING_LOCAL, null );
 166     }

 168     @Override
 169     public void dispose( )
 170     {
 171         // Unlink the input if it was linked
 172         IEditorInput input = getEditorInput( );
 173         if( input instanceof IFileEditorInput ) {
 174             IFile file = ( ( IFileEditorInput ) input ).getFile( );
 175             if( file.isLinked( ) ) {
 176                 file.getProject( ).getName( ).equals( AUTOLINK_PROJECT_NAME );
 177                 try {
 178                     // if the editor is disposed because workbench is shutting
 179                     // down, it is NOT a good
 180                     // idea to delete the link
 181                     if( PlatformUI.isWorkbenchRunning( )
 182                         && ! PlatformUI.getWorkbench( ).isClosing( ) ) {
 183                         file.delete( true, null );
 184                     }
 185                 } catch( CoreException e ) {
 186                     // Nothing to do really, it will be recreated/refreshed
 187                     // later if ever used - some garbage may stay behind...
 188                     Logger.getInstance( ).logException( e );
 189                 }
 190             }
 191         }
 192         super.dispose( );
 193     }

 208      * Translates an incoming IEditorInput being an FilestoreEditorInput, or
 209      * IURIEditorInput
 210      * that is not also a IFileEditorInput.
 211      * FilestoreEditorInput is used when opening external files in an IDE
 212      * environment.
 213      * The result is that the regular XtextEditor gets an IEFSEditorInput which
 214      * is also an
 215      * IStorageEditorInput.
 216      */
 217     @Override
 218     public void init( IEditorSite site, IEditorInput input )
 219         throws PartInitException
 220     {
 221         // THE ISSUE HERE:
 222         // In the IDE, the File Open Dialog (and elsewhere) uses a
 223         // FilestoreEditorInput class
 224         // which is an IDE specific implementation.
 225         // The state at this point:
 226         // 1. When creating a file, the IEditorInput is an IURIEditorInput
 227         // 2. The only (non IDE specific) interface implemented by
 228         // FilestoreEditorInput is IURIEditorInput
 229         // 3. The creation of a file is however also an IFileEditorInput
 230         //
 231         // Remedy:
 232         if( input instanceof IURIEditorInput
 233             && ! ( input instanceof IFileEditorInput ) ) {
 234             java.net.URI uri = ( ( IURIEditorInput ) input ).getURI( );
 235             String name = ( ( IURIEditorInput ) input ).getName( );
 236             // Check if this is linkable input
 237             if( uri.getScheme( ).equals( "file" ) ) { //$NON-NLS-1$
 238                 IFile linkedFile = obtainLink( uri );
 239                 IFileEditorInput linkedInput = new LinkedFileEditorInput(
 240                     linkedFile );
 241                 super.init( site, linkedInput );
 242 
 243             }
 244             else {
 245                 // use EMF URI (readonly) input - will open without validation
 246                 // though...
 247                 // (Could be improved, i.e. perform a download, make readonly,
 248                 // and keep in project,
 249                 // or stored in tmp, and processed as the other linked
 250                 // resources..
 251                 URIEditorInput uriInput = new URIEditorInput(
 252                     URI.createURI( uri.toString( ) ), name );
 253                 super.init( site, uriInput );
 254                 return;
 255             }
 256             return;
 257         }
 258         super.init( site, input );
 259     }

 263      * 
 264      * @param uri

 267     private IFile obtainLink( java.net.URI uri )
 268     {
 269         try {
 270             IWorkspace ws = ResourcesPlugin.getWorkspace( );
 271             // get, or create project if non existing
 272             IProject project = ws.getRoot( ).getProject( AUTOLINK_PROJECT_NAME );
 273             boolean newProject = false;
 274             if( ! project.exists( ) ) {
 275                 project.create( null );
 276                 newProject = true;
 277             }
 278             if( ! project.isOpen( ) ) {
 279                 project.open( null );
 280                 project.setHidden( true );
 281             }
 282 
 283             if( newProject ) {
 284                 project.setDefaultCharset( ENCODING_UTF8,
 285                     new NullProgressMonitor( ) );
 286             }
 287 
 288             // path in project that is the same as the external file's path
 289             IFile linkFile = project.getFile( uri.getPath( ) );
 290             if( linkFile.exists( ) ) {
 291                 linkFile.refreshLocal( 1, null ); // don't know if needed (or
 292             }
 293             else {
 294                 // create the link
 295                 createLink( project, linkFile, uri );
 296                 // linkFile.createLink(uri, IResource.ALLOW_MISSING_LOCAL,
 297                 // null);
 298             }
 299             return linkFile;
 300 
 301             // IPath location = new Path(name);
 302             // IFile file = project.getFile(location.lastSegment());
 303             // file.createLink(location, IResource.NONE, null);
 304         } catch( CoreException e ) {
 305             throw new WrappedException( e );
 306         }

 309     // SaveAs support for linked files - saves them on local disc, not to
 310     // workspace if file is in special
 311     // hidden external file link project.
 312     @Override
 313     protected void performSaveAs( IProgressMonitor progressMonitor )
 314     {
 315 
 316         Shell shell = getSite( ).getShell( );
 317         final IEditorInput input = getEditorInput( );
 318 
 319         // Customize save as if the file is linked, and it is in the special
 320         // external link project
 321         //
 322         if( input instanceof IFileEditorInput
 323             && ( ( IFileEditorInput ) input ).getFile( ).isLinked( )
 324             && ( ( IFileEditorInput ) input ).getFile( ).getProject( )
 325                 .getName( ).equals( AUTOLINK_PROJECT_NAME ) ) {
 326             final IEditorInput newInput;
 327             IDocumentProvider provider = getDocumentProvider( );
 328 
 329             FileDialog dialog = new FileDialog( shell, SWT.SAVE );
 330             IPath oldPath = URIUtil.toPath( ( ( IURIEditorInput ) input )
 331                 .getURI( ) );
 332             if( oldPath != null ) {
 333                 dialog.setFileName( oldPath.lastSegment( ) );
 334                 dialog.setFilterPath( oldPath.toOSString( ) );
 335             }
 336 
 337             dialog.setFilterExtensions( new String[] { "*.b3", "*.*" } ); //$NON-NLS-1$ //$NON-NLS-2$
 338             String path = dialog.open( );
 339             if( path == null ) {
 340                 if( progressMonitor != null ) {
 341                     progressMonitor.setCanceled( true );
 342                 }
 343                 return;
 344             }
 345 
 346             // Check whether file exists and if so, confirm overwrite
 347             final File localFile = new File( path );
 348             if( localFile.exists( ) ) {
 349                 MessageDialog overwriteDialog = new MessageDialog( shell,
 350                     Messages.WMLEditor_0, null,
 351                     path + Messages.WMLEditor_1, MessageDialog.WARNING,
 352                     new String[] { IDialogConstants.YES_LABEL,
 353                         IDialogConstants.NO_LABEL }, 1 ); // 'No' is the
 354                                                           // default
 355                 if( overwriteDialog.open( ) != Window.OK ) {
 356                     if( progressMonitor != null ) {
 357                         progressMonitor.setCanceled( true );
 358                         return;
 359                     }
 360                 }
 361             }
 362 
 363             IFileStore fileStore;
 364             try {
 365                 fileStore = EFS.getStore( localFile.toURI( ) );
 366             } catch( CoreException ex ) {
 367                 Logger.getInstance( ).logException( ex );
 368                 String title = Messages.WMLEditor_2;
 369                 String msg = Messages.WMLEditor_3 + ex.getMessage( );
 370                 MessageDialog.openError( shell, title, msg );
 371                 return;
 372             }
 373 
 374             IFile file = getWorkspaceFile( fileStore );
 375             if( file != null ) {
 376                 newInput = new FileEditorInput( file );
 377             }
 378             else {
 379                 IURIEditorInput uriInput = new FileStoreEditorInput( fileStore );
 380                 java.net.URI uri = uriInput.getURI( );
 381                 IFile linkedFile = obtainLink( uri );
 382 
 383                 newInput = new FileEditorInput( linkedFile );
 384             }
 385 
 386             if( provider == null ) {
 387                 // editor has programmatically been closed while the dialog was
 388                 // open
 389                 return;
 390             }
 391 
 392             boolean success = false;
 393             try {
 394 
 395                 provider.aboutToChange( newInput );
 396                 provider.saveDocument( progressMonitor, newInput,
 397                     provider.getDocument( input ), true );
 398                 success = true;
 399 
 400             } catch( CoreException x ) {
 401                 Logger.getInstance( ).logException( x );
 402                 final IStatus status = x.getStatus( );
 403                 if( status == null || status.getSeverity( ) != IStatus.CANCEL ) {
 404                     String title = Messages.WMLEditor_4;
 405                     String msg = Messages.WMLEditor_5 + x.getMessage( );
 406                     MessageDialog.openError( shell, title, msg );
 407                 }
 408             } finally {
 409                 provider.changed( newInput );
 410                 if( success ) {
 411                     setInput( newInput );
 412                 }
 413             }
 414 
 415             if( progressMonitor != null ) {
 416                 progressMonitor.setCanceled( ! success );
 417             }
 418 
 419             return;
 420         }
 421 
 422         super.performSaveAs( progressMonitor );
 423     }
