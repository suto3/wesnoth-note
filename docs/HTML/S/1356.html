 172                         } else {

 174                         }

 176                         if($char eq "\\") {

 178                         } elsif($char eq '"') {

 182                         }
 183                 } elsif($state eq 'VALUE_BACKSLASH') {
 184                         $token .= $char;
 185                         $state = 'VALUE_QUOTE';

 239 sub write_word
 240 {
 241         my $res = '';
 242         my ($schema,$schema_lookup,$word) = @_;
 243         if(exists $schema_lookup->{$word}) {
 244                 return chr($wml::first_word + $schema_lookup->{$word});
 245         }
 246 
 247         my $pos = scalar(@$schema);
 248         if($pos < $wml::max_schema_size and length($word) < $wml::max_schema_item_length) {
 249                 push @$schema, $word;
 250                 $schema_lookup->{$word} = $pos;
 251                 return chr($wml::schema_item) . &write_word_literal($word) . chr($wml::first_word + $pos);
 252         } else {
 253                 return chr($wml::literal_word) . &write_word_literal($word);
 254         }

 385         my $attr2 = $doc2->{'attr'};

 389

 408 sub get_diff
 409 {
 410         my ($doc1, $doc2, $diff_name) = @_;
 411         my $diffs = {'name' => $diff_name, 'children' => [], 'attr' => {}};
 412         my $attr1 = $doc1->{'attr'};
 413         my $attr2 = $doc2->{'attr'};
 414         my $d = 0;
 415         my @keys1 = keys %$attr1;
 416         my @keys2 = keys %$attr2;
 417 
 418         foreach my $key (@keys1) {
 419                 if($attr1->{$key} ne $attr2->{$key}) {
 420                         my $child = {'name' => 'insert', 'children' => [], 'attr' => {$key => $attr1->{$key}}};
 421                         push @{$diffs->{'children'}}, $child;
 422                         ++$d;
 423                 }
 424         }
 425 
 426         foreach my $key (@keys2) {
 427                 if(undef $attr1->{$key}) {
 428                         my $child = {'name' => 'delete', 'children' => [], 'attr' => {$key => 'x'}};
 429                         push @{$diffs->{'children'}}, $child;
 430                         ++$d;
 431                 }
 432         }
 433 
 434         my $children1 = $doc1->{'children'};
 435         my $children2 = $doc2->{'children'};
 436 
 437         my %childmap1 = {};
 438         my %childmap2 = {};
 439         my %entities = {};
 440 
 441         foreach my $child (@$children1) {
 442                 my $a = $childmap1{$child->{'name'}};
 443                 if(not $a) {
 444                         $a = [$child];
 445                         $childmap1{$child->{'name'}} = $a;
 446                         $entities{$child->{'name'}} = 1;
 447                 } else {
 448                         push @$a, $child;
 449                 }
 450         }
 451 
 452         foreach my $child (@$children2) {
 453                 my $a = $childmap2{$child->{'name'}};
 454                 if(not $a) {
 455                         $a = [$child];
 456                         $childmap2{$child->{'name'}} = $a;
 457                         $entities{$child->{'name'}} = 1;
 458                 } else {
 459                         push @$a, $child;
 460                 }
 461         }
 462 
 463         foreach my $key (keys %entities) {
 464                 my $ndeletes = 0;
 465                 my $a1 = ($childmap1{$key} or []);
 466                 my $a2 = ($childmap2{$key} or []);
 467                 for(my $n = 0; $n < @$a1 or $n < @$a2; ++$n) {
 468                         if($n < @$a1 and $n < @$a2) {
 469                                 if(not &docs_equal($a1->[$n], $a2->[$n])) {
 470                                         my $diff = &get_diff($a1->[$n], $a2->[$n], $key);
 471                                         my $change = {'name' => 'change_child', 'children' => $diff->{'children'}, 'attr' => {'index' => $n}};
 472                                         push @{$diffs->{'children'}}, $change;
 473                                         ++$d;
 474                                 }
 475                         } elsif($n < @$a1) {
 476                                 my $insert = {'name' => 'insert_child', 'children' => [$a1->[$n]], 'attr' => {'index' => $n}};
 477 
 478                                 push @{$diffs->{'children'}}, $insert;
 479                                 ++$d;
 480                         } else {
 481                                 my $attr = $a2->[$n];
 482                                 my $delete = {'name' => 'delete_child', 'children' => [{'name' => $attr->{'name'}, 'children' => [], 'attr' => {}}], 'attr' => {'index' => $n - $ndeletes}};
 483                                 ++$ndeletes;
 484                                 push @{$diffs->{'children'}}, $delete;
 485                                 ++$d;
 486                         }
 487                 }
 488         }
 489 
 490         return undef unless $d;
 491         return {'name' => '', 'children' => [$diffs], 'attr' => {}};
 492 }
