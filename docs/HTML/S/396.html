 106         struct filter_transform
 107         {
 108                 explicit filter_transform(const std::vector<std::string>& filtertext) : filtertext_(filtertext) {}
 109                 bool operator()(const config& cfg) const
 110                 {
 111                         for(const auto& filter : filtertext_)
 112                         {
 113                                 bool found = false;
 114                                 for(const auto& attribute : cfg.attribute_range())
 115                                 {
 116                                         std::string val = attribute.second.str();
 117                                         if(std::search(val.begin(),
 118                                                 val.end(),
 119                                                 filter.begin(),
 120                                                 filter.end(),
 121                                                 chars_equal_insensitive)
 122                                                 != val.end())
 123                                         {
 124                                                 found = true;
 125                                                 break;
 126                                         }
 127                                 }
 128                                 if(!found) {
 129                                         return false;
 130                                 }
 131                         }
 132                         return true;
 133                 }
 134                 const std::vector<std::string> filtertext_;
 135         };

 137         /**
 138          * Retrieves an element from the given associative container or dies in some
 139          * way.
 140          *
 141          * It fails an @a assert() check or throws an exception if the requested element
 142          * does not exist.
 143          *
 144          * @return An element from the container that is guranteed to have existed
 145          *         before running this function.
 146          */
 147         template <typename MapT>
 148         typename MapT::mapped_type const& const_at(typename MapT::key_type const& key,
 149                                                                                    MapT const& map)
 150         {
 151                 typename MapT::const_iterator it = map.find(key);
 152                 if(it == map.end()) {
 153                         assert(it != map.end());
 154                         throw std::out_of_range(
 155                                         "const_at()"); // Shouldn't get here without disabling assert()
 156                 }
 157                 return it->second;
 158         }

 160         std::string make_display_dependencies(
 161                 const std::string& addon_id,
 162                 const addons_list& addons_list,
 163                 const addons_tracking_list& addon_states)
 164         {
 165                 const addon_info& addon = const_at(addon_id, addons_list);
 166                 std::string str;
 167 
 168                 const std::set<std::string>& deps = addon.resolve_dependencies(addons_list);
 169 
 170                 for(const auto& dep_id : deps) {
 171                         addon_info dep;
 172                         addon_tracking_info depstate;
 173 
 174                         addons_list::const_iterator ali = addons_list.find(dep_id);
 175                         addons_tracking_list::const_iterator tli = addon_states.find(dep_id);
 176 
 177                         if(ali == addons_list.end()) {
 178                                 dep.id = dep_id; // Build dummy addon_info.
 179                         } else {
 180                                 dep = ali->second;
 181                         }
 182 
 183                         if(tli == addon_states.end()) {
 184                                 depstate = get_addon_tracking_info(dep);
 185                         } else {
 186                                 depstate = tli->second;
 187                         }
 188 
 189                         if(!str.empty()) {
 190                                 str += ", ";
 191                         }
 192 
 193                         str += addon_list::colorify_addon_state_string(dep.display_title(), depstate.state);
 194                 }
 195 
 196                 return str;
 197         }

 250         addon_list& addons = find_widget<addon_list>(textbox->get_window(), "addons", true);

 252         boost::dynamic_bitset<> res;

 254         const config::const_child_itors& addon_cfgs = cfg_.child_range("campaign");
 255 
 256         for(const auto& a : addons_)

 262                 });

 264                 res.push_back(filter(addon_cfg));
 265         }
 266 
 267         addons.set_addon_shown(res);

 270 static std::string describe_status_verbose(const addon_tracking_info& state)
 271 {
 272         std::string s;
 273 
 274         utils::string_map i18n_symbols {{"local_version", state.installed_version.str()}};
 275 
 276         switch(state.state) {
 277                 case ADDON_NONE:
 278                         s = !state.can_publish
 279                                 ? _("addon_state^Not installed")
 280                                 : _("addon_state^Published, not installed");
 281                         break;
 282                 case ADDON_INSTALLED:
 283                         s = !state.can_publish
 284                                 ? _("addon_state^Installed")
 285                                 : _("addon_state^Published");
 286                         break;
 287                 case ADDON_NOT_TRACKED:
 288                         s = !state.can_publish
 289                                 ? _("addon_state^Installed, not tracking local version")
 290                                 // Published add-ons often don't have local status information,
 291                                 // hence untracked. This should be considered normal.
 292                                 : _("addon_state^Published, not tracking local version");
 293                         break;
 294                 case ADDON_INSTALLED_UPGRADABLE: {
 295                         const std::string vstr = !state.can_publish
 296                                 ? _("addon_state^Installed ($local_version|), upgradable")
 297                                 : _("addon_state^Published ($local_version| installed), upgradable");
 298 
 299                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 300                 } break;
 301                 case ADDON_INSTALLED_OUTDATED: {
 302                         const std::string vstr = !state.can_publish
 303                                 ? _("addon_state^Installed ($local_version|), outdated on server")
 304                                 : _("addon_state^Published ($local_version| installed), outdated on server");
 305 
 306                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 307                 } break;
 308                 case ADDON_INSTALLED_LOCAL_ONLY:
 309                         s = !state.can_publish
 310                                 ? _("addon_state^Installed, not ready to publish")
 311                                 : _("addon_state^Ready to publish");
 312                         break;
 313                 case ADDON_INSTALLED_BROKEN:
 314                         s = !state.can_publish
 315                                 ? _("addon_state^Installed, broken")
 316                                 : _("addon_state^Published, broken");
 317                         break;
 318                 default:
 319                         s = _("addon_state^Unknown");
 320         }
 321 
 322         return addon_list::colorify_addon_state_string(s, state.state);
 323 }

 455 void addon_manager::load_addon_list(window& window)
 456 {
 457         if(need_wml_cache_refresh_) {
 458                 refresh_addon_version_info_cache();
 459         }
 460 
 461         client_.request_addons_list(cfg_);
 462         if(!cfg_) {
 463                 show_error_message(window.video(), _("An error occurred while downloading the add-ons list from the server."));
 464                 window.close();
 465         }
 466 
 467         read_addons_list(cfg_, addons_);
 468 
 469         std::vector<std::string> publishable_addons = available_addons();
 470 
 471         for(std::string id : publishable_addons) {
 472                 if(addons_.find(id) == addons_.end()) {
 473                         // Get a config from the addon's pbl file
 474                         // Note that the name= key is necessary or stuff breaks, since the filter code uses this key
 475                         // to match add-ons in the config list. It also fills in addon_info's id field.
 476                         config pbl_cfg = get_addon_pbl_info(id);
 477                         pbl_cfg["name"] = id;
 478 
 479                         // Add the add-on to the list.
 480                         addon_info addon(pbl_cfg);
 481                         addon.local_only = true;
 482                         addons_[id] = addon;
 483 
 484                         // Add the addon to the config entry
 485                         cfg_.add_child("campaign", pbl_cfg);
 486                 }
 487         }
 488 
 489         if(addons_.empty()) {
 490                 show_transient_message(window.video(), _("No Add-ons Available"), _("There are no add-ons available for download from this server."));
 491                 window.close();
 492         }
 493 
 494         addon_list& list = find_widget<addon_list>(&window, "addons", false);
 495         list.set_addons(addons_);
 496 
 497         bool has_upgradable_addons = false;
 498         for(const auto& a : addons_) {
 499                 tracking_info_[a.first] = get_addon_tracking_info(a.second);
 500 
 501                 if(tracking_info_[a.first].state == ADDON_INSTALLED_UPGRADABLE) {
 502                         has_upgradable_addons = true;
 503                 }
 504         }
 505 
 506         find_widget<button>(&window, "update_all", false).set_active(has_upgradable_addons);
 507 
 508         apply_filters(window);
 509 }

 520 boost::dynamic_bitset<> addon_manager::get_status_filter_visibility(const window& window) const
 521 {
 522         const menu_button& status_filter = find_widget<const menu_button>(&window, "install_status_filter", false);
 523         const ADDON_STATUS_FILTER selection = status_filter_types_[status_filter.get_value()].first;
 524 
 525         boost::dynamic_bitset<> res;
 526         for(const auto& a : addons_) {
 527                 const ADDON_STATUS state = tracking_info_.at(a.second.id).state;
 528 
 529                 res.push_back(
 530                         (selection == FILTER_ALL) ||
 531                         (selection == FILTER_INSTALLED     && is_installed_addon_status(state)) ||
 532                         (selection == FILTER_UPGRADABLE    && state == ADDON_INSTALLED_UPGRADABLE) ||
 533                         (selection == FILTER_NOT_INSTALLED && state == ADDON_NONE)
 534                 );
 535         }
 536 
 537         return res;
 538 }

 540 boost::dynamic_bitset<> addon_manager::get_type_filter_visibility(const window& window) const
 541 {
 542         const multimenu_button& type_filter = find_widget<const multimenu_button>(&window, "type_filter", false);
 543 
 544         boost::dynamic_bitset<> toggle_states = type_filter.get_toggle_states();
 545         if(toggle_states.none()) {
 546                 // Nothing selected. It means that *all* add-ons are shown.
 547                 boost::dynamic_bitset<> res_flipped(addons_.size());
 548                 return ~res_flipped;
 549         } else {
 550                 boost::dynamic_bitset<> res;
 551 
 552                 for(const auto& a : addons_) {
 553                         int index = std::find_if(type_filter_types_.begin(), type_filter_types_.end(),
 554                                 [&a](const std::pair<ADDON_TYPE, std::string>& entry) {
 555                                         return entry.first == a.second.type;
 556                                 }) - type_filter_types_.begin();
 557                         res.push_back(toggle_states[index]);
 558                 }
 559 
 560                 return res;
 561         }
 562 }

 583 void addon_manager::install_addon(const addon_info& addon, window& window)
 584 {
 585         addons_client::install_result result = client_.install_addon_with_checks(addons_, addon);
 586 
 587         // Take note if any wml_changes occurred
 588         need_wml_cache_refresh_ |= result.wml_changed;
 589 
 590         if(result.outcome != addons_client::install_outcome::abort) {
 591                 reload_list_and_reselect_item(addon.id, window);
 592         }
 593 }

 595 void addon_manager::uninstall_addon(const addon_info& addon, window& window)
 596 {
 597         if(have_addon_pbl_info(addon.id) || have_addon_in_vcs_tree(addon.id)) {
 598                 show_error_message(window.video(),
 599                         _("The following add-on appears to have publishing or version control information stored locally, and will not be removed:") + " " +
 600                                 addon.display_title());
 601                 return;
 602         }
 603 
 604         bool success = remove_local_addon(addon.id);
 605 
 606         if(!success) {
 607                 show_error_message(window.video(), _("The following add-on could not be deleted properly:") + " " + addon.display_title());
 608         } else {
 609                 need_wml_cache_refresh_ = true;
 610 
 611                 reload_list_and_reselect_item(addon.id, window);
 612         }
 613 }

 638 /** Performs all backend and UI actions for publishing the specified add-on. */
 639 void addon_manager::publish_addon(const addon_info& addon, window& window)
 640 {
 641         std::string server_msg;
 642 
 643         const std::string addon_id = addon.id;
 644         config cfg = get_addon_pbl_info(addon_id);
 645 
 646         const version_info& version_to_publish = cfg["version"].str();
 647 
 648         if(version_to_publish <= tracking_info_[addon_id].remote_version) {
 649                 const int res = gui2::show_message(window.video(), _("Warning"),
 650                         _("The remote version of this add-on is greater or equal to the version being uploaded. Do you really wish to continue?"),
 651                         gui2::dialogs::message::yes_no_buttons);
 652 
 653                 if(res != gui2::window::OK) {
 654                         return;
 655                 }
 656         }
 657 
 658         if(!::image::exists(cfg["icon"].str())) {
 659                 gui2::show_error_message(window.video(), _("Invalid icon path. Make sure the path points to a valid image."));
 660         } else if(!client_.request_distribution_terms(server_msg)) {
 661                 gui2::show_error_message(window.video(),
 662                         _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 663         } else if(gui2::show_message(window.video(), _("Terms"), server_msg, gui2::dialogs::message::ok_cancel_buttons) == gui2::window::OK) {
 664                 if(!client_.upload_addon(addon_id, server_msg, cfg)) {
 665                         gui2::show_error_message(window.video(),
 666                                 _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 667                 } else {
 668                         gui2::show_transient_message(window.video(), _("Response"), server_msg);
 669                 }
 670         }
 671 }

 673 /** Performs all backend and UI actions for taking down the specified add-on. */
 674 void addon_manager::delete_addon(const addon_info& addon, window& window)
 675 {
 676         const std::string addon_id = addon.id;
 677         const std::string& text = vgettext(
 678                 "Deleting '$addon|' will permanently erase its download and upload counts on the add-ons server. Do you really wish to continue?",
 679                 {{"addon", make_addon_title(addon_id)}} // FIXME: need the real title!
 680         );
 681 
 682         const int res = gui2::show_message(
 683                 window.video(), _("Confirm"), text, gui2::dialogs::message::yes_no_buttons);
 684 
 685         if(res != gui2::window::OK) {
 686                 return;
 687         }
 688 
 689         std::string server_msg;
 690         if(!client_.delete_remote_addon(addon_id, server_msg)) {
 691                 gui2::show_error_message(window.video(), _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 692         } else {
 693                 // FIXME: translation needed!
 694                 gui2::show_transient_message(window.video(), _("Response"), server_msg);
 695         }
 696 }

 698 /** Called when the player double-clicks an add-on. */
 699 void addon_manager::execute_default_action(const addon_info& addon, window& window)
 700 {
 701         switch(tracking_info_[addon.id].state) {
 702                 case ADDON_NONE:
 703                         install_addon(addon, window);
 704                         break;
 705                 case ADDON_INSTALLED:
 706                         if(!tracking_info_[addon.id].can_publish) {
 707                                 utils::string_map symbols{ { "addon", addon.display_title() } };
 708                                 int res = gui2::show_message(window.video(), _("Uninstall add-on"),
 709                                         vgettext("Do you want to uninstall '$addon|'?", symbols),
 710                                         gui2::dialogs::message::ok_cancel_buttons);
 711                                 if(res == gui2::window::OK) {
 712                                         uninstall_addon(addon, window);
 713                                 }
 714                         }
 715                         break;
 716                 case ADDON_INSTALLED_UPGRADABLE:
 717                         update_addon(addon, window);
 718                         break;
 719                 case ADDON_INSTALLED_LOCAL_ONLY:
 720                 case ADDON_INSTALLED_OUTDATED:
 721                         publish_addon(addon, window);
 722                         break;
 723                 default:
 724                         break;
 725         }
 726 }

 744 static std::string format_addon_time(time_t time)
 745 {
 746         if(time) {
 747                 std::ostringstream ss;
 748 
 749                 const char* format = preferences::use_twelve_hour_clock_format()
 750                         ? "%Y-%m-%d %I:%M %p"
 751                         : "%Y-%m-%d %H:%M";
 752 
 753                 ss << utils::put_time(std::localtime(&time), format);
 754 
 755                 return ss.str();
 756         }
 757 
 758         return font::unicode_em_dash;
 759 }

 761 void addon_manager::on_addon_select(window& window)
 762 {
 763         const addon_info* info = find_widget<addon_list>(&window, "addons", false).get_selected_addon();
 764 
 765         if(info == nullptr) {
 766                 return;
 767         }
 768 
 769         widget* parent = &window;
 770         if(stacked_widget* stk = find_widget<stacked_widget>(&window, "main_stack", false, false)) {
 771                 parent = stk->get_layer_grid(1);
 772         }
 773 
 774         find_widget<drawing>(parent, "image", false).set_label(info->display_icon());
 775 
 776         find_widget<styled_widget>(parent, "title", false).set_label(info->display_title());
 777         find_widget<styled_widget>(parent, "description", false).set_label(info->description);
 778         find_widget<styled_widget>(parent, "version", false).set_label(info->version.str());
 779         find_widget<styled_widget>(parent, "author", false).set_label(info->author);
 780         find_widget<styled_widget>(parent, "type", false).set_label(info->display_type());
 781 
 782         styled_widget& status = find_widget<styled_widget>(parent, "status", false);
 783         status.set_label(describe_status_verbose(tracking_info_[info->id]));
 784         status.set_use_markup(true);
 785 
 786         find_widget<styled_widget>(parent, "size", false).set_label(size_display_string(info->size));
 787         find_widget<styled_widget>(parent, "downloads", false).set_label(std::to_string(info->downloads));
 788         find_widget<styled_widget>(parent, "created", false).set_label(format_addon_time(info->created));
 789         find_widget<styled_widget>(parent, "updated", false).set_label(format_addon_time(info->updated));
 790 
 791         find_widget<styled_widget>(parent, "dependencies", false).set_label(!info->depends.empty()
 792                 ? make_display_dependencies(info->id, addons_, tracking_info_)
 793                 : _("None"));
 794 
 795         std::string languages;
 796 
 797         for(const auto& lc : info->locales) {
 798                 const std::string& langlabel = langcode_to_string(lc);
 799                 if(!langlabel.empty()) {
 800                         if(!languages.empty()) {
 801                                 languages += ", ";
 802                         }
 803                         languages += langlabel;
 804                 }
 805         }
 806 
 807         find_widget<styled_widget>(parent, "translations", false).set_label(!languages.empty() ? languages : _("None"));
 808 
 809         const std::string& feedback_url = info->feedback_url;
 810 
 811         if(!feedback_url.empty()) {
 812                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(1);
 813                 find_widget<text_box>(parent, "url", false).set_value(feedback_url);
 814         } else {
 815                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(0);
 816         }
 817 
 818         bool installed = is_installed_addon_status(tracking_info_[info->id].state);
 819         bool updatable = tracking_info_[info->id].state == ADDON_INSTALLED_UPGRADABLE;
 820 
 821         stacked_widget& action_stack = find_widget<stacked_widget>(parent, "action_stack", false);
 822 
 823         if(!tracking_info_[info->id].can_publish) {
 824                 action_stack.select_layer(0);
 825 
 826                 stacked_widget& install_update_stack = find_widget<stacked_widget>(parent, "install_update_stack", false);
 827                 install_update_stack.select_layer(updatable ? 1 : 0);
 828 
 829                 if(!updatable) {
 830                         find_widget<button>(parent, "install", false).set_active(!installed);
 831                 } else {
 832                         find_widget<button>(parent, "update", false).set_active(true);
 833                 }
 834 
 835                 find_widget<button>(parent, "uninstall", false).set_active(installed);
 836         } else {
 837                 action_stack.select_layer(1);
 838 
 839                 // TODO: are these the right flags to check?
 840                 find_widget<button>(parent, "publish", false).set_active(installed);
 841                 find_widget<button>(parent, "delete", false).set_active(!installed);
 842         }
 843 }
