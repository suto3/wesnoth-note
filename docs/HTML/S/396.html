 105         struct filter_transform
 106         {
 107                 explicit filter_transform(const std::vector<std::string>& filtertext) : filtertext_(filtertext) {}
 108                 bool operator()(const config& cfg) const
 109                 {
 110                         for(const auto& filter : filtertext_)
 111                         {
 112                                 bool found = false;
 113                                 for(const auto& attribute : cfg.attribute_range())
 114                                 {
 115                                         std::string val = attribute.second.str();
 116                                         if(std::search(val.begin(),
 117                                                 val.end(),
 118                                                 filter.begin(),
 119                                                 filter.end(),
 120                                                 chars_equal_insensitive)
 121                                                 != val.end())
 122                                         {
 123                                                 found = true;
 124                                                 break;
 125                                         }
 126                                 }
 127                                 if(!found) {
 128                                         return false;
 129                                 }
 130                         }
 131                         return true;
 132                 }
 133                 const std::vector<std::string> filtertext_;
 134         };

 136         /**
 137          * Retrieves an element from the given associative container or dies in some
 138          * way.
 139          *
 140          * It fails an @a assert() check or throws an exception if the requested element
 141          * does not exist.
 142          *
 143          * @return An element from the container that is guranteed to have existed
 144          *         before running this function.
 145          */
 146         template <typename MapT>
 147         typename MapT::mapped_type const& const_at(typename MapT::key_type const& key,
 148                                                                                    MapT const& map)
 149         {
 150                 typename MapT::const_iterator it = map.find(key);
 151                 if(it == map.end()) {
 152                         assert(it != map.end());
 153                         throw std::out_of_range(
 154                                         "const_at()"); // Shouldn't get here without disabling assert()
 155                 }
 156                 return it->second;
 157         }

 159         std::string make_display_dependencies(
 160                 const std::string& addon_id,
 161                 const addons_list& addons_list,
 162                 const addons_tracking_list& addon_states)
 163         {
 164                 const addon_info& addon = const_at(addon_id, addons_list);
 165                 std::string str;
 166 
 167                 const std::set<std::string>& deps = addon.resolve_dependencies(addons_list);
 168 
 169                 for(const auto& dep_id : deps) {
 170                         addon_info dep;
 171                         addon_tracking_info depstate;
 172 
 173                         addons_list::const_iterator ali = addons_list.find(dep_id);
 174                         addons_tracking_list::const_iterator tli = addon_states.find(dep_id);
 175 
 176                         if(ali == addons_list.end()) {
 177                                 dep.id = dep_id; // Build dummy addon_info.
 178                         } else {
 179                                 dep = ali->second;
 180                         }
 181 
 182                         if(tli == addon_states.end()) {
 183                                 depstate = get_addon_tracking_info(dep);
 184                         } else {
 185                                 depstate = tli->second;
 186                         }
 187 
 188                         if(!str.empty()) {
 189                                 str += ", ";
 190                         }
 191 
 192                         str += addon_list::colorify_addon_state_string(dep.display_title(), depstate.state);
 193                 }
 194 
 195                 return str;
 196         }

 253 static std::string describe_status_verbose(const addon_tracking_info& state)
 254 {
 255         std::string s;
 256 
 257         utils::string_map i18n_symbols {{"local_version", state.installed_version.str()}};
 258 
 259         switch(state.state) {
 260                 case ADDON_NONE:
 261                         s = !state.can_publish
 262                                 ? _("addon_state^Not installed")
 263                                 : _("addon_state^Published, not installed");
 264                         break;
 265                 case ADDON_INSTALLED:
 266                         s = !state.can_publish
 267                                 ? _("addon_state^Installed")
 268                                 : _("addon_state^Published");
 269                         break;
 270                 case ADDON_NOT_TRACKED:
 271                         s = !state.can_publish
 272                                 ? _("addon_state^Installed, not tracking local version")
 273                                 // Published add-ons often don't have local status information,
 274                                 // hence untracked. This should be considered normal.
 275                                 : _("addon_state^Published, not tracking local version");
 276                         break;
 277                 case ADDON_INSTALLED_UPGRADABLE: {
 278                         const std::string vstr = !state.can_publish
 279                                 ? _("addon_state^Installed ($local_version|), upgradable")
 280                                 : _("addon_state^Published ($local_version| installed), upgradable");
 281 
 282                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 283                 } break;
 284                 case ADDON_INSTALLED_OUTDATED: {
 285                         const std::string vstr = !state.can_publish
 286                                 ? _("addon_state^Installed ($local_version|), outdated on server")
 287                                 : _("addon_state^Published ($local_version| installed), outdated on server");
 288 
 289                         s = utils::interpolate_variables_into_string(vstr, &i18n_symbols);
 290                 } break;
 291                 case ADDON_INSTALLED_LOCAL_ONLY:
 292                         s = !state.can_publish
 293                                 ? _("addon_state^Installed, not ready to publish")
 294                                 : _("addon_state^Ready to publish");
 295                         break;
 296                 case ADDON_INSTALLED_BROKEN:
 297                         s = !state.can_publish
 298                                 ? _("addon_state^Installed, broken")
 299                                 : _("addon_state^Published, broken");
 300                         break;
 301                 default:
 302                         s = _("addon_state^Unknown");
 303         }
 304 
 305         return addon_list::colorify_addon_state_string(s, state.state);
 306 }

 442 void addon_manager::load_addon_list(window& window)
 443 {
 444         if(need_wml_cache_refresh_) {
 445                 refresh_addon_version_info_cache();
 446         }
 447 
 448         read_addons_list(cfg_, addons_);
 449 
 450         std::vector<std::string> publishable_addons = available_addons();
 451 
 452         for(std::string id : publishable_addons) {
 453                 if(addons_.find(id) == addons_.end()) {
 454                         // Get a config from the addon's pbl file
 455                         // Note that the name= key is necessary or stuff breaks, since the filter code uses this key
 456                         // to match add-ons in the config list. It also fills in addon_info's id field.
 457                         config pbl_cfg = get_addon_pbl_info(id);
 458                         pbl_cfg["name"] = id;
 459 
 460                         // Add the add-on to the list.
 461                         addon_info addon(pbl_cfg);
 462                         addon.local_only = true;
 463                         addons_[id] = addon;
 464 
 465                         // Add the addon to the config entry
 466                         cfg_.add_child("campaign", pbl_cfg);
 467                 }
 468         }
 469 
 470         if(addons_.empty()) {
 471                 show_transient_message(window.video(), _("No Add-ons Available"), _("There are no add-ons available for download from this server."));
 472                 window.close();
 473         }
 474 
 475         addon_list& list = find_widget<addon_list>(&window, "addons", false);
 476         list.set_addons(addons_);
 477 
 478         bool has_upgradable_addons = false;
 479         for(const auto& a : addons_) {
 480                 tracking_info_[a.first] = get_addon_tracking_info(a.second);
 481 
 482                 if(tracking_info_[a.first].state == ADDON_INSTALLED_UPGRADABLE) {
 483                         has_upgradable_addons = true;
 484                 }
 485         }
 486 
 487         find_widget<button>(&window, "update_all", false).set_active(has_upgradable_addons);
 488 
 489         apply_filters(window);
 490 }

 501 boost::dynamic_bitset<> addon_manager::get_name_filter_visibility(const window& window) const
 502 {
 503         const text_box& name_filter = find_widget<const text_box>(&window, "filter", false);
 504         const std::string& text = name_filter.get_value();
 505 
 506         filter_transform filter(utils::split(text, ' '));
 507         boost::dynamic_bitset<> res;
 508 
 509         const config::const_child_itors& addon_cfgs = cfg_.child_range("campaign");
 510 
 511         for(const auto& a : addons_)
 512         {
 513                 const config& addon_cfg = *std::find_if(addon_cfgs.begin(), addon_cfgs.end(),
 514                         [&a](const config& cfg)
 515                 {
 516                         return cfg["name"] == a.first;
 517                 });
 518 
 519                 res.push_back(filter(addon_cfg));
 520         }
 521 
 522         return res;
 523 }

 525 boost::dynamic_bitset<> addon_manager::get_status_filter_visibility(const window& window) const
 526 {
 527         const menu_button& status_filter = find_widget<const menu_button>(&window, "install_status_filter", false);
 528         const ADDON_STATUS_FILTER selection = status_filter_types_[status_filter.get_value()].first;
 529 
 530         boost::dynamic_bitset<> res;
 531         for(const auto& a : addons_) {
 532                 const addon_tracking_info& info = tracking_info_.at(a.second.id);
 533 
 534                 res.push_back(
 535                         (selection == FILTER_ALL) ||
 536                         (selection == FILTER_INSTALLED     && is_installed_addon_status(info.state)) ||
 537                         (selection == FILTER_UPGRADABLE    && info.state == ADDON_INSTALLED_UPGRADABLE) ||
 538                         (selection == FILTER_PUBLISHABLE   && info.can_publish == true) ||
 539                         (selection == FILTER_NOT_INSTALLED && info.state == ADDON_NONE)
 540                 );
 541         }
 542 
 543         return res;
 544 }

 546 boost::dynamic_bitset<> addon_manager::get_type_filter_visibility(const window& window) const
 547 {
 548         const multimenu_button& type_filter = find_widget<const multimenu_button>(&window, "type_filter", false);
 549 
 550         boost::dynamic_bitset<> toggle_states = type_filter.get_toggle_states();
 551         if(toggle_states.none()) {
 552                 // Nothing selected. It means that *all* add-ons are shown.
 553                 boost::dynamic_bitset<> res_flipped(addons_.size());
 554                 return ~res_flipped;
 555         } else {
 556                 boost::dynamic_bitset<> res;
 557 
 558                 for(const auto& a : addons_) {
 559                         int index = std::find_if(type_filter_types_.begin(), type_filter_types_.end(),
 560                                 [&a](const std::pair<ADDON_TYPE, std::string>& entry) {
 561                                         return entry.first == a.second.type;
 562                                 }) - type_filter_types_.begin();
 563                         res.push_back(toggle_states[index]);
 564                 }
 565 
 566                 return res;
 567         }
 568 }

 579 template<void(addon_manager::*fptr)(const addon_info& addon, window& window)>
 580 void addon_manager::execute_action_on_selected_addon(window& window)
 581 {
 582         addon_list& addons = find_widget<addon_list>(&window, "addons", false);
 583         const addon_info* addon = addons.get_selected_addon();
 584 
 585         if(addon == nullptr) {
 586                 return;
 587         }
 588 
 589         (this->*fptr)(*addon, window);
 590 }

 592 void addon_manager::install_addon(const addon_info& addon, window& window)
 593 {
 594         addons_client::install_result result = client_.install_addon_with_checks(addons_, addon);
 595 
 596         // Take note if any wml_changes occurred
 597         need_wml_cache_refresh_ |= result.wml_changed;
 598 
 599         if(result.outcome != addons_client::install_outcome::abort) {
 600                 reload_list_and_reselect_item(addon.id, window);
 601         }
 602 }

 604 void addon_manager::uninstall_addon(const addon_info& addon, window& window)
 605 {
 606         if(have_addon_pbl_info(addon.id) || have_addon_in_vcs_tree(addon.id)) {
 607                 show_error_message(window.video(),
 608                         _("The following add-on appears to have publishing or version control information stored locally, and will not be removed:") + " " +
 609                                 addon.display_title());
 610                 return;
 611         }
 612 
 613         bool success = remove_local_addon(addon.id);
 614 
 615         if(!success) {
 616                 show_error_message(window.video(), _("The following add-on could not be deleted properly:") + " " + addon.display_title());
 617         } else {
 618                 need_wml_cache_refresh_ = true;
 619 
 620                 reload_list_and_reselect_item(addon.id, window);
 621         }
 622 }

 648 /** Performs all backend and UI actions for publishing the specified add-on. */
 649 void addon_manager::publish_addon(const addon_info& addon, window& window)
 650 {
 651         std::string server_msg;
 652 
 653         const std::string addon_id = addon.id;
 654         config cfg = get_addon_pbl_info(addon_id);
 655 
 656         const version_info& version_to_publish = cfg["version"].str();
 657 
 658         if(version_to_publish <= tracking_info_[addon_id].remote_version) {
 659                 const int res = gui2::show_message(window.video(), _("Warning"),
 660                         _("The remote version of this add-on is greater or equal to the version being uploaded. Do you really wish to continue?"),
 661                         gui2::dialogs::message::yes_no_buttons);
 662 
 663                 if(res != gui2::window::OK) {
 664                         return;
 665                 }
 666         }
 667 
 668         if(!::image::exists(cfg["icon"].str())) {
 669                 gui2::show_error_message(window.video(), _("Invalid icon path. Make sure the path points to a valid image."));
 670         } else if(!client_.request_distribution_terms(server_msg)) {
 671                 gui2::show_error_message(window.video(),
 672                         _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 673         } else if(gui2::show_message(window.video(), _("Terms"), server_msg, gui2::dialogs::message::ok_cancel_buttons) == gui2::window::OK) {
 674                 if(!client_.upload_addon(addon_id, server_msg, cfg)) {
 675                         gui2::show_error_message(window.video(),
 676                                 _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 677                 } else {
 678                         gui2::show_transient_message(window.video(), _("Response"), server_msg);
 679                         fetch_addons_list(window);
 680                         reload_list_and_reselect_item(addon_id, window);
 681                 }
 682         }
 683 }

 685 /** Performs all backend and UI actions for taking down the specified add-on. */
 686 void addon_manager::delete_addon(const addon_info& addon, window& window)
 687 {
 688         const std::string addon_id = addon.id;
 689         const std::string& text = vgettext(
 690                 "Deleting '$addon|' will permanently erase its download and upload counts on the add-ons server. Do you really wish to continue?",
 691                 {{"addon", make_addon_title(addon_id)}} // FIXME: need the real title!
 692         );
 693 
 694         const int res = gui2::show_message(
 695                 window.video(), _("Confirm"), text, gui2::dialogs::message::yes_no_buttons);
 696 
 697         if(res != gui2::window::OK) {
 698                 return;
 699         }
 700 
 701         std::string server_msg;
 702         if(!client_.delete_remote_addon(addon_id, server_msg)) {
 703                 gui2::show_error_message(window.video(), _("The server responded with an error:") + "\n" + client_.get_last_server_error());
 704         } else {
 705                 // FIXME: translation needed!
 706                 gui2::show_transient_message(window.video(), _("Response"), server_msg);
 707                 fetch_addons_list(window);
 708                 reload_list_and_reselect_item(addon_id, window);
 709         }

 712 /** Called when the player double-clicks an add-on. */
 713 void addon_manager::execute_default_action(const addon_info& addon, window& window)
 714 {
 715         switch(tracking_info_[addon.id].state) {
 716                 case ADDON_NONE:
 717                         install_addon(addon, window);
 718                         break;
 719                 case ADDON_INSTALLED:
 720                         if(!tracking_info_[addon.id].can_publish) {
 721                                 utils::string_map symbols{ { "addon", addon.display_title() } };
 722                                 int res = gui2::show_message(window.video(), _("Uninstall add-on"),
 723                                         vgettext("Do you want to uninstall '$addon|'?", symbols),
 724                                         gui2::dialogs::message::ok_cancel_buttons);
 725                                 if(res == gui2::window::OK) {
 726                                         uninstall_addon(addon, window);
 727                                 }
 728                         }
 729                         break;
 730                 case ADDON_INSTALLED_UPGRADABLE:
 731                         update_addon(addon, window);
 732                         break;
 733                 case ADDON_INSTALLED_LOCAL_ONLY:
 734                 case ADDON_INSTALLED_OUTDATED:
 735                         publish_addon(addon, window);
 736                         break;
 737                 default:
 738                         break;
 739         }
 740 }

 758 static std::string format_addon_time(time_t time)
 759 {
 760         if(time) {
 761                 std::ostringstream ss;
 762 
 763                 const char* format = preferences::use_twelve_hour_clock_format()
 764                         ? "%Y-%m-%d %I:%M %p"
 765                         : "%Y-%m-%d %H:%M";
 766 
 767                 ss << utils::put_time(std::localtime(&time), format);
 768 
 769                 return ss.str();
 770         }
 771 
 772         return font::unicode_em_dash;
 773 }

 775 void addon_manager::on_addon_select(window& window)
 776 {
 777         const addon_info* info = find_widget<addon_list>(&window, "addons", false).get_selected_addon();
 778 
 779         if(info == nullptr) {
 780                 return;
 781         }
 782 
 783         widget* parent = &window;
 784         if(stacked_widget* stk = find_widget<stacked_widget>(&window, "main_stack", false, false)) {
 785                 parent = stk->get_layer_grid(1);
 786         }
 787 
 788         find_widget<drawing>(parent, "image", false).set_label(info->display_icon());
 789 
 790         find_widget<styled_widget>(parent, "title", false).set_label(info->display_title());
 791         find_widget<styled_widget>(parent, "description", false).set_label(info->description);
 792         find_widget<styled_widget>(parent, "version", false).set_label(info->version.str());
 793         find_widget<styled_widget>(parent, "author", false).set_label(info->author);
 794         find_widget<styled_widget>(parent, "type", false).set_label(info->display_type());
 795 
 796         styled_widget& status = find_widget<styled_widget>(parent, "status", false);
 797         status.set_label(describe_status_verbose(tracking_info_[info->id]));
 798         status.set_use_markup(true);
 799 
 800         find_widget<styled_widget>(parent, "size", false).set_label(size_display_string(info->size));
 801         find_widget<styled_widget>(parent, "downloads", false).set_label(std::to_string(info->downloads));
 802         find_widget<styled_widget>(parent, "created", false).set_label(format_addon_time(info->created));
 803         find_widget<styled_widget>(parent, "updated", false).set_label(format_addon_time(info->updated));
 804 
 805         find_widget<styled_widget>(parent, "dependencies", false).set_label(!info->depends.empty()
 806                 ? make_display_dependencies(info->id, addons_, tracking_info_)
 807                 : _("None"));
 808 
 809         std::string languages;
 810 
 811         for(const auto& lc : info->locales) {
 812                 const std::string& langlabel = langcode_to_string(lc);
 813                 if(!langlabel.empty()) {
 814                         if(!languages.empty()) {
 815                                 languages += ", ";
 816                         }
 817                         languages += langlabel;
 818                 }
 819         }
 820 
 821         find_widget<styled_widget>(parent, "translations", false).set_label(!languages.empty() ? languages : _("None"));
 822 
 823         const std::string& feedback_url = info->feedback_url;
 824 
 825         if(!feedback_url.empty()) {
 826                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(1);
 827                 find_widget<text_box>(parent, "url", false).set_value(feedback_url);
 828         } else {
 829                 find_widget<stacked_widget>(parent, "feedback_stack", false).select_layer(0);
 830         }
 831 
 832         bool installed = is_installed_addon_status(tracking_info_[info->id].state);
 833         bool updatable = tracking_info_[info->id].state == ADDON_INSTALLED_UPGRADABLE;
 834 
 835         stacked_widget& action_stack = find_widget<stacked_widget>(parent, "action_stack", false);
 836 
 837         if(!tracking_info_[info->id].can_publish) {
 838                 action_stack.select_layer(0);
 839 
 840                 stacked_widget& install_update_stack = find_widget<stacked_widget>(parent, "install_update_stack", false);
 841                 install_update_stack.select_layer(updatable ? 1 : 0);
 842 
 843                 if(!updatable) {
 844                         find_widget<button>(parent, "install", false).set_active(!installed);
 845                 } else {
 846                         find_widget<button>(parent, "update", false).set_active(true);
 847                 }
 848 
 849                 find_widget<button>(parent, "uninstall", false).set_active(installed);
 850         } else {
 851                 action_stack.select_layer(1);
 852 
 853                 // Always enable the publish button, but disable the delete button if not yet published.
 854                 find_widget<button>(parent, "publish", false).set_active(true);
 855                 find_widget<button>(parent, "delete", false).set_active(!info->local_only);
 856         }
 857 }
