   9 package org.wesnoth.ui.syntax;
  10 
  11 import org.eclipse.emf.common.notify.impl.AdapterImpl;
  12 import org.eclipse.emf.ecore.EObject;
  13 
  14 /**
  15  * A simple WML adapter that holds a coloring id on the specified resource
  16  */
  17 public class WMLSyntaxColoringAdapter extends AdapterImpl
  18 {
  19     /**
  20      * A color id from the
  21      * {@link org.wesnoth.ui.syntax.WMLHighlightingConfiguration}
  22      */
  23     public String  ColorId;
  24 
  25     /**
  26      * The object to color
  27      */
  28     public EObject TargetEObject;
  29 
  30     /**
  31      * Creates a new {@link WMLSyntaxColoringAdapter}
  32      * 
  33      * @param id
  34      *        The color id defined in WMLHighlightingConfiguration
  35      * @param target
  36      *        The {@link EObject} to color
  37      */
  38     public WMLSyntaxColoringAdapter( String id, EObject target )
  39     {
  40         ColorId = id;
  41         TargetEObject = target;
  42     }
  43 }
