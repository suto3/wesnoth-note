  57 void statistics_dialog::pre_show(window& window)
  58 {
  59         //
  60         // Set title
  61         //
  62         label& title = find_widget<label>(&window, "title", false);
  63         title.set_label((formatter() << title.get_label() << " (" << current_team_.side_name() << ")").str());
  64 
  65         //
  66         // Set up scenario menu
  67         //
  68         std::vector<config> menu_items;
  69         for(const auto& scenario : scenarios_) {
  70                 menu_items.emplace_back(config {"label", *scenario.first});
  71         }
  72 
  73         menu_button& scenario_menu = find_widget<menu_button>(&window, "scenario_menu", false);
  74 
  75         scenario_menu.set_values(menu_items, scenario_index_);
  76         scenario_menu.connect_click_handler(std::bind(&statistics_dialog::on_scenario_select, this, std::ref(window)));
  77 
  78         //
  79         // Set up tab toggle
  80         //
  81         listbox& tab_bar = find_widget<listbox>(&window, "tab_bar", false);
  82         tab_bar.select_row(use_campaign);
  83 
  84 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  85         connect_signal_notify_modified(*tab_bar,
  86                 std::bind(&statistics_dialog::on_tab_select, *this, std::ref(window)));
  87 #else
  88         tab_bar.set_callback_value_change(
  89                 dialog_callback<statistics_dialog, &statistics_dialog::on_tab_select>);
  90 #endif
  91 
  92         //
  93         // Set up primary stats list
  94         //
  95         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
  96 
  97 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  98         connect_signal_notify_modified(*stat_list,
  99                 std::bind(&statistics_dialog::on_primary_list_select, *this, std::ref(window)));
 100 #else
 101         stat_list.set_callback_value_change(
 102                 dialog_callback<statistics_dialog, &statistics_dialog::on_primary_list_select>);
 103 #endif
 104 
 105         update_lists(window);
 106 }

 113 void statistics_dialog::add_stat_row(window& window, const std::string& type, const statistics::stats::str_int_map& value, const bool has_cost)
 114 {
 115         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
 116 
 117         std::map<std::string, string_map> data;
 118         string_map item;
 119 
 120         item["label"] = type;
 121         data.emplace("stat_type", item);
 122 
 123         item["label"] = std::to_string(statistics::sum_str_int_map(value));
 124         data.emplace("stat_detail", item);
 125 
 126         item["label"] = has_cost ? std::to_string(statistics::sum_cost_str_int_map(value)) : font::unicode_em_dash;
 127         data.emplace("stat_cost", item);
 128 
 129         stat_list.add_row(data);
 130 
 131         main_stat_table_.push_back(&value);
 132 }

 134 void statistics_dialog::add_damage_row(
 135                 window& window,
 136                 const std::string& type,
 137                 const long long& damage,
 138                 const long long& expected,
 139                 const long long& turn_damage,
 140                 const long long& turn_expected,
 141                 const bool show_this_turn)
 142 {
 143         listbox& damage_list = find_widget<listbox>(&window, "stats_list_damage", false);
 144 
 145         std::map<std::string, string_map> data;
 146         string_map item;
 147 
 148         item["label"] = type;
 149         data.emplace("damage_type", item);
 150 
 151         const int shift = statistics::stats::decimal_shift;
 152 
 153         const long long dsa = shift * damage      - expected;
 154         const long long dst = shift * turn_damage - turn_expected;
 155 
 156         std::ostringstream str;
 157         str << damage << " / "
 158                 << (expected * 10 + shift / 2) / shift * 0.1
 159                 << "    " // TODO: should probably make this two columns
 160                 << ((dsa < 0) ^ (expected < 0) ? "" : "+")
 161                 << (expected == 0 ? 0 : 100 * dsa / expected) << '%';
 162 
 163         item["label"] = str.str();
 164         data.emplace("damage_overall", item);
 165 
 166         str.str("");
 167 
 168         if(show_this_turn) {
 169                 str << turn_damage << " / "
 170                         << (turn_expected * 10 + shift / 2) / shift * 0.1
 171                         << "    " // TODO: should probably make this two columns
 172                         << ((dst < 0) ^ (turn_expected < 0) ? "" : "+")
 173                         << (turn_expected == 0 ? 0 : 100 * dst / turn_expected) << '%';
 174 
 175                 item["label"] = str.str();
 176                 data.emplace("damage_this_turn", item);
 177         }
 178 
 179         damage_list.add_row(data);
 180 }

 182 void statistics_dialog::update_lists(window& window)
 183 {
 184         //
 185         // Update primary stats list
 186         //
 187         listbox& stat_list = find_widget<listbox>(&window, "stats_list_main", false);
 188 
 189         stat_list.clear();
 190         main_stat_table_.clear();
 191 
 192         const statistics::stats& stats = current_stats();
 193 
 194         add_stat_row(window, _("Recruits"),     stats.recruits);
 195         add_stat_row(window, _("Recalls"),      stats.recalls);
 196         add_stat_row(window, _("Advancements"), stats.advanced_to, false);
 197         add_stat_row(window, _("Losses"),       stats.deaths);
 198         add_stat_row(window, _("Kills"),        stats.killed);
 199 
 200         // Update unit count list
 201         on_primary_list_select(window);
 202 
 203         //
 204         // Update damage stats list
 205         //
 206         const bool show_this_turn = use_campaign || scenario_index_ + 1 == scenarios_.size();
 207 
 208         listbox& damage_list = find_widget<listbox>(&window, "stats_list_damage", false);
 209 
 210         damage_list.clear();
 211 
 212         add_damage_row(window, _("Inflicted"),
 213                 stats.damage_inflicted,
 214                 stats.expected_damage_inflicted,
 215                 stats.turn_damage_inflicted,
 216                 stats.turn_expected_damage_inflicted,
 217                 show_this_turn
 218         );
 219 
 220         add_damage_row(window, _("Taken"),
 221                 stats.damage_taken,
 222                 stats.expected_damage_taken,
 223                 stats.turn_damage_taken,
 224                 stats.turn_expected_damage_taken,
 225                 show_this_turn
 226         );
 227 }

 229 void statistics_dialog::on_tab_select(window& window)
 230 {
 231         const bool is_campaign_tab = find_widget<listbox>(&window, "tab_bar", false).get_selected_row() == 1;
 232 
 233         if(use_campaign != is_campaign_tab) {
 234                 use_campaign = is_campaign_tab;
 235 
 236                 update_lists(window);
 237         }
 238 }

 240 void statistics_dialog::on_scenario_select(window& window)
 241 {
 242         const size_t new_index = find_widget<menu_button>(&window, "scenario_menu", false).get_value();
 243 
 244         if(scenario_index_ != new_index) {
 245                 scenario_index_ = new_index;
 246 
 247                 update_lists(window);
 248         }
 249 }

 251 void statistics_dialog::on_primary_list_select(window& window)
 252 {
 253         const int selected_row = find_widget<listbox>(&window, "stats_list_main", false).get_selected_row();
 254         if(selected_row == -1) {
 255                 return;
 256         }
 257 
 258         listbox& unit_list = find_widget<listbox>(&window, "stats_list_units", false);
 259 
 260         unit_list.clear();
 261 
 262         for(const auto& i : *main_stat_table_[selected_row]) {
 263                 const unit_type* type = unit_types.find(i.first);
 264                 if(!type) {
 265                         continue;
 266                 }
 267 
 268                 std::map<std::string, string_map> data;
 269                 string_map item;
 270 
 271                 item["label"] = (formatter() << type->image() << "~RC(" << type->flag_rgb() << ">" << current_team_.color() << ")").str();
 272                 data.emplace("unit_image", item);
 273 
 274                 item["label"] = type->type_name();
 275                 data.emplace("unit_name", item);
 276 
 277                 item["label"] = (formatter() << i.second << font::unicode_multiplication_sign).str();
 278                 data.emplace("unit_count", item);
 279 
 280                 unit_list.add_row(data);
 281         }
 282 }
