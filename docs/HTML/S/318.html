  67 void wesnothd_connection::handle_connect(
  68                 const boost::system::error_code& ec,
  69                 resolver::iterator iterator
  70                 )
  71 {
  72         if(ec) {
  73                 WRN_NW << "Failed to connect to " <<
  74                         iterator->endpoint().address() << ": " <<
  75                         ec.message() << '\n';
  76                 socket_.close();
  77                 if(++iterator == resolver::iterator()) {
  78                         ERR_NW << "Tried all IPs. Giving up" << std::endl;
  79                         throw system_error(ec);
  80                 }
  81                 else {
  82                         connect(iterator);
  83                 }
  84         } else {
  85                 LOG_NW << "Connected to " << iterator->endpoint().address() << '\n';
  86                 handshake();
  87         }
  88 }

  90 void wesnothd_connection::handshake()
  91 {
  92         static const uint32_t handshake = 0;
  93         boost::asio::async_write(socket_,
  94                 boost::asio::buffer(reinterpret_cast<const char*>(&handshake), 4),
  95                 [](const error_code& ec, std::size_t) { if (ec) throw system_error(ec); }
  96         );
  97         boost::asio::async_read(socket_,
  98                 boost::asio::buffer(&handshake_response_.binary, 4),
  99                 std::bind(&wesnothd_connection::handle_handshake, this, _1)

 158 std::size_t wesnothd_connection::is_read_complete(
 159                 const boost::system::error_code& ec,
 160                 std::size_t bytes_transferred
 161                 )
 162 {
 163         if(ec) {
 164                 throw system_error(ec);
 165         }
 166         bytes_read_ = bytes_transferred;
 167         if(bytes_transferred < 4) {
 168                 return 4;
 169         } else {
 170                 if(!bytes_to_read_) {
 171                         std::istream is(&read_buf_);
 172                         union { char binary[4]; uint32_t num; } data_size;
 173                         is.read(data_size.binary, 4);
 174                         bytes_to_read_ = ntohl(data_size.num) + 4;
 175                         //Close immediately if we receive an invalid length
 176                         if (bytes_to_read_ < 4)
 177                                 bytes_to_read_ = bytes_transferred;
 178                 }
 179                 return bytes_to_read_ - bytes_transferred;
 180         }
 181 }

 183 void wesnothd_connection::handle_read(
 184         const boost::system::error_code& ec,
 185         std::size_t bytes_transferred
 186         )
 187 {
 188         DBG_NW << "Read " << bytes_transferred << " bytes.\n";
 189         bytes_to_read_ = 0;
 190         if(ec && ec != boost::asio::error::eof)
 191                 throw system_error(ec);
 192         std::istream is(&read_buf_);
 193         recv_queue_.emplace_back();
 194         read_gz(recv_queue_.back(), is);
 195         DBG_NW << "Received " << recv_queue_.back() << " bytes.\n";
 196 
 197         recv();
 198 }

 200 void wesnothd_connection::send()
 201 {
 202         auto& buf = send_queue_.front();
 203         size_t buf_size = buf.size();
 204         bytes_to_write_ = buf_size + 4;
 205         bytes_written_ = 0;
 206         payload_size_ = htonl(buf_size);
 207 
 208         boost::asio::streambuf::const_buffers_type gzipped_data = buf.data();
 209         std::deque<boost::asio::const_buffer> bufs(gzipped_data.begin(), gzipped_data.end());
 210         bufs.push_front(boost::asio::buffer(reinterpret_cast<const char*>(&payload_size_), 4));
 211         boost::asio::async_write(socket_, bufs,
 212                 std::bind(&wesnothd_connection::is_write_complete, this, _1, _2),
 213                 std::bind(&wesnothd_connection::handle_write, this, _1, _2)
 214         );
 215 }

 225 std::size_t wesnothd_connection::poll()
 226 {
 227         try {
 228                 return io_service_.poll();
 229         }
 230         catch (const boost::system::system_error& err) {
 231                 if(
 232                         err.code() == boost::asio::error::operation_aborted ||
 233                         err.code() == boost::asio::error::eof
 234                 )
 235                         return 1;
 236                 throw error(err.code());
 237         }

 239 bool wesnothd_connection::receive_data(config& result)
 240 {
 241         poll();
 242         if (recv_queue_.empty()) {
 243                 return false;
 244         }
 245         else {
 246                 result.swap(recv_queue_.front());
 247                 recv_queue_.pop_front();
 248                 return true;
 249         }
