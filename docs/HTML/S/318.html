 101 // main thread
 102 void wesnothd_connection::handle_connect(
 103                 const boost::system::error_code& ec,
 104                 resolver::iterator iterator
 105                 )
 106 {
 107         MPTEST_LOG;
 108         if(ec) {
 109                 WRN_NW << "Failed to connect to " <<
 110                         iterator->endpoint().address() << ": " <<
 111                         ec.message() << '\n';
 112                 socket_.close();
 113                 if(++iterator == resolver::iterator()) {
 114                         ERR_NW << "Tried all IPs. Giving up" << std::endl;
 115                         throw system_error(ec);
 116                 }
 117                 else {
 118                         connect(iterator);
 119                 }
 120         } else {
 121                 LOG_NW << "Connected to " << iterator->endpoint().address() << '\n';
 122                 handshake();
 123         }
 124 }
 125 // main thread
 126 void wesnothd_connection::handshake()
 127 {
 128         MPTEST_LOG;
 129         static const uint32_t handshake = 0;
 130         boost::asio::async_write(socket_,
 131                 boost::asio::buffer(reinterpret_cast<const char*>(&handshake), 4),
 132                 [](const error_code& ec, std::size_t) { if (ec) throw system_error(ec); }
 133         );
 134         boost::asio::async_read(socket_,
 135                 boost::asio::buffer(&handshake_response_.binary, 4),
 136                 std::bind(&wesnothd_connection::handle_handshake, this, _1)
 137         );
 138 }

 140 // main thread
 141 void wesnothd_connection::handle_handshake(const error_code& ec)
 142 {
 143         MPTEST_LOG;
 144         if (ec) {
 145                 LOG_NW << __func__ << " Throwing: " << ec << "\n";
 146                 throw system_error(ec);
 147         }
 148         handshake_finished_ = true;
 149 
 150         recv();
 151         worker_thread_.reset(new boost::thread( [this](){
 152                 // worker thread
 153                 std::shared_ptr<wesnothd_connection> this_ptr = this->shared_from_this();
 154                 io_service_.run();
 155                 LOG_NW << "wesnothd_connection::io_service::run() returned\n";          
 156         } ));
 157 }

 200 std::size_t wesnothd_connection::is_write_complete(const boost::system::error_code& ec, size_t bytes_transferred)
 201 {
 202         MPTEST_LOG;
 203         if(ec)
 204         {
 205                 {
 206                         std::lock_guard<std::mutex> lock(last_error_mutex_);
 207                         last_error_ = ec;
 208                 }
 209                 LOG_NW << __func__ << " Error: " << ec << "\n";
 210                 io_service_.stop();
 211                 return bytes_to_write_ - bytes_transferred;
 212         }
 213         bytes_written_ = bytes_transferred;
 214         return bytes_to_write_ - bytes_transferred;
 215 }

 217 //worker thread
 218 void wesnothd_connection::handle_write(
 219         const boost::system::error_code& ec,
 220         std::size_t bytes_transferred
 221         )
 222 {
 223         MPTEST_LOG;
 224         DBG_NW << "Written " << bytes_transferred << " bytes.\n";
 225         send_queue_.pop_front();
 226         if(ec)
 227         {
 228                 {
 229                         std::lock_guard<std::mutex> lock(last_error_mutex_);
 230                         last_error_ = ec;
 231                 }
 232                 LOG_NW << __func__ << " Error: " << ec << "\n";
 233                 io_service_.stop();
 234                 return;
 235         }
 236         if (!send_queue_.empty()) {
 237                 send();
 238         }
 239 }

 241 //worker thread
 242 std::size_t wesnothd_connection::is_read_complete(
 243                 const boost::system::error_code& ec,
 244                 std::size_t bytes_transferred
 245                 )
 246 {
 247         //We use custom is_write/read_complete function to be able to see the current progress of the upload/download
 248         MPTEST_LOG;
 249         if(ec)
 250         {
 251                 {
 252                         std::lock_guard<std::mutex> lock(last_error_mutex_);
 253                         last_error_ = ec;
 254                 }
 255                 LOG_NW << __func__ << " Error: " << ec << "\n";
 256                 io_service_.stop();
 257                 return bytes_to_read_ - bytes_transferred;
 258         }
 259         bytes_read_ = bytes_transferred;
 260         if(bytes_transferred < 4) {
 261                 return 4;
 262         } else {
 263                 if(!bytes_to_read_) {
 264                         std::istream is(&read_buf_);
 265                         union { char binary[4]; uint32_t num; } data_size;
 266                         is.read(data_size.binary, 4);
 267                         bytes_to_read_ = ntohl(data_size.num) + 4;
 268                         //Close immediately if we receive an invalid length
 269                         if (bytes_to_read_ < 4)
 270                                 bytes_to_read_ = bytes_transferred;
 271                 }
 272                 return bytes_to_read_ - bytes_transferred;
 273         }
 274 }

 276 //worker thread
 277 void wesnothd_connection::handle_read(
 278         const boost::system::error_code& ec,
 279         std::size_t bytes_transferred
 280         )
 281 {
 282         MPTEST_LOG;
 283         DBG_NW << "Read " << bytes_transferred << " bytes.\n";
 284         bytes_to_read_ = 0;
 285         if(last_error_ && ec != boost::asio::error::eof)
 286         {
 287                 {
 288                         std::lock_guard<std::mutex> lock(last_error_mutex_);
 289                         last_error_ = ec;
 290                 }
 291                 LOG_NW << __func__ << " Error: " << ec << "\n";
 292                 io_service_.stop();
 293                 return;
 294         }
 295 
 296         std::istream is(&read_buf_);
 297         config data;
 298         read_gz(data, is);
 299         {
 300                 std::lock_guard<std::mutex> lock(recv_queue_mutex_);
 301                 recv_queue_.emplace_back(std::move(data));
 302         }
 303         DBG_NW << "Received " << recv_queue_.back() << " bytes.\n";
 304 
 305         recv();
 306 }

 308 //worker thread
 309 void wesnothd_connection::send()
 310 {
 311         MPTEST_LOG;
 312         auto& buf = *send_queue_.front();
 313         size_t buf_size = buf.size();
 314         bytes_to_write_ = buf_size + 4;
 315         bytes_written_ = 0;
 316         payload_size_ = htonl(buf_size);
 317 
 318         boost::asio::streambuf::const_buffers_type gzipped_data = buf.data();
 319         std::deque<boost::asio::const_buffer> bufs(gzipped_data.begin(), gzipped_data.end());
 320         bufs.push_front(boost::asio::buffer(reinterpret_cast<const char*>(&payload_size_), 4));
 321         boost::asio::async_write(socket_, bufs,
 322                 std::bind(&wesnothd_connection::is_write_complete, this, _1, _2),
 323                 std::bind(&wesnothd_connection::handle_write, this, _1, _2)
 324         );
 325 }
 326 //worker thread
 327 void wesnothd_connection::recv()
 328 {
 329         MPTEST_LOG;

 331                 std::bind(&wesnothd_connection::is_read_complete, this, _1, _2),
 332                 std::bind(&wesnothd_connection::handle_read, this, _1, _2)
 333         );

 336 // main thread, during handshake
 337 std::size_t wesnothd_connection::poll()
 338 {
 339         MPTEST_LOG;
 340         assert(!worker_thread_);
 341         try {
 342                 return io_service_.poll();
 343         }
 344         catch (const boost::system::system_error& err) {
 345                 if( err.code() == boost::asio::error::operation_aborted || err.code() == boost::asio::error::eof) {
 346                         return 1;
 347                 }
 348                 WRN_NW << __func__ << " Rehrowing: " << err.code() << "\n";
 349                 throw error(err.code());
 350         }
 351 }

 353 // main thread
 354 bool wesnothd_connection::receive_data(config& result)
 355 {
 356         MPTEST_LOG;
 357         {
 358                 std::lock_guard<std::mutex> lock(last_error_mutex_);
 359                 if (last_error_) {
 360                         throw error(last_error_);
 361                 }
 362         }
 363         std::lock_guard<std::mutex> lock(recv_queue_mutex_);
 364         if (recv_queue_.empty()) {
 365                 return false;
 366         }
 367         else {
 368                 result.swap(recv_queue_.front());
 369                 recv_queue_.pop_front();
 370                 return true;
 371         }
 372 }
