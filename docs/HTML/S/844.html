  93                 wesnoth_message_format(std::locale base, const std::set<std::string>& domains, const std::set<std::string>& paths)
  94                         : base_loc_(base)
  95                 {
  96                         const bl::info& inf = std::use_facet<bl::info>(base);
  97                         if(inf.language() == "c") {
  98                                 return;
  99                         }
 100                         std::string lang_name_short = inf.language();
 101                         std::string lang_name_long = lang_name_short;
 102                         if(!inf.country().empty()) {
 103                                 lang_name_long += '_';
 104                                 lang_name_long += inf.country();
 105                         }
 106                         if(!inf.variant().empty()) {
 107                                 lang_name_long += '@';
 108                                 lang_name_long += inf.variant();
 109                                 lang_name_short += '@';
 110                                 lang_name_short += inf.variant();
 111                         }
 112                         DBG_G << "Loading po files for language " << lang_name_long << '\n';
 113                         for(auto& domain : domains) {
 114                                 DBG_G << "Searching for po files for domain " << domain << '\n';
 115                                 std::string path;
 116                                 for(auto base_path : paths) {
 117                                         DBG_G << "Searching in dir " << base_path << '\n';
 118                                         if(base_path[base_path.length()-1] != '/') {
 119                                                 base_path += '/';
 120                                         }
 121                                         base_path += domain;
 122                                         base_path += '/';
 123                                         path = base_path + lang_name_long + ".po";
 124                                         DBG_G << "  Trying path " << path << '\n';
 125                                         if(filesystem::file_exists(path)) {
 126                                                 break;
 127                                         }
 128                                         path = base_path + lang_name_short + ".po";
 129                                         DBG_G << "  Trying path " << path << '\n';
 130                                         if(filesystem::file_exists(path)) {
 131                                                 break;
 132                                         }
 133                                 }
 134                                 if(!filesystem::file_exists(path)) {
 135                                         continue;
 136                                 }
 137                                 std::ifstream po_file;
 138                                 po_file.exceptions(std::ios::badbit);
 139                                 LOG_G << "Loading language file from " << path << '\n';
 140                                 try {
 141                                         po_file.open(path);
 142                                         const po_catalog& cat = po_catalog::from_istream(po_file);
 143                                         extra_messages_.emplace(get_base().domain(domain), cat);
 144                                 } catch(spirit_po::catalog_exception& e) {
 145                                         throw_po_error(lang_name_long, domain, e.what());
 146                                 } catch(std::ios::failure&) {
 147                                         throw_po_error(lang_name_long, domain, strerror(errno));
 148                                 }
 149                         }
 150                 }

 160                 const char* get(int domain_id, const char* ctx, const char* msg_id) const override
 161                 {
 162                         auto& base = get_base();
 163                         const char* msg = base.get(domain_id, ctx, msg_id);
 164                         if(msg == nullptr) {
 165                                 auto iter = extra_messages_.find(domain_id);
 166                                 if(iter == extra_messages_.end()) {
 167                                         return nullptr;
 168                                 }
 169                                 auto& catalog = iter->second;
 170                                 const char* lookup = ctx ? catalog.pgettext(ctx, msg_id) : catalog.gettext(msg_id);
 171                                 if(lookup != msg_id) {
 172                                         // (p)gettext returns the input pointer if the string was not found
 173                                         msg = lookup;
 174                                 }
 175                         }
 176                         return msg;
 177                 }
 178 
 179                 const char* get(int domain_id, const char* ctx, const char* sid, int n) const override
 180                 {
 181                         auto& base = get_base();
 182                         const char* msg = base.get(domain_id, ctx, sid, n);
 183                         if(msg == nullptr) {
 184                                 auto iter = extra_messages_.find(domain_id);
 185                                 if(iter == extra_messages_.end()) {
 186                                         return nullptr;
 187                                 }
 188                                 auto& catalog = iter->second;
 189                                 const char* lookup = ctx ? catalog.npgettext(ctx, sid, sid, n) : catalog.ngettext(sid, sid, n);
 190                                 if(lookup != sid) {
 191                                         // n(p)gettext returns one of the input pointers if the string was not found
 192                                         msg = lookup;
 193                                 }
 194                         }
 195                         return msg;
 196                 }

 220                 translation_manager()
 221                         : loaded_paths_()
 222                         , loaded_domains_()
 223                         , current_language_(default_utf8_locale_name::name())
 224                         , generator_()
 225                         , current_locale_()
 226                         , is_dirty_(true)
 227                 {
 228                         const bl::localization_backend_manager& g_mgr = bl::localization_backend_manager::global();
 229                         for(const std::string& name : g_mgr.get_all_backends())
 230                         {
 231                                 LOG_G << "Found boost locale backend: '" << name << "'\n";
 232                         }
 233 
 234                         generator_.use_ansi_encoding(false);
 235                         generator_.categories(bl::message_facet | bl::information_facet | bl::collation_facet | bl::formatting_facet);
 236                         generator_.characters(bl::char_facet);
 237                         // We cannot have current_locale_ be a non boost-generated locale since it might not supply
 238                         // the bl::info facet. As soon as we add message paths, update_locale_internal might fail,
 239                         // for example because of invalid .mo files. So make sure we call it at least once before adding paths/domains
 240                         update_locale_internal();
 241                 }

 243                 void add_messages_domain(const std::string& domain)
 244                 {
 245                         if(loaded_domains_.find(domain) != loaded_domains_.end())
 246                         {
 247                                 return;
 248                         }
 249 
 250                         if(domain.find('/') != std::string::npos)
 251                         {
 252                                 // Forward slash has a specific meaning in Boost.Locale domain
 253                                 // names, specifying the encoding. We use UTF-8 for everything
 254                                 // so we can't possibly support that, and odds are it's a user
 255                                 // mistake (as in bug #23839).
 256                                 ERR_G << "illegal textdomain name '" << domain
 257                                           << "', skipping textdomain\n";
 258                                 return;
 259                         }
 260 
 261                         generator_.add_messages_domain(domain);
 262                         loaded_domains_.insert(domain);
 263                 }
 264

 281                 void set_language(const std::string& language)
 282                 {
 283                         std::string::size_type at_pos = language.rfind('@');
 284                         if(language.empty())
 285                         {
 286                                 current_language_ = default_utf8_locale_name::name();
 287                         }
 288                         else if(at_pos  != std::string::npos)
 289                         {
 290                                 current_language_ = language.substr(0, at_pos) + ".UTF-8" + language.substr(at_pos);
 291                         }
 292                         else
 293                         {
 294                                 current_language_ = language + ".UTF-8";
 295                         }
 296                         update_locale();
 297                 }

 299                 void update_locale()
 300                 {
 301                         is_dirty_ = true;
 302                 }

 304                 void update_locale_internal()
 305                 {
 306                         try
 307                         {
 308                                 LOG_G << "attempting to generate locale by name '" << current_language_ << "'\n";
 309                                 current_locale_ = generator_.generate(current_language_);
 310                                 current_locale_ = std::locale(current_locale_, new wesnoth_message_format(current_locale_, loaded_domains_, loaded_paths_));
 311                                 const bl::info& info = std::use_facet<bl::info>(current_locale_);
 312                                 LOG_G << "updated locale to '" << current_language_ << "' locale is now '" << current_locale_.name() << "' ( "
 313                                       << "name='" << info.name()
 314                                       << "' country='"  << info.country()
 315                                       << "' language='"  << info.language()
 316                                       << "' encoding='"  << info.encoding()
 317                                       << "' variant='"  << info.variant() << "')\n";
 318                         }
 319                         catch(const bl::conv::conversion_error&)
 320                         {
 321                                 assert(std::has_facet<bl::info>(current_locale_));
 322                                 const bl::info& info = std::use_facet<bl::info>(current_locale_);
 323                                 ERR_G << "Failed to update locale due to conversion error, locale is now: "
 324                                       << "name='" << info.name()
 325                                       << "' country='" << info.country()
 326                                       << "' language='" << info.language()
 327                                       << "' encoding='" << info.encoding()
 328                                       << "' variant='" << info.variant()
 329                                       << "'" << std::endl;
 330                         }
 331                         is_dirty_ = false;
 332                 }
 333 
 334                 const std::locale& get_locale()
 335                 {
 336                         if(is_dirty_)
 337                         {
 338                                 update_locale_internal();
 339                         }
 340                         return current_locale_;
 341                 }

 372 std::string dsgettext (const char * domainname, const char *msgid)
 373 {
 374         std::string msgval = dgettext (domainname, msgid);
 375         if (msgval == msgid) {
 376                 const char* firsthat = std::strrchr (msgid, '^');
 377                 if (firsthat == nullptr)
 378                         msgval = msgid;
 379                 else
 380                         msgval = firsthat + 1;
 381         }
 382         return msgval;
 383 }

 386 {
 387         std::string msgval = bl::dngettext(domainname, singular, plural, n, get_manager().get_locale());
 388         if (msgval == singular) {
 389                 const char* firsthat = std::strrchr (singular, '^');
 390                 if (firsthat == nullptr)
 391                         msgval = singular;
 392                 else
 393                         msgval = firsthat + 1;
 394         }
 395         return msgval;
