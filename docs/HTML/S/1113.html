  34                            const config& in,
  35                            const std::string& type,
  36                            bool copy_force_key=false)
  37 {
  38         if (in.has_attribute("allow_" + type)) {

  41                 out["disallow_" + type] = in["disallow_" + type];

  45                 out["ignore_incompatible_" + type] = in["ignore_incompatible_" + type];
  46         }
  47 
  48         if (copy_force_key) {
  49                 if (in.has_attribute("force_" + type)) {
  50                         out["force_" + type] = in["force_" + type];
  51                 }
  52         }

  62 manager::manager(const config& gamecfg, bool mp, CVideo& video)
  63         : video_(video)
  64         , depinfo_()
  65         , era_()
  66         , scenario_()
  67         , mods_()
  68         , prev_era_()
  69         , prev_scenario_()
  70         , prev_mods_()
  71 {
  72         DBG_MP << "Initializing the dependency manager" << std::endl;
  73         for (const config& cfg : gamecfg.child_range("modification")) {
  74                 component_availabilty type = cfg["type"].to_enum<component_availabilty>(component_availabilty::HYBRID);
  75                 if((type != component_availabilty::MP || mp) && (type != component_availabilty::SP || !mp) ) {
  76                         config info;
  77                         info["id"] = cfg["id"];
  78                         info["name"] = cfg["name"];
  79 
  80                         copy_keys(info, cfg, "scenario");
  81                         copy_keys(info, cfg, "era");
  82                         copy_keys(info, cfg, "modification");
  83 
  84                         depinfo_.add_child("modification", info);
  85                 }
  86         }
  87 
  88         for (const config& cfg : gamecfg.child_range("era")) {
  89                 component_availabilty type = cfg["type"].to_enum<component_availabilty>(component_availabilty::MP);
  90                 if((type != component_availabilty::MP || mp) && (type != component_availabilty::SP || !mp) ) {
  91                         config info;
  92                         info["id"] = cfg["id"];
  93                         info["name"] = cfg["name"];
  94 
  95                         copy_keys(info, cfg, "scenario");
  96                         copy_keys(info, cfg, "modification", true);
  97 
  98                         depinfo_.add_child("era", info);
  99                 }
 100         }
 101 
 102         for (const config& cfg : gamecfg.child_range("multiplayer")) {
 103                 if (cfg["allow_new_game"].to_bool(true)) {
 104                         config info;
 105                         info["id"] = cfg["id"];
 106                         info["name"] = cfg["name"];
 107 
 108                         copy_keys(info, cfg, "era");
 109                         copy_keys(info, cfg, "modification", true);
 110 
 111                         depinfo_.add_child("scenario", info);
 112                 }
 113         }
 114 }

 211 bool manager::conflicts(const elem& elem1, const elem& elem2, bool directonly) const
 212 {
 213         if (elem1 == elem2) {
 214                 return false;
 215         }
 216 
 217         // We ignore inexistent elements at this point, they will generate
 218         // errors in change_era()/change_scenario() anyways.
 219         if (!exists(elem1) || !exists(elem2)) {
 220                 return false;
 221         }
 222 
 223         config data1 = depinfo_.find_child(elem1.type, "id", elem1.id);
 224         config data2 = depinfo_.find_child(elem2.type, "id", elem2.id);
 225 
 226         // Whether we should skip the check entirely
 227         if (data1.has_attribute("ignore_incompatible_" + elem2.type)) {
 228                 std::vector<std::string> ignored =
 229                                                 utils::split(data1["ignore_incompatible_" + elem2.type]);
 230 
 231                 if ( utils::contains(ignored, elem2.id) ) {
 232                         return false;
 233                 }
 234         }
 235 
 236         if (data2.has_attribute("ignore_incompatible_" + elem1.type)) {
 237                 std::vector<std::string> ignored =
 238                                                 utils::split(data2["ignore_incompatible_" + elem1.type]);
 239 
 240                 if ( utils::contains(ignored, elem1.id) ) {
 241                         return false;
 242                 }
 243         }
 244 
 245         bool result = false;
 246 
 247         // Checking for direct conflicts between elem1 and elem2
 248         if (data1.has_attribute("allow_" + elem2.type)) {
 249                 std::vector<std::string> allowed =
 250                                                 utils::split(data1["allow_" + elem2.type]);
 251 
 252                 result = !utils::contains(allowed, elem2.id) && !requires(elem1, elem2);
 253         } else if (data1.has_attribute("disallow_" + elem2.type)) {
 254                 std::vector<std::string> disallowed =
 255                                                 utils::split(data1["disallow_" + elem2.type]);
 256 
 257                 result = utils::contains(disallowed, elem2.id);
 258         }
 259 
 260         if (data2.has_attribute("allow_" + elem1.type)) {
 261                 std::vector<std::string> allowed =
 262                                                 utils::split(data2["allow_" + elem1.type]);
 263 
 264                 result = result || (!utils::contains(allowed, elem1.id) && !requires(elem2, elem1));
 265         } else if (data2.has_attribute("disallow_" + elem1.type)) {
 266                 std::vector<std::string> disallowed =
 267                                                 utils::split(data2["disallow_" + elem1.type]);
 268 
 269                 result = result || utils::contains(disallowed, elem1.id);
 270         }
 271 
 272         if (result) {
 273                 return true;
 274         }
 275 
 276         // Checking for indirect conflicts (i.e. conflicts between dependencies)
 277         if (!directonly) {
 278                 std::vector<std::string>        req1 = get_required(elem1),
 279                                                                         req2 = get_required(elem2);
 280 
 281                 for (const std::string& s : req1) {
 282                         elem m(s, "modification");
 283 
 284                         if (conflicts(elem2, m, true)) {
 285                                 return true;
 286                         }
 287                 }
 288 
 289                 for (const std::string& s : req2) {
 290                         elem m(s, "modification");
 291 
 292                         if (conflicts(elem1, m, true)) {
 293                                 return true;
 294                         }
 295                 }
 296 
 297                 for (const std::string& id1 : req1) {
 298                         elem m1(id1, "modification");
 299 
 300                         for (const std::string& id2 : req2) {
 301                                 elem m2(id2, "modification");
 302 
 303                                 if (conflicts(m1, m2)) {
 304                                         return true;
 305                                 }
 306                         }
 307                 }
 308         }
 309 
 310         return false;
 311 }

 314 bool manager::requires(const elem& elem1, const elem& elem2) const
 315 {
 316         if (elem2.type != "modification") {
 317                 return false;
 318         }
 319 
 320         config data = depinfo_.find_child(elem1.type, "id", elem1.id);
 321 
 322         if (data.has_attribute("force_modification")) {
 323                 std::vector<std::string> required =
 324                                                         utils::split(data["force_modification"]);
 325 
 326                 return utils::contains(required, elem2.id);
 327         }
 328 
 329         return false;
 330 }

 365 void manager::try_modification_by_index(int index, bool activate, bool force)
 366 {
 367         std::string id = depinfo_.child("modification", index)["id"];
 368         std::vector<std::string> mods_copy = mods_;
 369         if (activate) {
 370                 if (std::find(mods_copy.begin(), mods_copy.end(), id) == mods_copy.end()) {
 371                         mods_copy.push_back(id);
 372                 }
 373         } else {
 374                 std::vector<std::string>::iterator pos = std::find(mods_copy.begin(), mods_copy.end(), id);
 375                 if (pos != mods_copy.end()) {
 376                         mods_copy.erase(pos);
 377                 }
 378         }
 379 
 380         try_modifications(mods_copy, force);
 381 }

 513 bool manager::change_scenario(const std::string& id)
 514 {
 515         // Checking for missing dependencies
 516         if (!get_required_not_installed(elem(id, "scenario")).empty()) {
 517                 std::string msg =
 518                         _("Scenario can't be activated. Some dependencies are missing: ");
 519 
 520                 msg +=
 521                         utils::join(get_required_not_installed(elem(id, "scenario")), ", ");
 522 
 523                 failure_dialog(msg);
 524                 return false;
 525         }
 526 
 527         scenario_ = id;
 528 
 529         elem scen = elem(id, "scenario");
 530         std::string scen_name = find_name_for(scen);
 531 
 532         // Firstly, we check if we have to enable/disable any mods
 533         std::vector<std::string> req = get_required_not_enabled(scen);
 534         std::vector<std::string> con = get_conflicting_enabled(scen);
 535 
 536         if (!req.empty()) {
 537                 if (!enable_mods_dialog(req, scen_name)) {
 538                         return false;
 539                 }
 540         }
 541 
 542         if (!con.empty()) {
 543                 if (!disable_mods_dialog(con, scen_name)) {
 544                         return false;
 545                 }
 546         }
 547 
 548         std::vector<std::string> newmods = req;
 549         for (const std::string& i : mods_) {
 550                 if ( !utils::contains(con, i) ) {
 551                         newmods.push_back(i);
 552                 }
 553         }
 554 
 555 
 556         mods_ = newmods;
 557 
 558         // Now checking if the currently selected era conflicts the scenario
 559         // and changing era if necessary
 560         if (!conflicts(scen, elem(era_, "era"))) {
 561                 return true;
 562         }
 563 
 564         std::vector<std::string> compatible;
 565         for (const config& i : depinfo_.child_range("era")) {
 566                 if (!conflicts(scen, elem(i["id"], "era"))) {
 567                         compatible.push_back(i["id"]);
 568                 }
 569         }
 570 
 571         if (!compatible.empty()) {
 572                 era_ = change_era_dialog(compatible);
 573         } else {
 574                 failure_dialog(_("No compatible eras found."));
 575                 return false;
 576         }
 577 
 578         if (era_.empty()) {
 579                 return false;
 580         }
 581 
 582         return change_era(era_);
 583 }

 585 bool manager::change_era(const std::string& id)
 586 {
 587         // Checking for missing dependenciess
 588         if (!get_required_not_installed(elem(id, "era")).empty()) {
 589                 std::string msg =
 590                                 _("Era can't be activated. Some dependencies are missing: ");
 591 
 592                 msg += utils::join(get_required_not_installed(elem(id, "era")), ", ");
 593                 failure_dialog(msg);
 594                 return false;
 595         }
 596 
 597         era_ = id;
 598 
 599         elem era = elem(id, "era");
 600         std::string era_name = find_name_for(era);
 601 
 602         std::vector<std::string> req = get_required_not_enabled(era);
 603         std::vector<std::string> con = get_conflicting_enabled(era);
 604 
 605         // Firstly, we check if we have to enable/disable any mods
 606         if (!req.empty()) {
 607                 if (!enable_mods_dialog(req, era_name)) {
 608                         return false;
 609                 }
 610         }
 611 
 612         if (!con.empty()) {
 613                 if (!disable_mods_dialog(con, era_name)) {
 614                         return false;
 615                 }
 616         }
 617 
 618         std::vector<std::string> newmods = req;
 619         for (const std::string& i : mods_) {
 620                 if ( !utils::contains(con, i) ) {
 621                         newmods.push_back(i);
 622                 }
 623         }
 624 
 625         mods_ = newmods;
 626 
 627         // Now checking if the currently selected scenario conflicts the era
 628         // and changing scenario if necessary
 629         if (!conflicts(era, elem(scenario_, "scenario"))) {
 630                 return true;
 631         }
 632 
 633         std::vector<std::string> compatible;
 634         for (const config& i : depinfo_.child_range("scenario")) {
 635                 if (!conflicts(era, elem(i["id"], "scenario"))) {
 636                         compatible.push_back(i["id"]);
 637                 }
 638         }
 639 
 640         if (!compatible.empty()) {
 641                 scenario_ = change_scenario_dialog(compatible);
 642         } else {
 643                 failure_dialog(_("No compatible scenarios found."));
 644                 return false;
 645         }
 646 
 647         if (scenario_.empty()) {
 648                 return false;
 649         }
 650 
 651         return change_scenario(scenario_);
 652 }

 654 bool manager::change_modifications
 655                                         (const std::vector<std::string>& modifications)
 656 {
 657         // Checking if the selected combination of mods is valid at all
 658         std::vector<std::string> filtered;
 659         for (const std::string& i : modifications) {
 660                 bool ok = true;
 661                 elem ei(i, "modification");
 662                 for (const std::string& j : filtered) {
 663                         ok = ok && !conflicts(ei, elem(j, "modification"));
 664                 }
 665 
 666                 if (ok) {
 667                         filtered.push_back(i);
 668                 }
 669         }
 670 
 671         if (filtered.size() != modifications.size()) {
 672                 failure_dialog(_("Not all of the chosen modifications are compatible." \
 673                                                  " Some of them will be disabled."));
 674         }
 675 
 676         mods_ = filtered;
 677 
 678         // Checking if the currently selected era is compatible with the set
 679         // modifications, and changing era if necessary
 680         std::vector<std::string> compatible;
 681         for (const config& c : depinfo_.child_range("era")) {
 682                 elem era(c["id"], "era");
 683                 bool ok = true;
 684                 for (const std::string& s : mods_) {
 685                         ok = ok && !conflicts(era, elem(s, "modification"));
 686                 }
 687 
 688                 if (ok) {
 689                         compatible.push_back(era.id);
 690                 }
 691         }
 692 
 693         if ( !utils::contains(compatible, era_) ) {
 694                 if (!compatible.empty()) {
 695                         era_ = change_era_dialog(compatible);
 696                 } else {
 697                         failure_dialog(_("No compatible eras found."));
 698                         return false;
 699                 }
 700 
 701                 if (era_.empty()) {
 702                         return false;
 703                 }
 704 
 705                 if (!change_era(era_)) {
 706                         return false;
 707                 }
 708         } else {
 709                 if (!change_era(era_)) {
 710                         return false;
 711                 }
 712         }
 713 
 714         compatible.clear();
 715 
 716         // Checking if the currently selected scenario is compatible with
 717         // the set modifications, and changing scenario if necessary
 718         for (const config& c : depinfo_.child_range("scenario")) {
 719                 elem scen(c["id"], "scenario");
 720                 bool ok = true;
 721                 for (const std::string& s : mods_) {
 722                         ok = ok && !conflicts(scen, elem(s, "modification"));
 723                 }
 724 
 725                 if (ok) {
 726                         compatible.push_back(scen.id);
 727                 }
 728         }
