 145 /**
 146  * The base class for all savegame stuff.
 147  * This should not be used directly, as it does not directly produce usable saves.
 148  * Instead, use one of the derived classes.
 149  */
 150 class savegame
 151 {
 152 protected:
 153         /** The only constructor of savegame. The title parameter is only necessary if you
 154                 intend to do interactive saves. */
 155         savegame(saved_game& gamestate, const compression::format compress_saves, const std::string& title = "Save");
 156 
 157 public:
 158         enum DIALOG_TYPE { YES_NO, OK_CANCEL};
 159 
 160         virtual ~savegame() {}
 161 
 162         /** Saves a game without user interaction, unless the file exists and it should be asked
 163                 to overwrite it. The return value denotes, if the save was successful or not.
 164                 This is used by automatically generated replays, start-of-scenario saves, autosaves,
 165                 and saves from the console (e.g. ":w").
 166         */
 167         bool save_game_automatic(CVideo& video, bool ask_for_overwrite = false, const std::string& filename = "");
 168 
 169         /** Save a game interactively through the savegame dialog. Used for manual midgame and replay
 170                 saves. The return value denotes, if the save was successful or not. */
 171         bool save_game_interactive(CVideo& video, const std::string& message,
 172                 DIALOG_TYPE dialog_type);
 173 
 174         const std::string& filename() const { return filename_; }
 175 
 176 protected:
 177         /**
 178                 Save a game without any further user interaction. If you want notifying messages
 179                 or error messages to appear, you have to provide the gui parameter.
 180                 The return value denotes, if the save was successful or not.
 181         */
 182         bool save_game(CVideo* video = nullptr, const std::string& filename = "");
 183 
 184         /** Sets the filename and removes invalid characters. Don't set the filename directly but
 185                 use this method instead. */
 186         void set_filename(std::string filename);
 187         /** Check, if the filename contains illegal constructs like ".gz". */
 188         bool check_filename(const std::string& filename, CVideo& video);
 189 
 190         /** Customize the standard error message */
 191         void set_error_message(const std::string& error_message) { error_message_ = error_message; }
 192 
 193         const std::string& title() const { return title_; }
 194         const saved_game& gamestate() const { return gamestate_; }
 195 
 196         /** If there needs to be some data fiddling before saving the game, this is the place to go. */
 197         void before_save();
 198 
 199         /** Writing the savegame config to a file. */
 200         virtual void write_game(config_writer &out);
 201 
 202 private:
 203         /** Checks if a certain character is allowed in a savefile name. */
 204         static bool is_illegal_file_char(char c);
 205 
 206         /** Build the filename according to the specific savegame's needs. Subclasses will have to
 207                 override this to take effect. */
 208         virtual void create_filename() {}
 209         /** Display the save game dialog. */
 210         virtual int show_save_dialog(CVideo& video, const std::string& message, DIALOG_TYPE dialog_type);
 211         /** Ask the user if an existing file should be overwritten. */
 212         bool check_overwrite(CVideo& video);
 213 
 214         /** The actual method for saving the game to disk. All interactive filename choosing and
 215                 data manipulation has to happen before calling this method. */
 216         void write_game_to_disk(const std::string& filename);
 217 
 218         /** Update the save_index */
 219         void finish_save_game(const config_writer &out);
 220         /** Throws game::save_game_failed. */
 221         filesystem::scoped_ostream open_save_game(const std::string &label);
 222         friend class save_info;
 223         //before_save (write replay data) changes this so it cannot be const
 224         saved_game& gamestate_;
 225         /** Filename of the savegame file on disk */
 226         std::string filename_;
 227 
 228         const std::string title_; /** Title of the savegame dialog */
 229 
 230         std::string error_message_; /** Error message to be displayed if the savefile could not be generated. */
 231 
 232         bool show_confirmation_; /** Determines if a confirmation of successful saving the game is shown. */
 233 
 234         compression::format compress_saves_; /** Determines, what compression format is used for the savegame file */
 235 };
