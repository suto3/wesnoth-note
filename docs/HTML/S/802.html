  68 int floating_label::xpos(size_t width) const
  69 {
  70         int xpos = int(xpos_);

  80 surface floating_label::create_surface()
  81 {
  82         if (surf_.null()) {
  83                 font::pango_text text;
  84                 text.set_foreground_color(color_);
  85                 text.set_font_size(font_size_);
  86                 text.set_maximum_width(width_ < 0 ? clip_rect_.w : width_);
  87                 text.set_maximum_height(height_ < 0 ? clip_rect_.h : height_, true);
  88 
  89                 //ignore last '\n'
  90                 if(!text_.empty() && *(text_.rbegin()) == '\n'){
  91                         text.set_text(std::string(text_.begin(), text_.end()-1), use_markup_);
  92                 } else {
  93                         text.set_text(text_, use_markup_);
  94                 }
  95 
  96                 surface foreground = text.render();
  97 
  98                 if(foreground == nullptr) {
  99                         ERR_FT << "could not create floating label's text" << std::endl;
 100                         return nullptr;
 101                 }
 102 
 103                 // combine foreground text with its background
 104                 if(bgalpha_ != 0) {
 105                         // background is a dark tooltip box
 106                         surface background = create_neutral_surface(foreground->w + border_*2, foreground->h + border_*2);
 107 
 108                         if (background == nullptr) {
 109                                 ERR_FT << "could not create tooltip box" << std::endl;
 110                                 return surf_ = foreground;
 111                         }
 112 
 113                         Uint32 color = SDL_MapRGBA(foreground->format, bgcolor_.r,bgcolor_.g, bgcolor_.b, bgalpha_);
 114                         sdl::fill_rect(background,nullptr, color);
 115 
 116                         // we make the text less transparent, because the blitting on the
 117                         // dark background will darken the anti-aliased part.
 118                         // This 1.13 value seems to restore the brightness of version 1.4
 119                         // (where the text was blitted directly on screen)
 120                         adjust_surface_alpha(foreground, ftofxp(1.13));
 121 
 122                         SDL_Rect r {border_, border_, 0, 0};
 123                         adjust_surface_alpha(foreground, SDL_ALPHA_OPAQUE);
 124                         sdl_blit(foreground, nullptr, background, &r);
 125 
 126                         surf_ = background;
 127                 }
 128                 else {
 129                         // background is blurred shadow of the text
 130                         surface background = create_neutral_surface
 131                                 (foreground->w + 4, foreground->h + 4);
 132                         sdl::fill_rect(background, nullptr, 0);
 133                         SDL_Rect r { 2, 2, 0, 0 };
 134                         sdl_blit(foreground, nullptr, background, &r);
 135                         background = shadow_image(background);
 136 
 137                         if (background == nullptr) {
 138                                 ERR_FT << "could not create floating label's shadow" << std::endl;
 139                                 return surf_ = foreground;
 140                         }
 141                         sdl_blit(foreground, nullptr, background, &r);
 142                         surf_ = background;
 143                 }
 144         }
 145 
 146         return surf_;
 147 }

 149 void floating_label::draw(surface screen)
 150 {
 151         if(!visible_) {
 152                 buf_.assign(nullptr);
 153                 return;
 154         }
 155 
 156         if(screen == nullptr) {
 157                 return;
 158         }
 159 
 160         create_surface();
 161         if(surf_ == nullptr) {
 162                 return;
 163         }
 164 
 165         if(buf_ == nullptr) {
 166                 buf_.assign(create_compatible_surface(screen, surf_->w, surf_->h));
 167                 if(buf_ == nullptr) {
 168                         return;
 169                 }
 170         }
 171 
 172         SDL_Rect rect = sdl::create_rect(xpos(surf_->w), ypos_, surf_->w, surf_->h);
 173         const clip_rect_setter clip_setter(screen, &clip_rect_);
 174         sdl_copy_portion(screen,&rect,buf_,nullptr);
 175         sdl_blit(surf_,nullptr,screen,&rect);
 176 }

 178 void floating_label::undraw(surface screen)
 179 {
 180         if(screen == nullptr || buf_ == nullptr) {
 181                 return;
 182         }
 183         SDL_Rect rect = sdl::create_rect(xpos(surf_->w), ypos_, surf_->w, surf_->h);
 184         const clip_rect_setter clip_setter(screen, &clip_rect_);
 185         sdl_blit(buf_,nullptr,screen,&rect);
 186 
 187         move(xmove_,ymove_);
 188         if(lifetime_ > 0) {
 189                 --lifetime_;
 190                 if(alpha_change_ != 0 && (xmove_ != 0.0 || ymove_ != 0.0) && surf_ != nullptr) {
 191                         // fade out moving floating labels
 192                         surf_.assign(adjust_surface_alpha_add(surf_,alpha_change_));
 193                 }
 194         }
 195 }

 197 int add_floating_label(const floating_label& flabel)
 198 {
 199         if(label_contexts.empty()) {
 200                 return 0;
 201         }
 202 
 203         ++label_id;
 204         labels.emplace(label_id, flabel);
 205         label_contexts.top().insert(label_id);
 206         return label_id;
 207 }

 209 void move_floating_label(int handle, double xmove, double ymove)
 210 {
 211         const label_map::iterator i = labels.find(handle);
 212         if(i != labels.end()) {
 213                 i->second.move(xmove,ymove);
 214         }

 217 void scroll_floating_labels(double xmove, double ymove)
 218 {
 219         for(label_map::iterator i = labels.begin(); i != labels.end(); ++i) {
 220                 if(i->second.scroll() == ANCHOR_LABEL_MAP) {
 221                         i->second.move(xmove,ymove);
 222                 }
 223         }
 224 }

 226 void remove_floating_label(int handle)
 227 {
 228         const label_map::iterator i = labels.find(handle);
 229         if(i != labels.end()) {
 230                 if(label_contexts.empty() == false) {
 231                         label_contexts.top().erase(i->first);
 232                 }
 233 
 234                 labels.erase(i);
 235         }
 236 }

 246 SDL_Rect get_floating_label_rect(int handle)
 247 {
 248         const label_map::iterator i = labels.find(handle);
 249         if(i != labels.end()) {
 250                 const surface surf = i->second.create_surface();
 251                 if(surf != nullptr) {
 252                         return {0, 0, surf->w, surf->h};
 253                 }
 254         }
 255         return sdl::empty_rect;

 258 floating_label_context::floating_label_context()

 261 
 262         label_contexts.push(std::set<int>());
 263 }

 265 floating_label_context::~floating_label_context()
 266 {
 267         const std::set<int>& context = label_contexts.top();
 268         while (!context.empty())
 269         {
 270                 // Remove_floating_label removes the passed label from the context.
 271                 // This loop removes a different label in every iteration.
 272                 remove_floating_label(*context.begin());
 273         }
 274 
 275         label_contexts.pop();
 276 
 277         const surface screen = nullptr;
 278 }

 280 void draw_floating_labels(surface screen)
 281 {
 282         if(label_contexts.empty()) {
 283                 return;
 284         }
 285 
 286         const std::set<int>& context = label_contexts.top();
 287 
 288         //draw the labels in the order they were added, so later added labels (likely to be tooltips)
 289         //are displayed over earlier added labels.
 290         for(label_map::iterator i = labels.begin(); i != labels.end(); ++i) {
 291                 if(context.count(i->first) > 0) {
 292                         i->second.draw(screen);
 293                 }
 294         }
 295 }

 297 void undraw_floating_labels(surface screen)
 298 {
 299         if(label_contexts.empty()) {
 300                 return;
 301         }
 302 
 303         std::set<int>& context = label_contexts.top();
 304 
 305         //undraw labels in reverse order, so that a LIFO process occurs, and the screen is restored
 306         //into the exact state it started in.
 307         for(label_map::reverse_iterator i = labels.rbegin(); i != labels.rend(); ++i) {
 308                 if(context.count(i->first) > 0) {
 309                         i->second.undraw(screen);
 310                 }
 311         }
 312 
 313         //remove expired labels
 314         for(label_map::iterator j = labels.begin(); j != labels.end(); ) {
 315                 if(context.count(j->first) > 0 && j->second.expired()) {
 316                         context.erase(j->first);
 317                         labels.erase(j++);
 318                 } else {
 319                         ++j;
 320                 }
 321         }
 322 }
