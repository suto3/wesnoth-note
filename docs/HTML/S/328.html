  44 static size_t compute(std::string expr, size_t ref1, size_t ref2 = 0)
  45 {
  46         size_t ref = 0;
  47         if(expr[0] == '=') {
  48                 ref = ref1;
  49                 expr = expr.substr(1);
  50         } else if((expr[0] == '+') || (expr[0] == '-')) {
  51                 ref = ref2;
  52         }
  53 
  54         return ref + atoi(expr.c_str());
  55 }

  57 // If x2 or y2 are not specified, use x1 and y1 values
  58 static _rect read_rect(const config& cfg)
  59 {
  60         _rect rect {0, 0, 0, 0};
  61         std::vector<std::string> items = utils::split(cfg["rect"].str());
  62         if(items.size() >= 1)
  63                 rect.x1 = atoi(items[0].c_str());
  64 
  65         if(items.size() >= 2)
  66                 rect.y1 = atoi(items[1].c_str());
  67 
  68         if(items.size() >= 3)
  69                 rect.x2 = atoi(items[2].c_str());
  70         else
  71                 rect.x2 = rect.x1;
  72 
  73         if(items.size() >= 4)
  74                 rect.y2 = atoi(items[3].c_str());
  75         else
  76                 rect.y2 = rect.y1;
  77 
  78         return rect;
  79 }

  81 static SDL_Rect read_sdl_rect(const config& cfg)
  82 {
  83         SDL_Rect sdlrect;
  84         const _rect rect = read_rect(cfg);
  85         sdlrect.x = rect.x1;
  86         sdlrect.y = rect.y1;
  87         sdlrect.w = (rect.x2 > rect.x1) ? (rect.x2 - rect.x1) : 0;
  88         sdlrect.h = (rect.y2 > rect.y1) ? (rect.y2 - rect.y1) : 0;
  89 
  90         return sdlrect;
  91 }

  93 static std::string resolve_rect(const std::string& rect_str)
  94 {
  95         _rect rect {0, 0, 0, 0};
  96         std::stringstream resolved;
  97         const std::vector<std::string> items = utils::split(rect_str.c_str());
  98         if(items.size() >= 1) {
  99                 rect.x1 = compute(items[0], ref_rect.x1, ref_rect.x2);
 100                 resolved << rect.x1;
 101         }
 102         if(items.size() >= 2) {
 103                 rect.y1 = compute(items[1], ref_rect.y1, ref_rect.y2);
 104                 resolved << "," << rect.y1;
 105         }
 106         if(items.size() >= 3) {
 107                 rect.x2 = compute(items[2], ref_rect.x2, rect.x1);
 108                 resolved << "," << rect.x2;
 109         }
 110         if(items.size() >= 4) {
 111                 rect.y2 = compute(items[3], ref_rect.y2, rect.y1);
 112                 resolved << "," << rect.y2;
 113         }
 114 
 115         // DBG_DP << "Rect " << rect_str << "\t: " << resolved.str() << "\n";
 116 
 117         ref_rect = rect;
 118         return resolved.str();
 119 }

 121 static config& find_ref(const std::string& id, config& cfg, bool remove = false)
 122 {
 123         static config empty_config;
 124 
 125         config::all_children_itors itors = cfg.all_children_range();
 126         for(config::all_children_iterator i = itors.begin(); i != itors.end(); ++i) {
 127                 config& icfg = i->cfg;
 128                 if(i->cfg["id"] == id) {
 129                         if(remove) {
 130                                 cfg.erase(i);
 131                                 return empty_config;
 132                         } else {
 133                                 return icfg;
 134                         }
 135                 }
 136 
 137                 // Recursively look in children.
 138                 config& c = find_ref(id, icfg, remove);
 139                 if(&c != &empty_config) {
 140                         return c;
 141                 }
 142         }
 143 
 144         // Not found.
 145         return empty_config;
 146 }

 165 /**
 166  * Returns a copy of the wanted resolution.
 167  *
 168  * The function returns a copy since our caller uses a copy of this resolution
 169  * as base to expand a partial resolution.
 170  *
 171  * @param resolutions             A config object containing the expanded
 172  *                                resolutions.
 173  * @param id                      The id of the resolution to return.
 174  *
 175  * @throw config::error           If the @p id is not found.
 176  *
 177  * @returns                       A copy of the resolution config.
 178  */
 179 static config get_resolution(const config& resolutions, const std::string& id)
 180 {
 181         for(const auto& resolution : resolutions.child_range("resolution")) {
 182                 if(resolution["id"] == id) {
 183                         return resolution;
 184                 }
 185         }
 186 
 187         throw config::error("[partialresolution] refers to non-existent [resolution] " + id);
 188 }

 190 /**
 191  * Returns a config with all partial resolutions of a theme expanded.
 192  *
 193  * @param theme                   The original object, whose objects need to be
 194  *                                expanded.
 195  *
 196  * @returns                       A new object with the expanded resolutions in
 197  *                                a theme. This object no longer contains
 198  *                                partial resolutions.
 199  */
 200 static config expand_partialresolution(const config& theme)
 201 {
 202         config result;
 203 
 204         // Add all the resolutions
 205         for(const auto& resolution : theme.child_range("resolution")) {
 206                 result.add_child("resolution", resolution);
 207         }
 208 
 209         // Resolve all the partialresolutions
 210         for(const auto& part : theme.child_range("partialresolution")) {
 211                 config resolution = get_resolution(result, part["inherits"]);
 212                 resolution.merge_attributes(part);
 213 
 214                 for(const auto& remove : part.child_range("remove")) {
 215                         VALIDATE(!remove["id"].empty(), missing_mandatory_wml_key("[theme][partialresolution][remove]", "id"));
 216 
 217                         find_ref(remove["id"], resolution, true);
 218                 }
 219 
 220                 for(const auto& change : part.child_range("change")) {
 221                         VALIDATE(!change["id"].empty(), missing_mandatory_wml_key("[theme][partialresolution][change]", "id"));
 222 
 223                         config& target = find_ref(change["id"], resolution, false);
 224                         target.merge_attributes(change);
 225                 }
 226 
 227                 // cannot add [status] sub-elements, but who cares
 228                 for(const auto& add : part.child_range("add")) {
 229                         for(const auto& child : add.all_children_range()) {
 230                                 resolution.add_child(child.key, child.cfg);
 231                         }
 232                 }
 233 
 234                 result.add_child("resolution", resolution);
 235         }
 236 
 237         return result;
 238 }

 240 static void do_resolve_rects(const config& cfg, config& resolved_config, config* resol_cfg = nullptr)
 241 {
 242         // recursively resolve children
 243         for(const config::any_child& value : cfg.all_children_range()) {
 244                 config& childcfg = resolved_config.add_child(value.key);
 245                 do_resolve_rects(value.cfg, childcfg, value.key == "resolution" ? &childcfg : resol_cfg);
 246         }
 247 
 248         // copy all key/values
 249         resolved_config.merge_attributes(cfg);
 250 
 251         // override default reference rect with "ref" parameter if any
 252         if(!cfg["ref"].empty()) {
 253                 if(resol_cfg == nullptr) {
 254                         ERR_DP << "Use of ref= outside a [resolution] block" << std::endl;
 255                 } else {
 256                         // DBG_DP << ">> Looking for " << cfg["ref"] << "\n";
 257                         const config& ref = find_ref(cfg["ref"], *resol_cfg);
 258 
 259                         if(ref["id"].empty()) {
 260                                 ERR_DP << "Reference to non-existent rect id \"" << cfg["ref"] << "\"" << std::endl;
 261                         } else if(ref["rect"].empty()) {
 262                                 ERR_DP << "Reference to id \"" << cfg["ref"] << "\" which does not have a \"rect\"\n";
 263                         } else {
 264                                 ref_rect = read_rect(ref);
 265                         }
 266                 }
 267         }
 268         // resolve the rect value to absolute coordinates
 269         if(!cfg["rect"].empty()) {
 270                 resolved_config["rect"] = resolve_rect(cfg["rect"]);
 271         }
 272 }

 311 SDL_Rect& theme::object::location(const SDL_Rect& screen) const
 312 {
 313         if(last_screen_ == screen && !location_modified_)
 314                 return relative_loc_;
 315 
 316         last_screen_ = screen;
 317 
 318         switch(xanchor_) {
 319                 case FIXED:
 320                         relative_loc_.x = loc_.x;
 321                         relative_loc_.w = loc_.w;
 322                         break;
 323                 case TOP_ANCHORED:
 324                         relative_loc_.x = loc_.x;
 325                         relative_loc_.w = screen.w - std::min<size_t>(XDim - loc_.w, screen.w);
 326                         break;
 327                 case BOTTOM_ANCHORED:
 328                         relative_loc_.x = screen.w - std::min<size_t>(XDim - loc_.x, screen.w);
 329                         relative_loc_.w = loc_.w;
 330                         break;
 331                 case PROPORTIONAL:
 332                         relative_loc_.x = (loc_.x * screen.w) / XDim;
 333                         relative_loc_.w = (loc_.w * screen.w) / XDim;
 334                         break;
 335                 default:
 336                         assert(false);
 337         }
 338 
 339         switch(yanchor_) {
 340                 case FIXED:
 341                         relative_loc_.y = loc_.y;
 342                         relative_loc_.h = loc_.h;
 343                         break;
 344                 case TOP_ANCHORED:
 345                         relative_loc_.y = loc_.y;
 346                         relative_loc_.h = screen.h - std::min<size_t>(YDim - loc_.h, screen.h);
 347                         break;
 348                 case BOTTOM_ANCHORED:
 349                         relative_loc_.y = screen.h - std::min<size_t>(YDim - loc_.y, screen.h);
 350                         relative_loc_.h = loc_.h;
 351                         break;
 352                 case PROPORTIONAL:
 353                         relative_loc_.y = (loc_.y * screen.h) / YDim;
 354                         relative_loc_.h = (loc_.h * screen.h) / YDim;
 355                         break;
 356                 default:
 357                         assert(false);
 358         }
 359 
 360         relative_loc_.x = std::min<int>(relative_loc_.x, screen.w);
 361         relative_loc_.w = std::min<int>(relative_loc_.w, screen.w - relative_loc_.x);
 362         relative_loc_.y = std::min<int>(relative_loc_.y, screen.h);
 363         relative_loc_.h = std::min<int>(relative_loc_.h, screen.h - relative_loc_.y);
 364 
 365         return relative_loc_;
 366 }

 368 theme::object::ANCHORING theme::object::read_anchor(const std::string& str)
 369 {
 370         static const std::string top_anchor = "top", left_anchor = "left", bot_anchor = "bottom", right_anchor = "right",
 371                                                          proportional_anchor = "proportional";
 372         if(str == top_anchor || str == left_anchor)
 373                 return TOP_ANCHORED;
 374         else if(str == bot_anchor || str == right_anchor)
 375                 return BOTTOM_ANCHORED;
 376         else if(str == proportional_anchor)
 377                 return PROPORTIONAL;
 378         else
 379                 return FIXED;
 380 }

 391 void theme::object::modify_location(std::string rect_str, SDL_Rect location_ref_rect)
 392 {
 393         _rect rect {0, 0, 0, 0};
 394         const std::vector<std::string> items = utils::split(rect_str.c_str());
 395         if(items.size() >= 1) {
 396                 rect.x1 = compute(items[0], location_ref_rect.x, location_ref_rect.x + location_ref_rect.w);
 397         }
 398         if(items.size() >= 2) {
 399                 rect.y1 = compute(items[1], location_ref_rect.y, location_ref_rect.y + location_ref_rect.h);
 400         }
 401         if(items.size() >= 3) {
 402                 rect.x2 = compute(items[2], location_ref_rect.x + location_ref_rect.w, rect.x1);
 403         }
 404         if(items.size() >= 4) {
 405                 rect.y2 = compute(items[3], location_ref_rect.y + location_ref_rect.h, rect.y1);
 406         }
 407         modify_location(rect);
 408 }

 585 bool theme::set_resolution(const SDL_Rect& screen)
 586 {
 587         bool result = false;
 588 
 589         int current_rating = 1000000;
 590         const config* current = nullptr;
 591         for(const config& i : cfg_.child_range("resolution")) {
 592                 int width = i["width"];
 593                 int height = i["height"];
 594                 LOG_DP << "comparing resolution " << screen.w << "," << screen.h << " to " << width << "," << height << "\n";
 595                 if(screen.w >= width && screen.h >= height) {
 596                         LOG_DP << "loading theme: " << width << "," << height << "\n";
 597                         current = &i;
 598                         result = true;
 599                         break;
 600                 }
 601 
 602                 const int rating = width * height;
 603                 if(rating < current_rating) {
 604                         current = &i;
 605                         current_rating = rating;
 606                 }
 607         }
 608 
 609         if(!current) {
 610                 if(cfg_.child_count("resolution")) {
 611                         ERR_DP << "No valid resolution found" << std::endl;
 612                 }
 613                 return false;
 614         }
 615 
 616         std::map<std::string, std::string> title_stash_menus;
 617         std::vector<theme::menu>::iterator m;
 618         for(m = menus_.begin(); m != menus_.end(); ++m) {
 619                 if(!m->title().empty() && !m->get_id().empty())
 620                         title_stash_menus[m->get_id()] = m->title();
 621         }
 622 
 623         std::map<std::string, std::string> title_stash_actions;
 624         std::vector<theme::action>::iterator a;
 625         for(a = actions_.begin(); a != actions_.end(); ++a) {
 626                 if(!a->title().empty() && !a->get_id().empty())
 627                         title_stash_actions[a->get_id()] = a->title();
 628         }
 629 
 630         panels_.clear();
 631         labels_.clear();
 632         status_.clear();
 633         menus_.clear();
 634         actions_.clear();
 635         sliders_.clear();
 636 
 637         add_object(*current);
 638 
 639         for(m = menus_.begin(); m != menus_.end(); ++m) {
 640                 if(title_stash_menus.find(m->get_id()) != title_stash_menus.end())
 641                         m->set_title(title_stash_menus[m->get_id()]);
 642         }
 643 
 644         for(a = actions_.begin(); a != actions_.end(); ++a) {
 645                 if(title_stash_actions.find(a->get_id()) != title_stash_actions.end())
 646                         a->set_title(title_stash_actions[a->get_id()]);
 647         }
 648 
 649         theme_reset_event_.notify_observers();
 650 
 651         return result;
 652 }

 654 void theme::add_object(const config& cfg)
 655 {
 656         if(const config& c = cfg.child("main_map")) {
 657                 main_map_ = object(c);
 658         }
 659 
 660         if(const config& c = cfg.child("mini_map")) {
 661                 mini_map_ = object(c);
 662         }
 663 
 664         if(const config& c = cfg.child("palette")) {
 665                 palette_ = object(c);
 666         }
 667 
 668         if(const config& status_cfg = cfg.child("status")) {
 669                 for(const config::any_child& i : status_cfg.all_children_range()) {
 670                         status_.emplace(i.key, status_item(i.cfg));
 671                 }
 672                 if(const config& unit_image_cfg = status_cfg.child("unit_image")) {
 673                         unit_image_ = object(unit_image_cfg);
 674                 } else {
 675                         unit_image_ = object();
 676                 }
 677         }
 678 
 679         for(const config& p : cfg.child_range("panel")) {
 680                 panel new_panel(p);
 681                 set_object_location(new_panel, p["rect"], p["ref"]);
 682                 panels_.push_back(new_panel);
 683         }
 684 
 685         for(const config& lb : cfg.child_range("label")) {
 686                 label new_label(lb);
 687                 set_object_location(new_label, lb["rect"], lb["ref"]);
 688                 labels_.push_back(new_label);
 689         }
 690 
 691         for(const config& m : cfg.child_range("menu")) {
 692                 menu new_menu(m);
 693                 DBG_DP << "adding menu: " << (new_menu.is_context() ? "is context" : "not context") << "\n";
 694                 if(new_menu.is_context())
 695                         context_ = new_menu;
 696                 else {
 697                         set_object_location(new_menu, m["rect"], m["ref"]);
 698                         menus_.push_back(new_menu);
 699                 }
 700 
 701                 DBG_DP << "done adding menu...\n";
 702         }
 703 
 704         for(const config& a : cfg.child_range("action")) {
 705                 action new_action(a);
 706                 DBG_DP << "adding action: " << (new_action.is_context() ? "is context" : "not context") << "\n";
 707                 if(new_action.is_context())
 708                         action_context_ = new_action;
 709                 else {
 710                         set_object_location(new_action, a["rect"], a["ref"]);
 711                         actions_.push_back(new_action);
 712                 }
 713 
 714                 DBG_DP << "done adding action...\n";
 715         }
 716 
 717         for(const config& s : cfg.child_range("slider")) {
 718                 slider new_slider(s);
 719                 DBG_DP << "adding slider\n";
 720                 set_object_location(new_slider, s["rect"], s["ref"]);
 721                 sliders_.push_back(new_slider);
 722 
 723                 DBG_DP << "done adding slider...\n";
 724         }
 725 
 726         if(const config& c = cfg.child("main_map_border")) {
 727                 border_ = border_t(c);
 728         }
 729 }

 731 void theme::remove_object(const std::string& id)
 732 {
 733         for(std::vector<theme::panel>::iterator p = panels_.begin(); p != panels_.end(); ++p) {
 734                 if(p->get_id() == id) {
 735                         panels_.erase(p);
 736                         return;
 737                 }
 738         }
 739         for(std::vector<theme::label>::iterator l = labels_.begin(); l != labels_.end(); ++l) {
 740                 if(l->get_id() == id) {
 741                         labels_.erase(l);
 742                         return;
 743                 }
 744         }
 745         for(std::vector<theme::menu>::iterator m = menus_.begin(); m != menus_.end(); ++m) {
 746                 if(m->get_id() == id) {
 747                         menus_.erase(m);
 748                         return;
 749                 }
 750         }
 751         for(std::vector<theme::action>::iterator a = actions_.begin(); a != actions_.end(); ++a) {
 752                 if(a->get_id() == id) {
 753                         actions_.erase(a);
 754                         return;
 755                 }
 756         }
 757         for(std::vector<theme::slider>::iterator s = sliders_.begin(); s != sliders_.end(); ++s) {
 758                 if(s->get_id() == id) {
 759                         sliders_.erase(s);
 760                         return;
 761                 }
 762         }
 763 }

 765 void theme::set_object_location(theme::object& element, std::string rect_str, std::string ref_id)
 766 {
 767         theme::object ref_element = element;
 768         if(ref_id.empty()) {
 769                 ref_id = element.get_id();
 770         } else {
 771                 ref_element = find_element(ref_id);
 772         }
 773         if(ref_element.get_id() == ref_id) {
 774                 SDL_Rect location_ref_rect = ref_element.get_location();
 775                 element.modify_location(rect_str, location_ref_rect);
 776         }

 779 void theme::modify(const config& cfg)
 780 {
 781         std::map<std::string, std::string> title_stash;
 782         std::vector<theme::menu>::iterator m;
 783         for(m = menus_.begin(); m != menus_.end(); ++m) {
 784                 if(!m->title().empty() && !m->get_id().empty())
 785                         title_stash[m->get_id()] = m->title();
 786         }
 787 
 788         std::vector<theme::action>::iterator a;
 789         for(a = actions_.begin(); a != actions_.end(); ++a) {
 790                 if(!a->title().empty() && !a->get_id().empty())
 791                         title_stash[a->get_id()] = a->title();
 792         }
 793 
 794         // Change existing theme objects.
 795         for(const config& c : cfg.child_range("change")) {
 796                 std::string id = c["id"];
 797                 std::string ref_id = c["ref"];
 798                 theme::object& element = find_element(id);
 799                 if(element.get_id() == id)
 800                         set_object_location(element, c["rect"], ref_id);
 801         }
 802 
 803         // Add new theme objects.
 804         for(const config& c : cfg.child_range("add")) {
 805                 add_object(c);
 806         }
 807 
 808         // Remove existent theme objects.
 809         for(const config& c : cfg.child_range("remove")) {
 810                 remove_object(c["id"]);
 811         }
 812 
 813         for(m = menus_.begin(); m != menus_.end(); ++m) {
 814                 if(title_stash.find(m->get_id()) != title_stash.end())
 815                         m->set_title(title_stash[m->get_id()]);
 816         }
 817         for(a = actions_.begin(); a != actions_.end(); ++a) {
 818                 if(title_stash.find(a->get_id()) != title_stash.end())
 819                         a->set_title(title_stash[a->get_id()]);
 820         }
 821 }

 823 theme::object& theme::find_element(const std::string& id)
 824 {
 825         static theme::object empty_object;
 826         theme::object* res = &empty_object;
 827         for(std::vector<theme::panel>::iterator p = panels_.begin(); p != panels_.end(); ++p) {
 828                 if(p->get_id() == id) {
 829                         res = &(*p);
 830                 }
 831         }
 832         for(std::vector<theme::label>::iterator l = labels_.begin(); l != labels_.end(); ++l) {
 833                 if(l->get_id() == id) {
 834                         res = &(*l);
 835                 }
 836         }
 837         for(std::vector<theme::menu>::iterator m = menus_.begin(); m != menus_.end(); ++m) {
 838                 if(m->get_id() == id) {
 839                         res = &(*m);
 840                 }
 841         }
 842         for(std::vector<theme::action>::iterator a = actions_.begin(); a != actions_.end(); ++a) {
 843                 if(a->get_id() == id) {
 844                         res = &(*a);
 845                 }
 846         }
 847         if(id == "main-map") {
 848                 res = &main_map_;
 849         }
 850         if(id == "mini-map") {
 851                 res = &mini_map_;
 852         }
 853         if(id == "palette") {
 854                 res = &palette_;
 855         }
 856         if(id == "unit-image") {
 857                 res = &unit_image_;
 858         }
 859         return *res;
 860 }
