  44 template<typename Map, typename Key>
  45 typename Map::mapped_type& map_get(Map& map, Key&& key)
  46 {
  47         auto res = map.lower_bound(key);
  48 
  49         if (res == map.end() || key != res->first) {
  50                 res = map.emplace_hint(res, std::piecewise_construct, std::forward_as_tuple(key), std::tuple<>());
  51         }
  52 
  53         return res->second;
  54 }

 423 config &config::add_child_at(config_key_type key, const config &val, unsigned index)
 424 {
 425         check_valid(val);
 426 
 427         child_list& v = map_get(children_, key);
 428         if(index > v.size()) {
 429                 throw error("illegal index to add child at");
 430         }
 431 
 432         v.emplace(v.begin()+index,new config(val));
 433 
 434         bool inserted = false;
 435 
 436         const child_pos value(children_.find(key),index);
 437 
 438         std::vector<child_pos>::iterator ord = ordered_children.begin();
 439         for(; ord != ordered_children.end(); ++ord) {
 440                 if (ord->pos != value.pos) continue;
 441                 if (!inserted && ord->index == index) {
 442                         ord = ordered_children.insert(ord,value);
 443                         inserted = true;
 444                 } else if (ord->index >= index) {
 445                         ord->index++;
 446                 }
 447         }
 448 
 449         if(!inserted) {
 450                 ordered_children.push_back(value);
 451         }
 452 
 453         return *v[index];
 454 }

 483 void config::splice_children(config &src, const std::string &key)
 484 {
 485         check_valid(src);
 486 
 487         child_map::iterator i_src = src.children_.find(key);
 488         if (i_src == src.children_.end()) return;
 489 
 490         src.ordered_children.erase(std::remove_if(src.ordered_children.begin(),
 491                 src.ordered_children.end(), remove_ordered(i_src)),
 492                 src.ordered_children.end());
 493 
 494         child_list &dst = map_get(children_, key);
 495         child_map::iterator i_dst = children_.find(key);
 496 
 497         unsigned before = dst.size();
 498         dst.insert(dst.end(), std::make_move_iterator(i_src->second.begin()), std::make_move_iterator(i_src->second.end()));
 499         src.children_.erase(i_src);
 500         // key might be a reference to i_src->first, so it is no longer usable.
 501 
 502         for (unsigned j = before; j < dst.size(); ++j) {
 503                 ordered_children.emplace_back(i_dst, j);
 504         }
 505 }

 520 std::vector<config::child_pos>::iterator config::remove_child(
 521         const child_map::iterator &pos, unsigned index)
 522 {
 523         /* Find the position with the correct index and decrement all the
 524            indices in the ordering that are above this index. */
 525         unsigned found = 0;
 526         for (child_pos &p : ordered_children)
 527         {
 528                 if (p.pos != pos) continue;
 529                 if (p.index == index)
 530                         found = &p - &ordered_children.front();
 531                 else if (p.index > index)
 532                         --p.index;
 533         }
 534 
 535         // Remove from the child map.
 536         pos->second.erase(pos->second.begin() + index);
 537 
 538         // Erase from the ordering and return the next position.
 539         return ordered_children.erase(ordered_children.begin() + found);
 540 }

 762 void config::get_diff(const config& c, config& res) const
 763 {
 764         check_valid(c);
 765         check_valid(res);
 766 
 767         config* inserts = nullptr;
 768 
 769         attribute_map::const_iterator i;
 770         for(i = values_.begin(); i != values_.end(); ++i) {
 771                 if(i->second.blank()) continue;
 772                 const attribute_map::const_iterator j = c.values_.find(i->first);
 773                 if(j == c.values_.end() || (i->second != j->second && !i->second.blank() )) {
 774                         if(inserts == nullptr) {
 775                                 inserts = &res.add_child("insert");
 776                         }
 777 
 778                         (*inserts)[i->first] = i->second;
 779                 }
 780         }
 781 
 782         config* deletes = nullptr;
 783 
 784         for(i = c.values_.begin(); i != c.values_.end(); ++i) {
 785                 if(i->second.blank()) continue;
 786                 const attribute_map::const_iterator itor = values_.find(i->first);
 787                 if(itor == values_.end() || itor->second.blank()) {
 788                         if(deletes == nullptr) {
 789                                 deletes = &res.add_child("delete");
 790                         }
 791 
 792                         (*deletes)[i->first] = "x";
 793                 }
 794         }
 795 
 796         std::vector<std::string> entities;
 797 
 798         child_map::const_iterator ci;
 799         for(ci = children_.begin(); ci != children_.end(); ++ci) {
 800                 entities.push_back(ci->first);
 801         }
 802 
 803         for(ci = c.children_.begin(); ci != c.children_.end(); ++ci) {
 804                 if(children_.count(ci->first) == 0) {
 805                         entities.push_back(ci->first);
 806                 }
 807         }
 808 
 809         for(std::vector<std::string>::const_iterator itor = entities.begin(); itor != entities.end(); ++itor) {
 810 
 811                 const child_map::const_iterator itor_a = children_.find(*itor);
 812                 const child_map::const_iterator itor_b = c.children_.find(*itor);
 813 
 814                 static const child_list dummy;
 815 
 816                 // Get the two child lists. 'b' has to be modified to look like 'a'.
 817                 const child_list& a = itor_a != children_.end() ? itor_a->second : dummy;
 818                 const child_list& b = itor_b != c.children_.end() ? itor_b->second : dummy;
 819 
 820                 size_t ndeletes = 0;
 821                 size_t ai = 0, bi = 0;
 822                 while(ai != a.size() || bi != b.size()) {
 823                         // If the two elements are the same, nothing needs to be done.
 824                         if(ai < a.size() && bi < b.size() && *a[ai] == *b[bi]) {
 825                                 ++ai;
 826                                 ++bi;
 827                         } else {
 828                                 // We have to work out what the most appropriate operation --
 829                                 // delete, insert, or change is the best to get b[bi] looking like a[ai].
 830                                 std::stringstream buf;
 831 
 832                                 // If b has more elements than a, then we assume this element
 833                                 // is an element that needs deleting.
 834                                 if(b.size() - bi > a.size() - ai) {
 835                                         config& new_delete = res.add_child("delete_child");
 836                                         buf << bi - ndeletes;
 837                                         new_delete.values_["index"] = buf.str();
 838                                         new_delete.add_child(*itor);
 839 
 840                                         ++ndeletes;
 841                                         ++bi;
 842                                 }
 843 
 844                                 // If b has less elements than a, then we assume this element
 845                                 // is an element that needs inserting.
 846                                 else if(b.size() - bi < a.size() - ai) {
 847                                         config& new_insert = res.add_child("insert_child");
 848                                         buf << ai;
 849                                         new_insert.values_["index"] = buf.str();
 850                                         new_insert.add_child(*itor,*a[ai]);
 851 
 852                                         ++ai;
 853                                 }
 854 
 855                                 // Otherwise, they have the same number of elements,
 856                                 // so try just changing this element to match.
 857                                 else {
 858                                         config& new_change = res.add_child("change_child");
 859                                         buf << bi;
 860                                         new_change.values_["index"] = buf.str();
 861                                         new_change.add_child(*itor,a[ai]->get_diff(*b[bi]));
 862 
 863                                         ++ai;
 864                                         ++bi;
 865                                 }
 866                         }
 867                 }
 868         }
 869 }

 871 void config::apply_diff(const config& diff, bool track /* = false */)
 872 {
 873         check_valid(diff);
 874 
 875         if (track) values_[diff_track_attribute] = "modified";
 876 
 877         if (const config &inserts = diff.child("insert")) {
 878                 for (const attribute &v : inserts.attribute_range()) {
 879                         values_[v.first] = v.second;
 880                 }
 881         }
 882 
 883         if (const config &deletes = diff.child("delete")) {
 884                 for (const attribute &v : deletes.attribute_range()) {
 885                         values_.erase(v.first);
 886                 }
 887         }
 888 
 889         for (const config &i : diff.child_range("change_child"))
 890         {
 891                 const size_t index = lexical_cast<size_t>(i["index"].str());
 892                 for (const any_child &item : i.all_children_range())
 893                 {
 894                         if (item.key.empty()) {
 895                                 continue;
 896                         }
 897 
 898                         const child_map::iterator itor = children_.find(item.key);
 899                         if(itor == children_.end() || index >= itor->second.size()) {
 900                                 throw error("error in diff: could not find element '" + item.key + "'");
 901                         }
 902 
 903                         itor->second[index]->apply_diff(item.cfg, track);
 904                 }
 905         }
 906 
 907         for (const config &i : diff.child_range("insert_child"))
 908         {
 909                 const size_t index = lexical_cast<size_t>(i["index"].str());
 910                 for (const any_child &item : i.all_children_range()) {
 911                         config& inserted = add_child_at(item.key, item.cfg, index);
 912                         if (track) inserted[diff_track_attribute] = "new";
 913                 }
 914         }
 915 
 916         for (const config &i : diff.child_range("delete_child"))
 917         {
 918                 const size_t index = lexical_cast<size_t>(i["index"].str());
 919                 for (const any_child &item : i.all_children_range()) {
 920                         if (!track) {
 921                                 remove_child(item.key, index);
 922                         } else {
 923                                 const child_map::iterator itor = children_.find(item.key);
 924                                 if(itor == children_.end() || index >= itor->second.size()) {
 925                                         throw error("error in diff: could not find element '" + item.key + "'");
 926                                 }
 927                                 itor->second[index]->values_[diff_track_attribute] = "deleted";
 928                         }
 929                 }
 930         }
 931 }

 933 void config::clear_diff_track(const config& diff)
 934 {
 935         remove_attribute(diff_track_attribute);
 936         for (const config &i : diff.child_range("delete_child"))
 937         {
 938                 const size_t index = lexical_cast<size_t>(i["index"].str());
 939                 for (const any_child &item : i.all_children_range()) {
 940                         remove_child(item.key, index);
 941                 }
 942         }
 943 
 944         for (const config &i : diff.child_range("change_child"))
 945         {
 946                 const size_t index = lexical_cast<size_t>(i["index"].str());
 947                 for (const any_child &item : i.all_children_range())
 948                 {
 949                         if (item.key.empty()) {
 950                                 continue;
 951                         }
 952 
 953                         const child_map::iterator itor = children_.find(item.key);
 954                         if(itor == children_.end() || index >= itor->second.size()) {
 955                                 throw error("error in diff: could not find element '" + item.key + "'");
 956                         }
 957 
 958                         itor->second[index]->clear_diff_track(item.cfg);
 959                 }
 960         }
 961         for(std::pair<const std::string, child_list>& p : children_) {
 962                 for(auto& cfg : p.second) {
 963                         cfg->remove_attribute(diff_track_attribute);
 964                 }
 965         }

 968 /**
 969  * Merge config 'c' into this config, overwriting this config's values.
 970  */
 971 void config::merge_with(const config& c)
 972 {
 973         check_valid(c);
 974 
 975         std::vector<child_pos> to_remove;
 976         std::map<std::string, unsigned> visitations;
 977 
 978         // Merge attributes first
 979         merge_attributes(c);
 980 
 981         // Now merge shared tags
 982         all_children_iterator::Itor i, i_end = ordered_children.end();
 983         for(i = ordered_children.begin(); i != i_end; ++i) {
 984                 const std::string& tag = i->pos->first;
 985                 const child_map::const_iterator j = c.children_.find(tag);
 986                 if (j != c.children_.end()) {
 987                         unsigned &visits = visitations[tag];
 988                         if(visits < j->second.size()) {
 989                                 // Get a const config so we do not add attributes.
 990                                 const config & merge_child = *j->second[visits++];
 991 
 992                                 if ( merge_child["__remove"].to_bool() ) {
 993                                         to_remove.push_back(*i);
 994                                 } else
 995                                         (i->pos->second[i->index])->merge_with(merge_child);
 996                         }
 997                 }
 998         }
 999 
1000         // Now add any unvisited tags
1001         for(const auto& pair : c.children_) {
1002                 const std::string& tag = pair.first;
1003                 unsigned &visits = visitations[tag];
1004                 while(visits < pair.second.size()) {
1005                         add_child(tag, *pair.second[visits++]);
1006                 }
1007         }
1008 
1009         // Remove those marked so
1010         std::map<std::string, unsigned> removals;
1011         for (const child_pos& pos : to_remove) {
1012                 const std::string& tag = pos.pos->first;
1013                 unsigned &removes = removals[tag];
1014                 remove_child(tag, pos.index - removes++);
1015         }
1016 
1017 }

1021  */
1022 void config::inherit_from(const config& c)
1023 {
1024         // Using a scratch config and merge_with() seems to execute about as fast
1025         // as direct coding of this merge.
1026         config scratch(c);
1027         scratch.merge_with(*this);
1028         swap(scratch);
1029 }

1031 bool config::matches(const config &filter) const
1032 {
1033         check_valid(filter);
1034 
1035         for (const attribute &i : filter.attribute_range())
1036         {
1037                 const attribute_value *v = get(i.first);
1038                 if (!v || *v != i.second) return false;
1039         }
1040 
1041         for (const any_child &i : filter.all_children_range())
1042         {
1043                 if (i.key == "not") {
1044                         if (matches(i.cfg)) return false;
1045                         continue;
1046                 }
1047                 bool found = false;
1048                 for (const config &j : child_range(i.key)) {
1049                         if (j.matches(i.cfg)) {
1050                                 found = true;
1051                                 break;
1052                         }
1053                 }
1054                 if(!found) return false;
1055         }
1056         return true;
1057 }

1089 std::string config::hash() const
1090 {
1091         check_valid();
1092 
1093         static const unsigned int hash_length = 128;
1094         static const char hash_string[] =
1095                 "+-,.<>0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
1096         char hash_str[hash_length + 1];
1097 
1098         unsigned int i;
1099         for(i = 0; i != hash_length; ++i) {
1100                 hash_str[i] = 'a';
1101         }
1102         hash_str[hash_length] = 0;
1103 
1104         i = 0;
1105         for (const attribute &val : values_)
1106         {
1107                 if (val.second.blank()) {
1108                         continue;
1109                 }
1110                 for (char c : val.first) {
1111                         hash_str[i] ^= c;
1112                         if (++i == hash_length) i = 0;
1113                 }
1114                 std::string base_str = val.second.t_str().base_str();
1115                 for (std::string::const_iterator c = base_str.begin(); c != base_str.end(); ++c) {
1116                         hash_str[i] ^= *c;
1117                         if (++i == hash_length) i = 0;
1118                 }
1119         }
1120 
1121         for (const any_child &ch : all_children_range())
1122         {
1123                 std::string child_hash = ch.cfg.hash();
1124                 for (char c : child_hash) {
1125                         hash_str[i] ^= c;
1126                         ++i;
1127                         if(i == hash_length) {
1128                                 i = 0;
1129                         }
1130                 }
1131         }
1132 
1133         for(i = 0; i != hash_length; ++i) {
1134                 hash_str[i] = hash_string[
1135                         static_cast<unsigned>(hash_str[i]) % strlen(hash_string)];
1136         }
1137 
1138         return std::string(hash_str);
1139 }
