 192     def make_string(self, indent = 0, textdomain = ""):
 193         """Write the data object to the given file object."""
 194         ifdef = 0
 195         result = []
 196 
 197         self.clean_empty_ifdefs()
 198 
 199         for item in self.data:
 200             if ifdef:
 201                 if not isinstance(item, DataIfDef) or not item.type == "else":
 202                     result.append("#endif\n")
 203                 ifdef = 0
 204 
 205             if isinstance(item, DataIfDef):
 206                 if item.type == "else":
 207                     result.append("#else\n")
 208                 else:
 209                     result.append("#ifdef %s\n" % item.name)
 210                 result.append( item.make_string(indent + 4, textdomain) )
 211                 ifdef = 1
 212 
 213             elif isinstance(item, DataSub):
 214                 result.append(" " * indent)
 215                 result.append("[%s]\n" % item.name)
 216                 result.append(item.make_string(indent + 4, textdomain))
 217                 result.append(" " * indent)
 218                 close = item.name
 219                 if close[0] == "+": close = close[1:]
 220                 result.append("[/%s]\n" % close)
 221 
 222             elif isinstance(item, DataText):
 223                 result.append(" " * indent)
 224                 text = item.data.replace('"', r'""')
 225                 # We always enclosed in quotes
 226                 # In theory, the parser will just remove then on reading in, so
 227                 # the actual value is 1:1 preserved
 228                 # TODO: is there no catch?
 229                 if textdomain and item.textdomain == "wesnoth":
 230                     result.append("#textdomain wesnoth\n")
 231                     result.append(" " * indent)
 232 
 233                 # FIXME: don't compile regex's if they are one shots
 234                 if item.translatable:
 235                     if "=" in text:
 236                         text = re.compile("=(.*?)(?=[=;]|$)"
 237                             ).sub("=\" + _\"\\1\" + \"", text)
 238                         text = '"' + text + '"'
 239                         text = re.compile(r'\+ _""').sub("", text)
 240                         result.append('%s=%s\n' % (item.name, text))
 241                     else:
 242                         result.append('%s=_"%s"\n' % (item.name, text))
 243 
 244                 else:
 245                     result.append('%s="%s"\n' % (item.name, text))
 246 
 247                 if textdomain and item.textdomain == "wesnoth":
 248                     result.append(" " * indent)
 249                     result.append("#textdomain %s\n" % textdomain)
 250 
 251             elif isinstance(item, DataMacro):
 252                 result.append(" " * indent)
 253                 result.append("%s\n" % item.data)
 254 
 255             elif isinstance(item, DataComment):
 256                 result.append("%s\n" % item.data)
 257 
 258             elif isinstance(item, DataDefine):
 259                 result.append("#define %s %s\n%s#enddef\n" % (item.name, item.params,
 260                     item.data))
 261 
 262             elif isinstance(item, DataClosingTag):
 263                 result.append("[/%s]\n" % item.name)
 264 
 265             elif isinstance(item, DataBinary):
 266                 data = item.data.replace('"', r'""')
 267                 result.append("%s=\"%s\"\n" % (
 268                     item.name, data))
 269 
 270             else:
 271                 raise WMLException("Unknown item: %s" % item.__class__.__name__)
 272 
 273         if ifdef:
 274             result.append("#endif\n")
 275

 277         for r in result:
 278             if r != None:
 279                 # For networking, we need actual bytestream here, not unicode.
 280                 if type(r) is unicode: r = r.encode("utf8")
 281                 bytes += str(r)

 283         return bytes

 314     def insert_after(self, previous, data):
 315         """Insert after given node, or else insert as first."""
 316         if not previous in self.data: return self.insert_first(data)
 317         # completely rebuild list and dict
 318         new_childs = []
 319         self.dict = {}
 320         for child in self.data:
 321             new_childs += [child]
 322             self.place_dict(child)
 323             if child == previous:
 324                 new_childs += [data]
 325                 self.place_dict(data)
 326         self.data = new_childs

 328     def insert_at(self, pos, data):
 329         """Insert at given index (or as last)."""
 330         if pos >= len(self.data): return self.insert(data)
 331         # completely rebuild list and dict
 332         new_childs = []
 333         self.dict = {}
 334         i = 0
 335         for child in self.data:
 336             if i == pos:
 337                 new_childs += [data]
 338                 self.place_dict(data)
 339             new_childs += [child]
 340             self.place_dict(child)
 341             i += 1
 342 
 343         self.data = new_childs

 588     def set_quantity(self, name, difficulty, value):
 589         """Sets one of 3 values of a quantity. If it doesn't exist yet, also the
 590         other difficulties get the same value.
 591         """
 592         value = str(value)
 593         # read existing values
 594         q = []
 595         for d in xrange(3):
 596             q += [self.get_quantity(name, d, value)]
 597         q[difficulty] = value
 598 
 599         # remove current tags
 600         self.remove_text(name)
 601         for ifdef in self.get_ifdefs("EASY") + self.get_ifdefs("NORMAL") + self.get_ifdefs("HARD"):
 602             ifdef.remove_text(name)
 603 
 604         # insert updated item
 605         if q[0] == q[1] == q[2]:
 606             self.set_text_val(name, value)
 607         else:
 608             for d in xrange(3):
 609                 ifdef = self.get_or_create_ifdef(["EASY", "NORMAL", "HARD"][d])
