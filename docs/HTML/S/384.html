 427 template<typename widget>
 428 void mp_create_game::on_filter_change(window& window, const std::string& id)
 429 {
 430         create_engine_.apply_level_filter(find_widget<widget>(&window, id, false).get_value());
 431 
 432         listbox& game_list = find_widget<listbox>(&window, "games_list", false);
 433 
 434         boost::dynamic_bitset<> filtered(game_list.get_item_count());
 435         for(const size_t i : create_engine_.get_filtered_level_indices(create_engine_.current_level_type())) {
 436                 filtered[i] = true;
 437         }
 438 
 439         game_list.set_row_shown(filtered);
 440 
 441         on_game_select(window);
 442 }

 444 void mp_create_game::on_game_select(window& window)
 445 {
 446         const int selected_game = find_widget<listbox>(&window, "games_list", false).get_selected_row();
 447 
 448         if(selected_game == selected_game_index_) {
 449                 return;
 450         }
 451 
 452         // Convert the absolute-index get_selected_row to a relatve index for the create_engine to handle
 453         selected_game_index_ = convert_to_game_filtered_index(selected_game);
 454 
 455         create_engine_.set_current_level(selected_game_index_);
 456         update_details(window);
 457 
 458         // General settings
 459         stacked_widget& stack = find_widget<stacked_widget>(&window, "pager", false);
 460 
 461         const bool can_select_era = create_engine_.current_level().allow_era_choice();
 462 
 463         menu_button& era_combo = find_widget<menu_button>(stack.get_layer_grid(TAB_GENERAL), "eras", false);
 464         if(!can_select_era) {
 465                 era_combo.set_label(_("No eras available for this game."));
 466         } else {
 467                 era_combo.set_selected(era_combo.get_value());
 468         }
 469 
 470         era_combo.set_active(can_select_era);
 471 
 472         // Custom options
 473         options_manager_->update_game_options();
 474 
 475         // Game settings
 476         update_map_settings(window);
 477 }

 479 void mp_create_game::on_tab_select(window& window)
 480 {
 481         const int i = find_widget<listbox>(&window, "tab_bar", false).get_selected_row();
 482         find_widget<stacked_widget>(&window, "pager", false).select_layer(i);
 483 
 484         /* HACK: the GUI2 field functions always internally save the correct value when set_widget_value is called
 485          *       - ie, when on_game_select calls update_map_settings, the correct values will be stored in the field,
 486          *       but if the settings tab isn't selected the widgets will not display the correct values. This forces
 487          *       an update when we switch to that tab so the widgets correctly display their values.
 488          *
 489          *       A possible better fix would be storing a pointer to the widget in question in the field object. It
 490          *       seems widgets will still correctly update even if they are not on the currently selected page if a
 491          *       pointer already exists.
 492          *
 493          *       Another possible fix would be allowing stacked_widget to return widgets on any page, not just on the
 494          *       currently visible one.
 495          */
 496         if(i == TAB_SETTINGS) {
 497                 update_map_settings(window);
 498         }
 499 }

 543 void mp_create_game::display_games_of_type(window& window, ng::level::TYPE type, const std::string& level)
 544 {
 545         create_engine_.set_current_level_type(type);
 546 
 547         listbox& list = find_widget<listbox>(&window, "games_list", false);
 548 
 549         list.clear();
 550 
 551         for(const auto& game : create_engine_.get_levels_by_type_unfiltered(type)) {
 552                 if(!game->can_launch_game()) {
 553                         continue;
 554                 }
 555 
 556                 std::map<std::string, string_map> data;
 557                 string_map item;
 558 
 559                 if(type == ng::level::TYPE::CAMPAIGN || type == ng::level::TYPE::SP_CAMPAIGN) {
 560                         item["label"] = game->icon();
 561                         data.emplace("game_icon", item);
 562                 }
 563 
 564                 item["label"] = game->name();
 565                 data.emplace("game_name", item);
 566 
 567                 list.add_row(data);
 568         }
 569 
 570         if(!level.empty() && !list.get_rows_shown().empty()) {
 571                 // Recalculate which rows should be visible
 572                 on_filter_change<slider>(window, "num_players");
 573                 on_filter_change<text_box>(window, "game_filter");
 574 
 575                 int level_index = create_engine_.find_level_by_id(level);
 576                 if(level_index >= 0 && size_t(level_index) < list.get_item_count()) {
 577                         list.select_row(level_index);
 578                 }
 579         }
 580 
 581         const bool is_random_map = type == ng::level::TYPE::RANDOM_MAP;
 582 
 583         find_widget<button>(&window, "random_map_regenerate", false).set_active(is_random_map);
 584         find_widget<button>(&window, "random_map_settings", false).set_active(is_random_map);
 585 
 586         // Override the last selection so on_game_select selects the new level
 587         selected_game_index_ = -1;
 588 
 589         on_game_select(window);
 590 }

 612 void mp_create_game::update_details(window& win)
 613 {
 614         styled_widget& players = find_widget<styled_widget>(&win, "map_num_players", false);
 615         styled_widget& map_size = find_widget<styled_widget>(&win, "map_size", false);
 616 
 617         if(create_engine_.current_level_type() == ng::level::TYPE::RANDOM_MAP) {
 618                 // If the current random map doesn't have data, generate it
 619                 if(create_engine_.generator_assigned() && create_engine_.current_level().data()["map_data"].empty()) {
 620                         create_engine_.init_generated_level_data();
 621                 }
 622 
 623                 find_widget<button>(&win, "random_map_settings", false).set_active(create_engine_.generator_has_settings());
 624         }
 625 
 626         create_engine_.current_level().set_metadata();
 627 
 628         // Reset the config_engine with new values
 629         config_engine_.reset(new ng::configure_engine(create_engine_.get_state()));
 630         config_engine_->update_initial_cfg(create_engine_.current_level().data());
 631         config_engine_->set_default_values();
 632 
 633         // Set the title, with newlines replaced. Newlines are sometimes found in SP Campaign names
 634         std::string title = create_engine_.current_level().name();
 635         boost::replace_all(title, "\n", " " + font::unicode_em_dash + " ");
 636         find_widget<styled_widget>(&win, "game_title", false).set_label(title);
 637 
 638         show_description(win, create_engine_.current_level().description());
 639 
 640         switch(create_engine_.current_level_type().v) {
 641                 case ng::level::TYPE::SCENARIO:
 642                 case ng::level::TYPE::USER_MAP:
 643                 case ng::level::TYPE::USER_SCENARIO:
 644                 case ng::level::TYPE::RANDOM_MAP: {
 645                         ng::scenario* current_scenario = dynamic_cast<ng::scenario*>(&create_engine_.current_level());
 646 
 647                         assert(current_scenario);
 648 
 649                         create_engine_.get_state().classification().campaign = "";
 650 
 651                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(0);
 652                         find_widget<minimap>(&win, "minimap", false).set_map_data(current_scenario->data()["map_data"]);
 653 
 654                         players.set_label(std::to_string(current_scenario->num_players()));
 655                         map_size.set_label(current_scenario->map_size());
 656 
 657                         break;
 658                 }
 659                 case ng::level::TYPE::CAMPAIGN:
 660                 case ng::level::TYPE::SP_CAMPAIGN: {
 661                         ng::campaign* current_campaign = dynamic_cast<ng::campaign*>(&create_engine_.current_level());
 662 
 663                         assert(current_campaign);
 664 
 665                         create_engine_.get_state().classification().campaign = current_campaign->data()["id"].str();
 666 
 667                         const std::string img = formatter() << current_campaign->data()["image"] << "~SCALE_INTO(265,265)";
 668 
 669                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(1);
 670                         find_widget<image>(&win, "campaign_image", false).set_image(img);
 671 
 672                         std::stringstream players_str;
 673                         players_str << current_campaign->min_players();
 674 
 675                         if(current_campaign->max_players() != current_campaign->min_players()) {
 676                                 players_str << " to " << current_campaign->max_players();
 677                         }
 678 
 679                         players.set_label(players_str.str());
 680                         map_size.set_label(font::unicode_em_dash);
 681 
 682                         break;
 683                 }
 684         }
 685 }

 728 void mp_create_game::load_game_callback(window& window)
 729 {
 730         savegame::loadgame load(window.video(), cfg_, create_engine_.get_state());
 731 
 732         if(!load.load_multiplayer_game()) {
 733                 return;
 734         }
 735 
 736         if(load.data().cancel_orders) {
 737                 create_engine_.get_state().cancel_orders();
 738         }
 739 
 740         window.set_retval(LOAD_GAME);
 741 }

 753 void mp_create_game::post_show(window& window)
 754 {
 755         plugins_context_.reset();
 756 
 757         // Show all tabs so that find_widget works correctly
 758         find_widget<stacked_widget>(&window, "pager", false).select_layer(-1);
 759 
 760         if(get_retval() == LOAD_GAME) {
 761                 create_engine_.prepare_for_saved_game();
 762                 return;
 763         }
 764 
 765         if(get_retval() == window::OK) {
 766                 prefs::set_modifications(create_engine_.active_mods());
 767                 prefs::set_level_type(create_engine_.current_level_type().v);
 768                 prefs::set_level(create_engine_.current_level().id());
 769                 prefs::set_era(create_engine_.current_extra(ng::create_engine::ERA).id);
 770 
 771                 create_engine_.prepare_for_era_and_mods();
 772 
 773                 if(create_engine_.current_level_type() == ng::level::TYPE::CAMPAIGN ||
 774                         create_engine_.current_level_type() == ng::level::TYPE::SP_CAMPAIGN) {
 775                         create_engine_.prepare_for_campaign();
 776                 } else if(create_engine_.current_level_type() == ng::level::TYPE::SCENARIO) {
 777                         create_engine_.prepare_for_scenario();
 778                 } else {
 779                         // This means define= doesn't work for randomly generated scenarios
 780                         create_engine_.prepare_for_other();
 781                 }
 782 
 783                 create_engine_.get_parameters();
 784 
 785                 create_engine_.prepare_for_new_level();
 786 
 787                 std::vector<const config*> entry_points;
 788                 std::vector<std::string> entry_point_titles;
 789 
 790                 const auto& tagname = create_engine_.get_state().classification().get_tagname();
 791 
 792                 if(tagname == "scenario") {
 793                         for(const config& scenario : game_config_manager::get()->game_config().child_range(tagname)) {
 794                                 if(scenario["allow_new_game"].to_bool(true) || game_config::debug) {
 795                                         const std::string& title = !scenario["new_game_title"].empty()
 796                                                 ? scenario["new_game_title"]
 797                                                 : scenario["name"];
 798 
 799                                         entry_points.push_back(&scenario);
 800                                         entry_point_titles.push_back(title);
 801                                 }
 802                         }
 803                 }
 804 
 805                 if(entry_points.size() > 1) {
 806                         gui2::dialogs::simple_item_selector dlg(_("Choose Starting Scenario"), _("Select at which point to begin this campaign."), entry_point_titles);
 807 
 808                         dlg.set_single_button(true);
 809                         dlg.show(window.video());
 810 
 811                         const config& scenario = *entry_points[dlg.selected_index()];
 812 
 813                         create_engine_.get_state().mp_settings().hash = scenario.hash();
 814                         create_engine_.get_state().set_scenario(scenario);
 815                 }
 816 
 817                 config_engine_->set_use_map_settings(use_map_settings_->get_widget_value(window));
 818 
 819                 if(!config_engine_->force_lock_settings()) {
 820                         // Max slider value (in this case, 100) means 'unlimited turns', so pass the value -1
 821                         const int num_turns = turns_->get_widget_value(window);
 822                         config_engine_->set_num_turns(num_turns < ::settings::turns_max ? num_turns : - 1);
 823                         config_engine_->set_village_gold(gold_->get_widget_value(window));
 824                         config_engine_->set_village_support(support_->get_widget_value(window));
 825                         config_engine_->set_xp_modifier(experience_->get_widget_value(window));
 826                         config_engine_->set_random_start_time(start_time_->get_widget_value(window));
 827                         config_engine_->set_fog_game(fog_->get_widget_value(window));
 828                         config_engine_->set_shroud_game(shroud_->get_widget_value(window));
 829 
 830                         config_engine_->write_parameters();
 831                 }
 832 
 833                 config_engine_->set_mp_countdown(time_limit_->get_widget_value(window));
 834                 config_engine_->set_mp_countdown_init_time(init_turn_limit_->get_widget_value(window));
 835                 config_engine_->set_mp_countdown_turn_bonus(turn_bonus_->get_widget_value(window));
 836                 config_engine_->set_mp_countdown_reservoir_time(reservoir_->get_widget_value(window));
 837                 config_engine_->set_mp_countdown_action_bonus(action_bonus_->get_widget_value(window));
 838 
 839                 config_engine_->set_allow_observers(observers_->get_widget_value(window));
 840                 config_engine_->set_registered_users_only(registered_users_->get_widget_value(window));
 841                 config_engine_->set_oos_debug(strict_sync_->get_widget_value(window));
 842                 config_engine_->set_shuffle_sides(shuffle_sides_->get_widget_value(window));
 843 
 844                 config_engine_->set_random_faction_mode(rfm_types_[selected_rfm_index_]);
 845 
 846                 // Since we don't have a field handling this option, we need to save the value manually
 847                 prefs::set_random_faction_mode(mp_game_settings::RANDOM_FACTION_MODE::enum_to_string(rfm_types_[selected_rfm_index_]));
 848 
 849                 // Save custom option settings
 850                 const config options_config = options_manager_->get_options_config();
 851 
 852                 config_engine_->set_options(options_config);
 853                 prefs::set_options(options_config);
 854 
 855                 // Set game name
 856                 const std::string name = find_widget<text_box>(&window, "game_name", false).get_value();
 857                 if(!name.empty() && (name != ng::configure_engine::game_name_default())) {
 858                         config_engine_->set_game_name(name);
 859                 }
 860 
 861                 // Set game password
 862                 const std::string password = find_widget<text_box>(&window, "game_password", false).get_value();
 863                 if(!password.empty()) {
 864                         config_engine_->set_game_password(password);
 865                 }
 866         }
