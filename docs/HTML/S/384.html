 426 template<typename widget>
 427 void mp_create_game::on_filter_change(window& window, const std::string& id)
 428 {
 429         create_engine_.apply_level_filter(find_widget<widget>(&window, id, false).get_value());
 430 
 431         listbox& game_list = find_widget<listbox>(&window, "games_list", false);
 432 
 433         boost::dynamic_bitset<> filtered(game_list.get_item_count());
 434         for(const size_t i : create_engine_.get_filtered_level_indices(create_engine_.current_level_type())) {
 435                 filtered[i] = true;
 436         }
 437 
 438         game_list.set_row_shown(filtered);
 439 
 440         on_game_select(window);
 441 }

 443 void mp_create_game::on_game_select(window& window)
 444 {
 445         const int selected_game = find_widget<listbox>(&window, "games_list", false).get_selected_row();
 446 
 447         if(selected_game == selected_game_index_) {
 448                 return;
 449         }
 450 
 451         // Convert the absolute-index get_selected_row to a relatve index for the create_engine to handle
 452         selected_game_index_ = convert_to_game_filtered_index(selected_game);
 453 
 454         create_engine_.set_current_level(selected_game_index_);
 455         update_details(window);
 456 
 457         // General settings
 458         stacked_widget& stack = find_widget<stacked_widget>(&window, "pager", false);
 459 
 460         const bool can_select_era = create_engine_.current_level().allow_era_choice();
 461 
 462         menu_button& era_combo = find_widget<menu_button>(stack.get_layer_grid(TAB_GENERAL), "eras", false);
 463         if(!can_select_era) {
 464                 era_combo.set_label(_("No eras available for this game."));
 465         } else {
 466                 era_combo.set_selected(era_combo.get_value());
 467         }
 468 
 469         era_combo.set_active(can_select_era);
 470 
 471         // Custom options
 472         options_manager_->update_game_options();
 473 
 474         // Game settings
 475         update_map_settings(window);
 476 }

 478 void mp_create_game::on_tab_select(window& window)
 479 {
 480         const int i = find_widget<listbox>(&window, "tab_bar", false).get_selected_row();
 481         find_widget<stacked_widget>(&window, "pager", false).select_layer(i);
 482 
 483         /* HACK: the GUI2 field functions always internally save the correct value when set_widget_value is called
 484          *       - ie, when on_game_select calls update_map_settings, the correct values will be stored in the field,
 485          *       but if the settings tab isn't selected the widgets will not display the correct values. This forces
 486          *       an update when we switch to that tab so the widgets correctly display their values.
 487          *
 488          *       A possible better fix would be storing a pointer to the widget in question in the field object. It
 489          *       seems widgets will still correctly update even if they are not on the currently selected page if a
 490          *       pointer already exists.
 491          *
 492          *       Another possible fix would be allowing stacked_widget to return widgets on any page, not just on the
 493          *       currently visible one.
 494          */
 495         if(i == TAB_SETTINGS) {
 496                 update_map_settings(window);
 497         }
 498 }

 542 void mp_create_game::display_games_of_type(window& window, ng::level::TYPE type, const std::string& level)
 543 {
 544         create_engine_.set_current_level_type(type);
 545 
 546         listbox& list = find_widget<listbox>(&window, "games_list", false);
 547 
 548         list.clear();
 549 
 550         for(const auto& game : create_engine_.get_levels_by_type_unfiltered(type)) {
 551                 if(!game->can_launch_game()) {
 552                         continue;
 553                 }
 554 
 555                 std::map<std::string, string_map> data;
 556                 string_map item;
 557 
 558                 if(type == ng::level::TYPE::CAMPAIGN || type == ng::level::TYPE::SP_CAMPAIGN) {
 559                         item["label"] = game->icon();
 560                         data.emplace("game_icon", item);
 561                 }
 562 
 563                 item["label"] = game->name();
 564                 data.emplace("game_name", item);
 565 
 566                 list.add_row(data);
 567         }
 568 
 569         if(!level.empty() && !list.get_rows_shown().empty()) {
 570                 // Recalculate which rows should be visible
 571                 on_filter_change<slider>(window, "num_players");
 572                 on_filter_change<text_box>(window, "game_filter");
 573 
 574                 int level_index = create_engine_.find_level_by_id(level);
 575                 if(level_index >= 0 && size_t(level_index) < list.get_item_count()) {
 576                         list.select_row(level_index);
 577                 }
 578         }
 579 
 580         const bool is_random_map = type == ng::level::TYPE::RANDOM_MAP;
 581 
 582         find_widget<button>(&window, "random_map_regenerate", false).set_active(is_random_map);
 583         find_widget<button>(&window, "random_map_settings", false).set_active(is_random_map);
 584 
 585         // Override the last selection so on_game_select selects the new level
 586         selected_game_index_ = -1;
 587 
 588         on_game_select(window);
 589 }

 611 void mp_create_game::update_details(window& win)
 612 {
 613         styled_widget& players = find_widget<styled_widget>(&win, "map_num_players", false);
 614         styled_widget& map_size = find_widget<styled_widget>(&win, "map_size", false);
 615 
 616         if(create_engine_.current_level_type() == ng::level::TYPE::RANDOM_MAP) {
 617                 // If the current random map doesn't have data, generate it
 618                 if(create_engine_.generator_assigned() && create_engine_.current_level().data()["map_data"].empty()) {
 619                         create_engine_.init_generated_level_data();
 620                 }
 621 
 622                 find_widget<button>(&win, "random_map_settings", false).set_active(create_engine_.generator_has_settings());
 623         }
 624 
 625         create_engine_.current_level().set_metadata();
 626 
 627         // Reset the config_engine with new values
 628         config_engine_.reset(new ng::configure_engine(create_engine_.get_state()));
 629         config_engine_->update_initial_cfg(create_engine_.current_level().data());
 630         config_engine_->set_default_values();
 631 
 632         // Set the title, with newlines replaced. Newlines are sometimes found in SP Campaign names
 633         std::string title = create_engine_.current_level().name();
 634         boost::replace_all(title, "\n", " " + font::unicode_em_dash + " ");
 635         find_widget<styled_widget>(&win, "game_title", false).set_label(title);
 636 
 637         show_description(win, create_engine_.current_level().description());
 638 
 639         switch(create_engine_.current_level_type().v) {
 640                 case ng::level::TYPE::SCENARIO:
 641                 case ng::level::TYPE::USER_MAP:
 642                 case ng::level::TYPE::USER_SCENARIO:
 643                 case ng::level::TYPE::RANDOM_MAP: {
 644                         ng::scenario* current_scenario = dynamic_cast<ng::scenario*>(&create_engine_.current_level());
 645 
 646                         assert(current_scenario);
 647 
 648                         create_engine_.get_state().classification().campaign = "";
 649 
 650                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(0);
 651                         find_widget<minimap>(&win, "minimap", false).set_map_data(current_scenario->data()["map_data"]);
 652 
 653                         players.set_label(std::to_string(current_scenario->num_players()));
 654                         map_size.set_label(current_scenario->map_size());
 655 
 656                         break;
 657                 }
 658                 case ng::level::TYPE::CAMPAIGN:
 659                 case ng::level::TYPE::SP_CAMPAIGN: {
 660                         ng::campaign* current_campaign = dynamic_cast<ng::campaign*>(&create_engine_.current_level());
 661 
 662                         assert(current_campaign);
 663 
 664                         create_engine_.get_state().classification().campaign = current_campaign->data()["id"].str();
 665 
 666                         const std::string img = formatter() << current_campaign->data()["image"] << "~SCALE_INTO(265,265)";
 667 
 668                         find_widget<stacked_widget>(&win, "minimap_stack", false).select_layer(1);
 669                         find_widget<image>(&win, "campaign_image", false).set_image(img);
 670 
 671                         std::stringstream players_str;
 672                         players_str << current_campaign->min_players();
 673 
 674                         if(current_campaign->max_players() != current_campaign->min_players()) {
 675                                 players_str << " to " << current_campaign->max_players();
 676                         }
 677 
 678                         players.set_label(players_str.str());
 679                         map_size.set_label(font::unicode_em_dash);
 680 
 681                         break;
 682                 }
 683         }
 684 }

 727 void mp_create_game::load_game_callback(window& window)
 728 {
 729         savegame::loadgame load(window.video(), cfg_, create_engine_.get_state());
 730 
 731         if(!load.load_multiplayer_game()) {
 732                 return;
 733         }
 734 
 735         if(load.data().cancel_orders) {
 736                 create_engine_.get_state().cancel_orders();
 737         }
 738 
 739         window.set_retval(LOAD_GAME);
 740 }

 752 void mp_create_game::post_show(window& window)
 753 {
 754         plugins_context_.reset();
 755 
 756         // Show all tabs so that find_widget works correctly
 757         find_widget<stacked_widget>(&window, "pager", false).select_layer(-1);
 758 
 759         if(get_retval() == LOAD_GAME) {
 760                 create_engine_.prepare_for_saved_game();
 761                 return;
 762         }
 763 
 764         if(get_retval() == window::OK) {
 765                 prefs::set_modifications(create_engine_.active_mods());
 766                 prefs::set_level_type(create_engine_.current_level_type().v);
 767                 prefs::set_level(create_engine_.current_level().id());
 768                 prefs::set_era(create_engine_.current_extra(ng::create_engine::ERA).id);
 769 
 770                 create_engine_.prepare_for_era_and_mods();
 771 
 772                 if(create_engine_.current_level_type() == ng::level::TYPE::CAMPAIGN ||
 773                         create_engine_.current_level_type() == ng::level::TYPE::SP_CAMPAIGN) {
 774                         create_engine_.prepare_for_campaign();
 775                 } else if(create_engine_.current_level_type() == ng::level::TYPE::SCENARIO) {
 776                         create_engine_.prepare_for_scenario();
 777                 } else {
 778                         // This means define= doesn't work for randomly generated scenarios
 779                         create_engine_.prepare_for_other();
 780                 }
 781 
 782                 create_engine_.get_parameters();
 783 
 784                 create_engine_.prepare_for_new_level();
 785 
 786                 std::vector<const config*> entry_points;
 787                 std::vector<std::string> entry_point_titles;
 788 
 789                 const auto& tagname = create_engine_.get_state().classification().get_tagname();
 790 
 791                 if(tagname == "scenario") {
 792                         for(const config& scenario : game_config_manager::get()->game_config().child_range(tagname)) {
 793                                 if(scenario["allow_new_game"].to_bool(true) || game_config::debug) {
 794                                         const std::string& title = !scenario["new_game_title"].empty()
 795                                                 ? scenario["new_game_title"]
 796                                                 : scenario["name"];
 797 
 798                                         entry_points.push_back(&scenario);
 799                                         entry_point_titles.push_back(title);
 800                                 }
 801                         }
 802                 }
 803 
 804                 if(entry_points.size() > 1) {
 805                         gui2::dialogs::simple_item_selector dlg(_("Choose Starting Scenario"), _("Select at which point to begin this campaign."), entry_point_titles);
 806 
 807                         dlg.set_single_button(true);
 808                         dlg.show(window.video());
 809 
 810                         const config& scenario = *entry_points[dlg.selected_index()];
 811 
 812                         create_engine_.get_state().mp_settings().hash = scenario.hash();
 813                         create_engine_.get_state().set_scenario(scenario);
 814                 }
 815 
 816                 config_engine_->set_use_map_settings(use_map_settings_->get_widget_value(window));
 817 
 818                 if(!config_engine_->force_lock_settings()) {
 819                         // Max slider value (in this case, 100) means 'unlimited turns', so pass the value -1
 820                         const int num_turns = turns_->get_widget_value(window);
 821                         config_engine_->set_num_turns(num_turns < ::settings::turns_max ? num_turns : - 1);
 822                         config_engine_->set_village_gold(gold_->get_widget_value(window));
 823                         config_engine_->set_village_support(support_->get_widget_value(window));
 824                         config_engine_->set_xp_modifier(experience_->get_widget_value(window));
 825                         config_engine_->set_random_start_time(start_time_->get_widget_value(window));
 826                         config_engine_->set_fog_game(fog_->get_widget_value(window));
 827                         config_engine_->set_shroud_game(shroud_->get_widget_value(window));
 828 
 829                         config_engine_->write_parameters();
 830                 }
 831 
 832                 config_engine_->set_mp_countdown(time_limit_->get_widget_value(window));
 833                 config_engine_->set_mp_countdown_init_time(init_turn_limit_->get_widget_value(window));
 834                 config_engine_->set_mp_countdown_turn_bonus(turn_bonus_->get_widget_value(window));
 835                 config_engine_->set_mp_countdown_reservoir_time(reservoir_->get_widget_value(window));
 836                 config_engine_->set_mp_countdown_action_bonus(action_bonus_->get_widget_value(window));
 837 
 838                 config_engine_->set_allow_observers(observers_->get_widget_value(window));
 839                 config_engine_->set_registered_users_only(registered_users_->get_widget_value(window));
 840                 config_engine_->set_oos_debug(strict_sync_->get_widget_value(window));
 841                 config_engine_->set_shuffle_sides(shuffle_sides_->get_widget_value(window));
 842 
 843                 config_engine_->set_random_faction_mode(rfm_types_[selected_rfm_index_]);
 844 
 845                 // Since we don't have a field handling this option, we need to save the value manually
 846                 prefs::set_random_faction_mode(mp_game_settings::RANDOM_FACTION_MODE::enum_to_string(rfm_types_[selected_rfm_index_]));
 847 
 848                 // Save custom option settings
 849                 const config options_config = options_manager_->get_options_config();
 850 
 851                 config_engine_->set_options(options_config);
 852                 prefs::set_options(options_config);
 853 
 854                 // Set game name
 855                 const std::string name = find_widget<text_box>(&window, "game_name", false).get_value();
 856                 if(!name.empty() && (name != ng::configure_engine::game_name_default())) {
 857                         config_engine_->set_game_name(name);
 858                 }
 859 
 860                 // Set game password
 861                 const std::string password = find_widget<text_box>(&window, "game_password", false).get_value();
 862                 if(!password.empty()) {
 863                         config_engine_->set_game_password(password);
 864                 }
 865         }
