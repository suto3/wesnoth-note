  91 void replay_controller::add_replay_theme()
  92 {
  93         const config &theme_cfg = controller_.get_theme(game_config_manager::get()->game_config(), controller_.theme());
  94         if (const config &res = theme_cfg.child("resolution"))
  95         {
  96                 if (const config &replay_theme_cfg = res.child("replay")) {
  97                         controller_.get_display().get_theme().modify(replay_theme_cfg);
  98                 }
  99                 //Make sure we get notified if the theme is redrawn completely. That way we have
 100                 //a chance to restore the replay controls of the theme as well.
 101                 controller_.get_display().invalidate_theme();
 102         }

 151 void replay_controller::update_replay_ui()
 152 {
 153         //check if we have all buttons - if someone messed with theme then some buttons may be missing

 198 void replay_controller::replay_ui_playback_should_stop()
 199 {
 200         if(!replay_ui_has_all_buttons())
 201                 return;
 202 
 203         if(!resources::recorder->at_end()) {
 204                 play_button()->enable(true);
 205                 reset_button()->enable(true);
 206                 play_turn_button()->enable(true);
 207                 play_side_button()->enable(true);
 208                 play_move_button()->enable(true);
 209 
 210                 play_button()->release();
 211                 play_turn_button()->release();
 212                 play_side_button()->release();
 213                 play_move_button()->release();
 214         } else {
 215                 reset_button()->enable(true);
 216                 stop_button()->enable(false);
 217         }
 218 
 219         if(stop_condition_->should_stop()) {
 220                 //user interrupted
 221                 stop_button()->release();
 222         }
 223 }

 291 REPLAY_RETURN replay_controller::play_side_impl()
 292 {
 293         while(!return_to_play_side_ && !static_cast<playsingle_controller&>(controller_).get_player_type_changed())
 294         {
 295                 if(!stop_condition_->should_stop())
 296                 {
 297                         if(resources::recorder->at_end()) {
 298                                 //Gather more replay data
 299                                 on_end_replay_();
 300                         }
 301                         else {
 302                                 REPLAY_RETURN res = do_replay(true);
 303                                 if(res == REPLAY_FOUND_END_MOVE) {
 304                                         stop_condition_->move_done();
 305                                 }
 306                                 if(res == REPLAY_FOUND_END_TURN) {
 307                                         return res;
 308                                 }
 309                                 if(res == REPLAY_RETURN_AT_END) {
 310                                         stop_replay();
 311                                 }
 312                                 if(res == REPLAY_FOUND_INIT_TURN)
 313                                 {
 314                                         stop_condition_->new_side_turn(controller_.current_side(), controller_.gamestate().tod_manager_.turn());
 315                                 }
 316                         }
 317                         controller_.play_slice(false);
 318                 }
 319                 else
 320                 {
 321                         controller_.play_slice(true);
 322                         replay_ui_playback_should_stop();
 323                 }
 324         }
 325         return REPLAY_FOUND_END_MOVE;
 326 }
 327 bool replay_controller::can_execute_command(const hotkey::hotkey_command& cmd, int) const
 328 {
 329         hotkey::HOTKEY_COMMAND command = cmd.id;
 330 
 331         switch(command) {
 332         case hotkey::HOTKEY_REPLAY_SKIP_ANIMATION:
 333                 return true;
 334         case hotkey::HOTKEY_REPLAY_SHOW_EVERYTHING:
 335         case hotkey::HOTKEY_REPLAY_SHOW_EACH:
 336         case hotkey::HOTKEY_REPLAY_SHOW_TEAM1:
 337                 return is_controlling_view();
 338         //commands we only can do before the end of the replay
 339         case hotkey::HOTKEY_REPLAY_STOP:
 340                 return !recorder_at_end();
 341         case hotkey::HOTKEY_REPLAY_PLAY:
 342         case hotkey::HOTKEY_REPLAY_NEXT_TURN:
 343         case hotkey::HOTKEY_REPLAY_NEXT_SIDE:
 344         case hotkey::HOTKEY_REPLAY_NEXT_MOVE:
 345                 //we have one events_disabler when starting the replay_controller and a second when entering the synced context.
 346                 return should_stop() && (events::commands_disabled <= 1 ) && !recorder_at_end();
 347         case hotkey::HOTKEY_REPLAY_RESET:
 348                 return allow_reset_replay() && events::commands_disabled <= 1;
 349         default:
 350                 assert(false);
 351                 return false;
 352         }
 353 }
