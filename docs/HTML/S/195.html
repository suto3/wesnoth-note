  41 /**
  42  * Returns a random numer, same interface as math.random.
  43  */
  44 static int intf_random(lua_State *L)
  45 {
  46         std::mt19937& rng = lua_kernel_base::get_lua_kernel<mapgen_lua_kernel>(L).get_default_rng();
  47         if(lua_isnoneornil(L, 1)) {
  48                 double r = double (rng());
  49                 double r_max = double (rng.max());
  50                 lua_push(L, r / (r_max + 1));
  51                 return 1;
  52         }
  53         else {
  54                 int32_t min;
  55                 int32_t max;
  56                 if(lua_isnumber(L, 2)) {
  57                         min = lua_check<int32_t>(L, 1);
  58                         max = lua_check<int32_t>(L, 2);
  59                 }
  60                 else {
  61                         min = 1;
  62                         max = lua_check<int32_t>(L, 1);
  63                 }
  64                 if(min > max) {
  65                         return luaL_argerror(L, 1, "min > max");
  66                 }
  67                 lua_push(L, min + static_cast<int>(rng() % (max - min + 1)));
  68                 return 1;
  69         }
  70 }

  72 /**
  73  * Finds a path between two locations.
  74  * - Args 1,2: source location.
  75  * - Args 3,4: destination.
  76  * - Arg 5: cost function
  77  * - Args 6,7 size of map.
  78  * - Arg 8 include border.
  79  * - Ret 1: array of pairs containing path steps.
  80  * - Ret 2: path cost.
  81  */
  82 static int intf_find_path(lua_State *L)
  83 {
  84         int arg = 1;
  85         map_location src, dst;
  86         src.set_wml_x(luaL_checkinteger(L, 1));
  87         src.set_wml_y(luaL_checkinteger(L, 2));
  88         dst.set_wml_x(luaL_checkinteger(L, 3));
  89         dst.set_wml_y(luaL_checkinteger(L, 4));
  90         if(lua_isfunction(L, arg)) {
  91                 const char *msg = lua_pushfstring(L, "%s expected, got %s", lua_typename(L, LUA_TFUNCTION), luaL_typename(L, 5));
  92                 return luaL_argerror(L, 5, msg);
  93         }
  94         lua_pathfind_cost_calculator calc(L, 5);
  95         int width = luaL_checkinteger(L, 6);
  96         int height = luaL_checkinteger(L, 7);
  97         bool border = false;
  98         if(lua_isboolean(L, 8)) {
  99                 border = luaW_toboolean(L, 8);
 100         }
 101         pathfind::plain_route res = pathfind::a_star_search(src, dst, 10000, calc, width, height, nullptr, border);
 102 
 103         int nb = res.steps.size();
 104         lua_createtable(L, nb, 0);
 105         for (int i = 0; i < nb; ++i)
 106         {
 107                 lua_createtable(L, 2, 0);
 108                 lua_pushinteger(L, res.steps[i].wml_x());
 109                 lua_rawseti(L, -2, 1);
 110                 lua_pushinteger(L, res.steps[i].wml_y());
 111                 lua_rawseti(L, -2, 2);
 112                 lua_rawseti(L, -2, i + 1);
 113         }
 114         lua_pushinteger(L, res.move_cost);
 115 
 116         return 2;
 117 }

 120 mapgen_lua_kernel::mapgen_lua_kernel()
 121         : lua_kernel_base()
 122         , random_seed_()
 123         , default_rng_()
 124 {
 125         lua_State *L = mState;
 126 
 127         // Unset wesnoth.random. This guarantees that the mapgen_lua_kernel version
 128         // of wesnoth.random overrides the lua_kernel_base version.
 129         lua_getglobal(L, "wesnoth");
 130         lua_pushnil(L);
 131         lua_setfield(L, -2, "random");
 132 
 133         lua_settop(L, 0);
 134 
 135         static luaL_Reg const callbacks[] {
 136                 { "find_path",           &intf_find_path           },
 137                 { "random",              &intf_random              },
 138                 { nullptr, nullptr }
 139         };
 140 
 141         lua_getglobal(L, "wesnoth");
 142         assert(lua_istable(L,-1));
 143         luaL_setfuncs(L, callbacks, 0);
 144         lua_pop(L, 1);
 145         assert(lua_gettop(L) == 0);
 146 }

 148 void mapgen_lua_kernel::run_generator(const char * prog, const config & generator)
 149 {
 150         load_string(prog, std::bind(&lua_kernel_base::throw_exception, this, _1, _2));
 151         luaW_pushconfig(mState, generator);
 152         protected_call(1, 1, std::bind(&lua_kernel_base::throw_exception, this, _1, _2));
 153 }

 160 std::string mapgen_lua_kernel::create_map(const char * prog, const config & generator, boost::optional<uint32_t> seed) // throws game::lua_error
 161 {
 162         random_seed_ = seed;
 163         run_generator(prog, generator);
 164 
 165         if (!lua_isstring(mState,-1)) {
 166                 std::string msg = "expected a string, found a ";
 167                 msg += lua_typename(mState, lua_type(mState, -1));
 168                 lua_pop(mState, 1);
 169                 throw game::lua_error(msg.c_str(),"bad return value");
 170         }
 171 
 172         return lua_tostring(mState, -1);
 173 }

 175 config mapgen_lua_kernel::create_scenario(const char * prog, const config & generator, boost::optional<uint32_t> seed) // throws game::lua_error
 176 {
 177         random_seed_ = seed;
 178         run_generator(prog, generator);
 179 
 180         if (!lua_istable(mState, -1)) {
 181                 std::string msg = "expected a config (table), found a ";
 182                 msg += lua_typename(mState, lua_type(mState, -1));
 183                 lua_pop(mState, 1);
 184                 throw game::lua_error(msg.c_str(),"bad return value");
 185         }
 186         config result;
 187         if (!luaW_toconfig(mState, -1, result)) {
 188                 std::string msg = "expected a config, but it is malformed ";
 189                 lua_pop(mState, 1);
 190                 throw game::lua_error(msg.c_str(),"bad return value");
 191         }
 192         return result;
 193 }
 194 uint32_t mapgen_lua_kernel::get_random_seed()
 195 {
 196         if(uint32_t* pint = random_seed_.get_ptr()) {
 197                 return (*pint)++;
 198         }
 199         else {
 200                 return lua_kernel_base::get_random_seed();
 201         }
 202 }

 204 std::mt19937& mapgen_lua_kernel::get_default_rng()
 205 {

 207                 default_rng_ = std::mt19937(get_random_seed());
 208         }
