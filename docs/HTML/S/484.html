  92 void unit_create::pre_show(window& window)
  93 {
  94         toggle_button& male_toggle
  95                         = find_widget<toggle_button>(&window, "male_toggle", false);
  96         toggle_button& female_toggle
  97                         = find_widget<toggle_button>(&window, "female_toggle", false);
  98 
  99         gender_toggle.add_member(&male_toggle, unit_race::MALE);
 100         gender_toggle.add_member(&female_toggle, unit_race::FEMALE);
 101 
 102         gender_toggle.set_member_states(last_gender);
 103 
 104         gender_toggle.set_callback_on_value_change(
 105                         dialog_callback<unit_create, &unit_create::gender_toggle_callback>);
 106 
 107         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 108 
 109         text_box* filter
 110                         = find_widget<text_box>(&window, "filter_box", false, true);
 111 
 112         filter->set_text_changed_callback(
 113                         std::bind(&unit_create::filter_text_changed, this, _1, _2));
 114 
 115         window.keyboard_capture(filter);
 116         window.add_to_keyboard_chain(&list);
 117 
 118         connect_signal_notify_modified(list, std::bind(&unit_create::list_item_clicked, this, std::ref(window)));
 119 
 120         list.clear();
 121 
 122         for(const auto & i : unit_types.types())
 123         {
 124                 if(i.second.do_not_list())
 125                         continue;
 126 
 127                 // Make sure this unit type is built with the data we need.
 128                 unit_types.build_unit_type(i.second, unit_type::FULL);
 129 
 130                 units_.push_back(&i.second);
 131 
 132                 std::map<std::string, string_map> row_data;
 133                 string_map column;
 134 
 135                 column["label"] = units_.back()->race()->plural_name();
 136                 row_data.emplace("race", column);
 137 
 138                 column["label"] = units_.back()->type_name();
 139                 row_data.emplace("unit_type", column);
 140 
 141                 list.add_row(row_data);
 142 
 143                 // Select the previous choice, if any.
 144                 if(choice_.empty() != true && choice_ == i.first) {
 145                         list.select_row(list.get_item_count() - 1);
 146                 }
 147         }
 148 
 149         if(units_.empty()) {
 150                 ERR_GUI_G << "no unit types found for unit create dialog; not good"
 151                                   << std::endl;
 152         }
 153 
 154         list.register_sorting_option(0, [this](const int i) { return (*units_[i]).race()->plural_name().str(); });
 155         list.register_sorting_option(1, [this](const int i) { return (*units_[i]).type_name().str(); });
 156 
 157         list_item_clicked(window);
 158 }

 160 void unit_create::post_show(window& window)
 161 {
 162         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 163 
 164         choice_ = "";
 165 
 166         if(get_retval() != window::OK) {
 167                 return;
 168         }
 169 
 170         const int selected_row = list.get_selected_row();
 171         if(selected_row < 0) {
 172                 return;
 173         } else if(static_cast<size_t>(selected_row) >= units_.size()) {
 174                 // FIXME: maybe assert?
 175                 ERR_GUI_G << "unit create dialog has more list items than known unit "
 176                                          "types; not good\n";
 177                 return;
 178         }
 179 
 180         last_chosen_type_id = choice_ = units_[selected_row]->id();
 181         last_gender = gender_;
 182 }

 195 void unit_create::list_item_clicked(window& window)
 196 {
 197         const int selected_row
 198                 = find_widget<listbox>(&window, "unit_type_list", false).get_selected_row();
 199 
 200         if(selected_row == -1) {
 201                 return;
 202         }
 203 
 204         update_displayed_type();
 205 
 206         gender_toggle.set_members_enabled([&](const unit_race::GENDER& gender)->bool {
 207                 return units_[selected_row]->has_gender_variation(gender);
 208         });
 209 }

 211 void unit_create::filter_text_changed(text_box_base* textbox, const std::string& text)
 212 {
 213         window& window = *textbox->get_window();
 214 
 215         listbox& list = find_widget<listbox>(&window, "unit_type_list", false);
 216 
 217         const std::vector<std::string> words = utils::split(text, ' ');
 218 
 219         if(words == last_words_)
 220                 return;
 221         last_words_ = words;
 222 
 223         boost::dynamic_bitset<> show_items;
 224         show_items.resize(list.get_item_count(), true);
 225 
 226         if(!text.empty()) {
 227                 for(unsigned int i = 0; i < list.get_item_count(); i++) {
 228                         grid* row = list.get_row_grid(i);
 229 
 230                         grid::iterator it = row->begin();
 231                         label& type_label
 232                                         = find_widget<label>(*it, "unit_type", false);
 233 
 234                         bool found = false;
 235                         for(const auto & word : words)
 236                         {
 237                                 found = std::search(type_label.get_label().str().begin(),
 238                                                                         type_label.get_label().str().end(),
 239                                                                         word.begin(),
 240                                                                         word.end(),
 241                                                                         chars_equal_insensitive)
 242                                                 != type_label.get_label().str().end();
 243 
 244                                 if(!found) {
 245                                         // one word doesn't match, we don't reach words.end()
 246                                         break;
 247                                 }
 248                         }
 249 
 250                         show_items[i] = found;
 251                 }
 252         }
 253 
 254         list.set_row_shown(show_items);
 255 }
