  89 void text_box_base::set_maximum_length(const size_t maximum_length)
  90 {
  91         if(maximum_length == 0) {
  92                 return;
  93         }
  94 
  95         const bool need_update = text_.get_length() > maximum_length;
  96 
  97         text_.set_maximum_length(maximum_length);
  98 
  99         if(need_update) {
 100                 if(selection_start_ > maximum_length) {
 101                         selection_start_ = maximum_length;
 102                         selection_length_ = 0;
 103                 } else if(selection_start_ + selection_length_ > maximum_length) {
 104                         selection_length_ = maximum_length - selection_start_;
 105                 }
 106                 update_canvas();
 107                 set_is_dirty(true);
 108         }
 109 }

 124 void text_box_base::set_cursor(const size_t offset, const bool select)
 125 {
 126         reset_cursor_state();
 127 
 128         if(select) {
 129 
 130                 if(selection_start_ == offset) {
 131                         selection_length_ = 0;
 132                 } else {
 133                         selection_length_ = -static_cast<int>(selection_start_ - offset);
 134                 }
 135 
 136 #ifdef __unix__
 137                 // selecting copies on UNIX systems.
 138                 copy_selection(true);
 139 #endif
 140                 update_canvas();
 141                 set_is_dirty(true);
 142 
 143         } else {
 144                 assert(offset <= text_.get_length());
 145                 selection_start_ = offset;
 146                 selection_length_ = 0;
 147 
 148                 update_canvas();
 149                 set_is_dirty(true);
 150         }
 151 }

 175 void text_box_base::copy_selection(const bool mouse)
 176 {
 177         if(selection_length_ == 0) {
 178                 return;
 179         }
 180 
 181         unsigned end, start = selection_start_;
 182         const utf8::string txt = text_.text();
 183 
 184         if(selection_length_ > 0) {
 185                 end = utf8::index(txt, start + selection_length_);
 186                 start = utf8::index(txt, start);
 187         } else {
 188                 // inverse selection: selection_start_ is in fact the end
 189                 end = utf8::index(txt, start);
 190                 start = utf8::index(txt, start + selection_length_);
 191         }
 192         desktop::clipboard::copy_to_clipboard(txt.substr(start, end - start), mouse);
 193 }

 227 void text_box_base::set_selection(size_t start, int length)
 228 {
 229         const size_t text_size = text_.get_length();
 230 
 231         if(start >= text_size) {
 232                 start = text_size;
 233         }
 234 
 235         if(length == 0) {
 236                 set_cursor(start, false);
 237                 return;
 238         }
 239 
 240         // The text pos/size type differs in both signedness and size with the
 241         // selection length. Such is life.
 242         const int sel_start = std::min<size_t>(start, std::numeric_limits<int>::max());
 243         const int sel_max_length = std::min<size_t>(text_size - start, std::numeric_limits<int>::max());
 244 
 245         const bool backwards = length < 0;
 246 
 247         if(backwards && -length > sel_start) {
 248                 length = -sel_start;
 249         } else if(!backwards && length > sel_max_length) {
 250                 length = sel_max_length;
 251         }
 252 
 253         set_selection_start(start);
 254         set_selection_length(length);
 255 
 256         update_canvas();
 257 }

 267 void text_box_base::toggle_cursor_timer(bool enable)
 268 {
 269         if(!cursor_blink_rate_ms_) {
 270                 return;
 271         }
 272 
 273         if(cursor_timer_) {
 274                 remove_timer(cursor_timer_);
 275         }
 276 
 277         cursor_timer_ = enable
 278                         ? add_timer(cursor_blink_rate_ms_, std::bind(&text_box_base::cursor_timer_callback, this), true)
 279                         : 0;

 282 void text_box_base::cursor_timer_callback()
 283 {
 284         unsigned was_alpha = cursor_alpha_;
 285         switch(state_) {
 286                 case DISABLED:
 287                         cursor_alpha_ = 0;
 288                         return;
 289                 case ENABLED:
 290                         cursor_alpha_ = 255;
 291                         return;
 292                 default:
 293                         if(get_window() != open_window_stack.back()) {
 294                                 cursor_alpha_ = 0;
 295                         } else {
 296                                 cursor_alpha_ = (~cursor_alpha_) & 0xFF;
 297                         }
 298         }
 299 
 300         if(was_alpha == cursor_alpha_) {
 301                 return;
 302         }
 303 
 304         for(auto& tmp : get_canvases()) {
 305                 tmp.set_variable("cursor_alpha", wfl::variant(cursor_alpha_));
 306         }
 307 
 308         set_is_dirty(true);
 309 }

 401 void text_box_base::handle_commit(bool& handled, const utf8::string& unicode)
 402 {
 403         DBG_GUI_E << LOG_SCOPE_HEADER << '\n';
 404 
 405         if(unicode.size() > 1 || unicode[0] != 0) {
 406                 handled = true;
 407                 if(ime_in_progress_) {
 408                         selection_start_ = ime_start_point_;
 409                         selection_length_ = ime_length_;
 410                         ime_in_progress_ = false;
 411                         ime_length_ = 0;
 412                 }
 413                 insert_char(unicode);
 414                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 415 
 416                 if(text_changed_callback_) {
 417                         text_changed_callback_(this, this->text());
 418                 }
 419         }
 420 }

 422 void text_box_base::handle_editing(bool& handled, const utf8::string& unicode, int32_t start, int32_t len)
 423 {
 424         if(unicode.size() > 1 || unicode[0] != 0) {
 425                 handled = true;
 426                 size_t new_len = utf8::size(unicode);
 427                 if(!ime_in_progress_) {
 428                         ime_in_progress_ = true;
 429                         delete_selection();
 430                         ime_start_point_ = selection_start_;
 431                         text_cached_ = text_.text();
 432                         SDL_Rect rect = get_rectangle();
 433                         if(new_len > 0) {
 434                                 rect.x += get_cursor_position(ime_start_point_).x;
 435                                 rect.w = get_cursor_position(ime_start_point_ + new_len).x - rect.x;
 436                         } else {
 437                                 rect.x += get_cursor_position(ime_start_point_ + new_len).x;
 438                                 rect.w = get_cursor_position(ime_start_point_).x - rect.x;
 439                         }
 440                         SDL_SetTextInputRect(&rect);
 441                 }
 442                 ime_cursor_ = start;
 443                 ime_length_ = new_len;
 444                 std::string new_text(text_cached_);
 445                 utf8::insert(new_text, ime_start_point_, unicode);
 446                 text_.set_text(new_text, false);
 447 
 448                 // Update status
 449                 set_cursor(ime_start_point_ + ime_cursor_, false);
 450                 if(len > 0) {
 451                         set_cursor(ime_start_point_ + ime_cursor_ + len, true);
 452                 }
 453                 update_canvas();
 454                 set_is_dirty(true);
 455         }
 456 }

 468 void text_box_base::signal_handler_sdl_key_down(const event::ui_event event,
 469                                                                                  bool& handled,
 470                                                                                  const SDL_Keycode key,
 471                                                                                  SDL_Keymod modifier)
 472 {
 473 
 474         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 475 
 476 // For copy/paste we use a different key on the MAC. Other ctrl modifiers won't
 477 // be modifed seems not to be required when I read the comment in
 478 // widgets/textbox.cpp:516. Would be nice if somebody on a MAC would test it.
 479 #ifdef __APPLE__
 480         const unsigned copypaste_modifier = KMOD_LGUI | KMOD_RGUI;
 481 #else
 482         const unsigned copypaste_modifier = KMOD_CTRL;
 483 #endif
 484 
 485         switch(key) {
 486 
 487                 case SDLK_LEFT:
 488                         handle_key_left_arrow(modifier, handled);
 489                         break;
 490 
 491                 case SDLK_RIGHT:
 492                         handle_key_right_arrow(modifier, handled);
 493                         break;
 494 
 495                 case SDLK_UP:
 496                         handle_key_up_arrow(modifier, handled);
 497                         break;
 498 
 499                 case SDLK_DOWN:
 500                         handle_key_down_arrow(modifier, handled);
 501                         break;
 502 
 503                 case SDLK_PAGEUP:
 504                         handle_key_page_up(modifier, handled);
 505                         break;
 506 
 507                 case SDLK_PAGEDOWN:
 508                         handle_key_page_down(modifier, handled);
 509                         break;
 510 
 511                 case SDLK_a:
 512                         if(!(modifier & KMOD_CTRL)) {
 513                                 return;
 514                         }
 515 
 516                         // If ctrl-a is used for home drop the styled_widget modifier
 517                         modifier = static_cast<SDL_Keymod>(modifier & ~KMOD_CTRL);
 518                         FALLTHROUGH;
 519 
 520                 case SDLK_HOME:
 521                         handle_key_home(modifier, handled);
 522                         break;
 523 
 524                 case SDLK_e:
 525                         if(!(modifier & KMOD_CTRL)) {
 526                                 return;
 527                         }
 528 
 529                         // If ctrl-e is used for end drop the styled_widget modifier
 530                         modifier = static_cast<SDL_Keymod>(modifier & ~KMOD_CTRL);
 531                         FALLTHROUGH;
 532 
 533                 case SDLK_END:
 534                         handle_key_end(modifier, handled);
 535                         break;
 536 
 537                 case SDLK_BACKSPACE:
 538                         handle_key_backspace(modifier, handled);
 539                         break;
 540 
 541                 case SDLK_u:
 542                         if(!(modifier & KMOD_CTRL)) {
 543                                 return;
 544                         }
 545                         handle_key_clear_line(modifier, handled);
 546                         break;
 547 
 548                 case SDLK_DELETE:
 549                         handle_key_delete(modifier, handled);
 550                         break;
 551 
 552                 case SDLK_c:
 553                         if(!(modifier & copypaste_modifier)) {
 554                                 return;
 555                         }
 556 
 557                         // atm we don't care whether there is something to copy or paste
 558                         // if nothing is there we still don't want to be chained.
 559                         copy_selection(false);
 560                         handled = true;
 561                         break;
 562 
 563                 case SDLK_x:
 564                         if(!(modifier & copypaste_modifier)) {
 565                                 return;
 566                         }
 567 
 568                         copy_selection(false);
 569                         delete_selection();
 570                         handled = true;
 571                         break;
 572 
 573                 case SDLK_v:
 574                         if(!(modifier & copypaste_modifier)) {
 575                                 return;
 576                         }
 577 
 578                         paste_selection(false);
 579                         handled = true;
 580                         break;
 581 
 582                 case SDLK_RETURN:
 583                 case SDLK_KP_ENTER:
 584                         if(!ime_in_progress_ || (modifier & (KMOD_CTRL | KMOD_ALT | KMOD_GUI | KMOD_SHIFT))) {
 585                                 return;
 586                         }
 587                         // The IME will handle it, we just need to make sure nothing else handles it too.
 588                         handled = true;
 589                         break;
 590                 
 591                 case SDLK_ESCAPE:
 592                         if(!ime_in_progress_ || (modifier & (KMOD_CTRL | KMOD_ALT | KMOD_GUI | KMOD_SHIFT))) {
 593                                 return;
 594                         }
 595                         interrupt_composition();
 596                         handled = true;
 597                         break;
 598                 
 599                 default:
 600                         // Don't call the text changed callback if nothing happened.
 601                         return;
 602         }
 603 
 604         if(text_changed_callback_) {
 605                 text_changed_callback_(this, this->text());
 606         }
 607 }
