  86 void text_box_base::set_maximum_length(const size_t maximum_length)
  87 {
  88         if(maximum_length == 0) {
  89                 return;
  90         }
  91 
  92         const bool need_update = text_.get_length() > maximum_length;
  93 
  94         text_.set_maximum_length(maximum_length);
  95 
  96         if(need_update) {
  97                 if(selection_start_ > maximum_length) {
  98                         selection_start_ = maximum_length;
  99                         selection_length_ = 0;
 100                 } else if(selection_start_ + selection_length_ > maximum_length) {
 101                         selection_length_ = maximum_length - selection_start_;
 102                 }
 103                 update_canvas();
 104                 set_is_dirty(true);
 105         }
 106 }

 121 void text_box_base::set_cursor(const size_t offset, const bool select)
 122 {
 123         reset_cursor_state();
 124 
 125         if(select) {
 126 
 127                 if(selection_start_ == offset) {
 128                         selection_length_ = 0;
 129                 } else {
 130                         selection_length_ = -static_cast<int>(selection_start_ - offset);
 131                 }
 132 
 133 #ifdef __unix__
 134                 // selecting copies on UNIX systems.
 135                 copy_selection(true);
 136 #endif
 137                 update_canvas();
 138                 set_is_dirty(true);
 139 
 140         } else {
 141                 assert(offset <= text_.get_length());
 142                 selection_start_ = offset;
 143                 selection_length_ = 0;
 144 
 145                 update_canvas();
 146                 set_is_dirty(true);
 147         }
 148 }

 163 void text_box_base::copy_selection(const bool mouse)
 164 {
 165         if(selection_length_ == 0) {
 166                 return;
 167         }
 168 
 169         unsigned end, start = selection_start_;
 170         const utf8::string txt = text_.text();
 171 
 172         if(selection_length_ > 0) {
 173                 end = utf8::index(txt, start + selection_length_);
 174                 start = utf8::index(txt, start);
 175         } else {
 176                 // inverse selection: selection_start_ is in fact the end
 177                 end = utf8::index(txt, start);
 178                 start = utf8::index(txt, start + selection_length_);
 179         }
 180         desktop::clipboard::copy_to_clipboard(txt.substr(start, end - start), mouse);
 181 }

 183 void text_box_base::paste_selection(const bool mouse)
 184 {
 185         const std::string& text = desktop::clipboard::copy_from_clipboard(mouse);
 186         if(text.empty()) {

 188         }

 194         update_canvas();
 195         set_is_dirty(true);

 215 void text_box_base::set_selection(size_t start, int length)
 216 {
 217         const size_t text_size = text_.get_length();
 218 
 219         if(start >= text_size) {
 220                 start = text_size;
 221         }
 222 
 223         if(length == 0) {
 224                 set_cursor(start, false);
 225                 return;
 226         }
 227 
 228         // The text pos/size type differs in both signedness and size with the
 229         // selection length. Such is life.
 230         const int sel_start = std::min<size_t>(start, std::numeric_limits<int>::max());
 231         const int sel_max_length = std::min<size_t>(text_size - start, std::numeric_limits<int>::max());
 232 
 233         const bool backwards = length < 0;
 234 
 235         if(backwards && -length > sel_start) {
 236                 length = -sel_start;
 237         } else if(!backwards && length > sel_max_length) {
 238                 length = sel_max_length;
 239         }
 240 
 241         set_selection_start(start);
 242         set_selection_length(length);
 243 
 244         update_canvas();
 245 }

 255 void text_box_base::toggle_cursor_timer(bool enable)
 256 {
 257         if(!cursor_blink_rate_ms_) {
 258                 return;
 259         }
 260 
 261         if(cursor_timer_) {
 262                 remove_timer(cursor_timer_);
 263         }
 264 
 265         cursor_timer_ = enable
 266                         ? add_timer(cursor_blink_rate_ms_, std::bind(&text_box_base::cursor_timer_callback, this), true)
 267                         : 0;
 268 }

 270 void text_box_base::cursor_timer_callback()
 271 {
 272         unsigned was_alpha = cursor_alpha_;
 273         switch(state_) {
 274                 case DISABLED:
 275                         cursor_alpha_ = 0;
 276                         return;
 277                 case ENABLED:
 278                         cursor_alpha_ = 255;
 279                         return;
 280                 default:
 281                         if(get_window() != open_window_stack.back()) {
 282                                 cursor_alpha_ = 0;
 283                         } else {
 284                                 cursor_alpha_ = (~cursor_alpha_) & 0xFF;
 285                         }
 286         }
 287 
 288         if(was_alpha == cursor_alpha_) {
 289                 return;
 290         }
 291 
 292         for(auto& tmp : get_canvases()) {
 293                 tmp.set_variable("cursor_alpha", wfl::variant(cursor_alpha_));
 294         }
 295 
 296         set_is_dirty(true);
 297 }

 413 void text_box_base::signal_handler_sdl_key_down(const event::ui_event event,
 414                                                                                  bool& handled,
 415                                                                                  const SDL_Keycode key,
 416                                                                                  SDL_Keymod modifier,
 417                                                                                  const utf8::string& unicode)
 418 {
 419 
 420         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 421 
 422 // For copy/paste we use a different key on the MAC. Other ctrl modifiers won't
 423 // be modifed seems not to be required when I read the comment in
 424 // widgets/textbox.cpp:516. Would be nice if somebody on a MAC would test it.
 425 #ifdef __APPLE__
 426         const unsigned copypaste_modifier = KMOD_LGUI | KMOD_RGUI;
 427 #else
 428         const unsigned copypaste_modifier = KMOD_CTRL;
 429 #endif
 430 
 431         switch(key) {
 432 
 433                 case SDLK_LEFT:
 434                         handle_key_left_arrow(modifier, handled);
 435                         break;
 436 
 437                 case SDLK_RIGHT:
 438                         handle_key_right_arrow(modifier, handled);
 439                         break;
 440 
 441                 case SDLK_UP:
 442                         handle_key_up_arrow(modifier, handled);
 443                         break;
 444 
 445                 case SDLK_DOWN:
 446                         handle_key_down_arrow(modifier, handled);
 447                         break;
 448 
 449                 case SDLK_PAGEUP:
 450                         handle_key_page_up(modifier, handled);
 451                         break;
 452 
 453                 case SDLK_PAGEDOWN:
 454                         handle_key_page_down(modifier, handled);
 455                         break;
 456 
 457                 case SDLK_a:
 458                         if(!(modifier & KMOD_CTRL)) {
 459                                 handle_key_default(handled, key, modifier, unicode);
 460                                 break;
 461                         }
 462 
 463                         // If ctrl-a is used for home drop the styled_widget modifier
 464                         modifier = static_cast<SDL_Keymod>(modifier & ~KMOD_CTRL);
 465                         FALLTHROUGH;
 466 
 467                 case SDLK_HOME:
 468                         handle_key_home(modifier, handled);
 469                         break;
 470 
 471                 case SDLK_e:
 472                         if(!(modifier & KMOD_CTRL)) {
 473                                 handle_key_default(handled, key, modifier, unicode);
 474                                 break;
 475                         }
 476 
 477                         // If ctrl-e is used for end drop the styled_widget modifier
 478                         modifier = static_cast<SDL_Keymod>(modifier & ~KMOD_CTRL);
 479                         FALLTHROUGH;
 480 
 481                 case SDLK_END:
 482                         handle_key_end(modifier, handled);
 483                         break;
 484 
 485                 case SDLK_BACKSPACE:
 486                         handle_key_backspace(modifier, handled);
 487                         break;
 488 
 489                 case SDLK_u:
 490                         if(modifier & KMOD_CTRL) {
 491                                 handle_key_clear_line(modifier, handled);
 492                         } else {
 493                                 handle_key_default(handled, key, modifier, unicode);
 494                         }
 495                         break;
 496 
 497                 case SDLK_DELETE:
 498                         handle_key_delete(modifier, handled);
 499                         break;
 500 
 501                 case SDLK_c:
 502                         if(!(modifier & copypaste_modifier)) {
 503                                 handle_key_default(handled, key, modifier, unicode);
 504                                 break;
 505                         }
 506 
 507                         // atm we don't care whether there is something to copy or paste
 508                         // if nothing is there we still don't want to be chained.
 509                         copy_selection(false);
 510                         handled = true;
 511                         break;
 512 
 513                 case SDLK_x:
 514                         if(!(modifier & copypaste_modifier)) {
 515                                 handle_key_default(handled, key, modifier, unicode);
 516                                 break;
 517                         }
 518 
 519                         copy_selection(false);
 520                         delete_selection();
 521                         handled = true;
 522                         break;
 523 
 524                 case SDLK_v:
 525                         if(!(modifier & copypaste_modifier)) {
 526                                 handle_key_default(handled, key, modifier, unicode);
 527                                 break;
 528                         }
 529 
 530                         paste_selection(false);
 531                         handled = true;
 532                         break;
 533 
 534                 default:
 535                         handle_key_default(handled, key, modifier, unicode);
 536         }
 537 
 538         if(text_changed_callback_) {
 539                 text_changed_callback_(this, this->text());
 540         }
 541 }
