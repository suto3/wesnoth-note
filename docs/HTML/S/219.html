  39 addons_client::addons_client(CVideo& v, const std::string& address)
  40         : v_(v)
  41         , addr_(address)
  42         , host_()
  43         , port_()
  44         , conn_(nullptr)
  45         , stat_(nullptr)
  46         , last_error_()
  47 {
  48         const std::vector<std::string>& address_components =
  49                 utils::split(addr_, ':');
  50 
  51         if(address_components.empty()) {
  52                 throw invalid_server_address();
  53         }
  54 
  55         // FIXME: this parsing will break IPv6 numeric addresses! */
  56         host_ = address_components[0];
  57         port_ = address_components.size() == 2 ?
  58                 address_components[1] : std::to_string(default_campaignd_port);
  59 }

 107 bool addons_client::upload_addon(const std::string& id, std::string& response_message, config& cfg)
 108 {
 109         LOG_ADDONS << "preparing to upload " << id << '\n';
 110 
 111         response_message.clear();
 112 
 113         utils::string_map i18n_symbols;
 114         i18n_symbols["addon_title"] = cfg["title"];
 115         if(i18n_symbols["addon_title"].empty()) {
 116                 i18n_symbols["addon_title"] = make_addon_title(id);
 117         }
 118 
 119         std::string passphrase = cfg["passphrase"];
 120         // generate a random passphrase and write it to disk
 121         // if the .pbl file doesn't provide one already
 122         if(passphrase.empty()) {
 123                 passphrase.resize(8);
 124                 for(size_t n = 0; n != 8; ++n) {
 125                         passphrase[n] = 'a' + (rand()%26);
 126                 }
 127                 cfg["passphrase"] = passphrase;
 128                 set_addon_pbl_info(id, cfg);
 129 
 130                 LOG_ADDONS << "automatically generated an initial passphrase for " << id << '\n';
 131         }
 132 
 133         cfg["name"] = id;
 134 
 135         config addon_data;
 136         archive_addon(id, addon_data);
 137 
 138         config request_buf, response_buf;
 139         request_buf.add_child("upload", cfg).add_child("data", addon_data);
 140 
 141         LOG_ADDONS << "sending " << id << '\n';
 142 
 143         this->send_request(request_buf, response_buf);
 144         this->wait_for_transfer_done(vgettext("Sending add-on <i>$addon_title</i>...", i18n_symbols
 145         ), true);
 146 
 147         if(const config& message_cfg = response_buf.child("message")) {
 148                 response_message = message_cfg["message"].str();
 149                 LOG_ADDONS << "server response: " << response_message << '\n';
 150         }
 151 
 152         return !this->update_last_error(response_buf);
 153 
 154 }

 209 bool addons_client::install_addon(config& archive_cfg, const addon_info& info)
 210 {
 211         const cursor::setter cursor_setter(cursor::WAIT);
 212 
 213         utils::string_map i18n_symbols;
 214         i18n_symbols["addon_title"] = info.title;
 215 
 216         if(!check_names_legal(archive_cfg)) {
 217                 gui2::show_error_message(v_,
 218                         vgettext("The add-on <i>$addon_title</i> has an invalid file or directory "
 219                                 "name and cannot be installed.", i18n_symbols));
 220                 return false;
 221         }
 222 
 223         // Add local version information before unpacking
 224 
 225         config* maindir = &archive_cfg.find_child("dir", "name", info.id);
 226         if(!*maindir) {
 227                 LOG_ADDONS << "downloaded add-on '" << info.id << "' is missing its directory in the archive; creating it\n";
 228                 maindir = &archive_cfg.add_child("dir");
 229                 (*maindir)["name"] = info.id;
 230         }
 231 
 232         LOG_ADDONS << "generating version info for add-on '" << info.id << "'\n";
 233 
 234         std::ostringstream info_contents;
 235         config wml;
 236 
 237         info_contents <<
 238                 "#\n"
 239                 "# File automatically generated by Wesnoth to keep track\n"
 240                 "# of version information on installed add-ons. DO NOT EDIT!\n"
 241                 "#\n";
 242 
 243         info.write_minimal(wml.add_child("info"));
 244         write(info_contents, wml);
 245 
 246         config file;
 247         file["name"] = "_info.cfg";
 248         file["contents"] = info_contents.str();
 249 
 250         maindir->add_child("file", file);
 251 
 252         LOG_ADDONS << "unpacking " << info.id << '\n';
 253 
 254         // Remove any previously installed versions
 255         if(!remove_local_addon(info.id)) {
 256                 WRN_ADDONS << "failed to uninstall previous version of " << info.id << "; the add-on may not work properly!" << std::endl;
 257         }
 258 
 259         unarchive_addon(archive_cfg);
 260         LOG_ADDONS << "unpacking finished\n";
 261 
 262         return true;
 263 }

 265 bool addons_client::try_fetch_addon(const addon_info & addon)
 266 {
 267         config archive;
 268 
 269         if(!(
 270                 download_addon(archive, addon.id, addon.title, !is_addon_installed(addon.id)) &&
 271                 install_addon(archive, addon)
 272                 )) {
 273                 const std::string& server_error = get_last_server_error();
 274                 if(!server_error.empty()) {
 275                         gui2::show_error_message(v_,
 276                                 _("The server responded with an error:") + "\n" + server_error);
 277                 }
 278                 return false;
 279         } else {
 280                 return true;

 284 addons_client::install_result addons_client::do_resolve_addon_dependencies(const addons_list& addons, const addon_info& addon)
 285 {
 286         install_result result;
 287         result.outcome = install_outcome::success;
 288         result.wml_changed = false;
 289 
 290         std::unique_ptr<cursor::setter> cursor_setter(new cursor::setter(cursor::WAIT));
 291 
 292         // TODO: We don't currently check for the need to upgrade. I'll probably
 293         // work on that when implementing dependency tiers later.
 294 
 295         const std::set<std::string>& deps = addon.resolve_dependencies(addons);
 296 
 297         std::vector<std::string> missing_deps;
 298         std::vector<std::string> broken_deps;
 299 
 300         for(const std::string& dep : deps) {
 301                 if(!is_addon_installed(dep)) {
 302                         if(addons.find(dep) != addons.end()) {
 303                                 missing_deps.push_back(dep);
 304                         } else {
 305                                 broken_deps.push_back(dep);
 306                         }
 307                 }
 308         }
 309 
 310         cursor_setter.reset();
 311 
 312         if(!broken_deps.empty()) {
 313                 std::string broken_deps_report;
 314 
 315                 broken_deps_report = _n(
 316                         "The selected add-on has the following dependency, which is not currently installed or available from the server. Do you wish to continue?",
 317                         "The selected add-on has the following dependencies, which are not currently installed or available from the server. Do you wish to continue?",
 318                         broken_deps.size());
 319                 broken_deps_report += "\n";
 320 
 321                 for(const std::string& broken_dep_id : broken_deps) {
 322                         broken_deps_report += "\n    " + font::unicode_bullet + " " + make_addon_title(broken_dep_id);
 323                 }
 324 
 325                 if(gui2::show_message(v_, _("Broken Dependencies"), broken_deps_report, gui2::dialogs::message::yes_no_buttons) != gui2::window::OK) {
 326                         result.outcome = install_outcome::abort;
 327                         return result; // canceled by user
 328                 }
 329         }
 330 
 331         if(missing_deps.empty()) {
 332                 // No dependencies to install, carry on.
 333                 return result;
 334         }
 335 
 336         {
 337                 addons_list options;
 338                 for(const std::string& dep : missing_deps) {
 339                         const addon_info& missing_addon = addons.at(dep);
 340                         options[dep] = missing_addon;
 341                 }
 342 
 343                 gui2::dialogs::install_dependencies dlg(options);
 344                 bool cont = dlg.show(v_);
 345                 if(!cont) {
 346                         return result; // the user has chosen to continue without installing anything.
 347                 }
 348         }
 349 
 350         //
 351         // Install dependencies now.
 352         //
 353 
 354         std::vector<std::string> failed_titles;
 355 
 356         for(const std::string& dep : missing_deps) {
 357                 const addon_info& missing_addon = addons.at(dep);
 358 
 359                 if(!try_fetch_addon(missing_addon)) {
 360                         failed_titles.push_back(missing_addon.title);
 361                 } else {
 362                         result.wml_changed = true;
 363                 }
 364         }
 365 
 366         if(!failed_titles.empty()) {
 367                 const std::string& failed_deps_report = _n(
 368                         "The following dependency could not be installed. Do you still wish to continue?",
 369                         "The following dependencies could not be installed. Do you still wish to continue?",
 370                         failed_titles.size()) + std::string("\n\n") + utils::bullet_list(failed_titles);
 371 
 372                 result.outcome = gui2::show_message(v_, _("Dependencies Installation Failed"), failed_deps_report, gui2::dialogs::message::yes_no_buttons) == gui2::window::OK ? install_outcome::success : install_outcome::abort; // If the user cancels, return abort. Otherwise, return success, since the user chose to ignore the failure.
 373                 return result;
 374         }
 375 
 376         return result;
 377 }

 379 bool addons_client::do_check_before_overwriting_addon(const addon_info& addon)
 380 {
 381         const std::string& addon_id = addon.id;
 382 
 383         const bool pbl = have_addon_pbl_info(addon_id);
 384         const bool vcs = have_addon_in_vcs_tree(addon_id);
 385 
 386         if(!pbl && !vcs) {
 387                 return true;
 388         }
 389 
 390         utils::string_map symbols;
 391         symbols["addon"] = addon.title;
 392         std::string text;
 393         std::vector<std::string> extra_items;
 394 
 395         text = vgettext("The add-on '$addon|' is already installed and contains additional information that will be permanently lost if you continue:", symbols);
 396         text += "\n\n";
 397 
 398         if(pbl) {
 399                 extra_items.push_back(_("Publishing information file (.pbl)"));
 400         }
 401 
 402         if(vcs) {
 403                 extra_items.push_back(_("Version control system (VCS) information"));
 404         }
 405 
 406         text += utils::bullet_list(extra_items) + "\n\n";
 407         text += _("Do you really wish to continue?");
 408 
 409         return gui2::show_message(v_, _("Confirm"), text, gui2::dialogs::message::yes_no_buttons) == gui2::window::OK;
 410 }

 412 addons_client::install_result addons_client::install_addon_with_checks(const addons_list& addons, const addon_info& addon)
 413 {
 414         if(!(do_check_before_overwriting_addon(addon))) {
 415                 // Just do nothing and leave.
 416                 install_result result;
 417                 result.outcome = install_outcome::abort;
 418                 result.wml_changed = false;
 419 
 420                 return result;
 421         }
 422 
 423         // Resolve any dependencies
 424         install_result res = do_resolve_addon_dependencies(addons, addon);
 425         if(res.outcome != install_outcome::success) { // this function only returns SUCCESS and ABORT as outcomes
 426                 return res; // user aborted
 427         }
 428 
 429         if(!try_fetch_addon(addon)) {
 430                 res.outcome = install_outcome::failure;
 431                 return res; //wml_changed should have whatever value was obtained in resolving dependencies
 432         } else {
 433                 res.wml_changed = true;
 434                 return res; //we successfully installed something, so now the wml was definitely changed
 435         }
 436 }

 493 void addons_client::wait_for_transfer_done(const std::string& status_message, bool track_upload)
 494 {
 495         check_connected();
 496         std::unique_ptr<network_transmission::connection_data> cd;
 497         if(track_upload)
 498                 cd.reset(new write_addon_connection_data{ *conn_ });
 499         else
 500                 cd.reset(new read_addon_connection_data{ *conn_ });
 501         if(!stat_) {
 502                 stat_ = new network_transmission(*cd, _("Add-ons Manager"), status_message);
 503         } else {
 504                 stat_->set_subtitle(status_message);
 505                 stat_->set_connection_data(*cd);
 506         }
 507 
 508         if(!stat_->show(v_)) {
 509                 // Notify the caller chain that the user aborted the operation.
 510                 throw user_exit();
 511         }
 512 }
