  14 class RecursionManager(object):
  15     """Keeps track of the levels of recursion and whether expansion should happen or not """
  16     default = 2
  17     curr = 0
  18 
  19     # @classmethod
  20     # def __init__(cls, val):
  21     #     cls.default = val
  22     #     cls.curr=0;
  23 
  24     @classmethod
  25     def get_level(cls):
  26         return cls.default

  28     @classmethod
  29     def set_level(cls, val):
  30         cls.curr = 0
  31         if val >= 0:
  32             cls.default = val
  33         return cls.default

  37         cls.curr = 0

  40     @classmethod
  41     def should_display(cls):
  42         return cls.curr <= cls.default

 154             elif attr_type == Atype.INT:

 162             else:
 163                 yield 'unknown', "unknown type code = " + ('%d' % attr_type)

 175 class BoostUnorderedMapPrinter(object):
 176     """ Print a boost unordered map"""
 177 
 178     class _iterator:
 179         def __init__(self, val):
 180             self.buckets = val['table_']['buckets_']
 181             if self.buckets != 0:
 182                 t_key = val.type.template_argument(0)
 183                 t_val = val.type.template_argument(1)
 184                 self.bucket_count = val['table_']['bucket_count_']
 185                 self.current_bucket = 0
 186                 pair = "std::pair<%s const, %s>" % (t_key, t_val)
 187                 self.pair_pointer = gdb.lookup_type(pair).pointer()
 188                 self.base_pointer = gdb.lookup_type("boost::unordered_detail::value_base< %s >" % pair).pointer()
 189                 self.node_pointer = gdb.lookup_type("boost::unordered_detail::hash_node<std::allocator< %s >, boost::unordered_detail::ungrouped>" % pair).pointer()
 190                 self.node = self.buckets[self.current_bucket]['next_']
 191 
 192         def __iter__(self):
 193             return self
 194 
 195         def next(self):
 196             if self.buckets == 0 or not RecursionManager.should_display():
 197                 raise StopIteration
 198             while not self.node:
 199                 self.current_bucket += 1
 200                 if self.current_bucket >= self.bucket_count:
 201                     raise StopIteration
 202                 self.node = self.buckets[self.current_bucket]['next_']
 203 
 204             iterator = self.node.cast(self.node_pointer).cast(self.base_pointer).cast(self.pair_pointer).dereference()
 205             self.node = self.node['next_']
 206 
 207             return ('%s' % iterator['first']), iterator['second']
 208 
 209     def __init__(self, val):
 210         self.val = val
 211         self.buckets = val['table_']['buckets_']
 212         self.descended = False
 213 
 214     def __del__(self):
 215         if self.descended:
 216             RecursionManager.dec()
 217 
 218     def children(self):
 219         self.descended = True
 220         RecursionManager.inc()
 221         return self._iterator(self.val)
 222 
 223     def to_string(self):
 224         ret = "boost::unordered_map"
 225         if self.buckets == 0:
 226             ret += " EMPTY"
 227         return ret
