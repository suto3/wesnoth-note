 102 void mouse_handler::mouse_motion(int x, int y, const bool browse, bool update, map_location new_hex)
 103 {
 104         // we ignore the position coming from event handler
 105         // because it's always a little obsolete and we don't need
 106         // to highlight all the hexes where the mouse passed.
 107         // Also, sometimes it seems to have one *very* obsolete
 108         // and isolated mouse motion event when using drag&drop
 109         SDL_GetMouseState(&x,&y);  // <-- modify x and y
 110 
 111         if (mouse_handler_base::mouse_motion_default(x, y, update)) return;
 112 
 113         game_board & board = pc_.gamestate().board_;
 114 
 115         if (new_hex == map_location::null_location())
 116                 new_hex = gui().hex_clicked_on(x,y);
 117 
 118         if(new_hex != last_hex_) {
 119                 if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 120                         lk->mouse_over_hex_callback(new_hex);
 121                 }
 122                 update = true;
 123                 if ( pc_.get_map_const().on_board(last_hex_) ) {
 124                         // we store the previous hexes used to propose attack direction
 125                         previous_hex_ = last_hex_;
 126                         // the hex of the selected unit is also "free"
 127                         { // start planned unit map scope
 128                                 wb::future_map_if_active raii;
 129                                 if (last_hex_ == selected_hex_ || !find_unit(last_hex_)) {
 130                                         previous_free_hex_ = last_hex_;
 131                                 }
 132                         } // end planned unit map scope
 133                 }
 134                 last_hex_ = new_hex;
 135         }
 136 
 137         if (reachmap_invalid_) update = true;
 138 
 139         if (!update) return;
 140 
 141         if (reachmap_invalid_) {
 142                 reachmap_invalid_ = false;
 143                 if (!current_paths_.destinations.empty() && !show_partial_move_) {
 144                         bool selected_hex_has_unit;
 145                         { // start planned unit map scope
 146                                 wb::future_map_if_active planned_unit_map;
 147                                 selected_hex_has_unit = find_unit(selected_hex_).valid();
 148                         } // end planned unit map scope
 149                         if(selected_hex_.valid() && selected_hex_has_unit ) {
 150                                 // reselect the unit without firing events (updates current_paths_)
 151                                 select_hex(selected_hex_, true);
 152                         }
 153                         // we do never deselect here, mainly because of canceled attack-move
 154                 }
 155         }
 156 
 157         // reset current_route_ and current_paths if not valid anymore
 158         // we do it before cursor selection, because it uses current_paths_
 159         if( !pc_.get_map_const().on_board(new_hex) ) {
 160                 current_route_.steps.clear();
 161                 gui().set_route(nullptr);
 162                 pc_.get_whiteboard()->erase_temp_move();
 163         }
 164 
 165         if(unselected_paths_) {
 166                 unselected_paths_ = false;
 167                 current_paths_ = pathfind::paths();
 168                 gui().unhighlight_reach();
 169         } else if(over_route_) {
 170                 over_route_ = false;
 171                 current_route_.steps.clear();
 172                 gui().set_route(nullptr);
 173                 pc_.get_whiteboard()->erase_temp_move();
 174         }
 175 
 176         gui().highlight_hex(new_hex);
 177         pc_.get_whiteboard()->on_mouseover_change(new_hex);
 178 
 179         unit_map::iterator selected_unit;
 180         unit_map::iterator mouseover_unit;
 181         map_location attack_from;
 182 
 183         { // start planned unit map scope
 184                 wb::future_map_if_active planned_unit_map;
 185                 selected_unit = find_unit(selected_hex_);
 186                 mouseover_unit = find_unit(new_hex);
 187 
 188                 // we search if there is an attack possibility and where
 189                 attack_from = current_unit_attacks_from(new_hex);
 190 
 191                 //see if we should show the normal cursor, the movement cursor, or
 192                 //the attack cursor
 193                 //If the cursor is on WAIT, we don't change it and let the setter
 194                 //of this state end it
 195                 if (cursor::get() != cursor::WAIT) {
 196                         if (selected_unit &&
 197                                         selected_unit->side() == side_num_ &&
 198                                         !selected_unit->incapacitated() && !browse)
 199                         {
 200                                 if (attack_from.valid()) {
 201                                         cursor::set(dragging_started_ ? cursor::ATTACK_DRAG : cursor::ATTACK);
 202                                 }
 203                                 else if (!mouseover_unit &&
 204                                                 current_paths_.destinations.contains(new_hex))
 205                                 {
 206                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 207                                 } else {
 208                                         // selected unit can't attack or move there
 209                                         cursor::set(cursor::NORMAL);
 210                                 }
 211                         } else {
 212                                 // no selected unit or we can't move it
 213 
 214                                 if ( selected_hex_.valid() && mouseover_unit
 215                                                 && mouseover_unit->side() == side_num_ ) {
 216                                         // empty hex field selected and unit on our site under the cursor
 217                                         cursor::set(dragging_started_ ? cursor::MOVE_DRAG : cursor::MOVE);
 218                                 } else {
 219                                         cursor::set(cursor::NORMAL);
 220                                 }
 221                         }
 222                 }
 223         } // end planned unit map scope
 224 
 225         // show (or cancel) the attack direction indicator
 226         if (attack_from.valid() && (!browse || pc_.get_whiteboard()->is_active())) {
 227                 gui().set_attack_indicator(attack_from, new_hex);
 228         } else {
 229                 gui().clear_attack_indicator();
 230         }
 231 
 232         unit_ptr un; //will later point to unit at mouseover_hex_
 233 
 234         // the destination is the pointed hex or the adjacent hex
 235         // used to attack it
 236         map_location dest;
 237         unit_map::const_iterator dest_un;
 238         { // start planned unit map scope
 239                 wb::future_map_if_active raii;
 240                 if (attack_from.valid()) {
 241                         dest = attack_from;
 242                         dest_un = find_unit(dest);
 243                 }       else {
 244                         dest = new_hex;
 245                         dest_un = find_unit(new_hex);
 246                 }
 247 
 248                 if(dest == selected_hex_ || dest_un) {
 249                         current_route_.steps.clear();
 250                         gui().set_route(nullptr);
 251                         pc_.get_whiteboard()->erase_temp_move();
 252                 }
 253                 else if (!current_paths_.destinations.empty() &&
 254                                 board.map().on_board(selected_hex_) && board.map().on_board(new_hex))
 255                 {
 256                         if (selected_unit && !selected_unit->incapacitated()) {
 257                                 // Show the route from selected unit to mouseover hex
 258                                 current_route_ = get_route(&*selected_unit, dest, viewing_team());
 259 
 260                                 pc_.get_whiteboard()->create_temp_move();
 261 
 262                                 if(!browse) {
 263                                         gui().set_route(&current_route_);
 264                                 }
 265                         }
 266                 }
 267 
 268                 if(board.map().on_board(selected_hex_)
 269                                 && !selected_unit
 270                                 && mouseover_unit.valid()
 271                                 && mouseover_unit) {
 272                         // Show the route from selected hex to mouseover unit
 273                         current_route_ = get_route(&*mouseover_unit, selected_hex_, viewing_team());
 274 
 275                         pc_.get_whiteboard()->create_temp_move();
 276 
 277                         if(!browse) {
 278                                 gui().set_route(&current_route_);
 279                         }
 280                 } else if (!selected_unit) {
 281                         current_route_.steps.clear();
 282                         gui().set_route(nullptr);
 283                         pc_.get_whiteboard()->erase_temp_move();
 284                 }
 285 
 286                 unit_map::iterator iter = mouseover_unit;
 287                 if (iter)
 288                         un = iter.get_shared_ptr();
 289                 else
 290                         un.reset();
 291         } //end planned unit map scope
 292 
 293         if ( (!selected_hex_.valid()) && un && current_paths_.destinations.empty() &&
 294                         !gui().fogged(un->get_location()))
 295         {
 296                 if (un->side() == side_num_) {
 297                         //unit is on our team, show path if the unit has one
 298                         const map_location go_to = un->get_goto();
 299                         if(board.map().on_board(go_to)) {
 300                                 pathfind::marked_route route;
 301                                 { // start planned unit map scope
 302                                         wb::future_map_if_active raii;
 303                                         route = get_route(un.get(), go_to, current_team());
 304                                 } // end planned unit map scope
 305                                 gui().set_route(&route);
 306                         }
 307                         over_route_ = true;
 308 
 309                         wb::future_map_if_active raii;
 310                         current_paths_ = pathfind::paths(*un, false, true,
 311                                         viewing_team(), path_turns_);
 312                 } else {
 313                         //unit under cursor is not on our team
 314                         //Note: planned unit map must be activated after this is done,
 315                         //since the future state includes changes to units' movement.
 316                         unit_movement_resetter move_reset(*un);
 317 
 318                         wb::future_map_if_active raii;
 319                         current_paths_ = pathfind::paths(*un, false, true,
 320                                         viewing_team(), path_turns_);
 321                 }
 322 
 323                 unselected_paths_ = true;
 324                 gui().highlight_reach(current_paths_);
 325         }
 326 
 327 }

 353 map_location mouse_handler::current_unit_attacks_from(const map_location& loc) const
 354 {
 355         if(loc == selected_hex_)
 356                 return map_location();
 357 
 358         bool wb_active = pc_.get_whiteboard()->is_active();
 359 
 360         {
 361                 // Check the unit SOURCE of the attack
 362 
 363                 // Check that there's a selected unit
 364                 const unit_map::const_iterator source_unit = find_unit(selected_hex_);
 365                 bool source_eligible = source_unit.valid();
 366                 if (!source_eligible) return map_location();
 367 
 368                 // The selected unit must at least belong to the player currently controlling this client.
 369                 source_eligible &= source_unit->side() == gui_->viewing_side();
 370                 if (!source_eligible) return map_location();
 371 
 372                 // In addition:
 373                 // - If whiteboard is enabled, we allow planning attacks outside of player's turn
 374                 // - If whiteboard is disabled, it must be the turn of the player controlling this client
 375                 if(!wb_active) {
 376                         source_eligible &= gui_->viewing_side() == pc_.current_side();
 377                         if (!source_eligible) return map_location();
 378                 }
 379 
 380                 // Unit must have attacks left
 381                 source_eligible &= source_unit->attacks_left() != 0;
 382                 if (!source_eligible) return map_location();
 383 
 384 
 385                 // Check the unit TARGET of the attack
 386 
 387                 team const& viewer = viewing_team();
 388 
 389                 // Check that there's a unit at the target location
 390                 const unit_map::const_iterator target_unit = find_unit(loc);
 391                 bool target_eligible = target_unit.valid();
 392                 if (!target_eligible) return map_location();
 393 
 394                 // The player controlling this client must be an enemy of the target unit's side
 395                 target_eligible &= viewer.is_enemy(target_unit->side());
 396                 if (!target_eligible) return map_location();
 397 
 398                 // Sanity check: source and target of the attack shouldn't be on the same team
 399                 assert(source_unit->side() != target_unit->side());
 400 
 401                 target_eligible &= !target_unit->incapacitated();
 402                 if (!target_eligible) return map_location();
 403         }
 404 
 405         const map_location::DIRECTION preferred = loc.get_relative_dir(previous_hex_);
 406         const map_location::DIRECTION second_preferred = loc.get_relative_dir(previous_free_hex_);
 407 
 408         int best_rating = 100;//smaller is better
 409         map_location res;
 410         map_location adj[6];
 411         get_adjacent_tiles(loc,adj);
 412 
 413         for(size_t n = 0; n != 6; ++n) {
 414                 if(pc_.gamestate().board_.map().on_board(adj[n]) == false) {
 415                         continue;
 416                 }
 417 
 418                 if(adj[n] != selected_hex_ && find_unit(adj[n])) {
 419                         continue;
 420                 }
 421 
 422                 if (current_paths_.destinations.contains(adj[n]))
 423                 {
 424                         static const size_t NDIRECTIONS = map_location::NDIRECTIONS;
 425                         unsigned int difference = std::abs(int(preferred - n));
 426                         if(difference > NDIRECTIONS/2) {
 427                                 difference = NDIRECTIONS - difference;
 428                         }
 429                         unsigned int second_difference = std::abs(int(second_preferred - n));
 430                         if(second_difference > NDIRECTIONS/2) {
 431                                 second_difference = NDIRECTIONS - second_difference;
 432                         }
 433                         const int rating = difference * 2 + (second_difference > difference);
 434                         if(rating < best_rating || res.valid() == false) {
 435                                 best_rating = rating;
 436                                 res = adj[n];
 437                         }
 438                 }
 439         }
 440 
 441         return res;
 442 }

 444 pathfind::marked_route mouse_handler::get_route(const unit* un, map_location go_to, team &team) const
 445 {
 446         game_board & board = pc_.gamestate().board_;
 447 
 448         // The pathfinder will check unit visibility (fogged/stealthy).
 449         const pathfind::shortest_path_calculator calc(*un, team, board.teams(), board.map());
 450 
 451         pathfind::teleport_map allowed_teleports = pathfind::get_teleport_locations(*un, viewing_team());
 452 
 453         pathfind::plain_route route;
 454 
 455         route = pathfind::a_star_search(un->get_location(), go_to, 10000.0, calc, board.map().w(), board.map().h(), &allowed_teleports);
 456 
 457         return mark_route(route);
 458 }

 471 void mouse_handler::select_or_action(bool browse)
 472 {
 473         if (!pc_.get_map_const().on_board(last_hex_)){
 474                 tooltips::click(drag_from_x_, drag_from_y_);
 475                 return;
 476         }
 477 
 478         // Load whiteboard partial moves
 479         wb::future_map_if_active planned_unit_map;
 480 
 481         unit_map::iterator clicked_u = find_unit(last_hex_);
 482         unit_map::iterator selected_u = find_unit(selected_hex_);
 483         if ( clicked_u &&
 484                  (!selected_u ||
 485                   selected_u->side() != side_num_ ||
 486                   (clicked_u->side() == side_num_ && clicked_u->id() != selected_u->id())) )
 487         {
 488                 select_hex(last_hex_, false);
 489         }
 490         else
 491         {
 492                 move_action(browse);
 493         }
 494 }

 496 void mouse_handler::move_action(bool browse)
 497 {
 498         // Lock whiteboard activation state to avoid problems due to
 499         // its changing while an animation takes place.
 500         wb::whiteboard_lock wb_lock = pc_.get_whiteboard()->get_activation_state_lock();
 501 
 502         //we use the last registered highlighted hex
 503         //since it's what update our global state
 504         map_location hex = last_hex_;
 505 
 506         // TODO
 507         //      // Clicks on border hexes mean to deselect.
 508         //      // (Check this before doing processing that might not be needed.)
 509         //      if ( !pc_.get_map_const().on_board(hex) ) {
 510         //              deselect_hex();
 511         //              return false;
 512         //      }
 513         if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 514                 lk->select_hex_callback(last_hex_);
 515         }
 516         unit_map::iterator u;
 517         unit_map::iterator clicked_u;
 518         map_location src;
 519         pathfind::paths orig_paths;
 520         map_location attack_from;
 521         { // start planned unit map scope
 522                 wb::future_map_if_active planned_unit_map;
 523                 u = find_unit(selected_hex_);
 524 
 525                 //if the unit is selected and then itself clicked on,
 526                 //any goto command is canceled
 527                 if (u && !browse && selected_hex_ == hex && u->side() == side_num_) {
 528                         u->set_goto(map_location());
 529                 }
 530 
 531                 clicked_u = find_unit(hex);
 532 
 533                 src = selected_hex_;
 534                 orig_paths = current_paths_;
 535                 attack_from = current_unit_attacks_from(hex);
 536         } // end planned unit map scope
 537 
 538         //see if we're trying to do a attack or move-and-attack
 539         if((!browse || pc_.get_whiteboard()->is_active()) && attack_from.valid()) {
 540 
 541                 // Ignore this command if commands are disabled.
 542                 if ( commands_disabled )
 543                         return;
 544 
 545                 if (((u.valid() && u->side() == side_num_) || pc_.get_whiteboard()->is_active()) && clicked_u.valid() ) {
 546                         if (attack_from == selected_hex_) { //no move needed
 547                                 int choice = -1;
 548                                 { wb::future_map_if_active planned_unit_map; //start planned unit map scope
 549                                         choice = show_attack_dialog(attack_from, clicked_u->get_location());
 550                                 } // end planned unit map scope
 551                                 if (choice >=0 ) {
 552                                         if (pc_.get_whiteboard()->is_active()) {
 553                                                 save_whiteboard_attack(attack_from, clicked_u->get_location(), choice);
 554                                         } else {
 555                                                 // clear current unit selection so that any other unit selected
 556                                                 // triggers a new selection
 557                                                 selected_hex_ = map_location();
 558 
 559                                                 attack_enemy(u->get_location(), clicked_u->get_location(), choice);
 560                                         }
 561                                 }
 562                                 return;
 563                         }
 564                         else {
 565 
 566                                 int choice = -1; //for the attack dialog
 567 
 568                                 { wb::future_map_if_active planned_unit_map; //start planned unit map scope
 569                                         // we will now temporary move next to the enemy
 570                                         pathfind::paths::dest_vect::const_iterator itor =
 571                                                         current_paths_.destinations.find(attack_from);
 572                                         if(itor == current_paths_.destinations.end()) {
 573                                                 // can't reach the attacking location
 574                                                 // not supposed to happen, so abort
 575                                                 return;
 576                                         }
 577 
 578                                         // block where we temporary move the unit
 579                                         {
 580                                                 temporary_unit_mover temp_mover(pc_.gamestate().board_.units_, src, attack_from,
 581                                                                                 itor->move_left);
 582                                                 choice = show_attack_dialog(attack_from, clicked_u->get_location());
 583                                         }
 584 
 585                                         if (choice < 0) {
 586                                                 // user hit cancel, don't start move+attack
 587                                                 return;
 588                                         }
 589                                 } // end planned unit map scope
 590 
 591                                 if (pc_.get_whiteboard()->is_active()) {
 592                                         save_whiteboard_attack(attack_from, hex, choice);
 593                                 }
 594                                 else {
 595                                         bool not_interrupted = move_unit_along_current_route();
 596                                         bool alt_unit_selected = (selected_hex_ != src);
 597                                         src = selected_hex_;
 598                                         // clear current unit selection so that any other unit selected
 599                                         // triggers a new selection
 600                                         selected_hex_ = map_location();
 601 
 602                                         if (not_interrupted)
 603                                                 attack_enemy(attack_from, hex, choice); // Fight !!
 604 
 605                                         //TODO: Maybe store the attack choice so "press t to continue"
 606                                         //      can also continue the attack?
 607 
 608                                         if (alt_unit_selected && !selected_hex_.valid()) {
 609                                                 //reselect other unit if selected during movement animation
 610                                                 select_hex(src, browse);
 611                                         }
 612                                 }
 613 
 614                                 return;
 615                         }
 616                 }
 617         }
 618         //otherwise we're trying to move to a hex
 619         else if (
 620                         // the old use case: move selected unit to mouse hex field
 621                         ( (!browse || pc_.get_whiteboard()->is_active()) &&
 622                                         selected_hex_.valid() && selected_hex_ != hex &&
 623                                         u.valid() &&
 624                                         (u->side() == side_num_ || pc_.get_whiteboard()->is_active()) &&
 625                                         !clicked_u &&
 626                                         !current_route_.steps.empty() &&
 627                                         current_route_.steps.front() == selected_hex_
 628                         )
 629                         || // the new use case: move mouse unit to selected hex field
 630                         ( (!browse || pc_.get_whiteboard()->is_active()) &&
 631                                         selected_hex_.valid() && selected_hex_ != hex &&
 632                                         clicked_u &&
 633                                         !current_route_.steps.empty() &&
 634                                         current_route_.steps.back() == selected_hex_
 635                                         && !u
 636                                         && clicked_u->side() == side_num_
 637                         )
 638         ) {
 639 
 640                 // Ignore this command if commands are disabled.
 641                 if ( commands_disabled )
 642                         return;
 643 
 644                 // If the whiteboard is active, it intercepts any unit movement
 645                 if (pc_.get_whiteboard()->is_active()) {
 646                                 // deselect the current hex, and create planned move for whiteboard
 647                                 selected_hex_ = map_location();
 648                                 gui().select_hex(map_location());
 649                                 gui().clear_attack_indicator();
 650                                 gui().set_route(nullptr);
 651                                 show_partial_move_ = false;
 652                                 gui().unhighlight_reach();
 653                                 current_paths_ = pathfind::paths();
 654                                 current_route_.steps.clear();
 655 
 656                                 pc_.get_whiteboard()->save_temp_move();
 657 
 658                 // Otherwise proceed to normal unit movement
 659                 } else {
 660                         //Don't move if the unit already has actions
 661                         //from the whiteboard.
 662                         if (pc_.get_whiteboard()->unit_has_actions(
 663                                         u ? &*u : &*clicked_u )) {
 664                                 return;
 665                         }
 666 
 667                         move_unit_along_current_route();
 668                         // during the move, we may have selected another unit
 669                         // (but without triggering a select event (command was disabled)
 670                         // in that case reselect it now to fire the event (+ anim & sound)
 671                         if (selected_hex_ != src) {
 672                                 select_hex(selected_hex_, browse);
 673                         }
 674                 }
 675                 return;
 676         }
 677 }

 679 void mouse_handler::select_hex(const map_location& hex, const bool browse, const bool highlight, const bool fire_event) {
 680 
 681         selected_hex_ = hex;
 682         if(game_lua_kernel* lk = pc_.gamestate().lua_kernel_.get()) {
 683                 lk->select_hex_callback(last_hex_);
 684         }
 685

 687         gui().clear_attack_indicator();
 688         gui().set_route(nullptr);
 689         show_partial_move_ = false;
 690 
 691         wb::future_map_if_active planned_unit_map; //lasts for whole method
 692 
 693         unit_map::iterator unit = find_unit(selected_hex_);
 694 
 695         if (selected_hex_.valid() && unit.valid() && !unit->get_hidden()) {
 696 
 697                 next_unit_ = unit->get_location();
 698 
 699                 {
 700                         current_paths_ = pathfind::paths(*unit, false, true, viewing_team(), path_turns_);

 702                 if(highlight) {
 703                         show_attack_options(unit);
 704                         gui().highlight_reach(current_paths_);

 706                 // the highlight now comes from selection
 707                 // and not from the mouseover on an enemy
 708                 unselected_paths_ = false;
 709                 gui().set_route(nullptr);
 710 
 711                 // selection have impact only if we are not observing and it's our unit
 712                 if ((!commands_disabled || pc_.get_whiteboard()->is_active()) && unit->side() == gui().viewing_side()) {
 713                         if (!(browse || pc_.get_whiteboard()->unit_has_actions(&*unit)))
 714                         {
 715                                 sound::play_UI_sound("select-unit.wav");
 716                                 unit->anim_comp().set_selecting();
 717                                 if(fire_event) {
 718                                         // ensure unit map is back to normal while event is fired
 719                                         wb::real_map srum;
 720                                         pc_.pump().fire("select", hex);
 721                                         //end forced real unit map
 722                                 }
 723                         }
 724                 }
 725                 return;
 726         }
 727 
 728         if (selected_hex_.valid() && !unit) {
 729                 // compute unit in range of the empty selected_hex field
 730 
 731                 gui_->unhighlight_reach();
 732 
 733                 pathfind::paths reaching_unit_locations;
 734 
 735                 pathfind::paths clicked_location;
 736                 clicked_location.destinations.insert(hex);
 737 
 738                 for(unit_map::iterator u = pc_.gamestate().board_.units_.begin(); u != pc_.gamestate().board_.units_.end(); ++u) {
 739                         bool invisible = u->invisible(u->get_location(), gui_->get_disp_context());
 740 
 741                         if (!gui_->fogged(u->get_location()) && !u->incapacitated() && !invisible)
 742                         {
 743 
 744                                 const pathfind::paths& path = pathfind::paths(*u, false, true,
 745                                                 viewing_team(), path_turns_, false, false);
 746 
 747                                 if (path.destinations.find(hex) != path.destinations.end()) {
 748                                         reaching_unit_locations.destinations.insert(u->get_location());
 749                                         gui_->highlight_another_reach(clicked_location);
 750                                 }
 751                         }
 752                 }
 753                 gui_->highlight_another_reach(reaching_unit_locations);
 754         } else {
 755                 if (!pc_.gamestate().board_.units_.find(last_hex_)) {
 756                         gui_->unhighlight_reach();
 757                 }
 758                 current_paths_ = pathfind::paths();
 759                 current_route_.steps.clear();
 760 
 761                 pc_.get_whiteboard()->on_deselect_hex();
 762         }

 769 /**
 770  * Moves a unit along the currently cached route.
 771  *
 772  * @returns  true if the end of the route was reached and no information was
 773  *           uncovered that would warrant interrupting a chain of actions;
 774  *           false otherwise.
 775  */
 776 bool mouse_handler::move_unit_along_current_route()
 777 {
 778         // Copy the current route to ensure it remains valid throughout the animation.
 779         const std::vector<map_location> steps = current_route_.steps;
 780 
 781         // do not show footsteps during movement
 782         gui().set_route(nullptr);
 783         gui().unhighlight_reach();
 784 
 785         // do not keep the hex highlighted that we started from
 786         selected_hex_ = map_location();
 787         gui().select_hex(map_location());
 788 
 789         bool interrupted = false;
 790         if ( steps.size() > 1 )
 791         {
 792                 size_t num_moves = move_unit_along_route(steps, interrupted);
 793 
 794                 interrupted =  interrupted || num_moves + 1 < steps.size();
 795                 next_unit_ = steps[num_moves];
 796         }
 797 
 798         // invalid after the move
 799         current_paths_ = pathfind::paths();
 800         current_route_.steps.clear();
 801 
 802         return !interrupted;
 803 }

 805 /**
 806  * Moves a unit across the board for a player.
 807  * This is specifically for movement at the time it is initiated by a player,
 808  * whether via a mouse click or executing whiteboard actions. Continued moves
 809  * (including goto execution) can bypass this and call actions::move_unit() directly.
 810  * This function call may include time for an animation, so make sure the
 811  * provided route will remain unchanged (the caller should probably make a local
 812  * copy).
 813  *
 814  * @param[in]   steps           The route to be traveled. The unit to be moved is at the beginning of this route.
 815  * @param[out]  interrupted     This is set to true if information was uncovered that warrants interrupting a chain of actions (and set to false otherwise).
 816  *
 817  * @returns The number of hexes entered. This can safely be used as an index
 818  *          into steps to get the location where movement ended, provided
 819  *          steps is not empty (the return value is guaranteed to be less
 820  *          than steps.size() ).
 821  */
 822 size_t mouse_handler::move_unit_along_route(const std::vector<map_location> & steps,
 823                                             bool & interrupted)
 824 {
 825         if(steps.empty()) {
 826                 interrupted = false;
 827                 return 0;
 828         }
 829 
 830         // Default return value.
 831         interrupted = true;
 832 
 833         //If this is a leader on a keep, ask permission to the whiteboard to move it
 834         //since otherwise it may cause planned recruits to be erased.
 835         if ( pc_.get_map_const().is_keep(steps.front()) )
 836         {
 837                 unit_map::const_iterator const u = pc_.gamestate().board_.units().find(steps.front());
 838 
 839                 if ( u &&  u->can_recruit()  &&
 840                      u->side() == gui().viewing_side()        &&
 841                      !pc_.get_whiteboard()->allow_leader_to_move(*u) )
 842                 {
 843                         gui2::show_transient_message(gui_->video(), "",
 844                                 _("You cannot move your leader away from the keep with some planned recruits or recalls left."));
 845                         return 0;
 846                 }
 847         }
 848 
 849         LOG_NG << "move unit along route  from " << steps.front() << " to " << steps.back() << "\n";
 850         size_t moves = actions::move_unit_and_record(steps, &pc_.get_undo_stack(), false, true, &interrupted);
 851 
 852         cursor::set(cursor::NORMAL);
 853         gui().invalidate_game_status();
 854 
 855         if(moves == 0)
 856                 return 0;
 857 
 858         if ( interrupted  &&  moves + 1 < steps.size() ) {
 859                 // reselect the unit (for "press t to continue")
 860                 select_hex(steps[moves], false, false, false);
 861                 // the new discovery is more important than the new movement range
 862                 show_partial_move_ = true;
 863         }
 864 
 865         return moves;
 866 }

 899 int mouse_handler::fill_weapon_choices(std::vector<battle_context>& bc_vector, unit_map::iterator attacker, unit_map::iterator defender)
 900 {
 901         int best = 0;
 902         for (unsigned int i = 0; i < attacker->attacks().size(); i++) {
 903                 // skip weapons with attack_weight=0
 904                 if (attacker->attacks()[i].attack_weight() > 0) {
 905                         battle_context bc(pc_.gamestate().board_.units_, attacker->get_location(), defender->get_location(), i);
 906                         if (!bc_vector.empty() && bc.better_attack(bc_vector[best], 0.5)) {
 907                                 // as some weapons can be hidden, i is not a valid index into the resulting vector
 908                                 best = bc_vector.size();
 909                         }
 910                         bc_vector.push_back(bc);
 911                 }
 912         }
 913         return best;
 914 }

 942                           attacker
 943                         , defender

1036 /**
1037  * Causes attackable hexes to be highlighted.
1038  *
1039  * This checks the hexes that the provided unit can attack. If there is a valid
1040  * target there, that location is inserted into current_paths_.destinations.
1041  */
1042 void mouse_handler::show_attack_options(const unit_map::const_iterator &u)
1043 {
1044         // Cannot attack if no attacks are left.
1045         if (u->attacks_left() == 0)
1046               return;
1047 
1048         // Get the teams involved.
1049         const team & cur_team = current_team();
1050         const team & u_team = pc_.gamestate().board_.teams_[u->side()-1];
1051 
1052         // Check each adjacent hex.
1053         map_location adj[6];
1054         get_adjacent_tiles(u->get_location(), adj);
1055         for(const map_location &loc : adj)
1056         {
1057                 // No attack option shown if no visible unit present.
1058                 // (Visible to current team, not necessarily the unit's team.)
1059                 if (!pc_.get_map_const().on_board(loc)) continue;
1060                 unit_map::const_iterator i = pc_.gamestate().board_.units().find(loc);
1061                 if ( !i ||  !i->is_visible_to_team(cur_team, gui_->get_disp_context(), false) )
1062                         continue;
1063                 const unit &target = *i;
1064                 // Can only attack non-petrified enemies.
1065                 if ( u_team.is_enemy(target.side())  &&  !target.incapacitated() )
1066                         current_paths_.destinations.insert(loc);
1067         }
1068 }

1070 bool mouse_handler::unit_in_cycle(unit_map::const_iterator it)
1071 {
1072         game_board & board = pc_.gamestate().board_;
1073 
1074         if (!it) {
1075                 return false;
1076         }
1077 
1078         if (it->side() != side_num_ || it->user_end_turn()
1079             || gui().fogged(it->get_location()) || !board.unit_can_move(*it))
1080                 return false;
1081 
1082         if (current_team().is_enemy(int(gui().viewing_team()+1)) &&
1083             it->invisible(it->get_location(), gui().get_disp_context()))
1084                 return false;
1085 
1086         if (it->get_hidden())
1087                 return false;
1088 
1089         return true;
1090

1093 void mouse_handler::cycle_units(const bool browse, const bool reverse)
1094 {
1095         game_board & board = pc_.gamestate().board_;
1096 
1097         if (board.units().begin() == board.units().end()) {
1098                 return;
1099         }
1100 
1101         unit_map::const_iterator it = find_unit(next_unit_);
1102         if (!it)
1103                 it = board.units().begin();
1104         const unit_map::const_iterator itx = it;
1105 
1106         do {
1107                 if (reverse) {
1108                         if (it == board.units().begin())
1109                                 it = board.units().end();
1110                         --it;
1111                 } else {
1112                         if (it == board.units().end())
1113                                 it = board.units().begin();
1114                         else
1115                                 ++it;
1116                 }
1117         } while (it != itx && !unit_in_cycle(it));
1118 
1119         if (unit_in_cycle(it)) {
1120                 gui().scroll_to_tile(it->get_location(), game_display::WARP);
1121                 select_hex(it->get_location(), browse);
1122 //              mouse_update(browse);
1123         }
1124 }
