  38 SDL_Rect panel::get_client_rect() const
  39 {
  40         std::shared_ptr<const panel_definition::resolution> conf
  41                         = std::static_pointer_cast<const panel_definition::resolution>(
  42                                         config());
  43         assert(conf);
  44 
  45         SDL_Rect result = get_rectangle();
  46         result.x += conf->left_border;
  47         result.y += conf->top_border;
  48         result.w -= conf->left_border + conf->right_border;
  49         result.h -= conf->top_border + conf->bottom_border;
  50 
  51         return result;
  52 }
