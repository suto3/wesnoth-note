  52 void faction_select::pre_show(window& window)
  53 {
  54         window.set_escape_disabled(true);
  55 
  56         find_widget<label>(&window, "starting_pos", false).set_label(std::to_string(side_));
  57 
  58         //
  59         // Set up gender radio buttons
  60         //
  61         toggle_button& gender_rand = find_widget<toggle_button>(&window, "gender_random", false);
  62         toggle_button& gender_male = find_widget<toggle_button>(&window, "gender_male", false);
  63         toggle_button& gender_female = find_widget<toggle_button>(&window, "gender_female", false);
  64 
  65         gender_toggle_.add_member(&gender_rand,   "random");
  66         gender_toggle_.add_member(&gender_male,   unit_race::s_male);
  67         gender_toggle_.add_member(&gender_female, unit_race::s_female);
  68 
  69         gender_toggle_.set_member_states("random");
  70 
  71         gender_toggle_.set_callback_on_value_change(
  72                         dialog_callback<faction_select, &faction_select::on_gender_select>);
  73 
  74         //
  75         // Set up leader menu button
  76         //
  77         find_widget<menu_button>(&window, "leader_menu", false).connect_click_handler(
  78                 std::bind(&faction_select::on_leader_select, this, std::ref(window)));
  79 
  80         //
  81         // Set up faction list
  82         //
  83         listbox& list = find_widget<listbox>(&window, "faction_list", false);
  84 
  85         window.keyboard_capture(&list);
  86 
  87 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  88         connect_signal_notify_modified(*list,
  89            std::bind(&faction_select::on_faction_select,
  90                         *this, std::ref(window)));
  91 #else
  92         list.set_callback_value_change(
  93                 dialog_callback<faction_select, &faction_select::on_faction_select>);
  94 #endif
  95 
  96         for(const config *s : flg_manager_.choosable_factions()) {
  97                 const config& side = *s;
  98 
  99                 std::map<std::string, string_map> data;
 100                 string_map item;
 101 
 102                 // flag_rgb here is unrelated to any handling in the unit class
 103                 const std::string flag_rgb = !side["flag_rgb"].empty() ? side["flag_rgb"].str() : "magenta";
 104 
 105                 item["label"] = (formatter() << side["image"] << "~RC(" << flag_rgb << ">" << tc_color_ << ")").str();
 106                 data.emplace("faction_image", item);
 107 
 108                 item["label"] = side["name"];
 109                 data.emplace("faction_name", item);
 110 
 111                 list.add_row(data);
 112         }
 113 
 114         list.select_row(flg_manager_.current_faction_index());
 115 
 116         on_faction_select(window);
 117 }

 119 void faction_select::on_faction_select(window& window)
 120 {
 121         const int selected_row = find_widget<listbox>(&window, "faction_list", false).get_selected_row();
 122 
 123         if(selected_row == -1) {
 124                 return;
 125         }
 126 
 127         // Since set_current_faction overrides the current leader, save a copy of the previous leader index so the
 128         // leader dropdown can be set to the appropriate initial selection.
 129         const int previous_leader_selection = flg_manager_.current_leader_index();
 130 
 131         flg_manager_.set_current_faction(selected_row);
 132 
 133         std::vector<config> leaders;
 134 
 135         for(const std::string& leader : flg_manager_.choosable_leaders()) {
 136                 const unit_type* unit = unit_types.find(leader);
 137 
 138                 if(unit) {
 139                         const std::string icon = formatter() << unit->image() << "~RC(" << unit->flag_rgb() << ">" << tc_color_ << ")";
 140                         leaders.emplace_back(config {"label", unit->type_name(), "icon", icon});
 141                 } else if(leader == "random") {
 142                         leaders.emplace_back(config {"label", _("Random"), "icon", ng::random_enemy_picture});
 143                 } else if(leader == "null") {
 144                         leaders.emplace_back(config {"label", font::unicode_em_dash});
 145                 } else {
 146                         leaders.emplace_back(config {"label", "?"});
 147                 }
 148         }
 149 
 150         menu_button& leader_dropdown = find_widget<menu_button>(&window, "leader_menu", false);
 151 
 152         leader_dropdown.set_values(leaders, std::min<int>(leaders.size() - 1, previous_leader_selection));
 153         leader_dropdown.set_active(leaders.size() > 1 && !flg_manager_.is_saved_game());
 154 
 155         on_leader_select(window);
 156 
 157         // Print recruits
 158         const std::vector<std::string> recruit_list = utils::split(flg_manager_.current_faction()["recruit"]);
 159         std::vector<t_string> recruit_names;
 160 
 161         for(const auto& recruit : recruit_list) {
 162                 if(const unit_type* rt = unit_types.find(recruit)) {
 163                         recruit_names.push_back(font::unicode_bullet + " " + rt->type_name());
 164                 }
 165         }
 166 
 167         std::sort(recruit_names.begin(), recruit_names.end(), [](const std::string& s1, const std::string& s2) {
 168                 return translation::compare(s1, s2) < 0;
 169         });
 170 
 171         find_widget<styled_widget>(&window, "recruits", false).set_label(utils::join(recruit_names, "\n"));
 172 }

 174 void faction_select::on_leader_select(window& window)
 175 {
 176         flg_manager_.set_current_leader(find_widget<menu_button>(&window, "leader_menu", false).get_value());
 177 
 178 
 179         // TODO: should we decouple this from the flg manager and instead just check the unit type directly?
 180         // If that's done so, we'd need to come up with a different check for Random availability.
 181         gender_toggle_.set_members_enabled([this](const std::string& gender)->bool {
 182                 const std::vector<std::string>& genders = flg_manager_.choosable_genders();
 183                 return std::find(genders.begin(), genders.end(), gender) != genders.end();
 184         });
 185 
 186         update_leader_image(window);
 187 }

 196 void faction_select::update_leader_image(window& window)
 197 {
 198         std::string leader_image = ng::random_enemy_picture;
 199 
 200         if(const unit_type* ut = unit_types.find(flg_manager_.current_leader())) {
 201                 const unit_type& utg = ut->get_gender_unit_type(flg_manager_.current_gender());
 202                 leader_image = formatter() << utg.image() << "~RC(" << utg.flag_rgb() << ">" << tc_color_ << ")" << "~SCALE_INTO_SHARP(144,144)";
 203         }
 204 
 205         find_widget<image>(&window, "leader_image", false).set_label(leader_image);
 206 }
