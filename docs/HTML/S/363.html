  53 void faction_select::pre_show(window& window)
  54 {
  55         window.set_escape_disabled(true);
  56 
  57         find_widget<label>(&window, "starting_pos", false).set_label(std::to_string(side_));
  58 
  59         //
  60         // Set up gender radio buttons
  61         //
  62         toggle_button& gender_rand = find_widget<toggle_button>(&window, "gender_random", false);
  63         toggle_button& gender_male = find_widget<toggle_button>(&window, "gender_male", false);
  64         toggle_button& gender_female = find_widget<toggle_button>(&window, "gender_female", false);
  65 
  66         gender_toggle_.add_member(&gender_rand,   "random");
  67         gender_toggle_.add_member(&gender_male,   unit_race::s_male);
  68         gender_toggle_.add_member(&gender_female, unit_race::s_female);
  69 
  70         gender_toggle_.set_member_states("random");
  71 
  72         gender_toggle_.set_callback_on_value_change(
  73                         dialog_callback<faction_select, &faction_select::on_gender_select>);
  74 
  75         //
  76         // Set up leader menu button
  77         //
  78         find_widget<menu_button>(&window, "leader_menu", false).connect_click_handler(
  79                 std::bind(&faction_select::on_leader_select, this, std::ref(window)));
  80 
  81         //
  82         // Set up faction list
  83         //
  84         listbox& list = find_widget<listbox>(&window, "faction_list", false);
  85 
  86         window.keyboard_capture(&list);
  87 
  88 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  89         connect_signal_notify_modified(*list,
  90            std::bind(&faction_select::on_faction_select,
  91                         *this, std::ref(window)));
  92 #else
  93         list.set_callback_value_change(
  94                 dialog_callback<faction_select, &faction_select::on_faction_select>);
  95 #endif
  96 
  97         for(const config *s : flg_manager_.choosable_factions()) {
  98                 const config& side = *s;
  99 
 100                 std::map<std::string, string_map> data;
 101                 string_map item;
 102 
 103                 // flag_rgb here is unrelated to any handling in the unit class
 104                 const std::string flag_rgb = !side["flag_rgb"].empty() ? side["flag_rgb"].str() : "magenta";
 105 
 106                 item["label"] = (formatter() << side["image"] << "~RC(" << flag_rgb << ">" << tc_color_ << ")").str();
 107                 data.emplace("faction_image", item);
 108 
 109                 item["label"] = side["name"];
 110                 data.emplace("faction_name", item);
 111 
 112                 list.add_row(data);
 113         }
 114 
 115         list.select_row(flg_manager_.current_faction_index());
 116 
 117         on_faction_select(window);

 120 void faction_select::on_faction_select(window& window)
 121 {
 122         const int selected_row = find_widget<listbox>(&window, "faction_list", false).get_selected_row();
 123 
 124         if(selected_row == -1) {
 125                 return;
 126         }
 127 
 128         // Since set_current_faction overrides the current leader, save a copy of the previous leader index so the
 129         // leader dropdown can be set to the appropriate initial selection.
 130         const int previous_leader_selection = flg_manager_.current_leader_index();
 131 
 132         flg_manager_.set_current_faction(selected_row);
 133 
 134         std::vector<config> leaders;
 135 
 136         for(const std::string& leader : flg_manager_.choosable_leaders()) {
 137                 const unit_type* unit = unit_types.find(leader);
 138 
 139                 if(unit) {
 140                         const std::string icon = formatter() << unit->image() << "~RC(" << unit->flag_rgb() << ">" << tc_color_ << ")";
 141                         leaders.emplace_back(config_of("label", unit->type_name())("icon", icon));
 142                 } else if(leader == "random") {
 143                         leaders.emplace_back(config_of("label", _("Random"))("icon", ng::random_enemy_picture));
 144                 } else if(leader == "null") {
 145                         leaders.emplace_back(config_of("label", font::unicode_em_dash));
 146                 } else {
 147                         leaders.emplace_back(config_of("label", "?"));
 148                 }
 149         }
 150 
 151         menu_button& leader_dropdown = find_widget<menu_button>(&window, "leader_menu", false);
 152 
 153         leader_dropdown.set_values(leaders, std::min<int>(leaders.size() - 1, previous_leader_selection));
 154         leader_dropdown.set_active(leaders.size() > 1 && !flg_manager_.is_saved_game());
 155 
 156         on_leader_select(window);
 157 
 158         // Print recruits
 159         const std::vector<std::string> recruit_list = utils::split(flg_manager_.current_faction()["recruit"]);
 160         std::vector<t_string> recruit_names;
 161 
 162         for(const auto& recruit : recruit_list) {
 163                 if(const unit_type* rt = unit_types.find(recruit)) {
 164                         recruit_names.push_back(font::unicode_bullet + " " + rt->type_name());
 165                 }
 166         }
 167 
 168         std::sort(recruit_names.begin(), recruit_names.end(), [](const std::string& s1, const std::string& s2) {
 169                 return translation::compare(s1, s2) < 0;
 170         });
 171 
 172         find_widget<styled_widget>(&window, "recruits", false).set_label(utils::join(recruit_names, "\n"));
 173 }

 175 void faction_select::on_leader_select(window& window)
 176 {
 177         flg_manager_.set_current_leader(find_widget<menu_button>(&window, "leader_menu", false).get_value());
 178 
 179 
 180         // TODO: should we decouple this from the flg manager and instead just check the unit type directly?
 181         // If that's done so, we'd need to come up with a different check for Random availability.
 182         gender_toggle_.set_members_enabled([this](const std::string& gender)->bool {
 183                 const std::vector<std::string>& genders = flg_manager_.choosable_genders();
 184                 return std::find(genders.begin(), genders.end(), gender) != genders.end();
 185         });
 186 
 187         update_leader_image(window);
 188 }

 197 void faction_select::update_leader_image(window& window)
 198 {
 199         std::string leader_image = ng::random_enemy_picture;
 200 
 201         if(const unit_type* ut = unit_types.find(flg_manager_.current_leader())) {
 202                 const unit_type& utg = ut->get_gender_unit_type(flg_manager_.current_gender());
 203                 leader_image = formatter() << utg.image() << "~RC(" << utg.flag_rgb() << ">" << tc_color_ << ")" << "~SCALE_INTO_SHARP(144,144)";
 204         }
 205 
 206         find_widget<image>(&window, "leader_image", false).set_label(leader_image);
 207 }
