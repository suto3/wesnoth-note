 140     private void updatePageIsComplete( )
 141     {
 142         IResource container = ResourcesPlugin.getWorkspace( ).getRoot( )
 143             .findMember( new Path( getDirectoryName( ) ) );
 144         setPageComplete( false );
 145         String fileName = getFileName( );
 146 
 147         if( getDirectoryName( ).isEmpty( ) ) {
 148             setErrorMessage( Messages.EraPage0_10 );
 149             return;
 150         }
 151 
 152         if( container == null || ! container.exists( )
 153             || ! ( container instanceof IContainer ) ) {
 154             setErrorMessage( Messages.EraPage0_11 );
 155             return;
 156         }
 157 
 158         if( fileName.isEmpty( ) ) {
 159             setErrorMessage( Messages.EraPage0_12 );
 160             return;
 161         }
 162 
 163         if( fileName.replace( '\\', '/' ).indexOf( '/', 1 ) > 0 ) {
 164             setErrorMessage( Messages.EraPage0_13 );
 165             return;
 166         }
 167 
 168         int dotLoc = fileName.lastIndexOf( '.' );
 169         if( dotLoc == - 1
 170             || fileName.substring( dotLoc + 1 ).equalsIgnoreCase( "cfg" ) == false ) //$NON-NLS-1$
 171         {
 172             setErrorMessage( Messages.EraPage0_15 );
 173             return;
 174         }
 175 
 176         if( getEraID( ).isEmpty( ) ) {
 177             setErrorMessage( Messages.EraPage0_16 );
 178             return;
 179         }
 180 
 181         if( getEraName( ).isEmpty( ) ) {
 182             setErrorMessage( Messages.EraPage0_17 );
 183             return;
 184         }
 185 
 186         setErrorMessage( null );
 187         setPageComplete( true );
 188     }
