  33 function ca_herding_attack_close_enemy:evaluation(cfg)
  34     -- Any enemy within attention_distance (default = 8) hexes of a sheep will get the dogs' attention
  35     -- with enemies within attack_distance (default: 4) being attacked
  36     if not get_sheep(cfg)[1] then return 0 end
  37     if not get_dogs(cfg)[1] then return 0 end
  38 
  39     local radius = cfg.attention_distance or 8
  40     if get_enemies(cfg, radius)[1] then return cfg.ca_score end
  41     return 0
  42 end

  44 function ca_herding_attack_close_enemy:execution(cfg)
  45     local sheep = get_sheep(cfg)
  46     local dogs = get_dogs(cfg)
  47 
  48     -- We start with enemies within attack_distance (default: 4) hexes, which will be attacked
  49     local radius = cfg.attack_distance or 4
  50     local enemies = get_enemies(cfg, radius)
  51 
  52     max_rating, best_dog, best_enemy, best_hex = -9e99
  53     for _,enemy in ipairs(enemies) do
  54         for _,dog in ipairs(dogs) do
  55             local reach_map = AH.get_reachable_unocc(dog)
  56             reach_map:iter( function(x, y, v)
  57                 -- Most important: distance from enemy
  58                 local rating = -M.distance_between(x, y, enemy.x, enemy.y) * 100.
  59                 -- 2nd: distance from any sheep
  60                 for _,single_sheep in ipairs(sheep) do
  61                     rating = rating - M.distance_between(x, y, single_sheep.x, single_sheep.y)
  62                 end
  63                 -- 3rd: most distant dog goes first
  64                 rating = rating + M.distance_between(enemy.x, enemy.y, dog.x, dog.y) / 100.
  65                 reach_map:insert(x, y, rating)
  66 
  67                 if (rating > max_rating) then
  68                     max_rating = rating
  69                     best_dog, best_enemy, best_hex = dog, enemy, { x, y }
  70                 end
  71             end)
  72         end
  73     end
  74 
  75     -- If we found a move, we do it, and attack if possible
  76     if best_dog then
  77         AH.robust_move_and_attack(ai, best_dog, best_hex, best_enemy)
  78         return
  79     end
  80 
  81     -- If we got here, no enemies to attack where found, so we go on to block other enemies
  82     local radius = cfg.attention_distance or 8
  83     local enemies = get_enemies(cfg, radius)
  84 
  85     -- We also need to remove dogs that have no moves left, since selection was done on attacks_left
  86     for i=#dogs,1,-1 do
  87         if (dogs[i].moves == 0) then
  88             table.remove(dogs, i)
  89         end
  90     end
  91     if (not dogs[1]) then return end
  92 
  93     -- Find closest sheep/enemy pair first
  94     local min_dist, closest_sheep, closest_enemy = 9e99
  95     for _,enemy in ipairs(enemies) do
  96         for _,single_sheep in ipairs(sheep) do
  97             local dist = M.distance_between(enemy.x, enemy.y, single_sheep.x, single_sheep.y)
  98             if dist < min_dist then
  99                 min_dist = dist
 100                 closest_sheep, closest_enemy = single_sheep, enemy
 101             end
 102         end
 103     end
 104 
 105     -- Move dogs in between enemies and sheep
 106     max_rating, best_dog, best_hex = -9e99
 107     for _,dog in ipairs(dogs) do
 108         local reach_map = AH.get_reachable_unocc(dog)
 109         reach_map:iter( function(x, y, v)
 110             -- We want equal distance between enemy and closest sheep
 111             local rating = - math.abs(
 112                 M.distance_between(x, y, closest_sheep.x, closest_sheep.y)
 113                 - M.distance_between(x, y, closest_enemy.x, closest_enemy.y)
 114             ) * 100
 115             -- 2nd: closeness to sheep
 116             rating = rating - M.distance_between(x, y, closest_sheep.x, closest_sheep.y)
 117             reach_map:insert(x, y, rating)
 118             -- 3rd: most distant dog goes first
 119             rating = rating + M.distance_between(closest_enemy.x, closest_enemy.y, dog.x, dog.y) / 100.
 120             reach_map:insert(x, y, rating)
 121 
 122             if (rating > max_rating) then
 123                 max_rating, best_hex, best_dog = rating, { x, y }, dog
 124             end
 125         end)
 126     end
 127 
 128     AH.movefull_stopunit(ai, best_dog, best_hex)
 129 end
