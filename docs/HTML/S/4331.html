  21 function ca_big_animals:execution(cfg)
  22     -- Big animals just move toward a goal that gets (re)set occasionally
  23     -- and attack whatever is in their range (except for some units that they avoid)
  24 
  25     local unit = get_big_animals(cfg)[1]
  26 
  27     local avoid_tag = H.get_child(cfg, "avoid_unit")
  28     local avoid_map = LS.create()
  29     if avoid_tag then
  30         local enemies_to_be_avoided = AH.get_attackable_enemies(avoid_tag)
  31         for _,enemy in ipairs(enemies_to_be_avoided) do
  32             avoid_map:insert(enemy.x, enemy.y)
  33             for xa,ya in H.adjacent_tiles(enemy.x, enemy.y) do
  34                 avoid_map:insert(xa, ya)
  35             end
  36         end
  37     end
  38 
  39     local goal = MAIUV.get_mai_unit_variables(unit, cfg.ai_id)
  40 
  41     -- Unit gets a new goal if none is set or on any move with a 10% random chance
  42     local r = math.random(10)
  43     if (not goal.goal_x) or (r == 1) then
  44         local locs = AH.get_passable_locations(H.get_child(cfg, "filter_location") or {})
  45         local rand = math.random(#locs)
  46 
  47         goal.goal_x, goal.goal_y = locs[rand][1], locs[rand][2]
  48         MAIUV.set_mai_unit_variables(unit, cfg.ai_id, goal)
  49     end
  50 
  51     local reach_map = AH.get_reachable_unocc(unit)
  52     local wander_terrain = H.get_child(cfg, "filter_location_wander") or {}
  53     reach_map:iter( function(x, y, v)
  54         -- Remove tiles that do not comform to the wander terrain filter
  55         if (not wesnoth.match_location(x, y, wander_terrain)) then
  56             reach_map:remove(x, y)
  57         end
  58     end)
  59 
  60     -- Now find the one of these hexes that is closest to the goal
  61     local max_rating, best_hex = -9e99
  62     reach_map:iter( function(x, y, v)
  63         local rating = -wesnoth.map.distance_between(x, y, goal.goal_x, goal.goal_y)
  64 
  65         -- Proximity to an enemy unit is a plus
  66         local enemy_hp = 500
  67         for xa,ya in H.adjacent_tiles(x, y) do
  68             local enemy = wesnoth.get_unit(xa, ya)
  69             if AH.is_attackable_enemy(enemy) then
  70                 if (enemy.hitpoints < enemy_hp) then enemy_hp = enemy.hitpoints end
  71             end
  72         end
  73         rating = rating + 500 - enemy_hp  -- Prefer attack on weakest enemy
  74 
  75         -- Hexes reachable by units to be be avoided get a massive negative hit
  76         if avoid_map:get(x, y) then rating = rating - 1000 end
  77 
  78         if (rating > max_rating) then
  79             max_rating, best_hex = rating, { x, y }
  80         end
  81     end)
  82 
  83     if (best_hex[1] ~= unit.x) or (best_hex[2] ~= unit.y) then
  84         AH.checked_move(ai, unit, best_hex[1], best_hex[2])  -- Partial move only
  85         if (not unit) or (not unit.valid) then return end
  86     else  -- If unit did not move, we need to stop it (also delete the goal)
  87         AH.checked_stopunit_moves(ai, unit)
  88         if (not unit) or (not unit.valid) then return end
  89         MAIUV.delete_mai_unit_variables(unit, cfg.ai_id)
  90     end
  91 
  92     -- If this gets the unit to the goal, we also delete the goal
  93     if (unit.x == goal.goal_x) and (unit.y == goal.goal_y) then
  94         MAIUV.delete_mai_unit_variables(unit, cfg.ai_id)
  95     end
  96 
  97     -- Finally, if the unit ended up next to enemies, attack the weakest of those
  98     local min_hp, target = 9e99
  99     for xa,ya in H.adjacent_tiles(unit.x, unit.y) do
 100         local enemy = wesnoth.get_unit(xa, ya)
 101         if AH.is_attackable_enemy(enemy) then
 102             if (enemy.hitpoints < min_hp) then
 103                 min_hp, target = enemy.hitpoints, enemy
 104             end
 105         end
 106     end
 107 
 108     if target then
 109         AH.checked_attack(ai, unit, target)
 110     end
 111 end
