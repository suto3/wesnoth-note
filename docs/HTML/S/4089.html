  63 def comma_split(csstring, list=None, strip="r"):
  64     "Split a comma-separated string, and append the entries to a list if specified."
  65     vallist = [x.lstrip() for x in csstring.split(",") if x.lstrip()]
  66     # strip=: utils::split will remove trailing whitespace from items in comma-
  67     # separated lists but the wml-tags.lua split function only removes leading
  68     # whitespace. So two flags are offered to change default behavior: one to
  69     # lstrip() only, the other to warn about trailing whitespace.
  70     if 'w' in strip:
  71         for item in vallist:
  72             if re.search('\s$', item):
  73                 print('Trailing whitespace may be problematic: "%s" in "%s"' % (item, csstring))
  74     if 'l' not in strip:
  75         vallist = [x.rstrip() for x in vallist]
  76     if list is not None:
  77         list.extend(vallist)
  78     else:
  79         return vallist

  81 def parse_attribute(line):
  82     "Parse a WML key-value pair from a line."
  83     if '=' not in line or line.find("#") > -1 and line.find("#") < line.find("="):
  84         return None
  85     where = line.find("=")
  86     leader = line[:where]
  87     after = line[where+1:]
  88     after = after.lstrip()
  89     if re.search("\s#", after):
  90         where = len(re.split("\s+#", after)[0])
  91         value = after[:where]
  92         comment = after[where:]
  93     else:
  94         value = after.rstrip()
  95         comment = ""
  96     # Return four fields: stripped key, part of line before value,
  97     # value, trailing whitespace and comment.
  98     return (leader.strip(), leader+"=", string_strip(value), comment)

 215 def parse_macroref(start, line):
 216     brackdepth = parendepth = 0
 217     instring = False
 218     args = []
 219     arg = ""
 220     for i in range(start, len(line)):
 221         if instring:
 222             if line[i] == '"':
 223                 instring = False
 224             arg += line[i]
 225         elif line[i] == '"':
 226             instring = not instring
 227             arg += line[i]
 228         elif line[i] == "{":
 229             if brackdepth > 0:
 230                 arg += line[i]
 231             brackdepth += 1
 232         elif line[i] == "}":
 233             brackdepth -= 1
 234             if brackdepth == 0:
 235                 if not line[i-1].isspace():
 236                     arg = arg.strip()
 237                     if arg.startswith('"') and arg.endswith('"'):
 238                         arg = arg[1:-1].strip()
 239                     args.append(arg)
 240                     arg = ""
 241                 break
 242             else:
 243                 arg += line[i]
 244         elif line[i] == "(":
 245             parendepth += 1
 246         elif line[i] == ")":
 247             parendepth -= 1
 248         elif not line[i-1].isspace() and \
 249              line[i].isspace() and \
 250              brackdepth == 1 and \
 251              parendepth == 0:
 252             arg = arg.strip()
 253             if arg.startswith('"') and arg.endswith('"'):
 254                 arg = arg[1:-1].strip()
 255             args.append(arg)
 256             arg = ""
 257         elif not line[i].isspace() or parendepth > 0:
 258             arg += line[i]
 259     return (args, brackdepth, parendepth)

 310 def actualtype(a):
 311     if a is None:
 312         return None
 313     # Deduce the type of the actual
 314     if a.isdigit() or a.startswith("-") and a[1:].isdigit():
 315         atype = "numeric"
 316     elif re.match(r"0\.[0-9]+\Z", a):
 317         atype = "percentage"
 318     elif re.match(r"-?[0-9]+,-?[0-9]+\Z", a):
 319         atype = "position"
 320     elif re.match(r"([0-9]+\-[0-9]+,?|[0-9]+,?)+\Z", a):
 321         atype = "span"
 322     elif a in ("melee", "ranged"):
 323         atype = "range"
 324     elif a in ("lawful", "neutral", "chaotic", "liminal"):
 325         atype = "alignment"
 326     elif a.startswith("{") or a.endswith("}") or a.startswith("$"):
 327         atype = None # Can't tell -- it's a macro expansion
 328     elif re.match(image_reference, a) or a == "unit_image":
 329         atype = "image"
 330     elif re.match(r"(\*|[A-Z][a-z]+)\^([A-Z][a-z\\|/]+\Z)?", a):
 331         atype = "terrain_code"
 332     elif a.endswith(".wav") or a.endswith(".ogg"):
 333         atype = "sound"
 334     elif a.startswith('"') and a.endswith('"') or (a.startswith("_") and a[1] not in string.ascii_lowercase):
 335         atype = "stringliteral"
 336     elif "=" in a:
 337         atype = "filter"
 338     elif re.match(r"[A-Z][a-z][a-z]?\Z", a):
 339         atype = "shortname"
 340     elif a == "":
 341         atype = "empty"
 342     elif not ' ' in a:
 343         atype = "name"
 344     else:
 345         atype = "string"
 346     return atype

 348 def argmatch(formals, actuals):
 349     if len(formals) != len(actuals):
 350         return False
 351     for (f, a) in zip(formals, actuals):
 352         # Here's the compatibility logic.  First, we catch the situations
 353         # in which a more restricted actual type matches a more general
 354         # formal one.  Then we have a fallback rule checking for type
 355         # equality or wildcarding.
 356         ftype = formaltype(f)
 357         atype = actualtype(a)
 358         if ftype == "any":
 359             pass
 360         elif (atype == "numeric" or a == "global") and ftype == "side":
 361             pass
 362         elif atype in ("filter", "empty") and ftype == "wml":
 363             pass
 364         elif atype in ("numeric", "position") and ftype == "span":
 365             pass
 366         elif atype in ("shortname", "name", "empty", "stringliteral") and ftype == "affix":
 367             pass
 368         elif atype in ("shortname", "name", "stringliteral") and ftype == "string":
 369             pass
 370         elif atype in ("shortname", "name", "string", "stringliteral", "empty") and ftype == "optional_string":
 371             pass
 372         elif atype in ("shortname",) and ftype == "terrain_code":
 373             pass
 374         elif atype in ("numeric", "position", "span", "empty") and ftype == "alliance":
 375             pass
 376         elif atype in ("terrain_code", "shortname", "name") and ftype == "terrain_pattern":
 377             pass
 378         elif atype in ("string", "shortname", "name") and ftype == "types":
 379             pass
 380         elif atype in ("numeric", "percentage") and ftype == "percentage":
 381             pass
 382         elif atype == "range" and ftype == "name":
 383             pass
 384         elif atype != ftype and ftype is not None and atype is not None:
 385             return False
 386     return True

 459     def visible_from(self, defn, fn, n):
 460         "Is specified definition visible from the specified file and line?"
 461         if isinstance(defn, basestring):
 462             defn = self.fileref[defn]
 463         if defn.undef is not None:
 464             # Local macros are only visible in the file where they were defined
 465             return defn.filename == fn and n >= defn.lineno and n <= defn.undef
 466         if self.exports(defn.namespace):
 467             # Macros and resources in subtrees with export=yes are global
 468             return True
 469         elif not self.filelist.neighbors(defn.filename, fn):
 470             # Otherwise, must be in the same subtree.
 471             return False
 472         else:
 473             # If the two files are in the same subtree, assume visibility.
 474             # This doesn't match the actual preprocessor semantics.
 475             # It means any macro without an undef is visible anywhere in the
 476             # same argument directory.
 477             #
 478             # We can't do better than this without a lot of hairy graph-
 479             # coloring logic to simulate include path interpretation.
 480             # If that logic ever gets built, it will go here.
 481             return True

 539                 if line.strip().startswith("#define"):
 540                     tokens = line.split()
 541                     if len(tokens) < 2:
 542                         print('"%s", line %d: malformed #define' \
 543                               % (filename, n+1), file=sys.stderr)
 544                     else:
 545                         name = tokens[1]
 546                         here = Reference(namespace, filename, n+1, line, args=tokens[2:])
 547                         here.hash = hashlib.md5()
 548                         here.docstring = line.lstrip()[8:] # Strip off #define_
 549                         state = "macro_header"
 550                     continue
 551                 elif state != 'outside' and line.strip().endswith("#enddef"):
 552                     here.hash.update(line.encode("utf8"))
 553                     here.hash = here.hash.digest()
 554                     if name in self.xref:
 555                         for defn in self.xref[name]:
 556                             if not self.visible_from(defn, filename, n+1):
 557                                 continue
 558                             elif conditionalsflag:
 559                                 continue
 560                             elif defn.hash != here.hash:
 561                                 print("%s: overrides different %s definition at %s" \
 562                                         % (here, name, defn), file=sys.stderr)
 563                             elif self.warnlevel > 0:
 564                                 print("%s: duplicates %s definition at %s" \
 565                                         % (here, name, defn), file=sys.stderr)
 566                     if name not in self.xref:
 567                         self.xref[name] = []
 568                     self.xref[name].append(here)
 569                     state = "outside"
 570                 elif state == "macro_header" and line.strip() and line.strip()[0] != "#":
 571                     state = "macro_body"
 572                 if state == "macro_header":
 573                     # Ignore macro header commends that are pragmas
 574                     if "wmlscope" not in line and "wmllint:" not in line:
 575                         here.docstring += line.lstrip()[1:]
 576                 if state in ("macro_header", "macro_body"):
 577                     here.hash.update(line.encode("utf8"))
 578                 elif line.strip().startswith("#undef"):
 579                     tokens = line.split()
 580                     name = tokens[1]
 581                     if name in self.xref and self.xref[name]:
 582                         self.xref[name][-1].undef = n+1
 583                     else:
 584                         print("%s: unbalanced #undef on %s" \
 585                               % (Reference(namespace, filename, n+1), name))
 586                 if state == 'outside':
 587                     if '[unit_type]' in line:
 588                         latch_unit = True
 589                     elif '[/unit_type]' in line:
 590                         latch_unit = False
 591                     elif '[base_unit]' in line:
 592                         in_base_unit = True
 593                     elif '[/base_unit]' in line:
 594                         in_base_unit = False
 595                     elif '[theme]' in line:
 596                         in_theme = True
 597                     elif '[/theme]' in line:
 598                         in_theme = False
 599                     elif latch_unit and not in_base_unit and not in_theme and "id" in line:
 600                         m = CrossRef.tag_parse.search(line)
 601                         if m and m.group(1) == "id":
 602                             uid = m.group(2)
 603                             if uid not in self.unit_ids:
 604                                 self.unit_ids[uid] = []
 605                             self.unit_ids[uid].append(Reference(namespace, filename, n+1))
 606                             latch_unit= False
 607     def __init__(self, dirpath=[], exclude="", warnlevel=0, progress=False):
 608         "Build cross-reference object from the specified filelist."
 609         self.filelist = Forest(dirpath, exclude)
 610         self.dirpath = [x for x in dirpath if not re.search(exclude, x)]
 611         self.warnlevel = warnlevel
 612         self.xref = {}
 613         self.fileref = {}
 614         self.noxref = False
 615         self.properties = {}
 616         self.unit_ids = {}
 617         all_in = []
 618         if self.warnlevel >=2 or progress:
 619             print("*** Beginning definition-gathering pass...")
 620         for (namespace, filename) in self.filelist.generator():
 621             all_in.append((namespace, filename))
 622             if self.warnlevel > 1:
 623                 print(filename + ":")
 624             if progress:
 625                 print(filename)
 626             if isresource(filename):
 627                 self.fileref[filename] = Reference(namespace, filename)
 628             elif iswml(filename):
 629                 # It's a WML file, scan for macro definitions
 630                 self.scan_for_definitions(namespace, filename)
 631             elif filename.endswith(".def"):
 632                 # It's a list of names to be considered defined
 633                 self.noxref = True
 634                 with codecs.open(filename, "r", "utf8") as dfp:
 635                     for line in dfp:
 636                         self.xref[line.strip()] = True
 637         # Next, decorate definitions with all references from the filelist.
 638         self.unresolved = []
 639         self.missing = []
 640         formals = []
 641         state = "outside"
 642         if self.warnlevel >=2 or progress:
 643             print("*** Beginning reference-gathering pass...")
 644         for (ns, fn) in all_in:
 645             if progress:
 646                 print(filename)
 647             if iswml(fn):

 649                     attack_name = None
 650                     beneath = 0

 659                             if "# wmlscope: start ignoring" in line:
 660                                 if self.warnlevel > 1:
 661                                     print('"%s", line %d: starting ignoring (reference pass)' \
 662                                           % (fn, n+1))
 663                                 ignoreflag = True
 664                             elif "# wmlscope: stop ignoring" in line:
 665                                 if self.warnlevel > 1:
 666                                     print('"%s", line %d: stopping ignoring (reference pass)' \
 667                                           % (fn, n+1))
 668                                 ignoreflag = False
 669                             m = re.search("# *wmlscope: self.warnlevel ([0-9]*)", line)
 670                             if m:
 671                                 self.warnlevel = int(m.group(1))
 672                                 print('"%s", line %d: self.warnlevel set to %d (reference-gathering pass)' \
 673                                      % (fn, n+1, self.warnlevel))
 674                                 continue
 675                             fields = line.split('#')
 676                             line = fields[0]
 677                             if len(fields) > 1:
 678                                 comment = fields[1]
 679                         if ignoreflag or not line:
 680                             continue
 681                         # Find references to macros
 682                         for match in re.finditer(CrossRef.macro_reference, line):
 683                             name = match.group(1)
 684                             candidates = []
 685                             if self.warnlevel >=2:
 686                                 print('"%s", line %d: seeking definition of %s' \
 687                                       % (fn, n+1, name))
 688                             if name in formals:
 689                                 continue
 690                             elif name in self.xref:
 691                                 # Count the number of actual arguments.
 692                                 # Set args to None if the call doesn't
 693                                 # close on this line
 694                                 (args, brackdepth, parendepth) = parse_macroref(match.start(0), line)
 695                                 if brackdepth > 0 or parendepth > 0:
 696                                     args = None
 697                                 else:

 699                                 #if args:

 701                                 #          % (fn, n+1, name, args))
 702                                 # Figure out which macros might resolve this
 703                                 for defn in self.xref[name]:
 704                                     if self.visible_from(defn, fn, n+1):
 705                                         defn.append(fn, n+1, args)
 706                                         candidates.append(str(defn))
 707                                 if len(candidates) > 1:
 708                                     print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns, fn, n), name, "; ".join(candidates)))
 709                             if len(candidates) == 0:
 710                                 self.unresolved.append((name,Reference(ns,fn,n+1)))
 711                         # Don't be fooled by HTML image references in help strings.
 712                         if "<img>" in line:
 713                             continue
 714                         # Find references to resource files
 715                         for match in re.finditer(CrossRef.file_reference, line):
 716                             name = match.group(0)
 717                             # Catches maps that look like macro names.
 718                             if (name.endswith(".map") or name.endswith(".mask")) and name[0] == '{':
 719                                 name = name[1:]
 720                             if os.sep == "\\":
 721                                 name = name.replace("/", "\\")
 722                             key = None
 723                             # If name is already in our resource list, it's easy.
 724                             if name in self.fileref and self.visible_from(name, fn, n):
 725                                 self.fileref[name].append(fn, n+1)
 726                                 continue
 727                             # If the name contains substitutable parts, count
 728                             # it as a reference to everything the substitutions
 729                             # could potentially match.
 730                             elif '{' in name or '@' in name:
 731                                 pattern = re.sub(r"(\{[^}]*\}|@R0|@V)", '.*', name)
 732                                 key = self.mark_matching_resources(pattern, fn,n+1)
 733                                 if key:
 734                                     self.fileref[key].append(fn, n+1)
 735                             else:
 736                                 candidates = []
 737                                 for trial in self.fileref:
 738                                     if trial.endswith(os.sep + name) and self.visible_from(trial, fn, n):
 739                                         key = trial
 740                                         self.fileref[trial].append(fn, n+1)
 741                                         candidates.append(trial)
 742                                 if len(candidates) > 1:
 743                                     print("%s: more than one resource matching %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 744                             if not key:
 745                                 self.missing.append((name, Reference(ns,fn,n+1)))
 746                         # Notice implicit references through attacks
 747                         if state == "outside":
 748                             if "[attack]" in line:
 749                                 beneath = 0
 750                                 attack_name = default_icon = None
 751                                 have_icon = False
 752                             elif "name=" in line and not "no-icon" in comment:
 753                                 attack_name = line[line.find("name=")+5:].strip()
 754                                 default_icon = os.path.join("attacks", attack_name  + ".png")
 755                             elif "icon=" in line and beneath == 0:
 756                                 have_icon = True
 757                             elif "[/attack]" in line:
 758                                 if attack_name and not have_icon:
 759                                     candidates = []
 760                                     key = None
 761                                     for trial in self.fileref:
 762                                         if trial.endswith(os.sep + default_icon) and self.visible_from(trial, fn, n):
 763                                             key = trial
 764                                             self.fileref[trial].append(fn, n+1)
 765                                             candidates.append(trial)
 766                                     if len(candidates) > 1:
 767                                         print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 768                                 if not key:
 769                                     self.missing.append((default_icon, Reference(ns,fn,n+1)))
 770                             elif line.strip().startswith("[/"):
 771                                 beneath -= 1
 772                             elif line.strip().startswith("["):
 773                                 beneath += 1
 774         # Check whether each namespace has a defined export property
 775         for namespace in self.dirpath:
 776             if namespace not in self.properties or "export" not in self.properties[namespace]:
 777                 print("warning: %s has no export property" % namespace)

 814 class Translation(dict):
 815     "Parses a po file to create a translation dictionary."
 816     def __init__(self, textdomain, isocode, topdir=""):
 817         self.textdomain = textdomain
 818         self.isocode = isocode
 819         self.gettext = {}
 820         if self.isocode != "C":
 821             isocode2 = isocode[:isocode.rfind("_")]
 822             for code in [isocode, isocode2]:
 823                 fn = "po/%s/%s.po" % (textdomain, code)
 824                 if topdir: fn = os.path.join(topdir, fn)
 825                 try:
 826                     f = file(fn)
 827                     break
 828                 except IOError:
 829                     pass
 830             else:
 831                 raise TranslationError(textdomain, self.isocode)
 832 
 833             expect = False
 834             fuzzy = "#, fuzzy\n"
 835             gettext = f.read().decode("utf8")
 836             matches = re.compile("""(msgid|msgstr)((\s*".*?")+)""").finditer(gettext)
 837             msgid = ""
 838             for match in matches:
 839                 text = "".join(re.compile('"(.*?)"').findall(match.group(2)))
 840                 if match.group(1) == "msgid":
 841                     msgid = text.replace("\\n", "\n")
 842                     expect = gettext[match.start(1) - len(fuzzy):match.start(1)] != fuzzy
 843                 elif expect:
 844                     self.gettext[msgid] = text.replace("\\n", "\n")
 845     def get(self, key, dflt):
 846         if self.isocode == "C":
 847             if key:
 848                 return key[key.find("^") + 1:]
 849             return "?"
 850         else:
 851             t = self.gettext.get(key, dflt)
 852             if not t:
 853                 if key:
 854                     return key[key.find("^") + 1:]
 855                 return "?"
 856             return t
 857     def __getitem__(self, key):
 858         if self.isocode == "C":
 859             return key
 860         else:
 861             return self.gettext[key]
 862     def __contains__(self, key):
 863         if self.isocode == "C":
 864             return True
 865         else:
 866             return key in self.gettext

 868 class Translations:
 869     "Wraps around Translation to support multiple languages and domains."
 870     def __init__(self, topdir = ""):
 871         self.translations = {}
 872         self.topdir = topdir
 873     def get(self, textdomain, isocode, key, default):
 874         t = (textdomain, isocode)
 875         if not t in self.translations:
 876             try:
 877                 self.translations[t] = Translation(textdomain, isocode, self.topdir)
 878             except TranslationError as e:
 879                 print(str(e), file=sys.stderr)
 880                 self.translations[t] = Translation(textdomain, "C", self.topdir)
 881         result = self.translations[t].get(key, default)
