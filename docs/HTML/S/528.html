  70 /**
  71  * SDL_AddTimer() callback for the draw event.
  72  *
  73  * When this callback is called it pushes a new draw event in the event queue.
  74  *
  75  * @returns                       The new timer interval, 0 to stop.
  76  */
  77 static Uint32 timer_sdl_draw_event(Uint32, void*)
  78 {
  79         // DBG_GUI_E << "Pushing draw event in queue.\n";
  80 
  81         SDL_Event event;
  82         SDL_UserEvent data;
  83 
  84         data.type = DRAW_EVENT;
  85         data.code = 0;
  86         data.data1 = nullptr;
  87         data.data2 = nullptr;
  88 
  89         event.type = DRAW_EVENT;
  90         event.user = data;
  91 
  92         SDL_PushEvent(&event);
  93         return draw_interval;
  94 }

  97  * SDL_AddTimer() callback for the poll event.

 348 void sdl_event_handler::handle_event(const SDL_Event& event)
 349 {
 350         /** No dispatchers drop the event. */
 351         if(dispatchers_.empty()) {
 352                 return;
 353         }
 354 
 355         switch(event.type) {
 356                 case SDL_MOUSEMOTION:
 357                         mouse(SDL_MOUSE_MOTION, {event.motion.x, event.motion.y});
 358                         break;
 359 
 360                 case SDL_MOUSEBUTTONDOWN:
 361                         mouse_button_down({event.button.x, event.button.y}, event.button.button);
 362                         break;
 363 
 364                 case SDL_MOUSEBUTTONUP:
 365                         mouse_button_up({event.button.x, event.button.y}, event.button.button);
 366                         break;
 367 
 368                 case SDL_MOUSEWHEEL:
 369                         mouse_wheel(get_mouse_position(), event.wheel.x, event.wheel.y);
 370                         break;
 371 
 372                 case SHOW_HELPTIP_EVENT:
 373                         mouse(SHOW_HELPTIP, get_mouse_position());
 374                         break;
 375 
 376                 case HOVER_REMOVE_POPUP_EVENT:
 377                         // remove_popup();
 378                         break;
 379 
 380                 case DRAW_EVENT:
 381                         draw(false);
 382                         break;
 383                 case DRAW_ALL_EVENT:
 384                         draw(true);
 385                         break;
 386 
 387                 case TIMER_EVENT:
 388                         execute_timer(reinterpret_cast<size_t>(event.user.data1));
 389                         break;
 390 
 391                 case CLOSE_WINDOW_EVENT:
 392                         close_window(event.user.code);
 393                         break;
 394 
 395                 case SDL_JOYBUTTONDOWN:
 396                         button_down(event);
 397                         break;
 398 
 399                 case SDL_JOYBUTTONUP:
 400                         break;
 401 
 402                 case SDL_JOYAXISMOTION:
 403                         break;
 404 
 405                 case SDL_JOYHATMOTION:
 406                         hat_motion(event);
 407                         break;
 408 
 409                 case SDL_KEYDOWN:
 410                         key_down(event);
 411                         break;
 412 
 413                 case SDL_WINDOWEVENT:
 414                         switch(event.window.event) {
 415                                 case SDL_WINDOWEVENT_EXPOSED:
 416                                         draw(true);
 417                                         break;
 418 
 419                                 case SDL_WINDOWEVENT_RESIZED:
 420                                         video_resize({event.window.data1, event.window.data2});
 421                                         break;
 422 
 423                                 case SDL_WINDOWEVENT_ENTER:
 424                                 case SDL_WINDOWEVENT_FOCUS_GAINED:
 425                                         activate();
 426                                         break;
 427                         }
 428 
 429                         break;
 430 
 431                 case SDL_TEXTINPUT:
 432                         key_down(event);
 433                         break;
 434 
 435                 case SDL_TEXTEDITING:
 436                         text_editing(event.edit.text, event.edit.start, event.edit.length);
 437                         break;
 438 
 439                 case SDL_FINGERMOTION:
 440                         touch_motion(point(event.tfinger.x, event.tfinger.y), point(event.tfinger.dx, event.tfinger.dy));
 441                         break;
 442 
 443 #if(defined(_X11) && !defined(__APPLE__)) || defined(_WIN32)
 444                 case SDL_SYSWMEVENT:
 445                         /* DO NOTHING */
 446                         break;
 447 #endif
 448 
 449                 // Silently ignored events.
 450                 case SDL_KEYUP:
 451                 case DOUBLE_CLICK_EVENT:
 452                 case SDL_FINGERUP:
 453                 case SDL_FINGERDOWN:
 454                         break;
 455 
 456                 default:
 457 #ifdef GUI2_SHOW_UNHANDLED_EVENT_WARNINGS
 458                         WRN_GUI_E << "Unhandled event " << static_cast<Uint32>(event.type)
 459                                   << ".\n";
 460 #endif
 461                         break;
 462         }
 463 
 464         raw_event(event);
 465 }

 472 void sdl_event_handler::connect(dispatcher* dispatcher)
 473 {
 474         assert(std::find(dispatchers_.begin(), dispatchers_.end(), dispatcher)
 475                    == dispatchers_.end());
 476 
 477         if(dispatchers_.empty()) {
 478                 event_context = new events::event_context();
 479                 join();
 480         }
 481 
 482         dispatchers_.push_back(dispatcher);
 483 }
 484 
 485 void sdl_event_handler::disconnect(dispatcher* disp)
 486 {
 487         /***** Validate pre conditions. *****/
 488         auto itor = std::find(dispatchers_.begin(), dispatchers_.end(), disp);
 489         assert(itor != dispatchers_.end());
 490 
 491         /***** Remove dispatcher. *****/
 492         dispatchers_.erase(itor);
 493 
 494         if(disp == mouse_focus) {
 495                 mouse_focus = nullptr;
 496         }
 497         if(disp == keyboard_focus_) {
 498                 keyboard_focus_ = nullptr;
 499         }
 500 
 501         /***** Set proper state for the other dispatchers. *****/
 502         for(auto d : dispatchers_)
 503         {
 504                 dynamic_cast<widget&>(*d).set_is_dirty(true);
 505         }
 506 
 507         activate();
 508 
 509         /***** Validate post conditions. *****/
 510         assert(std::find(dispatchers_.begin(), dispatchers_.end(), disp)
 511                    == dispatchers_.end());
 512 
 513         if(dispatchers_.empty()) {
 514                 leave();
 515                 delete event_context;
 516                 event_context = nullptr;
 517         }
 518 }

 528 void sdl_event_handler::draw(const bool force)
 529 {
 530         // Don't display this event since it floods the screen
 531         // DBG_GUI_E << "Firing " << DRAW << ".\n";
 532 
 533         /*
 534          * In normal draw mode the first window in not forced to be drawn the
 535          * others are. So for forced mode we only need to force the first window to
 536          * be drawn the others are already handled.
 537          */
 538         bool first = !force;
 539 
 540         /**
 541          * @todo Need to evaluate which windows really to redraw.
 542          *
 543          * For now we use a hack, but would be nice to rewrite it for 1.9/1.11.
 544          */
 545         for(auto dispatcher : dispatchers_)
 546         {
 547                 if(!first) {
 548                         /*
 549                          * This leaves glitches on window borders if the window beneath it
 550                          * has changed, on the other hand invalidating twindown::restorer_
 551                          * causes black borders around the window. So there's the choice
 552                          * between two evils.
 553                          */
 554                         dynamic_cast<widget&>(*dispatcher).set_is_dirty(true);
 555                 } else {
 556                         first = false;
 557                 }
 558 
 559                 dispatcher->fire(DRAW, dynamic_cast<widget&>(*dispatcher));
 560         }
 561 
 562         if(!dispatchers_.empty()) {
 563                 CVideo& video = dynamic_cast<window&>(*dispatchers_.back()).video();
 564 
 565                 video.flip();
 566         }
 567 }

 588 void sdl_event_handler::mouse(const ui_event event, const point& position)
 589 {
 590         DBG_GUI_E << "Firing: " << event << ".\n";
 591 
 592         if(mouse_focus) {
 593                 mouse_focus->fire(event, dynamic_cast<widget&>(*mouse_focus), position);
 594                 return;
 595         }
 596 
 597         for(auto& dispatcher : boost::adaptors::reverse(dispatchers_)) {
 598                 if(dispatcher->get_mouse_behavior() == dispatcher::all) {
 599                         dispatcher->fire(event, dynamic_cast<widget&>(*dispatcher), position);
 600                         break;
 601                 }
 602 
 603                 if(dispatcher->get_mouse_behavior() == dispatcher::none) {
 604                         continue;
 605                 }
 606 
 607                 if(dispatcher->is_at(position)) {
 608                         dispatcher->fire(event, dynamic_cast<widget&>(*dispatcher), position);
 609                         break;
 610                 }
 611         }
 612 }
