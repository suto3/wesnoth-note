  43 /**
  44  * The event send to the dispatcher.
  45  *
  46  * Events prefixed by SDL are (semi)-real SDL events. The handler does some
  47  * minor decoding like splitting the button down event to the proper event but
  48  * nothing more. Events without an SDL prefix are generated by another signal
  49  * eg the windows signal handler for SDL_MOUSE_MOTION can generate a
  50  * MOUSE_ENTER, MOUSE_MOTION and MOUSE_LEAVE event and send that to it's
  51  * children.
  52  *
  53  * @note When adding a new entry to the enum also add a unit test.
  54  */
  55 enum ui_event {
  56         SDL_ACTIVATE,                   /**< The main application window is activated. */
  57         DRAW,                           /**< Periodic redraw request. */
  58         CLOSE_WINDOW,                   /**< A request to close the current window. */
  59         SDL_VIDEO_RESIZE,               /**< An SDL resize request, coordinate is the new window size. */
  60 
  61         SDL_MOUSE_MOTION,               /**< An SDL mouse motion event. */
  62         MOUSE_ENTER,                    /**< A mouse enter event for a widget. */
  63         MOUSE_MOTION,                   /**< A mouse motion event for a widget. */
  64         MOUSE_LEAVE,                    /**< A mouse leave event for a widget. */
  65 
  66         SDL_LEFT_BUTTON_DOWN,           /**< An SDL left mouse button down event. */
  67         SDL_LEFT_BUTTON_UP,             /**< An SDL left mouse button up event. */
  68         LEFT_BUTTON_DOWN,               /**< A left mouse button down event for a widget. */
  69         LEFT_BUTTON_UP,                 /**< A left mouse button up event for a widget. */
  70         LEFT_BUTTON_CLICK,              /**< A left mouse button click event for a widget. */
  71         LEFT_BUTTON_DOUBLE_CLICK,       /**< A left mouse button double click event for a widget. */
  72 
  73         SDL_MIDDLE_BUTTON_DOWN,         /**< An SDL middle mouse button down event. */
  74         SDL_MIDDLE_BUTTON_UP,           /**< An SDL middle mouse button up event. */
  75         MIDDLE_BUTTON_DOWN,             /**< See LEFT_BUTTON_DOWN. */
  76         MIDDLE_BUTTON_UP,               /**< See LEFT_BUTTON_UP. */
  77         MIDDLE_BUTTON_CLICK,            /**< See LEFT_BUTTON_CLICK. */
  78         MIDDLE_BUTTON_DOUBLE_CLICK,     /**< See LEFT_BUTTON_DOUBLE_CLICK. */
  79 
  80         SDL_RIGHT_BUTTON_DOWN,          /**< An SDL right mouse button down event. */
  81         SDL_RIGHT_BUTTON_UP,            /**< An SDL right mouse button up event. */
  82         RIGHT_BUTTON_DOWN,              /**< See LEFT_BUTTON_DOWN. */
  83         RIGHT_BUTTON_UP,                /**< See LEFT_BUTTON_UP. */
  84         RIGHT_BUTTON_CLICK,             /**< See LEFT_BUTTON_CLICK. */
  85         RIGHT_BUTTON_DOUBLE_CLICK,      /**< See LEFT_BUTTON_DOUBLE_CLICK. */
  86 
  87         SDL_WHEEL_LEFT,                 /**< An SDL wheel left event. */
  88         SDL_WHEEL_RIGHT,                /**< An SDL wheel right event. */
  89         SDL_WHEEL_UP,                   /**< An SDL wheel up event. */
  90         SDL_WHEEL_DOWN,                 /**< An SDL wheel down event. */
  91         SDL_KEY_DOWN,                   /**< An SDL key down event. */
  92 
  93         NOTIFY_REMOVAL,                 /**< Sent by a widget to notify others it's being destroyed. */
  94         NOTIFY_MODIFIED,                /**<
  95                                          * Sent by a widget to notify others its contents or state are modified.
  96                                          *
  97                                          * What modified means is documented per widget. If not documented the request
  98                                          * means nothing.
  99                                          */
 100         REQUEST_PLACEMENT,              /**<
 101                                          * Request to place the widget. This may also trigger an update of other
 102                                          * layout components.
 103                                          */
 104 
 105         RECEIVE_KEYBOARD_FOCUS,         /**< Widget gains keyboard focus. */
 106         LOSE_KEYBOARD_FOCUS,            /**< Widget loses keyboard focus. */
 107 
 108         SHOW_TOOLTIP,                   /**< Request the widget to show its hover tooltip. */
 109         NOTIFY_REMOVE_TOOLTIP,          /**< Request the widget to remove its hover tooltip. */
 110         MESSAGE_SHOW_TOOLTIP,           /**< Request to show a tooltip based on the data sent. */
 111         SHOW_HELPTIP,                   /**< Request the widget to show its hover helptip. */
 112         MESSAGE_SHOW_HELPTIP,           /**< Request to show a helptip based on the data sent. */
 113 
 114         SDL_TOUCH_MOTION,
 115         SDL_TOUCH_UP,
 116         SDL_TOUCH_DOWN,
 117 
 118         SDL_RAW_EVENT                                   /**< Raw SDL event. */
 119 };

 121 /**
 122  * Helper for catching use error of dispatcher::connect_signal.
 123  *
 124  * This helper is needed as a user can't supply the wrong kind of callback
 125  * functions to dispatcher::connect_signal. If a wrong callback would be send
 126  * it will never get called.
 127  *
 128  * This version is for callbacks without extra parameters.
 129  * NOTE some mouse functions like MOUSE_ENTER don't send the mouse coordinates
 130  * to the callback function so they are also in this category.
 131  */
 132 typedef boost::mpl::set<boost::mpl::int_<DRAW>,
 133                                                 boost::mpl::int_<CLOSE_WINDOW>,
 134                                                 boost::mpl::int_<MOUSE_ENTER>,
 135                                                 boost::mpl::int_<MOUSE_LEAVE>,
 136                                                 boost::mpl::int_<LEFT_BUTTON_DOWN>,
 137                                                 boost::mpl::int_<LEFT_BUTTON_UP>,
 138                                                 boost::mpl::int_<LEFT_BUTTON_CLICK>,
 139                                                 boost::mpl::int_<LEFT_BUTTON_DOUBLE_CLICK>,
 140                                                 boost::mpl::int_<MIDDLE_BUTTON_DOWN>,
 141                                                 boost::mpl::int_<MIDDLE_BUTTON_UP>,
 142                                                 boost::mpl::int_<MIDDLE_BUTTON_CLICK>,
 143                                                 boost::mpl::int_<MIDDLE_BUTTON_DOUBLE_CLICK>,
 144                                                 boost::mpl::int_<RIGHT_BUTTON_DOWN>,
 145                                                 boost::mpl::int_<RIGHT_BUTTON_UP>,
 146                                                 boost::mpl::int_<RIGHT_BUTTON_CLICK>,
 147                                                 boost::mpl::int_<RIGHT_BUTTON_DOUBLE_CLICK> >
 148 set_event;
 149

 163                                                 boost::mpl::int_<SDL_RIGHT_BUTTON_UP>,
 164                                                 boost::mpl::int_<SHOW_TOOLTIP>,

 189 /**
 190  * Helper for catching use error of dispatcher::connect_signal.
 191  *
 192  * This version is for callbacks with a sender aka notification messages. Like
 193  *the
 194  * ones in set_event it has no extra parameters, but this version is only

 205 /**
 206  * Helper for catching use error of dispatcher::connect_signal.
 207  *
 208  * This version is for callbacks with a sender aka notification messages.
 209  * Unlike the notifications this message is send through the chain. The event
 210  * is send from a widget all the way up to the window, who always is the

 239 /**
 240  * Initializes the location of the mouse.
 241  *
 242  * After a layout of the window the mouse location needs to be updated to
 243  * test whether it entered or left a widget. Also after closing a window it's
 244  * needed to send a dummy mouse move.
 245  */
 246 void init_mouse_location();
 247 
 248 /**
 249  * Captures the mouse.
 250  *
 251  * A dispatcher can capture the mouse, when for example it's pressed on a
 252  * button, this means all mouse events after that are send to that widget.
 253  *
 254  * @param dispatcher              The dispatcher which should get the mouse
 255  *                                focus.
 256  */
 257 void capture_mouse(dispatcher* dispatcher);
 258 
 259 /**
 260  * Releases a captured mouse.
 261  *
 262  * @param dispatcher              The dispatcher which should release the mouse
 263  *                                capture.
 264  */
 265 void release_mouse(dispatcher* dispatcher);
 266 
 267 /**
 268  * Captures the keyboard.
 269  *
 270  * A dispatcher can capture the keyboard, when for example it's pressed on a
 271  * button, this means all keyboard events after that are send to that widget.
 272  *
 273  * @param dispatcher              The dispatcher which should get the keyboard
 274  *                                focus.
 275  */
 276 void capture_keyboard(dispatcher* dispatcher);
