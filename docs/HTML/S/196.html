  58 /**
  59  * Creates a t_string object (__call metamethod).
  60  * - Arg 1: userdata containing the domain.
  61  * - Arg 2: string to translate.
  62  * - Ret 1: string containing the translatable string.
  63  */
  64 static int impl_gettext(lua_State *L)
  65 {
  66         char const *m = luaL_checkstring(L, 2);
  67         char const *d = static_cast<char *>(lua_touserdata(L, 1));
  68         // Hidden metamethod, so d has to be a string. Use it to create a t_string.
  69         if(lua_isstring(L, 3)) {
  70                 const char* pl = luaL_checkstring(L, 3);
  71                 int count = luaL_checkinteger(L, 4);
  72                 luaW_pushtstring(L, t_string(m, pl, count, d));
  73         } else {
  74                 luaW_pushtstring(L, t_string(m, d));
  75         }
  76         return 1;
  77 }

  96 /**
  97  * Converts a Lua value at position @a src and appends it to @a dst.
  98  * @note This function is private to lua_tstring_concat. It expects two things.
  99  *       First, the t_string metatable is at the top of the stack on entry. (It
 100  *       is still there on exit.) Second, the caller hasn't any valuable object
 101  *       with dynamic lifetime, since they would be leaked on error.
 102  */
 103 static void tstring_concat_aux(lua_State *L, t_string &dst, int src)
 104 {
 105         switch (lua_type(L, src)) {
 106                 case LUA_TNUMBER:
 107                 case LUA_TSTRING:
 108                         dst += lua_tostring(L, src);
 109                         return;
 110                 case LUA_TUSERDATA:
 111                         // Compare its metatable with t_string's metatable.
 112                         if (t_string * src_ptr = static_cast<t_string *> (luaL_testudata(L, src, tstringKey))) {
 113                                 dst += *src_ptr;
 114                                 return;
 115                         }
 116                         //intentional fall-through
 117                 default:
 118                         luaW_type_error(L, src, "string");
 119         }
 120 }

 190 /**
 191  * Gets the parsed field of a vconfig object (_index metamethod).
 192  * Special fields __literal, __shallow_literal, __parsed, and
 193  * __shallow_parsed, return Lua tables.
 194  */
 195 static int impl_vconfig_get(lua_State *L)
 196 {
 197         vconfig *v = static_cast<vconfig *>(lua_touserdata(L, 1));
 198 
 199         if (lua_isnumber(L, 2))
 200         {
 201                 vconfig::all_children_iterator i = v->ordered_begin();
 202                 unsigned len = std::distance(i, v->ordered_end());
 203                 unsigned pos = lua_tointeger(L, 2) - 1;
 204                 if (pos >= len) return 0;
 205                 std::advance(i, pos);
 206 
 207                 lua_createtable(L, 2, 0);
 208                 lua_pushstring(L, i.get_key().c_str());
 209                 lua_rawseti(L, -2, 1);
 210                 luaW_pushvconfig(L, i.get_child());
 211                 lua_rawseti(L, -2, 2);
 212                 return 1;
 213         }
 214 
 215         char const *m = luaL_checkstring(L, 2);
 216         if (strcmp(m, "__literal") == 0) {
 217                 luaW_pushconfig(L, v->get_config());
 218                 return 1;
 219         }
 220         if (strcmp(m, "__parsed") == 0) {
 221                 luaW_pushconfig(L, v->get_parsed_config());
 222                 return 1;
 223         }
 224 
 225         bool shallow_literal = strcmp(m, "__shallow_literal") == 0;
 226         if (shallow_literal || strcmp(m, "__shallow_parsed") == 0)
 227         {
 228                 lua_newtable(L);
 229                 for (const config::attribute &a : v->get_config().attribute_range()) {
 230                         if (shallow_literal)
 231                                 luaW_pushscalar(L, a.second);
 232                         else
 233                                 luaW_pushscalar(L, v->expand(a.first));
 234                         lua_setfield(L, -2, a.first.c_str());
 235                 }
 236                 vconfig::all_children_iterator i = v->ordered_begin(),
 237                         i_end = v->ordered_end();
 238                 if (shallow_literal) {
 239                         i.disable_insertion();
 240                         i_end.disable_insertion();
 241                 }
 242                 for (int j = 1; i != i_end; ++i, ++j)
 243                 {
 244                         lua_createtable(L, 2, 0);
 245                         lua_pushstring(L, i.get_key().c_str());
 246                         lua_rawseti(L, -2, 1);
 247                         luaW_pushvconfig(L, i.get_child());
 248                         lua_rawseti(L, -2, 2);
 249                         lua_rawseti(L, -2, j);
 250                 }
 251                 return 1;
 252         }
 253 
 254         if (v->null() || !v->has_attribute(m)) return 0;
 255         luaW_pushscalar(L, (*v)[m]);
 256         return 1;
 257 }

 283 static int impl_vconfig_pairs_iter(lua_State *L)
 284 {
 285         vconfig vcfg = luaW_checkvconfig(L, 1);
 286         void* p = luaL_checkudata(L, lua_upvalueindex(1), vconfigpairsKey);
 287         config::const_attr_itors& range = *static_cast<config::const_attr_itors*>(p);
 288         if (range.empty()) {

 294         luaW_pushscalar(L, vcfg[value.first]);

 326 /**
 327  * Iterate through the subtags of a vconfig
 328  */
 329 static int impl_vconfig_ipairs_iter(lua_State *L)
 330 {
 331         luaW_checkvconfig(L, 1);
 332         int i = luaL_checkinteger(L, 2);
 333         void* p = luaL_checkudata(L, lua_upvalueindex(1), vconfigipairsKey);
 334         vconfig_child_range& range = *static_cast<vconfig_child_range*>(p);
 335         if (range.first == range.second) {
 336                 return 0;
 337         }
 338         std::pair<std::string, vconfig> value = *range.first++;
 339         lua_pushinteger(L, i + 1);
 340         lua_createtable(L, 2, 0);
 341         lua_pushlstring(L, value.first.c_str(), value.first.length());
 342         lua_rawseti(L, -2, 1);
 343         luaW_pushvconfig(L, value.second);
 344         lua_rawseti(L, -2, 2);
 345         return 2;
 346 }

 483 bool luaW_getmetafield(lua_State *L, int idx, const char* key)
 484 {
 485         if(key == nullptr) {
 486                 return false;
 487         }
 488         int n = strlen(key);
 489         if(n == 0) {
 490                 return false;
 491         }
 492         if(n >= 2 && key[0] == '_' && key[1] == '_') {
 493                 return false;
 494         }
 495         return luaL_getmetafield(L, idx, key) != 0;

 539 bool luaW_toscalar(lua_State *L, int index, config::attribute_value& v)
 540 {
 541         switch (lua_type(L, index)) {
 542                 case LUA_TBOOLEAN:
 543                         v = luaW_toboolean(L, -1);
 544                         break;
 545                 case LUA_TNUMBER:
 546                         v = lua_tonumber(L, -1);
 547                         break;
 548                 case LUA_TSTRING:
 549                         v = lua_tostring(L, -1);
 550                         break;
 551                 case LUA_TUSERDATA:
 552                 {
 553                         if (t_string * tptr = static_cast<t_string *>(luaL_testudata(L, -1, tstringKey))) {
 554                                 v = *tptr;
 555                                 break;
 556                         } else {
 557                                 return false;
 558                         }
 559                 }
 560                 default:
 561                         return false;
 562         }
 563         return true;
 564 }

 566 bool luaW_totstring(lua_State *L, int index, t_string &str)
 567 {
 568         switch (lua_type(L, index)) {
 569                 case LUA_TBOOLEAN:
 570                         str = lua_toboolean(L, index) ? "yes" : "no";
 571                         break;
 572                 case LUA_TNUMBER:
 573                 case LUA_TSTRING:
 574                         str = lua_tostring(L, index);
 575                         break;
 576                 case LUA_TUSERDATA:
 577                 {
 578                         if (t_string * tstr = static_cast<t_string *> (luaL_testudata(L, index, tstringKey))) {
 579                                 str = *tstr;
 580                                 break;
 581                         } else {
 582                                 return false;
 583                         }
 584                 }
 585                 default:
 586                         return false;
 587         }
 588         return true;

 599 void luaW_filltable(lua_State *L, config const &cfg)
 600 {
 601         if (!lua_checkstack(L, LUA_MINSTACK))
 602                 return;
 603 
 604         int k = 1;
 605         for (const config::any_child &ch : cfg.all_children_range())
 606         {
 607                 lua_createtable(L, 2, 0);
 608                 lua_pushstring(L, ch.key.c_str());
 609                 lua_rawseti(L, -2, 1);
 610                 lua_newtable(L);
 611                 luaW_filltable(L, ch.cfg);
 612                 lua_rawseti(L, -2, 2);
 613                 lua_rawseti(L, -2, k++);
 614         }
 615         for (const config::attribute &attr : cfg.attribute_range())
 616         {
 617                 luaW_pushscalar(L, attr.second);
 618                 lua_setfield(L, -2, attr.first.c_str());
 619         }

 633 bool luaW_tolocation(lua_State *L, int index, map_location& loc) {
 634         if (!lua_checkstack(L, LUA_MINSTACK)) {
 635                 return false;
 636         }
 637         if (lua_isnoneornil(L, index)) {
 638                 // Need this special check because luaW_tovconfig returns true in this case
 639                 return false;
 640         }
 641 
 642         vconfig dummy_vcfg = vconfig::unconstructed_vconfig();
 643 
 644         index = lua_absindex(L, index);
 645 
 646         if (lua_istable(L, index) || luaW_tounit(L, index) || luaW_tovconfig(L, index, dummy_vcfg)) {
 647                 map_location result;
 648                 int x_was_num = 0, y_was_num = 0;
 649                 lua_getfield(L, index, "x");
 650                 result.set_wml_x(lua_tonumberx(L, -1, &x_was_num));
 651                 lua_getfield(L, index, "y");
 652                 result.set_wml_y(lua_tonumberx(L, -1, &y_was_num));
 653                 lua_pop(L, 2);
 654                 if (!x_was_num || !y_was_num) {
 655                         // If we get here and it was userdata, checking numeric indices won't help
 656                         // (It won't help if it was a config either, but there's no easy way to check that.)
 657                         if (lua_isuserdata(L, index)) {
 658                                 return false;
 659                         }
 660                         lua_rawgeti(L, index, 1);
 661                         result.set_wml_x(lua_tonumberx(L, -1, &x_was_num));
 662                         lua_rawgeti(L, index, 2);
 663                         result.set_wml_y(lua_tonumberx(L, -1, &y_was_num));
 664                         lua_pop(L, 2);
 665                 }
 666                 if (x_was_num && y_was_num) {
 667                         loc = result;
 668                         return true;
 669                 }
 670         } else if (lua_isnumber(L, index) && lua_isnumber(L, index + 1)) {
 671                 // If it's a number, then we consume two elements on the stack
 672                 // Since we have no way of notifying the caller that we have
 673                 // done this, we remove the first number from the stack.
 674                 loc.set_wml_x(lua_tonumber(L, index));
 675                 lua_remove(L, index);
 676                 loc.set_wml_y(lua_tonumber(L, index));
 677                 return true;
 678         }
 679         return false;
 680 }

 685         if (!luaW_tolocation(L, index, result))
 686                 luaW_type_error(L, index, "location");
 687         return result;

 699 #define return_misformed() \
 700   do { lua_settop(L, initial_top); return false; } while (0)
 701 
 702 bool luaW_toconfig(lua_State *L, int index, config &cfg)
 703 {
 704         if (!lua_checkstack(L, LUA_MINSTACK))
 705                 return false;
 706 
 707         // Get the absolute index of the table.
 708         index = lua_absindex(L, index);
 709         int initial_top = lua_gettop(L);
 710 
 711         switch (lua_type(L, index))
 712         {
 713                 case LUA_TTABLE:
 714                         break;
 715                 case LUA_TUSERDATA:
 716                 {
 717                         if (vconfig * ptr = static_cast<vconfig *> (luaL_testudata(L, index, vconfigKey))) {
 718                                 cfg = ptr->get_parsed_config();
 719                                 return true;
 720                         } else {
 721                                 return false;
 722                         }
 723                 }
 724                 case LUA_TNONE:
 725                 case LUA_TNIL:
 726                         return true;
 727                 default:
 728                         return false;
 729         }
 730 
 731         // First convert the children (integer indices).
 732         for (int i = 1, i_end = lua_rawlen(L, index); i <= i_end; ++i)
 733         {
 734                 lua_rawgeti(L, index, i);
 735                 if (!lua_istable(L, -1)) return_misformed();
 736                 lua_rawgeti(L, -1, 1);
 737                 char const *m = lua_tostring(L, -1);
 738                 if (!m) return_misformed();
 739                 lua_rawgeti(L, -2, 2);
 740                 if (!luaW_toconfig(L, -1, cfg.add_child(m)))
 741                         return_misformed();
 742                 lua_pop(L, 3);
 743         }
 744 
 745         // Then convert the attributes (string indices).
 746         for (lua_pushnil(L); lua_next(L, index); lua_pop(L, 1))
 747         {
 748                 if (lua_isnumber(L, -2)) continue;
 749                 if (!lua_isstring(L, -2)) return_misformed();
 750                 config::attribute_value &v = cfg[lua_tostring(L, -2)];
 751                 if (lua_istable(L, -1)) {
 752                         int subindex = lua_absindex(L, -1);
 753                         std::ostringstream str;
 754                         for (int i = 1, i_end = lua_rawlen(L, subindex); i <= i_end; ++i, lua_pop(L, 1)) {
 755                                 lua_rawgeti(L, -1, i);
 756                                 config::attribute_value item;
 757                                 if (!luaW_toscalar(L, -1, item)) return_misformed();
 758                                 if (i > 1) str << ',';
 759                                 str << item;
 760                         }
 761                         // If there are any string keys, it's misformed
 762                         for (lua_pushnil(L); lua_next(L, subindex); lua_pop(L, 1)) {
 763                                 if (!lua_isnumber(L, -2)) return_misformed();
 764                         }
 765                         v = str.str();
 766                 } else if (!luaW_toscalar(L, -1, v)) return_misformed();
 767         }
 768 
 769         lua_settop(L, initial_top);
 770         return true;
 771 }
 772 
 773 #undef return_misformed

 793 bool luaW_tovconfig(lua_State *L, int index, vconfig &vcfg)
 794 {
 795         switch (lua_type(L, index))
 796         {
 797                 case LUA_TTABLE:
 798                 {
 799                         config cfg;
 800                         bool ok = luaW_toconfig(L, index, cfg);
 801                         if (!ok) return false;
 802                         vcfg = vconfig(cfg, true);
 803                         break;
 804                 }
 805                 case LUA_TUSERDATA:
 806                         if (vconfig * ptr = static_cast<vconfig *> (luaL_testudata(L, index, vconfigKey))) {
 807                                 vcfg = *ptr;
 808                         } else {
 809                                 return false;
 810                         }
 811                 case LUA_TNONE:
 812                 case LUA_TNIL:
 813                         break;
 814                 default:
 815                         return false;
 816         }
 817         return true;
 818 }

 828 bool luaW_getglobal(lua_State *L, const std::vector<std::string>& path)
 829 {
 830         lua_pushglobaltable(L);
 831         for (const std::string& s : path)
 832         {
 833                 if (!lua_istable(L, -1)) goto discard;
 834                 lua_pushlstring(L, s.c_str(), s.size());
 835                 lua_rawget(L, -2);
 836                 lua_remove(L, -2);
 837         }
 838 
 839         if (lua_isnil(L, -1)) {
 840                 discard:
 841                 lua_pop(L, 1);
 842                 return false;
 843         }
 844         return true;
 845 }

 852 bool luaW_pushvariable(lua_State *L, variable_access_const& v)
 853 {
 854         try
 855         {
 856                 if(v.exists_as_attribute())
 857                 {
 858                         luaW_pushscalar(L, v.as_scalar());
 859                         return true;
 860                 }
 861                 else if(v.exists_as_container())
 862                 {
 863                         lua_newtable(L);
 864                         if (luaW_toboolean(L, 2))
 865                                 luaW_filltable(L, v.as_container());
 866                         return true;
 867                 }
 868                 else
 869                 {
 870                         lua_pushnil(L);
 871                         return true;
 872                 }
 873         }
 874         catch (const invalid_variablename_exception&)
 875         {
 876                 WRN_LUA << v.get_error_message();
 877                 return false;
 878         }
 879 }

 881 bool luaW_checkvariable(lua_State *L, variable_access_create& v, int n)
 882 {
 883         int variabletype = lua_type(L, n);
 884         try
 885         {
 886                 switch (variabletype) {
 887                 case LUA_TBOOLEAN:
 888                         v.as_scalar() = luaW_toboolean(L, n);
 889                         return true;
 890                 case LUA_TNUMBER:
 891                         v.as_scalar() = lua_tonumber(L, n);
 892                         return true;
 893                 case LUA_TSTRING:
 894                         v.as_scalar() = lua_tostring(L, n);
 895                         return true;
 896                 case LUA_TUSERDATA:
 897                         if (t_string * t_str = static_cast<t_string*> (luaL_testudata(L, n, tstringKey))) {
 898                                 v.as_scalar() = *t_str;
 899                                 return true;
 900                         }
 901                         goto default_explicit;
 902                 case LUA_TTABLE:
 903                         {
 904                                 config &cfg = v.as_container();
 905                                 cfg.clear();
 906                                 if (luaW_toconfig(L, n, cfg)) {
 907                                         return true;
 908                                 }
 909                                 FALLTHROUGH;
 910                         }
 911                 default:
 912                 default_explicit:
 913                         return luaW_type_error(L, n, "WML table or scalar") != 0;
 914 
 915                 }
 916         }
 917         catch (const invalid_variablename_exception&)
 918         {
 919                 WRN_LUA << v.get_error_message() << " when attempting to write a '" << lua_typename(L, variabletype) << "'\n";
 920                 return false;
 921         }
 922 }

 937 int luaW_pcall_internal(lua_State *L, int nArgs, int nRets)
 938 {
 939         // Load the error handler before the function and its arguments.
 940         lua_getfield(L, LUA_REGISTRYINDEX, executeKey);
 941         lua_insert(L, -2 - nArgs);
 942 
 943         int error_handler_index = lua_gettop(L) - nArgs - 1;
 944 
 945         // Call the function.
 946         int errcode = lua_pcall(L, nArgs, nRets, -2 - nArgs);
 947 
 948         lua_jailbreak_exception::rethrow();
 949 
 950         // Remove the error handler.
 951         lua_remove(L, error_handler_index);
 952 
 953         return errcode;
 954 }

 956 #ifdef _MSC_VER
 957 #pragma warning (push)
 958 #pragma warning (disable: 4706)
 959 #endif
 960 bool luaW_pcall(lua_State *L, int nArgs, int nRets, bool allow_wml_error)
 961 {
 962         int res = luaW_pcall_internal(L, nArgs, nRets);
 963 
 964         if (res)
 965         {
 966                 /*
 967                  * When an exception is thrown which doesn't derive from
 968                  * std::exception m will be nullptr pointer.
 969                  */
 970                 char const *m = lua_tostring(L, -1);
 971                 if(m) {
 972                         if (allow_wml_error && strncmp(m, "~wml:", 5) == 0) {
 973                                 m += 5;
 974                                 char const *e = strstr(m, "stack traceback");
 975                                 lg::wml_error() << std::string(m, e ? e - m : strlen(m));
 976                         } else if (allow_wml_error && strncmp(m, "~lua:", 5) == 0) {
 977                                 m += 5;
 978                                 char const *e = nullptr, *em = m;
 979                                 while (em[0] && ((em = strstr(em + 1, "stack traceback"))))
 980 #ifdef _MSC_VER
 981 #pragma warning (pop)
 982 #endif
 983                                         e = em;
 984                                 chat_message("Lua error", std::string(m, e ? e - m : strlen(m)));
 985                         } else {
 986                                 ERR_LUA << m << '\n';
 987                                 chat_message("Lua error", m);
 988                         }
 989                 } else {
 990                         chat_message("Lua caught unknown exception", "");
 991                 }
 992                 lua_pop(L, 1);
 993                 return false;
 994         }
 995 
 996         return true;
 997 }
