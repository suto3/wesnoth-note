 328     # When an enemy side has less than half of its units left, some undead
 329     # appear at the next turn start in a location from which they can't reach
 330     # any player unit in one turn.
 331     #
 332     # NOTE: if you modify anything in the inner event, make extra sure you
 333     # handle delayed/undelayed variable substitution right!

 335     [event]
 336         name=die
 337         first_time_only=no
 338 
 339         [filter]
 340             side=2,3,4
 341         [/filter]
 342 
 343         [filter_condition]
 344             [variable]
 345                 name=side_$unit.side|_undead_backup_triggered
 346                 boolean_not_equals=yes
 347             [/variable]
 348         [/filter_condition]
 349 
 350         # This bit tracks the highest number of units the side has had; in most
 351         # cases this will likely be the number of units the side has when its
 352         # first unit dies.
 353         #
 354         # If this counter isn't perfectly accurate in every situation, it
 355         # shouldn't matter.
 356         [store_unit]
 357             [filter]
 358                 side=$unit.side
 359 
 360                 [not]
 361                     x,y=$x1,$y1
 362                 [/not]
 363             [/filter]
 364 
 365             kill=no
 366             variable=this_side_enemies
 367         [/store_unit]
 368         [if]
 369             [variable]
 370                 name=side_$unit.side|_top_enemy_count
 371                 less_than_equal_to=$this_side_enemies.length
 372             [/variable]
 373 
 374             [then]
 375                 {VARIABLE side_$unit.side|_top_enemy_count "$($this_side_enemies.length + 1)"}
 376             [/then]
 377         [/if]
 378 
 379         [if]
 380             [variable]
 381                 name=this_side_enemies.length
 382                 less_than_equal_to="$($side_$unit.side|_top_enemy_count / 2)"
 383             [/variable]
 384 
 385             [then]
 386                 #{DEBUG_MSG "Side $unit.side only has $this_side_enemies.length|/$side_$unit.side|_top_enemy_count units left, triggering undead reinforcements..."}
 387 
 388                 {VARIABLE side_$unit.side|_undead_backup_triggered yes}
 389 
 390                 [store_starting_location]
 391                     side=$unit.side
 392                     variable=search_center
 393                 [/store_starting_location]
 394 
 395                 [event]
 396                     name=new turn
 397                     delayed_variable_substitution=no
 398 
 399                     # This loop searches for a location which satisfies the
 400                     # following conditions:
 401                     # 1. Within exactly 12 hexes from any player unit
 402                     # 2. As close as possible to the location where this unit
 403                     #    was killed, but only through G*,R*,C*,K*,S*
 404                     # 3. Not next to any existing (enemy) unit
 405 
 406                     {VARIABLE search_radius 1}
 407                     [while]
 408                         [variable]
 409                             name=possible_undead_locs.length
 410                             less_than=1
 411                         [/variable]
 412 
 413                         [variable]
 414                             name=search_radius
 415                             less_than=30
 416                         [/variable]
 417 
 418                         [do]
 419                             [store_locations]
 420                                 terrain=!,H*,M*,*^F*,Q*
 421 
 422                                 [not]
 423                                     [filter]
 424                                     [/filter]
 425 
 426                                     radius=1
 427                                 [/not]
 428 
 429                                 [not]
 430                                     [filter]
 431                                         side=1
 432                                     [/filter]
 433 
 434                                     radius=11
 435                                 [/not]
 436 
 437                                 [and]
 438                                     [filter]
 439                                         side=1
 440                                     [/filter]
 441 
 442                                     radius=12
 443                                 [/and]
 444 
 445                                 [and]
 446                                     x,y=$search_center.x,$search_center.y
 447                                     radius=$|search_radius
 448 
 449                                     [filter_radius]
 450                                         terrain=G*,R*,C*,K*,S*
 451                                     [/filter_radius]
 452                                 [/and]
 453 
 454                                 variable=possible_undead_locs
 455                             [/store_locations]
 456 
 457                             {VARIABLE_OP search_radius add 1}
 458                         [/do]
 459                     [/while]
 460                     [if]
 461                         [variable]
 462                             name=possible_undead_locs.length
 463                             greater_than_equal_to=1
 464                         [/variable]
 465 
 466                         [then]
 467                             {VARIABLE_OP undead_loc_i rand "0..$|($|possible_undead_locs.length - 1)"}
 468                             {VARIABLE spawn_x $|possible_undead_locs[$|undead_loc_i].x}
 469                             {VARIABLE spawn_y $|possible_undead_locs[$|undead_loc_i].y}
 470 
 471                             {VARIABLE_OP undead_spawns_so_far add 1}
 472 
 473                             [scroll_to]
 474                                 x,y=$|possible_undead_locs[$|undead_loc_i].x,$|possible_undead_locs[$|undead_loc_i].y
 475                             [/scroll_to]
 476 
 477                             #{DEBUG_MSG "spawning at $|spawn_x|,$|spawn_y"}
 478 
 479                             [switch]
 480                                 variable=undead_spawns_so_far
 481 
 482                                 [case]
 483                                     value=1
 484 
 485                                     {NAMED_LOYAL_UNIT $unit.side Deathblade $|spawn_x $|spawn_y Norte _"Norte"}
 486 
 487                                     {LOYAL_UNIT $unit.side (Chocobone) $|spawn_x $|spawn_y}
 488 #ifndef EASY
 489                                     {LOYAL_UNIT $unit.side (Chocobone) $|spawn_x $|spawn_y}
 490 #endif
 491 
 492                                     {CLEAR_FOG 1 $|possible_undead_locs[$|undead_loc_i].x $|possible_undead_locs[$|undead_loc_i].y 2}
 493 
 494                                     [message]
 495                                         speaker=Norte
 496                                         message= _ "You will be felled by my hand, useless fleshbags!"
 497                                     [/message]
 498                                 [/case]
 499

 501                                     value=2
 502

 504
