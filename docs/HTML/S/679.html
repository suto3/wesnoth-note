  44 // std::map::operator[] does not support heterogenous lookup so we need this to work around.
  45 template<typename Map, typename Key>
  46 typename Map::mapped_type& map_get(Map& map, Key&& key)
  47 {
  48         auto res = map.lower_bound(key);
  49 
  50         if(res == map.end() || key != res->first) {
  51                 res = map.emplace_hint(res, std::piecewise_construct, std::forward_as_tuple(key), std::tuple<>());
  52         }
  53 
  54         return res->second;
  55 }
  56 
  57 // std::map::erase does not support heterogenous lookup so we need this to work around.
  58 template<typename Map, typename Key>
  59 int map_erase_key(Map& map, Key&& key)
  60 {
  61         auto i = map.find(key);
  62         if(i != map.end()) {
  63                 map.erase(i);
  64                 return 1;
  65         }
  66         return 0;
  67 }

 440 config& config::add_child_at(config_key_type key, const config& val, unsigned index)
 441 {
 442         check_valid(val);
 443 
 444         child_list& v = map_get(children_, key);
 445         if(index > v.size()) {
 446                 throw error("illegal index to add child at");
 447         }
 448 
 449         v.emplace(v.begin() + index, new config(val));
 450 
 451         bool inserted = false;
 452 
 453         const child_pos value(children_.find(key), index);
 454 
 455         std::vector<child_pos>::iterator ord = ordered_children.begin();
 456         for(; ord != ordered_children.end(); ++ord) {
 457                 if(ord->pos != value.pos)
 458                         continue;
 459                 if(!inserted && ord->index == index) {
 460                         ord = ordered_children.insert(ord, value);
 461                         inserted = true;
 462                 } else if(ord->index >= index) {
 463                         ord->index++;
 464                 }
 465         }
 466 
 467         if(!inserted) {
 468                 ordered_children.push_back(value);
 469         }
 470 
 471         return *v[index];
 472 }

 508 void config::splice_children(config& src, const std::string& key)
 509 {
 510         check_valid(src);
 511 
 512         child_map::iterator i_src = src.children_.find(key);
 513         if(i_src == src.children_.end()) {
 514                 return;
 515         }
 516 
 517         src.ordered_children.erase(
 518                 std::remove_if(src.ordered_children.begin(), src.ordered_children.end(), remove_ordered(i_src)),
 519                 src.ordered_children.end());
 520 
 521         child_list& dst = map_get(children_, key);
 522         child_map::iterator i_dst = children_.find(key);
 523 
 524         unsigned before = dst.size();
 525         dst.insert(dst.end(), std::make_move_iterator(i_src->second.begin()), std::make_move_iterator(i_src->second.end()));
 526         src.children_.erase(i_src);
 527         // key might be a reference to i_src->first, so it is no longer usable.
 528 
 529         for(unsigned j = before; j < dst.size(); ++j) {
 530                 ordered_children.emplace_back(i_dst, j);
 531         }
 532 }

 547 std::vector<config::child_pos>::iterator config::remove_child(const child_map::iterator& pos, unsigned index)
 548 {
 549         /* Find the position with the correct index and decrement all the
 550            indices in the ordering that are above this index. */
 551         unsigned found = 0;
 552         for(child_pos& p : ordered_children) {
 553                 if(p.pos != pos) {
 554                         continue;
 555                 }
 556 
 557                 if(p.index == index) {
 558                         found = &p - &ordered_children.front();
 559                 } else if(p.index > index) {
 560                         --p.index;
 561                 }
 562         }
 563 
 564         // Remove from the child map.
 565         pos->second.erase(pos->second.begin() + index);
 566 
 567         // Erase from the ordering and return the next position.
 568         return ordered_children.erase(ordered_children.begin() + found);
 569 }

 693 config& config::find_child(config_key_type key, const std::string& name, const std::string& value)
 694 {
 695         check_valid();
 696 
 697         const child_map::iterator i = children_.find(key);
 698         if(i == children_.end()) {
 699                 DBG_CF << "Key »" << name << "« value »" << value << "« pair not found as child of key »" << key << "«.\n";
 700 
 701                 return invalid;
 702         }
 703 
 704         const child_list::iterator j = std::find_if(i->second.begin(), i->second.end(),
 705                 [&](const std::unique_ptr<config>& pcfg) {
 706                         const config& cfg = *pcfg;
 707                         return cfg[name] == value;
 708                 }
 709         );
 710 
 711         if(j != i->second.end()) {
 712                 return **j;
 713         }
 714 
 715         DBG_CF << "Key »" << name << "« value »" << value << "« pair not found as child of key »" << key << "«.\n";
 716 
 717         return invalid;

 800 void config::get_diff(const config& c, config& res) const
 801 {
 802         check_valid(c);
 803         check_valid(res);
 804 
 805         config* inserts = nullptr;
 806 
 807         for(const auto& v : values_) {
 808                 if(v.second.blank()) {
 809                         continue;
 810                 }
 811 
 812                 const attribute_map::const_iterator j = c.values_.find(v.first);
 813                 if(j == c.values_.end() || (v.second != j->second && !v.second.blank())) {
 814                         if(inserts == nullptr) {
 815                                 inserts = &res.add_child("insert");
 816                         }
 817 
 818                         (*inserts)[v.first] = v.second;
 819                 }
 820         }
 821 
 822         config* deletes = nullptr;
 823 
 824         for(const auto& v : c.values_) {
 825                 if(v.second.blank()) {
 826                         continue;
 827                 }
 828 
 829                 const attribute_map::const_iterator itor = values_.find(v.first);
 830                 if(itor == values_.end() || itor->second.blank()) {
 831                         if(deletes == nullptr) {
 832                                 deletes = &res.add_child("delete");
 833                         }
 834 
 835                         (*deletes)[v.first] = "x";
 836                 }
 837         }
 838 
 839         std::vector<std::string> entities;
 840 
 841         for(const auto& child : children_) {
 842                 entities.push_back(child.first);
 843         }
 844 
 845         for(const auto& child : c.children_) {
 846                 if(children_.count(child.first) == 0) {
 847                         entities.push_back(child.first);
 848                 }
 849         }
 850 
 851         for(const std::string& entity : entities) {
 852                 const child_map::const_iterator itor_a = children_.find(entity);
 853                 const child_map::const_iterator itor_b = c.children_.find(entity);
 854 
 855                 static const child_list dummy;
 856 
 857                 // Get the two child lists. 'b' has to be modified to look like 'a'.
 858                 const child_list& a = itor_a != children_.end() ? itor_a->second : dummy;
 859                 const child_list& b = itor_b != c.children_.end() ? itor_b->second : dummy;
 860 
 861                 size_t ndeletes = 0;
 862                 size_t ai = 0, bi = 0;
 863                 while(ai != a.size() || bi != b.size()) {
 864                         // If the two elements are the same, nothing needs to be done.
 865                         if(ai < a.size() && bi < b.size() && *a[ai] == *b[bi]) {
 866                                 ++ai;
 867                                 ++bi;
 868                         } else {
 869                                 // We have to work out what the most appropriate operation --
 870                                 // delete, insert, or change is the best to get b[bi] looking like a[ai].
 871                                 std::stringstream buf;
 872 
 873                                 // If b has more elements than a, then we assume this element
 874                                 // is an element that needs deleting.
 875                                 if(b.size() - bi > a.size() - ai) {
 876                                         config& new_delete = res.add_child("delete_child");
 877                                         buf << bi - ndeletes;
 878                                         new_delete.values_["index"] = buf.str();
 879                                         new_delete.add_child(entity);
 880 
 881                                         ++ndeletes;
 882                                         ++bi;
 883                                 }
 884 
 885                                 // If b has less elements than a, then we assume this element
 886                                 // is an element that needs inserting.
 887                                 else if(b.size() - bi < a.size() - ai) {
 888                                         config& new_insert = res.add_child("insert_child");
 889                                         buf << ai;
 890                                         new_insert.values_["index"] = buf.str();
 891                                         new_insert.add_child(entity, *a[ai]);
 892 
 893                                         ++ai;
 894                                 }
 895 
 896                                 // Otherwise, they have the same number of elements,
 897                                 // so try just changing this element to match.
 898                                 else {
 899                                         config& new_change = res.add_child("change_child");
 900                                         buf << bi;
 901                                         new_change.values_["index"] = buf.str();
 902                                         new_change.add_child(entity, a[ai]->get_diff(*b[bi]));
 903 
 904                                         ++ai;
 905                                         ++bi;
 906                                 }
 907                         }
 908                 }
 909         }
 910 }

 912 void config::apply_diff(const config& diff, bool track /* = false */)
 913 {
 914         check_valid(diff);
 915 
 916         if(track) {
 917                 values_[diff_track_attribute] = "modified";
 918         }
 919 
 920         if(const config& inserts = diff.child("insert")) {
 921                 for(const attribute& v : inserts.attribute_range()) {
 922                         values_[v.first] = v.second;
 923                 }
 924         }
 925 
 926         if(const config& deletes = diff.child("delete")) {
 927                 for(const attribute& v : deletes.attribute_range()) {
 928                         values_.erase(v.first);
 929                 }
 930         }
 931 
 932         for(const config& i : diff.child_range("change_child")) {
 933                 const size_t index = lexical_cast<size_t>(i["index"].str());
 934                 for(const any_child& item : i.all_children_range()) {
 935                         if(item.key.empty()) {
 936                                 continue;
 937                         }
 938 
 939                         const child_map::iterator itor = children_.find(item.key);
 940                         if(itor == children_.end() || index >= itor->second.size()) {
 941                                 throw error("error in diff: could not find element '" + item.key + "'");
 942                         }
 943 
 944                         itor->second[index]->apply_diff(item.cfg, track);
 945                 }
 946         }
 947 
 948         for(const config& i : diff.child_range("insert_child")) {
 949                 const size_t index = lexical_cast<size_t>(i["index"].str());
 950                 for(const any_child& item : i.all_children_range()) {
 951                         config& inserted = add_child_at(item.key, item.cfg, index);
 952                         if(track) {
 953                                 inserted[diff_track_attribute] = "new";
 954                         }
 955                 }
 956         }
 957 
 958         for(const config& i : diff.child_range("delete_child")) {
 959                 const size_t index = lexical_cast<size_t>(i["index"].str());
 960                 for(const any_child& item : i.all_children_range()) {
 961                         if(!track) {
 962                                 remove_child(item.key, index);
 963                         } else {
 964                                 const child_map::iterator itor = children_.find(item.key);
 965                                 if(itor == children_.end() || index >= itor->second.size()) {
 966                                         throw error("error in diff: could not find element '" + item.key + "'");
 967                                 }
 968 
 969                                 itor->second[index]->values_[diff_track_attribute] = "deleted";
 970                         }
 971                 }
 972         }
 973 }

 975 void config::clear_diff_track(const config& diff)
 976 {
 977         remove_attribute(diff_track_attribute);
 978         for(const config& i : diff.child_range("delete_child")) {
 979                 const size_t index = lexical_cast<size_t>(i["index"].str());
 980                 for(const any_child& item : i.all_children_range()) {
 981                         remove_child(item.key, index);
 982                 }
 983         }
 984 
 985         for(const config& i : diff.child_range("change_child")) {
 986                 const size_t index = lexical_cast<size_t>(i["index"].str());
 987                 for(const any_child& item : i.all_children_range()) {
 988                         if(item.key.empty()) {
 989                                 continue;
 990                         }
 991 
 992                         const child_map::iterator itor = children_.find(item.key);
 993                         if(itor == children_.end() || index >= itor->second.size()) {
 994                                 throw error("error in diff: could not find element '" + item.key + "'");
 995                         }
 996 
 997                         itor->second[index]->clear_diff_track(item.cfg);
 998                 }
 999         }
1000 
1001         for(std::pair<const std::string, child_list>& p : children_) {
1002                 for(auto& cfg : p.second) {
1003                         cfg->remove_attribute(diff_track_attribute);
1004                 }
1005         }
1006 }

1008 /**
1009  * Merge config 'c' into this config, overwriting this config's values.
1010  */
1011 void config::merge_with(const config& c)
1012 {
1013         check_valid(c);
1014 
1015         std::vector<child_pos> to_remove;
1016         std::map<std::string, unsigned> visitations;
1017 
1018         // Merge attributes first
1019         merge_attributes(c);
1020 
1021         // Now merge shared tags
1022         all_children_iterator::Itor i, i_end = ordered_children.end();
1023         for(i = ordered_children.begin(); i != i_end; ++i) {
1024                 const std::string& tag = i->pos->first;
1025                 const child_map::const_iterator j = c.children_.find(tag);
1026 
1027                 if(j != c.children_.end()) {
1028                         unsigned& visits = visitations[tag];
1029 
1030                         if(visits < j->second.size()) {
1031                                 // Get a const config so we do not add attributes.
1032                                 const config& merge_child = *j->second[visits++];
1033 
1034                                 if(merge_child["__remove"].to_bool()) {
1035                                         to_remove.push_back(*i);
1036                                 } else {
1037                                         (i->pos->second[i->index])->merge_with(merge_child);
1038                                 }
1039                         }
1040                 }
1041         }
1042 
1043         // Now add any unvisited tags
1044         for(const auto& pair : c.children_) {
1045                 const std::string& tag = pair.first;
1046                 unsigned& visits = visitations[tag];
1047                 while(visits < pair.second.size()) {
1048                         add_child(tag, *pair.second[visits++]);
1049                 }
1050         }
1051 
1052         // Remove those marked so
1053         std::map<std::string, unsigned> removals;
1054         for(const child_pos& pos : to_remove) {
1055                 const std::string& tag = pos.pos->first;
1056                 unsigned& removes = removals[tag];
1057                 remove_child(tag, pos.index - removes++);
1058         }
1059 }

1063  */
1064 void config::inherit_from(const config& c)
1065 {
1066         // Using a scratch config and merge_with() seems to execute about as fast
1067         // as direct coding of this merge.
1068         config scratch(c);
1069         scratch.merge_with(*this);
1070         swap(scratch);

1073 bool config::matches(const config& filter) const
1074 {
1075         check_valid(filter);
1076 
1077         for(const attribute& i : filter.attribute_range()) {
1078                 const attribute_value* v = get(i.first);
1079                 if(!v || *v != i.second) {
1080                         return false;
1081                 }
1082         }
1083 
1084         for(const any_child& i : filter.all_children_range()) {
1085                 if(i.key == "not") {
1086                         if(matches(i.cfg)) {
1087                                 return false;
1088                         }
1089 
1090                         continue;
1091                 }
1092 
1093                 bool found = false;
1094                 for(const config& j : child_range(i.key)) {
1095                         if(j.matches(i.cfg)) {
1096                                 found = true;
1097                                 break;
1098                         }
1099                 }
1100 
1101                 if(!found) {
1102                         return false;
1103                 }
1104         }
1105 
1106         return true;
1107 }

1154 std::string config::hash() const
1155 {
1156         check_valid();
1157 
1158         static const unsigned int hash_length = 128;
1159         static const char hash_string[] = "+-,.<>0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
1160         char hash_str[hash_length + 1];
1161 
1162         unsigned int i;
1163         for(i = 0; i != hash_length; ++i) {
1164                 hash_str[i] = 'a';
1165         }
1166 
1167         hash_str[hash_length] = 0;
1168 
1169         i = 0;
1170         for(const attribute& val : values_) {
1171                 if(val.second.blank()) {
1172                         continue;
1173                 }
1174 
1175                 for(char c : val.first) {
1176                         hash_str[i] ^= c;
1177                         if(++i == hash_length) {
1178                                 i = 0;
1179                         }
1180                 }
1181 
1182                 std::string base_str = val.second.t_str().base_str();
1183                 for(const char c : base_str) {
1184                         hash_str[i] ^= c;
1185                         if(++i == hash_length) {
1186                                 i = 0;
1187                         }
1188                 }
1189         }
1190 
1191         for(const any_child& ch : all_children_range()) {
1192                 std::string child_hash = ch.cfg.hash();
1193                 for(char c : child_hash) {
1194                         hash_str[i] ^= c;
1195                         ++i;
1196                         if(i == hash_length) {
1197                                 i = 0;
1198                         }
1199                 }
1200         }
1201 
1202         for(i = 0; i != hash_length; ++i) {
1203                 hash_str[i] = hash_string[static_cast<unsigned>(hash_str[i]) % strlen(hash_string)];
1204         }
1205 
1206         return std::string(hash_str);
1207 }
