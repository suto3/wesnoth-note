  75     re_record_start = '^\s*'
  76     # There must be whitespace between the field separator (&).
  77     # However if there is an empty field eg '...foo & & bar...'
  78     # The first field separator eats the whitespace before the second
  79     # which would \s+ cause to fail.
  80     # The solution is to make the whitespace optional and assert the
  81     # character before the ampersand is whitespace.
  82     re_field_separator = '\s*(?:(?<=\s))&\s+'
  83     # Same issue as with the re_field_separator but then with '...foo & $'
  84     re_record_end = '\s*(?:(?<=\s))\$$'

  86     re_variable = '([a-zA-Z]\w*)'

  99     def reindent(data):
 100         """Converts the raw input to an easier to use format.
 101 
 102         Lines starting with 8 spaces are concatenated with the previous line.
 103         The start of line ' *' are removed and if another space exists it's also
 104         removed."""
 105 
 106         # concatenate
 107         data = re.sub(r'\n \*(?: ){8,}', " ", data)
 108 
 109         # strip
 110         data = re.sub(" \*(?: |)", "", data)
 111 
 112         #annotation
 113         data = re.sub(r'@(?:begin|end|allow|remove)\{(?:parent|tag|link|global|type|key)\}(?:\{.*\})', "", data)
 114         return data

 150     def format(data):
 151         """Formats the data for the wiki.
 152 
 153         Replaces the following:
 154         -  An end of line and its surrounding whitespace to a single space.
 155         - @$ -> $
 156         - @& -> &
 157         - @* -> \n* needed in a list.
 158         - @- -> \n needed to add text after a list.
 159         """
 160         data = re.sub(r'\s*\n\s*', ' ', data)
 161         data = re.sub(r'@\$', '$', data)
 162         data = re.sub(r'@&', '&', data)
 163         data = re.sub(r'@\*', "\n*", data)
 164         data = re.sub(r'@\-', "\n", data)
 165         return data

 167     def create_config_table(data):
 168         """Creates a table for data in a config table.
 169 
 170         A config table is a table with info about WML configuration key value pairs.
 171         """
 172 
 173         # matches a line like
 174         # x1 & f_unsigned & 0 &            The x coordinate of the
 175         # startpoint. $
 176         # x1 & f_unsigned & &              The x coordinate of the
 177         # startpoint. $
 178         regex = re.compile("([A-Za-z]\w*) +& +([A-Za-z]\w*) +& +([^&]*?) *& +(.*) +\$")
 179         res = regex.findall(data)
 180 
 181         regex  = re_record_start
 182         regex += re_variable           # 0 variable id
 183         regex += re_field_separator
 184         regex += re_variable           # 1 variable type
 185         regex += re_field_separator
 186         regex += '(.*?)'               # 2 optional default value, if omitted mandatory field
 187         regex += re_field_separator
 188         regex += re_string             # 3 description
 189         regex += re_record_end
 190 
 191         regex = re.compile(regex, re.DOTALL | re.MULTILINE)
 192         res = regex.findall(data)
 193 
 194         if is_empty(res, data):
 195             return "Empty table."
 196 
 197         result = '{| class="wikitable"'
 198         result += "\n!key\n!type\n!default\n!description\n"
 199         for i in res:
 200             result += "|-\n"
 201             result += "| %s\n" % i[0]
 202             result += "| [[GUIVariable#%s|%s]]\n" % (i[1], i[1])
 203             if not i[2]:
 204                 result += "| mandatory\n"
 205             else:
 206                 result += "| %s\n" % i[2]
 207             result += "| %s\n" % format(i[3])
 208         result += "|}"
 209 
 210         return result

 372     def validate_table(table):
 373         """Validates a table.
 374 
 375         At the moments tests for whitespace around separators."""
 376 
 377         regex = '((?<![@\s])&)|([^@]&(?!\s))'
 378         regex = re.compile(regex)
 379         invalid_field_separators = regex.findall(table)
 380 
 381         # Not the last test should actually be (\$(?![\s\Z]) but that fails as
 382         # work-around add a space to the input so when it would match the end
 383         # of the string it now matches the added space.
 384         regex = '((?<![@\s])\$)|([^@]\$(?!\s))'
 385         regex = re.compile(regex)
 386         invalid_record_terminator = regex.findall(table + ' ')
 387 
 388         if len(invalid_field_separators) or len(invalid_record_terminator):
 389             result = "Found %i invalid field separators " % (len(invalid_field_separators))
 390             result += "and %i invalid record separators " % (len(invalid_record_terminator))
 391             result += "in:\n%s\n\n" % (table)
 392             return result
 393         else:
 394             return None

 419     def create_wml_reference_description(data, nest_level):
 420         """ Creates a description for wml reference purposes."""
 421         # Matches a line like:
 422         # name & string & "" & 1.5 & description
 423         # [relative] & node & * & & description
 424 
 425         regex = re.compile("(.*) +& +(.*) +& *(.*) *& +(.*) *& *(.*)")
 426         res = regex.findall(data)
 427 
 428         if is_empty(res, data):
 429             return "Empty description"
 430 
 431         i = res[0]
 432         result = "%s '''''%s''''' " % ("*" * nest_level, i[0].strip())
 433         if i[1] != 'node':
 434             result += "(%s, ''%s'') " % (i[1], "default " + i[2].rstrip() if i[2] else "'mandatory'")
 435         else:
 436             result += "(%s) " % {'*': "zero or more times", '+': "one or more times", '?': "zero or one times", '1': "one time", '': "empty string bug."}[i[2].rstrip()]
 437         result += "%s" % i[4]
 438         if i[3]:
 439             result += " {{DevFeature%s}}" % i[3].replace(' ', '')
 440         return result + "\n"

 442     def create_description(description):
 443         """Wrapper for creating descriptions."""
 444 
 445         # List of descriptions used.
 446         descriptions = {
 447             "wml_reference": create_wml_reference_description}
 448 
 449         # We use a list because it's easier to handle nested elements
 450         type = [description.group(1)]
 451 
 452         # The $ are the separators between rows of a description
 453         parseme = description.group(2).split("$")
 454         # We strip everything that we don't need
 455         parseme = [i.replace("\n", "").strip() for i in parseme]
 456 
 457         nest_level = 1
 458         data = ""
 459 
 460         for i in parseme:
 461 
 462             # If it's a new block, we need to handle it.
 463             if i.startswith("@begin"):
 464                 nest_level += 1
 465                 i = i.lstrip("@begin{description}{")
 466                 type += [i[:i.find("}")]]
 467                 i = i.strip(type[nest_level - 1] + "}")
 468 
 469             # We also handle end of blocks.
 470             elif i.startswith("@end") and i != "@end{description}":
 471                 nest_level -= 1
 472                 i = i.lstrip("@end{description}")
 473                 type = type[:-1]
 474 
 475             # Send content unless special "@end" case.
 476             if not i == "@end{description}":
 477                 data += descriptions[type[nest_level-1]](i + "\n", nest_level)
 478 
 479         return data

 484         The body needs to be stripped of known markup values.

 494     def process(data):
 495         """Processes a wiki block."""
 496 
 497         data = replace_macros(data)
 498 
 499         res = re.compile("(.*?)\n\n(.*)", re.S).findall(data)
 500         if res:
 501             header = process_header(res[0][0])
 502             body = process_body(res[0][1])
 503         else:
 504             print "File: " + current_file
 505             print "Block:\n" + current_block
 506             print "\n\nInvalid wiki block, discarded."
 507             return
 508 
 509         if not header[0]:
 510             print "File: " + current_file
 511             print "Block:\n" + current_block
 512             print "\n\nNo page defined, dropped."
 513             return
 514 
 515         if not header[0] in file_map:
 516             file_map[header[0]] = []
 517 
 518         file_map[header[0]].append([header[1], body])

 568     def instanciate_macro(macro):
 569         """Replaces a macro text by the macro value."""

 571         global macro_map

 579     def replace_macros(data):
 580         """Replaces all macros found in the data.
 581 
 582         If a macro isn't defined, the text is left as is."""
 583 
 584         macro_regex = re.compile("@macro *= *(\w+)", re.M | re.S)
 585         data = macro_regex.sub(lambda match: instanciate_macro(match), data)
