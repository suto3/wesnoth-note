 290 bool game_launcher::init_joystick()
 291 {
 292         if (!preferences::joystick_support_enabled())
 293                 return false;
 294 
 295         if(SDL_WasInit(SDL_INIT_JOYSTICK) == 0)
 296                 if(SDL_InitSubSystem(SDL_INIT_JOYSTICK) == -1)
 297                         return false;
 298 
 299         int joysticks = SDL_NumJoysticks();
 300         if (joysticks == 0) return false;
 301 
 302         SDL_JoystickEventState(SDL_ENABLE);
 303 
 304         bool joystick_found = false;
 305         for (int i = 0; i<joysticks; i++)  {
 306 
 307                 if (SDL_JoystickOpen(i))
 308                         joystick_found = true;
 309         }
 310         return joystick_found;
 311 }

 313 bool game_launcher::init_language()
 314 {
 315         if(!::load_language_list())
 316                 return false;
 317 
 318         language_def locale;
 319         if(cmdline_opts_.language) {
 320                 std::vector<language_def> langs = get_languages();
 321                 for(const language_def & def : langs) {
 322                         if(def.localename == *cmdline_opts_.language) {
 323                                 locale = def;
 324                                 break;
 325                         }
 326                 }
 327                 if(locale.localename.empty()) {
 328                         std::cerr << "Language symbol '" << *cmdline_opts_.language << "' not found.\n";
 329                         return false;
 330                 }
 331         } else {
 332                 locale = get_locale();
 333         }
 334         ::set_language(locale);
 335 
 336         return true;
 337 }

 368 bool game_launcher::init_lua_script()
 369 {
 370         bool error = false;
 371 
 372         std::cerr << "Checking lua scripts... ";
 373 
 374         if (cmdline_opts_.script_unsafe_mode) {
 375                 plugins_manager::get()->get_kernel_base()->load_package(); //load the "package" package, so that scripts can get what packages they want
 376         }
 377 
 378         // get the application lua kernel, load and execute script file, if script file is present
 379         if (cmdline_opts_.script_file)
 380         {
 381                 filesystem::scoped_istream sf = filesystem::istream_file(*cmdline_opts_.script_file);
 382 
 383                 if (!sf->fail()) {
 384                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 385                         jump_to_multiplayer_ = false;
 386                         jump_to_editor_ = false;
 387                         jump_to_campaign_.jump_ = false;
 388 
 389                         std::string full_script((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 390 
 391                         std::cerr << "\nRunning lua script: " << *cmdline_opts_.script_file << std::endl;
 392 
 393                         plugins_manager::get()->get_kernel_base()->run(full_script.c_str());
 394                 } else {
 395                         std::cerr << "Encountered failure when opening script '" << *cmdline_opts_.script_file << "'\n";
 396                         error = true;
 397                 }
 398         }
 399 
 400         if (cmdline_opts_.plugin_file)
 401         {
 402                 std::string filename = *cmdline_opts_.plugin_file;
 403 
 404                 std::cerr << "Loading a plugin file'" << filename << "'...\n";
 405 
 406                 filesystem::scoped_istream sf = filesystem::istream_file(filename);
 407 
 408                 try {
 409                         if (sf->fail()) {
 410                                 throw std::runtime_error("failed to open plugin file");
 411                         }
 412 
 413                         /* Cancel all "jumps" to editor / campaign / multiplayer */
 414                         jump_to_multiplayer_ = false;
 415                         jump_to_editor_ = false;
 416                         jump_to_campaign_.jump_ = false;
 417 
 418                         std::string full_plugin((std::istreambuf_iterator<char>(*sf)), std::istreambuf_iterator<char>());
 419 
 420                         plugins_manager & pm = *plugins_manager::get();
 421 
 422                         size_t i = pm.add_plugin(filename, full_plugin);
 423 
 424                         for (size_t j = 0 ; j < pm.size(); ++j) {
 425                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 426                         }
 427 
 428                         std::cerr << "Starting a plugin...\n";
 429                         pm.start_plugin(i);
 430 
 431                         for (size_t j = 0 ; j < pm.size(); ++j) {
 432                                 std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 433                         }
 434 
 435                         plugins_context pc("init");
 436 
 437                         for (size_t repeat = 0; repeat < 5; ++repeat) {
 438                                 std::cerr << "Playing a slice...\n";
 439                                 pc.play_slice();
 440 
 441                                 for (size_t j = 0 ; j < pm.size(); ++j) {
 442                                         std::cerr << j << ": " << pm.get_name(j) << " -- " << pm.get_detailed_status(j) << std::endl;
 443                                 }
 444                         }
 445 
 446                         return true;
 447                 } catch (std::exception & e) {
 448                         gui2::show_error_message(video(), std::string("When loading a plugin, error:\n") + e.what());
 449                         error = true;
 450                 }
 451         }
 452 
 453         if (!error) {
 454                 std::cerr << "ok\n";
 455         }
 456 
 457         return !error;
 458 }

 474 bool game_launcher::play_test()
 475 {
 476         static bool first_time = true;
 477

 482                 return false;

 497                 return true;
 498         }
 499 
 500         return false;

 503 // Same as play_test except that we return the results of play_game.
 504 int game_launcher::unit_test()
 505 {
 506         static bool first_time_unit = true;
 507 
 508         if(!cmdline_opts_.unit_test) {
 509                 return 0;
 510         }
 511         if(!first_time_unit)
 512                 return 0;
 513 
 514         first_time_unit = false;
 515 
 516         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::TEST;
 517         state_.classification().campaign_define = "TEST";
 518         state_.set_carryover_sides_start(
 519                 config {"next_scenario", test_scenario_}
 520         );
 521 
 522 
 523         game_config_manager::get()->
 524                 load_game_config_for_game(state_.classification());
 525 
 526         try {
 527                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 528                 LEVEL_RESULT res = ccontroller.play_game();
 529                 if (!(res == LEVEL_RESULT::VICTORY) || lg::broke_strict()) {
 530                         return 1;
 531                 }
 532         } catch(wml_exception& e) {
 533                 std::cerr << "Caught WML Exception:" << e.dev_message << std::endl;
 534                 return 1;
 535         }
 536 
 537         savegame::clean_saves(state_.classification().label);
 538 
 539         if (cmdline_opts_.noreplaycheck)
 540                 return 0; //we passed, huzzah!
 541 
 542         savegame::replay_savegame save(state_, compression::NONE);
 543         save.save_game_automatic(video(), false, "unit_test_replay"); //false means don't check for overwrite
 544 
 545         load_data_.reset(new savegame::load_game_metadata{ "unit_test_replay" , "", true, true, false });
 546 
 547         if (!load_game()) {
 548                 std::cerr << "Failed to load the replay!" << std::endl;
 549                 return 3; //failed to load replay
 550         }
 551 
 552         try {
 553                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types(), true);
 554                 LEVEL_RESULT res = ccontroller.play_replay();
 555                 if (!(res == LEVEL_RESULT::VICTORY)) {
 556                         std::cerr << "Observed failure on replay" << std::endl;
 557                         return 4;
 558                 }
 559         } catch(wml_exception& e) {
 560                 std::cerr << "WML Exception while playing replay: " << e.dev_message << std::endl;
 561                 return 4; //failed with an error during the replay
 562         }
 563 
 564         return 0; //we passed, huzzah!
 565 }

 600         std::string outfile = "wesnoth_image.bmp";
 601 
 602         // If a output path was given as an argument, use that instead
 603         if (cmdline_opts_.render_image_dst) {
 604                 outfile = *cmdline_opts_.render_image_dst;
 605         }
 606 
 607         if (!image::save_image(*cmdline_opts_.render_image, outfile)) {
 608                 exit(1);
 609         }
 610 
 611         return false;
 612 }

 619 bool game_launcher::load_game()
 620 {
 621         assert(game_config_manager::get());
 622 
 623         DBG_GENERAL << "Current campaign type: " << state_.classification().campaign_type << std::endl;
 624 
 625         savegame::loadgame load(video(), game_config_manager::get()->game_config(), state_);
 626         if (load_data_) {
 627                 std::unique_ptr<savegame::load_game_metadata> load_data = std::move(load_data_);
 628                 load.data() = std::move(*load_data);
 629         }
 630 
 631         try {
 632                 if(!load.load_game()) {
 633                         return false;
 634                 }
 635 
 636                 try {
 637                         game_config_manager::get()->
 638                                 load_game_config_for_game(state_.classification());
 639                 } catch(config::error&) {
 640                         return false;
 641                 }
 642 
 643                 load.set_gamestate();
 644 
 645         } catch(config::error& e) {
 646                 if(e.message.empty()) {
 647                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 648                 }
 649                 else {
 650                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 651                 }
 652                 return false;
 653         } catch(wml_exception& e) {
 654                 e.show(video());
 655                 return false;
 656         } catch(filesystem::io_exception& e) {
 657                 if(e.message.empty()) {
 658                         gui2::show_error_message(video(), _("File I/O Error while reading the game"));
 659                 } else {
 660                         gui2::show_error_message(video(), _("File I/O Error while reading the game: '") + e.message + '\'');
 661                 }
 662                 return false;
 663         } catch(game::error& e) {
 664                 if(e.message.empty()) {
 665                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt"));
 666                 }
 667                 else {
 668                         gui2::show_error_message(video(), _("The file you have tried to load is corrupt: '") + e.message + '\'');
 669                 }
 670                 return false;
 671         }
 672 
 673         play_replay_ = load.data().show_replay;
 674         LOG_CONFIG << "is middle game savefile: " << (state_.is_mid_game_save() ? "yes" : "no") << "\n";
 675         LOG_CONFIG << "show replay: " << (play_replay_ ? "yes" : "no") << "\n";
 676         // in case load.data().show_replay && !state_.is_mid_game_save()
 677         // there won't be any turns to replay, but the
 678         // user gets to watch the intro sequence again ...
 679 
 680         if(state_.is_mid_game_save() && load.data().show_replay)
 681         {
 682                 statistics::clear_current_scenario();
 683         }
 684 
 685         if(state_.classification().campaign_type == game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 686                 state_.unify_controllers();
 687         }
 688 
 689         if (load.data().cancel_orders) {
 690                 state_.cancel_orders();
 691         }
 692 
 693         return true;
 694 }

 773 void game_launcher::start_wesnothd()
 774 {
 775         const std::string wesnothd_program =
 776                 preferences::get_mp_server_program_name().empty() ?
 777                 filesystem::get_program_invocation("wesnothd") : preferences::get_mp_server_program_name();
 778 
 779         std::string config = filesystem::get_user_config_dir() + "/lan_server.cfg";
 780         if (!filesystem::file_exists(config)) {
 781                 // copy file if it isn't created yet
 782                 filesystem::write_file(config, filesystem::read_file(filesystem::get_wml_location("lan_server.cfg")));
 783         }
 784 
 785 #ifndef _WIN32
 786         std::string command = "\"" + wesnothd_program +"\" -c \"" + config + "\" -d -t 2 -T 5";
 787 #else
 788         // start wesnoth as background job
 789         std::string command = "cmd /C start \"wesnoth server\" /B \"" + wesnothd_program + "\" -c \"" + config + "\" -t 2 -T 5";
 790         // Make sure wesnothd's console output is visible on the console window by
 791         // disabling SDL's stdio redirection code for this and future child
 792         // processes. No need to bother cleaning this up because it's only
 793         // meaningful to SDL applications during pre-main initialization.
 794         SetEnvironmentVariableA("SDL_STDIO_REDIRECT", "0");
 795 #endif
 796         LOG_GENERAL << "Starting wesnothd: "<< command << "\n";
 797         if (std::system(command.c_str()) == 0) {
 798                 // Give server a moment to start up
 799                 SDL_Delay(50);
 800                 return;
 801         }
 802         preferences::set_mp_server_program_name("");
 803 
 804         // Couldn't start server so throw error
 805         WRN_GENERAL << "Failed to run server start script" << std::endl;
 806         throw game::mp_server_error("Starting MP server failed!");
 807 }

 809 bool game_launcher::play_multiplayer(mp_selection res)
 810 {
 811         state_ = saved_game();
 812         state_.classification().campaign_type = game_classification::CAMPAIGN_TYPE::MULTIPLAYER;
 813 
 814         try {
 815                 if (res == MP_HOST)
 816                 {
 817                         try {
 818                                 start_wesnothd();
 819                         } catch(game::mp_server_error&)
 820                         {
 821                                 preferences::show_wesnothd_server_search(video());
 822 
 823                                 try {
 824                                         start_wesnothd();
 825                                 } catch(game::mp_server_error&)
 826                                 {
 827                                         return false;
 828                                 }
 829                         }
 830 
 831 
 832                 }
 833                 //create_engine already calls game_config_manager::get()->load_config but maybe its better to have MULTIPLAYER defined while we are in the lobby.
 834                 game_config_manager::get()->load_game_config_for_create(true);
 835 
 836                 events::discard_input(); // prevent the "keylogger" effect
 837                 cursor::set(cursor::NORMAL);
 838 
 839                 if(res == MP_LOCAL) {
 840                         mp::start_local_game(video(),
 841                             game_config_manager::get()->game_config(), state_);
 842                 } else {
 843                         mp::start_client(video(), game_config_manager::get()->game_config(),
 844                                 state_, multiplayer_server_);
 845                         multiplayer_server_.clear();
 846                 }
 847 
 848         } catch(game::mp_server_error& e) {
 849                 gui2::show_error_message(video(), _("Error while starting server: ") + e.message);
 850         } catch(game::load_game_failed& e) {
 851                 gui2::show_error_message(video(), _("The game could not be loaded: ") + e.message);
 852         } catch(game::game_error& e) {
 853                 gui2::show_error_message(video(), _("Error while playing the game: ") + e.message);
 854         } catch (mapgen_exception& e) {
 855                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 856         } catch(wesnothd_error& e) {
 857                 if(e.message != "") {
 858                         ERR_NET << "caught network error: " << e.message << std::endl;
 859                         gui2::show_transient_message(video()
 860                                         , ""
 861                                         , translation::gettext(e.message.c_str()));
 862                 } else {
 863                         ERR_NET << "caught network error" << std::endl;
 864                 }
 865         } catch(config::error& e) {
 866                 if(e.message != "") {
 867                         ERR_CONFIG << "caught config::error: " << e.message << std::endl;
 868                         gui2::show_transient_message(video(), "", e.message);
 869                 } else {
 870                         ERR_CONFIG << "caught config::error" << std::endl;
 871                 }
 872         } catch(incorrect_map_format_error& e) {
 873                 gui2::show_error_message(video(), _("The game map could not be loaded: ") + e.message);
 874         } catch (savegame::load_game_exception & e) {
 875                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 876                 //this will make it so next time through the title screen loop, this game is loaded
 877         } catch(wml_exception& e) {
 878                 e.show(video());
 879         } catch (game::error & e) {
 880                 std::cerr << "caught game::error...\n";
 881                 gui2::show_error_message(video(), _("Error: ") + e.message);
 882         }
 883 
 884         return false;
 885 }

 929 void game_launcher::launch_game(RELOAD_GAME_DATA reload)
 930 {
 931         assert(!load_data_);
 932         if(play_replay_)
 933         {
 934                 play_replay();
 935                 return;
 936         }
 937 
 938         gui2::dialogs::loading_screen::display(video(), [this, reload]() {
 939 
 940                 gui2::dialogs::loading_screen::progress("load data");
 941                 if(reload == RELOAD_DATA) {
 942                         try {
 943                                 game_config_manager::get()->
 944                                         load_game_config_for_game(state_.classification());
 945                         } catch(config::error&) {
 946                                 return;
 947                         }
 948                 }
 949         });
 950 
 951         try {
 952                 campaign_controller ccontroller(video(), state_, game_config_manager::get()->game_config(), game_config_manager::get()->terrain_types());
 953                 LEVEL_RESULT result = ccontroller.play_game();
 954                 // don't show The End for multiplayer scenario
 955                 // change this if MP campaigns are implemented
 956                 if(result == LEVEL_RESULT::VICTORY && !state_.classification().is_normal_mp_game()) {
 957                         preferences::add_completed_campaign(state_.classification().campaign, state_.classification().difficulty);
 958 
 959                         gui2::dialogs::outro::display(state_.classification().end_text, state_.classification().end_text_duration, video());
 960 
 961                         if(state_.classification().end_credits) {
 962                                 gui2::dialogs::end_credits::display(video(), state_.classification().campaign);
 963                         }
 964                 }
 965         } catch (savegame::load_game_exception &e) {
 966                 load_data_.reset(new savegame::load_game_metadata(std::move(e.data_)));
 967                 //this will make it so next time through the title screen loop, this game is loaded
 968         } catch(wml_exception& e) {
 969                 e.show(video());
 970         } catch(mapgen_exception& e) {
 971                 gui2::show_error_message(video(), _("Map generator error: ") + e.message);
 972         }
 973 }

 989 editor::EXIT_STATUS game_launcher::start_editor(const std::string& filename)
 990 {
 991         while(true){
 992                 game_config_manager::get()->load_game_config_for_editor();
 993 
 994                 ::init_textdomains(game_config_manager::get()->game_config());
 995 
 996                 editor::EXIT_STATUS res = editor::start(
 997                     game_config_manager::get()->game_config(), video(), filename);
 998 
 999                 if(res != editor::EXIT_RELOAD_DATA)
1000                         return res;
1001 
1002                 game_config_manager::get()->reload_changed_game_config();
1003                 image::flush_cache();
1004         }
1005         return editor::EXIT_ERROR; // not supposed to happen
1006 }
