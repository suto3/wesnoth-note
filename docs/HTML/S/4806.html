   3 # During the plot of this campaign Delfador changes recall lists more
   4 # often than most Wesnoth characters change their underwear.  This file
   5 # exists to encapsulate all that machinery and the comments explaining
   6 # what it does.
   7 
   8 # That means: *ALL* recall list manipulation should be done *HERE*!
   9 # The point is to have it in one place so the complexity can be audited.

  11 #define MEMOIRS_STORE_MATCHING_UNITS WML VAR
  12     # Store filtered units on Defador's side, except him, into an array variable
  13     [store_unit]
  14         kill=yes
  15         variable={VAR}
  16         [filter]
  17             side=1
  18             [and]
  19                 [not]
  20                     id=Delfador
  21                 [/not]
  22             [/and]
  23             {WML}
  24         [/filter]
  25     [/store_unit]
  26 #enddef
  27 
  28 #define MEMOIRS_STORE_UNITS VAR
  29     # Store all units on Delfador's side, except him, into an array variable
  30     {MEMOIRS_STORE_MATCHING_UNITS () {VAR}}
  31 #enddef
  32 
  33 #define MEMOIRS_UNSTORE_UNITS VAR
  34     # Unstore units from an array back into existence
  35     [foreach]
  36         array={VAR}
  37         [do]
  38             [unstore_unit]
  39                 variable=this_item
  40                 find_vacant=yes
  41                 x=recall
  42                 y=recall
  43             [/unstore_unit]
  44             [heal_unit]
  45                 [filter]
  46                     id=$this_item.id
  47                 [/filter]
  48                 amount=999
  49             [/heal_unit]
  50         [/do]
  51     [/foreach]
  52 
  53     [clear_variable]
  54         name={VAR}
  55     [/clear_variable]
  56 #enddef

  58 #define SUBSUME_SIDE_AND_VILLAGES SIDE
  59     # first change village ownership
  60     [store_villages]
  61         owner_side={SIDE}
  62 
  63         variable=side_villages
  64     [/store_villages]
  65     [foreach]
  66         array=side_villages
  67         [do]
  68             [capture_village]
  69                 side=1
  70                 x,y=$this_item.x,$this_item.y
  71             [/capture_village]
  72         [/do]
  73     [/foreach]
  74     {CLEAR_VARIABLE side_villages}
  75 
  76     [modify_unit]
  77         [filter]
  78             side={SIDE}
  79         [/filter]
  80         moves=$this_unit.max_moves
  81     [/modify_unit]
  82     #next, turn off canrecruit for the leader and change all allegiances.
  83     {MODIFY_UNIT (side={SIDE}
  84     canrecruit=yes) canrecruit no}
  85     {MODIFY_UNIT side={SIDE} side 1}
  86 
  87     [modify_side]
  88         side={SIDE}
  89         hidden=yes
  90     [/modify_side]
  91     [redraw]
  92         side=1
  93     [/redraw]
  94 #enddef
