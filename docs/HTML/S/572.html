  77 void multi_page::remove_page(const unsigned page, unsigned count)
  78 {
  79         assert(generator_);
  80 
  81         if(page >= get_page_count()) {
  82                 return;
  83         }
  84 
  85         if(!count || count > get_page_count()) {
  86                 count = get_page_count();
  87         }
  88 
  89         for(; count; --count) {
  90                 generator_->delete_item(page);
  91         }
  92 }

 255 builder_multi_page::builder_multi_page(const config& cfg)
 256         : implementation::builder_styled_widget(cfg), builders(), data()
 257 {
 258         for (const config& page : cfg.child_range("page_definition"))
 259         {
 260                 auto builder = std::make_shared<builder_grid>(page);
 261                 assert(builder);
 262                 builders[page["id"]] = builder;
 263         }
 264         VALIDATE(!builders.empty(), _("No page defined."));
 265 
 266         /** @todo This part is untested. */
 267         const config& d = cfg.child("page_data");
 268         if(!d) {
 269                 return;
 270         }
 271 
 272         auto builder = builders.begin()->second;
 273         for(const auto & row : d.child_range("row"))
 274         {
 275                 unsigned col = 0;
 276 
 277                 for(const auto & column : row.child_range("column"))
 278                 {
 279                         data.emplace_back();
 280                         for(const auto & i : column.attribute_range())
 281                         {
 282                                 data.back()[i.first] = i.second;
 283                         }
 284                         ++col;
 285                 }
 286 
 287                 VALIDATE(col == builder->cols,
 288                                  _("'list_data' must have "
 289                                    "the same number of columns as the 'list_definition'."));
 290         }
 291 }

 293 widget* builder_multi_page::build() const
 294 {
 295         multi_page* widget = new multi_page();
 296 
 297         init_control(widget);
 298 
 299         widget->set_page_builders(builders);
 300 
 301         DBG_GUI_G << "Window builder: placed multi_page '" << id
 302                           << "' with definition '" << definition << "'.\n";
 303 
 304         std::shared_ptr<const multi_page_definition::resolution>
 305         conf = std::static_pointer_cast<const multi_page_definition::resolution>(
 306                                         widget->config());
 307         assert(conf);
 308 
 309         widget->init_grid(conf->grid);
 310 
 311         widget->finalize(data);
 312 
 313         return widget;
 314 }
