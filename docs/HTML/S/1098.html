  59 static void verify(const unit_map& units, const config& cfg) {
  60         std::stringstream errbuf;
  61         LOG_REPLAY << "verifying unit structure...\n";
  62 
  63         const size_t nunits = cfg["num_units"].to_size_t();
  64         if(nunits != units.size()) {
  65                 errbuf << "SYNC VERIFICATION FAILED: number of units from data source differ: "
  66                            << nunits << " according to data source. " << units.size() << " locally\n";
  67 
  68                 std::set<map_location> locs;
  69                 for (const config &u : cfg.child_range("unit"))
  70                 {
  71                         const map_location loc(u);
  72                         locs.insert(loc);
  73 
  74                         if(units.count(loc) == 0) {
  75                                 errbuf << "data source says there is a unit at "
  76                                            << loc << " but none found locally\n";
  77                         }
  78                 }
  79 
  80                 for(unit_map::const_iterator j = units.begin(); j != units.end(); ++j) {
  81                         if (locs.count(j->get_location()) == 0) {
  82                                 errbuf << "local unit at " << j->get_location()
  83                                            << " but none in data source\n";
  84                         }
  85                 }
  86                 replay::process_error(errbuf.str());
  87                 errbuf.clear();
  88         }
  89 
  90         for (const config &un : cfg.child_range("unit"))
  91         {
  92                 const map_location loc(un);
  93                 const unit_map::const_iterator u = units.find(loc);
  94                 if(u == units.end()) {
  95                         errbuf << "SYNC VERIFICATION FAILED: data source says there is a '"
  96                                    << un["type"] << "' (side " << un["side"] << ") at "
  97                                    << loc << " but there is no local record of it\n";
  98                         replay::process_error(errbuf.str());
  99                         errbuf.clear();
 100                 }
 101 
 102                 config u_cfg;
 103                 u->write(u_cfg);
 104 
 105                 bool is_ok = true;
 106                 static const std::string fields[] {"type","hitpoints","experience","side",""};
 107                 for(const std::string* str = fields; str->empty() == false; ++str) {
 108                         if (u_cfg[*str] != un[*str]) {
 109                                 errbuf << "ERROR IN FIELD '" << *str << "' for unit at "
 110                                            << loc << " data source: '" << un[*str]
 111                                            << "' local: '" << u_cfg[*str] << "'\n";
 112                                 is_ok = false;
 113                         }
 114                 }
 115 
 116                 if(!is_ok) {
 117                         errbuf << "(SYNC VERIFICATION FAILED)\n";
 118                         replay::process_error(errbuf.str());
 119                         errbuf.clear();
 120                 }
 121         }
 122 
 123         LOG_REPLAY << "verification passed\n";
 124 }

 126 static time_t get_time(const config &speak)
 127 {
 128         time_t time;
 129         if (!speak["time"].empty())
 130         {
 131                 std::stringstream ss(speak["time"].str());
 132                 ss >> time;
 133         }
 134         else
 135         {
 136                 //fallback in case sender uses wesnoth that doesn't send timestamps
 137                 time = ::time(nullptr);
 138         }
 139         return time;
 140 }

 387 config replay::get_data_range(int cmd_start, int cmd_end, DATA_TYPE data_type) const
 388 {
 389         config res;
 390 
 391         for (int cmd = cmd_start; cmd < cmd_end; ++cmd)
 392         {
 393                 config &c = command(cmd);
 394                 //prevent creating 'blank' attribute values during checks
 395                 const config &cc = c;
 396                 if ((data_type == ALL_DATA || !cc["undo"].to_bool(true)) && !cc["sent"].to_bool(false))
 397                 {
 398                         res.add_child("command", c);
 399                         if (data_type == NON_UNDO_DATA) c["sent"] = true;

 443 /// fixes a rename command when undoing a earlier command.
 444 /// @return: true if the command should be removed.
 445 static bool fix_rename_command(const config& c, config& async_child)
 446 {
 447         if (const config &child = c.child("move"))
 448         {
 449                 // A unit's move is being undone.
 450                 // Repair unsynced cmds whose locations depend on that unit's location.
 451                 std::vector<map_location> steps;
 452 
 453                 try {
 454                         read_locations(child,steps);
 455                 } catch (bad_lexical_cast &) {
 456                         WRN_REPLAY << "Warning: Path data contained something which could not be parsed to a sequence of locations:" << "\n config = " << child.debug() << std::endl;
 457                 }
 458 
 459                 if (steps.empty()) {
 460                         ERR_REPLAY << "trying to undo a move using an empty path";
 461                 }
 462                 else {
 463                         const map_location &src = steps.front();
 464                         const map_location &dst = steps.back();
 465                         map_location aloc(async_child);
 466                         if (dst == aloc) src.write(async_child);
 467                 }
 468         }
 469         else
 470         {
 471                 const config *chld = &c.child("recruit");
 472                 if (!*chld) chld = &c.child("recall");
 473                 if (*chld) {
 474                         // A unit is being un-recruited or un-recalled.
 475                         // Remove unsynced commands that would act on that unit.
 476                         map_location src(*chld);
 477                         map_location aloc(async_child);
 478                         if (src == aloc) {
 479                                 return true;
 480                         }
 481                 }
 482         }
 483         return false;
 484 }

 486 void replay::undo_cut(config& dst)
 487 {
 488         assert(dst.empty());
 489         //assert that we are not undoing a command which we didn't execute yet.
 490         assert(at_end());
 491 
 492         //calculate the index of the last synced user action (which we want to undo).
 493         int cmd_index = ncommands() - 1;
 494         for (; cmd_index >= 0; --cmd_index)
 495         {
 496                 //"undo"=no means speak/label/remove_label, especialy attack, recruits etc. have "undo"=yes
 497                 //"async"=yes means rename_unit
 498                 //"dependent"=true means user input
 499                 const config &c = command(cmd_index);
 500 
 501                 if(c["undo"].to_bool(true) && !c["async"].to_bool(false) && !c["dependent"].to_bool(false))
 502                 {
 503                         if(c["sent"].to_bool(false))
 504                         {
 505                                 ERR_REPLAY << "trying to undo a command that was already sent.\n";
 506                                 return;
 507                         }
 508                         else
 509                         {
 510                                 break;
 511                         }
 512                 }
 513         }
 514 
 515         if (cmd_index < 0)
 516         {
 517                 ERR_REPLAY << "trying to undo a command but no command was found.\n";
 518                 return;
 519         }
 520         //Fix the [command]s after the undone action. This includes dependent commands for that user actions and async user action.
 521         for(int i = ncommands() - 1; i >= cmd_index; --i)
 522         {
 523                 config &c = command(i);
 524                 const config &cc = c;
 525                 if(!cc["undo"].to_bool(true))
 526                 {
 527                         //Leave these commands on the replay.
 528                 }
 529                 else if(cc["async"].to_bool(false))
 530                 {
 531                         if(config& rename = c.child("rename"))
 532                         {
 533                                 if(fix_rename_command(command(cmd_index), rename))
 534                                 {
 535                                         //remove the command from the replay if fix_rename_command requested it.
 536                                         remove_command(i);
 537                                 }
 538                         }
 539                 }
 540                 else if(cc["dependent"].to_bool(false) || i == cmd_index)
 541                 {
 542                         //we loop backwars so we must insert new insert at beginning to preserve order.
 543                         dst.add_child_at("command", config(), 0).swap(c);
 544                         remove_command(i);
 545                 }
 546                 else
 547                 {
 548                         ERR_REPLAY << "Coudn't handle command:\n" << cc << "\nwhen undoing.\n";
 549                 }
 550         }
 551         set_to_end();
 552 }

 663 REPLAY_RETURN do_replay(bool one_move)
 664 {
 665         log_scope("do replay");
 666 
 667         if (!resources::controller->is_skipping_replay()) {
 668                 resources::screen->recalculate_minimap();
 669         }
 670 
 671         update_locker lock_update(resources::screen->video(), resources::controller->is_skipping_replay());
 672         return do_replay_handle(one_move);
 673 }
 674 /**
 675         @returns:
 676                 if we expect a user choice and found something that prevents us from moving on we return REPLAY_FOUND_DEPENDENT (even if it is not a dependent command)
 677                 else if we found an [end_turn] we return REPLAY_FOUND_END_TURN
 678                 else if we found a player action and one_move=true we return REPLAY_FOUND_END_MOVE
 679                 else (<=> we reached the end of the replay) we return REPLAY_RETURN_AT_END
 680 */
 681 REPLAY_RETURN do_replay_handle(bool one_move)
 682 {
 683 
 684         //team &current_team = resources::gameboard->get_team(side_num);
 685 
 686         const int side_num = resources::controller->current_side();
 687         while(true)
 688         {
 689                 const config *cfg = resources::recorder->get_next_action();
 690                 const bool is_synced = synced_context::is_synced();
 691 
 692                 DBG_REPLAY << "in do replay with is_synced=" << is_synced << "\n";
 693 
 694                 if (cfg != nullptr)
 695                 {
 696                         DBG_REPLAY << "Replay data:\n" << *cfg << "\n";
 697                 }
 698                 else
 699                 {
 700                         DBG_REPLAY << "Replay data at end\n";
 701                         return REPLAY_RETURN_AT_END;
 702                 }
 703 
 704 
 705                 const config::const_all_children_itors ch_itors = cfg->all_children_range();
 706                 //if there is an empty command tag or a start tag
 707                 if (ch_itors.empty() || cfg->has_child("start"))
 708                 {
 709                         //this shouldn't happen anymore because replaycontroller now moves over the [start] with get_next_action
 710                         //also we removed the the "add empty replay entry at scenario reload" behavior.
 711                         ERR_REPLAY << "found "<<  cfg->debug() <<" in replay" << std::endl;
 712                         //do nothing
 713                 }
 714                 else if (const config &speak = cfg->child("speak"))
 715                 {
 716                         const std::string &team_name = speak["team_name"];
 717                         const std::string &speaker_name = speak["id"];
 718                         const std::string &message = speak["message"];
 719                         //if (!preferences::parse_should_show_lobby_join(speaker_name, message)) return;
 720                         bool is_whisper = (speaker_name.find("whisper: ") == 0);
 721                         if(resources::recorder->add_chat_message_location()) {
 722                                 DBG_REPLAY << "tried to add a chat message twice.\n";
 723                                 if (!resources::controller->is_skipping_replay() || is_whisper) {
 724                                         int side = speak["side"];
 725                                         resources::screen->get_chat_manager().add_chat_message(get_time(speak), speaker_name, side, message,
 726                                                 (team_name.empty() ? events::chat_handler::MESSAGE_PUBLIC
 727                                                 : events::chat_handler::MESSAGE_PRIVATE),
 728                                                 preferences::message_bell());
 729                                 }
 730                         }
 731                 }
 732                 else if (const config &label_config = cfg->child("label"))
 733                 {
 734                         terrain_label label(resources::screen->labels(), label_config);
 735 
 736                         resources::screen->labels().set_label(label.location(),
 737                                                 label.text(),
 738                                                 label.creator(),
 739                                                 label.team_name(),
 740                                                 label.color());
 741                 }
 742                 else if (const config &clear_labels = cfg->child("clear_labels"))
 743                 {
 744                         resources::screen->labels().clear(std::string(clear_labels["team_name"]), clear_labels["force"].to_bool());
 745                 }
 746                 else if (const config &rename = cfg->child("rename"))
 747                 {
 748                         const map_location loc(rename);
 749                         const std::string &name = rename["name"];
 750 
 751                         unit_map::iterator u = resources::gameboard->units().find(loc);
 752                         if (u.valid() && !u->unrenamable()) {
 753                                 u->rename(name);
 754                         } else {
 755                                 // Users can rename units while it's being killed or at another machine.
 756                                 // This since the player can rename units when it's not his/her turn.
 757                                 // There's not a simple way to prevent that so in that case ignore the
 758                                 // rename instead of throwing an OOS.
 759                                 // The same way it is possible that an unrenamable unit moves to a
 760                                 // hex where previously a renamable unit was.
 761                                 WRN_REPLAY << "attempt to rename unit at location: "
 762                                    << loc << (u.valid() ? ", which is unrenamable" : ", where none exists (anymore)") << "\n";
 763                         }
 764                 }
 765 
 766                 else if (cfg->child("init_side"))
 767                 {
 768 
 769                         if(is_synced)
 770                         {
 771                                 replay::process_error("found side initialization in replay expecting a user choice\n" );
 772                                 resources::recorder->revert_action();
 773                                 return REPLAY_FOUND_DEPENDENT;
 774                         }
 775                         else
 776                         {
 777                                 resources::controller->do_init_side();
 778                                 if (one_move) {
 779                                         return REPLAY_FOUND_INIT_TURN;
 780                                 }
 781                         }
 782                 }
 783 
 784                 //if there is an end turn directive
 785                 else if (cfg->child("end_turn"))
 786                 {
 787                         if(is_synced)
 788                         {
 789                                 replay::process_error("found turn end in replay while expecting a user choice\n" );
 790                                 resources::recorder->revert_action();
 791                                 return REPLAY_FOUND_DEPENDENT;
 792                         }
 793                         else
 794                         {
 795                                 if (const config &cfg_verify = cfg->child("verify")) {
 796                                         verify(resources::gameboard->units(), cfg_verify);
 797                                 }
 798 
 799                                 return REPLAY_FOUND_END_TURN;
 800                         }
 801                 }
 802                 else if (const config &countdown_update = cfg->child("countdown_update"))
 803                 {
 804                         int val = countdown_update["value"];
 805                         int tval = countdown_update["team"];
 806                         if (tval <= 0  || tval > int(resources::gameboard->teams().size())) {
 807                                 std::stringstream errbuf;
 808                                 errbuf << "Illegal countdown update \n"
 809                                         << "Received update for :" << tval << " Current user :"
 810                                         << side_num << "\n" << " Updated value :" << val;
 811 
 812                                 replay::process_error(errbuf.str());
 813                         } else {
 814                                 resources::gameboard->get_team(tval).set_countdown_time(val);
 815                         }
 816                 }
 817                 else if ((*cfg)["dependent"].to_bool(false))
 818                 {
 819                         if(!is_synced)
 820                         {
 821                                 replay::process_error("found dependent command in replay while is_synced=false\n" );
 822                                 //ignore this command
 823                                 continue;
 824                         }
 825                         //this means user choice.
 826                         // it never makes sense to try to execute a user choice.
 827                         // but we are called from
 828                         // the only other option for "dependent" command is checksum wich is already checked.
 829                         assert(cfg->all_children_count() == 1);
 830                         std::string child_name = cfg->all_children_range().front().key;
 831                         DBG_REPLAY << "got an dependent action name = " << child_name <<"\n";
 832                         resources::recorder->revert_action();
 833                         return REPLAY_FOUND_DEPENDENT;
 834                 }
 835                 else
 836                 {
 837                         //we checked for empty commands at the beginning.
 838                         const std::string & commandname = cfg->ordered_begin()->key;
 839                         config data = cfg->ordered_begin()->cfg;
 840 
 841                         if(is_synced)
 842                         {
 843                                 replay::process_error("found [" + commandname + "] command in replay expecting a user choice\n" );
 844                                 resources::recorder->revert_action();
 845                                 return REPLAY_FOUND_DEPENDENT;
 846                         }
 847                         else
 848                         {
 849                                 LOG_REPLAY << "found commandname " << commandname << "in replay";
 850 
 851                                 if((*cfg)["from_side"].to_int(0) != resources::controller->current_side()) {
 852                                         ERR_REPLAY << "recieved a synced [command] from side " << (*cfg)["from_side"].to_int(0) << ". Expacted was a [command] from side " << resources::controller->current_side() << "\n";
 853                                 }
 854                                 else if((*cfg)["side_invalid"].to_bool(false)) {
 855                                         ERR_REPLAY << "recieved a synced [command] from side " << (*cfg)["from_side"].to_int(0) << ". Sent from wrong client.\n";
 856                                 }
 857                                 /*
 858                                         we need to use the undo stack during replays in order to make delayed shroud updated work.
 859                                 */
 860                                 synced_context::run(commandname, data, true, !resources::controller->is_skipping_replay(), show_oos_error_error_function);
 861                                 if(resources::controller->is_regular_game_end()) {
 862                                         return REPLAY_FOUND_END_LEVEL;
 863                                 }
 864                                 if (one_move) {
 865                                         return REPLAY_FOUND_END_MOVE;
 866                                 }
 867                         }
 868                 }
 869 
 870                 if (const config &child = cfg->child("verify")) {
 871                         verify(resources::gameboard->units(), child);
 872                 }
 873         }
 874 }

 907
