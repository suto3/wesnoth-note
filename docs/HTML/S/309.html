  51 static map_location legacy_negation(const map_location& me)
  52 {
  53         return map_location(-me.x, -me.y);
  54 }

  56 static map_location& legacy_sum_assign(map_location& me, const map_location& a)
  57 {
  58         bool parity = (me.x & 1) != 0;
  59         me.x += a.x;
  60         me.y += a.y;
  61         if((a.x > 0) && (a.x % 2) && parity)
  62                 me.y++;
  63         if((a.x < 0) && (a.x % 2) && !parity)
  64                 me.y--;
  65 
  66         return me;
  67 }

  69 static map_location legacy_sum(const map_location& me, const map_location& a)
  70 {
  71         map_location ret(me);
  72         legacy_sum_assign(ret, a);
  73         return ret;
  74 }

  76 static map_location legacy_difference(const map_location& me, const map_location& a)
  77 {
  78         return legacy_sum(me, legacy_negation(a));

 112 void terrain_builder::tile::rebuild_cache(const std::string& tod, logs* log)
 113 {
 114         images_background.clear();
 115         images_foreground.clear();
 116 
 117         if(!sorted_images) {
 118                 // sort images by their layer (and basey)
 119                 // but use stable to keep the insertion order in equal cases
 120                 std::stable_sort(images.begin(), images.end());
 121                 sorted_images = true;
 122         }
 123 
 124         for(const rule_image_rand& ri : images) {
 125                 bool is_background = ri->is_background();
 126                 bool animate = (!ri.ri->is_water || preferences::animate_water());
 127 
 128                 imagelist& img_list = is_background ? images_background : images_foreground;
 129 
 130                 for(const rule_image_variant& variant : ri->variants) {
 131                         if(!variant.has_flag.empty()) {
 132                                 bool has_flag_match = true;
 133                                 for(const std::string& s : variant.has_flag) {
 134                                         // If a flag listed in "has_flag" is not present, this variant does not match
 135                                         if(flags.find(s) == flags.end()) {
 136                                                 has_flag_match = false;
 137                                                 break;
 138                                         }
 139                                 }
 140 
 141                                 if(!has_flag_match) {
 142                                         continue;
 143                                 }
 144                         }
 145 
 146                         if(!variant.tods.empty() && variant.tods.find(tod) == variant.tods.end())
 147                                 continue;
 148 
 149                         // need to break parity pattern in RNG
 150                         /** @todo improve this */
 151                         unsigned int rnd = ri.rand / 7919; // just the 1000th prime
 152                         const animated<image::locator>& anim = variant.images[rnd % variant.images.size()];
 153 
 154                         bool is_empty = true;
 155                         for(size_t i = 0; i < anim.get_frames_count(); ++i) {
 156                                 if(!image::is_empty_hex(anim.get_frame(i))) {
 157                                         is_empty = false;
 158                                         break;
 159                                 }
 160                         }
 161 
 162                         if(is_empty)
 163                                 continue;
 164 
 165                         img_list.push_back(anim);
 166 
 167                         if(variant.random_start)
 168                                 img_list.back().set_animation_time(ri.rand % img_list.back().get_animation_duration());
 169 
 170                         if(!animate) {
 171                                 img_list.back().pause_animation();
 172                         }
 173 
 174                         if(log) {
 175                                 log->emplace_back(&ri, &variant);
 176                         }
 177 
 178                         break; // found a matching variant
 179                 }
 180         }
 181 }

 193 static unsigned int get_noise(const map_location& loc, unsigned int index)
 194 {
 195         unsigned int a = (loc.x + 92872973) ^ 918273;
 196         unsigned int b = (loc.y + 1672517) ^ 128123;
 197         unsigned int c = (index + 127390) ^ 13923787;
 198         unsigned int abc = a * b * c + a * b + b * c + a * c + a + b + c;
 199         return abc * abc;
 200 }

 263 }

 403 static std::vector<std::string> get_variations(const std::string& base, const std::string& variations)
 404 {
 405         /** @todo optimize this function */
 406         std::vector<std::string> res;
 407         if(variations.empty()) {
 408                 res.push_back(base);
 409                 return res;
 410         }
 411         std::string::size_type pos = base.find("@V", 0);
 412         if(pos == std::string::npos) {
 413                 res.push_back(base);
 414                 return res;
 415         }
 416         std::vector<std::string> vars = utils::split(variations, ';', 0);
 417 
 418         for(const std::string& v : vars) {
 419                 res.push_back(base);
 420                 pos = 0;
 421                 while((pos = res.back().find("@V", pos)) != std::string::npos) {
 422                         res.back().replace(pos, 2, v);
 423                         pos += v.size();
 424                 }
 425         }
 426         return res;
 427 }

 429 bool terrain_builder::load_images(building_rule& rule)
 430 {
 431         // If the rule has no constraints, it is invalid
 432         if(rule.constraints.empty())
 433                 return false;
 434 
 435         // Parse images and animations data
 436         // If one is not valid, return false.
 437         for(terrain_constraint& constraint : rule.constraints) {
 438                 for(rule_image& ri : constraint.images) {
 439                         for(rule_image_variant& variant : ri.variants) {
 440                                 std::vector<std::string> var_strings = get_variations(variant.image_string, variant.variations);
 441                                 for(const std::string& var : var_strings) {
 442                                         /** @todo improve this, 99% of terrains are not animated. */
 443                                         std::vector<std::string> frames = utils::square_parenthetical_split(var, ',');
 444                                         animated<image::locator> res;
 445 
 446                                         for(const std::string& frame : frames) {
 447                                                 const std::vector<std::string> items = utils::split(frame, ':');
 448                                                 const std::string& str = items.front();
 449 
 450                                                 const size_t tilde = str.find('~');
 451                                                 bool has_tilde = tilde != std::string::npos;
 452                                                 const std::string filename = "terrain/" + (has_tilde ? str.substr(0, tilde) : str);
 453 
 454                                                 if(!image_exists(filename)) {
 455                                                         continue; // ignore missing frames
 456                                                 }
 457 
 458                                                 const std::string modif = (has_tilde ? str.substr(tilde + 1) : "");
 459 
 460                                                 int time = 100;
 461                                                 if(items.size() > 1) {
 462                                                         try {
 463                                                                 time = std::stoi(items.back());
 464                                                         } catch(std::invalid_argument) {
 465                                                                 ERR_NG << "Invalid 'time' value in terrain image builder: " << items.back() << "\n";
 466                                                         }
 467                                                 }
 468                                                 image::locator locator;
 469                                                 if(ri.global_image) {
 470                                                         locator = image::locator(filename, constraint.loc, ri.center_x, ri.center_y, modif);
 471                                                 } else {
 472                                                         locator = image::locator(filename, modif);
 473                                                 }
 474                                                 res.add_frame(time, locator);
 475                                         }
 476                                         if(res.get_frames_count() == 0)
 477                                                 break; // no valid images, don't register it
 478 
 479                                         res.start_animation(0, true);
 480                                         variant.images.push_back(std::move(res));
 481                                 }
 482                                 if(variant.images.empty())
 483                                         return false; // no valid images, rule is invalid
 484                         }
 485                 }
 486         }
 487 
 488         return true;
 489 }

 491 void terrain_builder::rotate(terrain_constraint& ret, int angle)
 492 {
 493         static const struct
 494         {
 495                 int ii;
 496                 int ij;
 497                 int ji;
 498                 int jj;
 499         } rotations[6] {{1, 0, 0, 1}, {1, 1, -1, 0}, {0, 1, -1, -1}, {-1, 0, 0, -1}, {-1, -1, 1, 0}, {0, -1, 1, 1}};
 500 
 501         // The following array of matrices is intended to rotate the (x,y)
 502         // coordinates of a point in a wesnoth hex (and wesnoth hexes are not
 503         // regular hexes :) ).
 504         // The base matrix for a 1-step rotation with the wesnoth tile shape
 505         // is:
 506         //
 507         // r = s^-1 * t * s
 508         //
 509         // with s = [[ 1   0         ]
 510         //           [ 0   -sqrt(3)/2 ]]
 511         //
 512         // and t =  [[ -1/2       sqrt(3)/2 ]
 513         //           [ -sqrt(3)/2  1/2        ]]
 514         //
 515         // With t being the rotation matrix (pi/3 rotation), and s a matrix
 516         // that transforms the coordinates of the wesnoth hex to make them
 517         // those of a regular hex.
 518         //
 519         // (demonstration left as an exercise for the reader)
 520         //
 521         // So we have
 522         //
 523         // r = [[ 1/2  -3/4 ]
 524         //      [ 1    1/2  ]]
 525         //
 526         // And the following array contains I(2), r, r^2, r^3, r^4, r^5
 527         // (with r^3 == -I(2)), which are the successive rotations.
 528         static const struct
 529         {
 530                 double xx;
 531                 double xy;
 532                 double yx;
 533                 double yy;
 534         } xyrotations[6] {
 535                 { 1.,         0.,  0., 1.    },
 536                 { 1./2. , -3./4.,  1., 1./2. },
 537                 { -1./2., -3./4.,   1, -1./2.},
 538                 { -1.   ,     0.,  0., -1.   },
 539                 { -1./2.,  3./4., -1., -1./2.},
 540                 { 1./2. ,  3./4., -1., 1./2. },
 541         };
 542 
 543         assert(angle >= 0);
 544 
 545         angle %= 6;
 546 
 547         // Vector i is going from n to s, vector j is going from ne to sw.
 548         int vi = ret.loc.y - ret.loc.x / 2;
 549         int vj = ret.loc.x;
 550 
 551         int ri = rotations[angle].ii * vi + rotations[angle].ij * vj;
 552         int rj = rotations[angle].ji * vi + rotations[angle].jj * vj;
 553 
 554         ret.loc.x = rj;
 555         ret.loc.y = ri + (rj >= 0 ? rj / 2 : (rj - 1) / 2);
 556 
 557         for(rule_imagelist::iterator itor = ret.images.begin(); itor != ret.images.end(); ++itor) {
 558                 double vx, vy, rx, ry;
 559 
 560                 vx = double(itor->basex) - double(tilewidth_) / 2;
 561                 vy = double(itor->basey) - double(tilewidth_) / 2;
 562 
 563                 rx = xyrotations[angle].xx * vx + xyrotations[angle].xy * vy;
 564                 ry = xyrotations[angle].yx * vx + xyrotations[angle].yy * vy;
 565 
 566                 itor->basex = int(rx + tilewidth_ / 2);
 567                 itor->basey = int(ry + tilewidth_ / 2);
 568 
 569                 // std::cerr << "Rotation: from " << vx << ", " << vy << " to " << itor->basex <<
 570                 //      ", " << itor->basey << "\n";
 571         }
 572 }

 574 void terrain_builder::replace_rotate_tokens(std::string& s, int angle, const std::vector<std::string>& replacement)
 575 {
 576         std::string::size_type pos = 0;
 577         while((pos = s.find("@R", pos)) != std::string::npos) {
 578                 if(pos + 2 >= s.size())
 579                         return;
 580                 unsigned i = s[pos + 2] - '0' + angle;
 581                 if(i >= 6)
 582                         i -= 6;
 583                 if(i >= 6) {
 584                         pos += 2;
 585                         continue;
 586                 }
 587                 const std::string& r = replacement[i];
 588                 s.replace(pos, 3, r);
 589                 pos += r.size();
 590         }
 591 }

 627 void terrain_builder::rotate_rule(building_rule& ret, int angle, const std::vector<std::string>& rot)
 628 {
 629         if(rot.size() != 6) {
 630                 ERR_NG << "invalid rotations" << std::endl;
 631                 return;
 632         }
 633 
 634         for(terrain_constraint& cons : ret.constraints) {
 635                 rotate(cons, angle);
 636         }
 637 
 638         // Normalize the rotation, so that it starts on a positive location
 639         int minx = INT_MAX;
 640         int miny = INT_MAX;
 641 
 642         for(const terrain_constraint& cons : ret.constraints) {
 643                 minx = std::min<int>(cons.loc.x, minx);
 644                 miny = std::min<int>(2 * cons.loc.y + (cons.loc.x & 1), miny);
 645         }
 646 
 647         if((miny & 1) && (minx & 1) && (minx < 0))
 648                 miny += 2;
 649         if(!(miny & 1) && (minx & 1) && (minx > 0))
 650                 miny -= 2;
 651 
 652         for(terrain_constraint& cons : ret.constraints) {
 653                 legacy_sum_assign(cons.loc, map_location(-minx, -((miny - 1) / 2)));
 654         }
 655 
 656         replace_rotate_tokens(ret, angle, rot);
 657 }

 680 void terrain_builder::add_images_from_config(rule_imagelist& images, const config& cfg, bool global, int dx, int dy)
 681 {
 682         for(const config& img : cfg.child_range("image")) {
 683                 int layer = img["layer"];
 684 
 685                 int basex = tilewidth_ / 2 + dx, basey = tilewidth_ / 2 + dy;
 686                 if(const config::attribute_value* base_ = img.get("base")) {
 687                         std::vector<std::string> base = utils::split(*base_);
 688                         if(base.size() >= 2) {
 689                                 try {
 690                                         basex = std::stoi(base[0]);
 691                                         basey = std::stoi(base[1]);
 692                                 } catch(std::invalid_argument) {
 693                                         ERR_NG << "Invalid 'base' value in terrain image builder: " << base[0] << ", " << base[1] << "\n";
 694                                 }
 695                         }
 696                 }
 697 
 698                 int center_x = -1, center_y = -1;
 699                 if(const config::attribute_value* center_ = img.get("center")) {
 700                         std::vector<std::string> center = utils::split(*center_);
 701                         if(center.size() >= 2) {
 702                                 try {
 703                                         center_x = std::stoi(center[0]);
 704                                         center_y = std::stoi(center[1]);
 705                                 } catch(std::invalid_argument) {
 706                                         ERR_NG << "Invalid 'center' value in terrain image builder: " << center[0] << ", " << center[1]
 707                                                    << "\n";
 708                                 }
 709                         }
 710                 }
 711 
 712                 bool is_water = img["is_water"].to_bool();
 713 
 714                 images.push_back(rule_image(layer, basex - dx, basey - dy, global, center_x, center_y, is_water));
 715 
 716                 // Adds the other variants of the image
 717                 for(const config& variant : img.child_range("variant")) {
 718                         const std::string& name = variant["name"];
 719                         const std::string& variations = img["variations"];
 720                         const std::string& tod = variant["tod"];
 721                         const std::string& has_flag = variant["has_flag"];
 722                         bool random_start = variant["random_start"].to_bool(true);
 723 
 724                         images.back().variants.push_back(rule_image_variant(name, variations, tod, has_flag, random_start));
 725                 }
 726 
 727                 // Adds the main (default) variant of the image at the end,
 728                 // (will be used only if previous variants don't match)
 729                 const std::string& name = img["name"];
 730                 const std::string& variations = img["variations"];
 731                 bool random_start = img["random_start"].to_bool(true);
 732                 images.back().variants.push_back(rule_image_variant(name, variations, random_start));
 733         }
 734 }

 736 terrain_builder::terrain_constraint& terrain_builder::add_constraints(terrain_builder::constraint_set& constraints,
 737                 const map_location& loc,
 738                 const t_translation::ter_match& type,
 739                 const config& global_images)
 740 {
 741         terrain_constraint* cons = nullptr;
 742         for(terrain_constraint& c : constraints) {
 743                 if(c.loc == loc) {
 744                         cons = &c;
 745                         break;
 746                 }
 747         }
 748 
 749         if(!cons) {
 750                 // The terrain at the current location did not exist, so create it
 751                 constraints.emplace_back(loc);
 752                 cons = &constraints.back();
 753         }
 754 
 755         if(!type.terrain.empty()) {
 756                 cons->terrain_types_match = type;
 757         }
 758 
 759         int x = loc.x * tilewidth_ * 3 / 4;
 760         int y = loc.y * tilewidth_ + (loc.x % 2) * tilewidth_ / 2;
 761         add_images_from_config(cons->images, global_images, true, x, y);
 762 
 763         return *cons;
 764 }

 793 void terrain_builder::parse_mapstring(
 794                 const std::string& mapstring, struct building_rule& br, anchormap& anchors, const config& global_images)
 795 {
 796         const t_translation::ter_map map = t_translation::read_builder_map(mapstring);
 797 
 798         // If there is an empty map leave directly.
 799         // Determine after conversion, since a
 800         // non-empty string can return an empty map.
 801         if(map.data.empty()) {
 802                 return;
 803         }
 804 
 805         int lineno = (map.get(0, 0) == t_translation::NONE_TERRAIN) ? 1 : 0;
 806         int x = lineno;
 807         int y = 0;
 808         for(int y_off = 0; y_off < map.w; ++y_off) {
 809                 for(int x_off = x; x_off < map.h; ++x_off) {
 810                         const t_translation::terrain_code terrain = map.get(y_off, x_off);
 811 
 812                         if(terrain.base == t_translation::TB_DOT) {
 813                                 // Dots are simple placeholders,
 814                                 // which do not represent actual terrains.
 815                         } else if(terrain.overlay != 0) {
 816                                 anchors.emplace(terrain.overlay, map_location(x, y));
 817                         } else if(terrain.base == t_translation::TB_STAR) {
 818                                 add_constraints(br.constraints, map_location(x, y), t_translation::STAR, global_images);
 819                         } else {
 820                                 ERR_NG << "Invalid terrain (" << t_translation::write_terrain_code(terrain) << ") in builder map"
 821                                            << std::endl;
 822                                 assert(false);
 823                                 return;
 824                         }
 825                         x += 2;
 826                 }
 827 
 828                 if(lineno % 2 == 1) {
 829                         ++y;
 830                         x = 0;
 831                 } else {
 832                         x = 1;
 833                 }
 834                 ++lineno;
 835         }
 836 }

 845 void terrain_builder::add_rotated_rules(building_ruleset& rules, building_rule& tpl, const std::string& rotations)
 846 {
 847         if(rotations.empty()) {
 848                 // Adds the parsed built terrain to the list
 849 
 850                 add_rule(rules, tpl);
 851         } else {
 852                 const std::vector<std::string>& rot = utils::split(rotations, ',');
 853 
 854                 for(size_t angle = 0; angle < rot.size(); ++angle) {
 855                         /* Only 5% of the rules have valid images, so most of
 856                            them will be discarded. If the ratio was higher,
 857                            it would be more efficient to insert a copy of the
 858                            template rule into the ruleset, modify it in place,
 859                            and remove it if invalid. But since the ratio is so
 860                            low, the speedup is not worth the extra multiset
 861                            manipulations. */
 862 
 863                         if(rot.at(angle) == "skip") {
 864                                 continue;
 865                         }
 866 
 867                         building_rule rule = tpl;
 868                         rotate_rule(rule, angle, rot);
 869                         add_rule(rules, rule);
 870                 }
 871         }
 872 }

 874 void terrain_builder::parse_config(const config& cfg, bool local)
 875 {
 876         log_scope("terrain_builder::parse_config");
 877         int n = 0;
 878 
 879         // Parses the list of building rules (BRs)
 880         for(const config& br : cfg.child_range("terrain_graphics")) {
 881                 building_rule pbr; // Parsed Building rule
 882                 pbr.local = local;
 883 
 884                 // add_images_from_config(pbr.images, **br);
 885 
 886                 pbr.location_constraints = map_location(br["x"].to_int() - 1, br["y"].to_int() - 1);
 887 
 888                 pbr.modulo_constraints = map_location(br["mod_x"].to_int(), br["mod_y"].to_int());
 889 
 890                 pbr.probability = br["probability"].to_int(100);
 891 
 892                 // Mapping anchor indices to anchor locations.
 893                 anchormap anchors;
 894 
 895                 // Parse the map= , if there is one (and fill the anchors list)
 896                 parse_mapstring(br["map"], pbr, anchors, br);
 897 
 898                 // Parses the terrain constraints (TCs)
 899                 for(const config& tc : br.child_range("tile")) {
 900                         // Adds the terrain constraint to the current built terrain's list
 901                         // of terrain constraints, if it does not exist.
 902                         map_location loc;
 903                         if(const config::attribute_value* v = tc.get("x")) {
 904                                 loc.x = *v;
 905                         }
 906                         if(const config::attribute_value* v = tc.get("y")) {
 907                                 loc.y = *v;
 908                         }
 909                         if(loc.valid()) {
 910                                 add_constraints(pbr.constraints, loc, tc, br);
 911                         }
 912                         if(const config::attribute_value* v = tc.get("pos")) {
 913                                 int pos = *v;
 914                                 if(anchors.find(pos) == anchors.end()) {
 915                                         WRN_NG << "Invalid anchor!" << std::endl;
 916                                         continue;
 917                                 }
 918 
 919                                 std::pair<anchormap::const_iterator, anchormap::const_iterator> range = anchors.equal_range(pos);
 920 
 921                                 for(; range.first != range.second; ++range.first) {
 922                                         loc = range.first->second;
 923                                         add_constraints(pbr.constraints, loc, tc, br);
 924                                 }
 925                         }
 926                 }
 927 
 928                 const std::vector<std::string> global_set_flag = utils::split(br["set_flag"]);
 929                 const std::vector<std::string> global_no_flag = utils::split(br["no_flag"]);
 930                 const std::vector<std::string> global_has_flag = utils::split(br["has_flag"]);
 931                 const std::vector<std::string> global_set_no_flag = utils::split(br["set_no_flag"]);
 932 
 933                 for(terrain_constraint& constraint : pbr.constraints) {
 934                         constraint.set_flag.insert(constraint.set_flag.end(), global_set_flag.begin(), global_set_flag.end());
 935                         constraint.no_flag.insert(constraint.no_flag.end(), global_no_flag.begin(), global_no_flag.end());
 936                         constraint.has_flag.insert(constraint.has_flag.end(), global_has_flag.begin(), global_has_flag.end());
 937                         constraint.set_flag.insert(constraint.set_flag.end(), global_set_no_flag.begin(), global_set_no_flag.end());
 938                         constraint.no_flag.insert(constraint.no_flag.end(), global_set_no_flag.begin(), global_set_no_flag.end());
 939                 }
 940 
 941                 // Handles rotations
 942                 const std::string& rotations = br["rotations"];
 943 
 944                 pbr.precedence = br["precedence"];
 945 
 946                 add_rotated_rules(building_rules_, pbr, rotations);
 947 
 948                 n++;
 949                 if(n % 10 == 0) {
 950                         gui2::dialogs::loading_screen::progress();
 951                 }
 952         }
 953 
 954 // Debug output for the terrain rules
 955 #if 0
 956         std::cerr << "Built terrain rules: \n";
 957 
 958         building_ruleset::const_iterator rule;
 959         for(rule = building_rules_.begin(); rule != building_rules_.end(); ++rule) {
 960                 std::cerr << ">> New rule: image_background = "
 961                         << "\n>> Location " << rule->second.location_constraints
 962                         << "\n>> Probability " << rule->second.probability
 963 
 964                 for(constraint_set::const_iterator constraint = rule->second.constraints.begin();
 965                     constraint != rule->second.constraints.end(); ++constraint) {
 966 
 967                         std::cerr << ">>>> New constraint: location = (" << constraint->second.loc
 968                                   << "), terrain types = '" << t_translation::write_list(constraint->second.terrain_types_match.terrain) << "'\n";
 969 
 970                         std::vector<std::string>::const_iterator flag;
 971 
 972                         for(flag  = constraint->second.set_flag.begin(); flag != constraint->second.set_flag.end(); ++flag) {
 973                                 std::cerr << ">>>>>> Set_flag: " << *flag << "\n";
 974                         }
 975 
 976                         for(flag = constraint->second.no_flag.begin(); flag != constraint->second.no_flag.end(); ++flag) {
 977                                 std::cerr << ">>>>>> No_flag: " << *flag << "\n";
 978                         }

1009 bool terrain_builder::rule_matches(const terrain_builder::building_rule& rule,
1010                 const map_location& loc,
1011                 const terrain_constraint* type_checked) const
1012 {
1013         // Don't match if the location isn't a multiple of mod_x and mod_y
1014         if(rule.modulo_constraints.x > 0 && (loc.x % rule.modulo_constraints.x != 0)) {
1015                 return false;
1016         }
1017         if(rule.modulo_constraints.y > 0 && (loc.y % rule.modulo_constraints.y != 0)) {
1018                 return false;
1019         }
1020 
1021         if(rule.location_constraints.valid() && rule.location_constraints != loc) {
1022                 return false;
1023         }
1024 
1025         if(rule.probability != 100) {
1026                 unsigned int random = get_noise(loc, rule.get_hash()) % 100;
1027                 if(random > static_cast<unsigned int>(rule.probability)) {
1028                         return false;
1029                 }
1030         }
1031 
1032         for(const terrain_constraint& cons : rule.constraints) {
1033                 // Translated location
1034                 const map_location tloc = legacy_sum(loc, cons.loc);
1035 
1036                 if(!tile_map_.on_map(tloc)) {
1037                         return false;
1038                 }
1039 
1040                 // std::cout << "testing..." << builder_letter(map().get_terrain(tloc))
1041 
1042                 // check if terrain matches except if we already know that it does
1043                 if(&cons != type_checked && !terrain_matches(map().get_terrain(tloc), cons.terrain_types_match)) {
1044                         return false;
1045                 }
1046 
1047                 const std::set<std::string>& flags = tile_map_[tloc].flags;
1048 
1049                 for(const std::string& s : cons.no_flag) {
1050                         // If a flag listed in "no_flag" is present, the rule does not match
1051                         if(flags.find(s) != flags.end()) {
1052                                 return false;
1053                         }
1054                 }
1055                 for(const std::string& s : cons.has_flag) {
1056                         // If a flag listed in "has_flag" is not present, this rule does not match
1057                         if(flags.find(s) == flags.end()) {
1058                                 return false;
1059                         }
1060                 }
1061         }
1062 
1063         return true;
1064 }

1066 void terrain_builder::apply_rule(const terrain_builder::building_rule& rule, const map_location& loc)
1067 {
1068         unsigned int rand_seed = get_noise(loc, rule.get_hash());
1069 
1070         for(const terrain_constraint& constraint : rule.constraints) {
1071                 const map_location tloc = legacy_sum(loc, constraint.loc);
1072                 if(!tile_map_.on_map(tloc)) {
1073                         return;
1074                 }
1075 
1076                 tile& btile = tile_map_[tloc];
1077 
1078                 if(!constraint.no_draw) {
1079                         for(const rule_image& img : constraint.images) {
1080                                 btile.images.push_back(tile::rule_image_rand(&img, rand_seed));
1081                         }
1082                 }
1083 
1084                 // Sets flags
1085                 for(const std::string& flag : constraint.set_flag) {
1086                         btile.flags.insert(flag);
1087                 }
1088         }
1089 }

1091 // copied from text_surface::hash()
1092 // but keep it separated because the needs are different
1093 // and changing it will modify the map random variations
1094 static unsigned int hash_str(const std::string& str)
1095 {
1096         unsigned int h = 0;
1097         for(std::string::const_iterator it = str.begin(), it_end = str.end(); it != it_end; ++it)
1098                 h = ((h << 9) | (h >> (sizeof(int) * 8 - 9))) ^ (*it);
1099         return h;
1100 }
1101 
1102 unsigned int terrain_builder::building_rule::get_hash() const
1103 {
1104         if(hash_ != DUMMY_HASH)
1105                 return hash_;
1106 
1107         for(const terrain_constraint& constraint : constraints) {
1108                 for(const rule_image& ri : constraint.images) {
1109                         for(const rule_image_variant& variant : ri.variants) {
1110                                 // we will often hash the same string, but that seems fast enough
1111                                 hash_ += hash_str(variant.image_string);
1112                         }
1113                 }
1114         }
1115 
1116         // don't use the reserved dummy hash
1117         if(hash_ == DUMMY_HASH)
1118                 hash_ = 105533; // just a random big prime number
1119 
1120         return hash_;
1121 }

1123 void terrain_builder::build_terrains()
1124 {
1125         log_scope("terrain_builder::build_terrains");
1126 
1127         // Builds the terrain_by_type_ cache
1128         for(int x = -2; x <= map().w(); ++x) {
1129                 for(int y = -2; y <= map().h(); ++y) {
1130                         const map_location loc(x, y);
1131                         const t_translation::terrain_code t = map().get_terrain(loc);
1132 
1133                         terrain_by_type_[t].push_back(loc);
1134 
1135                         // Flag all hexes according to whether they're on the border or not,
1136                         // to make it easier for WML to draw the borders
1137                         if(!map().on_board(loc)) {
1138                                 tile_map_[loc].flags.insert("_border");
1139                         } else {
1140                                 tile_map_[loc].flags.insert("_board");
1141                         }
1142                 }
1143         }
1144 
1145         for(const building_rule& rule : building_rules_) {
1146                 // Find the constraint that contains the less terrain of all terrain rules.
1147                 // We will keep a track of the matching terrains of this constraint
1148                 // and later try to apply the rule only on them
1149                 size_t min_size = INT_MAX;
1150                 t_translation::ter_list min_types = t_translation::ter_list(); // <-- This must be explicitly initialized, just
1151                                                                                                                                            // as min_constraint is, at start of loop, or we
1152                                                                                                                                            // get a null pointer dereference when we go
1153                                                                                                                                            // through on later times.
1154                 const terrain_constraint* min_constraint = nullptr;
1155 
1156                 for(const terrain_constraint& constraint : rule.constraints) {
1157                         const t_translation::ter_match& match = constraint.terrain_types_match;
1158                         t_translation::ter_list matching_types;
1159                         size_t constraint_size = 0;
1160 
1161                         for(terrain_by_type_map::iterator type_it = terrain_by_type_.begin(); type_it != terrain_by_type_.end();
1162                                         ++type_it) {
1163                                 const t_translation::terrain_code t = type_it->first;
1164                                 if(terrain_matches(t, match)) {
1165                                         const size_t match_size = type_it->second.size();
1166                                         constraint_size += match_size;
1167                                         if(constraint_size >= min_size) {
1168                                                 break; // not a minimum, bail out
1169                                         }
1170                                         matching_types.push_back(t);
1171                                 }
1172                         }
1173 
1174                         if(constraint_size < min_size) {
1175                                 min_size = constraint_size;
1176                                 min_types = matching_types;
1177                                 min_constraint = &constraint;
1178                                 if(min_size == 0) {
1179                                         // a constraint is never matched on this map
1180                                         // we break with a empty type list
1181                                         break;
1182                                 }
1183                         }
1184                 }
1185 
1186                 // NOTE: if min_types is not empty, we have found a valid min_constraint;
1187                 for(t_translation::ter_list::const_iterator t = min_types.begin(); t != min_types.end(); ++t) {
1188                         const std::vector<map_location>* locations = &terrain_by_type_[*t];
1189 
1190                         for(std::vector<map_location>::const_iterator itor = locations->begin(); itor != locations->end(); ++itor) {
1191                                 const map_location loc = legacy_difference(*itor, min_constraint->loc);
1192 
1193                                 if(rule_matches(rule, loc, min_constraint)) {
1194                                         apply_rule(rule, loc);
1195                                 }
1196                         }
1197                 }
1198         }
1199 }
