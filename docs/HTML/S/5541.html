  11 local function get_image(cfg, speaker)
  12         local image = cfg.image
  13         local left_side = true
  14 
  15         if speaker and (image == nil or image == "") and (cfg.second_image == nil or cfg.second_image == "") then
  16                 image = speaker.portrait
  17         end
  18 
  19         if image == "none" or image == nil then
  20                 return "", true
  21         end
  22 
  23         -- Note: This is deprecated except for use to set default alignment in portraits
  24         -- (Move it into the first if statement later, with a nil check)
  25         if image:find("~RIGHT%(%)") then
  26                 left_side = false
  27                 -- The percent signs escape the parentheses for a literal match
  28                 image = image:gsub("~RIGHT%(%)", "")
  29         end
  30 
  31         if cfg.image_pos == 'left' then
  32                 left_side = true
  33         elseif cfg.image_pos == 'right' then
  34                 left_side = false
  35         elseif cfg.image_pos ~= nil then
  36                 helper.wml_error('Invalid [message]image_pos - should be left or right')
  37         end
  38 
  39         return image, left_side
  40 end

  52 local function get_pango_color(color)
  53         local pango_color = "#"
  54         
  55         -- if a hex color was passed in
  56         -- or if a color string was passed in - contains no non-letter characters
  57         -- just use that
  58         if string.sub(color, 1, 1) == "#" or not string.match(color, "%A+") then
  59                 pango_color = color
  60         -- decimal color was passed in, convert to hex color for pango
  61         else
  62                 for s in string.gmatch(color, "%d+") do
  63                         pango_color = pango_color .. tonumber(s, 16)
  64                 end
  65         end
  66         
  67         return pango_color
  68 end

 177 local function get_speaker(cfg)
 178         local speaker
 179         local context = wesnoth.current.event_context
 180 
 181         if cfg.speaker == "narrator" then
 182                 speaker = "narrator"
 183         elseif cfg.speaker == "unit" then
 184                 speaker = wesnoth.get_unit(context.x1 or 0, context.y1 or 0)
 185         elseif cfg.speaker == "second_unit" then
 186                 speaker = wesnoth.get_unit(context.x2 or 0, context.y2 or 0)
 187         else
 188                 speaker = wesnoth.get_units(cfg)[1]
 189         end
 190 
 191         return speaker
 192 end

 194 local function message_user_choice(cfg, speaker, options, text_input, sound, voice)
 195         local image, left_side = get_image(cfg, speaker)
 196         local caption = get_caption(cfg, speaker)
 197 
 198         local msg_cfg = {
 199                 left_side = left_side,
 200                 title = caption,
 201                 message = cfg.message,
 202                 portrait = image,
 203                 mirror = cfg.mirror,
 204                 second_portrait = cfg.second_image,
 205                 second_mirror = cfg.second_mirror,
 206         }
 207         
 208         if speaker ~= nil then
 209                 if cfg.male_message ~= nil and speaker.gender == "male" then
 210                         msg_cfg.message = cfg.male_message
 211                 elseif cfg.female_message ~= nil and speaker.gender == "female" then
 212                         msg_cfg.message = cfg.female_message
 213                 end
 214         end
 215         
 216         -- add formatting
 217         msg_cfg.message = add_formatting(cfg, msg_cfg.message)
 218         
 219         -- Parse input text, if not available all fields are empty
 220         if text_input then
 221                 local input_max_size = tonumber(text_input.max_length) or 256
 222                 if input_max_size > 1024 or input_max_size < 1 then
 223                         log("Invalid maximum size for input " .. input_max_size, "warning")
 224                         input_max_size = 256
 225                 end
 226 
 227                 -- This roundabout method is because text_input starts out
 228                 -- as an immutable userdata value
 229                 text_input = {
 230                         label = text_input.label or "",
 231                         text  = text_input.text  or "",
 232                         max_length = input_max_size,
 233                 }
 234         end
 235 
 236         return function()
 237                 if sound then wesnoth.play_sound(sound) end
 238                 if voice then
 239                         local speech = {
 240                                 id = "wml_message_speaker",
 241                                 sounds = voice,
 242                                 loops = 0,
 243                                 delay = 0,
 244                         }
 245                         if speaker then
 246                                 speech.x = speaker.x
 247                                 speech.y = speaker.y
 248                         end
 249                         wesnoth.add_sound_source(speech)
 250                 end
 251 
 252                 local option_chosen, ti_content = wesnoth.show_message_dialog(msg_cfg, options, text_input)
 253 
 254                 if voice then
 255                         wesnoth.remove_sound_source("wml_message_speaker")
 256                 end
 257 
 258                 if option_chosen == -2 then -- Pressed Escape (only if no input)
 259                         wesnoth.skip_messages()
 260                 end
 261 
 262                 local result_cfg = {}
 263 
 264                 if #options > 0 then
 265                         result_cfg.value = option_chosen
 266                 end
 267 
 268                 if text_input ~= nil then
 269                         result_cfg.text = ti_content
 270                 end
 271 
 272                 return result_cfg
 273         end

 276 function wesnoth.wml_actions.message(cfg)
 277         local show_if = helper.get_child(cfg, "show_if") or {}
 278         if not wesnoth.eval_conditional(show_if) then
 279                 log("[message] skipped because [show_if] did not pass", "debug")
 280                 return
 281         end
 282 
 283         -- Only the first text_input tag is considered
 284         local text_input
 285         for cfg in helper.child_range(cfg, "text_input") do
 286                 if text_input ~= nil then
 287                         log("Too many [text_input] tags, only one accepted", "warning")
 288                         break
 289                 end
 290                 text_input = cfg
 291         end
 292 
 293         local options, option_events = {}, {}
 294         for option in helper.child_range(cfg, "option") do
 295                 local condition = helper.get_child(option, "show_if") or {}
 296 
 297                 if wesnoth.eval_conditional(condition) then
 298                         if option.message and not option.image and not option.label then
 299                                 local message = tostring(option.message)
 300                                 if message:find("&") or message:find("=") or message:find("*") == 1 then
 301                                         -- Perhaps someday this should be promoted from "debug" to "warning", but for now, we're keeping it quiet.
 302                                         log('[option]message="&image=col1=col2" is deprecated, use new DescriptionWML instead (default, image, label, description)', "debug")
 303                                 end
 304                                 -- Legacy format
 305                                 table.insert(options, option.message)
 306                         else
 307                                 local opt = {
 308                                         label = option.label,
 309                                         description = option.description,
 310                                         image = option.image,
 311                                         default = option.default,
 312                                         value = option.value
 313                                 }
 314                                 if option.message then
 315                                         if not option.label then
 316                                                 -- Support either message or description
 317                                                 opt.label = option.message
 318                                         else
 319                                                 log("[option] has both label= and message=, ignoring the latter", "warning")
 320                                         end
 321                                 end
 322                                 table.insert(options, opt)
 323                         end
 324                         table.insert(option_events, {})
 325 
 326                         for cmd in helper.child_range(option, "command") do
 327                                 table.insert(option_events[#option_events], cmd)
 328                         end
 329                 end
 330         end
 331 
 332         -- Check if there is any input to be made, if not the message may be skipped
 333         local has_input = text_input ~= nil or #options > 0
 334 
 335         if not has_input and wesnoth.is_skipping_messages() then
 336                 -- No input to get and the user is not interested either
 337                 log("Skipping [message] because user not interested", "debug")
 338                 return
 339         end
 340 
 341         local sides_for = cfg.side_for
 342         if sides_for and not has_input then
 343                 local show_for_side = false
 344 
 345                 -- Sanity checks on side number and controller
 346                 for side in utils.split(sides_for) do
 347                         side = tonumber(side)
 348                         if side > 0 and side < #wesnoth.sides and wesnoth.sides[side].controller == "human" then
 349                                 show_for_side = true
 350                                 break
 351                         end
 352                 end
 353 
 354                 if not show_for_side then
 355                         -- Player isn't controlling side which should see the message
 356                         log("Player isn't controlling side that should see [message]", "debug")
 357                         return
 358                 end
 359         end
 360 
 361         local speaker = get_speaker(cfg)
 362         if not speaker then
 363                 -- No matching unit found, continue onto the next message
 364                 log("No speaker found for [message]", "debug")
 365                 return
 366         elseif cfg.highlight == false then
 367                 -- Nothing to do here
 368         elseif speaker == "narrator" then
 369                 -- Narrator, so deselect units
 370                 wesnoth.deselect_hex()
 371                 -- The speaker is expected to be either nil or a unit later
 372                 speaker = nil
 373         else
 374                 -- Check ~= false, because the default if omitted should be true
 375                 if cfg.scroll ~= false then
 376                         wesnoth.scroll_to_tile(speaker.x, speaker.y, false, false, true)
 377                 end
 378 
 379                 wesnoth.highlight_hex(speaker.x, speaker.y)
 380                 wesnoth.fire("redraw")
 381         end
 382 
 383         local msg_dlg = message_user_choice(cfg, speaker, options, text_input, cfg.sound, cfg.voice)
 384 
 385         local option_chosen
 386         if not has_input then
 387                 -- Always show the dialog if it has no input, whether we are replaying or not
 388                 msg_dlg()
 389         else
 390                 local wait_description = cfg.wait_description or _("input")
 391                 if type(sides_for) ~= "number" then
 392                         -- 0 means currently playing side.
 393                         sides_for = 0
 394                 end
 395                 local choice = wesnoth.synchronize_choice(wait_description, msg_dlg, sides_for)
 396 
 397                 option_chosen = tonumber(choice.value)
 398 
 399                 if text_input ~= nil then
 400                         -- Implement the consequences of the choice
 401                         wesnoth.set_variable(text_input.variable or "input", choice.text)
 402                 end
 403         end
 404         
 405         -- Unhilight the speaker
 406         if speaker and not cfg.highlight == false then
 407                 wesnoth.deselect_hex()
 408         end
 409 
 410         if #options > 0 then
 411                 if option_chosen > #options then
 412                         log("invalid choice (" .. option_chosen .. ") was specified, choice 1 to " ..
 413                                 #options .. " was expected", "debug")
 414                         return
 415                 end
 416 
 417                 if cfg.variable ~= nil then
 418                         if options[option_chosen].value == nil then
 419                                 wesnoth.set_variable(cfg.variable, option_chosen)
 420                         else
 421                                 wesnoth.set_variable(cfg.variable, options[option_chosen].value)
 422                         end
 423                 end
 424 
 425                 for i, cmd in ipairs(option_events[option_chosen]) do
 426                         local action = utils.handle_event_commands(cmd, "plain")
 427                         if action ~= "none" then break end
 428                 end
 429         end
 430 end
