  49 void controller_base::handle_event(const SDL_Event& event)
  50 {
  51         if(gui::in_dialog()) {
  52                 return;
  53         }
  54         static const hotkey::hotkey_command& quit_hotkey = hotkey::hotkey_command::get_command_by_command(hotkey::HOTKEY_QUIT_GAME);
  55 
  56         switch(event.type) {
  57         case SDL_TEXTINPUT:
  58                 if(have_keyboard_focus()) {
  59                         hotkey::key_event(event, get_hotkey_command_executor());
  60                 }
  61                 break;
  62         case SDL_KEYDOWN:
  63                 // Detect key press events, unless there something that has keyboard focus
  64                 // in which case the key press events should go only to it.
  65                 if(have_keyboard_focus()) {
  66                         if(event.key.keysym.sym == SDLK_ESCAPE) {
  67                                 hotkey::execute_command(quit_hotkey, get_hotkey_command_executor());
  68                                 break;
  69                         }
  70 
  71                         process_keydown_event(event);
  72                         hotkey::key_event(event, get_hotkey_command_executor());
  73                         process_keyup_event(event);
  74                 } else {
  75                         process_focus_keydown_event(event);
  76                 }
  77                 break;
  78         case SDL_KEYUP:
  79                 process_keyup_event(event);
  80                 hotkey::key_event(event, get_hotkey_command_executor());
  81                 break;
  82         case SDL_JOYBUTTONDOWN:
  83                 process_keydown_event(event);
  84                 hotkey::jbutton_event(event, get_hotkey_command_executor());
  85                 break;
  86         case SDL_JOYHATMOTION:
  87                 process_keydown_event(event);
  88                 hotkey::jhat_event(event, get_hotkey_command_executor());
  89                 break;
  90         case SDL_MOUSEMOTION:
  91                 // Ignore old mouse motion events in the event queue
  92                 SDL_Event new_event;
  93                 if(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {
  94                         while(SDL_PeepEvents(&new_event, 1, SDL_GETEVENT, SDL_MOUSEMOTION, SDL_MOUSEMOTION) > 0) {};
  95                         get_mouse_handler_base().mouse_motion_event(new_event.motion, is_browsing());
  96                 } else {
  97                         get_mouse_handler_base().mouse_motion_event(event.motion, is_browsing());
  98                 }
  99                 break;
 100         case SDL_MOUSEBUTTONDOWN:
 101                 process_keydown_event(event);
 102                 get_mouse_handler_base().mouse_press(event.button, is_browsing());
 103                 if (get_mouse_handler_base().get_show_menu()){
 104                         show_menu(get_display().get_theme().context_menu()->items(),event.button.x,event.button.y,true, get_display());
 105                 }
 106                 hotkey::mbutton_event(event, get_hotkey_command_executor());
 107                 break;
 108         case SDL_MOUSEBUTTONUP:
 109                 get_mouse_handler_base().mouse_press(event.button, is_browsing());
 110                 if (get_mouse_handler_base().get_show_menu()){
 111                         show_menu(get_display().get_theme().context_menu()->items(),event.button.x,event.button.y,true, get_display());
 112                 }
 113                 break;
 114         case SDL_MOUSEWHEEL:
 115                 get_mouse_handler_base().mouse_wheel(event.wheel.x, event.wheel.y, is_browsing());
 116                 break;
 117         default:
 118                 break;
 119         }
 120 }

 139 bool controller_base::handle_scroll(int mousex, int mousey, int mouse_flags, double x_axis, double y_axis)
 140 {
 141         sdl::window* window = CVideo::get_singleton().get_window();
 142         bool mouse_in_window = (window != nullptr && (window->get_flags() & SDL_WINDOW_MOUSE_FOCUS) != 0)
 143                 || preferences::get("scroll_when_mouse_outside", true);
 144         int scroll_speed = preferences::scroll_speed();
 145         int dx = 0, dy = 0;
 146         int scroll_threshold = (preferences::mouse_scroll_enabled())
 147                 ? preferences::mouse_scroll_threshold() : 0;
 148         for (const theme::menu& m : get_display().get_theme().menus()) {
 149                 if (sdl::point_in_rect(mousex, mousey, m.get_location())) {
 150                         scroll_threshold = 0;
 151                 }
 152         }
 153 
 154         // apply keyboard scrolling
 155         dy -= scroll_up_ * scroll_speed;
 156         dy += scroll_down_ * scroll_speed;
 157         dx -= scroll_left_ * scroll_speed;
 158         dx += scroll_right_ * scroll_speed;
 159 
 160         // scroll if mouse is placed near the edge of the screen
 161         if (mouse_in_window) {
 162                 if (mousey < scroll_threshold) {
 163                         dy -= scroll_speed;
 164                 }
 165                 if (mousey > get_display().h() - scroll_threshold) {
 166                         dy += scroll_speed;
 167                 }
 168                 if (mousex < scroll_threshold) {
 169                         dx -= scroll_speed;
 170                 }
 171                 if (mousex > get_display().w() - scroll_threshold) {
 172                         dx += scroll_speed;
 173                 }
 174         }
 175 
 176         // scroll with middle-mouse if enabled
 177         if ((mouse_flags & SDL_BUTTON_MMASK) != 0 && preferences::middle_click_scrolls()) {
 178                 const SDL_Point original_loc = get_mouse_handler_base().get_scroll_start();
 179 
 180                 if (get_mouse_handler_base().scroll_started()) {
 181                         const SDL_Rect& rect = get_display().map_outside_area();
 182                         if (sdl::point_in_rect(mousex, mousey,rect) &&
 183                                 get_mouse_handler_base().scroll_started()) {
 184                                 // Scroll speed is proportional from the distance from the first
 185                                 // middle click and scrolling speed preference.
 186                                 const double speed = 0.04 * sqrt(static_cast<double>(scroll_speed));
 187                                 const double snap_dist = 16; // Snap to horizontal/vertical scrolling
 188                                 const double x_diff = (mousex - original_loc.x);
 189                                 const double y_diff = (mousey - original_loc.y);
 190 
 191                                 if (std::fabs(x_diff) > snap_dist || std::fabs(y_diff) <= snap_dist) dx += speed * x_diff;
 192                                 if (std::fabs(y_diff) > snap_dist || std::fabs(x_diff) <= snap_dist) dy += speed * y_diff;
 193                         }
 194                 }
 195                 else { // Event may fire mouse down out of order with respect to initial click
 196                         get_mouse_handler_base().set_scroll_start(mousex, mousey);
 197                 }
 198         }
 199 
 200         // scroll with joystick
 201         dx += round_double( x_axis * scroll_speed);
 202         dy += round_double( y_axis * scroll_speed);
 203 
 204         return get_display().scroll(dx, dy);
 205 }

 207 void controller_base::play_slice(bool is_delay_enabled)
 208 {
 209         CKey key;
 210 
 211         if (plugins_context *l = get_plugins_context()) {
 212                 l->play_slice();
 213         }
 214 
 215         events::pump();
 216         events::raise_process_event();
 217         events::raise_draw_event();
 218 
 219         // Update sound sources before scrolling
 220         if (soundsource::manager *l = get_soundsource_man()) {
 221                 l->update();
 222         }
 223 
 224         const theme::menu* const m = get_display().menu_pressed();
 225         if(m != nullptr) {
 226                 const SDL_Rect& menu_loc = m->location(get_display().screen_area());
 227                 show_menu(m->items(),menu_loc.x+1,menu_loc.y + menu_loc.h + 1,false, get_display());
 228 
 229                 return;
 230         }
 231         const theme::action* const a = get_display().action_pressed();
 232         if(a != nullptr) {
 233                 const SDL_Rect& action_loc = a->location(get_display().screen_area());
 234                 execute_action(a->items(), action_loc.x+1, action_loc.y + action_loc.h + 1,false);
 235 
 236                 return;
 237         }
 238         auto str_vec = additional_actions_pressed();
 239         if (!str_vec.empty()) {
 240                 execute_action(str_vec, 0, 0, false);
 241                 return;
 242         }
 243 
 244         bool was_scrolling = scrolling_;
 245 
 246         std::pair<double, double> values = joystick_manager_.get_scroll_axis_pair();
 247         const double joystickx = values.first;
 248         const double joysticky = values.second;
 249 
 250         int mousex, mousey;
 251         Uint8 mouse_flags = SDL_GetMouseState(&mousex, &mousey);
 252 
 253         /* TODO fendrin enable after an axis choosing mechanism is implemented
 254         std::pair<double, double> values = joystick_manager_.get_mouse_axis_pair();
 255         mousex += values.first * 10;
 256         mousey += values.second * 10;
 257         SDL_WarpMouse(mousex, mousey);
 258         */
 259         scrolling_ = handle_scroll(mousex, mousey, mouse_flags, joystickx, joysticky);
 260 
 261         map_location highlighted_hex = get_display().mouseover_hex();
 262 
 263         /* TODO fendrin enable when the relative cursor movement is implemented well enough
 264         const map_location& selected_hex = get_display().selected_hex();
 265 
 266         if (selected_hex != map_location::null_location()) {
 267                 if (joystick_manager_.next_highlighted_hex(highlighted_hex, selected_hex)) {
 268                         get_mouse_handler_base().mouse_motion(0,0, true, true, highlighted_hex);
 269                         get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 270                         scrolling_ = true;
 271                 }
 272         } else */
 273 
 274         if (joystick_manager_.update_highlighted_hex(highlighted_hex)
 275                         && get_display().get_map().on_board(highlighted_hex)) {
 276                 get_mouse_handler_base().mouse_motion(0,0, true, true, highlighted_hex);
 277                 get_display().scroll_to_tile(highlighted_hex, display::ONSCREEN_WARP, false, true);
 278                 scrolling_ = true;
 279                 }
 280 
 281         get_display().draw();
 282 
 283         // be nice when window is not visible
 284         // NOTE should be handled by display instead, to only disable drawing
 285         sdl::window* window = CVideo::get_singleton().get_window();
 286         if (window != nullptr && is_delay_enabled && (window->get_flags() & SDL_WINDOW_SHOWN) == 0) {
 287                 CVideo::delay(200);
 288         }
 289 
 290         if (!scrolling_ && was_scrolling) {
 291                 // scrolling ended, update the cursor and the brightened hex
 292                 get_mouse_handler_base().mouse_update(is_browsing(), highlighted_hex);
 293         }
 294 }

 296 void controller_base::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool context_menu, display& disp)
 297 {
 298         hotkey::command_executor* cmd_exec = get_hotkey_command_executor();
 299         if(!cmd_exec) {
 300                 return;
 301         }
 302 
 303         std::vector<config> items;
 304         for(const config& c : items_arg) {
 305                 const std::string& id = c["id"];
 306                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
 307 
 308                 if(cmd_exec->can_execute_command(command) && (!context_menu || in_context_menu(command.id))) {
 309                         items.emplace_back(config {"id", id});
 310                 }
 311         }
 312 
 313         if(items.empty()) {
 314                 return;
 315         }
 316 
 317         cmd_exec->show_menu(items, xloc, yloc, context_menu, disp);
 318 }

 320 void controller_base::execute_action(const std::vector<std::string>& items_arg, int xloc, int yloc, bool context_menu)
 321 {
 322         hotkey::command_executor * cmd_exec = get_hotkey_command_executor();
 323         if (!cmd_exec) {
 324                 return;
 325         }
 326 
 327         std::vector<std::string> items;
 328         for (const std::string& item : items_arg) {
 329 
 330                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(item);
 331                 if(cmd_exec->can_execute_command(command))
 332                         items.push_back(item);
 333         }
 334 
 335         if(items.empty())
 336                 return;
 337         cmd_exec->execute_action(items, xloc, yloc, context_menu, get_display());
 338 }

 347 const config& controller_base::get_theme(const config& game_config, std::string theme_name)
 348 {
 349         if (theme_name.empty()) theme_name = preferences::theme();
 350 
 351         if (const config &c = game_config.find_child("theme", "id", theme_name))
 352                 return c;
 353 
 354         ERR_DP << "Theme '" << theme_name << "' not found. Trying the default theme." << std::endl;
 355 
 356         if (const config &c = game_config.find_child("theme", "id", "Default"))
 357                 return c;
 358 
 359         ERR_DP << "Default theme not found." << std::endl;
 360 
 361         static config empty;
 362         return empty;
 363 }
