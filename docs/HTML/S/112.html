 425 void add_color_info(const config& v, bool build_defaults)
 426 {
 427         if(build_defaults) {
 428                 default_colors.clear();
 429         }
 430 
 431         for(const config& teamC : v.child_range("color_range")) {
 432                 const config::attribute_value* a1 = teamC.get("id"), *a2 = teamC.get("rgb");
 433                 if(!a1 || !a2) {
 434                         continue;
 435                 }
 436 
 437                 std::string id = *a1;
 438                 std::vector<color_t> temp;
 439 
 440                 for(const auto& s : utils::split(*a2)) {
 441                         try {
 442                                 temp.push_back(color_t::from_hex_string(s));
 443                         } catch(std::invalid_argument&) {
 444                                 std::stringstream ss;
 445                                 ss << "can't parse color string:\n" << teamC.debug() << "\n";
 446                                 throw config::error(ss.str());
 447                         }
 448                 }
 449 
 450                 team_rgb_range.emplace(id, color_range(temp));
 451                 team_rgb_name.emplace(id, teamC["name"].t_str());
 452 
 453                 LOG_NG << "registered color range '" << id << "': " << team_rgb_range[id].debug() << '\n';
 454 
 455                 // Ggenerate palette of same name;
 456                 std::vector<color_t> tp = palette(team_rgb_range[id]);
 457                 if(!tp.empty()) {
 458                         team_rgb_colors.emplace(id, tp);
 459                 }
 460 
 461                 if(build_defaults && teamC["default"].to_bool()) {
 462                         default_colors.push_back(*a1);
 463                 }
 464         }
 465 
 466         for(const config &cp : v.child_range("color_palette")) {
 467                 for(const config::attribute& rgb : cp.attribute_range()) {
 468                         std::vector<color_t> temp;
 469                         for(const auto& s : utils::split(rgb.second)) {
 470                                 try {
 471                                         temp.push_back(color_t::from_hex_string(s));
 472                                 } catch(std::invalid_argument&) {
 473                                         ERR_NG << "Invalid color in palette: " << s << std::endl;
 474                                 }
 475                         }
 476 
 477                         team_rgb_colors.emplace(rgb.first, temp);
 478                         LOG_NG << "registered color palette: " << rgb.first << '\n';
 479                 }
 480         }
 481 }

 491 const color_range& color_info(const std::string& name)
 492 {
 493         auto i = team_rgb_range.find(name);
 494         if(i != team_rgb_range.end()) {
 495                 return i->second;
 496         }
 497 
 498         std::vector<color_t> temp;
 499         for(const auto& s : utils::split(name)) {
 500                 try {
 501                         temp.push_back(color_t::from_hex_string(s));
 502                 } catch(std::invalid_argument&) {
 503                         throw config::error(_("Invalid color in range: ") + s);
 504                 }
 505         }
 506 
 507         team_rgb_range.emplace(name, color_range(temp));
 508         return color_info(name);
 509 }
 510 
 511 const std::vector<color_t>& tc_info(const std::string& name)
 512 {
 513         auto i = team_rgb_colors.find(name);
 514         if(i != team_rgb_colors.end()) {
 515                 return i->second;
 516         }
 517 
 518         std::vector<color_t> temp;
 519         for(const auto& s : utils::split(name)) {
 520                 try {
 521                         temp.push_back(color_t::from_hex_string(s));
 522                 } catch(std::invalid_argument&) {
 523                         static std::vector<color_t> stv;
 524                         ERR_NG << "Invalid color in palette: " << s << std::endl;
 525                         return stv;
 526                 }
 527         }
 528 
 529         team_rgb_colors.emplace(name, temp);
 530         return tc_info(name);
 531 }

 533 color_t red_to_green(int val, bool for_text)
 534 {
 535         const std::vector<color_t>& color_scale = for_text ? red_green_scale_text : red_green_scale;
 536 
 537         val = utils::clamp(val, 0, 100);
 538         const int lvl = (color_scale.size() - 1) * val / 100;
 539 
 540         return color_scale[lvl];
 541 }
