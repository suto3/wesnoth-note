 102 /**
 103  * Helper to generate a widget from a WML widget instance.
 104  *
 105  * Mainly used as functor for @ref register_builder_widget.
 106  *
 107  * @param cfg                     The config with the information to
 108  *                                Instantiate the widget.
 109  *
 110  * @returns                       A generic widget builder pointer.
 111  */
 112 template <class T>
 113 builder_widget_ptr build_widget(const config& cfg)

 115         return std::make_shared<T>(cfg);
