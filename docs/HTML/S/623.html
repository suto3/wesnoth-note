 100 void listbox::remove_row(const unsigned row, unsigned count)
 101 {
 102         assert(generator_);
 103 
 104         if(row >= get_item_count()) {
 105                 return;
 106         }
 107 
 108         if(!count || count + row > get_item_count()) {
 109                 count = get_item_count() - row;
 110         }
 111 
 112         int height_reduced = 0;
 113         int width_reduced = 0;
 114         //TODO: Fix this for horizinal listboxes
 115         //Note the we have to use content_grid_ and cannot use "_list_grid" which is what generator_ uses.
 116         int row_pos_y = is_horizonal_ ? -1 : generator_->item(row).get_y()  - content_grid_->get_y();
 117         int row_pos_x = is_horizonal_ ? -1 : 0;
 118         for(; count; --count) {
 119                 if(generator_->item(row).get_visible() != visibility::invisible) {
 120                         if(is_horizonal_) {
 121                                 width_reduced += generator_->item(row).get_width();
 122                         }
 123                         else {
 124                                 height_reduced += generator_->item(row).get_height();
 125                         }
 126                 }
 127                 generator_->delete_item(row);
 128         }
 129 
 130         if((height_reduced != 0 || width_reduced != 0) && get_item_count() != 0) {
 131                 resize_content(-width_reduced, -height_reduced, row_pos_x, row_pos_y);
 132         } else {
 133                 update_content_size();
 134         }
 135 }

 155 void listbox::set_row_shown(const unsigned row, const bool shown)
 156 {
 157         assert(generator_);
 158 
 159         window* window = get_window();
 160         assert(window);
 161 
 162         const int selected_row = get_selected_row();
 163 
 164         bool resize_needed = false;
 165         {
 166                 window::invalidate_layout_blocker invalidate_layout_blocker(*window);
 167 
 168                 generator_->set_item_shown(row, shown);
 169                 point best_size = generator_->calculate_best_size();
 170                 generator_->place(generator_->get_origin(), { std::max(best_size.x, content_visible_area().w), best_size.y });
 171                 resize_needed = !content_resize_request();
 172         }
 173 
 174         if(resize_needed) {
 175                 window->invalidate_layout();
 176         } else {
 177                 content_grid_->set_visible_rectangle(content_visible_area());
 178                 set_is_dirty(true);
 179         }
 180 
 181         if(selected_row != get_selected_row() && callback_value_changed_) {
 182                 callback_value_changed_(*this);
 183         }
 184 }

 278 void listbox::list_item_clicked(widget& caller)
 279 {
 280         assert(generator_);
 281 
 282         /** @todo Hack to capture the keyboard focus. */
 283         get_window()->keyboard_capture(this);
 284 
 285         for(size_t i = 0; i < generator_->get_item_count(); ++i) {
 286 
 287                 if(generator_->item(i).has_widget(caller)) {
 288                         toggle_button* checkbox = dynamic_cast<toggle_button*>(&caller);
 289                         if(checkbox != nullptr) {
 290                                 generator_->select_item(i, checkbox->get_value_bool());
 291                         } else {
 292                                 generator_->toggle_item(i);
 293                         }
 294 
 295                         if(callback_item_changed_) {
 296                                 callback_item_changed_(i);
 297                         }
 298                         if(callback_value_changed_) {
 299                                 callback_value_changed_(*this);
 300                         }
 301                         return;
 302                 }
 303         }
 304         assert(false);
 305 }

 339 void listbox::place(const point& origin, const point& size)
 340 {
 341         boost::optional<unsigned> vertical_scrollbar_position, horizontal_scrollbar_position;
 342         // Check if this is the first time placing the list box
 343         if (get_origin() != point{-1, -1})
 344         {
 345                 vertical_scrollbar_position = get_vertical_scrollbar_item_position();
 346                 horizontal_scrollbar_position = get_horizontal_scrollbar_item_position();
 347         }
 348 
 349         // Inherited.
 350         scrollbar_container::place(origin, size);
 351 
 352         const int selected_item = generator_->get_selected_item();
 353         if (vertical_scrollbar_position && horizontal_scrollbar_position)
 354         {
 355                 LOG_GUI_L << LOG_HEADER << " restoring scroll position" << std::endl;
 356                 set_vertical_scrollbar_item_position(*vertical_scrollbar_position);
 357                 set_horizontal_scrollbar_item_position(*horizontal_scrollbar_position);
 358         }
 359         else if (selected_item != -1)
 360         {
 361                 LOG_GUI_L << LOG_HEADER << " making the initially selected item visible" << std::endl;
 362                 const SDL_Rect& visible = content_visible_area();
 363                 SDL_Rect rect = generator_->item(selected_item).get_rectangle();
 364 
 365                 rect.x = visible.x;
 366                 rect.w = visible.w;
 367 
 368                 show_content_rect(rect);
 369         }
 370 }

 376 void listbox::resize_content(const int width_modification,
 377                                                           const int height_modification,
 378                                                           const int width_modification_pos,
 379                                                           const int height_modification_pos)
 380 {
 381         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 382                           << " width_modification " << width_modification
 383                           << " height_modification " << height_modification << ".\n";
 384 
 385         if(content_resize_request(width_modification, height_modification, width_modification_pos, height_modification_pos)) {
 386 
 387                 // Calculate new size.
 388                 point size = content_grid()->get_size();
 389                 size.x += width_modification;
 390                 size.y += height_modification;
 391 
 392                 // Set new size.
 393                 content_grid()->set_size(size);
 394 
 395                 // Set status.
 396                 need_layout_ = true;
 397                 // If the content grows assume it "overwrites" the old content.
 398                 if(width_modification < 0 || height_modification < 0) {
 399                         set_is_dirty(true);
 400                 }
 401                 DBG_GUI_L << LOG_HEADER << " succeeded.\n";
 402         } else {
 403                 DBG_GUI_L << LOG_HEADER << " failed.\n";
 404         }
 405 }

 407 void listbox::resize_content(const widget& row)
 408 {
 409         if(row.get_visible() == visibility::invisible) {
 410                 return;
 411         }
 412 
 413         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 414                           << " row size " << row.get_best_size() << ".\n";
 415 
 416         const point content = content_grid()->get_size();
 417         point size = row.get_best_size();
 418         if(size.x < content.x) {
 419                 size.x = 0;
 420         } else {
 421                 size.x -= content.x;
 422         }
 423 
 424         resize_content(size.x, size.y);
 425 }

 552 void listbox::finalize(builder_grid_const_ptr header,
 553                                                 builder_grid_const_ptr footer,
 554                                                 const std::vector<std::map<std::string, string_map>>& list_data)
 555 {
 556         // "Inherited."
 557         scrollbar_container::finalize_setup();
 558 
 559         assert(generator_);
 560 
 561         if(header) {
 562                 swap_grid(&get_grid(), content_grid(), header->build(), "_header_grid");
 563         }
 564         grid& p = find_widget<grid>(this, "_header_grid", false);
 565         for(unsigned i = 0, max = std::max(p.get_cols(), p.get_rows()); i < max; ++i) {
 566                 if(selectable_item* selectable = find_widget<selectable_item>(&p, "sort_" +  std::to_string(i), false, false)) {
 567                         selectable->set_callback_state_change(std::bind(&listbox::order_by_column, this, i, _1));
 568                         if(orders_.size() < max ) {
 569                                 orders_.resize(max);
 570                         }
 571                         orders_[i].first = selectable;
 572                 }
 573         }
 574         if(footer) {
 575                 swap_grid(&get_grid(), content_grid(), footer->build(), "_footer_grid");
 576         }
 577 
 578         generator_->create_items(
 579                         -1, list_builder_, list_data, callback_list_item_clicked);
 580         swap_grid(nullptr, content_grid(), generator_, "_list_grid");
 581 }

 589 void listbox::order_by_column(unsigned column, widget& widget)
 590 {
 591         selectable_item& selectable = dynamic_cast<selectable_item&>(widget);
 592         if(column >= orders_.size()) {
 593                 return;
 594         }
 595         for(auto& pair : orders_)
 596         {
 597                 if(pair.first != nullptr && pair.first != &selectable) {
 598                         pair.first->set_value(SORT_NONE);
 599                 }
 600         }
 601         if(selectable.get_value() > orders_[column].second.size()) {
 602                 return;
 603         }
 604         if(selectable.get_value() == SORT_NONE) {
 605                 order_by(generator_base::torder_func(&default_sort));
 606         }
 607         else {
 608                 order_by(orders_[column].second[selectable.get_value() - 1]);
 609         }
 610 }

 643 const listbox::order_pair listbox::get_active_sorting_option()
 644 {
 645         const auto iter = std::find_if(orders_.begin(), orders_.end(), [](const std::pair<selectable_item*, generator_sort_array>& option) {
 646                 return option.first != nullptr && option.first->get_value() != SORT_NONE;
 647         });
 648 
 649         if(iter != orders_.end()) {
 650                 return {iter - orders_.begin(), static_cast<SORT_ORDER>(iter->first->get_value())};
 651         }
 652 
 653         return {-1, SORT_NONE};
 654 }

 901 static std::vector<std::map<std::string, string_map>> parse_list_data(const config& data, const unsigned int req_cols)
 902 {
 903         std::vector<std::map<std::string, string_map>> list_data;
 904         for(const auto & row : data.child_range("row"))
 905         {
 906                 auto cols = row.child_range("column");
 907                 VALIDATE(static_cast<unsigned>(cols.size()) == req_cols, _("'list_data' must have the same number of columns as the 'list_definition'."));
 908 
 909                 for(const auto & c : cols)
 910                 {
 911                         list_data.emplace_back();
 912                         for(const auto & i : c.attribute_range())
 913                         {
 914                                 list_data.back()[""][i.first] = i.second;
 915                         }
 916                         for(const auto& w : c.child_range("widget"))
 917                         {
 918                                 VALIDATE(w.has_attribute("id"), missing_mandatory_wml_key("[list_data][row][column][widget]", "id"));
 919                                 for(const auto& i : w.attribute_range()) {
 920                                         list_data.back()[w["id"]][i.first] = i.second;
 921                                 }
 922                         }
 923                 }
 924         }
 925         return list_data;
 926 }
