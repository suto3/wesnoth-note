  59         /** The way to handle the showing or hiding of the scrollbar. */
  60         enum scrollbar_mode {
  61                 /**
  62                  * The scrollbar is always shown, whether needed or not.
  63                  */
  64                 ALWAYS_VISIBLE,
  65 
  66                 /**
  67                  * The scrollbar is never shown even notwhen needed. There's also no space
  68                  * reserved for the scrollbar.
  69                  */
  70                 ALWAYS_INVISIBLE,
  71 
  72                 /**
  73                  * The scrollbar is shown when the number of items is larger as the visible items.
  74                  * The space for the scrollbar is always reserved, just in case it's needed after
  75                  * the initial sizing (due to adding items).
  76                  */
  77                 AUTO_VISIBLE,
  78 
  79                 /**
  80                  * Like AUTO_VISIBLE, but when not needed upon the initial layout phase, the bars
  81                  * are not shown and no space is reserved for them. (The algorithm hides them by
  82                  * default.
  83                  */
  84                 AUTO_VISIBLE_FIRST_RUN,
  85         };

 270         /**
 271          * Notification if the content of a child needs a resize.
 272          *
 273          * When a resize is required the container first can try to handle it
 274          * itself. If it can't honor the request the function will call @ref
 275          * window::invalidate_layout().
 276          *
 277          * @note Calling this function on a widget with size == (0, 0) results
 278          * false but doesn't call invalidate_layout, the engine expects to be in
 279          * build up phase with the layout already invalidated.
 280          *
 281          * @param force_sizing        If the contents fit do we want to force a
 282          *                            resize? This is needed in the MP lobby since
 283          *                            items might not be properly placed yet.
 284          *                            (The listboxes with the player info need it.)
 285          *
 286          * @returns                   True if the resize is handled, false
 287          *                            otherwise.
 288          */
 289         bool content_resize_request(const bool force_sizing = false);
 290 
 291         /**
 292          * Request from the content to modify the size of the container.
 293          *
 294          * When the wanted resize fails the function will call @ref
 295          * window::invalidate_layout().
 296          *
 297          *
 298          * @note Calling this function on a widget with size == (0, 0) results
 299          * false but doesn't call invalidate_layout, the engine expects to be in
 300          * build up phase with the layout already invalidated.
 301          *
 302          * @note If @ref window::get_need_layout() is true the function returns
 303          * false and doesn't try to fit the contents since a layout phase will be
 304          * triggered anyway.
 305          *
 306          * @note This function might replace the @ref content_resize_request above.
 307          *
 308          * @param width_modification  The wanted modification to the width:
 309          *                            * negative values reduce width.
 310          *                            * zero leave width as is.
 311          *                            * positive values increase width.
 312          * @param height_modification The wanted modification to the height:
 313          *                            * negative values reduce height.
 314          *                            * zero leave height as is.
 315          *                            * positive values increase height.
 316          * @param width_modification_pos
 317          *                            The position where the additional content was
 318          *                            inserted/removed, defaults to -1 which means
 319          *                            'at end'
 320          * @param height_modification_pos
 321          *                            The position where the additional content was
 322          *                            inserted/removed, defaults to -1 which means
 323          *                            'at end'
 324          *
 325          * @returns                   True is wanted modification is accepted false
 326          *                            otherwise.
 327          */
 328         bool content_resize_request(const int width_modification,
 329                         const int height_modification,
 330                         const int width_modification_pos = -1,
 331                         const int height_modification_pos = -1);
