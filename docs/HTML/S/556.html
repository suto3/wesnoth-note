  40 void password_box::set_value(const std::string& text)
  41 {
  42         real_value_ = text;
  43         size_t sz = utf8::size(text);
  44         utf8::string passwd;
  45         for(size_t i = 0; i < sz; i++) {
  46                 passwd.append(font::unicode_bullet);
  47         }
  48         text_box::set_value(passwd);
  49 }

  51 void password_box::delete_selection()
  52 {
  53         int len = get_selection_length();
  54         if(len == 0) {
  55                 return;
  56         }
  57         unsigned start = get_selection_start();
  58         if(len < 0) {
  59                 len = -len;
  60                 start -= len;
  61         }
  62 
  63         utf8::erase(real_value_, start, len);
  64         set_value(real_value_);
  65         set_cursor(start, false);
  66 }
  67 
  68 void password_box::insert_char(const utf8::string& unicode)
  69 {
  70         int len = get_selection_length();
  71         unsigned sel = get_selection_start();
  72         if(len < 0) {
  73                 len = -len;
  74                 sel -= len;
  75         }
  76 
  77         size_t sz = utf8::size(unicode);
  78         if(sz == 1) {
  79                 text_box::insert_char(font::unicode_bullet);
  80         } else {
  81                 utf8::string passwd;
  82                 for(size_t i = 0; i < sz; i++) {
  83                         passwd.append(font::unicode_bullet);
  84                 }
  85                 text_box::insert_char(passwd);
  86         }
  87         utf8::insert(real_value_, sel, unicode);
  88 }
  89 
  90 void password_box::paste_selection(const bool mouse)
  91 {
  92         const std::string& text = desktop::clipboard::copy_from_clipboard(mouse);
  93         if(text.empty()) {
  94                 return;
  95         }
  96         insert_char(text);
  97 }
