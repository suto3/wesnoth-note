  61 static std::string format_level_string(const int level)
  62 {
  63         std::string lvl = std::to_string(level);
  64 
  65         if(level < 1) {
  66                 return "<span color='#969696'>" + lvl + "</span>";
  67         } else if(level == 1) {
  68                 return lvl;
  69         } else if(level == 2) {
  70                 return "<b>" + lvl + "</b>";
  71         } else if(level > 2 ) {
  72                 return"<b><span color='#ffffff'>" + lvl + "</span></b>";
  73         }
  74 
  75         return lvl;
  76 }

  78 static std::string format_if_leader(unit_const_ptr u, const std::string str)
  79 {
  80         return (*u).can_recruit() ? "<span color='#cdad00'>" + str + "</span>" : str;
  81 }

  83 static std::string format_movement_string(unit_const_ptr u)
  84 {
  85         const int moves_left = (*u).movement_left();
  86         const int moves_max  = (*u).total_movement();
  87 
  88         std::string color = "#00ff00";
  89 
  90         if(moves_left == 0) {
  91                 color = "#ff0000";
  92         } else if(moves_left < moves_max) {
  93                 color = "#ffff00";
  94         }
  95 
  96         return formatter() << "<span color='" << color << "'>" << moves_left << "/" << moves_max << "</span>";
  97 }

  99 void unit_list::pre_show(window& window)
 100 {
 101         listbox& list = find_widget<listbox>(&window, "units_list", false);
 102 
 103 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 104         connect_signal_notify_modified(*list,
 105                         std::bind(&unit_list::list_item_clicked,
 106                                 *this, std::ref(window)));
 107 #else
 108         list.set_callback_value_change(
 109                         dialog_callback<unit_list, &unit_list::list_item_clicked>);
 110 #endif
 111 
 112         list.clear();
 113 
 114         window.keyboard_capture(&list);
 115 
 116         for(const unit_const_ptr& unit : unit_list_) {
 117                 std::map<std::string, string_map> row_data;
 118                 string_map column;
 119 
 120                 column["use_markup"] = "true";
 121 
 122                 column["label"] = format_if_leader(unit, unit->type_name());
 123                 row_data.emplace("unit_type", column);
 124 
 125                 const std::string& name = !unit->name().empty() ? format_if_leader(unit, unit->name().str()) : font::unicode_en_dash;
 126                 column["label"] = name;
 127                 row_data.emplace("unit_name", column);
 128 
 129                 column["label"] = format_movement_string(unit);
 130                 row_data.emplace("unit_moves", column);
 131 
 132                 std::stringstream hp_str;
 133                 hp_str << font::span_color(unit->hp_color()) << unit->hitpoints() << "/" << unit->max_hitpoints() << "</span>";
 134 
 135                 column["label"] = hp_str.str();
 136                 row_data.emplace("unit_hp", column);
 137 
 138                 column["label"] = format_level_string(unit->level());
 139                 row_data.emplace("unit_level", column);
 140 
 141                 std::stringstream exp_str;
 142                 exp_str << font::span_color(unit->xp_color()) << unit->experience() << "/"
 143                         << (unit->can_advance() ? std::to_string(unit->max_experience()) : font::unicode_en_dash) << "</span>";
 144 
 145                 column["label"] = exp_str.str();
 146                 row_data.emplace("unit_experience", column);
 147 
 148                 column["label"] = utils::join(unit->trait_names(), ", ");
 149                 row_data.emplace("unit_traits", column);
 150 
 151                 grid* row_grid = &list.add_row(row_data);
 152 
 153                 // NOTE: this needs to be done *after* the row is added
 154                 // TODO: show custom statuses
 155                 if(!unit->get_state(unit::STATE_PETRIFIED)) {
 156                         find_widget<image>(row_grid, "unit_status_petrified", false).set_visible(widget::visibility::invisible);
 157                 }
 158 
 159                 if(!unit->get_state(unit::STATE_POISONED)) {
 160                         find_widget<image>(row_grid, "unit_status_poisoned", false).set_visible(widget::visibility::invisible);
 161                 }
 162 
 163                 if(!unit->get_state(unit::STATE_SLOWED)) {
 164                         find_widget<image>(row_grid, "unit_status_slowed", false).set_visible(widget::visibility::invisible);
 165                 }
 166 
 167                 if(!unit->invisible(unit->get_location(), *resources::gameboard, false)) {
 168                         find_widget<image>(row_grid, "unit_status_invisible", false).set_visible(widget::visibility::invisible);
 169                 }
 170         }
 171 
 172         list.register_sorting_option(0, [this](const int i) { return unit_list_[i]->type_name().str(); });
 173         list.register_sorting_option(1, [this](const int i) { return unit_list_[i]->name().str(); });
 174         list.register_sorting_option(2, [this](const int i) { return unit_list_[i]->movement_left(); });
 175         list.register_sorting_option(3, [this](const int i) { return unit_list_[i]->hitpoints(); });
 176         list.register_sorting_option(4, [this](const int i) { return unit_list_[i]->level(); });
 177         list.register_sorting_option(5, [this](const int i) { return unit_list_[i]->experience(); });
 178         list.register_sorting_option(6, [this](const int i) {
 179                 return !unit_list_[i]->trait_names().empty() ? unit_list_[i]->trait_names().front().str() : ""; });
 180 
 181         list_item_clicked(window);
 182 }

 184 void unit_list::list_item_clicked(window& window)
 185 {
 186         const int selected_row
 187                 = find_widget<listbox>(&window, "units_list", false).get_selected_row();
 188 
 189         if(selected_row == -1) {
 190                 return;
 191         }
 192 
 193         find_widget<unit_preview_pane>(&window, "unit_details", false)
 194                 .set_displayed_unit(*unit_list_[selected_row].get());
 195 }

 197 void unit_list::post_show(window& window)
 198 {
 199         if(get_retval() == window::OK) {
 200                 const int selected_row = find_widget<listbox>(&window, "units_list", false).get_selected_row();
 201 
 202                 scroll_to_ = unit_list_[selected_row]->get_location();
 203         }

 206 void show_unit_list(display& gui)
 207 {
 208         unit_ptr_vector unit_list;
 209         map_location scroll_to;
 210 
 211         const unit_map& units = gui.get_units();
 212         for(unit_map::const_iterator i = units.begin(); i != units.end(); ++i) {
 213                 if(i->side() != gui.viewing_side()) {
 214                         continue;
 215                 }
 216 
 217                 unit_list.push_back(i.get_shared_ptr());
 218         }
 219 
 220         if(unit_list::execute(unit_list, scroll_to, gui.video())) {
 221                 gui.scroll_to_tile(scroll_to, display::WARP);
 222                 gui.select_hex(scroll_to);
 223         }
 224 }
