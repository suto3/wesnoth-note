  84 void mp_change_control::pre_show(window& window)
  85 {
  86         listbox& sides_list = find_widget<listbox>(&window, "sides_list", false);
  87         listbox& nicks_list = find_widget<listbox>(&window, "nicks_list", false);
  88 
  89 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  90         connect_signal_notify_modified(sides_list,
  91                 std::bind(&mp_change_control::handle_sides_list_item_clicked, this, std::ref(window)));
  92 
  93         connect_signal_notify_modified(nicks_list,
  94                 std::bind(&mp_change_control::handle_nicks_list_item_clicked, this, std::ref(window)));
  95 #else
  96         sides_list.set_callback_value_change(
  97                 dialog_callback<mp_change_control, &mp_change_control::handle_sides_list_item_clicked>);
  98 
  99         nicks_list.set_callback_value_change(
 100                 dialog_callback<mp_change_control, &mp_change_control::handle_nicks_list_item_clicked>);
 101 #endif
 102 
 103         //
 104         // Initialize sides list
 105         //
 106         const unsigned int num_sides = menu_handler_.board().teams().size();
 107 
 108         for(unsigned int side = 1; side <= num_sides; ++side) {
 109                 if(menu_handler_.board().get_team(side).hidden()) {
 110                         continue;
 111                 }
 112 
 113                 sides_.push_back(side);
 114 
 115                 std::map<std::string, string_map> data;
 116                 string_map item;
 117 
 118                 std::string side_str = vgettext("Side $side", {{"side", std::to_string(side)}});
 119                 side_str = font::span_color(team::get_side_color(side)) + side_str + "</span>";
 120 
 121                 item["id"] = (formatter() << "side_" << side).str();
 122                 item["label"] = side_str;
 123                 item["use_markup"] = "true";
 124                 data.emplace("side", item);
 125 
 126                 sides_list.add_row(data);
 127         }
 128 
 129         //
 130         // Prepare set of available nicknames
 131         //
 132         std::set<std::string> temp_nicks;
 133         for(const auto& team : menu_handler_.board().teams()) {
 134                 if(!team.is_local_ai() && !team.is_network_ai() && !team.is_idle()
 135                         && !team.is_empty() && !team.current_player().empty())
 136                 {
 137                         temp_nicks.insert(team.current_player());
 138                 }
 139         }
 140 
 141         const std::set<std::string>& observers = resources::screen->observers();
 142         temp_nicks.insert(observers.begin(), observers.end());
 143 
 144         // In case we are an observer, it isn't in the observers set and has to be added manually.
 145         temp_nicks.insert(preferences::login());
 146 
 147         //
 148         // Initialize nick list
 149         //
 150         for(const std::string& nick : temp_nicks) {
 151                 nicks_.push_back(nick);
 152 
 153                 std::map<std::string, string_map> data;
 154                 string_map item;
 155 
 156                 item["id"] = nick;
 157                 item["label"] = nick;
 158                 item["use_markup"] = "true";
 159                 data.emplace("nick", item);
 160 
 161                 nicks_list.add_row(data);
 162         }
 163 
 164         handle_sides_list_item_clicked(window);
 165         handle_nicks_list_item_clicked(window);
 166 }

 168 void mp_change_control::handle_sides_list_item_clicked(window& window)
 169 {
 170         selected_side_ = find_widget<listbox>(&window, "sides_list", false).get_selected_row();
 171 
 172         highlight_side_nick(window);
 173 }
 174 
 175 void mp_change_control::handle_nicks_list_item_clicked(window& window)
 176 {
 177         selected_nick_ = find_widget<listbox>(&window, "nicks_list", false).get_selected_row();
 178 }
 179 
 180 void mp_change_control::highlight_side_nick(window& window)
 181 {
 182         listbox& nicks_list = find_widget<listbox>(&window, "nicks_list", false);
 183         const auto& teams = menu_handler_.board().teams();
 184 
 185         int i = 0;
 186         for(const std::string& nick : nicks_) {
 187                 std::string label_str = "";
 188 
 189                 if(selected_side_ <= static_cast<unsigned int>(teams.size()) && teams.at(selected_side_).current_player() == nick) {
 190                         label_str = formatter() << "<b>" << nick << "</b>";
 191                 } else {
 192                         label_str = nick;
 193                 }
 194 
 195                 grid* row_grid = nicks_list.get_row_grid(i);
 196                 find_widget<label>(row_grid, nick, false).set_label(label_str);
 197 
 198                 ++i;
 199         }
 200 }

 202 void mp_change_control::post_show(window& window)
 203 {
 204         if(window.get_retval() == window::OK) {
 205                 DBG_GUI << "Main: changing control of side "
 206                         << sides_[selected_side_] << " to nick "
 207                         << nicks_[selected_nick_] << std::endl;
 208 
 209                 menu_handler_.request_control_change(
 210                         sides_[selected_side_],
 211                         nicks_[selected_nick_]
 212                 );
 213         }
 214 }
