  72 void connection::handle_connect(
  73                 const boost::system::error_code& ec,
  74                 resolver::iterator iterator
  75                 )
  76 {
  77         if(ec) {
  78                 WRN_NW << "Failed to connect to " <<
  79                         iterator->endpoint().address() << ": " <<
  80                         ec.message() << '\n';
  81                 socket_.close();
  82                 if(++iterator == resolver::iterator()) {
  83                         ERR_NW << "Tried all IPs. Giving up" << std::endl;
  84                         throw system_error(ec);
  85                 } else
  86                         connect(iterator);
  87         } else {
  88                 LOG_NW << "Connected to " << iterator->endpoint().address() << '\n';
  89                 handshake();
  90         }
  91 }

  93 void connection::handshake()
  94 {
  95         static const uint32_t handshake = 0;
  96         boost::asio::async_write(socket_,
  97                 boost::asio::buffer(reinterpret_cast<const char*>(&handshake), 4),
  98                 std::bind(&connection::handle_write, this, _1, _2)

 100         boost::asio::async_read(socket_,
 101                 boost::asio::buffer(&handshake_response_.binary, 4),

 115 void connection::transfer(const config& request, config& response)
 116 {
 117         io_service_.reset();
 118         done_ = false;
 119 
 120         std::ostream os(&write_buf_);
 121         write_gz(os, request);
 122         bytes_to_write_ = write_buf_.size() + 4;
 123         bytes_written_ = 0;
 124         payload_size_ = htonl(bytes_to_write_ - 4);
 125         boost::asio::streambuf::const_buffers_type gzipped_data = write_buf_.data();
 126         std::deque<boost::asio::const_buffer> bufs(gzipped_data.begin(), gzipped_data.end());
 127         bufs.push_front(boost::asio::buffer(reinterpret_cast<const char*>(&payload_size_), 4));
 128         boost::asio::async_write(socket_, bufs,
 129                 std::bind(&connection::is_write_complete, this, _1, _2),
 130                 std::bind(&connection::handle_write, this, _1, _2)
 131                 );
 132         boost::asio::async_read(socket_, read_buf_,
 133                 std::bind(&connection::is_read_complete, this, _1, _2),
 134                 std::bind(&connection::handle_read, this, _1, _2, std::ref(response))
 135                 );
 136 }

 138 void connection::cancel()
 139 {

 149 std::size_t connection::is_write_complete(
 150                 const boost::system::error_code& ec,
 151                 std::size_t bytes_transferred
 152                 )
 153 {
 154         if(ec)
 155                 throw system_error(ec);

 171 std::size_t connection::is_read_complete(
 172                 const boost::system::error_code& ec,
 173                 std::size_t bytes_transferred
 174                 )
 175 {
 176         if(ec) {
 177                 throw system_error(ec);
 178         }
 179         bytes_read_ = bytes_transferred;
 180         if(bytes_transferred < 4) {
 181                 return 4;
 182         } else {
 183                 if(!bytes_to_read_) {
 184                         std::istream is(&read_buf_);
 185                         union { char binary[4]; uint32_t num; } data_size;
 186                         is.read(data_size.binary, 4);
 187                         bytes_to_read_ = ntohl(data_size.num) + 4;
 188                         //Close immediately if we receive an invalid length
 189                         if (bytes_to_read_ < 4)
 190                                 bytes_to_read_ = bytes_transferred;
 191                 }
 192                 return bytes_to_read_ - bytes_transferred;
 193         }
 194 }

 196 void connection::handle_read(
 197         const boost::system::error_code& ec,
 198         std::size_t bytes_transferred,
 199         config& response
 200         )
 201 {
 202         DBG_NW << "Read " << bytes_transferred << " bytes.\n";
 203         bytes_to_read_ = 0;
 204         bytes_to_write_ = 0;
 205         done_ = true;
 206         if(ec && ec != boost::asio::error::eof)
 207                 throw system_error(ec);
 208         std::istream is(&read_buf_);
 209         read_gz(response, is);
 210 }
