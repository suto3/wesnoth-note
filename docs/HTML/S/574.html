 115 void chatbox::send_message_button_callback()
 116 {
 117         const std::string& input = chat_input_->get_value();
 118         if(input.empty()) {
 119                 return;
 120         }
 121 
 122         if(input[0] == '/') {
 123                 // TODO: refactor do_speak so it uses context information about
 124                 //      opened window, so e.g. /ignore in a whisper session ignores
 125                 //      the other party without having to specify it's nick.
 126                 chat_handler::do_speak(input);
 127         } else {
 128                 lobby_chat_window& t = open_windows_[active_window_];
 129 
 130                 if(t.whisper) {
 131                         send_whisper(t.name, input);
 132                         add_whisper_sent(t.name, input);
 133                 } else {
 134                         send_chat_room_message(t.name, input);
 135                         add_chat_room_message_sent(t.name, input);
 136                 }
 137         }
 138 
 139         chat_input_->save_to_history();
 140         chat_input_->set_value("");
 141 }

 144 void chatbox::chat_input_keypress_callback(bool& handled, bool& halt, const SDL_Keycode key)
 145 {
 146         if(key == SDLK_RETURN || key == SDLK_KP_ENTER) {
 147                 send_message_button_callback();
 148                 handled = true;
 149                 halt = true;
 150         } else if (key == SDLK_TAB) {
 151                 std::string text = chat_input_->get_value();
 152 
 153                 std::vector<std::string> matches;
 154                 for(const auto& ui : lobby_info_->users()) {
 155                         if(ui.name != preferences::login()) {
 156                                 matches.push_back(ui.name);
 157                         }
 158                 }
 159 
 160                 const bool line_start = utils::word_completion(text, matches);
 161 
 162                 if(matches.empty()) {
 163                         return;
 164                 }
 165 
 166                 if(matches.size() == 1) {
 167                         text.append(line_start ? ": " : " ");
 168                 } else {
 169                         std::string completion_list = utils::join(matches, " ");
 170                         append_to_chatbox(completion_list);
 171                 }
 172 
 173                 chat_input_->set_value(text);
 174 
 175                 handled = true;
 176                 halt = true;
 177         }
 178 }

 270 void chatbox::add_whisper_received(const std::string& sender, const std::string& message)
 271 {
 272         bool can_go_to_active = !preferences::whisper_friends_only() || preferences::is_friend(sender);
 273         bool can_open_new = preferences::auto_open_whisper_windows() && can_go_to_active;
 274 
 275         lobby_info_->get_whisper_log(sender).add_message(sender, message);
 276 
 277         if(whisper_window_open(sender, can_open_new)) {
 278                 if(whisper_window_active(sender)) {
 279                         add_active_window_message(sender, message);
 280 
 281                         do_notify(mp::NOTIFY_WHISPER, sender, message);
 282                 } else {
 283                         add_whisper_window_whisper(sender, message);
 284                         increment_waiting_whsipers(sender);
 285 
 286                         do_notify(mp::NOTIFY_WHISPER_OTHER_WINDOW, sender, message);
 287                 }
 288         } else if(can_go_to_active) {
 289                 add_active_window_whisper(sender, message);
 290                 do_notify(mp::NOTIFY_WHISPER, sender, message);
 291         } else {
 292                 LOG_LB << "Ignoring whisper from " << sender << "\n";
 293         }
 294 }

 296 void chatbox::add_chat_room_message_sent(const std::string& room,
 297         const std::string& message)
 298 {
 299         lobby_chat_window* t = room_window_open(room, false);
 300         if(!t) {
 301                 LOG_LB << "Cannot add sent message to ui for room " << room << ", player not in the room\n";
 302                 return;
 303         }
 304 
 305         // Do not open room window here. The player should be in the room before sending messages
 306         mp::room_info* ri = lobby_info_->get_room(room);
 307         assert(ri);
 308 
 309         if(!room_window_active(room)) {
 310                 switch_to_window(t);
 311         }
 312 
 313         ri->log().add_message(preferences::login(), message);
 314         add_active_window_message(preferences::login(), message, true);
 315 }

 317 void chatbox::add_chat_room_message_received(const std::string& room,
 318         const std::string& speaker,
 319         const std::string& message)
 320 {
 321         mp::room_info* ri = lobby_info_->get_room(room);
 322         if(!ri) {
 323                 LOG_LB << "Discarding message to room " << room << " from " << speaker << " (room not open)\n";
 324                 return;
 325         }
 326 
 327         mp::notify_mode notify_mode = mp::NOTIFY_NONE;
 328         ri->log().add_message(speaker, message);
 329 
 330         if(room_window_active(room)) {
 331                 add_active_window_message(speaker, message);
 332                 notify_mode = mp::NOTIFY_MESSAGE;
 333         } else {
 334                 add_room_window_message(room, speaker, message);
 335                 increment_waiting_messages(room);
 336                 notify_mode = mp::NOTIFY_MESSAGE_OTHER_WINDOW;
 337         }
 338 
 339         if(speaker == "server") {
 340                 notify_mode = mp::NOTIFY_SERVER_MESSAGE;
 341         } else if (utils::word_match(message, preferences::login())) {
 342                 notify_mode = mp::NOTIFY_OWN_NICK;
 343         } else if (preferences::is_friend(speaker)) {
 344                 notify_mode = mp::NOTIFY_FRIEND_MESSAGE;
 345         }
 346 
 347         do_notify(notify_mode, speaker, message);
 348 }

 372 lobby_chat_window* chatbox::search_create_window(const std::string& name,
 373         const bool whisper,
 374         const bool open_new,
 375         const bool allow_close)
 376 {
 377         for(auto& t : open_windows_) {
 378                 if(t.name == name && t.whisper == whisper) {
 379                         return &t;
 380                 }
 381         }
 382 
 383         if(!open_new) {
 384                 return nullptr;
 385         }
 386 
 387         open_windows_.push_back(lobby_chat_window(name, whisper));
 388 
 389         std::map<std::string, string_map> data;
 390         string_map item;
 391 
 392         item["use_markup"] = "true";
 393 
 394         if(whisper) {
 395                 item["label"] = vgettext("Whisper session with <i>“$name”</i> started. "
 396                         "If you do not want to receive messages from this user, type <i>/ignore $name</i>\n", {{"name", name}});
 397                 data.emplace("log_text", item);
 398         } else {
 399                 item["label"] = vgettext("Room <i>“$name”</i> joined", {{"name", name}});
 400                 data.emplace("log_text", item);
 401 
 402                 lobby_info_->open_room(name);
 403         }
 404 
 405         chat_log_container_->add_page(data);
 406 
 407         data.clear();
 408 
 409         item["label"] = whisper ? font::escape_text("<" + name + ">") : name;
 410         data.emplace("room", item);
 411 
 412         grid& row_grid = roomlistbox_->add_row(data);
 413 
 414         button& close_button = find_widget<button>(&row_grid, "close_window", false);
 415         connect_signal_mouse_left_click(close_button,
 416                 std::bind(&chatbox::close_window_button_callback, this, open_windows_.back(), _3, _4));
 417 
 418         if(!allow_close) {
 419                 close_button.set_visible(styled_widget::visibility::hidden);
 420         }
 421 
 422         return &open_windows_.back();
 423 }

 497 void chatbox::close_window(size_t idx)
 498 {
 499         const lobby_chat_window& t = open_windows_[idx];
 500         bool active_changed = idx == active_window_;
 501         DBG_LB << "Close window " << idx << " - " << t.name << "\n";
 502 
 503         if((t.name == "lobby" && t.whisper == false) || open_windows_.size() == 1) {
 504                 return;
 505         }
 506 
 507         if(t.whisper == false) {
 508                 // closing a room window -- send a part to the server
 509                 ::config data, msg;
 510                 msg["room"] = t.name;
 511                 msg["player"] = preferences::login();
 512                 data.add_child("room_part", msg);
 513                 send_to_server(data);
 514         }
 515 
 516         if(active_window_ == open_windows_.size() - 1) {
 517                 active_window_--;
 518         }
 519 
 520         if(t.whisper) {
 521                 lobby_info_->get_whisper_log(t.name).clear();
 522         } else {
 523                 lobby_info_->close_room(t.name);
 524         }
 525 
 526         open_windows_.erase(open_windows_.begin() + idx);
 527 
 528         roomlistbox_->remove_row(idx);
 529         roomlistbox_->select_row(active_window_);
 530 
 531         chat_log_container_->remove_page(idx);
 532         chat_log_container_->select_page(active_window_);
 533 
 534         if(active_changed) {
 535                 active_window_changed();
 536         }
 537 }

 572 void chatbox::process_room_join(const ::config& data)
 573 {
 574         const std::string& room = data["room"];
 575         const std::string& player = data["player"];
 576 
 577         DBG_LB << "room join: " << room << " " << player << "\n";
 578 
 579         mp::room_info* r = lobby_info_->get_room(room);
 580         if(r) {
 581                 if(player == preferences::login()) {
 582                         if(const auto& members = data.child("members")) {
 583                                 r->process_room_members(members);
 584                         }
 585                 } else {
 586                         r->add_member(player);
 587 
 588                         /* TODO: add/use preference */
 589                         add_room_window_message(room, "server", vgettext("$player has entered the room", {{"player", player}}));
 590                 }
 591 
 592                 if(r == active_window_room()) {
 593                         active_window_changed_callback_();
 594                 }
 595         } else {
 596                 if(player == preferences::login()) {
 597                         lobby_chat_window* t = room_window_open(room, true);
 598                         lobby_info_->open_room(room);
 599                         r = lobby_info_->get_room(room);
 600                         assert(r);
 601                         if(const auto& members = data.child("members")) {
 602                                 r->process_room_members(members);
 603                         }
 604                         switch_to_window(t);
 605 
 606                         const std::string& topic = data["topic"];
 607                         if(!topic.empty()) {
 608                                 add_chat_room_message_received("room", "server", room + ": " + topic);
 609                         }
 610                 } else {
 611                         LOG_LB << "Discarding join info for a room the player is not in\n";
 612                 }
 613         }
 614 }

 616 void chatbox::process_room_part(const ::config& data)
 617 {
 618         // TODO: close room window when the part message is sent
 619         const std::string& room = data["room"];
 620         const std::string& player = data["player"];
 621 
 622         DBG_LB << "Room part: " << room << " " << player << "\n";
 623 
 624         if(mp::room_info* r = lobby_info_->get_room(room)) {
 625                 r->remove_member(player);
 626 
 627                 /* TODO: add/use preference */
 628                 add_room_window_message(room, "server", vgettext("$player has left the room", {{"player", player}}));
 629                 if(active_window_room() == r) {
 630                         active_window_changed_callback_();
 631                 }
 632         } else {
 633                 LOG_LB << "Discarding part info for a room the player is not in\n";
 634         }
 635 }

 637 void chatbox::process_room_query_response(const ::config& data)
 638 {
 639         const std::string& room = data["room"];
 640         const std::string& message = data["message"];
 641 
 642         DBG_LB << "room query response: " << room << " " << message << "\n";
 643 
 644         if(room.empty()) {
 645                 if(!message.empty()) {
 646                         add_active_window_message("server", message);
 647                 }
 648 
 649                 if(const ::config& rooms = data.child("rooms")) {
 650                         // TODO: this should really open a nice join room dialog instead
 651                         std::stringstream ss;
 652                         ss << "Rooms:";
 653 
 654                         for(const auto & r : rooms.child_range("room")) {
 655                                 ss << " " << r["name"];
 656                         }
 657 
 658                         add_active_window_message("server", ss.str());
 659                 }
 660         } else {
 661                 if(room_window_open(room, false)) {
 662                         if(!message.empty()) {
 663                                 add_chat_room_message_received(room, "server", message);
 664                         }
 665 
 666                         if(const ::config& members = data.child("members")) {
 667                                 mp::room_info* r = lobby_info_->get_room(room);
 668                                 assert(r);
 669                                 r->process_room_members(members);
 670                                 if(r == active_window_room()) {
 671                                         active_window_changed_callback_();
 672                                 }
 673                         }
 674                 } else {
 675                         if(!message.empty()) {
 676                                 add_active_window_message("server", room + ": " + message);
 677                         }
 678                 }
 679         }
 680 }

 682 void chatbox::process_message(const ::config& data, bool whisper /*= false*/)
 683 {
 684         std::string sender = data["sender"];
 685         DBG_LB << "process message from " << sender << " " << (whisper ? "(w)" : "")
 686                 << ", len " << data["message"].str().size() << '\n';
 687 
 688         if(preferences::is_ignored(sender)) {
 689                 return;
 690         }
 691 
 692         const std::string& message = data["message"];
 693         preferences::parse_admin_authentication(sender, message);
 694 
 695         if(whisper) {
 696                 add_whisper_received(sender, message);
 697         } else {
 698                 std::string room = data["room"];
 699                 if(room.empty()) {
 700                         LOG_LB << "Message without a room from " << sender << ", assuming lobby\n";
 701                         room = "lobby";
 702                 }
 703 
 704                 add_chat_room_message_received(room, sender, message);
 705         }
 706 
 707         // Notify plugins about the message
 708         ::config plugin_data = data;
 709         plugin_data["whisper"] = whisper;
 710         plugins_manager::get()->notify_event("chat", plugin_data);
 711 }

 713 bool chatbox::process_network_data(const ::config& data)
 714 {
 715         if(const ::config& message = data.child("message")) {
 716                 process_message(message);
 717         } else if(const ::config& whisper = data.child("whisper")) {
 718                 process_message(whisper, true);
 719         } else if(const ::config& room_join = data.child("room_join")) {
 720                 process_room_join(room_join);
 721         } else if(const ::config& room_part = data.child("room_part")) {
 722                 process_room_part(room_part);
 723         } else if(const ::config& room_query_response = data.child("room_query_response")) {
 724                 process_room_query_response(room_query_response);
 725         }
 726 
 727         return false;
 728 }
