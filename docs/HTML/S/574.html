 114 void chatbox::send_message_button_callback()
 115 {
 116         const std::string& input = chat_input_->get_value();
 117         if(input.empty()) {
 118                 return;
 119         }
 120 
 121         if(input[0] == '/') {
 122                 // TODO: refactor do_speak so it uses context information about
 123                 //      opened window, so e.g. /ignore in a whisper session ignores
 124                 //      the other party without having to specify it's nick.
 125                 chat_handler::do_speak(input);
 126         } else {
 127                 lobby_chat_window& t = open_windows_[active_window_];
 128 
 129                 if(t.whisper) {
 130                         send_whisper(t.name, input);
 131                         add_whisper_sent(t.name, input);
 132                 } else {
 133                         send_chat_room_message(t.name, input);
 134                         add_chat_room_message_sent(t.name, input);
 135                 }
 136         }
 137 
 138         chat_input_->save_to_history();
 139         chat_input_->set_value("");
 140 }

 143 void chatbox::chat_input_keypress_callback(bool& handled, bool& halt, const SDL_Keycode key)
 144 {
 145         if(key == SDLK_RETURN || key == SDLK_KP_ENTER) {
 146                 send_message_button_callback();
 147                 handled = true;
 148                 halt = true;
 149         } else if (key == SDLK_TAB) {
 150                 std::string text = chat_input_->get_value();
 151 
 152                 std::vector<std::string> matches;
 153                 for(const auto& ui : lobby_info_->users()) {
 154                         if(ui.name != preferences::login()) {
 155                                 matches.push_back(ui.name);
 156                         }
 157                 }
 158 
 159                 const bool line_start = utils::word_completion(text, matches);
 160 
 161                 if(matches.empty()) {
 162                         return;
 163                 }
 164 
 165                 if(matches.size() == 1) {
 166                         text.append(line_start ? ": " : " ");
 167                 } else {
 168                         std::string completion_list = utils::join(matches, " ");
 169                         append_to_chatbox(completion_list);
 170                 }
 171 
 172                 chat_input_->set_value(text);
 173 
 174                 handled = true;
 175                 halt = true;
 176         }
 177 }

 269 void chatbox::add_whisper_received(const std::string& sender, const std::string& message)
 270 {
 271         bool can_go_to_active = !preferences::whisper_friends_only() || preferences::is_friend(sender);
 272         bool can_open_new = preferences::auto_open_whisper_windows() && can_go_to_active;
 273 
 274         lobby_info_->get_whisper_log(sender).add_message(sender, message);
 275 
 276         if(whisper_window_open(sender, can_open_new)) {
 277                 if(whisper_window_active(sender)) {
 278                         add_active_window_message(sender, message);
 279 
 280                         do_notify(mp::NOTIFY_WHISPER, sender, message);
 281                 } else {
 282                         add_whisper_window_whisper(sender, message);
 283                         increment_waiting_whsipers(sender);
 284 
 285                         do_notify(mp::NOTIFY_WHISPER_OTHER_WINDOW, sender, message);
 286                 }
 287         } else if(can_go_to_active) {
 288                 add_active_window_whisper(sender, message);
 289                 do_notify(mp::NOTIFY_WHISPER, sender, message);
 290         } else {
 291                 LOG_LB << "Ignoring whisper from " << sender << "\n";
 292         }
 293 }

 295 void chatbox::add_chat_room_message_sent(const std::string& room,
 296         const std::string& message)
 297 {
 298         lobby_chat_window* t = room_window_open(room, false);
 299         if(!t) {
 300                 LOG_LB << "Cannot add sent message to ui for room " << room << ", player not in the room\n";
 301                 return;
 302         }
 303 
 304         // Do not open room window here. The player should be in the room before sending messages
 305         mp::room_info* ri = lobby_info_->get_room(room);
 306         assert(ri);
 307 
 308         if(!room_window_active(room)) {
 309                 switch_to_window(t);
 310         }
 311 
 312         ri->log().add_message(preferences::login(), message);
 313         add_active_window_message(preferences::login(), message, true);
 314 }

 316 void chatbox::add_chat_room_message_received(const std::string& room,
 317         const std::string& speaker,
 318         const std::string& message)
 319 {
 320         mp::room_info* ri = lobby_info_->get_room(room);
 321         if(!ri) {
 322                 LOG_LB << "Discarding message to room " << room << " from " << speaker << " (room not open)\n";
 323                 return;
 324         }
 325 
 326         mp::notify_mode notify_mode = mp::NOTIFY_NONE;
 327         ri->log().add_message(speaker, message);
 328 
 329         if(room_window_active(room)) {
 330                 add_active_window_message(speaker, message);
 331                 notify_mode = mp::NOTIFY_MESSAGE;
 332         } else {
 333                 add_room_window_message(room, speaker, message);
 334                 increment_waiting_messages(room);
 335                 notify_mode = mp::NOTIFY_MESSAGE_OTHER_WINDOW;
 336         }
 337 
 338         if(speaker == "server") {
 339                 notify_mode = mp::NOTIFY_SERVER_MESSAGE;
 340         } else if (utils::word_match(message, preferences::login())) {
 341                 notify_mode = mp::NOTIFY_OWN_NICK;
 342         } else if (preferences::is_friend(speaker)) {
 343                 notify_mode = mp::NOTIFY_FRIEND_MESSAGE;
 344         }
 345 
 346         do_notify(notify_mode, speaker, message);
 347 }

 371 lobby_chat_window* chatbox::search_create_window(const std::string& name,
 372         const bool whisper,
 373         const bool open_new,
 374         const bool allow_close)
 375 {
 376         for(auto& t : open_windows_) {
 377                 if(t.name == name && t.whisper == whisper) {
 378                         return &t;
 379                 }
 380         }
 381 
 382         if(!open_new) {
 383                 return nullptr;
 384         }
 385 
 386         open_windows_.push_back(lobby_chat_window(name, whisper));
 387 
 388         std::map<std::string, string_map> data;
 389         string_map item;
 390 
 391         item["use_markup"] = "true";
 392 
 393         if(whisper) {
 394                 item["label"] = vgettext("Whisper session with <i>$name</i> started. "
 395                         "If you do not want to receive messages from this user, type <i>/ignore $name</i>\n", {{"name", name}});
 396                 data.emplace("log_text", item);
 397         } else {
 398                 item["label"] = vgettext("Room <i>$name</i> joined", {{"name", name}});
 399                 data.emplace("log_text", item);
 400 
 401                 lobby_info_->open_room(name);
 402         }
 403 
 404         chat_log_container_->add_page(data);
 405 
 406         data.clear();
 407 
 408         item["label"] = whisper ? font::escape_text("<" + name + ">") : name;
 409         data.emplace("room", item);
 410 
 411         grid& row_grid = roomlistbox_->add_row(data);
 412 
 413         button& close_button = find_widget<button>(&row_grid, "close_window", false);
 414         connect_signal_mouse_left_click(close_button,
 415                 std::bind(&chatbox::close_window_button_callback, this, open_windows_.back(), _3, _4));
 416 
 417         if(!allow_close) {
 418                 close_button.set_visible(styled_widget::visibility::hidden);
 419         }
 420 
 421         return &open_windows_.back();
 422 }

 496 void chatbox::close_window(size_t idx)
 497 {
 498         const lobby_chat_window& t = open_windows_[idx];
 499         bool active_changed = idx == active_window_;
 500         DBG_LB << "Close window " << idx << " - " << t.name << "\n";
 501 
 502         if((t.name == "lobby" && t.whisper == false) || open_windows_.size() == 1) {
 503                 return;
 504         }
 505 
 506         if(t.whisper == false) {
 507                 // closing a room window -- send a part to the server
 508                 ::config data, msg;
 509                 msg["room"] = t.name;
 510                 msg["player"] = preferences::login();
 511                 data.add_child("room_part", msg);
 512                 send_to_server(data);
 513         }
 514 
 515         if(active_window_ == open_windows_.size() - 1) {
 516                 active_window_--;
 517         }
 518 
 519         if(t.whisper) {
 520                 lobby_info_->get_whisper_log(t.name).clear();
 521         } else {
 522                 lobby_info_->close_room(t.name);
 523         }
 524 
 525         open_windows_.erase(open_windows_.begin() + idx);
 526 
 527         roomlistbox_->remove_row(idx);
 528         roomlistbox_->select_row(active_window_);
 529 
 530         chat_log_container_->remove_page(idx);
 531         chat_log_container_->select_page(active_window_);
 532 
 533         if(active_changed) {
 534                 active_window_changed();
 535         }
 536 }

 571 void chatbox::process_room_join(const ::config& data)
 572 {
 573         const std::string& room = data["room"];
 574         const std::string& player = data["player"];
 575 
 576         DBG_LB << "room join: " << room << " " << player << "\n";
 577 
 578         mp::room_info* r = lobby_info_->get_room(room);
 579         if(r) {
 580                 if(player == preferences::login()) {
 581                         if(const auto& members = data.child("members")) {
 582                                 r->process_room_members(members);
 583                         }
 584                 } else {
 585                         r->add_member(player);
 586 
 587                         /* TODO: add/use preference */
 588                         add_room_window_message(room, "server", vgettext("$player has entered the room", {{"player", player}}));
 589                 }
 590 
 591                 if(r == active_window_room()) {
 592                         active_window_changed_callback_();
 593                 }
 594         } else {
 595                 if(player == preferences::login()) {
 596                         lobby_chat_window* t = room_window_open(room, true);
 597                         lobby_info_->open_room(room);
 598                         r = lobby_info_->get_room(room);
 599                         assert(r);
 600                         if(const auto& members = data.child("members")) {
 601                                 r->process_room_members(members);
 602                         }
 603                         switch_to_window(t);
 604 
 605                         const std::string& topic = data["topic"];
 606                         if(!topic.empty()) {
 607                                 add_chat_room_message_received("room", "server", room + ": " + topic);
 608                         }
 609                 } else {
 610                         LOG_LB << "Discarding join info for a room the player is not in\n";
 611                 }
 612         }
 613 }

 627                 add_room_window_message(room, "server", vgettext("$player has left the room", {{"player", player}}));
 628                 if(active_window_room() == r) {
 629                         active_window_changed_callback_();
 630                 }
 631         } else {
 632                 LOG_LB << "Discarding part info for a room the player is not in\n";
 633         }
 634 }

 636 void chatbox::process_room_query_response(const ::config& data)
 637 {
 638         const std::string& room = data["room"];
 639         const std::string& message = data["message"];
 640 
 641         DBG_LB << "room query response: " << room << " " << message << "\n";
 642 
 643         if(room.empty()) {
 644                 if(!message.empty()) {
 645                         add_active_window_message("server", message);
 646                 }
 647 
 648                 if(const ::config& rooms = data.child("rooms")) {
 649                         // TODO: this should really open a nice join room dialog instead
 650                         std::stringstream ss;
 651                         ss << "Rooms:";
 652 
 653                         for(const auto & r : rooms.child_range("room")) {
 654                                 ss << " " << r["name"];
 655                         }
 656 
 657                         add_active_window_message("server", ss.str());
 658                 }
 659         } else {
 660                 if(room_window_open(room, false)) {
 661                         if(!message.empty()) {
 662                                 add_chat_room_message_received(room, "server", message);
 663                         }
 664 
 665                         if(const ::config& members = data.child("members")) {
 666                                 mp::room_info* r = lobby_info_->get_room(room);
 667                                 assert(r);
 668                                 r->process_room_members(members);
 669                                 if(r == active_window_room()) {
 670                                         active_window_changed_callback_();
 671                                 }
 672                         }
 673                 } else {
 674                         if(!message.empty()) {
 675                                 add_active_window_message("server", room + ": " + message);
 676                         }
 677                 }
 678         }
 679 }

 681 void chatbox::process_message(const ::config& data, bool whisper /*= false*/)
 682 {
 683         std::string sender = data["sender"];
 684         DBG_LB << "process message from " << sender << " " << (whisper ? "(w)" : "")
 685                 << ", len " << data["message"].str().size() << '\n';
 686 
 687         if(preferences::is_ignored(sender)) {
 688                 return;
 689         }
 690 
 691         const std::string& message = data["message"];
 692         preferences::parse_admin_authentication(sender, message);
 693 
 694         if(whisper) {
 695                 add_whisper_received(sender, message);
 696         } else {
 697                 std::string room = data["room"];
 698                 if(room.empty()) {
 699                         LOG_LB << "Message without a room from " << sender << ", assuming lobby\n";
 700                         room = "lobby";
 701                 }
 702 
 703                 add_chat_room_message_received(room, sender, message);
 704         }
 705 
 706         // Notify plugins about the message
 707         ::config plugin_data = data;
 708         plugin_data["whisper"] = whisper;
 709         plugins_manager::get()->notify_event("chat", plugin_data);
 710 }

 712 bool chatbox::process_network_data(const ::config& data)
 713 {
 714         if(const ::config& message = data.child("message")) {
 715                 process_message(message);
 716         } else if(const ::config& whisper = data.child("whisper")) {
 717                 process_message(whisper, true);
 718         } else if(const ::config& room_join = data.child("room_join")) {
 719                 process_room_join(room_join);
 720         } else if(const ::config& room_part = data.child("room_part")) {
 721                 process_room_part(room_part);
 722         } else if(const ::config& room_query_response = data.child("room_query_response")) {
 723                 process_room_query_response(room_query_response);
 724         }
 725 
 726         return false;
 727 }
