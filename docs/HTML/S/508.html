  71 window* build(CVideo& video, const builder_window::window_resolution* definition)
  72 {
  73         // We set the values from the definition since we can only determine the
  74         // best size (if needed) after all widgets have been placed.
  75         window* win = new window(video, definition);
  76         assert(win);
  77 
  78         for(const auto& lg : definition->linked_groups) {
  79                 if(win->has_linked_size_group(lg.id)) {
  80                         t_string msg = vgettext("Linked '$id' group has multiple definitions.", {{"id", lg.id}});
  81 
  82                         FAIL(msg);
  83                 }
  84 
  85                 win->init_linked_size_group(lg.id, lg.fixed_width, lg.fixed_height);
  86         }
  87 
  88         win->set_click_dismiss(definition->click_dismiss);
  89 
  90         const auto conf = win->cast_config_to<window_definition>();
  91         assert(conf);
  92 
  93         if(conf->grid) {
  94                 win->init_grid(conf->grid);
  95                 win->finalize(definition->grid);
  96         } else {
  97                 win->init_grid(definition->grid);
  98         }
  99 
 100         win->add_to_keyboard_chain(win);
 101 
 102         return win;
 103 }

 126 builder_widget_ptr create_builder_widget(const config& cfg)
 127 {
 128         config::const_all_children_itors children = cfg.all_children_range();
 129         VALIDATE(children.size() == 1, "Grid cell does not have exactly 1 child.");
 130 
 131         for(const auto& item : builder_widget_lookup()) {
 132                 if(item.first == "window" || item.first == "tooltip") {
 133                         continue;
 134                 }
 135                 if(const config& c = cfg.child(item.first)) {
 136                         return item.second(c);
 137                 }
 138         }
 139 
 140         if(const config& c = cfg.child("grid")) {
 141                 return std::make_shared<builder_grid>(c);
 142         }
 143 
 144         if(const config& instance = cfg.child("instance")) {
 145                 return std::make_shared<implementation::builder_instance>(instance);
 146         }
 147 
 148         if(const config& pane = cfg.child("pane")) {
 149                 return std::make_shared<implementation::builder_pane>(pane);
 150         }
 151 
 152         if(const config& viewport = cfg.child("viewport")) {
 153                 return std::make_shared<implementation::builder_viewport>(viewport);
 154         }

 209         VALIDATE(iter != builder_widget_lookup().end(), "Invalid widget type '" + type + "'");

 356 builder_window::window_resolution::window_resolution(const config& cfg)
 357         : window_width(cfg["window_width"])
 358         , window_height(cfg["window_height"])
 359         , automatic_placement(cfg["automatic_placement"].to_bool(true))
 360         , x(cfg["x"])
 361         , y(cfg["y"])
 362         , width(cfg["width"])
 363         , height(cfg["height"])
 364         , reevaluate_best_size(cfg["reevaluate_best_size"])
 365         , functions()
 366         , vertical_placement(implementation::get_v_align(cfg["vertical_placement"]))
 367         , horizontal_placement(implementation::get_h_align(cfg["horizontal_placement"]))
 368         , maximum_width(cfg["maximum_width"])
 369         , maximum_height(cfg["maximum_height"])
 370         , click_dismiss(cfg["click_dismiss"].to_bool())
 371         , definition(cfg["definition"])
 372         , linked_groups()
 373         , tooltip(cfg.child_or_empty("tooltip"), "tooltip")
 374         , helptip(cfg.child_or_empty("helptip"), "helptip")
 375         , grid(nullptr)
 376 {
 377         if(!cfg["functions"].empty()) {
 378                 wfl::formula(cfg["functions"], &functions).evaluate();
 379         }
 380 
 381         const config& c = cfg.child("grid");
 382 
 383         VALIDATE(c, _("No grid defined."));
 384 
 385         grid = std::make_shared<builder_grid>(builder_grid(c));
 386 
 387         if(!automatic_placement) {
 388                 VALIDATE(width.has_formula() || width(), missing_mandatory_wml_key("resolution", "width"));
 389                 VALIDATE(height.has_formula() || height(), missing_mandatory_wml_key("resolution", "height"));
 390         }
 391 
 392         DBG_GUI_P << "Window builder: parsing resolution " << window_width << ',' << window_height << '\n';
 393 
 394         if(definition.empty()) {
 395                 definition = "default";
 396         }
 397 
 398         linked_groups = parse_linked_group_definitions(cfg);
 399 }

 440  *     border & border & "" &          Where to place the border in this grid

 445  *                                     the grid cell. (This value is ignored if
 446  *                                     vertical_grow is true.) $
 447  *     horizontal_alignment & h_align & "" &
 448  *                                     The horizontal alignment of the widget in
 449  *                                     the grid cell.(This value is ignored if
 450  *                                     horizontal_grow is true.) $
 451  *
 452  *     vertical_grow & bool & false &    Does the widget grow in vertical
 453  *                                     direction when the grid cell grows in the
 454  *                                     vertical direction. This is used if the
 455  *                                     grid cell is wider as the best width for

 457  *     horizontal_grow & bool & false &  Does the widget grow in horizontal
 458  *                                     direction when the grid cell grows in the
 459  *                                     horizontal direction. This is used if the
 460  *                                     grid cell is higher as the best width for
 461  *                                     the widget. $
 462  * @end{table}
 463  * @end{tag}{name="column"}
 464  * @end{tag}{name="row"}
 465  * @end{tag}{name="grid"}
 466  * @end{parent}{name="gui/window/resolution/"}
 467  *

 469 builder_grid::builder_grid(const config& cfg)
 470         : builder_widget(cfg)
 471         , rows(0)
 472         , cols(0)
 473         , row_grow_factor()
 474         , col_grow_factor()
 475         , flags()
 476         , border_size()
 477         , widgets()
 478 {
 479         log_scope2(log_gui_parse, "Window builder: parsing a grid");
 480 
 481         for(const auto& row : cfg.child_range("row")) {
 482                 unsigned col = 0;
 483 
 484                 row_grow_factor.push_back(row["grow_factor"]);
 485 
 486                 for(const auto& c : row.child_range("column")) {
 487                         flags.push_back(implementation::read_flags(c));
 488                         border_size.push_back(c["border_size"]);
 489                         if(rows == 0) {
 490                                 col_grow_factor.push_back(c["grow_factor"]);
 491                         }
 492 
 493                         widgets.push_back(create_builder_widget(c));
 494 
 495                         ++col;
 496                 }
 497 
 498                 ++rows;
 499                 if(rows == 1) {
 500                         cols = col;
 501                 } else {
 502                         VALIDATE(col, _("A row must have a column."));
 503                         VALIDATE(col == cols, _("Number of columns differ."));
 504                 }
 505         }
 506 
 507         DBG_GUI_P << "Window builder: grid has " << rows << " rows and " << cols << " columns.\n";
 508 }

 522 grid* builder_grid::build(grid* grid) const
 523 {
 524         grid->set_id(id);
 525         grid->set_linked_group(linked_group);
 526         grid->set_rows_cols(rows, cols);
 527 
 528         log_scope2(log_gui_general, "Window builder: building grid");
 529 
 530         DBG_GUI_G << "Window builder: grid '" << id << "' has " << rows << " rows and " << cols << " columns.\n";
 531 
 532         for(unsigned x = 0; x < rows; ++x) {
 533                 grid->set_row_grow_factor(x, row_grow_factor[x]);
 534 
 535                 for(unsigned y = 0; y < cols; ++y) {
 536                         if(x == 0) {
 537                                 grid->set_column_grow_factor(y, col_grow_factor[y]);
 538                         }
 539 
 540                         DBG_GUI_G << "Window builder: adding child at " << x << ',' << y << ".\n";
 541 
 542                         const unsigned int i = x * cols + y;
 543 
 544                         widget* widget = widgets[i]->build();
 545                         grid->set_child(widget, x, y, flags[i], border_size[i]);
 546                 }
 547         }
 548 
 549         return grid;
 550 }
