  57 void context::add_handler(sdl_handler* ptr)
  58 {
  59         /* Add new handlers to the staging list initially.
  60         This ensures that if an event handler adds more handlers, the new handlers
  61         won't be called for the event that caused them to be added. */
  62         staging_handlers.push_back(ptr);

  65 bool context::remove_handler(sdl_handler* ptr)
  66 {
  67         static int depth = 0;
  68         ++depth;
  69 
  70         // The handler is most likely on the back of the events list,
  71         // so look there first, otherwise do a complete search.
  72         if(!handlers.empty() && handlers.back() == ptr) {
  73                 if(focused_handler != handlers.end() && *focused_handler == ptr) {
  74                         focused_handler = handlers.end();
  75                 }
  76 
  77                 handlers.pop_back();
  78         } else {
  79                 const handler_list::iterator i = std::find(handlers.begin(), handlers.end(), ptr);
  80 
  81                 if(i == handlers.end()) {
  82                         --depth;
  83 
  84                         // The handler may be in the staging area. Search it from there.
  85                         auto j = std::find(staging_handlers.begin(), staging_handlers.end(), ptr);
  86                         if (j != staging_handlers.end())
  87                         {
  88                                 staging_handlers.erase(j);
  89                                 return true;
  90                         }
  91                         else
  92                         {
  93                                 return false;
  94                         }
  95                 }
  96 
  97                 if(i == focused_handler) {
  98                         focused_handler != handlers.begin() ? --focused_handler : ++focused_handler;
  99                 }
 100 
 101                 handlers.erase(i);
 102         }
 103 
 104         --depth;
 105 
 106         if(depth == 0) {
 107                 cycle_focus();
 108         } else {
 109                 focused_handler = handlers.end();
 110         }
 111 
 112         return true;
 113 }

 115 void context::cycle_focus()
 116 {
 117         if (handlers.begin() == handlers.end()) {
 118                 return;
 119         }
 120 
 121         handler_list::iterator current = focused_handler;
 122         handler_list::iterator last = focused_handler;
 123 
 124         if (last != handlers.begin()) {
 125                 --last;
 126         }
 127 
 128         if (current == handlers.end()) {
 129                 current = handlers.begin();
 130         } else {
 131                 ++current;
 132         }
 133 
 134         while (current != last) {
 135 
 136                 if(current != handlers.end() && (*current)->requires_event_focus()) {
 137                         focused_handler = current;
 138                         break;
 139                 }
 140 
 141                 if (current == handlers.end()) {
 142                         current = handlers.begin();
 143                 } else {
 144                         ++current;
 145                 }
 146         }
 147 }

 241 void sdl_handler::join(context &c)
 242 {
 243         if (has_joined_global_) {
 244                 leave_global();
 245         }
 246 
 247         if(has_joined_) {
 248                 leave(); // should not be in multiple event contexts
 249         }
 250 
 251         //join self
 252         c.add_handler(this);
 253         has_joined_ = true;
 254 
 255         //instruct members to join
 256         sdl_handler_vector members = handler_members();
 257         if(!members.empty()) {
 258                 for(sdl_handler_vector::iterator i = members.begin(); i != members.end(); ++i) {
 259                         (*i)->join(c);
 260                 }
 261         }

 264 void sdl_handler::join_same(sdl_handler* parent)
 265 {
 266         if(has_joined_) {
 267                 leave(); // should not be in multiple event contexts
 268         }
 269 
 270         for(std::deque<context>::reverse_iterator i = event_contexts.rbegin(); i != event_contexts.rend(); ++i) {
 271                 handler_list& handlers = (*i).handlers;
 272                 if (std::find(handlers.begin(), handlers.end(), parent) != handlers.end()) {
 273                         join(*i);
 274                         return;
 275                 }
 276         }
 277 
 278         join(event_contexts.back());

 281 void sdl_handler::leave()
 282 {
 283         sdl_handler_vector members = handler_members();
 284         if(!members.empty()) {
 285                 for(sdl_handler_vector::iterator i = members.begin(); i != members.end(); ++i) {
 286                         (*i)->leave();
 287                 }
 288         } else {
 289                 assert(event_contexts.empty() == false);
 290         }
 291         for(std::deque<context>::reverse_iterator i = event_contexts.rbegin(); i != event_contexts.rend(); ++i) {
 292                 if(i->remove_handler(this)) {
 293                         break;
 294                 }
 295         }
 296         has_joined_ = false;
 297 }

 342 bool has_focus(const sdl_handler* hand, const SDL_Event* event)
 343 {
 344         if(event_contexts.empty()) {
 345                 return true;
 346         }
 347 
 348         if(hand->requires_event_focus(event) == false) {
 349                 return true;
 350         }
 351 
 352         const handler_list::iterator foc = event_contexts.back().focused_handler;
 353         auto& handlers = event_contexts.back().handlers;
 354 
 355         // If no-one has focus at the moment, this handler obviously wants
 356         // focus, so give it to it.
 357         if(foc == handlers.end()) {
 358                 focus_handler(hand);
 359                 return true;
 360         }
 361 
 362         sdl_handler* const foc_hand = *foc;
 363         if(foc_hand == hand){
 364                 return true;
 365         } else if(!foc_hand->requires_event_focus(event)) {
 366                 // If the currently focused handler doesn't need focus for this event
 367                 // allow the most recent interested handler to take care of it
 368                 for(auto i = handlers.rbegin(); i != handlers.rend(); ++i) {
 369                         sdl_handler* const thief_hand = *i;
 370 
 371                         if(thief_hand != foc_hand && thief_hand->requires_event_focus(event)) {
 372                                 // Steal focus
 373                                 focus_handler(thief_hand);
 374 
 375                                 // Position the previously focused handler to allow stealing back
 376                                 handlers.splice(handlers.end(), handlers, foc);
 377 
 378                                 return thief_hand == hand;
 379                         }
 380                 }
 381         }
 382         return false;
 383 }

 390 static bool remove_on_resize(const SDL_Event &a) {
 391         if (a.type == DRAW_EVENT || a.type == DRAW_ALL_EVENT) {
 392                 return true;
 393         }
 394         if (a.type == SHOW_HELPTIP_EVENT) {
 395                 return true;
 396         }
 397         if ((a.type == SDL_WINDOWEVENT) &&
 398                         (a.window.event == SDL_WINDOWEVENT_RESIZED ||
 399                                         a.window.event == SDL_WINDOWEVENT_SIZE_CHANGED ||
 400                                         a.window.event == SDL_WINDOWEVENT_EXPOSED)) {
 401                 return true;
 402         }
 403 
 404         return false;
 405 }

 407 // TODO: I'm uncertain if this is always safe to call at static init; maybe set in main() instead?
 408 static const boost::thread::id main_thread = boost::this_thread::get_id();
 409 void pump()
 410 {
 411         if(boost::this_thread::get_id() != main_thread) {
 412                 // Can only call this on the main thread!
 413                 return;
 414         }
 415         SDL_PumpEvents();
 416         peek_for_resize();
 417         pump_info info;
 418 
 419         //used to keep track of double click events
 420         static int last_mouse_down = -1;
 421         static int last_click_x = -1, last_click_y = -1;
 422 
 423         SDL_Event temp_event;
 424         int poll_count = 0;
 425         int begin_ignoring = 0;
 426         std::vector< SDL_Event > events;
 427         while(SDL_PollEvent(&temp_event)) {
 428 
 429                 if (temp_event.type == INVOKE_FUNCTION_EVENT) {
 430                         static_cast<invoked_function_data*>(temp_event.user.data1)->call();
 431                         continue;
 432                 }
 433 
 434                 ++poll_count;
 435                 peek_for_resize();
 436 
 437                 if(!begin_ignoring && temp_event.type == SDL_WINDOWEVENT
 438                                 && (temp_event.window.event == SDL_WINDOWEVENT_ENTER
 439                                                 || temp_event.window.event == SDL_WINDOWEVENT_FOCUS_GAINED))
 440                 {
 441                         begin_ignoring = poll_count;
 442                 } else if(begin_ignoring > 0 && is_input(temp_event)) {
 443                         //ignore user input events that occurred after the window was activated
 444                         continue;
 445                 }
 446                 events.push_back(temp_event);
 447         }
 448 
 449         std::vector<SDL_Event>::iterator ev_it = events.begin();
 450         for(int i=1; i < begin_ignoring; ++i){
 451                 if(is_input(*ev_it)) {
 452                         //ignore user input events that occurred before the window was activated
 453                         ev_it = events.erase(ev_it);
 454                 } else {
 455                         ++ev_it;
 456                 }
 457         }
 458 
 459         std::vector<SDL_Event>::iterator ev_end = events.end();
 460         bool resize_found = false;
 461         for(ev_it = events.begin(); ev_it != ev_end; ++ev_it){
 462                 SDL_Event &event = *ev_it;
 463                 if (event.type == SDL_WINDOWEVENT &&
 464                                 event.window.event == SDL_WINDOWEVENT_RESIZED) {
 465                         resize_found = true;
 466                         last_resize_event = event;
 467                         last_resize_event_used = false;
 468 
 469                 }
 470         }
 471         // remove all inputs, draw events and only keep the last of the resize events
 472         // This will turn horrible after ~38 days when the Uint32 wraps.
 473         if (resize_found || SDL_GetTicks() <= last_resize_event.window.timestamp + resize_timeout) {
 474                 events.erase(std::remove_if(events.begin(), events.end(), remove_on_resize), events.end());
 475         } else if(SDL_GetTicks() > last_resize_event.window.timestamp + resize_timeout && !last_resize_event_used) {
 476                 events.insert(events.begin(), last_resize_event);
 477                 last_resize_event_used = true;
 478         }
 479 
 480         // move all draw events to the end of the queue
 481         auto first_draw_event = std::stable_partition(events.begin(), events.end(),
 482                 [](const SDL_Event& e) {return e.type != DRAW_EVENT;});
 483         if(first_draw_event != events.end()) {
 484                 // remove all draw events except one
 485                 events.erase(first_draw_event + 1, events.end());
 486         }
 487 
 488         ev_end = events.end();
 489 
 490         for(ev_it = events.begin(); ev_it != ev_end; ++ev_it){
 491                 for (context& c : event_contexts)
 492                 {
 493                         c.add_staging_handlers();
 494                 }
 495 
 496                 SDL_Event &event = *ev_it;
 497                 switch(event.type) {
 498 
 499                         case SDL_WINDOWEVENT:
 500                                 switch(event.window.event) {
 501                                         case SDL_WINDOWEVENT_ENTER:
 502                                         case SDL_WINDOWEVENT_FOCUS_GAINED:
 503                                                 cursor::set_focus(1);
 504                                                 break;
 505 
 506                                         case SDL_WINDOWEVENT_LEAVE:
 507                                         case SDL_WINDOWEVENT_FOCUS_LOST:
 508                                                 cursor::set_focus(1);
 509                                                 break;
 510 
 511                                         case SDL_WINDOWEVENT_RESIZED:
 512                                                 info.resize_dimensions.first = event.window.data1;
 513                                                 info.resize_dimensions.second = event.window.data2;
 514                                                 break;
 515                                 }
 516                                 //make sure this runs in it's own scope.
 517                                 {
 518                                         flip_locker flip_lock(CVideo::get_singleton());
 519                                         for( std::deque<context>::iterator i = event_contexts.begin() ; i != event_contexts.end(); ++i) {
 520                                                 const handler_list& event_handlers = (*i).handlers;
 521                                                 for(auto handler : event_handlers) {
 522                                                         handler->handle_window_event(event);
 523                                                 }
 524                                         }
 525                                         const handler_list& event_handlers = event_contexts.front().handlers;
 526                                         for(auto handler : event_handlers) {
 527                                                 handler->handle_window_event(event);
 528                                         }
 529                                 }
 530 
 531                                 //This event was just distributed, don't re-distribute.
 532                                 continue;
 533 
 534                         case SDL_MOUSEMOTION: {
 535                                 //always make sure a cursor is displayed if the
 536                                 //mouse moves or if the user clicks
 537                                 cursor::set_focus(true);
 538                                 raise_help_string_event(event.motion.x,event.motion.y);
 539                                 break;
 540                         }
 541 
 542                         case SDL_MOUSEBUTTONDOWN: {
 543                                 //always make sure a cursor is displayed if the
 544                                 //mouse moves or if the user clicks
 545                                 cursor::set_focus(true);
 546                                 if(event.button.button == SDL_BUTTON_LEFT) {
 547                                         static const int DoubleClickTime = 500;
 548                                         static const int DoubleClickMaxMove = 3;
 549                                         if(last_mouse_down >= 0 && info.ticks() - last_mouse_down < DoubleClickTime &&
 550                                            std::abs(event.button.x - last_click_x) < DoubleClickMaxMove &&
 551                                            std::abs(event.button.y - last_click_y) < DoubleClickMaxMove) {
 552                                                 SDL_UserEvent user_event;
 553                                                 user_event.type = DOUBLE_CLICK_EVENT;
 554                                                 user_event.code = 0;
 555                                                 user_event.data1 = reinterpret_cast<void*>(event.button.x);
 556                                                 user_event.data2 = reinterpret_cast<void*>(event.button.y);
 557                                                 ::SDL_PushEvent(reinterpret_cast<SDL_Event*>(&user_event));
 558                                         }
 559                                         last_mouse_down = info.ticks();
 560                                         last_click_x = event.button.x;
 561                                         last_click_y = event.button.y;
 562                                 }
 563                                 break;
 564                         }
 565                         case DRAW_ALL_EVENT:
 566                         {
 567                                 flip_locker flip_lock(CVideo::get_singleton());
 568                                 /* iterate backwards as the most recent things will be at the top */
 569                                 for( std::deque<context>::iterator i = event_contexts.begin() ; i != event_contexts.end(); ++i) {
 570                                         handler_list& event_handlers = (*i).handlers;
 571                                         for( handler_list::iterator i1 = event_handlers.begin(); i1 != event_handlers.end(); ++i1) {
 572                                                 (*i1)->handle_event(event);
 573                                         }
 574                                 }
 575                                 continue; //do not do further handling here
 576                         }
 577 
 578 #ifndef __APPLE__
 579                         case SDL_KEYDOWN: {
 580                                 if(event.key.keysym.sym == SDLK_F4 && (event.key.keysym.mod == KMOD_RALT || event.key.keysym.mod == KMOD_LALT)) {
 581                                         quit_confirmation::quit_to_desktop();
 582                                         continue; // this event is already handled
 583                                 }
 584                                 break;
 585                         }
 586 #endif
 587 
 588 #if defined(_X11) && !defined(__APPLE__)
 589                         case SDL_SYSWMEVENT: {
 590                                 //clipboard support for X11
 591                                 desktop::clipboard::handle_system_event(event);
 592                                 break;
 593                         }
 594 #endif
 595 
 596 #if defined _WIN32
 597                         case SDL_SYSWMEVENT: {
 598                                 windows_tray_notification::handle_system_event(event);
 599                                 break;
 600                         }
 601 #endif
 602 
 603                         case SDL_QUIT: {
 604                                 quit_confirmation::quit_to_desktop();
 605                                 continue; //this event is already handled.
 606                         }
 607                 }
 608 
 609                 const handler_list& global_handlers = event_contexts.front().handlers;
 610                 for(auto handler : global_handlers) {
 611                         handler->handle_event(event);
 612                 }
 613 
 614                 if(event_contexts.empty() == false) {
 615 
 616                         const handler_list& event_handlers = event_contexts.back().handlers;
 617 
 618                         for(auto handler : event_handlers) {
 619                                 handler->handle_event(event);
 620                         }
 621                 }
 622 
 623         }
 624 
 625         //inform the pump monitors that an events::pump() has occurred
 626         for(size_t i1 = 0, i2 = pump_monitors.size(); i1 != i2 && i1 < pump_monitors.size(); ++i1) {
 627                 pump_monitors[i1]->process(info);
 628         }
 629 }

 765 void call_in_main_thread(const std::function<void(void)>& f)
 766 {
 767         if (boost::this_thread::get_id() == main_thread) {
 768                 // nothing special to do if called from the main thread.
 769                 f();
 770                 return;
 771         }
 772 
 773 
 774         invoked_function_data fdata { false, f };
 775 
 776         SDL_Event sdl_event;
 777         SDL_UserEvent sdl_userevent;
 778 
 779         sdl_userevent.type = INVOKE_FUNCTION_EVENT;
 780         sdl_userevent.code = 0;
 781         sdl_userevent.data1 = &fdata;
 782         sdl_userevent.data2 = nullptr;
 783 
 784         sdl_event.type = INVOKE_FUNCTION_EVENT;
 785         sdl_event.user = sdl_userevent;
 786 
 787         SDL_PushEvent(&sdl_event);
 788 
 789         while (!fdata.finished) {
 790                 SDL_Delay(10);
 791         }
 792 }
