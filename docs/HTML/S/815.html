  86 static void copy_persistent(const config& src, config& dst)
  87 {
  88         static const std::set<std::string> attrs {
  89                         "description",
  90                         "name",
  91                         "victory_when_enemies_defeated",
  92                         "remove_from_carryover_on_defeat",
  93                         "disallow_recall",
  94                         "experience_modifier",
  95                         "require_scenario"};
  96 
  97         static const std::set<std::string> tags {
  98                         "terrain_graphics",
  99                         "lua"};
 100 
 101         for (const std::string& attr : attrs)
 102         {
 103                 dst[attr] = src[attr];
 104         }
 105 
 106         for (const std::string& tag : tags)
 107         {
 108                 dst.append_children(src, tag);
 109         }
 110 }

 347 void play_controller::fire_prestart()
 348 {
 349         // pre-start events must be executed before any GUI operation,
 350         // as those may cause the display to be refreshed.
 351         update_locker lock_display(gui_->video());
 352         gamestate().gamedata_.set_phase(game_data::PRESTART);
 353 
 354         // Fire these right before prestart events, to catch only the units sides
 355         // have started with.
 356         for (const unit& u : gamestate().board_.units()) {
 357                 pump().fire("unit_placed", map_location(u.get_location()));
 358         }
 359 
 360         pump().fire("prestart");
 361         // prestart event may modify start turn with WML, reflect any changes.
 362         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 363 }

 365 void play_controller::fire_start()

 369         // start event may modify start turn with WML, reflect any changes.
 370         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 371         check_objectives();
 372         // prestart and start events may modify the initial gold amount,
 373         // reflect any changes.
 374         for (team& tm : gamestate().board_.teams_)
 375         {
 376                 tm.set_start_gold(tm.gold());
 377         }
 378         gamestate_->init_side_done() = false;
 379         gamestate().gamedata_.set_phase(game_data::PLAY);

 419 void play_controller::do_init_side()
 420 {
 421         set_scontext_synced sync;
 422         log_scope("player turn");
 423         // In case we might end up calling sync:network during the side turn events,
 424         // and we don't want do_init_side to be called when a player drops.
 425         gamestate_->init_side_done() = true;
 426         init_side_done_now_ = true;
 427 
 428         const std::string turn_num = std::to_string(turn());
 429         const std::string side_num = std::to_string(current_side());
 430 
 431         gamestate().gamedata_.get_variable("side_number") = current_side();
 432 
 433         // We might have skipped some sides because they were empty so it is not enough to check for side_num==1
 434         if(!gamestate().tod_manager_.has_turn_event_fired())
 435         {
 436                 pump().fire("turn_" + turn_num);
 437                 pump().fire("new_turn");
 438                 gamestate().tod_manager_.turn_event_fired();
 439         }
 440 
 441         pump().fire("side_turn");
 442         pump().fire("side_" + side_num + "_turn");
 443         pump().fire("side_turn_" + turn_num);
 444         pump().fire("side_" + side_num + "_turn_" + turn_num);
 445 
 446         // We want to work out if units for this player should get healed,
 447         // and the player should get income now.
 448         // Healing/income happen if it's not the first turn of processing,
 449         // or if we are loading a game.
 450         if (turn() > 1) {
 451                 gamestate().board_.new_turn(current_side());
 452                 current_team().new_turn();
 453 
 454                 // If the expense is less than the number of villages owned
 455                 // times the village support capacity,
 456                 // then we don't have to pay anything at all
 457                 int expense = gamestate().board_.side_upkeep(current_side()) -
 458                         current_team().support();
 459                 if(expense > 0) {
 460                         current_team().spend_gold(expense);
 461                 }
 462 
 463                 calculate_healing(current_side(), !is_skipping_replay());
 464         }
 465 
 466         // Prepare the undo stack.
 467         undo_stack().new_side_turn(current_side());
 468 
 469         pump().fire("turn_refresh");
 470         pump().fire("side_" + side_num + "_turn_refresh");
 471         pump().fire("turn_" + turn_num + "_refresh");
 472         pump().fire("side_" + side_num + "_turn_" + turn_num + "_refresh");
 473 
 474         // Make sure vision is accurate.
 475         actions::clear_shroud(current_side(), true);
 476         init_side_end();
 477         check_victory();
 478         sync.do_final_checkup();
 479 }

 481 void play_controller::init_side_end()
 482 {
 483         const time_of_day& tod = gamestate().tod_manager_.get_time_of_day();
 484 
 485         if (current_side() == 1 || !init_side_done_now_)
 486                 sound::play_sound(tod.sounds, sound::SOUND_SOURCES);
 487 
 488         if (!is_skipping_replay()){
 489                 gui_->invalidate_all();
 490         }
 491 
 492         if (!is_skipping_replay() && current_team().get_scroll_to_leader()){
 493                 gui_->scroll_to_leader(current_side(), game_display::ONSCREEN,false);
 494         }
 495         whiteboard_manager_->on_init_side();
 496 }

 516 void play_controller::finish_side_turn()
 517 {
 518         whiteboard_manager_->on_finish_side_turn(current_side());
 519 
 520         { //Block for set_scontext_synced
 521                 set_scontext_synced sync(1);
 522                 // Ending the turn commits all moves.
 523                 undo_stack().clear();
 524                 gamestate().board_.end_turn(current_side());
 525                 const std::string turn_num = std::to_string(turn());
 526                 const std::string side_num = std::to_string(current_side());
 527 
 528                 // Clear shroud, in case units had been slowed for the turn.
 529                 actions::clear_shroud(current_side());
 530 
 531                 pump().fire("side_turn_end");
 532                 pump().fire("side_"+ side_num + "_turn_end");
 533                 pump().fire("side_turn_" + turn_num + "_end");
 534                 pump().fire("side_" + side_num + "_turn_" + turn_num + "_end");
 535                 // This is where we refog, after all of a side's events are done.
 536                 actions::recalculate_fog(current_side());
 537                 check_victory();
 538                 sync.do_final_checkup();
 539         }
 540 
 541         mouse_handler_.deselect_hex();
 542         resources::gameboard->unit_id_manager().reset_fake();
 543         gamestate_->init_side_done() = false;
 544 }

 555 bool play_controller::enemies_visible() const
 556 {
 557         // If we aren't using fog/shroud, this is easy :)
 558         if(current_team().uses_fog() == false && current_team().uses_shroud() == false)
 559                 return true;
 560 
 561         // See if any enemies are visible
 562         for (const unit& u : gamestate().board_.units()) {
 563                 if (current_team().is_enemy(u.side()) && !gui_->fogged(u.get_location())) {
 564                         return true;
 565                 }
 566         }
 567 
 568         return false;
 569 }

 676 /// @returns: the number n in [min, min+mod ) so that (n - num) is a multiple of mod.
 677 static int modulo(int num, int mod, int min)
 678 {
 679         assert(mod > 0);
 680         int n = (num - min) % mod;
 681         if (n < 0)
 682                 n += mod;
 683         //n is now in [0, mod)
 684         n = n + min;
 685         return n;
 686         // the folowing properties are easy to verify:
 687         //  1) For all m: modulo(num, mod, min) == modulo(num + mod*m, mod, min)
 688         //  2) For all 0 <= m < mod: modulo(min + m, mod, min) == min + m
 689 }

 765 void play_controller::process_keyup_event(const SDL_Event& event)
 766 {
 767         // If the user has pressed 1 through 9, we want to show
 768         // how far the unit can move in that many turns
 769         if(event.key.keysym.sym >= '1' && event.key.keysym.sym <= '9') {
 770                 const int new_path_turns = (event.type == SDL_KEYDOWN) ?
 771                                            event.key.keysym.sym - '1' : 0;
 772 
 773                 if(new_path_turns != mouse_handler_.get_path_turns()) {
 774                         mouse_handler_.set_path_turns(new_path_turns);
 775 
 776                         const unit_map::iterator u = mouse_handler_.selected_unit();
 777 
 778                         if(u.valid()) {
 779                                 // if it's not the unit's turn, we reset its moves
 780                                 unit_movement_resetter move_reset(*u, u->side() != current_side());
 781 
 782                                 mouse_handler_.set_current_paths(pathfind::paths(*u, false,
 783                                                        true, gamestate().board_.teams_[gui_->viewing_team()],
 784                                                        mouse_handler_.get_path_turns()));
 785 
 786                                 gui_->highlight_reach(mouse_handler_.current_paths());
 787                         } else {
 788                                 mouse_handler_.select_hex(mouse_handler_.get_selected_hex(), false, false, false);
 789                         }
 790 
 791                 }
 792         } else if (event.key.keysym.sym == SDLK_TAB) {
 793                 static CKey keys;
 794                 if (!keys[SDLK_TAB]) {
 795                         whiteboard_manager_->set_invert_behavior(false);
 796                 }
 797         }
 798 }

 901 void play_controller::check_victory()
 902 {
 903         if(linger_)
 904         {
 905                 return;
 906         }
 907 
 908         if (is_regular_game_end()) {
 909                 return;
 910         }
 911         bool continue_level, found_player, found_network_player, invalidate_all;
 912         std::set<unsigned> not_defeated;
 913 
 914         gamestate().board_.check_victory(continue_level, found_player, found_network_player, invalidate_all, not_defeated, remove_from_carryover_on_defeat_);
 915 
 916         if (invalidate_all) {
 917                 gui_->invalidate_all();
 918         }
 919 
 920         if (continue_level) {
 921                 return ;
 922         }
 923 
 924         if (found_player || found_network_player) {
 925                 pump().fire("enemies_defeated");
 926                 if (is_regular_game_end()) {
 927                         return;
 928                 }
 929         }
 930 
 931         DBG_EE << "victory_when_enemies_defeated: " << victory_when_enemies_defeated_ << std::endl;
 932         DBG_EE << "found_player: " << found_player << std::endl;
 933         DBG_EE << "found_network_player: " << found_network_player << std::endl;
 934 
 935         if (!victory_when_enemies_defeated_ && (found_player || found_network_player)) {
 936                 // This level has asked not to be ended by this condition.
 937                 return;
 938         }
 939 
 940         if (gui_->video().non_interactive()) {
 941                 LOG_AIT << "winner: ";
 942                 for (unsigned l : not_defeated) {
 943                         std::string ai = ai::manager::get_active_ai_identifier_for_side(l);
 944                         if (ai.empty()) ai = "default ai";
 945                         LOG_AIT << l << " (using " << ai << ") ";
 946                 }
 947                 LOG_AIT << std::endl;
 948                 ai_testing::log_victory(not_defeated);
 949         }
 950 
 951         DBG_EE << "throwing end level exception..." << std::endl;
 952         //Also proceed to the next scenario when another player survived.
 953         end_level_data el_data;
 954         el_data.proceed_to_next_level = found_player || found_network_player;
 955         el_data.is_victory = found_player;
 956         set_end_level_data(el_data);
 957 }

1118 void play_controller::play_side()
1119 {
1120         //check for team-specific items in the scenario
1121         gui_->parse_team_overlays();
1122         do {
1123                 update_viewing_player();
1124                 {
1125                         save_blocker blocker;
1126                         maybe_do_init_side();
1127                         if(is_regular_game_end()) {
1128                                 return;
1129                         }
1130                 }
1131                 // This flag can be set by derived classes (in overridden functions).
1132                 player_type_changed_ = false;
1133 
1134                 statistics::reset_turn_stats(gamestate().board_.get_team(current_side()).save_id());
1135 
1136                 play_side_impl();
1137 
1138                 if(is_regular_game_end()) {
1139                         return;
1140                 }
1141 
1142         } while (player_type_changed_);
1143         // Keep looping if the type of a team (human/ai/networked)
1144         // has changed mid-turn
1145         sync_end_turn();
1146 }

1148 void play_controller::play_turn()
1149 {
1150         whiteboard_manager_->on_gamestate_change();
1151         gui_->new_turn();
1152         gui_->invalidate_game_status();
1153         events::raise_draw_event();
1154 
1155         LOG_NG << "turn: " << turn() << "\n";
1156 
1157         if(gui_->video().non_interactive()) {
1158                 LOG_AIT << "Turn " << turn() << ":" << std::endl;
1159         }
1160 
1161         for (; gamestate_->player_number_ <= int(gamestate().board_.teams().size()); ++gamestate_->player_number_)
1162         {
1163                 // If a side is empty skip over it.
1164                 if (current_team().is_empty()) {
1165                         continue;
1166                 }
1167                 init_side_begin();
1168                 if(gamestate_->init_side_done()) {
1169                         // This is the case in a reloaded game where the side was initialized before saving the game.
1170                         init_side_end();
1171                 }
1172 
1173                 ai_testing::log_turn_start(current_side());
1174                 play_side();
1175                 if(is_regular_game_end()) {
1176                         return;
1177                 }
1178                 finish_side_turn();
1179                 if(is_regular_game_end()) {
1180                         return;
1181                 }
1182                 if(gui_->video().non_interactive()) {
1183                         LOG_AIT << " Player " << current_side() << ": " <<
1184                                 current_team().villages().size() << " Villages" <<
1185                                 std::endl;
1186                         ai_testing::log_turn_end(current_side());
1187                 }
1188         }
1189         // If the loop exits due to the last team having been processed.
1190         gamestate_->player_number_ = gamestate().board_.teams().size();
1191 
1192         finish_turn();
1193 
1194         // Time has run out
1195         check_time_over();
1196 }

1198 void play_controller::check_time_over()
1199 {
1200         const bool time_left = gamestate().tod_manager_.next_turn(&gamestate().gamedata_);
1201 
1202         if(!time_left) {
1203                 LOG_NG << "firing time over event...\n";
1204                 set_scontext_synced_base sync;
1205                 pump().fire("time_over");
1206                 LOG_NG << "done firing time over event...\n";
1207                 // If turns are added while handling 'time over' event.
1208                 if (gamestate().tod_manager_.is_time_left()) {
1209                         return;
1210                 }
1211 
1212                 if(gui_->video().non_interactive()) {
1213                         LOG_AIT << "time over (draw)\n";
1214                         ai_testing::log_draw();
1215                 }
1216 
1217                 check_victory();
1218                 if (is_regular_game_end()) {
1219                         return;
1220                 }
1221                 end_level_data e;
1222                 e.proceed_to_next_level = false;
1223                 e.is_victory = false;
1224                 set_end_level_data(e);
1225         }
1226 }
