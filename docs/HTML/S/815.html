  87 static void copy_persistent(const config& src, config& dst)
  88 {
  89         static const std::set<std::string> attrs {
  90                         "description",
  91                         "name",
  92                         "victory_when_enemies_defeated",
  93                         "remove_from_carryover_on_defeat",
  94                         "disallow_recall",
  95                         "experience_modifier",
  96                         "require_scenario"};
  97 
  98         static const std::set<std::string> tags {
  99                         "terrain_graphics",
 100                         "lua"};
 101 
 102         for (const std::string& attr : attrs)
 103         {
 104                 dst[attr] = src[attr];
 105         }
 106 
 107         for (const std::string& tag : tags)
 108         {
 109                 dst.append_children(src, tag);
 110         }
 111 }

 288                 plugins_context_->set_accessor_string("scenario_name", [this](config) { return get_scenario_name(); });
 289         });

 348 void play_controller::fire_prestart()
 349 {
 350         // pre-start events must be executed before any GUI operation,
 351         // as those may cause the display to be refreshed.
 352         update_locker lock_display(gui_->video());
 353         gamestate().gamedata_.set_phase(game_data::PRESTART);
 354 
 355         // Fire these right before prestart events, to catch only the units sides
 356         // have started with.
 357         for (const unit& u : gamestate().board_.units()) {
 358                 pump().fire("unit_placed", map_location(u.get_location()));
 359         }
 360 
 361         pump().fire("prestart");
 362         // prestart event may modify start turn with WML, reflect any changes.
 363         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 364 }

 366 void play_controller::fire_start()
 367 {
 368         gamestate().gamedata_.set_phase(game_data::START);
 369         pump().fire("start");
 370         // start event may modify start turn with WML, reflect any changes.
 371         gamestate().gamedata_.get_variable("turn_number") = int(turn());
 372         check_objectives();
 373         // prestart and start events may modify the initial gold amount,
 374         // reflect any changes.
 375         for (team& tm : gamestate().board_.teams_)
 376         {
 377                 tm.set_start_gold(tm.gold());
 378         }
 379         gamestate_->init_side_done() = false;
 380         gamestate().gamedata_.set_phase(game_data::PLAY);
 381 }

 405 void play_controller::maybe_do_init_side()
 406 {
 407         //
 408         // We do side init only if not done yet for a local side when we are not replaying.
 409         // For all other sides it is recorded in replay and replay handler has to handle
 410         // calling do_init_side() functions.

 412         if (gamestate_->init_side_done() || !current_team().is_local() || gamestate().gamedata_.phase() != game_data::PLAY || is_replay() || !replay_->at_end()) {

 420 void play_controller::do_init_side()
 421 {
 422         set_scontext_synced sync;
 423         log_scope("player turn");
 424         // In case we might end up calling sync:network during the side turn events,
 425         // and we don't want do_init_side to be called when a player drops.
 426         gamestate_->init_side_done() = true;
 427         init_side_done_now_ = true;
 428 
 429         const std::string turn_num = std::to_string(turn());
 430         const std::string side_num = std::to_string(current_side());
 431 
 432         gamestate().gamedata_.get_variable("side_number") = current_side();
 433 
 434         // We might have skipped some sides because they were empty so it is not enough to check for side_num==1
 435         if(!gamestate().tod_manager_.has_turn_event_fired())
 436         {
 437                 pump().fire("turn_" + turn_num);
 438                 pump().fire("new_turn");
 439                 gamestate().tod_manager_.turn_event_fired();
 440         }
 441 
 442         pump().fire("side_turn");
 443         pump().fire("side_" + side_num + "_turn");
 444         pump().fire("side_turn_" + turn_num);
 445         pump().fire("side_" + side_num + "_turn_" + turn_num);
 446 
 447         // We want to work out if units for this player should get healed,
 448         // and the player should get income now.
 449         // Healing/income happen if it's not the first turn of processing,
 450         // or if we are loading a game.
 451         if (turn() > 1) {
 452                 gamestate().board_.new_turn(current_side());
 453                 current_team().new_turn();
 454 
 455                 // If the expense is less than the number of villages owned
 456                 // times the village support capacity,
 457                 // then we don't have to pay anything at all
 458                 int expense = gamestate().board_.side_upkeep(current_side()) -
 459                         current_team().support();
 460                 if(expense > 0) {
 461                         current_team().spend_gold(expense);
 462                 }
 463 
 464                 calculate_healing(current_side(), !is_skipping_replay());
 465         }
 466 
 467         // Prepare the undo stack.
 468         undo_stack().new_side_turn(current_side());
 469 
 470         pump().fire("turn_refresh");
 471         pump().fire("side_" + side_num + "_turn_refresh");
 472         pump().fire("turn_" + turn_num + "_refresh");
 473         pump().fire("side_" + side_num + "_turn_" + turn_num + "_refresh");
 474 
 475         // Make sure vision is accurate.
 476         actions::clear_shroud(current_side(), true);
 477         init_side_end();
 478         check_victory();
 479         sync.do_final_checkup();
 480 }

 482 void play_controller::init_side_end()
 483 {
 484         const time_of_day& tod = gamestate().tod_manager_.get_time_of_day();
 485 
 486         if (current_side() == 1 || !init_side_done_now_)
 487                 sound::play_sound(tod.sounds, sound::SOUND_SOURCES);
 488 
 489         if (!is_skipping_replay()){
 490                 gui_->invalidate_all();
 491         }
 492 
 493         if (!is_skipping_replay() && current_team().get_scroll_to_leader()){
 494                 gui_->scroll_to_leader(current_side(), game_display::ONSCREEN,false);
 495         }
 496         whiteboard_manager_->on_init_side();
 497 }

 517 void play_controller::finish_side_turn()
 518 {
 519         whiteboard_manager_->on_finish_side_turn(current_side());
 520 
 521         { //Block for set_scontext_synced
 522                 set_scontext_synced sync(1);
 523                 // Ending the turn commits all moves.
 524                 undo_stack().clear();
 525                 gamestate().board_.end_turn(current_side());
 526                 const std::string turn_num = std::to_string(turn());
 527                 const std::string side_num = std::to_string(current_side());
 528 
 529                 // Clear shroud, in case units had been slowed for the turn.
 530                 actions::clear_shroud(current_side());
 531 
 532                 pump().fire("side_turn_end");
 533                 pump().fire("side_"+ side_num + "_turn_end");
 534                 pump().fire("side_turn_" + turn_num + "_end");
 535                 pump().fire("side_" + side_num + "_turn_" + turn_num + "_end");
 536                 // This is where we refog, after all of a side's events are done.
 537                 actions::recalculate_fog(current_side());
 538                 check_victory();
 539                 sync.do_final_checkup();
 540         }
 541 
 542         mouse_handler_.deselect_hex();
 543         resources::gameboard->unit_id_manager().reset_fake();
 544         gamestate_->init_side_done() = false;
 545 }

 556 bool play_controller::enemies_visible() const
 557 {
 558         // If we aren't using fog/shroud, this is easy :)
 559         if(current_team().uses_fog() == false && current_team().uses_shroud() == false)
 560                 return true;
 561 
 562         // See if any enemies are visible
 563         for (const unit& u : gamestate().board_.units()) {
 564                 if (current_team().is_enemy(u.side()) && !gui_->fogged(u.get_location())) {
 565                         return true;
 566                 }
 567         }
 568 
 569         return false;
 570 }

 575                 return;
 576         }
 577 
 578         const std::string str = menu_handler_.get_textbox().box()->text();
 579         const unsigned int team_num = current_side();
 580         events::mouse_handler& mousehandler = mouse_handler_;
 581 
 582         switch(menu_handler_.get_textbox().mode()) {
 583         case gui::TEXTBOX_SEARCH:
 584                 menu_handler_.do_search(str);
 585                 menu_handler_.get_textbox().close(*gui_);
 586                 break;
 587         case gui::TEXTBOX_MESSAGE:
 588                 menu_handler_.do_speak();
 589                 menu_handler_.get_textbox().close(*gui_);  //need to close that one after executing do_speak() !
 590                 break;
 591         case gui::TEXTBOX_COMMAND:
 592                 menu_handler_.get_textbox().close(*gui_);
 593                 menu_handler_.do_command(str);
 594                 break;
 595         case gui::TEXTBOX_AI:
 596                 menu_handler_.get_textbox().close(*gui_);
 597                 menu_handler_.do_ai_formula(str, team_num, mousehandler);
 598                 break;
 599         default:

 601                 ERR_DP << "unknown textbox mode" << std::endl;
 602         }

 677 /// @returns: the number n in [min, min+mod ) so that (n - num) is a multiple of mod.
 678 static int modulo(int num, int mod, int min)
 679 {
 680         assert(mod > 0);
 681         int n = (num - min) % mod;
 682         if (n < 0)
 683                 n += mod;
 684         //n is now in [0, mod)
 685         n = n + min;
 686         return n;
 687         // the folowing properties are easy to verify:
 688         //  1) For all m: modulo(num, mod, min) == modulo(num + mod*m, mod, min)
 689         //  2) For all 0 <= m < mod: modulo(min + m, mod, min) == min + m
 690 }

 703 int play_controller::find_last_visible_team() const
 704 {
 705         assert(current_side() <= int(gamestate().board_.teams().size()));
 706         const int num_teams = gamestate().board_.teams().size();
 707         const bool is_observer = this->is_observer();

 713                 }
 714         }

 766 void play_controller::process_keyup_event(const SDL_Event& event)
 767 {
 768         // If the user has pressed 1 through 9, we want to show
 769         // how far the unit can move in that many turns
 770         if(event.key.keysym.sym >= '1' && event.key.keysym.sym <= '9') {
 771                 const int new_path_turns = (event.type == SDL_KEYDOWN) ?
 772                                            event.key.keysym.sym - '1' : 0;
 773 
 774                 if(new_path_turns != mouse_handler_.get_path_turns()) {
 775                         mouse_handler_.set_path_turns(new_path_turns);
 776 
 777                         const unit_map::iterator u = mouse_handler_.selected_unit();
 778 
 779                         if(u.valid()) {
 780                                 // if it's not the unit's turn, we reset its moves
 781                                 unit_movement_resetter move_reset(*u, u->side() != current_side());
 782 
 783                                 mouse_handler_.set_current_paths(pathfind::paths(*u, false,
 784                                                        true, gamestate().board_.teams_[gui_->viewing_team()],
 785                                                        mouse_handler_.get_path_turns()));
 786 
 787                                 gui_->highlight_reach(mouse_handler_.current_paths());
 788                         } else {
 789                                 mouse_handler_.select_hex(mouse_handler_.get_selected_hex(), false, false, false);
 790                         }
 791 
 792                 }
 793         } else if (event.key.keysym.sym == SDLK_TAB) {
 794                 static CKey keys;
 795                 if (!keys[SDLK_TAB]) {
 796                         whiteboard_manager_->set_invert_behavior(false);
 797                 }
 798         }
 799 }

 902 void play_controller::check_victory()
 903 {
 904         if(linger_)
 905         {
 906                 return;
 907         }
 908 
 909         if (is_regular_game_end()) {
 910                 return;
 911         }
 912         bool continue_level, found_player, found_network_player, invalidate_all;
 913         std::set<unsigned> not_defeated;
 914 
 915         gamestate().board_.check_victory(continue_level, found_player, found_network_player, invalidate_all, not_defeated, remove_from_carryover_on_defeat_);
 916 
 917         if (invalidate_all) {
 918                 gui_->invalidate_all();
 919         }
 920 
 921         if (continue_level) {
 922                 return ;
 923         }
 924 
 925         if (found_player || found_network_player) {
 926                 pump().fire("enemies_defeated");
 927                 if (is_regular_game_end()) {
 928                         return;
 929                 }
 930         }
 931 
 932         DBG_EE << "victory_when_enemies_defeated: " << victory_when_enemies_defeated_ << std::endl;
 933         DBG_EE << "found_player: " << found_player << std::endl;
 934         DBG_EE << "found_network_player: " << found_network_player << std::endl;
 935 
 936         if (!victory_when_enemies_defeated_ && (found_player || found_network_player)) {
 937                 // This level has asked not to be ended by this condition.
 938                 return;
 939         }
 940 
 941         if (gui_->video().non_interactive()) {
 942                 LOG_AIT << "winner: ";
 943                 for (unsigned l : not_defeated) {
 944                         std::string ai = ai::manager::get_active_ai_identifier_for_side(l);
 945                         if (ai.empty()) ai = "default ai";
 946                         LOG_AIT << l << " (using " << ai << ") ";
 947                 }
 948                 LOG_AIT << std::endl;
 949                 ai_testing::log_victory(not_defeated);
 950         }
 951 
 952         DBG_EE << "throwing end level exception..." << std::endl;
 953         //Also proceed to the next scenario when another player survived.
 954         end_level_data el_data;
 955         el_data.proceed_to_next_level = found_player || found_network_player;
 956         el_data.is_victory = found_player;
 957         set_end_level_data(el_data);
 958 }

1119 void play_controller::play_side()
1120 {
1121         //check for team-specific items in the scenario
1122         gui_->parse_team_overlays();
1123         do {
1124                 update_viewing_player();
1125                 {
1126                         save_blocker blocker;
1127                         maybe_do_init_side();
1128                         if(is_regular_game_end()) {
1129                                 return;
1130                         }
1131                 }
1132                 // This flag can be set by derived classes (in overridden functions).
1133                 player_type_changed_ = false;
1134

1136

1138

1140                         return;
1141                 }
1142

1144         // Keep looping if the type of a team (human/ai/networked)
1145         // has changed mid-turn
1146         sync_end_turn();

1149 void play_controller::play_turn()
1150 {
1151         whiteboard_manager_->on_gamestate_change();
1152         gui_->new_turn();
1153         gui_->invalidate_game_status();
1154         events::raise_draw_event();
1155 
1156         LOG_NG << "turn: " << turn() << "\n";
1157 
1158         if(gui_->video().non_interactive()) {
1159                 LOG_AIT << "Turn " << turn() << ":" << std::endl;
1160         }
1161 
1162         for (; gamestate_->player_number_ <= int(gamestate().board_.teams().size()); ++gamestate_->player_number_)
1163         {
1164                 // If a side is empty skip over it.
1165                 if (current_team().is_empty()) {
1166                         continue;
1167                 }
1168                 init_side_begin();
1169                 if(gamestate_->init_side_done()) {
1170                         // This is the case in a reloaded game where the side was initialized before saving the game.
1171                         init_side_end();
1172                 }
1173 
1174                 ai_testing::log_turn_start(current_side());
1175                 play_side();
1176                 if(is_regular_game_end()) {
1177                         return;
1178                 }
1179                 finish_side_turn();
1180                 if(is_regular_game_end()) {
1181                         return;
1182                 }
1183                 if(gui_->video().non_interactive()) {
1184                         LOG_AIT << " Player " << current_side() << ": " <<
1185                                 current_team().villages().size() << " Villages" <<
1186                                 std::endl;
1187                         ai_testing::log_turn_end(current_side());
1188                 }
1189         }
1190         // If the loop exits due to the last team having been processed.
1191         gamestate_->player_number_ = gamestate().board_.teams().size();
1192 
1193         finish_turn();
1194 
1195         // Time has run out
1196         check_time_over();
1197 }

1199 void play_controller::check_time_over()
1200 {
1201         const bool time_left = gamestate().tod_manager_.next_turn(&gamestate().gamedata_);
1202 
1203         if(!time_left) {
1204                 LOG_NG << "firing time over event...\n";
1205                 set_scontext_synced_base sync;
1206                 pump().fire("time_over");
1207                 LOG_NG << "done firing time over event...\n";
1208                 // If turns are added while handling 'time over' event.
1209                 if (gamestate().tod_manager_.is_time_left()) {
1210                         return;
1211                 }
1212 
1213                 if(gui_->video().non_interactive()) {
1214                         LOG_AIT << "time over (draw)\n";
1215                         ai_testing::log_draw();
1216                 }
1217 
1218                 check_victory();
1219                 if (is_regular_game_end()) {
1220                         return;
1221                 }
1222                 end_level_data e;
1223                 e.proceed_to_next_level = false;
1224                 e.is_victory = false;
1225                 set_end_level_data(e);
1226         }
