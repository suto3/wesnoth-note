 211 #define REPEAT NUMBER BODY_WML
 212     # Macro to execute some WML a defined number of times.
 213     #
 214     # Example that causes screen to quake 5 times:
 215     #! {REPEAT 5 (
 216     #!     {QUAKE "rumble.ogg"}
 217     #! )}
 218     {VARIABLE REPEAT_i 0}
 219 
 220     [while]
 221         [variable]
 222             name=REPEAT_i
 223             less_than={NUMBER}
 224         [/variable]
 225 
 226         [do]
 227             {BODY_WML}
 228 
 229             {VARIABLE_OP REPEAT_i add 1}
 230         [/do]
 231     [/while]
 232 
 233     {CLEAR_VARIABLE REPEAT_i}
 234 #enddef

 236 #define LOOKUP_INDEX FROM_ARRAY_NAME WHERE_KEY_NAME WHERE_VALUE SAVE_AS_NAME
 237     # Call this to lookup an array element that has a particular key-value pair
 238     # then it saves the index of that element, or
 239     # if the key-value pair is not found it saves the array's length
 240     {VARIABLE {SAVE_AS_NAME} 0}
 241     [while]
 242         [variable]
 243             name={SAVE_AS_NAME}
 244             less_than=${FROM_ARRAY_NAME}.length
 245         [/variable]
 246         [variable]
 247             name={FROM_ARRAY_NAME}[${SAVE_AS_NAME}].{WHERE_KEY_NAME}
 248             not_equals={WHERE_VALUE}
 249         [/variable]
 250         [do]
 251             {VARIABLE_OP {SAVE_AS_NAME} add 1}
 252         [/do]
 253     [/while]
 254 #enddef
 255 
 256 #define LOOKUP_VALUE FROM_ARRAY_NAME WHERE_KEY_NAME WHERE_VALUE SAVE_KEY_NAME DEFAULT_VALUE SAVE_AS_NAME
 257     # Call this to look up an array element that has a particular key-value pair
 258     # then it saves another key from that same element.
 259     {LOOKUP_INDEX {FROM_ARRAY_NAME} {WHERE_KEY_NAME} {WHERE_VALUE} {SAVE_AS_NAME}}
 260     [if]
 261         [variable]
 262             name={SAVE_AS_NAME}
 263             numerical_equals=${FROM_ARRAY_NAME}.length
 264         [/variable]
 265         [then]
 266             {VARIABLE {SAVE_AS_NAME} {DEFAULT_VALUE}}
 267         [/then]
 268         [else]
 269             {VARIABLE {SAVE_AS_NAME} ${FROM_ARRAY_NAME}[${SAVE_AS_NAME}].{SAVE_KEY_NAME}}
 270         [/else]
 271     [/if]
 272 #enddef

 496 #define FORCE_CHANCE_TO_HIT FILTER SECOND_FILTER CTH_NUMBER EXTRA_CONDITIONS_WML
 497     # Invisibly forces certain units to always have a specific chance to hit
 498     # when fighting against certain other units.
 499     #
 500     # Note that the player still only sees the regular damage calculations, so
 501     # this is useful if you need to give an invisible helping hand to the player
 502     # or AI. For example, if the player is forced to attack with only a couple
 503     # of units at the beginning of a scenario, you can use this to ensure that
 504     # simply having bad luck cannot ruin their attempt so easily. Also you might
 505     # have enemy leaders which the player is not supposed to fight or be able to
 506     # defeat due to storyline reasons, but could theoretically still kill with
 507     # some clever trick, AI mistake or sheer exceptional luck.
 508     #
 509     # An example which forces Konrad's attacks to always hit Li'sar, but only
 510     # after turn 10:
 511     #! {FORCE_CHANCE_TO_HIT id=Konrad id="Li'sar" 100 (
 512     #!     [variable]
 513     #!         name=turn_number
 514     #!         greater_than=10
 515     #!     [/variable]
 516     #! )}
 517     [event]
 518         name=attack
 519         first_time_only=no
 520 
 521         [filter]
 522             {FILTER}
 523         [/filter]
 524 
 525         [filter_second]
 526             {SECOND_FILTER}
 527         [/filter_second]
 528 
 529         [filter_condition]
 530             [and]
 531                 {EXTRA_CONDITIONS_WML}
 532             [/and]
 533         [/filter_condition]
 534 
 535         [foreach]
 536             array=unit.attack
 537             [do]
 538                 [if]
 539                     #This is to mute a warning message about retrieving a member of non-existant wml container.
 540                     [variable]
 541                         name=this_item.specials.length
 542                         greater_than=0
 543                     [/variable]
 544 
 545                     [variable]
 546                         name=this_item.specials.chance_to_hit.length
 547                         greater_than=0
 548                     [/variable]
 549 
 550                     [then]
 551                         [set_variables]
 552                             name=this_item.specials.original_chance_to_hit
 553                             to_variable=this_item.specials.chance_to_hit
 554                         [/set_variables]
 555 
 556                         {CLEAR_VARIABLE this_item.specials.chance_to_hit}
 557                     [/then]
 558                 [/if]
 559 
 560                 [set_variables]
 561                     name=this_item.specials.chance_to_hit
 562 
 563                     [value]
 564                         id=forced_cth
 565                         value={CTH_NUMBER}
 566                         cumulative=no
 567                     [/value]
 568                 [/set_variables]
 569             [/do]
 570         [/foreach]

 572         [unstore_unit]
 573             variable=unit
 574             find_vacant=no
 575         [/unstore_unit]

 577         [event]
 578             name=attack end
 579             delayed_variable_substitution=yes
 580 
 581             [foreach]
 582                 array=unit.attack
 583                 [do]
 584                     {CLEAR_VARIABLE this_item.specials.chance_to_hit}
 585 
 586                     [set_variables]
 587                         name=this_item.specials.chance_to_hit
 588                         to_variable=this_item.specials.original_chance_to_hit
 589                     [/set_variables]
 590
