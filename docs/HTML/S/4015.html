  72 #define PICKUPPABLE_ITEM ID X Y CAN_TAKE_FILTER_WML IMAGE TEXT TAKE_IT_STRING LEAVE_IT_STRING CANNOT_TAKE_TEXT OBJECT_WML
  73     # This places an item on the map which can be picked up once, but allows the
  74     # player to choose whether the unit stepping on it should take it or not.
  75     # The "object" doesn't necessarily need to be an [object], any action works.
  76     #
  77     # An example which shows the:
  78     # ! {PICKUPPABLE_ITEM ring_of_hp 8 12 race=human items/ring-red.png
  79     # !     _"$unit.name finds a pretty ring. Should he pick it up?"
  80     # !     _"ring of HP^Take it"
  81     # !     _"ring of HP^Leave it"
  82     # !     _"$unit.name finds a pretty ring. But only a human can take it!" (
  83     # !     [object]
  84     # !         name= _ "Ring of HP"
  85     # !         image=items/ring-red.png
  86     # !         description= _ "This ring grants the wearer +10 max hp!"
  87     # !
  88     # !         [effect]
  89     # !             apply_to=hitpoints
  90     # !             increase_total=10
  91     # !         [/effect]
  92     # !     [/object]
  93     # ! )}
  94 
  95     [item]
  96         x,y={X},{Y}
  97         image={IMAGE}
  98     [/item]
  99 
 100     [event]
 101         name=moveto
 102         first_time_only=no
 103 
 104         [filter]
 105             x,y={X},{Y}
 106         [/filter]
 107 
 108         [if]
 109             [have_unit]
 110                 x,y={X},{Y}
 111                 {CAN_TAKE_FILTER_WML}
 112             [/have_unit]
 113 
 114             [variable]
 115                 name=item_{ID}_picked_up
 116                 not_equals=yes
 117             [/variable]
 118 
 119             [then]
 120                 [message]
 121                     speaker=narrator
 122                     message={TEXT}
 123                     image={IMAGE}
 124 
 125                     [option]
 126                         label={TAKE_IT_STRING}
 127 
 128                         [command]
 129                             {OBJECT_WML}
 130 
 131                             [remove_item]
 132                                 x,y={X},{Y}
 133                                 image={IMAGE}
 134                             [/remove_item]
 135 
 136                             [set_variable]
 137                                 name=item_{ID}_picked_up
 138                                 value=yes
 139                             [/set_variable]
 140                         [/command]
 141                     [/option]
 142 
 143                     [option]
 144                         label={LEAVE_IT_STRING}
 145 
 146                         [command]
 147                             [allow_undo]
 148                             [/allow_undo]
 149                         [/command]
 150                     [/option]
 151                 [/message]
 152             [/then]
 153 
 154             [else]
 155                 [if]
 156                     [variable]
 157                         name=item_{ID}_picked_up
 158                         not_equals=yes
 159                     [/variable]
 160 
 161                     [then]
 162                         [message]
 163                             speaker=narrator
 164                             message={CANNOT_TAKE_TEXT}
 165                             image={IMAGE}
 166                             side_for=$side_number
 167                             # the above is used to prevent an AI side from
 168                             # accidentally triggering this dialog
 169                         [/message]
 170                     [/then]
 171                 [/if]
 172 
 173                 [allow_undo]
 174                 [/allow_undo]
 175             [/else]
 176         [/if]
 177     [/event]
 178 
 179     [event]
 180         name=victory
 181 
 182         [clear_variable]
 183             name=item_{ID}_picked_up
 184         [/clear_variable]
 185     [/event]
 186 #enddef
