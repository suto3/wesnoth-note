 361 void command_executor::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool /*context_menu*/, display& gui)
 362 {
 363         std::vector<config> items = items_arg;
 364         if (items.empty()) return;
 365 
 366         get_menu_images(gui, items);
 367 
 368         int res = -1;
 369         {
 370                 SDL_Rect pos {xloc, yloc, 1, 1};
 371                 gui2::dialogs::drop_down_menu mmenu(pos, items, -1, true, false); // TODO: last value should be variable
 372                 mmenu.show(gui.video());
 373                 if(mmenu.get_retval() == gui2::window::OK) {
 374                         res = mmenu.selected_item();
 375                 }
 376         } // This will kill the dialog.
 377         if (res < 0 || size_t(res) >= items.size()) return;
 378 
 379         const theme::menu* submenu = gui.get_theme().get_menu_item(items[res]["id"]);
 380         if (submenu) {
 381                 int y,x;
 382                 SDL_GetMouseState(&x,&y);
 383                 this->show_menu(submenu->items(), x, y, submenu->is_context(), gui);
 384         } else {
 385                 const hotkey::hotkey_command& cmd = hotkey::get_hotkey_command(items[res]["id"]);
 386                 hotkey::execute_command(cmd,this,res);
 387                 set_button_state();
 388         }
 389 }

 391 void command_executor::execute_action(const std::vector<std::string>& items_arg, int /*xloc*/, int /*yloc*/, bool /*context_menu*/, display&)
 392 {
 393         std::vector<std::string> items = items_arg;
 394         if (items.empty()) {
 395                 return;
 396         }
 397 
 398         std::vector<std::string>::iterator i = items.begin();
 399         while(i != items.end()) {
 400                 const hotkey_command &command = hotkey::get_hotkey_command(*i);
 401                 if (can_execute_command(command)) {
 402                         hotkey::execute_command(command, this);
 403                         set_button_state();
 404                 }
 405                 ++i;
 406         }
 407 }

 409 std::string command_executor::get_menu_image(display& disp, const std::string& command, int index) const {
 410 
 411         // TODO: Find a way to do away with the fugly special markup
 412         if(command[0] == '&') {
 413                 size_t n = command.find_first_of('=');
 414                 if(n != std::string::npos)
 415                         return command.substr(1, n - 1);
 416         }
 417 
 418         const std::string base_image_name = "icons/action/" + command + "_25.png";
 419         const std::string pressed_image_name = "icons/action/" + command + "_25-pressed.png";
 420 
 421         const hotkey::HOTKEY_COMMAND hk = hotkey::get_id(command);
 422         const hotkey::ACTION_STATE state = get_action_state(hk, index);
 423 
 424         const theme::menu* menu = disp.get_theme().get_menu_item(command);
 425         if (menu) {
 426                 return "icons/arrows/short_arrow_right_25.png~CROP(3,3,18,18)"; // TODO should not be hardcoded
 427         }
 428 
 429         if (filesystem::file_exists(game_config::path + "/images/" + base_image_name)) {
 430                 switch (state) {
 431                         case ACTION_ON:
 432                         case ACTION_SELECTED:
 433                                 return pressed_image_name + "~CROP(3,3,18,18)";
 434                         default:
 435                                 return base_image_name + "~CROP(3,3,18,18)";
 436                 }
 437         }
 438 
 439         switch (get_action_state(hk, index)) {
 440                 case ACTION_ON:
 441                         return game_config::images::checked_menu;
 442                 case ACTION_OFF:
 443                         return game_config::images::unchecked_menu;
 444                 case ACTION_SELECTED:
 445                         return game_config::images::selected_menu;
 446                 case ACTION_DESELECTED:
 447                         return game_config::images::deselected_menu;
 448                 default: return get_action_image(hk, index);
 449         }
 450 }

 452 void command_executor::get_menu_images(display& disp, std::vector<config>& items)
 453 {
 454         for(size_t i = 0; i < items.size(); ++i) {
 455                 config& item = items[i];
 456 
 457                 const std::string& item_id = item["id"];
 458                 const hotkey::HOTKEY_COMMAND hk = hotkey::get_id(item_id);
 459 
 460                 //see if this menu item has an associated image
 461                 std::string img(get_menu_image(disp, item_id, i));
 462                 if (img.empty() == false) {
 463                         item["icon"] = img;
 464                 }
 465 
 466                 const theme::menu* menu = disp.get_theme().get_menu_item(item_id);
 467                 if(menu) {
 468                         item["label"] = menu->title();
 469                 } else if(hk != hotkey::HOTKEY_NULL) {
 470                         std::string desc = hotkey::get_description(item_id);
 471                         if(hk == HOTKEY_ENDTURN) {
 472                                 const theme::action *b = disp.get_theme().get_action_item("button-endturn");
 473                                 if (b) {
 474                                         desc = b->title();
 475                                 }
 476                         }
 477 
 478                         item["label"] = desc;
 479                         item["details"] = hotkey::get_names(item_id);
 480                 } else if(item["label"].empty()) {
 481                         // If no matching hotkey was found and a custom label wasn't already set, treat
 482                         // the id as a plaintext description. This is because either type of value can
 483                         // be written to the id field by the WMI manager. The plaintext description is
 484                         // used in the case the menu item specifies the relevant entry is *not* a hotkey.
 485                         item["label"] = item_id;
 486                 }
 487         }
 488 }

 511 static void event_execute( const SDL_Event& event, command_executor* executor)
 512 {
 513         if (!executor) return;
 514         const hotkey_ptr hk = get_hotkey(event);
 515         if (!hk->active() || hk->is_disabled()) {
 516                 return;
 517         }
 518 
 519         bool press = event.type == SDL_KEYDOWN ||
 520                         event.type == SDL_JOYBUTTONDOWN ||
 521                         event.type == SDL_MOUSEBUTTONDOWN ||
 522                         event.type == SDL_TEXTINPUT;
 523 
 524         execute_command(hotkey::get_hotkey_command(hk->get_command()), executor, -1, press);
 525         executor->set_button_state();
 526 }

 528 void execute_command(const hotkey_command& command, command_executor* executor, int index, bool press)
 529 {
 530         assert(executor != nullptr);
 531 
 532         if (!executor->can_execute_command(command, index)
 533                         || executor->execute_command(command, index, press)) {
 534                 return;
 535         }
 536 
 537         if (!press) {
 538                 return; // none of the commands here respond to a key release
 539     }
 540 
 541         switch (command.id) {
 542 
 543                 case HOTKEY_MINIMAP_DRAW_TERRAIN:
 544                         preferences::toggle_minimap_draw_terrain();
 545                         executor->recalculate_minimap();
 546                         break;
 547                 case HOTKEY_MINIMAP_CODING_TERRAIN:
 548                         preferences::toggle_minimap_terrain_coding();
 549                         executor->recalculate_minimap();
 550                         break;
 551                 case HOTKEY_MINIMAP_CODING_UNIT:
 552                         preferences::toggle_minimap_movement_coding();
 553                         executor->recalculate_minimap();
 554                         break;
 555                 case HOTKEY_MINIMAP_DRAW_UNITS:
 556                         preferences::toggle_minimap_draw_units();
 557                         executor->recalculate_minimap();
 558                         break;
 559                 case HOTKEY_MINIMAP_DRAW_VILLAGES:
 560                         preferences::toggle_minimap_draw_villages();
 561                         executor->recalculate_minimap();
 562                         break;
 563                 case HOTKEY_FULLSCREEN:
 564                         executor->get_video().set_fullscreen(!preferences::fullscreen());
 565                         break;
 566                 case HOTKEY_SCREENSHOT:
 567                         make_screenshot(_("Screenshot"), executor->get_video(), &::screenshot);
 568                         break;
 569                 case HOTKEY_ANIMATE_MAP:
 570                         preferences::set_animate_map(!preferences::animate_map());
 571                         break;
 572                 case HOTKEY_MOUSE_SCROLL:
 573                         preferences::enable_mouse_scroll(!preferences::mouse_scroll_enabled());
 574                         break;
 575                 case HOTKEY_MUTE:
 576                         {
 577                                 // look if both is not playing
 578                                 static struct before_muted_s
 579                                 {
 580                                         bool playing_sound,playing_music;
 581                                         before_muted_s() : playing_sound(false),playing_music(false){}
 582                                 } before_muted;
 583                                 if (preferences::music_on() || preferences::sound_on())
 584                                 {
 585                                         // then remember settings and mute both
 586                                         before_muted.playing_sound = preferences::sound_on();
 587                                         before_muted.playing_music = preferences::music_on();
 588                                         preferences::set_sound(false);
 589                                         preferences::set_music(false);
 590                                 }
 591                                 else
 592                                 {
 593                                         // then set settings before mute
 594                                         preferences::set_sound(before_muted.playing_sound);
 595                                         preferences::set_music(before_muted.playing_music);
 596                                 }
 597                         }
 598                         break;
 599                 default:
 600                         DBG_G << "command_executor: unknown command number " << command.id << ", ignoring.\n";
 601                         break;
 602         }
 603 }

 605 void command_executor_default::set_button_state()
 606 {
 607         display& disp = get_display();
 608         for (const theme::menu& menu : disp.get_theme().menus()) {
 609 
 610                 std::shared_ptr<gui::button> button = disp.find_menu_button(menu.get_id());
 611                 if (!button) continue;
 612                 bool enabled = false;
 613                 for (const auto& command : menu.items()) {
 614 
 615                         const hotkey::hotkey_command& command_obj = hotkey::get_hotkey_command(command["id"]);
 616                         bool can_execute = can_execute_command(command_obj);
 617                         if (can_execute) {
 618                                 enabled = true;
 619                                 break;
 620                         }
 621                 }
 622                 button->enable(enabled);
 623         }
 624 
 625         for (const theme::action& action : disp.get_theme().actions()) {
 626 
 627                 std::shared_ptr<gui::button> button = disp.find_action_button(action.get_id());
 628                 if (!button) continue;
 629                 bool enabled = false;
 630                 int i = 0;
 631                 for (const std::string& command : action.items()) {
 632 
 633                         const hotkey::hotkey_command& command_obj = hotkey::get_hotkey_command(command);
 634                         std::string tooltip = action.tooltip(i);
 635                         if (filesystem::file_exists(game_config::path + "/images/icons/action/" + command + "_25.png" ))
 636                                 button->set_overlay("icons/action/" + command);
 637                         if (!tooltip.empty())
 638                                 button->set_tooltip_string(tooltip);
 639 
 640                         bool can_execute = can_execute_command(command_obj);
 641                         i++;
 642                         if (!can_execute) continue;
 643                         enabled = true;
 644 
 645                         ACTION_STATE state = get_action_state(command_obj.id, -1);
 646                         switch (state) {
 647                         case ACTION_SELECTED:
 648                         case ACTION_ON:
 649                                 button->set_check(true);
 650                                 break;
 651                         case ACTION_OFF:
 652                         case ACTION_DESELECTED:
 653                                 button->set_check(false);
 654                                 break;
 655                         case ACTION_STATELESS:
 656                                 break;
 657                         default:
 658                                 break;
 659                         }
 660 
 661                         break;
 662                 }
 663                 button->enable(enabled);
 664         }
 665 }
