  67 void clean_saves(const std::string& label)
  68 {
  69         std::vector<save_info> games = get_saves_list();
  70         std::string prefix = label + "-" + _("Auto-Save");
  71         LOG_SAVE << "Cleaning saves with prefix '" << prefix << "'\n";
  72         for (std::vector<save_info>::iterator i = games.begin(); i != games.end(); ++i) {
  73                 if (i->name().compare(0, prefix.length(), prefix) == 0) {
  74                         LOG_SAVE << "Deleting savegame '" << i->name() << "'\n";
  75                         delete_game(i->name());
  76                 }

  87 bool loadgame::show_difficulty_dialog()
  88 {
  89         if(load_data_.summary["corrupt"].to_bool()) {

  91         }
  92 
  93         std::string campaign_id = load_data_.summary["campaign"];
  94 
  95         for(const config &campaign : game_config_.child_range("campaign"))

  98                         continue;
  99                 }

 109                 load_data_.difficulty = difficulty_dlg.selected_difficulty();

 111                 // Exit loop
 112                 break;

 118 // Called only by play_controller to handle in-game attempts to load. Instead of returning true,
 119 // throws a "load_game_exception" to signal a resulting load game request.
 120 bool loadgame::load_game_ingame()
 121 {
 122         if (video_.faked()) {
 123                 return false;
 124         }
 125 
 126         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 127                 return false;
 128         }
 129 
 130         if(load_data_.filename.empty()) {
 131                 return false;
 132         }
 133 
 134         if(load_data_.select_difficulty) {
 135                 if(!show_difficulty_dialog()) {
 136                         return false;
 137                 }
 138         }
 139 
 140         load_data_.show_replay |= is_replay_save(load_data_.summary);
 141 
 142         // Confirm the integrity of the file before throwing the exception.
 143         // Use the summary in the save_index for this.
 144         const config & summary = save_index_manager.get(load_data_.filename);
 145 
 146         if (summary["corrupt"].to_bool(false)) {
 147                 gui2::show_error_message(video_,
 148                                 _("The file you have tried to load is corrupt: '"));
 149                 return false;
 150         }
 151 
 152         if (!loadgame::check_version_compatibility(summary["version"].str(), video_)) {
 153                 return false;
 154         }
 155 
 156         throw load_game_exception(std::move(load_data_));
 157 }

 159 bool loadgame::load_game()
 160 {
 161         bool skip_version_check = true;
 162 
 163         if(load_data_.filename.empty()){
 164                 if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 165                         return false;
 166                 }
 167                 skip_version_check = false;
 168                 load_data_.show_replay |= is_replay_save(load_data_.summary);
 169         }
 170 
 171         if(load_data_.filename.empty()) {
 172                 return false;
 173         }
 174 
 175         if(load_data_.select_difficulty) {
 176                 if(!show_difficulty_dialog()) {
 177                         return false;
 178                 }
 179         }
 180 
 181         std::string error_log;
 182         read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 183 
 184         convert_old_saves(load_data_.load_config);
 185         
 186         for (config& side : load_data_.load_config.child_range("side")) {
 187                 side.remove_attribute("is_local");
 188         }
 189 
 190         if(!error_log.empty()) {
 191         try {
 192                     gui2::show_error_message(video_,
 193                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 194                                     error_log);
 195         } catch (utf8::invalid_utf8_exception&) {
 196                     gui2::show_error_message(video_,
 197                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 198                     std::string("(UTF-8 ERROR)"));
 199         }
 200         }
 201 
 202         if (!load_data_.difficulty.empty()){
 203                 load_data_.load_config["difficulty"] = load_data_.difficulty;
 204         }
 205         // read classification to for loading the game_config config object.
 206         gamestate_.classification() = game_classification(load_data_.load_config);
 207 
 208         if (skip_version_check) {
 209                 return true;
 210         }
 211 
 212         return check_version_compatibility();
 213 }

 220 bool loadgame::check_version_compatibility(const version_info & save_version, CVideo & video)
 221 {
 222         if (save_version == game_config::version) {
 223                 return true;
 224         }
 225 
 226         const version_info &wesnoth_version = game_config::wesnoth_version;
 227 
 228         // Even minor version numbers indicate stable releases which are
 229         // compatible with each other.
 230         if (wesnoth_version.minor_version() % 2 == 0 &&
 231             wesnoth_version.major_version() == save_version.major_version() &&
 232             wesnoth_version.minor_version() == save_version.minor_version())
 233         {
 234                 return true;
 235         }
 236 
 237         // Do not load if too old. If either the savegame or the current
 238         // game has the version 'test', load. This 'test' version is never
 239         // supposed to occur, except when Soliton is testing MP servers.
 240         if (save_version < game_config::min_savegame_version &&
 241             save_version != game_config::test_version &&
 242             wesnoth_version != game_config::test_version)
 243         {
 244                 const std::string message = _("This save is from an old, unsupported version ($version_number|) and cannot be loaded.");
 245                 utils::string_map symbols;
 246                 symbols["version_number"] = save_version.str();
 247                 gui2::show_error_message(video, utils::interpolate_variables_into_string(message, &symbols));
 248                 return false;
 249         }
 250 
 251         if(preferences::confirm_load_save_from_different_version()) {
 252                 const std::string message = _("This save is from a different version of the game ($version_number|). Do you wish to try to load it?");
 253                 utils::string_map symbols;
 254                 symbols["version_number"] = save_version.str();
 255                 const int res = gui2::show_message(video, _("Load Game"), utils::interpolate_variables_into_string(message, &symbols),
 256                         gui2::dialogs::message::yes_no_buttons);
 257                 return res == gui2::window::OK;
 258         }
 259 
 260         return true;
 261 }

 268 bool loadgame::load_multiplayer_game()
 269 {
 270         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 271                 return false;
 272         }
 273 
 274 
 275         load_data_.show_replay |= is_replay_save(load_data_.summary);
 276         if(load_data_.filename.empty()) {
 277                 return false;
 278         }
 279 
 280         // read_save_file needs to be called before we can verify the classification so the data has
 281         // been populated. Since we do that, we report any errors in that process first.
 282         std::string error_log;
 283         {
 284                 cursor::setter cur(cursor::WAIT);
 285                 log_scope("load_game");
 286 
 287                 read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 288                 copy_era(load_data_.load_config);
 289         }
 290 
 291         if(!error_log.empty()) {
 292                 gui2::show_error_message(video_,
 293                                 _("The file you have tried to load is corrupt: '") +
 294                                 error_log);
 295                 return false;
 296         }
 297 
 298         if(is_replay_save(load_data_.summary)) {
 299                 gui2::show_transient_message(video_, _("Load Game"), _("Replays are not supported in multiplayer mode."));
 300                 return false;
 301         }
 302 
 303         // We want to verify the game classification before setting the data, so we don't check on
 304         // gamestate_.classification() and instead construct a game_classification object manually.
 305         if(game_classification(load_data_.load_config).campaign_type != game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 306                 gui2::show_transient_error_message(video_, _("This is not a multiplayer save."));
 307                 return false;
 308         }
 309 
 310         set_gamestate();
 311 
 312         return check_version_compatibility();
 313 }

 440 bool savegame::save_game(CVideo* video, const std::string& filename)
 441 {
 442 
 443         try {
 444                 Uint32 start, end;
 445                 start = SDL_GetTicks();
 446 
 447                 if (filename_ == "")
 448                         filename_ = filename;
 449 
 450                 before_save();
 451 
 452                 write_game_to_disk(filename_);
 453                 if (resources::persist != nullptr) {
 454                         resources::persist->end_transaction();
 455                         resources::persist ->start_transaction();
 456                 }
 457 
 458                 end = SDL_GetTicks();
 459                 LOG_SAVE << "Milliseconds to save " << filename_ << ": " << end - start << std::endl;
 460 
 461                 if (video != nullptr && show_confirmation_)
 462                         gui2::show_transient_message(*video, _("Saved"), _("The game has been saved."));
 463                 return true;
 464         } catch(game::save_game_failed& e) {
 465                 ERR_SAVE << error_message_ << e.message << std::endl;
 466                 if (video != nullptr){
 467                         gui2::show_error_message(*video, error_message_ + e.message);
 468                         //do not bother retrying, since the user can just try to save the game again
 469                         //maybe show a yes-no dialog for "disable autosaves now"?
 470                 }
 471 
 472                 return false;
 473         };
 474 }

 476 void savegame::write_game_to_disk(const std::string& filename)
 477 {
 478         LOG_SAVE << "savegame::save_game" << std::endl;
 479 
 480         filename_ = filename;
 481         filename_ += compression::format_extension(compress_saves_);
 482 
 483         std::stringstream ss;
 484         {
 485                 config_writer out(ss, compress_saves_);
 486                 write_game(out);
 487                 finish_save_game(out);
 488         }
 489         filesystem::scoped_ostream os(open_save_game(filename_));
 490         (*os) << ss.str();
 491 
 492         if (!os->good()) {
 493                 throw game::save_game_failed(_("Could not write to file"));
 494         }
 495 }

 658                 config carryover;
 659                 //copy rng and menu items from toplevel to new carryover_sides
 660                 carryover["random_seed"] = cfg["random_seed"];
 661                 carryover["random_calls"] = cfg["random_calls"];
 662                 for(const config& menu_item : cfg.child_range("menu_item")) {
 663                         carryover.add_child("menu_item", menu_item);
 664                 }
 665                 carryover["difficulty"] = cfg["difficulty"];
 666                 carryover["random_mode"] = cfg["random_mode"];
 667                 //the scenario to be played is always stored as next_scenario in carryover_sides_start
 668                 carryover["next_scenario"] = cfg["scenario"];
 669 
 670                 config carryover_start = carryover;
 671 
 672                 //copy sides from either snapshot or replay_start to new carryover_sides
 673                 if(!snapshot.empty()){
 674                         for(const config& side : snapshot.child_range("side")) {
 675                                 carryover.add_child("side", side);
 676                         }
 677                         //for compatibility with old savegames that use player instead of side
 678                         for(const config& side : snapshot.child_range("player")) {
 679                                 carryover.add_child("side", side);
 680                         }
 681                         //save the sides from replay_start in carryover_sides_start
 682                         for(const config& side : replay_start.child_range("side")) {
 683                                 carryover_start.add_child("side", side);
 684                         }
 685                         //for compatibility with old savegames that use player instead of side
 686                         for(const config& side : replay_start.child_range("player")) {
 687                                 carryover_start.add_child("side", side);
 688                         }
 689                 } else if (!replay_start.empty()){
 690                         for(const config& side : replay_start.child_range("side")) {
 691                                 carryover.add_child("side", side);
 692                                 carryover_start.add_child("side", side);
 693                         }
 694                         //for compatibility with old savegames that use player instead of side
 695                         for(const config& side : replay_start.child_range("player")) {
 696                                 carryover.add_child("side", side);
 697                                 carryover_start.add_child("side", side);
 698                         }
 699                 }
 700 
 701                 //get variables according to old hierarchy and copy them to new carryover_sides
 702                 if(!snapshot.empty()){
 703                         if(const config& variables_from_snapshot = snapshot.child("variables")){
 704                                 carryover.add_child("variables", variables_from_snapshot);
 705                                 carryover_start.add_child("variables", replay_start.child_or_empty("variables"));
 706                         } else if (const config& variables_from_cfg = cfg.child("variables")){
 707                                 carryover.add_child("variables", variables_from_cfg);
 708                                 carryover_start.add_child("variables", variables_from_cfg);
 709                         }
 710                 } else if (!replay_start.empty()){
 711                         if(const config& variables = replay_start.child("variables")){
 712                                 carryover.add_child("variables", variables);
 713                                 carryover_start.add_child("variables", variables);
 714                         }
 715                 } else {
 716                         carryover.add_child("variables", cfg.child("variables"));
 717                         carryover_start.add_child("variables", cfg.child("variables"));
 718                 }
 719 
 720                 cfg.add_child("carryover_sides", carryover);
 721                 cfg.add_child("carryover_sides_start", carryover_start);
 722         }
 723 
 724         //if replay and snapshot are empty we've got a start of scenario save and don't want replay_start either
 725         if(replay.empty() && snapshot.empty()){
 726                 LOG_RG<<"removing replay_start \n";
 727                 cfg.clear_children("replay_start");
 728         }
 729 
 730         //remove empty replay or snapshot so type of save can be detected more easily
 731         if(replay.empty()){
 732                 LOG_RG<<"removing replay \n";
 733                 cfg.clear_children("replay");
 734         }
 735 
 736         if(snapshot.empty()){
 737                 LOG_RG<<"removing snapshot \n";
 738                 cfg.clear_children("snapshot");
 739         }
 740 }

 742 static void convert_old_saves_1_13_0(config& cfg)
 743 {
 744         if(config& carryover_sides_start = cfg.child("carryover_sides_start"))
 745         {
 746                 if(!carryover_sides_start.has_attribute("next_underlying_unit_id"))
 747                 {
 748                         carryover_sides_start["next_underlying_unit_id"] = cfg["next_underlying_unit_id"];
 749                 }
 750         }
 751         if(cfg.child_or_empty("snapshot").empty())
 752         {
 753                 cfg.clear_children("snapshot");
 754         }
 755         if(cfg.child_or_empty("replay_start").empty())
 756         {
 757                 cfg.clear_children("replay_start");
 758         }
 759         if(config& snapshot = cfg.child("snapshot"))
 760         {
 761                 //make [end_level] -> [end_level_data] since its alo called [end_level_data] in the carryover.
 762                 if(config& end_level = cfg.child("end_level") )
 763                 {
 764                         snapshot.add_child("end_level_data", end_level);
 765                         snapshot.clear_children("end_level");
 766                 }
 767                 //if we have a snapshot then we already applied carryover so there is no reason to keep this data.
 768                 if(cfg.has_child("carryover_sides_start"))
 769                 {
 770                         cfg.clear_children("carryover_sides_start");
 771                 }
 772         }
 773         if(!cfg.has_child("snapshot") && !cfg.has_child("replay_start"))
 774         {
 775                 cfg.clear_children("carryover_sides");
 776         }
 777         //This code is needed because for example otherwise it won't find the (empty) era
 778         if(!cfg.has_child("multiplayer")) {
 779                 cfg.add_child("multiplayer", config {
 780                         "mp_era", "era_blank",
 781                         "show_connect", false,
 782                         "show_configure", false,
 783                         "mp_use_map_settings", true,
 784                 });
 785         }
 786 }

 790 static void convert_old_saves_1_13_1(config& cfg)
 791 {
 792         if(config& multiplayer = cfg.child("multiplayer")) {
 793                 if(multiplayer["mp_era"] == "era_blank") {
 794                         multiplayer["mp_era"] = "era_default";
 795                 }
 796         }
 797         //This currently only fixes start-of-scenario saves.
 798         if(config& carryover_sides_start = cfg.child("carryover_sides_start"))
 799         {
 800                 for(config& side : carryover_sides_start.child_range("side"))
 801                 {
 802                         for(config& unit : side.child_range("unit"))
 803                         {
 804                                 if(config& modifications = unit.child("modifications"))
 805                                 {
 806                                         for(config& advancement : modifications.child_range("advance"))
 807                                         {
 808                                                 modifications.add_child("advancement", advancement);
 809                                         }
 810                                         modifications.clear_children("advance");
 811                                 }
 812                         }
 813                 }
 814         }
 815         for(config& snapshot : cfg.child_range("snapshot")) {
 816                 if (snapshot.has_attribute("used_items")) {
 817                         config used_items;
 818                         for(const std::string& item : utils::split(snapshot["used_items"])) {
 819                                 used_items[item] = true;
 820                         }
 821                         snapshot.remove_attribute("used_items");
 822                         snapshot.add_child("used_items", used_items);
 823                 }
 824         }

 827 void convert_old_saves(config& cfg)
 828 {
 829         version_info loaded_version(cfg["version"]);
 830         if(loaded_version < version_info("1.12.0"))
 831         {
 832                 convert_old_saves_1_11_0(cfg);
 833         }
 834         // '<= version_info("1.13.0")' doesn't work
 835         //because version_info cannot handle 1.13.0-dev versions correctly.
 836         if(loaded_version < version_info("1.13.1"))
 837         {
 838                 convert_old_saves_1_13_0(cfg);
 839         }
 840         if(loaded_version <= version_info("1.13.1"))
 841         {
 842                 convert_old_saves_1_13_1(cfg);
 843         }
 844         LOG_RG<<"cfg after conversion "<<cfg<<"\n";
 845 }
