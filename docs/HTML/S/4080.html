  57     def update_addon(addon_obj, addon_name, addon_server, temp_dir):
  58         """Update the add-on from addon-server
  59 
  60         addon_obj       Github.Addon object for the add-on
  61         addon_name      Name of the add-on
  62         addon_server    Url to the add-on server
  63         temp_dir        The directory where the unpacked campaign can be stored.
  64 
  65         returns         Whether anything was changed.
  66         """
  67         # Grab timestamp from server
  68         server_timestamp = get_timestamp(addon_server, addon_name)
  69 
  70         # Check local timestamp to see if the add-on server version is newer
  71         if os.path.exists(os.path.join(addon_obj.get_dir(), ADDONVER_FILE)):
  72             with open(os.path.join(addon_obj.get_dir(), ADDONVER_FILE), "r") as stamp_file:
  73                 str_timestamp = stamp_file.read()
  74             try:
  75                 local_timestamp = int(str_timestamp)
  76                 if local_timestamp == server_timestamp:
  77                     logging.info("Addon '%s' is up-to-date.", addon_name)
  78                     return False
  79             except:
  80                 pass
  81 
  82         # Download the addon.
  83         extract(addon_server, addon_name, temp_dir)
  84 
  85         # Translation needs to be prevented from the campaign to overwrite
  86         # the ones in wescamp.
  87         # The other files are present in wescamp and shouldn't be deleted.
  88         ignore_list = ["translations", "po", "campaign.def",
  89             "config.status", "Makefile", BUILDSYS_FILE, ADDONVER_FILE]
  90         if addon_obj.sync_from(temp_dir, ignore_list):
  91             # Store add-on timestamp
  92             with open(os.path.join(addon_obj.get_dir(), ADDONVER_FILE), "w") as timestamp_file:
  93                 timestamp_file.write(str(server_timestamp))
  94             addon_obj._execute(["git", "add", ADDONVER_FILE])
  95 
  96             addon_obj.commit("wescamp.py: Update from add-on server")
  97             logging.info("New version of addon '%s' uploaded.", addon_name)
  98             return True
  99         else:
 100             logging.info("Addon '%s' hasn't been modified, thus not uploaded.",
 101                 addon_name)
 102             return False

 104     def update_build_system(addon_obj, addon_name, wescamp_dir, build_sys_dir):
 105         """Initialize or update the build-system.
 106 
 107         addon_obj           libgithub.Addon objectof the addon.
 108         addon_name          Name of the addon.
 109         wescamp_dir         The directory containing a checkout of wescamp.
 110         build_sys_dir       Possible directory containing a checkout of build-system.
 111         returns             Boolean indicating whether the add-on now has a build-system.
 112         """
 113         logging.info("Checking if build system for add-on {0} needs to be updated".format(addon_name))
 114         previously_initialized = os.path.exists(os.path.join(addon_obj.get_dir(), "Makefile"))
 115 
 116         # Grab the build system
 117         below_branch = os.path.basename(wescamp_dir.rstrip(os.sep))
 118         possible_build_paths = []
 119         if build_sys_dir:
 120             possible_build_paths.append(build_sys_dir)
 121         possible_build_paths.append(os.path.join(below_branch, "build-system"))
 122         build_system = libgithub.get_build_system(possible_build_paths)
 123         init_script = os.path.join(build_system.get_dir(), "init-build-sys.sh")
 124 
 125         # Grab master build system's version
 126         # Ugliness
 127         out, err, res = build_system._execute(["git", "show", "--pretty=oneline", "--summary"])
 128         build_system_version = out.split()[0]
 129         if len(build_system_version) != 40:
 130             logging.warn("Incorrect SHA1 for build system checkout: {0}".format(build_system_version))
 131 
 132         # Check if build system version in add-on is up-to-date
 133         if os.path.exists(os.path.join(addon_obj.get_dir(), BUILDSYS_FILE)):
 134             with open(os.path.join(addon_obj.get_dir(), BUILDSYS_FILE), "r") as stamp_file:
 135                 addon_build_version = stamp_file.read()
 136             if addon_build_version == build_system_version:
 137                 logging.info("Build system for add-on {0} is up-to-date".format(addon_name))
 138                 return True
 139 
 140         # Ugliness
 141         out, err, res = addon_obj._execute([init_script, "--{0}".format(git_version), addon_name, "."], check_error=False)
 142         if len(err):
 143             logging.warn("init-build-sys.sh in add-on {0}:\n{1}".format(addon_name, err))
 144 
 145         if not out.strip().endswith("Done.") or res != 0:
 146             logging.error("Failed to init the build-system for add-on {0}".format(addon_name))
 147             addon_obj._execute(["rm", "-rf", "po", "campaign.def", "Makefile"])
 148             addon_obj._execute(["git", "reset", "--hard"])
 149             return False
 150 
 151         # Store build system version
 152         with open(os.path.join(addon_obj.get_dir(), BUILDSYS_FILE), "w") as version_file:
 153             version_file.write(build_system_version)
 154 
 155         addon_obj._execute(["git", "add", "po", "campaign.def", "Makefile", BUILDSYS_FILE], check_error=True)
 156         if previously_initialized:
 157             logging.info("Updated build system for add-on {0}".format(addon_name))
 158             addon_obj.commit("wescamp.py: Update build-system")
 159         else:
 160             logging.info("Initialized build system for add-on {0}".format(addon_name))
 161             addon_obj.commit("wescamp.py: Initialize build-system")
 162         return True

 164     def pot_update(addon_obj, addon_name):
 165         """Update the translation catalogs.
 166 
 167         addon_obj           libgithub.Addon objectof the addon.
 168         addon_name          Name of the addon.
 169         returns             Boolean indicating whether the operation was successful.
 170         """
 171         if not os.path.exists(os.path.join(addon_obj.get_dir(), "Makefile")):
 172             logging.warn("Cannot pot-update: build system does not exist for add-on {0}.".format(addon_name))
 173             return False
 174         # Ugliness, again
 175         out, err, res = addon_obj._execute(["make"])
 176         if len(err):
 177             logging.warn("pot-update in addon {0}:\n{1}".format(addon_name, err))
 178             # TODO: bail?
 179 
 180         if res != 0:
 181             logging.error("Failed to pot-update for add-on {0}".format(addon_name))
 182             addon_obj._execute(["rm", "-rf", "po", addon_name])
 183             addon_obj._execute(["git", "reset", "--hard"])
 184             return False
 185 
 186         outlines = addon_obj._status()
 187 
 188         to_rm = []
 189         to_add = []
 190         longname = "wesnoth-{0}".format(addon_name)
 191         for line in outlines:
 192             mod, name = line.split()
 193             if mod == "D":
 194                 to_rm.append(name)
 195             elif mod == "M" and name.endswith((".po", "LC_MESSAGES/{0}.mo".format(longname), "po/{0}.pot".format(longname))):
 196                 to_add.append(name)
 197             else:
 198                 logging.info("Ignoring {0}".format(line))
 199         if to_rm:
 200             addon_obj._execute(["git", "rm"] + to_rm, check_error=True)
 201         if to_add:
 202             addon_obj._execute(["git", "add"] + to_add, check_error=True)
 203         addon_obj.commit("wescamp.py: pot-update")
 204         return True

 223     def list_addons(server, translatable_only):
 224         """Get a list of addons on the server.
 225 
 226         server              The url of the addon server eg
 227                             add-ons.wesnoth.org:15005.
 228         translatable_only   If True only returns translatable addons.
 229         returns             A dictionary with the addon as key and the translatable
 230                             status as value.
 231         """
 232 
 233         logging.debug("list addons server = '%s' translatable_only = %s",
 234             server, translatable_only)
 235 
 236         wml = libwml.CampaignClient(server, quiet_libwml)
 237         data = wml.list_campaigns()
 238 
 239         # Item [0] hardcoded seems to work
 240         campaigns = data.data[0]
 241         result = {}
 242         for c in campaigns.get_all("campaign"):
 243             translatable = c.get_text_val("translate")
 244             if(translatable == "yes" or translatable == "on" or translatable == "true" or translatable == "1"):
 245                 result[c.get_text_val("name")] = True
 246             else:
 247                 # when the campaign isn't marked for translation skip it
 248                 if(translatable_only):
 249                     continue
 250                 else:
 251                     result[c.get_text_val("name")] = False
 252 
 253         return result

 255     def get_timestamp(server, addon):
 256         """Get the timestamp of a campaign on the server.
 257 
 258         server              The url of the addon server eg
 259                             add-ons.wesnoth.org:15005.
 260         addon               The name of the addon.
 261         returns             The timestamp of the campaign, -1 if not on the server.
 262         """
 263 
 264         logging.debug("get_timestamp server = '%s' addon = %s",
 265             server, addon)
 266 
 267         wml = libwml.CampaignClient(server, quiet_libwml)
 268         data = wml.list_campaigns(addon)
 269 
 270         # Item [0] hardcoded seems to work
 271         campaigns = data.data[0]
 272         result = {}
 273         for c in campaigns.get_all("campaign"):
 274             if(c.get_text_val("name") != addon):
 275                 continue
 276 
 277             return int(c.get_text_val("timestamp"))
 278 
 279         return -1

 281     def upload(server, addon, temp_dir, wescamp_dir, build_sys_dir=None):
 282         """Upload a addon from the server to wescamp.
 283 
 284         server              The url of the addon server eg
 285                             add-ons.wesnoth.org:15005.
 286         addon               The name of the addon.
 287         temp_dir            The directory where the unpacked campaign can be stored.
 288         wescamp_dir         The directory containing a checkout of wescamp.
 289         build_sys_dir       Possible directory containing a checkout of build-system.
 290         """
 291 
 292         logging.debug("upload addon to wescamp server = '%s' addon = '%s' "
 293             + "temp_dir = '%s' wescamp_dir = '%s'",
 294             server, addon, temp_dir, wescamp_dir)
 295 
 296         # Is the addon in the list with campaigns to be translated.
 297         campaigns = list_addons(server, True)
 298         if((addon in campaigns) == False):
 299             logging.info("Addon '%s' is not marked as translatable "
 300                 + "upload aborted.", addon)
 301             return
 302 
 303         github = libgithub.GitHub(wescamp_dir, git_version, authorization=git_auth)
 304 
 305         has_updated = False
 306 
 307         # If the checkout doesn't exist we need to create it.
 308         if not os.path.isdir(os.path.join(wescamp_dir, addon)):
 309 
 310             logging.info("Checking out '%s'.",
 311                 os.path.join(wescamp_dir, addon))
 312 
 313             if not github.addon_exists(addon):
 314                 github.create_addon(addon)
 315 
 316         # Update the directory
 317         addon_obj = github.addon(addon)
 318         addon_obj.update()
 319 
 320         has_updated = update_addon(addon_obj, addon, server, temp_dir)
 321 
 322         has_build_system = update_build_system(addon_obj, addon, wescamp_dir, build_sys_dir)
 323 
 324         if has_updated and has_build_system:
 325             pot_update(addon_obj, addon)
