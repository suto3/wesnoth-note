  30 void size_lock::place(const point& origin, const point& size)
  31 {
  32         point content_size = widget_->get_best_size();
  33 
  34         if (content_size.x > size.x)
  35         {
  36                 reduce_width(size.x);
  37                 content_size = widget_->get_best_size();
  38         }
  39 
  40         if (content_size.y > size.y)
  41         {
  42                 reduce_height(size.y);
  43                 content_size = widget_->get_best_size();
  44         }
  45 
  46         container_base::place(origin, size);
  47 }

 138 widget* builder_size_lock::build() const
 139 {
 140         size_lock* widget = new size_lock();
 141 
 142         init_control(widget);
 143 
 144         DBG_GUI_G << "Window builder: placed fixed size widget '" << id <<
 145                 "' with definition '" << definition << "'.\n";
 146 
 147         auto conf = std::static_pointer_cast<const size_lock_definition::resolution>(widget->config());
 148         assert(conf != nullptr);
 149 
 150         widget->init_grid(conf->grid);
 151 
 152         wfl::map_formula_callable size = get_screen_size_variables();
 153 
 154         const unsigned width = width_(size);
 155         const unsigned height = height_(size);
 156 
 157         VALIDATE(width > 0 || height > 0, _("Invalid size."));
 158 
 159         widget->set_target_size(point(width, height));
 160 
 161         widget->finalize(content_);
 162 
 163         return widget;
 164 }
