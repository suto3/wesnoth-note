   1 #textdomain wesnoth-lib
   2 ###
   3 ### Definition of the title label for the story viewer.
   4 ###
   5 
   6 # We can't get the exact size of the text outside a [text] element, so estimate it instead.
   7 # Using an average character width of 12 seems to give good results.
   8 # Then we centre the text in the estimated width rather than in the entire width.
   9 
  10 #define _GUI_ESTIMATE_TEXT_WIDTH
  11 (length(text) * 12)#enddef
  12 
  13 #define _GUI_CENTER_STORY_TEXT
  14 (10 + max((box_width - text_width) / 2, 0)
  15 where box_width = {_GUI_ESTIMATE_TEXT_WIDTH})#enddef
  16 
  17 #define _GUI_BACKGROUND
  18         [round_rectangle]
  19                 x = 10
  20                 y = 10
  21                 w = "{_GUI_ESTIMATE_TEXT_WIDTH}"
  22                 h = "(height - 20)"
  23                 corner_radius = 5
  24                 fill_color = "0, 0, 0, 96"
  25         [/round_rectangle]
  26         [round_rectangle]
  27                 x = 13
  28                 y = 13
  29                 w = "(text_width - 6 where text_width = {_GUI_ESTIMATE_TEXT_WIDTH})"
  30                 h = "(height - 26)"
  31                 corner_radius = 5
  32                 fill_color = "0, 0, 0, 96"
  33         [/round_rectangle]
  34 #enddef
  35 
  36 #define _GUI_TEXT MODE
  37         [text]
  38                 #x = "{_GUI_CENTER_STORY_TEXT}"
  39                 x = 20
  40                 #y = {GUI__TEXT_VERTICALLY_CENTRED}
  41                 y = 20
  42                 w = "(width)"
  43                 h = "(text_height)"
  44                 maximum_width = "(width)"
  45                 font_size = {GUI_FONT_SIZE_LARGE}
  46                 color = {GUI__FONT_COLOR_{MODE}__DEFAULT}
  47                 text = "(text)"
  48                 text_markup = "(text_markup)"
  49                 text_alignment = "(text_alignment)"
  50                 text_link_aware = "(text_link_aware)"
  51                 text_link_color = "(text_link_color)"
  52         [/text]
  53 #enddef
  54 
  55 [label_definition]
  56         id = "story_screen_title"
  57         description = "Large label with translucent background for story screen"
  58 
  59         [resolution]
  60                 {GUI_NORMAL__RESOLUTION}
  61 
  62                 min_width = 0
  63                 min_height = 0
  64 
  65                 default_width = 0
  66                 default_height = 100
  67 
  68                 max_width = 0
  69                 max_height = 0
  70 
  71                 text_font_size = {GUI_FONT_SIZE_LARGE}
  72 
  73                 link_aware = false
  74 
  75                 [state_enabled]
  76 
  77                         [draw]
  78                                 #{_GUI_BACKGROUND}
  79                                 {_GUI_TEXT ENABLED}
  80                         [/draw]
  81 
  82                 [/state_enabled]
  83 
  84                 [state_disabled]
  85 
  86                         [draw]
  87                                 #{_GUI_BACKGROUND}
  88                                 {_GUI_TEXT DISABLED}
  89                         [/draw]
  90 
  91                 [/state_disabled]
  92 
  93         [/resolution]
  94 
  95 [/label_definition]
  96 
  97 #undef _GUI_ESTIMATE_TEXT_WIDTH
  98 #undef _GUI_CENTER_STORY_TEXT
  99 #undef _GUI_BACKGROUND
 100 #undef _GUI_TEXT
