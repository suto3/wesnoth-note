 141 static int os_execute (lua_State *L) {
 142   const char *cmd = luaL_optstring(L, 1, NULL);
 143   int stat = system(cmd);
 144   if (cmd != NULL)
 145     return luaL_execresult(L, stat);
 146   else {
 147     lua_pushboolean(L, stat);  /* true if there is a shell */
 148     return 1;
 149   }
 150 }

 211 ** Set all fields from structure 'tm' in the table on top of the stack
 212 */
 213 static void setallfields (lua_State *L, struct tm *stm) {
 214   setfield(L, "sec", stm->tm_sec);
 215   setfield(L, "min", stm->tm_min);
 216   setfield(L, "hour", stm->tm_hour);
 217   setfield(L, "day", stm->tm_mday);
 218   setfield(L, "month", stm->tm_mon + 1);
 219   setfield(L, "year", stm->tm_year + 1900);
 220   setfield(L, "wday", stm->tm_wday + 1);
 221   setfield(L, "yday", stm->tm_yday + 1);
 222   setboolfield(L, "isdst", stm->tm_isdst);
 223 }

 226 static int getboolfield (lua_State *L, const char *key) {
 227   int res;
 228   res = (lua_getfield(L, -1, key) == LUA_TNIL) ? -1 : lua_toboolean(L, -1);
 229   lua_pop(L, 1);
 230   return res;
 231 }

 236 #define L_MAXDATEFIELD  (INT_MAX / 2)

 238 
 239 static int getfield (lua_State *L, const char *key, int d, int delta) {
 240   int isnum;
 241   int t = lua_getfield(L, -1, key);  /* get field and its type */
 242   lua_Integer res = lua_tointegerx(L, -1, &isnum);
 243   if (!isnum) {  /* field is not an integer? */
 244     if (t != LUA_TNIL)  /* some other value? */
 245       return luaL_error(L, "field '%s' is not an integer", key);
 246     else if (d < 0)  /* absent field; no default? */
 247       return luaL_error(L, "field '%s' missing in date table", key);
 248     res = d;
 249   }
 250   else {
 251     if (!(-L_MAXDATEFIELD <= res && res <= L_MAXDATEFIELD))
 252       return luaL_error(L, "field '%s' is out-of-bound", key);
 253     res -= delta;
 254   }
 255   lua_pop(L, 1);
 256   return (int)res;
 257 }

 260 static const char *checkoption (lua_State *L, const char *conv, char *buff) {
 261   const char *option;
 262   int oplen = 1;
 263   int convlen = (int)strlen(conv);
 264   for (option = LUA_STRFTIMEOPTIONS; *option != '\0' && oplen <= convlen; option += oplen) {
 265     if (*option == '|')  /* next block? */
 266       oplen++;  /* next length */
 267     else if (memcmp(conv, option, oplen) == 0) {  /* match? */
 268       memcpy(buff, conv, oplen);  /* copy valid option to buffer */
 269       buff[oplen] = '\0';
 270       return conv + oplen;  /* return next item */
 271     }
 272   }
 273   luaL_argerror(L, 1,
 274     lua_pushfstring(L, "invalid conversion specifier '%%%s'", conv));
 275   return conv;  /* to avoid warnings */
 276 }

 283 static int os_date (lua_State *L) {
 284   const char *s = luaL_optstring(L, 1, "%c");
 285   time_t t = luaL_opt(L, l_checktime, 2, time(NULL));
 286   struct tm tmr, *stm;
 287   if (*s == '!') {  /* UTC? */
 288     stm = l_gmtime(&t, &tmr);
 289     s++;  /* skip '!' */
 290   }
 291   else
 292     stm = l_localtime(&t, &tmr);
 293   if (stm == NULL)  /* invalid date? */
 294     luaL_error(L, "time result cannot be represented in this installation");
 295   if (strcmp(s, "*t") == 0) {
 296     lua_createtable(L, 0, 9);  /* 9 = number of fields */
 297     setallfields(L, stm);
 298   }
 299   else {
 300     char cc[4];  /* buffer for individual conversion specifiers */
 301     luaL_Buffer b;
 302     cc[0] = '%';
 303     luaL_buffinit(L, &b);
 304     while (*s) {
 305       if (*s != '%')  /* not a conversion specifier? */
 306         luaL_addchar(&b, *s++);
 307       else {
 308         size_t reslen;
 309         char *buff = luaL_prepbuffsize(&b, SIZETIMEFMT);
 310         s = checkoption(L, s + 1, cc + 1);  /* copy specifier to 'cc' */
 311         reslen = strftime(buff, SIZETIMEFMT, cc, stm);
 312         luaL_addsize(&b, reslen);
 313       }
 314     }
 315     luaL_pushresult(&b);
 316   }
 317   return 1;
 318 }

 321 static int os_time (lua_State *L) {
 322   time_t t;
 323   if (lua_isnoneornil(L, 1))  /* called without args? */
 324     t = time(NULL);  /* get current time */
 325   else {
 326     struct tm ts;
 327     luaL_checktype(L, 1, LUA_TTABLE);
 328     lua_settop(L, 1);  /* make sure table is at the top */
 329     ts.tm_sec = getfield(L, "sec", 0, 0);
 330     ts.tm_min = getfield(L, "min", 0, 0);
 331     ts.tm_hour = getfield(L, "hour", 12, 0);
 332     ts.tm_mday = getfield(L, "day", -1, 0);
 333     ts.tm_mon = getfield(L, "month", -1, 1);
 334     ts.tm_year = getfield(L, "year", -1, 1900);
 335     ts.tm_isdst = getboolfield(L, "isdst");
 336     t = mktime(&ts);
 337     setallfields(L, &ts);  /* update fields with normalized values */
 338   }
 339   if (t != (time_t)(l_timet)t || t == (time_t)(-1))
 340     luaL_error(L, "time result cannot be represented in this installation");
 341   l_pushtime(L, t);
 342   return 1;
 343 }

 356 static int os_setlocale (lua_State *L) {
 357   static const int cat[] = {LC_ALL, LC_COLLATE, LC_CTYPE, LC_MONETARY,
 358                       LC_NUMERIC, LC_TIME};
 359   static const char *const catnames[] = {"all", "collate", "ctype", "monetary",
 360      "numeric", "time", NULL};
 361   const char *l = luaL_optstring(L, 1, NULL);
 362   int op = luaL_checkoption(L, 2, "all", catnames);
 363   lua_pushstring(L, setlocale(cat[op], l));
 364   return 1;
 365 }

 368 static int os_exit (lua_State *L) {
 369   int status;
 370   if (lua_isboolean(L, 1))
 371     status = (lua_toboolean(L, 1) ? EXIT_SUCCESS : EXIT_FAILURE);
 372   else
 373     status = (int)luaL_optinteger(L, 1, EXIT_SUCCESS);
 374   if (lua_toboolean(L, 2))
 375     lua_close(L);
 376   if (L) exit(status);  /* 'if' to avoid warnings for unreachable 'return' */
 377   return 0;
 378 }
