 197         template<typename T>
 198         utils::enable_if_t<is_pair<T>::value && !std::is_const<typename T::first_type>::value, T>
 199         lua_check(lua_State *L, int n)
 200         {
 201                 T result;
 202                 if (lua_istable(L, n)) {
 203                         lua_rawgeti(L, n, 1);
 204                         result.first = lua_check<const typename T::first_type&>(L, -1);
 205                         lua_rawgeti(L, n, 2);
 206                         result.second = lua_check<const typename T::second_type&>(L, -1);
 207                         lua_pop(L, 2);
 208                 }
 209                 return result;
 210         }
 211         template<typename T>
 212         utils::enable_if_t<is_pair<T>::value && !std::is_const<typename T::first_type>::value, void>
 213         lua_push(lua_State *L, const T& val)
 214         {
 215                 lua_newtable(L);

 222         //std::vector and similar but not std::string
 223         template<typename T>
 224         utils::enable_if_t<is_container<T>::value && !std::is_same<T, std::string>::value, T>
 225         lua_check(lua_State * L, int n)
 226         {
 227                 if (lua_istable(L, n))
 228                 {
 229                         T res;
 230                         for (int i = 1, i_end = lua_rawlen(L, n); i <= i_end; ++i)
 231                         {
 232                                 lua_rawgeti(L, n, i);
 233                                 res.push_back(lua_check_impl::lua_check<remove_constref<typename T::reference>>(L, -1));
 234                                 lua_pop(L, 1);
 235                         }
 236                         return res;
 237                 }
 238                 else
 239                 {
 240                         luaL_argerror(L, n, "Table expected");
 241                         throw "luaL_argerror returned"; //shouldnt happen, luaL_argerror always throws.
 242                 }
 243         }

 252         //also accepts things like std::vector<int>() | std::adaptors::transformed(..)
 253         template<typename T>
 254         utils::enable_if_t<
 255                 is_container<T>::value && !std::is_same<T, std::string>::value && !is_map<T>::value
 256                 , void
 257         >
 258         lua_push(lua_State * L, const T& list )
 259         {
 260                 // NOTE: T might be some boost::iterator_range type where size might be < 0. (unfortunately in this case size() does not return T::size_type)
 261                 assert(list.size() >= 0);
 262                 lua_createtable(L, list.size(), 0);
 263                 int i = 1;
 264                 for(typename T::const_iterator iter = list.begin(); iter != list.end(); ++iter) {
 265                         lua_check_impl::lua_push<remove_constref<typename T::reference>>(L, *iter);
 266                         lua_rawseti(L, -2, i++);
 267                 }
 268         }
