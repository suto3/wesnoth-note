   1 # Tests that when a side wins by killing the
   2 # enemy leader with an attack, the game ends
   3 # correctly. (This is different from the
   4 # test_check_victory tests because we use
   5 # [do_command] attack, and not [kill]).
   6 #
   7 # Also we test that killing on retal works
   8 # the same.
   9 
  10 #define ALICE_MOVES_TO_BOB
  11     [do_command]
  12         [move]
  13             x=7,8,9,10,11,12,13
  14             y=3,3,4, 4, 4, 4, 4
  15         [/move]
  16     [/do_command]
  17 #enddef
  18 
  19 #define BOB_1_HP
  20     [harm_unit]
  21         [filter]
  22             x,y=13,3
  23         [/filter]
  24         amount = 100
  25         kill = no
  26     [/harm_unit]
  27 #enddef
  28 
  29 #define ALICE_1_HP
  30     [harm_unit]
  31         [filter]
  32             x,y=13,4
  33         [/filter]
  34         amount = 100
  35         kill = no
  36     [/harm_unit]
  37 #enddef
  38 
  39 #define BOTH_0_DEF
  40     [terrain_mask]
  41         x,y=13,3
  42         border=no
  43         mask="Xv"
  44     [/terrain_mask]
  45 
  46     [terrain_mask]
  47         x,y=13,4
  48         border=no
  49         mask="Xv"
  50     [/terrain_mask]
  51 #enddef
  52 
  53 #define BOTH_MAX_XP
  54     [store_unit]
  55         [filter]
  56             id=alice
  57         [/filter]
  58         variable=temp
  59         kill=yes
  60     [/store_unit]
  61     {VARIABLE temp.max_experience 8}
  62     [unstore_unit]
  63         variable=temp
  64         animate=no
  65         fire_event=no
  66     [/unstore_unit]
  67 
  68     [store_unit]
  69         [filter]
  70             id=bob
  71         [/filter]
  72         variable=temp
  73         kill=yes
  74     [/store_unit]
  75     {VARIABLE temp.max_experience 1}
  76     [unstore_unit]
  77         variable=temp
  78         animate=no
  79         fire_event=no
  80     [/unstore_unit]
  81 #enddef
  82 
  83 {GENERIC_UNIT_TEST "alice_kills_bob" (
  84     [event]
  85         name = start
  86 
  87         {ALICE_MOVES_TO_BOB}
  88         {BOB_1_HP}
  89         {BOTH_0_DEF}
  90 
  91         [do_command]
  92             [attack]
  93                 weapon = 1
  94                 defender_weapon = -1
  95                 [source]
  96                     x,y=13,4
  97                 [/source]
  98                 [destination]
  99                     x,y=13,3
 100                 [/destination]
 101             [/attack]
 102         [/do_command]
 103     [/event]
 104 
 105     [event]
 106         name= side turn end
 107         {FAIL}
 108     [/event]
 109 )}
