  74     /**
  75      * Gets the cache for the specified project. If the cache doesn't
  76      * exist, a new one will be created.
  77      * 
  78      * @param project
  79      * @return A {@link ProjectCache} instance. It will never be null.
  80      */
  81     public static ProjectCache getCacheForProject( IProject project )
  82     {
  83         ProjectCache cache = projectCache_.get( project );
  84 
  85         if( cache == null ) {
  86             cache = new ProjectCache( project );
  87             cache.loadCache( );
  88             projectCache_.put( project, cache );

  93     /**
  94      * Creates a folder that link to the project's wesnoth installation's
  95      * 'data/core' directory
  96      * 
  97      * @param updateFlags
  98      *        bit-wise or of update flag constants
  99      *        (IResource.ALLOW_MISSING_LOCAL, IResource.REPLACE,
 100      *        IResource.BACKGROUND_REFRESH, and IResource.HIDDEN)
 101      * @param project
 102      *        The project to create the link for
 103      * @return True if the creation was successfully, false otherwise
 104      */
 105     public static boolean createCoreLibraryFolder( IProject project,
 106         int updateFlags )
 107     {
 108         ProjectCache cache = getCacheForProject( project );
 109         Paths paths = Preferences.getPaths( cache.getInstallName( ) );
 110 
 111         if( ! paths.isValid( ) ) {
 112             return false;
 113         }
 114 
 115         IFolder coreLibrary = project
 116             .getFolder( WesnothProjectsExplorer.CORE_LIBRARY_NAME );
 117 
 118         if( ResourcesPlugin.getWorkspace( ).validateLinkLocation( coreLibrary,
 119             paths.getCoreDirPath( ) ).getCode( ) != IStatus.ERROR ) {
 120             try {
 121                 if( coreLibrary.exists( ) ) {
 122                     // the new Core Library will point to the same location.
 123                     // Skip it then.
 124                     if( coreLibrary.getLocation( ).equals( paths.getCoreDirPath( ) ) ) {
 125                         return true;
 126                     }
 127 
 128                     coreLibrary.delete( true, new NullProgressMonitor( ) );
 129                 }
 130 
 131                 coreLibrary.createLink( paths.getCoreDirPath( ), updateFlags,
 132                     new NullProgressMonitor( ) );
 133 
 134                 coreLibrary.setDerived( true, new NullProgressMonitor( ) );
 135             } catch( CoreException e ) {
 136                 Logger.getInstance( ).logException( e );
 137                 return false;
 138             }
 139         }
 140         else {
 141             Logger.getInstance( ).log(
 142                 "Couldn't create link on:" + paths.getCoreDir( ) //$NON-NLS-1$
 143                     + "; project: " + project.getName( ), //$NON-NLS-1$
 144                 "Cannot create the Wesnoth Core Library folder for project " //$NON-NLS-1$
 145                     + project.getName( ) + "!" ); //$NON-NLS-1$
 146             return false;
 147         }
 148 
 149         return true;
 150     }

 174     /**
 175      * Creates a new wesnoth project with the specified name
 176      * and on the specified location on disk
 177      * 
 178      * @param name
 179      *        The name of the new project
 180      * @param location
 181      *        The location of the new project
 182      * @param installName
 183      *        The name of the install this project belongs to
 184      * @param skipCreateBuildXML
 185      *        True to skip creating a build xml even if needed
 186      * @param monitor
 187      *        The monitor to monitor the progress
 188      * @return A project handle
 189      * @throws CoreException
 190      */
 191     public static IProject createWesnothProject( String name, String location,
 192         String installName, boolean skipCreateBuildXML,
 193         IProgressMonitor monitor )
 194     {
 195         IWorkspaceRoot root = ResourcesPlugin.getWorkspace( ).getRoot( );
 196         IProject newProject = root.getProject( name );
 197         IProjectDescription description = null;
 198 
 199         // set the path only if the project's location is outside the workspace
 200         // root
 201         if( location != null
 202             && ! location.equals( root.getLocation( ).toOSString( ) ) ) {
 203             description = ResourcesPlugin.getWorkspace( )
 204                 .newProjectDescription( name );
 205             description.setLocation( new Path( location ) );
 206         }
 207 
 208         createWesnothProject( newProject, description, installName,
 209             skipCreateBuildXML, monitor );
 210 
 211         return newProject;
 212     }

 214     /**
 215      * Creates a project that has associated the wesnoth nature using
 216      * the specified handle. If the project is created there will be
 217      * no modifications done by this method.
 218      * 
 219      * @param handle
 220      *        the handle to the project
 221      * @param description
 222      *        the default description used when the project is created
 223      * @param installName
 224      *        The name of the install this project belongs to
 225      * @param skipCreateBuildXML
 226      *        True to skip creating a build xml even if needed
 227      * @param monitor
 228      *        the monitor will do a 30 worked amount in the method
 229      * @return
 230      *         0 - project was created successfully.
 231      *         -1 - project already exists or the handle is null
 232      *         1 - there was an error.
 233      * 
 234      * @throws CoreException
 235      */
 236     public static int createWesnothProject( IProject handle,
 237         IProjectDescription description, String installName,
 238         boolean skipCreateBuildXML, IProgressMonitor monitor )
 239     {
 240         if( handle == null || handle.exists( ) ) {
 241             return - 1;
 242         }
 243 
 244         if( installName.isEmpty( ) ) {
 245             GUIUtils.showErrorMessageBox( Messages.ProjectUtils_7 );
 246             return - 1;
 247         }
 248 
 249         try {
 250             String projectPath = null;
 251 
 252             if( handle.getLocation( ) == null && description != null ) {
 253                 projectPath = description.getLocationURI( ).getPath( )
 254                     .substring( 1 );
 255             }
 256             else if( handle.getLocation( ) != null ) {
 257                 projectPath = handle.getLocation( ).toOSString( );
 258             }
 259             else {
 260                 // project is in workspace
 261                 projectPath = ResourcesPlugin.getWorkspace( ).getRoot( )
 262                     .getLocation( ).toOSString( )
 263                     + "/" + handle.getProject( ).getName( ); //$NON-NLS-1$
 264             }
 265 
 266             monitor.subTask( Messages.ProjectUtils_0 );
 267 
 268             // cleanup existing files
 269             ResourceUtils.removeFile( projectPath + "/.wesnoth" ); //$NON-NLS-1$
 270             ResourceUtils.removeFile( projectPath + "/.project" ); //$NON-NLS-1$
 271             ResourceUtils.removeFile( projectPath + "/.build.xml" ); //$NON-NLS-1$
 272             monitor.worked( 5 );
 273 
 274             monitor.subTask( String.format( Messages.ProjectUtils_4,
 275                 handle.getName( ) ) );
 276 
 277             // create the project
 278             if( description == null ) {
 279                 handle.create( monitor );
 280             }
 281             else {
 282                 handle.create( description, monitor );
 283             }
 284 
 285             handle.open( monitor );
 286             monitor.worked( 10 );
 287 
 288             monitor.subTask( Messages.ProjectUtils_6 );
 289             // add wesnoth nature
 290             IProjectDescription tmpDescription = handle.getDescription( );
 291             tmpDescription
 292                 .setNatureIds( new String[] { WesnothProjectNature.ID_NATURE } );
 293             handle.setDescription( tmpDescription, monitor );
 294             monitor.worked( 5 );
 295 
 296             if( ! skipCreateBuildXML ) {
 297                 // add the build.xml file, but only if the project is not
 298                 // located
 299                 // into the data/campaigns or user's /addons directory
 300                 Paths paths = Preferences.getPaths( installName );
 301                 String normalizedPath = StringUtils.normalizePath( projectPath );
 302                 if( ! normalizedPath.contains( StringUtils.normalizePath( paths
 303                     .getCampaignDir( ) ) )
 304                     && ! normalizedPath.contains( StringUtils
 305                         .normalizePath( paths.getAddonsDir( ) ) ) ) {
 306 
 307                     ArrayList< ReplaceableParameter > param = new ArrayList< ReplaceableParameter >( );
 308                     param.add( new ReplaceableParameter( "$$project_name", handle.getName( ) ) ); //$NON-NLS-1$
 309                     param.add( new ReplaceableParameter( "$$project_dir_name", handle.getName( ) ) ); //$NON-NLS-1$
 310                     ResourceUtils.createFile( handle, "build.xml", //$NON-NLS-1$
 311                         TemplateProvider.getInstance( ).getProcessedTemplate( "build_xml", param ), true ); //$NON-NLS-1$
 312                 }
 313             }
 314             monitor.worked( 10 );
 315 
 316             // save the install name
 317             ProjectCache cache = new ProjectCache( handle );
 318             cache.setInstallName( installName );
 319             cache.loadCache( );
 320             projectCache_.put( handle, cache );
 321 
 322             // create the Core library link
 323             createCoreLibraryFolder( handle, IResource.NONE );
 324             monitor.worked( 10 );
 325 
 326             handle.refreshLocal( IResource.DEPTH_ONE, monitor );
 327         } catch( CoreException e ) {
 328             Logger.getInstance( ).logException( e );
 329             return 1;
 330         }
 331         return 0;
 332     }
