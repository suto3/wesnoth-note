   9 function wml_actions.spread_bandit_villages(cfg)
  10         local x = cfg.x or helper.wml_error("[spread_bandit_villages] missing required x= attribute.")
  11         local y = cfg.y or helper.wml_error("[spread_bandit_villages] missing required y= attribute.")
  12         local count = cfg.count or helper.wml_error("[spread_bandit_villages] missing required count= attribute.")
  13         local types = cfg.types or helper.wml_error("[spread_bandit_villages] missing required types= attribute.")
  14 
  15         wesnoth.set_variable("villages_visited", 0)
  16         wesnoth.set_variable("boss_found", false)
  17 
  18         wesnoth.set_variable("bandit_types", types)
  19 
  20         local villages = wesnoth.get_villages(cfg)
  21 
  22         -- Shouldn't happen in the scenario, but a failsafe is always nice.
  23         if count > #villages then count = #villages end
  24 
  25         local village_i
  26 
  27         for i = 0, (count - 1) do
  28                 village_i = helper.rand("1.."..#villages)
  29 
  30                 wesnoth.set_variable(string.format("bandit_villages[%d].x", i), villages[village_i][1])
  31                 wesnoth.set_variable(string.format("bandit_villages[%d].y", i), villages[village_i][2])
  32                 table.remove(villages, village_i)
  33         end
  34 end

  36 local function bandits_found(x,y)
  37         local bandit_types = wesnoth.get_variable("bandit_types")
  38         local bandit_villages = helper.get_variable_array("bandit_villages")
  39         local boss_found = wesnoth.get_variable("boss_found")
  40         local visited = wesnoth.get_variable("villages_visited")
  41         local rand1 = helper.rand("3,4")
  42         local rand2 = helper.rand("2.."..rand1)
  43 
  44         for i=1,rand2 do
  45                 local radius = 1
  46                 local locs
  47                 repeat
  48                         locs = wesnoth.get_locations({T["not"] { T.filter {} } , T["and"] { x = x, y = y, radius = radius } })
  49                         radius = radius + 1
  50                 until locs[1]
  51 
  52                 local bandit = helper.rand(bandit_types)
  53                 local loc_i = helper.rand("1.."..#locs)
  54 
  55                 wml_actions.move_unit_fake({x = string.format("%d,%d", x, locs[loc_i][1]), y = string.format("%d,%d", y, locs[loc_i][2]), type = bandit, side = "4"})
  56                 wesnoth.put_unit(locs[loc_i][1], locs[loc_i][2], { type = bandit, side = "4", random_traits = "yes", generate_name = "yes", upkeep = "loyal" })
  57         end
  58 
  59         if not boss_found and visited > 2 then
  60                 local boss_chance = (100 / #bandit_villages)
  61                 local rand3 = helper.rand("1..100")
  62 
  63                 if rand3 <= boss_chance or #bandit_villages < 3 then
  64                         wesnoth.set_variable("boss_found", true)
  65                         local loc = wesnoth.get_locations({T["not"] { T.filter {} } , T["and"] { x = x, y = y, radius = 2 } })[1]
  66                         wesnoth.fire_event("boss_found", x, y, loc[1], loc[2])
  67                 end
  68         end
  69 end

  71 function wml_actions.bandit_village_capture(cfg)

  74         local y = cfg.y or helper.wml_error("[bandit_village_capture] missing required y= attribute.")
  75 
  76         for i=1,#bandit_villages do
  77                 if bandit_villages[i].x == x and bandit_villages[i].y == y then
  78                         wesnoth.set_variable(string.format("bandit_villages[%d]", i - 1))
  79 
  80                         local visited = wesnoth.get_variable("villages_visited")
  81                         wesnoth.set_variable("villages_visited", visited + 1)
  82 
  83                         wesnoth.fire("message" , { x = x , y = y , message = _"They're here!"})
  84 
  85                         bandits_found(x,y)
  86                         return
  87                 end
  88         end
  89 
  90         wesnoth.fire("message" , { x = x , y = y , message = _"No outlaws in this village."})
  91 end
