  43 surface getMinimap(int w, int h, const gamemap &map, const team *vw, const std::map<map_location,unsigned int> *reach_map)
  44 {
  45         const terrain_type_data & tdata = *map.tdata();
  46 
  47 
  48         const bool preferences_minimap_draw_terrain = preferences::minimap_draw_terrain();
  49         const bool preferences_minimap_terrain_coding = preferences::minimap_terrain_coding();
  50         const bool preferences_minimap_draw_villages = preferences::minimap_draw_villages();
  51         const bool preferences_minimap_unit_coding = preferences::minimap_movement_coding();
  52 
  53         const int scale = (preferences_minimap_draw_terrain && preferences_minimap_terrain_coding) ? 24 : 4;
  54 
  55         DBG_DP << "creating minimap " << int(map.w()*scale*0.75) << "," << map.h()*scale << "\n";
  56 
  57         const size_t map_width = map.w()*scale*3/4;
  58         const size_t map_height = map.h()*scale;
  59         if(map_width == 0 || map_height == 0) {
  60                 return surface(nullptr);
  61         }
  62 
  63         if(!preferences_minimap_draw_villages && !preferences_minimap_draw_terrain)
  64         {
  65                 //return if there is nothing to draw.
  66                 //(optimisation)
  67                 double ratio = std::min<double>( w*1.0 / map_width, h*1.0 / map_height);
  68                 return create_neutral_surface(map_width * ratio, map_height * ratio);
  69         }
  70 
  71         surface minimap(create_neutral_surface(map_width, map_height));
  72         if(minimap == nullptr)
  73                 return surface(nullptr);
  74 
  75         typedef mini_terrain_cache_map cache_map;
  76         cache_map *normal_cache = &mini_terrain_cache;
  77         cache_map *fog_cache = &mini_fogged_terrain_cache;
  78         cache_map *highlight_cache = &mini_highlighted_terrain_cache;
  79 
  80         for(int y = 0; y <= map.total_height(); ++y)
  81                 for(int x = 0; x <= map.total_width(); ++x) {
  82 
  83                         const map_location loc(x,y);
  84                         if(!map.on_board_with_border(loc))
  85                                 continue;
  86 
  87                         const bool highlighted = reach_map && reach_map->count(loc) != 0;
  88 
  89                         const bool shrouded = (resources::screen != nullptr && resources::screen->is_blindfolded()) || (vw != nullptr && vw->shrouded(loc));
  90                         // shrouded hex are not considered fogged (no need to fog a black image)
  91                         const bool fogged = (vw != nullptr && !shrouded && vw->fogged(loc));
  92 
  93                         const t_translation::terrain_code terrain = shrouded ?
  94                                         t_translation::VOID_TERRAIN : map[loc];
  95                         const terrain_type& terrain_info = tdata.get_terrain_info(terrain);
  96 
  97                         // we need a balanced shift up and down of the hexes.
  98                         // if not, only the bottom half-hexes are clipped
  99                         // and it looks asymmetrical.
 100 
 101                         SDL_Rect maprect = sdl::create_rect(
 102                                         x * scale * 3 / 4 - (scale / 4)
 103                                         , y * scale + scale / 4 * (is_odd(x) ? 1 : -1) - (scale / 4)
 104                                         , 0
 105                                         , 0);
 106 
 107                         if (preferences_minimap_draw_terrain) {
 108 
 109                                 if (preferences_minimap_terrain_coding) {
 110 
 111                                         surface surf(nullptr);
 112 
 113                                         bool need_fogging = false;
 114                                         bool need_highlighting = false;
 115 
 116                                         cache_map* cache = fogged ? fog_cache : normal_cache;
 117                                         if (highlighted)
 118                                                 cache = highlight_cache;
 119                                         cache_map::iterator i = cache->find(terrain);
 120 
 121                                         if (fogged && i == cache->end()) {
 122                                                 // we don't have the fogged version in cache
 123                                                 // try the normal cache and ask fogging the image
 124                                                 cache = normal_cache;
 125                                                 i = cache->find(terrain);
 126                                                 need_fogging = true;
 127                                         }
 128 
 129                                         if (highlighted && i == cache->end()) {
 130                                                 // we don't have the highlighted version in cache
 131                                                 // try the normal cache and ask fogging the image
 132                                                 cache = normal_cache;
 133                                                 i = cache->find(terrain);
 134                                                 need_highlighting = true;
 135                                         }
 136 
 137                                         if(i == cache->end() && !terrain_info.minimap_image().empty()) {
 138                                                 std::string base_file =
 139                                                                 "terrain/" + terrain_info.minimap_image() + ".png";
 140                                                 surface tile = get_image(base_file,image::HEXED);
 141 
 142                                                 //Compose images of base and overlay if necessary
 143                                                 // NOTE we also skip overlay when base is missing (to avoid hiding the error)
 144                                                 if(tile != nullptr && tdata.get_terrain_info(terrain).is_combined() && !terrain_info.minimap_image_overlay().empty()) {
 145                                                         std::string overlay_file =
 146                                                                         "terrain/" + terrain_info.minimap_image_overlay() + ".png";
 147                                                         surface overlay = get_image(overlay_file,image::HEXED);
 148 
 149                                                         if(overlay != nullptr && overlay != tile) {
 150                                                                 surface combined = create_neutral_surface(tile->w, tile->h);
 151                                                                 SDL_Rect r = sdl::create_rect(0,0,0,0);
 152                                                                 sdl_blit(tile, nullptr, combined, &r);
 153                                                                 r.x = std::max(0, (tile->w - overlay->w)/2);
 154                                                                 r.y = std::max(0, (tile->h - overlay->h)/2);
 155                                                                 surface overlay_neutral = make_neutral_surface(overlay);
 156                                                                 sdl_blit(overlay_neutral, nullptr, combined, &r);
 157                                                                 tile = combined;
 158                                                         }
 159                                                 }
 160 
 161                                                 surf = scale_surface_sharp(tile, scale, scale);
 162 
 163                                                 i = normal_cache->insert(cache_map::value_type(terrain,surf)).first;
 164                                         }
 165 
 166                                         if (i != cache->end())
 167                                         {
 168                                                 surf = i->second;
 169 
 170                                                 if (need_fogging) {
 171                                                         surf = adjust_surface_color(surf, -50, -50, -50);
 172                                                         fog_cache->insert(cache_map::value_type(terrain, surf));
 173                                                 }
 174 
 175                                                 if (need_highlighting) {
 176                                                         surf = adjust_surface_color(surf, 50, 50, 50);
 177                                                         highlight_cache->insert(cache_map::value_type(terrain, surf));
 178                                                 }
 179                                         }
 180 
 181                                         if(surf != nullptr)
 182                                                 sdl_blit(surf, nullptr, minimap, &maprect);
 183 
 184                                 } else {
 185 
 186                                         color_t col;
 187                                         std::map<std::string, color_range>::const_iterator it = game_config::team_rgb_range.find(terrain_info.id());
 188                                         if (it == game_config::team_rgb_range.end()) {
 189                                                 col = color_t(0,0,0,0);
 190                                         } else
 191                                                 col = it->second.rep();
 192 
 193                                         bool first = true;
 194                                         const t_translation::ter_list& underlying_terrains = tdata.underlying_union_terrain(terrain);
 195                                         for(const t_translation::terrain_code& underlying_terrain : underlying_terrains) {
 196 
 197                                                 const std::string& terrain_id = tdata.get_terrain_info(underlying_terrain).id();
 198                                                 it = game_config::team_rgb_range.find(terrain_id);
 199                                                 if (it == game_config::team_rgb_range.end())
 200                                                         continue;
 201 
 202                                                 color_t tmp = it->second.rep();
 203 
 204                                                 if (fogged) {
 205                                                         if (tmp.b < 50) tmp.b = 0;
 206                                                         else tmp.b -= 50;
 207                                                         if (tmp.g < 50) tmp.g = 0;
 208                                                         else tmp.g -= 50;
 209                                                         if (tmp.r < 50) tmp.r = 0;
 210                                                         else tmp.r -= 50;
 211                                                 }
 212 
 213                                                 if (highlighted) {
 214                                                         if (tmp.b > 205) tmp.b = 255;
 215                                                         else tmp.b += 50;
 216                                                         if (tmp.g > 205) tmp.g = 255;
 217                                                         else tmp.g += 50;
 218                                                         if (tmp.r > 205) tmp.r = 255;
 219                                                         else tmp.r += 50;
 220                                                 }
 221 
 222                                                 if (first) {
 223                                                         first = false;
 224                                                         col = tmp;
 225                                                 } else {
 226                                                         col.r = col.r - (col.r - tmp.r)/2;
 227                                                         col.g = col.g - (col.g - tmp.g)/2;
 228                                                         col.b = col.b - (col.b - tmp.b)/2;
 229                                                 }
 230                                         }
 231                                         SDL_Rect fillrect = sdl::create_rect(maprect.x, maprect.y, scale * 3/4, scale);
 232                                         const Uint32 mapped_col = SDL_MapRGB(minimap->format,col.r,col.g,col.b);
 233                                         sdl::fill_rect(minimap, &fillrect, mapped_col);
 234                                 }
 235                         }
 236 
 237                         if (terrain_info.is_village() && preferences_minimap_draw_villages) {
 238 
 239                                 int side = (resources::gameboard ? resources::gameboard->village_owner(loc) : -1); //check needed for mp create dialog
 240 
 241                                 // TODO: Add a key to [game_config][colors] for this
 242                                 auto iter = game_config::team_rgb_range.find("white");
 243                                 color_t col(255,255,255);
 244                                 if(iter != game_config::team_rgb_range.end()) {
 245                                         col = iter->second.min();
 246                                 }
 247 
 248                                 if (!fogged) {
 249                                         if (side > -1) {
 250 
 251                                                 if (preferences_minimap_unit_coding || !vw ) {
 252                                                         col = team::get_minimap_color(side + 1);
 253                                                 } else {
 254 
 255                                                         if (vw->owns_village(loc))
 256                                                                 col = game_config::color_info(preferences::unmoved_color()).rep();
 257                                                         else if (vw->is_enemy(side + 1))
 258                                                                 col = game_config::color_info(preferences::enemy_color()).rep();
 259                                                         else
 260                                                                 col = game_config::color_info(preferences::allied_color()).rep();
 261                                                 }
 262                                         }
 263                                 }
 264 
 265                                 SDL_Rect fillrect = sdl::create_rect(
 266                                                 maprect.x
 267                                                 , maprect.y
 268                                                 , scale * 3/4
 269                                                 , scale
 270                                 );
 271 
 272                                 const Uint32 mapped_col = SDL_MapRGB(minimap->format,col.r,col.g,col.b);
 273                                 sdl::fill_rect(minimap, &fillrect, mapped_col);
 274 
 275                         }
 276 
 277                 }
 278 
 279         double wratio = w*1.0 / minimap->w;
 280         double hratio = h*1.0 / minimap->h;
 281         double ratio = std::min<double>(wratio, hratio);
 282 
 283         minimap = scale_surface_sharp(minimap,
 284                 static_cast<int>(minimap->w * ratio), static_cast<int>(minimap->h * ratio));
 285 
 286         DBG_DP << "done generating minimap\n";
 287 
 288         return minimap;
 289 }
