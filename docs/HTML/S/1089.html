 339 inline bool tiles_adjacent(const map_location& a, const map_location& b)
 340 {
 341         // Two tiles are adjacent:
 342         // if y is different by 1, and x by 0,
 343         // or if x is different by 1 and y by 0,
 344         // or if x and y are each different by 1,
 345         // and the x value of the hex with the greater y value is even.
 346 
 347         switch (a.y - b.y) {
 348                 case 1 :
 349                         switch (a.x - b.x) {
 350                                 case 1:
 351                                 case -1:
 352                                         return (a.x & 1) == 0;
 353                                 case 0:
 354                                         return true;
 355                                 default:
 356                                         return false;
 357                         }
 358                 case -1 :
 359                         switch (a.x - b.x) {
 360                                 case 1:
 361                                 case -1:
 362                                         return (b.x & 1) == 0;
 363                                 case 0:
 364                                         return true;
 365                                 default:
 366                                         return false;
 367                         }
 368                 case 0 :
 369                         return ((a.x - b.x) == 1) || ((a.x - b.x) == - 1);
 370                 default:
 371                         return false;
 372         }

 382 inline size_t distance_between(const map_location& a, const map_location& b)
 383 {
 384         const size_t hdistance = std::abs(a.x - b.x);
 385 
 386         const size_t vpenalty = ( (((a.x & 1)==0) && ((b.x & 1)==1) && (a.y < b.y))
 387                 || (((b.x & 1)==0) && ((a.x & 1)==1) && (b.y < a.y)) ) ? 1 : 0;
 388 
 389 /* Don't want to include util.hpp in this header
 390         const size_t vpenalty = ( (is_even(a.x) && is_odd(b.x) && (a.y < b.y))
 391                 || (is_even(b.x) && is_odd(a.x) && (b.y < a.y)) ) ? 1 : 0;
 392 */
 393         // For any non-negative integer i, i - i/2 - i%2 == i/2
 394         // previously returned (hdistance + vdistance - vsavings)
 395         // = hdistance + vdistance - minimum(vdistance,hdistance/2+hdistance%2)
 396         // = maximum(hdistance, vdistance+hdistance-hdistance/2-hdistance%2)
 397         // = maximum(hdistance,std::abs(a.y-b.y)+vpenalty+hdistance/2)
 398 
 399         return std::max<int>(hdistance, std::abs(a.y - b.y) + vpenalty + hdistance/2);
 400 }
