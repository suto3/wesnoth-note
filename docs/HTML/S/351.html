 116 // Helper function to refresh resolution list
 117 void preferences_dialog::set_resolution_list(menu_button& res_list, CVideo& video)
 118 {
 119         resolutions_ = video.get_available_resolutions(true);
 120 
 121         std::vector<config> options;
 122         for(const auto& res : resolutions_)
 123         {
 124                 config option;
 125                 option["label"] = formatter() << res.first << font::unicode_multiplication_sign << res.second;
 126 
 127                 const int div = boost::math::gcd(res.first, res.second);
 128                 const int ratio[2] {res.first/div, res.second/div};
 129                 if(ratio[0] <= 10 || ratio[1] <= 10) {
 130                         option["details"] = formatter() << "<span color='#777777'>(" << ratio[0] << ':' << ratio[1] << ")</span>";
 131                 }
 132 
 133                 options.push_back(option);
 134         }
 135 
 136         const unsigned current_res = std::find(resolutions_.begin(), resolutions_.end(),
 137                 video.current_resolution()) - resolutions_.begin();
 138 
 139         res_list.set_values(options, current_res);
 140 }

 142 std::map<std::string, string_map> preferences_dialog::get_friends_list_row_data(const acquaintance& entry)
 143 {
 144         std::map<std::string, string_map> data;
 145         string_map item;
 146 
 147         std::string image = "friend.png";
 148         std::string descriptor = _("friend");
 149         std::string notes;
 150 
 151         if(entry.get_status() == "ignore") {
 152                 image = "ignore.png";
 153                 descriptor = _("ignored");
 154         }
 155 
 156         if(!entry.get_notes().empty()) {
 157                 notes = " <small>(" + entry.get_notes() + ")</small>";
 158         }
 159 
 160         item["use_markup"] = "true";
 161 
 162         item["label"] = "misc/status-" + image;
 163         data.emplace("friend_icon", item);
 164 
 165         item["label"] = entry.get_nick() + notes;
 166         data.emplace("friend_name", item);
 167 
 168         item["label"] = "<small>" + descriptor + "</small>";
 169         data.emplace("friend_status", item);
 170 
 171         return data;
 172 }

 174 void preferences_dialog::on_friends_list_select(listbox& list, text_box& textbox)
 175 {
 176         const int num_friends = get_acquaintances().size();
 177         const int sel = list.get_selected_row();
 178 
 179         if(sel < 0 || sel >= num_friends) {
 180                 return;
 181         }
 182 
 183         std::map<std::string, acquaintance>::const_iterator who = get_acquaintances().begin();
 184         std::advance(who, sel);
 185 
 186         textbox.set_value(who->second.get_nick() + " " + who->second.get_notes());
 187 }

 203 void preferences_dialog::add_friend_list_entry(const bool is_friend, text_box& textbox, window& window)
 204 {
 205         std::string username = textbox.text();
 206         if(username.empty()) {
 207                 gui2::show_transient_message(window.video(), "", _("No username specified"), "", false, false, true);
 208                 return;
 209         }
 210 
 211         std::string reason;
 212 
 213         size_t pos = username.find_first_of(' ');
 214         if(pos != std::string::npos) {
 215                 reason = username.substr(pos + 1);
 216                 username = username.substr(0, pos);
 217         }
 218 
 219         acquaintance* entry = add_acquaintance(username, (is_friend ? "friend": "ignore"), reason);
 220         if(!entry) {
 221                 gui2::show_transient_message(window.video(), _("Error"), _("Invalid username"), "", false, false, true);
 222                 return;
 223         }
 224 
 225         textbox.clear();
 226 
 227         listbox& list = find_widget<listbox>(&window, "friends_list", false);
 228         list.add_row(get_friends_list_row_data(*entry));
 229 
 230         update_friends_list_controls(window, list);
 231 }

 233 void preferences_dialog::remove_friend_list_entry(listbox& friends_list, text_box& textbox, window& window)
 234 {
 235         const int selected_row = std::max(0, friends_list.get_selected_row());
 236 
 237         std::map<std::string, acquaintance>::const_iterator who = get_acquaintances().begin();
 238         std::advance(who, selected_row);
 239 
 240         const std::string to_remove = !textbox.text().empty() ? textbox.text() : who->second.get_nick();
 241 
 242         if(to_remove.empty()) {
 243                 gui2::show_transient_message(window.video(), "", _("No username specified"), "", false, false, true);
 244                 return;
 245         }
 246 
 247         if(!remove_acquaintance(to_remove)) {
 248                 gui2::show_transient_error_message(window.video(), _("Not on friends or ignore lists"));
 249                 return;
 250         }
 251 
 252         textbox.clear();
 253 
 254         listbox& list = find_widget<listbox>(&window, "friends_list", false);
 255         list.remove_row(selected_row);
 256 
 257         update_friends_list_controls(window, list);
 258 }

 799 void preferences_dialog::add_hotkey_callback(listbox& hotkeys)
 800 {
 801         CVideo& video = hotkeys.get_window()->video();
 802         int row_number = hotkeys.get_selected_row();
 803         const hotkey::hotkey_command& hotkey_item = *visible_hotkeys_[row_number];
 804 
 805         gui2::dialogs::hotkey_bind bind_dlg(hotkey_item.command);
 806         bind_dlg.show(video);
 807 
 808         hotkey::hotkey_ptr newhk = bind_dlg.get_new_binding();
 809         hotkey::hotkey_ptr oldhk;
 810 
 811         // only if not cancelled.
 812         if(newhk.get() == nullptr) {
 813                 return;
 814         }
 815 
 816         for(const hotkey::hotkey_ptr& hk : hotkey::get_hotkeys()) {
 817                 if(!hk->is_disabled() && newhk->bindings_equal(hk)) {
 818                         oldhk = hk;
 819                 }
 820         }
 821 
 822         hotkey::scope_changer scope_restorer;
 823         hotkey::set_active_scopes(hotkey_item.scope);
 824 
 825         if(oldhk && oldhk->get_command() == hotkey_item.command) {
 826                 return;
 827         }
 828 
 829         if(oldhk && oldhk->get_command() != "null") {
 830                 const std::string text = vgettext("“<b>$hotkey_sequence|</b>” is in use by “<b>$old_hotkey_action|</b>”.\nDo you wish to reassign it to “<b>$new_hotkey_action|</b>”?", {
 831                         {"hotkey_sequence",   oldhk->get_name()},
 832                         {"old_hotkey_action", hotkey::get_description(oldhk->get_command())},
 833                         {"new_hotkey_action", hotkey::get_description(newhk->get_command())}
 834                 });
 835 
 836                 const int res = gui2::show_message(video, _("Reassign Hotkey"), text, gui2::dialogs::message::yes_no_buttons, true);
 837                 if(res != gui2::window::OK) {
 838                         return;
 839                 }
 840         }
 841 
 842         hotkey::add_hotkey(newhk);
 843 
 844         // We need to recalculate all hotkey names in because we might have removed a hotkey from another command.
 845         for(size_t i = 0; i < hotkeys.get_item_count(); ++i) {
 846                 const hotkey::hotkey_command& hotkey_item_row = *visible_hotkeys_[i];
 847                 find_widget<label>(hotkeys.get_row_grid(i), "lbl_hotkey", false).set_label(hotkey::get_names(hotkey_item_row.command));
 848         }
 849 }

 873 void preferences_dialog::hotkey_type_filter_callback(window& window) const
 874 {
 875         const multimenu_button& hotkey_menu = find_widget<const multimenu_button>(&window, "hotkey_category_menu", false);
 876 
 877         boost::dynamic_bitset<> toggle_states = hotkey_menu.get_toggle_states();
 878         boost::dynamic_bitset<> res(visible_hotkeys_.size());
 879 
 880         if(!toggle_states.none()) {
 881                 for(size_t h = 0; h < visible_hotkeys_.size(); ++h) {
 882                         int index = 0;
 883 
 884                         for(size_t i = 0; i < cat_names_.size(); ++i) {
 885                                 hotkey::HOTKEY_CATEGORY cat = hotkey::HOTKEY_CATEGORY(i);
 886 
 887                                 if(visible_hotkeys_[h]->category == cat) {
 888                                         index = i;
 889                                         break;
 890                                 }
 891                         }
 892 
 893                         res[h] = toggle_states[index];
 894                 }
 895         } else {
 896                 // Nothing selected. It means that *all* categories are shown.
 897                 res = ~res;
 898         }
 899 
 900         find_widget<listbox>(&window, "list_hotkeys", false).set_row_shown(res);
 901 }

 958 static int index_in_pager_range(const int& first, const stacked_widget& pager)
 959 {
 960         // Ensure the specified index is between 0 and one less than the max
 961         // number of pager layers (since get_layer_count returns one-past-end).
 962         return std::min<int>(std::max(0, first), pager.get_layer_count() - 1);
 963 }

 965 void preferences_dialog::pre_show(window& window)
 966 {
 967         set_always_save_fields(true);
 968 
 969         //
 970         // Status labels
 971         // These need to be set here in pre_show, once the fields are initialized. For some reason, this
 972         // is not the case for those in Advanced
 973         //
 974 
 975         gui2::bind_status_label<slider>(window, "max_saves_slider", [](slider& s)->std::string {
 976                 return s.get_value() == INFINITE_AUTO_SAVES ? _("∞") : s.get_value_label().str();
 977         });
 978 
 979         gui2::bind_status_label<slider>(window, "turbo_slider",     [](slider& s)->std::string {
 980                 return s.get_value_label();
 981         });
 982 
 983         gui2::bind_status_label<slider>(window, "scaling_slider",   [](slider& s)->std::string {
 984                 return s.get_value_label() + "%";
 985         });
 986 
 987         listbox& selector = find_widget<listbox>(&window, "selector", false);
 988         stacked_widget& pager = find_widget<stacked_widget>(&window, "pager", false);
 989 
 990 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 991         connect_signal_notify_modified(selector, std::bind(
 992                         &preferences_dialog::on_page_select,
 993                         this,
 994                         std::ref(window)));
 995 #else
 996         selector.set_callback_value_change(dialog_callback
 997                         <preferences_dialog, &preferences_dialog::on_page_select>);
 998 #endif
 999         window.keyboard_capture(&selector);
1000 
1001         VALIDATE(selector.get_item_count() == pager.get_layer_count(),
1002                 "The preferences pager and its selector listbox do not have the same number of items.");
1003 
1004         const int main_index = index_in_pager_range(initial_index_.first, pager);
1005 
1006         // Loops through each pager layer and checks if it has both a tab bar
1007         // and stack. If so, it initilizes the options for the former and
1008         // selects the specified layer of the latter.
1009         for(unsigned int i = 0; i < pager.get_layer_count(); ++i) {
1010                 listbox* tab_selector = find_widget<listbox>(
1011                         pager.get_layer_grid(i), "tab_selector", false, false);
1012 
1013                 stacked_widget* tab_pager = find_widget<stacked_widget>(
1014                         pager.get_layer_grid(i), "tab_pager", false, false);
1015 
1016                 if(tab_pager && tab_selector) {
1017                         const int ii = static_cast<int>(i);
1018                         const int tab_index = index_in_pager_range(initial_index_.second, *tab_pager);
1019                         const int to_select = (ii == main_index ? tab_index : 0);
1020 
1021                         // Initialize tabs for this page
1022                         initialize_tabs(window, *tab_selector);
1023 
1024                         tab_selector->select_row(to_select);
1025                         tab_pager->select_layer(to_select);
1026                 }
1027         }
1028 
1029         // Finally, select the initial main page
1030         selector.select_row(main_index);
1031         pager.select_layer(main_index);
1032 }
