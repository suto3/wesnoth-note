  50 void unit_advance::pre_show(window& window)
  51 {
  52         listbox& list = find_widget<listbox>(&window, "advance_choice", false);
  53 
  54 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  55         connect_signal_notify_modified(*list,
  56                 std::bind(&unit_advance::list_item_clicked,
  57                 *this,
  58                 std::ref(window)));
  59 #else
  60         list.set_callback_value_change(
  61                 dialog_callback<unit_advance, &unit_advance::list_item_clicked>);
  62 #endif
  63 
  64         window.keyboard_capture(&list);
  65 
  66         connect_signal_mouse_left_click(
  67                 find_widget<button>(&window, "show_help", false),
  68                 std::bind(&unit_advance::show_help, this, std::ref(window)));
  69 
  70         for(size_t i = 0; i < previews_.size(); i++) {
  71                 const unit& sample = *previews_[i];
  72 
  73                 std::map<std::string, string_map> row_data;
  74                 string_map column;
  75 
  76                 std::string image_string, name = sample.type_name();
  77 
  78                 // This checks if we've finished iterating over the last unit type advancements
  79                 // and are into the modification-based advancements.
  80                 if(i >= last_real_advancement_) {
  81                         auto iter = sample.get_modifications().child_range("advancement").end();
  82                         iter--;
  83 
  84                         if(iter->has_attribute("image")) {
  85                                 image_string = iter->get("image")->str();
  86                         }
  87 
  88                         name = iter->get("description")->str();
  89                 }
  90 
  91                 if(image_string.empty()) {
  92                         image_string = sample.type().image() + sample.image_mods();
  93                 }
  94 
  95                 column["label"] = image_string;
  96                 row_data.emplace("advancement_image", column);
  97 
  98                 column["label"] = name;
  99                 row_data.emplace("advancement_name", column);
 100 
 101                 list.add_row(row_data);
 102         }
 103 
 104         list_item_clicked(window);
 105 
 106         // Disable ESC existing
 107         window.set_escape_disabled(true);
 108 }
 109 
 110 void unit_advance::list_item_clicked(window& window)
 111 {
 112         const int selected_row
 113                 = find_widget<listbox>(&window, "advance_choice", false).get_selected_row();
 114 
 115         if(selected_row == -1) {
 116                 return;
 117         }
 118 
 119         find_widget<unit_preview_pane>(&window, "advancement_details", false)
 120                 .set_displayed_unit(*previews_[selected_row]);
 121 }
 122 
 123 void unit_advance::show_help(window& window)
 124 {
 125         help::show_help(window.video(), "advancement");
 126 }
 127 
 128 void unit_advance::post_show(window& window)
 129 {
 130         if(get_retval() == window::OK) {
 131                 selected_index_ = find_widget<listbox>(&window, "advance_choice", false)
 132                         .get_selected_row();
 133         }
 134 }
