1081             : function_expression("zip", args, 1, -1)
1082         {}

1084         struct indexer {
1085                 size_t i;
1086                 explicit indexer(size_t i) : i(i) {}
1087                 variant operator()(const variant& v) const {
1088                         if(i >= v.num_elements()) {
1089                                 return variant();
1090                         } else {
1091                                 return v[i];
1092                         }
1093                 }
1094         };
1095         struct comparator {
1096                 bool operator()(const variant& a, const variant& b) const {
1097                         return a.num_elements() < b.num_elements();
1098                 }
1099         };
1100         std::vector<variant> get_input(const formula_callable& variables, formula_debugger* fdb) const {
1101                 if(args().size() == 1) {

1104                 } else {
1105                         std::vector<variant> input;
1106                         input.reserve(args().size());
1107                         for(expression_ptr expr : args()) {
1108                                 input.push_back(expr->evaluate(variables, fdb));
1109                         }
1110                         return input;
1111                 }
1112         }

1114                 const std::vector<variant> input = get_input(variables, fdb);
1115                 std::vector<variant> output;
1116                 // So basically this does [[a,b,c],[d,e,f],[x,y,z]] -> [[a,d,x],[b,e,y],[c,f,z]]
1117                 // Or [[a,b,c,d],[x,y,z]] -> [[a,x],[b,y],[c,z],[d,null()]]
1118                 size_t max_i = std::max_element(input.begin(), input.end(), comparator())->num_elements();
1119                 output.reserve(max_i);
1120                 for(size_t i = 0; i < max_i; i++) {
1121                         std::vector<variant> elem(input.size());
1122                         std::transform(input.begin(), input.end(), elem.begin(), indexer(i));
1123                         output.emplace_back(elem);
1124                 }

1524   : function_expression(name, args, arg_names.size(), arg_names.size()),

1536 variant formula_function_expression::execute(const formula_callable& variables, formula_debugger *fdb) const
1537 {
1538         static std::string indent;
1539         indent += "  ";
1540         DBG_NG << indent << "executing '" << formula_->str() << "'\n";
1541         const int begin_time = SDL_GetTicks();
1542         map_formula_callable callable;
1543         for(size_t n = 0; n != arg_names_.size(); ++n) {
1544                 variant var = args()[n]->evaluate(variables,fdb);
1545                 callable.add(arg_names_[n], var);
1546                 if(static_cast<int>(n) == star_arg_) {
1547                         callable.set_fallback(var.as_callable());
1548                 }
1549         }
1550 
1551         if(precondition_) {
1552                 if(!precondition_->evaluate(callable,fdb).as_bool()) {
1553                         DBG_NG << "FAILED function precondition for function '" << formula_->str() << "' with arguments: ";
1554                         for(size_t n = 0; n != arg_names_.size(); ++n) {
1555                                 DBG_NG << "  arg " << (n+1) << ": " << args()[n]->evaluate(variables,fdb).to_debug_string() << "\n";
1556                         }
1557                 }
1558         }
1559 
1560         variant res = formula_->evaluate(callable,fdb);
1561         const int taken = SDL_GetTicks() - begin_time;
1562         DBG_NG << indent << "returning: " << taken << "\n";
1563         indent.resize(indent.size() - 2);
1564 
1565         return res;
1566 }
