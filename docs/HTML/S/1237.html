  66 attack::attack(config const& cfg, bool hidden)
  67         : move(cfg,hidden)
  68         , target_hex_(cfg.child("target_hex_")["x"],cfg.child("target_hex_")["y"], wml_loc())
  69         , weapon_choice_(cfg["weapon_choice_"].to_int(-1)) //default value: -1
  70         , attack_movement_cost_()
  71         , temp_movement_subtracted_(0)
  72 {
  73         // Validate target_hex
  74         if(!tiles_adjacent(target_hex_,get_dest_hex()))
  75                 throw action::ctor_err("attack: Invalid target_hex_");
  76 
  77         // Validate weapon_choice_
  78         if(weapon_choice_ < 0 || weapon_choice_ >= static_cast<int>(get_unit()->attacks().size()))
  79                 throw action::ctor_err("attack: Invalid weapon_choice_");
  80 
  81         // Construct attack_movement_cost_
  82         attack_movement_cost_ = get_unit()->attacks()[weapon_choice_].movement_used();
  83 
  84         this->init();
  85 }

 113 void attack::execute(bool& success, bool& complete)
 114 {
 115         if(!valid()) {
 116                 success = false;
 117                 //Setting complete to true signifies to side_actions to delete the planned action: nothing more to do with it.
 118                 complete = true;
 119                 return;
 120         }
 121 
 122         LOG_WB << "Executing: " << shared_from_this() << "\n";
 123 
 124         if (route_->steps.size() >= 2)
 125         {
 126                 bool m_success, m_complete;
 127                 move::execute(m_success,m_complete);
 128                 if(!m_success) {
 129                         //Move failed for some reason, so don't attack.
 130                         success = false;
 131                         complete = true;
 132                         return;
 133                 }
 134         }
 135 
 136         resources::controller->get_mouse_handler_base().attack_enemy(get_dest_hex(), get_target_hex(), weapon_choice_);
 137         complete = true;
 138 
 139         //check that attacking unit is still alive, if not, consider the attack a failure
 140         unit_map::const_iterator survivor = resources::gameboard->units().find(get_dest_hex());
 141         if(!survivor.valid() || survivor->id() != unit_id_)
 142         {
 143                 success = false;
 144         }
 145 
 146         success = true;
 147 }

 149 void attack::apply_temp_modifier(unit_map& unit_map)
 150 {
 151         move::apply_temp_modifier(unit_map);
 152         assert(get_unit());
 153         unit& unit = *get_unit();
 154         DBG_WB << unit.name() << " [" << unit.id()
 155                                         << "] has " << unit.attacks_left() << " attacks, decreasing by one" << "\n";
 156         assert(unit.attacks_left() > 0);
 157         unit.set_attacks(unit.attacks_left() - 1);
 158 
 159         //Calculate movement to subtract
 160         temp_movement_subtracted_ = unit.movement_left() >= attack_movement_cost_ ? attack_movement_cost_ : 0 ;
 161         DBG_WB << "Attack: Changing movement points for unit " << unit.name() << " [" << unit.id()
 162                                 << "] from " << unit.movement_left() << " to "
 163                                 << unit.movement_left() - temp_movement_subtracted_ << ".\n";
 164         unit.set_movement(unit.movement_left() - temp_movement_subtracted_, true);
 165 
 166         //Update status of fake unit (not undone by remove_temp_modifiers)
 167         //@todo this contradicts the name "temp_modifiers"
 168         if (fake_unit_) { //Attacks that are not attack-moves don't have fake units
 169                 fake_unit_->set_movement(unit.movement_left(), true);
 170                 fake_unit_->set_attacks(unit.attacks_left());
 171         }
 172 }

 174 void attack::remove_temp_modifier(unit_map& unit_map)
 175 {
 176         assert(get_unit());
 177         unit& unit = *get_unit();
 178         DBG_WB << unit.name() << " [" << unit.id()
 179                                         << "] has " << unit.attacks_left() << " attacks, increasing by one" << "\n";
 180         unit.set_attacks(unit.attacks_left() + 1);
 181         DBG_WB << "Attack: Changing movement points for unit " << unit.name() << " [" << unit.id()
 182                                 << "] from " << unit.movement_left() << " to "
 183                                 << unit.movement_left() + temp_movement_subtracted_ << ".\n";
 184         unit.set_movement(unit.movement_left() + temp_movement_subtracted_, true);
 185         temp_movement_subtracted_ = 0;
 186         move::remove_temp_modifier(unit_map);
 187 }

 189 void attack::draw_hex(const map_location& hex)
 190 {
 191         if (hex == get_dest_hex() || hex == target_hex_) //draw attack indicator
 192         {
 193                 //@todo: replace this by either the use of transparency + LAYER_ATTACK_INDICATOR,
 194                 //or a dedicated layer
 195                 const display::drawing_layer layer = display::LAYER_FOOTSTEPS;
 196 
 197                 //calculate direction (valid for both hexes)
 198                 std::string direction_text = map_location::write_direction(
 199                                 get_dest_hex().get_relative_dir(target_hex_));
 200 
 201                 if (hex == get_dest_hex()) //add symbol to attacker hex
 202                 {
 203                         int xpos = resources::screen->get_location_x(get_dest_hex());
 204                         int ypos = resources::screen->get_location_y(get_dest_hex());
 205 
 206                         resources::screen->drawing_buffer_add(layer, get_dest_hex(), xpos, ypos,
 207                                         image::get_image("whiteboard/attack-indicator-src-" + direction_text + ".png", image::SCALED_TO_HEX));
 208                 }
 209                 else if (hex == target_hex_) //add symbol to defender hex
 210                 {
 211                         int xpos = resources::screen->get_location_x(target_hex_);
 212                         int ypos = resources::screen->get_location_y(target_hex_);
 213 
 214                         resources::screen->drawing_buffer_add(layer, target_hex_, xpos, ypos,
 215                                         image::get_image("whiteboard/attack-indicator-dst-" + direction_text + ".png", image::SCALED_TO_HEX));
 216                 }
 217         }
 218 }

 226 action::error attack::check_validity() const
 227 {
 228         // Verify that the unit that planned this attack exists
 229         if(!get_unit()) {
 230                 return NO_UNIT;
 231         }
 232         // Verify that the target hex is still valid
 233         if(!target_hex_.valid()) {
 234                 return INVALID_LOCATION;
 235         }
 236         // Verify that the target hex isn't empty
 237         if(resources::gameboard->units().find(target_hex_) == resources::gameboard->units().end()){
 238                 return NO_TARGET;
 239         }
 240         // Verify that the attacking unit has attacks left
 241         if(get_unit()->attacks_left() <= 0) {
 242                 return NO_ATTACK_LEFT;
 243         }
 244         // Verify that the attacker and target are enemies
 245         if(!resources::gameboard->get_team(get_unit()->side()).is_enemy(resources::gameboard->units().find(target_hex_)->side())){
 246                 return NOT_AN_ENEMY;
 247         }
 248         //@todo: (maybe) verify that the target hex contains the same unit that before,
 249         // comparing for example the unit ID
 250 
 251         return move::check_validity();
 252 }
