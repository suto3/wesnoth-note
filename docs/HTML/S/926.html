  67 void clean_saves(const std::string& label)
  68 {
  69         std::vector<save_info> games = get_saves_list();
  70         std::string prefix = label + "-" + _("Auto-Save");
  71         LOG_SAVE << "Cleaning saves with prefix '" << prefix << "'\n";
  72         for (std::vector<save_info>::iterator i = games.begin(); i != games.end(); ++i) {
  73                 if (i->name().compare(0, prefix.length(), prefix) == 0) {
  74                         LOG_SAVE << "Deleting savegame '" << i->name() << "'\n";
  75                         delete_game(i->name());
  76                 }

  87 bool loadgame::show_difficulty_dialog()
  88 {
  89         if(load_data_.summary["corrupt"].to_bool()) {

  92 
  93         std::string campaign_id = load_data_.summary["campaign"];
  94 
  95         for(const config &campaign : game_config_.child_range("campaign"))

  97                 if(campaign["id"] != campaign_id) {
  98                         continue;
  99                 }

 102                 difficulty_dlg.show(video_);

 109                 load_data_.difficulty = difficulty_dlg.selected_difficulty();

 111                 // Exit loop
 112                 break;

 118 // Called only by play_controller to handle in-game attempts to load. Instead of returning true,
 119 // throws a "load_game_exception" to signal a resulting load game request.
 120 bool loadgame::load_game_ingame()
 121 {
 122         if (video_.faked()) {
 123                 return false;
 124         }
 125 
 126         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 127                 return false;
 128         }
 129 
 130         if(load_data_.filename.empty()) {
 131                 return false;
 132         }
 133 
 134         if(load_data_.select_difficulty) {
 135                 if(!show_difficulty_dialog()) {
 136                         return false;
 137                 }
 138         }
 139 
 140         load_data_.show_replay |= is_replay_save(load_data_.summary);
 141 
 142         // Confirm the integrity of the file before throwing the exception.
 143         // Use the summary in the save_index for this.
 144         const config & summary = save_index_manager.get(load_data_.filename);
 145 
 146         if (summary["corrupt"].to_bool(false)) {
 147                 gui2::show_error_message(video_,
 148                                 _("The file you have tried to load is corrupt: '"));
 149                 return false;
 150         }
 151 
 152         if (!loadgame::check_version_compatibility(summary["version"].str(), video_)) {
 153                 return false;
 154         }
 155 
 156         throw load_game_exception(std::move(load_data_));
 157 }

 159 bool loadgame::load_game()
 160 {
 161         bool skip_version_check = true;
 162 
 163         if(load_data_.filename.empty()){
 164                 if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 165                         return false;
 166                 }
 167                 skip_version_check = false;
 168                 load_data_.show_replay |= is_replay_save(load_data_.summary);
 169         }
 170 
 171         if(load_data_.filename.empty()) {
 172                 return false;
 173         }
 174 
 175         if(load_data_.select_difficulty) {
 176                 if(!show_difficulty_dialog()) {
 177                         return false;
 178                 }
 179         }
 180 
 181         std::string error_log;
 182         read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 183 
 184         convert_old_saves(load_data_.load_config);
 185 
 186         if(!error_log.empty()) {
 187         try {
 188                     gui2::show_error_message(video_,
 189                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 190                                     error_log);
 191         } catch (utf8::invalid_utf8_exception&) {
 192                     gui2::show_error_message(video_,
 193                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 194                     std::string("(UTF-8 ERROR)"));
 195         }
 196         }
 197 
 198         if (!load_data_.difficulty.empty()){
 199                 load_data_.load_config["difficulty"] = load_data_.difficulty;
 200         }
 201         // read classification to for loading the game_config config object.
 202         gamestate_.classification() = game_classification(load_data_.load_config);
 203 
 204         if (skip_version_check) {
 205                 return true;
 206         }
 207 
 208         return check_version_compatibility();
 209 }

 216 bool loadgame::check_version_compatibility(const version_info & save_version, CVideo & video)
 217 {
 218         if (save_version == game_config::version) {
 219                 return true;
 220         }
 221 
 222         const version_info &wesnoth_version = game_config::wesnoth_version;
 223 
 224         // Even minor version numbers indicate stable releases which are
 225         // compatible with each other.
 226         if (wesnoth_version.minor_version() % 2 == 0 &&
 227             wesnoth_version.major_version() == save_version.major_version() &&
 228             wesnoth_version.minor_version() == save_version.minor_version())
 229         {
 230                 return true;
 231         }
 232 
 233         // Do not load if too old. If either the savegame or the current
 234         // game has the version 'test', load. This 'test' version is never
 235         // supposed to occur, except when Soliton is testing MP servers.
 236         if (save_version < game_config::min_savegame_version &&
 237             save_version != game_config::test_version &&
 238             wesnoth_version != game_config::test_version)
 239         {
 240                 const std::string message = _("This save is from an old, unsupported version ($version_number|) and cannot be loaded.");
 241                 utils::string_map symbols;
 242                 symbols["version_number"] = save_version.str();
 243                 gui2::show_error_message(video, utils::interpolate_variables_into_string(message, &symbols));
 244                 return false;
 245         }
 246 
 247         if(preferences::confirm_load_save_from_different_version()) {
 248                 const std::string message = _("This save is from a different version of the game ($version_number|). Do you wish to try to load it?");
 249                 utils::string_map symbols;
 250                 symbols["version_number"] = save_version.str();
 251                 const int res = gui2::show_message(video, _("Load Game"), utils::interpolate_variables_into_string(message, &symbols),
 252                         gui2::dialogs::message::yes_no_buttons);
 253                 return res == gui2::window::OK;
 254         }
 255 
 256         return true;
 257 }

 264 bool loadgame::load_multiplayer_game()
 265 {
 266         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 267                 return false;
 268         }
 269 
 270 
 271         load_data_.show_replay |= is_replay_save(load_data_.summary);
 272         if(load_data_.filename.empty()) {
 273                 return false;
 274         }
 275 
 276         // read_save_file needs to be called before we can verify the classification so the data has
 277         // been populated. Since we do that, we report any errors in that process first.
 278         std::string error_log;
 279         {
 280                 cursor::setter cur(cursor::WAIT);
 281                 log_scope("load_game");
 282 
 283                 read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 284                 copy_era(load_data_.load_config);
 285         }
 286 
 287         if(!error_log.empty()) {
 288                 gui2::show_error_message(video_,
 289                                 _("The file you have tried to load is corrupt: '") +
 290                                 error_log);
 291                 return false;
 292         }
 293 
 294         if(is_replay_save(load_data_.summary)) {
 295                 gui2::show_transient_message(video_, _("Load Game"), _("Replays are not supported in multiplayer mode."));
 296                 return false;
 297         }
 298 
 299         // We want to verify the game classification before setting the data, so we don't check on
 300         // gamestate_.classification() and instead construct a game_classification object manually.
 301         if(game_classification(load_data_.load_config).campaign_type != game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 302                 gui2::show_transient_error_message(video_, _("This is not a multiplayer save."));
 303                 return false;
 304         }
 305 
 306         set_gamestate();
 307 
 308         return check_version_compatibility();
 309 }

 436 bool savegame::save_game(CVideo* video, const std::string& filename)
 437 {
 438 
 439         try {
 440                 Uint32 start, end;
 441                 start = SDL_GetTicks();
 442 
 443                 if (filename_ == "")
 444                         filename_ = filename;
 445 
 446                 before_save();
 447 
 448                 write_game_to_disk(filename_);
 449                 if (resources::persist != nullptr) {
 450                         resources::persist->end_transaction();
 451                         resources::persist ->start_transaction();
 452                 }
 453 
 454                 end = SDL_GetTicks();
 455                 LOG_SAVE << "Milliseconds to save " << filename_ << ": " << end - start << std::endl;
 456 
 457                 if (video != nullptr && show_confirmation_)
 458                         gui2::show_transient_message(*video, _("Saved"), _("The game has been saved."));
 459                 return true;
 460         } catch(game::save_game_failed& e) {
 461                 ERR_SAVE << error_message_ << e.message << std::endl;
 462                 if (video != nullptr){
 463                         gui2::show_error_message(*video, error_message_ + e.message);
 464                         //do not bother retrying, since the user can just try to save the game again
 465                         //maybe show a yes-no dialog for "disable autosaves now"?
 466                 }
 467 
 468                 return false;
 469         };
 470 }

 472 void savegame::write_game_to_disk(const std::string& filename)
 473 {
 474         LOG_SAVE << "savegame::save_game" << std::endl;
 475 
 476         filename_ = filename;
 477         filename_ += compression::format_extension(compress_saves_);
 478 
 479         std::stringstream ss;
 480         {
 481                 config_writer out(ss, compress_saves_);
 482                 write_game(out);
 483                 finish_save_game(out);
 484         }
 485         filesystem::scoped_ostream os(open_save_game(filename_));
 486         (*os) << ss.str();
 487 
 488         if (!os->good()) {
 489                 throw game::save_game_failed(_("Could not write to file"));
 490         }
 491 }

 643 static void convert_old_saves_1_11_0(config& cfg)
 644 {
 645         if(!cfg.has_child("snapshot")){
 646                 return;
 647         }
 648 
 649         const config& snapshot = cfg.child("snapshot");
 650         const config& replay_start = cfg.child("replay_start");
 651         const config& replay = cfg.child("replay");
 652 
 653         if(!cfg.has_child("carryover_sides") && !cfg.has_child("carryover_sides_start")){
 654                 config carryover;
 655                 //copy rng and menu items from toplevel to new carryover_sides
 656                 carryover["random_seed"] = cfg["random_seed"];
 657                 carryover["random_calls"] = cfg["random_calls"];
 658                 for(const config& menu_item : cfg.child_range("menu_item")) {
 659                         carryover.add_child("menu_item", menu_item);
 660                 }
 661                 carryover["difficulty"] = cfg["difficulty"];
 662                 carryover["random_mode"] = cfg["random_mode"];
 663                 //the scenario to be played is always stored as next_scenario in carryover_sides_start
 664                 carryover["next_scenario"] = cfg["scenario"];
 665 
 666                 config carryover_start = carryover;
 667 
 668                 //copy sides from either snapshot or replay_start to new carryover_sides
 669                 if(!snapshot.empty()){
 670                         for(const config& side : snapshot.child_range("side")) {
 671                                 carryover.add_child("side", side);
 672                         }
 673                         //for compatibility with old savegames that use player instead of side
 674                         for(const config& side : snapshot.child_range("player")) {
 675                                 carryover.add_child("side", side);
 676                         }
 677                         //save the sides from replay_start in carryover_sides_start
 678                         for(const config& side : replay_start.child_range("side")) {
 679                                 carryover_start.add_child("side", side);
 680                         }
 681                         //for compatibility with old savegames that use player instead of side
 682                         for(const config& side : replay_start.child_range("player")) {
 683                                 carryover_start.add_child("side", side);
 684                         }
 685                 } else if (!replay_start.empty()){
 686                         for(const config& side : replay_start.child_range("side")) {
 687                                 carryover.add_child("side", side);
 688                                 carryover_start.add_child("side", side);
 689                         }
 690                         //for compatibility with old savegames that use player instead of side
 691                         for(const config& side : replay_start.child_range("player")) {
 692                                 carryover.add_child("side", side);
 693                                 carryover_start.add_child("side", side);
 694                         }
 695                 }
 696 
 697                 //get variables according to old hierarchy and copy them to new carryover_sides
 698                 if(!snapshot.empty()){
 699                         if(const config& variables_from_snapshot = snapshot.child("variables")){
 700                                 carryover.add_child("variables", variables_from_snapshot);
 701                                 carryover_start.add_child("variables", replay_start.child_or_empty("variables"));
 702                         } else if (const config& variables_from_cfg = cfg.child("variables")){
 703                                 carryover.add_child("variables", variables_from_cfg);
 704                                 carryover_start.add_child("variables", variables_from_cfg);
 705                         }
 706                 } else if (!replay_start.empty()){
 707                         if(const config& variables = replay_start.child("variables")){
 708                                 carryover.add_child("variables", variables);
 709                                 carryover_start.add_child("variables", variables);
 710                         }
 711                 } else {
 712                         carryover.add_child("variables", cfg.child("variables"));
 713                         carryover_start.add_child("variables", cfg.child("variables"));
 714                 }
 715 
 716                 cfg.add_child("carryover_sides", carryover);
 717                 cfg.add_child("carryover_sides_start", carryover_start);
 718         }
 719 
 720         //if replay and snapshot are empty we've got a start of scenario save and don't want replay_start either
 721         if(replay.empty() && snapshot.empty()){
 722                 LOG_RG<<"removing replay_start \n";
 723                 cfg.clear_children("replay_start");
 724         }
 725 
 726         //remove empty replay or snapshot so type of save can be detected more easily
 727         if(replay.empty()){
 728                 LOG_RG<<"removing replay \n";
 729                 cfg.clear_children("replay");
 730         }
 731 
 732         if(snapshot.empty()){
 733                 LOG_RG<<"removing snapshot \n";
 734                 cfg.clear_children("snapshot");
 735         }
 736 }

 738 static void convert_old_saves_1_13_0(config& cfg)
 739 {
 740         if(config& carryover_sides_start = cfg.child("carryover_sides_start"))
 741         {
 742                 if(!carryover_sides_start.has_attribute("next_underlying_unit_id"))
 743                 {
 744                         carryover_sides_start["next_underlying_unit_id"] = cfg["next_underlying_unit_id"];
 745                 }
 746         }
 747         if(cfg.child_or_empty("snapshot").empty())
 748         {
 749                 cfg.clear_children("snapshot");
 750         }
 751         if(cfg.child_or_empty("replay_start").empty())
 752         {
 753                 cfg.clear_children("replay_start");
 754         }
 755         if(config& snapshot = cfg.child("snapshot"))
 756         {
 757                 //make [end_level] -> [end_level_data] since its alo called [end_level_data] in the carryover.
 758                 if(config& end_level = cfg.child("end_level") )
 759                 {
 760                         snapshot.add_child("end_level_data", end_level);
 761                         snapshot.clear_children("end_level");
 762                 }
 763                 //if we have a snapshot then we already applied carryover so there is no reason to keep this data.
 764                 if(cfg.has_child("carryover_sides_start"))
 765                 {
 766                         cfg.clear_children("carryover_sides_start");
 767                 }
 768         }
 769         if(!cfg.has_child("snapshot") && !cfg.has_child("replay_start"))
 770         {
 771                 cfg.clear_children("carryover_sides");
 772         }
 773         //This code is needed because for example otherwise it won't find the (empty) era
 774         if(!cfg.has_child("multiplayer")) {
 775                 cfg.add_child("multiplayer", config {
 776                         "mp_era", "era_blank",
 777                         "show_connect", false,
 778                         "show_configure", false,
 779                         "mp_use_map_settings", true,
 780                 });
 781         }
 782 }

 786 static void convert_old_saves_1_13_1(config& cfg)
 787 {
 788         if(config& multiplayer = cfg.child("multiplayer")) {
 789                 if(multiplayer["mp_era"] == "era_blank") {

 792         }

 795         {
 796                 for(config& side : carryover_sides_start.child_range("side"))
 797                 {

 799                         {
 800                                 if(config& modifications = unit.child("modifications"))
 801                                 {
 802                                         for(config& advancement : modifications.child_range("advance"))
 803                                         {
 804                                                 modifications.add_child("advancement", advancement);
 805                                         }
 806                                         modifications.clear_children("advance");
 807                                 }
 808                         }
 809                 }
 810         }
 811         for(config& snapshot : cfg.child_range("snapshot")) {
 812                 if (snapshot.has_attribute("used_items")) {
 813                         config used_items;
 814                         for(const std::string& item : utils::split(snapshot["used_items"])) {
 815                                 used_items[item] = true;
 816                         }
 817                         snapshot.remove_attribute("used_items");
 818                         snapshot.add_child("used_items", used_items);
 819                 }
 820         }

 823 void convert_old_saves(config& cfg)
 824 {
 825         version_info loaded_version(cfg["version"]);

 827         {
 828                 convert_old_saves_1_11_0(cfg);
 829         }
 830         // '<= version_info("1.13.0")' doesn't work
 831         //because version_info cannot handle 1.13.0-dev versions correctly.

 833         {
 834                 convert_old_saves_1_13_0(cfg);
 835         }
 836         if(loaded_version <= version_info("1.13.1"))
 837         {
 838                 convert_old_saves_1_13_1(cfg);
 839         }
 840         LOG_RG<<"cfg after conversion "<<cfg<<"\n";
 841 }
