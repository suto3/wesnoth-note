  96 static ng::side_engine* create_side_engine(const config& defaults,
  97         test_connect_engine* connect_engine)
  98 {
  99         config side_cfg = connect_engine->current_config()->child("side");
 100         side_cfg.remove_attributes("faction");
 101         side_cfg.clear_children("default_faction");
 102         side_cfg.append(defaults);
 103 
 104         return new ng::side_engine(side_cfg, *connect_engine, 0);
 105 }

 325 BOOST_AUTO_TEST_CASE( flg_no_map_settings )
 326 {
 327         // Set up side_engine and its dependencies.
 328         state->mp_settings().use_map_settings = false;
 329         state->mp_settings().saved_game = false;
 330         const std::unique_ptr<test_connect_engine>
 331                 connect_engine(create_test_connect_engine());
 332         ng::side_engine_ptr side_engine;
 333         config side;
 334 
 335         // Recruit list with no faction.
 336         side.clear();
 337         side["recruit"] = "Elvish Archer";
 338         side_engine.reset(create_side_engine(side, connect_engine.get()));
 339         BOOST_CHECK( side_engine->flg().choosable_factions().size() >  1 );
 340         //BOOST_CHECK_EQUAL( side_engine->flg().current_faction()["id"], "Custom" );
 341 
 342         // Custom faction, no recruits.
 343         side.clear();
 344         side["faction"] = "Custom";
 345         side_engine.reset(create_side_engine(side, connect_engine.get()));
 346         BOOST_CHECK( side_engine->flg().choosable_factions().size() >  1 );
 347         BOOST_CHECK_EQUAL( side_engine->flg().current_faction()["id"], "Custom" );
 348         BOOST_CHECK_EQUAL( side_engine->new_config()["recruit"].empty(), true );
 349 
 350         // Carried over recruits.
 351         side.clear();
 352         side["previous_recruits"] = "Elvish Archer";
 353         side_engine.reset(create_side_engine(side, connect_engine.get()));
 354         BOOST_CHECK( side_engine->flg().choosable_factions().size() >  1 );
 355         BOOST_CHECK_EQUAL( side_engine->flg().current_faction()["id"], "Custom" );
 356         BOOST_CHECK_EQUAL( side_engine->new_config()["previous_recruits"],
 357                 "Elvish Archer" );
 358 
 359         // Explicit leader for faction with multiple leaders.
 360         side.clear();
 361         side["type"] = "Goblin Impaler";
 362         side_engine.reset(create_side_engine(side, connect_engine.get()));
 363         side_engine->flg().set_current_faction("Rebels");
 364         BOOST_CHECK( side_engine->flg().choosable_leaders().size() > 1 );
 365 
 366         // Duplicate leaders.
 367         side.clear();
 368         side["faction"] = "Custom";
 369         side["type"] = "Swordsman";
 370         side_engine.reset(create_side_engine(side, connect_engine.get()));
 371         BOOST_CHECK( side_engine->flg().choosable_leaders().size() > 1 );
 372         const std::vector<std::string>& leaders =
 373                 side_engine->flg().choosable_leaders();
 374         BOOST_CHECK_EQUAL( std::count(leaders.begin(), leaders.end(), "Swordsman"),
 375                 1 );
 376 
 377         // Explicit gender for unit with both genders available.
 378         side.clear();
 379         side["gender"] = "female";
 380         side_engine.reset(create_side_engine(side, connect_engine.get()));
 381         side_engine->flg().set_current_faction("Rebels");
 382         side_engine->flg().set_current_leader("Elvish Ranger");
 383         BOOST_CHECK_EQUAL( side_engine->flg().current_gender(), "random" );
 384 }
