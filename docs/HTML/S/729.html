 268 static std::pair<map_location , map_location> mirror_walk( std::pair<map_location,map_location> p, map_location::DIRECTION d)
 269 {
 270         p.first = p.first.get_direction(d);
 271         p.second = p.second.get_direction(map_location::get_opposite_dir(d));
 272         BOOST_CHECK_EQUAL(p.first, p.second.vector_negation());
 273         return p;
 274 }

 326 static map_location::DIRECTION legacy_get_opposite_dir(map_location::DIRECTION d)
 327 {
 328         switch (d) {
 329                 case map_location::NORTH:
 330                         return map_location::SOUTH;
 331                 case map_location::NORTH_EAST:
 332                         return map_location::SOUTH_WEST;
 333                 case map_location::SOUTH_EAST:
 334                         return map_location::NORTH_WEST;
 335                 case map_location::SOUTH:
 336                         return map_location::NORTH;
 337                 case map_location::SOUTH_WEST:
 338                         return map_location::NORTH_EAST;
 339                 case map_location::NORTH_WEST:
 340                         return map_location::SOUTH_EAST;
 341                 case map_location::NDIRECTIONS:
 342                 default:
 343                         return map_location::NDIRECTIONS;
 344         }
 345 }

 383 static void rotate_around_centers ( const std::vector<map_location> & locs )
 384 {
 385         for (std::vector<map_location>::const_iterator it_a = locs.begin(); it_a != locs.end(); ++it_a) {
 386                 for (std::vector<map_location>::const_iterator it_b = it_a + 1; it_b != locs.end(); ++it_b) {
 387                         const map_location & a = *it_a;
 388                         const map_location & b = *it_b;
 389 
 390                         a.rotate_right_around_center(b,1);
 391                         a.rotate_right_around_center(b,-1);
 392                         a.rotate_right_around_center(b,2);
 393                         a.rotate_right_around_center(b,-2);
 394                         a.rotate_right_around_center(b,3);
 395                         a.rotate_right_around_center(b,0);
 396                 }
 397         }
 398 }
