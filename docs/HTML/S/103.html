 210 /// Stores all information related to functions that can be bound to hotkeys.
 211 /// this is currently a semi struct: it haves a constructor, but only const-public members.
 212 struct hotkey_command {
 213 public:
 214         /// the compiler want me to make a default constructor
 215         /// since most member are const, calling the default constructor is normally no use.
 216         hotkey_command();
 217         hotkey_command(HOTKEY_COMMAND cmd, const std::string& id, const t_string& desc, bool hidden, hk_scopes scope, HOTKEY_CATEGORY category, const t_string& tooltip);
 218         /// the names are strange: the "hotkey::HOTKEY_COMMAND" is named id, and the string to identify the object is called "command"
 219         /// there is some inconstancy with that names in this file.
 220         /// This binds the command to a function. Does not need to be unique.
 221         const HOTKEY_COMMAND id;
 222         /// The command is unique.
 223         const std::string command;
 224         // since the wml_menu hotkey_command s can have different textdomains we need t_string now.
 225         const t_string description;
 226         /// If hidden then don't show the command in the hotkey preferences.
 227         const bool hidden;
 228         /// The visibility scope of the command.
 229         const hk_scopes scope;
 230         /// The category of the command.
 231         const HOTKEY_CATEGORY category;
 232 
 233         const t_string tooltip;
 234 
 235         /// checks weather this is the null hotkey_command
 236         bool null() const;
 237         /// returns the command that is treated as null
 238         static const hotkey_command& null_command();
 239         /// the execute_command argument was changed from HOTKEY_COMMAND to hotkey_command,
 240         /// to be able to call it with HOTKEY_COMMAND, this function was created
 241         static const hotkey_command& get_command_by_command(HOTKEY_COMMAND command);
 242 };
