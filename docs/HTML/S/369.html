  75 void mp_staging::pre_show(window& window)
  76 {
  77         window.set_enter_disabled(true);
  78 
  79         //
  80         // Set title and status widget states
  81         //
  82         label& title = find_widget<label>(&window, "title", false);
  83         title.set_label((formatter() << title.get_label() << " " << font::unicode_em_dash << " " << connect_engine_.scenario()["name"].t_str()).str());
  84 
  85         update_status_label_and_buttons(window);
  86 
  87         //
  88         // Set up sides list
  89         //
  90         for(const auto& side : connect_engine_.side_engines()) {
  91                 if(!side->allow_player() && !game_config::debug) {
  92                         continue;
  93                 }
  94 
  95                 add_side_node(window, side);
  96         }
  97 
  98         //
  99         // Initialize chatbox and game rooms
 100         //
 101         chatbox& chat = find_widget<chatbox>(&window, "chat", false);
 102 
 103         chat.set_lobby_info(lobby_info_);
 104 
 105         if(wesnothd_connection_) {
 106                 chat.set_wesnothd_connection(*wesnothd_connection_);
 107         }
 108 
 109         chat.room_window_open("this game", true, false); // TODO: better title?
 110         chat.active_window_changed();
 111 
 112         //
 113         // Set up player list
 114         //
 115         update_player_list(window);
 116 
 117         //
 118         // Set up the network handling
 119         //
 120         update_timer_ = add_timer(game_config::lobby_network_timer, std::bind(&mp_staging::network_handler, this, std::ref(window)), true);
 121 
 122         network_handler(window);
 123 
 124         //
 125         // Set up the Lua plugin context
 126         //
 127         plugins_context_.reset(new plugins_context("Multiplayer Staging"));
 128 
 129         plugins_context_->set_callback("launch", [&window](const config&) { window.set_retval(window::OK); }, false);
 130         plugins_context_->set_callback("quit",   [&window](const config&) { window.set_retval(window::CANCEL); }, false);
 131         plugins_context_->set_callback("chat",   [&chat](const config& cfg) { chat.send_chat_message(cfg["message"], false); }, true);
 132 }

 134 void mp_staging::add_side_node(window& window, ng::side_engine_ptr side)
 135 {
 136         tree_view& tree = find_widget<tree_view>(&window, "side_list", false);
 137         static const std::map<std::string, string_map> empty_map;
 138 
 139         std::map<std::string, string_map> data;
 140         string_map item;
 141 
 142         item["label"] = std::to_string(side->index() + 1);
 143         data.emplace("side_number", item);
 144 
 145         // TODO: don't hardcode magenta?
 146         item["label"] = "units/unknown-unit.png~RC(magenta>" + std::to_string(side->color() + 1) + ")";
 147         data.emplace("leader_image", item);
 148 
 149         item["label"] = "icons/icon-random.png";
 150         data.emplace("leader_gender", item);
 151 
 152         // Check to see whether we've added a toplevel tree node for this team. If not, add one
 153         if(team_tree_map_.find(side->team_name()) == team_tree_map_.end()) {
 154                 std::map<std::string, string_map> tree_data;
 155                 string_map tree_item;
 156 
 157                 tree_item["label"] = (formatter() << _("Team:") << " " << side->user_team_name()).str();
 158                 tree_data.emplace("tree_view_node_label", tree_item);
 159 
 160                 tree_view_node& team_node = tree.add_node("team_header", tree_data);
 161                 team_node.add_sibling("side_spacer", empty_map);
 162 
 163                 team_tree_map_[side->team_name()] = &team_node;
 164         }
 165 
 166         // Find an appropriate position to insert this node. This ensures the side nodes are always
 167         // arranged by descending index order in each team group.
 168         int position = 0;
 169         for(const auto& side_engine : connect_engine_.side_engines()) {
 170                 if(side->team() == side_engine->team() && side->index() > side_engine->index()) {
 171                         ++position;
 172                 }
 173         }
 174 
 175         // Must be *after* the above if block, or the node ptr could be invalid
 176         tree_view_node& node = team_tree_map_[side->team_name()]->add_child("side_panel", data, position);
 177 
 178         side_tree_map_[side] = &node;
 179 
 180         grid& row_grid = node.get_grid();
 181 
 182         update_leader_display(side, row_grid);
 183 
 184         // Status variables
 185         const bool fls = connect_engine_.force_lock_settings();
 186         const bool ums = connect_engine_.params().use_map_settings;
 187 
 188         const bool lock_gold   = side->cfg()["gold_lock"].to_bool(fls);
 189         const bool lock_income = side->cfg()["income_lock"].to_bool(fls);
 190         const bool lock_team   = side->cfg()["team_lock"].to_bool(fls);
 191         const bool lock_color  = side->cfg()["color_lock"].to_bool(fls);
 192 
 193         const bool saved_game = connect_engine_.params().saved_game;
 194 
 195         //
 196         // AI Algorithm
 197         //
 198         int selection = 0;
 199 
 200         // We use an index-based loop in order to get the index of the selected option
 201         std::vector<config> ai_options;
 202         for(unsigned i = 0; i < ai_algorithms_.size(); i++) {
 203                 ai_options.emplace_back(config {"label", ai_algorithms_[i]->text});
 204 
 205                 if(ai_algorithms_[i]->id == side->ai_algorithm()) {
 206                         selection = i;
 207                 }
 208         }
 209 
 210         menu_button& ai_selection = find_widget<menu_button>(&row_grid, "ai_controller", false);
 211 
 212         ai_selection.set_values(ai_options, selection);
 213         ai_selection.connect_click_handler(std::bind(&mp_staging::on_ai_select, this, side, std::ref(ai_selection)));
 214 
 215         on_ai_select(side, ai_selection);
 216 
 217         //
 218         // Controller
 219         //
 220         std::vector<config> controller_names;
 221         for(const auto& controller : side->controller_options()) {
 222                 controller_names.emplace_back(config {"label", controller.second});
 223         }
 224 
 225         menu_button& controller_selection = find_widget<menu_button>(&row_grid, "controller", false);
 226 
 227         controller_selection.set_values(controller_names, side->current_controller_index());
 228         controller_selection.set_active(controller_names.size() > 1);
 229         controller_selection.connect_click_handler(std::bind(&mp_staging::on_controller_select, this, side, std::ref(row_grid)));
 230 
 231         on_controller_select(side, row_grid);
 232 
 233         //
 234         // Leader controls
 235         //
 236         button& leader_select = find_widget<button>(&row_grid, "select_leader", false);
 237 
 238         leader_select.set_active(!saved_game);
 239 
 240         connect_signal_mouse_left_click(leader_select,
 241                 std::bind(&mp_staging::select_leader_callback, this, std::ref(window), side, std::ref(row_grid)));
 242 
 243         //
 244         // Team
 245         //
 246         std::vector<config> team_names;
 247         for(const auto& team : side->player_teams()) {
 248                 team_names.emplace_back(config {"label", team});
 249         }
 250 
 251         menu_button& team_selection = find_widget<menu_button>(&row_grid, "side_team", false);
 252 
 253         // HACK: side->team() does not get its index from side->player_teams(), but rather side->team_names().
 254         // As such, the index is off if there is only 1 playable team. This is a hack to make sure the menu_button
 255         // widget doesn't assert with the invalid initial selection. The connect_engine should be fixed once the GUI1
 256         // dialog is dropped
 257         team_selection.set_values(team_names, std::min<int>(team_names.size() - 1, side->team()));
 258         team_selection.set_active(!saved_game);
 259         team_selection.connect_click_handler(std::bind(&mp_staging::on_team_select, this, std::ref(window), side, std::ref(team_selection), _3, _4));
 260 
 261         //
 262         // Colors
 263         //
 264         std::vector<config> color_options;
 265         for(const auto& color : side->color_options()) {
 266                 color_options.emplace_back(config {
 267                         "label", font::get_color_string_pango(color),
 268                         "icon", (formatter() << "misc/status.png~RC(magenta>" << color << ")").str(),
 269                 });
 270         }
 271 
 272         menu_button& color_selection = find_widget<menu_button>(&row_grid, "side_color", false);
 273 
 274         color_selection.set_values(color_options, side->color());
 275         color_selection.set_active(!saved_game);
 276         color_selection.set_use_markup(true);
 277         color_selection.connect_click_handler(std::bind(&mp_staging::on_color_select, this, side, std::ref(row_grid)));
 278 
 279         //
 280         // Gold and Income
 281         //
 282         slider& slider_gold = find_widget<slider>(&row_grid, "side_gold_slider", false);
 283         slider_gold.set_value(side->cfg()["gold"].to_int(100));
 284 
 285         connect_signal_notify_modified(slider_gold, std::bind(
 286                 &mp_staging::on_side_slider_change<&ng::side_engine::set_gold>, this, side, std::ref(slider_gold)));
 287 
 288         slider& slider_income = find_widget<slider>(&row_grid, "side_income_slider", false);
 289         slider_income.set_value(side->cfg()["income"]);
 290 
 291         connect_signal_notify_modified(slider_income, std::bind(
 292                 &mp_staging::on_side_slider_change<&ng::side_engine::set_income>, this, side, std::ref(slider_income)));
 293 
 294         // TODO: maybe display the saved values
 295         if(saved_game) {
 296                 slider_gold.set_visible(widget::visibility::invisible);
 297                 slider_income.set_visible(widget::visibility::invisible);
 298         }
 299 
 300         //
 301         // Gold, income, team, and color are only suggestions unless explicitly locked
 302         //
 303         if(!saved_game && ums) {
 304                 team_selection.set_active(!lock_team);
 305                 color_selection.set_active(!lock_color);
 306 
 307                 slider_gold.set_active(!lock_gold);
 308                 slider_income.set_active(!lock_income);
 309         }
 310 }

 312 void mp_staging::update_player_list(window& window)
 313 {
 314         listbox& player_list = find_widget<listbox>(&window, "player_list", false);
 315 
 316         player_list.clear();
 317 
 318         for(const auto& player : connect_engine_.connected_users()) {
 319                 std::map<std::string, string_map> data;
 320                 string_map item;
 321 
 322                 item["label"] = player;

 324 
 325                 player_list.add_row(data);

 357 void mp_staging::on_team_select(window& window, ng::side_engine_ptr side, menu_button& team_menu, bool& handled, bool& halt)
 358 {
 359         if(static_cast<int>(team_menu.get_value()) == side->team()) {
 360                 return;
 361         }
 362 
 363         side->set_team(team_menu.get_value());
 364 
 365         // First, remove the node from the tree
 366         find_widget<tree_view>(&window, "side_list", false).remove_node(side_tree_map_[side]);
 367 
 368         // Then add a new node as a child to the appropriate team's node
 369         add_side_node(window, side);
 370 
 371         set_state_changed();
 372 
 373         handled = true;
 374         halt = true;
 375 }

 397 void mp_staging::update_leader_display(ng::side_engine_ptr side, grid& row_grid)
 398 {
 399         const std::string current_faction = (*side->flg().choosable_factions()[side->flg().current_faction_index()])["name"];
 400 
 401         // BIG FAT TODO: get rid of this shitty "null" string value in the FLG manager
 402         std::string current_leader = side->flg().current_leader() != "null" ? side->flg().current_leader() : font::unicode_em_dash;
 403         const std::string current_gender = side->flg().current_gender() != "null" ? side->flg().current_gender() : font::unicode_em_dash;
 404 
 405         // Sprite
 406         std::string new_image;
 407 
 408         if(side->flg().is_random_faction() || current_leader == "random") {
 409                 new_image = ng::random_enemy_picture;
 410         }
 411 
 412         if(const unit_type* ut = unit_types.find(current_leader)) {
 413                 const unit_type& type = ut->get_gender_unit_type(current_gender);
 414 
 415                 new_image = formatter() << type.image() << "~RC(magenta>" << side->color_id() << ")";
 416 
 417                 // We don't need the unit type id anymore, and can now replace this variable with the type name
 418                 current_leader = type.type_name();
 419         }
 420 
 421         find_widget<image>(&row_grid, "leader_image", false).set_label(new_image);
 422 
 423         // Faction and leader
 424         if(!side->cfg()["name"].empty()) {
 425                 current_leader = formatter() << side->cfg()["name"] << " (<i>" << current_leader << "</i>)";
 426         }
 427 
 428         find_widget<label>(&row_grid, "leader_type", false).set_label(current_leader);
 429         find_widget<label>(&row_grid, "leader_faction", false).set_label("<span color='#a69275'>" + current_faction + "</span>");
 430 
 431         // Gender
 432         if(current_gender != font::unicode_em_dash) {
 433                 const std::string gender_icon = formatter() << "icons/icon-" << current_gender << ".png";
 434 
 435                 image& icon = find_widget<image>(&row_grid, "leader_gender", false);
 436 
 437                 icon.set_label(gender_icon);
 438                 icon.set_tooltip(current_gender);
 439         }
 440 }

 453 void mp_staging::network_handler(window& window)
 454 {
 455         // First, send off any changes if they've been accumulated
 456         if(state_changed_) {
 457                 connect_engine_.update_and_send_diff();
 458         }
 459 
 460         // Next, check for any incoming changes
 461         config data;
 462         if(!state_changed_ && (!wesnothd_connection_ || !wesnothd_connection_->receive_data(data))) {
 463                 return;
 464         }
 465 
 466         // Update chat
 467         find_widget<chatbox>(&window, "chat", false).process_network_data(data);
 468 
 469         // TODO: why is this needed...
 470         const bool was_able_to_start = connect_engine_.can_start_game();
 471 
 472         bool quit_signal_received;
 473         std::tie(quit_signal_received, std::ignore) = connect_engine_.process_network_data(data);
 474 
 475         if(quit_signal_received) {
 476                 window.set_retval(window::CANCEL);
 477         }
 478 
 479         // Update side leader displays
 480         // This is basically only needed when a new player joins and selects their faction
 481         for(auto& tree_entry : side_tree_map_) {
 482                 ng::side_engine_ptr side = tree_entry.first;
 483 
 484                 grid& row_grid = tree_entry.second->get_grid();
 485 
 486                 update_leader_display(side, row_grid);
 487 
 488                 std::vector<config> controller_names;
 489                 for(const auto& controller : side->controller_options()) {
 490                         controller_names.emplace_back(config {"label", controller.second});
 491                 }
 492 
 493                 menu_button& controller_selection = find_widget<menu_button>(&row_grid, "controller", false);
 494 
 495                 controller_selection.set_values(controller_names, side->current_controller_index());
 496                 controller_selection.set_active(controller_names.size() > 1);
 497         }
 498 
 499         // Update player list
 500         // TODO: optimally, it wouldn't regenerate the entire list every single refresh cycle
 501         update_player_list(window);
 502 
 503         // Update status label and buttons
 504         update_status_label_and_buttons(window);
 505 
 506         if(!was_able_to_start && connect_engine_.can_start_game()) {
 507                 mp_ui_alerts::ready_for_start();
 508         }
 509 
 510         state_changed_ = false;
 511 }

 513 void mp_staging::post_show(window& window)
 514 {
 515         if(update_timer_ != 0) {
 516                 remove_timer(update_timer_);
 517                 update_timer_ = 0;
 518         }
 519 
 520         if(window.get_retval() == window::OK) {
 521                 connect_engine_.start_game();
 522         } else {
 523                 connect_engine_.leave_game();
 524         }
 525 }
