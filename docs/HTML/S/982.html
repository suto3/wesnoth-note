  35 bool is_neutral(const surface& surf)
  36 {
  37         return (surf->format->BytesPerPixel == 4 &&
  38                         surf->format->Rmask == SDL_RED_MASK &&
  39                         (surf->format->Amask | SDL_ALPHA_MASK) == SDL_ALPHA_MASK);
  40 }

  88 surface stretch_surface_horizontal(
  89                 const surface& surf, const unsigned w)
  90 {
  91         // Since SDL version 1.1.5 0 is transparent, before 255 was transparent.
  92         assert(SDL_ALPHA_TRANSPARENT==0);
  93 
  94         if(surf == nullptr)
  95                 return nullptr;
  96 
  97         if(static_cast<int>(w) == surf->w) {
  98                 return surf;
  99         }
 100         assert(w > 0);
 101 
 102         surface dst(create_neutral_surface(w, surf->h));
 103 
 104         surface src(make_neutral_surface(surf));
 105         // Now both surfaces are always in the "neutral" pixel format
 106 
 107         if(src == nullptr || dst == nullptr) {
 108                 std::cerr << "Could not create surface to scale onto\n";
 109                 return nullptr;
 110         }
 111 
 112         {
 113                 // Extra scoping used for the surface_lock.
 114                 const_surface_lock src_lock(src);
 115                 surface_lock dst_lock(dst);
 116 
 117                 const Uint32* const src_pixels = src_lock.pixels();
 118                 Uint32* dst_pixels = dst_lock.pixels();
 119 
 120                 for(unsigned y = 0; y < static_cast<unsigned>(src->h); ++y) {
 121                         const Uint32 pixel = src_pixels [y * src->w];
 122                         for(unsigned x = 0; x < w; ++x) {
 123 
 124                                 *dst_pixels++ = pixel;
 125 
 126                         }
 127                 }
 128         }
 129 
 130         return dst;

 254 // NOTE: Don't pass this function 0 scaling arguments.
 255 surface scale_surface(const surface &surf, int w, int h)
 256 {
 257         // Since SDL version 1.1.5 0 is transparent, before 255 was transparent.
 258         assert(SDL_ALPHA_TRANSPARENT==0);
 259 
 260         if(surf == nullptr)
 261                 return nullptr;
 262 
 263         if(w == surf->w && h == surf->h) {
 264                 return surf;
 265         }
 266         assert(w >= 0);
 267         assert(h >= 0);
 268 
 269         surface dst(create_neutral_surface(w,h));
 270 
 271         if (w == 0 || h ==0) {
 272                 std::cerr << "Create an empty image\n";
 273                 return dst;
 274         }
 275 
 276         surface src(make_neutral_surface(surf));
 277         // Now both surfaces are always in the "neutral" pixel format
 278 
 279         if(src == nullptr || dst == nullptr) {
 280                 std::cerr << "Could not create surface to scale onto\n";
 281                 return nullptr;
 282         }
 283 
 284         {
 285                 const_surface_lock src_lock(src);
 286                 surface_lock dst_lock(dst);
 287 
 288                 const Uint32* const src_pixels = src_lock.pixels();
 289                 Uint32* const dst_pixels = dst_lock.pixels();
 290 
 291                 fixed_t xratio = fxpdiv(surf->w,w);
 292                 fixed_t yratio = fxpdiv(surf->h,h);
 293 
 294                 fixed_t ysrc = ftofxp(0.0);
 295                 for(int ydst = 0; ydst != h; ++ydst, ysrc += yratio) {
 296                         fixed_t xsrc = ftofxp(0.0);
 297                         for(int xdst = 0; xdst != w; ++xdst, xsrc += xratio) {
 298                                 const int xsrcint = fxptoi(xsrc);
 299                                 const int ysrcint = fxptoi(ysrc);
 300 
 301                                 const Uint32* const src_word = src_pixels + ysrcint*src->w + xsrcint;
 302                                 Uint32* const dst_word = dst_pixels +    ydst*dst->w + xdst;
 303                                 const int dx = (xsrcint + 1 < src->w) ? 1 : 0;
 304                                 const int dy = (ysrcint + 1 < src->h) ? src->w : 0;
 305 
 306                                 Uint8 r,g,b,a;
 307                                 Uint32 rr,gg,bb,aa, temp;
 308 
 309                                 Uint32 pix[4], bilin[4];
 310 
 311                                 // This next part is the fixed point
 312                                 // equivalent of "take everything to
 313                                 // the right of the decimal point."
 314                                 // These fundamental weights decide
 315                                 // the contributions from various
 316                                 // input pixels. The labels assume
 317                                 // that the upper left corner of the
 318                                 // screen ("northeast") is 0,0 but the
 319                                 // code should still be consistent if
 320                                 // the graphics origin is actually
 321                                 // somewhere else.
 322                                 //
 323                                 // That is, the bilin array holds the
 324                                 // "geometric" weights. I.E. If I'm scaling
 325                                 // a 2 x 2 block a 10 x 10 block, then for
 326                                 // pixel (2,2) of ouptut, the upper left
 327                                 // pixel should be 10:1 more influential than
 328                                 // the upper right, and also 10:1 more influential
 329                                 // than lower left, and 100:1 more influential
 330                                 // than lower right.
 331 
 332                                 const fixed_t e = 0x000000FF & xsrc;
 333                                 const fixed_t s = 0x000000FF & ysrc;
 334                                 const fixed_t n = 0xFF - s;
 335                                 // Not called "w" to avoid hiding a function parameter
 336                                 // (would cause a compiler warning in MSVC2015 with /W4)
 337                                 const fixed_t we = 0xFF - e;
 338 
 339                                 pix[0] = *src_word;              // northwest
 340                                 pix[1] = *(src_word + dx);       // northeast
 341                                 pix[2] = *(src_word + dy);       // southwest
 342                                 pix[3] = *(src_word + dx + dy);  // southeast
 343 
 344                                 bilin[0] = n*we;
 345                                 bilin[1] = n*e;
 346                                 bilin[2] = s*we;
 347                                 bilin[3] = s*e;
 348 
 349                                 int loc;
 350                                 rr = bb = gg = aa = 0;
 351                                 for (loc=0; loc<4; loc++) {
 352                                   a = pix[loc] >> 24;
 353                                   r = pix[loc] >> 16;
 354                                   g = pix[loc] >> 8;
 355                                   b = pix[loc] >> 0;
 356 
 357                                   //We also have to implement weighting by alpha for the RGB components
 358                                   //If a unit has some parts solid and some parts translucent,
 359                                   //i.e. a red cloak but a dark shadow, then when we scale in
 360                                   //the shadow shouldn't appear to become red at the edges.
 361                                   //This part also smoothly interpolates between alpha=0 being
 362                                   //transparent and having no contribution, vs being opaque.
 363                                   temp = (a * bilin[loc]);
 364                                   rr += r * temp;
 365                                   gg += g * temp;
 366                                   bb += b * temp;
 367                                   aa += temp;
 368                                 }
 369 
 370                                 a = aa >> (16); // we average the alphas, they don't get weighted by any other factor besides bilin
 371                                 if (a != 0) {
 372                                         rr /= a;        // finish alpha weighting: divide by sum of alphas
 373                                         gg /= a;
 374                                         bb /= a;
 375                                 }
 376                                 r = rr >> (16); // now shift over by 16 for the bilin part
 377                                 g = gg >> (16);
 378                                 b = bb >> (16);
 379                                 *dst_word = (a << 24) + (r << 16) + (g << 8) + b;
 380                         }
 381                 }
 382         }
 383 
 384         return dst;
 385 }

 390         assert(SDL_ALPHA_TRANSPARENT==0);

 399         assert(h >= 0);
 400

 402 
 403         surface src(make_neutral_surface(surf));
 404         // Now both surfaces are always in the "neutral" pixel format
 405 
 406         if(src == NULL || dst == NULL) {
 407                 std::cerr << "Could not create surface to scale onto\n";
 408                 return NULL;
 409         }
 410 
 411         {
 412                 const_surface_lock src_lock(src);
 413                 surface_lock dst_lock(dst);
 414 
 415                 const Uint32* const src_pixels = src_lock.pixels();
 416                 Uint32* const dst_pixels = dst_lock.pixels();
 417 
 418                 fixed_t xratio = fxpdiv(surf->w,w);
 419                 fixed_t yratio = fxpdiv(surf->h,h);
 420 
 421                 fixed_t ysrc = ftofxp(0.0);
 422                 for(int ydst = 0; ydst != h; ++ydst, ysrc += yratio) {
 423                         fixed_t xsrc = ftofxp(0.0);
 424                         for(int xdst = 0; xdst != w; ++xdst, xsrc += xratio) {
 425                                 const int xsrcint = fxptoi(xsrc);
 426                                 const int ysrcint = fxptoi(ysrc);
 427 
 428                                 const Uint32* const src_word = src_pixels + ysrcint*src->w + xsrcint;
 429                                 Uint32* const dst_word = dst_pixels +    ydst*dst->w + xdst;
 430                                 const int dx = (xsrcint + 1 < src->w) ? 1 : 0;
 431                                 const int dy = (ysrcint + 1 < src->h) ? src->w : 0;
 432 
 433                                 Uint8 r,g,b,a;
 434                                 Uint32 rr,gg,bb,aa;
 435                                 Uint16 avg_r, avg_g, avg_b, avg_a;
 436                                 Uint32 pix[4], bilin[4];
 437 
 438                                 // This next part is the fixed point
 439                                 // equivalent of "take everything to
 440                                 // the right of the decimal point."
 441                                 // These fundamental weights decide
 442                                 // the contributions from various
 443                                 // input pixels. The labels assume
 444                                 // that the upper left corner of the
 445                                 // screen ("northeast") is 0,0 but the
 446                                 // code should still be consistent if
 447                                 // the graphics origin is actually
 448                                 // somewhere else.
 449 
 450                                 const fixed_t e = 0x000000FF & xsrc;
 451                                 const fixed_t s = 0x000000FF & ysrc;
 452                                 const fixed_t n = 0xFF - s;
 453                                 const fixed_t w = 0xFF - e;
 454 
 455                                 pix[0] = *src_word;              // northwest
 456                                 pix[1] = *(src_word + dx);       // northeast
 457                                 pix[2] = *(src_word + dy);       // southwest
 458                                 pix[3] = *(src_word + dx + dy);  // southeast
 459 
 460                                 bilin[0] = n*w;
 461                                 bilin[1] = n*e;
 462                                 bilin[2] = s*w;
 463                                 bilin[3] = s*e;
 464 
 465                                 // Scope out the neighboorhood, see
 466                                 // what the pixel values are like.
 467 
 468                                 int count = 0;
 469                                 avg_r = avg_g = avg_b = avg_a = 0;
 470                                 int loc;
 471                                 for (loc=0; loc<4; loc++) {
 472                                   a = pix[loc] >> 24;
 473                                   r = pix[loc] >> 16;
 474                                   g = pix[loc] >> 8;
 475                                   b = pix[loc] >> 0;
 476                                   if (a != 0) {
 477                                     avg_r += r;
 478                                     avg_g += g;
 479                                     avg_b += b;
 480                                     avg_a += a;
 481                                     count++;
 482                                   }
 483                                 }
 484                                 if (count>0) {
 485                                   avg_r /= count;
 486                                   avg_b /= count;
 487                                   avg_g /= count;
 488                                   avg_a /= count;
 489                                 }
 490 
 491                                 // Perform modified bilinear interpolation.
 492                                 // Don't trust any color information from
 493                                 // an RGBA sample when the alpha channel
 494                                 // is set to fully transparent.
 495                                 //
 496                                 // Some of the input images are hex tiles,
 497                                 // created using a hexagon shaped alpha channel
 498                                 // that is either set to full-on or full-off.
 499 
 500                                 rr = gg = bb = aa = 0;
 501                                 for (loc=0; loc<4; loc++) {
 502                                   a = pix[loc] >> 24;
 503                                   r = pix[loc] >> 16;
 504                                   g = pix[loc] >> 8;
 505                                   b = pix[loc] >> 0;
 506                                   if (a == 0) {
 507                                     r = static_cast<Uint8>(avg_r);
 508                                     g = static_cast<Uint8>(avg_g);
 509                                     b = static_cast<Uint8>(avg_b);
 510                                   }
 511                                   rr += r * bilin[loc];
 512                                   gg += g * bilin[loc];
 513                                   bb += b * bilin[loc];
 514                                   aa += a * bilin[loc];
 515                                 }
 516                                 r = rr >> 16;
 517                                 g = gg >> 16;
 518                                 b = bb >> 16;
 519                                 a = aa >> 16;
 520                                 *dst_word = (a << 24) + (r << 16) + (g << 8) + b;
 521                         }
 522                 }
 523         }
 524 
 525         return dst;
 526 }

 529 surface scale_surface_sharp(const surface& surf, int w, int h)
 530 {
 531         // Since SDL version 1.1.5 0 is transparent, before 255 was transparent.
 532         assert(SDL_ALPHA_TRANSPARENT==0);
 533 
 534         if(surf == nullptr)
 535                 return nullptr;
 536 
 537         if(w == surf->w && h == surf->h) {
 538                 return surf;
 539         }
 540         assert(w >= 0);
 541         assert(h >= 0);
 542 
 543         surface dst(create_neutral_surface(w,h));
 544 
 545         if (w == 0 || h ==0) {
 546                 std::cerr << "Create an empty image\n";
 547                 return dst;
 548         }
 549 
 550         surface src(make_neutral_surface(surf));
 551         // Now both surfaces are always in the "neutral" pixel format
 552 
 553         if(src == nullptr || dst == nullptr) {
 554                 std::cerr << "Could not create surface to scale onto\n";
 555                 return nullptr;
 556         }
 557 
 558         {
 559                 const_surface_lock src_lock(src);
 560                 surface_lock dst_lock(dst);
 561 
 562                 const Uint32* const src_pixels = src_lock.pixels();
 563                 Uint32* const dst_pixels = dst_lock.pixels();
 564 
 565                 float xratio = static_cast<float>(surf->w) / w;
 566                 float yratio = static_cast<float>(surf->h) / h;
 567 
 568                 float ysrc = 0.0f;
 569                 for(int ydst = 0; ydst != h; ++ydst, ysrc += yratio) {
 570                         float xsrc = 0.0f;
 571                         for(int xdst = 0; xdst != w; ++xdst, xsrc += xratio) {
 572                                 float red = 0.0f, green = 0.0f, blue = 0.0f, alpha = 0.0f;
 573 
 574                                 float summation = 0.0f;
 575 
 576                                 // We now have a rectangle, (xsrc,ysrc,xratio,yratio)
 577                                 // which we want to derive the pixel from
 578                                 for(float xloc = xsrc; xloc < xsrc+xratio; xloc += 1) {
 579                                         const float xsize = std::min<float>(std::floor(xloc+1)-xloc,xsrc+xratio-xloc);
 580 
 581                                         for(float yloc = ysrc; yloc < ysrc+yratio; yloc += 1) {
 582                                                 const int xsrcint = std::max<int>(0,std::min<int>(src->w-1,static_cast<int>(xsrc)));
 583                                                 const int ysrcint = std::max<int>(0,std::min<int>(src->h-1,static_cast<int>(ysrc)));
 584                                                 const float ysize = std::min<float>(std::floor(yloc+1)-yloc,ysrc+yratio-yloc);
 585 
 586                                                 Uint8 r,g,b,a;
 587 
 588                                                 SDL_GetRGBA(src_pixels[ysrcint*src->w + xsrcint],src->format,&r,&g,&b,&a);
 589                                                 float value = xsize * ysize;
 590                                                 summation += value;
 591                                                 if (!a) continue;
 592                                                 value *= a;
 593                                                 alpha += value;
 594                                                 red += r * value;
 595                                                 green += g * value;
 596                                                 blue += b * value;
 597                                         }
 598                                 }
 599 
 600                                 if (alpha != 0) {
 601                                         red = red / alpha + 0.5f;
 602                                         green = green / alpha + 0.5f;
 603                                         blue = blue / alpha + 0.5f;
 604                                         alpha = alpha / summation + 0.5f;
 605                                 }
 606 
 607                                 dst_pixels[ydst*dst->w + xdst] = SDL_MapRGBA(
 608                                 dst->format
 609                                 , static_cast<uint8_t>(red)
 610                                 , static_cast<uint8_t>(green)
 611                                 , static_cast<uint8_t>(blue)
 612                                 , static_cast<uint8_t>(alpha));
 613                         }
 614 
 615                 }
 616         }
 617 
 618         return dst;
 619 }

1805 /* Simplified RotSprite algorithm.
1806  * http://en.wikipedia.org/wiki/Image_scaling#RotSprite
1807  * Lifted from: http://github.com/salmonmoose/SpriteRotator
1808  * 1) Zoom the source image by a certain factor.
1809  * 2) Scan the zoomed source image at every step=offset and put it in the result. */
1810 surface rotate_any_surface(const surface& surf, float angle, int zoom, int offset)
1811 {
1812         int src_w, src_h, dst_w, dst_h;
1813         float min_x, min_y, sine, cosine;
1814         {
1815                 float max_x, max_y;
1816                 // convert angle to radiant (angle * 2 * PI) / 360
1817                 const float radians = angle * boost::math::constants::pi<float>() / 180;
1818                 cosine = cos(radians);
1819                 sine   = sin(radians);
1820                 // calculate the size of the dst image
1821                 src_w = surf->w * zoom;
1822                 src_h = surf->h * zoom;
1823                 /* See http://en.wikipedia.org/wiki/Rotation_(mathematics) */
1824                 const float point_1x = src_h * -sine;
1825                 const float point_1y = src_h * cosine;
1826                 const float point_2x = src_w * cosine - src_h * sine;
1827                 const float point_2y = src_h * cosine + src_w * sine;
1828                 const float point_3x = src_w * cosine;
1829                 const float point_3y = src_w * sine;
1830                 /* After the rotation, the new image has different dimensions.
1831                  * E.g.: The maximum height equals the former diagonal in case the angle is 45, 135, 225 or 315 degree.
1832                  * See http://en.wikipedia.org/wiki/File:Rotation_illustration2.svg to get the idea. */
1833                 min_x = std::min(0.0F, std::min(point_1x, std::min(point_2x, point_3x)));
1834                 min_y = std::min(0.0F, std::min(point_1y, std::min(point_2y, point_3y)));
1835                 max_x = (angle >  90 && angle < 180) ? 0 : std::max(point_1x, std::max(point_2x, point_3x));
1836                 max_y = (angle > 180 && angle < 270) ? 0 : std::max(point_1y, std::max(point_2y, point_3y));
1837                 dst_w = static_cast<int>(ceil(std::abs(max_x) - min_x)) / zoom;
1838                 dst_h = static_cast<int>(ceil(std::abs(max_y) - min_y)) / zoom;
1839         }
1840         surface dst(create_neutral_surface(dst_w, dst_h));
1841         {
1842                 surface_lock dst_lock(dst);
1843                 const surface src = scale_surface(surf, src_w, src_h);
1844                 const_surface_lock src_lock(src);
1845                 const float scale =   1.f / zoom;
1846                 const int   max_x = dst_w * zoom;
1847                 const int   max_y = dst_h * zoom;
1848                 /* Loop through the zoomed src image,
1849                  * take every pixel in steps with offset distance and place it in the dst image. */
1850                 for (int x = 0; x < max_x; x += offset)
1851                         for (int y = 0; y < max_y; y += offset) {
1852                                 // calculate the src pixel that fits in the dst
1853                                 const float source_x = (x + min_x)*cosine + (y + min_y)*sine;
1854                                 const float source_y = (y + min_y)*cosine - (x + min_x)*sine;
1855                                 // if the pixel exists on the src surface
1856                                 if (source_x >= 0 && source_x < src_w
1857                                                 && source_y >= 0 && source_y < src_h)
1858                                         // get it from the src surface and place it on the dst surface
1859                                         put_pixel(dst, dst_lock, x*scale , y*scale, // multiply with scale
1860                                                         get_pixel(src, src_lock, source_x, source_y));
1861                         }
1862         }
1863 
1864         return dst;
1865 }
