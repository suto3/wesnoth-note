 170 void manager::set_active(bool active)
 171 {
 172         if(!can_activate())
 173         {
 174                 active_ = false;
 175                 LOG_WB << "Whiteboard can't be activated now.\n";
 176         }
 177         else if (active != active_)
 178         {
 179                 active_ = active;
 180                 erase_temp_move();
 181 
 182                 if (active_)
 183                 {
 184                         if(should_clear_undo())
 185                                 resources::undo_stack->clear();
 186                         validate_viewer_actions();
 187                         LOG_WB << "Whiteboard activated! " << *viewer_actions() << "\n";
 188                         create_temp_move();
 189                 } else {
 190                         LOG_WB << "Whiteboard deactivated!\n";
 191                 }
 192         }

 195 void manager::set_invert_behavior(bool invert)
 196 {
 197         //any more than one reference means a lock on whiteboard state was requested
 198         if(!activation_state_lock_.unique())
 199                 return;
 200 
 201         bool block_whiteboard_activation = false;
 202         if(!can_activate())
 203         {
 204                  block_whiteboard_activation = true;
 205         }
 206 
 207         if (invert)
 208         {
 209                 if (!inverted_behavior_)
 210                 {
 211                         if (active_)
 212                         {
 213                                 DBG_WB << "Whiteboard deactivated temporarily.\n";
 214                                 inverted_behavior_ = true;
 215                                 set_active(false);
 216                         }
 217                         else if (!block_whiteboard_activation)
 218                         {
 219                                 DBG_WB << "Whiteboard activated temporarily.\n";
 220                                 inverted_behavior_ = true;
 221                                 set_active(true);
 222                         }
 223                 }
 224         }
 225         else
 226         {
 227                 if (inverted_behavior_)
 228                 {
 229                         if (active_)
 230                         {
 231                                 DBG_WB << "Whiteboard set back to deactivated status.\n";
 232                                 inverted_behavior_ = false;
 233                                 set_active(false);
 234                         }
 235                         else if (!block_whiteboard_activation)
 236                         {
 237                                 DBG_WB << "Whiteboard set back to activated status.\n";
 238                                 inverted_behavior_ = false;
 239                                 set_active(true);
 240                         }
 241                 }
 242         }
 243 }

 261 bool manager::allow_leader_to_move(unit const& leader) const
 262 {
 263         if(!has_actions())
 264                 return true;
 265 
 266         //Look for another leader on another keep in the same castle
 267         { wb::future_map future; // start planned unit map scope
 268                 if(!has_planned_unit_map()) {
 269                         WRN_WB << "Unable to build future map to determine whether leader's allowed to move." << std::endl;
 270                 }
 271                 if(find_backup_leader(leader))
 272                         return true;
 273         } // end planned unit map scope
 274 
 275         if(viewer_actions()->empty()) {
 276                 return true;
 277         }
 278 
 279         //Look for planned recruits that depend on this leader
 280         for(action_const_ptr action : *viewer_actions())
 281         {
 282                 recruit_const_ptr recruit = std::dynamic_pointer_cast<class recruit const>(action);
 283                 recall_const_ptr recall = std::dynamic_pointer_cast<class recall const>(action);
 284                 if(recruit || recall)
 285                 {
 286                         map_location const target_hex = recruit?recruit->get_recruit_hex():recall->get_recall_hex();
 287                         if ( dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(leader, target_hex) )
 288                                 return false;
 289                 }
 290         }
 291         return true;
 292 }

 326 void manager::post_delete_action(action_ptr action)
 327 {
 328         // The fake unit representing the destination of a chain of planned moves should have the regular animation.
 329         // If the last remaining action of the unit that owned this move is a move as well,
 330         // adjust its appearance accordingly.
 331 
 332         side_actions_ptr side_actions = resources::gameboard->teams().at(action->team_index()).get_side_actions();
 333 
 334         unit_ptr actor = action->get_unit();
 335         if(actor) { // The unit might have died following the execution of an attack
 336                 side_actions::iterator action_it = side_actions->find_last_action_of(*actor);
 337                 if(action_it != side_actions->end()) {
 338                         move_ptr move = std::dynamic_pointer_cast<class move>(*action_it);
 339                         if(move && move->get_fake_unit()) {
 340                                 move->get_fake_unit()->anim_comp().set_standing(true);
 341                         }
 342                 }
 343         }
 344 }

 354 void manager::update_plan_hiding(size_t team_index)
 355 {
 356         //We don't control the "viewing" side ... we're probably an observer
 357         if(!resources::gameboard->teams().at(team_index).is_local_human())
 358                 hide_all_plans();
 359         else // normal circumstance
 360         {
 361                 for(team& t : resources::gameboard->teams())
 362                 {
 363                         //make sure only appropriate teams are hidden
 364                         if(!t.is_network_human())
 365                                 team_plans_hidden_[t.side()-1] = false;
 366 
 367                         if(t.is_enemy(team_index+1) || team_plans_hidden_[t.side()-1])
 368                                 t.get_side_actions()->hide();
 369                         else
 370                                 t.get_side_actions()->show();
 371                 }
 372         }
 373         validate_viewer_actions();
 374 }

 384 void manager::on_change_controller(int side, const team& t)
 385 {
 386         wb::side_actions& sa = *t.get_side_actions();
 387         if(t.is_local_human()) // we own this side now
 388         {
 389                 //tell everyone to clear this side's actions -- we're starting anew
 390                 resources::whiteboard->queue_net_cmd(sa.team_index(),sa.make_net_cmd_clear());
 391                 sa.clear();
 392                 //refresh the hidden_ attribute of every team's side_actions
 393                 update_plan_hiding();
 394         }
 395         else if(t.is_local_ai() || t.is_network_ai()) // no one owns this side anymore
 396                 sa.clear(); // clear its plans away -- the ai doesn't plan ... yet
 397         else if(t.is_network()) // Another client is taking control of the side
 398         {
 399                 if(side==viewer_side()) // They're taking OUR side away!
 400                         hide_all_plans(); // give up knowledge of everyone's plans, in case we became an observer
 401 
 402                 //tell them our plans -- they may not have received them up to this point
 403                 size_t num_teams = resources::gameboard->teams().size();
 404                 for(size_t i=0; i<num_teams; ++i)
 405                 {
 406                         team& local_team = resources::gameboard->teams().at(i);
 407                         if(local_team.is_local_human() && !local_team.is_enemy(side))
 408                                 resources::whiteboard->queue_net_cmd(i,local_team.get_side_actions()->make_net_cmd_refresh());
 409                 }
 410         }
 411 }

 435 static void draw_numbers(map_location const& hex, side_actions::numbers_t numbers)
 436 {
 437         std::vector<int>& numbers_to_draw = numbers.numbers_to_draw;
 438         std::vector<size_t>& team_numbers = numbers.team_numbers;
 439         int& main_number = numbers.main_number;
 440         std::set<size_t>& secondary_numbers = numbers.secondary_numbers;
 441 
 442         const double x_offset_base = 0.0;
 443         const double y_offset_base = 0.2;
 444         //position 0,0 in the hex is the upper left corner
 445         //0.8 = horizontal coord., close to the right side of the hex
 446         const double x_origin = 0.8 - numbers_to_draw.size() * x_offset_base;
 447         //0.5 = halfway in the hex vertically
 448         const double y_origin = 0.5 - numbers_to_draw.size() * (y_offset_base / 2);
 449         double x_offset = 0, y_offset = 0;
 450 
 451         size_t size = numbers_to_draw.size();
 452         for(size_t i=0; i<size; ++i)
 453         {
 454                 int number = numbers_to_draw[i];
 455 
 456                 std::string number_text = std::to_string(number);
 457                 size_t font_size;
 458                 if (int(i) == main_number) font_size = 19;
 459                 else if (secondary_numbers.find(i)!=secondary_numbers.end()) font_size = 17;
 460                 else font_size = 15;
 461 
 462                 color_t color = team::get_side_color(static_cast<int>(team_numbers[i]+1));
 463                 const double x_in_hex = x_origin + x_offset;
 464                 const double y_in_hex = y_origin + y_offset;
 465                 resources::screen->draw_text_in_hex(hex, display::LAYER_ACTIONS_NUMBERING,
 466                                 number_text, font_size, color, x_in_hex, y_in_hex);
 467                 x_offset += x_offset_base;
 468                 y_offset += y_offset_base;
 469         }
 470 }

 475         //Helper struct that finds all units teams whose planned actions are currently visible
 476         //Only used by manager::pre_draw().
 477         //Note that this structure is used as a functor.
 478         struct move_owners_finder: public visitor
 479         {
 480 
 481         public:
 482                 move_owners_finder(): move_owners_() { }
 483 
 484                 void operator()(action* action) {
 485                         action->accept(*this);
 486                 }
 487 
 488                 std::set<size_t> const& get_units_owning_moves() {
 489                         return move_owners_;
 490                 }
 491 
 492                 virtual void visit(move_ptr move) {
 493                         if(size_t id = move->get_unit_id()) {
 494                                 move_owners_.insert(id);
 495                         }
 496                 }
 497 
 498                 virtual void visit(attack_ptr attack) {
 499                         //also add attacks if they have an associated move
 500                         if(attack->get_route().steps.size() >= 2) {
 501                                 if(size_t id = attack->get_unit_id()) {
 502                                         move_owners_.insert(id);
 503                                 }
 504                         }
 505                 }
 506                 virtual void visit(recruit_ptr){}
 507                 virtual void visit(recall_ptr){}
 508                 virtual void visit(suppose_dead_ptr){}
 509 
 510         private:
 511                 std::set<size_t> move_owners_;
 512         };
 513 }

 515 void manager::pre_draw()
 516 {
 517         if (can_modify_game_state() && has_actions() && unit_map_lock_.unique()) {
 518                 move_owners_finder move_finder;
 519                 for_each_action(std::ref(move_finder));
 520                 units_owning_moves_ = move_finder.get_units_owning_moves();
 521 
 522                 for (size_t unit_id : units_owning_moves_) {
 523                         unit_map::iterator unit_iter = resources::gameboard->units().find(unit_id);
 524                         assert(unit_iter.valid());
 525                         ghost_owner_unit(&*unit_iter);
 526                 }
 527         }
 528 }
 529 
 530 void manager::post_draw()
 531 {
 532         for (size_t unit_id : units_owning_moves_)
 533         {
 534                 unit_map::iterator unit_iter = resources::gameboard->units().find(unit_id);
 535                 if (unit_iter.valid()) {
 536                         unghost_owner_unit(&*unit_iter);
 537                 }
 538         }
 539         units_owning_moves_.clear();
 540 }

 542 void manager::draw_hex(const map_location& hex)
 543 {
 544         /**
 545          * IMPORTANT: none of the code in this method can call anything which would
 546          * cause a hex to be invalidated (i.e. by calling in turn any variant of display::invalidate()).
 547          * Doing so messes up the iterator currently going over the list of invalidated hexes to draw.
 548          */
 549 
 550         if (!wait_for_side_init_ && has_actions())
 551         {
 552                 //call draw() for all actions
 553                 for_each_action(std::bind(&action::draw_hex, std::placeholders::_1, hex));
 554 
 555                 //Info about the action numbers to be displayed on screen.
 556                 side_actions::numbers_t numbers;
 557                 for (team& t : resources::gameboard->teams())
 558                 {
 559                         side_actions& sa = *t.get_side_actions();
 560                         if(!sa.hidden())
 561                                 sa.get_numbers(hex,numbers);
 562                 }
 563                 draw_numbers(hex,numbers); // helper fcn
 564         }
 565 
 566 }

 600 void manager::send_network_data()
 601 {
 602         size_t size = net_buffer_.size();
 603         for(size_t team_index=0; team_index<size; ++team_index)
 604         {
 605                 config& buf_cfg = net_buffer_[team_index];
 606 
 607                 if(buf_cfg.empty())
 608                         continue;
 609 
 610                 config packet;
 611                 config& wb_cfg = packet.add_child("whiteboard",buf_cfg);
 612                 wb_cfg["side"] = static_cast<int>(team_index+1);
 613                 wb_cfg["to_sides"] = resources::gameboard->teams().at(team_index).allied_human_teams();
 614 
 615                 buf_cfg = config();

 643 void manager::create_temp_move()
 644 {
 645         route_.reset();
 646 
 647         /*
 648          * CHECK PRE-CONDITIONS
 649          * (This section has multiple return paths.)
 650          */
 651 
 652         if ( !active_ || !can_modify_game_state() )
 653                 return;
 654 
 655         pathfind::marked_route const& route =
 656                         resources::controller->get_mouse_handler_base().get_current_route();
 657 
 658         if (route.steps.empty() || route.steps.size() < 2) return;
 659 
 660         unit* temp_moved_unit =
 661                         future_visible_unit(resources::controller->get_mouse_handler_base().get_selected_hex(), viewer_side());
 662         if (!temp_moved_unit) temp_moved_unit =
 663                         future_visible_unit(resources::controller->get_mouse_handler_base().get_last_hex(), viewer_side());
 664         if (!temp_moved_unit) return;
 665         if (temp_moved_unit->side() != resources::screen->viewing_side()) return;
 666 
 667         /*
 668          * DONE CHECKING PRE-CONDITIONS, CREATE THE TEMP MOVE
 669          * (This section has only one return path.)
 670          */
 671 
 672         temp_move_unit_underlying_id_ = temp_moved_unit->underlying_id();
 673 
 674         //@todo: May be appropriate to replace these separate components by a temporary
 675         //      wb::move object
 676 
 677         route_.reset(new pathfind::marked_route(route));
 678         //NOTE: route_->steps.back() = dst, and route_->steps.front() = src
 679 
 680         size_t turn = 0;
 681         std::vector<map_location>::iterator prev_itor = route.steps.begin();
 682         std::vector<map_location>::iterator curr_itor = prev_itor;
 683         std::vector<map_location>::iterator end_itor  = route.steps.end();
 684         for(; curr_itor!=end_itor; ++curr_itor)
 685         {
 686                 const map_location& hex = *curr_itor;
 687 
 688                 //search for end-of-turn marks
 689                 pathfind::marked_route::mark_map::const_iterator w =
 690                                 route.marks.find(hex);
 691                 if(w != route.marks.end() && w->second.turns > 0)
 692                 {
 693                         turn = w->second.turns-1;
 694 
 695                         if(turn >= move_arrows_.size())
 696                                 move_arrows_.resize(turn+1);
 697                         if(turn >= fake_units_.size())
 698                                 fake_units_.resize(turn+1);
 699 
 700                         arrow_ptr& move_arrow = move_arrows_[turn];
 701                         fake_unit_ptr& fake_unit = fake_units_[turn];
 702 
 703                         if(!move_arrow)
 704                         {
 705                                 // Create temp arrow
 706                                 move_arrow.reset(new arrow());
 707                                 move_arrow->set_color(team::get_side_color_index(
 708                                                 viewer_side()));
 709                                 move_arrow->set_style(arrow::STYLE_HIGHLIGHTED);
 710                         }
 711 
 712                         arrow_path_t path(prev_itor,curr_itor+1);
 713                         move_arrow->set_path(path);
 714 
 715                         if(path.size() >= 2)
 716                         {
 717                                 // Bug #20299 demonstrates a situation where an incorrect fake/ghosted unit can be used.
 718                                 // So before assuming that a pre-existing fake_unit can be re-used, check that its ID matches the unit being moved.
 719                                 if(!fake_unit || fake_unit.get_unit_ptr()->id() != temp_moved_unit->id())
 720                                 {
 721                                         // Create temp ghost unit
 722                                         fake_unit = fake_unit_ptr(unit_ptr (new unit(*temp_moved_unit)), resources::fake_units);
 723                                         fake_unit->anim_comp().set_ghosted(true);
 724                                 }
 725 
 726                                 unit_display::move_unit(path, fake_unit.get_unit_ptr(), false); //get facing right
 727                                 fake_unit->anim_comp().invalidate(*game_display::get_singleton());
 728                                 fake_unit->set_location(*curr_itor);
 729                                 fake_unit->anim_comp().set_ghosted(true);
 730                         }
 731                         else //zero-hex path -- don't bother drawing a fake unit
 732                                 fake_unit.reset();
 733 
 734                         prev_itor = curr_itor;
 735                 }
 736         }
 737         //in case path shortens on next step and one ghosted unit has to be removed
 738         int ind = fake_units_.size() - 1;
 739         fake_units_[ind]->anim_comp().invalidate(*game_display::get_singleton());
 740         //toss out old arrows and fake units
 741         move_arrows_.resize(turn+1);
 742         fake_units_.resize(turn+1);
 743 }

 746 {
 747         move_arrows_.clear();
 748         for(fake_unit_ptr const& tmp : fake_units_) {
 749                 if(tmp) {
 750                         tmp->anim_comp().invalidate(*game_display::get_singleton());
 751                 }
 752         }
 753         fake_units_.clear();
 754         route_.reset();
 755         temp_move_unit_underlying_id_ = 0;

 758 void manager::save_temp_move()
 759 {
 760         if (has_temp_move() && !executing_actions_ && !resources::controller->is_linger_mode())
 761         {
 762                 side_actions& sa = *viewer_actions();
 763                 unit* u = future_visible_unit(route_->steps.front());
 764                 assert(u);
 765                 size_t first_turn = sa.get_turn_num_of(*u);
 766 
 767                 validate_viewer_actions();
 768 
 769                 assert(move_arrows_.size() == fake_units_.size());
 770                 size_t size = move_arrows_.size();
 771                 for(size_t i=0; i<size; ++i)
 772                 {
 773                         arrow_ptr move_arrow = move_arrows_[i];
 774                         if(!arrow::valid_path(move_arrow->get_path()))
 775                                 continue;
 776 
 777                         size_t turn = first_turn + i;
 778                         fake_unit_ptr fake_unit = fake_units_[i];
 779 
 780                         //@todo Using a marked_route here is wrong, since right now it's not marked
 781                         //either switch over to a plain route for planned moves, or mark it correctly
 782                         pathfind::marked_route route;
 783                         route.steps = move_arrow->get_path();
 784                         route.move_cost = path_cost(route.steps,*u);
 785 
 786                         sa.queue_move(turn,*u,route,move_arrow,fake_unit);
 787                 }
 788                 erase_temp_move();
 789 
 790                 LOG_WB << *viewer_actions() << "\n";
 791                 print_help_once();
 792         }
 793 }

 800 void manager::save_temp_attack(const map_location& attacker_loc, const map_location& defender_loc, int weapon_choice)
 801 {
 802         if (active_ && !executing_actions_ && !resources::controller->is_linger_mode())
 803         {
 804                 assert(weapon_choice >= 0);
 805 
 806                 arrow_ptr move_arrow;
 807                 fake_unit_ptr fake_unit;
 808                 map_location source_hex;
 809 
 810                 if (route_ && !route_->steps.empty())
 811                 {
 812                         //attack-move
 813                         assert(move_arrows_.size() == 1);
 814                         assert(fake_units_.size() == 1);
 815                         move_arrow = move_arrows_.front();
 816                         fake_unit = fake_units_.front();
 817 
 818                         assert(route_->steps.back() == attacker_loc);
 819                         source_hex = route_->steps.front();
 820 
 821                         fake_unit->anim_comp().set_disabled_ghosted(true);
 822                 }
 823                 else
 824                 {
 825                         //simple attack
 826                         move_arrow.reset(new arrow);
 827                         source_hex = attacker_loc;
 828                         route_.reset(new pathfind::marked_route);
 829                         // We'll pass as parameter a one-hex route with no marks.
 830                         route_->steps.push_back(attacker_loc);
 831                 }
 832 
 833                 unit* attacking_unit = future_visible_unit(source_hex);
 834                 assert(attacking_unit);
 835 
 836                 validate_viewer_actions();
 837 
 838                 side_actions& sa = *viewer_actions();
 839                 sa.queue_attack(sa.get_turn_num_of(*attacking_unit),*attacking_unit,defender_loc,weapon_choice,*route_,move_arrow,fake_unit);
 840 
 841                 print_help_once();
 842 
 843                 resources::screen->invalidate(defender_loc);
 844                 resources::screen->invalidate(attacker_loc);
 845                 erase_temp_move();
 846                 LOG_WB << *viewer_actions() << "\n";
 847         }

 914 void manager::contextual_execute()
 915 {
 916         validate_viewer_actions();
 917         if (can_enable_execution_hotkeys())
 918         {
 919                 erase_temp_move();
 920 
 921                 //For exception-safety, this struct sets executing_actions_ to false on destruction.
 922                 variable_finalizer<bool> finally(executing_actions_, false);
 923 
 924                 action_ptr action;
 925                 side_actions::iterator it = viewer_actions()->end();
 926                 unit const* selected_unit = future_visible_unit(resources::controller->get_mouse_handler_base().get_selected_hex(), viewer_side());
 927                 if (selected_unit &&
 928                                 (it = viewer_actions()->find_first_action_of(*selected_unit)) != viewer_actions()->end())
 929                 {
 930                         executing_actions_ = true;
 931                         viewer_actions()->execute(it);
 932                 }
 933                 else if (highlighter_ && (action = highlighter_->get_execute_target()) &&
 934                                  (it = viewer_actions()->get_position_of(action)) != viewer_actions()->end())
 935                 {
 936                         executing_actions_ = true;
 937                         viewer_actions()->execute(it);
 938                 }
 939                 else //we already check above for viewer_actions()->empty()
 940                 {
 941                         executing_actions_ = true;
 942                         viewer_actions()->execute_next();
 943                 }
 944         } //Finalizer struct sets executing_actions_ to false
 945 }

 953 bool manager::execute_all_actions()
 954 {
 955         //exception-safety: finalizers set variables to false on destruction
 956         //i.e. when method exits naturally or exception is thrown
 957         variable_finalizer<bool> finalize_executing_actions(executing_actions_, false);
 958         variable_finalizer<bool> finalize_executing_all_actions(executing_all_actions_, false);
 959 
 960         validate_viewer_actions();
 961         if(viewer_actions()->empty() || viewer_actions()->turn_size(0) == 0)
 962         {
 963                 //No actions to execute, job done.
 964                 return true;
 965         }
 966 
 967         assert(can_enable_execution_hotkeys());
 968 
 969         erase_temp_move();
 970 
 971         // Build unit map once to ensure spent gold and other calculations are refreshed
 972         set_planned_unit_map();
 973         assert(has_planned_unit_map());
 974         set_real_unit_map();
 975 
 976         executing_actions_ = true;
 977         executing_all_actions_ = true;
 978 
 979         side_actions_ptr sa = viewer_actions();
 980 
 981         if (resources::whiteboard->has_planned_unit_map())
 982         {
 983                 ERR_WB << "Modifying action queue while temp modifiers are applied!!!" << std::endl;
 984         }
 985 
 986         //LOG_WB << "Before executing all actions, " << *sa << "\n";
 987 
 988         while (sa->turn_begin(0) != sa->turn_end(0))
 989         {
 990                 bool action_successful = sa->execute(sa->begin());
 991 
 992                 // Interrupt on incomplete action
 993                 if (!action_successful)
 994                 {
 995                         return false;
 996                 }
 997         }
 998         return true;
 999 }

1117         case 1:
1118                 for(team* t : allies) {
1119                         team_plans_hidden_[t->side()-1]=true;
1120                 }
1121                 break;
1122         default:
1123                 if(selection > 1)
1124                 {
1125                         size_t t_index = allies[selection-2]->side()-1;
1126                         //toggle ...
1127                         bool hidden = team_plans_hidden_[t_index];
1128                         team_plans_hidden_[t_index] = !hidden;
1129                 }
1130                 break;
1131         }
