 194         template<class T>
 195         void test_resolutions(const resolution_list& resolutions)
 196         {
 197                 for(const resolution& resolution : resolutions) {
 198                         CVideo& video = test_utils::get_fake_display(resolution.first, resolution.second).video();
 199 
 200                         dialog_tester<T> ctor;
 201                         const std::unique_ptr<modal_dialog> dlg(ctor.create());
 202                         BOOST_REQUIRE_MESSAGE(dlg.get(), "Failed to create a dialog.");
 203 
 204                         const std::string id = unit_test_mark_as_tested(*(dlg.get()));
 205 
 206                         std::string exception;
 207                         try {
 208                                 dlg->show(video, 1);
 209                         } catch(gui2::layout_exception_width_modified&) {
 210                                 exception = "gui2::layout_exception_width_modified";
 211                         } catch(gui2::layout_exception_width_resize_failed&) {
 212                                 exception = "gui2::layout_exception_width_resize_failed";
 213                         } catch(gui2::layout_exception_height_resize_failed&) {
 214                                 exception = "gui2::layout_exception_height_resize_failed";
 215                         } catch(wml_exception& e) {
 216                                 exception = e.dev_message;
 217                         } catch(std::exception& e) {
 218                                 exception = e.what();
 219                         } catch(...) {
 220                                 exception = "unknown";
 221                         }
 222                         BOOST_CHECK_MESSAGE(exception.empty(),
 223                                         "Test for '" << id
 224                                         << "' Failed\nnew widgets = " << gui2::new_widgets
 225                                         << " resolution = " << resolution.first
 226                                         << 'x' << resolution.second
 227                                         << "\nException caught: " << exception << '.');
 228                 }
