 106 static void add_widget(gui2::grid& grid
 107                 , gui2::widget* widget
 108                 , const std::string& id
 109                 , const unsigned row
 110                 , const unsigned column)
 111 {
 112         BOOST_REQUIRE_NE(widget, static_cast<gui2::widget*>(nullptr));
 113 
 114         widget->set_id(id);
 115         grid.set_child(widget
 116                         , row
 117                         , column
 118                         , gui2::grid::VERTICAL_GROW_SEND_TO_CLIENT
 119                                 | gui2::grid::HORIZONTAL_GROW_SEND_TO_CLIENT
 120                         , 0);
 121 }

 123 template<class T>
 124 static void test_control()
 125 {
 126         T control;
 127 
 128         {
 129                 gui2::iteration::iterator< gui2::iteration::policy::order::top_down<
 130                                 true
 131                                 , true
 132                                 , true> >
 133                         iterator(control);
 134 
 135                 /***** INITIAL STATE *****/
 136 
 137                 BOOST_CHECK_EQUAL(iterator.at_end(), false);
 138 
 139                 BOOST_CHECK_EQUAL(&*iterator, &control);
 140 
 141                 /***** POST END *****/
 142 
 143                 BOOST_CHECK_EQUAL(iterator.next(), false);
 144 
 145                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 146 
 147         }
 148         {
 149                 gui2::iteration::iterator< gui2::iteration::policy::order::top_down<
 150                                 false
 151                                 , true
 152                                 , true> >
 153                         iterator(control);
 154 
 155                 /***** INITIAL STATE *****/
 156 
 157                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 158         }
 159 
 160         {
 161                 gui2::iteration::iterator<gui2::iteration::policy::order::bottom_up<true, true, true> > iterator(control);
 162                 BOOST_CHECK_EQUAL(iterator.at_end(), false);
 163         }
 164         {
 165                 gui2::iteration::iterator<gui2::iteration::policy::order::bottom_up<false, false, false> > iterator(control);
 166                 BOOST_CHECK_EQUAL(iterator.at_end(), true);
 167         }
 168 }
