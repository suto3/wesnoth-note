  75 /**
  76   * Finalizer class to help with exception safety
  77   * sets variable to value on destruction
  78   */
  79 template <typename T>
  80 class variable_finalizer
  81 {
  82 public:
  83         variable_finalizer(T & variable, T value):
  84                 variable_(&variable),
  85                 value_(value)
  86         {}
  87         ~variable_finalizer()
  88         {
  89                 if(variable_ != nullptr) {
  90                         *variable_ = value_;
  91                 }
  92         }
  93         /** Stop tracking the variable, i.e. this object won't do anything on destruction. */
  94         void clear()
  95         {
  96                 variable_ = nullptr;
  97         }
  98 private:
  99         T * variable_;
 100         T value_;
 101 };
