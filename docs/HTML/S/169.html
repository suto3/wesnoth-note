  36 /**
  37  * Gets some data on a unit type (__index metamethod).
  38  * - Arg 1: table containing an "id" field.
  39  * - Arg 2: string containing the name of the property.
  40  * - Ret 1: something containing the attribute.
  41  */
  42 static int impl_unit_type_get(lua_State *L)
  43 {
  44         const unit_type& ut = luaW_checkunittype(L, 1);
  45         char const *m = luaL_checkstring(L, 2);
  46 
  47         // Find the corresponding attribute.
  48         return_tstring_attrib("name", ut.type_name());
  49         return_string_attrib("id", ut.id());
  50         return_string_attrib("alignment", ut.alignment().to_string());
  51         return_string_attrib("race", ut.race_id());
  52         return_string_attrib("image", ut.image());
  53         return_string_attrib("icon", ut.icon());
  54         return_int_attrib("max_hitpoints", ut.hitpoints());
  55         return_int_attrib("max_moves", ut.movement());
  56         return_int_attrib("max_experience", ut.experience_needed());
  57         return_int_attrib("cost", ut.cost());
  58         return_int_attrib("level", ut.level());
  59         return_int_attrib("recall_cost", ut.recall_cost());
  60         return_cfgref_attrib("__cfg", ut.get_cfg());
  61         if (strcmp(m, "traits") == 0) {
  62                 lua_newtable(L);
  63                 for (const config& trait : ut.possible_traits()) {
  64                         const std::string& id = trait["id"];
  65                         lua_pushlstring(L, id.c_str(), id.length());
  66                         luaW_pushconfig(L, trait);
  67                         lua_rawset(L, -3);
  68                 }
  69                 return 1;
  70         }
  71         if (strcmp(m, "abilities") == 0) {
  72                 lua_push(L, ut.get_ability_list());
  73                 return 1;
  74         }
  75         if (strcmp(m, "attacks") == 0) {
  76                 push_unit_attacks_table(L, 1);
  77                 return 1;
  78         }
  79         // TODO: Should this only exist for base units?
  80         if(strcmp(m, "variations") == 0) {
  81                 *new(L) const unit_type* = &ut;
  82                 luaL_setmetatable(L, UnitTypeTable);
  83                 return 1;
  84         }
  85         return 0;
  86 }

  99 static int impl_unit_type_lookup(lua_State* L)
 100 {
 101         std::string id = luaL_checkstring(L, 2);
 102         const unit_type* ut;
 103         if(const unit_type* base = *static_cast<const unit_type**>(luaL_testudata(L, 1, UnitTypeTable))) {
 104                 if(id == "male" || id == "female") {
 105                         ut = &base->get_gender_unit_type(id);
 106                 } else {
 107                         ut = &base->get_variation(id);
 108                 }
 109         } else {
 110                 ut = unit_types.find(id);
 111         }
 112         if(ut) {
 113                 luaW_pushunittype(L, *ut);
 114                 return 1;
 115         }
 116         return 0;
 117 }

 127 static int impl_unit_type_count(lua_State* L)
 128 {
 129         lua_pushstring(L, "base");
 130         lua_rawget(L, 1);
 131         if(const unit_type* base = luaW_tounittype(L, -1)) {
 132                 lua_pushnumber(L, base->variations().size() + base->genders().size());
 133         } else {
 134                 lua_pushnumber(L, unit_types.types().size());

 139 static int impl_unit_type_next(lua_State* L)
 140 {
 141         const unit_type* base = *static_cast<const unit_type**>(luaL_testudata(L, 1, UnitTypeTable));
 142         auto unit_map = base ? base->variation_types() : unit_types.types();
 143         decltype(unit_map)::const_iterator it = unit_map.end();
 144         if(lua_isnoneornil(L, 2)) {
 145                 if(base) {
 146                         if(base->has_gender_variation(unit_race::MALE)) {
 147                                 lua_pushstring(L, "male");
 148                                 luaW_pushunittype(L, base->get_gender_unit_type(unit_race::MALE));
 149                                 return 2;
 150                         } else if(base->has_gender_variation(unit_race::FEMALE)) {
 151                                 lua_pushstring(L, "female");
 152                                 luaW_pushunittype(L, base->get_gender_unit_type(unit_race::FEMALE));
 153                                 return 2;
 154                         }
 155                 }
 156                 it = unit_map.begin();
 157         } else {
 158                 const std::string id = luaL_checkstring(L, 2);
 159                 if(base) {
 160                         if(id == "male" && base->has_gender_variation(unit_race::FEMALE)) {
 161                                 lua_pushstring(L, "female");
 162                                 luaW_pushunittype(L, base->get_gender_unit_type(unit_race::FEMALE));
 163                                 return 2;
 164                         } else if(id == "male" || id == "female") {
 165                                 it = unit_map.begin();
 166                         }
 167                 }
 168                 if(it == unit_map.end()) {
 169                         it = unit_map.find(id);
 170                 }
 171                 if(it == unit_map.end()) {
 172                         return 0;
 173                 }
 174                 ++it;
 175         }
 176         if (it == unit_map.end()) {
 177                 return 0;
 178         }
 179         lua_pushlstring(L, it->first.c_str(), it->first.size());
 180         luaW_pushunittype(L, it->second);
 181         return 2;
 182 }

 184 static int impl_unit_type_pairs(lua_State* L) {
 185         lua_pushcfunction(L, &impl_unit_type_next);
 186         lua_pushvalue(L, -2);
 187         lua_pushnil(L);

 206         std::string register_table(lua_State* L)
 207         {
 208                 lua_getglobal(L, "wesnoth");
 209                 *new(L) unit_type* = nullptr;
 210                 luaL_newmetatable(L, UnitTypeTable);
 211                 lua_pushcfunction(L, impl_unit_type_lookup);
 212                 lua_setfield(L, -2, "__index");
 213                 lua_pushcfunction(L, impl_unit_type_new);
 214                 lua_setfield(L, -2, "__newindex");
 215                 lua_pushcfunction(L, impl_unit_type_count);
 216                 lua_setfield(L, -2, "__len");
 217                 lua_pushcfunction(L, impl_unit_type_pairs);
 218                 lua_setfield(L, -2, "__pairs");
 219                 lua_pushstring(L, UnitTypeTable);
 220                 lua_setfield(L, -2, "__metatable");
 221                 lua_setmetatable(L, -2);
 222                 lua_setfield(L, -2, "unit_types");
 223                 lua_pop(L, 1);
 224 
 225                 return "Adding unit_types table...\n";
 226         }

 229 void luaW_pushunittype(lua_State *L, const unit_type& ut)
 230 {
 231         *static_cast<const unit_type**>(lua_newuserdata(L, sizeof(unit_type*))) = &ut;
 232         luaL_setmetatable(L, UnitType);
 233 }

 235 const unit_type* luaW_tounittype(lua_State* L, int idx)
 236 {
 237         if(void* p = luaL_testudata(L, idx, UnitType)) {
 238                 return *static_cast<const unit_type**>(p);
 239         }
 240         return nullptr;
 241 }

 243 const unit_type& luaW_checkunittype(lua_State* L, int idx)
 244 {
 245         return **static_cast<const unit_type**>(luaL_checkudata(L, idx, UnitType));;
