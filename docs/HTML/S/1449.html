 170      * @param window
 171      *        The workbench windows from where to get the current selection
 172      * @return An {@link IStructuredSelection} instance
 173      */
 174     public static IStructuredSelection getSelectedStructuredSelection(
 175         final IWorkbenchWindow window )
 176     {
 177         if( window == null ) {
 178             return null;
 179         }
 180 
 181         RunnableWithResult< IStructuredSelection > runnable = new RunnableWithResult< IStructuredSelection >( ) {
 182             @Override
 183             public void run( )
 184             {
 185                 try {
 186                     setResult( null );
 187 
 188                     if( ! ( window.getSelectionService( ).getSelection( ) instanceof IStructuredSelection ) ) {
 189                         return;
 190                     }
 191 
 192                     setResult( ( IStructuredSelection ) window
 193                         .getSelectionService( ).getSelection( ) );
 194                 } catch( Exception e ) {
 195                     e.printStackTrace( );
 196                 }
 197             }
 198         };
 199         Display.getDefault( ).syncExec( runnable );
 200         return runnable.getResult( );
 201     }

 339     /**
 340      * Setups the workspace, by checking:
 341      * 1) The user has set all plugin's preferences.
 342      * If not, the preferences window will open
 343      * 
 344      * @param guided
 345      *        True to guide the user through the setup routine.
 346      */
 347     public static void setupWorkspace( final boolean guided )
 348     {
 349         if( guided ) {
 350             GUIUtils.showInfoMessageBox( Messages.WorkspaceUtils_0 );
 351         }
 352 
 353         if( ! checkPathsAreSet( Preferences.getDefaultInstallName( ), false ) ) {
 354             PreferenceDialog pref = PreferencesUtil.createPreferenceDialogOn(
 355                 WesnothPlugin.getShell( ), "org.wesnoth.preferences.InstallsPage", null, null ); //$NON-NLS-1$
 356             pref.open( );
 357             if( ! checkPathsAreSet( Preferences.getDefaultInstallName( ), true ) ) {
 358                 GUIUtils.showErrorMessageBox( Messages.WorkspaceUtils_7 );
 359                 return;
 360             }
 361         }
 362 
 363         if( guided ) {
 364             GUIUtils.showInfoMessageBox( Messages.WorkspaceUtils_9 );
 365         }
 366 
 367         WorkspaceJob job = new WorkspaceJob( Messages.WorkspaceUtils_13 ) {
 368             @Override
 369             public IStatus runInWorkspace( IProgressMonitor monitor )
 370             {
 371                 try {
 372                     // automatically import 'special' folders as projects
 373                     List< File > userAddonsFiles = new ArrayList< File >( );
 374                     Paths paths = Preferences.getPaths( Preferences
 375                         .getDefaultInstallName( ) );
 376 
 377                     if( GUIUtils.showMessageBox( Messages.WorkspaceUtils_18,
 378                         SWT.ICON_QUESTION | SWT.YES | SWT.NO ) == SWT.YES ) {
 379                         // useraddons/add-ons/data
 380                         File[] tmp = new File( paths.getAddonsDir( ) )
 381                             .listFiles( );
 382                         if( tmp != null ) {
 383                             userAddonsFiles.addAll( Arrays.asList( tmp ) );
 384                         }
 385                     }
 386 
 387                     monitor.beginTask( Messages.WorkspaceUtils_22,
 388                         userAddonsFiles.size( ) * 10 );
 389 
 390                     for( File file: userAddonsFiles ) {
 391                         if( file.isDirectory( ) == false
 392                             || file.getName( ).startsWith( "." ) ) {
 393                             continue;
 394                         }
 395 
 396                         String projectName = file.getName( );
 397                         if( ResourceUtils.isCampaignDirPath( paths,
 398                             file.getAbsolutePath( ) ) ) {
 399                             projectName = "_Mainline_" + file.getName( ); //$NON-NLS-1$
 400                         }
 401 
 402                         try {
 403                             ProjectUtils.createWesnothProject( projectName,
 404                                 file.getAbsolutePath( ),
 405                                 paths.getInstallName( ), monitor );
 406                         } catch( OperationCanceledException e ) {
 407                             // do nothing
 408                         }
 409 
 410 
 411                         monitor.worked( 10 );
 412                     }
 413 
 414                     if( guided ) {
 415                         GUIUtils
 416                             .showInfoMessageBox( Messages.WorkspaceUtils_25 );
 417                     }
 418                     else {
 419                         Logger.getInstance( ).log( "setupWorkspace was successful", //$NON-NLS-1$
 420                             Messages.WorkspaceUtils_29 );
 421                     }
 422                 } catch( Exception e ) {
 423                     Logger.getInstance( ).logException( e );
 424                     GUIUtils.showErrorMessageBox( Messages.WorkspaceUtils_30 );
 425                 }
 426 
 427                 monitor.done( );
 428                 return Status.OK_STATUS;
 429             }
 430         };
 431         job.schedule( );
 432     }

 434     /**
 435      * Checks if the user has set some needed preferences and if the workspace
 436      * is setup
 437      * 
 438      * @param installName
 439      *        The install name to use when checking the paths
 440      * @param displayWarning
 441      *        true to display a messagebox warning
 442      *        the user if conditions are not met
 443      * @return True if all paths are set, false otherwise.
 444      */
 445     public static boolean checkPathsAreSet( String installName,
 446         boolean displayWarning )
 447     {
 448         if( ! ResourceUtils.isValidFilePath( Preferences.getPaths( installName )
 449             .getWesnothExecutablePath( ) )
 450             || ! ResourceUtils.isValidFilePath( Preferences.getPaths(
 451                 installName ).getUserDir( ) )
 452             || ! ResourceUtils.isValidFilePath( Preferences.getPaths(
 453                 installName ).getWMLToolsDir( ) )
 454             || ! ResourceUtils.isValidFilePath( Preferences.getPaths(
 455                 installName ).getWorkingDir( ) ) ) {
 456             if( displayWarning ) {
 457                 GUIUtils.showWarnMessageBox( Messages.WorkspaceUtils_33 );
 458             }
 459             return false;
 460         }
 461 
 462         return true;
 463     }
