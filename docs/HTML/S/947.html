  25 /**
  26  * Function that will add to @a result all locations exactly @a radius tiles
  27  * from @a center (or nothing if @a radius is not positive). @a result must be
  28  * a std::vector of locations.
  29  */
  30 void get_tile_ring(const map_location& center, const int radius,
  31                    std::vector<map_location>& result)
  32 {
  33         if ( radius <= 0 ) {
  34                 return;
  35         }
  36 
  37         map_location loc = center.get_direction(map_location::SOUTH_WEST, radius);
  38 
  39         for(int n = 0; n != 6; ++n) {
  40                 const map_location::DIRECTION dir = static_cast<map_location::DIRECTION>(n);
  41                 for(int i = 0; i != radius; ++i) {
  42                         result.push_back(loc);
  43                         loc = loc.get_direction(dir, 1);
  44                 }
  45         }
  46 }

  49 /**
  50  * Function that will add to @a result all locations within @a radius tiles
  51  * of @a center (excluding @a center itself). @a result must be a std::vector
  52  * of locations.
  53  */
  54 void get_tiles_in_radius(const map_location& center, const int radius,
  55                          std::vector<map_location>& result)
  56 {
  57         for(int n = 1; n <= radius; ++n) {
  58                 get_tile_ring(center, n, result);
  59         }

  82         // Ranges of rows are stored as pairs of a row number and a number of rows.
  83         typedef std::pair<int, size_t> row_range;
  84         // This is a map from column numbers to sets of ranges of rows.
  85         typedef std::map<int, std::set<row_range> > column_ranges;
  86

  88         /**
  89          * Function that will collect all locations within @a radius tiles of an
  90          * element of @a locs, subject to the restriction col_begin <= x < col_end.
  91          */
  92         // Complexity: O(nr lg(nr)), where n = locs.size() and r = radius.
  93         // In this formula, r is bound by col_end-col_begin (but that is
  94         // probably a rare event).
  95         void get_column_ranges(column_ranges & collected_tiles,
  96                                const std::vector<map_location>& locs,
  97                                const size_t radius,
  98                                const int col_begin, const int col_end)
  99         {
 100                 // Shorter names for the directions we'll use.
 101                 const map_location::DIRECTION NORTH_WEST = map_location::NORTH_WEST;
 102                 const map_location::DIRECTION NORTH_EAST = map_location::NORTH_EAST;
 103                 const map_location::DIRECTION SOUTH_EAST = map_location::SOUTH_EAST;
 104 
 105                 // Perform this conversion once.
 106                 const int radius_i = static_cast<int>(radius);
 107 
 108                 for (const map_location &loc : locs)
 109                         if ( loc != map_location::null_location() )
 110                         {
 111                                 // Calculate the circle of hexes around this one.
 112                                 size_t height = radius;
 113                                 map_location top = loc.get_direction(NORTH_WEST, radius_i);
 114                                 // Don't start off the map edge.
 115                                 if ( top.x < col_begin ) {
 116                                         const int col_shift = std::min(col_begin, loc.x) - top.x;
 117                                         top = top.get_direction(NORTH_EAST, col_shift);
 118                                         height += col_shift;
 119                                 }
 120                                 // The left side.
 121                                 const int end_l = std::min(loc.x, col_end);
 122                                 for ( ; top.x < end_l; top = top.get_direction(NORTH_EAST, 1) )
 123                                         collected_tiles[top.x].insert(row_range(top.y, ++height));
 124                                 // Extra increment so the middle column is tall enough.
 125                                 height += 2;
 126                                 // Don't start off the map edge (we allow loc to be off-board).
 127                                 if ( top.x < col_begin ) {
 128                                         const int col_shift = col_begin - top.x;
 129                                         top = top.get_direction(SOUTH_EAST, col_shift);
 130                                         height -= col_shift;
 131                                 }
 132                                 // The middle column and right side.
 133                                 const int end_r = std::min(loc.x + radius_i + 1, col_end);
 134                                 for ( ; top.x < end_r; top = top.get_direction(SOUTH_EAST, 1) )
 135                                         collected_tiles[top.x].insert(row_range(top.y, --height));
 136                         }
 137         }

 139         /**
 140          * Function that interprets @a collected_tiles and adds to @a result those
 141          * whose y-coordinate satisifies row_begin <= y < row_end.
 142          * When passed to this function, @a result must not be empty. (This allows
 143          * a code simplification and is currently always the case anyway.)
 144          */
 145         // Complexity: O(number of distinct hexes collected), assuming that the
 146         // insertion hint makes insertions O(1). Furthermore, hexes outside the
 147         // interval [row_begin, row_end) are skipped and do not count towards the
 148         // complexity.
 149         void ranges_to_tiles(std::set<map_location> & result,
 150                              const column_ranges & collected_tiles,
 151                              int row_begin, int row_end)
 152         {
 153                 // This should help optimize the insertions (since we will be
 154                 // processing hexes in their lexicographical order).
 155                 std::set<map_location>::const_iterator insert_hint = result.begin();
 156                 // Note: This hint will get incremented later, which is the only
 157                 // reason we require result to be initially non-empty.
 158 
 159                 for (const column_ranges::value_type & column : collected_tiles)
 160                 {
 161                         // For this loop, the order within the set is crucial; we need
 162                         // rows.first to be non-decreasing with each iteration.
 163                         // Loop invariant: within this column, all rows before next_row
 164                         // have been processed and either added to result or skipped.
 165                         // There is no going back (nor a need to).
 166                         int next_row = row_begin;
 167                         for (const row_range &rows : column.second)
 168                         {
 169                                 // Skipping some rows?
 170                                 if ( next_row < rows.first )
 171                                         next_row = rows.first;
 172 
 173                                 // Add this range of hexes.
 174                                 const int end = std::min(rows.first + static_cast<int>(rows.second),
 175                                                          row_end);
 176                                 for ( ; next_row < end; ++next_row )
 177                                         insert_hint = result.insert(++insert_hint,
 178                                                               map_location(column.first, next_row));
 179 
 180                                 // Have we reached the end of the board?
 181                                 if ( next_row >= row_end )
 182                                         break;
 183                         }
 184                 }
 185         }

 190 /**
 191  * Function that will add to @a result all elements of @a locs, plus all
 192  * on-board locations that are within @a radius tiles of an element of locs.
 193  * @a result must be a std::set of locations.
 194  */
 195 // Complexity: O(nr lg(nr) + nr^2), where n = locs.size(), r = radius.
 196 // The nr^2 term is bounded by the size of the board.
 197 void get_tiles_radius(const gamemap& map, const std::vector<map_location>& locs,
 198                       size_t radius, std::set<map_location>& result,
 199                       bool with_border)
 200 {
 201         // Make sure the provided locations are included.
 202         // This would be needed in case some of the provided locations are off-map.
 203         // It also allows simpler processing if the radius is zero.
 204         // For efficiency, do this first since locs is potentially unsorted.
 205         result.insert(locs.begin(), locs.end());
 206 
 207         if ( radius != 0  &&  !locs.empty() )
 208         {
 209                 const int border = with_border ? map.border_size() : 0;
 210                 column_ranges collected_tiles;
 211 
 212                 // Collect the hexes within the desired disks into collected_tiles.
 213                 // This maps each x-value to a set of ranges of y-values that
 214                 // are covered by the disks around each element of locs.
 215                 // (So the data size at this point is proportional to the number
 216                 // of x-values involved, which is O(nr). The lg(nr) factor comes
 217                 // from the data being sorted.)
 218                 get_column_ranges(collected_tiles, locs, radius, -border, map.w() + border);
 219 
 220                 // Now that all the tiles have been collected, add them to result.
 221                 // (There are O(nr^2) hexes to add.) By collecting before adding, each
 222                 // hex will be processed only once, even when disks overlap. This is
 223                 // how we can get good performance if there is significant overlap, and
 224                 // how the work required can be bound by the size of the board.
 225                 ranges_to_tiles(result, collected_tiles, -border, map.h() + border);
 226         }
 227 }

 230 /**
 231  * Function that will add to @a result all elements of @a locs, plus all
 232  * on-board locations matching @a pred that are connected to elements of
 233  * locs by a chain of at most @a radius tiles, each of which matches @a pred.
 234  * @a result must be a std::set of locations.
 235  */
 236 void get_tiles_radius(gamemap const &map, std::vector<map_location> const &locs,
 237                       size_t radius, std::set<map_location> &result,
 238                       bool with_border, xy_pred const &pred)
 239 {
 240         typedef std::set<map_location> location_set;
 241 
 242         location_set must_visit, filtered_out;
 243         location_set not_visited(locs.begin(), locs.end());
 244 
 245         for ( ; radius != 0  &&  !not_visited.empty(); --radius )
 246         {
 247                 location_set::const_iterator it = not_visited.begin();
 248                 location_set::const_iterator it_end = not_visited.end();
 249 
 250                 result.insert(it, it_end);
 251                 for(; it != it_end; ++it) {
 252                         map_location adj[6];
 253                         get_adjacent_tiles(*it, adj);
 254                         for(size_t i = 0; i != 6; ++i) {
 255                                 map_location const &loc = adj[i];
 256                                 if ( with_border ? map.on_board_with_border(loc) :
 257                                                    map.on_board(loc) ) {
 258                                         if ( !result.count(loc) && !filtered_out.count(loc) ) {
 259                                                 if ( pred(loc) )
 260                                                         must_visit.insert(loc);
 261                                                 else
 262                                                         filtered_out.insert(loc);
 263                                         }
 264                                 }
 265                         }
 266                 }
 267 
 268                 not_visited.swap(must_visit);
 269                 must_visit.clear();
 270         }
 271 
 272         result.insert(not_visited.begin(), not_visited.end());
 273 }
