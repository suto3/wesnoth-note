  36 static void validate(boost::any& v, const std::vector<std::string>& values,
  37               two_strings*, int)
  38 {
  39         two_strings ret_val;
  40         if (values.size() != 2) {
  41                 throw po::validation_error(po::validation_error::invalid_option_value);
  42         }
  43         ret_val.first = values.at(0);
  44         ret_val.second = values.at(1);
  45         v = ret_val;
  46 }

 479 void commandline_options::parse_log_domains_(const std::string &domains_string, const int severity)
 480 {
 481         const std::vector<std::string> domains = utils::split(domains_string, ',');
 482         for (const std::string& domain : domains)
 483         {
 484                 if (!log)
 485                         log = std::vector<std::pair<int, std::string> >();
 486                 log->emplace_back(severity, domain);
 487         }

 490 void commandline_options::parse_log_strictness (const std::string & severity ) {
 491         static lg::logger const *loggers[] { &lg::err(), &lg::warn(), &lg::info(), &lg::debug() };
 492         for (const lg::logger * l : loggers ) {
 493                 if (severity == l->get_name()) {
 494                         lg::set_strict_severity(*l);
 495                         return ;
 496                 }
 497         }
 498         std::cerr << "Unrecognized argument to --log-strict : " << severity << " . \nDisabling strict mode logging." << std::endl;
 499         lg::set_strict_severity(-1);
 500 }

 502 void commandline_options::parse_resolution_ ( const std::string& resolution_string )
 503 {
 504         const std::vector<std::string> tokens = utils::split(resolution_string, 'x');
 505         if (tokens.size() != 2) {
 506                 throw bad_commandline_resolution(resolution_string);
 507         }
 508 
 509         int xres, yres;
 510 
 511         try {
 512                 xres = std::stoi(tokens[0]);
 513                 yres = std::stoi(tokens[1]);
 514         } catch(std::invalid_argument &) {
 515                 throw bad_commandline_resolution(resolution_string);
 516         }
 517 
 518         resolution = std::make_pair(xres, yres);
 519 }

 521 std::vector<std::pair<unsigned int,std::string> > commandline_options::parse_to_uint_string_tuples_(const std::vector<std::string> &strings, char separator)
 522 {
 523         std::vector<std::pair<unsigned int,std::string> > vec;
 524         const std::string& expected_format
 525                         = std::string() + "UINT" + separator + "STRING";
 526 
 527         for (const std::string &s : strings)
 528         {
 529                 const std::vector<std::string> tokens = utils::split(s, separator);
 530                 if(tokens.size() != 2) {
 531                         throw bad_commandline_tuple(s, expected_format);
 532                 }
 533 
 534                 unsigned int temp;
 535                 try {
 536                         temp = lexical_cast<unsigned int>(tokens[0]);
 537                 } catch (bad_lexical_cast &) {
 538                         throw bad_commandline_tuple(s, expected_format);
 539                 }
 540 
 541                 vec.emplace_back(temp, tokens[1]);
 542         }
 543         return vec;
 544 }

 546 std::vector<std::tuple<unsigned int,std::string,std::string> > commandline_options::parse_to_uint_string_string_tuples_(const std::vector<std::string> &strings, char separator)
 547 {
 548         std::vector<std::tuple<unsigned int,std::string,std::string> > vec;
 549         const std::string& expected_format
 550                         = std::string() + "UINT" + separator + "STRING" + separator + "STRING";
 551 
 552         for (const std::string &s : strings)
 553         {
 554                 const std::vector<std::string> tokens = utils::split(s, separator);
 555                 if(tokens.size() != 3) {
 556                         throw bad_commandline_tuple(s, expected_format);
 557                 }
 558 
 559                 unsigned int temp;
 560                 try {
 561                         temp = lexical_cast<unsigned int>(tokens[0]);
 562                 } catch (bad_lexical_cast &) {
 563                         throw bad_commandline_tuple(s, expected_format);
 564                 }
 565 
 566                 vec.emplace_back(temp, tokens[1], tokens[2]);
 567         }
 568         return vec;
 569 }
