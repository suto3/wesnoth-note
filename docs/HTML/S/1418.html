  19 /**
  20  * The main running point of the UMC IDE

  22 public class WesnothApplication implements IApplication

  24     @Override
  25     public Object start( IApplicationContext context )
  26     {
  27         Display display = PlatformUI.createDisplay( );
  28         Logger.getInstance( ).startLogger( );
  29         try {
  30             int returnCode = PlatformUI.createAndRunWorkbench( display,
  31                 new WesnothWorkbenchAdvisor( ) );
  32             if( returnCode == PlatformUI.RETURN_RESTART ) {
  33                 return IApplication.EXIT_RESTART;
  34             }
  35             else {
  36                 return IApplication.EXIT_OK;
  37             }
  38         } catch( Exception e ) {
  39             Logger.getInstance( ).logException( e );
  40             return IApplication.EXIT_OK;
  41         } finally {
  42             display.dispose( );
  43         }
  44     }
  45 
  46     @Override
  47     public void stop( )
  48     {
  49         if( ! PlatformUI.isWorkbenchRunning( ) ) {
  50             return;
  51         }
  52 
  53         final IWorkbench workbench = PlatformUI.getWorkbench( );
  54         final Display display = workbench.getDisplay( );
  55         display.syncExec( new Runnable( ) {
  56             @Override
  57             public void run( )
  58             {
  59                 if( ! display.isDisposed( ) ) {
  60                     workbench.close( );
  61                 }
  62             }
  63         } );
  64     }
  65 }
