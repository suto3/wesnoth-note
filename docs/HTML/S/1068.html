  54 /** Erases the item with id @a key. */
  55 wmi_container::size_type wmi_container::erase(const std::string & id)
  56 {
  57         // Locate the item to erase.
  58         const map_t::iterator iter = wml_menu_items_.find(id);
  59 
  60         if ( iter == wml_menu_items_.end() ) {
  61                 WRN_NG << "Trying to remove non-existent menu item '" << id << "'; ignoring." << std::endl;
  62                 // No such item.
  63                 return 0;
  64         }
  65 
  66         // Clean up our bookkeeping.
  67         iter->second->finish_handler();
  68 
  69         // Remove the item from the map.
  70         wml_menu_items_.erase(iter);
  71 
  72         return 1; // Erased one item.
  73 }

  75 /**
  76  * Fires the menu item with the given @a id.
  77  * @returns true if a matching item was found (even if it could not be fired).
  78  * NOTE: The return value could be altered if it is decided that
  79  * play_controller::execute_command() needs something different.
  80  */
  81 bool wmi_container::fire_item(const std::string & id, const map_location & hex, game_data & gamedata, filter_context & fc, unit_map & units) const
  82 {
  83         // Does this item exist?
  84         const_iterator iter = find(id);
  85         if ( iter == end() ) {
  86                 return false;
  87         }
  88         const wml_menu_item & wmi = **iter;
  89 
  90         // Prepare for can show().
  91         gamedata.get_variable("x1") = hex.wml_x();
  92         gamedata.get_variable("y1") = hex.wml_y();
  93         scoped_xy_unit highlighted_unit("unit", hex, units);
  94 
  95         // Can this item be shown?
  96         if ( wmi.can_show(hex, gamedata, fc) ) {
  97                 wmi.fire_event(hex, gamedata);
  98         }
  99         return true;
 100 }

 102 /**
 103  * Returns the menu items that can be shown for the given location.
 104  *
 105  * @param[out] items        Pointers to applicable menu items will be pushed onto @a items.
 106  * @param[out] descriptions Menu item text will be pushed onto @a descriptions (in the same order as @a items).
 107  */
 108 void wmi_container::get_items(const map_location& hex,
 109                               std::vector<std::shared_ptr<const wml_menu_item>>& items,
 110                               std::vector<config>& descriptions,
 111                               filter_context& fc, game_data& gamedata, unit_map& units) const
 112 {
 113         if ( empty() ) {
 114                 // Nothing to do (skip setting game variables).
 115                 return;
 116         }
 117 
 118         // Prepare for can show().
 119         gamedata.get_variable("x1") = hex.wml_x();
 120         gamedata.get_variable("y1") = hex.wml_y();
 121         scoped_xy_unit highlighted_unit("unit", hex, units);
 122 
 123         // Check each menu item.
 124         for (const item_ptr & item : *this)
 125         {
 126                 // Can this item be shown?
 127                 if ( item->use_wml_menu() && (!item->is_synced() || resources::controller->can_use_synced_wml_menu()) && item->can_show(hex, gamedata, fc) )
 128                 {
 129                         // Include this item.
 130                         items.push_back(item);
 131                         descriptions.emplace_back(config_of("id", item->menu_text()));
 132                 }
 133         }
 134         return;
 135 }

 137 /**
 138  * Initializes the implicit event handlers for inlined [command]s.
 139  */
 140 void wmi_container::init_handlers() const
 141 {
 142         // Applying default hotkeys here currently does not work because
 143         // the hotkeys are reset by play_controler::init_managers() ->
 144         // display_manager::display_manager, which is called after this.
 145         // The result is that default wml hotkeys will be ignored if wml
 146         // hotkeys are set to default in the preferences menu. (They are
 147         // still reapplied if set_menu_item is called again, for example
 148         // by starting a new campaign.) Since it isn't that important
 149         // I'll just leave it for now.
 150 
 151         unsigned wmi_count = 0;
 152 
 153         // Loop through each menu item.
 154         for (const item_ptr & wmi : *this) {
 155                 // If this menu item has a [command], add a handler for it.
 156                 wmi->init_handler();
 157                 // Count the menu items (for the diagnostic message).
 158                 ++wmi_count;
 159         }
 160 
 161         // Diagnostic:
 162         if ( wmi_count > 0 ) {
 163                 LOG_NG << wmi_count << " WML menu items found, loaded." << std::endl;
 164         }
 165 }

 167 void wmi_container::to_config(config& cfg) const
 168 {
 169         // Loop through our items.
 170         for (const item_ptr & item : *this)
 171         {
 172                 // Add this item as a child of cfg.
 173                 item->to_config(cfg.add_child("menu_item"));
 174         }

 177 /**
 178  * Updates or creates (as appropriate) the menu item with the given @a id.
 179  */
 180 void wmi_container::set_item(const std::string& id, const vconfig& menu_item)
 181 {
 182         // Try to insert a dummy value. This combines looking for an existing
 183         // entry with insertion.
 184         map_t::iterator add_it = wml_menu_items_.insert(map_t::value_type(id, item_ptr())).first;
 185 
 186         if ( add_it->second )
 187                 // Create a new menu item based on the old. This leaves the old item
 188                 // alone in case someone else is holding on to (and processing) it.
 189                 add_it->second.reset(new wml_menu_item(id, menu_item, *add_it->second));
 190         else
 191                 // This is a new menu item.
 192                 add_it->second.reset(new wml_menu_item(id, menu_item));
 193 }

 195 /**
 196  * Sets the current menu items to the "menu_item" children of @a cfg.
 197  */
 198 void wmi_container::set_menu_items(const config& cfg)
 199 {
 200         wml_menu_items_.clear();
 201         for (const config &item : cfg.child_range("menu_item"))
 202         {
 203                 if(!item.has_attribute("id")){ continue; }
 204 
 205                 std::string id = item["id"];
 206                 item_ptr & mref = wml_menu_items_[id];
 207                 if ( !mref ) {
 208                         mref.reset(new wml_menu_item(id, item));
 209                 } else {
 210                         WRN_NG << "duplicate menu item (" << id << ") while loading from config" << std::endl;
 211                 }
 212         }
 213 }
