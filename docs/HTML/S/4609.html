 502     # The variable "Caladon_leader_status" can be one of several values:
 503     # <empty>:   Caladon is not a leader. He is still a unit on side 1
 504     #
 505     # "passive": His side still has "controller=null" set. (He is
 506     # giving the mermen time to leave before attacking them.)
 507     #
 508     # "impatient": Caladon’s side now has "controller=ai" set. (He's
 509     # tired of waiting for the mermen to leave.) This is actually
 510     # *set* on the players turn, because Caladon’s side doesn't *get*
 511     # a turn until it is set. On his first turn after "impatient" is
 512     # set, he will say something and summon fire guardians.
 513     #
 514     # "angry": Player attacked Caladon. (...and Caladon summoned the
 515     # first fire guardians immediately, so he no longer has to on his
 516     # turn.)
 517     #
 518     # "leader": He is a normal leader now. He has started his turns,
 519     # and is summoning two fire guardians per turn.
 520     #
 521     # "dead":     Keeps any of the other events from happening.
 522

 524     [modify_side]
 525         side=4
 526         controller=ai
 527     [/modify_side]
