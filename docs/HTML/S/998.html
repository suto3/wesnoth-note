 124 bool affects_side(const config& cfg, const std::vector<team>& teams, size_t side, size_t other_side)
 125 {
 126         if (side == other_side)
 127                 return cfg["affect_allies"].to_bool(true);
 128         if (teams[side - 1].is_enemy(other_side))
 129                 return cfg["affect_enemies"].to_bool();
 130         else
 131                 return cfg["affect_allies"].to_bool();

 137 bool unit::get_ability_bool(const std::string& tag_name, const map_location& loc, const display_context& dc) const
 138 {
 139         for (const config &i : this->abilities_.child_range(tag_name)) {
 140                 if (ability_active(tag_name, i, loc) &&
 141                         ability_affects_self(tag_name, i, loc))
 142                 {
 143                         return true;
 144                 }
 145         }
 146 
 147         assert(resources::units);
 148         const unit_map& units = *resources::units;
 149 
 150         map_location adjacent[6];
 151         get_adjacent_tiles(loc,adjacent);
 152         for(int i = 0; i != 6; ++i) {
 153                 const unit_map::const_iterator it = units.find(adjacent[i]);
 154                 if (it == units.end() || it->incapacitated())
 155                         continue;
 156                 // Abilities may be tested at locations other than the unit's current
 157                 // location. This is intentional to allow for less messing with the unit
 158                 // map during calculations, particularly with regards to movement.
 159                 // Thus, we need to make sure the adjacent unit (*it) is not actually
 160                 // ourself.
 161                 if ( &*it == this )
 162                         continue;
 163                 for (const config &j : it->abilities_.child_range(tag_name)) {
 164                         if (affects_side(j, dc.teams(), side(), it->side()) &&
 165                             it->ability_active(tag_name, j, adjacent[i]) &&
 166                             ability_affects_adjacent(tag_name,  j, i, loc, *it))
 167                         {
 168                                 return true;
 169                         }
 170                 }
 171         }
 172 
 173 
 174         return false;
 175 }

 303 bool unit::ability_active(const std::string& ability,const config& cfg,const map_location& loc) const
 304 {
 305         bool illuminates = ability == "illuminates";
 306 
 307         if (const config &afilter = cfg.child("filter"))
 308                 if ( !unit_filter(vconfig(afilter), resources::filter_con, illuminates).matches(*this, loc) )
 309                         return false;
 310 
 311         map_location adjacent[6];
 312         get_adjacent_tiles(loc,adjacent);
 313 
 314         assert(resources::units);
 315         const unit_map& units = *resources::units;
 316 
 317         for (const config &i : cfg.child_range("filter_adjacent"))
 318         {
 319                 size_t count = 0;
 320                 const unit_filter ufilt(vconfig(i), resources::filter_con, illuminates);
 321                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 322                 for (const map_location::DIRECTION index : dirs)
 323                 {
 324                         if (index == map_location::NDIRECTIONS)
 325                                 continue;
 326                         unit_map::const_iterator unit = units.find(adjacent[index]);
 327                         if (unit == units.end())
 328                                 return false;
 329                         if (!ufilt(*unit, *this))
 330                                 return false;
 331                         if (i.has_attribute("is_enemy")) {
 332                                 const display_context& dc = resources::filter_con->get_disp_context();
 333                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(side_)) {
 334                                         continue;
 335                                 }
 336                         }
 337                         count++;
 338                 }
 339                 if (i["count"].empty() && count != dirs.size()) {
 340                         return false;
 341                 }
 342                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 343                         return false;
 344                 }
 345         }
 346 
 347         for (const config &i : cfg.child_range("filter_adjacent_location"))
 348         {
 349                 size_t count = 0;
 350                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 351                 adj_filter.flatten(illuminates);
 352 
 353                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 354                 for (const map_location::DIRECTION index : dirs)
 355                 {
 356                         if (index == map_location::NDIRECTIONS) {
 357                                 continue;
 358                         }
 359                         if(!adj_filter.match(adjacent[index])) {
 360                                 return false;
 361                         }
 362                         count++;
 363                 }
 364                 if (i["count"].empty() && count != dirs.size()) {
 365                         return false;
 366                 }
 367                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 368                         return false;
 369                 }
 370         }
 371         return true;
 372 }

 374 bool unit::ability_affects_adjacent(const std::string& ability, const config& cfg,int dir,const map_location& loc,const unit& from) const
 375 {
 376         bool illuminates = ability == "illuminates";
 377 
 378         assert(dir >=0 && dir <= 5);
 379         map_location::DIRECTION direction = static_cast<map_location::DIRECTION>(dir);
 380 
 381         for (const config &i : cfg.child_range("affect_adjacent"))
 382         {
 383                 if (i.has_attribute("adjacent")) { //key adjacent defined
 384                         std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 385                         if (std::find(dirs.begin(), dirs.end(), direction) == dirs.end()) {
 386                                 continue;
 387                         }
 388                 }
 389                 const config &filter = i.child("filter");
 390                 if (!filter || //filter tag given
 391                         unit_filter(vconfig(filter), resources::filter_con, illuminates).matches(*this, loc, from) ) {
 392                         return true;
 393                 }
 394         }
 395         return false;
 396 }

 414 template<typename T, typename TFuncFormula>
 415 class get_ability_value_visitor : public boost::static_visitor<T>
 416 {
 417 public:
 418         // Constructor stores the default value.
 419         get_ability_value_visitor(T def, const TFuncFormula& formula_handler) : def_(def), formula_handler_(formula_handler) {}
 420 
 421         T operator()(boost::blank const &) const { return def_; }
 422         T operator()(bool)                 const { return def_; }
 423         T operator()(int i)                const { return static_cast<T>(i); }
 424         T operator()(unsigned long long u) const { return static_cast<T>(u); }
 425         T operator()(double d)             const { return static_cast<T>(d); }
 426         T operator()(t_string const &)     const { return def_; }
 427         T operator()(const std::string& s) const
 428         {
 429                 if(s.size() >= 2 && s[0] == '(') {
 430                         return formula_handler_(s);
 431                 }
 432                 return lexical_cast_default<T>(s, def_);
 433         }
 434 
 435 private:
 436         const T def_;
 437         const TFuncFormula& formula_handler_;
 438 };
 439 template<typename T, typename TFuncFormula>
 440 get_ability_value_visitor<T, TFuncFormula> make_get_ability_value_visitor(T def, const TFuncFormula& formula_handler)
 441 {
 442         return get_ability_value_visitor<T, TFuncFormula>(def, formula_handler);

 444 template<typename T, typename TFuncFormula>
 445 T get_single_ability_value(const config::attribute_value& v, T def, const map_location& sender_loc, const map_location& receiver_loc, const TFuncFormula& formula_handler)
 446 {
 447         return v.apply_visitor(make_get_ability_value_visitor(def, [&](const std::string& s) {
 448                 
 449                         try {
 450                                 assert(resources::units);
 451                                 auto u_itor = resources::units->find(sender_loc);
 452 
 453                                 if(u_itor == resources::units->end()) {
 454                                         return def;
 455                                 }
 456                                 wfl::map_formula_callable callable(std::make_shared<wfl::unit_callable>(*u_itor));
 457                                 u_itor = resources::units->find(receiver_loc);
 458                                 if(u_itor != resources::units->end()) {
 459                                         callable.add("other", wfl::variant(std::make_shared<wfl::unit_callable>(*u_itor)));
 460                                 }
 461                                 return formula_handler(wfl::formula(s), callable);
 462                         } catch(wfl::formula_error& e) {
 463                                 lg::wml_error() << "Formula error in ability or weapon special: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 464                                 return def;
 465                         }
 466         }));
 467 }

 470 template<typename TComp>
 471 std::pair<int,map_location> unit_ability_list::get_extremum(const std::string& key, int def, const TComp& comp) const
 472 {
 473         if ( cfgs_.empty() ) {
 474                 return std::make_pair(def, map_location());
 475         }
 476         // The returned location is the best non-cumulative one, if any,
 477         // the best absolute cumulative one otherwise.
 478         map_location best_loc;
 479         bool only_cumulative = true;
 480         int abs_max = 0;
 481         int flat = 0;
 482         int stack = 0;
 483         for (unit_ability const &p : cfgs_)
 484         {
 485                 int value = get_single_ability_value((*p.first)[key], def, p.second, loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
 486                         return formula.evaluate(callable).as_int();
 487                 });
 488 
 489                 if ((*p.first)["cumulative"].to_bool()) {
 490                         stack += value;
 491                         if (value < 0) value = -value;
 492                         if (only_cumulative && !comp(value, abs_max)) {
 493                                 abs_max = value;
 494                                 best_loc = p.second;
 495                         }
 496                 } else if (only_cumulative || comp(flat, value)) {
 497                         only_cumulative = false;
 498                         flat = value;
 499                         best_loc = p.second;
 500                 }
 501         }
 502         return std::make_pair(flat + stack, best_loc);
 503 }

 538         /**
 539          * Gets the children of @parent (which should be the specials for an
 540          * attack_type) and places the ones whose tag or id= matches @a id into
 541          * @a result.
 542          * If @a just_peeking is set to true, then @a result is not touched;
 543          * instead the return value is used to indicate if any matching children
 544          * were found.
 545          *
 546          * @returns  true if @a just_peeking is true and a match was found;
 547          *           false otherwise.
 548          */
 549         bool get_special_children(std::vector<const config*>& result, const config& parent,
 550                                    const std::string& id, bool just_peeking=false) {
 551                 for (const config::any_child &sp : parent.all_children_range())
 552                 {
 553                         if (sp.key == id || sp.cfg["id"] == id) {
 554                                 if(just_peeking) {
 555                                         return true; // peek succeeded; done
 556                                 } else {
 557                                         result.push_back(&sp.cfg);
 558                                 }
 559                         }
 560                 }
 561                 return false;
 562         }
 563 }

 565 /**
 566  * Returns whether or not @a *this has a special with a tag or id equal to
 567  * @a special. If @a simple_check is set to true, then the check is merely
 568  * for being present. Otherwise (the default), the check is for a special
 569  * active in the current context (see set_specials_context), including
 570  * specials obtained from the opponent's attack.
 571  */
 572 bool attack_type::get_special_bool(const std::string& special, bool simple_check) const
 573 {
 574         {
 575                 std::vector<const config*> list;
 576                 if ( get_special_children(list, specials_, special, simple_check) ) {
 577                         return true;
 578                 }
 579                 // If we make it to here, then either list.empty() or !simple_check.
 580                 // So if the list is not empty, then this is not a simple check and
 581                 // we need to check each special in the list to see if any are active.
 582                 for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 583                         if ( special_active(**i, AFFECT_SELF) ) {
 584                                 return true;
 585                         }
 586                 }
 587         }
 588         // Skip checking the opponent's attack?
 589         if ( simple_check || !other_attack_ ) {
 590                 return false;
 591         }
 592 
 593         std::vector<const config*> list;
 594         get_special_children(list, other_attack_->specials_, special);
 595         for (std::vector<const config*>::iterator i = list.begin(), i_end = list.end(); i != i_end; ++i) {
 596                 if ( other_attack_->special_active(**i, AFFECT_OTHER) ) {
 597                         return true;
 598                 }
 599         }
 600         return false;
 601 }

 603 /**
 604  * Returns the currently active specials as an ability list, given the current
 605  * context (see set_specials_context).
 606  */
 607 unit_ability_list attack_type::get_specials(const std::string& special) const
 608 {
 609         //log_scope("get_specials");
 610         unit_ability_list res(self_loc_);
 611         for (const config &i : specials_.child_range(special)) {
 612                 if ( special_active(i, AFFECT_SELF) )
 613                         res.push_back(unit_ability(&i, self_loc_));
 614         }
 615         if (!other_attack_) return res;
 616         for (const config &i : other_attack_->specials_.child_range(special)) {
 617                 if ( other_attack_->special_active(i, AFFECT_OTHER) )
 618                         res.push_back(unit_ability(&i, other_loc_));
 619         }
 620         return res;
 621 }

 623 /**
 624  * Returns a vector of names and descriptions for the specials of *this.
 625  * Each std::pair in the vector has first = name and second = description.
 626  *
 627  * This uses either the active or inactive name/description for each special,
 628  * based on the current context (see set_specials_context), provided
 629  * @a active_list is not nullptr. Otherwise specials are assumed active.
 630  * If the appropriate name is empty, the special is skipped.
 631  */
 632 std::vector<std::pair<t_string, t_string> > attack_type::special_tooltips(
 633         boost::dynamic_bitset<>* active_list) const
 634 {
 635         //log_scope("special_tooltips");
 636         std::vector<std::pair<t_string, t_string> > res;
 637         if ( active_list )
 638                 active_list->clear();
 639 
 640         for (const config::any_child &sp : specials_.all_children_range())
 641         {
 642                 if ( !active_list || special_active(sp.cfg, AFFECT_EITHER) ) {
 643                         const t_string &name = sp.cfg["name"];
 644                         if (!name.empty()) {
 645                                 res.emplace_back(name, sp.cfg["description"].t_str() );
 646                                 if ( active_list )
 647                                         active_list->push_back(true);
 648                         }
 649                 } else {
 650                         t_string const &name = default_value(sp.cfg, "name_inactive", "name").t_str();
 651                         if (!name.empty()) {
 652                                 res.emplace_back(name, default_value(sp.cfg, "description_inactive", "description").t_str() );
 653                                 active_list->push_back(false);
 654                         }
 655                 }
 656         }
 657         return res;
 658 }

 729 /**
 730  * Calculates the number of attacks this weapon has, considering specials.
 731  * This returns two numbers because of the swarm special. The actual number of
 732  * attacks depends on the unit's health and should be:
 733  *   min_attacks + (max_attacks - min_attacks) * (current hp) / (max hp)
 734  * c.f. swarm_blows()
 735  */
 736 void attack_type::modified_attacks(bool is_backstab, unsigned & min_attacks,
 737                                    unsigned & max_attacks) const
 738 {
 739         // Apply [attacks].
 740         unit_abilities::effect attacks_effect(get_specials("attacks"),
 741                                               num_attacks(), is_backstab);
 742         int attacks_value = attacks_effect.get_composite_value();
 743         if ( attacks_value < 0 ) {
 744                 attacks_value = num_attacks();
 745                 ERR_NG << "negative number of strikes after applying weapon specials" << std::endl;
 746         }
 747 
 748         // Apply [swarm].
 749         unit_ability_list swarm_specials = get_specials("swarm");
 750         if ( !swarm_specials.empty() ) {
 751                 min_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_min").first);
 752                 max_attacks = std::max<int>(0, swarm_specials.highest("swarm_attacks_max", attacks_value).first);
 753         } else {
 754                 min_attacks = max_attacks = attacks_value;
 755         }
 756 }

 782                         return false;

 786                         return true;
 787                 if ( is_attacker  &&  apply_to == "defender" )

 790                         return true;

 804                         return true;

 816         /**
 817          * Determines if a unit/weapon combination matches the specified child
 818          * (normally a [filter_*] child) of the provided filter.
 819          * @param[in]  un_it      The unit to filter.
 820          * @param[in]  loc        The presumed location of @a un_it.
 821          * @param[in]  weapon     The attack_type to filter.
 822          * @param[in]  filter     The filter containing the child filter to use.
 823          * @param[in]  child_tag  The tag of the child filter to use.
 824          */
 825         static bool special_unit_matches(const unit_map::const_iterator & un_it,
 826                                                                          const unit_map::const_iterator & u2,
 827                                              const map_location & loc,
 828                                              const_attack_ptr weapon,
 829                                              const config & filter,
 830                                                                          const bool for_listing,
 831                                              const std::string & child_tag)
 832         {
 833                 if (for_listing && !loc.valid())
 834                         // The special's context was set to ignore this unit, so assume we pass.
 835                         // (This is used by reports.cpp to show active specials when the
 836                         // opponent is not known. From a player's perspective, the special
 837                         // is active, in that it can be used, even though the player might
 838                         // need to select an appropriate opponent.)
 839                         return true;
 840 
 841                 const config & filter_child = filter.child(child_tag);
 842                 if ( !filter_child )
 843                         // The special does not filter on this unit, so we pass.
 844                         return true;
 845 
 846                 // If the primary unit doesn't exist, there's nothing to match
 847                 if (!un_it.valid()) {
 848                         return false;
 849                 }
 850 
 851                 unit_filter ufilt(vconfig(filter_child), resources::filter_con);
 852 
 853                 // If the other unit doesn't exist, try matching without it
 854                 if (!u2.valid()) {
 855                         return ufilt.matches(*un_it, loc);
 856                 }
 857 
 858                 // Check for a unit match.
 859                 if (!ufilt.matches(*un_it, loc, *u2)) {
 860                         return false;
 861                 }
 862 
 863                 // Check for a weapon match.
 864                 if ( const config & filter_weapon = filter_child.child("filter_weapon") ) {
 865                         if ( !weapon || !weapon->matches_filter(filter_weapon) )
 866                                 return false;
 867                 }
 868 
 869                 // Passed.
 870                 return true;
 871         }

 875 /**
 876  * Returns whether or not the given special is active for the specified unit,
 877  * based on the current context (see set_specials_context).
 878  * @param[in] special           a weapon special WML structure
 879  * @param[in] whom              specifies which combatant we care about
 880  * @param[in] include_backstab  false if backstab specials should not be active
 881  *                              (usually true since backstab is usually accounted
 882  *                              for elsewhere)
 883  */
 884 bool attack_type::special_active(const config& special, AFFECTS whom,
 885                                  bool include_backstab) const
 886 {
 887         //log_scope("special_active");
 888 
 889         // Backstab check
 890         if ( !include_backstab )
 891                 if ( special["backstab"].to_bool() )
 892                         return false;
 893 
 894         // Does this affect the specified unit?
 895         if ( whom == AFFECT_SELF ) {
 896                 if ( !special_affects_self(special, is_attacker_) )
 897                         return false;
 898         }
 899         if ( whom == AFFECT_OTHER ) {
 900                 if ( !special_affects_opponent(special, is_attacker_) )
 901                         return false;
 902         }
 903 
 904         // Is this active on attack/defense?
 905         const std::string & active_on = special["active_on"];
 906         if ( !active_on.empty() ) {
 907                 if ( is_attacker_  &&  active_on != "offense" )
 908                         return false;
 909                 if ( !is_attacker_  &&  active_on != "defense" )
 910                         return false;
 911         }
 912 
 913         // Get the units involved.
 914         assert(resources::units);
 915         const unit_map& units = *resources::units;
 916 
 917         unit_map::const_iterator self  = units.find(self_loc_);
 918         unit_map::const_iterator other = units.find(other_loc_);
 919 
 920         // Make sure they're facing each other.
 921         temporary_facing self_facing(self, self_loc_.get_relative_dir(other_loc_));
 922         temporary_facing other_facing(other, other_loc_.get_relative_dir(self_loc_));
 923 
 924         // Translate our context into terms of "attacker" and "defender".
 925         unit_map::const_iterator & att = is_attacker_ ? self : other;
 926         unit_map::const_iterator & def = is_attacker_ ? other : self;
 927         const map_location & att_loc   = is_attacker_ ? self_loc_ : other_loc_;
 928         const map_location & def_loc   = is_attacker_ ? other_loc_ : self_loc_;
 929         const_attack_ptr att_weapon = is_attacker_ ? shared_from_this() : other_attack_;
 930         const_attack_ptr def_weapon = is_attacker_ ? other_attack_ : shared_from_this();
 931 
 932         // Filter the units involved.
 933         if (!special_unit_matches(self, other, self_loc_, shared_from_this(), special, is_for_listing_, "filter_self"))
 934                 return false;
 935         if (!special_unit_matches(other, self, other_loc_, other_attack_, special, is_for_listing_, "filter_opponent"))
 936                 return false;
 937         if (!special_unit_matches(att, def, att_loc, att_weapon, special, is_for_listing_, "filter_attacker"))
 938                 return false;
 939         if (!special_unit_matches(def, att, def_loc, def_weapon, special, is_for_listing_, "filter_defender"))
 940                 return false;
 941 
 942         map_location adjacent[6];
 943         get_adjacent_tiles(self_loc_, adjacent);
 944 
 945         // Filter the adjacent units.
 946         for (const config &i : special.child_range("filter_adjacent"))
 947         {
 948                 size_t count = 0;
 949                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 950                 unit_filter filter(vconfig(i), resources::filter_con);
 951                 for (const map_location::DIRECTION index : dirs)
 952                 {
 953                         if (index == map_location::NDIRECTIONS)
 954                                 continue;
 955                         unit_map::const_iterator unit = units.find(adjacent[index]);
 956                         if (unit == units.end() || !filter.matches(*unit, adjacent[index], *self))
 957                                 return false;
 958                         if (i.has_attribute("is_enemy")) {
 959                                 const display_context& dc = resources::filter_con->get_disp_context();
 960                                 if (i["is_enemy"].to_bool() != dc.get_team(unit->side()).is_enemy(self->side())) {
 961                                         continue;
 962                                 }
 963                         }
 964                         count++;
 965                 }
 966                 if (i["count"].empty() && count != dirs.size()) {
 967                         return false;
 968                 }
 969                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 970                         return false;
 971                 }
 972         }
 973 
 974         // Filter the adjacent locations.
 975         for (const config &i : special.child_range("filter_adjacent_location"))
 976         {
 977                 size_t count = 0;
 978                 std::vector<map_location::DIRECTION> dirs = map_location::parse_directions(i["adjacent"]);
 979                 terrain_filter adj_filter(vconfig(i), resources::filter_con);
 980                 for (const map_location::DIRECTION index : dirs)
 981                 {
 982                         if (index == map_location::NDIRECTIONS)
 983                                 continue;
 984                         if(!adj_filter.match(adjacent[index])) {
 985                                 return false;
 986                         }
 987                         count++;
 988                 }
 989                 if (i["count"].empty() && count != dirs.size()) {
 990                         return false;
 991                 }
 992                 if (!in_ranges<int>(count, utils::parse_ranges(i["count"].str()))) {
 993                         return false;
 994                 }
 995         }
 996 
 997         return true;
 998 }

1013 bool filter_base_matches(const config& cfg, int def)
1014 {
1015         if (const config &apply_filter = cfg.child("filter_base_value")) {
1016                 config::attribute_value cond_eq = apply_filter["equals"];
1017                 config::attribute_value cond_ne = apply_filter["not_equals"];
1018                 config::attribute_value cond_lt = apply_filter["less_than"];
1019                 config::attribute_value cond_gt = apply_filter["greater_than"];
1020                 config::attribute_value cond_ge = apply_filter["greater_than_equal_to"];
1021                 config::attribute_value cond_le = apply_filter["less_than_equal_to"];
1022                 return  (cond_eq.empty() || def == cond_eq.to_int()) &&
1023                         (cond_ne.empty() || def != cond_ne.to_int()) &&
1024                         (cond_lt.empty() || def <  cond_lt.to_int()) &&
1025                         (cond_gt.empty() || def >  cond_gt.to_int()) &&
1026                         (cond_ge.empty() || def >= cond_ge.to_int()) &&
1027                         (cond_le.empty() || def <= cond_le.to_int());
1028         }
1029         return true;
1030 }
1031 
1032 effect::effect(const unit_ability_list& list, int def, bool backstab) :
1033         effect_list_(),
1034         composite_value_(0)
1035 {
1036 
1037         int value_set = def;
1038         bool value_is_set = false;
1039         std::map<std::string,individual_effect> values_add;
1040         std::map<std::string,individual_effect> values_mul;
1041         std::map<std::string,individual_effect> values_div;
1042 
1043         individual_effect set_effect;
1044 
1045         for (const unit_ability & ability : list) {
1046                 const config& cfg = *ability.first;
1047                 const std::string& effect_id = cfg[cfg["id"].empty() ? "name" : "id"];
1048 
1049                 if (!cfg["backstab"].blank()) {
1050                         lg::wml_error() << "The backstab= key in weapon specials is deprecated; use [filter_adjacent] instead\n";
1051                 }
1052 
1053                 if (!backstab && cfg["backstab"].to_bool())
1054                         continue;
1055                 if (!filter_base_matches(cfg, def))
1056                         continue;
1057 
1058                 if (const config::attribute_value *v = cfg.get("value")) {
1059                         int value = get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1060                                 callable.add("base_value", wfl::variant(def));
1061                                 return formula.evaluate(callable).as_int();
1062                         });
1063         
1064                         bool cumulative = cfg["cumulative"].to_bool();
1065                         if (!value_is_set && !cumulative) {
1066                                 value_set = value;
1067                                 set_effect.set(SET, value, ability.first, ability.second);
1068                         } else {
1069                                 if (cumulative) value_set = std::max<int>(value_set, def);
1070                                 if (value > value_set) {
1071                                         value_set = value;
1072                                         set_effect.set(SET, value, ability.first, ability.second);
1073                                 }
1074                         }
1075                         value_is_set = true;
1076                 }
1077 
1078                 if (const config::attribute_value *v = cfg.get("add")) {
1079                         int add = get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1080                                 callable.add("base_value", wfl::variant(def));
1081                                 return formula.evaluate(callable).as_int();
1082                         });
1083                         std::map<std::string,individual_effect>::iterator add_effect = values_add.find(effect_id);
1084                         if(add_effect == values_add.end() || add > add_effect->second.value) {
1085                                 values_add[effect_id].set(ADD, add, ability.first, ability.second);
1086                         }
1087                 }
1088                 if (const config::attribute_value *v = cfg.get("sub")) {
1089                         int sub = - get_single_ability_value(*v, def, ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1090                                 callable.add("base_value", wfl::variant(def));
1091                                 return formula.evaluate(callable).as_int();
1092                         });
1093                         std::map<std::string,individual_effect>::iterator sub_effect = values_add.find(effect_id);
1094                         if(sub_effect == values_add.end() || sub > sub_effect->second.value) {
1095                                 values_add[effect_id].set(ADD, sub, ability.first, ability.second);
1096                         }
1097                 }
1098                 if (const config::attribute_value *v = cfg.get("multiply")) {
1099                         int multiply = static_cast<int>(get_single_ability_value(*v, static_cast<double>(def), ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1100                                 callable.add("base_value", wfl::variant(def));
1101                                 return formula.evaluate(callable).as_decimal() / 1000.0 ;
1102                         }) * 100);
1103                         std::map<std::string,individual_effect>::iterator mul_effect = values_mul.find(effect_id);
1104                         if(mul_effect == values_mul.end() || multiply > mul_effect->second.value) {
1105                                 values_mul[effect_id].set(MUL, multiply, ability.first, ability.second);
1106                         }
1107                 }
1108                 if (const config::attribute_value *v = cfg.get("divide")) {
1109                         int divide = static_cast<int>(get_single_ability_value(*v, static_cast<double>(def), ability.second, list.loc(),[&](const wfl::formula& formula, wfl::map_formula_callable& callable) {
1110                                 callable.add("base_value", wfl::variant(def));
1111                                 return formula.evaluate(callable).as_decimal() / 1000.0 ;
1112                         }) * 100);
1113 
1114                         if (divide == 0) {
1115                                 ERR_NG << "division by zero with divide= in ability/weapon special " << effect_id << std::endl;
1116                         }
1117                         else {
1118                                 std::map<std::string,individual_effect>::iterator div_effect = values_div.find(effect_id);
1119                                 if(div_effect == values_div.end() || divide > div_effect->second.value) {
1120                                         values_div[effect_id].set(DIV, divide, ability.first, ability.second);
1121                                 }
1122                         }
1123                 }
1124         }
1125 
1126         if(value_is_set && set_effect.type != NOT_USED) {
1127                 effect_list_.push_back(set_effect);
1128         }
1129 
1130         /* Do multiplication with floating point values rather than integers
1131          * We want two places of precision for each multiplier
1132          * Using integers multiplied by 100 to keep precision causes overflow
1133          *   after 3-4 abilities for 32-bit values and ~8 for 64-bit
1134          * Avoiding the overflow by dividing after each step introduces rounding errors
1135          *   that may vary depending on the order effects are applied
1136          * As the final values are likely <1000 (always true for mainline), loss of less significant digits is not an issue
1137          */
1138         double multiplier = 1.0;
1139         double divisor = 1.0;
1140         std::map<std::string,individual_effect>::const_iterator e, e_end;
1141         for (e = values_mul.begin(), e_end = values_mul.end(); e != e_end; ++e) {
1142                 multiplier *= e->second.value/100.0;
1143                 effect_list_.push_back(e->second);
1144         }
1145         for (e = values_div.begin(), e_end = values_div.end(); e != e_end; ++e) {
1146                 divisor *= e->second.value/100.0;
1147                 effect_list_.push_back(e->second);
1148         }
1149         int addition = 0;
1150         for (e = values_add.begin(), e_end = values_add.end(); e != e_end; ++e) {
1151                 addition += e->second.value;
1152                 effect_list_.push_back(e->second);
1153         }
1154 
1155         composite_value_ = int((value_set + addition) * multiplier / divisor);
1156 }
