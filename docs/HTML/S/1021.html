  26 void playturn_network_adapter::read_from_network()
  27 {
  28         assert(!data_.empty());
  29 
  30         this->data_.emplace_back();
  31         config& back = data_.back();
  32         bool has_data = false;
  33         try
  34         {
  35                 has_data = this->network_reader_(back);
  36         }
  37         catch(...)
  38         {
  39                 //Readin from network can throw, we want to ignore the possibly corrupt packet in this case.
  40                 this->data_.pop_back();
  41                 throw;
  42         }
  43         //ping is handeled by network.cpp and we can ignore it.
  44         back.remove_attribute("ping");
  45         if((!has_data) || back.empty())
  46         {
  47                 this->data_.pop_back();
  48                 return;
  49         }
  50         assert(!data_.back().empty());
  51 
  52         if(back.has_attribute("side_drop"))
  53         {
  54                 config child;
  55                 child["side_num"] = back["side_drop"];
  56                 child["controller"] = back["controller"];
  57                 this->data_.emplace_back(config {"side_drop", child});
  58                 back.remove_attribute("side_drop");
  59                 back.remove_attribute("controller");
  60         }
  61         else if(!back.attribute_range().empty() )
  62         {
  63                 ERR_NW << "found unexpected attribute:" <<back.debug() << std::endl;
  64                 this->data_.pop_back();
  65                 //ignore those here
  66         }
  67         assert(!data_.back().empty());
  68         //there should be no attributes left.
  69 }

  71 bool playturn_network_adapter::is_at_end()
  72 {
  73         assert(!data_.empty());
  74         if (data_.size() > 1) return false;
  75         return this->next_ == data_.back().ordered_end();
  76 }

  78 bool playturn_network_adapter::read(config& dst)
  79 {
  80         assert(dst.empty());
  81         if(is_at_end())
  82         {
  83                 read_from_network();
  84         }
  85         if(is_at_end())
  86         {
  87                 //that means we couldn't read anything from the network.
  88                 return false;
  89         }
  90         //skip empty data.
  91         while(next_ == data_.begin()->ordered_end())
  92         {
  93                 data_.pop_front();
  94                 next_ = data_.front().ordered_begin();
  95                 assert(!is_at_end());
  96         }
  97         config& child = dst.add_child(next_->key);
  98         config& child_old = next_->cfg;
  99         if(next_->key == "turn")
 100         {
 101                 //split [turn] indo different [turn] for each child.
 102                 assert(next_->cfg.all_children_count() > next_command_num_);
 103                 config::all_children_iterator itor = child_old.ordered_begin();
 104                 std::advance(itor, next_command_num_);
 105                 config& childchild_old = itor->cfg;
 106                 config& childchild = child.add_child(itor->key);
 107                 childchild.swap(childchild_old);
 108 
 109                 ++next_command_num_;
 110                 if(next_->cfg.all_children_count() == next_command_num_)
 111                 {
 112                         next_command_num_ = 0;
 113                         ++next_;
 114                 }
 115                 return true;
 116         }
 117         else
 118         {
 119                 child.swap(child_old);
 120                 ++next_;
 121                 return true;
 122         }
 123 }

 151 static bool read_config(config& src, config& dst)
 152 {
 153         assert(dst.empty());
 154         if(!src.empty())
 155         {
 156                 src.swap(dst);
 157                 return true;
 158         }
 159         else
 160         {
 161                 return false;
 162         }
 163 }
