  45 def wrap_elem(line):
  46     """If the supplied line contains spaces, return it wrapped between double quotes"""
  47     if ' ' in line:
  48         return "\"{0}\"".format(line)
  49     return line

  51 def run_tool(tool,queue,command):
  52     """Runs a maintenance tool with the desired arguments and pushes the output in the supplied queue"""
  53     # set the encoding for the subprocess
  54     # otherwise, with the new Unicode literals used by the wml tools,
  55     # we may get UnicodeDecodeErros
  56     env=os.environ
  57     env['PYTHONIOENCODING']="utf8"
  58     if sys.platform=="win32":
  59         # Windows wants a string, Linux wants a list and Polly wants a cracker
  60         # Windows wants also strings flavoured with double quotes
  61         # since maps return iterators, we must cast them as lists, otherwise join won't work
  62         # not doing this causes an OSError: [WinError 87]
  63         # this doesn't happen on Python 2.7, because here map() returns a list
  64         wrapped_line=list(map(wrap_elem,command))
  65         queue.put_nowait(' '.join(wrapped_line)+"\n")
  66         si=subprocess.STARTUPINFO()
  67         si.dwFlags=subprocess.STARTF_USESHOWWINDOW|subprocess.SW_HIDE # to avoid showing a DOS prompt
  68         try:
  69             output=subprocess.check_output(' '.join(wrapped_line),stderr=subprocess.STDOUT,startupinfo=si,env=env)
  70             queue.put_nowait(str(output, "utf8"))
  71         except subprocess.CalledProcessError as error:
  72             # post the precise message and the remaining output as a tuple
  73             queue.put_nowait((tool,error.returncode,str(error.output, "utf8")))
  74     else: # STARTUPINFO is not available, nor needed, outside of Windows
  75         queue.put_nowait(' '.join(command)+"\n")
  76         try:
  77             output=subprocess.check_output(command,stderr=subprocess.STDOUT,env=env)
  78             queue.put_nowait(str(output, "utf8"))
  79         except subprocess.CalledProcessError as error:
  80             # post the precise message and the remaining output as a tuple
  81             queue.put_nowait((tool,error.returncode, str(error.output, "utf8")))

  92 def get_addons_directory():
  93     """Returns a string containing the path of the add-ons directory"""
  94     # os.path.expanduser gets the current user's home directory on every platform
  95     if sys.platform=="win32":
  96         # get userdata directory on Windows
  97         # it assumes that you choose to store userdata in the My Games directory
  98         # while installing Wesnoth
  99         userdata=os.path.join(os.path.expanduser("~"),
 100                               "Documents",
 101                               "My Games",
 102                               "Wesnoth"+WESNOTH_SERIES,
 103                               "data",
 104                               "add-ons")
 105     elif sys.platform.startswith("linux") or "bsd" in sys.platform:
 106         # we're on Linux or on a BSD system like FreeBSD
 107         userdata=os.path.join(os.path.expanduser("~"),
 108                               ".local",
 109                               "share",
 110                               "wesnoth",
 111                               WESNOTH_SERIES,
 112                               "data",
 113                               "add-ons")
 114     elif sys.platform=="darwin": # we're on MacOS
 115         # bear in mind that I don't have a Mac, so this point may be bugged
 116         userdata=os.path.join(os.path.expanduser("~"),
 117                               "Library",
 118                               "Application Support",
 119                               "Wesnoth_"+WESNOTH_SERIES,
 120                               "data",
 121                               "add-ons")
 122     else: # unknown system; if someone else wants to add other rules, be my guest
 123         userdata="."
 124 
 125     return userdata if os.path.exists(userdata) else "." # fallback in case the directory doesn't exist

 127 def attach_context_menu(widget,function):
 128     # on Mac the right button fires a Button-2 event, or so I'm told
 129     # some mice don't even have two buttons, so the user is forced
 130     # to use Command + the only button
 131     # bear in mind that I don't have a Mac, so this point may be bugged
 132     # bind also the context menu key, for those keyboards that have it
 133     # that is, most of the Windows and Linux ones (however, in Win it's
 134     # called App, while on Linux is called Menu)
 135     # Mac doesn't have a context menu key on its keyboards, so no binding
 136     # bind also the Shift+F10 shortcut (same as Menu/App key)
 137     # the call to tk windowingsystem is justified by the fact
 138     # that it is possible to install X11 over Darwin
 139     windowingsystem = widget.tk.call('tk', 'windowingsystem')
 140     if windowingsystem == "win32": # Windows, both 32 and 64 bit
 141         widget.bind("<Button-3>", function)
 142         widget.bind("<KeyPress-App>", function)
 143         widget.bind("<Shift-KeyPress-F10>", function)
 144     elif windowingsystem == "aqua": # MacOS with Aqua
 145         widget.bind("<Button-2>", function)
 146         widget.bind("<Control-Button-1>", function)
 147     elif windowingsystem == "x11": # Linux, FreeBSD, Darwin with X11
 148         widget.bind("<Button-3>", function)
 149         widget.bind("<KeyPress-Menu>", function)
 150         widget.bind("<Shift-KeyPress-F10>", function)

 152 def attach_select_all(widget,function):

 163 class Tooltip(Toplevel):
 164     def __init__(self,widget,text,tag=None):
 165         """A tooltip, or balloon. Displays the specified help text when the
 166 mouse pointer stays on the widget for more than 500 ms."""
 167         # the master attribute retrieves the window where our "parent" widget is
 168         super().__init__(widget.master)
 169         self.widget=widget
 170         self.preshow_id=None
 171         self.show_id=None
 172         self.label=Label(self,
 173                          text=text,
 174                          background="#ffffe1", # background color used on Windows
 175                          borderwidth=1,
 176                          relief=SOLID,
 177                          padding=1,
 178                          # Tk has a bunch of predefined fonts
 179                          # use the one specific for tooltips
 180                          font=font.nametofont("TkTooltipFont"))
 181         self.label.pack()
 182         self.overrideredirect(True)
 183         # allow binding the tooltips to tagged elements of a widget
 184         # only Text, Canvas and Treeview support tags
 185         # and as such, they have a tag_bind method
 186         # if the widget doesn't support tags, bind directly to it
 187         if tag and hasattr(widget, "tag_bind") and callable(widget.tag_bind):
 188             self.widget.tag_bind(tag,"<Enter>",self.preshow)
 189             self.widget.tag_bind(tag,"<Leave>",self.hide)
 190         else:
 191             self.widget.bind("<Enter>",self.preshow)
 192             self.widget.bind("<Leave>",self.hide)
 193         self.widget.bind("<ButtonPress>",self.hide)
 194         self.widget.bind("<ButtonRelease>",self.hide)
 195         self.withdraw()
 196     def preshow(self,event=None):
 197         self.after_cleanup()
 198         # 500 ms and 5000 ms are the default values used on Windows
 199         self.preshow_id=self.after(500,self.show)
 200     def show(self):
 201         self.after_cleanup()
 202         # check if the tooltip will end up out of the screen
 203         # and handle this case if so
 204         screen_width = self.winfo_screenwidth()
 205         tooltip_width = self.winfo_reqwidth()
 206         if tooltip_width + self.winfo_pointerx() > screen_width:
 207             # unfortunately, it seems like Tkinter doesn't have a way to check the pointer's size
 208             # so I'm using a value of 20px, which is enough for the usual 16px pointers
 209             self.geometry("+%d+%d" % (screen_width-tooltip_width,self.winfo_pointery()+20))
 210         else:
 211             self.geometry("+%d+%d" % (self.winfo_pointerx(),self.winfo_pointery()+20))
 212         # update_idletasks forces a geometry update
 213         self.update_idletasks()
 214         self.state("normal")
 215         self.lift()
 216         self.show_id=self.after(5000, self.hide)
 217     def hide(self,event=None):
 218         self.after_cleanup()
 219         self.withdraw()
 220     def after_cleanup(self):
 221         # each event should cleanup after itself,
 222         # to avoid having two .after() calls conflicting
 223         # for example, one previously scheduled .after() may
 224         # try to hide the tooltip before five seconds are passed
 225         if self.show_id:
 226             self.after_cancel(self.show_id)
 227             self.show_id=None
 228         if self.preshow_id:
 229             self.after_cancel(self.preshow_id)
 230             self.preshow_id=None
 231     def set_text(self,text):
 232         self.label.configure(text=text)
 233         self.update_idletasks()

 235 class Popup(Toplevel):
 236     def __init__(self,parent,tool,thread):
 237         """Creates a popup that informs the user that the desired tool is running.
 238 Self destroys when the tool thread is over"""
 239         self.thread=thread
 240         super().__init__(parent)
 241         self.transient(parent)
 242         self.grab_set()
 243         self.protocol("WM_DELETE_WINDOW",
 244                       lambda: None) # disable close button
 245         self.resizable(width=False,
 246                        height=False)
 247         frame=Frame(self)
 248         frame.pack(fill=BOTH,expand=YES)
 249         wait_label=Label(frame,
 250                          text="{0} is running\nPlease wait...".format(tool),
 251                          justify=CENTER)
 252         wait_label.grid(row=0,
 253                         column=0,
 254                         padx=5,
 255                         pady=5)
 256         wait_progress=Progressbar(frame,
 257                                   mode="indeterminate")
 258         wait_progress.grid(row=1,
 259                            column=0,
 260                            sticky=E+W,
 261                            padx=5,
 262                            pady=5)
 263         frame.columnconfigure(0,weight=1)
 264         # place the popup in the middle of the main window
 265         # get the main window position and dimension
 266         self.geometry("{0}x{1}+{2}+{3}".format(400,
 267                                                80,
 268                                                int(root.winfo_rootx()+(root.winfo_width()-400)/2),
 269                                                int(root.winfo_rooty()+(root.winfo_height()-80)/2)))
 270         wait_progress.start(10)
 271         self.check_thread_alive()
 272     def check_thread_alive(self):
 273         """Checks if the thread is still alive, and destroys the window if it isn't"""
 274         # placing this in a for or while cycle freezes the app
 275         # so we need to use the .after method and recursively call the function
 276         # that's one of the many quirks of Tkinter
 277         if self.thread.isAlive():
 278             self.after(100,self.check_thread_alive)
 279         else:
 280             self.after(1,self.destroy)

 282 class ContextMenu(Menu):
 283     def __init__(self,x,y,widget):
 284         """A subclass of Menu, used to display a context menu in Text and Entry widgets
 285 If the widget isn't active, some options do not appear"""
 286         super().__init__(None,tearoff=0) # otherwise Tk allows splitting it in a new window
 287         self.widget=widget
 288         # MacOS uses a key called Command, instead of the usual Control used by Windows and Linux
 289         # so prepare the accelerator strings accordingly
 290         # For future reference, Mac also uses Option instead of Alt
 291         # also, a little known fact about Python is that it *does* support using the ternary operator
 292         # like in this case
 293         control_key = "Command" if self.tk.call('tk', 'windowingsystem') == "aqua" else "Ctrl"
 294         # str is necessary because in some instances a Tcl_Obj is returned instead of a string
 295         if str(widget.cget('state')) in (ACTIVE,NORMAL): # do not add if state is readonly or disabled
 296             self.add_command(label="Cut",
 297                              image=ICONS['cut'],
 298                              compound=LEFT,
 299                              accelerator='%s+X' % (control_key),
 300                              command=lambda: self.widget.event_generate("<<Cut>>"))
 301         self.add_command(label="Copy",
 302                          image=ICONS['copy'],
 303                          compound=LEFT,
 304                          accelerator='%s+C' % (control_key),
 305                          command=lambda: self.widget.event_generate("<<Copy>>"))
 306         if str(widget.cget('state')) in (ACTIVE,NORMAL):
 307             self.add_command(label="Paste",
 308                              image=ICONS['paste'],
 309                              compound=LEFT,
 310                              accelerator='%s+V' % (control_key),
 311                              command=lambda: self.widget.event_generate("<<Paste>>"))
 312         self.add_separator()
 313         self.add_command(label="Select all",
 314                          image=ICONS['select_all'],
 315                          compound=LEFT,
 316                          accelerator='%s+A' % (control_key),
 317                          command=self.on_select_all)
 318         self.tk_popup(x,y) # self.post does not destroy the menu when clicking out of it
 319     def on_select_all(self):
 320         # disabled Text widgets have a different way to handle selection
 321         if isinstance(self.widget,Text):
 322             # adding a SEL tag to a chunk of text causes it to be selected
 323             self.widget.tag_add(SEL,"1.0",END)
 324         elif isinstance(self.widget,Entry) or \
 325              isinstance(self.widget,Combobox):
 326             # apparently, the <<SelectAll>> event doesn't fire correctly if the widget is readonly
 327             self.widget.select_range(0,END)
 328         elif isinstance(self.widget,Spinbox):
 329             self.widget.selection("range",0,END)
