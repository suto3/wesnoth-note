  27 /**
  28  * The list class.
  29  *
  30  * For now it's a generic for all kind of lists, horizontal, vertical etc.
  31  * Might be that there will be different types per class, not sure yet.
  32  */
  33 class list_view : public container_base
  34 {
  35         friend class debug_layout_graph;
  36 
  37 public:
  38         /**
  39          * Constructor.
  40          *
  41          * @param has_minimum         Does the listbox need to have one item
  42          *                            selected.
  43          * @param has_maximum         Can the listbox only have one item
  44          *                            selected.
  45          * @param placement           How are the items placed.
  46          * @param select              Select an item when selected, if false it
  47          *                            changes the visible state instead.
  48          */
  49         list_view(const bool has_minimum,
  50                   const bool has_maximum,
  51                   const generator_base::tplacement placement,
  52                   const bool select,
  53                   const builder_grid_const_ptr list_builder);
  54 
  55         /***** ***** ***** ***** Row handling. ***** ***** ****** *****/
  56         /**
  57          * When an item in the list is selected by the user we need to
  58          * update the state. We installed a callback handler which
  59          * calls us.
  60          *
  61          * @param item                The data send to the set_members of the
  62          *                            widgets.
  63          * @param index               The item before which to add the new item,
  64          *                            0 == begin, -1 == end.
  65          */
  66         void add_row(const string_map& item, const int index = -1);
  67 
  68         /**
  69          * Adds single row to the grid.
  70          *
  71          * This function expect a row to have multiple widgets (either multiple
  72          * columns or one column with multiple widgets).
  73          *
  74          *
  75          * @param data                The data to send to the set_members of the
  76          *                            widgets. If the member id is not an empty
  77          *                            string it is only send to the widget that has
  78          *                            the wanted id (if any). If the member id is an
  79          *                            empty string, it is send to all members.
  80          *                            Having both empty and non-empty id's gives
  81          *                            undefined behavior.
  82          * @param index               The item before which to add the new item,
  83          *                            0 == begin, -1 == end.
  84          */
  85         void add_row(const std::map<std::string /* widget id */, string_map>& data,
  86                                  const int index = -1);
  87 
  88         /**
  89          * Appends several rows to the grid.
  90          *
  91          * @param items               The data to send to the set_members of the
  92          *                            widgets.
  93          */
  94         void append_rows(const std::vector<string_map>& items);
  95 
  96         /**
  97          * Removes a row in the listbox.
  98          *
  99          * @param row                 The row to remove, when not in
 100          *                            range the function is ignored.
 101          * @param count               The number of rows to remove, 0 means all
 102          *                            rows (starting from row).
 103          */
 104         void remove_row(const unsigned row, unsigned count = 1);
 105 
 106         /** Removes all the rows in the listbox, clearing it. */
 107         void clear();
 108 
 109         /** Returns the number of items in the listbox. */
 110         unsigned get_item_count() const;
 111 
 112         /**
 113          * Makes a row active or inactive.
 114          *
 115          * NOTE this doesn't change the select status of the row.
 116          *
 117          * @param row                 The row to (de)activate.
 118          * @param active              true activate, false deactivate.
 119          */
 120         void set_row_active(const unsigned row, const bool active);
 121 
 122         /**
 123          * Makes a row visible or invisible.
 124          *
 125          * @param row                 The row to show or hide.
 126          * @param shown               true visible, false invisible.
 127          */
 128         void set_row_shown(const unsigned row, const bool shown);
 129 
 130         /**
 131          * Makes a row visible or invisible.
 132          *
 133          * Use this version if you want to show hide multiple items since it's
 134          * optimized for that purpose, for one it calls the selection changed
 135          * callback only once instead of several times.
 136          *
 137          * @param shown               A vector with the show hide status for every
 138          *                            row. The number of items in the vector must
 139          *                            be equal to the number of items in the
 140          *                            listbox.
 141          */
 142         void set_row_shown(const boost::dynamic_bitset<>& shown);
 143 
 144         /**
 145          * Returns the grid of the wanted row.
 146          *
 147          * There's only a const version since allowing callers to modify the grid
 148          * behind our backs might give problems. We return a pointer instead of a
 149          * reference since dynamic casting of pointers is easier (no try catch
 150          * needed).
 151          *
 152          * @param row                 The row to get the grid from, the caller has
 153          *                            to make sure the row is a valid row.
 154          * @returns                   The grid of the wanted row.
 155          */
 156         const grid* get_row_grid(const unsigned row) const;
 157 
 158         /**
 159          * The possibly-giving-problems nonconst version of get_row_grid
 160          *
 161          * @param row                 The row to get the grid from, the caller has
 162          *                            to make sure the row is a valid row.
 163          * @returns                   The grid of the wanted row.
 164          */
 165         grid* get_row_grid(const unsigned row);
 166 
 167         /**
 168          * Selectes a row.
 169          *
 170          * @param row                 The row to select.
 171          * @param select              Select or deselect the row.
 172          */
 173         bool select_row(const unsigned row, const bool select = true);
 174 
 175         /**
 176          * Returns the first selected row
 177          *
 178          * @returns                   The first selected row.
 179          * @retval -1                 No row selected.
 180          */
 181         int get_selected_row() const;

 197         bool update_content_size();

 205         virtual bool get_active() const override;

 208         virtual unsigned get_state() const override;

 211         virtual void place(const point& origin, const point& size) override;

 219         enum state_t {

 240         generator_base* generator_;
 241 
 242         /** Contains the builder for the new items. */
 243         builder_grid_const_ptr list_builder_;
 244 
 245         bool need_layout_;

 273         virtual void set_self_active(const bool active) override;

 275         /** See @ref styled_widget::get_control_type. */
 276         virtual const std::string& get_control_type() const override;

 278         /***** ***** ***** signal handlers ***** ****** *****/
 279 
 280         void signal_handler_left_button_down(const event::ui_event event);
 281 
 282         void signal_handler_pre_child_left_button_click(grid* grid,
 283                                                                                                         const event::ui_event event,
 284                                                                                                         bool& handled,
 285                                                                                                         bool& halt);

 287         void signal_handler_left_button_click(grid* grid,
 288                                                                                   const event::ui_event event);
