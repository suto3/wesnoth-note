  60 unsigned toggle_button::num_states() const
  61 {
  62         std::div_t res = std::div(this->config()->state.size(), COUNT);
  63         assert(res.rem == 0);
  64         assert(res.quot > 0);
  65         return res.quot;
  66 }

 113 void toggle_button::set_value(const unsigned selected)
 114 {
 115         if(selected == get_value()) {
 116                 return;
 117         }
 118         state_num_ = selected % num_states();
 119         set_is_dirty(true);
 120 
 121         // Check for get_window() is here to prevent the callback from
 122         // being called when the initial value is set.
 123         if(!get_window()) {
 124                 return;
 125         }
 126 
 127         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 128 }

 130 void toggle_button::set_retval(const int retval)
 131 {

 133                 return;

 164 void toggle_button::signal_handler_left_button_click(const event::ui_event event,
 165                                                                                                           bool& handled)
 166 {
 167         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 168 
 169         sound::play_UI_sound(settings::sound_toggle_button_click);
 170 
 171         set_value(get_value() + 1);
 172 
 173         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 174 
 175         handled = true;
 176 }
 177 
 178 void toggle_button::signal_handler_left_button_double_click(
 179                 const event::ui_event event, bool& handled)
 180 {
 181         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 182 
 183         if(retval_ == 0) {
 184                 return;
 185         }
 186 
 187         window* window = get_window();
 188         assert(window);
 189 
 190         window->set_retval(retval_);
 191 
 192         handled = true;
 193 }

 240 toggle_button_definition::resolution::resolution(const config& cfg)
 241         : resolution_definition(cfg)
 242 {
 243         // Note the order should be the same as the enum state_t in
 244         // toggle_button.hpp.
 245         for(const auto& c : cfg.child_range("state"))
 246         {
 247                 state.emplace_back(c.child("enabled"));
 248                 state.emplace_back(c.child("disabled"));
 249                 state.emplace_back(c.child("focused"));
 250         }
 251 }
