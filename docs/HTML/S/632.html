  44 std::string addon_list::colorify_addon_state_string(const std::string& str, ADDON_STATUS state, bool verbose)
  45 {
  46         color_t colorname = font::NORMAL_COLOR;
  47 
  48         switch(state) {
  49         case ADDON_NONE:
  50                 if(!verbose) {
  51                         return str;
  52                 }
  53                 colorname = font::weapon_details_color;
  54                 break;
  55         case ADDON_INSTALLED:
  56         case ADDON_INSTALLED_LOCAL_ONLY:
  57         case ADDON_NOT_TRACKED:
  58                 colorname = font::GOOD_COLOR;
  59                 break;
  60         case ADDON_INSTALLED_UPGRADABLE:
  61                 colorname = font::YELLOW_COLOR;
  62                 break;
  63         case ADDON_INSTALLED_OUTDATED:
  64                 colorname = color_outdated;
  65                 break;
  66         case ADDON_INSTALLED_BROKEN:
  67                 colorname = font::BAD_COLOR;
  68                 break;
  69         default:
  70                 colorname = font::GRAY_COLOR;
  71                 break;
  72         }
  73 
  74         return font::span_color(colorname) + str + "</span>";
  75 }

  77 std::string addon_list::describe_status(const addon_tracking_info& info)
  78 {
  79         std::string tc, tx;
  80 
  81         switch(info.state) {
  82         case ADDON_NONE:
  83                 tx = info.can_publish ? _("addon_state^Published, not installed") : _("addon_state^Not installed");
  84                 break;
  85 
  86         case ADDON_INSTALLED:
  87         case ADDON_NOT_TRACKED:
  88                 // Consider add-ons without version information as installed
  89                 // for the main display. Their Description info should elaborate
  90                 // on their status.
  91                 tx = info.can_publish ? _("addon_state^Published") : _("addon_state^Installed");
  92                 break;
  93         case ADDON_INSTALLED_LOCAL_ONLY:
  94                 tx = info.can_publish ? _("addon_state^Ready to publish") : _("addon_state^Installed, not ready to publish");
  95                 break;
  96 
  97         case ADDON_INSTALLED_UPGRADABLE:
  98                 tx = info.can_publish ? _("addon_state^Published, upgradable") : _("addon_state^Installed, upgradable");
  99                 break;
 100 
 101         case ADDON_INSTALLED_OUTDATED:
 102                 tx = info.can_publish ? _("addon_state^Published, outdated on server") : _("addon_state^Installed, outdated on server");
 103                 break;
 104 
 105         case ADDON_INSTALLED_BROKEN:
 106                 tx = info.can_publish ? _("addon_state^Published, broken") : _("addon_state^Installed, broken");
 107                 break;
 108 
 109         default:
 110                 tx = _("addon_state^Unknown");
 111         }
 112 
 113         return colorify_addon_state_string(tx, info.state, true);
 114 }

 116 void addon_list::set_addons(const addons_list& addons)
 117 {
 118         listbox& list = get_listbox();
 119         list.clear();
 120 
 121         addon_vector_.clear();
 122 
 123         for(const auto& a : addons) {
 124                 const addon_info& addon = a.second;
 125                 addon_tracking_info tracking_info = get_addon_tracking_info(addon);
 126 
 127                 addon_vector_.push_back(&addon);
 128 
 129                 std::map<std::string, string_map> data;
 130                 string_map item;
 131 
 132                 item["use_markup"] = "true";
 133 
 134                 if(!tracking_info.can_publish) {
 135                         item["label"] = addon.display_icon();
 136                         data.emplace("icon", item);
 137 
 138                         item["label"] = addon.display_title();
 139                         data.emplace("name", item);
 140                 } else {
 141                         item["label"] = addon.display_icon() + "~BLIT(icons/icon-addon-publish.png)";
 142                         data.emplace("icon", item);
 143 
 144                         const std::string publish_name = formatter()
 145                                 << "<span color='#00ff00'>" // GOOD_COLOR
 146                                 << addon.display_title()
 147                                 << "</span>";
 148 
 149                         item["label"] = publish_name;
 150                         data.emplace("name", item);
 151                 }
 152 
 153                 item["label"] = describe_status(tracking_info);
 154                 data.emplace("installation_status", item);
 155 
 156                 item["label"] = addon.version.str();
 157                 data.emplace("version", item);
 158 
 159                 item["label"] = addon.author;
 160                 data.emplace("author", item);
 161 
 162                 item["label"] = size_display_string(addon.size);
 163                 data.emplace("size", item);
 164 
 165                 item["label"] = std::to_string(addon.downloads);
 166                 data.emplace("downloads", item);
 167 
 168                 item["label"] = addon.display_type();
 169                 data.emplace("type", item);
 170 
 171                 grid* row_grid = &list.add_row(data);
 172 
 173                 // Set special retval for the toggle panels
 174                 find_widget<toggle_panel>(row_grid, "list_panel", false).set_retval(DEFAULT_ACTION_RETVAL);
 175 
 176                 grid* control_grid = find_widget<grid>(row_grid, "single_install_buttons", false, false);
 177                 if(!control_grid) {
 178                         continue;
 179                 }
 180                 stacked_widget& install_update_stack = find_widget<stacked_widget>(row_grid, "install_update_stack", false);
 181 
 182                 if(!tracking_info.can_publish) {
 183                         const bool is_updatable = tracking_info.state == ADDON_INSTALLED_UPGRADABLE;
 184                         const bool is_installed =
 185                                 tracking_info.state == ADDON_INSTALLED || tracking_info.state == ADDON_INSTALLED_UPGRADABLE;
 186 
 187                         install_update_stack.select_layer(static_cast<int>(is_updatable));
 188 
 189                         if(!is_updatable) {
 190                                 find_widget<button>(control_grid, "single_install", false).set_active(!is_installed);
 191                                 if(install_function_ != nullptr) {
 192                                         gui2::event::connect_signal_mouse_left_click(
 193                                                 find_widget<button>(control_grid, "single_install", false),
 194                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 195                                         {
 196                                                 install_function_(addon);
 197                                                 handled = true;
 198                                                 halt = true;
 199                                         });
 200                                 }
 201                         } else {
 202                                 find_widget<button>(control_grid, "single_update", false).set_active(true);
 203                                 if(update_function_ != nullptr) {
 204                                         gui2::event::connect_signal_mouse_left_click(
 205                                                 find_widget<button>(control_grid, "single_update", false),
 206                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 207                                         {
 208                                                 update_function_(addon);
 209                                                 handled = true;
 210                                                 halt = true;
 211                                         });
 212                                 }
 213                         }
 214 
 215                         if(is_installed) {
 216                                 if(uninstall_function_ != nullptr) {
 217                                         gui2::event::connect_signal_mouse_left_click(
 218                                                 find_widget<button>(control_grid, "single_uninstall", false),
 219                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 220                                         {
 221                                                 uninstall_function_(addon);
 222                                                 handled = true;
 223                                                 halt = true;
 224                                         });
 225                                 }
 226                         }
 227 
 228                         find_widget<button>(control_grid, "single_uninstall", false).set_active(is_installed);
 229 
 230                         find_widget<grid>(control_grid, "single_install_buttons", false).set_visible(install_buttons_visibility_);
 231                         find_widget<label>(row_grid, "installation_status", false).set_visible(install_status_visibility_);
 232                 } else {
 233                         const bool is_updatable = tracking_info.state == ADDON_INSTALLED_OUTDATED;
 234 
 235                         button& install_button = find_widget<button>(control_grid, "single_install", false);
 236                         button& update_button = find_widget<button>(control_grid, "single_update", false);
 237                         button& uninstall_button = find_widget<button>(control_grid, "single_uninstall", false);
 238 
 239                         install_button.set_active(true);
 240                         update_button.set_active(true);
 241                         uninstall_button.set_active(tracking_info.state == ADDON_INSTALLED);
 242 
 243                         if(true) {
 244                                 gui2::event::connect_signal_mouse_left_click(
 245                                         install_button,
 246                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 247                                 {
 248                                         publish_function_(addon);
 249                                         handled = true;
 250                                         halt = true;
 251                                 });
 252 
 253                                 install_button.set_tooltip(_("Publish add-on"));
 254                         }
 255 
 256                         if(is_updatable) {
 257                                 gui2::event::connect_signal_mouse_left_click(
 258                                         update_button,
 259                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 260                                 {
 261                                         publish_function_(addon);
 262                                         handled = true;
 263                                         halt = true;
 264                                 });
 265 
 266                                 update_button.set_tooltip(_("Send new version to server"));
 267                         }
 268 
 269                         if(tracking_info.state == ADDON_INSTALLED) {
 270                                 gui2::event::connect_signal_mouse_left_click(
 271                                         uninstall_button,
 272                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 273                                 {
 274                                         delete_function_(addon);
 275                                         handled = true;
 276                                         halt = true;
 277                                 });
 278 
 279                                 uninstall_button.set_tooltip(_("Delete add-on from server"));
 280                         }
 281 
 282                         install_update_stack.select_layer(static_cast<int>(is_updatable));
 283                 }
 284         }
 285 }

 287 const addon_info* addon_list::get_selected_addon() const
 288 {
 289         const listbox& list = find_widget<const listbox>(&get_grid(), "addons", false);
 290 
 291         const int index = list.get_selected_row();
 292         if(index == -1 || index >= static_cast<int>(addon_vector_.size())) {
 293                 return nullptr;
 294         }
 295 
 296         return addon_vector_.at(index);
 297 }

 299 std::string addon_list::get_remote_addon_id()
 300 {
 301         const addon_info* addon = get_selected_addon();
 302         if(addon == nullptr || !get_addon_tracking_info(*addon).can_publish) {
 303                 return "";
 304         } else {
 305                 return addon->id;
 306         }

 309 void addon_list::select_addon(const std::string& id)
 310 {
 311         listbox& list = get_listbox();
 312 
 313         const addon_info& info = **std::find_if(addon_vector_.begin(), addon_vector_.end(),
 314                 [&id](const addon_info* a)
 315         {
 316                 return a->id == id;
 317         });
 318 
 319         for(unsigned int i = 0u; i < list.get_item_count(); ++i)
 320         {
 321                 grid* row = list.get_row_grid(i);
 322                 const label& name_label = find_widget<label>(row, "name", false);
 323                 if(name_label.get_label().base_str() == info.display_title())
 324                 {
 325                         list.select_row(i);
 326                 }
 327         }
 328 }

 342 void addon_list::finalize_setup()
 343 {
 344         listbox& list = get_listbox();
 345 
 346         list.register_sorting_option(0, [this](const int i) { return addon_vector_[i]->title; });
 347         list.register_sorting_option(1, [this](const int i) { return addon_vector_[i]->author; });
 348         list.register_sorting_option(2, [this](const int i) { return addon_vector_[i]->size; });
 349         list.register_sorting_option(3, [this](const int i) { return addon_vector_[i]->downloads; });
 350         list.register_sorting_option(4, [this](const int i) { return addon_vector_[i]->display_type(); });
 351 }

 377 static widget::visibility parse_visibility(const std::string& str)
 378 {
 379         if(str == "visible") {
 380                 return widget::visibility::visible;
 381         } else if(str == "hidden") {
 382                 return widget::visibility::hidden;
 383         } else if(str == "invisible") {
 384                 return widget::visibility::invisible;
 385         } else {
 386                 FAIL("Invalid visibility value");
 387         }
 388 }
