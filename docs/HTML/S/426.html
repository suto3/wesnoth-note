 155 static void register_button(window& win, const std::string& id, hotkey::HOTKEY_COMMAND hk, btn_callback callback)
 156 {
 157         if(hk != hotkey::HOTKEY_NULL) {
 158                 win.register_hotkey(hk, [callback](event::dispatcher& w, hotkey::HOTKEY_COMMAND) {
 159                         callback(dynamic_cast<window&>(w));
 160                         return true;
 161                 });
 162         }
 163 
 164         event::connect_signal_mouse_left_click(find_widget<button>(&win, id, false),
 165                 [callback](event::dispatcher& w, event::ui_event, bool&, bool&) { callback(dynamic_cast<window&>(w)); });
 166 }

 182  * This function is used to test the tooltip placement algorithms as
 183  * described in the »Tooltip placement« section in the GUI2 design
 184  * document.

 186  * Use a 1024 x 768 screen size, set the maximum loop iteration to:

 193 static void debug_tooltip(window& window, bool& handled, const point& coordinate)
 194 {
 195         std::string message = "Hello world.";
 196 
 197         for(int i = 0; i < 0; ++i) {
 198                 message += " More greetings.";
 199         }
 200 
 201         gui2::tip::remove();
 202         gui2::tip::show(window.video(), "tooltip", message, coordinate);
 203 
 204         handled = true;
 205 }

 208 void title_screen::pre_show(window& win)
 209 {
 210         win.set_click_dismiss(false);
 211         win.set_enter_disabled(true);
 212         win.set_escape_disabled(true);
 213 
 214 #ifdef DEBUG_TOOLTIP
 215         win.connect_signal<event::SDL_MOUSE_MOTION>(
 216                         std::bind(debug_tooltip, std::ref(win), _3, _5),
 217                         event::dispatcher::front_child);
 218 #endif
 219 
 220         win.connect_signal<event::SDL_VIDEO_RESIZE>(std::bind(&title_screen::on_resize, this, std::ref(win)));
 221 
 222         //
 223         // General hotkeys
 224         //
 225         win.register_hotkey(hotkey::TITLE_SCREEN__RELOAD_WML, [](event::dispatcher& w, hotkey::HOTKEY_COMMAND) {
 226                 dynamic_cast<window&>(w).set_retval(RELOAD_GAME_DATA);
 227                 return true;
 228         });
 229 
 230         win.register_hotkey(hotkey::TITLE_SCREEN__TEST, [this](event::dispatcher& w, hotkey::HOTKEY_COMMAND) {
 231                 game_config_manager::get()->load_game_config_for_create(false, true);
 232                 std::vector<std::string> options;
 233                 for(const config &sc : game_config_manager::get()->game_config().child_range("test")) {
 234                         if(!sc["is_unit_test"].to_bool(false)) {
 235                                 options.emplace_back(sc["id"]);
 236                         }
 237                 }
 238                 std::sort(options.begin(), options.end());
 239                 gui2::dialogs::simple_item_selector dlg(_("Choose Test"), "", options);
 240                 dlg.show(game_.video());
 241                 int choice = dlg.selected_index();
 242                 if(choice >= 0) {
 243                         game_.set_test(options[choice]);
 244                         dynamic_cast<window&>(w).set_retval(LAUNCH_GAME);
 245                 }
 246                 return true;
 247         });
 248 
 249         win.register_hotkey(hotkey::HOTKEY_FULLSCREEN, std::bind(fullscreen, std::ref(win.video())));
 250         win.register_hotkey(hotkey::LUA_CONSOLE, std::bind(&launch_lua_console, std::ref(win)));
 251 
 252         //
 253         // Background and logo images
 254         //
 255         if(game_config::images::game_title.empty()) {
 256                 ERR_CF << "No title image defined" << std::endl;
 257         }
 258 
 259         win.get_canvas(0).set_variable("title_image", wfl::variant(game_config::images::game_title));
 260 
 261         if(game_config::images::game_title_background.empty()) {
 262                 ERR_CF << "No title background image defined" << std::endl;
 263         }
 264 
 265         win.get_canvas(0).set_variable("background_image", wfl::variant(game_config::images::game_title_background));
 266 
 267         find_widget<image>(&win, "logo-bg", false).set_image(game_config::images::game_logo_background);
 268         find_widget<image>(&win, "logo", false).set_image(game_config::images::game_logo);
 269 
 270         //
 271         // Version string
 272         //
 273         const std::string version_string = formatter() << ("Version") << " " << game_config::revision;
 274 
 275         if(label* version_label = find_widget<label>(&win, "revision_number", false, false)) {
 276                 version_label->set_label(version_string);
 277         }
 278 
 279         win.get_canvas(0).set_variable("revision_number", wfl::variant(version_string));
 280 
 281         //
 282         // Tip-of-the-day browser
 283         //
 284         multi_page& tip_pages = find_widget<multi_page>(&win, "tips", false);
 285 
 286         std::vector<game_tip> tips(settings::get_tips());
 287         if(tips.empty()) {
 288                 WRN_CF << "There are no tips of day available." << std::endl;
 289         }
 290 
 291         for(const auto& tip : tips)     {
 292                 string_map widget;
 293                 std::map<std::string, string_map> page;
 294 
 295                 widget["use_markup"] = "true";
 296 
 297                 widget["label"] = tip.text();
 298                 page.emplace("tip", widget);
 299 
 300                 widget["label"] = tip.source();
 301                 page.emplace("source", widget);
 302 
 303                 tip_pages.add_page(page);
 304         }
 305 
 306         update_tip(win, true);
 307 
 308         register_button(win, "next_tip", hotkey::TITLE_SCREEN__NEXT_TIP,
 309                 std::bind(&title_screen::update_tip, this, std::ref(win), true));
 310         register_button(win, "previous_tip", hotkey::TITLE_SCREEN__PREVIOUS_TIP,
 311                 std::bind(&title_screen::update_tip, this, std::ref(win), false));
 312 
 313         //
 314         // Help
 315         //
 316         register_button(win, "help", hotkey::HOTKEY_HELP, [](window& w) {
 317                 if(gui2::new_widgets) {
 318                         gui2::dialogs::help_browser::display(w.video());
 319                 }
 320 
 321                 help::help_manager help_manager(&game_config_manager::get()->game_config());
 322                 help::show_help(w.video());
 323         });
 324 
 325         //
 326         // About
 327         //
 328         register_button(win, "about", hotkey::HOTKEY_NULL, std::bind(&game_version::display, std::ref(win.video())));
 329 
 330         //
 331         // Tutorial
 332         //
 333         register_button(win, "tutorial", hotkey::TITLE_SCREEN__TUTORIAL, [this](window& w) {
 334                 game_.set_tutorial();
 335                 w.set_retval(LAUNCH_GAME);
 336         });
 337 
 338         //
 339         // Campaign
 340         //
 341         register_button(win, "campaign", hotkey::TITLE_SCREEN__CAMPAIGN, [this](window& w) {
 342                 try{
 343                         if(game_.new_campaign()) {
 344                                 w.set_retval(LAUNCH_GAME);
 345                         }
 346                 } catch (const config::error& e) {
 347                         gui2::show_error_message(w.video(), e.what());
 348                 }
 349         });
 350 
 351         //
 352         // Multiplayer
 353         //
 354         register_button(win, "multiplayer", hotkey::TITLE_SCREEN__MULTIPLAYER, [this](window& w) {
 355                 while(true) {
 356                         gui2::dialogs::mp_method_selection dlg;
 357                         dlg.show(w.video());
 358 
 359                         if(dlg.get_retval() != gui2::window::OK) {
 360                                 return;
 361                         }
 362 
 363                         const int res = dlg.get_choice();
 364 
 365                         if(res == 2 && preferences::mp_server_warning_disabled() < 2) {
 366                                 if(!gui2::dialogs::mp_host_game_prompt::execute(w.video())) {
 367                                         continue;
 368                                 }
 369                         }
 370 
 371                         switch(res) {
 372                                 case 0:
 373                                         game_.select_mp_server(preferences::server_list().front().address);
 374                                         w.set_retval(MP_CONNECT);
 375                                         break;
 376                                 case 1:
 377                                         game_.select_mp_server("");
 378                                         w.set_retval(MP_CONNECT);
 379                                         break;
 380                                 case 2:
 381                                         game_.select_mp_server("localhost");
 382                                         w.set_retval(MP_HOST);
 383                                         break;
 384                                 case 3:
 385                                         w.set_retval(MP_LOCAL);
 386                                         break;
 387                         }
 388 
 389                         return;
 390                 }
 391         });
 392 
 393         //
 394         // Load game
 395         //
 396         register_button(win, "load", hotkey::HOTKEY_LOAD_GAME, [this](window& w) {
 397                 if(game_.load_game()) {
 398                         w.set_retval(LAUNCH_GAME);
 399                 } else {
 400                         game_.clear_loaded_game();
 401                 }
 402         });
 403 
 404         //
 405         // Addons
 406         //
 407         register_button(win, "addons", hotkey::TITLE_SCREEN__ADDONS, [](window& w) {
 408                 // NOTE: we need the help_manager to get access to the Add-ons section in the game help!
 409                 help::help_manager help_manager(&game_config_manager::get()->game_config());
 410 
 411                 if(manage_addons(w.video())) {
 412                         game_config_manager::get()->reload_changed_game_config();
 413                 }
 414         });
 415 
 416         //
 417         // Editor
 418         //
 419         register_button(win, "editor", hotkey::TITLE_SCREEN__EDITOR, [&](window& w) { w.set_retval(MAP_EDITOR); });
 420 
 421         //
 422         // Cores
 423         //
 424         register_button(win, "cores", hotkey::TITLE_SCREEN__CORES, [](window& w) {
 425                 int current = 0;
 426                 std::vector<config> cores;
 427                 for(const config& core : game_config_manager::get()->game_config().child_range("core")) {
 428                         cores.push_back(core);
 429 
 430                         if(core["id"] == preferences::core_id()) {
 431                                 current = cores.size() - 1;
 432                         }
 433                 }
 434 
 435                 gui2::dialogs::core_selection core_dlg(cores, current);
 436                 if(core_dlg.show(w.video())) {
 437                         const std::string& core_id = cores[core_dlg.get_choice()]["id"];
 438 
 439                         preferences::set_core_id(core_id);
 440                         game_config_manager::get()->reload_changed_game_config();
 441                 }
 442         });
 443 
 444         if(game_config_manager::get()->game_config().child_range("core").size() <= 1) {
 445                 find_widget<button>(&win, "cores", false).set_visible(window::visibility::invisible);
 446         }
 447 
 448         //
 449         // Language
 450         //
 451         register_button(win, "language", hotkey::HOTKEY_LANGUAGE, [this](window& w) {
 452                 try {
 453                         if(game_.change_language()) {
 454                                 t_string::reset_translations();
 455                                 ::image::flush_cache();
 456                                 on_resize(w);
 457                         }
 458                 } catch(std::runtime_error& e) {
 459                         gui2::show_error_message(w.video(), e.what());
 460                 }
 461         });
 462 
 463         //
 464         // Preferences
 465         //
 466         register_button(win, "preferences", hotkey::HOTKEY_PREFERENCES, [this](window&) { game_.show_preferences(); });
 467 
 468         //
 469         // Credits
 470         //
 471         register_button(win, "credits", hotkey::TITLE_SCREEN__CREDITS, [&](window& w) { w.set_retval(SHOW_ABOUT); });
 472 
 473         //
 474         // Quit
 475         //
 476         register_button(win, "quit", hotkey::HOTKEY_QUIT_TO_DESKTOP, [&](window& w) { w.set_retval(QUIT_GAME); });
 477 
 478         //
 479         // Debug clock
 480         //
 481         register_button(win, "clock", hotkey::HOTKEY_NULL,
 482                 std::bind(&title_screen::show_debug_clock_window, this, std::ref(win.video())));
 483 
 484         find_widget<button>(&win, "clock", false).set_visible(show_debug_clock_button
 485                 ? widget::visibility::visible
 486                 : widget::visibility::invisible);
 487 }

 494 void title_screen::update_tip(window& win, const bool previous)
 495 {
 496         multi_page& tips = find_widget<multi_page>(&win, "tips", false);
 497         if(tips.get_page_count() == 0) {
 498                 return;
 499         }
 500 
 501         int page = tips.get_selected_page();
 502         if(previous) {
 503                 if(page <= 0) {
 504                         page = tips.get_page_count();
 505                 }
 506                 --page;
 507         } else {
 508                 ++page;
 509                 if(static_cast<unsigned>(page) >= tips.get_page_count()) {
 510                         page = 0;
 511                 }
 512         }
 513 
 514         tips.select_page(page);
 515 
 516         /**
 517          * @todo Look for a proper fix.
 518          *
 519          * This dirtying is required to avoid the blurring to be rendered wrong.
 520          * Not entirely sure why, but since we plan to move to SDL2 that change
 521          * will probably fix this issue automatically.
 522          */
 523         win.set_is_dirty(true);
 524 }

 526 void title_screen::show_debug_clock_window(CVideo& video)
 527 {
 528         assert(show_debug_clock_button);
 529 
 530         if(debug_clock_) {
 531                 debug_clock_.reset(nullptr);
 532         } else {
 533                 debug_clock_.reset(new debug_clock());
 534                 debug_clock_->show(video, true);
 535         }
 536 }
