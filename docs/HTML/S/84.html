 116 void playsingle_controller::init_gui(){
 117         LOG_NG << "Initializing GUI... " << (SDL_GetTicks() - ticks()) << "\n";
 118         play_controller::init_gui();
 119 
 120         // Scroll to the starting position of the first team. If there is a
 121         // human team, use that team; otherwise use team 1. If the map defines
 122         // a starting position for the selected team, scroll to that tile. Note
 123         // this often does not matter since many scenario start with messages,
 124         // which will usually scroll to the speaker. Also note that the map
 125         // does not necessarily define the starting positions. While usually
 126         // best to use the map, the scenarion may explicitly set the positions,
 127         // overriding those found in the map (if any).
 128         if(map_start_.valid())
 129         {
 130                 gui_->scroll_to_tile(map_start_, game_display::WARP);
 131         }
 132         else
 133         {
 134                 int scroll_team = gamestate().first_human_team_ + 1;
 135                 if (scroll_team == 0) {
 136                         scroll_team = 1;
 137                 }
 138                 map_location loc(gamestate().board_.map().starting_position(scroll_team));
 139                 if ((loc.x >= 0) && (loc.y >= 0)) {
 140                         gui_->scroll_to_tile(loc, game_display::WARP);
 141                 }
 142         }
 143 
 144         update_locker lock_display(gui_->video(), is_skipping_replay());
 145         gui_->draw();
 146         get_hotkey_command_executor()->set_button_state();
 147         events::raise_draw_event();
 148 }

 151 void playsingle_controller::play_scenario_init()
 152 {
 153         // At the beginning of the scenario, save a snapshot as replay_start
 154         if(saved_game_.replay_start().empty()){
 155                 saved_game_.replay_start() = to_config();
 156         }
 157         start_game();
 158         if( saved_game_.classification().random_mode != "" && is_networked_mp()) {
 159                 // This won't cause errors later but we should notify the user about it in case he didn't knew it.
 160                 gui2::show_transient_message(
 161                         gui_->video(),
 162                         // TODO: find a better title
 163                         _("Game Error"),
 164                         _("This multiplayer game uses an alternative random mode, if you don't know what this message means, then most likely someone is cheating or someone reloaded a corrupt game.")
 165                 );
 166         }
 167         return;
 168 }

 170 void playsingle_controller::play_scenario_main_loop()
 171 {
 172         LOG_NG << "starting main loop\n" << (SDL_GetTicks() - ticks()) << "\n";
 173 
 174 
 175         // Avoid autosaving after loading, but still
 176         // allow the first turn to have an autosave.
 177         ai_testing::log_game_start();
 178         if(gamestate().board_.teams().empty())
 179         {
 180                 ERR_NG << "Playing game with 0 teams." << std::endl;
 181         }
 182         while(true) {
 183                 try {
 184                         play_turn();
 185                         if (is_regular_game_end()) {
 186                                 turn_data_.send_data();
 187                                 return;
 188                         }
 189                         gamestate_->player_number_ = 1;
 190                 }
 191                 catch(const reset_gamestate_exception& ex) {
 192                         //
 193                         // TODO:
 194                         //
 195                         // The MP replay feature still doesn't work properly (causes OOS)
 196                         // because:
 197                         //
 198                         // 1) The undo stack is not reset along with the gamestate (fixed).
 199                         // 2) The server_request_number_ is not reset along with the
 200                         //    gamestate (fixed).
 201                         // 3) chat and other unsynced actions are inserted in the middle of
 202                         //    the replay bringing the replay_pos in unorder (fixed).
 203                         // 4) untracked changes in side controllers are lost when resetting
 204                         //    gamestate (fixed).
 205                         // 5) The game should have a stricter check for whether the loaded
 206                         //    game is actually a parent of this game.
 207                         // 6) If an action was undone after a game was saved it can cause
 208                         //    OOS if the undone action is in the snapshot of the saved
 209                         //    game (luckily this is never the case for autosaves).
 210                         //
 211                         boost::dynamic_bitset<> local_players;
 212                         local_players.resize(gamestate().board_.teams().size(), true);
 213                         //Preserve side controllers, becasue we won't get the side controoller updates again when replaying.
 214                         for(size_t i = 0; i < local_players.size(); ++i) {
 215                                 local_players[i] = gamestate().board_.teams()[i].is_local();
 216                         }
 217                         reset_gamestate(*ex.level, (*ex.level)["replay_pos"]);
 218                         for(size_t i = 0; i < local_players.size(); ++i) {
 219                                 resources::gameboard->teams()[i].set_local(local_players[i]);
 220                         }
 221                         play_scenario_init();
 222                         replay_.reset(new replay_controller(*this, false, ex.level));
 223                         if(ex.start_replay) {
 224                                 replay_->play_replay();
 225                         }
 226                 }
 227         } //end for loop
 228 }

 230 LEVEL_RESULT playsingle_controller::play_scenario(const config& level)
 231 {
 232         LOG_NG << "in playsingle_controller::play_scenario()...\n";
 233 
 234         // Start music.
 235         for(const config &m : level.child_range("music")) {
 236                 sound::play_music_config(m);
 237         }
 238         sound::commit_music_changes();
 239 
 240         if(!this->is_skipping_replay()) {
 241                 // Combine all the [story] tags into a single config. Handle this here since
 242                 // storyscreen::controller doesn't have a default constructor.
 243                 config cfg;
 244                 for(const auto& iter : level.child_range("story")) {
 245                         cfg.append_children(iter);
 246                 }
 247 
 248                 if(!cfg.empty()) {
 249                         gui2::dialogs::story_viewer::display(get_scenario_name(), cfg, gui_->video());
 250                 }
 251         }
 252         gui_->labels().read(level);
 253 
 254         // Read sound sources
 255         assert(soundsources_manager_ != nullptr);
 256         for (const config &s : level.child_range("sound_source")) {
 257                 try {
 258                         soundsource::sourcespec spec(s);
 259                         soundsources_manager_->add(spec);
 260                 } catch (bad_lexical_cast &) {
 261                         ERR_NG << "Error when parsing sound_source config: bad lexical cast." << std::endl;
 262                         ERR_NG << "sound_source config was: " << s.debug() << std::endl;
 263                         ERR_NG << "Skipping this sound source..." << std::endl;
 264                 }
 265         }
 266         LOG_NG << "entering try... " << (SDL_GetTicks() - ticks()) << "\n";
 267         try {
 268                 play_scenario_init();
 269                 // clears level config;
 270                 this->saved_game_.remove_snapshot();
 271 
 272                 if (!is_regular_game_end() && !linger_) {
 273                         play_scenario_main_loop();
 274                 }
 275                 if (game_config::exit_at_end) {
 276                         exit(0);
 277                 }
 278                 const bool is_victory = get_end_level_data_const().is_victory;
 279 
 280                 if(gamestate().gamedata_.phase() <= game_data::PRESTART) {
 281                         gui_->video().clear_screen();
 282                 }
 283 
 284                 ai_testing::log_game_end();
 285 
 286                 const end_level_data& end_level = get_end_level_data_const();
 287 
 288                 if (gamestate().board_.teams().empty())
 289                 {
 290                         //store persistent teams
 291                         saved_game_.set_snapshot(config());
 292 
 293                         return LEVEL_RESULT::VICTORY; // this is probably only a story scenario, i.e. has its endlevel in the prestart event
 294                 }
 295                 if(linger_) {
 296                         LOG_NG << "resuming from loaded linger state...\n";
 297                         //as carryover information is stored in the snapshot, we have to re-store it after loading a linger state
 298                         saved_game_.set_snapshot(config());
 299                         if(!is_observer()) {
 300                                 persist_.end_transaction();
 301                         }
 302                         return LEVEL_RESULT::VICTORY;
 303                 }
 304                 pump().fire(is_victory ? "local_victory" : "local_defeat");
 305                 { // Block for set_scontext_synced_base
 306                         set_scontext_synced_base sync;
 307                         pump().fire(end_level.proceed_to_next_level ? "victory" : "defeat");
 308                         pump().fire("scenario_end");
 309                 }
 310                 if(end_level.proceed_to_next_level) {
 311                         gamestate().board_.heal_all_survivors();
 312                 }
 313                 if(is_observer()) {
 314                         gui2::show_transient_message(gui_->video(), _("Game Over"), _("The game is over."));
 315                         return LEVEL_RESULT::OBSERVER_END;
 316                 }
 317                 // If we're a player, and the result is victory/defeat, then send
 318                 // a message to notify the server of the reason for the game ending.
 319                 send_to_wesnothd(config {
 320                         "info", config {
 321                                 "type", "termination",
 322                                 "condition", "game over",
 323                                 "result", is_victory ? "victory" : "defeat",
 324                         },
 325                 });
 326                 // Play victory music once all victory events
 327                 // are finished, if we aren't observers and the
 328                 // carryover dialog isn't disabled.
 329                 //
 330                 // Some scenario authors may use 'continue'
 331                 // result for something that is not story-wise
 332                 // a victory, so let them use [music] tags
 333                 // instead should they want special music.
 334                 const std::string& end_music = select_music(is_victory);
 335                 if((!is_victory || end_level.transient.carryover_report) && !end_music.empty()) {
 336                         sound::empty_playlist();
 337                         sound::play_music_once(end_music);
 338                 }
 339                 persist_.end_transaction();
 340                 return is_victory ? LEVEL_RESULT::VICTORY : LEVEL_RESULT::DEFEAT;
 341         } catch(const savegame::load_game_exception &) {
 342                 // Loading a new game is effectively a quit.
 343                 saved_game_.clear();
 344                 throw;
 345         } catch(wesnothd_error& e) {
 346 
 347                 scoped_savegame_snapshot snapshot(*this);
 348                 savegame::ingame_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 349                 save.save_game_interactive(gui_->video(), _("A network disconnection has occurred, and the game cannot continue. Do you want to save the game?"), savegame::savegame::YES_NO);
 350                 if(dynamic_cast<ingame_wesnothd_error*>(&e)) {
 351                         return LEVEL_RESULT::QUIT;
 352                 } else {
 353                         throw;
 354                 }
 355         }
 356 }

 367 void playsingle_controller::play_side_impl()
 368 {
 369         if (!skip_next_turn_) {
 370                 end_turn_ = END_TURN_NONE;
 371         }
 372         if(replay_.get() != nullptr) {
 373                 REPLAY_RETURN res = replay_->play_side_impl();
 374                 if(res == REPLAY_FOUND_END_TURN) {
 375                         end_turn_ = END_TURN_SYNCED;
 376                 }
 377                 if (player_type_changed_) {
 378                         replay_.reset();
 379                 }
 380         } else if((current_team().is_local_human() && current_team().is_proxy_human())) {
 381                 LOG_NG << "is human...\n";
 382                 // If a side is dead end the turn, but play at least side=1's
 383                 // turn in case all sides are dead
 384                 if (gamestate().board_.side_units(current_side()) == 0 && !(gamestate().board_.units().size() == 0 && current_side() == 1)) {
 385                         end_turn_ = END_TURN_REQUIRED;
 386                 }
 387 
 388                 before_human_turn();
 389                 if (end_turn_ == END_TURN_NONE) {
 390                         play_human_turn();
 391                 }
 392                 if ( !player_type_changed_ && !is_regular_game_end()) {
 393                         after_human_turn();
 394                 }
 395                 LOG_NG << "human finished turn...\n";
 396 
 397         } else if(current_team().is_local_ai() || (current_team().is_local_human() && current_team().is_droid())) {
 398                 play_ai_turn();
 399         } else if(current_team().is_network()) {
 400                 play_network_turn();
 401         } else if(current_team().is_local_human() && current_team().is_idle()) {
 402                 end_turn_enable(false);
 403                 do_idle_notification();
 404                 before_human_turn();
 405                 if (end_turn_ == END_TURN_NONE) {
 406                         play_idle_loop();
 407                 }
 408         }
 409         else {
 410                 // we should have skipped over empty controllers before so this shouldn't be possible
 411                 ERR_NG << "Found invalid side controller " << current_team().controller().to_string() << " (" << current_team().proxy_controller().to_string() << ") for side " << current_team().side() << "\n";
 412         }
 413 }

 415 void playsingle_controller::before_human_turn()
 416 {
 417         log_scope("player turn");
 418         assert(!linger_);
 419         if(end_turn_ != END_TURN_NONE || is_regular_game_end()) {
 420                 return;
 421         }
 422 
 423         if(init_side_done_now_) {
 424                 scoped_savegame_snapshot snapshot(*this);
 425                 savegame::autosave_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 426                 save.autosave(game_config::disable_autosave, preferences::autosavemax(), preferences::INFINITE_AUTO_SAVES);
 427         }
 428 
 429         if(preferences::turn_bell()) {
 430                 sound::play_bell(game_config::sounds::turn_bell);
 431         }
 432 }

 478 void playsingle_controller::linger()
 479 {
 480         LOG_NG << "beginning end-of-scenario linger\n";
 481         linger_ = true;
 482 
 483         // If we need to set the status depending on the completion state
 484         // the key to it is here.
 485         gui_->set_game_mode(game_display::LINGER);
 486 
 487         // change the end-turn button text to its alternate label
 488         gui_->get_theme().refresh_title2("button-endturn", "title2");
 489         gui_->invalidate_theme();
 490         gui_->redraw_everything();
 491 
 492         // End all unit moves
 493         gamestate().board_.set_all_units_user_end_turn();
 494         try {
 495                 // Same logic as single-player human turn, but
 496                 // *not* the same as multiplayer human turn.
 497                 end_turn_enable(true);
 498                 end_turn_ = END_TURN_NONE;
 499                 while(end_turn_ == END_TURN_NONE) {
 500                         play_slice();
 501                         gui_->draw();
 502                 }
 503         } catch(const savegame::load_game_exception &) {
 504                 // Loading a new game is effectively a quit.
 505                 saved_game_.clear();
 506                 throw;
 507         }
 508 
 509         // revert the end-turn button text to its normal label
 510         gui_->get_theme().refresh_title2("button-endturn", "title");
 511         gui_->invalidate_theme();
 512         gui_->redraw_everything();
 513         gui_->set_game_mode(game_display::RUNNING);
 514 
 515         LOG_NG << "ending end-of-scenario linger\n";
 516 }

 532 void playsingle_controller::play_ai_turn()
 533 {
 534         LOG_NG << "is ai...\n";
 535 
 536         end_turn_enable(false);
 537         gui_->recalculate_minimap();
 538 
 539         const cursor::setter cursor_setter(cursor::WAIT);
 540 
 541         // Correct an oddball case where a human could have left delayed shroud
 542         // updates on before giving control to the AI. (The AI does not bother
 543         // with the undo stack, so it cannot delay shroud updates.)
 544         team & cur_team = current_team();
 545         if ( !cur_team.auto_shroud_updates() ) {
 546                 // We just took control, so the undo stack is empty. We still need
 547                 // to record this change for the replay though.
 548                 synced_context::run_and_store("auto_shroud", replay_helper::get_auto_shroud(true));
 549         }
 550         undo_stack().clear();
 551 
 552         turn_data_.send_data();
 553         try {
 554                 try {
 555                         if (!should_return_to_play_side()) {
 556                                 ai::manager::play_turn(current_side());
 557                         }
 558                 }
 559                 catch (return_to_play_side_exception&) {
 560                 }
 561                 catch (fallback_ai_to_human_exception&) {
 562                         current_team().make_human();
 563                         player_type_changed_ = true;
 564                         ai_fallback_ = true;
 565                 }
 566         }
 567         catch(...) {
 568                 turn_data_.sync_network();
 569                 throw;
 570         }
 571         if(!should_return_to_play_side()) {
 572                 end_turn_ = END_TURN_REQUIRED;
 573         }
 574         turn_data_.sync_network();
 575         gui_->recalculate_minimap();
 576         gui_->invalidate_unit();
 577         gui_->invalidate_game_status();
 578         gui_->invalidate_all();
 579         gui_->draw();
 580 }

 645 void playsingle_controller::sync_end_turn()
 646 {
 647         //We cannot add [end_turn] to the recorder while executing another action.
 648         assert(synced_context::synced_state() == synced_context::UNSYNCED);
 649         if(end_turn_ == END_TURN_REQUIRED && current_team().is_local())
 650         {
 651                 //TODO: we shodul also send this immideateley.
 652                 resources::recorder->end_turn();
 653                 end_turn_ = END_TURN_SYNCED;
 654         }
 655 
 656         assert(end_turn_ == END_TURN_SYNCED);
 657         skip_next_turn_ = false;
 658 
 659         if(ai_fallback_) {
 660                 current_team().make_ai();
 661                 ai_fallback_ = false;
 662         }
 663 }
