 118 void playsingle_controller::init_gui(){
 119         LOG_NG << "Initializing GUI... " << (SDL_GetTicks() - ticks()) << "\n";
 120         play_controller::init_gui();
 121 
 122         // Scroll to the starting position of the first team. If there is a
 123         // human team, use that team; otherwise use team 1. If the map defines
 124         // a starting position for the selected team, scroll to that tile. Note
 125         // this often does not matter since many scenario start with messages,
 126         // which will usually scroll to the speaker. Also note that the map
 127         // does not necessarily define the starting positions. While usually
 128         // best to use the map, the scenarion may explicitly set the positions,
 129         // overriding those found in the map (if any).
 130         {
 131                 int scroll_team = gamestate().first_human_team_ + 1;
 132                 if (scroll_team == 0) {
 133                         scroll_team = 1;
 134                 }
 135                 map_location loc(gamestate().board_.map().starting_position(scroll_team));
 136                 if ((loc.x >= 0) && (loc.y >= 0)) {
 137                         gui_->scroll_to_tile(loc, game_display::WARP);
 138                 }
 139         }
 140 
 141         update_locker lock_display(gui_->video(), is_skipping_replay());
 142         gui_->draw();
 143         get_hotkey_command_executor()->set_button_state();
 144         events::raise_draw_event();
 145 }

 148 void playsingle_controller::play_scenario_init()
 149 {
 150         // At the beginning of the scenario, save a snapshot as replay_start
 151         if(saved_game_.replay_start().empty()){
 152                 saved_game_.replay_start() = to_config();
 153         }
 154         start_game();
 155         if( saved_game_.classification().random_mode != "" && is_networked_mp()) {
 156                 // This won't cause errors later but we should notify the user about it in case he didn't knew it.
 157                 gui2::show_transient_message(
 158                         gui_->video(),
 159                         // TODO: find a better title
 160                         _("Game Error"),
 161                         _("This multiplayer game uses an alternative random mode, if you don't know what this message means, then most likely someone is cheating or someone reloaded a corrupt game.")
 162                 );
 163         }
 164         return;
 165 }

 167 void playsingle_controller::play_scenario_main_loop()
 168 {
 169         LOG_NG << "starting main loop\n" << (SDL_GetTicks() - ticks()) << "\n";
 170 
 171 
 172         // Avoid autosaving after loading, but still
 173         // allow the first turn to have an autosave.
 174         ai_testing::log_game_start();
 175         if(gamestate().board_.teams().empty())
 176         {
 177                 ERR_NG << "Playing game with 0 teams." << std::endl;
 178         }
 179         while(true) {
 180                 try {
 181                         play_turn();
 182                         if (is_regular_game_end()) {
 183                                 turn_data_.send_data();
 184                                 return;
 185                         }
 186                         gamestate_->player_number_ = 1;
 187                 }
 188                 catch(const reset_gamestate_exception& ex) {
 189                         //
 190                         // TODO:
 191                         //
 192                         // The MP replay feature still doesn't work properly (causes OOS)
 193                         // because:
 194                         //
 195                         // 1) The undo stack is not reset along with the gamestate (fixed).
 196                         // 2) The server_request_number_ is not reset along with the
 197                         //    gamestate (fixed).
 198                         // 3) chat and other unsynced actions are inserted in the middle of
 199                         //    the replay bringing the replay_pos in unorder (fixed).
 200                         // 4) untracked changes in side controllers are lost when resetting
 201                         //    gamestate (fixed).
 202                         // 5) The game should have a stricter check for whether the loaded
 203                         //    game is actually a parent of this game.
 204                         // 6) If an action was undone after a game was saved it can cause
 205                         //    OOS if the undone action is in the snapshot of the saved
 206                         //    game (luckily this is never the case for autosaves).
 207                         //
 208                         boost::dynamic_bitset<> local_players;
 209                         local_players.resize(gamestate().board_.teams().size(), true);
 210                         //Preserve side controllers, becasue we won't get the side controoller updates again when replaying.
 211                         for(size_t i = 0; i < local_players.size(); ++i) {
 212                                 local_players[i] = gamestate().board_.teams()[i].is_local();
 213                         }
 214                         reset_gamestate(*ex.level, (*ex.level)["replay_pos"]);
 215                         for(size_t i = 0; i < local_players.size(); ++i) {
 216                                 resources::gameboard->teams()[i].set_local(local_players[i]);
 217                         }
 218                         play_scenario_init();
 219                         replay_.reset(new replay_controller(*this, false, ex.level));
 220                         if(ex.start_replay) {
 221                                 replay_->play_replay();
 222                         }
 223                 }
 224         } //end for loop
 225 }

 227 LEVEL_RESULT playsingle_controller::play_scenario(const config& level)
 228 {
 229         LOG_NG << "in playsingle_controller::play_scenario()...\n";
 230 
 231         // Start music.
 232         for(const config &m : level.child_range("music")) {
 233                 sound::play_music_config(m);
 234         }
 235         sound::commit_music_changes();
 236 
 237         if(!this->is_skipping_replay()) {
 238                 // Combine all the [story] tags into a single config. Handle this here since
 239                 // storyscreen::controller doesn't have a default constructor.
 240                 config cfg;
 241                 for(const auto& iter : level.child_range("story")) {
 242                         cfg.append_children(iter);
 243                 }
 244 
 245                 if(!cfg.empty()) {
 246                         gui2::dialogs::story_viewer::display(get_scenario_name(), cfg, gui_->video());
 247                 }
 248         }
 249         gui_->labels().read(level);
 250 
 251         // Read sound sources
 252         assert(soundsources_manager_ != nullptr);
 253         for (const config &s : level.child_range("sound_source")) {
 254                 try {
 255                         soundsource::sourcespec spec(s);
 256                         soundsources_manager_->add(spec);
 257                 } catch (bad_lexical_cast &) {
 258                         ERR_NG << "Error when parsing sound_source config: bad lexical cast." << std::endl;
 259                         ERR_NG << "sound_source config was: " << s.debug() << std::endl;
 260                         ERR_NG << "Skipping this sound source..." << std::endl;
 261                 }
 262         }
 263         LOG_NG << "entering try... " << (SDL_GetTicks() - ticks()) << "\n";
 264         try {
 265                 play_scenario_init();
 266                 // clears level config;
 267                 this->saved_game_.remove_snapshot();
 268 
 269                 if (!is_regular_game_end() && !linger_) {
 270                         play_scenario_main_loop();
 271                 }
 272                 if (game_config::exit_at_end) {
 273                         exit(0);
 274                 }
 275                 const bool is_victory = get_end_level_data_const().is_victory;
 276 
 277                 if(gamestate().gamedata_.phase() <= game_data::PRESTART) {
 278                         sdl::draw_solid_tinted_rectangle(
 279                                 0, 0, gui_->video().getx(), gui_->video().gety(), 0, 0, 0, 1.0,
 280                                 gui_->video().getSurface()
 281                                 );
 282                 }
 283 
 284                 ai_testing::log_game_end();
 285 
 286                 const end_level_data& end_level = get_end_level_data_const();
 287 
 288                 if (gamestate().board_.teams().empty())
 289                 {
 290                         //store persistent teams
 291                         saved_game_.set_snapshot(config());
 292 
 293                         return LEVEL_RESULT::VICTORY; // this is probably only a story scenario, i.e. has its endlevel in the prestart event
 294                 }
 295                 if(linger_) {
 296                         LOG_NG << "resuming from loaded linger state...\n";
 297                         //as carryover information is stored in the snapshot, we have to re-store it after loading a linger state
 298                         saved_game_.set_snapshot(config());
 299                         if(!is_observer()) {
 300                                 persist_.end_transaction();
 301                         }
 302                         return LEVEL_RESULT::VICTORY;
 303                 }
 304                 pump().fire(is_victory ? "local_victory" : "local_defeat");
 305                 { // Block for set_scontext_synced_base
 306                         set_scontext_synced_base sync;
 307                         pump().fire(end_level.proceed_to_next_level ? "victory" : "defeat");
 308                         pump().fire("scenario_end");
 309                 }
 310                 if(end_level.proceed_to_next_level) {
 311                         gamestate().board_.heal_all_survivors();
 312                 }
 313                 if(is_observer()) {
 314                         gui2::show_transient_message(gui_->video(), _("Game Over"), _("The game is over."));
 315                         return LEVEL_RESULT::OBSERVER_END;
 316                 }
 317                 // If we're a player, and the result is victory/defeat, then send
 318                 // a message to notify the server of the reason for the game ending.
 319                 send_to_wesnothd(config_of
 320                         ("info", config_of
 321                                 ("type", "termination")
 322                                 ("condition", "game over")
 323                                 ("result", is_victory ? "victory" : "defeat")
 324                         ));
 325                 // Play victory music once all victory events
 326                 // are finished, if we aren't observers and the
 327                 // carryover dialog isn't disabled.
 328                 //
 329                 // Some scenario authors may use 'continue'
 330                 // result for something that is not story-wise
 331                 // a victory, so let them use [music] tags
 332                 // instead should they want special music.
 333                 const std::string& end_music = select_music(is_victory);
 334                 if((!is_victory || end_level.transient.carryover_report) && !end_music.empty()) {
 335                         sound::empty_playlist();
 336                         sound::play_music_once(end_music);
 337                 }
 338                 persist_.end_transaction();
 339                 return is_victory ? LEVEL_RESULT::VICTORY : LEVEL_RESULT::DEFEAT;
 340         } catch(const savegame::load_game_exception &) {
 341                 // Loading a new game is effectively a quit.
 342                 saved_game_ = saved_game();
 343                 throw;
 344         } catch(wesnothd_error& e) {
 345 
 346                 scoped_savegame_snapshot snapshot(*this);
 347                 savegame::ingame_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 348                 save.save_game_interactive(gui_->video(), _("A network disconnection has occurred, and the game cannot continue. Do you want to save the game?"), savegame::savegame::YES_NO);
 349                 if(dynamic_cast<ingame_wesnothd_error*>(&e)) {
 350                         return LEVEL_RESULT::QUIT;
 351                 } else {
 352                         throw;
 353                 }
 354         }
 355 }

 366 void playsingle_controller::play_side_impl()
 367 {
 368         if (!skip_next_turn_) {
 369                 end_turn_ = END_TURN_NONE;
 370         }
 371         if(replay_.get() != nullptr) {
 372                 REPLAY_RETURN res = replay_->play_side_impl();
 373                 if(res == REPLAY_FOUND_END_TURN) {
 374                         end_turn_ = END_TURN_SYNCED;
 375                 }
 376                 if (player_type_changed_) {
 377                         replay_.reset();
 378                 }
 379         } else if((current_team().is_local_human() && current_team().is_proxy_human())) {
 380                 LOG_NG << "is human...\n";
 381                 // If a side is dead end the turn, but play at least side=1's
 382                 // turn in case all sides are dead
 383                 if (gamestate().board_.side_units(current_side()) == 0 && !(gamestate().board_.units().size() == 0 && current_side() == 1)) {
 384                         end_turn_ = END_TURN_REQUIRED;
 385                 }
 386 
 387                 before_human_turn();
 388                 if (end_turn_ == END_TURN_NONE) {
 389                         play_human_turn();
 390                 }
 391                 if ( !player_type_changed_ && !is_regular_game_end()) {
 392                         after_human_turn();
 393                 }
 394                 LOG_NG << "human finished turn...\n";
 395 
 396         } else if(current_team().is_local_ai() || (current_team().is_local_human() && current_team().is_droid())) {
 397                 play_ai_turn();
 398         } else if(current_team().is_network()) {
 399                 play_network_turn();
 400         } else if(current_team().is_local_human() && current_team().is_idle()) {
 401                 end_turn_enable(false);
 402                 do_idle_notification();
 403                 before_human_turn();
 404                 if (end_turn_ == END_TURN_NONE) {
 405                         play_idle_loop();
 406                 }
 407         }
 408         else {
 409                 // we should have skipped over empty controllers before so this shouldn't be possible
 410                 ERR_NG << "Found invalid side controller " << current_team().controller().to_string() << " (" << current_team().proxy_controller().to_string() << ") for side " << current_team().side() << "\n";
 411         }
 412 }

 414 void playsingle_controller::before_human_turn()
 415 {
 416         log_scope("player turn");
 417         assert(!linger_);
 418         if(end_turn_ != END_TURN_NONE || is_regular_game_end()) {
 419                 return;
 420         }
 421 
 422         if(init_side_done_now_) {
 423                 scoped_savegame_snapshot snapshot(*this);
 424                 savegame::autosave_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 425                 save.autosave(game_config::disable_autosave, preferences::autosavemax(), preferences::INFINITE_AUTO_SAVES);
 426         }
 427 
 428         if(preferences::turn_bell()) {
 429                 sound::play_bell(game_config::sounds::turn_bell);
 430         }
 431 }

 477 void playsingle_controller::linger()
 478 {
 479         LOG_NG << "beginning end-of-scenario linger\n";
 480         linger_ = true;
 481 
 482         // If we need to set the status depending on the completion state
 483         // the key to it is here.
 484         gui_->set_game_mode(game_display::LINGER);
 485 
 486         // change the end-turn button text to its alternate label
 487         gui_->get_theme().refresh_title2("button-endturn", "title2");
 488         gui_->invalidate_theme();
 489         gui_->redraw_everything();
 490 
 491         // End all unit moves
 492         gamestate().board_.set_all_units_user_end_turn();
 493         try {
 494                 // Same logic as single-player human turn, but
 495                 // *not* the same as multiplayer human turn.
 496                 end_turn_enable(true);
 497                 end_turn_ = END_TURN_NONE;
 498                 while(end_turn_ == END_TURN_NONE) {
 499                         play_slice();
 500                         gui_->draw();
 501                 }
 502         } catch(const savegame::load_game_exception &) {
 503                 // Loading a new game is effectively a quit.
 504                 saved_game_ = saved_game();
 505                 throw;
 506         }
 507 
 508         // revert the end-turn button text to its normal label
 509         gui_->get_theme().refresh_title2("button-endturn", "title");
 510         gui_->invalidate_theme();
 511         gui_->redraw_everything();
 512         gui_->set_game_mode(game_display::RUNNING);
 513 
 514         LOG_NG << "ending end-of-scenario linger\n";
 515 }

 531 void playsingle_controller::play_ai_turn()
 532 {
 533         LOG_NG << "is ai...\n";
 534 
 535         end_turn_enable(false);
 536         gui_->recalculate_minimap();
 537 
 538         const cursor::setter cursor_setter(cursor::WAIT);
 539 
 540         // Correct an oddball case where a human could have left delayed shroud
 541         // updates on before giving control to the AI. (The AI does not bother
 542         // with the undo stack, so it cannot delay shroud updates.)
 543         team & cur_team = current_team();
 544         if ( !cur_team.auto_shroud_updates() ) {
 545                 // We just took control, so the undo stack is empty. We still need
 546                 // to record this change for the replay though.
 547                 synced_context::run_and_store("auto_shroud", replay_helper::get_auto_shroud(true));
 548         }
 549         undo_stack().clear();
 550 
 551         turn_data_.send_data();
 552         try {
 553                 try {
 554                         if (!should_return_to_play_side()) {
 555                                 ai::manager::play_turn(current_side());
 556                         }
 557                 }
 558                 catch (return_to_play_side_exception&) {
 559                 }
 560                 catch (fallback_ai_to_human_exception&) {
 561                         current_team().make_human();
 562                         player_type_changed_ = true;
 563                         ai_fallback_ = true;
 564                 }
 565         }
 566         catch(...) {
 567                 turn_data_.sync_network();
 568                 throw;
 569         }
 570         if(!should_return_to_play_side()) {
 571                 end_turn_ = END_TURN_REQUIRED;
 572         }
 573         turn_data_.sync_network();
 574         gui_->recalculate_minimap();
 575         gui_->invalidate_unit();
 576         gui_->invalidate_game_status();
 577         gui_->invalidate_all();
 578         gui_->draw();
 579 }

 610 void playsingle_controller::end_turn(){
 611         if (linger_)
 612                 end_turn_ = END_TURN_REQUIRED;

 615         }
 616 }

 644 void playsingle_controller::sync_end_turn()
 645 {
 646         //We cannot add [end_turn] to the recorder while executing another action.
 647         assert(synced_context::synced_state() == synced_context::UNSYNCED);
 648         if(end_turn_ == END_TURN_REQUIRED && current_team().is_local())
 649         {
 650                 //TODO: we shodul also send this immideateley.
 651                 resources::recorder->end_turn();
 652                 end_turn_ = END_TURN_SYNCED;
 653         }
 654 
 655         assert(end_turn_ == END_TURN_SYNCED);
 656         skip_next_turn_ = false;
 657 
 658         if(ai_fallback_) {
 659                 current_team().make_ai();
 660                 ai_fallback_ = false;
 661         }
 662 }

 696 bool playsingle_controller::should_return_to_play_side() const
