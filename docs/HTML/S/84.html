 116 void playsingle_controller::init_gui(){
 117         LOG_NG << "Initializing GUI... " << (SDL_GetTicks() - ticks()) << "\n";
 118         play_controller::init_gui();
 119 
 120         // Scroll to the starting position of the first team. If there is a
 121         // human team, use that team; otherwise use team 1. If the map defines
 122         // a starting position for the selected team, scroll to that tile. Note
 123         // this often does not matter since many scenario start with messages,
 124         // which will usually scroll to the speaker. Also note that the map
 125         // does not necessarily define the starting positions. While usually
 126         // best to use the map, the scenarion may explicitly set the positions,
 127         // overriding those found in the map (if any).
 128         {
 129                 int scroll_team = gamestate().first_human_team_ + 1;
 130                 if (scroll_team == 0) {
 131                         scroll_team = 1;
 132                 }
 133                 map_location loc(gamestate().board_.map().starting_position(scroll_team));
 134                 if ((loc.x >= 0) && (loc.y >= 0)) {
 135                         gui_->scroll_to_tile(loc, game_display::WARP);
 136                 }
 137         }
 138 
 139         update_locker lock_display(gui_->video(), is_skipping_replay());
 140         gui_->draw();
 141         get_hotkey_command_executor()->set_button_state();
 142         events::raise_draw_event();
 143 }

 146 void playsingle_controller::play_scenario_init()
 147 {
 148         // At the beginning of the scenario, save a snapshot as replay_start
 149         if(saved_game_.replay_start().empty()){
 150                 saved_game_.replay_start() = to_config();
 151         }
 152         start_game();
 153         if( saved_game_.classification().random_mode != "" && is_networked_mp()) {
 154                 // This won't cause errors later but we should notify the user about it in case he didn't knew it.
 155                 gui2::show_transient_message(
 156                         gui_->video(),
 157                         // TODO: find a better title
 158                         _("Game Error"),
 159                         _("This multiplayer game uses an alternative random mode, if you don't know what this message means, then most likely someone is cheating or someone reloaded a corrupt game.")
 160                 );
 161         }
 162         return;
 163 }

 165 void playsingle_controller::play_scenario_main_loop()
 166 {
 167         LOG_NG << "starting main loop\n" << (SDL_GetTicks() - ticks()) << "\n";
 168 
 169 
 170         // Avoid autosaving after loading, but still
 171         // allow the first turn to have an autosave.
 172         ai_testing::log_game_start();
 173         if(gamestate().board_.teams().empty())
 174         {
 175                 ERR_NG << "Playing game with 0 teams." << std::endl;
 176         }
 177         while(true) {
 178                 try {
 179                         play_turn();
 180                         if (is_regular_game_end()) {
 181                                 turn_data_.send_data();
 182                                 return;
 183                         }
 184                         gamestate_->player_number_ = 1;
 185                 }
 186                 catch(const reset_gamestate_exception& ex) {
 187                         //
 188                         // TODO:
 189                         //
 190                         // The MP replay feature still doesn't work properly (causes OOS)
 191                         // because:
 192                         //
 193                         // 1) The undo stack is not reset along with the gamestate (fixed).
 194                         // 2) The server_request_number_ is not reset along with the
 195                         //    gamestate (fixed).
 196                         // 3) chat and other unsynced actions are inserted in the middle of
 197                         //    the replay bringing the replay_pos in unorder (fixed).
 198                         // 4) untracked changes in side controllers are lost when resetting
 199                         //    gamestate (fixed).
 200                         // 5) The game should have a stricter check for whether the loaded
 201                         //    game is actually a parent of this game.
 202                         // 6) If an action was undone after a game was saved it can cause
 203                         //    OOS if the undone action is in the snapshot of the saved
 204                         //    game (luckily this is never the case for autosaves).
 205                         //
 206                         boost::dynamic_bitset<> local_players;
 207                         local_players.resize(gamestate().board_.teams().size(), true);
 208                         //Preserve side controllers, becasue we won't get the side controoller updates again when replaying.
 209                         for(size_t i = 0; i < local_players.size(); ++i) {
 210                                 local_players[i] = gamestate().board_.teams()[i].is_local();
 211                         }
 212                         reset_gamestate(*ex.level, (*ex.level)["replay_pos"]);
 213                         for(size_t i = 0; i < local_players.size(); ++i) {
 214                                 resources::gameboard->teams()[i].set_local(local_players[i]);
 215                         }
 216                         play_scenario_init();
 217                         replay_.reset(new replay_controller(*this, false, ex.level));
 218                         if(ex.start_replay) {
 219                                 replay_->play_replay();
 220                         }
 221                 }
 222         } //end for loop
 223 }

 225 LEVEL_RESULT playsingle_controller::play_scenario(const config& level)
 226 {
 227         LOG_NG << "in playsingle_controller::play_scenario()...\n";
 228 
 229         // Start music.
 230         for(const config &m : level.child_range("music")) {
 231                 sound::play_music_config(m);
 232         }
 233         sound::commit_music_changes();
 234 
 235         if(!this->is_skipping_replay()) {
 236                 // Combine all the [story] tags into a single config. Handle this here since
 237                 // storyscreen::controller doesn't have a default constructor.
 238                 config cfg;
 239                 for(const auto& iter : level.child_range("story")) {
 240                         cfg.append_children(iter);
 241                 }
 242 
 243                 if(!cfg.empty()) {
 244                         gui2::dialogs::story_viewer::display(get_scenario_name(), cfg, gui_->video());
 245                 }
 246         }
 247         gui_->labels().read(level);
 248 
 249         // Read sound sources
 250         assert(soundsources_manager_ != nullptr);
 251         for (const config &s : level.child_range("sound_source")) {
 252                 try {
 253                         soundsource::sourcespec spec(s);
 254                         soundsources_manager_->add(spec);
 255                 } catch (bad_lexical_cast &) {
 256                         ERR_NG << "Error when parsing sound_source config: bad lexical cast." << std::endl;
 257                         ERR_NG << "sound_source config was: " << s.debug() << std::endl;
 258                         ERR_NG << "Skipping this sound source..." << std::endl;
 259                 }
 260         }
 261         LOG_NG << "entering try... " << (SDL_GetTicks() - ticks()) << "\n";
 262         try {
 263                 play_scenario_init();
 264                 // clears level config;
 265                 this->saved_game_.remove_snapshot();
 266 
 267                 if (!is_regular_game_end() && !linger_) {
 268                         play_scenario_main_loop();
 269                 }
 270                 if (game_config::exit_at_end) {
 271                         exit(0);
 272                 }
 273                 const bool is_victory = get_end_level_data_const().is_victory;
 274 
 275                 if(gamestate().gamedata_.phase() <= game_data::PRESTART) {
 276                         if(sdl::window* window = gui_->video().get_window()) {
 277                                 window->fill(0, 0, 0, 255);
 278                         }
 279                 }
 280 
 281                 ai_testing::log_game_end();
 282 
 283                 const end_level_data& end_level = get_end_level_data_const();
 284 
 285                 if (gamestate().board_.teams().empty())
 286                 {
 287                         //store persistent teams
 288                         saved_game_.set_snapshot(config());
 289 
 290                         return LEVEL_RESULT::VICTORY; // this is probably only a story scenario, i.e. has its endlevel in the prestart event
 291                 }
 292                 if(linger_) {
 293                         LOG_NG << "resuming from loaded linger state...\n";
 294                         //as carryover information is stored in the snapshot, we have to re-store it after loading a linger state
 295                         saved_game_.set_snapshot(config());
 296                         if(!is_observer()) {
 297                                 persist_.end_transaction();
 298                         }
 299                         return LEVEL_RESULT::VICTORY;
 300                 }
 301                 pump().fire(is_victory ? "local_victory" : "local_defeat");
 302                 { // Block for set_scontext_synced_base
 303                         set_scontext_synced_base sync;
 304                         pump().fire(end_level.proceed_to_next_level ? "victory" : "defeat");
 305                         pump().fire("scenario_end");
 306                 }
 307                 if(end_level.proceed_to_next_level) {
 308                         gamestate().board_.heal_all_survivors();
 309                 }
 310                 if(is_observer()) {
 311                         gui2::show_transient_message(gui_->video(), _("Game Over"), _("The game is over."));
 312                         return LEVEL_RESULT::OBSERVER_END;
 313                 }
 314                 // If we're a player, and the result is victory/defeat, then send
 315                 // a message to notify the server of the reason for the game ending.
 316                 send_to_wesnothd(config {
 317                         "info", config {
 318                                 "type", "termination",
 319                                 "condition", "game over",
 320                                 "result", is_victory ? "victory" : "defeat",
 321                         },
 322                 });
 323                 // Play victory music once all victory events
 324                 // are finished, if we aren't observers and the
 325                 // carryover dialog isn't disabled.
 326                 //
 327                 // Some scenario authors may use 'continue'
 328                 // result for something that is not story-wise
 329                 // a victory, so let them use [music] tags
 330                 // instead should they want special music.
 331                 const std::string& end_music = select_music(is_victory);
 332                 if((!is_victory || end_level.transient.carryover_report) && !end_music.empty()) {
 333                         sound::empty_playlist();
 334                         sound::play_music_once(end_music);
 335                 }
 336                 persist_.end_transaction();
 337                 return is_victory ? LEVEL_RESULT::VICTORY : LEVEL_RESULT::DEFEAT;
 338         } catch(const savegame::load_game_exception &) {
 339                 // Loading a new game is effectively a quit.
 340                 saved_game_ = saved_game();
 341                 throw;
 342         } catch(wesnothd_error& e) {
 343 
 344                 scoped_savegame_snapshot snapshot(*this);
 345                 savegame::ingame_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 346                 save.save_game_interactive(gui_->video(), _("A network disconnection has occurred, and the game cannot continue. Do you want to save the game?"), savegame::savegame::YES_NO);
 347                 if(dynamic_cast<ingame_wesnothd_error*>(&e)) {
 348                         return LEVEL_RESULT::QUIT;
 349                 } else {
 350                         throw;
 351                 }
 352         }
 353 }

 364 void playsingle_controller::play_side_impl()
 365 {
 366         if (!skip_next_turn_) {
 367                 end_turn_ = END_TURN_NONE;
 368         }
 369         if(replay_.get() != nullptr) {
 370                 REPLAY_RETURN res = replay_->play_side_impl();
 371                 if(res == REPLAY_FOUND_END_TURN) {
 372                         end_turn_ = END_TURN_SYNCED;
 373                 }
 374                 if (player_type_changed_) {
 375                         replay_.reset();
 376                 }
 377         } else if((current_team().is_local_human() && current_team().is_proxy_human())) {
 378                 LOG_NG << "is human...\n";
 379                 // If a side is dead end the turn, but play at least side=1's
 380                 // turn in case all sides are dead
 381                 if (gamestate().board_.side_units(current_side()) == 0 && !(gamestate().board_.units().size() == 0 && current_side() == 1)) {
 382                         end_turn_ = END_TURN_REQUIRED;
 383                 }
 384 
 385                 before_human_turn();
 386                 if (end_turn_ == END_TURN_NONE) {
 387                         play_human_turn();
 388                 }
 389                 if ( !player_type_changed_ && !is_regular_game_end()) {
 390                         after_human_turn();
 391                 }
 392                 LOG_NG << "human finished turn...\n";
 393 
 394         } else if(current_team().is_local_ai() || (current_team().is_local_human() && current_team().is_droid())) {
 395                 play_ai_turn();
 396         } else if(current_team().is_network()) {
 397                 play_network_turn();
 398         } else if(current_team().is_local_human() && current_team().is_idle()) {
 399                 end_turn_enable(false);
 400                 do_idle_notification();
 401                 before_human_turn();
 402                 if (end_turn_ == END_TURN_NONE) {
 403                         play_idle_loop();
 404                 }
 405         }
 406         else {
 407                 // we should have skipped over empty controllers before so this shouldn't be possible
 408                 ERR_NG << "Found invalid side controller " << current_team().controller().to_string() << " (" << current_team().proxy_controller().to_string() << ") for side " << current_team().side() << "\n";
 409         }
 410 }

 412 void playsingle_controller::before_human_turn()
 413 {
 414         log_scope("player turn");
 415         assert(!linger_);
 416         if(end_turn_ != END_TURN_NONE || is_regular_game_end()) {
 417                 return;
 418         }
 419 
 420         if(init_side_done_now_) {
 421                 scoped_savegame_snapshot snapshot(*this);
 422                 savegame::autosave_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 423                 save.autosave(game_config::disable_autosave, preferences::autosavemax(), preferences::INFINITE_AUTO_SAVES);
 424         }
 425 
 426         if(preferences::turn_bell()) {
 427                 sound::play_bell(game_config::sounds::turn_bell);
 428         }
 429 }

 475 void playsingle_controller::linger()
 476 {
 477         LOG_NG << "beginning end-of-scenario linger\n";
 478         linger_ = true;
 479 
 480         // If we need to set the status depending on the completion state
 481         // the key to it is here.
 482         gui_->set_game_mode(game_display::LINGER);
 483 
 484         // change the end-turn button text to its alternate label
 485         gui_->get_theme().refresh_title2("button-endturn", "title2");
 486         gui_->invalidate_theme();
 487         gui_->redraw_everything();
 488 
 489         // End all unit moves
 490         gamestate().board_.set_all_units_user_end_turn();
 491         try {
 492                 // Same logic as single-player human turn, but
 493                 // *not* the same as multiplayer human turn.
 494                 end_turn_enable(true);
 495                 end_turn_ = END_TURN_NONE;
 496                 while(end_turn_ == END_TURN_NONE) {
 497                         play_slice();
 498                         gui_->draw();
 499                 }
 500         } catch(const savegame::load_game_exception &) {
 501                 // Loading a new game is effectively a quit.
 502                 saved_game_ = saved_game();
 503                 throw;
 504         }
 505 
 506         // revert the end-turn button text to its normal label
 507         gui_->get_theme().refresh_title2("button-endturn", "title");
 508         gui_->invalidate_theme();
 509         gui_->redraw_everything();
 510         gui_->set_game_mode(game_display::RUNNING);
 511 
 512         LOG_NG << "ending end-of-scenario linger\n";
 513 }

 529 void playsingle_controller::play_ai_turn()
 530 {
 531         LOG_NG << "is ai...\n";
 532 
 533         end_turn_enable(false);
 534         gui_->recalculate_minimap();
 535 
 536         const cursor::setter cursor_setter(cursor::WAIT);
 537 
 538         // Correct an oddball case where a human could have left delayed shroud
 539         // updates on before giving control to the AI. (The AI does not bother
 540         // with the undo stack, so it cannot delay shroud updates.)
 541         team & cur_team = current_team();
 542         if ( !cur_team.auto_shroud_updates() ) {
 543                 // We just took control, so the undo stack is empty. We still need
 544                 // to record this change for the replay though.
 545                 synced_context::run_and_store("auto_shroud", replay_helper::get_auto_shroud(true));
 546         }
 547         undo_stack().clear();
 548 
 549         turn_data_.send_data();
 550         try {
 551                 try {
 552                         if (!should_return_to_play_side()) {
 553                                 ai::manager::play_turn(current_side());
 554                         }
 555                 }
 556                 catch (return_to_play_side_exception&) {
 557                 }
 558                 catch (fallback_ai_to_human_exception&) {
 559                         current_team().make_human();
 560                         player_type_changed_ = true;
 561                         ai_fallback_ = true;
 562                 }
 563         }
 564         catch(...) {
 565                 turn_data_.sync_network();
 566                 throw;
 567         }
 568         if(!should_return_to_play_side()) {
 569                 end_turn_ = END_TURN_REQUIRED;
 570         }
 571         turn_data_.sync_network();
 572         gui_->recalculate_minimap();
 573         gui_->invalidate_unit();
 574         gui_->invalidate_game_status();
 575         gui_->invalidate_all();
 576         gui_->draw();
 577 }

 642 void playsingle_controller::sync_end_turn()
 643 {
 644         //We cannot add [end_turn] to the recorder while executing another action.
 645         assert(synced_context::synced_state() == synced_context::UNSYNCED);
 646         if(end_turn_ == END_TURN_REQUIRED && current_team().is_local())
 647         {
 648                 //TODO: we shodul also send this immideateley.
 649                 resources::recorder->end_turn();
 650                 end_turn_ = END_TURN_SYNCED;
 651         }
 652 
 653         assert(end_turn_ == END_TURN_SYNCED);
 654         skip_next_turn_ = false;
 655 
 656         if(ai_fallback_) {
 657                 current_team().make_ai();
 658                 ai_fallback_ = false;
 659         }
 660 }

 694 bool playsingle_controller::should_return_to_play_side() const
 695 {
 696         if(player_type_changed_ || is_regular_game_end()) {
 697                 return true;
 698         }
 699         else if (end_turn_ == END_TURN_NONE || replay_.get() != 0 || current_team().is_network()) {
 700                 return false;
 701         }
 702         else {
 703                 return true;
 704         }
