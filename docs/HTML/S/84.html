 116 void playsingle_controller::init_gui(){
 117         LOG_NG << "Initializing GUI... " << (SDL_GetTicks() - ticks()) << "\n";
 118         play_controller::init_gui();
 119 
 120         // Scroll to the starting position of the first team. If there is a
 121         // human team, use that team; otherwise use team 1. If the map defines
 122         // a starting position for the selected team, scroll to that tile. Note
 123         // this often does not matter since many scenario start with messages,
 124         // which will usually scroll to the speaker. Also note that the map
 125         // does not necessarily define the starting positions. While usually
 126         // best to use the map, the scenarion may explicitly set the positions,
 127         // overriding those found in the map (if any).
 128         {
 129                 int scroll_team = gamestate().first_human_team_ + 1;
 130                 if (scroll_team == 0) {
 131                         scroll_team = 1;
 132                 }
 133                 map_location loc(gamestate().board_.map().starting_position(scroll_team));
 134                 if ((loc.x >= 0) && (loc.y >= 0)) {
 135                         gui_->scroll_to_tile(loc, game_display::WARP);
 136                 }
 137         }
 138 
 139         update_locker lock_display(gui_->video(), is_skipping_replay());
 140         gui_->draw();
 141         get_hotkey_command_executor()->set_button_state();
 142         events::raise_draw_event();
 143 }

 146 void playsingle_controller::play_scenario_init()
 147 {
 148         // At the beginning of the scenario, save a snapshot as replay_start
 149         if(saved_game_.replay_start().empty()){
 150                 saved_game_.replay_start() = to_config();
 151         }
 152         start_game();
 153         if( saved_game_.classification().random_mode != "" && is_networked_mp()) {
 154                 // This won't cause errors later but we should notify the user about it in case he didn't knew it.
 155                 gui2::show_transient_message(
 156                         gui_->video(),
 157                         // TODO: find a better title
 158                         _("Game Error"),
 159                         _("This multiplayer game uses an alternative random mode, if you don't know what this message means, then most likely someone is cheating or someone reloaded a corrupt game.")
 160                 );
 161         }
 162         return;
 163 }

 165 void playsingle_controller::play_scenario_main_loop()
 166 {
 167         LOG_NG << "starting main loop\n" << (SDL_GetTicks() - ticks()) << "\n";
 168 
 169 
 170         // Avoid autosaving after loading, but still
 171         // allow the first turn to have an autosave.
 172         ai_testing::log_game_start();
 173         if(gamestate().board_.teams().empty())
 174         {
 175                 ERR_NG << "Playing game with 0 teams." << std::endl;
 176         }
 177         while(true) {
 178                 try {
 179                         play_turn();
 180                         if (is_regular_game_end()) {
 181                                 turn_data_.send_data();
 182                                 return;
 183                         }
 184                         gamestate_->player_number_ = 1;
 185                 }
 186                 catch(const reset_gamestate_exception& ex) {
 187                         //
 188                         // TODO:
 189                         //
 190                         // The MP replay feature still doesn't work properly (causes OOS)
 191                         // because:
 192                         //
 193                         // 1) The undo stack is not reset along with the gamestate (fixed).
 194                         // 2) The server_request_number_ is not reset along with the
 195                         //    gamestate (fixed).
 196                         // 3) chat and other unsynced actions are inserted in the middle of
 197                         //    the replay bringing the replay_pos in unorder (fixed).
 198                         // 4) untracked changes in side controllers are lost when resetting
 199                         //    gamestate (fixed).
 200                         // 5) The game should have a stricter check for whether the loaded
 201                         //    game is actually a parent of this game.
 202                         // 6) If an action was undone after a game was saved it can cause
 203                         //    OOS if the undone action is in the snapshot of the saved
 204                         //    game (luckily this is never the case for autosaves).
 205                         //
 206                         boost::dynamic_bitset<> local_players;
 207                         local_players.resize(gamestate().board_.teams().size(), true);
 208                         //Preserve side controllers, becasue we won't get the side controoller updates again when replaying.
 209                         for(size_t i = 0; i < local_players.size(); ++i) {
 210                                 local_players[i] = gamestate().board_.teams()[i].is_local();
 211                         }
 212                         reset_gamestate(*ex.level, (*ex.level)["replay_pos"]);
 213                         for(size_t i = 0; i < local_players.size(); ++i) {
 214                                 resources::gameboard->teams()[i].set_local(local_players[i]);
 215                         }
 216                         play_scenario_init();
 217                         replay_.reset(new replay_controller(*this, false, ex.level));
 218                         if(ex.start_replay) {
 219                                 replay_->play_replay();
 220                         }
 221                 }
 222         } //end for loop
 223 }

 225 LEVEL_RESULT playsingle_controller::play_scenario(const config& level)
 226 {
 227         LOG_NG << "in playsingle_controller::play_scenario()...\n";
 228 
 229         // Start music.
 230         for(const config &m : level.child_range("music")) {
 231                 sound::play_music_config(m);
 232         }
 233         sound::commit_music_changes();
 234 
 235         if(!this->is_skipping_replay()) {
 236                 // Combine all the [story] tags into a single config. Handle this here since
 237                 // storyscreen::controller doesn't have a default constructor.
 238                 config cfg;
 239                 for(const auto& iter : level.child_range("story")) {
 240                         cfg.append_children(iter);
 241                 }
 242 
 243                 if(!cfg.empty()) {
 244                         gui2::dialogs::story_viewer::display(get_scenario_name(), cfg, gui_->video());
 245                 }
 246         }
 247         gui_->labels().read(level);
 248 
 249         // Read sound sources
 250         assert(soundsources_manager_ != nullptr);
 251         for (const config &s : level.child_range("sound_source")) {
 252                 try {
 253                         soundsource::sourcespec spec(s);
 254                         soundsources_manager_->add(spec);
 255                 } catch (bad_lexical_cast &) {
 256                         ERR_NG << "Error when parsing sound_source config: bad lexical cast." << std::endl;
 257                         ERR_NG << "sound_source config was: " << s.debug() << std::endl;
 258                         ERR_NG << "Skipping this sound source..." << std::endl;
 259                 }
 260         }
 261         LOG_NG << "entering try... " << (SDL_GetTicks() - ticks()) << "\n";
 262         try {
 263                 play_scenario_init();
 264                 // clears level config;
 265                 this->saved_game_.remove_snapshot();
 266 
 267                 if (!is_regular_game_end() && !linger_) {
 268                         play_scenario_main_loop();
 269                 }
 270                 if (game_config::exit_at_end) {
 271                         exit(0);
 272                 }
 273                 const bool is_victory = get_end_level_data_const().is_victory;
 274 
 275                 if(gamestate().gamedata_.phase() <= game_data::PRESTART) {
 276                         sdl::draw_solid_tinted_rectangle(
 277                                 0, 0, gui_->video().getx(), gui_->video().gety(), 0, 0, 0, 1.0,
 278                                 gui_->video().getSurface()
 279                                 );
 280                 }
 281 
 282                 ai_testing::log_game_end();
 283 
 284                 const end_level_data& end_level = get_end_level_data_const();
 285 
 286                 if (gamestate().board_.teams().empty())
 287                 {
 288                         //store persistent teams
 289                         saved_game_.set_snapshot(config());
 290 
 291                         return LEVEL_RESULT::VICTORY; // this is probably only a story scenario, i.e. has its endlevel in the prestart event
 292                 }
 293                 if(linger_) {
 294                         LOG_NG << "resuming from loaded linger state...\n";
 295                         //as carryover information is stored in the snapshot, we have to re-store it after loading a linger state
 296                         saved_game_.set_snapshot(config());
 297                         if(!is_observer()) {
 298                                 persist_.end_transaction();
 299                         }
 300                         return LEVEL_RESULT::VICTORY;
 301                 }
 302                 pump().fire(is_victory ? "local_victory" : "local_defeat");
 303                 { // Block for set_scontext_synced_base
 304                         set_scontext_synced_base sync;
 305                         pump().fire(end_level.proceed_to_next_level ? "victory" : "defeat");
 306                         pump().fire("scenario_end");
 307                 }
 308                 if(end_level.proceed_to_next_level) {
 309                         gamestate().board_.heal_all_survivors();
 310                 }
 311                 if(is_observer()) {
 312                         gui2::show_transient_message(gui_->video(), _("Game Over"), _("The game is over."));
 313                         return LEVEL_RESULT::OBSERVER_END;
 314                 }
 315                 // If we're a player, and the result is victory/defeat, then send
 316                 // a message to notify the server of the reason for the game ending.
 317                 send_to_wesnothd(config {
 318                         "info", config {
 319                                 "type", "termination",
 320                                 "condition", "game over",
 321                                 "result", is_victory ? "victory" : "defeat",
 322                         },
 323                 });
 324                 // Play victory music once all victory events
 325                 // are finished, if we aren't observers and the
 326                 // carryover dialog isn't disabled.
 327                 //
 328                 // Some scenario authors may use 'continue'
 329                 // result for something that is not story-wise
 330                 // a victory, so let them use [music] tags
 331                 // instead should they want special music.
 332                 const std::string& end_music = select_music(is_victory);
 333                 if((!is_victory || end_level.transient.carryover_report) && !end_music.empty()) {
 334                         sound::empty_playlist();
 335                         sound::play_music_once(end_music);
 336                 }
 337                 persist_.end_transaction();
 338                 return is_victory ? LEVEL_RESULT::VICTORY : LEVEL_RESULT::DEFEAT;
 339         } catch(const savegame::load_game_exception &) {
 340                 // Loading a new game is effectively a quit.
 341                 saved_game_ = saved_game();
 342                 throw;
 343         } catch(wesnothd_error& e) {
 344 
 345                 scoped_savegame_snapshot snapshot(*this);
 346                 savegame::ingame_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 347                 save.save_game_interactive(gui_->video(), _("A network disconnection has occurred, and the game cannot continue. Do you want to save the game?"), savegame::savegame::YES_NO);
 348                 if(dynamic_cast<ingame_wesnothd_error*>(&e)) {
 349                         return LEVEL_RESULT::QUIT;
 350                 } else {
 351                         throw;
 352                 }
 353         }
 354 }

 365 void playsingle_controller::play_side_impl()
 366 {
 367         if (!skip_next_turn_) {
 368                 end_turn_ = END_TURN_NONE;
 369         }
 370         if(replay_.get() != nullptr) {
 371                 REPLAY_RETURN res = replay_->play_side_impl();
 372                 if(res == REPLAY_FOUND_END_TURN) {
 373                         end_turn_ = END_TURN_SYNCED;
 374                 }
 375                 if (player_type_changed_) {
 376                         replay_.reset();
 377                 }
 378         } else if((current_team().is_local_human() && current_team().is_proxy_human())) {
 379                 LOG_NG << "is human...\n";
 380                 // If a side is dead end the turn, but play at least side=1's
 381                 // turn in case all sides are dead
 382                 if (gamestate().board_.side_units(current_side()) == 0 && !(gamestate().board_.units().size() == 0 && current_side() == 1)) {
 383                         end_turn_ = END_TURN_REQUIRED;
 384                 }
 385 
 386                 before_human_turn();
 387                 if (end_turn_ == END_TURN_NONE) {
 388                         play_human_turn();
 389                 }
 390                 if ( !player_type_changed_ && !is_regular_game_end()) {
 391                         after_human_turn();
 392                 }
 393                 LOG_NG << "human finished turn...\n";
 394 
 395         } else if(current_team().is_local_ai() || (current_team().is_local_human() && current_team().is_droid())) {
 396                 play_ai_turn();
 397         } else if(current_team().is_network()) {
 398                 play_network_turn();
 399         } else if(current_team().is_local_human() && current_team().is_idle()) {
 400                 end_turn_enable(false);
 401                 do_idle_notification();
 402                 before_human_turn();
 403                 if (end_turn_ == END_TURN_NONE) {
 404                         play_idle_loop();
 405                 }
 406         }
 407         else {
 408                 // we should have skipped over empty controllers before so this shouldn't be possible
 409                 ERR_NG << "Found invalid side controller " << current_team().controller().to_string() << " (" << current_team().proxy_controller().to_string() << ") for side " << current_team().side() << "\n";
 410         }
 411 }

 413 void playsingle_controller::before_human_turn()
 414 {
 415         log_scope("player turn");
 416         assert(!linger_);
 417         if(end_turn_ != END_TURN_NONE || is_regular_game_end()) {
 418                 return;
 419         }
 420 
 421         if(init_side_done_now_) {
 422                 scoped_savegame_snapshot snapshot(*this);
 423                 savegame::autosave_savegame save(saved_game_, *gui_, preferences::save_compression_format());
 424                 save.autosave(game_config::disable_autosave, preferences::autosavemax(), preferences::INFINITE_AUTO_SAVES);
 425         }
 426 
 427         if(preferences::turn_bell()) {
 428                 sound::play_bell(game_config::sounds::turn_bell);
 429         }
 430 }

 476 void playsingle_controller::linger()
 477 {
 478         LOG_NG << "beginning end-of-scenario linger\n";
 479         linger_ = true;
 480 
 481         // If we need to set the status depending on the completion state
 482         // the key to it is here.
 483         gui_->set_game_mode(game_display::LINGER);
 484 
 485         // change the end-turn button text to its alternate label
 486         gui_->get_theme().refresh_title2("button-endturn", "title2");
 487         gui_->invalidate_theme();
 488         gui_->redraw_everything();
 489 
 490         // End all unit moves
 491         gamestate().board_.set_all_units_user_end_turn();
 492         try {
 493                 // Same logic as single-player human turn, but
 494                 // *not* the same as multiplayer human turn.
 495                 end_turn_enable(true);
 496                 end_turn_ = END_TURN_NONE;
 497                 while(end_turn_ == END_TURN_NONE) {
 498                         play_slice();
 499                         gui_->draw();
 500                 }
 501         } catch(const savegame::load_game_exception &) {
 502                 // Loading a new game is effectively a quit.
 503                 saved_game_ = saved_game();
 504                 throw;
 505         }
 506 
 507         // revert the end-turn button text to its normal label
 508         gui_->get_theme().refresh_title2("button-endturn", "title");
 509         gui_->invalidate_theme();
 510         gui_->redraw_everything();
 511         gui_->set_game_mode(game_display::RUNNING);
 512 
 513         LOG_NG << "ending end-of-scenario linger\n";
 514 }

 530 void playsingle_controller::play_ai_turn()
 531 {
 532         LOG_NG << "is ai...\n";
 533 
 534         end_turn_enable(false);
 535         gui_->recalculate_minimap();
 536 
 537         const cursor::setter cursor_setter(cursor::WAIT);
 538 
 539         // Correct an oddball case where a human could have left delayed shroud
 540         // updates on before giving control to the AI. (The AI does not bother
 541         // with the undo stack, so it cannot delay shroud updates.)
 542         team & cur_team = current_team();
 543         if ( !cur_team.auto_shroud_updates() ) {
 544                 // We just took control, so the undo stack is empty. We still need
 545                 // to record this change for the replay though.
 546                 synced_context::run_and_store("auto_shroud", replay_helper::get_auto_shroud(true));
 547         }
 548         undo_stack().clear();
 549 
 550         turn_data_.send_data();
 551         try {
 552                 try {
 553                         if (!should_return_to_play_side()) {
 554                                 ai::manager::play_turn(current_side());
 555                         }
 556                 }
 557                 catch (return_to_play_side_exception&) {
 558                 }
 559                 catch (fallback_ai_to_human_exception&) {
 560                         current_team().make_human();
 561                         player_type_changed_ = true;
 562                         ai_fallback_ = true;
 563                 }
 564         }
 565         catch(...) {
 566                 turn_data_.sync_network();
 567                 throw;
 568         }
 569         if(!should_return_to_play_side()) {
 570                 end_turn_ = END_TURN_REQUIRED;
 571         }
 572         turn_data_.sync_network();
 573         gui_->recalculate_minimap();
 574         gui_->invalidate_unit();
 575         gui_->invalidate_game_status();
 576         gui_->invalidate_all();
 577         gui_->draw();
 578 }

 609 void playsingle_controller::end_turn(){
 610         if (linger_)
 611                 end_turn_ = END_TURN_REQUIRED;
 612         else if (!is_browsing() && menu_handler_.end_turn(current_side())){
 613                 end_turn_ = END_TURN_REQUIRED;

 643 void playsingle_controller::sync_end_turn()
 644 {
 645         //We cannot add [end_turn] to the recorder while executing another action.
 646         assert(synced_context::synced_state() == synced_context::UNSYNCED);
 647         if(end_turn_ == END_TURN_REQUIRED && current_team().is_local())
 648         {
 649                 //TODO: we shodul also send this immideateley.
 650                 resources::recorder->end_turn();
 651                 end_turn_ = END_TURN_SYNCED;
 652         }
 653 
 654         assert(end_turn_ == END_TURN_SYNCED);
 655         skip_next_turn_ = false;
 656 
 657         if(ai_fallback_) {
 658                 current_team().make_ai();
 659                 ai_fallback_ = false;
 660         }
 661 }
