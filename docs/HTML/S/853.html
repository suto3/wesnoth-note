  66 SYNCED_COMMAND_HANDLER_FUNCTION(recruit, child, use_undo, show, error_handler)
  67 {
  68         int current_team_num = resources::controller->current_side();
  69         team &current_team = resources::gameboard->get_team(current_team_num);
  70 
  71         map_location loc(child, resources::gamedata);
  72         map_location from(child.child_or_empty("from"), resources::gamedata);
  73         // Validate "from".
  74         if ( !from.valid() ) {
  75                 // This will be the case for AI recruits in replays saved
  76                 // before 1.11.2, so it is not more severe than a warning.
  77                 // EDIT:  we borke compability with 1.11.2 anyway so we should give an error.
  78                 error_handler("Missing leader location for recruitment.\n", false);
  79         }
  80         else if ( resources::gameboard->units().find(from) == resources::gameboard->units().end() ) {
  81                 // Sync problem?
  82                 std::stringstream errbuf;
  83                 errbuf << "Recruiting leader not found at " << from << ".\n";
  84                 error_handler(errbuf.str(), false);
  85         }
  86 
  87         // Get the unit_type ID.
  88         std::string type_id = child["type"];
  89         if ( type_id.empty() ) {
  90                 error_handler("Recruitment is missing a unit type.", true);
  91                 return false;
  92         }
  93 
  94         const unit_type *u_type = unit_types.find(type_id);
  95         if (!u_type) {
  96                 std::stringstream errbuf;
  97                 errbuf << "Recruiting illegal unit: '" << type_id << "'.\n";
  98                 error_handler(errbuf.str(), true);
  99                 return false;
 100         }
 101 
 102         const std::string res = actions::find_recruit_location(current_team_num, loc, from, type_id);
 103         if(!res.empty())
 104         {
 105                 std::stringstream errbuf;
 106                 errbuf << "cannot recruit unit: " << res << "\n";
 107                 error_handler(errbuf.str(), true);
 108                 return false;
 109                 //we are already oos because the unit wasn't created, no need to keep the bookkeeping right...
 110         }
 111         const int beginning_gold = current_team.gold();
 112 
 113 
 114 
 115         if ( u_type->cost() > beginning_gold ) {
 116                 std::stringstream errbuf;
 117                 errbuf << "unit '" << type_id << "' is too expensive to recruit: "
 118                         << u_type->cost() << "/" << beginning_gold << "\n";
 119                 error_handler(errbuf.str(), false);
 120         }
 121 
 122         actions::recruit_unit(*u_type, current_team_num, loc, from, show, use_undo);
 123 
 124         LOG_REPLAY << "recruit: team=" << current_team_num << " '" << type_id << "' at (" << loc
 125                 << ") cost=" << u_type->cost() << " from gold=" << beginning_gold << ' '
 126                 << "-> " << current_team.gold() << "\n";
 127         return true;
 128 }

 148 SYNCED_COMMAND_HANDLER_FUNCTION(attack, child, /*use_undo*/, show, error_handler)
 149 {
 150         const config &destination = child.child("destination");
 151         const config &source = child.child("source");
 152         //check_checksums(*cfg);
 153 
 154         if (!destination) {
 155                 error_handler("no destination found in attack\n", true);
 156                 return false;
 157         }
 158 
 159         if (!source) {
 160                 error_handler("no source found in attack \n", true);
 161                 return false;
 162         }
 163 
 164         //we must get locations by value instead of by references, because the iterators
 165         //may become invalidated later
 166         const map_location src(source, resources::gamedata);
 167         const map_location dst(destination, resources::gamedata);
 168 
 169         int weapon_num = child["weapon"];
 170         // having defender_weapon in the replay fixes a bug (OOS) where one player (or observer) chooses a different defensive weapon.
 171         // Xan pointed out this was a possibility: we calculate defense weapon
 172         // now based on attack_prediction code, but this uses floating point
 173         // calculations, which means that in the case where results are close,
 174         // rounding differences can mean that both ends choose different weapons.
 175         int def_weapon_num = child["defender_weapon"].to_int(-2);
 176         if (def_weapon_num == -2) {
 177                 // Let's not gratuitously destroy backwards compatibility.
 178                 LOG_REPLAY << "Old data, having to guess weapon\n";
 179                 def_weapon_num = -1;
 180         }
 181 
 182         unit_map::iterator u = resources::gameboard->units().find(src);
 183         if (!u.valid()) {
 184                 error_handler("unfound location for source of attack\n", true);
 185                 return false;
 186         }
 187 
 188         if (child.has_attribute("attacker_type")) {
 189                 const std::string &att_type_id = child["attacker_type"];
 190                 if (u->type_id() != att_type_id) {
 191                         WRN_REPLAY << "unexpected attacker type: " << att_type_id << "(game state gives: " << u->type_id() << ")" << std::endl;
 192                 }
 193         }
 194 
 195         if (static_cast<unsigned>(weapon_num) >= u->attacks().size()) {
 196                 error_handler("illegal weapon type in attack\n", true);
 197                 return false;
 198         }
 199 
 200         unit_map::const_iterator tgt = resources::gameboard->units().find(dst);
 201 
 202         if (!tgt.valid()) {
 203                 std::stringstream errbuf;
 204                 errbuf << "unfound defender for attack: " << src << " -> " << dst << '\n';
 205                 error_handler(errbuf.str(), true);
 206                 return false;
 207         }
 208 
 209         if (child.has_attribute("defender_type")) {
 210                 const std::string &def_type_id = child["defender_type"];
 211                 if (tgt->type_id() != def_type_id) {
 212                         WRN_REPLAY << "unexpected defender type: " << def_type_id << "(game state gives: " << tgt->type_id() << ")" << std::endl;
 213                 }
 214         }
 215 
 216         if (def_weapon_num >= static_cast<int>(tgt->attacks().size())) {
 217 
 218                 error_handler("illegal defender weapon type in attack\n", true);
 219                 return false;
 220         }
 221 
 222         DBG_REPLAY << "Attacker XP (before attack): " << u->experience() << "\n";
 223 
 224         resources::undo_stack->clear();
 225         attack_unit_and_advance(src, dst, weapon_num, def_weapon_num, show);
 226         return true;
 227 }

 229 SYNCED_COMMAND_HANDLER_FUNCTION(disband, child, /*use_undo*/, /*show*/, error_handler)
 230 {
 231 
 232         int current_team_num = resources::controller->current_side();
 233         team &current_team = resources::gameboard->get_team(current_team_num);
 234 
 235         const std::string& unit_id = child["value"];
 236         size_t old_size = current_team.recall_list().size();
 237 
 238         // Find the unit in the recall list.
 239         unit_ptr dismissed_unit = current_team.recall_list().find_if_matches_id(unit_id);
 240         assert(dismissed_unit);
 241         //add dismissal to the undo stack
 242         resources::undo_stack->add_dismissal(dismissed_unit);
 243 
 244         current_team.recall_list().erase_if_matches_id(unit_id);
 245 
 246         if (old_size == current_team.recall_list().size()) {
 247                 error_handler("illegal disband\n", true);
 248                 return false;
 249         }
 250         return true;
 251 }

 253 SYNCED_COMMAND_HANDLER_FUNCTION(move, child,  use_undo, show, error_handler)
 254 {
 255         int current_team_num = resources::controller->current_side();
 256         team &current_team = resources::gameboard->get_team(current_team_num);
 257 
 258         std::vector<map_location> steps;
 259 
 260         try {
 261                 read_locations(child,steps);
 262         } catch (std::invalid_argument&) {
 263                 WRN_REPLAY << "Warning: Path data contained something which could not be parsed to a sequence of locations:" << "\n config = " << child.debug() << std::endl;
 264                 return false;
 265         }
 266 
 267         if(steps.empty())
 268         {
 269                 WRN_REPLAY << "Warning: Missing path data found in [move]" << std::endl;
 270                 return false;
 271         }
 272 
 273         const map_location& src = steps.front();
 274         const map_location& dst = steps.back();
 275 
 276         if (src == dst) {
 277                 WRN_REPLAY << "Warning: Move with identical source and destination. Skipping..." << std::endl;
 278                 return false;
 279         }
 280 
 281         // The nominal destination should appear to be unoccupied.
 282         unit_map::iterator u = resources::gameboard->find_visible_unit(dst, current_team);
 283         if ( u.valid() ) {
 284                 WRN_REPLAY << "Warning: Move destination " << dst << " appears occupied." << std::endl;
 285                 // We'll still proceed with this movement, though, since
 286                 // an event might intervene.
 287                 // 'event' doesn't mean wml event but rather it means 'hidden' units form the movers point of view.
 288         }
 289 
 290         u = resources::gameboard->units().find(src);
 291         if (!u.valid()) {
 292                 std::stringstream errbuf;
 293                 errbuf << "unfound location for source of movement: "
 294                         << src << " -> " << dst << '\n';
 295                 error_handler(errbuf.str(), true);
 296                 return false;
 297         }
 298         bool skip_sighted = false;
 299         bool skip_ally_sighted = false;
 300         if(child["skip_sighted"] == "all")
 301         {
 302                 skip_sighted = true;
 303         }
 304         else if(child["skip_sighted"] == "only_ally")
 305         {
 306                 skip_ally_sighted = true;
 307         }
 308 
 309         bool show_move = show;
 310         if ( current_team.is_local_ai() || current_team.is_network_ai())
 311         {
 312                 show_move = show_move && !preferences::skip_ai_moves();
 313         }
 314         actions::move_unit_from_replay(steps, use_undo ? resources::undo_stack : nullptr, skip_sighted, skip_ally_sighted, show_move);
 315 
 316         return true;
 317 }

 319 SYNCED_COMMAND_HANDLER_FUNCTION(fire_event, child,  use_undo, /*show*/, /*error_handler*/)
 320 {
 321         bool undoable = true;
 322 
 323         if(const config &last_select = child.child("last_select"))
 324         {
 325                 //the select event cannot clear the undo stack.
 326                 resources::game_events->pump().fire("select", map_location(last_select, resources::gamedata));
 327         }
 328         const std::string &event_name = child["raise"];
 329         if (const config &source = child.child("source")) {
 330                 undoable = undoable & !resources::game_events->pump().fire(event_name, map_location(source, resources::gamedata));
 331         } else {
 332                 undoable = undoable & !resources::game_events->pump().fire(event_name);
 333         }
 334 
 335         // Not clearing the undo stack here casues OOS because we added an entry to the replay but no entry to the undo stack.
 336         if(use_undo) {
 337                 if(!undoable) {
 338                         resources::undo_stack->clear();
 339                 } else {
 340                         resources::undo_stack->add_dummy();
 341                 }
 342         }
 343         return true;
 344 }

 354 SYNCED_COMMAND_HANDLER_FUNCTION(auto_shroud, child,  use_undo, /*show*/, /*error_handler*/)
 355 {
 356         assert(use_undo);
 357         team &current_team = resources::controller->current_team();
 358 
 359         bool active = child["active"].to_bool();
 360         // We cannot update shroud here like 'if(active) resources::undo_stack->commit_vision();'.
 361         // Becasue the undo.cpp code assumes exactly 1 entry in the undo stack per entry in the replay.
 362         // And doing so would create a second entry in the undo stack for this 'auto_shroud' entry.
 363         current_team.set_auto_shroud_updates(active);
 364         resources::undo_stack->add_auto_shroud(active);
 365         return true;
 366 }

 397 SYNCED_COMMAND_HANDLER_FUNCTION(debug_unit, child,  use_undo, /*show*/, /*error_handler*/)
 398 {
 399         if(use_undo) {
 400                 resources::undo_stack->clear();
 401         }
 402         debug_notification(":unit debug command was used during turn of $player");
 403         map_location loc(child);
 404         const std::string name = child["name"];
 405         const std::string value = child["value"];
 406 
 407         unit_map::iterator i = resources::gameboard->units().find(loc);
 408         if (i == resources::gameboard->units().end()) {
 409                 return false;
 410         }
 411         if (name == "advances" ) {
 412                 int int_value = std::stoi(value);
 413                 for (int levels=0; levels<int_value; levels++) {
 414                         i->set_experience(i->max_experience());
 415 
 416                         advance_unit_at(advance_unit_params(loc).force_dialog(true));
 417                         i = resources::gameboard->units().find(loc);
 418                         if (!i.valid()) {
 419                                 break;
 420                         }
 421                 }
 422         } else if (name == "status" ) {
 423                 for (std::string status : utils::split(value)) {
 424                         bool add = true;
 425                         if (status.length() >= 1 && status[0] == '-') {
 426                                 add = false;
 427                                 status = status.substr(1);
 428                         }
 429                         if (status.empty()) {
 430                                 continue;
 431                         }
 432                         i->set_state(status, add);
 433                 }
 434         } else {
 435                 config cfg;
 436                 i->write(cfg);
 437                 cfg[name] = value;
 438 
 439                 // Attempt to create a new unit. If there are error (such an invalid type key), exit.
 440                 try{
 441                         unit_ptr new_u(new unit(cfg, true));
 442                         new_u->set_location(loc);
 443                         // Don't remove the unit until after we've verified there are no errors in creating the new one,
 444                         // or else the unit would simply be removed from the map with no replacement.
 445                         resources::gameboard->units().erase(loc);
 446                         resources::gameboard->units().insert(new_u);
 447                 } catch(unit_type::error& e) {
 448                         ERR_REPLAY << e.what() << std::endl; // TODO: more appropriate error message log
 449                         return false;
 450                 }
 451         }
 452         if (name == "fail") { //testcase for bug #18488
 453                 assert(i.valid());
 454         }
 455         resources::screen->invalidate(loc);
 456         resources::screen->invalidate_unit();
 457 
 458         return true;
 459 }
 460 SYNCED_COMMAND_HANDLER_FUNCTION(debug_create_unit, child,  use_undo, /*show*/, error_handler)
 461 {
 462         if(use_undo) {
 463                 resources::undo_stack->clear();
 464         }
 465 
 466         debug_notification("A unit was created using debug command during turn of $player");
 467         map_location loc(child);
 468         const unit_race::GENDER gender = string_gender(child["gender"], unit_race::NUM_GENDERS);
 469         const unit_type *u_type = unit_types.find(child["type"]);
 470         if (!u_type) {
 471                 error_handler("Invalid unit type", true);
 472                 return false;
 473         }
 474 
 475         const int side_num = resources::controller
 476                         ? resources::controller->current_side() : 1;
 477 
 478         // Create the unit.
 479         unit_ptr created(new unit(*u_type, side_num, true, gender));
 480         created->new_turn();
 481         // Add the unit to the board.
 482         std::pair<unit_map::iterator, bool> add_result = resources::gameboard->units().replace(loc, created);
 483         resources::screen->invalidate_unit();
 484         resources::game_events->pump().fire("unit_placed", loc);
 485         unit_display::unit_recruited(loc);
 486 
 487         // Village capture?
 488         if ( resources::gameboard->map().is_village(loc) )
 489                 actions::get_village(loc, created->side());
 490 
 491         // Update fog/shroud.
 492         actions::shroud_clearer clearer;
 493         clearer.clear_unit(loc, *created);
 494         clearer.fire_events();
 495         if ( add_result.first.valid() ) // In case sighted events messed with the unit.
 496                 actions::actor_sighted(*add_result.first);
 497 
 498         return true;
 499 }

 516                 resources::undo_stack->clear();
 517         }
 518         debug_notification("kill debug command was used during turn of $player");
 519 
 520         const map_location loc(child["x"].to_int(), child["y"].to_int(), wml_loc());
 521         const unit_map::iterator i = resources::gameboard->units().find(loc);
 522         if (i != resources::gameboard->units().end()) {
 523                 const int dying_side = i->side();
 524                 resources::controller->pump().fire("last_breath", loc, loc);
 525                 if (i.valid()) {

 528                 resources::screen->redraw_minimap();

 530                         i->set_hitpoints(0);
 531                 }
 532                 resources::controller->pump().fire("die", loc, loc);
 533                 if (i.valid()) {
 534                         resources::gameboard->units().erase(i);
 535                 }
 536                 actions::recalculate_fog(dying_side);
 537         }
 538         return true;
