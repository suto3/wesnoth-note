  80 void tod_manager::resolve_random(randomness::rng& r)
  81 {
  82         //process the random_start_time string, which can be boolean yes/no true/false or a
  83         //comma-separated string of integers >= 1 referring to the times_ array indices
  84         std::vector<std::string> output_strings = utils::split(random_tod_.str());
  85         std::vector<int> output;
  86 
  87         try
  88         {
  89                 std::transform(output_strings.begin(), output_strings.end(), std::back_inserter(output),
  90                         [](const std::string& str)
  91                 {
  92                         return std::stoi(str);
  93                 });
  94         }
  95         catch (std::invalid_argument)
  96         {
  97                 // This happens if the random_start_time string is a boolean.
  98                 // Simply ignore the exception.
  99         }
 100 
 101         // Remove non-positive times
 102         output.erase(
 103                 std::remove_if(
 104                         output.begin(),
 105                         output.end(),
 106                         [](int time){ return time <= 0; }),
 107                 output.end());
 108 
 109         if(!output.empty())
 110         {
 111                 int chosen = output[r.next_random() % output.size()];
 112                 currentTime_ = calculate_current_time(times_.size(), turn_, chosen, true);
 113                 r.next_random();
 114         }
 115         else if (random_tod_.to_bool(false))
 116         {
 117                 currentTime_ = calculate_current_time(times_.size(), turn_, r.next_random(), true);
 118         }
 119         random_tod_ = false;
 120 }
 121 config tod_manager::to_config() const
 122 {
 123         config cfg;
 124         cfg["turn_at"] = turn_;
 125         cfg["turns"] = num_turns_;
 126         cfg["current_time"] = currentTime_;
 127         cfg["random_start_time"] = random_tod_;
 128         cfg["it_is_a_new_turn"] = !has_turn_event_fired_;
 129         std::vector<time_of_day>::const_iterator t;
 130         for(t = times_.begin(); t != times_.end(); ++t) {
 131                 t->write(cfg.add_child("time"));
 132         }
 133         for(std::vector<area_time_of_day>::const_iterator i = areas_.begin(); i != areas_.end(); ++i) {
 134                 config& area = cfg.add_child("time_area");
 135                 // if no ranges, then use hexes to generate ranges
 136                 if(i->xsrc.empty() && i->ysrc.empty()) {
 137                         write_location_range(i->hexes, area);
 138                 } else {
 139                         area["x"] = i->xsrc;
 140                         area["y"] = i->ysrc;
 141                 }
 142                 for(t = i->times.begin(); t != i->times.end(); ++t) {
 143                         t->write(area.add_child("time"));
 144                 }
 145                 area["current_time"] = i->currentTime;
 146                 if (!i->id.empty())
 147                         area["id"] = i->id;
 148         }
 149         return cfg;
 150 }

 190 const time_of_day& tod_manager::get_time_of_day(const map_location& loc, int n_turn) const
 191 {
 192         if(n_turn == 0)
 193                 n_turn = turn_;
 194 
 195         if ( loc != map_location::null_location() )
 196         {
 197                 for ( std::vector<area_time_of_day>::const_reverse_iterator
 198                       i = areas_.rbegin(), i_end = areas_.rend(); i != i_end; ++i )
 199                 {
 200                         if (i->hexes.find(loc) != i->hexes.end())
 201                                 return get_time_of_day_turn(i->times, n_turn, i->currentTime);
 202                 }
 203         }
 204 
 205         return get_time_of_day_turn(times_, n_turn, currentTime_);

 208 const time_of_day tod_manager::get_illuminated_time_of_day(const unit_map & units, const gamemap & map, const map_location& loc, int for_turn) const
 209 {
 210         // get ToD ignoring illumination
 211         time_of_day tod = get_time_of_day(loc, for_turn);
 212 
 213         if ( map.on_board_with_border(loc) )
 214         {
 215                 // Now add terrain illumination.
 216                 const int terrain_light = map.get_terrain_info(loc).light_bonus(tod.lawful_bonus);
 217 
 218                 std::vector<int> mod_list;
 219                 std::vector<int> max_list;
 220                 std::vector<int> min_list;
 221                 int most_add = 0;
 222                 int most_sub = 0;
 223 
 224                 // Find the "illuminates" effects from units that can affect loc.
 225                 map_location locs[7];
 226                 locs[0] = loc;
 227                 get_adjacent_tiles(loc,locs+1);
 228                 for ( size_t i = 0; i != 7; ++i ) {
 229                         const unit_map::const_iterator itor = units.find(locs[i]);
 230                         if (itor != units.end() &&
 231                             itor->get_ability_bool("illuminates", *resources::gameboard) &&
 232                             !itor->incapacitated())
 233                         {
 234                                 unit_ability_list illum = itor->get_abilities("illuminates");
 235                                 unit_abilities::effect illum_effect(illum, terrain_light, false);
 236                                 const int unit_mod = illum_effect.get_composite_value();
 237 
 238                                 // Record this value.
 239                                 mod_list.push_back(unit_mod);
 240                                 max_list.push_back(illum.highest("max_value").first);
 241                                 min_list.push_back(illum.lowest("min_value").first);
 242                                 if ( unit_mod > most_add )
 243                                         most_add = unit_mod;
 244                                 else if ( unit_mod < most_sub )
 245                                         most_sub = unit_mod;
 246                         }
 247                 }
 248                 const bool net_darker = most_add < -most_sub;
 249 
 250                 // Apply each unit's effect, tracking the best result.
 251                 int best_result = terrain_light;
 252                 const int base_light = terrain_light + (net_darker ? most_add : most_sub);
 253                 for ( size_t i = 0; i != mod_list.size(); ++i ) {
 254                         int result =
 255                                 bounded_add(base_light, mod_list[i], max_list[i], min_list[i]);
 256 
 257                         if ( net_darker  &&  result < best_result )
 258                                 best_result = result;
 259                         else if ( !net_darker  &&  result > best_result )
 260                                 best_result = result;
 261                 }
 262 
 263                 // Update the object we will return.
 264                 tod.bonus_modified = best_result - tod.lawful_bonus;
 265                 tod.lawful_bonus = best_result;
 266         }
 267 
 268         return tod;
 269 }

 345 void tod_manager::add_time_area(const gamemap & map, const config& cfg)

 371 void tod_manager::remove_time_area(const std::string& area_id)
 372 {
 373         if(area_id.empty()) {
 374                 areas_.clear();
 375         } else {
 376                 // search for all time areas that match the id.
 377                 std::vector<area_time_of_day>::iterator i = areas_.begin();
 378                 while(i != areas_.end()) {
 379                         if((*i).id == area_id) {
 380                                 i = areas_.erase(i);
 381                         } else {
 382                                 ++i;
 383                         }
 384                 }
 385         }
 386         has_tod_bonus_changed_ = true;
 387 }

 437 void tod_manager::set_turn(const int num, game_data* vars, const bool increase_limit_if_needed)
 438 {
 439         has_tod_bonus_changed_ = false;
 440         const int new_turn = std::max<int>(num, 1);
 441         LOG_NG << "changing current turn number from " << turn_ << " to " << new_turn << '\n';
 442         // Correct ToD
 443         set_new_current_times(new_turn);
 444 
 445         if(increase_limit_if_needed && (new_turn > num_turns_) && num_turns_ != -1) {
 446                 set_number_of_turns(new_turn);
 447         }
 448         turn_ = new_turn;
 449         if (vars)
 450                 vars->get_variable("turn_number") = new_turn;
 451 }

 470 int tod_manager::calculate_current_time(
 471         const int number_of_times,
 472         const int for_turn_number,
 473         const int current_time,
 474         const bool only_to_allowed_range) const
 475 {
 476         if (number_of_times == 0) return 0;
 477         int new_current_time = 0;
 478         if(only_to_allowed_range) new_current_time = current_time % number_of_times;
 479         else new_current_time = (current_time + for_turn_number - turn_) % number_of_times;
 480         while(new_current_time < 0) { new_current_time += number_of_times; }
 481         return new_current_time;
 482 }
