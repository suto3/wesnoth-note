 107 void teleport_group::get_teleport_pair(
 108                   teleport_pair& loc_pair
 109                 , const unit& u
 110                 , const bool ignore_units) const
 111 {
 112         const filter_context * fc = resources::filter_con;
 113         assert(fc);
 114 
 115         if (ignore_units) {
 116                 fc = new ignore_units_filter_context(*resources::filter_con);
 117         }
 118 
 119         vconfig filter(cfg_.child_or_empty("filter"), true);
 120         vconfig source(cfg_.child_or_empty("source"), true);
 121         vconfig target(cfg_.child_or_empty("target"), true);
 122         const unit_filter ufilt(filter, resources::filter_con); //Note: Don't use the ignore units filter context here, only for the terrain filters. (That's how it worked before the filter contexts were introduced)
 123         if (ufilt.matches(u)) {
 124                 terrain_filter source_filter(source, fc);
 125                 source_filter.get_locations(reversed_ ? loc_pair.second : loc_pair.first, u);
 126 
 127                 terrain_filter target_filter(target, fc);
 128                 target_filter.get_locations(reversed_ ? loc_pair.first : loc_pair.second, u);
 129         }
 130 
 131         if (ignore_units) {
 132                 delete fc;
 133         }
 134 }

 160 teleport_map::teleport_map(
 161                   const std::vector<teleport_group>& groups
 162                 , const unit& unit
 163                 , const team &viewing_team
 164                 , const bool see_all
 165                 , const bool ignore_units
 166                 , const bool check_vision)
 167         : teleport_map_()
 168         , sources_()
 169         , targets_()
 170 {
 171 
 172         for (const teleport_group& group : groups) {
 173 
 174                 teleport_pair locations;
 175 
 176                 if (check_vision && !group.allow_vision()) {
 177                         continue;
 178                 }
 179 
 180                 group.get_teleport_pair(locations, unit, ignore_units);
 181                 if (!see_all && !group.always_visible() && viewing_team.is_enemy(unit.side())) {
 182                         teleport_pair filter_locs;
 183                         for (const map_location &loc : locations.first) {
 184                                 if(!viewing_team.fogged(loc))
 185                                         filter_locs.first.insert(loc);
 186                         }
 187                         for (const map_location &loc : locations.second) {
 188                                 if(!viewing_team.fogged(loc))
 189                                         filter_locs.second.insert(loc);
 190                         }
 191                         locations.first.swap(filter_locs.first);
 192                         locations.second.swap(filter_locs.second);
 193                 }
 194 
 195                 if (!group.pass_allied_units() && !ignore_units && !check_vision) {
 196                         std::set<map_location>::iterator loc = locations.second.begin();
 197                         while(loc != locations.second.end()) {
 198                                 unit_map::iterator u;
 199                                 if (see_all) {
 200                                         u = resources::gameboard->units().find(*loc);
 201                                 } else {
 202                                         u = resources::gameboard->find_visible_unit(*loc, viewing_team);
 203                                 }
 204                                 if (u != resources::gameboard->units().end()) {
 205                                         loc = locations.second.erase(loc);
 206                                 } else {
 207                                         ++loc;
 208                                 }
 209                         }
 210                 }
 211 
 212                 std::string teleport_id = group.get_teleport_id();
 213                 std::set<map_location>::iterator source_it = locations.first.begin();
 214                 for (; source_it != locations.first.end(); ++source_it ) {
 215                         if(teleport_map_.count(*source_it) == 0) {
 216                                 std::set<std::string> id_set;
 217                                 id_set.insert(teleport_id);
 218                                 teleport_map_.emplace(*source_it, id_set);
 219                         } else {
 220                                 (teleport_map_.find(*source_it)->second).insert(teleport_id);
 221                         }
 222                 }
 223                 sources_.emplace(teleport_id, locations.first);
 224                 targets_.emplace(teleport_id, locations.second);
 225         }
 226 }

 228 void teleport_map::get_adjacents(std::set<map_location>& adjacents, map_location loc) const {
 229 
 230         if (teleport_map_.count(loc) == 0) {
 231                 return;
 232         } else {
 233                 const std::set<std::string>& keyset = (teleport_map_.find(loc)->second);
 234                 for(std::set<std::string>::const_iterator it = keyset.begin(); it != keyset.end(); ++it) {
 235 
 236                         const std::set<map_location>& target = targets_.find(*it)->second;
 237                         adjacents.insert(target.begin(), target.end());
 238                 }
 239         }
 240 }

 259 const teleport_map get_teleport_locations(const unit &u,
 260         const team &viewing_team,
 261         bool see_all, bool ignore_units, bool check_vision)
 262 {
 263         std::vector<teleport_group> groups;
 264 
 265         if (u.get_ability_bool("teleport", *resources::gameboard)) {
 266                 for (const unit_ability & teleport : u.get_abilities("teleport")) {
 267                         const int tunnel_count = (teleport.first)->child_count("tunnel");
 268                         for(int i = 0; i < tunnel_count; ++i) {
 269                                 config teleport_group_cfg = (teleport.first)->child("tunnel", i);
 270                                 groups.emplace_back(vconfig(teleport_group_cfg, true), false);
 271                         }
 272                 }
 273         }
 274 
 275         const std::vector<teleport_group>& global_groups = resources::tunnels->get();
 276         groups.insert(groups.end(), global_groups.begin(), global_groups.end());
 277 
 278         return teleport_map(groups, u, viewing_team, see_all, ignore_units, check_vision);
 279 }

 281 manager::manager(const config &cfg) : tunnels_(), id_(cfg["next_teleport_group_id"].to_int(0)) {
 282         const int tunnel_count = cfg.child_count("tunnel");
 283         for(int i = 0; i < tunnel_count; ++i) {
 284                 const config& t = cfg.child("tunnel", i);
 285                 if(!t["saved"].to_bool()) {
 286                         lg::wml_error() << "Do not use [tunnel] directly in a [scenario]. Use it in an [event] or [abilities] tag.\n";
 287                         continue;
 288                 }
 289                 const teleport_group tunnel(t);
 290                 this->add(tunnel);
 291         }
 292 }

 298 void manager::remove(const std::string &id) {
 299         std::vector<teleport_group>::iterator t = tunnels_.begin();
 300         for(;t != tunnels_.end();) {
 301                 if (t->get_teleport_id() == id || t->get_teleport_id() == id + reversed_suffix) {
 302                         t = tunnels_.erase(t);
 303                 } else {
 304                         ++t;
 305                 }
 306         }
 307 }

 313 config manager::to_config() const {
 314         config store;
 315 
 316         std::vector<teleport_group>::const_iterator tunnel = tunnels_.begin();
 317         for(; tunnel != tunnels_.end(); ++tunnel) {
 318                 store.add_child("tunnel", tunnel->to_config());
 319         }
 320         store["next_teleport_group_id"] = std::to_string(id_);
 321 
 322         return store;
