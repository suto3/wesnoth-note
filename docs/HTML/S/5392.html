  10         local y = cfg.y or helper.wml_error("[spawn_units] missing required y= attribute.")

  18                 if #locs == 0 then locs = wesnoth.get_locations({T["not"] { T.filter {} } , T["and"] { x = x, y = y, radius = 2 } }) end
  19                 if #locs == 0 then break end
  20 
  21                 done = done + 1
  22

  24                 local loc_i = helper.rand("1.."..#locs)
  25 
  26                 wml_actions.move_unit_fake({x = string.format("%d,%d", x, locs[loc_i][1]) , y = string.format("%d,%d", y, locs[loc_i][2]) , type = unit_type , side = side})
  27                 wesnoth.put_unit({ type = unit_type , side = side, random_traits = "yes", generate_name = "yes" , upkeep = "loyal" }, locs[loc_i][1], locs[loc_i][2])
  28         end
  29 
  30         if done > 0 then
  31                 for then_child in helper.child_range(cfg, "then") do
  32                         local action = utils.handle_event_commands(then_child, "conditional")
  33                         if action ~= "none" then return end
  34                 end
  35         end
  36 end
