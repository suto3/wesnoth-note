   4 local function create_game(context)
   5   local events, info
   6 
   7   context.create({})
   8 
   9   repeat
  10     events, context, info = coroutine.yield()
  11   until info.name == "Multiplayer Create"
  12 
  13   context.select_type({type = "scenario"})
  14   local s = info.find_level({id = "test1"})
  15   context.select_level({index = s.index})
  16   context.set_name({name = tostring(wesnoth.random(999999))})
  17   context.update_settings({registered_users = false})
  18 
  19   events, context, info = coroutine.yield()

  21   context.create({})
  22 end

  24 local function exit_game(context)

  26 
  27   repeat
  28     context.quit({})
  29     events, context, info = coroutine.yield()
  30   until info.name == "titlescreen"
  31 
  32   context.exit({code = 0})
  33   coroutine.yield()
  34 end

  37   local events, context, info
  38 
  39   repeat
  40     events, context, info = coroutine.yield()
  41   until info.name == "titlescreen" or info.name == "Multiplayer Lobby"
  42

  46   end

  52   -- Reached the lobby. Random delay before we start actually simulating activity.

  59   local in_staging = false

  61   while true do
  62     if info.name ~= "Multiplayer Lobby" and info.name ~= "Multiplayer Staging" then
  63       -- most often, this means that the server was terminated -> stop generating traffic
  64       exit_game(context)
  65     end
  66 
  67     if wesnoth.random() > 0.1 then
  68       -- chat message
  69       local messages = {"asdf", "qwerty", "zxc"}
  70       context.chat({message = messages[wesnoth.random(#messages)]})
  71     else
  72       -- toggle between creating a game and leaving it
  73       if not in_staging then

  77         repeat
  78           context.quit({})
  79           events, context, info = coroutine.yield()
  80         until info.name == "Multiplayer Lobby"
  81         in_staging = false
  82       end
  83     end

  85     wesnoth.delay(15000)

  88   end
