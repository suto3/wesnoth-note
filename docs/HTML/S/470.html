  75 void core_selection::core_selected(window& window)
  76 {
  77         const int selected_row
  78                         = find_widget<listbox>(&window, "core_list", false)
  79                                           .get_selected_row();
  80 
  81         multi_page& pages
  82                         = find_widget<multi_page>(&window, "core_details", false);
  83 
  84         pages.select_page(selected_row);
  85 }

  87 void core_selection::pre_show(window& window)
  88 {
  89         /***** Setup core list. *****/
  90         listbox& list = find_widget<listbox>(&window, "core_list", false);
  91 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  92         connect_signal_notify_modified(
  93                         list,
  94                         std::bind(&core_selection::core_selected,
  95                                                 this,
  96                                                 std::ref(window)));
  97 #else
  98         list.set_callback_value_change(
  99                         dialog_callback<core_selection,
 100                                                         &core_selection::core_selected>);
 101 #endif
 102         window.keyboard_capture(&list);
 103 
 104         /***** Setup core details. *****/
 105         multi_page& pages
 106                         = find_widget<multi_page>(&window, "core_details", false);
 107 
 108         for(const auto & core : cores_)
 109         {
 110                 /*** Add list item ***/
 111                 string_map list_item;
 112                 std::map<std::string, string_map> list_item_item;
 113 
 114                 list_item["label"] = core["image"];
 115                 list_item_item.emplace("image", list_item);
 116 
 117                 list_item["label"] = core["name"];
 118                 list_item_item.emplace("name", list_item);
 119 
 120                 grid* grid = &list.add_row(list_item_item);
 121                 assert(grid);
 122 
 123                 /*** Add detail item ***/
 124                 string_map detail_item;
 125                 std::map<std::string, string_map> detail_page;
 126 
 127                 detail_item["label"] = core["description"];
 128                 detail_item["use_markup"] = "true";
 129                 detail_page.emplace("description", detail_item);
 130 
 131                 pages.add_page(detail_page);
 132         }
 133         list.select_row(choice_, true);
 134 
 135         core_selected(window);
 136 }
 137 
 138 void core_selection::post_show(window& window)
 139 {
 140         choice_ = find_widget<listbox>(&window, "core_list", false)
 141                                           .get_selected_row();
 142 }
