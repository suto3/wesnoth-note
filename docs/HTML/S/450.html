  31 /**
  32  * This class implements the title screen.
  33  *
  34  * The menu buttons return a result back to the caller with the button pressed.
  35  * So at the moment it only handles the tips itself.
  36  */
  37 class title_screen : public modal_dialog
  38 {
  39 public:
  40         title_screen(game_launcher& game);
  41 
  42         ~title_screen();
  43 
  44         /**
  45          * Values for actions which leave the title screen.
  46          * Actions that merely show a dialog are not included here.
  47          */
  48         enum result {
  49                 // Window was resized, so needs redrawing
  50                 REDRAW_BACKGROUND = 0, // Needs to be 0, the value of gui2::window::NONE
  51                 // Start playing a single-player game, such as the tutorial or a campaign
  52                 LAUNCH_GAME,
  53                 // Connect to an MP server
  54                 MP_CONNECT,
  55                 // Host an MP game
  56                 MP_HOST,
  57                 // Start a local MP game
  58                 MP_LOCAL,
  59                 // Start the map/scenario editor
  60                 MAP_EDITOR,
  61                 // Show credits
  62                 SHOW_ABOUT,
  63                 // Exit to desktop
  64                 QUIT_GAME,
  65                 // Used to reload all game data
  66                 RELOAD_GAME_DATA,
  67         };
  68 
  69 private:
  70         game_launcher& game_;
  71 
  72         /** Inherited from modal_dialog, implemented by REGISTER_DIALOG. */
  73         virtual const std::string& window_id() const override;
  74 
  75         /** Inherited from modal_dialog. */
  76         virtual void pre_show(window& window) override;
  77 
  78         void on_resize(window& window);
  79 
  80         /** Holds the debug clock dialog. */
  81         std::unique_ptr<modeless_dialog> debug_clock_;
  82 
  83         /**
  84          * Updates the tip of day widget.
  85          *
  86          * @param window              The window being shown.
  87          * @param previous            Show the previous tip, else shows the next one.
  88          */
  89         void update_tip(window& window, const bool previous);
  90 
  91         /** Shows the debug clock. */
  92         void show_debug_clock_window(CVideo& video);
  93 };
  94 
  95 } // namespace dialogs
  96 } // namespace gui2
