   1 #define CHECK_RECRUIT CONTENT
   2     [event]
   3         name=recruit
   4         [lua]
   5             code =<<
   6                     local temp = wesnoth.get_variable("unit")
   7                     local result = wesnoth.synchronize_choice(
   8                       function()
   9                         return { value = temp.facing }
  10                       end)
  11                     wesnoth.set_variable("synced_temp", result.value)
  12                 >>
  13         [/lua]
  14         {ASSERT ({VARIABLE_CONDITIONAL unit.facing equals $synced_temp})}
  15         {CONTENT}
  16     [/event]
  17 #enddef
  18 
  19 #define CHECK_FACES FACEDIR

 152 #define RECALL_AND_CHECK TURN FACING LOC
 153     [event]
 154         name=side 1 turn {TURN} refresh
 155         [store_unit]
 156             [filter]
 157                 canrecruit=yes
 158             [/filter]
 159             variable=leader_store
 160         [/store_unit]
 161         {CHECK_FACES {FACING}}

 163             type=Spearman
 164             x,y=recall,recall
 165             side=1
 166             id=test_{TURN}
 167         [/unit]
 168         [do_command]
 169             [recall]
 170                 value=test_{TURN}
 171                 x,y={LOC}
 172             [/recall]
 173         [/do_command]
 174         [unstore_unit]
 175             variable=leader_store
 176         [/unstore_unit]
 177         [end_turn]
 178         [/end_turn]
 179     [/event]
