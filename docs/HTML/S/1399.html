 147     private void addNewItem( List targetList, WMLTag tag )
 148     {
 149         if( ( tag.is_Optional( ) || tag.is_Required( ) )
 150             && targetList.getItemCount( ) == 1 ) {
 151             GUIUtils.showWarnMessageBox( Messages.WizardGeneratorPageTag_12 );
 152             return;
 153         }
 154 
 155         WizardGenerator wizard = new WizardGenerator(
 156             Messages.WizardGeneratorPageTag_13 + tag.getName( ),
 157             tag.getName( ), ( byte ) ( indent_ + 1 ) );
 158         WizardUtils.launchWizard( wizard, getShell( ), null );
 159         if( wizard.isFinished( ) ) {
 160             targetList
 161                 .add( wizard.getObjectName( ) + targetList.getItemCount( ) );
 162             content_.get( tag.getName( ) ).add( wizard.getData( ).toString( ) );
 163         }
 164         updatePageIsComplete( );
 165     }
 166 
 167     private void removeItem( List targetList, String tagName )
 168     {
 169         if( targetList.getSelectionCount( ) == 0
 170             || targetList.getItemCount( ) == 0 ) {
 171             GUIUtils.showWarnMessageBox( Messages.WizardGeneratorPageTag_14 );
 172             return;
 173         }
 174 
 175         content_.get( tagName ).remove( targetList.getSelectionIndex( ) );
 176         targetList.remove( targetList.getSelectionIndex( ) );
 177         updatePageIsComplete( );
 178     }
 179 
 180     private void updatePageIsComplete( )
 181     {
 182         setPageComplete( false );
 183         for( Control control: container_.getChildren( ) ) {
 184             if( ! ( control instanceof Group ) ) {
 185                 continue;
 186             }
 187 
 188             int cnt = ( ( List ) control.getData( "list" ) ).getItemCount( ); //$NON-NLS-1$
 189             WMLTag tag = ( WMLTag ) control.getData( "tag" ); //$NON-NLS-1$
 190             if( cnt == 0 && tag.is_Required( ) ) {
 191                 setErrorMessage( String.format(
 192                     Messages.WizardGeneratorPageTag_0, tag.getName( ) ) );
 193                 return;
 194             }
 195         }
 196 
 197         setPageComplete( true );
 198         setErrorMessage( null );
 199     }

 201     /**
 202      * @return The WML Code generated by this page
 203      */
 204     public String getContent( )
 205     {
 206         StringBuilder result = new StringBuilder( );
 207         for( Entry< String, java.util.List< String >> tag: content_.entrySet( ) ) {
 208             result
 209                 .append( StringUtils.multiples( "\t", indent_ ) + "[" + tag.getKey( ) + "]\n" ); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
 210             result
 211                 .append( ListUtils.concatenateList( tag.getValue( ), "\n\t" ) ); //$NON-NLS-1$
 212             result
 213                 .append( StringUtils.multiples( "\t", indent_ ) + "[/" + tag.getKey( ) + "]\n" ); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
 214         }
 215         return result.toString( );
 216     }
