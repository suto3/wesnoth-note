 117 # Sets a specified flag to all hexes adjacent to a given type which is in turn
 118 # adjacent to another given type. Used to set the transition2 flag for sunken
 119 # keeps, so their cliff transition to water will be disabled.
 120 
 121 #define SET_ADJACENT_FLAG CENTER_TERRAIN ADJACENT_TERRAIN ADJACENT_TO_TERRAIN FLAG
 122     [terrain_graphics]
 123         map="
 124 , 2
 125 .,  .
 126 , 1
 127 .,  .
 128 , ."
 129 
 130         [tile]
 131             pos=1
 132             type={CENTER_TERRAIN}
 133         [/tile]
 134         [tile]
 135             pos=2
 136             type={ADJACENT_TERRAIN}
 137             set_flag="adjacent_to_{ADJACENT_TO_TERRAIN}"
 138             no_flag="adjacent_to_{ADJACENT_TO_TERRAIN}"
 139         [/tile]
 140 
 141         rotations=n,ne,se,s,sw,nw
 142     [/terrain_graphics]
 143 
 144     [terrain_graphics]
 145         map="
 146 , 4
 147 .,  .
 148 , 3
 149 .,  .
 150 , ."
 151 
 152         [tile]
 153             pos=3

 155             has_flag="adjacent_to_{ADJACENT_TO_TERRAIN}"
 156         [/tile]
