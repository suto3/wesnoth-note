 509 #define NEXT_LAVA
 510     [store_locations]
 511         variable=potential_locs
 512         terrain=Xu
 513         [filter_adjacent_location]
 514             terrain=U*, Re
 515             [filter_adjacent_location]
 516                 terrain=U*, Re
 517                 count=4-5
 518             [/filter_adjacent_location]
 519         [/filter_adjacent_location]
 520         [not]
 521             # don't start near the cave floor around the sceptre
 522             x=$sceptre_x
 523             y=$sceptre_y
 524             radius=5
 525         [/not]
 526         [not]
 527             # don't start near the existing lava
 528             terrain=Ql
 529             radius=5
 530         [/not]
 531     [/store_locations]
 532     [if]
 533         [variable]
 534             name=potential_locs.length
 535             greater_than=0
 536         [/variable]
 537         [then]
 538             {VARIABLE lava_count 0}
 539             {VARIABLE randrange $potential_locs.length}
 540             {VARIABLE_OP randrange sub 1}
 541             {VARIABLE randrange "0..$randrange"}
 542             {RANDOM $randrange}
 543             {CLEAR_VARIABLE lava_body}
 544             {VARIABLE lava_body.x $potential_locs[$random].x}
 545             {VARIABLE lava_body.y $potential_locs[$random].y}
 546             {CLEAR_VARIABLE random}
 547             [terrain]
 548                 x=$lava_body.x
 549                 y=$lava_body.y
 550                 terrain=Ql
 551             [/terrain]
 552             {CLEAR_VARIABLE randrange}
 553         [/then]
 554     [/if]
 555     {CLEAR_VARIABLE potential_locs}
 556 #enddef

 558 #define EXPAND_LAVA
 559     # the lava should expand to any cave floor that is not next to a cavewall,
 560     # or cave floor that is next to cavewall that is already next to lava
 561     # also if it comes near another lavabody, it should not expand in that direction
 562 
 563     # first we store cavewall that is not already next to this lava
 564     [store_locations]
 565         variable=far_walls
 566         terrain=Xu
 567         [not]
 568             find_in=lava_body
 569             radius=1
 570         [/not]
 571     [/store_locations]
 572 
 573     # next we store lava that is not part of this body
 574     [store_locations]
 575         variable=far_lava
 576         terrain=Ql
 577         [not]
 578             find_in=lava_body
 579         [/not]
 580     [/store_locations]
 581 
 582     # now we store cave floor nearby that doesn't approach the bad stuff
 583     [store_locations]
 584         variable=good_floor
 585         terrain=U*,Re
 586         [and]
 587             find_in=lava_body
 588             radius=1
 589         [/and]
 590         [not]
 591             find_in=far_walls
 592             radius=1
 593         [/not]
 594         [not]
 595             find_in=far_lava
 596             radius=1
 597         [/not]
 598         [not]
 599             # don't go anywhere near the cave floor around the sceptre
 600             x,y=$sceptre_x,$sceptre_y
 601             radius=4
 602             [filter_radius]
 603                 terrain=U*,Re
 604             [/filter_radius]
 605         [/not]
 606     [/store_locations]
 607 
 608     [if]
 609         [variable]
 610             name=good_floor.length
 611             greater_than=0
 612         [/variable]
 613         [then]
 614             [foreach]
 615                 array=good_floor
 616                 [do]
 617                     [terrain]
 618                         x=$this_item.x
 619                         y=$this_item.y
 620                         terrain=Ql
 621                     [/terrain]
 622                 [/do]
 623             [/foreach]
 624             [store_locations]
 625                 variable=lava_body
 626                 find_in=lava_body
 627                 [or]
 628                     find_in=good_floor
 629                 [/or]
 630             [/store_locations]
 631         [/then]
 632         [else]
 633             # this lava can't expand so force a new eruption
 634             {NEXT_LAVA}
 635         [/else]
 636     [/if]
 637     {CLEAR_VARIABLE far_walls}
 638     {CLEAR_VARIABLE far_lava}
 639     {CLEAR_VARIABLE good_floor}

 645         [/not]
 646         [filter_location]
 647             terrain=Ql
 648         [/filter_location]
 649         animate=yes
 650         fire_event=yes
 651     [/kill]
