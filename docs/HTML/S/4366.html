  70         local function get_reachable_enemy_leaders(unit)
  71             -- We're cheating a little here and also find hidden enemy leaders. That's
  72             -- because a human player could make a pretty good educated guess as to where
  73             -- the enemy leaders are likely to be while the AI does not know how to do that.
  74             local potential_enemy_leaders = AH.get_live_units { canrecruit = 'yes',
  75                 { "filter_side", { { "enemy_of", {side = wesnoth.current.side} } } }
  76             }
  77             local enemy_leaders = {}
  78             for j,e in ipairs(potential_enemy_leaders) do
  79                 local path, cost = wesnoth.find_path(unit, e.x, e.y, { ignore_units = true, viewing_side = 0 })
  80                 if cost < AH.no_path then
  81                     table.insert(enemy_leaders, e)
  82                 end
  83             end
  84 
  85             return enemy_leaders
  86         end

 101                 }[1]

 104                 self.data.leader_target = nil

 114                 if next_hop and next_hop[1] == self.data.leader_target[1]
 115                 and next_hop[2] == self.data.leader_target[2] then
 116                     return self.data.leader_score
 117                 end
 118             end

 122                 terrain = 'K*,K*^*,*^K*', -- Keeps
 123                 x = '1-'..width,
 124                 y = '1-'..height,

 126                 { "not", { radius = 6, {"filter", { canrecruit = 'yes',
 127                     { "filter_side", { { "enemy_of", {side = wesnoth.current.side} } } }
 128                 }} }}, -- That are not too close to an enemy leader
 129                 { "not", {
 130                     x = leader.x, y = leader.y, terrain = 'K*,K*^*,*^K*',
 131                     radius = 3,
 132                     { "filter_radius", { terrain = 'C*,K*,C*^*,K*^*,*^K*,*^C*' } }
 133                 }}, -- That are not close and connected to a keep the leader is on
 134                 { "filter_adjacent_location", {
 135                     terrain = 'C*,K*,C*^*,K*^*,*^K*,*^C*'
 136                 }} -- That are not one-hex keeps
 137             }
 138             if #keeps < 1 then
 139                 -- Skip if there aren't extra keeps to evaluate
 140                 -- In this situation we'd only switch keeps if we were running away
 141                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 142                 return 0
 143             end

 174                     x = leader.x,

 176                     radius = leader.max_moves
 177                     }},
 178                     owner_side = 0
 179                 })[1]

 182                     for j,e in ipairs(enemy_leaders) do
 183                         -- count all distances as three less than they actually are

 207                             local path_keep, cost_keep = wesnoth.find_path(dummy_leader, best_loc[1], best_loc[2])
 208                             local turns_from_keep = math.ceil(cost_keep/leader.max_moves)
 209                             if turns_from_keep < best_turns
 210                             or (turns_from_keep == 1 and wesnoth.sides[wesnoth.current.side].gold < cheapest_unit_cost)
 211                             then
 212                                 -- There is, go there instead

 214                                 break
 215                             end
 216                         end
 217                     end
 218                 end
 219 
 220                 self.data.leader_target = next_hop
 221 
 222                 -- if we're on a keep, wait until there are no movable units on the castle before moving off
 223                 self.data.leader_score = 290000
 224                 if wesnoth.get_terrain_info(wesnoth.get_terrain(leader.x, leader.y)).keep then
 225                     local castle = wesnoth.get_locations {

 240                         end
 241                     end

 244                     end
 245                 end

 267         function generic_rush:grab_villages_eval()
 268             local start_time, ca_name = wesnoth.get_time_stamp() / 1000., 'grab_villages'
 269             if AH.print_eval() then print_time('     - Evaluating grab_villages CA:') end
 270 
 271             -- Check if there are units with moves left
 272             local units = wesnoth.get_units { side = wesnoth.current.side, canrecruit = 'no',
 273                 formula = 'movement_left > 0'
 274             }
 275             if (not units[1]) then
 276                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 277                 return 0
 278             end
 279 
 280             local enemies = AH.get_attackable_enemies()
 281 
 282             local villages = wesnoth.get_villages()
 283             -- Just in case:
 284             if (not villages[1]) then
 285                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 286                 return 0
 287             end
 288             --print('#units, #enemies', #units, #enemies)
 289 
 290             -- First check if attacks are possible for any unit
 291             local return_value = 200000
 292             -- If one with > 50% chance of kill is possible, set return_value to lower than combat CA
 293             local attacks = ai.get_attacks()
 294             --print(#attacks)
 295             for i,a in ipairs(attacks) do
 296                 if (#a.movements == 1) and (a.chance_to_kill > 0.5) then
 297                     return_value = 90000
 298                     break
 299                 end
 300             end
 301 
 302             -- Also find which locations can be attacked by enemies
 303             local enemy_attack_map = BC.get_attack_map(enemies).units
 304 
 305             -- Now we go through the villages and units
 306             local max_rating, best_village, best_unit = -9e99, {}, {}
 307             local village_ratings = {}
 308             for j,v in ipairs(villages) do
 309                 -- First collect all information that only depends on the village
 310                 local village_rating = 0  -- This is the unit independent rating
 311 
 312                 local unit_in_way = wesnoth.get_unit(v[1], v[2])
 313 
 314                 -- If an enemy can get within one move of the village, we want to hold it
 315                 if enemy_attack_map:get(v[1], v[2]) then
 316                         --print('  within enemy reach', v[1], v[2])
 317                         village_rating = village_rating + 100
 318                 end
 319 
 320                 -- Unowned and enemy-owned villages get a large bonus
 321                 local owner = wesnoth.get_village_owner(v[1], v[2])
 322                 if (not owner) then
 323                     village_rating = village_rating + 10000
 324                 else
 325                     if wesnoth.is_enemy(owner, wesnoth.current.side) then village_rating = village_rating + 20000 end
 326                 end
 327 
 328                 local enemy_distance_from_village = AH.get_closest_enemy(v)
 329 
 330                 -- Now we go on to the unit-dependent rating
 331                 local best_unit_rating = -9e99
 332                 local reachable = false
 333                 for i,u in ipairs(units) do
 334                     -- Skip villages that have units other than 'u' itself on them
 335                     local village_occupied = false
 336                     if AH.is_visible_unit(wesnoth.current.side, unit_in_way) and ((unit_in_way ~= u)) then
 337                         village_occupied = true
 338                     end
 339 
 340                     -- Rate all villages that can be reached and are unoccupied by other units
 341                     if (not village_occupied) then
 342                         -- Path finding is expensive, so we do a first cut simply by distance
 343                         -- There is no way a unit can get to the village if the distance is greater than its moves
 344                         local dist = M.distance_between(u.x, u.y, v[1], v[2])
 345                         if (dist <= u.moves) then
 346                             local path, cost = wesnoth.find_path(u, v[1], v[2])
 347                             if (cost <= u.moves) then
 348                                 village_rating = village_rating - 1
 349                                 reachable = true
 350                                 --print('Can reach:', u.id, v[1], v[2], cost)
 351                                 local rating = 0
 352 
 353                                 -- Prefer strong units if enemies can reach the village, injured units otherwise
 354                                 if enemy_attack_map:get(v[1], v[2]) then
 355                                     rating = rating + u.hitpoints
 356                                 else
 357                                     rating = rating + u.max_hitpoints - u.hitpoints
 358                                 end
 359 
 360                                 -- Prefer not backtracking and moving more distant units to capture villages
 361                                 local enemy_distance_from_unit = AH.get_closest_enemy({u.x, u.y})
 362                                 rating = rating - (enemy_distance_from_village + enemy_distance_from_unit)/5
 363 
 364                                 if (rating > best_unit_rating) then
 365                                     best_unit_rating, best_unit = rating, u
 366                                 end
 367                                 --print('  rating:', rating)
 368                             end
 369                         end
 370                     end
 371                 end
 372                 village_ratings[v] = {village_rating, best_unit, reachable}
 373             end
 374             for j,v in ipairs(villages) do
 375                 local rating = village_ratings[v][1]
 376                 if village_ratings[v][3] and rating > max_rating then
 377                     max_rating, best_village, best_unit = rating, v, village_ratings[v][2]
 378                 end
 379             end
 380             --print('max_rating', max_rating)
 381 
 382             if (max_rating > -9e99) then
 383                 self.data.unit, self.data.village = best_unit, best_village
 384                 if (max_rating >= 1000) then
 385                     if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 386                     return return_value
 387                 else
 388                     if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 389                     return 0
 390                 end
 391             end
 392             if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 393             return 0
 394         end

 406         function generic_rush:spread_poison_eval()
 407             local start_time, ca_name = wesnoth.get_time_stamp() / 1000., 'spread_poison'
 408             if AH.print_eval() then print_time('     - Evaluating spread_poison CA:') end
 409 
 410             -- If a unit with a poisoned weapon can make an attack, we'll do that preferentially
 411             -- (with some exceptions)
 412             local poisoners = AH.get_live_units { side = wesnoth.current.side,
 413                 formula = 'attacks_left > 0',
 414                 { "filter_wml", {
 415                     { "attack", {
 416                         { "specials", {
 417                             { "poison", { } }
 418                         } }
 419                     } }
 420                 } },
 421                 canrecruit = 'no'
 422             }
 423             --print('#poisoners', #poisoners)
 424             if (not poisoners[1]) then
 425                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 426                 return 0
 427             end
 428 
 429             local attacks = AH.get_attacks(poisoners)
 430             --print('#attacks', #attacks)
 431             if (not attacks[1]) then
 432                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 433                 return 0
 434             end
 435 
 436             -- Go through all possible attacks with poisoners
 437             local max_rating, best_attack = -9e99, {}
 438             for i,a in ipairs(attacks) do
 439                 local attacker = wesnoth.get_unit(a.src.x, a.src.y)
 440                 local defender = wesnoth.get_unit(a.target.x, a.target.y)
 441 
 442                 -- Don't try to poison a unit that cannot be poisoned
 443                 local cant_poison = defender.status.poisoned or defender.status.unpoisonable
 444 
 445                 -- For now, we also simply don't poison units on villages (unless standard combat CA does it)
 446                 local on_village = wesnoth.get_terrain_info(wesnoth.get_terrain(defender.x, defender.y)).village
 447 
 448                 -- Also, poisoning units that would level up through the attack or could level on their turn as a result is very bad
 449                 local about_to_level = defender.max_experience - defender.experience <= (wesnoth.unit_types[attacker.type].level * 2)
 450 
 451                 if (not cant_poison) and (not on_village) and (not about_to_level) then
 452                     -- Strongest enemy gets poisoned first
 453                     local rating = defender.hitpoints
 454 
 455                     -- Always attack enemy leader, if possible
 456                     if defender.canrecruit then rating = rating + 1000 end
 457 
 458                     -- Enemies that can regenerate are not good targets
 459                     if wesnoth.unit_ability(defender, 'regenerate') then rating = rating - 1000 end
 460 
 461                     -- More priority to enemies on strong terrain
 462                     local defender_defense = 100 - wesnoth.unit_defense(defender, wesnoth.get_terrain(defender.x, defender.y))
 463                     rating = rating + defender_defense / 4.
 464 
 465                     -- For the same attacker/defender pair, go to strongest terrain
 466                     local attack_defense = 100 - wesnoth.unit_defense(attacker, wesnoth.get_terrain(a.dst.x, a.dst.y))
 467                     rating = rating + attack_defense / 2.
 468                     --print('rating', rating)
 469 
 470                     -- And from village everything else being equal
 471                     local is_village = wesnoth.get_terrain_info(wesnoth.get_terrain(a.dst.x, a.dst.y)).village
 472                     if is_village then rating = rating + 0.5 end
 473 
 474                     if rating > max_rating then
 475                         max_rating, best_attack = rating, a
 476                     end
 477                 end
 478             end
 479 
 480             if (max_rating > -9e99) then
 481                 self.data.attack = best_attack
 482                 if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 483                 return 190000
 484             end
 485             if AH.print_eval() then AH.done_eval_messages(start_time, ca_name) end
 486             return 0
 487         end

 548         -- our share is defined as being slightly more than the total/the number of sides
 549 
 550         function generic_rush:village_hunt_eval()
 551             local villages = wesnoth.get_villages()
 552 
 553             if not villages[1] then
 554                 return 0
 555             end
 556 
 557             local my_villages = wesnoth.get_villages { owner_side = wesnoth.current.side }
 558 
 559             if #my_villages > #villages / #wesnoth.sides then
 560                 return 0
 561             end
 562 
 563             local allied_villages = wesnoth.get_villages { {"filter_owner", { {"ally_of", { side = wesnoth.current.side }} }} }
 564             if #allied_villages == #villages then
 565                 return 0
 566             end
 567 
 568             local units = wesnoth.get_units {
 569                 side = wesnoth.current.side,
 570                 canrecruit = false,
 571                 formula = 'movement_left > 0'
 572             }
 573 
 574             if not units[1] then
 575                 return 0
 576             end
 577 
 578             return 30000
 579         end
 580 
 581         function generic_rush:village_hunt_exec()
 582             local unit = wesnoth.get_units({
 583                 side = wesnoth.current.side,
 584                 canrecruit = false,
 585                 formula = 'movement_left > 0'
 586             })[1]
 587 
 588             local villages = wesnoth.get_villages()
 589             local target, best_cost = nil, AH.no_path
 590             for i,v in ipairs(villages) do
 591                 if not wesnoth.match_location(v[1], v[2], { {"filter_owner", { {"ally_of", { side = wesnoth.current.side }} }} }) then
 592                     local path, cost = wesnoth.find_path(unit, v[1], v[2], { ignore_units = true, max_cost = best_cost })
 593                     if cost < best_cost then
 594                         target = v
 595                         best_cost = cost
 596                     end
 597                 end
 598             end
 599 
 600             if target then
 601                 local x, y = wesnoth.find_vacant_tile(target[1], target[2], unit)
 602                 local dest = AH.next_hop(unit, x, y)
 603                 AH.checked_move(ai, unit, dest[1], dest[2])
 604             end
 605         end
 606 
 607         return generic_rush
 608     end
