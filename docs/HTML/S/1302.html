   1 #!/bin/bash
   2 # Install isutf8 program (from package "moreutils" at least in linux mint)
   3 # in order to use this script
   4 #
   5 # This script assumes that the current working directory is the root of the
   6 # wesnoth repository.
   7 set -e
   8 find src/ -type f -print0 | xargs -0 isutf8 --
   9 find data/ -not -name "*.png" -not -name "*.ogg" -not -name "*.jpg" -not -name "*.wav" -not -name "*.gif" -not -name "*.xcf" -type f -print0 | xargs -0 isutf8 --
  10 find po/ -type f -print0 | xargs -0 isutf8 --
  11 isutf8 changelog
  12 isutf8 players_changelog
  13 isutf8 RELEASE_NOTES
