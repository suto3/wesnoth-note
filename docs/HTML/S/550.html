 180 void styled_widget::request_reduce_width(const unsigned maximum_width)
 181 {
 182         assert(config_);
 183 
 184         if(!label_.empty() && can_wrap()) {
 185 
 186                 point size = get_best_text_size(
 187                                 point(), point(maximum_width - config_->text_extra_width, 0));
 188 
 189                 size.x += config_->text_extra_width;
 190                 size.y += config_->text_extra_height;
 191 
 192                 set_layout_size(size);
 193 
 194                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 195                                   << "' maximum_width " << maximum_width << " result " << size
 196                                   << ".\n";
 197 
 198         } else if(label_.empty() || text_can_shrink()) {
 199                 point size = get_best_size();
 200                 point min_size = get_config_minimum_size();
 201                 size.x = std::min(size.x, std::max<int>(maximum_width, min_size.x));
 202                 set_layout_size(size);
 203 
 204                 DBG_GUI_L << LOG_HEADER << " styled_widget " << id()
 205                           << " maximum_width " << maximum_width << " result " << size
 206                           << ".\n";
 207         } else {
 208                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 209                                   << "' failed; either no label or wrapping not allowed.\n";
 210         }
 211 }

 229 point styled_widget::calculate_best_size() const
 230 {
 231         assert(config_);
 232         if(label_.empty()) {
 233                 DBG_GUI_L << LOG_HEADER << " empty label return default.\n";
 234                 return get_config_default_size();
 235         }
 236 
 237         const point minimum = get_config_default_size();
 238         const point maximum = get_config_maximum_size();
 239 
 240         /**
 241          * @todo The value send should subtract the border size
 242          * and read it after calculation to get the proper result.

 244         point result = get_best_text_size(minimum, maximum);
 245         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 246                           << "' result " << result << ".\n";
 247         return result;

 250 void styled_widget::place(const point& origin, const point& size)
 251 {
 252         // resize canvasses
 253         for(auto & canvas : canvas_)
 254         {
 255                 canvas.set_width(size.x);
 256                 canvas.set_height(size.y);
 257         }
 258 
 259         // Note we assume that the best size has been queried but otherwise it
 260         // should return false.
 261         if(renderer_.is_truncated() && use_tooltip_on_label_overflow_
 262            && tooltip_.empty()) {
 263 
 264                 set_tooltip(label_);
 265         }
 266 
 267         // inherited
 268         widget::place(origin, size);
 269 
 270         // update the state of the canvas after the sizes have been set.
 271         update_canvas();
 272 }

 353 void styled_widget::update_canvas()
 354 {
 355         const int max_width = get_text_maximum_width();
 356         const int max_height = get_text_maximum_height();
 357 
 358         // set label in canvases
 359         for(auto & canvas : canvas_)
 360         {
 361                 canvas.set_variable("text", wfl::variant(label_));
 362                 canvas.set_variable("text_markup", wfl::variant(use_markup_));
 363                 canvas.set_variable("text_link_aware", wfl::variant(get_link_aware()));
 364 
 365                 // Possible TODO: consider making a formula_callable for colors.
 366                 color_t link_color = get_link_color();
 367                 std::vector<wfl::variant> link_color_as_list {
 368                         wfl::variant(link_color.r),
 369                         wfl::variant(link_color.g),
 370                         wfl::variant(link_color.b),
 371                         wfl::variant(link_color.a)
 372                 };
 373 
 374                 canvas.set_variable("text_link_color", wfl::variant(link_color_as_list));
 375                 canvas.set_variable("text_alignment",
 376                                                         wfl::variant(encode_text_alignment(text_alignment_)));
 377                 canvas.set_variable("text_maximum_width", wfl::variant(max_width));
 378                 canvas.set_variable("text_maximum_height", wfl::variant(max_height));
 379                 canvas.set_variable("text_wrap_mode", wfl::variant(get_text_ellipse_mode()));
 380                 canvas.set_variable("text_characters_per_line",
 381                                                         wfl::variant(get_characters_per_line()));
 382         }
 383 }

 420 void styled_widget::definition_load_configuration(const std::string& control_type)
 421 {
 422         assert(!config_);
 423 
 424         set_config(get_control(control_type, definition_));
 425 
 426         /**
 427          * Fill in each canvas from the widget state definitons.
 428          *
 429          * Most widgets have a single canvas. However, some widgets such as toggle_panel
 430          * and toggle_button have a variable canvas count determined by their definitions.
 431          */
 432         for(const auto& widget_state : config_->state) {
 433                 canvas_.emplace_back(std::move(widget_state.canvas_));
 434         }
 435 
 436         update_canvas();
 437 }

 439 point styled_widget::get_best_text_size(point minimum_size,
 440                                                                         point maximum_size) const
 441 {
 442         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 443 
 444         assert(!label_.empty());
 445 
 446         const point border(config_->text_extra_width, config_->text_extra_height);
 447         point size = minimum_size - border;
 448 
 449         renderer_.set_link_aware(get_link_aware())
 450                         .set_link_color(get_link_color());
 451 
 452         renderer_.set_text(label_, use_markup_);
 453 
 454         renderer_.set_family_class(config_->text_font_family);
 455         renderer_.set_font_size(config_->text_font_size);
 456         renderer_.set_font_style(config_->text_font_style);
 457         renderer_.set_alignment(text_alignment_);
 458 
 459         // Try with the minimum wanted size.
 460         const int maximum_width = text_maximum_width_ != 0 ? text_maximum_width_
 461                                                                                                            : maximum_size.x;
 462 
 463         renderer_.set_maximum_width(maximum_width);
 464 
 465         renderer_.set_ellipse_mode(get_text_ellipse_mode());
 466 
 467         renderer_.set_characters_per_line(get_characters_per_line());
 468         if(get_characters_per_line() != 0 && !can_wrap()) {
 469                 WRN_GUI_L
 470                 << LOG_HEADER << " Limited the number of characters per line, "
 471                 << "but wrapping is not set, output may not be as expected.\n";
 472         }
 473 
 474         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 475                           << "' status: "
 476                           << " minimum_size " << minimum_size << " maximum_size "
 477                           << maximum_size << " text_maximum_width_ " << text_maximum_width_
 478                           << " can_wrap " << can_wrap() << " characters_per_line "
 479                           << get_characters_per_line() << " truncated "
 480                           << renderer_.is_truncated() << " renderer size "
 481                           << renderer_.get_size() << ".\n";
 482 
 483         // If doesn't fit try the maximum.
 484         if(renderer_.is_truncated() && !can_wrap()) {
 485                 // FIXME if maximum size is defined we should look at that
 486                 // but also we don't adjust for the extra text space yet!!!
 487                 maximum_size = point(config_->max_width, config_->max_height);
 488                 renderer_.set_maximum_width(maximum_size.x ? maximum_size.x - border.x
 489                                                                                                    : -1);
 490         }
 491 
 492         size = renderer_.get_size() + border;
 493 
 494         if(size.x < minimum_size.x) {
 495                 size.x = minimum_size.x;
 496         }
 497 
 498         if(size.y < minimum_size.y) {
 499                 size.y = minimum_size.y;
 500         }
 501 
 502         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 503                           << "' result " << size << ".\n";
 504         return size;
 505 }

 507 void styled_widget::signal_handler_show_tooltip(const event::ui_event event,
 508                                                                                    bool& handled,
 509                                                                                    const point& location)
 510 {
 511         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 512 
 513         if(!tooltip_.empty()) {
 514                 std::string tip = tooltip_;
 515                 if(!help_message_.empty()) {
 516                         utils::string_map symbols;
 517                         symbols["hotkey"] = hotkey::get_names(
 518                                         hotkey::hotkey_command::get_command_by_command(
 519                                                         hotkey::GLOBAL__HELPTIP).command);
 520 
 521                         tip = tooltip_ + utils::interpolate_variables_into_string(
 522                                                                          settings::has_helptip_message, &symbols);
 523                 }
 524 
 525                 event::message_show_tooltip message(tip, location, get_rectangle());
 526                 handled = fire(event::MESSAGE_SHOW_TOOLTIP, *this, message);
 527         }
 528 }

 662                         _("Found a widget with a helptip and without a tooltip."),
 663                         formatter() << "id '" << id << "' label '" << label_string
 664                                                  << "' helptip '" << help << "'.");
 665 
 666 
 667         DBG_GUI_P << "Window builder: found styled_widget with id '" << id
 668                           << "' and definition '" << definition << "'.\n";
 669 }
