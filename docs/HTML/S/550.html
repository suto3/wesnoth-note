 100 void styled_widget::set_members(const string_map& data)
 101 {
 102         /** @todo document this feature on the wiki. */
 103         /** @todo do we need to add the debug colors here as well? */
 104         string_map::const_iterator itor = data.find("id");

 111                 set_linked_group(itor->second);

 205 void styled_widget::request_reduce_width(const unsigned maximum_width)
 206 {
 207         assert(config_);
 208 
 209         if(!label_.empty() && can_wrap()) {
 210 
 211                 point size = get_best_text_size(
 212                                 point(), point(maximum_width - config_->text_extra_width, 0));
 213 
 214                 size.x += config_->text_extra_width;
 215                 size.y += config_->text_extra_height;
 216 
 217                 set_layout_size(size);
 218 
 219                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 220                                   << "' maximum_width " << maximum_width << " result " << size
 221                                   << ".\n";
 222 
 223         } else if(label_.empty() || text_can_shrink()) {
 224                 point size = get_best_size();
 225                 point min_size = get_config_minimum_size();
 226                 size.x = std::min(size.x, std::max<int>(maximum_width, min_size.x));
 227                 set_layout_size(size);
 228 
 229                 DBG_GUI_L << LOG_HEADER << " styled_widget " << id()
 230                           << " maximum_width " << maximum_width << " result " << size
 231                           << ".\n";
 232         } else {
 233                 DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 234                                   << "' failed; either no label or wrapping not allowed.\n";
 235         }
 236 }

 254 point styled_widget::calculate_best_size() const
 255 {
 256         assert(config_);
 257         if(label_.empty()) {
 258                 DBG_GUI_L << LOG_HEADER << " empty label return default.\n";
 259                 return get_config_default_size();
 260         }
 261 
 262         const point minimum = get_config_default_size();
 263         const point maximum = get_config_maximum_size();
 264 
 265         /**
 266          * @todo The value send should subtract the border size
 267          * and read it after calculation to get the proper result.
 268          */
 269         point result = get_best_text_size(minimum, maximum);
 270         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 271                           << "' result " << result << ".\n";
 272         return result;

 275 void styled_widget::place(const point& origin, const point& size)
 276 {
 277         // resize canvasses
 278         for(auto & canvas : canvas_)
 279         {
 280                 canvas.set_width(size.x);
 281                 canvas.set_height(size.y);
 282         }
 283 
 284         // Note we assume that the best size has been queried but otherwise it
 285         // should return false.
 286         if(renderer_.is_truncated() && use_tooltip_on_label_overflow_
 287            && tooltip_.empty()) {
 288 
 289                 set_tooltip(label_);
 290         }
 291 
 292         // inherited
 293         widget::place(origin, size);
 294 
 295         // update the state of the canvas after the sizes have been set.
 296         update_canvas();
 297 }

 400 void styled_widget::update_canvas()
 401 {
 402         const int max_width = get_text_maximum_width();
 403         const int max_height = get_text_maximum_height();
 404 
 405         // set label in canvases
 406         for(auto & canvas : canvas_)
 407         {
 408                 canvas.set_variable("text", wfl::variant(label_));
 409                 canvas.set_variable("text_markup", wfl::variant(use_markup_));
 410                 canvas.set_variable("text_link_aware", wfl::variant(get_link_aware()));
 411                 // Possible TODO: Consider making a formula_callable for colours
 412                 color_t link_color = get_link_color();
 413                 std::vector<wfl::variant> link_color_as_list{wfl::variant(link_color.r), wfl::variant(link_color.g), wfl::variant(link_color.b), wfl::variant(link_color.a)};
 414                 canvas.set_variable("text_link_color", wfl::variant(link_color_as_list));
 415                 canvas.set_variable("text_alignment",
 416                                                         wfl::variant(encode_text_alignment(text_alignment_)));
 417                 canvas.set_variable("text_maximum_width", wfl::variant(max_width));
 418                 canvas.set_variable("text_maximum_height", wfl::variant(max_height));
 419                 canvas.set_variable("text_wrap_mode", wfl::variant(get_text_ellipse_mode()));
 420                 canvas.set_variable("text_characters_per_line",
 421                                                         wfl::variant(get_characters_per_line()));
 422         }
 423 }

 460 void styled_widget::definition_load_configuration(const std::string& control_type)
 461 {
 462         assert(!config_);
 463 
 464         set_config(get_control(control_type, definition_));
 465 
 466         /**
 467          * Fill in each canvas from the widget state definitons.
 468          *
 469          * Most widgets have a single canvas. However, some widgets such as toggle_panel
 470          * and toggle_button have a variable canvas count determined by their definitions.
 471          */
 472         for(const auto& widget_state : config_->state) {
 473                 canvas_.emplace_back(std::move(widget_state.canvas_));
 474         }
 475 
 476         update_canvas();
 477 }

 479 point styled_widget::get_best_text_size(point minimum_size,
 480                                                                         point maximum_size) const
 481 {
 482         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 483 
 484         assert(!label_.empty());
 485 
 486         const point border(config_->text_extra_width, config_->text_extra_height);
 487         point size = minimum_size - border;
 488 
 489         renderer_.set_link_aware(get_link_aware())
 490                         .set_link_color(get_link_color());
 491 
 492         renderer_.set_text(label_, use_markup_);
 493 
 494         renderer_.set_family_class(config_->text_font_family);
 495         renderer_.set_font_size(config_->text_font_size);
 496         renderer_.set_font_style(config_->text_font_style);
 497         renderer_.set_alignment(text_alignment_);
 498 
 499         // Try with the minimum wanted size.
 500         const int maximum_width = text_maximum_width_ != 0 ? text_maximum_width_
 501                                                                                                            : maximum_size.x;
 502 
 503         renderer_.set_maximum_width(maximum_width);
 504 
 505         renderer_.set_ellipse_mode(get_text_ellipse_mode());
 506 
 507         renderer_.set_characters_per_line(get_characters_per_line());
 508         if(get_characters_per_line() != 0 && !can_wrap()) {
 509                 WRN_GUI_L
 510                 << LOG_HEADER << " Limited the number of characters per line, "
 511                 << "but wrapping is not set, output may not be as expected.\n";
 512         }
 513 
 514         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 515                           << "' status: "
 516                           << " minimum_size " << minimum_size << " maximum_size "
 517                           << maximum_size << " text_maximum_width_ " << text_maximum_width_
 518                           << " can_wrap " << can_wrap() << " characters_per_line "
 519                           << get_characters_per_line() << " truncated "
 520                           << renderer_.is_truncated() << " renderer size "
 521                           << renderer_.get_size() << ".\n";
 522 
 523         // If doesn't fit try the maximum.
 524         if(renderer_.is_truncated() && !can_wrap()) {
 525                 // FIXME if maximum size is defined we should look at that
 526                 // but also we don't adjust for the extra text space yet!!!
 527                 maximum_size = point(config_->max_width, config_->max_height);
 528                 renderer_.set_maximum_width(maximum_size.x ? maximum_size.x - border.x
 529                                                                                                    : -1);
 530         }
 531 
 532         size = renderer_.get_size() + border;
 533 
 534         if(size.x < minimum_size.x) {
 535                 size.x = minimum_size.x;
 536         }
 537 
 538         if(size.y < minimum_size.y) {
 539                 size.y = minimum_size.y;
 540         }
 541 
 542         DBG_GUI_L << LOG_HEADER << " label '" << debug_truncate(label_)
 543                           << "' result " << size << ".\n";
 544         return size;
 545 }

 547 void styled_widget::signal_handler_show_tooltip(const event::ui_event event,
 548                                                                                    bool& handled,
 549                                                                                    const point& location)
 550 {
 551         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 552 
 553         if(!tooltip_.empty()) {
 554                 std::string tip = tooltip_;
 555                 if(!help_message_.empty()) {
 556                         utils::string_map symbols;
 557                         symbols["hotkey"] = hotkey::get_names(
 558                                         hotkey::hotkey_command::get_command_by_command(
 559                                                         hotkey::GLOBAL__HELPTIP).command);
 560 
 561                         tip = tooltip_ + utils::interpolate_variables_into_string(
 562                                                                          settings::has_helptip_message, &symbols);
 563                 }
 564 
 565                 event::message_show_tooltip message(tip, location, get_rectangle());
 566                 handled = fire(event::MESSAGE_SHOW_TOOLTIP, *this, message);
 567         }
 568 }

 635  *

 639  *     label & t_string & "" &          Most widgets have some text associated

 641  *                                     of that text. Some widgets use this value
 642  *                                     for other purposes, this is documented
 643  *                                     at the widget. E.g. an image uses the
 644  *                                     filename in this field. $
 645  *
 646  *     tooltip & t_string & "" &        If you hover over a widget a while (the
 647  *                                     time it takes can differ per widget) a
 648  *                                     short help can show up.This defines the
 649  *                                     text of that message. This field may not
 650  *                                     be empty when 'help' is set. $
 651  *
 652  *

 655  *                                     tip) a help message can show up. This
 656  *                                     help message might be the same as the
 657  *                                     tooltip but in general (if used) this
 658  *                                     message should show more help. This
 659  *                                     defines the text of that message. $
 660  *
 661  *    use_tooltip_on_label_overflow & bool & true &
 662  *                                     If the text on the label is truncated and
 663  *                                     the tooltip is empty the label can be
 664  *                                     used for the tooltip. If this variable is
 665  *                                     set to true this will happen. $
 666  *
 667  *   debug_border_mode & unsigned & 0 &
 668  *                                     The mode for showing the debug border.
 669  *                                     This border shows the area reserved for
 670  *                                     a widget. This function is only meant
 671  *                                     for debugging and might not be
 672  *                                     available in all Wesnoth binaries.
 673  *                                     Available modes:
 674  *                                     @* 0 no border.
 675  *                                     @* 1 1 pixel border.
 676  *                                     @* 2 floodfill the widget area. $
 677  *
 678  *   debug_border_color & color & "" & The color of the debug border. $
 679  * @end{table}
 680  * @end{tag}{name="widget_instance"}
 681  * @end{parent}{name="generic/"}

 692         , help(cfg["help"].t_str())

 704                                                  << "' helptip '" << help << "'.");

 708                           << "' and definition '" << definition << "'.\n";
