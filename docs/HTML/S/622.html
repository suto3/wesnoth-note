  99 void listbox::remove_row(const unsigned row, unsigned count)
 100 {
 101         assert(generator_);
 102 
 103         if(row >= get_item_count()) {
 104                 return;
 105         }
 106 
 107         if(!count || count + row > get_item_count()) {
 108                 count = get_item_count() - row;
 109         }
 110 
 111         int height_reduced = 0;
 112         int width_reduced = 0;
 113         //TODO: Fix this for horizinal listboxes
 114         //Note the we have to use content_grid_ and cannot use "_list_grid" which is what generator_ uses.
 115         int row_pos_y = is_horizonal_ ? -1 : generator_->item(row).get_y()  - content_grid_->get_y();
 116         int row_pos_x = is_horizonal_ ? -1 : 0;
 117         for(; count; --count) {
 118                 if(generator_->item(row).get_visible() != visibility::invisible) {
 119                         if(is_horizonal_) {
 120                                 width_reduced += generator_->item(row).get_width();
 121                         }
 122                         else {
 123                                 height_reduced += generator_->item(row).get_height();
 124                         }
 125                 }
 126                 generator_->delete_item(row);
 127         }
 128 
 129         if((height_reduced != 0 || width_reduced != 0) && get_item_count() != 0) {
 130                 resize_content(-width_reduced, -height_reduced, row_pos_x, row_pos_y);
 131         } else {
 132                 update_content_size();
 133         }
 134 }

 154 void listbox::set_row_shown(const unsigned row, const bool shown)
 155 {
 156         assert(generator_);
 157 
 158         window* window = get_window();
 159         assert(window);
 160 
 161         const int selected_row = get_selected_row();
 162 
 163         bool resize_needed = false;
 164         {
 165                 window::invalidate_layout_blocker invalidate_layout_blocker(*window);
 166 
 167                 generator_->set_item_shown(row, shown);
 168                 point best_size = generator_->calculate_best_size();
 169                 generator_->place(generator_->get_origin(), { std::max(best_size.x, content_visible_area().w), best_size.y });
 170                 resize_needed = !content_resize_request();
 171         }
 172 
 173         if(resize_needed) {
 174                 window->invalidate_layout();
 175         } else {
 176                 content_grid_->set_visible_rectangle(content_visible_area());
 177                 set_is_dirty(true);
 178         }
 179 
 180         if(selected_row != get_selected_row() && callback_value_changed_) {
 181                 callback_value_changed_(*this);
 182         }
 183 }

 277 void listbox::list_item_clicked(widget& caller)
 278 {
 279         assert(generator_);
 280 
 281         /** @todo Hack to capture the keyboard focus. */
 282         get_window()->keyboard_capture(this);
 283 
 284         for(size_t i = 0; i < generator_->get_item_count(); ++i) {
 285 
 286                 if(generator_->item(i).has_widget(caller)) {
 287                         toggle_button* checkbox = dynamic_cast<toggle_button*>(&caller);
 288                         if(checkbox != nullptr) {
 289                                 generator_->select_item(i, checkbox->get_value_bool());
 290                         } else {
 291                                 generator_->toggle_item(i);
 292                         }
 293 
 294                         if(callback_item_changed_) {
 295                                 callback_item_changed_(i);
 296                         }
 297                         if(callback_value_changed_) {
 298                                 callback_value_changed_(*this);
 299                         }
 300                         return;
 301                 }
 302         }
 303         assert(false);
 304 }

 338 void listbox::place(const point& origin, const point& size)
 339 {
 340         boost::optional<unsigned> vertical_scrollbar_position, horizontal_scrollbar_position;
 341         // Check if this is the first time placing the list box
 342         if (get_origin() != point{-1, -1})
 343         {
 344                 vertical_scrollbar_position = get_vertical_scrollbar_item_position();
 345                 horizontal_scrollbar_position = get_horizontal_scrollbar_item_position();
 346         }
 347 
 348         // Inherited.
 349         scrollbar_container::place(origin, size);
 350 
 351         const int selected_item = generator_->get_selected_item();
 352         if (vertical_scrollbar_position && horizontal_scrollbar_position)
 353         {
 354                 LOG_GUI_L << LOG_HEADER << " restoring scroll position" << std::endl;
 355                 set_vertical_scrollbar_item_position(*vertical_scrollbar_position);
 356                 set_horizontal_scrollbar_item_position(*horizontal_scrollbar_position);
 357         }
 358         else if (selected_item != -1)
 359         {
 360                 LOG_GUI_L << LOG_HEADER << " making the initially selected item visible" << std::endl;
 361                 const SDL_Rect& visible = content_visible_area();
 362                 SDL_Rect rect = generator_->item(selected_item).get_rectangle();
 363 
 364                 rect.x = visible.x;
 365                 rect.w = visible.w;
 366 
 367                 show_content_rect(rect);
 368         }
 369 }

 375 void listbox::resize_content(const int width_modification,
 376                                                           const int height_modification,
 377                                                           const int width_modification_pos,
 378                                                           const int height_modification_pos)
 379 {
 380         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 381                           << " width_modification " << width_modification
 382                           << " height_modification " << height_modification << ".\n";
 383 
 384         if(content_resize_request(width_modification, height_modification, width_modification_pos, height_modification_pos)) {
 385 
 386                 // Calculate new size.
 387                 point size = content_grid()->get_size();
 388                 size.x += width_modification;
 389                 size.y += height_modification;
 390 
 391                 // Set new size.
 392                 content_grid()->set_size(size);
 393 
 394                 // Set status.
 395                 need_layout_ = true;
 396                 // If the content grows assume it "overwrites" the old content.
 397                 if(width_modification < 0 || height_modification < 0) {
 398                         set_is_dirty(true);
 399                 }
 400                 DBG_GUI_L << LOG_HEADER << " succeeded.\n";
 401         } else {
 402                 DBG_GUI_L << LOG_HEADER << " failed.\n";
 403         }
 404 }

 406 void listbox::resize_content(const widget& row)
 407 {
 408         if(row.get_visible() == visibility::invisible) {
 409                 return;
 410         }
 411 
 412         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 413                           << " row size " << row.get_best_size() << ".\n";
 414 
 415         const point content = content_grid()->get_size();
 416         point size = row.get_best_size();
 417         if(size.x < content.x) {
 418                 size.x = 0;
 419         } else {
 420                 size.x -= content.x;
 421         }
 422 
 423         resize_content(size.x, size.y);
 424 }

 554 /**
 555  * Swaps an item in a grid for another one.*/
 556 void swap_grid(grid* g,
 557                            grid* content_grid,
 558                            widget* wgt,
 559                            const std::string& id)
 560 {
 561         assert(content_grid);
 562         assert(wgt);
 563 
 564         // Make sure the new child has same id.
 565         wgt->set_id(id);
 566 
 567         // Get the container containing the wanted widget.
 568         grid* parent_grid = nullptr;
 569         if(g) {
 570                 parent_grid = find_widget<grid>(g, id, false, false);
 571         }
 572         if(!parent_grid) {
 573                 parent_grid = find_widget<grid>(content_grid, id, true, false);
 574         }
 575         parent_grid = dynamic_cast<grid*>(parent_grid->parent());
 576         assert(parent_grid);
 577 
 578         // Replace the child.
 579         auto old = parent_grid->swap_child(id, wgt, false);
 580         assert(old);
 581 }

 585 void listbox::finalize(builder_grid_const_ptr header,
 586                                                 builder_grid_const_ptr footer,
 587                                                 const std::vector<std::map<std::string, string_map>>& list_data)
 588 {
 589         // "Inherited."
 590         scrollbar_container::finalize_setup();
 591 
 592         assert(generator_);
 593 
 594         if(header) {
 595                 swap_grid(&get_grid(), content_grid(), header->build(), "_header_grid");
 596         }
 597         grid& p = find_widget<grid>(this, "_header_grid", false);
 598         for(unsigned i = 0, max = std::max(p.get_cols(), p.get_rows()); i < max; ++i) {
 599                 if(selectable_item* selectable = find_widget<selectable_item>(&p, "sort_" +  std::to_string(i), false, false)) {
 600                         selectable->set_callback_state_change(std::bind(&listbox::order_by_column, this, i, _1));
 601                         if(orders_.size() < max ) {
 602                                 orders_.resize(max);
 603                         }
 604                         orders_[i].first = selectable;
 605                 }
 606         }
 607         if(footer) {
 608                 swap_grid(&get_grid(), content_grid(), footer->build(), "_footer_grid");
 609         }
 610 
 611         generator_->create_items(
 612                         -1, list_builder_, list_data, callback_list_item_clicked);
 613         swap_grid(nullptr, content_grid(), generator_, "_list_grid");
 614 }

 622 void listbox::order_by_column(unsigned column, widget& widget)
 623 {
 624         selectable_item& selectable = dynamic_cast<selectable_item&>(widget);
 625         if(column >= orders_.size()) {
 626                 return;
 627         }
 628         for(auto& pair : orders_)
 629         {
 630                 if(pair.first != nullptr && pair.first != &selectable) {
 631                         pair.first->set_value(SORT_NONE);
 632                 }
 633         }
 634         if(selectable.get_value() > orders_[column].second.size()) {
 635                 return;
 636         }
 637         if(selectable.get_value() == SORT_NONE) {
 638                 order_by(generator_base::torder_func(&default_sort));
 639         }
 640         else {
 641                 order_by(orders_[column].second[selectable.get_value() - 1]);
 642         }
 643 }

 676 const listbox::order_pair listbox::get_active_sorting_option()
 677 {
 678         const auto iter = std::find_if(orders_.begin(), orders_.end(), [](const std::pair<selectable_item*, generator_sort_array>& option) {
 679                 return option.first != nullptr && option.first->get_value() != SORT_NONE;
 680         });
 681 
 682         if(iter != orders_.end()) {
 683                 return {iter - orders_.begin(), static_cast<SORT_ORDER>(iter->first->get_value())};
 684         }
 685 
 686         return {-1, SORT_NONE};

 700 void listbox::layout_children(const bool force)
 701 {
 702         assert(content_grid());
 703 
 704         if(need_layout_ || force) {
 705                 const int selected_item = generator_->get_selected_item();
 706 
 707                 content_grid()->place(content_grid()->get_origin(),
 708                                                           content_grid()->get_size());
 709 
 710                 const SDL_Rect& visible = content_visible_area_;
 711 
 712                 content_grid()->set_visible_rectangle(visible);
 713 
 714                 if(selected_item != -1) {
 715                         SDL_Rect rect = generator_->item(selected_item).get_rectangle();
 716 
 717                         rect.x = visible.x;
 718                         rect.w = visible.w;
 719 
 720                         show_content_rect(rect);
 721                 }
 722 
 723                 need_layout_ = false;
 724                 set_is_dirty(true);
 725         }

 934 static std::vector<std::map<std::string, string_map>> parse_list_data(const config& data, const unsigned int req_cols)
 935 {
 936         std::vector<std::map<std::string, string_map>> list_data;
 937         for(const auto & row : data.child_range("row"))
 938         {
 939                 auto cols = row.child_range("column");
 940                 VALIDATE(static_cast<unsigned>(cols.size()) == req_cols, _("'list_data' must have the same number of columns as the 'list_definition'."));
 941 
 942                 for(const auto & c : cols)
 943                 {
 944                         list_data.emplace_back();
 945                         for(const auto & i : c.attribute_range())
 946                         {
 947                                 list_data.back()[""][i.first] = i.second;
 948                         }
 949                         for(const auto& w : c.child_range("widget"))
 950                         {
 951                                 VALIDATE(w.has_attribute("id"), missing_mandatory_wml_key("[list_data][row][column][widget]", "id"));
 952                                 for(const auto& i : w.attribute_range()) {
 953                                         list_data.back()[w["id"]][i.first] = i.second;
 954                                 }
 955                         }
 956                 }
 957         }
 958         return list_data;
 959 }
