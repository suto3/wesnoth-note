 132 void lobby_player_info::update_relation()
 133 {
 134         add_to_friends_->set_active(false);
 135         add_to_ignores_->set_active(false);
 136         remove_from_list_->set_active(false);
 137         switch(info_.relation) {
 138                 case mp::user_info::FRIEND:
 139                         relation_->set_label(_("On friends list"));
 140                         add_to_ignores_->set_active(true);
 141                         remove_from_list_->set_active(true);
 142                         break;
 143                 case mp::user_info::IGNORED:
 144                         relation_->set_label(_("On ignores list"));
 145                         add_to_friends_->set_active(true);
 146                         remove_from_list_->set_active(true);
 147                         break;
 148                 case mp::user_info::NEUTRAL:
 149                         relation_->set_label(_("Neither a friend nor ignored"));
 150                         add_to_friends_->set_active(true);
 151                         add_to_ignores_->set_active(true);
 152                         break;
 153                 case mp::user_info::ME:
 154                         relation_->set_label(_("You"));
 155                         break;
 156                 default:
 157                         relation_->set_label(_("Error"));
 158         }
 159 }

 206 void lobby_player_info::do_kick_ban(bool ban)
 207 {
 208         std::stringstream ss;
 209         ss << (ban ? "kban" : "kick ") << info_.name;
 210         if(ban && !time_->get_value().empty()) {
 211                 ss << " " << time_->get_value();
 212         }
 213         if(!reason_->get_value().empty()) {
 214                 ss << " " << reason_->get_value();
 215         }
 216 
 217         chat_.send_command("query", ss.str());
 218 }
