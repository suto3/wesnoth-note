 101 bool terrain_filter::match_internal(const map_location& loc, const unit* ref_unit, const bool ignore_xy) const
 102 {
 103         if (!this->fc_->get_disp_context().map().on_board_with_border(loc)) {
 104                 return false;
 105         }
 106 
 107         std::string lua_function = cfg_["lua_function"];
 108         if (!lua_function.empty() && fc_->get_lua_kernel()) {
 109                 if (!fc_->get_lua_kernel()->run_filter(lua_function.c_str(), loc)) {
 110                         return false;
 111                 }
 112         }
 113 
 114         //Filter Areas
 115         if (cfg_.has_attribute("area") &&
 116                 fc_->get_tod_man().get_area_by_id(cfg_["area"]).count(loc) == 0)
 117                 return false;
 118 
 119         if(cfg_.has_attribute("terrain")) {
 120                 if(cache_.parsed_terrain == nullptr) {
 121                         cache_.parsed_terrain = new t_translation::ter_match(cfg_["terrain"]);
 122                 }
 123                 if(!cache_.parsed_terrain->is_empty) {
 124                         const t_translation::terrain_code letter = fc_->get_disp_context().map().get_terrain_info(loc).number();
 125                         if(!t_translation::terrain_matches(letter, *cache_.parsed_terrain)) {
 126                                 return false;
 127                         }
 128                 }
 129         }
 130 
 131         //Allow filtering on location ranges
 132         if (!ignore_xy) {
 133                 if (!loc.matches_range(cfg_["x"], cfg_["y"])) {
 134                         return false;
 135                 }
 136                 //allow filtering by searching a stored variable of locations
 137                 if (cfg_.has_attribute("find_in")) {
 138                         if (const game_data * gd = fc_->get_game_data()) {
 139                                 try
 140                                 {
 141                                         variable_access_const vi = gd->get_variable_access_read(cfg_["find_in"]);
 142 
 143                                         bool found = false;
 144                                         for (const config &cfg : vi.as_array()) {
 145                                                 if (map_location(cfg, nullptr) == loc) {
 146                                                         found = true;
 147                                                         break;
 148                                                 }
 149                                         }
 150                                         if (!found) return false;
 151                                 }
 152                                 catch (const invalid_variablename_exception&)
 153                                 {
 154                                         return false;
 155                                 }
 156                         }
 157                 }
 158                 if (cfg_.has_attribute("location_id")) {
 159                         if (loc != fc_->get_disp_context().map().special_location(cfg_["location_id"])) {
 160                                 return false;
 161                         }
 162                 }
 163         }
 164         //Allow filtering on unit
 165         if(cfg_.has_child("filter")) {
 166                 const unit_map::const_iterator u = fc_->get_disp_context().units().find(loc);
 167                 if (!u.valid())
 168                         return false;
 169                 if (!cache_.ufilter_)
 170                         cache_.ufilter_.reset(new unit_filter(cfg_.child("filter"), fc_, flat_));
 171                 if (!cache_.ufilter_->matches(*u, loc))
 172                         return false;
 173         }
 174 
 175         // Allow filtering on visibility to a side
 176         if (cfg_.has_child("filter_vision")) {
 177                 const vconfig::child_list& vis_filt = cfg_.get_children("filter_vision");
 178                 vconfig::child_list::const_iterator i, i_end = vis_filt.end();
 179                 for (i = vis_filt.begin(); i != i_end; ++i) {
 180                         bool visible = (*i)["visible"].to_bool(true);
 181                         bool respect_fog = (*i)["respect_fog"].to_bool(true);
 182 
 183                         side_filter ssf(*i, fc_);
 184                         std::vector<int> sides = ssf.get_teams();
 185 
 186                         bool found = false;
 187                         for (const int side : sides) {
 188                                 const team &viewing_team = fc_->get_disp_context().get_team(side);
 189                                 bool viewer_sees = respect_fog ? !viewing_team.fogged(loc) : !viewing_team.shrouded(loc);
 190                                 if (visible == viewer_sees) {
 191                                         found = true;
 192                                         break;
 193                                 }
 194                         }
 195                         if (!found) {return false;}
 196                 }
 197         }
 198 
 199         //Allow filtering on adjacent locations
 200         if(cfg_.has_child("filter_adjacent_location")) {
 201                 map_location adjacent[6];
 202                 get_adjacent_tiles(loc, adjacent);
 203                 const vconfig::child_list& adj_cfgs = cfg_.get_children("filter_adjacent_location");
 204                 vconfig::child_list::const_iterator i, i_end, i_begin = adj_cfgs.begin();
 205                 for (i = i_begin, i_end = adj_cfgs.end(); i != i_end; ++i) {
 206                         int match_count = 0;
 207                         vconfig::child_list::difference_type index = i - i_begin;
 208                         std::vector<map_location::DIRECTION> dirs = (*i).has_attribute("adjacent")
 209                                 ? map_location::parse_directions((*i)["adjacent"]) : map_location::default_dirs();
 210                         std::vector<map_location::DIRECTION>::const_iterator j, j_end = dirs.end();
 211                         for (j = dirs.begin(); j != j_end; ++j) {
 212                                 map_location &adj = adjacent[*j];
 213                                 if (fc_->get_disp_context().map().on_board(adj)) {
 214                                         if(cache_.adjacent_matches == nullptr) {
 215                                                 while(index >= std::distance(cache_.adjacent_match_cache.begin(), cache_.adjacent_match_cache.end())) {
 216                                                         const vconfig& adj_cfg = adj_cfgs[cache_.adjacent_match_cache.size()];
 217                                                         std::pair<terrain_filter, std::map<map_location,bool> > amc_pair(
 218                                                                 terrain_filter(adj_cfg, *this),
 219                                                                 std::map<map_location,bool>());
 220                                                         cache_.adjacent_match_cache.push_back(amc_pair);
 221                                                 }
 222                                                 terrain_filter &amc_filter = cache_.adjacent_match_cache[index].first;
 223                                                 std::map<map_location,bool> &amc = cache_.adjacent_match_cache[index].second;
 224                                                 std::map<map_location,bool>::iterator lookup = amc.find(adj);
 225                                                 if(lookup == amc.end()) {
 226                                                         if(amc_filter(adj)) {
 227                                                                 amc[adj] = true;
 228                                                                 ++match_count;
 229                                                         } else {
 230                                                                 amc[adj] = false;
 231                                                         }
 232                                                 } else if(lookup->second) {
 233                                                         ++match_count;
 234                                                 }
 235                                         } else {
 236                                                 assert(index < std::distance(cache_.adjacent_matches->begin(), cache_.adjacent_matches->end()));
 237                                                 std::set<map_location> &amc = (*cache_.adjacent_matches)[index];
 238                                                 if(amc.find(adj) != amc.end()) {
 239                                                         ++match_count;
 240                                                 }
 241                                         }
 242                                 }
 243                         }
 244                         static std::vector<std::pair<int,int> > default_counts = utils::parse_ranges("1-6");
 245                         std::vector<std::pair<int,int> > counts = (*i).has_attribute("count")
 246                                 ? utils::parse_ranges((*i)["count"]) : default_counts;
 247                         if(!in_ranges(match_count, counts)) {
 248                                 return false;
 249                         }
 250                 }
 251         }
 252 
 253         const t_string& t_tod_type = cfg_["time_of_day"];
 254         const t_string& t_tod_id = cfg_["time_of_day_id"];
 255         const std::string& tod_type = t_tod_type;
 256         const std::string& tod_id = t_tod_id;
 257         if(!tod_type.empty() || !tod_id.empty()) {
 258                 // creating a time_of_day is expensive, only do it if we will use it
 259                 time_of_day tod;
 260 
 261                 if(flat_) {
 262                         tod = fc_->get_tod_man().get_time_of_day(loc);
 263                 } else {
 264                         tod = fc_->get_tod_man().get_illuminated_time_of_day(fc_->get_disp_context().units(), fc_->get_disp_context().map(),loc);
 265                 }
 266 
 267                 if(!tod_type.empty()) {
 268                         const std::vector<std::string>& vals = utils::split(tod_type);
 269                         if(tod.lawful_bonus<0) {
 270                                 if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::CHAOTIC)) == vals.end()) {
 271                                         return false;
 272                                 }
 273                         } else if(tod.lawful_bonus>0) {
 274                                 if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::LAWFUL)) == vals.end()) {
 275                                         return false;
 276                                 }
 277                         } else if(std::find(vals.begin(),vals.end(),unit_type::ALIGNMENT::enum_to_string(unit_type::ALIGNMENT::NEUTRAL)) == vals.end()) {
 278                                 return false;
 279                         }
 280                 }
 281 
 282                 if(!tod_id.empty()) {
 283                         if(tod_id != tod.id) {
 284                                 if(std::find(tod_id.begin(),tod_id.end(),',') != tod_id.end() &&
 285                                         std::search(tod_id.begin(),tod_id.end(),
 286                                         tod.id.begin(),tod.id.end()) != tod_id.end()) {
 287                                         const std::vector<std::string>& vals = utils::split(tod_id);
 288                                         if(std::find(vals.begin(),vals.end(),tod.id) == vals.end()) {
 289                                                 return false;
 290                                         }
 291                                 } else {
 292                                         return false;
 293                                 }
 294                         }
 295                 }
 296         }
 297 
 298         //allow filtering on owner (for villages)
 299         const config::attribute_value &owner_side = cfg_["owner_side"];
 300         const vconfig& filter_owner = cfg_.child("filter_owner");
 301         if(!filter_owner.null()) {
 302                 if(!owner_side.empty()) {
 303                         WRN_NG << "duplicate side information in a SLF, ignoring inline owner_side=" << std::endl;
 304                 }
 305                 if(!fc_->get_disp_context().map().is_village(loc))
 306                         return false;
 307                 side_filter ssf(filter_owner, fc_);
 308                 const std::vector<int>& sides = ssf.get_teams();
 309                 bool found = false;
 310                 if(sides.empty() && fc_->get_disp_context().village_owner(loc) == -1)
 311                         found = true;
 312                 for(const int side : sides) {
 313                         if(fc_->get_disp_context().get_team(side).owns_village(loc)) {
 314                                 found = true;
 315                                 break;
 316                         }
 317                 }
 318                 if(!found)
 319                         return false;
 320         }
 321         else if(!owner_side.empty()) {
 322                 const int side_index = owner_side.to_int(0) - 1;
 323                 if(fc_->get_disp_context().village_owner(loc) != side_index) {
 324                         return false;
 325                 }
 326         }
 327 
 328         if(cfg_.has_attribute("formula")) {
 329                 try {
 330                         const wfl::terrain_callable main(fc_->get_disp_context(), loc);
 331                         wfl::map_formula_callable callable(main.fake_ptr());
 332                         if(ref_unit) {
 333                                 std::shared_ptr<wfl::unit_callable> ref(new wfl::unit_callable(*ref_unit));
 334                                 callable.add("teleport_unit", wfl::variant(ref));
 335                                 // It's not destroyed upon scope exit because the variant holds a reference
 336                         }
 337                         const wfl::formula form(cfg_["formula"], new wfl::gamestate_function_symbol_table);
 338                         if(!form.evaluate(callable).as_bool()) {
 339                                 return false;
 340                         }
 341                         return true;
 342                 } catch(wfl::formula_error& e) {
 343                         lg::wml_error() << "Formula error in location filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 344                         // Formulae with syntax errors match nothing
 345                         return false;
 346                 }
 347         }
 348 
 349         return true;
 350 }

 362 bool terrain_filter::match_impl(const map_location& loc, const unit* ref_unit) const
 363 {
 364         if(cfg_["x"] == "recall" && cfg_["y"] == "recall") {
 365                 return !fc_->get_disp_context().map().on_board(loc);
 366         }
 367         std::set<map_location> hexes;
 368         std::vector<map_location> loc_vec(1, loc);
 369 
 370         std::unique_ptr<scoped_wml_variable> ref_unit_var;
 371         if(ref_unit) {
 372                 if(fc_->get_disp_context().map().on_board(ref_unit->get_location())) {
 373                         ref_unit_var.reset(new scoped_xy_unit("teleport_unit", ref_unit->get_location(), fc_->get_disp_context().units()));
 374                 } else {
 375                         // Possible TODO: Support recall list units?
 376                 }
 377         }
 378 
 379         //handle radius
 380         size_t radius = cfg_["radius"].to_size_t(0);
 381         if(radius > max_loop_) {
 382                 ERR_NG << "terrain_filter: radius greater than " << max_loop_
 383                 << ", restricting\n";
 384                 radius = max_loop_;
 385         }
 386         if ( radius == 0 )
 387                 hexes.insert(loc_vec.begin(), loc_vec.end());
 388         else if ( cfg_.has_child("filter_radius") ) {
 389                 terrain_filter r_filter(cfg_.child("filter_radius"), *this);
 390                 if(ref_unit) {
 391                         get_tiles_radius(fc_->get_disp_context().map(), loc_vec, radius, hexes, false, filter_with_unit(r_filter, *ref_unit));
 392                 } else {
 393                         get_tiles_radius(fc_->get_disp_context().map(), loc_vec, radius, hexes, false, r_filter);
 394                 }
 395         } else {
 396                 get_tiles_radius(fc_->get_disp_context().map(), loc_vec, radius, hexes);
 397         }
 398 
 399         size_t loop_count = 0;
 400         std::set<map_location>::const_iterator i;
 401         for(i = hexes.begin(); i != hexes.end(); ++i) {
 402                 bool matches = match_internal(*i, ref_unit, false);
 403 
 404                 //handle [and], [or], and [not] with in-order precedence
 405                 vconfig::all_children_iterator cond = cfg_.ordered_begin();
 406                 vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 407                 while(cond != cond_end)
 408                 {
 409                         const std::string& cond_name = cond.get_key();
 410                         const vconfig& cond_cfg = cond.get_child();
 411 
 412                         //handle [and]
 413                         if(cond_name == "and")
 414                         {
 415                                 matches = matches && terrain_filter(cond_cfg, *this).match_impl(*i, ref_unit);
 416                         }
 417                         //handle [or]
 418                         else if(cond_name == "or")
 419                         {
 420                                 matches = matches || terrain_filter(cond_cfg, *this).match_impl(*i, ref_unit);
 421                         }
 422                         //handle [not]
 423                         else if(cond_name == "not")
 424                         {
 425                                 matches = matches && !terrain_filter(cond_cfg, *this).match_impl(*i, ref_unit);
 426                         }
 427                         ++cond;
 428                 }
 429                 if(matches) {
 430                         return true;
 431                 }
 432                 if(++loop_count > max_loop_) {
 433                         std::set<map_location>::const_iterator temp = i;
 434                         if(++temp != hexes.end()) {
 435                                 ERR_NG << "terrain_filter: loop count greater than " << max_loop_
 436                                 << ", aborting\n";
 437                                 break;
 438                         }
 439                 }
 440         }
 441         return false;
 442 }

 444 class terrain_filterimpl
 445 {
 446 public:
 447         using location_set = std::set<map_location>;
 448         struct no_start_set_yet {};
 449         struct no_filter
 450         {
 451                 bool operator()(const map_location&) const { return true; }
 452         };
 453 
 454         template<typename T, typename F1, typename F2, typename F3>
 455         static void filter_final(T&& src, location_set& dest, const terrain_filter&, const F1& f1, const F2& f2, const F3& f3)
 456         {
 457                 for (const map_location &loc : src) {
 458                         if (f1(loc) && f2(loc) && f3(loc)) {
 459                                 dest.insert(loc);
 460                         }
 461                 }
 462         }
 463 
 464         template<typename T, typename F1, typename F2>
 465         static void filter_special_loc(T&& src, location_set& dest, const terrain_filter& filter, const F1& f1, const F2& f2)
 466         {
 467                 if (filter.cfg_.has_attribute("location_id")) {
 468                         map_location loc2 = filter.fc_->get_disp_context().map().special_location(filter.cfg_["location_id"]);
 469                         filter_final(src, dest, filter, f1, f2, [loc2](const map_location& loc) { return loc == loc2; });
 470                 }
 471                 else {
 472                         filter_final(src, dest, filter, f1, f2, no_filter());
 473                 }
 474         }
 475 
 476         template<typename T, typename F1>
 477         static void filter_area(T&& src, location_set& dest, const terrain_filter& filter, const F1& f1)
 478         {
 479                 if (filter.cfg_.has_attribute("area")) {
 480                         const std::set<map_location>& area = filter.fc_->get_tod_man().get_area_by_id(filter.cfg_["area"]);
 481                         filter_special_loc(src, dest, filter, f1, [&area](const map_location& loc) { return area.find(loc) != area.end(); });
 482                 }
 483                 else {
 484                         filter_special_loc(src, dest, filter, f1, no_filter());
 485                 }
 486         }
 487 
 488         template<typename T>
 489         static void filter_xy(T&& src, location_set& dest, const terrain_filter& filter, bool with_border)
 490         {
 491                 if (filter.cfg_.has_attribute("x") || filter.cfg_.has_attribute("y")) {
 492                         std::vector<map_location> xy_vector = filter.fc_->get_disp_context().map().parse_location_range(filter.cfg_["x"], filter.cfg_["y"], with_border);
 493                         filter_area(src, dest, filter, [&xy_vector](const map_location& loc) { return std::find(xy_vector.begin(), xy_vector.end(), loc) != xy_vector.end(); });
 494                 }
 495                 else {
 496                         filter_area(src, dest, filter, no_filter());
 497                 }
 498         }
 499 };

 506 void terrain_filter::get_locs_impl(std::set<map_location>& locs, const unit* ref_unit, bool with_border) const
 507 {
 508         std::unique_ptr<scoped_wml_variable> ref_unit_var;
 509         if(ref_unit) {
 510                 if(fc_->get_disp_context().map().on_board(ref_unit->get_location())) {
 511                         ref_unit_var.reset(new scoped_xy_unit("teleport_unit", ref_unit->get_location(), fc_->get_disp_context().units()));
 512                 } else {
 513                         // Possible TODO: Support recall list units?
 514                 }
 515         }
 516 
 517         std::set<map_location> match_set;
 518 
 519         // See if the caller provided an override to with_border
 520         with_border = cfg_["include_borders"].to_bool(with_border);
 521 
 522         if (cfg_.has_attribute("find_in")) {
 523 
 524                 if (const game_data * gd = fc_->get_game_data()) {
 525                         try
 526                         {
 527                                 auto ar = gd->get_variable_access_read(cfg_["find_in"]).as_array();
 528                                 terrain_filterimpl::filter_xy(ar | boost::adaptors::transformed(cfg_to_loc()), match_set, *this, with_border);
 529                         }
 530                         catch (const invalid_variablename_exception&)
 531                         {
 532                                 //Do nothing
 533                         }
 534                 }
 535         }
 536         else if (cfg_.has_attribute("x") || cfg_.has_attribute("y")) {
 537                 std::vector<map_location> xy_vector = fc_->get_disp_context().map().parse_location_range(cfg_["x"], cfg_["y"], with_border);
 538                 terrain_filterimpl::filter_area(xy_vector, match_set, *this, terrain_filterimpl::no_filter());
 539         }
 540         else if (cfg_.has_attribute("area")) {
 541                 const std::set<map_location>& area = fc_->get_tod_man().get_area_by_id(cfg_["area"]);
 542                 terrain_filterimpl::filter_special_loc(area, match_set, *this, terrain_filterimpl::no_filter(), terrain_filterimpl::no_filter());
 543         }
 544         else if (cfg_.has_attribute("location_id")) {
 545                 map_location loc2 = fc_->get_disp_context().map().special_location(cfg_["location_id"]);
 546                 if (loc2.valid()) {
 547                         match_set.insert(loc2);
 548                 }
 549         }
 550         else {
 551                 //consider all locations on the map
 552                 int bs = fc_->get_disp_context().map().border_size();
 553                 int w = with_border ? fc_->get_disp_context().map().w() + bs : fc_->get_disp_context().map().w();
 554                 int h = with_border ? fc_->get_disp_context().map().h() + bs : fc_->get_disp_context().map().h();
 555                 for (int x = with_border ? 0 - bs : 0; x < w; ++x) {
 556                         for (int y = with_border ? 0 - bs : 0; y < h; ++y) {
 557                                 match_set.insert(map_location(x, y));
 558                         }
 559                 }
 560         }
 561 
 562         //handle location filter
 563         if(cfg_.has_child("filter_adjacent_location")) {
 564                 if(cache_.adjacent_matches == nullptr) {
 565                         cache_.adjacent_matches = new std::vector<std::set<map_location> >();
 566                 }
 567                 const vconfig::child_list& adj_cfgs = cfg_.get_children("filter_adjacent_location");
 568                 for (unsigned i = 0; i < adj_cfgs.size(); ++i) {
 569                         std::set<map_location> adj_set;
 570                         /* GCC-3.3 doesn't like operator[] so use at(), which has the same result */
 571                         terrain_filter(adj_cfgs.at(i), *this).get_locations(adj_set, with_border);
 572                         cache_.adjacent_matches->push_back(adj_set);
 573                         if(i >= max_loop_ && i+1 < adj_cfgs.size()) {
 574                                 ERR_NG << "terrain_filter: loop count greater than " << max_loop_
 575                                 << ", aborting\n";
 576                                 break;
 577                         }
 578                 }
 579         }
 580         std::set<map_location>::iterator loc_itor = match_set.begin();
 581         while(loc_itor != match_set.end()) {
 582                 if(match_internal(*loc_itor, ref_unit, true)) {
 583                         ++loc_itor;
 584                 } else {
 585                         loc_itor = match_set.erase(loc_itor);
 586                 }
 587         }
 588 
 589         //handle [and], [or], and [not] with in-order precedence
 590         vconfig::all_children_iterator cond = cfg_.ordered_begin();
 591         vconfig::all_children_iterator cond_end = cfg_.ordered_end();
 592         int ors_left = std::count_if(cond, cond_end, cfg_isor());
 593         while(cond != cond_end)
 594         {
 595                 //if there are no locations or [or] conditions left, go ahead and return empty
 596                 if(match_set.empty() && ors_left <= 0) {
 597                         return;
 598                 }
 599 
 600                 const std::string& cond_name = cond.get_key();
 601                 const vconfig& cond_cfg = cond.get_child();
 602 
 603                 //handle [and]
 604                 if(cond_name == "and") {
 605                         std::set<map_location> intersect_hexes;
 606                         terrain_filter(cond_cfg, *this).get_locations(intersect_hexes, with_border);
 607                         std::set<map_location>::iterator intersect_itor = match_set.begin();
 608                         while(intersect_itor != match_set.end()) {
 609                                 if(intersect_hexes.find(*intersect_itor) == intersect_hexes.end()) {
 610                                         match_set.erase(*intersect_itor++);
 611                                 } else {
 612                                         ++intersect_itor;
 613                                 }
 614                         }
 615                 }
 616                 //handle [or]
 617                 else if(cond_name == "or") {
 618                         std::set<map_location> union_hexes;
 619                         terrain_filter(cond_cfg, *this).get_locations(union_hexes, with_border);
 620                         //match_set.insert(union_hexes.begin(), union_hexes.end()); //doesn't compile on MSVC
 621                         std::set<map_location>::iterator insert_itor = union_hexes.begin();
 622                         while(insert_itor != union_hexes.end()) {
 623                                 match_set.insert(*insert_itor++);
 624                         }
 625                         --ors_left;
 626                 }
 627                 //handle [not]
 628                 else if(cond_name == "not") {
 629                         std::set<map_location> removal_hexes;
 630                         terrain_filter(cond_cfg, *this).get_locations(removal_hexes, with_border);
 631                         std::set<map_location>::iterator erase_itor = removal_hexes.begin();
 632                         while(erase_itor != removal_hexes.end()) {
 633                                 match_set.erase(*erase_itor++);
 634                         }
 635                 }
 636                 ++cond;
 637         }
 638         if(match_set.empty()) {
 639                 return;
 640         }
 641 
 642         //handle radius
 643         size_t radius = cfg_["radius"].to_size_t(0);
 644         if(radius > max_loop_) {
 645                 ERR_NG << "terrain_filter: radius greater than " << max_loop_
 646                 << ", restricting\n";
 647                 radius = max_loop_;
 648         }
 649         if(radius > 0) {
 650                 std::vector<map_location> xy_vector (match_set.begin(), match_set.end());
 651                 if(cfg_.has_child("filter_radius")) {
 652                         terrain_filter r_filter(cfg_.child("filter_radius"), *this);
 653                         get_tiles_radius(fc_->get_disp_context().map(), xy_vector, radius, locs, with_border, r_filter);
 654                 } else {
 655                         get_tiles_radius(fc_->get_disp_context().map(), xy_vector, radius, locs, with_border);
 656                 }
 657         } else {
 658                 locs.insert(match_set.begin(), match_set.end());
 659         }
 660 }
 661 
 662 config terrain_filter::to_config() const
 663 {
 664         return cfg_.get_config();
 665 }

 669         delete adjacent_matches;
 670 }
