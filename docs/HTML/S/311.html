  49         /**
  50          * A terrain string which is converted to a terrain is a string with 1 or 2 layers
  51          * the layers are separated by a caret and each group consists of 2 to 4 characters
  52          * if no second layer is defined it is stored as 0xFFFFFFFF, if the second layer
  53          * is empty (needed for matching) the layer has the value 0.
  54          */
  55         struct terrain_code {
  56                 terrain_code(const std::string& b, const std::string& o);
  57                 terrain_code(const std::string& b, ter_layer o = NO_LAYER);
  58                 terrain_code(ter_layer b, ter_layer o) : base(b), overlay(o) {}
  59                 terrain_code() : base(0), overlay(NO_LAYER) {}
  60 
  61                 ter_layer base;
  62                 ter_layer overlay;
  63         };
  64         const terrain_code NONE_TERRAIN = terrain_code();
  65 
  66         inline bool operator<(const terrain_code& a, const terrain_code& b)
  67                 { return a.base < b.base ||  (a.base == b.base && a.overlay < b.overlay); }
  68 
  69         inline bool operator==(const terrain_code& a, const terrain_code& b)
  70                 { return a.base == b.base && a.overlay == b.overlay; }
  71 
  72         inline bool operator!=(const terrain_code& a, const terrain_code& b)
  73                 { return a.base != b.base || a.overlay != b.overlay; }
  74 
  75         inline terrain_code operator&(const terrain_code& a, const terrain_code& b)
  76                 { return terrain_code(a.base & b.base, a.overlay & b.overlay); }
  77 
  78         inline terrain_code operator|(const terrain_code& a, const terrain_code& b)
  79                 { return terrain_code(a.base | b.base, a.overlay | b.overlay); }

 274         /**
 275          * Tests whether a specific terrain matches a list of expressions.
 276          * The list can use wildcard matching with *.
 277          * It also has an inversion function.
 278          * When a ! is found the result of the match is inverted.
 279          * The matching stops at the first match (regardless of the ! found)
 280          * the data is match from start to end.
 281          *
 282          * Example:
 283          * Ww, W*               does match and returns true
 284          * Ww, {!, W*}  does match and returns false (due to the !)
 285          * WW, Ww               doesn't match and return false
 286          *
 287          * Multilayer rules:
 288          * If a terrain has multiple layers, each layer will be matched separately,
 289          * returning true only if both layers match.
 290          *
 291          * Example:
 292          * A*^*     matches Abcd but also Abcd^Abcd
 293          * A*^      matches Abcd but *not* Abcd^Abcd
 294          * A*^Abcd  does not match Abcd but matches Abcd^Abcd
 295          *
 296          * Note: If an expression doesn't specify a second layer (i.e. it contains
 297          * no caret) the second layer will be filled in with a default value
 298          * (See read_terrain_code and read_list).
 299          *
 300          * In the terrain building code, the second layer will default to the wildcard,
 301          * so both A* and A*^* will match Abcd^Abcd
 302          *
 303          * @param src   the value to match (may not contain wildcards)
 304          * @param dest  the list of expressions to match against
 305          *
 306          * @returns             the result of the match (depending on the !'s)
 307          */
 308         bool terrain_matches(const terrain_code& src, const ter_list& dest);
