 130 /** May select only one item. */
 131 struct one_item : public virtual generator_base
 132 {
 133         /**
 134          * Called when an item is selected.
 135          *
 136          * This function can deselect other items to obey the policy. This
 137          * function should always call do_select_item() so the new item does get
 138          * selected.
 139          *
 140          * Since this funcion controls the maximum selection count it should only
 141          * be used to select items, not to deselect them.
 142          *
 143          * @pre                       @p select == @c true
 144          *
 145          * @param index               The item to select.
 146          */
 147         void select_item(const unsigned index, const bool select) override
 148         {
 149                 assert(select);
 150 
 151                 if(get_selected_item_count() == 1) {
 152                         // deselect current.
 153                         do_deselect_item(get_selected_item());
 154                 }
 155                 // select new.
 156                 do_select_item(index);
 157         }
 158 };

 554 /**
 555  * Basic template class to generate new items.
 556  *
 557  * The class is policy based so the behavior can be selected.
 558  */
 559 template <class minimum_selection,
 560                   class maximum_selection,
 561                   class my_placement,
 562                   class select_action>
 563 class generator : public minimum_selection,
 564                                    public maximum_selection,
 565                                    public my_placement,
 566                                    public select_action
 567 {
 568 public:
 569         generator()
 570                 : minimum_selection()
 571                 , maximum_selection()
 572                 , my_placement()
 573                 , select_action()
 574                 , selected_item_count_(0)
 575                 , last_selected_item_(-1)
 576                 , items_()
 577                 , order_()
 578                 , order_dirty_(true)
 579                 , order_func_()
 580         {
 581         }
 582 
 583         ~generator()
 584         {
 585                 clear();
 586         }
 587 
 588         /***** ***** ***** inherited ***** ****** *****/
 589 
 590         /** Inherited from generator_base. */
 591         void delete_item(const unsigned index) override
 592         {
 593                 assert(index < items_.size());
 594 
 595                 // Might be other parts of the engine want to know about the
 596                 // deselection, if minimum fails it gets another chance later on,
 597                 // since it deletes the item.
 598                 if(is_selected(index)) {
 599                         select_item(index, false);
 600                 }
 601 
 602                 minimum_selection::delete_item(index);
 603 
 604                 items_.erase(items_.begin() + index);
 605                 order_dirty_ = true;
 606         }
 607 
 608         /** Inherited from generator_base. */
 609         void clear() override
 610         {
 611                 items_.clear();
 612                 order_dirty_ = true;
 613                 selected_item_count_ = 0;
 614         }
 615 
 616         /** Inherited from generator_base. */
 617         void select_item(const unsigned index, const bool select = true) override
 618         {
 619                 assert(index < items_.size());
 620 
 621                 if(select && !is_selected(index)) {
 622                         maximum_selection::select_item(index, true);
 623                         last_selected_item_ = index;
 624                 } else if(is_selected(index)) {
 625                         if(!minimum_selection::deselect_item(index)) {
 626                                 // Some items might have deseleted themselves so
 627                                 // make sure they do get selected again.
 628                                 select_action::select(item(index), true);
 629                         }
 630                 }
 631         }
 632 
 633         /** Inherited from generator_base. */
 634         bool is_selected(const unsigned index) const override
 635         {
 636                 assert(index < items_.size());
 637                 return (*items_[index]).selected;
 638         }
 639 
 640         /** Inherited from generator_base. */
 641         void set_item_shown(const unsigned index, const bool show) override
 642         {
 643                 assert(index < items_.size());
 644                 if(items_[index]->shown != show) {
 645 
 646                         /*** Set the proper visible state. ***/
 647                         items_[index]->shown = show;
 648                         items_[index]
 649                                         ->child_grid.set_visible(show ? widget::visibility::visible
 650                                                                                         : widget::visibility::invisible);
 651 
 652                         /*** Update the selection. ***/
 653                         minimum_selection::set_item_shown(index, show);
 654                 }
 655         }
 656 
 657         /** Inherited from generator_base. */
 658         virtual bool get_item_shown(const unsigned index) const override
 659         {
 660                 assert(index < items_.size());
 661                 return items_[index]->shown && items_[index]->child_grid.get_visible() != window::visibility::invisible;
 662         }
 663 
 664 
 665         /** Inherited from generator_base. */
 666         unsigned get_item_count() const override
 667         {
 668                 return items_.size();
 669         }
 670 
 671         /** Inherited from generator_base. */
 672         unsigned get_selected_item_count() const override
 673         {
 674                 return selected_item_count_;
 675         }
 676 
 677         /** Inherited from generator_base. */
 678         int get_selected_item() const override
 679         {
 680 
 681                 if(selected_item_count_ == 0) {
 682                         return -1;
 683                 } else if(last_selected_item_ != -1
 684                                   && last_selected_item_ < static_cast<int>(items_.size())
 685                                   && (*items_[last_selected_item_]).selected) {
 686 
 687                         return last_selected_item_;
 688 
 689                 } else {
 690                         for(size_t i = 0; i < items_.size(); ++i) {
 691                                 if((*items_[i]).selected) {
 692                                         return i;
 693                                 }
 694                         }
 695                         FAIL_WITH_DEV_MESSAGE("No item selected.",
 696                                 "selected_item_count_ was non-zero, yet no selected item was found.");
 697                 }
 698         }
 699 
 700         /** Inherited from generator_base. */
 701         grid& item(const unsigned index) override
 702         {
 703                 assert(index < items_.size());
 704                 return items_[index]->child_grid;
 705         }
 706

1010         struct calculate_order_helper
1011         {
1012                 const torder_func& order_func_;
1013                 const child_list& items_;
1014 
1015                 calculate_order_helper(const torder_func& order_func, const child_list& items)
1016                         : order_func_(order_func)
1017                         , items_(items)
1018                 {
1019                 }
1020

1022                 {

1041         void calculate_order() const
1042         {
1043                 if(order_dirty_) {
1044                         if(order_.size() != items_.size()) {
1045                                 order_.resize(items_.size());
1046                                 for(size_t i = 0; i < items_.size(); ++i) {
1047                                         order_[i] = i;
1048                                 }
1049                         }
1050                         if(order_func_) {
1051                                 std::stable_sort(order_.begin(), order_.end(), calculate_order_helper(order_func_, items_));
1052                         }
1053                         for(size_t i = 0; i < order_.size(); ++i) {
1054                                 items_[order_[i]]->ordered_index = i;
1055                         }
1056 
1057                         order_dirty_ = false;
1058                 }
1059                 else {
1060                         assert(order_.size() == items_.size());
1061                 }
1062         }
