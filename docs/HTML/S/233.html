  41 void configuration::init(const config &game_config)
  42 {
  43         ai_configurations_.clear();
  44         era_ai_configurations_.clear();
  45         mod_ai_configurations_.clear();
  46 
  47         const config &ais = game_config.child("ais");
  48         default_config_ = ais.child("default_config");
  49         if (!default_config_) {
  50                 ERR_AI_CONFIGURATION << "Missing AI [default_config]. Therefore, default_config_ set to empty." << std::endl;
  51                 default_config_ = config();
  52         }
  53 
  54 
  55         for (const config &ai_configuration : ais.child_range("ai")) {
  56                 const std::string &id = ai_configuration["id"];
  57                 if (id.empty()){
  58 
  59                         ERR_AI_CONFIGURATION << "skipped AI config due to missing id" << ". Config contains:"<< std::endl << ai_configuration << std::endl;
  60                         continue;
  61                 }
  62                 if (ai_configurations_.count(id)>0){
  63                         ERR_AI_CONFIGURATION << "skipped AI config due to duplicate id [" << id << "]. Config contains:"<< std::endl << ai_configuration << std::endl;
  64                         continue;
  65                 }
  66 
  67                 description desc;
  68                 desc.id=id;
  69                 desc.text = ai_configuration["description"].t_str();
  70                 desc.cfg=ai_configuration;
  71 
  72                 ai_configurations_.emplace(id, desc);
  73                 LOG_AI_CONFIGURATION << "loaded AI config: " << ai_configuration["description"] << std::endl;
  74         }
  75 }

 117 std::vector<description*> configuration::get_available_ais(){
 118         std::vector<description*> ais_list;
 119         for(description_map::iterator desc = ai_configurations_.begin(); desc!=ai_configurations_.end(); ++desc) {
 120                 ais_list.push_back(&desc->second);
 121                 DBG_AI_CONFIGURATION << "has ai with config: "<< std::endl << desc->second.cfg<< std::endl;
 122         }

 128                 ais_list.push_back(&desc->second);
 129                 DBG_AI_CONFIGURATION << "has ai with config: "<< std::endl << desc->second.cfg<< std::endl;

 134 const config& configuration::get_ai_config_for(const std::string &id)
 135 {
 136         description_map::iterator cfg_it = ai_configurations_.find(id);
 137         if (cfg_it==ai_configurations_.end()){
 138                 description_map::iterator era_cfg_it = era_ai_configurations_.find(id);
 139                 if (era_cfg_it==era_ai_configurations_.end()){
 140                         description_map::iterator mod_cfg_it = mod_ai_configurations_.find(id);
 141                         if (mod_cfg_it==mod_ai_configurations_.end()) {
 142                                 return default_config_;
 143                         } else {
 144                                 return mod_cfg_it->second.cfg;
 145                         }
 146                 } else {
 147                         return era_cfg_it->second.cfg;
 148                 }
 149         }
 150         return cfg_it->second.cfg;
 151 }

 159 bool configuration::get_side_config_from_file(const std::string& file, config& cfg ){
 160         try {
 161                 filesystem::scoped_istream stream = preprocess_file(filesystem::get_wml_location(file));
 162                 read(cfg, *stream);
 163                 LOG_AI_CONFIGURATION << "Reading AI configuration from file '" << file  << "'" << std::endl;
 164         } catch(config::error &) {
 165                 ERR_AI_CONFIGURATION << "Error while reading AI configuration from file '" << file  << "'" << std::endl;
 166                 return false;
 167         }
 168         LOG_AI_CONFIGURATION << "Successfully read AI configuration from file '" << file  << "'" << std::endl;
 169         return true;

 178 bool configuration::parse_side_config(side_number side, const config& original_cfg, config &cfg )
 179 {
 180         LOG_AI_CONFIGURATION << "side "<< side <<": parsing AI configuration from config" << std::endl;
 181 
 182         //leave only the [ai] children
 183         cfg = config();
 184         for (const config &aiparam : original_cfg.child_range("ai")) {
 185                 cfg.add_child("ai",aiparam);
 186         }
 187 
 188         //backward-compatibility hack: put ai_algorithm if it is present
 189         if (const config::attribute_value *v = original_cfg.get("ai_algorithm")) {
 190                 config ai_a;
 191                 ai_a["ai_algorithm"] = *v;
 192                 cfg.add_child("ai",ai_a);
 193         }
 194         DBG_AI_CONFIGURATION << "side " << side << ": config contains:"<< std::endl << cfg << std::endl;
 195 
 196         //insert default config at the beginning
 197         if (default_config_) {
 198                 DBG_AI_CONFIGURATION << "side "<< side <<": applying default configuration" << std::endl;
 199                 cfg.add_child_at("ai",default_config_,0);
 200         } else {
 201                 ERR_AI_CONFIGURATION << "side "<< side <<": default configuration is not available, not applying it" << std::endl;
 202         }
 203 
 204         LOG_AI_CONFIGURATION << "side "<< side << ": expanding simplified aspects into full facets"<< std::endl;
 205         expand_simplified_aspects(side, cfg);
 206 
 207         //construct new-style integrated config
 208         LOG_AI_CONFIGURATION << "side "<< side << ": doing final operations on AI config"<< std::endl;
 209         config parsed_cfg = config();
 210 
 211         LOG_AI_CONFIGURATION << "side "<< side <<": merging AI configurations"<< std::endl;
 212         for (const config &aiparam : cfg.child_range("ai")) {
 213                 parsed_cfg.append(aiparam);
 214         }
 215 
 216 
 217         LOG_AI_CONFIGURATION << "side "<< side <<": merging AI aspect with the same id"<< std::endl;
 218         parsed_cfg.merge_children_by_attribute("aspect","id");
 219 
 220         LOG_AI_CONFIGURATION << "side "<< side <<": removing duplicate [default] tags from aspects"<< std::endl;
 221         for (config &aspect_cfg : parsed_cfg.child_range("aspect")) {
 222                 if (aspect_cfg["name"] != "composite_aspect") {
 223                         // No point in warning about Lua or standard aspects lacking [default]
 224                         continue;
 225                 }
 226                 if (!aspect_cfg.child("default")) {
 227                         WRN_AI_CONFIGURATION << "side "<< side <<": aspect with id=["<<aspect_cfg["id"]<<"] lacks default config facet!" <<std::endl;
 228                         continue;
 229                 }
 230                 aspect_cfg.merge_children("default");
 231                 config& dflt = aspect_cfg.child("default");
 232                 if (dflt.has_child("value")) {
 233                         while (dflt.child_count("value") > 1) {
 234                                 dflt.remove_child("value", 0);
 235                         }
 236                 }
 237         }
 238 
 239         DBG_AI_CONFIGURATION << "side "<< side <<": done parsing side config, it contains:"<< std::endl << parsed_cfg << std::endl;
 240         LOG_AI_CONFIGURATION << "side "<< side <<": done parsing side config"<< std::endl;
 241 
 242         cfg = parsed_cfg;
 243         return true;
 244 
 245 }

 251 void configuration::expand_simplified_aspects(side_number side, config &cfg) {
 252         std::string algorithm;
 253         config base_config, parsed_config;
 254         for (const config &aiparam : cfg.child_range("ai")) {
 255                 std::string turns, time_of_day, engine = "cpp";
 256                 if (aiparam.has_attribute("turns")) {
 257                         turns = aiparam["turns"].str();
 258                 }
 259                 if (aiparam.has_attribute("time_of_day")) {
 260                         time_of_day = aiparam["time_of_day"].str();
 261                 }
 262                 if (aiparam.has_attribute("engine")) {
 263                         engine = aiparam["engine"].str();
 264                 }
 265                 if (aiparam.has_attribute("ai_algorithm")) {
 266                         if (algorithm.empty()) {
 267                                 algorithm = aiparam["ai_algorithm"].str();
 268                                 base_config = get_ai_config_for(algorithm);
 269                         } else if(algorithm != aiparam["ai_algorithm"]) {
 270                                 lg::wml_error() << "side " << side << " has two [ai] tags with contradictory ai_algorithm - the first one will take precedence.\n";
 271                         }
 272                 }
 273                 std::deque<std::pair<std::string, config> > facet_configs;
 274                 for (const config::attribute &attr : aiparam.attribute_range()) {
 275                         if (non_aspect_attributes.count(attr.first)) {
 276                                 continue;
 277                         }
 278                         config facet_config;
 279                         facet_config["engine"] = engine;
 280                         facet_config["name"] = "standard_aspect";
 281                         facet_config["turns"] = turns;
 282                         facet_config["time_of_day"] = time_of_day;
 283                         facet_config["value"] = attr.second;
 284                         facet_configs.emplace_back(attr.first, facet_config);
 285                 }
 286                 for (const config::any_child &child : aiparam.all_children_range()) {
 287                         if (just_copy_tags.count(child.key)) {
 288                                 // These aren't simplified, so just copy over unchanged.
 289                                 parsed_config.add_child(child.key, child.cfg);
 290                                 continue;
 291                         } else if(old_goal_tags.count(child.key)) {
 292                                 // A simplified goal, mainly kept around just for backwards compatibility.
 293                                 config goal_config, criteria_config = child.cfg;
 294                                 goal_config["name"] = child.key;
 295                                 goal_config["turns"] = turns;
 296                                 goal_config["time_of_day"] = time_of_day;
 297                                 if(child.key.substr(0,7) == "protect" && criteria_config.has_attribute("protect_radius")) {
 298                                         goal_config["protect_radius"] = criteria_config["protect_radius"];
 299                                         criteria_config.remove_attribute("protect_radius");
 300                                 }
 301                                 if(criteria_config.has_attribute("value")) {
 302                                         goal_config["value"] = criteria_config["value"];
 303                                         criteria_config.remove_attribute("value");
 304                                 }
 305                                 parsed_config.add_child("goal", goal_config);
 306                                 continue;
 307                         }
 308                         // Now there's two possibilities. If the tag is [attacks] or contains either value= or [value],
 309                         // then it can be copied verbatim as a [facet] tag.
 310                         // Otherwise, it needs to be placed as a [value] within a [facet] tag.
 311                         if (child.key == "attacks" || child.cfg.has_attribute("value") || child.cfg.has_child("value")) {
 312                                 facet_configs.emplace_back(child.key, child.cfg);
 313                         } else {
 314                                 config facet_config;
 315                                 facet_config["engine"] = engine;
 316                                 facet_config["name"] = "standard_aspect";
 317                                 facet_config["turns"] = turns;
 318                                 facet_config["time_of_day"] = time_of_day;
 319                                 facet_config.add_child("value", child.cfg);
 320                                 if (child.key == "leader_goal" && !child.cfg["id"].empty()) {
 321                                         // Use id= attribute (if present) as the facet ID
 322                                         const std::string& id = child.cfg["id"];
 323                                         if(id != "*" && id.find_first_not_of("0123456789") != std::string::npos) {
 324                                                 facet_config["id"] = child.cfg["id"];
 325                                         }
 326                                 }
 327                                 facet_configs.emplace_back(child.key, facet_config);
 328                         }
 329                 }
 330                 std::map<std::string, config> aspect_configs;
 331                 while (!facet_configs.empty()) {
 332                         const std::string &aspect = facet_configs.front().first;
 333                         const config &facet_config = facet_configs.front().second;
 334                         aspect_configs[aspect]["id"] = aspect; // Will sometimes be redundant assignment
 335                         aspect_configs[aspect]["name"] = "composite_aspect";
 336                         aspect_configs[aspect].add_child("facet", facet_config);
 337                         facet_configs.pop_front();
 338                 }
 339                 typedef std::map<std::string, config>::value_type aspect_pair;
 340                 for (const aspect_pair& p : aspect_configs) {
 341                         parsed_config.add_child("aspect", p.second);
 342                 }
 343         }
 344         // Support old recruitment aspect syntax
 345         for(auto& child : parsed_config.child_range("aspect")) {
 346                 if(child["id"] == "recruitment") {
 347                         child["id"] = "recruitment_instructions";
 348                 }
 349         }
 350         if (algorithm.empty() && !parsed_config.has_child("stage")) {
 351                 base_config = get_ai_config_for("ai_default_rca");
 352         }
 353         for (const config::any_child &child : parsed_config.all_children_range()) {
 354                 base_config.add_child(child.key, child.cfg);
 355         }
 356         cfg.clear_children("ai");
 357         cfg.add_child("ai", base_config);
 358 }
