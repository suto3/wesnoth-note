 218 #define FLAG_VARIANT NAME
 219     # Enable the specified flag variant: knalgan, loyalist, ragged,
 220     # long, undead, or wood-elvish. Use within a [side] block; only
 221     # affects that side.
 222     # Since the ragged variant has six frames instead of the usual four,
 223     # for that flag it's better to use FLAG_VARIANT6 below instead.
 224     # wmlscope: start ignoring
 225     flag=flags/{NAME}-flag-[1~4].png:150
 226     flag_icon=flags/{NAME}-flag-icon.png
 227     # wmlscope: stop ignoring
 228 #enddef
 229 
 230 #define FLAG_VARIANT6 NAME
 231     # Like FLAG_VARIANT, but this supports flags with six frames.
 232     # Currently only the ragged flag has six frames.
 233     # wmlscope: start ignoring
 234     flag=flags/{NAME}-flag-[1~6].png:150
 235     flag_icon=flags/{NAME}-flag-icon.png
 236     # wmlscope: stop ignoring
 237 #enddef
