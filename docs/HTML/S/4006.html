  10 # Use this to restrict a filter to on-map units, not the recall list.
  11 #define NOT_ON_RECALL_LIST
  12     [not]
  13         x,y=recall,recall
  14     [/not]
  15 #enddef

  17 # Rationale for the naming scheme of these generators:
  18 # All generators take SIDE X Y for consistency.
  19 # Those that begin with NAMED_ additionally take ID_STRING NAME_STRING
  20 # The LOYAL_ macros set the loyalty trait.
  21 
  22 #define UNIT SIDE TYPE X Y WML
  23     # Place a unit with embedded WML specified.
  24     #
  25     [unit]
  26         side={SIDE}
  27         type={TYPE}
  28         x={X}
  29         y={Y}
  30         {WML}
  31     [/unit]
  32 #enddef

 200 #define STORE_UNIT_VAR FILTER VAR TO_VAR_NAME
 201     # Stores an attribute of a unit to the given variable.
 202     #
 203     # Example where this is used to flip all orcs to whatever side James is on:
 204     #! {STORE_UNIT_VAR (id=James) side side_of_James}
 205     #! {MODIFY_UNIT (race=orc) side $side_of_James}
 206     [store_unit]
 207         [filter]
 208             {FILTER}
 209         [/filter]
 210 
 211         kill=no
 212         variable=STORE_UNIT_VAR_store
 213     [/store_unit]
 214 
 215     [set_variable]
 216         name={TO_VAR_NAME}
 217         value=$STORE_UNIT_VAR_store.{VAR}
 218     [/set_variable]
 219 
 220     [clear_variable]
 221         name=STORE_UNIT_VAR_store
 222     [/clear_variable]
 223 #enddef

 225 #define ADVANCE_UNIT FILTER ID_STRING
 226     # Advances all units matching the filter to ID_STRING or if missing, their normal advancement
 227     #
 228     # Example to advance all spearmen to a cavalrymen:
 229     #! {ADVANCE_UNIT type=Spearman Cavalryman}
 230     # Example to advance your leader normally:
 231     #! {ADVANCE_UNIT canrecruit,side=yes,1 ""}
 232 
 233     [modify_unit]
 234         [filter]
 235             {FILTER}
 236         [/filter]
 237         type={ID_STRING}
 238     [/modify_unit]
 239 #enddef
 240 
 241 #define TRANSFORM_UNIT FILTER TYPE
 242     # Transforms all units matching the filter into TYPE or if missing, their normal advancement
 243     # Keeps the unit's hitpoints, experience and status intact.
 244     #
 245     # Example to turn all spearmen into cavalrymen:
 246     #! {TRANSFORM_UNIT type=Spearman Cavalryman}
 247 
 248     [transform_unit]
 249         {FILTER}
 250         transform_to={TYPE}
 251     [/transform_unit]
 252 #enddef
