 163         [lua]
 164             code=<<
 165 -- The only parameter right now is the coverage (real 0.0 to 1.0, incl.)
 166 -- For prime-time use the filter terrain needs to come in and the convert
 167 -- table needs to be VERY complete. Or, rename this to morph_terrain and
 168 -- have the conversion table come in as an argument, too.
 169 
 170 local locations = wesnoth.get_locations{terrain = 'Ce,Ch,Chr,Co,Gd,Gd^Vc,Gd^Vh,Gd^Vo,Gll^Fp,Hhd,Hhd^Fp,Hhd^Vo,Ke,Khr,Ko,Md,Mm,Rd,Re,Re^Vo,Rr,Ww'}
 171 
 172 local coverage = wesnoth.get_variable("snowCoverage")
 173 if coverage == nil or coverage < 0 then
 174     coverage = 0
 175 elseif coverage > 1 then
 176     coverage = 1
 177 end
 178 local snowNeeded = (#locations * coverage) / 3
 179 -- Use 1/3rd of the coverage from Northern Winter. Spring is approaching. We need to watch
 180 -- this; if it adds too much snow, the player may not be able to make it to the exit in time.
 181 
 182 local convert = {
 183     ['Ce'] = 'Cea',
 184     ['Ch'] = 'Cha',
 185     ['Chr'] = 'Cha',
 186     ['Co'] = 'Coa',
 187     ['Gd'] = 'Aa',
 188     ['Gd^Vc'] = 'Aa^Vca',
 189     ['Gd^Vh'] = 'Aa^Vha',
 190     ['Gd^Vo'] = 'Aa^Voa',
 191     ['Gll^Fp'] = 'Aa^Fpa',
 192     ['Hhd'] = 'Ha',
 193     ['Hhd^Fp'] = 'Ha^Fpa',
 194     ['Hhd^Vo'] = 'Ha^Voa',
 195     ['Ke'] = 'Kea',
 196     ['Khr'] = 'Kha',
 197     ['Ko'] = 'Koa',
 198     ['Md'] = 'Ms',
 199     ['Mm'] = 'Ms',
 200     ['Rd'] = 'Aa',
 201     ['Re'] = 'Aa',
 202     ['Re^Vo'] = 'Aa^Voa',
 203     ['Rr'] = 'Aa',
 204     ['Ww'] = 'Ai'
 205 }
 206 
 207 local loopCounter
 208 for loopCounter = 1, snowNeeded do
 209     local locationsIndex = wesnoth.random(#locations)
 210     local coordinate = locations[locationsIndex]
 211     local terrainCode = wesnoth.get_terrain(coordinate[1], coordinate[2])
 212     wesnoth.set_terrain(coordinate[1], coordinate[2], (convert[terrainCode] or terrainCode))
 213     table.remove(locations, locationsIndex)
 214 end
