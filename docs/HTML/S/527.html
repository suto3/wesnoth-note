  70 /**
  71  * SDL_AddTimer() callback for the draw event.
  72  *
  73  * When this callback is called it pushes a new draw event in the event queue.
  74  *
  75  * @returns                       The new timer interval, 0 to stop.
  76  */
  77 static Uint32 timer_sdl_draw_event(Uint32, void*)
  78 {
  79         // DBG_GUI_E << "Pushing draw event in queue.\n";
  80 
  81         SDL_Event event;
  82         SDL_UserEvent data;
  83 
  84         data.type = DRAW_EVENT;
  85         data.code = 0;
  86         data.data1 = nullptr;
  87         data.data2 = nullptr;
  88 
  89         event.type = DRAW_EVENT;
  90         event.user = data;
  91 
  92         SDL_PushEvent(&event);
  93         return draw_interval;
  94 }
  95 
  96 /**
  97  * SDL_AddTimer() callback for the poll event.
  98  *
  99  * When this callback is called it will run the events in the SDL event queue.

 105         try
 106         {
 107                 events::pump();
 108         }
 109         catch(CVideo::quit&)

 111                 return 0;

 334 void sdl_event_handler::handle_event(const SDL_Event& event)
 335 {
 336         /** No dispatchers drop the event. */
 337         if(dispatchers_.empty()) {
 338                 return;
 339         }
 340 
 341         switch(event.type) {
 342                 case SDL_MOUSEMOTION:
 343                         mouse(SDL_MOUSE_MOTION, {event.motion.x, event.motion.y});
 344                         break;
 345 
 346                 case SDL_MOUSEBUTTONDOWN:
 347                         mouse_button_down({event.button.x, event.button.y}, event.button.button);
 348                         break;
 349 
 350                 case SDL_MOUSEBUTTONUP:
 351                         mouse_button_up({event.button.x, event.button.y}, event.button.button);
 352                         break;
 353 
 354                 case SDL_MOUSEWHEEL:
 355                         mouse_wheel(get_mouse_position(), event.wheel.x, event.wheel.y);
 356                         break;
 357 
 358                 case SHOW_HELPTIP_EVENT:
 359                         mouse(SHOW_HELPTIP, get_mouse_position());
 360                         break;
 361 
 362                 case HOVER_REMOVE_POPUP_EVENT:
 363                         // remove_popup();
 364                         break;
 365 
 366                 case DRAW_EVENT:
 367                         draw(false);
 368                         break;
 369                 case DRAW_ALL_EVENT:
 370                         draw(true);
 371                         break;
 372 
 373                 case TIMER_EVENT:
 374                         execute_timer(reinterpret_cast<size_t>(event.user.data1));
 375                         break;
 376 
 377                 case CLOSE_WINDOW_EVENT: {
 378                         /** @todo Convert this to a proper new style event. */
 379                         DBG_GUI_E << "Firing " << CLOSE_WINDOW << ".\n";
 380 
 381                         window* window = window::window_instance(event.user.code);
 382                         if(window) {
 383                                 window->set_retval(window::AUTO_CLOSE);
 384                         }
 385                 } break;
 386 
 387                 case SDL_JOYBUTTONDOWN:
 388                         button_down(event);
 389                         break;
 390 
 391                 case SDL_JOYBUTTONUP:
 392                         break;
 393 
 394                 case SDL_JOYAXISMOTION:
 395                         break;
 396 
 397                 case SDL_JOYHATMOTION:
 398                         hat_motion(event);
 399                         break;
 400 
 401                 case SDL_KEYDOWN:
 402                         key_down(event);
 403                         break;
 404 
 405                 case SDL_WINDOWEVENT:
 406                         switch(event.window.event) {
 407                                 case SDL_WINDOWEVENT_EXPOSED:
 408                                         draw(true);
 409                                         break;
 410 
 411                                 case SDL_WINDOWEVENT_RESIZED:
 412                                         video_resize({event.window.data1, event.window.data2});
 413                                         break;
 414 
 415                                 case SDL_WINDOWEVENT_ENTER:
 416                                 case SDL_WINDOWEVENT_FOCUS_GAINED:
 417                                         activate();
 418                                         break;
 419                         }
 420 
 421                         break;
 422 
 423                 case SDL_TEXTINPUT:
 424                         text_input(event.text.text);
 425                         break;
 426 
 427                 case SDL_FINGERMOTION:
 428                         touch_motion(point(event.tfinger.x, event.tfinger.y), point(event.tfinger.dx, event.tfinger.dy));
 429                         break;
 430 
 431 #if(defined(_X11) && !defined(__APPLE__)) || defined(_WIN32)
 432                 case SDL_SYSWMEVENT:
 433                         /* DO NOTHING */
 434                         break;
 435 #endif
 436 
 437                 // Silently ignored events.
 438                 case SDL_KEYUP:
 439                 case DOUBLE_CLICK_EVENT:
 440                 case SDL_FINGERUP:
 441                 case SDL_FINGERDOWN:
 442                         break;
 443 
 444                 default:
 445 #ifdef GUI2_SHOW_UNHANDLED_EVENT_WARNINGS
 446                         WRN_GUI_E << "Unhandled event " << static_cast<Uint32>(event.type)
 447                                   << ".\n";
 448 #endif
 449                         break;
 450         }
 451 
 452         raw_event(event);
 453 }

 460 void sdl_event_handler::connect(dispatcher* dispatcher)
 461 {
 462         assert(std::find(dispatchers_.begin(), dispatchers_.end(), dispatcher)
 463                    == dispatchers_.end());
 464 
 465         if(dispatchers_.empty()) {
 466                 event_context = new events::event_context();
 467                 join();
 468         }
 469 
 470         dispatchers_.push_back(dispatcher);
 471 }
 472 
 473 void sdl_event_handler::disconnect(dispatcher* disp)
 474 {
 475         /***** Validate pre conditions. *****/
 476         auto itor = std::find(dispatchers_.begin(), dispatchers_.end(), disp);
 477         assert(itor != dispatchers_.end());
 478 
 479         /***** Remove dispatcher. *****/
 480         dispatchers_.erase(itor);
 481 
 482         if(disp == mouse_focus) {
 483                 mouse_focus = nullptr;
 484         }
 485         if(disp == keyboard_focus_) {
 486                 keyboard_focus_ = nullptr;
 487         }
 488 
 489         /***** Set proper state for the other dispatchers. *****/
 490         for(auto d : dispatchers_)
 491         {
 492                 dynamic_cast<widget&>(*d).set_is_dirty(true);
 493         }
 494 
 495         activate();
 496 
 497         /***** Validate post conditions. *****/
 498         assert(std::find(dispatchers_.begin(), dispatchers_.end(), disp)
 499                    == dispatchers_.end());
 500 
 501         if(dispatchers_.empty()) {
 502                 leave();
 503                 delete event_context;
 504                 event_context = nullptr;
 505         }
 506 }

 516 void sdl_event_handler::draw(const bool force)
 517 {
 518         // Don't display this event since it floods the screen
 519         // DBG_GUI_E << "Firing " << DRAW << ".\n";
 520 
 521         /*
 522          * In normal draw mode the first window in not forced to be drawn the
 523          * others are. So for forced mode we only need to force the first window to
 524          * be drawn the others are already handled.
 525          */
 526         bool first = !force;
 527 
 528         /**
 529          * @todo Need to evaluate which windows really to redraw.
 530          *
 531          * For now we use a hack, but would be nice to rewrite it for 1.9/1.11.
 532          */
 533         for(auto dispatcher : dispatchers_)
 534         {
 535                 if(!first) {
 536                         /*
 537                          * This leaves glitches on window borders if the window beneath it
 538                          * has changed, on the other hand invalidating twindown::restorer_
 539                          * causes black borders around the window. So there's the choice
 540                          * between two evils.
 541                          */
 542                         dynamic_cast<widget&>(*dispatcher).set_is_dirty(true);
 543                 } else {
 544                         first = false;
 545                 }
 546 
 547                 dispatcher->fire(DRAW, dynamic_cast<widget&>(*dispatcher));
 548         }
 549 
 550         if(!dispatchers_.empty()) {
 551                 CVideo& video = dynamic_cast<window&>(*dispatchers_.back()).video();
 552 
 553                 video.flip();
 554         }
 555 }

 576 void sdl_event_handler::mouse(const ui_event event, const point& position)
 577 {
 578         DBG_GUI_E << "Firing: " << event << ".\n";
 579 
 580         if(mouse_focus) {
 581                 mouse_focus->fire(event, dynamic_cast<widget&>(*mouse_focus), position);
 582                 return;
 583         }
 584 
 585         for(auto& dispatcher : boost::adaptors::reverse(dispatchers_)) {
 586                 if(dispatcher->get_mouse_behavior() == dispatcher::all) {
 587                         dispatcher->fire(event, dynamic_cast<widget&>(*dispatcher), position);
 588                         break;
 589                 }
 590 
 591                 if(dispatcher->get_mouse_behavior() == dispatcher::none) {
 592                         continue;
 593                 }
 594 
 595                 if(dispatcher->is_at(position)) {
 596                         dispatcher->fire(event, dynamic_cast<widget&>(*dispatcher), position);
 597                         break;
 598                 }
 599         }
 600 }
