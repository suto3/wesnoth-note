 108 void game_display::new_turn()
 109 {
 110         const time_of_day& tod = tod_manager_->get_time_of_day();
 111 
 112         if( !first_turn_) {
 113                 const time_of_day& old_tod = tod_manager_->get_previous_time_of_day();
 114 
 115                 if(old_tod.image_mask != tod.image_mask) {
 116                         surface old_mask(image::get_image(old_tod.image_mask,image::SCALED_TO_HEX));
 117                         surface new_mask(image::get_image(tod.image_mask,image::SCALED_TO_HEX));
 118 
 119                         const int niterations = static_cast<int>(10/turbo_speed());
 120                         const int frame_time = 30;
 121                         const int starting_ticks = SDL_GetTicks();
 122                         for(int i = 0; i != niterations; ++i) {
 123 
 124                                 if(old_mask != nullptr) {
 125                                         const fixed_t proportion = ftofxp(1.0) - fxpdiv(i,niterations);
 126                                         adjust_surface_alpha(old_mask, proportion);
 127                                         tod_hex_mask1.assign(old_mask);
 128                                 }
 129 
 130                                 if(new_mask != nullptr) {
 131                                         const fixed_t proportion = fxpdiv(i,niterations);
 132                                         adjust_surface_alpha(new_mask, proportion);
 133                                         tod_hex_mask2.assign(new_mask);
 134                                 }
 135 
 136                                 invalidate_all();
 137                                 draw();
 138 
 139                                 const int cur_ticks = SDL_GetTicks();
 140                                 const int wanted_ticks = starting_ticks + i*frame_time;
 141                                 if(cur_ticks < wanted_ticks) {
 142                                         SDL_Delay(wanted_ticks - cur_ticks);
 143                                 }
 144                         }
 145                 }
 146 
 147                 tod_hex_mask1.assign(nullptr);
 148                 tod_hex_mask2.assign(nullptr);
 149         }
 150 
 151         first_turn_ = false;
 152 
 153         display::update_tod();
 154 
 155         invalidate_all();
 156         draw();
 157 }

 169 void game_display::highlight_hex(map_location hex)
 170 {
 171         wb::future_map_if future(!synced_context::is_synced()); /**< Lasts for whole method. */
 172 
 173         const unit *u = resources::gameboard->get_visible_unit(hex, dc_->teams()[viewing_team()], !dont_show_all_);
 174         if (u) {
 175                 displayedUnitHex_ = hex;
 176                 invalidate_unit();
 177         } else {
 178                 u = resources::gameboard->get_visible_unit(mouseoverHex_, dc_->teams()[viewing_team()], !dont_show_all_);
 179                 if (u) {
 180                         // mouse moved from unit hex to non-unit hex
 181                         if (dc_->units().count(selectedHex_)) {
 182                                 displayedUnitHex_ = selectedHex_;
 183                                 invalidate_unit();
 184                         }
 185                 }
 186         }
 187 
 188         display::highlight_hex(hex);
 189         invalidate_game_status();
 190 }

 243 void game_display::draw_invalidated()
 244 {
 245         halo_man_->unrender(invalidated_);
 246         display::draw_invalidated();
 247         if (fake_unit_man_->empty()) {
 248                 return;
 249         }
 250         unit_drawer drawer = unit_drawer(*this);
 251 
 252         for (const unit* temp_unit : *fake_unit_man_) {
 253                 const map_location& loc = temp_unit->get_location();
 254                 exclusive_unit_draw_requests_t::iterator request = exclusive_unit_draw_requests_.find(loc);
 255                 if (invalidated_.find(loc) != invalidated_.end()
 256                                 && (request == exclusive_unit_draw_requests_.end() || request->second == temp_unit->id()))
 257                         drawer.redraw_unit(*temp_unit);

 266 void game_display::draw_hex(const map_location& loc)
 267 {
 268         const bool on_map = get_map().on_board(loc);
 269         const bool is_shrouded = shrouded(loc);
 270 //      const bool is_fogged = fogged(loc);
 271         const int xpos = get_location_x(loc);
 272         const int ypos = get_location_y(loc);
 273 
 274 //      image::TYPE image_type = get_image_type(loc);
 275 
 276         display::draw_hex(loc);
 277 
 278         if(cursor::get() == cursor::WAIT) {
 279                 // Interaction is disabled, so we don't need anything else
 280                 return;
 281         }
 282 
 283         if(on_map && loc == mouseoverHex_) {
 284                 drawing_layer hex_top_layer = LAYER_MOUSEOVER_BOTTOM;
 285                 const unit *u = resources::gameboard->get_visible_unit(loc, dc_->teams()[viewing_team()] );
 286                 if( u != nullptr ) {
 287                         hex_top_layer = LAYER_MOUSEOVER_TOP;
 288                 }
 289                 if(u == nullptr) {
 290                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 291                                         image::get_image("misc/hover-hex-top.png~RC(magenta>gold)", image::SCALED_TO_HEX));
 292                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 293                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>gold)", image::SCALED_TO_HEX));
 294                 } else if(dc_->teams()[currentTeam_].is_enemy(u->side())) {
 295                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 296                                         image::get_image("misc/hover-hex-enemy-top.png~RC(magenta>red)", image::SCALED_TO_HEX));
 297                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 298                                         image::get_image("misc/hover-hex-enemy-bottom.png~RC(magenta>red)", image::SCALED_TO_HEX));
 299                 } else if(dc_->teams()[currentTeam_].side() == u->side()) {
 300                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 301                                         image::get_image("misc/hover-hex-top.png~RC(magenta>green)", image::SCALED_TO_HEX));
 302                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 303                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>green)", image::SCALED_TO_HEX));
 304                 } else {
 305                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 306                                         image::get_image("misc/hover-hex-top.png~RC(magenta>lightblue)", image::SCALED_TO_HEX));
 307                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 308                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>lightblue)", image::SCALED_TO_HEX));
 309                 }
 310         }
 311 
 312 
 313 
 314         // Draw reach_map information.
 315         // We remove the reachability mask of the unit
 316         // that we want to attack.
 317         if (!is_shrouded && !reach_map_.empty()
 318                         && reach_map_.find(loc) == reach_map_.end() && loc != attack_indicator_dst_) {
 319                 static const image::locator unreachable(game_config::images::unreachable);
 320                 drawing_buffer_add(LAYER_REACHMAP, loc, xpos, ypos,
 321                                 image::get_image(unreachable,image::SCALED_TO_HEX));
 322         }
 323 
 324         if (std::shared_ptr<wb::manager> w = wb_.lock()) {
 325                 w->draw_hex(loc);
 326 
 327                 if (!(w->is_active() && w->has_temp_move()))
 328                 {
 329                         std::vector<surface> footstepImages = footsteps_images(loc, route_, dc_);
 330                         if (!footstepImages.empty()) {
 331                                 drawing_buffer_add(LAYER_FOOTSTEPS, loc, xpos, ypos, footsteps_images(loc, route_, dc_));
 332                         }
 333                 }
 334         }
 335         // Draw the attack direction indicator
 336         if(on_map && loc == attack_indicator_src_) {
 337                 drawing_buffer_add(LAYER_ATTACK_INDICATOR, loc, xpos, ypos,
 338                         image::get_image("misc/attack-indicator-src-" + attack_indicator_direction() + ".png", image::SCALED_TO_HEX));
 339         } else if (on_map && loc == attack_indicator_dst_) {
 340                 drawing_buffer_add(LAYER_ATTACK_INDICATOR, loc, xpos, ypos,
 341                         image::get_image("misc/attack-indicator-dst-" + attack_indicator_direction() + ".png", image::SCALED_TO_HEX));
 342         }
 343 
 344         // Linger overlay unconditionally otherwise it might give glitches
 345         // so it's drawn over the shroud and fog.
 346         if(mode_ != RUNNING) {
 347                 static const image::locator linger(game_config::images::linger);
 348                 drawing_buffer_add(LAYER_LINGER_OVERLAY, loc, xpos, ypos,
 349                         image::get_image(linger, image::TOD_COLORED));
 350         }
 351 
 352         if(on_map && loc == selectedHex_ && !game_config::images::selected.empty()) {
 353                 static const image::locator selected(game_config::images::selected);
 354                 drawing_buffer_add(LAYER_SELECTED_HEX, loc, xpos, ypos,
 355                                 image::get_image(selected, image::SCALED_TO_HEX));
 356         }
 357 
 358         // Show def% and turn to reach info
 359         if(!is_shrouded && on_map) {
 360                 draw_movement_info(loc);
 361         }
 362 
 363         if(game_config::debug) {
 364                 int debugH = debugHighlights_[loc];
 365                 if (debugH) {
 366                         std::string txt = std::to_string(debugH);
 367                         draw_text_in_hex(loc, LAYER_MOVE_INFO, txt, 18, font::BAD_COLOR);
 368                 }
 369         }
 370         //simulate_delay += 1;
 371 }

 413 void game_display::draw_movement_info(const map_location& loc)
 414 {
 415         // Search if there is a mark here
 416         pathfind::marked_route::mark_map::iterator w = route_.marks.find(loc);
 417 
 418         std::shared_ptr<wb::manager> wb = wb_.lock();
 419 
 420         // Don't use empty route or the first step (the unit will be there)
 421         if(w != route_.marks.end()
 422                                 && !route_.steps.empty() && route_.steps.front() != loc) {
 423                 const unit_map::const_iterator un =
 424                                 (wb && wb->get_temp_move_unit().valid()) ?
 425                                                 wb->get_temp_move_unit() : dc_->units().find(route_.steps.front());
 426                 if(un != dc_->units().end()) {
 427                         // Display the def% of this terrain
 428                         int def =  100 - un->defense_modifier(get_map().get_terrain(loc));
 429                         std::stringstream def_text;
 430                         def_text << def << "%";
 431 
 432                         color_t color = game_config::red_to_green(def, false);
 433 
 434                         // simple mark (no turn point) use smaller font
 435                         int def_font = w->second.turns > 0 ? 18 : 16;
 436                         draw_text_in_hex(loc, LAYER_MOVE_INFO, def_text.str(), def_font, color);
 437 
 438                         int xpos = get_location_x(loc);
 439                         int ypos = get_location_y(loc);
 440                         if (w->second.invisible) {
 441                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 442                                         image::get_image("misc/hidden.png", image::SCALED_TO_HEX));
 443                         }
 444 
 445                         if (w->second.zoc) {
 446                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 447                                         image::get_image("misc/zoc.png", image::SCALED_TO_HEX));
 448                         }
 449 
 450                         if (w->second.capture) {
 451                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 452                                         image::get_image("misc/capture.png", image::SCALED_TO_HEX));
 453                         }
 454 
 455                         //we display turn info only if different from a simple last "1"
 456                         if (w->second.turns > 1 || (w->second.turns == 1 && loc != route_.steps.back())) {
 457                                 std::stringstream turns_text;
 458                                 turns_text << w->second.turns;
 459                                 draw_text_in_hex(loc, LAYER_MOVE_INFO, turns_text.str(), 17, font::NORMAL_COLOR, 0.5,0.8);
 460                         }
 461 
 462                         // The hex is full now, so skip the "show enemy moves"
 463                         return;
 464                 }
 465         }
 466         // When out-of-turn, it's still interesting to check out the terrain defs of the selected unit
 467         else if (selectedHex_.valid() && loc == mouseoverHex_)
 468         {
 469                 const unit_map::const_iterator selectedUnit = resources::gameboard->find_visible_unit(selectedHex_,dc_->teams()[currentTeam_]);
 470                 const unit_map::const_iterator mouseoveredUnit = resources::gameboard->find_visible_unit(mouseoverHex_,dc_->teams()[currentTeam_]);
 471                 if(selectedUnit != dc_->units().end() && mouseoveredUnit == dc_->units().end()) {
 472                         // Display the def% of this terrain
 473                         int def =  100 - selectedUnit->defense_modifier(get_map().get_terrain(loc));
 474                         std::stringstream def_text;
 475                         def_text << def << "%";
 476 
 477                         color_t color = game_config::red_to_green(def, false);
 478 
 479                         // use small font
 480                         int def_font = 16;
 481                         draw_text_in_hex(loc, LAYER_MOVE_INFO, def_text.str(), def_font, color);
 482                 }
 483         }
 484 
 485         if (!reach_map_.empty()) {
 486                 reach_map::iterator reach = reach_map_.find(loc);
 487                 if (reach != reach_map_.end() && reach->second > 1) {
 488                         const std::string num = std::to_string(reach->second);
 489                         draw_text_in_hex(loc, LAYER_MOVE_INFO, num, 16, font::YELLOW_COLOR);
 490                 }
 491         }
 492 }

 494 std::vector<surface> footsteps_images(const map_location& loc, const pathfind::marked_route & route_, const display_context * dc_)
 495 {
 496         std::vector<surface> res;
 497 
 498         if (route_.steps.size() < 2) {
 499                 return res; // no real "route"
 500         }
 501 
 502         std::vector<map_location>::const_iterator i =
 503                  std::find(route_.steps.begin(),route_.steps.end(),loc);
 504 
 505         if( i == route_.steps.end()) {
 506                 return res; // not on the route
 507         }
 508 
 509         // Check which footsteps images of game_config we will use
 510         int move_cost = 1;
 511         const unit_map::const_iterator u = dc_->units().find(route_.steps.front());
 512         if(u != dc_->units().end()) {
 513                 move_cost = u->movement_cost(dc_->map().get_terrain(loc));
 514         }
 515         int image_number = std::min<int>(move_cost, game_config::foot_speed_prefix.size());
 516         if (image_number < 1) {
 517                 return res; // Invalid movement cost or no images
 518         }
 519         const std::string foot_speed_prefix = game_config::foot_speed_prefix[image_number-1];
 520 
 521         surface teleport = nullptr;
 522 
 523         // We draw 2 half-hex (with possibly different directions),
 524         // but skip the first for the first step.
 525         const int first_half = (i == route_.steps.begin()) ? 1 : 0;
 526         // and the second for the last step
 527         const int second_half = (i+1 == route_.steps.end()) ? 0 : 1;
 528 
 529         for (int h = first_half; h <= second_half; ++h) {
 530                 const std::string sense( h==0 ? "-in" : "-out" );
 531 
 532                 if (!tiles_adjacent(*(i+(h-1)), *(i+h))) {
 533                         std::string teleport_image =
 534                         h==0 ? game_config::foot_teleport_enter : game_config::foot_teleport_exit;
 535                         teleport = image::get_image(teleport_image, image::SCALED_TO_HEX);
 536                         continue;
 537                 }
 538 
 539                 // In function of the half, use the incoming or outgoing direction
 540                 map_location::DIRECTION dir = (i+(h-1))->get_relative_dir(*(i+h));
 541 
 542                 std::string rotate;
 543                 if (dir > map_location::SOUTH_EAST) {
 544                         // No image, take the opposite direction and do a 180 rotation
 545                         dir = i->get_opposite_dir(dir);
 546                         rotate = "~FL(horiz)~FL(vert)";
 547                 }
 548 
 549                 const std::string image = foot_speed_prefix
 550                         + sense + "-" + i->write_direction(dir)
 551                         + ".png" + rotate;
 552 
 553                 res.push_back(image::get_image(image, image::SCALED_TO_HEX));
 554         }
 555 
 556         // we draw teleport image (if any) in last
 557         if (teleport != nullptr) res.push_back(teleport);
 558 
 559         return res;
 560 }

 588 {
 589         for(std::vector<map_location>::const_iterator i = route_.steps.begin();

 591                 invalidate(*i);
 592         }

 595 void game_display::set_route(const pathfind::marked_route *route)
 596 {
 597         invalidate_route();
 598 
 599         if(route != nullptr) {
 600                 route_ = *route;
 601         } else {
 602                 route_.steps.clear();
 603                 route_.marks.clear();
 604         }

 606         invalidate_route();

 609 void game_display::float_label(const map_location& loc, const std::string& text, const color_t& color)
 610 {
 611         if(preferences::show_floating_labels() == false || fogged(loc)) {
 612                 return;
 613         }
 614 
 615         font::floating_label flabel(text);
 616         flabel.set_font_size(font::SIZE_XLARGE);
 617         flabel.set_color(color);
 618         flabel.set_position(get_location_x(loc)+zoom_/2, get_location_y(loc));
 619         flabel.set_move(0, -2 * turbo_speed());
 620         flabel.set_lifetime(60/turbo_speed());
 621         flabel.set_scroll_mode(font::ANCHOR_LABEL_MAP);
 622 
 623         font::add_floating_label(flabel);
 624 }
