  77 void editor_edit_side::pre_show(window& window)
  78 {
  79         controller_group.add_member(&find_widget<toggle_button>(&window, "controller_human", false), team::CONTROLLER::HUMAN);
  80         controller_group.add_member(&find_widget<toggle_button>(&window, "controller_ai", false),    team::CONTROLLER::AI);
  81         controller_group.add_member(&find_widget<toggle_button>(&window, "controller_null", false),  team::CONTROLLER::EMPTY);
  82 
  83         controller_group.set_member_states(controller_);
  84 
  85         vision_group.add_member(&find_widget<toggle_button>(&window, "vision_all", false),    team::SHARE_VISION::ALL);
  86         vision_group.add_member(&find_widget<toggle_button>(&window, "vision_shroud", false), team::SHARE_VISION::SHROUD);
  87         vision_group.add_member(&find_widget<toggle_button>(&window, "vision_null", false),   team::SHARE_VISION::NONE);
  88 
  89         vision_group.set_member_states(share_vision_);
  90 
  91         window.add_to_tab_order(find_widget<text_box>(&window, "team_name", false, false));
  92         window.add_to_tab_order(find_widget<text_box>(&window, "user_team_name", false, false));
  93 }
  94 
  95 void editor_edit_side::post_show(window&)
  96 {
  97         controller_ = controller_group.get_active_member_value();
  98         share_vision_ = vision_group.get_active_member_value();
  99 }
