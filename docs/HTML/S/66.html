  67 /**
  68  * Utility class to properly refresh the display when the map context object is replaced
  69  * without duplicating code.
  70  */
  71 class map_context_refresher
  72 {
  73 public:
  74         map_context_refresher(context_manager& ec)
  75                 : context_manager_(ec)
  76                 , refreshed_(false)
  77         {
  78         }
  79 
  80         ~map_context_refresher()
  81         {
  82                 if(!refreshed_) refresh();
  83         }
  84 
  85         void refresh()
  86         {
  87                 context_manager_.gui().change_display_context(&context_manager_.get_map_context());
  88 
  89                 // TODO register the tod_manager with the gui?
  90                 resources::tod_manager = context_manager_.get_map_context().get_time_manager();
  91                 resources::units = &context_manager_.get_map_context().get_units();
  92                 resources::filter_con = &context_manager_.gui();
  93 
  94                 context_manager_.gui().replace_overlay_map(&context_manager_.get_map_context().get_overlays());
  95 
  96                 resources::classification = &context_manager_.get_map_context().get_classification();
  97 
  98                 context_manager_.gui().init_flags();
  99 
 100                 context_manager_.reload_map();
 101 
 102                 // Enable the labels of the current context;
 103                 context_manager_.get_labels().enable(true);
 104 
 105                 refreshed_ = true;
 106         }
 107 
 108 private:
 109         context_manager& context_manager_;
 110         bool refreshed_;
 111 };

 184 size_t context_manager::modified_maps(std::string& message)
 185 {
 186         std::vector<std::string> modified;
 187         for(auto& mc : map_contexts_) {
 188                 if(mc->modified()) {
 189                         if(!mc->get_name().empty()) {
 190                                 modified.push_back(mc->get_name());
 191                         } else if(!mc->get_filename().empty()) {
 192                                 modified.push_back(mc->get_filename());
 193                         } else {
 194                                 modified.push_back(mc->get_default_context_name());
 195                         }
 196                 }
 197         }
 198 
 199         for(std::string& str : modified) {
 200                 message += "\n" + font::unicode_bullet + " " + str;
 201         }
 202 
 203         return modified.size();
 204 }

 298 void context_manager::expand_open_maps_menu(std::vector<config>& items, int i)
 299 {
 300         auto pos = items.erase(items.begin() + i);
 301         std::vector<config> contexts;
 302 
 303         for(size_t mci = 0; mci < map_contexts_.size(); ++mci) {
 304                 map_context& mc = *map_contexts_[mci];
 305 
 306                 std::string filename;
 307                 if(mc.is_pure_map()) {
 308                         filename = filesystem::base_name(mc.get_filename());
 309                 } else {
 310                         filename = mc.get_name();
 311                 }
 312 
 313                 if(filename.empty()) {
 314                         filename = mc.get_default_context_name();
 315                 }
 316 
 317                 std::ostringstream ss;
 318                 ss << "[" << mci + 1 << "] ";
 319 
 320                 const bool changed = mc.modified();
 321 
 322                 if(changed) {
 323                         ss << "<i>" << filename << "</i>";
 324                 } else {
 325                         ss << filename;
 326                 }
 327 
 328                 if(mc.is_embedded()) {
 329                         ss << " (E)";
 330                 }
 331 
 332                 const std::string label = ss.str();
 333                 const std::string details = get_menu_marker(changed);
 334 
 335                 contexts.emplace_back(config_of("label", label)("details", details));
 336         }
 337 
 338         items.insert(pos, contexts.begin(), contexts.end());
 339 }

 341 void context_manager::expand_load_mru_menu(std::vector<config>& items, int i)
 342 {
 343         std::vector<std::string> mru = preferences::editor::recent_files();
 344 
 345         auto pos = items.erase(items.begin() + i);
 346 
 347         if(mru.empty()) {
 348                 items.insert(pos, config_of("label", _("No Recent Files")));
 349                 return;
 350         }
 351 
 352         for(std::string& path : mru) {
 353                 // TODO: add proper leading ellipsization instead, since otherwise
 354                 // it'll be impossible to tell apart files with identical names and
 355                 // different parent paths.
 356                 path = filesystem::base_name(path);
 357         }
 358 
 359         std::vector<config> temp;
 360         std::transform(mru.begin(), mru.end(), std::back_inserter(temp), [](const std::string& str) {
 361                 return config_of("label", str);
 362         });
 363 
 364         items.insert(pos, temp.begin(), temp.end());
 365 }

 392                         mci == static_cast<size_t>(get_map_context().get_active_area())

 534 void context_manager::refresh_after_action(bool drag_part)
 535 {
 536         if(get_map_context().needs_reload()) {
 537                 reload_map();
 538                 return;
 539         }
 540 
 541         const std::set<map_location>& changed_locs = get_map_context().changed_locations();
 542 
 543         if(get_map_context().needs_terrain_rebuild()) {
 544                 if((auto_update_transitions_ == preferences::editor::TransitionUpdateMode::on)
 545                 || ((auto_update_transitions_ == preferences::editor::TransitionUpdateMode::partial)
 546                 && (!drag_part || get_map_context().everything_changed())))
 547                 {
 548                         gui_.rebuild_all();
 549                         get_map_context().set_needs_terrain_rebuild(false);
 550                         gui_.invalidate_all();
 551                 } else {
 552                         for(const map_location& loc : changed_locs) {
 553                                 gui_.rebuild_terrain(loc);
 554                         }
 555                         gui_.invalidate(changed_locs);
 556                 }
 557         } else {
 558                 if(get_map_context().everything_changed()) {
 559                         gui_.invalidate_all();
 560                 } else {
 561                         gui_.invalidate(changed_locs);
 562                 }
 563         }
 564 
 565         if(get_map_context().needs_labels_reset()) {
 566                 get_map_context().reset_starting_position_labels(gui_);
 567         }
 568 
 569         get_map_context().clear_changed_locations();
 570         gui_.recalculate_minimap();
 571 }

 573 void context_manager::resize_map_dialog()
 574 {
 575         int w = get_map().w();
 576         int h = get_map().h();
 577 
 578         gui2::dialogs::editor_resize_map::EXPAND_DIRECTION dir = gui2::dialogs::editor_resize_map::EXPAND_DIRECTION();
 579         bool copy = false;
 580 
 581         if(!gui2::dialogs::editor_resize_map::execute(w, h, dir, copy, gui_.video())) {
 582                 return;
 583         }
 584 
 585         if(w != get_map().w() || h != get_map().h()) {
 586                 t_translation::terrain_code fill = get_selected_bg_terrain();
 587                 if(copy) {
 588                         fill = t_translation::NONE_TERRAIN;
 589                 }
 590 
 591                 int x_offset = get_map().w() - w;
 592                 int y_offset = get_map().h() - h;
 593 
 594                 switch (dir) {
 595                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM_RIGHT:
 596                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM:
 597                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM_LEFT:
 598                                 y_offset = 0;
 599                                 break;
 600                         case gui2::dialogs::editor_resize_map::EXPAND_RIGHT:
 601                         case gui2::dialogs::editor_resize_map::EXPAND_CENTER:
 602                         case gui2::dialogs::editor_resize_map::EXPAND_LEFT:
 603                                 y_offset /= 2;
 604                                 break;
 605                         case gui2::dialogs::editor_resize_map::EXPAND_TOP_RIGHT:
 606                         case gui2::dialogs::editor_resize_map::EXPAND_TOP:
 607                         case gui2::dialogs::editor_resize_map::EXPAND_TOP_LEFT:
 608                                 break;
 609                         default:
 610                                 y_offset = 0;
 611                                 WRN_ED << "Unknown resize expand direction" << std::endl;
 612                                 break;
 613                 }
 614 
 615                 switch (dir) {
 616                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM_RIGHT:
 617                         case gui2::dialogs::editor_resize_map::EXPAND_RIGHT:
 618                         case gui2::dialogs::editor_resize_map::EXPAND_TOP_RIGHT:
 619                                 x_offset = 0;
 620                                 break;
 621                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM:
 622                         case gui2::dialogs::editor_resize_map::EXPAND_CENTER:
 623                         case gui2::dialogs::editor_resize_map::EXPAND_TOP:
 624                                 x_offset /= 2;
 625                                 break;
 626                         case gui2::dialogs::editor_resize_map::EXPAND_BOTTOM_LEFT:
 627                         case gui2::dialogs::editor_resize_map::EXPAND_LEFT:
 628                         case gui2::dialogs::editor_resize_map::EXPAND_TOP_LEFT:
 629                                 break;
 630                         default:
 631                                 x_offset = 0;
 632                                 break;
 633                 }
 634 
 635                 editor_action_resize_map a(w, h, x_offset, y_offset, fill);
 636                 perform_refresh(a);
 637         }
 638 }

 749 void context_manager::save_all_maps(bool auto_save_windows)
 750 {
 751         int current = current_context_index_;
 752         saved_windows_.clear();
 753         for(size_t i = 0; i < map_contexts_.size(); ++i) {
 754                 switch_context(i);
 755                 std::string name = get_map_context().get_filename();
 756                 if(auto_save_windows) {
 757                         if(name.empty() || filesystem::is_directory(name)) {
 758                                 std::ostringstream s;
 759                                 s << default_dir_ << "/" << "window_" << i + 1;
 760                                 name = s.str();
 761                                 get_map_context().set_filename(name);
 762                         }
 763                 }
 764                 saved_windows_.push_back(name);
 765                 save_map();
 766         }
 767 
 768         switch_context(current);
 769 }

 887 void context_manager::load_map(const std::string& filename, bool new_context)
 888 {
 889         if(new_context && check_switch_open_map(filename)) {
 890                 return;
 891         }
 892 
 893         LOG_ED << "Load map: " << filename << (new_context ? " (new)" : " (same)") << "\n";
 894         try {
 895                 {
 896                         context_ptr mc(new map_context(game_config_, filename));
 897                         if(mc->get_filename() != filename) {
 898                                 if(new_context && check_switch_open_map(mc->get_filename())) {
 899                                         return;
 900                                 }
 901                         }
 902 
 903                         if(new_context) {
 904                                 int new_id = add_map_context_of(std::move(mc));
 905                                 switch_context(new_id);
 906                         } else {
 907                                 replace_map_context_with(std::move(mc));
 908                         }
 909                 }
 910 
 911                 if(get_map_context().is_embedded()) {
 912                         const std::string& msg = _("Loaded embedded map data");
 913                         gui2::show_transient_message(gui_.video(), _("Map loaded from scenario"), msg);
 914                 } else {
 915                         if(get_map_context().get_filename() != filename) {
 916                                 if(get_map_context().get_map_data_key().empty()) {
 917                                         ERR_ED << "Internal error, map context filename changed: "
 918                                                 << filename << " -> " << get_map_context().get_filename()
 919                                                 << " with no apparent scenario load\n";
 920                                 } else {
 921                                         utils::string_map symbols;
 922                                         symbols["old"] = filename;
 923                                         const std::string& msg = _("Loaded referenced map file:\n$new");
 924                                         symbols["new"] = get_map_context().get_filename();
 925                                         symbols["map_data"] = get_map_context().get_map_data_key();
 926                                         gui2::show_transient_message(gui_.video(), _("Map loaded from scenario"),
 927                                                 //TODO: msg is already translated does vgettext make sense?
 928                                                 vgettext(msg.c_str(), symbols));
 929                                 }
 930                         }
 931                 }
 932         } catch (editor_map_load_exception& e) {
 933                 gui2::show_transient_message(gui_.video(), _("Error loading map"), e.what());
 934                 return;
 935         }
 936 }

1032 void context_manager::close_current_context()
1033 {
1034         if(!confirm_discard()) return;
1035 
1036         if(map_contexts_.size() == 1) {
1037                 create_default_context();
1038                 map_contexts_.erase(map_contexts_.begin());
1039         } else if(current_context_index_ == static_cast<int>(map_contexts_.size()) - 1) {
1040                 map_contexts_.pop_back();
1041                 current_context_index_--;
1042         } else {
1043                 map_contexts_.erase(map_contexts_.begin() + current_context_index_);
1044         }
1045 
1046         map_context_refresher(*this);
1047         set_window_title();
1048 }

1050 void context_manager::switch_context(const int index, const bool force)
1051 {
1052         if(index < 0 || static_cast<size_t>(index) >= map_contexts_.size()) {
1053                 WRN_ED << "Invalid index in switch map context: " << index << std::endl;
1054                 return;
1055         }
1056 
1057         if(index == current_context_index_ && !force) {
1058                 return;
1059         }
1060 
1061         // Disable the labels of the current context before switching.
1062         // The refresher handles enabling the new ones.
1063         get_labels().enable(false);
1064 
1065         map_context_refresher mcr(*this);
1066         current_context_index_ = index;
1067 
1068         set_window_title();
1069 }
