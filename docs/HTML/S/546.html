 339 const std::string& gui_definition::read(const config& cfg)
 340 {
 341         id = cfg["id"].str();
 342         description = cfg["description"];
 343 
 344         VALIDATE(!id.empty(), missing_mandatory_wml_key("gui", "id"));
 345         VALIDATE(!description.empty(),
 346                          missing_mandatory_wml_key("gui", "description"));
 347 
 348         DBG_GUI_P << "Parsing gui " << id << '\n';
 349 
 350         /***** Control definitions *****/
 351 
 352         for(auto& widget_type : registered_widget_types())
 353         {
 354                 std::vector<styled_widget_definition_ptr> definitions;
 355                 for(const auto& definition :
 356                         cfg.child_range(widget_type.second.key ? widget_type.second.key : widget_type.first + "_definition"))
 357                 {
 358                                 definitions.push_back(widget_type.second.parser(definition));
 359                 }
 360                 load_widget_definitions(widget_type.first, definitions);
 361         }
 362 
 363         /***** Window types *****/
 364         for(const auto& w : cfg.child_range("window"))
 365         {
 366                 std::pair<std::string, builder_window> child;
 367                 child.first = child.second.read(w);
 368                 window_types.insert(child);
 369         }
 370 
 371         if(id == "default") {
 372                 // The default gui needs to define all window types since we're the
 373                 // fallback in case another gui doesn't define the window type.
 374                 for(std::vector<std::string>::const_iterator itor
 375                         = registered_window_types().begin();
 376                         itor != registered_window_types().end();
 377                         ++itor) {
 378 
 379                         const std::string error_msg(
 380                                         "Window not defined in WML: '" + *itor
 381                                         + "'. Perhaps a mismatch between data and source versions."
 382                                           " Try --data-dir <trunk-dir>");
 383                         VALIDATE(window_types.find(*itor) != window_types.end(), error_msg);
 384                 }
 385         }
 386 
 387         /***** settings *****/
 388 
 389         /**
 390          * @todo Regarding sounds:
 391          * Need to evaluate but probably we want the widget definition be able to:
 392          * - Override the default (and clear it). This will allow toggle buttons in
 393          * a
 394          *   listbox to sound like a toggle panel.
 395          * - Override the default and above per instance of the widget, some buttons
 396          *   can give a different sound.
 397          */
 398         const config& settings = cfg.child("settings");
 399 
 400         popup_show_delay_ = settings["popup_show_delay"];
 401         popup_show_time_ = settings["popup_show_time"];
 402         help_show_time_ = settings["help_show_time"];
 403         double_click_time_ = settings["double_click_time"];
 404 
 405         repeat_button_repeat_time_ = settings["repeat_button_repeat_time"];
 406 
 407         VALIDATE(double_click_time_,
 408                          missing_mandatory_wml_key("settings", "double_click_time"));
 409 
 410         sound_button_click_ = settings["sound_button_click"].str();
 411         sound_toggle_button_click_ = settings["sound_toggle_button_click"].str();
 412         sound_toggle_panel_click_ = settings["sound_toggle_panel_click"].str();
 413         sound_slider_adjust_ = settings["sound_slider_adjust"].str();
 414 
 415         has_helptip_message_ = settings["has_helptip_message"];
 416 
 417         VALIDATE(!has_helptip_message_.empty(),
 418                          missing_mandatory_wml_key("[settings]", "has_helptip_message"));
 419 
 420         tips_ = tip_of_the_day::load(cfg);
 421 
 422         return id;
 423 }

 440 void gui_definition::load_widget_definitions(
 441                 const std::string& definition_type,
 442                 const std::vector<styled_widget_definition_ptr>& definitions)
 443 {
 444         for(const auto& def : definitions)
 445         {
 446                 if(control_definition[definition_type].find(def->id) != control_definition[definition_type].end()) {
 447                         ERR_GUI_P << "Skipping duplicate styled_widget definition '" << def->id << "' for '" << definition_type << "'\n";
 448                         continue;
 449                 }
 450 
 451                 control_definition[definition_type].emplace(def->id, def);
 452         }
 453 
 454         // The default GUI needs to ensure each widget has a default definition, but non-default GUIs can just fall back to the default definition in the default GUI.
 455         if(this->id != "default") {
 456                 return;
 457         }
 458 
 459         t_string msg(vgettext("Widget definition '$definition' doesn't contain the definition for '$id'.", {
 460                 {"definition", definition_type},
 461                 {"id", "default"}
 462         }));
 463 
 464         VALIDATE(control_definition[definition_type].find("default")
 465                          != control_definition[definition_type].end(),
 466                          msg);
 467 }

 498 void load_settings()
 499 {
 500         LOG_GUI_G << "Setting: init gui.\n";
 501 
 502         // Init.
 503         window::update_screen_size();
 504 
 505         // Read file.
 506         config cfg;
 507         try
 508         {
 509                 schema_validation::schema_validator validator(
 510                                 filesystem::get_wml_location("gui/schema.cfg"));
 511                 preproc_map preproc(
 512                                 game_config::config_cache::instance().get_preproc_map());
 513                 filesystem::scoped_istream stream = preprocess_file(
 514                                 filesystem::get_wml_location("gui/_main.cfg"), &preproc);
 515 
 516                 read(cfg, *stream, &validator);
 517         }
 518         catch(config::error& e)
 519         {
 520                 ERR_GUI_P << e.what() << '\n';
 521                 ERR_GUI_P << "Setting: could not read file 'data/gui/_main.cfg'."
 522                                   << std::endl;
 523         }
 524         catch(const abstract_validator::error& e)
 525         {
 526                 ERR_GUI_P << "Setting: could not read file 'data/gui/schema.cfg'."
 527                                   << std::endl;
 528                 ERR_GUI_P << e.message;
 529         }
 530         // Parse guis
 531         for(const auto& g : cfg.child_range("gui"))
 532         {
 533                 std::pair<std::string, gui_definition> child;
 534                 child.first = child.second.read(g);
 535                 guis.insert(child);
 536         }
 537 
 538         default_gui = guis.find("default");
 539         VALIDATE(default_gui != guis.end(), _("No default gui defined."));
 540 
 541         std::string current_theme = preferences::gui_theme();
 542         current_gui = current_theme.empty() ? default_gui : guis.find(current_theme);
 543         if(current_gui == guis.end()) {
 544                 ERR_GUI_P << "Missing [gui] definition for '" << current_theme << "'\n";
 545                 current_gui = default_gui;
 546         }
 547         current_gui->second.activate();
 548 }

 592 resolution_definition_ptr get_control(const std::string& control_type,
 593                                                                            const std::string& definition)
 594 {
 595         const gui_definition::styled_widget_definition_map::const_iterator
 596 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 597         control_definition
 598                         = (control_type == "list")
 599                                           ? current_gui->second.control_definition.find("listbox")
 600                                           : current_gui->second.control_definition.find(
 601                                                                 control_type);
 602 #else
 603         control_definition
 604                         = current_gui->second.control_definition.find(control_type);
 605 #endif
 606 
 607         std::map<std::string, styled_widget_definition_ptr>::const_iterator control;
 608 
 609         if(control_definition == current_gui->second.control_definition.end()) {
 610                 goto fallback;
 611         }
 612 
 613         control = control_definition->second.find(definition);
 614 
 615         if(control == control_definition->second.end()) {
 616         fallback:
 617                 bool found_fallback = false;
 618                 if(current_gui != default_gui) {
 619 #ifdef GUI2_EXPERIMENTAL_LISTBOX
 620                         auto default_control_definition = (control_type == "list")
 621                                         ? default_gui->second.control_definition.find("listbox")
 622                                         : default_gui->second.control_definition.find(control_type);
 623 #else
 624                         auto default_control_definition = default_gui->second.control_definition.find(control_type);
 625 #endif
 626 
 627                         VALIDATE(control_definition != current_gui->second.control_definition.end(), formatter() << "Type '" << control_type << "' is unknown.");
 628 
 629                         control = default_control_definition->second.find(definition);
 630                         found_fallback = control != default_control_definition->second.end();
 631                 }
 632                 if(!found_fallback) {
 633                         if(definition != "default") {
 634                                 LOG_GUI_G << "Control: type '" << control_type << "' definition '"
 635                                                   << definition << "' not found, falling back to 'default'.\n";
 636                                 return get_control(control_type, "default");
 637                         }
 638                         FAIL(formatter() << "default definition not found for styled_widget " << control_type);
 639                 }
 640         }
 641 
 642         for(std::vector<resolution_definition_ptr>::const_iterator itor
 643                 = (*control->second).resolutions.begin(),
 644                 end = (*control->second).resolutions.end();
 645                 itor != end;
 646                 ++itor) {
 647 
 648                 if(settings::screen_width <= (**itor).window_width
 649                    && settings::screen_height <= (**itor).window_height) {
 650 
 651                         return *itor;
 652                 } else if(itor == end - 1) {
 653                         return *itor;
 654                 }
 655         }
 656 
 657         FAIL(formatter() << "Control: type '" << control_type << "' definition '" << definition << "' has no resolutions.\n");
 658 }

 660 std::vector<builder_window::window_resolution>::const_iterator
 661 get_window_builder(const std::string& type)
 662 {
 663         window::update_screen_size();
 664 
 665         std::map<std::string, builder_window>::const_iterator window
 666                         = current_gui->second.window_types.find(type);
 667 
 668         if(window == current_gui->second.window_types.end()) {
 669                 if(current_gui != default_gui) {
 670                         window = default_gui->second.window_types.find(type);
 671                         if(window == default_gui->second.window_types.end()) {
 672                                 throw window_builder_invalid_id();
 673                         }
 674                 } else if(window == current_gui->second.window_types.end()) {
 675                         throw window_builder_invalid_id();
 676                 }
 677         }
 678 
 679         for(std::vector<builder_window::window_resolution>::const_iterator itor
 680                 = window->second.resolutions.begin(),
 681                 end = window->second.resolutions.end();
 682                 itor != end;
 683                 ++itor) {
 684 
 685                 if(settings::screen_width <= itor->window_width
 686                    && settings::screen_height <= itor->window_height) {
 687 
 688                         return itor;
 689                 } else if(itor == end - 1) {
 690                         return itor;
 691                 }
 692         }
 693 
 694         FAIL(formatter() << "Window '" << type << "' has no resolutions.\n");
 695 }

 706 bool add_single_widget_definition(const std::string& widget_type, const std::string& definition_id, const config& cfg)
 707 {
 708         auto& gui = default_gui->second;
 709         auto parser = registered_widget_types().find(widget_type);
 710 
 711         if (parser == registered_widget_types().end()) {
 712                 throw std::invalid_argument("widget '" + widget_type  + "' doesn't exist");
 713         }
 714 
 715         if (gui.control_definition[widget_type].find(definition_id) != gui.control_definition[widget_type].end()) {
 716                 return false;
 717         }
 718 
 719         gui.control_definition[widget_type].emplace(definition_id, parser->second.parser(cfg));
 720         return true;
 721 }
 722 
 723 void remove_single_widget_definition(const std::string& widget_type, const std::string& definition_id)
 724 {
 725         auto& gui = default_gui->second;
 726         auto it = gui.control_definition[widget_type].find(definition_id);
 727 
 728         if ( it != gui.control_definition[widget_type].end()) {
 729                 gui.control_definition[widget_type].erase(it);
 730         }
 731 }
