  49         struct key {
  50                 /// Instructions for converting a map_t iterator to an item_ptr.
  51                 static const item_ptr & eval(const map_t::const_iterator & iter)
  52                 { return iter->second; }
  53         };
  54 
  55 public:

  63 
  64         typedef utils::iterator_extend      <value_type, map_t, key, key> iterator;
  65         typedef utils::const_iterator_extend<value_type, map_t, key, key> const_iterator;
