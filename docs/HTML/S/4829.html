 143             # Put a wose in a random location with some constraints:
 144             [do]
 145                 # [do] keep choosing a location [while] the location is no good.
 146                 [while]
 147                     [have_location]
 148                         x=$wose_x
 149                         y=$wose_y
 150                         terrain=!,*^F*  # No good. Not a forest hex.
 151 
 152                         [or]
 153                             terrain=Ke
 154                             [and]
 155                                 x=$wose_x
 156                                 y=$wose_y
 157                                 radius=7  # Too close to the player’s keep.
 158                             [/and]
 159                         [/or]
 160 
 161                         [or]
 162                             [filter]
 163                                 side=1
 164                             [/filter]
 165                             [and]
 166                                 x=$wose_x
 167                                 y=$wose_y
 168                                 radius=2
 169                                 # The wose would be adjacent to one of the
 170                                 # player's units after it moved one hex.
 171                             [/and]
 172                         [/or]
 173 
 174                         [or]
 175                             x=$wose_x
 176                             y=$wose_y
 177                             [filter]
 178                             [/filter]  # Another unit is there already.
 179                         [/or]
 180                     [/have_location]
 181 
 182                     [do]
 183                         # Choose a random location, which gets closer to the
 184                         # top corner after turn 8.
 185                         [if]
 186                             [variable]
 187                                 name=turn_number
 188                                 greater_than=8
 189                             [/variable]
 190                             [then]
 191                                 [set_variable]
 192                                     name=wose_x
 193                                     rand=13..27
 194                                 [/set_variable]
 195                                 [set_variable]
 196                                     name=wose_y
 197                                     rand=1..13
 198                                 [/set_variable]
 199                             [/then]
 200                             [else]
 201                                 [set_variable]
 202                                     name=wose_x
 203                                     rand=1..27
 204                                 [/set_variable]
 205                                 [set_variable]
 206                                     name=wose_y
 207                                     rand=1..21
 208                                 [/set_variable]
 209                             [/else]
 210                         [/if]
 211                     [/do]
 212                 [/while]

 214                 [if]
