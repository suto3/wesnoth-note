 141 template <typename filter>
 142 static void encode(const std::string & input_file, const std::string & output_file)
 143 {
 144         try {
 145                 std::ifstream ifile(input_file.c_str(),
 146                                 std::ios_base::in | std::ios_base::binary);
 147                 ifile.peek(); // We need to touch the stream to set the eof bit
 148                 if(!ifile.good()) {
 149                         std::cerr << "Input file " << input_file << " is not good for reading. Exiting to prevent bzip2 from segfaulting\n";
 150                         safe_exit(1);
 151                 }
 152                 std::ofstream ofile(output_file.c_str(), std::ios_base::out
 153                                 | std::ios_base::binary);
 154                 boost::iostreams::filtering_stream<boost::iostreams::output> stream;
 155                 stream.push(filter());
 156                 stream.push(ofile);
 157                 boost::iostreams::copy(ifile, stream);
 158                 ifile.close();
 159                 safe_exit(remove(input_file.c_str()));
 160         }  catch(filesystem::io_exception& e) {
 161                 std::cerr << "IO error: " << e.what() << "\n";
 162         }
 163 }

 165 template <typename filter>
 166 static void decode(const std::string & input_file, const std::string & output_file)
 167 {
 168         try {
 169                 std::ofstream ofile(output_file.c_str(), std::ios_base::out
 170                                 | std::ios_base::binary);
 171                 std::ifstream ifile(input_file.c_str(),
 172                                 std::ios_base::in | std::ios_base::binary);
 173                 boost::iostreams::filtering_stream<boost::iostreams::input> stream;
 174                 stream.push(filter());
 175                 stream.push(ifile);
 176                 boost::iostreams::copy(stream, ofile);
 177                 ifile.close();
 178                 safe_exit(remove(input_file.c_str()));
 179         }  catch(filesystem::io_exception& e) {
 180                 std::cerr << "IO error: " << e.what() << "\n";
 181         }
 182 }

 204 static void handle_preprocess_command(const commandline_options& cmdline_opts)
 205 {
 206         preproc_map input_macros;
 207 
 208         if( cmdline_opts.preprocess_input_macros ) {
 209                 std::string file = *cmdline_opts.preprocess_input_macros;
 210                 if ( filesystem::file_exists( file ) == false )
 211                 {
 212                         std::cerr << "please specify an existing file. File "<< file <<" doesn't exist.\n";
 213                         return;
 214                 }
 215 
 216                 std::cerr << SDL_GetTicks() << " Reading cached defines from: " << file << "\n";
 217 
 218                 config cfg;
 219 
 220                 try {
 221                         filesystem::scoped_istream stream = filesystem::istream_file( file );
 222                         read( cfg, *stream );
 223                 } catch (config::error & e) {
 224                         std::cerr << "Caught a config error while parsing file '" << file << "':\n" << e.message << std::endl;
 225                 }
 226 
 227                 int read = 0;
 228 
 229                 // use static preproc_define::read_pair(config) to make a object
 230                 for (const config::any_child &value : cfg.all_children_range()) {
 231                         const preproc_map::value_type def = preproc_define::read_pair( value.cfg );
 232                         input_macros[def.first] = def.second;
 233                         ++read;
 234                 }
 235                 std::cerr << SDL_GetTicks() << " Read " << read << " defines.\n";
 236         }
 237 
 238         const std::string resourceToProcess(*cmdline_opts.preprocess_path);
 239         const std::string targetDir(*cmdline_opts.preprocess_target);
 240 
 241         Uint32 startTime = SDL_GetTicks();
 242         // if the users add the SKIP_CORE define we won't preprocess data/core
 243         bool skipCore = false;
 244         bool skipTerrainGFX = false;
 245         // the 'core_defines_map' is the one got from data/core macros
 246         preproc_map defines_map( input_macros );
 247 
 248         if ( cmdline_opts.preprocess_defines ) {
 249 
 250                 // add the specified defines
 251                 for (const std::string &define : *cmdline_opts.preprocess_defines) {
 252                         if (define.empty()){
 253                                 std::cerr << "empty define supplied\n";
 254                                 continue;
 255                         }
 256 
 257                         LOG_PREPROC << "adding define: " << define << '\n';
 258                         defines_map.emplace(define, preproc_define(define));
 259 
 260                         if (define == "SKIP_CORE")
 261                         {
 262                                 std::cerr << "'SKIP_CORE' defined.\n";
 263                                 skipCore = true;
 264                         }
 265                         else if (define == "NO_TERRAIN_GFX")
 266                         {
 267                                 std::cerr << "'NO_TERRAIN_GFX' defined." << std::endl;
 268                                 skipTerrainGFX = true;
 269                         }
 270                 }
 271         }
 272 
 273         // add the WESNOTH_VERSION define
 274         defines_map["WESNOTH_VERSION"] = preproc_define(game_config::wesnoth_version.str());
 275 
 276         std::cerr << "added " << defines_map.size() << " defines.\n";
 277 
 278         // preprocess core macros first if we don't skip the core
 279         if (skipCore == false) {
 280                 std::cerr << "preprocessing common macros from 'data/core' ...\n";
 281 
 282                 // process each folder explicitly to gain speed
 283                 preprocess_resource(game_config::path + "/data/core/macros",&defines_map);
 284                 if (skipTerrainGFX == false)
 285                         preprocess_resource(game_config::path + "/data/core/terrain-graphics",&defines_map);
 286 
 287                 std::cerr << "acquired " << (defines_map.size() - input_macros.size())
 288                         << " 'data/core' defines.\n";
 289         }
 290         else
 291                 std::cerr << "skipped 'data/core'\n";
 292 
 293         // preprocess resource
 294         std::cerr << "preprocessing specified resource: "
 295                 << resourceToProcess << " ...\n";
 296         preprocess_resource(resourceToProcess, &defines_map, true,true, targetDir);
 297         std::cerr << "acquired " << (defines_map.size() - input_macros.size())
 298                 << " total defines.\n";
 299 
 300         if ( cmdline_opts.preprocess_output_macros )
 301         {
 302                 std::string outputFileName = "_MACROS_.cfg";
 303                 if (!cmdline_opts.preprocess_output_macros->empty()) {
 304                         outputFileName = *cmdline_opts.preprocess_output_macros;
 305                 }
 306 
 307                 std::string outputPath = targetDir + "/" + outputFileName;
 308 
 309                 std::cerr << "writing '" << outputPath << "' with "
 310                         << defines_map.size() << " defines.\n";
 311 
 312                 filesystem::scoped_ostream out = filesystem::ostream_file(outputPath);
 313                 if (!out->fail())
 314                 {
 315                         config_writer writer(*out,false);
 316 
 317                         for(preproc_map::iterator itor = defines_map.begin();
 318                                 itor != defines_map.end(); ++itor)
 319                         {
 320                                 (*itor).second.write(writer, (*itor).first);
 321                         }
 322                 }
 323                 else
 324                         std::cerr << "couldn't open the file.\n";
 325         }
 326 
 327         std::cerr << "preprocessing finished. Took "<< SDL_GetTicks() - startTime << " ticks.\n";
 328 }

 330 /** Process commandline-arguments */
 331 static int process_command_args(const commandline_options& cmdline_opts) {
 332 
 333         // Options that don't change behavior based on any others should be checked alphabetically below.
 334 
 335         if(cmdline_opts.userconfig_dir) {
 336                 filesystem::set_user_config_dir(*cmdline_opts.userconfig_dir);
 337         }
 338         if(cmdline_opts.userconfig_path) {
 339                 std::cout << filesystem::get_user_config_dir() << '\n';
 340                 return 0;
 341         }
 342         if(cmdline_opts.userdata_dir) {
 343                 filesystem::set_user_data_dir(*cmdline_opts.userdata_dir);
 344         }
 345         if(cmdline_opts.userdata_path) {
 346                 std::cout << filesystem::get_user_data_dir() << '\n';
 347                 return 0;
 348         }
 349         if(cmdline_opts.data_dir) {
 350                 const std::string datadir = *cmdline_opts.data_dir;
 351 #ifdef _WIN32
 352                 // use c_str to ensure that index 1 points to valid element since c_str() returns null-terminated string
 353                 if(datadir.c_str()[1] == ':') {
 354 #else
 355                 if(datadir[0] == '/') {
 356 #endif
 357                         game_config::path = datadir;
 358                 } else {
 359                         game_config::path = filesystem::get_cwd() + '/' + datadir;
 360                 }
 361                 game_config::path = filesystem::normalize_path(game_config::path, true, true);
 362                 std::cerr << "Overriding data directory with " << game_config::path << std::endl;
 363 
 364                 if(!filesystem::is_directory(game_config::path)) {
 365                         std::cerr << "Could not find directory '" << game_config::path << "'\n";
 366                         throw config::error("directory not found");
 367                 }
 368         // don't update font as we already updating it in game ctor
 369         //font_manager_.update_font_path();
 370         }
 371         if(cmdline_opts.data_path) {
 372                 std::cout << game_config::path << '\n';
 373                 return 0;
 374         }
 375         if(cmdline_opts.debug_lua) {
 376                 game_config::debug_lua = true;
 377         }
 378         if(cmdline_opts.gunzip) {
 379                 const std::string input_file(*cmdline_opts.gunzip);
 380                 if(!filesystem::is_gzip_file(input_file)) {
 381                         std::cerr << "file '" << input_file << "'isn't a .gz file\n";
 382                         return 2;
 383                 }
 384                 const std::string output_file(
 385                         input_file, 0, input_file.length() - 3);
 386                 gzip_decode(input_file, output_file);
 387         }
 388         if(cmdline_opts.bunzip2) {
 389                 const std::string input_file(*cmdline_opts.bunzip2);
 390                 if(!filesystem::is_bzip2_file(input_file)) {
 391                         std::cerr << "file '" << input_file << "'isn't a .bz2 file\n";
 392                         return 2;
 393                 }
 394                 const std::string output_file(
 395                         input_file, 0, input_file.length() - 4);
 396                 bzip2_decode(input_file, output_file);
 397         }
 398         if(cmdline_opts.gzip) {
 399                 const std::string input_file(*cmdline_opts.gzip);
 400                 const std::string output_file(*cmdline_opts.gzip + ".gz");
 401                 gzip_encode(input_file, output_file);
 402         }
 403         if(cmdline_opts.bzip2) {
 404                 const std::string input_file(*cmdline_opts.bzip2);
 405                 const std::string output_file(*cmdline_opts.bzip2 + ".bz2");
 406                 bzip2_encode(input_file, output_file);
 407         }
 408         if(cmdline_opts.help) {
 409                 std::cout << cmdline_opts;
 410                 return 0;
 411         }
 412         if(cmdline_opts.log) {
 413                 for(std::vector<std::pair<int, std::string> >::const_iterator it=cmdline_opts.log->begin(); it!=cmdline_opts.log->end(); ++it)
 414                 {
 415                         const std::string log_domain = it->second;
 416                         const int severity = it->first;
 417                         if (!lg::set_log_domain_severity(log_domain, severity))
 418                         {
 419                                 std::cerr << "unknown log domain: " << log_domain << '\n';
 420                                 return 2;
 421                         }
 422                 }
 423         }
 424         if(cmdline_opts.logdomains) {
 425                 std::cout << lg::list_logdomains(*cmdline_opts.logdomains);
 426                 return 0;
 427         }
 428         if(cmdline_opts.path) {
 429                 std::cout <<  game_config::path << "\n";
 430                 return 0;
 431         }
 432         if(cmdline_opts.log_precise_timestamps) {
 433                 lg::precise_timestamps(true);
 434         }
 435         if(cmdline_opts.rng_seed) {
 436                 srand(*cmdline_opts.rng_seed);
 437         }
 438         if(cmdline_opts.screenshot || cmdline_opts.render_image) {
 439                 SDL_setenv("SDL_VIDEODRIVER", "dummy", 1);
 440         }
 441         if(cmdline_opts.strict_validation) {
 442                 strict_validation_enabled = true;
 443         }
 444         if(cmdline_opts.version) {
 445                 std::cout << "Battle for Wesnoth" << " " << game_config::version << "\n\n";
 446                 std::cout << "Library versions:\n" << game_config::library_versions_report() << '\n';
 447                 std::cout << "Optional features:\n" << game_config::optional_features_report();
 448 
 449                 return 0;
 450         }
 451         if(cmdline_opts.report) {
 452                 std::cout << "\n========= BUILD INFORMATION =========\n\n"
 453                                   << game_config::full_build_report();
 454                 return 0;
 455         }
 456 
 457         // Options changing their behavior dependent on some others should be checked below.
 458 
 459         if ( cmdline_opts.preprocess ) {
 460                 handle_preprocess_command(cmdline_opts);
 461                 return 0;
 462         }
 463 
 464         // Not the most intuitive solution, but I wanted to leave current semantics for now
 465         return -1;
 466 }

 523 #ifdef _MSC_VER
 524 static void check_fpu()
 525 {
 526         uint32_t f_control;
 527         if(_controlfp_s(&f_control, 0, 0) == 0) {
 528                 uint32_t unused;
 529                 uint32_t rounding_mode = f_control & _MCW_RC;
 530                 uint32_t precision_mode = f_control & _MCW_PC;
 531                 if(rounding_mode != _RC_NEAR) {
 532                         std::cerr << "Floating point rounding mode is currently '" <<
 533                                 ((rounding_mode == _RC_CHOP) ? "chop" :
 534                                 (rounding_mode == _RC_UP) ? "up" :
 535                                 (rounding_mode == _RC_DOWN) ? "down" :
 536                                 (rounding_mode == _RC_NEAR) ? "near" :
 537                                 "unknown") << "' setting to 'near'\n";
 538                         if(_controlfp_s(&unused, _RC_NEAR, _MCW_RC)) {
 539                                 std::cerr << "failed to set floating point rounding type to 'near'\n";
 540                         }
 541                 }
 542                 if(precision_mode != _PC_53) {
 543                         std::cerr << "Floating point precision mode is currently '" <<
 544                                 ((precision_mode == _PC_53) ? "double" :
 545                                 (precision_mode == _PC_24) ? "single" :
 546                                 (precision_mode == _PC_64 ) ? "double extended" :
 547                                 "unknown") << "' setting to 'double'\n";
 548                         if(_controlfp_s(&unused, _PC_53, _MCW_PC)) {
 549                                 std::cerr << "failed to set floating point precision type to 'double'\n";
 550                         }
 551                 }
 552         }
 553         else {
 554                 std::cerr << "_controlfp_s failed.\n";
 555         }
 556 }
 557 #else
 558 static void check_fpu()
 559 {
 560         switch (fegetround()) {
 561         case FE_TONEAREST: break;
 562         case FE_DOWNWARD: std::cerr << "Floating point precision mode is currently 'downward'"; goto reset_fpu;
 563         case FE_TOWARDZERO: std::cerr << "Floating point precision mode is currently 'toward-zero'"; goto reset_fpu;
 564         case FE_UPWARD: std::cerr << "Floating point precision mode is currently 'upward'"; goto reset_fpu;
 565         default: std::cerr << "Floating point precision mode is currently 'unknown'"; goto reset_fpu;
 566         reset_fpu:
 567                 std::cerr << "setting to 'nearest'";
 568                 fesetround(FE_TONEAREST);
 569                 break;
 570         }
 571 }

 577 static int do_gameloop(const std::vector<std::string>& args)
 578 {
 579         srand(time(nullptr));
 580 
 581         commandline_options cmdline_opts = commandline_options(args);
 582         game_config::wesnoth_program_dir = filesystem::directory_name(args[0]);
 583         int finished = process_command_args(cmdline_opts);
 584         if(finished != -1) {
 585                 return finished;
 586         }
 587 
 588         const std::unique_ptr<game_launcher> game(
 589                 new game_launcher(cmdline_opts,args[0].c_str()));
 590         const int start_ticks = SDL_GetTicks();
 591 
 592         init_locale();
 593 
 594         bool res;
 595 
 596         // do initialize fonts before reading the game config, to have game
 597         // config error messages displayed. fonts will be re-initialized later
 598         // when the language is read from the game config.
 599         res = font::load_font_config();
 600         if(res == false) {
 601                 std::cerr << "could not initialize fonts\n";
 602                 // The most common symptom of a bogus data dir path -- warn the user.
 603                 warn_early_init_failure();
 604                 return 1;
 605         }
 606 
 607         res = game->init_language();
 608         if(res == false) {
 609                 std::cerr << "could not initialize the language\n";
 610                 return 1;
 611         }
 612 
 613         res = game->init_video();
 614         if(res == false) {
 615                 std::cerr << "could not initialize display\n";
 616                 return 1;
 617         }
 618 
 619         res = image::update_from_preferences();
 620         if(res == false) {
 621                 std::cerr << "could not initialize image preferences\n";
 622                 return 1;
 623         }
 624 
 625         if(preferences::joystick_support_enabled()) {
 626                 res = game->init_joystick();
 627                 if(res == false) {
 628                         std::cerr << "could not initialize joystick\n";
 629                 }
 630         }
 631 
 632         check_fpu();
 633         const cursor::manager cursor_manager;
 634         cursor::set(cursor::WAIT);
 635 
 636 #if (defined(_X11) && !defined(__APPLE__)) || defined(_WIN32)
 637         SDL_EventState(SDL_SYSWMEVENT, SDL_ENABLE);
 638 #endif
 639 
 640         gui2::init();
 641         const gui2::event::manager gui_event_manager;
 642 
 643         game_config_manager config_manager(cmdline_opts, game->video(),
 644             game->jump_to_editor());
 645 
 646         gui2::dialogs::loading_screen::display(game->video(), [&res, &config_manager]() {
 647                 gui2::dialogs::loading_screen::progress("load config");
 648                 res = config_manager.init_game_config(game_config_manager::NO_FORCE_RELOAD);
 649 
 650                 if(res == false) {
 651                         std::cerr << "could not initialize game config\n";
 652                         return;
 653                 }
 654                 gui2::dialogs::loading_screen::progress("init fonts");
 655 
 656                 res = font::load_font_config();
 657                 if(res == false) {
 658                         std::cerr << "could not re-initialize fonts for the current language\n";
 659                         return;
 660         }
 661 
 662                 gui2::dialogs::loading_screen::progress("refresh addons");
 663                 refresh_addon_version_info_cache();
 664         });
 665 
 666         if(res == false) {
 667                 return 1;
 668         }
 669 
 670         LOG_CONFIG << "time elapsed: "<<  (SDL_GetTicks() - start_ticks) << " ms\n";
 671 
 672         plugins_manager plugins_man(new application_lua_kernel);
 673 
 674         plugins_context::Reg const callbacks[] {
 675                 { "play_multiplayer",           std::bind(&game_launcher::play_multiplayer, game.get(), game_launcher::MP_CONNECT)},
 676         };
 677         plugins_context::aReg const accessors[] {
 678                 { "command_line",               std::bind(&commandline_options::to_config, &cmdline_opts)},
 679         };
 680 
 681         plugins_context plugins("titlescreen", callbacks, accessors);
 682 
 683         plugins.set_callback("exit", [](const config& cfg) { safe_exit(cfg["code"].to_int(0)); }, false);
 684 
 685         for (;;)
 686         {
 687                 // reset the TC, since a game can modify it, and it may be used
 688                 // by images in add-ons or campaigns dialogs
 689                 image::set_team_colors();
 690 
 691                 statistics::fresh_stats();
 692 
 693                 if (!game->is_loading()) {
 694                         const config &cfg =
 695                             config_manager.game_config().child("titlescreen_music");
 696                         if (cfg) {
 697                                 sound::play_music_repeatedly(game_config::title_music);
 698                                 for (const config &i : cfg.child_range("music")) {
 699                                         sound::play_music_config(i);
 700                                 }
 701                                 sound::commit_music_changes();
 702                         } else {
 703                                 sound::empty_playlist();
 704                                 sound::stop_music();
 705                         }
 706                 }
 707 
 708                 handle_lua_script_args(&*game,cmdline_opts);
 709 
 710                 plugins.play_slice();
 711                 plugins.play_slice();
 712 
 713                 if(cmdline_opts.unit_test) {
 714                         if(cmdline_opts.timeout) {
 715                                 std::cerr << "The wesnoth built-in timeout feature has been removed.\n" << std::endl;
 716                                 std::cerr << "Please use a platform-specific script which will kill the overtime process instead.\n" << std::endl;
 717                                 std::cerr << "For examples in bash, or in windows cmd, see the forums, or the wesnoth repository." << std::endl;
 718                                 std::cerr << "The bash script is called `run_wml_tests`, the windows script is part of the VC project.\n" << std::endl;
 719                         }
 720                         int worker_result = game->unit_test();
 721                         std::cerr << ((worker_result == 0) ? "PASS TEST " : "FAIL TEST ")
 722                                 << ((worker_result == 3) ? "(INVALID REPLAY)" : "")
 723                                 << ((worker_result == 4) ? "(ERRORED REPLAY)" : "")
 724                                 << ": "<<*cmdline_opts.unit_test << std::endl;
 725                         return worker_result;
 726                 }
 727 
 728                 if(game->play_test() == false) {
 729                         return 0;
 730                 }
 731 
 732                 if(game->play_screenshot_mode() == false) {
 733                         return 0;
 734                 }
 735 
 736                 if(game->play_render_image_mode() == false) {
 737                         return 0;
 738                 }
 739 
 740                 //Start directly a campaign
 741                 if(game->goto_campaign() == false){
 742                         if (game->jump_to_campaign_id().empty())
 743                                 continue; //Go to main menu
 744                         else
 745                                 return 1; //we got an error starting the campaign from command line
 746                 }
 747 
 748                 //Start directly a multiplayer
 749                 //Eventually with a specified server
 750                 if(game->goto_multiplayer() == false){
 751                         continue; //Go to main menu
 752                 }
 753 
 754                 //Start directly a commandline multiplayer game
 755                 if(game->play_multiplayer_commandline() == false) {
 756                         return 0;
 757                 }
 758 
 759                 if (game->goto_editor() == false) {
 760                         return 0;
 761                 }
 762 
 763                 preferences::load_hotkeys();
 764 
 765                 const font::floating_label_context label_manager;
 766 
 767                 cursor::set(cursor::NORMAL);
 768 
 769                 game_launcher::RELOAD_GAME_DATA should_reload = game_launcher::RELOAD_DATA;
 770 
 771                 // If loading a game, skip the titlescreen entirely
 772                 if(game->is_loading()) {
 773                         if(!game->load_game()) {
 774                                 game->clear_loaded_game();
 775                         }
 776 
 777                         game->launch_game(should_reload);
 778                         continue;
 779                 }
 780 
 781                 gui2::dialogs::title_screen dlg(*game);
 782 
 783                 /*
 784                  * Quick explanation of the titlscreen loop:
 785                  *
 786                  * The dialog's redraw_background_ flag is initialized as true in the constructor, so the dialog will always display
 787                  * at least once when this loop is executed. Each time it's opened, the aforementioned flag is set to false, and any
 788                  * selection that results in leaving the dialog simply sets the window's retval and proceeds to the appropriate action.
 789                  *
 790                  * Certain actions (such as window resizing) set the flag to true, which allows the dialog to reopen with any layout
 791                  * changes such as those dictated by window resolution.
 792                  */
 793                 while(dlg.get_retval() == gui2::dialogs::title_screen::REDRAW_BACKGROUND) {
 794                         dlg.show(game->video());
 795                 }
 796 
 797                 switch(dlg.get_retval()) {
 798                 case gui2::dialogs::title_screen::QUIT_GAME:
 799                         LOG_GENERAL << "quitting game...\n";
 800                         return 0;
 801                 case gui2::dialogs::title_screen::MP_CONNECT:
 802                         game_config::debug = game_config::mp_debug;
 803                         if(!game->play_multiplayer(game_launcher::MP_CONNECT)) {
 804                                 continue;
 805                         }
 806                         break;
 807                 case gui2::dialogs::title_screen::MP_HOST:
 808                         game_config::debug = game_config::mp_debug;
 809                         if(!game->play_multiplayer(game_launcher::MP_HOST)) {
 810                                 continue;
 811                         }
 812                         break;
 813                 case gui2::dialogs::title_screen::MP_LOCAL:
 814                         game_config::debug = game_config::mp_debug;
 815                         if(!game->play_multiplayer(game_launcher::MP_LOCAL)) {
 816                                 continue;
 817                         }
 818                         break;
 819                 case gui2::dialogs::title_screen::RELOAD_GAME_DATA:
 820                         gui2::dialogs::loading_screen::display(game->video(), [&config_manager]() {
 821                                 config_manager.reload_changed_game_config();
 822                                 image::flush_cache();
 823                         });
 824                         break;
 825                 case gui2::dialogs::title_screen::MAP_EDITOR:
 826                         game->start_editor();
 827                         break;
 828                 case gui2::dialogs::title_screen::SHOW_ABOUT:
 829                         gui2::dialogs::end_credits::display(game->video());
 830                         break;
 831                 case gui2::dialogs::title_screen::LAUNCH_GAME:
 832                         game->launch_game(should_reload);
 833                         break;
 834                 case gui2::dialogs::title_screen::REDRAW_BACKGROUND:
 835                         break;
 836                 }
 837         }
 838 }

 840 static bool parse_commandline_argument(const char*& next, const char* end, std::string& res)
 841 {
 842         //strip leading whitespace
 843         while(next != end && *next == ' ')
 844                 ++next;
 845         if(next == end)
 846                 return false;
 847 
 848         bool is_excaped = false;
 849 
 850         for(;next != end; ++next)
 851         {
 852                 if(*next == ' ' && !is_excaped) {
 853                         break;
 854                 }
 855                 else if(*next == '"' && !is_excaped) {
 856                         is_excaped = true;
 857                         continue;
 858                 }
 859                 else if(*next == '"' && is_excaped && next + 1 != end && *(next + 1) == '"') {
 860                         res.push_back('"');
 861                         ++next;
 862                         continue;
 863                 }
 864                 else if(*next == '"' && is_excaped ) {
 865                         is_excaped = false;
 866                         continue;
 867                 }
 868                 else {
 869                         res.push_back(*next);
 870                 }
 871         }
 872         return true;
 873 }

 875 static std::vector<std::string> parse_commandline_arguments(std::string input)
 876 {
 877         const char* start = &input[0];
 878         const char* end = start + input.size();
 879         std::string buffer;
 880         std::vector<std::string> res;
 881 
 882         while(parse_commandline_argument(start, end, buffer))
 883         {
 884                 res.emplace_back();
 885                 res.back().swap(buffer);
 886         }
 887         return res;
 888 }

 897 #if defined(_OPENMP) && _MSC_VER >= 1600
 898 static void restart_process(const std::vector<std::string>& commandline)
 899 {
 900         wchar_t process_path[MAX_PATH];
 901         SetLastError(ERROR_SUCCESS);
 902         GetModuleFileNameW(nullptr, process_path, MAX_PATH);
 903         if (GetLastError() != ERROR_SUCCESS)
 904         {
 905                 throw std::runtime_error("Failed to retrieve the process path");
 906         }
 907 
 908         std::wstring commandline_str = unicode_cast<std::wstring>(utils::join(commandline, " "));
 909         // CreateProcessW is allowed to modify the passed command line.
 910         // Therefore we need to copy it.
 911         wchar_t* commandline_c_str = new wchar_t[commandline_str.length() + 1];
 912         commandline_str.copy(commandline_c_str, commandline_str.length());
 913         commandline_c_str[commandline_str.length()] = L'\0';
 914 
 915         STARTUPINFOW startup_info;
 916         ZeroMemory(&startup_info, sizeof(startup_info));
 917         startup_info.cb = sizeof(startup_info);
 918         PROCESS_INFORMATION process_info;
 919         ZeroMemory(&process_info, sizeof(process_info));
 920 
 921         CreateProcessW(process_path, commandline_c_str, nullptr, nullptr,
 922                 false, 0u, nullptr, nullptr, &startup_info, &process_info);
 923 
 924         CloseHandle(process_info.hProcess);
 925         CloseHandle(process_info.hThread);
 926 
 927         std::exit(EXIT_SUCCESS);
 928 }

 944 #ifdef __APPLE__
 945 extern "C" int wesnoth_main(int argc, char** argv);
 946 int wesnoth_main(int argc, char** argv)
 947 #else
 948 int main(int argc, char** argv)
 949 #endif
 950 {
 951 #ifdef _WIN32
 952         (void)argc;
 953         (void)argv;
 954 
 955         //windows argv is ansi encoded by default
 956         std::vector<std::string> args = parse_commandline_arguments(unicode_cast<std::string>(std::wstring(GetCommandLineW())));
 957 
 958         // HACK: we don't parse command line arguments using program_options until
 959         //       the startup banner is printed. We need to get a console up and
 960         //       running before then if requested, so just perform a trivial search
 961         //       here and let program_options ignore the switch later.
 962         for(size_t k = 0; k < args.size(); ++k) {
 963                 if(args[k] == "--wconsole" || args[k] == "--help" || args[k] == "--logdomains" || args[k] == "--path" || args[k] == "--render-image" || args[k] == "--screenshot" || args[k] == "--data-path" || args[k] == "--userdata-path" || args[k] == "--userconfig-path" || args[k] == "--version") {
 964                         lg::enable_native_console_output();
 965                         break;
 966                 }
 967         }
 968 
 969         lg::early_log_file_setup();
 970 #else
 971         std::vector<std::string> args;
 972         for(int i = 0; i < argc; ++i)
 973         {
 974                 args.push_back(std::string(argv[i]));
 975         }
 976 #endif
 977         assert(!args.empty());
 978 
 979 #ifdef _OPENMP
 980         // Wesnoth is a special case for OMP
 981         // OMP wait strategy is to have threads busy-loop for 100ms
 982         // if there is nothing to do, they then go to sleep.
 983         // this avoids the scheduler putting the thread to sleep when work
 984         // is about to be available
 985         //
 986         // However Wesnoth has a lot of very small jobs that need to be done
 987         // at each redraw => 50fps every 2ms.
 988         // All the threads are thus busy-waiting all the time, hogging the CPU
 989         // To avoid that problem, we need to set the OMP_WAIT_POLICY env var
 990         // but that var is read by OMP at library loading time (before main)
 991         // thus the relaunching of ourselves after setting the variable.
 992 #if !defined(_WIN32) && !defined(__APPLE__)
 993         if (!getenv("OMP_WAIT_POLICY")) {
 994                 setenv("OMP_WAIT_POLICY", "PASSIVE", 1);
 995                 execv(argv[0], argv);
 996         }
 997 #elif _MSC_VER >= 1600
 998         if (!getenv("OMP_WAIT_POLICY")) {
 999                 _putenv_s("OMP_WAIT_POLICY", "PASSIVE");
1000                 args[0] = utils::quote(args[0]);
1001                 restart_process(args);
1002         }
1003 #endif
1004 #endif //_OPENMP
1005 
1006         if(SDL_Init(SDL_INIT_TIMER) < 0) {
1007                 fprintf(stderr, "Couldn't initialize SDL: %s\n", SDL_GetError());
1008                 return(1);
1009         }
1010 
1011 #ifndef _WIN32
1012         struct sigaction terminate_handler;
1013         terminate_handler.sa_handler = wesnoth_terminate_handler;
1014         terminate_handler.sa_flags = 0;
1015         sigemptyset(&terminate_handler.sa_mask);
1016         sigaction(SIGTERM, &terminate_handler, nullptr);
1017         sigaction(SIGINT, &terminate_handler, nullptr);
1018 #endif
1019 
1020         //declare this here so that it will always be at the front of the event queue.
1021         events::event_context global_context;
1022 
1023         SDL_StartTextInput();
1024 
1025         try {
1026                 std::cerr << "Battle for Wesnoth v" << game_config::revision << '\n';
1027                 const time_t t = time(nullptr);
1028                 std::cerr << "Started on " << ctime(&t) << "\n";
1029 
1030                 const std::string& exe_dir = filesystem::get_exe_dir();
1031                 if(!exe_dir.empty()) {
1032                         // Try to autodetect the location of the game data dir. Note that
1033                         // the root of the source tree currently doubles as the data dir.
1034                         std::string auto_dir;
1035 
1036                         // scons leaves the resulting binaries at the root of the source
1037                         // tree by default.
1038                         if(filesystem::file_exists(exe_dir + "/data/_main.cfg")) {
1039                                 auto_dir = exe_dir;
1040                         }
1041                         // cmake encourages creating a subdir at the root of the source
1042                         // tree for the build, and the resulting binaries are found in it.
1043                         else if(filesystem::file_exists(exe_dir + "/../data/_main.cfg")) {
1044                                 auto_dir = filesystem::normalize_path(exe_dir + "/..");
1045                         }
1046                         // In Windows debug builds, the EXE is placed away from the game data dir
1047                         // (in projectfiles\VCx\Debug), but the working directory is set to the
1048                         // game data dir. Thus, check if the working dir is the game data dir.
1049                         else if(filesystem::file_exists(filesystem::get_cwd() + "/data/_main.cfg")) {
1050                                 auto_dir = filesystem::get_cwd();
1051                         }
1052 
1053                         if(!auto_dir.empty()) {
1054                                 std::cerr << "Automatically found a possible data directory at "
1055                                                   << auto_dir << '\n';
1056                                 game_config::path = auto_dir;
1057                         }
1058                 }
1059 
1060                 const int res = do_gameloop(args);
1061                 safe_exit(res);
1062         } catch(boost::program_options::error& e) {
1063                 std::cerr << "Error in command line: " << e.what() << '\n';
1064                 error_exit(1);
1065         } catch(CVideo::error & e) {
1066                 std::cerr << "Could not initialize video.\n\n" << e.what() << "\n\nExiting.\n";
1067                 error_exit(1);
1068         } catch(font::error & e) {
1069                 std::cerr << "Could not initialize fonts.\n\n" << e.what() << "\n\nExiting.\n";
1070                 error_exit(1);
1071         } catch(config::error& e) {
1072                 std::cerr << e.message << "\n";
1073                 error_exit(1);
1074         } catch(gui::button::error&) {
1075                 std::cerr << "Could not create button: Image could not be found\n";
1076                 error_exit(1);
1077         } catch(CVideo::quit&) {
1078                 //just means the game should quit
1079         } catch(return_to_play_side_exception&) {
1080                 std::cerr << "caught return_to_play_side_exception, please report this bug (quitting)\n";
1081         } catch(quit_game_exception&) {
1082                 std::cerr << "caught quit_game_exception (quitting)\n";
1083         } catch(wml_exception& e) {
1084                 std::cerr << "WML exception:\nUser message: "
1085                         << e.user_message << "\nDev message: " << e.dev_message << '\n';
1086                 error_exit(1);
1087         } catch(wfl::formula_error& e) {
1088                 std::cerr << e.what()
1089                         << "\n\nGame will be aborted.\n";
1090                 error_exit(1);
1091         } catch(const sdl::exception& e) {
1092                 std::cerr << e.what();
1093                 error_exit(1);
1094         } catch(game::error &) {
1095                 // A message has already been displayed.
1096                 error_exit(1);
1097         } catch(std::bad_alloc&) {
1098                 std::cerr << "Ran out of memory. Aborted.\n";
1099                 error_exit(ENOMEM);
1100 #if !defined(NO_CATCH_AT_GAME_END)
1101         } catch(const std::exception & e) {
1102                 // Try to catch unexpected exceptions.
1103                 std::cerr << "Caught general '" << typeid(e).name() << "' exception:\n" << e.what() << std::endl;
1104                 error_exit(1);
1105         } catch(std::string & e) {
1106                 std::cerr << "Caught a string thrown as an exception:\n" << e << std::endl;
1107                 error_exit(1);
1108         } catch(const char * e) {
1109                 std::cerr << "Caught a string thrown as an exception:\n" << e << std::endl;
1110                 error_exit(1);
1111         } catch(...) {
1112                 // Ensure that even when we terminate with `throw 42`, the exception
1113                 // is caught and all destructors are actually called. (Apparently,
1114                 // some compilers will simply terminate without calling destructors if
1115                 // the exception isn't caught.)
1116                 std::cerr << "Caught unspecified general exception. Terminating." << std::endl;
1117                 error_exit(1);
1118 #endif
1119         }
1120 
1121         return 0;
1122 } // end main
