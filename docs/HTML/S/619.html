 134 void multimenu_button::signal_handler_left_button_click(const event::ui_event event, bool& handled)
 135 {
 136         assert(get_window());
 137         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 138 
 139         sound::play_UI_sound(settings::sound_button_click);
 140 
 141         // If a button has a retval do the default handling.
 142         dialogs::drop_down_menu droplist(this->get_rectangle(), this->values_, -1, this->get_use_markup(), true,
 143                 std::bind(&multimenu_button::toggle_state_changed, this));
 144 
 145         droplist_ = &droplist;
 146         droplist.show(get_window()->video());
 147         droplist_ = nullptr;
 148 
 149         if(retval_ != 0) {
 150                 if(window* window = get_window()) {
 151                         window->set_retval(retval_);
 152                         return;
 153                 }
 154         }
 155 
 156         /* In order to allow toggle button states to be specified by verious dialogs in the values config, we write the state
 157          * bools to the values_ config here, but only if a checkbox= key was already provided. The value of the checkbox= key
 158          * is handled by the drop_down_menu widget.
 159          *
 160          * Passing the dynamic_bitset directly to the drop_down_menu ctor would mean bool values would need to be passed to this
 161          * class independently of the values config by dialogs that use this widget. However, the bool states are also saved
 162          * in a dynamic_bitset class member which can be fetched for other uses if necessary.
 163          */
 164         update_config_from_toggle_states();
 165 
 166         handled = true;
 167 }

 169 void multimenu_button::update_label()
 170 {
 171         std::vector<t_string> selected;
 172         for(size_t i = 0; i < toggle_states_.size() && i < values_.size(); i++) {
 173                 if(!toggle_states_[i]) {
 174                         continue;
 175                 }
 176 
 177                 selected.push_back(values_[i]["label"]);
 178         }
 179 
 180         if(selected.size() == 0) {
 181                 set_label(_("multimenu^None Selected"));
 182         } else if(selected.size() == 1) {
 183                 set_label(selected[0]);
 184         } else if(selected.size() == 2) {
 185                 const string_map variables {{"first", selected[0]}, {"second", selected[1]}, {"excess", "1"}};
 186                 if(max_shown_ == 1) {
 187                         set_label(VNGETTEXT("multimenu^$first and 1 other","multimenu^$first and $excess others", 1, variables));
 188                 } else {
 189                         set_label(VGETTEXT("multimenu^$first and $second", variables));
 190                 }
 191         } else if(selected.size() == values_.size()) {
 192                 set_label(_("multimenu^All Selected"));
 193         } else {
 194                 const int excess = selected.size() - max_shown_;
 195                 if(max_shown_ > 0 && excess > 0) {
 196                         selected.resize(max_shown_);
 197                 }
 198 
 199                 std::string first = selected[0];
 200                 for(size_t i = 1; i < selected.size() - 1; i++) {
 201                         const string_map variables {{"first", first}, {"second", selected[i]}};
 202                         first = VGETTEXT("multimenu^$first, $second", variables);
 203                 }
 204 
 205                 if(max_shown_ > 0 && excess > 0) {
 206                         const string_map variables {{"first", first}, {"excess", std::to_string(excess + 1)}};
 207                         set_label(VNGETTEXT("multimenu^$first and 1 other","$first and $excess others", excess + 1, variables));
 208                 } else {
 209                         const string_map variables {{"first", first}, {"second", selected.back()}};
 210                         set_label(VGETTEXT("multimenu^$first and $second", variables));
 211                 }
 212         }
 213 }

 215 void multimenu_button::update_config_from_toggle_states()
 216 {
 217         for(unsigned i = 0; i < values_.size(); i++) {
 218                 ::config& c = values_[i];
 219 
 220                 c["checkbox"] = toggle_states_[i];
 221         }
 222 }

 231 void multimenu_button::toggle_state_changed()
 232 {
 233         assert(droplist_ != nullptr);
 234 
 235         toggle_states_ = droplist_->get_toggle_states();
 236         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 237         update_label();
 238 
 239         if(callback_toggle_state_change_ != nullptr) {
 240                 callback_toggle_state_change_(toggle_states_);
 241         }
 242 }

 244 void multimenu_button::select_option(const unsigned option, const bool selected)
 245 {
 246         assert(option < values_.size());
 247 
 248         if(option < toggle_states_.size()) {
 249                 toggle_states_.resize(option + 1);
 250         }
 251         toggle_states_[option] = selected;
 252         update_config_from_toggle_states();
 253         update_label();
 254 }
 255 
 256 void multimenu_button::select_options(boost::dynamic_bitset<> states)
 257 {
 258         assert(states.size() == values_.size());
 259         toggle_states_ = states;
 260         update_config_from_toggle_states();
 261         update_label();
 262 }

 264 void multimenu_button::set_values(const std::vector<::config>& values)
 265 {
 266         set_is_dirty(true);
 267 
 268         values_ = values;
 269         toggle_states_.resize(values_.size(), false);
 270         toggle_states_.reset();
 271 
 272         set_label(_("multimenu^None Selected"));
 273 }

 378 widget* builder_multimenu_button::build() const
 379 {
 380         multimenu_button* widget = new multimenu_button();
 381 
 382         init_control(widget);
 383 
 384         widget->set_retval(get_retval(retval_id_, retval_, id));
 385         widget->set_max_shown(max_shown_);
 386         if(!options_.empty()) {
 387                 widget->set_values(options_);
 388         }

 391                   << "' with definition '" << definition << "'.\n";
