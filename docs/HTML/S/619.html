 133 void multimenu_button::signal_handler_left_button_click(const event::ui_event event, bool& handled)
 134 {
 135         assert(get_window());
 136         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 137 
 138         sound::play_UI_sound(settings::sound_button_click);
 139 
 140         // If a button has a retval do the default handling.
 141         dialogs::drop_down_menu droplist(this->get_rectangle(), this->values_, -1, this->get_use_markup(), true,
 142                 std::bind(&multimenu_button::toggle_state_changed, this));
 143 
 144         droplist_ = &droplist;
 145         droplist.show(get_window()->video());
 146         droplist_ = nullptr;
 147 
 148         if(retval_ != 0) {
 149                 if(window* window = get_window()) {
 150                         window->set_retval(retval_);
 151                         return;
 152                 }
 153         }
 154 
 155         /* In order to allow toggle button states to be specified by verious dialogs in the values config, we write the state
 156          * bools to the values_ config here, but only if a checkbox= key was already provided. The value of the checkbox= key
 157          * is handled by the drop_down_menu widget.
 158          *
 159          * Passing the dynamic_bitset directly to the drop_down_menu ctor would mean bool values would need to be passed to this
 160          * class independently of the values config by dialogs that use this widget. However, the bool states are also saved
 161          * in a dynamic_bitset class member which can be fetched for other uses if necessary.
 162          */
 163         update_config_from_toggle_states();
 164 
 165         handled = true;
 166 }

 168 void multimenu_button::update_label()
 169 {
 170         std::vector<t_string> selected;
 171         for(size_t i = 0; i < toggle_states_.size() && i < values_.size(); i++) {
 172                 if(!toggle_states_[i]) {
 173                         continue;
 174                 }
 175 
 176                 selected.push_back(values_[i]["label"]);
 177         }
 178 
 179         if(selected.size() == values_.size()) {
 180                 set_label(_("multimenu^All Selected"));
 181         } else {
 182                 if(selected.size() > static_cast<size_t>(max_shown_)) {
 183                         const int excess = selected.size() - max_shown_;
 184                         selected.resize(max_shown_ + 1);
 185                         selected.back() = VNGETTEXT("multimenu excess^1 other", "$excess others", excess, {{"excess", std::to_string(excess)}});
 186                 }
 187                 set_label(utils::format_conjunct_list(_("multimenu^None Selected"), selected));
 188         }
 189 }

 191 void multimenu_button::update_config_from_toggle_states()
 192 {
 193         for(unsigned i = 0; i < values_.size(); i++) {
 194                 ::config& c = values_[i];
 195 
 196                 c["checkbox"] = toggle_states_[i];
 197         }
 198 }

 207 void multimenu_button::toggle_state_changed()
 208 {
 209         assert(droplist_ != nullptr);
 210 
 211         toggle_states_ = droplist_->get_toggle_states();
 212         fire(event::NOTIFY_MODIFIED, *this, nullptr);
 213         update_label();
 214 
 215         if(callback_toggle_state_change_ != nullptr) {
 216                 callback_toggle_state_change_(toggle_states_);
 217         }
 218 }

 220 void multimenu_button::select_option(const unsigned option, const bool selected)
 221 {
 222         assert(option < values_.size());
 223 
 224         if(option < toggle_states_.size()) {
 225                 toggle_states_.resize(option + 1);
 226         }
 227         toggle_states_[option] = selected;
 228         update_config_from_toggle_states();
 229         update_label();
 230 }
 231 
 232 void multimenu_button::select_options(boost::dynamic_bitset<> states)
 233 {
 234         assert(states.size() == values_.size());
 235         toggle_states_ = states;
 236         update_config_from_toggle_states();
 237         update_label();
 238 }

 240 void multimenu_button::set_values(const std::vector<::config>& values)
 241 {
 242         set_is_dirty(true);
 243 
 244         values_ = values;
 245         toggle_states_.resize(values_.size(), false);
 246         toggle_states_.reset();
 247 
 248         set_label(_("multimenu^None Selected"));
 249 }

 342 builder_multimenu_button::builder_multimenu_button(const config& cfg)
 343         : builder_styled_widget(cfg)
 344         , retval_id_(cfg["return_value_id"])
 345         , retval_(cfg["return_value"])
 346         , max_shown_(cfg["maximum_shown"])
 347         , options_()
 348 {
 349         for(const auto& option : cfg.child_range("option")) {
 350                 options_.push_back(option);
 351         }
 352 }
 353 
 354 widget* builder_multimenu_button::build() const
 355 {
 356         multimenu_button* widget = new multimenu_button();
 357 
 358         init_control(widget);
 359 
 360         widget->set_retval(get_retval(retval_id_, retval_, id));
 361         widget->set_max_shown(max_shown_);
 362         if(!options_.empty()) {
 363                 widget->set_values(options_);
 364         }
 365 
 366         DBG_GUI_G << "Window builder: placed multimenu_button '" << id
 367                   << "' with definition '" << definition << "'.\n";
 368 
 369         return widget;
 370 }
