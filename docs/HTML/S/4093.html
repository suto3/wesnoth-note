 114     def send_packet(self, packet):
 115         """
 116         Send binary data to the server.
 117         """
 118         # Compress the packet before we send it
 119         fio = io.BytesIO()
 120         z = gzip.GzipFile(mode = "wb", fileobj = fio)
 121         z.write(packet)
 122         z.close()
 123         zdata = fio.getvalue()
 124 
 125         zpacket = struct.pack("!I", len(zdata)) + zdata
 126         self.sock.sendall(zpacket)

 128     def read_packet(self):
 129         """
 130         Read binary data from the server.
 131         """
 132         packet = b""
 133         while len(packet) < 4 and not self.canceled:
 134             r = self.sock.recv(4 - len(packet))
 135             if not r:
 136                 return None
 137             packet += r
 138         if self.canceled:
 139             return None
 140 
 141         self.length = l = struct.unpack("!I", packet)[0]
 142 
 143         if not self.quiet:
 144             sys.stderr.write("Receiving %d bytes.\n" % self.length)
 145 
 146         packet = b""
 147         while len(packet) < l and not self.canceled:
 148             r = self.sock.recv(l - len(packet))
 149             if not r:
 150                 return None
 151             packet += r
 152             self.counter = len(packet)
 153         if self.canceled:
 154             return None
 155 
 156         if not self.quiet:
 157             sys.stderr.write("Received %d bytes.\n" % len(packet))
 158 
 159         if packet.startswith(b"\x1F\x8B"):
 160             if self.verbose:
 161                 sys.stderr.write("GZIP compression found...\n")
 162             bio = io.BytesIO(packet)
 163             z = gzip.GzipFile("rb", fileobj = bio)
 164             unzip = z.read()
 165             z.close()
 166             packet = unzip
 167 
 168         elif packet.startswith(b'\x78\x9C' ):
 169             if self.verbose:
 170                 sys.stderr.write("ZLIB compression found...\n")
 171             packet = zlib.decompres( packet )
 172 
 173         return packet

 182     def unescape(self, data):
 183         data2 = bytearray()
 184         # 01 is used as escape character
 185         pos = 0
 186         while True:
 187             i = data.find(b"\x01", pos)
 188             if i < 0:
 189                 break
 190             data2 += data[pos:i]
 191             data2 += bytes([data[i + 1] - 1])
 192             pos = i + 2
 193         data2 += data[pos:]
 194         return data2

 287         def put_file(name, f):
 288             for ig in ign:
 289                 if ig and ig[-1] != "/" and fnmatch.fnmatch(name, ig):
 290                     print(("Ignored file", name))
 291                     return None
 292             fileNode = append_tag(None, "file")
 293 
 294             # Order in which we apply escape sequences matters.
 295             contents = f.read()
 296             contents = contents.replace(b"\x01", b"\x01\x02" )
 297             contents = contents.replace(b"\x00", b"\x01\x01")
 298             contents = contents.replace(b"\x0d", b"\x01\x0e")
 299             contents = contents.replace(b"\xfe", b"\x01\xff")
 300 
 301             append_attributes(fileNode, name = name)
 302             append_attributes(fileNode, contents = contents)

 304             return fileNode

 306         def put_dir(name, path):
 307             for ig in ign:

 312             dataNode = append_tag("dir")

 399     def unpackdir(self, data, path, i = 0, verbose = False):
 400         """
 401         Call this to unpack a campaign contained in a WML object
 402         to the filesystem. The data parameter is the WML object,
 403         path is the path under which it will be placed.
 404         """
 405 
 406         try:
 407             os.mkdir(path)
 408         except OSError:
 409             pass
 410         for f in data.get_all(tag = "file"):
 411             name = f.get_text_val("name", "?")
 412             contents = f.get_binary("contents")
 413             if not contents:
 414                 contents = b""
 415                 if not self.quiet:
 416                     sys.stderr.write("File %s is empty.\n" % name)
 417                     sys.stderr.write(f.debug() + "\n")
 418             if verbose:
 419                 sys.stderr.write(i * " " + name + " (" +
 420                       str(len(contents)) + ")\n")
 421             save = open( os.path.join(path, name), "wb")
 422 
 423             # We MUST un-escape our data
 424             # Order we apply escape sequences matter here
 425             contents = self.unescape(contents)
 426             save.write(contents)
 427             save.close()
 428 
 429         for dir in data.get_all(tag = "dir"):
 430             name = dir.get_text_val("name", "?")
 431             shutil.rmtree(os.path.join(path, name), True)
 432             os.mkdir(os.path.join(path, name))

 434                 sys.stderr.write(i * " " + name + "\n")
