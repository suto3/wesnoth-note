   5 This parser uses the --preprocess option of wesnoth so a working
   6 wesnoth executable must be available at runtime if the WML to parse
   7 contains preprocessing directives.
   8 
   9 Pure WML can be parsed as is.
  10 
  11 For example:
  12 
  13     wml = ""
  14     [unit]
  15         id=elve
  16         name=Elve
  17         [abilities]
  18             [damage]
  19                 id=Ensnare
  20             [/dama  ge]
  21         [/abilities]
  22     [/unit]

 148 class TagNode:
 149     """
 150     A WML tag. For example the "unit" in this example:
 151         [unit]
 152             id=Elfish Archer
 153         [/unit]
 154     """
 155     def __init__(self, name, location=None):
 156         self.name = name
 157         self.location = location
 158         # List of child elements, which are either of type TagNode or
 159         # AttributeNode.
 160         self.data = []
 161 
 162         self.speedy_tags = {}
 163 
 164     def wml(self) -> bytes:
 165         """
 166         Returns a (binary) WML representation of the entire node.
 167         All attribute values are enclosed in quotes and quotes are
 168         escaped (as double quotes). Note that no other escaping is
 169         performed (see the BinaryWML specification for additional
 170         escaping you may require).
 171         """
 172         s = b"[" + self.name + b"]\n"
 173         for sub in self.data:
 174             s += sub.wml() + b"\n"
 175         s += b"[/" + self.name + b"]\n"
 176         return s
 177 
 178     def debug(self):
 179         s = "[%s]\n" % self.name.decode("utf8")
 180         for sub in self.data:
 181             for subline in sub.debug().splitlines():
 182                 s += "    %s\n" % subline
 183         s += "[/%s]\n" % self.name.decode("utf8")
 184         return s

 186     def get_all(self, **kw):
 187         """
 188         This gets all child tags or child attributes of the tag.
 189         For example:
 190 
 191         [unit]
 192             name=A
 193             name=B
 194             [attack]
 195             [/attack]
 196             [attack]
 197             [/attack]
 198         [/unit]
 199 
 200         unit.get_all(att = "name")
 201         will return two nodes for "name=A" and "name=B"
 202 
 203         unit.get_all(tag = "attack")
 204         will return two nodes for the two [attack] tags.
 205 
 206         unit.get_all()
 207         will return 4 nodes for all 4 sub-elements.
 208 
 209         unit.get_all(att = "")
 210         Will return the two attribute nodes.
 211 
 212         unit.get_all(tag = "")
 213         Will return the two tag nodes.
 214 
 215         If no elements are found an empty list is returned.
 216         """
 217         if len(kw) == 1 and "tag" in kw and kw["tag"]:
 218             return self.speedy_tags.get(kw["tag"].encode("utf8"), [])
 219 
 220         r = []
 221         for sub in self.data:
 222             ok = True
 223             for k, v in list(kw.items()):
 224                 v = v.encode("utf8")
 225                 if k == "tag":
 226                     if not isinstance(sub, TagNode): ok = False
 227                     elif v != b"" and sub.name != v: ok = False
 228                 elif k == "att":
 229                     if not isinstance(sub, AttributeNode): ok = False
 230                     elif v != b"" and sub.name != v: ok = False
 231             if ok:
 232                 r.append(sub)
 233         return r

 235     def get_text_val(self, name, default=None, translation=None, val=-1):
 236         """
 237         Returns the value of the specified attribute. If the attribute
 238         is given multiple times, the value number val is returned (default
 239         behaviour being to return the last value). If the
 240         attribute is not found, the default parameter is returned.
 241 
 242         If a translation is specified, it should be a function which
 243         when passed a unicode string and text-domain returns a
 244         translation of the unicode string. The easiest way is to pass
 245         it to gettext.translation if you have the binary message
 246         catalogues loaded.
 247         """
 248         x = self.get_all(att = name)
 249         if not x: return default
 250         return x[val].get_text(translation)

 350         """
 351         This is called by the parse functions to preprocess the
 352         input from a normal WML .cfg file into a preprocessed
 353         .plain file.
 354         """
 355         if self.keep_temp_dir:
 356             output = self.keep_temp_dir

 358             output = tempfile.mkdtemp(prefix="wmlparser_")
 359             tempdirs_to_clean.append(output)

 361         self.temp_dir = output
 362         commandline = [self.wesnoth_exe]
 363         if self.data_dir:
 364             commandline += ["--data-dir", self.data_dir]
 365         if self.config_dir:

 367         commandline += ["--preprocess", self.path, output]

 387     def parse_line_without_commands_loop(self, line : str) -> str:
 388         """
 389         Once the .plain commands are handled WML lines are passed to
 390         this.
 391         """
 392         if not line: return
 393 
 394         if line.strip():
 395             self.skip_newlines_after_plus = False
 396 
 397         if self.in_tag:
 398             self.handle_tag(line)
 399             return
 400 
 401         if self.in_arrows:
 402             arrows = line.find(b'>>')
 403             if arrows >= 0:
 404                 self.in_arrows = False
 405                 self.temp_string += line[:arrows]
 406                 self.temp_string_node = StringNode(self.temp_string)
 407                 self.temp_string = b""
 408                 self.temp_key_nodes[self.commas].value.append(
 409                     self.temp_string_node)
 410                 self.in_arrows = False
 411                 return line[arrows + 2:]
 412             else:
 413                 self.temp_string += line
 414             return
 415 
 416         quote = line.find(b'"')
 417 
 418         if not self.in_string:
 419             arrows = line.find(b'<<')
 420             if arrows >= 0 and (quote < 0 or quote > arrows):
 421                 self.parse_line_without_commands(line[:arrows])
 422                 self.in_arrows = True
 423                 return line[arrows + 2:]
 424 
 425         if quote >= 0:
 426             if self.in_string:
 427                 # double quote
 428                 if quote < len(line) - 1 and line[quote + 1] == b'"'[0]:
 429                     self.temp_string += line[:quote + 1]
 430                     return line[quote + 2:]
 431                 self.temp_string += line[:quote]
 432                 self.temp_string_node = StringNode(self.temp_string)
 433                 if self.translatable:
 434                     self.temp_string_node.textdomain = self.textdomain
 435                     self.translatable = False
 436                 self.temp_string = b""
 437                 if not self.temp_key_nodes:
 438                     raise WMLError(self, "Unexpected string value.")
 439 
 440                 self.temp_key_nodes[self.commas].value.append(
 441                     self.temp_string_node)
 442 
 443                 self.in_string = False
 444                 return line[quote + 1:]
 445             else:
 446                 self.parse_outside_strings(line[:quote])
 447                 self.in_string = True
 448                 return line[quote + 1:]
 449         else:
 450             if self.in_string:
 451                 self.temp_string += line
 452             else:
 453                 self.parse_outside_strings(line)

 455     def parse_line_without_commands(self, line):
 456         while True:
 457             line = self.parse_line_without_commands_loop(line)
 458             if not line:
 459                 break

 461     def parse_outside_strings(self, line):
 462         """
 463         Parse a WML fragment outside of strings.
 464         """
 465         if not line: return
 466         if not self.temp_key_nodes:
 467             line = line.lstrip()
 468             if not line: return
 469 
 470             if line.startswith(b"#textdomain "):
 471                 self.textdomain = line[12:].strip().decode("utf8")
 472                 return
 473 
 474             # Is it a tag?
 475             if line.startswith(b"["):
 476                 self.handle_tag(line)
 477             # No tag, must be an attribute.
 478             else:
 479                 self.handle_attribute(line)
 480         else:
 481             for i, segment in enumerate(line.split(b"+")):
 482                 segment = segment.lstrip(b" ")
 483 
 484                 if i > 0:
 485                     # If the last segment is empty (there was a plus sign
 486                     # at the end) we need to skip newlines.
 487                     self.skip_newlines_after_plus = not segment.strip()
 488 
 489                 if not segment: continue
 490 
 491                 if segment.rstrip(b" ") == b"_":
 492                     self.translatable = True
 493                     segment = segment[1:].lstrip(b" ")
 494                     if not segment: continue
 495                 self.handle_value(segment)

 498     def handle_tag(self, line):
 499         end = line.find(b"]")
 500         if end < 0:
 501             if line.endswith(b"\n"):
 502                 raise WMLError(self, "Expected closing bracket.")
 503             self.in_tag += line
 504             return
 505         tag = (self.in_tag + line[:end])[1:]
 506         self.in_tag = b""
 507         if tag.startswith(b"/"):
 508             self.parent_node = self.parent_node[:-1]
 509         else:
 510             node = TagNode(tag, location=(self.line_in_file, self.chunk_start))
 511             if self.parent_node:
 512                 self.parent_node[-1].append(node)
 513             self.parent_node.append(node)
 514         self.parse_outside_strings(line[end + 1:])

 516     def handle_attribute(self, line):
 517         assign = line.find(b"=")
 518         remainder = None
 519         if assign >= 0:
 520             remainder = line[assign + 1:]
 521             line = line[:assign]
 522 
 523         self.commas = 0
 524         self.temp_key_nodes = []
 525         for att in line.split(b","):
 526             att = att.strip()
 527             node = AttributeNode(att, location=(self.line_in_file, self.chunk_start))
 528             self.temp_key_nodes.append(node)
 529             if self.parent_node:
 530                 self.parent_node[-1].append(node)
 531 
 532         if remainder:
 533             self.parse_outside_strings(remainder)

 535     def handle_value(self, segment):
 536         def add_text(segment):
 537             segment = segment.rstrip()
 538             if not segment: return
 539             n = len(self.temp_key_nodes)
 540             maxsplit = n - self.commas - 1
 541             if maxsplit < 0: maxsplit = 0
 542             for subsegment in segment.split(b",", maxsplit):
 543                 self.temp_string += subsegment.strip()
 544                 self.temp_string_node = StringNode(self.temp_string)
 545                 self.temp_string = b""
 546                 self.temp_key_nodes[self.commas].value.append(
 547                     self.temp_string_node)
 548                 if self.commas < n - 1:
 549                     self.commas += 1
 550 
 551         # Finish assignment on newline, except if there is a
 552         # plus sign before the newline.
 553         add_text(segment)
 554         if segment.endswith(b"\n") and not self.skip_newlines_after_plus:
 555             self.temp_key_nodes = []

 557     def parse(self) -> RootNode:
 558         """
 559         Parse preprocessed WML into a tree of tags and attributes.
 560         """
 561 
 562         # parsing state
 563         self.temp_string = b""
 564         self.temp_string_node = None
 565         self.commas = 0
 566         self.temp_key_nodes = []
 567         self.in_string = False
 568         self.in_arrows = False
 569         self.textdomain = "wesnoth"
 570         self.translatable = False
 571         self.root = RootNode()
 572         self.parent_node = [self.root]
 573         self.skip_newlines_after_plus = False
 574         self.in_tag = b""
 575 
 576         command_marker_byte = bytes([254])
 577 
 578         input = self.preprocessed
 579         if not input: input = self.path
 580 
 581         for rawline in open(input, "rb"):
 582             compos = rawline.find(command_marker_byte)
 583             self.parser_line += 1
 584             # Everything from chr(254) to newline is the command.
 585             if compos != 0:
 586                 self.line_in_file += 1
 587             if compos >= 0:
 588                 self.parse_line_without_commands(rawline[:compos])
 589                 self.handle_command(rawline[compos + 1:-1])
 590             else:
 591                 self.parse_line_without_commands(rawline)
 592 
 593         if self.keep_temp_dir is None and self.temp_dir:
 594             if self.verbose:
 595                 print(("removing " + self.temp_dir))
 596             shutil.rmtree(self.temp_dir, ignore_errors=True)
 597 
 598         return self.root

 600     def handle_command(self, com):
 601         if com.startswith(b"line "):
 602             self.last_wml_line = com[5:]
 603             _ = self.last_wml_line.split(b" ")
 604             self.chunk_start = [(_[i+1], int(_[i])) for i in range(0, len(_), 2)]
 605             self.line_in_file = self.chunk_start[0][1]
 606         elif com.startswith(b"textdomain "):
 607             self.textdomain = com[11:].decode("utf8")
 608         else:
 609             raise WMLError(self, "Unknown parser command: " + com)
