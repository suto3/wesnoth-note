 242 bool play_controller::hotkey_handler::execute_command(const hotkey::hotkey_command& cmd, int index, bool press)
 243 {
 244         hotkey::HOTKEY_COMMAND command = cmd.id;
 245         if(index >= 0) {
 246                 unsigned i = static_cast<unsigned>(index);
 247                 if(i < savenames_.size() && !savenames_[i].empty()) {
 248                         // Load the game by throwing load_game_exception
 249                         load_autosave(savenames_[i]);
 250 
 251                 } else if ( i < wml_commands_.size()  &&  wml_commands_[i] ) {
 252                         wml_commands_[i]->fire_event(mouse_handler_.get_last_hex(), gamestate().gamedata_);
 253                         return true;
 254                 }
 255         }
 256         int prefixlen = wml_menu_hotkey_prefix.length();
 257         if(command == hotkey::HOTKEY_WML && cmd.command.compare(0, prefixlen, wml_menu_hotkey_prefix) == 0)
 258         {
 259                 std::string name = cmd.command.substr(prefixlen);
 260                 const map_location& hex = mouse_handler_.get_last_hex();
 261 
 262                 gamestate().get_wml_menu_items().fire_item(name, hex, gamestate().gamedata_, gamestate(), gamestate().board_.units_);
 263                 /// @todo Shouldn't the function return at this point?
 264         }
 265         return command_executor::execute_command(cmd, index, press);
 266 }

 322         case hotkey::HOTKEY_SCROLL_UP:
 323         case hotkey::HOTKEY_SCROLL_DOWN:
 324         case hotkey::HOTKEY_SCROLL_LEFT:
 325         case hotkey::HOTKEY_SCROLL_RIGHT:
 326                 return true;
 327 
 328         // Commands that have some preconditions:
 329         case hotkey::HOTKEY_SAVE_GAME:
 330                 return !events::commands_disabled;
 331 
 332         case hotkey::HOTKEY_SHOW_ENEMY_MOVES:
 333         case hotkey::HOTKEY_BEST_ENEMY_MOVES:
 334                 return !linger() && play_controller_.enemies_visible();
 335 
 336         case hotkey::HOTKEY_LOAD_GAME:
 337                 return !play_controller_.is_networked_mp(); // Can only load games if not in a network game
 338 
 339         case hotkey::HOTKEY_CHAT_LOG:
 340                 return true;
 341 
 342         case hotkey::HOTKEY_REDO:
 343                 return play_controller_.can_redo();
 344         case hotkey::HOTKEY_UNDO:
 345                 return play_controller_.can_undo();
 346 
 347         case hotkey::HOTKEY_UNIT_DESCRIPTION:
 348                 return menu_handler_.current_unit().valid();
 349 
 350         case hotkey::HOTKEY_TERRAIN_DESCRIPTION:
 351                 return mouse_handler_.get_last_hex().valid();
 352 
 353         case hotkey::HOTKEY_RENAME_UNIT:
 354                 return !events::commands_disabled &&
 355                         menu_handler_.current_unit().valid() &&
 356                         !(menu_handler_.current_unit()->unrenamable()) &&
 357                         menu_handler_.current_unit()->side() == gui()->viewing_side() &&
 358                         play_controller_.get_teams_const()[menu_handler_.current_unit()->side() - 1].is_local_human();
 359 
 360         default:
 361                 return false;
 362         }
 363 }

 365 template<typename T>
 366 static void trim_items(std::vector<T>& newitems)
 367 {
 368         if(newitems.size() > 5) {
 369                 std::vector<T> subitems;
 370                 subitems.push_back(std::move(newitems[0]));
 371                 subitems.push_back(std::move(newitems[1]));
 372                 subitems.push_back(std::move(newitems[newitems.size() / 3]));
 373                 subitems.push_back(std::move(newitems[newitems.size() * 2 / 3]));
 374                 subitems.push_back(std::move(newitems.back()));
 375                 newitems = subitems;
 376         }
 377 }

 379 void play_controller::hotkey_handler::expand_autosaves(std::vector<config>& items, int i)
 380 {
 381         const compression::format comp_format = preferences::save_compression_format();
 382 
 383         savenames_.resize(i);
 384 
 385         auto pos = items.erase(items.begin() + i);
 386         std::vector<config> newitems;
 387         std::vector<std::string> newsaves;
 388 
 389         const std::string& start_name = saved_game_.classification().label;
 390 
 391         for(unsigned int turn = play_controller_.turn(); turn != 0; turn--) {
 392                 const std::string name = formatter() << start_name << "-" << _("Auto-Save") << turn;
 393 
 394                 if(savegame::save_game_exists(name, comp_format)) {
 395                         newsaves.emplace_back(name + compression::format_extension(comp_format));
 396                         newitems.emplace_back(config_of("label", _("Back to Turn ") + std::to_string(turn)));
 397                 }
 398         }
 399 
 400         if(savegame::save_game_exists(start_name, comp_format)) {
 401                 newsaves.emplace_back(start_name + compression::format_extension(comp_format));
 402                 newitems.emplace_back(config_of("label", _("Back to Start")));
 403         }
 404 
 405         // Make sure list doesn't get too long: keep top two, midpoint and bottom.
 406         trim_items(newitems);
 407         trim_items(newsaves);
 408 
 409         items.insert(pos, newitems.begin(), newitems.end());
 410         savenames_.insert(savenames_.end(), newsaves.begin(), newsaves.end());
 411 }

 413 void play_controller::hotkey_handler::expand_wml_commands(std::vector<config>& items, int i)
 414 {
 415         // Pad the commands with null pointers (keeps the indices of items and wml_commands_ synced).
 416         wml_commands_.resize(i);
 417 
 418         auto pos = items.erase(items.begin() + i);
 419         std::vector<config> newitems;
 420 
 421         gamestate().get_wml_menu_items().get_items(mouse_handler_.get_last_hex(), wml_commands_, newitems,
 422                 gamestate(), gamestate().gamedata_, gamestate().board_.units_);
 423 
 424         // Replace this placeholder entry with available menu items.
 425         items.insert(pos, newitems.begin(), newitems.end());
 426 }

 428 void play_controller::hotkey_handler::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool context_menu, display& disp)
 429 {
 430         if(context_menu) {
 431                 last_context_menu_x_ = xloc;
 432                 last_context_menu_y_ = yloc;
 433         }
 434 
 435         std::vector<config> items;
 436         for(const auto& item : items_arg) {
 437                 const std::string& id = item["id"];
 438                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
 439 
 440                 if(id == "wml" || (can_execute_command(command) && (!context_menu || in_context_menu(command.id)))) {
 441                         items.emplace_back(config_of("id", id));
 442                 }
 443         }
 444 
 445         // Add special non-hotkey items to the menu and remember their indices
 446         // Iterate in reverse to avoid also iterating over the new inserted items
 447         savenames_.clear();
 448         wml_commands_.clear();
 449 
 450         for(int i = items.size() - 1; i >= 0; i--) {
 451                 if(items[i]["id"] == "AUTOSAVES") {
 452                         expand_autosaves(items, i);
 453                 } else if(items[i]["id"] == "wml") {
 454                         expand_wml_commands(items, i);
 455                 }
 456         }
 457 
 458         if(items.empty()) {
 459                 return;
 460         }
 461 
 462         command_executor::show_menu(items, xloc, yloc, context_menu, disp);
 463 }

 465 bool play_controller::hotkey_handler::in_context_menu(hotkey::HOTKEY_COMMAND command) const
 466 {
 467         switch(command) {
 468         // Only display these if the mouse is over a castle or keep tile
 469         case hotkey::HOTKEY_RECRUIT:
 470         case hotkey::HOTKEY_REPEAT_RECRUIT:
 471         case hotkey::HOTKEY_RECALL: {
 472                 // last_hex_ is set by mouse_events::mouse_motion
 473                 const map_location & last_hex = mouse_handler_.get_last_hex();
 474                 const int viewing_side = gui()->viewing_side();
 475 
 476                 // A quick check to save us having to create the future map and
 477                 // possibly loop through all units.
 478                 if ( !play_controller_.get_map_const().is_keep(last_hex)  &&
 479                      !play_controller_.get_map_const().is_castle(last_hex) )
 480                         return false;
 481 
 482                 wb::future_map future; /* lasts until method returns. */
 483 
 484                 return gamestate().side_can_recruit_on(viewing_side, last_hex);
 485         }
 486         default:
 487                 return true;
 488         }
 489 }

 502 hotkey::ACTION_STATE play_controller::hotkey_handler::get_action_state(hotkey::HOTKEY_COMMAND command, int /*index*/) const
 503 {
 504         switch(command) {
 505 
 506         case hotkey::HOTKEY_MINIMAP_DRAW_VILLAGES:
 507                 return (preferences::minimap_draw_villages()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 508         case hotkey::HOTKEY_MINIMAP_CODING_UNIT:
 509                 return (preferences::minimap_movement_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 510         case hotkey::HOTKEY_MINIMAP_CODING_TERRAIN:
 511                 return (preferences::minimap_terrain_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 512         case hotkey::HOTKEY_MINIMAP_DRAW_UNITS:
 513                 return (preferences::minimap_draw_units()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 514         case hotkey::HOTKEY_MINIMAP_DRAW_TERRAIN:
 515                 return (preferences::minimap_draw_terrain()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 516         case hotkey::HOTKEY_ZOOM_DEFAULT:
 517                 return (gui()->get_zoom_factor() == 1.0) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 518         case hotkey::HOTKEY_DELAY_SHROUD:
 519                 return viewing_team().auto_shroud_updates() ? hotkey::ACTION_OFF : hotkey::ACTION_ON;
 520         default:
 521                 return hotkey::ACTION_STATELESS;
 522         }
 523 }
