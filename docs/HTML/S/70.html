  93 map_context::map_context(const config& game_config, const std::string& filename)
  94         : filename_(filename)
  95         , map_data_key_()
  96         , embedded_(false)
  97         , pure_map_(false)
  98         , map_(game_config)
  99         , undo_stack_()
 100         , redo_stack_()
 101         , actions_since_save_(0)
 102         , starting_position_label_locs_()
 103         , needs_reload_(false)
 104         , needs_terrain_rebuild_(false)
 105         , needs_labels_reset_(false)
 106         , changed_locations_()
 107         , everything_changed_(false)
 108         , scenario_id_()
 109         , scenario_name_()
 110         , scenario_description_()
 111         , xp_mod_(70)
 112         , victory_defeated_(true)
 113         , random_time_(false)
 114         , active_area_(-1)
 115         , labels_(nullptr)
 116         , units_()
 117         , teams_()
 118         , tod_manager_(new tod_manager(game_config.find_child("editor_times", "id", "default")))
 119         , mp_settings_()
 120         , game_classification_()
 121         , music_tracks_()
 122 {
 123         /*
 124          * Overview of situations possibly found in the file:
 125          *
 126          * 0. Not a scenario or map file.
 127          *    0.1 File not found
 128          *    0.2 Map file empty
 129          *    0.3 No valid data
 130          * 1. It's a file containing only pure map data.
 131          *    * embedded_ = false
 132          *    * pure_map_ = true
 133          * 2. A scenario embedding the map
 134          *    * embedded_ = true
 135          *    * pure_map_ = true
 136          *    The data/scenario-test.cfg for example.
 137          *    The map is written back to the file.
 138          * 3. The map file is referenced by map_data={MACRO_ARGUEMENT}.
 139          *    * embedded_ = false
 140          *    * pure_map_ = true
 141          * 4. The file contains an editor generated scenario file.
 142          *    * embedded_ = false
 143          *    * pure_map_ = false
 144          */
 145 
 146         log_scope2(log_editor, "Loading file " + filename);
 147 
 148         // 0.1 File not found
 149         if (!filesystem::file_exists(filename) || filesystem::is_directory(filename))
 150                 throw editor_map_load_exception(filename, _("File not found"));
 151 
 152         std::string file_string = filesystem::read_file(filename);
 153 
 154         // 0.2 Map file empty
 155         if (file_string.empty()) {
 156                 std::string message = _("Empty file");
 157                 throw editor_map_load_exception(filename, message);
 158         }
 159 
 160         // 1.0 Pure map data
 161         boost::regex rexpression_map_data(R"""(map_data\s*=\s*"(.+?)")""");
 162         boost::smatch matched_map_data;
 163         if (!boost::regex_search(file_string, matched_map_data, rexpression_map_data,
 164                         boost::regex_constants::match_not_dot_null)) {
 165                 map_ = editor_map::from_string(game_config, file_string); //throws on error
 166                 pure_map_ = true;
 167 
 168                 add_to_recent_files();
 169                 return;
 170         }
 171 
 172         // 2.0 Embedded map
 173         const std::string& map_data = matched_map_data[1];
 174         boost::regex rexpression_macro(R"""(\{(.+?)\})""");
 175         boost::smatch matched_macro;
 176         if (!boost::regex_search(map_data, matched_macro, rexpression_macro)) {
 177                 // We have a map_data string but no macro ---> embedded or scenario
 178 
 179                 boost::regex rexpression_scenario(R"""(\[(scenario|test|multiplayer|tutorial)\])""");
 180                 if (!boost::regex_search(file_string, rexpression_scenario)) {
 181                         LOG_ED << "Loading generated scenario file" << std::endl;
 182                         // 4.0 editor generated scenario
 183                         try {
 184                                 load_scenario(game_config);
 185                         } catch (config::error & e) {
 186                                 throw editor_map_load_exception("load_scenario", e.message); //we already caught and rethrew this exception in load_scenario
 187                         }
 188                 } else {
 189                         LOG_ED << "Loading embedded map file" << std::endl;
 190                         embedded_ = true;
 191                         pure_map_ = true;
 192                         map_ = editor_map::from_string(game_config, map_data);
 193                 }
 194 
 195                 add_to_recent_files();
 196                 return;
 197         }
 198 
 199         // 3.0 Macro referenced pure map
 200         const std::string& macro_argument = matched_macro[1];
 201         LOG_ED << "Map looks like a scenario, trying {" << macro_argument << "}" << std::endl;
 202         std::string new_filename = filesystem::get_wml_location(macro_argument,
 203                         filesystem::directory_name(filesystem::get_short_wml_path(filename_)));
 204         if (new_filename.empty()) {
 205                 std::string message = _("The map file looks like a scenario, "
 206                                 "but the map_data value does not point to an existing file")
 207                                                                                                 + std::string("\n") + macro_argument;
 208                 throw editor_map_load_exception(filename, message);
 209         }
 210         LOG_ED << "New filename is: " << new_filename << std::endl;
 211         filename_ = new_filename;
 212         file_string = filesystem::read_file(filename_);
 213         map_ = editor_map::from_string(game_config, file_string);
 214         pure_map_ = true;
 215 
 216         add_to_recent_files();
 217 }

 294 void map_context::load_scenario(const config& game_config)
 295 {
 296         config scenario;
 297 
 298         try {
 299                 read(scenario, *(preprocess_file(filename_)));
 300         } catch (config::error & e) {
 301                 LOG_ED << "Caught a config error while parsing file: '" << filename_ << "'\n" << e.message << std::endl;
 302                 throw e;
 303         }
 304 
 305         scenario_id_   = scenario["id"].str();
 306         scenario_name_ = scenario["name"].str();
 307         scenario_description_ = scenario["description"].str();
 308 
 309         xp_mod_ = scenario["experience_modifier"].to_int();
 310 
 311         victory_defeated_ = scenario["victory_when_enemies_defeated"].to_bool(true);
 312         random_time_ = scenario["random_start_time"].to_bool(false);
 313 
 314         map_ = editor_map::from_string(game_config, scenario["map_data"]); //throws on error
 315 
 316         labels_.read(scenario);
 317 
 318         tod_manager_.reset(new tod_manager(scenario));
 319         for(const config &time_area : scenario.child_range("time_area")) {
 320                 tod_manager_->add_time_area(map_,time_area);
 321         }
 322 
 323         for(const config& item : scenario.child_range("item")) {
 324                 const map_location loc(item);
 325                 overlays_.emplace(loc, overlay(item));
 326         }
 327 
 328         for(const config& music : scenario.child_range("music")) {
 329                 music_tracks_.emplace(music["name"], sound::music_track(music));
 330         }
 331 
 332         int i = 1;
 333         for(config &side : scenario.child_range("side"))
 334         {
 335                 teams_.emplace_back();
 336 
 337                 side["side"] = i;
 338                 teams_.back().build(side, map_);
 339 
 340                 for(config &a_unit : side.child_range("unit")) {
 341                         map_location loc(a_unit, nullptr);
 342                         a_unit["side"] = i;
 343                         units_.add(loc, unit(a_unit, true) );
 344                 }
 345                 i++;
 346         }
 347 }

 369 void map_context::draw_terrain_actual(const t_translation::terrain_code & terrain,
 370         const map_location& loc, bool one_layer_only)
 371 {
 372         if (!map_.on_board_with_border(loc)) {
 373                 //requests for painting off the map are ignored in set_terrain anyway,
 374                 //but ideally we should not have any
 375                 LOG_ED << "Attempted to draw terrain off the map (" << loc << ")\n";
 376                 return;
 377         }
 378         t_translation::terrain_code old_terrain = map_.get_terrain(loc);
 379         if (terrain != old_terrain) {
 380                 if (terrain.base == t_translation::NO_LAYER) {
 381                         map_.set_terrain(loc, terrain, terrain_type_data::OVERLAY);
 382                 } else if (one_layer_only) {
 383                         map_.set_terrain(loc, terrain, terrain_type_data::BASE);
 384                 } else {
 385                         map_.set_terrain(loc, terrain);
 386                 }
 387                 add_changed_location(loc);
 388         }
 389 }

 449 config map_context::to_config()
 450 {
 451         config scenario;
 452 
 453         scenario["id"] = scenario_id_;
 454         scenario["name"] = t_string(scenario_name_);
 455         scenario["description"] = scenario_description_;
 456 
 457         scenario["experience_modifier"] = xp_mod_;
 458         scenario["victory_when_enemies_defeated"] = victory_defeated_;
 459         scenario["random_starting_time"] = random_time_;
 460 
 461         scenario.append(tod_manager_->to_config());
 462         scenario.remove_attribute("turn_at");
 463 
 464         scenario["map_data"] = map_.write();
 465 
 466         labels_.write(scenario);
 467 
 468         overlay_map::const_iterator it;
 469         for (it = overlays_.begin(); it != overlays_.end(); ++it) {
 470 
 471                         config& item = scenario.add_child("item");
 472                         it->first.write(item);
 473                         item["image"] = it->second.image;
 474                         item["id"] = it->second.id;
 475                         item["halo"] = it->second.halo;
 476                         item["visible_in_fog"] = it->second.visible_in_fog;
 477                         item["name"] = it->second.name;
 478                         item["team_name"] = it->second.team_name;
 479         }
 480 
 481         for(const music_map::value_type& track : music_tracks_) {
 482                 track.second.write(scenario, true);
 483         }
 484 
 485         for(std::vector<team>::const_iterator t = teams_.begin(); t != teams_.end(); ++t) {
 486                 int side_num = t - teams_.begin() + 1;
 487 
 488                 config& side = scenario.add_child("side");
 489 
 490                 side["side"] = side_num;
 491                 side["hidden"] = t->hidden();
 492 
 493                 side["controller"] = t->controller();
 494                 side["no_leader"] = t->no_leader();
 495 
 496                 side["team_name"] = t->team_name();
 497                 side["user_team_name"] = t->user_team_name();
 498 
 499                 // TODO
 500                 // side["allow_player"] = "yes";
 501 
 502                 side["fog"] = t->uses_fog();
 503                 side["shroud"] = t->uses_shroud();
 504                 side["share_vision"] = t->share_vision();
 505 
 506                 side["gold"] = t->gold();
 507                 side["income"] = t->base_income();
 508 
 509                 for(const map_location& village : t->villages()) {
 510                         village.write(side.add_child("village"));
 511                 }
 512 
 513                 //current visible units
 514                 for(unit_map::const_iterator i = units_.begin(); i != units_.end(); ++i) {
 515                         if(i->side() == side_num) {
 516                                 config& u = side.add_child("unit");
 517                                 i->get_location().write(u);
 518                                 u["type"] = i->type_id();
 519                                 u["canrecruit"] = i->can_recruit();
 520                                 u["unrenamable"] = i->unrenamable();
 521                                 if(!boost::regex_match(i->id(), boost::regex(".*-[0-9]+")))
 522                                 {
 523                                         u["id"] = i->id();
 524                                 }
 525                                 u["name"] = i->name();
 526                                 u["extra_recruit"] = utils::join(i->recruits());
 527                                 u["facing"] = map_location::write_direction(i->facing());
 528                         }
 529                 }
 530         }
 531 
 532         return scenario;
 533 }

 535 bool map_context::save_scenario()
 536 {
 537         assert(!is_embedded());
 538 
 539         if (scenario_id_.empty())
 540                 scenario_id_ = filesystem::base_name(filename_);
 541         if (scenario_name_.empty())
 542                 scenario_name_ = scenario_id_;
 543 
 544         try {
 545                 std::stringstream wml_stream;
 546                 {
 547                         config_writer out(wml_stream, false);
 548                         out.write(to_config());
 549                 }
 550                 if (!wml_stream.str().empty())
 551                         filesystem::write_file(get_filename(), wml_stream.str());
 552                 clear_modified();
 553         } catch (filesystem::io_exception& e) {
 554                 utils::string_map symbols;
 555                 symbols["msg"] = e.what();
 556                 const std::string msg = vgettext("Could not save the scenario: $msg", symbols);
 557                 throw editor_map_save_exception(msg);
 558         }
 559         // TODO the return value of this method does not need to be boolean.
 560         // We either return true or there is an exception thrown.
 561         return true;
 562 }

 564 bool map_context::save_map()
 565 {
 566         std::string map_data = map_.write();
 567 
 568         try {
 569                 if (!is_embedded()) {
 570                         filesystem::write_file(get_filename(), map_data);
 571                 } else {
 572                         std::string map_string = filesystem::read_file(get_filename());
 573                         boost::regex rexpression_map_data(R"""((.*map_data\s*=\s*")(.+?)(".*))""");
 574                         boost::smatch matched_map_data;
 575                         if (boost::regex_search(map_string, matched_map_data, rexpression_map_data,
 576                                         boost::regex_constants::match_not_dot_null)) {
 577                                 std::stringstream ss;
 578                                 ss << matched_map_data[1];
 579                                 ss << map_data;
 580                                 ss << matched_map_data[3];
 581                                 filesystem::write_file(get_filename(), ss.str());
 582                         } else {
 583                                 throw editor_map_save_exception(_("Could not save into scenario"));
 584                         }
 585                 }
 586                 add_to_recent_files();
 587                 clear_modified();
 588         } catch (filesystem::io_exception& e) {
 589                 utils::string_map symbols;
 590                 symbols["msg"] = e.what();
 591                 const std::string msg = vgettext("Could not save the map: $msg", symbols);
 592                 throw editor_map_save_exception(msg);
 593         }
 594         //TODO the return value of this method does not need to be boolean.
 595         //We either return true or there is an exception thrown.
 596         return true;
 597 }

 609 void map_context::perform_action(const editor_action& action)
 610 {
 611         LOG_ED << "Performing action " << action.get_id() << ": " << action.get_name()
 612                 << ", actions count is " << action.get_instance_count() << std::endl;
 613         editor_action* undo = action.perform(*this);
 614         if (actions_since_save_ < 0) {
 615                 //set to a value that will make it impossible to get to zero, as at this point
 616                 //it is no longer possible to get back the original map state using undo/redo
 617                 actions_since_save_ = 1 + undo_stack_.size();
 618         }
 619         actions_since_save_++;
 620         undo_stack_.push_back(undo);
 621         trim_stack(undo_stack_);
 622         clear_stack(redo_stack_);
 623 }

 625 void map_context::perform_partial_action(const editor_action& action)
 626 {
 627         LOG_ED << "Performing (partial) action " << action.get_id() << ": " << action.get_name()
 628                 << ", actions count is " << action.get_instance_count() << std::endl;
 629         if (!can_undo()) {
 630                 throw editor_logic_exception("Empty undo stack in perform_partial_action()");
 631         }
 632         editor_action_chain* undo_chain = dynamic_cast<editor_action_chain*>(last_undo_action());
 633         if (undo_chain == nullptr) {
 634                 throw editor_logic_exception("Last undo action not a chain in perform_partial_action()");
 635         }
 636         editor_action* undo = action.perform(*this);
 637         //actions_since_save_ += action.action_count();
 638         undo_chain->prepend_action(undo);
 639         clear_stack(redo_stack_);
 640 }

 687 void map_context::undo()
 688 {
 689         LOG_ED << "undo() beg, undo stack is " << undo_stack_.size() << ", redo stack " << redo_stack_.size() << std::endl;
 690         if (can_undo()) {
 691                 perform_action_between_stacks(undo_stack_, redo_stack_);
 692                 actions_since_save_--;
 693         } else {
 694                 WRN_ED << "undo() called with an empty undo stack" << std::endl;
 695         }
 696         LOG_ED << "undo() end, undo stack is " << undo_stack_.size() << ", redo stack " << redo_stack_.size() << std::endl;
 697 }

 699 void map_context::redo()
 700 {
 701         LOG_ED << "redo() beg, undo stack is " << undo_stack_.size() << ", redo stack " << redo_stack_.size() << std::endl;
 702         if (can_redo()) {
 703                 perform_action_between_stacks(redo_stack_, undo_stack_);
 704                 actions_since_save_++;
 705         } else {
 706                 WRN_ED << "redo() called with an empty redo stack" << std::endl;
 707         }
 708         LOG_ED << "redo() end, undo stack is " << undo_stack_.size() << ", redo stack " << redo_stack_.size() << std::endl;
 709 }

 711 void map_context::partial_undo()
 712 {
 713         //callers should check for these conditions
 714         if (!can_undo()) {
 715                 throw editor_logic_exception("Empty undo stack in partial_undo()");
 716         }
 717         editor_action_chain* undo_chain = dynamic_cast<editor_action_chain*>(last_undo_action());
 718         if (undo_chain == nullptr) {
 719                 throw editor_logic_exception("Last undo action not a chain in partial undo");
 720         }
 721         //a partial undo performs the first action form the current action's action_chain that would be normally performed
 722         //i.e. the *first* one.
 723         const std::unique_ptr<editor_action> first_action_in_chain(undo_chain->pop_first_action());
 724         if (undo_chain->empty()) {
 725                 actions_since_save_--;
 726                 delete undo_chain;
 727                 undo_stack_.pop_back();
 728         }
 729         redo_stack_.push_back(first_action_in_chain->perform(*this));
 730         //actions_since_save_ -= last_redo_action()->action_count();
 731 }

 739 void map_context::trim_stack(action_stack& stack)
 740 {
 741         if (stack.size() > max_action_stack_size_) {
 742                 delete stack.front();
 743                 stack.pop_front();
 744         }
 745 }

 747 void map_context::clear_stack(action_stack& stack)
 748 {
 749         for (editor_action* a : stack) {
 750                 delete a;
 751         }
 752         stack.clear();
 753 }

 755 void map_context::perform_action_between_stacks(action_stack& from, action_stack& to)
 756 {
 757         assert(!from.empty());
 758         const std::unique_ptr<editor_action> action(from.back());
 759         from.pop_back();
 760         editor_action* reverse_action = action->perform(*this);
 761         to.push_back(reverse_action);
 762         trim_stack(to);
 763 }
