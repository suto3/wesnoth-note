 170 /**
 171  * Returns whether or not *this is applicable given the context.
 172  * Assumes game variables x1, y1, and unit have been set.
 173  * @param[in]  hex  The hex where the menu will appear.
 174  */
 175 bool wml_menu_item::can_show(const map_location & hex, const game_data & data, filter_context & filter_con) const
 176 {
 177         // Failing the [show_if] tag means no show.
 178         if ( !show_if_.empty() && !conditional_passed(show_if_) )
 179                 return false;
 180 
 181         // Failing the [fiter_location] tag means no show.
 182         if ( !filter_location_.empty() &&
 183              !terrain_filter(filter_location_, &filter_con)(hex) )
 184                 return false;
 185 
 186         // Failing to have a required selection means no show.
 187         if ( needs_select_ && !data.last_selected.valid() )
 188                 return false;
 189 
 190         // Passed all tests.
 191         return true;
 192 }

 194 /**
 195  * Causes the event associated with this item to fire.
 196  * Also records the event.
 197  * This includes recording the previous select event, if applicable.
 198  * The undo stack will be cleared if the event mutated the gamestate.
 199  *
 200  * @param[in] event_hex    The location of the event (where the menu was opened).
 201  * @param[in] last_select  The location of the most recent "select" event.
 202  */
 203 #define STR(X) #X
 204 void wml_menu_item::fire_event(const map_location & event_hex, const game_data & data) const
 205 {
 206         if(!this->is_synced())
 207         {
 208                 //It is possible to for example show a help menu during a [delay] of a synced event.
 209                 set_scontext_unsynced leave_synced_context;
 210                 assert(resources::game_events != nullptr);
 211                 resources::game_events->pump().fire(event_name_, event_hex);
 212                 return;
 213         }
 214         const map_location & last_select = data.last_selected;
 215 
 216         // No new player-issued commands allowed while this is firing.
 217         const events::command_disabler disable_commands;
 218 
 219         // instead of adding a second "select" event like it was done before, we just fire the select event again, and this time in a synced context.
 220         // note that there coudn't be a user choice during the last "select" event because it didn't run in a synced context.
 221         if ( needs_select_  &&  last_select.valid() )
 222         {
 223                 synced_context::run_and_throw("fire_event",  replay_helper::get_event(event_name_, event_hex, &last_select));
 224         }
 225         else
 226         {
 227                 synced_context::run_in_synced_context_if_not_already("fire_event",  replay_helper::get_event(event_name_, event_hex, nullptr));
 228         }
 229 }

 231 /**
 232  * Removes the implicit event handler for an inlined [command].
 233  */
 234 void wml_menu_item::finish_handler() const
 235 {
 236         if ( !command_.empty() ) {
 237                 assert(resources::game_events);
 238                 resources::game_events->remove_event_handler(command_["id"]);
 239         }
 240 
 241         // Hotkey support
 242         if ( use_hotkey_ )
 243                 hotkey::remove_wml_hotkey(hotkey_id_);
 244 }

 246 /**
 247  * Initializes the implicit event handler for an inlined [command].
 248  */
 249 void wml_menu_item::init_handler() const
 250 {
 251         // If this menu item has a [command], add a handler for it.
 252         if ( !command_.empty() ) {
 253                 assert(resources::game_events);
 254                 resources::game_events->add_event_handler(command_, true);
 255         }
 256 
 257         // Hotkey support
 258         if ( use_hotkey_ ) {
 259                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 260         }
 261 }

 263 /**
 264  * Writes *this to the provided config.
 265  * This is the reverse of the constructor from a config and corresponds to
 266  * what will appear in [menu_item].
 267  */
 268 void wml_menu_item::to_config(config & cfg) const
 269 {
 270         cfg["id"] = item_id_;
 271         cfg["image"] = image_;
 272         cfg["description"] = description_;
 273         cfg["needs_select"] = needs_select_;
 274         cfg["synced"] = is_synced_;
 275 
 276         if ( use_hotkey_ && use_wml_menu_ )
 277                 cfg["use_hotkey"] = true;
 278         if ( use_hotkey_ && !use_wml_menu_ )
 279                 cfg["use_hotkey"] = "only";
 280         if ( !use_hotkey_ && use_wml_menu_ )
 281                 cfg["use_hotkey"] = false;
 282         if ( !use_hotkey_ && !use_wml_menu_ )
 283         {
 284                 ERR_NG << "Bad data: wml_menu_item with both use_wml_menu and use_hotkey set to false is not supposed to be possible.";
 285                 cfg["use_hotkey"] = false;
 286         }
 287         if ( !show_if_.empty() )
 288                 cfg.add_child("show_if", show_if_.get_config());
 289         if ( !filter_location_.empty() )
 290                 cfg.add_child("filter_location", filter_location_.get_config());
 291         if ( !command_.empty() )
 292                 cfg.add_child("command", command_);
 293         if ( !default_hotkey_.empty() )
 294                 cfg.add_child("default_hotkey", default_hotkey_);
 295 }

 297 /**
 298  * Updates *this based on @a vcfg.
 299  * This corresponds to what can appear in [set_menu_item].
 300  */
 301 void wml_menu_item::update(const vconfig & vcfg)
 302 {
 303         const bool old_use_hotkey = use_hotkey_;
 304         // Tracks whether or not the hotkey has been updated.
 305         bool hotkey_updated = false;
 306 
 307         if ( vcfg.has_attribute("image") )
 308                 image_ = vcfg["image"].str();
 309 
 310         if ( vcfg.has_attribute("description") ) {
 311                 description_ = vcfg["description"].t_str();
 312                 hotkey_updated = true;
 313         }
 314 
 315         if ( vcfg.has_attribute("needs_select") ) {
 316                 ERR_NG << "needs_select= is deprecated\n";
 317                 needs_select_ = vcfg["needs_select"].to_bool();
 318         }
 319         if ( vcfg.has_attribute("synced") )
 320                 is_synced_ = vcfg["synced"].to_bool(true);
 321 
 322         if ( const vconfig & child = vcfg.child("show_if") ) {
 323                 show_if_ = child;
 324                 show_if_.make_safe();
 325         }
 326 
 327         if ( const vconfig & child = vcfg.child("filter_location") ) {
 328                 filter_location_ = child;
 329                 filter_location_.make_safe();
 330         }
 331 
 332         if ( const vconfig & child = vcfg.child("default_hotkey") ) {
 333                 default_hotkey_ = child.get_parsed_config();
 334                 hotkey_updated = true;
 335         }
 336 
 337         if ( vcfg.has_attribute("use_hotkey") ) {
 338                 const config::attribute_value & use_hotkey_av = vcfg["use_hotkey"];
 339 
 340                 use_hotkey_ = use_hotkey_av.to_bool(true);
 341                 use_wml_menu_ = use_hotkey_av.str() != "only";
 342         }
 343 
 344         if ( const vconfig & cmd = vcfg.child("command") ) {
 345                 const bool delayed = cmd["delayed_variable_substitution"].to_bool(true);
 346                 update_command(delayed ? cmd.get_config() : cmd.get_parsed_config());
 347         }
 348 
 349 
 350         // Update the registered hotkey?
 351 
 352         if ( use_hotkey_ && !old_use_hotkey )
 353                 // The hotkey needs to be enabled.
 354                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 355 
 356         else if ( use_hotkey_ && hotkey_updated )
 357                 // The hotkey needs to be updated.
 358                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 359 
 360         else if ( !use_hotkey_ && old_use_hotkey )
 361                 // The hotkey needs to be disabled.
 362                 hotkey::remove_wml_hotkey(hotkey_id_);
 363 }

 365 /**
 366  * Updates our command to @a new_command.
 367  */
 368 void wml_menu_item::update_command(const config & new_command)
 369 {
 370         // If there is an old command, remove it from the event handlers.
 371         if ( !command_.empty() ) {
 372                 assert(resources::game_events);
 373                 manager::iteration iter(event_name_, *resources::game_events);
 374                 while ( handler_ptr hand = *iter ) {
 375                         if ( hand->is_menu_item() ) {
 376                                 LOG_NG << "Removing command for " << event_name_ << ".\n";
 377                                 resources::game_events->remove_event_handler(command_["id"].str());
 378                         }
 379                         ++iter;
 380                 }
 381         }
 382 
 383         // Update our stored command.
 384         if ( new_command.empty() )
 385                 command_.clear();
 386         else {
 387                 command_ = new_command;
 388 
 389                 // Set some fields required by event processing.
 390                 config::attribute_value & event_id = command_["id"];
 391                 if ( event_id.empty() && !item_id_.empty() ) {
 392                         event_id = item_id_;
 393                 }
 394                 command_["name"] = event_name_;
 395                 command_["first_time_only"] = false;
 396 
 397                 // Register the event.
 398                 LOG_NG << "Setting command for " << event_name_ << " to:\n" << command_;
 399                 assert(resources::game_events);
 400                 resources::game_events->add_event_handler(command_, true);
 401         }
 402 }
