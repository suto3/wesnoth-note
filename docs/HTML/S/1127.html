 101 std::string dump_games_map(const lobby_info::game_info_map& games)
 102 {
 103         std::stringstream ss;
 104         for(const auto & v : games) {
 105                 const game_info& game = *v.second;
 106                 ss << "G" << game.id << "(" << game.name << ") "
 107                    << game.display_status_string() << " ";
 108         }
 109         ss << "\n";
 110         return ss.str();

 126 void lobby_info::process_gamelist(const config& data)
 127 {
 128         SCOPE_LB;
 129         gamelist_ = data;
 130         gamelist_initialized_ = true;
 131         delete_games();
 132         games_by_id_.clear();
 133 
 134         for(const auto & c : gamelist_.child("gamelist").child_range("game")) {
 135                 game_info* game = new game_info(c, game_config_, installed_addons_);
 136                 games_by_id_[game->id] = game;
 137         }
 138 
 139         DBG_LB << dump_games_map(games_by_id_);
 140         DBG_LB << dump_games_config(gamelist_.child("gamelist"));
 141 
 142         process_userlist();
 143 }

 145 bool lobby_info::process_gamelist_diff(const config& data)
 146 {
 147         SCOPE_LB;
 148         if(!gamelist_initialized_) {
 149                 return false;
 150         }
 151 
 152         DBG_LB << "prediff " << dump_games_config(gamelist_.child("gamelist"));
 153         try {
 154                 gamelist_.apply_diff(data, true);
 155         } catch(config::error& e) {
 156                 ERR_LB << "Error while applying the gamelist diff: '" << e.message
 157                        << "' Getting a new gamelist.\n";
 158                 return false;
 159         }
 160 
 161         DBG_LB << "postdiff " << dump_games_config(gamelist_.child("gamelist"));
 162         DBG_LB << dump_games_map(games_by_id_);
 163 
 164         for(config& c : gamelist_.child("gamelist").child_range("game")) {
 165                 DBG_LB << "data process: " << c["id"] << " (" << c[config::diff_track_attribute] << ")\n";
 166                 const int game_id = c["id"];
 167                 if(game_id == 0) {
 168                         ERR_LB << "game with id 0 in gamelist config" << std::endl;
 169                         return false;
 170                 }
 171 
 172                 game_info_map::iterator current_i = games_by_id_.find(game_id);
 173 
 174                 const std::string& diff_result = c[config::diff_track_attribute];
 175                 if(diff_result == "new" || diff_result == "modified") {
 176                         if(current_i == games_by_id_.end()) {
 177                                 games_by_id_.emplace(game_id, new game_info(c, game_config_, installed_addons_));
 178                                 continue;
 179                         }
 180 
 181                         // Had a game with that id, so update it and mark it as such
 182                         *(current_i->second) = game_info(c, game_config_, installed_addons_);
 183                         current_i->second->display_status = game_info::UPDATED;
 184                 } else if(diff_result == "deleted") {
 185                         if(current_i == games_by_id_.end()) {
 186                                 WRN_LB << "Would have to delete a game that I don't have: " << game_id << "\n";
 187                                 continue;
 188                         }
 189 
 190                         if(current_i->second->display_status == game_info::NEW) {
 191                                 // This means the game never made it through to the user interface,
 192                                 // so just deleting it is fine
 193                                 // TODO: use std::unique_ptr instead of deleting manually.
 194                                 delete current_i->second;
 195                                 games_by_id_.erase(current_i);
 196                         } else {
 197                                 current_i->second->display_status = game_info::DELETED;
 198                         }
 199                 }
 200         }
 201 
 202         DBG_LB << dump_games_map(games_by_id_);
 203         try     {
 204                 gamelist_.clear_diff_track(data);
 205         } catch(config::error& e) {
 206                 ERR_LB << "Error while applying the gamelist diff (2): '" << e.message
 207                            << "' Getting a new gamelist.\n";
 208                 return false;
 209         }
 210 
 211         DBG_LB << "postclean " << dump_games_config(gamelist_.child("gamelist"));
 212         process_userlist();
 213 
 214         return true;
 215 }

 217 void lobby_info::process_userlist()
 218 {
 219         SCOPE_LB;
 220         users_.clear();
 221         for(const auto & c : gamelist_.child_range("user")) {
 222                 users_.push_back(user_info(c));
 223         }
 224 
 225         for(auto & ui : users_) {
 226                 if(ui.game_id == 0) {
 227                         continue;
 228                 }
 229 
 230                 game_info* g = get_game_by_id(ui.game_id);
 231                 if(!g) {
 232                         WRN_NG << "User " << ui.name << " has unknown game_id: " << ui.game_id << "\n";
 233                         continue;
 234                 }
 235 
 236                 switch(ui.relation) {
 237                         case user_info::FRIEND:
 238                                 g->has_friends = true;
 239                                 break;
 240                         case user_info::IGNORED:
 241                                 g->has_ignored = true;
 242                                 break;
 243                         default:
 244                                 break;
 245                 }
 246         }
 247 }

 249 void lobby_info::sync_games_display_status()
 250 {
 251         DBG_LB << "lobby_info::sync_games_display_status";
 252         DBG_LB << "games_by_id_ size: " << games_by_id_.size();
 253 
 254         game_info_map::iterator i = games_by_id_.begin();
 255         while(i != games_by_id_.end()) {
 256                 if(i->second->display_status == game_info::DELETED) {
 257                         delete i->second;
 258                         i = games_by_id_.erase(i);
 259                 } else {
 260                         i->second->display_status = game_info::CLEAN;
 261                         ++i;
 262                 }
 263         }
 264 
 265         DBG_LB << " -> " << games_by_id_.size() << "\n";
 266         make_games_vector();
 267 }

 369 void lobby_info::apply_game_filter()
 370 {
 371         games_filtered_.clear();
 372         games_visibility_.clear();
 373         for(auto g : games_) {
 374                 game_info& gi = *g;
 375 
 376                 bool show = true;
 377                 for(const auto& filter_func : game_filters_) {
 378                         show = filter_func(gi);
 379                         if(!show) {
 380                                 break;
 381                         }
 382                 }
 383 
 384                 if(game_filter_invert_) {
 385                         show = !show;
 386                 }
 387 
 388                 games_visibility_.push_back(show);
 389                 if(show) {
 390                         games_filtered_.push_back(&gi);
 391                 }
 392         }
 393 }

 402 void lobby_info::sort_users(bool by_name, bool by_relation)
 403 {
 404         users_sorted_.clear();
 405         for(auto& u : users_) {
 406                 users_sorted_.push_back(&u);
 407         }
 408 
 409         // TODO: is this the simplest way to represent the sorting method?
 410         std::sort(users_sorted_.begin(), users_sorted_.end(), [&](const user_info* u1, const user_info* u2) {
 411                 if(by_name && by_relation) {
 412                         return u1->relation < u2->relation || (u1->relation == u2->relation && u1->name < u2->name);
 413                 }
 414 
 415                 return (by_name ? u1->name < u2->name : true) && (by_relation ? u1->relation < u2->relation : true);
 416         });
 417 }
