  92     /**
  93      * Loads the templates from the file system
  94      */
  95     public void loadTemplates( )
  96     {
  97         templates_.clear( );
  98         try {
  99             Logger.getInstance( ).log( "reading templates from: " + //$NON-NLS-1$
 100                 WesnothPlugin.PLUGIN_FULL_PATH + TEMPLATES_FILENAME );
 101 
 102             BufferedReader reader = new BufferedReader( new FileReader(
 103                 WesnothPlugin.PLUGIN_FULL_PATH + TEMPLATES_FILENAME ) );
 104             BufferedReader tmpReader;
 105             String line, tmpLine;
 106             StringBuilder content = new StringBuilder( );
 107 
 108             // read the main "templatesIndex.txt" file
 109             while( ( line = reader.readLine( ) ) != null ) {
 110                 // comment
 111                 if( line.startsWith( "#" ) || line.isEmpty( ) ) {
 112                     continue;
 113                 }
 114 
 115                 // 0 - template name | 1 - template file
 116                 String[] tokensStrings = line.split( " " ); //$NON-NLS-1$
 117 
 118                 if( tokensStrings.length != 2 ) {
 119                     Logger.getInstance( ).logWarn(
 120                         "TemplateIndex line " + line
 121                             + "is not properly formatted" );
 122                     continue;
 123                 }
 124 
 125                 content.setLength( 0 );
 126 
 127                 if( new File( WesnothPlugin.PLUGIN_FULL_PATH + tokensStrings[1] )
 128                     .exists( ) ) {
 129                     tmpReader = new BufferedReader( new FileReader(
 130                         WesnothPlugin.PLUGIN_FULL_PATH + tokensStrings[1] ) );
 131                     while( ( tmpLine = tmpReader.readLine( ) ) != null ) {
 132                         content.append( tmpLine + '\n' );
 133                     }
 134                     templates_.put( tokensStrings[0], content.toString( ) );
 135                     tmpReader.close( );
 136                 }
 137             }
 138             reader.close( );
 139         } catch( IOException e ) {
 140             Logger.getInstance( ).logException( e );
 141         }
 142     }

 144     /**
 145      * Gets a string of the processed specified template
 146      * 
 147      * @param templateName
 148      *        The name of the template to process
 149      * @param parameters
 150      *        The parameters to replace into the template
 151      * @return A {@link String} that represents the processed template.
 152      */
 153     public String getProcessedTemplate( String templateName,
 154         List< ReplaceableParameter > parameters )
 155     {
 156         String tmpTemplate = TemplateProvider.getInstance( ).getTemplate(
 157             templateName );
 158         if( tmpTemplate == null || parameters == null ) {
 159             return null;
 160         }
 161 
 162         StringBuilder result = new StringBuilder( );
 163         String[] template = StringUtils.getLines( tmpTemplate );
 164         boolean skipLine;
 165 
 166         for( int i = 0; i < template.length; ++i ) {
 167             skipLine = false;
 168             for( ReplaceableParameter param: parameters ) {
 169                 if( template[i].contains( param.paramName ) ) {
 170                     template[i] = StringUtils.replaceWithIndent( template[i],
 171                         param.paramName, param.paramValue );
 172 
 173                     if( ! templateName.equals( "build_xml" ) && //$NON-NLS-1$
 174                         ( param.paramValue == null || param.paramValue
 175                             .isEmpty( ) ) ) {
 176                         // we don't have any value supplied -
 177                         // let's comment that line (if it's not already
 178                         // commented)
 179                         // or remove it if it's empty
 180                         if( ! ( StringUtils.startsWith( template[i], "#" ) ) ) //$NON-NLS-1$
 181                         {
 182                             template[i] = "#" + template[i]; //$NON-NLS-1$
 183                             Pattern pattern = Pattern.compile( "#[\t ]*" ); //$NON-NLS-1$
 184                             Matcher matcher = pattern.matcher( template[i] );
 185                             if( matcher.matches( ) ) {
 186                                 skipLine = true;
 187                             }
 188                         }
 189                     }
 190                 }
 191             }
 192 
 193             if( skipLine == false ) {
 194                 result.append( template[i] + "\n" ); //$NON-NLS-1$
 195             }
 196         }
 197         return result.toString( );
 198     }

 233     /**
 234      * Gets the lists of the specified structure template. The first return
 235      * value is a list of <String, String> that consist of <Filename, Template
 236      * used for file contents> and the second return value is a list of String
 237      * with directories names
 238      * 
 239      * @param structureTemplate
 240      *        the template
 241      * @return A pair with the files and directories.
 242      */
 243     public Pair< List< Pair< String, String >>, List< String >> getFilesDirectories(
 244         String structureTemplate )
 245     {
 246         List< Pair< String, String >> files = new ArrayList< Pair< String, String >>( );
 247         List< String > dirs = new ArrayList< String >( );
 248 
 249         for( String line: StringUtils.getLines( structureTemplate ) ) {
 250             if( StringUtils.startsWith( line, "#" ) ) {
 251                 continue;
 252             }
 253 
 254             if( line.contains( ":" ) ) // file with template //$NON-NLS-1$
 255             {
 256                 String[] tmpLine = line.split( ":" ); //$NON-NLS-1$
 257 
 258                 // oops. error
 259                 if( tmpLine.length != 2 ) {
 260                     Logger
 261                         .getInstance( )
 262                         .logError(
 263                             String
 264                                 .format( "error parsing 'structure template' (%s) on line %s", //$NON-NLS-1$
 265                                     structureTemplate, line ) );
 266                     continue;
 267                 }
 268 
 269                 files.add( new Pair< String, String >( tmpLine[0].trim( ),
 270                     tmpLine[1].trim( ) ) );
 271             }
 272             else {
 273                 dirs.add( line.trim( ) );
 274             }
 275         }
 276 
 277         return new Pair< List< Pair< String, String >>, List< String >>( files,
 278             dirs );
 279     }
