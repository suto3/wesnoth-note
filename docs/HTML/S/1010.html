 123 /// The basic unit filter gives a generic implementation of the match fcn
 124 class basic_unit_filter_impl : public unit_filter_abstract_impl {
 125 public:
 126         basic_unit_filter_impl(const vconfig & vcfg, const filter_context & fc, bool flat_tod)
 127                 : fc_(fc)
 128                 , vcfg(vcfg)
 129                 , use_flat_tod_(flat_tod)
 130                 , cond_children_()
 131                 , cond_child_types_()
 132         {
 133                 // Handle [and], [or], and [not] with in-order precedence
 134                 vconfig::all_children_iterator cond = vcfg.ordered_begin();
 135                 vconfig::all_children_iterator cond_end = vcfg.ordered_end();
 136                 while(cond != cond_end)
 137                 {
 138                         const std::string& cond_name = cond.get_key();
 139                         conditional::TYPE type;
 140                         if(type.parse(cond_name)) {
 141                                 const vconfig& cond_filter = cond.get_child();
 142 
 143                                 cond_children_.emplace_back(cond_filter, &fc_, use_flat_tod_);
 144                                 cond_child_types_.push_back(type);
 145                         }
 146                         else {
 147                                 static const int NUM_VALID_TAGS = 5;
 148                                 static const std::string valid_tags[NUM_VALID_TAGS] {
 149                                         "filter_vision",
 150                                         "filter_adjacent",
 151                                         "filter_location",
 152                                         "filter_side",
 153                                         "filter_wml",
 154                                 };
 155                                 static const std::string* const valid_tags_end = valid_tags + NUM_VALID_TAGS;
 156 
 157                                 if (std::find(valid_tags, valid_tags_end, cond_name) == valid_tags_end){
 158                                         std::stringstream errmsg;
 159                                         errmsg << "encountered a child [" << cond_name << "] of a standard unit filter, it is being ignored";
 160                                         DBG_CF << errmsg.str() << std::endl; //FAIL( errmsg.str() );
 161                                 }
 162 
 163                         }
 164                         ++cond;
 165                 }
 166                 this->vcfg.make_safe();
 167         }
 168 
 169         virtual bool matches(const unit & u, const map_location & loc, const unit * u2) const;
 170         virtual std::vector<const unit *> all_matches_on_map(unsigned max_matches) const;
 171         virtual unit_const_ptr first_match_on_map() const;
 172         config to_config() const {
 173                 return vcfg.get_config();
 174         }
 175 
 176         virtual ~basic_unit_filter_impl() {}
 177 private:
 178         const filter_context & fc_;
 179         const vconfig vcfg;
 180         bool use_flat_tod_;
 181 
 182         std::vector<unit_filter> cond_children_;
 183         std::vector<conditional::TYPE> cond_child_types_;
 184 
 185         bool internal_matches_filter(const unit & u, const map_location & loc, const unit* u2) const;
 186 };

 192 static std::shared_ptr<unit_filter_abstract_impl> construct(const vconfig & vcfg, const filter_context & fc, bool flat_tod)
 193 {
 194         if (vcfg.empty()) {
 195                 return std::make_shared<null_unit_filter_impl> (fc);
 196         }
 197         if (vcfg.get_config().attribute_count() == 1 && vcfg.get_config().all_children_count() == 0 && vcfg.has_attribute("limit")) {
 198                 return std::make_shared<null_unit_filter_impl> (fc);
 199         }
 200         return std::make_shared<basic_unit_filter_impl>(vcfg, fc, flat_tod);
 201         //TODO: Add more efficient implementations for special cases
 202 }

 223 bool basic_unit_filter_impl::matches(const unit & u, const map_location& loc, const unit * u2) const
 224 {
 225         bool matches = true;
 226 
 227         if(loc.valid()) {
 228                 scoped_xy_unit auto_store("this_unit", loc, fc_.get_disp_context().units());
 229                 if (u2) {
 230                         const map_location& loc2 = u2->get_location();
 231                         scoped_xy_unit u2_auto_store("other_unit", loc2, fc_.get_disp_context().units());
 232                         matches = internal_matches_filter(u, loc, u2);
 233                 } else {
 234                         matches = internal_matches_filter(u, loc, u2);
 235                 }
 236         } else {
 237                 // If loc is invalid, then this is a recall list unit (already been scoped)
 238                 matches = internal_matches_filter(u, loc, nullptr);
 239         }
 240 
 241         // Handle [and], [or], and [not] with in-order precedence
 242         for (size_t i = 0; i < cond_children_.size(); i++) {
 243                 switch (cond_child_types_[i].v) {
 244                         case conditional::TYPE::AND:
 245                                 matches = matches && cond_children_[i].matches(u,loc);
 246                                 break;
 247                         case conditional::TYPE::OR:
 248                                 matches = matches || cond_children_[i].matches(u,loc);
 249                                 break;
 250                         case conditional::TYPE::NOT:
 251                                 matches = matches && !cond_children_[i].matches(u,loc);
 252                 }
 253         }
 254         return matches;
 255 }

 257 bool basic_unit_filter_impl::internal_matches_filter(const unit & u, const map_location& loc, const unit* u2) const
 258 {
 259         if (!vcfg["name"].blank() && vcfg["name"].t_str() != u.name()) {
 260                 return false;
 261         }
 262 
 263         if (!vcfg["id"].blank()) {
 264                 std::vector<std::string> id_list = utils::split(vcfg["id"]);
 265                 if (std::find(id_list.begin(), id_list.end(), u.id()) == id_list.end()) {
 266                         return false;
 267                 }
 268         }
 269 
 270         // Allow 'speaker' as an alternative to id, since people use it so often
 271         if (!vcfg["speaker"].blank() && vcfg["speaker"].str() != u.id()) {
 272                 return false;
 273         }
 274 
 275         if (vcfg.has_child("filter_location")) {
 276                 if (vcfg.count_children("filter_location") > 1) {
 277                         FAIL("Encountered multiple [filter_location] children of a standard unit filter. "
 278                                  "This is not currently supported and in all versions of wesnoth would have "
 279                                  "resulted in the later children being ignored. You must use [and] or similar "
 280                                  "to achieve the desired result.");
 281                 }
 282                 terrain_filter filt(vcfg.child("filter_location"), &fc_, use_flat_tod_);
 283                 if (!filt.match(loc)) {
 284                         return false;
 285                 }
 286         }
 287 
 288         if(vcfg.has_child("filter_side")) {
 289                 if (vcfg.count_children("filter_side") > 1) {
 290                         FAIL("Encountered multiple [filter_side] children of a standard unit filter. "
 291                                  "This is not currently supported and in all versions of wesnoth would have "
 292                                  "resulted in the later children being ignored. You must use [and] or similar "
 293                                  "to achieve the desired result.");
 294                 }
 295                 side_filter filt(vcfg.child("filter_side"), &fc_);
 296                 if(!filt.match(u.side()))
 297                         return false;
 298         }
 299 
 300         // Also allow filtering on location ranges outside of the location filter
 301         if (!vcfg["x"].blank() || !vcfg["y"].blank()){
 302                 if(vcfg["x"] == "recall" && vcfg["y"] == "recall") {
 303                         //locations on the map are considered to not be on a recall list
 304                         if (fc_.get_disp_context().map().on_board(loc))
 305                         {
 306                                 return false;
 307                         }
 308                 } else if(vcfg["x"].empty() && vcfg["y"].empty()) {
 309                         return false;
 310                 } else if(!loc.matches_range(vcfg["x"], vcfg["y"])) {
 311                         return false;
 312                 }
 313         }
 314 
 315         // The type could be a comma separated list of types
 316         if (!vcfg["type"].empty()) {
 317                 std::vector<std::string> types = utils::split(vcfg["type"]);
 318                 if (std::find(types.begin(), types.end(), u.type_id()) == types.end()) {
 319                         return false;
 320                 }
 321         }
 322 
 323         // Shorthand for all advancements of a given type
 324         if (!vcfg["type_adv_tree"].empty()) {
 325                 std::set<std::string> types;
 326                 for(const std::string type : utils::split(vcfg["type_adv_tree"])) {
 327                         if(types.count(type)) {
 328                                 continue;
 329                         }
 330                         if(const unit_type* ut = unit_types.find(type)) {
 331                                 const auto& tree = ut->advancement_tree();
 332                                 types.insert(tree.begin(), tree.end());
 333                                 types.insert(type);
 334                         }
 335                 }
 336                 if(types.find(u.type_id()) == types.end()) {
 337                         return false;
 338                 }
 339         }
 340 
 341         // The variation_type could be a comma separated list of types
 342         if (!vcfg["variation"].empty())
 343         {
 344                 std::vector<std::string> types = utils::split(vcfg["variation"]);
 345                 if (std::find(types.begin(), types.end(), u.variation()) == types.end()) {
 346                         return false;
 347                 }
 348         }
 349 
 350         // The has_variation_type could be a comma separated list of types
 351         if (!vcfg["has_variation"].empty())
 352         {
 353                 bool match = false;
 354                 // If this unit is a variation itself then search in the base unit's variations.
 355                 const unit_type* const type = u.variation().empty() ? &u.type() : unit_types.find(u.type().base_id());
 356                 assert(type);
 357 
 358                 for (const std::string& variation_id : utils::split(vcfg["has_variation"])) {
 359                         if (type->has_variation(variation_id)) {
 360                                 match = true;
 361                                 break;
 362                         }
 363                 }
 364                 if (!match) return false;
 365         }
 366 
 367         if (!vcfg["ability"].empty())
 368         {
 369                 bool match = false;
 370 
 371                 for (const std::string& ability_id : utils::split(vcfg["ability"])) {
 372                         if (u.has_ability_by_id(ability_id)) {
 373                                 match = true;
 374                                 break;
 375                         }
 376                 }
 377                 if (!match) return false;
 378         }
 379 
 380         if (!vcfg["ability_type"].empty())
 381         {
 382                 bool match = false;
 383 
 384                 for (const std::string& ability : utils::split(vcfg["ability_type"])) {
 385                         if (u.has_ability_type(ability)) {
 386                                 match = true;
 387                                 break;
 388                         }
 389                 }
 390                 if (!match) return false;
 391         }
 392 
 393         if(!vcfg["ability_type_active"].empty()) {
 394                 bool match = false;
 395 
 396                 for(const std::string& ability : utils::split(vcfg["ability_type_active"])) {
 397                         if(!u.get_abilities(ability, loc).empty()) {
 398                                 match = true;
 399                                 break;
 400                         }
 401                 }
 402                 if(!match) {
 403                         return false;
 404                 }
 405         }
 406 
 407         if(!vcfg["trait"].empty()) {
 408                 std::vector<std::string> check_traits = utils::split(vcfg["trait"]), have_traits = u.get_traits_list(), isect;
 409                 std::sort(check_traits.begin(), check_traits.end());
 410                 std::sort(have_traits.begin(), have_traits.end());
 411                 std::set_intersection(check_traits.begin(), check_traits.end(), have_traits.begin(), have_traits.end(), std::back_inserter(isect));
 412                 if(isect.empty()) {
 413                         return false;
 414                 }
 415         }
 416 
 417         if (!vcfg["race"].empty()) {
 418                 std::vector<std::string> races = utils::split(vcfg["race"]);
 419                 if (std::find(races.begin(), races.end(), u.race()->id()) == races.end()) {
 420                         return false;
 421                 }
 422         }
 423 
 424         if (!vcfg["gender"].blank() && string_gender(vcfg["gender"]) != u.gender()) {
 425                 return false;
 426         }
 427 
 428         if (!vcfg["side"].empty() && vcfg["side"].to_int(-999) != u.side()) {
 429                 std::vector<std::string> sides = utils::split(vcfg["side"]);
 430                 const std::string u_side = std::to_string(u.side());
 431                 if (std::find(sides.begin(), sides.end(), u_side) == sides.end()) {
 432                         return false;
 433                 }
 434         }
 435 
 436         // handle statuses list
 437         if (!vcfg["status"].empty()) {
 438                 bool status_found = false;
 439 
 440                 for (const std::string status : utils::split(vcfg["status"])) {
 441                         if(u.get_state(status)) {
 442                                 status_found = true;
 443                                 break;
 444                         }
 445                 }
 446 
 447                 if(!status_found) {
 448                         return false;
 449                 }
 450         }
 451 
 452         if (vcfg.has_child("has_attack")) {
 453                 const vconfig& weap_filter = vcfg.child("has_attack");
 454                 bool has_weapon = false;
 455                 for(const attack_type& a : u.attacks()) {
 456                         if(a.matches_filter(weap_filter.get_parsed_config())) {
 457                                 has_weapon = true;
 458                                 break;
 459                         }
 460                 }
 461                 if(!has_weapon) {
 462                         return false;
 463                 }
 464         } else if (!vcfg["has_weapon"].blank()) {
 465                 std::string weapon = vcfg["has_weapon"];
 466                 bool has_weapon = false;
 467                 for(const attack_type& a : u.attacks()) {
 468                         if(a.id() == weapon) {
 469                                 has_weapon = true;
 470                                 break;
 471                         }
 472                 }
 473                 if(!has_weapon) {
 474                         return false;
 475                 }
 476         }
 477 
 478         if (!vcfg["role"].blank() && vcfg["role"].str() != u.get_role()) {
 479                 return false;
 480         }
 481 
 482         if (!vcfg["ai_special"].blank() && ((vcfg["ai_special"].str() == "guardian") != u.get_state(unit::STATE_GUARDIAN))) {
 483                 return false;
 484         }
 485 
 486         if (!vcfg["canrecruit"].blank() && vcfg["canrecruit"].to_bool() != u.can_recruit()) {
 487                 return false;
 488         }
 489 
 490         if (!vcfg["recall_cost"].blank() && vcfg["recall_cost"].to_int(-1) != u.recall_cost()) {
 491                 return false;
 492         }
 493 
 494         if (!vcfg["level"].blank() && vcfg["level"].to_int(-1) != u.level()) {
 495                 return false;
 496         }
 497 
 498         if (!vcfg["defense"].blank() && vcfg["defense"].to_int(-1) != u.defense_modifier(fc_.get_disp_context().map().get_terrain(loc))) {
 499                 return false;
 500         }
 501 
 502         if (!vcfg["movement_cost"].blank() && vcfg["movement_cost"].to_int(-1) != u.movement_cost(fc_.get_disp_context().map().get_terrain(loc))) {
 503                 return false;
 504         }
 505 
 506         // Now start with the new WML based comparison.
 507         // If a key is in the unit and in the filter, they should match
 508         // filter only => not for us
 509         // unit only => not filtered
 510         config unit_cfg; // No point in serializing the unit once for each [filter_wml]!
 511         for (const vconfig& wmlcfg : vcfg.get_children("filter_wml")) {
 512                         config fwml = wmlcfg.get_parsed_config();
 513                         /* Check if the filter only cares about variables.
 514                            If so, no need to serialize the whole unit. */
 515                         config::all_children_itors ci = fwml.all_children_range();
 516                         if (fwml.all_children_count() == 1 &&
 517                                 fwml.attribute_count() == 1 &&
 518                             ci.front().key == "variables") {
 519                                 if (!u.variables().matches(ci.front().cfg))
 520                                         return false;
 521                         } else {
 522                                 if (unit_cfg.empty())
 523                                         u.write(unit_cfg);
 524                                 if (!unit_cfg.matches(fwml))
 525                                         return false;
 526                         }
 527         }
 528 
 529         for (const vconfig& vision : vcfg.get_children("filter_vision")) {
 530                 std::set<int> viewers;
 531 
 532                 // Use standard side filter
 533                 side_filter ssf(vision, &fc_);
 534                 std::vector<int> sides = ssf.get_teams();
 535                 viewers.insert(sides.begin(), sides.end());
 536 
 537                 bool found = false;
 538                 for (const int viewer : viewers) {
 539                         bool fogged = fc_.get_disp_context().get_team(viewer).fogged(loc);
 540                         bool hiding = u.invisible(loc, fc_.get_disp_context())
 541                                 && fc_.get_disp_context().get_team(viewer).is_enemy(u.side());
 542                         bool unit_hidden = fogged || hiding;
 543                         if (vision["visible"].to_bool(true) != unit_hidden) {
 544                                 found = true;
 545                                 break;
 546                         }
 547                 }
 548                 if (!found) {return false;}
 549         }
 550 
 551         if (vcfg.has_child("filter_adjacent")) {
 552                 const unit_map& units = fc_.get_disp_context().units();
 553                 map_location adjacent[6];
 554                 get_adjacent_tiles(loc, adjacent);
 555 
 556                 for (const vconfig& adj_cfg : vcfg.get_children("filter_adjacent")) {
 557                         int match_count=0;
 558                         unit_filter filt(adj_cfg, &fc_, use_flat_tod_);
 559 
 560                         config::attribute_value i_adjacent = adj_cfg["adjacent"];
 561                         std::vector<map_location::DIRECTION> dirs;
 562                         if (i_adjacent.blank()) {
 563                                 dirs = map_location::default_dirs();
 564                         } else {
 565                                 dirs = map_location::parse_directions(i_adjacent);
 566                         }
 567 
 568                         std::vector<map_location::DIRECTION>::const_iterator j, j_end = dirs.end();
 569                         for (j = dirs.begin(); j != j_end; ++j) {
 570                                 unit_map::const_iterator unit_itor = units.find(adjacent[*j]);
 571                                 if (unit_itor == units.end() || !filt(*unit_itor, u)) {
 572                                         continue;
 573                                 }
 574                                 boost::optional<bool> is_enemy;
 575                                 if (!adj_cfg["is_enemy"].blank()) {
 576                                         is_enemy = adj_cfg["is_enemy"].to_bool();
 577                                 }
 578                                 if (!is_enemy || *is_enemy ==
 579                                     fc_.get_disp_context().get_team(u.side()).is_enemy(unit_itor->side())) {
 580                                         ++match_count;
 581                                 }
 582                         }
 583 
 584                         static std::vector<std::pair<int,int> > default_counts = utils::parse_ranges("1-6");
 585                         config::attribute_value i_count = adj_cfg["count"];
 586                         if(!in_ranges(match_count, !i_count.blank() ? utils::parse_ranges(i_count) : default_counts)) {
 587                                 return false;
 588                         }
 589                 }
 590         }
 591 
 592         if (!vcfg["find_in"].blank()) {
 593                 // Allow filtering by searching a stored variable of units
 594                 if (const game_data * gd = fc_.get_game_data()) {
 595                         try
 596                         {
 597                                 variable_access_const vi = gd->get_variable_access_read(vcfg["find_in"]);
 598                                 bool found_id = false;
 599                                 for (const config& c : vi.as_array())
 600                                 {
 601                                         if(c["id"] == u.id())
 602                                                 found_id = true;
 603                                 }
 604                                 if(!found_id)
 605                                 {
 606                                         return false;
 607                                 }
 608                         }
 609                         catch(const invalid_variablename_exception&)
 610                         {
 611                                 return false;
 612                         }
 613                 }
 614         }
 615         if (!vcfg["formula"].blank()) {
 616                 try {
 617                         const wfl::unit_callable main(loc,u);
 618                         wfl::map_formula_callable callable(main.fake_ptr());
 619                         if (u2) {
 620                                 std::shared_ptr<wfl::unit_callable> secondary(new wfl::unit_callable(*u2));
 621                                 callable.add("other", wfl::variant(secondary));
 622                                 // It's not destroyed upon scope exit because the variant holds a reference
 623                         }
 624                         const wfl::formula form(vcfg["formula"]);
 625                         if(!form.evaluate(callable).as_bool()) {
 626                                 return false;
 627                         }
 628                         return true;
 629                 } catch(wfl::formula_error& e) {
 630                         lg::wml_error() << "Formula error in unit filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 631                         // Formulae with syntax errors match nothing
 632                         return false;
 633                 }
 634         }
 635 
 636         if (!vcfg["lua_function"].blank()) {
 637                 if (game_lua_kernel * lk = fc_.get_lua_kernel()) {
 638                         bool b = lk->run_filter(vcfg["lua_function"].str().c_str(), u);
 639                         if (!b) return false;
 640                 }
 641         }
 642 
 643         return true;
 644 }

 646 std::vector<const unit *> basic_unit_filter_impl::all_matches_on_map(unsigned max_matches) const {
 647         std::vector<const unit *> ret;
 648         for (const unit & u : fc_.get_disp_context().units()) {
 649                 if (matches(u, u.get_location(), nullptr)) {
 650                         if(max_matches == 0) {
 651                                 return ret;
 652                         }
 653                         --max_matches;
 654                         ret.push_back(&u);
 655                 }
 656         }
 657         return ret;
 658 }

 660 unit_const_ptr basic_unit_filter_impl::first_match_on_map() const {
 661         const unit_map & units = fc_.get_disp_context().units();
 662         for(unit_map::const_iterator u = units.begin(); u != units.end(); u++) {
 663                 if (matches(*u,u->get_location(),nullptr)) {
 664                         return u.get_shared_ptr();
 665                 }
 666         }
 667         return unit_const_ptr();
 668 }
