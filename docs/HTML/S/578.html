  91 scrollbar_panel_definition::resolution::resolution(const config& cfg)
  92         : resolution_definition(cfg), grid()
  93 {
  94         // The panel needs to know the order.
  95         state.emplace_back(cfg.child("background"));
  96         state.emplace_back(cfg.child("foreground"));
  97 
  98         const config& child = cfg.child("grid");
  99         VALIDATE(child, _("No grid defined."));
 100 
 101         grid = std::make_shared<builder_grid>(child);
 102 }

 138 builder_scrollbar_panel::builder_scrollbar_panel(const config& cfg)
 139         : builder_styled_widget(cfg)
 140         , vertical_scrollbar_mode(
 141                           get_scrollbar_mode(cfg["vertical_scrollbar_mode"]))
 142         , horizontal_scrollbar_mode(
 143                           get_scrollbar_mode(cfg["horizontal_scrollbar_mode"]))
 144         , grid_(nullptr)
 145 {
 146         const config& grid_definition = cfg.child("definition");
 147 
 148         VALIDATE(grid_definition, _("No list defined."));
 149         grid_ = std::make_shared<builder_grid>(grid_definition);
 150         assert(grid_);
 151 }

 153 widget* builder_scrollbar_panel::build() const
 154 {
 155         scrollbar_panel* panel = new scrollbar_panel();
 156 
 157         init_control(panel);
 158 
 159         panel->set_vertical_scrollbar_mode(vertical_scrollbar_mode);
 160         panel->set_horizontal_scrollbar_mode(horizontal_scrollbar_mode);
 161 
 162         DBG_GUI_G << "Window builder: placed scrollbar_panel '" << id
 163                           << "' with definition '" << definition << "'.\n";
 164 
 165         std::shared_ptr<const scrollbar_panel_definition::resolution> conf
 166                         = std::static_pointer_cast<const scrollbar_panel_definition::resolution>(
 167                                 panel->config());
 168         assert(conf);
 169 
 170         panel->init_grid(conf->grid);
 171         panel->finalize_setup();
 172 
 173         /*** Fill the content grid. ***/
 174         grid* content_grid = panel->content_grid();
 175         assert(content_grid);
 176 
 177         const unsigned rows = grid_->rows;
 178         const unsigned cols = grid_->cols;
 179 
 180         content_grid->set_rows_cols(rows, cols);
 181 
 182         for(unsigned x = 0; x < rows; ++x) {
 183                 content_grid->set_row_grow_factor(x, grid_->row_grow_factor[x]);
 184                 for(unsigned y = 0; y < cols; ++y) {
 185 
 186                         if(x == 0) {
 187                                 content_grid->set_column_grow_factor(y,
 188                                                                                                          grid_->col_grow_factor[y]);
 189                         }
 190 
 191                         widget* widget = grid_->widgets[x * cols + y]->build();
 192                         content_grid->set_child(widget,
 193                                                                         x,
 194                                                                         y,
 195                                                                         grid_->flags[x * cols + y],
 196                                                                         grid_->border_size[x * cols + y]);
 197                 }
 198         }
 199 
 200         return panel;
 201 }
