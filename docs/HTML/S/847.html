 107 void teleport_group::get_teleport_pair(
 108                   teleport_pair& loc_pair
 109                 , const unit& u
 110                 , const bool ignore_units) const
 111 {
 112         const map_location &loc = u.get_location();
 113 
 114         const filter_context * fc = resources::filter_con;
 115         assert(fc);
 116 
 117         if (ignore_units) {
 118                 fc = new ignore_units_filter_context(*resources::filter_con);
 119         }
 120 
 121         vconfig filter(cfg_.child_or_empty("filter"), true);
 122         vconfig source(cfg_.child_or_empty("source"), true);
 123         vconfig target(cfg_.child_or_empty("target"), true);
 124         const unit_filter ufilt(filter, resources::filter_con); //Note: Don't use the ignore units filter context here, only for the terrain filters. (That's how it worked before the filter contexts were introduced)
 125         if (ufilt.matches(u, loc)) {
 126 
 127                 scoped_xy_unit teleport_unit("teleport_unit", loc, resources::gameboard->units());
 128 
 129                 terrain_filter source_filter(source, fc);
 130                 source_filter.get_locations(reversed_ ? loc_pair.second : loc_pair.first);
 131 
 132                 terrain_filter target_filter(target, fc);
 133                 target_filter.get_locations(reversed_ ? loc_pair.first : loc_pair.second);
 134         }
 135 
 136         if (ignore_units) {
 137                 delete fc;
 138         }
 139 }

 165 teleport_map::teleport_map(
 166                   const std::vector<teleport_group>& groups
 167                 , const unit& unit
 168                 , const team &viewing_team
 169                 , const bool see_all
 170                 , const bool ignore_units
 171                 , const bool check_vision)
 172         : teleport_map_()
 173         , sources_()
 174         , targets_()
 175 {
 176 
 177         for (const teleport_group& group : groups) {
 178 
 179                 teleport_pair locations;
 180 
 181                 if (check_vision && !group.allow_vision()) {
 182                         continue;
 183                 }
 184 
 185                 group.get_teleport_pair(locations, unit, ignore_units);
 186                 if (!see_all && !group.always_visible() && viewing_team.is_enemy(unit.side())) {
 187                         teleport_pair filter_locs;
 188                         for (const map_location &loc : locations.first) {
 189                                 if(!viewing_team.fogged(loc))
 190                                         filter_locs.first.insert(loc);
 191                         }
 192                         for (const map_location &loc : locations.second) {
 193                                 if(!viewing_team.fogged(loc))
 194                                         filter_locs.second.insert(loc);
 195                         }
 196                         locations.first.swap(filter_locs.first);
 197                         locations.second.swap(filter_locs.second);
 198                 }
 199 
 200                 if (!group.pass_allied_units() && !ignore_units && !check_vision) {
 201                         std::set<map_location>::iterator loc = locations.second.begin();
 202                         while(loc != locations.second.end()) {
 203                                 unit_map::iterator u;
 204                                 if (see_all) {
 205                                         u = resources::gameboard->units().find(*loc);
 206                                 } else {
 207                                         u = resources::gameboard->find_visible_unit(*loc, viewing_team);
 208                                 }
 209                                 if (u != resources::gameboard->units().end()) {
 210                                         loc = locations.second.erase(loc);
 211                                 } else {
 212                                         ++loc;
 213                                 }
 214                         }
 215                 }
 216 
 217                 std::string teleport_id = group.get_teleport_id();
 218                 std::set<map_location>::iterator source_it = locations.first.begin();
 219                 for (; source_it != locations.first.end(); ++source_it ) {
 220                         if(teleport_map_.count(*source_it) == 0) {
 221                                 std::set<std::string> id_set;
 222                                 id_set.insert(teleport_id);
 223                                 teleport_map_.emplace(*source_it, id_set);
 224                         } else {
 225                                 (teleport_map_.find(*source_it)->second).insert(teleport_id);
 226                         }
 227                 }
 228                 sources_.emplace(teleport_id, locations.first);
 229                 targets_.emplace(teleport_id, locations.second);
 230         }
 231 }

 233 void teleport_map::get_adjacents(std::set<map_location>& adjacents, map_location loc) const {
 234 
 235         if (teleport_map_.count(loc) == 0) {
 236                 return;
 237         } else {
 238                 const std::set<std::string>& keyset = (teleport_map_.find(loc)->second);
 239                 for(std::set<std::string>::const_iterator it = keyset.begin(); it != keyset.end(); ++it) {
 240 
 241                         const std::set<map_location>& target = targets_.find(*it)->second;
 242                         adjacents.insert(target.begin(), target.end());
 243                 }
 244         }
 245 }

 264 const teleport_map get_teleport_locations(const unit &u,
 265         const team &viewing_team,
 266         bool see_all, bool ignore_units, bool check_vision)
 267 {
 268         std::vector<teleport_group> groups;
 269 
 270         if (u.get_ability_bool("teleport", *resources::gameboard)) {
 271                 for (const unit_ability & teleport : u.get_abilities("teleport")) {
 272                         const int tunnel_count = (teleport.first)->child_count("tunnel");
 273                         for(int i = 0; i < tunnel_count; ++i) {
 274                                 config teleport_group_cfg = (teleport.first)->child("tunnel", i);
 275                                 groups.emplace_back(vconfig(teleport_group_cfg, true), false);
 276                         }
 277                 }
 278         }
 279 
 280         const std::vector<teleport_group>& global_groups = resources::tunnels->get();
 281         groups.insert(groups.end(), global_groups.begin(), global_groups.end());
 282 
 283         return teleport_map(groups, u, viewing_team, see_all, ignore_units, check_vision);
 284 }

 286 manager::manager(const config &cfg) : tunnels_(), id_(cfg["next_teleport_group_id"].to_int(0)) {
 287         const int tunnel_count = cfg.child_count("tunnel");
 288         for(int i = 0; i < tunnel_count; ++i) {
 289                 const config& t = cfg.child("tunnel", i);
 290                 if(!t["saved"].to_bool()) {
 291                         lg::wml_error() << "Do not use [tunnel] directly in a [scenario]. Use it in an [event] or [abilities] tag.\n";
 292                         continue;
 293                 }
 294                 const teleport_group tunnel(t);
 295                 this->add(tunnel);
 296         }
 297 }

 303 void manager::remove(const std::string &id) {
 304         std::vector<teleport_group>::iterator t = tunnels_.begin();
 305         for(;t != tunnels_.end();) {
 306                 if (t->get_teleport_id() == id || t->get_teleport_id() == id + reversed_suffix) {
 307                         t = tunnels_.erase(t);
 308                 } else {
 309                         ++t;
 310                 }
 311         }
 312 }

 318 config manager::to_config() const {
 319         config store;
 320 
 321         std::vector<teleport_group>::const_iterator tunnel = tunnels_.begin();
 322         for(; tunnel != tunnels_.end(); ++tunnel) {
 323                 store.add_child("tunnel", tunnel->to_config());
 324         }
 325         store["next_teleport_group_id"] = std::to_string(id_);
 326 
 327         return store;
