 117 bool set_log_domain_severity(const std::string& name, int severity)
 118 {
 119         std::string::size_type s = name.size();
 120         if (name == "all") {
 121                 for(logd &l : *domains) {
 122                         l.second = severity;
 123                 }
 124         } else if (s > 2 && name.compare(s - 2, 2, "/*") == 0) {
 125                 for(logd &l : *domains) {
 126                         if (l.first.compare(0, s - 1, name, 0, s - 1) == 0)
 127                                 l.second = severity;
 128                 }
 129         } else {
 130                 domain_map::iterator it = domains->find(name);
 131                 if (it == domains->end())
 132                         return false;
 133                 it->second = severity;
 134         }
 135         return true;
 136 }
 137 bool set_log_domain_severity(const std::string& name, const logger &lg) {
 138         return set_log_domain_severity(name, lg.get_severity());
 139 }

 141 bool get_log_domain_severity(const std::string& name, int &severity)
 142 {
 143         domain_map::iterator it = domains->find(name);
 144         if (it == domains->end())
 145                 return false;
 146         severity = it->second;
 147         return true;

 181 std::string get_timespan(const time_t& t) {
 182         std::ostringstream sout;
 183         // There doesn't seem to be any library function for this
 184         const time_t minutes = t / 60;
 185         const time_t days = minutes / 60 / 24;
 186         if(t <= 0) {
 187                 sout << "expired";
 188         } else if(minutes == 0) {
 189                 sout << t << " seconds";
 190         } else if(days == 0) {
 191                 sout << minutes / 60 << " hours, " << minutes % 60 << " minutes";
 192         } else {
 193                 sout << days << " days, " << (minutes / 60) % 24 << " hours, " << minutes % 60 << " minutes";
 194         }
 195         return sout.str();
 196 }

 207 std::ostream &logger::operator()(log_domain const &domain, bool show_names, bool do_indent) const
 208 {
 209         if (severity_ > domain.domain_->second) {
 210                 return null_ostream;
 211         } else {
 212                 if (!strict_threw_ && (severity_ <= strict_level_)) {
 213                         std::stringstream ss;
 214                         ss << "Error (strict mode, strict_level = " << strict_level_ << "): wesnoth reported on channel " << name_ << " " << domain.domain_->first;
 215                         std::cerr << ss.str() << std::endl;
 216                         strict_threw_ = true;
 217                 }
 218                 std::ostream& stream = output();
 219                 if(do_indent) {
 220                         for(int i = 0; i != indent; ++i)
 221                                 stream << "  ";
 222                         }
 223                 if (timestamp) {
 224                         if(precise_timestamp) {
 225                                 print_precise_timestamp(stream);
 226                         } else {
 227                                 stream << get_timestamp(time(nullptr));
 228                         }
 229                 }
 230                 if (show_names) {
 231                         stream << name_ << ' ' << domain.domain_->first << ": ";
 232                 }
 233                 return stream;
 234         }
 235 }

 237 void scope_logger::do_log_entry(log_domain const &domain, const std::string& str)
 238 {
 239         output_ = &debug()(domain, false, true);
 240         str_ = str;
 241         ticks_ = boost::posix_time::microsec_clock::local_time();
 242         (*output_) << "{ BEGIN: " << str_ << "\n";
 243         ++indent;
 244 }
 245 
 246 void scope_logger::do_log_exit()
 247 {
 248         const long ticks = (boost::posix_time::microsec_clock::local_time() - ticks_).total_milliseconds();
 249         --indent;
 250         do_indent();
 251         if (timestamp) (*output_) << get_timestamp(time(nullptr));
 252         (*output_) << "} END: " << str_ << " (took " << ticks << "ms)\n";
 253 }
 254 
 255 void scope_logger::do_indent() const
 256 {
 257         for(int i = 0; i != indent; ++i)
 258                 (*output_) << "  ";
