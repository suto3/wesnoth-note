 121 side_actions_container::iterator side_actions_container::push_front(size_t turn, action_ptr action){
 122         if(turn_size(turn) == 0) {
 123                 return queue(turn, action);
 124         }
 125 
 126         iterator res = insert(turn_begin(turn), action);
 127         if(res != end()) {
 128                 bool current_turn_unplanned = turn_size(0) == 0;
 129                 turn_beginnings_[turn] = res;
 130 
 131                 if(current_turn_unplanned && turn == 1) {
 132                         turn_beginnings_.front() = res;
 133                 }
 134         }
 135         return res;
 136 }

 138 side_actions_container::iterator side_actions_container::insert(iterator position, action_ptr action)
 139 {
 140         assert(position <= end());
 141 
 142         bool first = position == begin();
 143 
 144         std::pair<iterator,bool> res = actions_.insert(position, action);
 145         if(!res.second) {
 146                 return end();
 147         }
 148         if(first) {
 149                 // If we are inserting before the first action, then the inserted action should became the first of turn 0.
 150                 turn_beginnings_.front() = begin();
 151         }
 152         return res.first;
 153 }

 155 side_actions_container::iterator side_actions_container::queue(size_t turn_num, action_ptr action)
 156 {
 157         // Are we inserting an action in the future while the current turn is empty?
 158         // That is, are we in the sole case where an empty turn can be followed by a non-empty one.
 159         bool future_only = turn_num == 1 && num_turns() == 0;
 160 
 161         bool current_turn_unplanned = turn_size(0) == 0;
 162 
 163         //for a little extra safety, since we should never resize by much at a time
 164         assert(turn_num <= num_turns() || future_only);
 165 
 166         std::pair<iterator,bool> res = actions_.insert(turn_end(turn_num), action);
 167         if(!res.second) {
 168                 return end();
 169         }
 170 
 171         if(future_only) {
 172                 // No action are planned for the current turn but we are planning an action for turn 1 (the next turn).
 173                 turn_beginnings_.push_back(res.first);
 174         }
 175         if(turn_num >= num_turns()) {
 176                 turn_beginnings_.push_back(res.first);
 177         } else if(current_turn_unplanned && turn_num == 0) {
 178                 // We are planning the first action of the current turn while others actions are planned in the future.
 179                 turn_beginnings_.front() = res.first;
 180         }
 181 
 182         return res.first;
 183 }

 203 side_actions_container::iterator side_actions_container::erase(iterator position)
 204 {
 205         //precondition
 206         assert(position < end());
 207 
 208         //prepare
 209         iterator next = position + 1;
 210         bool deleting_last_element = next == end();
 211 
 212         // pre-processing (check if position is at the beginning of a turn)
 213         action_limits::iterator beginning = std::find(turn_beginnings_.begin(), turn_beginnings_.end(), position);
 214         if(beginning != turn_beginnings_.end()) {
 215                 if(deleting_last_element) {
 216                         if(size() == 1) {
 217                                 // If we are deleting our sole action, we can clear turn_beginnings_ (and we have to if this last action is in turn 1)
 218                                 turn_beginnings_.clear();
 219                         } else {
 220                                 // Otherwise, we just delete the last turn
 221                                 turn_beginnings_.pop_back();
 222                         }
 223                 } else {
 224                         size_t turn_of_position = beginning - turn_beginnings_.begin();
 225                         // If we still have action this turn
 226                         if(get_turn(next) == turn_of_position) {
 227                                 *beginning = next; // We modify the beginning of the turn
 228                         } else {
 229                                 assert(turn_of_position == 0);
 230                                 *beginning = turn_end(0); // Otherwise, we are emptying the current turn.
 231                         }
 232                 }
 233         }
 234 
 235         //erase!
 236         return actions_.erase(position);
 237 }

 267 void side_actions::get_numbers(const map_location& hex, numbers_t& result)
 268 {
 269         if(empty()) {
 270                 return;
 271         }
 272 
 273         std::vector<int>& numbers_to_draw = result.numbers_to_draw;
 274         std::vector<size_t>& team_numbers = result.team_numbers;
 275         int& main_number = result.main_number;
 276         std::set<size_t>& secondary_numbers = result.secondary_numbers;
 277         std::shared_ptr<highlighter> hlighter = resources::whiteboard->get_highlighter().lock();
 278 
 279         for(const_iterator it = begin(); it != end(); ++it) {
 280                 if((*it)->is_numbering_hex(hex)) {
 281                         //store number corresponding to iterator's position + 1
 282                         size_t number = (it - begin()) + 1;
 283                         size_t index = numbers_to_draw.size();
 284                         numbers_to_draw.push_back(number);
 285                         team_numbers.push_back(team_index());
 286 
 287                         if(hlighter) {
 288                                 if(hlighter->get_main_highlight().lock() == *it) {
 289                                         main_number = index;
 290                                 }
 291 
 292                                 for(weak_action_ptr action : hlighter->get_secondary_highlights()) {
 293                                         if(action.lock() == *it) {
 294                                                 secondary_numbers.insert(index);
 295                                         }
 296                                 }
 297                         }
 298                 }
 299         }
 300 }

 311 bool side_actions::execute(side_actions::iterator position)
 312 {
 313         if(resources::whiteboard->has_planned_unit_map()) {
 314                 ERR_WB << "Modifying action queue while temp modifiers are applied!!!" << std::endl;
 315         }
 316 
 317         if(actions_.empty() || position == actions_.end()) {
 318                 return false;
 319         }
 320 
 321         assert(position < turn_end(0)); //can't execute actions from future turns
 322 
 323         LOG_WB << "Before execution, " << *this << "\n";
 324 
 325         action_ptr action = *position;
 326 
 327         if(!action->valid()) {
 328                 LOG_WB << "Invalid action sent to execution, deleting.\n";
 329                 synced_erase(position);
 330                 return true;
 331         }
 332 
 333         bool action_successful;
 334         // Determines whether action should be deleted. Interrupted moves return action_complete == false.
 335         bool action_complete;
 336         try     {
 337                  action->execute(action_successful, action_complete);
 338         } catch (return_to_play_side_exception&) {
 339                 synced_erase(position);
 340                 LOG_WB << "End turn exception caught during execution, deleting action. " << *this << "\n";
 341                 //validate actions at next map rebuild
 342                 resources::whiteboard->on_gamestate_change();
 343                 throw;
 344         }
 345 
 346         if(resources::whiteboard->should_clear_undo()) {
 347                 resources::undo_stack->clear();
 348         }
 349 
 350         std::stringstream ss;
 351         ss << "After " << (action_successful? "successful": "failed") << " execution ";
 352         if(action_complete) {
 353                 ss << "with deletion, ";
 354                 synced_erase(position);
 355         }
 356         else { //action may have revised itself; let's tell our allies.
 357                 ss << "without deletion, ";
 358                 resources::whiteboard->queue_net_cmd(team_index_,make_net_cmd_replace(position,*position));
 359 
 360                 //Idea that needs refining: move action at the end of the queue if it failed executing:
 361                         //actions_.erase(position);
 362                         //actions_.insert(end(), action);
 363         }
 364         ss << *this << "\n";
 365         LOG_WB << ss.str();
 366 
 367         resources::whiteboard->validate_viewer_actions();
 368         return action_successful;
 369 }

 421         /**
 422          * Check whether a move is swapable with a given action.
 423          */
 424         struct swapable_with_move: public visitor
 425         {
 426         public:
 427                 swapable_with_move(side_actions &sa, side_actions::iterator position, move_ptr second): sa_(sa), valid_(false), position_(position), second_(second) {}
 428                 bool valid() const { return valid_; }
 429 
 430                 void visit(move_ptr first) {
 431                         valid_ = second_->get_dest_hex() != first->get_source_hex();
 432                 }
 433 
 434                 void visit(attack_ptr first) {
 435                         visit(std::static_pointer_cast<move>(first));
 436                 }
 437 
 438                 void visit(recruit_ptr first) {
 439                         check_recruit_recall(first->get_recruit_hex());
 440                 }
 441 
 442                 void visit(recall_ptr first) {
 443                         check_recruit_recall(first->get_recall_hex());
 444                 }
 445 
 446                 void visit(suppose_dead_ptr) {
 447                         valid_ = true;
 448                 }
 449 
 450         private:
 451                 side_actions &sa_;
 452                 bool valid_;
 453                 side_actions::iterator position_;
 454                 move_ptr second_;
 455 
 456                 void check_recruit_recall(const map_location &loc) {
 457                         const unit_const_ptr leader = second_->get_unit();
 458                         if(leader->can_recruit() && dynamic_cast<game_state*>(resources::filter_con)->can_recruit_on(*leader, loc)) {
 459                                 if(const unit_const_ptr backup_leader = find_backup_leader(*leader)) {
 460                                         side_actions::iterator it = sa_.find_first_action_of(*backup_leader);
 461                                         if(!(it == sa_.end() || position_ < it)) {
 462                                                 return; //backup leader but he moves before us, refuse bump
 463                                         }
 464                                 } else {
 465                                         return; //no backup leader, refuse bump
 466                                 }
 467                         }
 468                         valid_ = true;
 469                 }
 470         };
 471 }

 473 //move action toward front of queue
 474 side_actions::iterator side_actions::bump_earlier(side_actions::iterator position)
 475 {
 476         if(resources::whiteboard->has_planned_unit_map()) {
 477                 ERR_WB << "Modifying action queue while temp modifiers are applied!!!" << std::endl;
 478         }
 479 
 480         assert(position <= end());
 481 
 482         //Don't allow bumping the very first action any earlier, of course.
 483         //Also, don't allow bumping an action into a previous turn queue
 484         if(actions_.position_in_turn(position) == 0) {
 485                 return end();
 486         }
 487 
 488         side_actions::iterator previous = position - 1;
 489 
 490         //Verify we're not moving an action out-of-order compared to other action of the same unit
 491         const unit_const_ptr previous_ptr = (*previous)->get_unit();
 492         const unit_const_ptr current_ptr = (*position)->get_unit();
 493         if(previous_ptr && current_ptr && previous_ptr.get() == current_ptr.get()) {
 494                 return end();
 495         }
 496 
 497         if(move_ptr second = std::dynamic_pointer_cast<move>(*position)) {
 498                 swapable_with_move check(*this, position, second);
 499                 (*previous)->accept(check);
 500                 if(!check.valid()) {
 501                         return end();
 502                 }
 503         }
 504 
 505         LOG_WB << "Before bumping earlier, " << *this << "\n";
 506 
 507         int turn_number = get_turn(position);
 508         int action_number = actions_.position_in_turn(position);
 509         int last_position = turn_size(turn_number) - 1;
 510         LOG_WB << "In turn #" << turn_number
 511                         << ", bumping action #" << action_number << "/" << last_position
 512                         << " to position #" << action_number - 1  << "/" << last_position << ".\n";
 513 
 514         resources::whiteboard->queue_net_cmd(team_index_, make_net_cmd_bump_later(position - 1));
 515 
 516         actions_.bump_earlier(position);
 517 
 518         LOG_WB << "After bumping earlier, " << *this << "\n";
 519         return position - 1;
 520 }

 522 //move action toward back of queue
 523 side_actions::iterator side_actions::bump_later(side_actions::iterator position)
 524 {
 525         assert(position < end());
 526 
 527         ++position;

 529                 return end();
 530         }
 531         position = bump_earlier(position);
 532         if(position == end()) {
 533                 return end();
 534         }
 535         return position + 1;

 655 side_actions::iterator side_actions::synced_enqueue(size_t turn_num, action_ptr act)
 656 {
 657         //raw_enqueue() creates actions_[turn_num] if it doesn't exist already, so we
 658         //have to do it first -- before subsequently calling actions_[turn_num].size().
 659         iterator result = actions_.queue(turn_num, act);
 660         if(result != end()) {
 661                 resources::whiteboard->queue_net_cmd(team_index_, make_net_cmd_insert(turn_num, turn_size(turn_num) - 1, act));
 662                 // The insert position is turn_size(turn_num)-1 since we already inserted the action.
 663         }
 664         return result;
 665 }

 705 void side_actions::execute_net_cmd(net_cmd const& cmd)
 706 {
 707         std::string type = cmd["type"];
 708 
 709         if(type=="insert") {
 710                 size_t turn = cmd["turn"].to_int();
 711                 size_t pos = cmd["pos"].to_int();
 712                 action_ptr act = action::from_config(cmd.child("action"), hidden_);
 713                 if(!act) {
 714                         ERR_WB << "side_actions::execute_network_command(): received invalid action data!" << std::endl;
 715                         return;
 716                 }
 717 
 718                 iterator itor = safe_insert(turn, pos, act);
 719                 if(itor >= end()) {
 720                         ERR_WB << "side_actions::execute_network_command(): received invalid insertion position!" << std::endl;
 721                         return;
 722                 }
 723 
 724                 LOG_WB << "Command received: action inserted on turn #" << turn << ", position #" << pos << ": " << act << "\n";
 725 
 726                 //update numbering hexes as necessary
 727                 ++itor;
 728                 for(iterator end_itor = end(); itor != end_itor; ++itor) {
 729                         resources::screen->invalidate((*itor)->get_numbering_hex());
 730                 }
 731         } else if(type=="replace") {
 732                 size_t turn = cmd["turn"].to_int();
 733                 size_t pos = cmd["pos"].to_int();
 734                 action_ptr act = action::from_config(cmd.child("action"), hidden_);
 735                 if(!act) {
 736                         ERR_WB << "side_actions::execute_network_command(): received invalid action data!" << std::endl;
 737                         return;
 738                 }
 739 
 740                 iterator itor = turn_begin(turn) + pos;
 741                 if(itor >= end() || get_turn(itor) != turn) {
 742                         ERR_WB << "side_actions::execute_network_command(): received invalid pos!" << std::endl;
 743                         return;
 744                 }
 745 
 746                 if(!actions_.replace(itor, act)){
 747                         ERR_WB << "side_actions::execute_network_command(): replace failed!" << std::endl;
 748                         return;
 749                 }
 750 
 751                 LOG_WB << "Command received: action replaced on turn #" << turn << ", position #" << pos << ": " << act << "\n";
 752         } else if(type=="remove") {
 753                 size_t turn = cmd["turn"].to_int();
 754                 size_t pos = cmd["pos"].to_int();
 755 
 756                 iterator itor = turn_begin(turn) + pos;
 757                 if(itor >= end() || get_turn(itor) != turn) {
 758                         ERR_WB << "side_actions::execute_network_command(): received invalid pos!" << std::endl;
 759                         return;
 760                 }
 761 
 762                 itor = safe_erase(itor);
 763 
 764                 LOG_WB << "Command received: action removed on turn #" << turn << ", position #" << pos << "\n";
 765 
 766                 //update numbering hexes as necessary
 767                 for(iterator end_itor = end(); itor != end_itor; ++itor) {
 768                         resources::screen->invalidate((*itor)->get_numbering_hex());
 769                 }
 770         } else if(type=="bump_later") {
 771                 size_t turn = cmd["turn"].to_int();
 772                 size_t pos = cmd["pos"].to_int();
 773 
 774                 iterator itor = turn_begin(turn) + pos;
 775                 if(itor+1 >= end() || get_turn(itor) != turn) {
 776                         ERR_WB << "side_actions::execute_network_command(): received invalid pos!" << std::endl;
 777                         return;
 778                 }
 779 
 780                 action_ptr first_action = *itor;
 781                 action_ptr second_action = itor[1];
 782                 bump_later(itor);
 783 
 784                 LOG_WB << "Command received: action bumped later from turn #" << turn << ", position #" << pos << "\n";
 785 
 786                 //update numbering hexes as necessary
 787                 resources::screen->invalidate(first_action->get_numbering_hex());
 788                 resources::screen->invalidate(second_action->get_numbering_hex());
 789         } else if(type=="clear") {
 790                 LOG_WB << "Command received: clear\n";
 791                 clear();
 792         } else if(type=="refresh") {
 793                 LOG_WB << "Command received: refresh\n";
 794                 clear();
 795                 for(net_cmd const& sub_cmd : cmd.child_range("net_cmd"))
 796                         execute_net_cmd(sub_cmd);
 797         } else {
 798                 ERR_WB << "side_actions::execute_network_command(): received invalid type!" << std::endl;
 799                 return;
 800         }
 801 
 802         resources::whiteboard->validate_viewer_actions();
 803 }

 866 void side_actions::raw_turn_shift()
 867 {
 868         //find units who still have plans for turn 0 (i.e. were too lazy to finish their jobs)
 869         std::set<unit_const_ptr> lazy_units;
 870         for(action_ptr const& act : iter_turn(0)) {
 871                 unit_const_ptr u = act->get_unit();
 872                 if(u) {
 873                         lazy_units.insert(u);
 874                 }
 875         }
 876 
 877         //push their plans back one turn
 878         std::set<unit_const_ptr>::iterator lazy_end = lazy_units.end();
 879         iterator itor = end();
 880         while(itor != begin()) {
 881                 --itor;
 882                 action_ptr act = *itor;
 883 
 884                 if(lazy_units.find(act->get_unit()) != lazy_end) {
 885                         safe_insert(get_turn(itor)+1, 0, act);
 886                         itor = actions_.erase(itor);
 887                 }
 888         }
 889 
 890         //push any remaining first-turn plans into the second turn
 891         for(iterator act=turn_begin(0), end=turn_end(0); act!=end; ++act) {
 892                 safe_insert(1, 0, *act);
 893         }
 894 
 895         //shift everything forward one turn
 896         actions_.erase(turn_begin(0), turn_end(0));
 897         actions_.turn_shift();
 898 }

 900 void side_actions::synced_turn_shift()
 901 {
 902         raw_turn_shift();
 903         resources::whiteboard->queue_net_cmd(team_index(), make_net_cmd_refresh());
 904 }
