 114     /**
 115      * Adds a new node in the PDL
 116      * 
 117      * @param file
 118      *        The file to add
 119      * @return The newly created node
 120      */
 121     public DependencyListNode addNode( IFile file )
 122     {
 123         // TODO: create better heuristics for the indexes
 124         // in case we need to add multiple subsequent files...
 125         DependencyListNode backupPrevious = previous_, newNode = null;
 126         String fileParentProjectPath = file.getParent( )
 127             .getProjectRelativePath( ).toString( );
 128         if( ! fileParentProjectPath.isEmpty( ) ) {
 129             fileParentProjectPath = "/" + fileParentProjectPath;
 130         }
 131 
 132         // we add a file in an existing processed directory.
 133         int dirEntryIndex = directories_.indexOf( fileParentProjectPath );
 134         if( dirEntryIndex != - 1 ) {
 135 
 136             DirectoryIncludeEntry entry = directoriesEntries_
 137                 .get( dirEntryIndex );
 138             DependencyListNode tmpNode = entry.FirstNode;
 139 
 140             // had any files in dir?
 141             if( tmpNode != null ) {
 142                 String fileName = file.getName( );
 143 
 144                 /*
 145                  * check if the file is _main.cfg. If yes,
 146                  * all current nodes from this directory need to be erased
 147                  * and processed by the includes from other _main.cfg
 148                  */
 149                 if( fileName.equals( "_main.cfg" ) ) {
 150 
 151                     // save the previous
 152                     previous_ = tmpNode.getPrevious( );
 153 
 154                     internal_removeContainer( fileParentProjectPath );
 155 
 156                     // create the node
 157                     newNode = internal_addNode( file );
 158                 }
 159                 else {
 160                     // search for the correct place in current dir
 161                     DependencyListNode prevTmpNode = null;
 162                     while( tmpNode != null ) {
 163 
 164                         // we found the place?
 165                         if( ResourceUtils.wmlFileNameCompare( tmpNode.getFile( )
 166                             .getName( ), fileName ) > 0 ) {
 167 
 168                             previous_ = tmpNode.getPrevious( );
 169 
 170                             newNode = internal_addNode( file );
 171                             break;
 172                         }
 173 
 174                         prevTmpNode = tmpNode;
 175                         tmpNode = tmpNode.getNext( );
 176                     }
 177 
 178                     // we arrived at the end
 179                     if( newNode == null ) {
 180                         previous_ = prevTmpNode;
 181                         newNode = internal_addNode( file );
 182                     }
 183                 }
 184             }
 185             else {
 186                 // previous_ should be the first non-null node in previous
 187                 // directories.
 188 
 189                 while( dirEntryIndex > 0 && tmpNode == null ) {
 190                     --dirEntryIndex;
 191                     tmpNode = directoriesEntries_.get( dirEntryIndex ).FirstNode;
 192                 }
 193 
 194                 previous_ = tmpNode;
 195                 newNode = internal_addNode( file );
 196             }
 197 
 198             // now, parse the file to check if we should include other
 199             // dirs/files
 200             internal_addIncludes( newNode.getMacroIncludes( true ) );
 201         }
 202         else {
 203             // didn't found any place to put it. where shall we?
 204             // the place should be dictated by other cfg via a macro include
 205         }
 206 
 207         // restore old previous
 208         previous_ = backupPrevious;
 209 
 210         return newNode;
 211     }

 234     {

 236         if( file.exists( ) ) {
 237             DependencyListNode node = internal_addNode( file );

 246     /**
 247      * Add the container and it's contents to the list
 248      * 
 249      * @param containerPath
 250      *        The path of the container
 251      */
 252     private void internal_addContainer( String containerPath )
 253     {
 254         IContainer container = null;
 255         if( StringUtils.isNullOrEmpty( containerPath ) ) {
 256             container = project_;
 257         }
 258         else {
 259             container = project_.getFolder( containerPath );
 260         }
 261 
 262         // skip core library
 263         if( container.getName( ).equals(
 264             WesnothProjectsExplorer.CORE_LIBRARY_NAME ) ) {
 265             return;
 266         }
 267 
 268         IResource main_cfg = container.findMember( "_main.cfg" ); //$NON-NLS-1$
 269         if( main_cfg != null ) {
 270             // add main.cfg to list
 271             DependencyListNode newNode = internal_addNode( ( IFile ) main_cfg );
 272 
 273             // add any included files/folders
 274             internal_addIncludes( newNode.getMacroIncludes( true ) );
 275         }
 276         else {
 277             // no main.cfg, just follow WML reading rules
 278             List< IResource > members = null;
 279             try {
 280                 members = Arrays.asList( container.members( ) );
 281             } catch( CoreException e ) {
 282                 Logger.getInstance( ).logException( e );
 283                 return;
 284             }
 285 
 286             Collections.sort( members, new WMLFilesComparator( ) );
 287 
 288             boolean toAddDirectoryEntry = false;
 289             if( ! directories_.contains( containerPath ) ) {
 290                 directories_.add( containerPath );
 291                 directoriesEntries_.add( new DirectoryIncludeEntry(
 292                     containerPath,
 293                     null, null ) );
 294 
 295                 toAddDirectoryEntry = true;
 296             }
 297             else {
 298                 // update the includes
 299                 for( DirectoryIncludeEntry entry: directoriesEntries_ ) {
 300                     if( entry.DirectoryPath.equals( containerPath ) ) {
 301                         ++entry.Includes;
 302                         break;
 303                     }
 304                 }
 305             }
 306 
 307             if( members.isEmpty( ) ) {
 308                 return;
 309             }
 310 
 311             DependencyListNode firstNewNode = null;
 312             DependencyListNode lastNode = null;
 313 
 314             for( IResource resource: members ) {
 315                 if( resource instanceof IContainer ) {
 316                     internal_addContainer( resource.getProjectRelativePath( )
 317                         .toString( ) );
 318                 }
 319                 else {
 320                     // just config files.
 321                     if( ! ResourceUtils.isConfigFile( resource ) ) {
 322                         continue;
 323                     }
 324 
 325                     lastNode = internal_addNode( ( IFile ) resource );
 326                     if( firstNewNode == null ) {
 327                         firstNewNode = lastNode;
 328                     }
 329                 }
 330             }
 331 
 332             if( firstNewNode != null && lastNode != null ) {
 333                 if( toAddDirectoryEntry ) {
 334                     // update the first directory node
 335                     directoriesEntries_.set( directories_.size( ) - 1,
 336                         new DirectoryIncludeEntry( containerPath,
 337                             firstNewNode, lastNode ) );
 338                 }
 339                 else {
 340                     // if the current entry has null nodes
 341                     // or the indexes are greater/lower than the current ones
 342                     // we need to update the references nodes
 343 
 344                     DirectoryIncludeEntry entry = directoriesEntries_
 345                         .get( directories_.indexOf( containerPath ) );
 346 
 347                     if( entry.FirstNode == null
 348                         || firstNewNode.getIndex( ) < entry.FirstNode
 349                             .getIndex( ) ) {
 350                         entry.FirstNode = firstNewNode;
 351                     }
 352 
 353                     if( entry.LastNode == null
 354                         || lastNode.getIndex( ) > entry.LastNode.getIndex( ) ) {
 355                         entry.LastNode = lastNode;
 356                     }
 357                 }
 358             }
 359         }
 360     }

 362     /**
 363      * Adds a new node to this list
 364      * 
 365      * @param file
 366      *        The file to add
 367      * @return The newly created node
 368      */
 369     private DependencyListNode internal_addNode( IFile file )
 370     {
 371         // don't add a file more than 1 time
 372         if( list_.containsKey( file.getProjectRelativePath( ).toString( ) ) ) {
 373             return getNode( file );
 374         }
 375 
 376         DependencyListNode newNode = new DependencyListNode( file, - 1 );
 377 
 378         if( previous_ != null ) {
 379 
 380             // inserting is done between 2 nodes
 381             if( previous_.getNext( ) != null ) {
 382                 int newIndex = ( previous_.getIndex( ) + previous_.getNext( )
 383                     .getIndex( ) ) / 2;
 384 
 385                 if( newIndex > previous_.getIndex( )
 386                     + DependencyListNode.INDEX_STEP ) {
 387                     newIndex = previous_.getIndex( )
 388                         + DependencyListNode.INDEX_STEP;
 389                 }
 390 
 391                 newNode.setIndex( newIndex );
 392 
 393                 newNode.setNext( previous_.getNext( ) );
 394                 previous_.getNext( ).setPrevious( newNode );
 395             }
 396             else {
 397                 newNode.setIndex( currentIndex_ );
 398                 currentIndex_ += DependencyListNode.INDEX_STEP;
 399             }
 400 
 401             previous_.setNext( newNode );
 402             newNode.setPrevious( previous_ );
 403         }
 404         else {
 405             // no previous yet (== null)
 406             // so we're making this the root node for this list
 407 
 408             // check if we had a previous root node
 409             DependencyListNode root = list_.get( ROOT_NODE_KEY );
 410             if( root != null ) {
 411                 root.setPrevious( newNode );
 412                 newNode.setNext( root );
 413 
 414                 newNode.setIndex( root.getIndex( )
 415                     - DependencyListNode.INDEX_STEP );
 416             }
 417             else {
 418                 newNode.setIndex( currentIndex_ );
 419                 currentIndex_ += DependencyListNode.INDEX_STEP;
 420             }
 421 
 422             list_.put( ROOT_NODE_KEY, newNode );
 423         }
 424 
 425         list_.put( file.getProjectRelativePath( ).toString( ), newNode );
 426         previous_ = newNode;
 427         return newNode;
 428     }

 441     /**
 442      * Removes the specified node from the list
 443      * 
 444      * @param node
 445      *        The node to remove from the list
 446      */
 447     public void removeNode( DependencyListNode node )
 448     {
 449         // the node didn't even exist in the list!?
 450         if( node == null ) {
 451             return;
 452         }
 453 
 454         if( node.getPrevious( ) != null ) {
 455             node.getPrevious( ).setNext( node.getNext( ) );
 456         }
 457         if( node.getNext( ) != null ) {
 458             node.getNext( ).setPrevious( node.getPrevious( ) );
 459         }
 460 
 461         IFile file = node.getFile( );
 462 
 463         if( node.equals( list_.get( ROOT_NODE_KEY ) ) ) {
 464             list_.remove( ROOT_NODE_KEY );
 465         }
 466 
 467         list_.remove( file.getProjectRelativePath( ).toString( ) );
 468 
 469         // if we're at last node, decrease currentIndex_ to make economy on
 470         // indexes
 471         if( node.getNext( ) == null ) {
 472             if( node.getPrevious( ) != null ) {
 473                 currentIndex_ = node.getPrevious( ).getIndex( )
 474                     + DependencyListNode.INDEX_STEP;
 475             }
 476         }
 477 
 478         // if removing a _main.cfg, we need to add the parent container
 479         // back to the list along with it's directories_ entry
 480         if( file.getName( ).equals( "_main.cfg" ) ) {
 481             DependencyListNode backupPrevious = previous_;
 482 
 483             previous_ = node.getPrevious( );
 484             internal_addContainer( file.getParent( ).getProjectRelativePath( )
 485                 .toString( ) );
 486 
 487             previous_ = backupPrevious;
 488         }
 489     }

 508     /**
 509      * Removes the container and all it's contents from the list
 510      * 
 511      * @param path
 512      *        The container's path
 513      */
 514     private void internal_removeContainer( String path )
 515     {
 516         int dirEntryIndex = directories_.indexOf( path );
 517         if( dirEntryIndex == - 1 ) {
 518             return;
 519         }
 520 
 521         DirectoryIncludeEntry entry = directoriesEntries_.get( dirEntryIndex );
 522 
 523         if( entry == null ) {
 524             return;
 525         }
 526 
 527         --entry.Includes;
 528 
 529         // we shouldn't delete entries that are used more than 1 time
 530         if( entry.Includes > 0 ) {
 531             return;
 532         }
 533 
 534         DependencyListNode firstNode = entry.FirstNode;
 535         DependencyListNode lastNode = entry.LastNode.getNext( );
 536 
 537         // now delete all nodes that are in the same folder
 538         while( firstNode != null && firstNode != lastNode ) {
 539             removeNode( firstNode );
 540             firstNode = firstNode.getNext( );
 541         }
 542 
 543         // clear the directories entry since we can't use it anymore
 544         directories_.remove( dirEntryIndex );
 545         directoriesEntries_.remove( dirEntryIndex );
 546     }

 548     /**
 549      * Updates the specified node in the list
 550      * 
 551      * @param node
 552      *        The node to update
 553      */
 554     public void updateNode( DependencyListNode node )
 555     {
 556         // check the includes, to see if they changed
 557         List< String > previousIncludes = node.getMacroIncludes( false );
 558         int prevLength = previousIncludes.size( );
 559 
 560         List< String > newIncludes = node.getMacroIncludes( true );
 561         int newLength = newIncludes.size( );
 562 
 563         List< String > processedIncludes = new ArrayList< String >( );
 564 
 565         for( int prevIndex = 0, newIndex = 0; prevIndex < prevLength
 566             || newIndex < newLength; ) {
 567             String prevInclude = null;
 568             String newInclude = null;
 569             if( prevIndex < prevLength ) {
 570                 prevInclude = previousIncludes.get( prevIndex );
 571             }
 572             if( newIndex < newLength ) {
 573                 newInclude = newIncludes.get( newIndex );
 574             }
 575 
 576             // nothing changed
 577             if( prevInclude != null &&
 578                 prevInclude.equals( newInclude ) ) {
 579                 ++prevIndex;
 580                 ++newIndex;
 581                 continue;
 582             }
 583 
 584             boolean newIsNew = prevIndex >= prevLength ||
 585                 ( newInclude != null &&
 586                 ! previousIncludes.contains( newInclude ) );
 587             boolean prevDeleted = newIndex >= newLength ||
 588                 ( prevInclude != null &&
 589                 ! newIncludes.contains( prevInclude ) );
 590 
 591             if( newIsNew ) {
 592                 // add the new include before the previous included dir (if
 593                 // any)
 594 
 595                 DependencyListNode backupPrevious = previous_;
 596 
 597                 previous_ = null;
 598 
 599                 // find the node for the previous include
 600                 if( newIndex > 0 ) {
 601 
 602                     int dirIndex = directories_.indexOf( newIncludes
 603                         .get( newIndex - 1 ) );
 604 
 605                     if( dirIndex != - 1 ) {
 606                         // previous include was a directory
 607                         DirectoryIncludeEntry entry = directoriesEntries_
 608                             .get( dirIndex );
 609 
 610                         if( entry != null ) {
 611                             previous_ = entry.LastNode;
 612                         }
 613                     }
 614                     else {
 615                         previous_ = fileIncludes_.get( newIncludes
 616                             .get( newIndex - 1 ) );
 617                     }
 618                 }
 619 
 620                 internal_addInclude( newInclude );
 621 
 622                 previous_ = backupPrevious;
 623                 ++newIndex;
 624             }
 625             else if( prevDeleted ) {
 626                 // the previous include was deleted
 627                 // it's not present in the new includes list
 628                 internal_removeInclude( prevInclude );
 629 
 630                 ++prevIndex;
 631             }
 632             else {
 633                 // the previous include has changed it's index (in the
 634                 // includes list)
 635                 // there are 3 types of swaps:
 636                 // 1 ) file - file
 637                 // 2 ) directory - file
 638                 // 3 ) directory - directory
 639 
 640                 // don't reprocess this pair if we already did
 641                 if( ! processedIncludes.contains( prevInclude ) ) {
 642 
 643                     // see what type of swap we need to do.
 644 
 645                     int prevDirIndex = directories_.indexOf( prevInclude );
 646                     int newDirIndex = directories_.indexOf( newInclude );
 647 
 648                     if( prevDirIndex == - 1 && newDirIndex == - 1 ) {
 649                         // file <-> file. Just swap the nodes
 650                         DependencyListNode prevNode = fileIncludes_
 651                             .get( prevInclude );
 652                         DependencyListNode newNode = fileIncludes_
 653                             .get( newInclude );
 654 
 655                         DependencyListNode tmpSwapNode = prevNode.getPrevious( );
 656                         prevNode.setPrevious( newNode.getPrevious( ) );
 657                         if( newNode.getPrevious( ) != null ) {
 658                             newNode.getPrevious( ).setNext( prevNode );
 659                         }
 660                         newNode.setPrevious( tmpSwapNode );
 661                         if( tmpSwapNode != null ) {
 662                             tmpSwapNode.setNext( newNode );
 663                         }
 664 
 665                         tmpSwapNode = prevNode.getNext( );
 666                         prevNode.setNext( newNode.getNext( ) );
 667                         if( newNode.getNext( ) != null ) {
 668                             newNode.getNext( ).setPrevious( prevNode );
 669                         }
 670                         newNode.setNext( tmpSwapNode );
 671                         if( tmpSwapNode != null ) {
 672                             tmpSwapNode.setPrevious( newNode );
 673                         }
 674                     }
 675                     else {
 676                         // directory <-> directory and
 677                         // directory <-> file
 678                         // This is code is a bit ugly and weird, but for the
 679                         // moment I haven't found any better solution
 680                         // If you want to understand it, you'd better use
 681                         // a pen and paper. That's what I did when writing it!
 682                         DirectoryIncludeEntry prevEntry = null;
 683                         DirectoryIncludeEntry newEntry = null;
 684                         DependencyListNode prevNode = null;
 685                         DependencyListNode newNode = null;
 686                         DependencyListNode middleNode = null;
 687 
 688                         boolean prevIsDir = false, newIsDir = false;
 689 
 690                         // create a list for easier swap
 691                         List< DependencyListNode > nodes = new ArrayList< DependencyListNode >( );
 692 
 693                         if( prevDirIndex != - 1 ) {
 694                             prevIsDir = true;
 695                             prevEntry = directoriesEntries_.get( prevDirIndex );
 696 
 697                             nodes.add( prevEntry.FirstNode.getPrevious( ) );
 698                             nodes.add( prevEntry.FirstNode );
 699                             nodes.add( prevEntry.LastNode );
 700 
 701                             middleNode = prevEntry.LastNode.getNext( );
 702                         }
 703                         else {
 704                             prevNode = fileIncludes_.get( prevInclude );
 705 
 706                             nodes.add( prevNode.getPrevious( ) );
 707                             nodes.add( prevNode );
 708 
 709                             middleNode = prevNode.getNext( );
 710                         }
 711 
 712                         if( newDirIndex != - 1 ) {
 713                             newIsDir = true;
 714                             newEntry = directoriesEntries_.get( newDirIndex );
 715 
 716                             if( middleNode != newEntry.FirstNode ) {
 717                                 nodes.add( middleNode );
 718                                 nodes.add( newEntry.FirstNode.getPrevious( ) );
 719                             }
 720 
 721                             nodes.add( newEntry.FirstNode );
 722                             nodes.add( newEntry.LastNode );
 723                             nodes.add( newEntry.LastNode.getNext( ) );
 724                         }
 725                         else {
 726                             newNode = fileIncludes_.get( newInclude );
 727 
 728                             if( middleNode != newNode ) {
 729                                 nodes.add( middleNode );
 730                                 nodes.add( newNode.getPrevious( ) );
 731                             }
 732 
 733                             nodes.add( newNode );
 734                             nodes.add( newNode.getNext( ) );
 735                         }
 736 
 737                         int nodesSize = nodes.size( );
 738 
 739                         // now swap the nodes
 740 
 741                         if( prevIsDir && newIsDir ) {
 742                             // dir <-> dir
 743                             int swapIndex = ( nodesSize == 8 ? 5: 3 );
 744 
 745                             DependencyListNode tmp = nodes.get( swapIndex );
 746                             nodes.set( swapIndex, nodes.get( 1 ) );
 747                             nodes.set( 1, tmp );
 748 
 749                             swapIndex = ( nodesSize == 8 ? 6: 4 );
 750 
 751                             tmp = nodes.get( swapIndex );
 752                             nodes.set( swapIndex, nodes.get( 2 ) );
 753                             nodes.set( 2, tmp );
 754 
 755                             // update the links
 756                             for( int i = 0; i < nodesSize - 1; i += 2 ) {
 757                                 DependencyListNode fst = nodes.get( i );
 758                                 DependencyListNode lst = nodes.get( i + 1 );
 759 
 760                                 if( fst != null ) {
 761                                     fst.setNext( lst );
 762                                 }
 763 
 764                                 if( lst != null ) {
 765                                     lst.setPrevious( fst );
 766                                 }
 767                             }
 768                         }
 769                         else {
 770                             // transform the list in the oposite case
 771                             if( prevIsDir ) {
 772                                 Collections.reverse( nodes );
 773                             }
 774 
 775                             List< DependencyListNode > tmpNodes =
 776                                 new ArrayList< DependencyListNode >( nodes );
 777 
 778                             if( nodesSize == 7 ) {
 779                                 nodes.set( 1, tmpNodes.get( 4 ) );
 780                                 nodes.set( 2, tmpNodes.get( 5 ) );
 781                                 nodes.set( 3, tmpNodes.get( 2 ) );
 782                                 nodes.set( 4, tmpNodes.get( 3 ) );
 783                                 nodes.set( 5, tmpNodes.get( 1 ) );
 784                             }
 785                             else {
 786                                 nodes.set( 1, tmpNodes.get( 2 ) );
 787                                 nodes.set( 2, tmpNodes.get( 3 ) );
 788                                 nodes.set( 3, tmpNodes.get( 1 ) );
 789                             }
 790 
 791                             // reverse to the original order
 792                             if( prevIsDir ) {
 793                                 Collections.reverse( nodes );
 794                             }
 795 
 796                             // update the links
 797                             for( int i = 0; i < nodesSize - 1; ) {
 798                                 DependencyListNode fst = nodes.get( i );
 799                                 DependencyListNode lst = nodes.get( i + 1 );
 800 
 801                                 if( fst != null ) {
 802                                     fst.setNext( lst );
 803                                 }
 804 
 805                                 if( lst != null ) {
 806                                     lst.setPrevious( fst );
 807                                 }
 808 
 809                                 if( i == 0 && prevIsDir ) {
 810                                     ++i;
 811                                 }
 812                                 else if( i == nodesSize - 3 && ! prevIsDir ) {
 813                                     ++i;
 814                                 }
 815                                 else {
 816                                     i += 2;
 817                                 }
 818                             }
 819                         }
 820                     }
 821                 }
 822 
 823                 processedIncludes.add( prevInclude );
 824                 processedIncludes.add( newInclude );
 825                 ++prevIndex;
 826                 ++newIndex;
 827             }
 828         }
 829     }
