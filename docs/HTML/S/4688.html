 223                 [set_variables]
 224                     name=this_guard_traits
 225                     to_variable=traits_without_quick[$first_trait].trait
 226                 [/set_variables]
 227 
 228                 # To prevent the same trait from being picked twice, we need to find
 229                 # and remove the first picked trait from the array.
 230                 [for]
 231                     array=traits_without_quick
 232                     variable=j
 233                     [do]
 234                         [if]

 236                                 name=traits_without_quick[$j].trait.id
 237                                 equals=$traits_without_quick[$first_trait].trait.id
 238                             [/variable]
 239 
 240                             [then]
 241                                 {CLEAR_VARIABLE traits_without_quick[$first_trait]}
 242                             [/then]
 243                         [/if]
 244                     [/do]
 245                 [/for]

 247                 {VARIABLE_OP second_trait rand "0..2"}
 248                 [set_variables]
 249                     name=this_guard_traits
