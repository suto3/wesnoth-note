  64  * @page = GUIWindowDefinitionWML
  65  * @order = 2_game_version
  66  *
  67  * == Game paths ==
  68  *
  69  * Dialog displaying the various paths used by the game to locate
  70  * resource and configuration files.
  71  *
  72  * There are several item types used to build widget ids in this dialog.
  73  * All references to TYPE below refer to the following suffixes:
  74  * datadir, config, userdata, saves, addons, cache.
  75  *
  76  * @begin{table}{dialog_widgets}
  77  *
  78  * path_TYPE & & text_box & m &
  79  *        Textbox containing the filesystem path for the given item. $
  80  *
  81  * copy_TYPE & & button & m &
  82  *        Copies the given item's path to clipboard. $
  83  *
  84  * browse_TYPE & & button & m &
  85  *        Launches the default file browser on the given item's path. $
  86  *
  87  * @end{table}
  88  */

  92 game_version::game_version()
  93         : path_wid_stem_("path_")
  94         , copy_wid_stem_("copy_")
  95         , browse_wid_stem_("browse_")
  96         , path_map_()
  97 #ifdef _WIN32
  98         , log_path_(lg::log_file_path())
  99 #endif
 100         , deps_()
 101         , opts_(game_config::optional_features_table())
 102         , report_()
 103 {
 104         // NOTE: these path_map_ entries are referenced by the GUI2 WML
 105         // definition of this dialog using preprocessor macros.
 106         path_map_["datadir"] = game_config::path;
 107         path_map_["config"] = filesystem::get_user_config_dir();
 108         path_map_["userdata"] = filesystem::get_user_data_dir();
 109         path_map_["saves"] = filesystem::get_saves_dir();
 110         path_map_["addons"] = filesystem::get_addons_dir();
 111         path_map_["cache"] = filesystem::get_cache_dir();
 112 
 113         for(unsigned k = 0; k < game_config::LIB_COUNT; ++k) {
 114                 const game_config::LIBRARY_ID lib = game_config::LIBRARY_ID(k);
 115 
 116                 deplist_entry e;
 117                 e[0] = game_config::library_name(lib);
 118                 if(e[0].empty()) {
 119                         continue;
 120                 }
 121                 e[1] = game_config::library_build_version(lib);
 122                 e[2] = game_config::library_runtime_version(lib);
 123                 deps_.push_back(e);
 124         }
 125 
 126         generate_plain_text_report();
 127 }

 129 void game_version::pre_show(window& window)
 130 {
 131         string_map i18n_syms;
 132 
 133         //
 134         // General information.
 135         //
 136 
 137         styled_widget& version_label = find_widget<styled_widget>(&window, "version", false);
 138         i18n_syms["version"] = game_config::revision;
 139         version_label.set_label(VGETTEXT("Version $version", i18n_syms));
 140 
 141         styled_widget& os_label = find_widget<styled_widget>(&window, "os", false);
 142         i18n_syms["os"] = desktop::os_version();
 143         os_label.set_label(VGETTEXT("Running on $os", i18n_syms));
 144 
 145         button& copy_all = find_widget<button>(&window, "copy_all", false);
 146         connect_signal_mouse_left_click(
 147                         copy_all,
 148                         std::bind(&game_version::report_copy_callback, this));
 149 
 150         //
 151         // Game paths tab.
 152         //
 153 
 154         for(const auto & path_ent : path_map_)
 155         {
 156                 const std::string& path_id = path_ent.first;
 157                 const std::string& path_path = path_ent.second;
 158 
 159                 text_box_base& path_w
 160                                 = find_widget<text_box_base>(&window, path_wid_stem_ + path_id, false);
 161                 button& copy_w = find_widget<button>(
 162                                 &window, copy_wid_stem_ + path_id, false);
 163                 button& browse_w = find_widget<button>(
 164                                 &window, browse_wid_stem_ + path_id, false);
 165 
 166                 path_w.set_value(path_path);
 167                 path_w.set_active(false);
 168 
 169                 connect_signal_mouse_left_click(
 170                                 copy_w,
 171                                 std::bind(&game_version::copy_to_clipboard_callback,
 172                                                         this,
 173                                                         path_path));
 174                 connect_signal_mouse_left_click(
 175                                 browse_w,
 176                                 std::bind(&game_version::browse_directory_callback,
 177                                                         this,
 178                                                         path_path));
 179 
 180                 if(!desktop::open_object_is_supported()) {
 181                         // No point in displaying these on platforms that can't do
 182                         // open_object().
 183                         browse_w.set_visible(widget::visibility::invisible);
 184                 }
 185 
 186                 if(!desktop::clipboard::available()) {
 187                         copy_w.set_active(false);
 188                         copy_w.set_tooltip(_("Clipboard support not found, contact your packager"));
 189                 }
 190         }
 191 
 192 #ifndef _WIN32
 193         grid& w32_options_grid
 194                         = find_widget<grid>(&window, "win32_paths", false);
 195         w32_options_grid.set_visible(widget::visibility::invisible);
 196 #else
 197         button& stderr_button
 198                         = find_widget<button>(&window, "open_stderr", false);
 199         connect_signal_mouse_left_click(
 200                         stderr_button,
 201                         std::bind(&game_version::browse_directory_callback,
 202                                                 this,
 203                                                 log_path_));
 204         stderr_button.set_active(!log_path_.empty());
 205 #endif
 206 
 207         //
 208         // Build info tab.
 209         //
 210 
 211         std::map<std::string, string_map> list_data;
 212 
 213         listbox& deps_listbox
 214                         = find_widget<listbox>(&window, "deps_listbox", false);
 215 
 216         for(const auto & dep : deps_)
 217         {
 218                 list_data["dep_name"]["label"] = dep[0];
 219 
 220                 list_data["dep_build_version"]["label"] = dep[1];
 221 
 222                 // The build version is always known, but runtime version isn't, esp.
 223                 // for header-only libraries like Boost for which the concept does not
 224                 // apply.
 225                 if(!dep[2].empty()) {
 226                         list_data["dep_rt_version"]["label"] = dep[2];
 227                 } else {
 228                         list_data["dep_rt_version"]["label"] = _("version^N/A");
 229                 }
 230 
 231                 deps_listbox.add_row(list_data);
 232         }
 233 
 234         deps_listbox.select_row(0);
 235         list_data.clear();
 236 
 237         //
 238         // Features tab.
 239         //
 240 
 241         listbox& opts_listbox
 242                         = find_widget<listbox>(&window, "opts_listbox", false);
 243 
 244         for(const auto & opt : opts_)
 245         {
 246                 list_data["opt_name"]["label"] = opt.name;
 247 
 248                 if(opt.enabled) {
 249                         list_data["opt_status"]["label"] = text_feature_on;
 250                 } else {
 251                         list_data["opt_status"]["label"] = text_feature_off;
 252                 }
 253                 list_data["opt_status"]["use_markup"] = "true";
 254 
 255                 opts_listbox.add_row(list_data);
 256         }
 257 
 258         opts_listbox.select_row(0);
 259         list_data.clear();
 260 
 261         //
 262         // Set-up page stack and auxiliary controls last.
 263         //
 264 
 265         stacked_widget& pager
 266                         = find_widget<stacked_widget>(&window, "tabs_container", false);
 267         pager.select_layer(0);
 268 
 269         listbox& tab_bar
 270                         = find_widget<listbox>(&window, "tab_bar", false);
 271 
 272         window.keyboard_capture(&tab_bar);
 273 
 274         const unsigned tab_count = tab_bar.get_item_count();
 275         VALIDATE(tab_count == pager.get_layer_count(), "Tab bar and container size mismatch");
 276 
 277         connect_signal_notify_modified(tab_bar,
 278                 std::bind(&game_version::tab_switch_callback, this, std::ref(window)));
 279 }

 281 void game_version::tab_switch_callback(window& window)
 282 {
 283         stacked_widget& pager
 284                         = find_widget<stacked_widget>(&window, "tabs_container", false);
 285         listbox& tab_bar
 286                         = find_widget<listbox>(&window, "tab_bar", false);
 287 
 288         pager.select_layer(std::max<int>(0, tab_bar.get_selected_row()));
 289 }
