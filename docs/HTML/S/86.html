  97 static int placing_score(const config& side, const gamemap& map, const map_location& pos)
  98 {
  99         int positions = 0, liked = 0;
 100         const t_translation::ter_list terrain = t_translation::read_list(side["terrain_liked"]);
 101 
 102         for(int i = -8; i != 8; ++i) {
 103                 for(int j = -8; j != +8; ++j) {
 104                         const map_location pos2  = pos.plus(i, j);
 105                         if(map.on_board(pos2)) {
 106                                 ++positions;
 107                                 if(std::count(terrain.begin(),terrain.end(),map[pos2])) {
 108                                         ++liked;
 109                                 }
 110                         }
 111                 }
 112         }
 113 
 114         return (100*liked)/positions;
 115 }

 133 void game_state::place_sides_in_preferred_locations(const config& level)
 134 {
 135         std::vector<placing_info> placings;
 136 
 137         int num_pos = board_.map().num_valid_starting_positions();
 138 
 139         int side_num = 1;
 140         for(const config &side : level.child_range("side"))
 141         {
 142                 for(int p = 1; p <= num_pos; ++p) {
 143                         const map_location& pos = board_.map().starting_position(p);
 144                         int score = placing_score(side, board_.map(), pos);
 145                         placing_info obj;
 146                         obj.side = side_num;
 147                         obj.score = score;
 148                         obj.pos = pos;
 149                         placings.push_back(obj);
 150                 }
 151                 ++side_num;
 152         }
 153 
 154         std::stable_sort(placings.begin(),placings.end());
 155         std::set<int> placed;
 156         std::set<map_location> positions_taken;
 157 
 158         for (std::vector<placing_info>::const_iterator i = placings.begin(); i != placings.end() && int(placed.size()) != side_num - 1; ++i) {
 159                 if(placed.count(i->side) == 0 && positions_taken.count(i->pos) == 0) {
 160                         placed.insert(i->side);
 161                         positions_taken.insert(i->pos);
 162                         board_.map_->set_starting_position(i->side,i->pos);
 163                         LOG_NG << "placing side " << i->side << " at " << i->pos << std::endl;
 164                 }
 165         }
 166 }

 168 void game_state::init(const config& level, play_controller & pc)
 169 {
 170         events_manager_->read_scenario(level);
 171         gui2::dialogs::loading_screen::progress("init teams");
 172         if (level["modify_placing"].to_bool()) {
 173                 LOG_NG << "modifying placing..." << std::endl;
 174                 place_sides_in_preferred_locations(level);
 175         }
 176 
 177         LOG_NG << "initialized time of day regions... "    << (SDL_GetTicks() - pc.ticks()) << std::endl;
 178         for (const config &t : level.child_range("time_area")) {
 179                 tod_manager_.add_time_area(board_.map(),t);
 180         }
 181 
 182         LOG_NG << "initialized teams... "    << (SDL_GetTicks() - pc.ticks()) << std::endl;
 183 
 184         board_.teams_.resize(level.child_count("side"));
 185 
 186         std::vector<team_builder_ptr> team_builders;
 187 
 188         int team_num = 0;
 189         for (const config &side : level.child_range("side"))
 190         {
 191                 if (first_human_team_ == -1) {
 192                         const std::string &controller = side["controller"];
 193                         if (controller == "human") {
 194                                 first_human_team_ = team_num;
 195                         }
 196                 }
 197                 ++team_num;
 198                 team_builder_ptr tb_ptr = create_team_builder(side,
 199                         board_.teams_, level, board_, team_num);
 200                 build_team_stage_one(tb_ptr);
 201                 team_builders.push_back(tb_ptr);
 202         }
 203         //Initilize the lua kernel before the units are created.
 204         lua_kernel_->initialize(level);
 205 
 206         {
 207                 //sync traits of start units and the random start time.
 208                 randomness::set_random_determinstic deterministic(gamedata_.rng());
 209 
 210                 tod_manager_.resolve_random(*randomness::generator);
 211 
 212                 for(team_builder_ptr tb_ptr : team_builders)
 213                 {
 214                         build_team_stage_two(tb_ptr);
 215                 }
 216                 for(size_t i = 0; i < board_.teams_.size(); i++) {
 217                         // Labels from players in your ignore list default to hidden
 218                         if(preferences::is_ignored(board_.teams_[i].current_player())) {
 219                                 std::string label_cat = "side:" + std::to_string(i + 1);
 220                                 board_.hidden_label_categories_ref().push_back(label_cat);
 221                         }
 222                 }
 223         }
 224 }

 231 void game_state::write(config& cfg) const
 232 {
 233         cfg["init_side_done"] = init_side_done_;
 234         if(gamedata_.phase() == game_data::PLAY) {
 235                 cfg["playing_team"] = player_number_ - 1;
 236         }
 237         cfg["server_request_number"] = server_request_number_;
 238         //Call the lua save_game functions
 239         lua_kernel_->save_game(cfg);
 240 
 241         //Write the game events.
 242         events_manager_->write_events(cfg);
 243 
 244         //Write the map, unit_map, and teams info
 245         board_.write_config(cfg);
 246 
 247         //Write the tod manager, and time areas
 248         cfg.merge_with(tod_manager_.to_config());
 249 
 250         //write out the current state of the map
 251         cfg.merge_with(pathfind_manager_->to_config());
 252 
 253         //Write the game data, including wml vars
 254         gamedata_.write_snapshot(cfg);
 255 
 256         // Preserve the undo stack so that fog/shroud clearing is kept accurate.
 257         undo_stack_->write(cfg.add_child("undo_stack"));
 258 
 259         if(end_level_data_.get_ptr() != nullptr) {
 260                 end_level_data_->write(cfg.add_child("end_level_data"));
 261         }
 262 }

 264 namespace {
 265         struct castle_cost_calculator : pathfind::cost_calculator
 266         {
 267                 castle_cost_calculator(const gamemap& map, const team & view_team) :
 268                         map_(map),
 269                         viewer_(view_team),
 270                         use_shroud_(view_team.uses_shroud())
 271                 {}
 272 
 273                 virtual double cost(const map_location& loc, const double) const
 274                 {
 275                         if(!map_.is_castle(loc))
 276                                 return 10000;
 277 
 278                         if ( use_shroud_ && viewer_.shrouded(loc) )
 279                                 return 10000;
 280 
 281                         return 1;
 282                 }
 283 
 284         private:
 285                 const gamemap& map_;
 286                 const team& viewer_;
 287                 const bool use_shroud_; // Allows faster checks when shroud is disabled.
 288         };

 292 /**
 293  * Checks to see if a leader at @a leader_loc could recruit somewhere.
 294  * This takes into account terrain, shroud (for side @a side), and the presence
 295  * of visible units.
 296  * The behavior for an invalid @a side is subject to change for future needs.
 297  */
 298 bool game_state::can_recruit_from(const map_location& leader_loc, int side) const
 299 {
 300         const gamemap& map = board_.map();
 301 
 302         if( !map.is_keep(leader_loc) )
 303                 return false;
 304 
 305         if ( side < 1  ||  board_.teams().size() < static_cast<size_t>(side) ) {
 306                 // Invalid side specified.
 307                 // Currently this cannot happen, but it could conceivably be used in
 308                 // the future to request that shroud and visibility be ignored. Until
 309                 // that comes to pass, just return.
 310                 return false;
 311         }
 312 
 313         return pathfind::find_vacant_tile(leader_loc, pathfind::VACANT_CASTLE, nullptr,
 314                                           &(board_.teams())[side-1])
 315                != map_location::null_location();
 316 }

 324 /**
 325  * Checks to see if a leader at @a leader_loc could recruit on @a recruit_loc.
 326  * This takes into account terrain, shroud (for side @a side), and whether or
 327  * not there is already a visible unit at recruit_loc.
 328  * The behavior for an invalid @a side is subject to change for future needs.
 329  */
 330 bool game_state::can_recruit_on(const map_location& leader_loc, const map_location& recruit_loc, int side) const
 331 {
 332         const gamemap& map = board_.map();
 333 
 334         if( !map.is_castle(recruit_loc) )
 335                 return false;
 336 
 337         if( !map.is_keep(leader_loc) )
 338                 return false;
 339 
 340         if ( side < 1  ||  board_.teams().size() < static_cast<size_t>(side) ) {
 341                 // Invalid side specified.
 342                 // Currently this cannot happen, but it could conceivably be used in
 343                 // the future to request that shroud and visibility be ignored. Until
 344                 // that comes to pass, just return.
 345                 return false;
 346         }
 347         const team & view_team = board_.get_team(side);
 348 
 349         if ( view_team.shrouded(recruit_loc) )
 350                 return false;
 351 
 352         if ( board_.has_visible_unit(recruit_loc, view_team) )
 353                 return false;
 354 
 355         castle_cost_calculator calc(map, view_team);
 356         // The limit computed in the third argument is more than enough for
 357         // any convex castle on the map. Strictly speaking it could be
 358         // reduced to sqrt(map.w()**2 + map.h()**2).
 359         pathfind::plain_route rt =
 360                 pathfind::a_star_search(leader_loc, recruit_loc, map.w()+map.h(), calc,
 361                                         map.w(), map.h());
 362         return !rt.steps.empty();
 363 }

 370 bool game_state::side_can_recruit_on(int side, map_location hex) const
 371 {
 372         unit_map::const_iterator leader = board_.units().find(hex);
 373         if ( leader != board_.units().end() ) {
 374                 return leader->can_recruit() && leader->side() == side && can_recruit_from(*leader);
 375         } else {
 376                 // Look for a leader who can recruit on last_hex.
 377                 for ( leader = board_.units().begin(); leader != board_.units().end(); ++leader) {
 378                         if ( leader->can_recruit() && leader->side() == side && can_recruit_on(*leader, hex) ) {
 379                                 return true;
 380                         }
 381                 }
 382         }
 383         // No leader found who can recruit at last_hex.
 384         return false;
 385 }
