 108 void game_display::new_turn()
 109 {
 110         const time_of_day& tod = tod_manager_->get_time_of_day();
 111 
 112         if( !first_turn_) {
 113                 const time_of_day& old_tod = tod_manager_->get_previous_time_of_day();
 114 
 115                 if(old_tod.image_mask != tod.image_mask) {
 116                         surface old_mask(image::get_image(old_tod.image_mask,image::SCALED_TO_HEX));
 117                         surface new_mask(image::get_image(tod.image_mask,image::SCALED_TO_HEX));
 118 
 119                         const int niterations = static_cast<int>(10/turbo_speed());
 120                         const int frame_time = 30;
 121                         const int starting_ticks = SDL_GetTicks();
 122                         for(int i = 0; i != niterations; ++i) {
 123 
 124                                 if(old_mask != nullptr) {
 125                                         const fixed_t proportion = ftofxp(1.0) - fxpdiv(i,niterations);
 126                                         adjust_surface_alpha(old_mask, proportion);
 127                                         tod_hex_mask1.assign(old_mask);
 128                                 }
 129 
 130                                 if(new_mask != nullptr) {
 131                                         const fixed_t proportion = fxpdiv(i,niterations);
 132                                         adjust_surface_alpha(new_mask, proportion);
 133                                         tod_hex_mask2.assign(new_mask);
 134                                 }
 135 
 136                                 invalidate_all();
 137                                 draw();
 138 
 139                                 const int cur_ticks = SDL_GetTicks();
 140                                 const int wanted_ticks = starting_ticks + i*frame_time;
 141                                 if(cur_ticks < wanted_ticks) {
 142                                         SDL_Delay(wanted_ticks - cur_ticks);
 143                                 }
 144                         }
 145                 }
 146 
 147                 tod_hex_mask1.assign(nullptr);
 148                 tod_hex_mask2.assign(nullptr);
 149         }
 150 
 151         first_turn_ = false;
 152 
 153         display::update_tod();
 154 
 155         invalidate_all();
 156         draw();
 157 }

 169 void game_display::highlight_hex(map_location hex)
 170 {
 171         wb::future_map_if future(!synced_context::is_synced()); /**< Lasts for whole method. */
 172 
 173         const unit *u = resources::gameboard->get_visible_unit(hex, dc_->teams()[viewing_team()], !dont_show_all_);

 175                 displayedUnitHex_ = hex;
 176                 invalidate_unit();
 177         } else {
 178                 u = resources::gameboard->get_visible_unit(mouseoverHex_, dc_->teams()[viewing_team()], !dont_show_all_);
 179                 if (u) {
 180                         // mouse moved from unit hex to non-unit hex
 181                         if (dc_->units().count(selectedHex_)) {
 182                                 displayedUnitHex_ = selectedHex_;
 183                                 invalidate_unit();
 184                         }
 185                 }
 186         }
 187 
 188         display::highlight_hex(hex);
 189         invalidate_game_status();
 190 }

 248 void game_display::draw_invalidated()

 250         halo_man_->unrender(invalidated_);
 251         display::draw_invalidated();
 252         if (fake_unit_man_->empty()) {
 253                 return;

 271 void game_display::draw_hex(const map_location& loc)
 272 {
 273         const bool on_map = get_map().on_board(loc);
 274         const bool is_shrouded = shrouded(loc);
 275 //      const bool is_fogged = fogged(loc);
 276         const int xpos = get_location_x(loc);
 277         const int ypos = get_location_y(loc);
 278 
 279 //      image::TYPE image_type = get_image_type(loc);
 280 
 281         display::draw_hex(loc);
 282 
 283         if(cursor::get() == cursor::WAIT) {
 284                 // Interaction is disabled, so we don't need anything else
 285                 return;
 286         }
 287 
 288         if(on_map && loc == mouseoverHex_) {
 289                 drawing_layer hex_top_layer = LAYER_MOUSEOVER_BOTTOM;
 290                 const unit *u = resources::gameboard->get_visible_unit(loc, dc_->teams()[viewing_team()] );
 291                 if( u != nullptr ) {
 292                         hex_top_layer = LAYER_MOUSEOVER_TOP;
 293                 }
 294                 if(u == nullptr) {
 295                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 296                                         image::get_image("misc/hover-hex-top.png~RC(magenta>gold)", image::SCALED_TO_HEX));
 297                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 298                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>gold)", image::SCALED_TO_HEX));
 299                 } else if(dc_->teams()[currentTeam_].is_enemy(u->side())) {
 300                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 301                                         image::get_image("misc/hover-hex-enemy-top.png~RC(magenta>red)", image::SCALED_TO_HEX));
 302                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 303                                         image::get_image("misc/hover-hex-enemy-bottom.png~RC(magenta>red)", image::SCALED_TO_HEX));
 304                 } else if(dc_->teams()[currentTeam_].side() == u->side()) {
 305                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 306                                         image::get_image("misc/hover-hex-top.png~RC(magenta>green)", image::SCALED_TO_HEX));
 307                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 308                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>green)", image::SCALED_TO_HEX));
 309                 } else {
 310                         drawing_buffer_add( hex_top_layer, loc, xpos, ypos,
 311                                         image::get_image("misc/hover-hex-top.png~RC(magenta>lightblue)", image::SCALED_TO_HEX));
 312                         drawing_buffer_add(LAYER_MOUSEOVER_BOTTOM, loc, xpos, ypos,
 313                                         image::get_image("misc/hover-hex-bottom.png~RC(magenta>lightblue)", image::SCALED_TO_HEX));
 314                 }
 315         }
 316 
 317 
 318 
 319         // Draw reach_map information.
 320         // We remove the reachability mask of the unit
 321         // that we want to attack.
 322         if (!is_shrouded && !reach_map_.empty()
 323                         && reach_map_.find(loc) == reach_map_.end() && loc != attack_indicator_dst_) {
 324                 static const image::locator unreachable(game_config::images::unreachable);
 325                 drawing_buffer_add(LAYER_REACHMAP, loc, xpos, ypos,
 326                                 image::get_image(unreachable,image::SCALED_TO_HEX));
 327         }
 328 
 329         if (std::shared_ptr<wb::manager> w = wb_.lock()) {
 330                 w->draw_hex(loc);
 331 
 332                 if (!(w->is_active() && w->has_temp_move()))
 333                 {
 334                         std::vector<surface> footstepImages = footsteps_images(loc, route_, dc_);
 335                         if (!footstepImages.empty()) {
 336                                 drawing_buffer_add(LAYER_FOOTSTEPS, loc, xpos, ypos, footsteps_images(loc, route_, dc_));
 337                         }
 338                 }
 339         }
 340         // Draw the attack direction indicator
 341         if(on_map && loc == attack_indicator_src_) {
 342                 drawing_buffer_add(LAYER_ATTACK_INDICATOR, loc, xpos, ypos,
 343                         image::get_image("misc/attack-indicator-src-" + attack_indicator_direction() + ".png", image::SCALED_TO_HEX));
 344         } else if (on_map && loc == attack_indicator_dst_) {
 345                 drawing_buffer_add(LAYER_ATTACK_INDICATOR, loc, xpos, ypos,
 346                         image::get_image("misc/attack-indicator-dst-" + attack_indicator_direction() + ".png", image::SCALED_TO_HEX));
 347         }
 348 
 349         // Linger overlay unconditionally otherwise it might give glitches
 350         // so it's drawn over the shroud and fog.
 351         if(mode_ != RUNNING) {
 352                 static const image::locator linger(game_config::images::linger);
 353                 drawing_buffer_add(LAYER_LINGER_OVERLAY, loc, xpos, ypos,
 354                         image::get_image(linger, image::TOD_COLORED));
 355         }
 356 
 357         if(on_map && loc == selectedHex_ && !game_config::images::selected.empty()) {
 358                 static const image::locator selected(game_config::images::selected);
 359                 drawing_buffer_add(LAYER_SELECTED_HEX, loc, xpos, ypos,
 360                                 image::get_image(selected, image::SCALED_TO_HEX));
 361         }
 362 
 363         // Show def% and turn to reach info
 364         if(!is_shrouded && on_map) {
 365                 draw_movement_info(loc);
 366         }
 367 
 368         if(game_config::debug) {
 369                 int debugH = debugHighlights_[loc];
 370                 if (debugH) {
 371                         std::string txt = std::to_string(debugH);
 372                         draw_text_in_hex(loc, LAYER_MOVE_INFO, txt, 18, font::BAD_COLOR);
 373                 }
 374         }
 375         //simulate_delay += 1;
 376 }

 418 void game_display::draw_movement_info(const map_location& loc)
 419 {
 420         // Search if there is a mark here
 421         pathfind::marked_route::mark_map::iterator w = route_.marks.find(loc);
 422 
 423         std::shared_ptr<wb::manager> wb = wb_.lock();
 424 
 425         // Don't use empty route or the first step (the unit will be there)
 426         if(w != route_.marks.end()
 427                                 && !route_.steps.empty() && route_.steps.front() != loc) {
 428                 const unit_map::const_iterator un =
 429                                 (wb && wb->get_temp_move_unit().valid()) ?
 430                                                 wb->get_temp_move_unit() : dc_->units().find(route_.steps.front());
 431                 if(un != dc_->units().end()) {
 432                         // Display the def% of this terrain
 433                         int def =  100 - un->defense_modifier(get_map().get_terrain(loc));
 434                         std::stringstream def_text;
 435                         def_text << def << "%";
 436 
 437                         color_t color = game_config::red_to_green(def, false);
 438 
 439                         // simple mark (no turn point) use smaller font
 440                         int def_font = w->second.turns > 0 ? 18 : 16;
 441                         draw_text_in_hex(loc, LAYER_MOVE_INFO, def_text.str(), def_font, color);
 442 
 443                         int xpos = get_location_x(loc);
 444                         int ypos = get_location_y(loc);
 445                         if (w->second.invisible) {
 446                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 447                                         image::get_image("misc/hidden.png", image::SCALED_TO_HEX));
 448                         }
 449 
 450                         if (w->second.zoc) {
 451                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 452                                         image::get_image("misc/zoc.png", image::SCALED_TO_HEX));
 453                         }
 454 
 455                         if (w->second.capture) {
 456                                 drawing_buffer_add(LAYER_MOVE_INFO, loc, xpos, ypos,
 457                                         image::get_image("misc/capture.png", image::SCALED_TO_HEX));
 458                         }
 459 
 460                         //we display turn info only if different from a simple last "1"
 461                         if (w->second.turns > 1 || (w->second.turns == 1 && loc != route_.steps.back())) {
 462                                 std::stringstream turns_text;
 463                                 turns_text << w->second.turns;
 464                                 draw_text_in_hex(loc, LAYER_MOVE_INFO, turns_text.str(), 17, font::NORMAL_COLOR, 0.5,0.8);
 465                         }
 466 
 467                         // The hex is full now, so skip the "show enemy moves"
 468                         return;
 469                 }
 470         }
 471         // When out-of-turn, it's still interesting to check out the terrain defs of the selected unit
 472         else if (selectedHex_.valid() && loc == mouseoverHex_)
 473         {
 474                 const unit_map::const_iterator selectedUnit = resources::gameboard->find_visible_unit(selectedHex_,dc_->teams()[currentTeam_]);
 475                 const unit_map::const_iterator mouseoveredUnit = resources::gameboard->find_visible_unit(mouseoverHex_,dc_->teams()[currentTeam_]);
 476                 if(selectedUnit != dc_->units().end() && mouseoveredUnit == dc_->units().end()) {
 477                         // Display the def% of this terrain
 478                         int def =  100 - selectedUnit->defense_modifier(get_map().get_terrain(loc));
 479                         std::stringstream def_text;
 480                         def_text << def << "%";
 481 
 482                         color_t color = game_config::red_to_green(def, false);
 483 
 484                         // use small font
 485                         int def_font = 16;
 486                         draw_text_in_hex(loc, LAYER_MOVE_INFO, def_text.str(), def_font, color);
 487                 }
 488         }
 489 
 490         if (!reach_map_.empty()) {
 491                 reach_map::iterator reach = reach_map_.find(loc);
 492                 if (reach != reach_map_.end() && reach->second > 1) {
 493                         const std::string num = std::to_string(reach->second);
 494                         draw_text_in_hex(loc, LAYER_MOVE_INFO, num, 16, font::YELLOW_COLOR);
 495                 }
 496         }
 497 }

 499 std::vector<surface> footsteps_images(const map_location& loc, const pathfind::marked_route & route_, const display_context * dc_)
 500 {
 501         std::vector<surface> res;
 502 
 503         if (route_.steps.size() < 2) {
 504                 return res; // no real "route"
 505         }
 506 
 507         std::vector<map_location>::const_iterator i =
 508                  std::find(route_.steps.begin(),route_.steps.end(),loc);
 509 
 510         if( i == route_.steps.end()) {
 511                 return res; // not on the route
 512         }
 513 
 514         // Check which footsteps images of game_config we will use
 515         int move_cost = 1;
 516         const unit_map::const_iterator u = dc_->units().find(route_.steps.front());
 517         if(u != dc_->units().end()) {
 518                 move_cost = u->movement_cost(dc_->map().get_terrain(loc));
 519         }
 520         int image_number = std::min<int>(move_cost, game_config::foot_speed_prefix.size());
 521         if (image_number < 1) {
 522                 return res; // Invalid movement cost or no images
 523         }
 524         const std::string foot_speed_prefix = game_config::foot_speed_prefix[image_number-1];
 525 
 526         surface teleport = nullptr;
 527 
 528         // We draw 2 half-hex (with possibly different directions),
 529         // but skip the first for the first step.
 530         const int first_half = (i == route_.steps.begin()) ? 1 : 0;
 531         // and the second for the last step
 532         const int second_half = (i+1 == route_.steps.end()) ? 0 : 1;
 533 
 534         for (int h = first_half; h <= second_half; ++h) {
 535                 const std::string sense( h==0 ? "-in" : "-out" );
 536 
 537                 if (!tiles_adjacent(*(i+(h-1)), *(i+h))) {
 538                         std::string teleport_image =
 539                         h==0 ? game_config::foot_teleport_enter : game_config::foot_teleport_exit;
 540                         teleport = image::get_image(teleport_image, image::SCALED_TO_HEX);
 541                         continue;
 542                 }
 543 
 544                 // In function of the half, use the incoming or outgoing direction
 545                 map_location::DIRECTION dir = (i+(h-1))->get_relative_dir(*(i+h));
 546 
 547                 std::string rotate;
 548                 if (dir > map_location::SOUTH_EAST) {
 549                         // No image, take the opposite direction and do a 180 rotation
 550                         dir = i->get_opposite_dir(dir);
 551                         rotate = "~FL(horiz)~FL(vert)";
 552                 }
 553 
 554                 const std::string image = foot_speed_prefix
 555                         + sense + "-" + i->write_direction(dir)
 556                         + ".png" + rotate;
 557 
 558                 res.push_back(image::get_image(image, image::SCALED_TO_HEX));
 559         }
 560 
 561         // we draw teleport image (if any) in last
 562         if (teleport != nullptr) res.push_back(teleport);
 563 
 564         return res;
 565 }

 614 void game_display::float_label(const map_location& loc, const std::string& text, const color_t& color)
 615 {
 616         if(preferences::show_floating_labels() == false || fogged(loc)) {
 617                 return;
 618         }
 619 
 620         font::floating_label flabel(text);
 621         flabel.set_font_size(font::SIZE_XLARGE);
 622         flabel.set_color(color);
 623         flabel.set_position(get_location_x(loc)+zoom_/2, get_location_y(loc));
 624         flabel.set_move(0, -2 * turbo_speed());
 625         flabel.set_lifetime(60/turbo_speed());
 626         flabel.set_scroll_mode(font::ANCHOR_LABEL_MAP);
 627 
 628         font::add_floating_label(flabel);
 629 }
