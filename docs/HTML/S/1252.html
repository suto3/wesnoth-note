  37 void resolve_negative_value(int size, int& val)
  38 {
  39         if(val < 0) {

  51 /**
  52  * Parses a ']' terminated string.
  53  * This is a important optimization of lexical_cast_default
  54  */
  55 int parse_index(const char* index_str)
  56 {
  57         char* endptr;
  58         int res = strtol(index_str, &endptr, 10);
  59 
  60         if(*endptr != ']' || res > static_cast<int>(game_config::max_loop) || endptr == index_str) {
  61                 throw invalid_variablename_exception();
  62         }
  63 
  64         return res;
  65 }

  71 /**
  72  * Visitor base class.
  73  *
  74  * This provides the interface for the main functions each visitor can implement. The default implementation of
  75  * each function simply throws @ref invalid_variablename_exception.
  76  *
  77  * This class also provides two type aliases corresponding to the function return value and argument types.
  78  * Note that visitors shouldn't inherit from this directly and instead use the @ref info_visitor and
  79  * @ref info_visitor_const wrappers, since both fully specify the parameter type.
  80  *
  81  * @tparam R                   Return value type.
  82  * @tparam P                   Argument type.
  83  */
  84 template<typename R, typename P>
  85 class info_visitor_base
  86 {
  87 public:
  88         using result_t = R;
  89         using param_t = P&;
  90 
  91 #define DEFAULTHANDLER(name)                                                                                           \
  92         result_t name(param_t) const                                                                                       \
  93         {                                                                                                                  \
  94                 throw invalid_variablename_exception();                                                                        \
  95         }
  96 
  97         /** For use if the variable name was previously empty. This can only happen during calculate_value. */
  98         DEFAULTHANDLER(from_start)
  99 
 100         /** For use if the variable ended with a .somename. */
 101         DEFAULTHANDLER(from_named)
 102 
 103         /** For use if the variable ended with .somename[someindex]. */
 104         DEFAULTHANDLER(from_indexed)
 105 
 106         /** For use if the variable is a readonly value (.somename.length). */
 107         DEFAULTHANDLER(from_temporary)
 108 
 109 #undef DEFAULTHANDLER
 110 };

 118 /** Adds a '.<key>' to the current variable. */
 119 template<typename V>
 120 class get_variable_key_visitor : public info_visitor<V, void>
 121 {
 122 public:
 123         // Import typedefs from base class.
 124         using param_t = typename get_variable_key_visitor::param_t;
 125 
 126         get_variable_key_visitor(const std::string& key)
 127                 : key_(key)
 128         {
 129                 if(!config::valid_id(key_)) {
 130                         throw invalid_variablename_exception();
 131                 }
 132         }
 133 
 134         void from_named(param_t state) const
 135         {
 136                 if(key_ == "length") {
 137                         state.temp_val_ = state.child_->child_count(state.key_);
 138                         state.type_ = state_temporary;
 139                         return;
 140                 }
 141 
 142                 return do_from_config(V::get_child_at(*state.child_, state.key_, 0), state);
 143         }
 144 
 145         void from_start(param_t state) const
 146         {
 147                 return do_from_config(*state.child_, state);
 148         }
 149 
 150         void from_indexed(param_t state) const
 151         {
 152                 // We do not support aaa[0].length
 153                 return do_from_config(V::get_child_at(*state.child_, state.key_, state.index_), state);
 154         }
 155 
 156 private:
 157         void do_from_config(maybe_const_t<config, V>& cfg, param_t state) const
 158         {
 159                 state.type_ = state_named;
 160                 state.key_ = key_;
 161                 state.child_ = &cfg;
 162         }

 164         const std::string& key_;

 385 /**
 386  * Replaces the child in [startindex, endindex) with 'source'
 387  * 'insert' and 'append' are subcases of this.
 388  */
 389 class replace_range_h
 390 {
 391 public:
 392         typedef config::child_itors result_t;
 393         replace_range_h(std::vector<config>& source)
 394                 : datasource_(source)
 395         {
 396         }
 397 
 398         result_t operator()(config& child, const std::string& key, int startindex, int endindex) const
 399         {
 400                 assert(endindex - startindex >= 0);
 401                 if(endindex > 0) {
 402                         // NOTE: currently this is only called from as_range_visitor_base<vi_policy_create>
 403                         // Based on that assumption we use vi_policy_create::get_child_at here instead of making this
 404             // a class template.
 405                         vi_policy_create::get_child_at(child, key, endindex - 1);
 406                 }
 407 
 408                 int size_diff = datasource_.size() - (endindex - startindex);
 409 
 410                 // remove configs first
 411                 while(size_diff < 0) {
 412                         child.remove_child(key, startindex);
 413                         ++size_diff;
 414                 }
 415 
 416                 size_t index = 0;
 417                 for(index = 0; index < static_cast<size_t>(size_diff); ++index) {
 418                         child.add_child_at(key, config(), startindex + index).swap(datasource_[index]);
 419                 }
 420 
 421                 for(; index < datasource_.size(); ++index) {
 422                         child.child(key, startindex + index).swap(datasource_[index]);
 423                 }
 424 
 425                 return get_child_range(child, key, startindex, datasource_.size());
 426         }
 427 
 428 private:
 429         std::vector<config>& datasource_;
 430 };

 465 class merge_range_h
 466 {
 467 public:
 468         typedef void result_t;
 469         merge_range_h(std::vector<config>& source)
 470                 : datasource_(source)
 471         {
 472         }
 473 
 474         void operator()(config& child, const std::string& key, int startindex, int /*endindex*/) const
 475         {
 476                 // The merge_with function only accepts configs so we convert vector -> config.
 477                 config datatemp;
 478 
 479                 // Add emtpy config to 'shift' the merge to startindex
 480                 for(int index = 0; index < startindex; ++index) {
 481                         datatemp.add_child(key);
 482                 }
 483 
 484                 // move datasource_ -> datatemp
 485                 for(size_t index = 0; index < datasource_.size(); ++index) {
 486                         datatemp.add_child(key).swap(datasource_[index]);
 487                 }
 488 
 489                 child.merge_with(datatemp);
 490         }
 491 
 492 private:
 493         std::vector<config>& datasource_;
 494 };
