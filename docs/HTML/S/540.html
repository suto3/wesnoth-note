  24 /**
  25  * Class for text input history.
  26  *
  27  * The history of text items can be stored in the preferences. This class
  28  * handles that. Every item needs an id by which the history is loaded and
  29  * saved.
  30  */
  31 class text_history
  32 {
  33 public:
  34         /**
  35          * Gets history that matches id.
  36          *
  37          * @param id                  The id of the history to look for.
  38          * @param enabled             The enabled state of the history.
  39          *
  40          * @returns                   The history object.
  41          */
  42         static text_history get_history(const std::string& id, const bool enabled);
  43 
  44         text_history() : history_(0), pos_(0), enabled_(false)
  45         {
  46         }
  47 
  48         /**
  49          * Push string into the history.
  50          *
  51          * If the string is empty or the same as the last item in the history this
  52          * function is a nop.
  53          *
  54          * @param text                   The text to push in the history.
  55          */
  56         void push(const std::string& text);
  57 
  58         /**
  59          * One step up in the history.
  60          *
  61          * Pushes text to the history if at the end.
  62          *
  63          * @param text                The text to push in the history.
  64          *
  65          * @returns                   The current value of the history.
  66          */
  67         std::string up(const std::string& text = "");
  68 
  69         /**
  70          * One step down in the history.
  71          *
  72          * Pushes text to the history if at the end.
  73          *
  74          * @param text                The text to push in the history.
  75          *
  76          * @returns                   The current value of the history.
  77          */
  78         std::string down(const std::string& text = "");
  79 
  80         /**
  81          * Gets the current history value.
  82          *
  83          * @returns                   If enabled return the current history
  84          *                            position, otherwise an empty string is
  85          *                            returned.
  86          */
  87         std::string get_value() const;
  88 
  89         /***** ***** ***** setters / getters for members ***** ****** *****/
  90 
  91         void set_enabled(bool enabled = true)
  92         {
  93                 enabled_ = enabled;
  94         }
  95         bool get_enabled() const
  96         {
  97                 return enabled_;
  98         }
  99 
 100 private:
 101         text_history(std::vector<std::string>* history, const bool enabled)
 102                 : history_(history), pos_(history->size()), enabled_(enabled)
 103         {
 104         }
 105 
 106         /** The items in the history. */
 107         std::vector<std::string>* history_;
 108 
 109         /** The current position in the history. */
 110         unsigned pos_;
 111 
 112         /** Is the history enabled. */
 113         bool enabled_;
 114 };
