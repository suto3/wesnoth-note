 112 * A struct to describe one possible combat scenario.
 113 * (Needed when the number of attacks can vary due to swarm.)

 115 struct combat_slice
 116 {
 117         // The hit point range this slice covers.
 118         unsigned begin_hp; // included in the range.
 119         unsigned end_hp;   // excluded from the range.
 120 
 121         // The probability of this slice.
 122         double prob;
 123 
 124         // The number of strikes applicable with this slice.
 125         unsigned strikes;
 126 
 127         combat_slice(
 128                         const summary_t& src_summary, unsigned begin, unsigned end, unsigned num_strikes);
 129         combat_slice(const summary_t& src_summary, unsigned num_strikes);
 130 };

 132 /**
 133 * Creates a slice from a summary, and associates a number of strikes.
 134 */
 135 combat_slice::combat_slice(
 136                 const summary_t& src_summary, unsigned begin, unsigned end, unsigned num_strikes)
 137         : begin_hp(begin)
 138         , end_hp(end)
 139         , prob(0.0)
 140         , strikes(num_strikes)
 141 {
 142         if(src_summary[0].empty()) {
 143                 // No summary; this should be the only slice.
 144                 prob = 1.0;
 145                 return;
 146         }
 147 
 148         // Avoid accessing beyond the end of the vectors.
 149         if(end > src_summary[0].size()) {
 150                 end = src_summary[0].size();
 151         }
 152 
 153         // Sum the probabilities in the slice.
 154         for(unsigned i = begin; i < end; ++i) {
 155                 prob += src_summary[0][i];
 156         }
 157 
 158         if(!src_summary[1].empty()) {
 159                 for(unsigned i = begin; i < end; ++i) {
 160                         prob += src_summary[1][i];
 161                 }
 162         }
 163 }

 177 /**
 178 * Returns the number of hit points greater than cur_hp, and at most
 179 * stats.max_hp+1, at which the unit would get another attack because
 180 * of swarm.
 181 * Helper function for split_summary().
 182 */
 183 unsigned hp_for_next_attack(unsigned cur_hp, const battle_context_unit_stats& stats)
 184 {
 185         unsigned old_strikes = stats.calc_blows(cur_hp);
 186 
 187         // A formula would have to deal with rounding issues; instead
 188         // loop until we find more strikes.
 189         while(++cur_hp <= stats.max_hp) {
 190                 if(stats.calc_blows(cur_hp) != old_strikes) {
 191                         break;
 192                 }
 193         }
 194 
 195         return cur_hp;
 196 }

 198 /**
 199 * Split the combat by number of attacks per combatant (for swarm).
 200 * This also clears the current summaries.
 201 */
 202 std::vector<combat_slice> split_summary(
 203                 const battle_context_unit_stats& unit_stats, summary_t& summary)
 204 {
 205         std::vector<combat_slice> result;
 206 
 207         if(unit_stats.swarm_min == unit_stats.swarm_max || summary[0].empty()) {
 208                 // We use the same number of blows for all possibilities.
 209                 result.emplace_back(summary, unit_stats.num_blows);
 210                 return result;
 211         }
 212 
 213         debug(("Slicing:\n"));
 214         // Loop through our slices.
 215         unsigned cur_end = 0;
 216         do {
 217                 // Advance to the next slice.
 218                 const unsigned cur_begin = cur_end;
 219                 cur_end = hp_for_next_attack(cur_begin, unit_stats);
 220 
 221                 // Add this slice.
 222                 combat_slice slice(summary, cur_begin, cur_end, unit_stats.calc_blows(cur_begin));
 223                 if(slice.prob != 0.0) {
 224                         result.push_back(slice);
 225                         debug(("\t%2u-%2u hp; strikes: %u; probability: %6.2f\n", cur_begin, cur_end, slice.strikes,
 226                                         slice.prob * 100.0));
 227                 }
 228         } while(cur_end <= unit_stats.max_hp);
 229 
 230         return result;
 231 }

 373 /**
 374  * Constructor.
 375  * @param  a_max          The maximum value we will track for A.
 376  * @param  b_max          The maximum value we will track for B.
 377  * @param  need_a_slowed  Set to true if there might be transfers to a "slow" plane for A.
 378  * @param  need_b_slowed  Set to true if there might be transfers to a "slow" plane for B.
 379  * @param  a_cur          The current value for A. (Ignored if a_initial[0] is not empty.)
 380  * @param  b_cur          The current value for B. (Ignored if b_initial[0] is not empty.)
 381  * @param  a_initial      The initial distribution of values for A. Element [0] is for normal A. while [1] is for slowed
 382  * A.
 383  * @param  b_initial      The initial distribution of values for B. Element [0] is for normal B. while [1] is for slowed
 384  * B.
 385  */
 386 prob_matrix::prob_matrix(unsigned int a_max,
 387                 unsigned int b_max,
 388                 bool need_a_slowed,
 389                 bool need_b_slowed,
 390                 unsigned int a_cur,
 391                 unsigned int b_cur,
 392                 const summary_t& a_initial,
 393                 const summary_t& b_initial)
 394         : rows_(a_max + 1)
 395         , cols_(b_max + 1)
 396         , plane_()
 397         , used_rows_()
 398         , used_cols_()
 399 {
 400         // Make sure we do not access the matrix in invalid positions.
 401         a_cur = std::min<unsigned int>(a_cur, rows_ - 1);
 402         b_cur = std::min<unsigned int>(b_cur, cols_ - 1);
 403 
 404         // It will be convenient to always consider row/col 0 to be used.
 405         for(unsigned plane = 0; plane != NUM_PLANES; ++plane) {
 406                 used_rows_[plane].insert(0u);
 407                 used_cols_[plane].insert(0u);
 408         }
 409 
 410         // We will need slowed planes if the initial vectors have them.
 411         need_a_slowed = need_a_slowed || !a_initial[1].empty();
 412         need_b_slowed = need_b_slowed || !b_initial[1].empty();
 413 
 414         // Allocate the needed planes.
 415         plane_[NEITHER_SLOWED] = new_plane();
 416         plane_[A_SLOWED] = !need_a_slowed ? nullptr : new_plane();
 417         plane_[B_SLOWED] = !need_b_slowed ? nullptr : new_plane();
 418         plane_[BOTH_SLOWED] = !(need_a_slowed && need_b_slowed) ? nullptr : new_plane();
 419 
 420         // Initialize the probability distribution.
 421         initialize_plane(NEITHER_SLOWED, a_cur, b_cur, a_initial[0], b_initial[0]);
 422 
 423         if(!a_initial[1].empty()) {
 424                 initialize_plane(A_SLOWED, a_cur, b_cur, a_initial[1], b_initial[0]);
 425         }
 426 
 427         if(!b_initial[1].empty()) {
 428                 initialize_plane(B_SLOWED, a_cur, b_cur, a_initial[0], b_initial[1]);
 429         }
 430 
 431         if(!a_initial[1].empty() && !b_initial[1].empty()) {
 432                 initialize_plane(BOTH_SLOWED, a_cur, b_cur, a_initial[1], b_initial[1]);
 433         }
 434 
 435         // Some debugging messages.
 436         if(!a_initial[0].empty()) {
 437                 debug(("A has fought before (or is slowed).\n"));
 438                 dump();
 439         }
 440 
 441         if(!b_initial[0].empty()) {
 442                 debug(("B has fought before (or is slowed).\n"));
 443                 dump();
 444         }
 445 }

 456 /**
 457  * Fills the indicated plane with its initial (non-zero) values.
 458  * (Part of construction)
 459  * @param  plane          The plane to initialize.
 460  * @param  a_cur          The current value for A. (Ignored if a_initial is not empty.)
 461  * @param  b_cur          The current value for B. (Ignored if b_initial is not empty.)
 462  * @param  a_initial      The initial distribution of values for A for this plane.
 463  * @param  b_initial      The initial distribution of values for B for this plane.
 464  */
 465 void prob_matrix::initialize_plane(unsigned plane,
 466                 unsigned a_cur,
 467                 unsigned b_cur,
 468                 const std::vector<double>& a_initial,
 469                 const std::vector<double>& b_initial)
 470 {
 471         if(!a_initial.empty()) {
 472                 unsigned row_count = std::min<unsigned>(a_initial.size(), rows_);
 473                 // The probabilities for each row are contained in a_initial.
 474                 for(unsigned row = 0; row < row_count; ++row) {
 475                         if(a_initial[row] != 0.0) {
 476                                 used_rows_[plane].insert(row);
 477                                 initialize_row(plane, row, a_initial[row], b_cur, b_initial);
 478                         }
 479                 }
 480         } else {
 481                 used_rows_[plane].insert(a_cur);
 482                 // Only the row indicated by a_cur is a possibility.
 483                 initialize_row(plane, a_cur, 1.0, b_cur, b_initial);
 484         }
 485 }

 487 /**
 488  * Fills the indicated row with its initial (non-zero) values.
 489  * (Part of construction)
 490  * @param  plane          The plane containing the row to initialize.
 491  * @param  row            The row to initialize.
 492  * @param  row_prob       The probability of A having the value for this row.
 493  * @param  b_cur          The current value for B. (Ignored if b_initial is not empty.)
 494  * @param  b_initial      The initial distribution of values for B for this plane.
 495  */
 496 void prob_matrix::initialize_row(
 497                 unsigned plane, unsigned row, double row_prob, unsigned b_cur, const std::vector<double>& b_initial)
 498 {
 499         if(!b_initial.empty()) {
 500                 unsigned col_count = std::min<unsigned>(b_initial.size(), cols_);
 501                 // The probabilities for each column are contained in b_initial.
 502                 for(unsigned col = 0; col < col_count; ++col) {
 503                         if(b_initial[col] != 0.0) {
 504                                 used_cols_[plane].insert(col);
 505                                 val(plane, row, col) = row_prob * b_initial[col];
 506                         }
 507                 }
 508         } else {
 509                 // Only the column indicated by b_cur is a possibility.
 510                 used_cols_[plane].insert(b_cur);
 511                 val(plane, row, b_cur) = row_prob;
 512         }
 513 }

 529 // xfer, shift_cols and shift_rows use up most of our time.  Careful!
 530 /**
 531  * Transfers a portion (value * prob) of one value in the matrix to another.
 532  */
 533 void prob_matrix::xfer(unsigned dst_plane,
 534                 unsigned src_plane,
 535                 unsigned row_dst,
 536                 unsigned col_dst,
 537                 unsigned row_src,
 538                 unsigned col_src,
 539                 double prob)
 540 {
 541         double& src = val(src_plane, row_src, col_src);
 542         if(src != 0.0) {
 543                 double diff = src * prob;
 544                 src -= diff;
 545 
 546                 double& dst = val(dst_plane, row_dst, col_dst);
 547                 if(dst == 0.0) {
 548                         // Track that this entry is now used.
 549                         used_rows_[dst_plane].insert(row_dst);
 550                         used_cols_[dst_plane].insert(col_dst);
 551                 }
 552                 dst += diff;
 553 
 554                 debug(("Shifted %4.3g from %s(%u,%u) to %s(%u,%u).\n", diff,
 555                         src_plane == NEITHER_SLOWED
 556                                 ? ""
 557                                 : src_plane == A_SLOWED
 558                                         ? "[A_SLOWED]"
 559                                         : src_plane == B_SLOWED
 560                                                 ? "[B_SLOWED]"
 561                                                 : src_plane == BOTH_SLOWED
 562                                                         ? "[BOTH_SLOWED]"
 563                                                         : "INVALID",
 564 
 565                         row_src, col_src,
 566                         dst_plane == NEITHER_SLOWED
 567                                 ? ""
 568                                 : dst_plane == A_SLOWED
 569                                         ? "[A_SLOWED]"
 570                                         : dst_plane == B_SLOWED
 571                                                 ? "[B_SLOWED]"
 572                                                 : dst_plane == BOTH_SLOWED
 573                                                         ? "[BOTH_SLOWED]"
 574                                                         : "INVALID",
 575                         row_dst, col_dst)
 576                 );
 577         }
 578 }

 584                 unsigned dst_plane, unsigned src_plane, unsigned row_dst, unsigned col_dst, unsigned row_src, unsigned col_src)
 585 {
 586         if(dst_plane == src_plane && row_dst == row_src && col_dst == col_src)
 587                 // Transferring to itself. Nothing to do.
 588                 return;
 589 
 590         double& src = val(src_plane, row_src, col_src);
 591         if(src != 0.0) {
 592                 debug(("Shifting %4.3g from %s(%u,%u) to %s(%u,%u).\n", src,
 593                         src_plane == NEITHER_SLOWED
 594                                 ? ""
 595                                 : src_plane == A_SLOWED
 596                                         ? "[A_SLOWED]"
 597                                         : src_plane == B_SLOWED
 598                                                 ? "[B_SLOWED]"
 599                                                 : src_plane == BOTH_SLOWED
 600                                                         ? "[BOTH_SLOWED]"
 601                                                         : "INVALID",
 602                         row_src, col_src,
 603                         dst_plane == NEITHER_SLOWED
 604                                 ? ""
 605                                 : dst_plane == A_SLOWED
 606                                         ? "[A_SLOWED]"
 607                                         : dst_plane == B_SLOWED
 608                                                 ? "[B_SLOWED]"
 609                                                 : dst_plane == BOTH_SLOWED
 610                                                         ? "[BOTH_SLOWED]"
 611                                                         : "INVALID",
 612                         row_dst, col_dst)
 613                 );
 614 
 615                 double& dst = val(dst_plane, row_dst, col_dst);
 616                 if(dst == 0.0) {
 617                         // Track that this entry is now used.
 618                         used_rows_[dst_plane].insert(row_dst);
 619                         used_cols_[dst_plane].insert(col_dst);
 620                 }
 621 
 622                 dst += src;
 623                 src = 0.0;
 624         }
 625 }

 627 /**
 628  * Transfers a portion (value * prob) of the values in a row to another.
 629  * Part of shift_cols().
 630  */
 631 void prob_matrix::shift_cols_in_row(unsigned dst,
 632                 unsigned src,
 633                 unsigned row,
 634                 const std::vector<unsigned>& cols,
 635                 unsigned damage,
 636                 double prob,
 637                 int drainmax,
 638                 int drain_constant,
 639                 int drain_percent)
 640 {
 641         // Some conversions to (signed) int.
 642         int row_i = static_cast<int>(row);
 643         int max_row = static_cast<int>(rows_) - 1;
 644 
 645         // cols[0] is excluded since that should be 0, representing already dead.
 646         unsigned col_x = 1;
 647 
 648         // Killing blows can have different drain amounts, so handle them first
 649         for(; col_x < cols.size() && cols[col_x] < damage; ++col_x) {
 650                 // These variables are not strictly necessary, but they make the
 651                 // calculation easier to parse.
 652                 int col_i = static_cast<int>(cols[col_x]);
 653                 int drain_amount = col_i * drain_percent / 100 + drain_constant;
 654                 unsigned newrow = utils::clamp(row_i + drain_amount, 1, max_row);
 655                 xfer(dst, src, newrow, 0, row, cols[col_x], prob);
 656         }
 657 
 658         // The remaining columns use the specified drainmax.
 659         unsigned newrow = utils::clamp(row_i + drainmax, 1, max_row);
 660         for(; col_x < cols.size(); ++col_x) {
 661                 xfer(dst, src, newrow, cols[col_x] - damage, row, cols[col_x], prob);
 662         }
 663 }

 665 /**
 666  * Transfers a portion (value * prob) of each column in a plane to another.
 667  * Each column in the @a src plane gets shifted @a damage columns towards 0, and
 668  * also shifted into the @a dst plane. In addition, the rows can shift if
 669  * @a drain constant or @a drain_percent is nonzero.
 670  */
 671 void prob_matrix::shift_cols(
 672                 unsigned dst, unsigned src, unsigned damage, double prob, int drain_constant, int drain_percent)
 673 {
 674         int drainmax = (drain_percent * (static_cast<signed>(damage)) / 100 + drain_constant);
 675 
 676         if(drain_constant || drain_percent) {
 677                 debug(("Drains %i (%i%% of %u plus %i)\n", drainmax, drain_percent, damage, drain_constant));
 678         }
 679 
 680         // Get lists of indices currently used in the source plane.
 681         // (This needs to be cached since we might add indices while shifting.)
 682         const std::vector<unsigned> rows(used_rows_[src].begin(), used_rows_[src].end());
 683         const std::vector<unsigned> cols(used_cols_[src].begin(), used_cols_[src].end());
 684 
 685         // Loop downwards if we drain positive, but upwards if we drain negative,
 686         // so we write behind us (for when src == dst).
 687         if(drainmax > 0) {
 688                 // rows[0] is excluded since that should be 0, representing already dead.
 689                 for(unsigned row_x = rows.size() - 1; row_x != 0; --row_x) {
 690                         shift_cols_in_row(dst, src, rows[row_x], cols, damage, prob, drainmax, drain_constant, drain_percent);
 691                 }
 692         } else {
 693                 // rows[0] is excluded since that should be 0, representing already dead.
 694                 for(unsigned row_x = 1; row_x != rows.size(); ++row_x) {
 695                         shift_cols_in_row(dst, src, rows[row_x], cols, damage, prob, drainmax, drain_constant, drain_percent);
 696                 }
 697         }
 698 }

 858 /**
 859  * Set all values to zero and clear the lists of used columns/rows.
 860  */
 861 void prob_matrix::clear()
 862 {
 863         for(unsigned int p = 0u; p < NUM_PLANES; ++p) {
 864                 if(!plane_used(p)) {
 865                         continue;
 866                 }
 867 
 868                 if(used_rows_[p].empty()) {
 869                         // Nothing to do
 870                         continue;
 871                 }
 872 
 873                 decltype(used_rows_[p].begin()) first_row, last_row;
 874                 std::tie(first_row, last_row) = std::minmax_element(used_rows_[p].begin(), used_rows_[p].end());
 875                 for(unsigned int r = *first_row; r <= *last_row; ++r) {
 876                         for(unsigned int c = 0u; c < cols_; ++c) {
 877                                 plane_[p][r * cols_ + c] = 0.0;
 878                         }
 879                 }
 880 
 881                 used_rows_[p].clear();
 882                 used_cols_[p].clear();
 883 
 884                 /* Row and column 0 are always considered to be used.
 885                 Functions like merge_col() access *used_rows_[plane].begin() without checking if there are
 886                 any used rows: thus, ensuring that there are used rows and columns is necessary to avoid
 887                 memory corruption. */
 888                 used_rows_[p].insert(0u);
 889                 used_cols_[p].insert(0u);
 890         }
 891 }

 906 /**
 907  * What is the chance that an indicated combatant (one of them) is at zero?
 908  */
 909 double prob_matrix::prob_of_zero(bool check_a, bool check_b) const
 910 {
 911         double prob = 0.0;
 912 
 913         for(unsigned p = 0; p < NUM_PLANES; ++p) {
 914                 if(!plane_used(p)) {
 915                         continue;
 916                 }
 917 
 918                 // Column 0 is where b is at zero.
 919                 if(check_b) {
 920                         for(const unsigned& row : used_rows_[p]) {
 921                                 prob += val(p, row, 0);
 922                         }
 923                 }
 924 
 925                 // Row 0 is where a is at zero.
 926                 if(check_a) {
 927                         for(const unsigned& col : used_cols_[p]) {
 928                                 prob += val(p, 0, col);
 929                         }
 930                 }
 931                 // Theoretically, if checking both, we should subtract the chance that
 932                 // both are dead, but that chance is zero, so don't worry about it.
 933         }
 934 
 935         return prob;
 936 }

1271 // Shift combat_matrix to reflect the probability 'hit_chance' that damage
1272 // is done to 'b'.
1273 void probability_combat_matrix::receive_blow_b(double hit_chance)
1274 {
1275         // Walk backwards so we don't copy already-copied matrix planes.
1276         unsigned src = NUM_PLANES;
1277         while(src-- != 0) {
1278                 if(!plane_used(src)) {
1279                         continue;
1280                 }
1281 
1282                 // If A slows us, we go from 0=>2, 1=>3, 2=>2 3=>3.
1283                 int dst = a_slows_ ? src | 2 : src;
1284 
1285                 // A is slow in planes 1 and 3.
1286                 unsigned damage = (src & 1) ? a_slow_damage_ : a_damage_;
1287 
1288                 shift_cols(dst, src, damage, hit_chance, a_drain_constant_, a_drain_percent_);
1289         }
1290 }

1463 void monte_carlo_combat_matrix::simulate()
1464 {
1465         for(unsigned int i = 0u; i < NUM_ITERATIONS; ++i) {
1466                 bool a_hit = false;
1467                 bool b_hit = false;
1468                 bool a_slowed = randomness::generator->get_random_bool(a_initially_slowed_chance_);
1469                 bool b_slowed = randomness::generator->get_random_bool(b_initially_slowed_chance_);
1470                 const std::vector<double>& a_initial = a_slowed ? a_initial_slowed_ : a_initial_;
1471                 const std::vector<double>& b_initial = b_slowed ? b_initial_slowed_ : b_initial_;
1472                 unsigned int a_hp = randomness::generator->get_random_element(a_initial.begin(), a_initial.end());
1473                 unsigned int b_hp = randomness::generator->get_random_element(b_initial.begin(), b_initial.end());
1474                 unsigned int a_strikes = calc_blows_a(a_hp);
1475                 unsigned int b_strikes = calc_blows_b(b_hp);
1476 
1477                 for(unsigned int j = 0u; j < rounds_ && a_hp > 0u && b_hp > 0u; ++j) {
1478                         for(unsigned int k = 0u; k < std::max(a_strikes, b_strikes); ++k) {
1479                                 if(k < a_strikes) {
1480                                         if(randomness::generator->get_random_bool(a_hit_chance_)) {
1481                                                 // A hits B
1482                                                 unsigned int damage = a_slowed ? a_slow_damage_ : a_damage_;
1483                                                 damage = std::min(damage, b_hp);
1484                                                 b_hit = true;
1485                                                 b_slowed |= a_slows_;
1486 
1487                                                 int drain_amount = (a_drain_percent_ * static_cast<signed>(damage) / 100 + a_drain_constant_);
1488                                                 a_hp = utils::clamp(a_hp + drain_amount, 1u, a_max_hp_);
1489 
1490                                                 b_hp -= damage;
1491 
1492                                                 if(b_hp == 0u) {
1493                                                         // A killed B
1494                                                         break;
1495                                                 }
1496                                         }
1497                                 }
1498 
1499                                 if(k < b_strikes) {
1500                                         if(randomness::generator->get_random_bool(b_hit_chance_)) {
1501                                                 // B hits A
1502                                                 unsigned int damage = b_slowed ? b_slow_damage_ : b_damage_;
1503                                                 damage = std::min(damage, a_hp);
1504                                                 a_hit = true;
1505                                                 a_slowed |= b_slows_;
1506 
1507                                                 int drain_amount = (b_drain_percent_ * static_cast<signed>(damage) / 100 + b_drain_constant_);
1508                                                 b_hp = utils::clamp(b_hp + drain_amount, 1u, b_max_hp_);
1509 
1510                                                 a_hp -= damage;
1511 
1512                                                 if(a_hp == 0u) {
1513                                                         // B killed A
1514                                                         break;
1515                                                 }
1516                                         }
1517                                 }
1518                         }
1519                 }
1520 
1521                 iterations_a_hit_ += a_hit ? 1 : 0;
1522                 iterations_b_hit_ += b_hit ? 1 : 0;
1523 
1524                 record_monte_carlo_result(a_hp, b_hp, a_slowed, b_slowed);
1525         }
1526 }

1609 void monte_carlo_combat_matrix::scale_probabilities(
1610                 const std::vector<double>& source, std::vector<double>& target, double divisor, unsigned int singular_hp)
1611 {
1612         if(divisor == 0.0) {
1613                 // Happens if the "target" HP distribution vector isn't used,
1614                 // in which case it's not necessary to scale the probabilities.
1615                 return;
1616         }
1617 
1618         if(source.empty()) {
1619                 target.resize(singular_hp + 1u, 0.0);
1620                 target[singular_hp] = 1.0;
1621         } else {
1622                 std::transform(
1623                                 source.begin(), source.end(), std::back_inserter(target), [=](double prob) { return prob / divisor; });
1624         }
1625 
1626         assert(std::abs(std::accumulate(target.begin(), target.end(), 0.0) - 1.0) < 0.001);
1627 }

1684 void forced_levelup(std::vector<double>& hp_dist)
1685 {
1686         /* If we survive the combat, we will level up. So the probability
1687            of death is unchanged, but all other cases get merged into the
1688            fully healed case. */
1689         auto i = hp_dist.begin();
1690         // Skip to the second value.
1691         for(++i; i != hp_dist.end(); ++i) {
1692                 *i = 0;
1693         }
1694 
1695         // Full heal unless dead.
1696         hp_dist.back() = 1 - hp_dist.front();
1697 }

1699 void conditional_levelup(std::vector<double>& hp_dist, double kill_prob)
1700 {
1701         /* If we kill, we will level up. So then the damage we had becomes
1702            less probable since it's now conditional on us not leveling up.
1703            This doesn't apply to the probability of us dying, of course. */
1704         double scalefactor = 0;
1705         const double chance_to_survive = 1 - hp_dist.front();
1706         if(chance_to_survive > DBL_MIN) {
1707                 scalefactor = 1 - kill_prob / chance_to_survive;
1708         }
1709 
1710         auto i = hp_dist.begin();
1711         // Skip to the second value.
1712         for(++i; i != hp_dist.end(); ++i) {
1713                 *i *= scalefactor;
1714         }
1715 
1716         // Full heal if leveled up.
1717         hp_dist.back() += kill_prob;
1718 }

2280                 /* The slowed probability depends on in how many rounds
2281                  * the combatant happened to be slowed.
2282                  * We need to recalculate it based on the HP distribution.
2283                  */
2284                 slowed = std::accumulate(summary[1].begin(), summary[1].end(), 0.0);
2285                 opponent.slowed = std::accumulate(opponent.summary[1].begin(), opponent.summary[1].end(), 0.0);
2286         }
2287 
2288         untouched *= self_not_hit;
2289         opponent.untouched *= opp_not_hit;
2290 }

2457 static void run(unsigned specific_battle)
2458 {
2459         using std::chrono::duration_cast;
2460         using std::chrono::microseconds;
2461 
2462         // N^2 battles
2463         struct battle_context_unit_stats* stats[NUM_UNITS];
2464         struct combatant* u[NUM_UNITS];
2465         unsigned int i, j, k, battle = 0;
2466         std::chrono::high_resolution_clock::time_point start, end;
2467 
2468         for(i = 0; i < NUM_UNITS; ++i) {
2469                 unsigned alt = i + 74; // To offset some cycles.
2470                 // To get somewhat realistic performance data, try to approximate
2471                 // hit point ranges for mainline units (say 25-60 max hitpoints?)
2472                 unsigned max_hp = (i * 2) % 23 + (i * 3) % 14 + 25;
2473                 unsigned hp = (alt * 5) % max_hp + 1;
2474                 stats[i] = new battle_context_unit_stats(alt % 8 + 2, // damage
2475                                 (alt % 19 + 3) / 4,                           // number of strikes
2476                                 hp, max_hp,
2477                                 (i % 6) * 10 + 30, // hit chance

2479                                 (i % 11) % 3 == 0, // slows
2480                                 false,             // slowed

2483                                 i % 5 == 0);       // swarm

2485                 list_combatant(*stats[i], i + 1);
2486         }
2487 
2488         start = std::chrono::high_resolution_clock::now();
2489         // Go through all fights with two attackers (j and k attacking i).
2490         for(i = 0; i < NUM_UNITS; ++i) {
2491                 for(j = 0; j < NUM_UNITS; ++j) {
2492                         if(i == j) {
2493                                 continue;
