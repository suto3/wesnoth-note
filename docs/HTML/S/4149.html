 218 # Research Complete Messages
 219 # (These are shown at the start of a player's turn - they are triggered by an event included in the auto-studying macro)

 221 #define FARMING_RESEARCH_COMPLETE
 222     [if]
 223         [variable]
 224             name=player_$side_number|.farming.progress
 225             greater_than_equal_to=$player_$side_number|.farming.target
 226         [/variable]
 227 
 228         [then]
 229             [set_variable]
 230                 name=player_$side_number|.farming.gold
 231                 add=1
 232             [/set_variable]
 233 
 234             [message]
 235                 speaker=narrator
 236                 caption= _ "Study Complete"
 237                 image=items/flower4.png
 238                 message= _ "$player_$side_number|.name|’s farms now produce $player_$side_number|.farming.gold gold."
 239             [/message]
 240 
 241             [set_variable]
 242                 name=player_$side_number|.farming.progress
 243                 sub=$player_$side_number|.farming.target
 244             [/set_variable]
 245             [set_variable]
 246                 name=player_$side_number|.farming.target
 247                 add=1
 248             [/set_variable]
 249         [/then]
 250     [/if]
 251 #enddef
