  26 const unit_animation* unit_animation_component::choose_animation(const display& disp, const map_location& loc,const std::string& event,
  27                 const map_location& second_loc,const int value,const unit_animation::hit_type hit,
  28                 const_attack_ptr attack, const_attack_ptr second_attack, int swing_num)
  29 {
  30         // Select one of the matching animations at random
  31         std::vector<const unit_animation*> options;
  32         int max_val = unit_animation::MATCH_FAIL;
  33         for(std::vector<unit_animation>::const_iterator i = animations_.begin(); i != animations_.end(); ++i) {
  34                 int matching = i->matches(disp,loc,second_loc,&u_,event,value,hit,attack,second_attack,swing_num);
  35                 if(matching > unit_animation::MATCH_FAIL && matching == max_val) {
  36                         options.push_back(&*i);
  37                 } else if(matching > max_val) {
  38                         max_val = matching;
  39                         options.clear();
  40                         options.push_back(&*i);
  41                 }
  42         }
  43 
  44         if(max_val == unit_animation::MATCH_FAIL) {
  45                 return nullptr;
  46         }
  47         return options[rand()%options.size()];
  48 }

  97 void unit_animation_component::start_animation (int start_time, const unit_animation *animation,
  98         bool with_bars,  const std::string &text, color_t text_color, STATE state)
  99 {
 100         const display * disp =  display::get_singleton();
 101         if (!animation) {
 102                 if (state == STATE_STANDING)
 103                         state_ = state;
 104                 if (!anim_ && state_ != STATE_STANDING)
 105                         set_standing(with_bars);
 106                 return ;
 107         }
 108         state_ = state;
 109         // everything except standing select and idle
 110         bool accelerate = (state != STATE_FORGET && state != STATE_STANDING);
 111         draw_bars_ =  with_bars;
 112         anim_.reset(new unit_animation(*animation));
 113         const int real_start_time = start_time == INT_MAX ? anim_->get_begin_time() : start_time;
 114         anim_->start_animation(real_start_time, u_.loc_, u_.loc_.get_direction(u_.facing_),
 115                  text, text_color, accelerate);
 116         frame_begin_time_ = anim_->get_begin_time() -1;
 117         if (disp->idle_anim()) {
 118                 next_idling_ = get_current_animation_tick()
 119                         + static_cast<int>((20000 + rand() % 20000) * disp->idle_anim_rate());
 120         } else {
 121                 next_idling_ = INT_MAX;
 122         }
 123 }

 125 void unit_animation_component::refresh()
 126 {
 127         if (state_ == STATE_FORGET && anim_ && anim_->animation_finished_potential())
 128         {
 129                 set_standing();
 130                 return;
 131         }
 132         display &disp = *display::get_singleton();
 133         if (state_ != STATE_STANDING || get_current_animation_tick() < next_idling_ ||
 134             !disp.tile_nearly_on_screen(u_.loc_) || u_.incapacitated())
 135         {
 136                 return;
 137         }
 138         if (get_current_animation_tick() > next_idling_ + 1000)
 139         {
 140                 // prevent all units animating at the same time
 141                 if (disp.idle_anim()) {
 142                         next_idling_ = get_current_animation_tick()
 143                                 + static_cast<int>((20000 + rand() % 20000) * disp.idle_anim_rate());
 144                 } else {
 145                         next_idling_ = INT_MAX;
 146                 }
 147         } else {
 148                 set_idling();
 149         }
 150 }

 158 bool unit_animation_component::invalidate (const display & disp)
 159 {
 160         bool result = false;
 161 
 162         // Very early calls, anim not initialized yet
 163         if(get_animation()) {
 164                 frame_parameters params;
 165                 const gamemap & map = disp.get_map();
 166                 const t_translation::terrain_code terrain = map.get_terrain(u_.loc_);
 167                 const terrain_type& terrain_info = map.get_terrain_info(terrain);
 168 
 169                 int height_adjust = static_cast<int>(terrain_info.unit_height_adjust() * disp.get_zoom_factor());
 170                 if (u_.is_flying() && height_adjust < 0) {
 171                         height_adjust = 0;
 172                 }
 173                 params.y -= height_adjust;
 174                 params.halo_y -= height_adjust;
 175                 params.image_mod = u_.image_mods();
 176                 params.halo_mod = u_.TC_image_mods();
 177                 params.image= u_.default_anim_image();
 178 
 179                 result |= get_animation()->invalidate(params);
 180         }
 181 
 182         return result;
 183 
 184 }

 196 void unit_animation_component::apply_new_animation_effect(const config & effect) {
 197         if(effect["id"].empty()) {
 198                 unit_animation::add_anims(animations_, effect);
 199         } else {
 200                 static std::map< std::string, std::vector<unit_animation> > animation_cache;
 201                 std::vector<unit_animation> &built = animation_cache[effect["id"]];
 202                 if(built.empty()) {
 203                         unit_animation::add_anims(built, effect);
 204                 }
 205                 animations_.insert(animations_.end(),built.begin(),built.end());
 206         }
 207 }

 209 std::vector<std::string> unit_animation_component::get_flags() {
 210         std::set<std::string> result;
 211         for(const auto& anim : animations_) {
 212                 const std::vector<std::string>& flags = anim.get_flags();
 213                 std::copy_if(flags.begin(), flags.end(), std::inserter(result, result.begin()), [](const std::string flag) {
 214                         return !(flag.empty() || (flag.front() == '_' && flag.back() == '_'));
 215                 });
 216         }
 217         return std::vector<std::string>(result.begin(), result.end());
 218 }
