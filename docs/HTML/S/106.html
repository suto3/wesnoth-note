 337 // Returns whether a hotkey was deleted.
 338 bool remove_wml_hotkey(const std::string& id)
 339 {
 340         const hotkey::hotkey_command& command = get_hotkey_command(id);
 341         if(command.id == hotkey::HOTKEY_NULL)
 342         {
 343                 LOG_G << "remove_wml_hotkey: command with id=" + id + " doesn't exist\n";
 344                 return false;
 345         }
 346         else if (command.id != hotkey::HOTKEY_WML)
 347         {
 348                 LOG_G << "remove_wml_hotkey: command with id=" + id + " cannot be removed because it is no wml menu hotkey\n";
 349                 return false;
 350         }
 351         else
 352         {
 353                 LOG_G << "removing wml hotkey with id=" + id + "\n";
 354                 for(boost::ptr_vector<hotkey_command>::iterator itor = known_hotkeys.begin(); itor != known_hotkeys.end(); ++itor)
 355                 {
 356                         if(itor->command == id)
 357                         {
 358                                 known_hotkeys.erase(itor);
 359                                 break;
 360                         }
 361                 }
 362                 //command_map_ might be all wrong now, so we need to rebuild.
 363                 command_map_.clear();
 364                 for(size_t index = 0; index < known_hotkeys.size(); index++)
 365                 {
 366                         command_map_[known_hotkeys[index].command] = index;
 367                 }
 368                 return true;
 369         }
 370 }

 377 void add_wml_hotkey(const std::string& id, const t_string& description, const config& default_hotkey)
 378 {
 379         if(id == "null")
 380         {
 381                 LOG_G << "Couldn't add wml hotkey with null id and description = '" << description << "'.\n";
 382                 return;
 383         }
 384         else
 385         {
 386                 if(has_hotkey_command(id))
 387                 {
 388                         LOG_G << "Hotkey with id '" << id << "' already exists. Deleting the old hotkey_command.\n";
 389                         remove_wml_hotkey(id);
 390                 }
 391                 DBG_G << "Added wml hotkey with id = '" << id << "' and description = '" << description << "'.\n";
 392                 known_hotkeys.push_back(new hotkey_command(hotkey::HOTKEY_WML, id, description, false, scope_game, HKCAT_CUSTOM, t_string("")));
 393 
 394                 command_map_[id] = known_hotkeys.size() - 1;
 395 
 396                 if(!default_hotkey.empty() && !has_hotkey_item(id))
 397                 {
 398                         hotkey::hotkey_ptr new_item = hotkey::load_from_config(default_hotkey);
 399                         new_item->set_command(id);
 400                         if(new_item->valid())
 401                         {
 402                                 DBG_G << "added default description for the wml hotkey with id=" + id;
 403                                 add_hotkey(new_item);
 404                         }
 405                         else
 406                         {
 407                                 ERR_CF << "failed to add default hotkey with id=" + id;
 408                         }
 409                 }
 410         }
 411 }

 432 bool hotkey_command::null() const
 433 {
 434         if(id == HOTKEY_NULL || command == "null")
 435         {
 436                 const hotkey_command& null_cmd = null_command();
 437                 if(id == null_cmd.id && command == null_cmd.command && scope == null_cmd.scope && description == null_cmd.description)
 438                         return true;
 439                 else
 440                 {
 441                         ERR_G << "the hotkey command seems to be the null command but it is not 100% sure. This shouldn't happen";
 442                         return true;
 443                 }
 444         }
 445         return false;
 446 }

 462 {
 463         //it is the last entry in that array, and the indexes in hotkey_list_ and known_hotkeys are the same.
 464         return known_hotkeys[sizeof(hotkey_list_) / sizeof(hotkey_list_[0])  - 1];

 467 void delete_all_wml_hotkeys()
 468 {
 469         while(known_hotkeys.back().id == hotkey::HOTKEY_WML)
 470         {
 471                 command_map_.erase(known_hotkeys.back().command);
 472                 //according to some page in the Internet  .back() returns a reference not an iterator, so i use this.
 473                 boost::ptr_vector<hotkey_command>::iterator last_element = known_hotkeys.end();
 474                 --last_element;
 475                 //boost::ptr_vector<hotkey_command> will manage the deleting of the object for me.
 476                 known_hotkeys.erase(last_element);
 477         }
 478 }

 492         //the size value is just random set.
 493         boost::ptr_vector<hotkey_command> known_hotkeys_temp(200);
 494         known_hotkeys = known_hotkeys_temp;
 495 
 496         size_t i = 0;
 497         for(hotkey_command_temp& cmd : hotkey_list_)
 498         {
 499                 known_hotkeys.push_back( new hotkey_command(cmd.id, cmd.command, t_string(cmd.description, "wesnoth-lib"), cmd.hidden, cmd.scope, cmd.category, t_string(cmd.tooltip, "wesnoth-lib")));
 500                 command_map_[cmd.command] = i;
 501                 i++;
 502         }
 503 }
