 110     private void initialize( )
 111     {
 112         if( selection != null && selection.isEmpty( ) == false
 113             && selection instanceof IStructuredSelection ) {
 114             IStructuredSelection ssel = ( IStructuredSelection ) selection;
 115             if( ssel.size( ) > 1 ) {
 116                 return;
 117             }
 118             Object obj = ssel.getFirstElement( );
 119             if( obj instanceof IResource ) {
 120                 IContainer container;
 121                 if( obj instanceof IContainer ) {
 122                     container = ( IContainer ) obj;
 123                 }
 124                 else {
 125                     container = ( ( IResource ) obj ).getParent( );
 126                 }
 127                 containerText.setText( container.getFullPath( ).toString( ) );
 128             }
 129         }
 130         fileText.setText( "new_file.cfg" ); //$NON-NLS-1$
 131     }

 155     private void dialogChanged( )
 156     {
 157         IResource container = ResourcesPlugin.getWorkspace( ).getRoot( )
 158             .findMember( new Path( getContainerName( ) ) );
 159         String fileName = getFileName( );
 160 
 161         if( getContainerName( ).length( ) == 0 ) {
 162             updateStatus( Messages.NewConfigFilePage0_8 );
 163             return;
 164         }
 165         if( container == null
 166             || ( container.getType( ) & ( IResource.PROJECT | IResource.FOLDER ) ) == 0 ) {
 167             updateStatus( Messages.NewConfigFilePage0_9 );
 168             return;
 169         }
 170         if( ! container.isAccessible( ) ) {
 171             updateStatus( Messages.NewConfigFilePage0_10 );
 172             return;
 173         }
 174         if( fileName.length( ) == 0 ) {
 175             updateStatus( Messages.NewConfigFilePage0_11 );
 176             return;
 177         }
 178         if( fileName.replace( '\\', '/' ).indexOf( '/', 1 ) > 0 ) {
 179             updateStatus( Messages.NewConfigFilePage0_12 );
 180             return;
 181         }
 182         int dotLoc = fileName.lastIndexOf( '.' );
 183         if( dotLoc != - 1 ) {
 184             String ext = fileName.substring( dotLoc + 1 );
 185             if( ext.equalsIgnoreCase( "cfg" ) == false ) { //$NON-NLS-1$
 186                 updateStatus( Messages.NewConfigFilePage0_14 );
 187                 return;
 188             }
 189         }
 190         updateStatus( null );
 191     }
