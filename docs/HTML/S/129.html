 109 void parser::operator()()
 110 {
 111         cfg_.clear();
 112         elements.push(element(&cfg_, ""));
 113 
 114         do {
 115                 tok_.next_token();
 116 
 117                 switch(tok_.current_token().type) {
 118                 case token::LF:
 119                         continue;
 120                 case '[':
 121                         parse_element();
 122                         break;
 123                 case token::STRING:
 124                         parse_variable();
 125                         break;
 126                 default:
 127                         if (static_cast<unsigned char>(tok_.current_token().value[0]) == 0xEF &&
 128                             static_cast<unsigned char>(tok_.next_token().value[0])    == 0xBB &&
 129                             static_cast<unsigned char>(tok_.next_token().value[0])    == 0xBF)
 130                         {
 131                                 utils::string_map i18n_symbols;
 132                                 std::stringstream ss;
 133                                 ss << tok_.get_start_line() << " " << tok_.get_file();
 134                                 ERR_CF << lineno_string(i18n_symbols,
 135                                                                                 ss.str(),
 136                                                                                 "Skipping over a utf8 BOM at $pos")
 137                                            << '\n';
 138                         } else {
 139                                 error(_("Unexpected characters at line start"));
 140                         }
 141                         break;
 142                 case token::END:
 143                         break;
 144                 }
 145         } while (tok_.current_token().type != token::END);
 146 
 147         // The main element should be there. If it is not, this is a parser error.
 148         assert(!elements.empty());
 149 
 150         if(elements.size() != 1) {
 151                 utils::string_map i18n_symbols;
 152                 i18n_symbols["tag"] = elements.top().name;
 153                 std::stringstream ss;
 154                 ss << elements.top().start_line << " " << elements.top().file;
 155                 error(lineno_string(i18n_symbols, ss.str(),
 156                                 _("Missing closing tag for tag [$tag]"),
 157                                 _("expected at $pos")), _("opened at $pos"));
 158         }
 159 }

 161 void parser::parse_element()
 162 {
 163         tok_.next_token();
 164         std::string elname;
 165         config* current_element = nullptr;
 166         switch(tok_.current_token().type) {
 167         case token::STRING: // [element]
 168                 elname = tok_.current_token().value;
 169                 if (tok_.next_token().type != ']')
 170                         error(_("Unterminated [element] tag"));
 171                 // Add the element
 172                 current_element = &(elements.top().cfg->add_child(elname));
 173                 elements.push(element(current_element, elname, tok_.get_start_line(), tok_.get_file()));
 174                 if (validator_){
 175                         validator_->open_tag(elname,tok_.get_start_line(),
 176                                                                   tok_.get_file());
 177                 }
 178                 break;
 179 
 180         case '+': // [+element]
 181                 if (tok_.next_token().type != token::STRING)
 182                         error(_("Invalid tag name"));
 183                 elname = tok_.current_token().value;
 184                 if (tok_.next_token().type != ']')
 185                         error(_("Unterminated [+element] tag"));
 186 
 187                 // Find the last child of the current element whose name is
 188                 // element
 189                 if (config &c = elements.top().cfg->child(elname, -1)) {
 190                         current_element = &c;
 191                         if (validator_){
 192                                 validator_->open_tag(elname,tok_.get_start_line(),
 193                                                                          tok_.get_file(),true);
 194                         }
 195                 } else {
 196                         current_element = &elements.top().cfg->add_child(elname);
 197                         if (validator_){
 198                                 validator_->open_tag(elname,tok_.get_start_line(),
 199                                                                          tok_.get_file());
 200                         }
 201                 }
 202                 elements.push(element(current_element, elname, tok_.get_start_line(), tok_.get_file()));
 203                 break;
 204 
 205         case '/': // [/element]
 206                 if(tok_.next_token().type != token::STRING)
 207                         error(_("Invalid closing tag name"));
 208                 elname = tok_.current_token().value;
 209                 if(tok_.next_token().type != ']')
 210                         error(_("Unterminated closing tag"));
 211                 if(elements.size() <= 1)
 212                         error(_("Unexpected closing tag"));
 213                 if(elname != elements.top().name) {
 214                         utils::string_map i18n_symbols;
 215                         i18n_symbols["tag1"] = elements.top().name;
 216                         i18n_symbols["tag2"] = elname;
 217                         std::stringstream ss;
 218                         ss << elements.top().start_line << " " << elements.top().file;
 219                         error(lineno_string(i18n_symbols, ss.str(),
 220                                         _("Found invalid closing tag [/$tag2] for tag [$tag1]"),
 221                                         _("opened at $pos")), _("closed at $pos"));
 222                 }
 223                 if(validator_){
 224                         element & el= elements.top();
 225                         validator_->validate(*el.cfg,el.name,el.start_line,el.file);
 226                         validator_->close_tag();
 227                 }
 228                 elements.pop();
 229                 break;
 230         default:
 231                 error(_("Invalid tag name"));
 232         }
 233 }

 235 void parser::parse_variable()
 236 {
 237         config& cfg = *elements.top().cfg;
 238         std::vector<std::string> variables;
 239         variables.emplace_back();
 240 
 241         while (tok_.current_token().type != '=') {
 242                 switch(tok_.current_token().type) {
 243                 case token::STRING:
 244                         if(!variables.back().empty())
 245                                 variables.back() += ' ';
 246                         variables.back() += tok_.current_token().value;
 247                         break;
 248                 case ',':
 249                         if(variables.back().empty()) {
 250                                 error(_("Empty variable name"));
 251                         } else {
 252                                 variables.emplace_back();
 253                         }
 254                         break;
 255                 default:
 256                         error(_("Unexpected characters after variable name (expected , or =)"));
 257                         break;
 258                 }
 259                 tok_.next_token();
 260         }
 261         if(variables.back().empty())
 262                 error(_("Empty variable name"));
 263 
 264         t_string_base buffer;
 265 
 266         std::vector<std::string>::const_iterator curvar = variables.begin();
 267 
 268         bool ignore_next_newlines = false, previous_string = false;
 269         while(1) {
 270                 tok_.next_token();
 271                 assert(curvar != variables.end());
 272 
 273                 switch (tok_.current_token().type) {
 274                 case ',':
 275                         if ((curvar+1) != variables.end()) {
 276                                 if (buffer.translatable())
 277                                         cfg[*curvar] = t_string(buffer);
 278                                 else
 279                                         cfg[*curvar] = buffer.value();
 280                                 if(validator_){
 281                                         validator_->validate_key (cfg,*curvar,buffer.value(),
 282                                                                                           tok_.get_start_line (),
 283                                                                                           tok_.get_file ());
 284                                 }
 285                                 buffer = t_string_base();
 286                                 ++curvar;
 287                         } else {
 288                                 buffer += ",";
 289                         }
 290                         break;
 291                 case '_':
 292                         tok_.next_token();
 293                         switch (tok_.current_token().type) {
 294                         case token::UNTERMINATED_QSTRING:
 295                                 error(_("Unterminated quoted string"));
 296                                 break;
 297                         case token::QSTRING:
 298                                 buffer += t_string_base(tok_.current_token().value, tok_.textdomain());
 299                                 break;
 300                         default:
 301                                 buffer += "_";
 302                                 buffer += tok_.current_token().value;
 303                                 break;
 304                         case token::END:
 305                         case token::LF:
 306                                 buffer += "_";
 307                                 goto finish;
 308                         }
 309                         break;
 310                 case '+':
 311                         ignore_next_newlines = true;
 312                         continue;
 313                 case token::STRING:
 314                         if(previous_string) {
 315                                 buffer += " ";
 316                         }
 317                         FALLTHROUGH;
 318                 default:
 319                         buffer += tok_.current_token().value;
 320                         break;
 321                 case token::QSTRING:
 322                         buffer += tok_.current_token().value;
 323                         break;
 324                 case token::UNTERMINATED_QSTRING:
 325                         error(_("Unterminated quoted string"));
 326                         break;
 327                 case token::LF:
 328                         if(ignore_next_newlines) {
 329                                 continue;
 330                         }
 331                         FALLTHROUGH;
 332                 case token::END:
 333                         goto finish;
 334                 }
 335 
 336                 previous_string = tok_.current_token().type == token::STRING;
 337                 ignore_next_newlines = false;
 338         }
 339 
 340         finish:
 341         if (buffer.translatable())
 342                 cfg[*curvar] = t_string(buffer);
 343         else
 344                 cfg[*curvar] = buffer.value();
 345         if(validator_){
 346                 validator_->validate_key (cfg,*curvar,buffer.value(),
 347                                                                   tok_.get_start_line (),
 348                                                                   tok_.get_file ());
 349         }
 350         while (++curvar != variables.end()) {
 351                 cfg[*curvar] = "";
 352         }
 353 }

 358 std::string parser::lineno_string(utils::string_map& i18n_symbols,
 359                                                                   const std::string& lineno,
 360                                                                   const std::string& error_string,
 361                                                                   const std::string& hint_string,
 362                                                                   const std::string& debug_string)
 363 {
 364         i18n_symbols["pos"] = ::lineno_string(lineno);
 365         std::string result = error_string;
 366 
 367         if(!hint_string.empty()) {
 368                 result += '\n' + hint_string;
 369         }
 370 
 371         if(!debug_string.empty()) {
 372                 result += '\n' + debug_string;
 373         }
 374 
 375         for(utils::string_map::value_type& var : i18n_symbols)
 376                 boost::algorithm::replace_all(result, std::string("$") + var.first, std::string(var.second));
 377         return result;
 378 }

 380 void parser::error(const std::string& error_type, const std::string& pos_format)
 381 {
 382         std::string hint_string = pos_format;
 383 
 384         if(hint_string.empty()) {
 385                 hint_string = _("at $pos");
 386         }
 387 
 388         utils::string_map i18n_symbols;
 389         i18n_symbols["error"] = error_type;
 390 
 391         std::stringstream ss;
 392         ss << tok_.get_start_line() << " " << tok_.get_file();
 393 
 394 #ifdef DEBUG_TOKENIZER
 395         i18n_symbols["value"] = tok_.current_token().value;
 396         i18n_symbols["previous_value"] = tok_.previous_token().value;
 397 
 398         const std::string& tok_state =
 399                 _("Value: '$value' Previous: '$previous_value'");
 400 #else
 401         const std::string& tok_state = "";
 402 #endif
 403 
 404         const std::string& message =
 405                 lineno_string(i18n_symbols, ss.str(), "$error", hint_string, tok_state);
 406 
 407         throw config::error(message);
 408 }

 423 template <typename decompressor>
 424 void read_compressed(config &cfg, std::istream &file, abstract_validator * validator)
 425 {
 426         //an empty gzip file seems to confuse boost on msvc
 427         //so return early if this is the case
 428         if (file.peek() == EOF) {
 429                 return;
 430         }
 431         boost::iostreams::filtering_stream<boost::iostreams::input> filter;
 432         filter.push(decompressor());
 433         filter.push(file);
 434 
 435 
 436 
 437         // This causes especially gzip_error (and the corresponding bz2 error), std::ios_base::failure to be thrown here.
 438         // save_index_class::data expects that and config_cache::read_cache and other functions are also capable of catching.
 439         // Note that parser(cuff, filter,validator)(); -> tokenizer::tokenizer can throw exeptions too (meaning this functions did already throw these exceptions before this patch).
 440         // We try to fix https://svn.boost.org/trac/boost/ticket/5237 by not creating empty gz files.
 441         filter.exceptions(filter.exceptions() | std::ios_base::badbit);
 442 
 443         /*
 444          * It sometimes seems the file is not empty but still no real data.
 445          * Filter that case here. It might be previous test is no longer required
 446          * but simply keep it.
 447          */
 448 
 449         // on msvc filter.peek() != EOF does not imply filter.good().
 450         // we never create empty compressed gzip files because boosts gzip fails at doing that.
 451         // but empty compressed bz2 files are possible.
 452         if(filter.peek() == EOF) {
 453                 LOG_CF << "Empty compressed file or error at reading a compressed file.";
 454                 return;
 455         }
 456 
 457 
 458         if(!filter.good()) {
 459                 LOG_CF << " filter.peek() != EOF but !filter.good(), this indicates a malformed gz stream, and can make wesnoth crash.";
 460         }
 461 
 462         parser(cfg, filter,validator)();
 463 }

 478         /**
 479          * Copies a string fragment and converts it to a suitable format for WML.
 480          * (I.e., quotes are doubled.)
 481          */
 482         std::string escaped_string(const std::string::const_iterator &begin,
 483                                    const std::string::const_iterator &end)
 484         {
 485                 std::string res;
 486                 std::string::const_iterator iter = begin;
 487                 while ( iter != end ) {
 488                         const char c = *iter;
 489                         res.append(c == '"' ? 2 : 1, c);
 490                         ++iter;
 491                 }
 492                 return res;
 493         }
 494         /**
 495          * Copies a string and converts it to a suitable format for WML.
 496          * (I.e., quotes are doubled.)
 497          */
 498         inline std::string escaped_string(const std::string &value)
 499         {
 500                 return escaped_string(value.begin(), value.end());
 501         }

 503         class write_key_val_visitor : public boost::static_visitor<void>
 504         {
 505                 std::ostream &out_;
 506                 const unsigned level_;
 507                 std::string &textdomain_;
 508                 const std::string &key_;
 509 
 510         public:
 511                 write_key_val_visitor(std::ostream &out, unsigned level,
 512                         std::string &textdomain, const std::string &key)
 513                         : out_(out), level_(level), textdomain_(textdomain), key_(key)
 514                 {}
 515 
 516                 // Generic visitor just streams "key=value".
 517                 template <typename T> void operator()(T const & v) const
 518                 { indent(); out_ << key_ << '=' << v << '\n'; }
 519 
 520                 // Specialized visitors for things that go in quotes:
 521                 void operator()(boost::blank const &) const
 522                 { /* treat blank values as nonexistent which fits better than treating them as empty strings.*/ }
 523                 void operator()(const std::string& s) const
 524                 { indent(); out_ << key_ << '=' << '"' << escaped_string(s) << '"' << '\n'; }
 525                 void operator()(t_string const &s) const;
 526 
 527         private:
 528                 void indent() const
 529                 { for ( unsigned i = 0; i < level_; ++i )  out_ << '\t'; }
 530         };

 532         /**
 533          * Writes all the parts of a translatable string.
 534          * @note If the first part is translatable and in the wrong textdomain,
 535          *       the textdomain change has to happen before the attribute name.
 536          *       That is the reason for not outputting the key beforehand and
 537          *       letting this function do it.
 538          */
 539         void write_key_val_visitor::operator()(t_string const &value) const
 540         {
 541                 bool first = true;
 542 
 543                 for (t_string::walker w(value); !w.eos(); w.next())
 544                 {
 545                         if (!first)
 546                                 out_ << " +\n";
 547 
 548                         if (w.translatable() && w.textdomain() != textdomain_) {
 549                                 textdomain_ = w.textdomain();
 550                                 out_ << "#textdomain " << textdomain_ << '\n';
 551                         }
 552 
 553                         indent();
 554 
 555                         if (first)
 556                                 out_ << key_ << '=';
 557                         else
 558                                 out_ << '\t';
 559 
 560                         if (w.translatable())
 561                                 out_ << '_';
 562 
 563                         out_ << '"' << escaped_string(w.begin(), w.end()) << '"';
 564                         first = false;
 565                 }
 566                 out_ << '\n';
 567         }

 587 static void write_internal(config const &cfg, std::ostream &out, std::string& textdomain, size_t tab = 0)
 588 {
 589         if (tab > max_recursion_levels)
 590                 throw config::error("Too many recursion levels in config write");
 591 
 592         for (const config::attribute &i : cfg.attribute_range()) {
 593                 if (!config::valid_id(i.first)) {
 594                         ERR_CF << "Config contains invalid attribute name '" << i.first << "', skipping...\n";
 595                         continue;
 596                 }
 597                 write_key_val(out, i.first, i.second, tab, textdomain);
 598         }
 599 
 600         for (const config::any_child &item : cfg.all_children_range())
 601         {
 602                 if (!config::valid_id(item.key)) {
 603                         ERR_CF << "Config contains invalid tag name '" << item.key << "', skipping...\n";
 604                         continue;
 605                 }
 606                 write_open_child(out, item.key, tab);
 607                 write_internal(item.cfg, out, textdomain, tab + 1);
 608                 write_close_child(out, item.key, tab);
 609         }
 610 }
 611 
 612 static void write_internal(configr_of const &cfg, std::ostream &out, std::string& textdomain, size_t tab = 0)
 613 {
 614         if (tab > max_recursion_levels)
 615                 throw config::error("Too many recursion levels in config write");
 616         if (cfg.data_) {
 617                 write_internal(*cfg.data_, out, textdomain, tab);
 618         }
 619 
 620         for (const auto &pair: cfg.subtags_)
 621         {
 622                 assert(pair.first && pair.second);
 623                 if (!config::valid_id(*pair.first)) {
 624                         ERR_CF << "Config contains invalid tag name '" << *pair.first << "', skipping...\n";
 625                         continue;
 626                 }
 627                 write_open_child(out, *pair.first, tab);
 628                 write_internal(*pair.second, out, textdomain, tab + 1);
 629                 write_close_child(out, *pair.first, tab);
 630         }
 631 }

 636         write_internal(cfg, out, textdomain, level);
