  34 mp_options_helper::mp_options_helper(window& window, ng::create_engine& create_engine)
  35         : create_engine_(create_engine)
  36         , options_tree_(find_widget<tree_view>(&window, "custom_options", false))
  37         , no_options_notice_(find_widget<styled_widget>(&window, "no_options_notice", false))

  40         , options_data_()

  44                         options_data_[c.cfg["id"]][saved_option["id"]] = saved_option["value"];
  45                 }
  46         }
  47 
  48         update_all_options();
  49 }

  62 void mp_options_helper::update_game_options()
  63 {
  64         const std::string type = create_engine_.current_level_type() == ng::level::TYPE::CAMPAIGN ? "campaign" : "multiplayer";
  65 
  66         // For game options, we check for both types and remove them. This is to prevent options from a game
  67         // of one type remaining visible when selecting a game of another type.
  68         int pos = remove_nodes_for_type("campaign");
  69             pos = remove_nodes_for_type("multiplayer");
  70 
  71         display_custom_options(type, pos, create_engine_.current_level().data());
  72 
  73         update_status_label();
  74 }

  87 void mp_options_helper::update_mod_options()
  88 {
  89         static const std::string type = "modification";
  90 
  91         int pos = remove_nodes_for_type(type);
  92 
  93         for(const auto& mod : create_engine_.active_mods_data()) {
  94                 display_custom_options(type, pos, *mod->cfg);
  95         }
  96 
  97         update_status_label();

 100 int mp_options_helper::remove_nodes_for_type(const std::string& type)
 101 {
 102         // Remove all visible options of the specified source type
 103         auto vo_iter = std::remove_if(visible_options_.begin(), visible_options_.end(), [&type](const option_source& source) {
 104                 return source.level_type == type;
 105         });
 106 
 107         visible_options_.erase(vo_iter, visible_options_.end());
 108 
 109         // Get the node data for this specific source type
 110         type_node_data* data;
 111 
 112         auto node_data_map_iter = node_data_map_.end();
 113         std::tie(node_data_map_iter, std::ignore) = node_data_map_.emplace(type, type_node_data());
 114 
 115         data = &node_data_map_iter->second;
 116 
 117         node_vector& type_node_vector = data->nodes;
 118 
 119         // The position to insert a new node of this type. If no nodes exist yet, the default value (-1) is
 120         // accepted by tree_view_node as meaning at-end.
 121         int& position = data->position;
 122 
 123         // Remove each node in reverse, so that in the end we have the position of the first node removed
 124         for(auto i = type_node_vector.rbegin(); i != type_node_vector.rend(); i++) {
 125                 position = options_tree_.remove_node(*i);
 126         }
 127 
 128         type_node_vector.clear();
 129 
 130         return position;
 131 }

 156 void mp_options_helper::reset_options_data(const option_source& source, bool& handled, bool& halt)
 157 {
 158         options_data_[source.id].clear();
 159 
 160         if(source.level_type == "campaign" || source.level_type == "multiplayer") {
 161                 update_game_options();
 162         } else if(source.level_type == "era") {
 163                 update_era_options();
 164         } else if(source.level_type == "modification") {
 165                 update_mod_options();
 166         }
 167 
 168         handled = true;
 169         halt = true;
 170 }

 172 template<typename T>
 173 std::pair<T*, config::attribute_value> mp_options_helper::add_node_and_get_widget(
 174                 tree_view_node& option_node, const std::string& id, data_map& data, const config& cfg)
 175 {
 176         tree_view_node& node = option_node.add_child(id + "_node", data);
 177 
 178         T* widget = dynamic_cast<T*>(node.find(id, true));
 179         VALIDATE(widget, missing_widget(id));
 180 
 181         const std::string widget_id = cfg["id"];
 182 
 183         auto& option_config = options_data_[visible_options_.back().id];
 184         if(!option_config.has_attribute(widget_id) || option_config[widget_id].empty()) {
 185                 option_config[widget_id] = cfg["default"];
 186         }
 187 
 188         widget->set_id(widget_id);
 189         widget->set_tooltip(cfg["description"]);
 190 
 191         return {widget, option_config[widget_id]};
 192 }

 194 void mp_options_helper::display_custom_options(const std::string& type, int node_position, const config& cfg)
 195 {
 196         // Needed since some compilers don't like passing just {}
 197         static const std::map<std::string, string_map> empty_map;
 198 
 199         // This ensures that any game, era, or mod with no options doesn't get an entry in the visible_options_
 200         // vector and prevents invalid options from different games, era, or mods being created when the options
 201         // config is created.
 202         if(!cfg.has_child("options")) {
 203                 return;
 204         }
 205 
 206         visible_options_.push_back({type, cfg["id"]});
 207 
 208         // Get the node vector for this specific source type
 209         node_vector& type_node_vector = node_data_map_[type].nodes;
 210 
 211         for(const auto& options : cfg.child_range("options")) {
 212                 std::map<std::string, string_map> data;
 213                 string_map item;
 214 
 215                 item["label"] = cfg["name"];
 216                 data.emplace("tree_view_node_label", item);
 217 
 218                 tree_view_node& option_node = options_tree_.add_node("option_node", data, node_position);
 219                 type_node_vector.push_back(&option_node);
 220 
 221                 for(const config::any_child& opt : options.all_children_range()) {
 222                         data.clear();
 223                         item.clear();
 224 
 225                         const config& option_cfg = opt.cfg;
 226 
 227                         const auto add_name = [&](const std::string& id) {
 228                                 item["label"] = option_cfg["name"];
 229                                 data.emplace(id, item);
 230                         };
 231 
 232                         config::attribute_value val;
 233 
 234                         if(opt.key == "checkbox") {
 235                                 add_name("option_checkbox");
 236 
 237                                 toggle_button* checkbox;
 238                                 std::tie(checkbox, val) = add_node_and_get_widget<toggle_button>(option_node, "option_checkbox", data, option_cfg);
 239 
 240                                 checkbox->set_value(val.to_bool());
 241                                 checkbox->set_callback_state_change(
 242                                         std::bind(&mp_options_helper::update_options_data_map<toggle_button>, this, checkbox, visible_options_.back()));
 243 
 244                         } else if(opt.key == "spacer") {
 245                                 option_node.add_child("options_spacer_node", empty_map);
 246 
 247                         } else if(opt.key == "choice" || opt.key == "combo") {
 248                                 if(opt.key == "combo") {
 249                                         lg::wml_error() << "[options][combo] is deprecated; use [choice] instead\n";
 250                                 }
 251 
 252                                 if(!option_cfg.has_child("item")) {
 253                                         continue;
 254                                 }
 255 
 256                                 add_name("menu_button_label");
 257 
 258                                 std::vector<config> combo_items;
 259                                 std::vector<std::string> combo_values;
 260 
 261                                 for(auto i : option_cfg.child_range("item")) {
 262                                         // Comboboxes expect this key to be 'label' not 'name'
 263                                         i["label"] = i["name"];
 264 
 265                                         combo_items.push_back(i);
 266                                         combo_values.push_back(i["value"]);
 267                                 }
 268 
 269                                 menu_button* menu;
 270                                 std::tie(menu, val) = add_node_and_get_widget<menu_button>(option_node, "option_menu_button", data, option_cfg);
 271 
 272                                 // Needs to be called before set_selected
 273                                 menu->set_values(combo_items);
 274 
 275                                 auto iter = std::find(combo_values.begin(), combo_values.end(), val.str());
 276 
 277                                 if(iter != combo_values.end()) {
 278                                         menu->set_selected(iter - combo_values.begin());
 279                                 }
 280 
 281                                 menu->connect_click_handler(
 282                                         std::bind(&mp_options_helper::update_options_data_map_menu_button, this, menu, visible_options_.back(), option_cfg));
 283 
 284                         } else if(opt.key == "slider") {
 285                                 add_name("slider_label");
 286 
 287                                 slider* slide;
 288                                 std::tie(slide, val) = add_node_and_get_widget<slider>(option_node, "option_slider", data, option_cfg);
 289 
 290                                 slide->set_maximum_value(option_cfg["max"].to_int());
 291                                 slide->set_minimum_value(option_cfg["min"].to_int());
 292                                 slide->set_step_size(option_cfg["step"].to_int(1));
 293                                 slide->set_value(val.to_int());
 294 
 295                                 connect_signal_notify_modified(*slide,
 296                                         std::bind(&mp_options_helper::update_options_data_map<slider>, this, slide, visible_options_.back()));
 297 
 298                         } else if(opt.key == "entry") {
 299                                 add_name("text_entry_label");
 300 
 301                                 text_box* textbox;
 302                                 std::tie(textbox, val) = add_node_and_get_widget<text_box>(option_node, "option_text_entry", data, option_cfg);
 303 
 304                                 textbox->set_value(val.str());
 305                                 textbox->set_text_changed_callback(
 306                                         std::bind(&mp_options_helper::update_options_data_map<text_box>, this, textbox, visible_options_.back()));
 307                         }
 308                 }
 309 
 310                 // Add the Defaults button at the end
 311                 tree_view_node& node = option_node.add_child("options_default_button", empty_map);
 312 
 313                 connect_signal_mouse_left_click(find_widget<button>(&node, "reset_option_values", false),
 314                         std::bind(&mp_options_helper::reset_options_data, this, visible_options_.back(),
 315                                 std::placeholders::_3, std::placeholders::_4));
 316         }
 317 }

 319 config mp_options_helper::get_options_config()
 320 {
 321         config options;
 322         for(const auto& source : visible_options_) {
 323                 config& mod = options.add_child(source.level_type);
 324                 mod["id"] = source.id;
 325 #if 0
 326                 // TODO: enable this as soon as we drop the old mp configure screen.
 327                 mod.add_child("options", options_data_[source.id]);
 328 #else
 329                 for(const auto& option : options_data_[source.id].attribute_range()) {
 330                         mod.add_child("option", config {"id", option.first, "value", option.second});
 331                 }
 332 #endif
 333         }
 334 
 335         return options;
 336 }
