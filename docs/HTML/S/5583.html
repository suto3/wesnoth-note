   1 # Test that clearing multiple variables works, also that
   2 # the WML `_G` analogue works.
   3 
   4 {GENERIC_UNIT_TEST "test_clear_one" (
   5     [event]
   6         name = start
   7 
   8         {VARIABLE A 5}
   9         {VARIABLE B 10}
  10 
  11         {CLEAR_VARIABLE A}
  12 
  13         [lua]
  14             code = << local a,b,c = false,false,false
  15                           for k,v in pairs(wesnoth.get_all_vars()) do
  16                             if k == "A" then
  17                               a = true
  18                             elseif (k == "B") and (v == 10) then
  19                               b = true
  20                             elseif (k == "C") then
  21                               c = true
  22                             end
  23                           end
  24                           wesnoth.set_variable("result", (not a) and b and (not c)) >>
  25         [/lua]
  26 
  27         {RETURN {VARIABLE_CONDITIONAL result boolean_equals true}}
  28     [/event]
  29 )}
