 102             @Override

 104                 throws InvocationTargetException
 105             {
 106                 try {
 107                     doFinish( containerName, fileName, monitor );
 108                 } catch( Exception e ) {
 109                     throw new InvocationTargetException( e );
 110                 } finally {
 111                     monitor.done( );
 112                 }
 113             }
 114         };
 115         try {
 116             getContainer( ).run( false, false, op );
 117         } catch( InterruptedException e ) {
 118             return false;
 119         } catch( InvocationTargetException e ) {
 120             Logger.getInstance( ).logException( e );
 121             return false;
 122         }
 123         return true;

 126     /**
 127      * The worker method. It will find the container, create the file if missing
 128      * or just replace its contents, and open the editor on the newly created
 129      * file.
 130      */
 131     private void doFinish( String containerName, String fileName,
 132         IProgressMonitor monitor ) throws Exception
 133     {
 134         monitor.beginTask( Messages.ScenarioNewWizard_2 + fileName, 2 );
 135         IWorkspaceRoot root = ResourcesPlugin.getWorkspace( ).getRoot( );
 136         IResource resource = root.findMember( new Path( containerName ) );
 137 
 138         if( ! resource.exists( ) || ! ( resource instanceof IContainer ) ) {
 139             throw new Exception(
 140                 "Container " + containerName + " does not exist." ); //$NON-NLS-1$ //$NON-NLS-2$
 141         }
 142 
 143         final IContainer container = ( IContainer ) resource;
 144         final IFile file = container.getFile( new Path( fileName ) );
 145 
 146         InputStream stream = getScenarioStream( container );
 147 
 148         try {
 149             if( stream == null ) {
 150                 return;
 151             }
 152 
 153             if( file.exists( ) ) {
 154                 file.setContents( stream, true, true, monitor );
 155             }
 156             else {
 157                 file.create( stream, true, monitor );
 158             }
 159 
 160             stream.close( );
 161             container.refreshLocal( IResource.DEPTH_INFINITE,
 162                 new NullProgressMonitor( ) );
 163         } catch( IOException e ) {
 164             Logger.getInstance( ).logException( e );
 165         }
 166 
 167         monitor.worked( 1 );
 168         monitor.setTaskName( Messages.ScenarioNewWizard_5 );
 169         getShell( ).getDisplay( ).asyncExec( new Runnable( ) {
 170             @Override
 171             public void run( )
 172             {
 173                 IWorkbenchPage page = PlatformUI.getWorkbench( )
 174                     .getActiveWorkbenchWindow( ).getActivePage( );
 175                 try {
 176                     IDE.openEditor( page, file, true );
 177                 } catch( PartInitException e ) {
 178                 }
 179             }
 180         } );
 181         monitor.worked( 1 );
 182         monitor.done( );
 183     }

 185     /**
 186      * Returns the scenario file contents as an InputStream
 187      * 
 188      * @throws Exception
 189      */
 190     private InputStream getScenarioStream( IContainer container )
 191         throws Exception
 192     {
 193         ArrayList< ReplaceableParameter > params = new ArrayList< ReplaceableParameter >( );
 194 
 195         // common variables (sp + mp)
 196         params.add( new ReplaceableParameter( "$$scenario_id", page0_.getScenarioId( ) ) ); //$NON-NLS-1$
 197         params.add( new ReplaceableParameter( "$$next_scenario_id", page0_.getNextScenarioId( ) ) ); //$NON-NLS-1$
 198         params.add( new ReplaceableParameter( "$$scenario_name", page0_.getScenarioName( ) ) ); //$NON-NLS-1$
 199 
 200         String mapData = ""; //$NON-NLS-1$
 201         if( ! page0_.getMapData( ).isEmpty( ) ) {
 202             String userMapPath = page0_.getMapData( ).replace( "~add-ons", //$NON-NLS-1$
 203                 container.getParent( ).getLocation( ).toOSString( ) + "/" ); //$NON-NLS-1$
 204             // trim the '{' and '}'
 205             userMapPath = userMapPath.substring( 1, userMapPath.length( ) - 1 );
 206 
 207             if( ! page0_.getIsMapEmbedded( ) ) {
 208                 mapData = page0_.getMapData( );
 209                 ResourceUtils.copyTo( new File( page0_.getRawMapPath( ) ),
 210                     new File( userMapPath ) );
 211             }
 212             else {
 213                 mapData = ResourceUtils
 214                     .getFileContents( new File( userMapPath ) );
 215             }
 216         }
 217         params.add( new ReplaceableParameter( "$$map_data", mapData ) ); //$NON-NLS-1$
 218         params.add( new ReplaceableParameter( "$$turns_number", String.valueOf( page0_.getTurnsNumber( ) ) ) ); //$NON-NLS-1$
 219 
 220         String startingGold = ""; //$NON-NLS-1$
 221         if( page1_ != null ) {
 222             startingGold = page1_.getStartingGoldByDifficulties( );
 223             if( startingGold == null ) {
 224                 throw new Exception( "incorrect arguments" ); //$NON-NLS-1$
 225             }
 226         }
 227         params
 228             .add( new ReplaceableParameter( "$$starting_gold", startingGold ) ); //$NON-NLS-1$
 229 
 230         // multiplayer only variables
 231         params.add( new ReplaceableParameter( "$$allow_new_game", page2_.getAllowNewGame( ) ) ); //$NON-NLS-1$
 232 
 233         String template = TemplateProvider
 234             .getInstance( )
 235             .getProcessedTemplate(
 236                 page2_.isMultiplayerScenario( ) ? "multiplayer": "scenario", params ); //$NON-NLS-1$ //$NON-NLS-2$
 237 
 238         if( template == null ) {
 239             GUIUtils.showMessageBox( WorkspaceUtils.getWorkbenchWindow( ),
 240                 Messages.ScenarioNewWizard_20 );
 241             return null;
 242         }
 243 
 244         return new ByteArrayInputStream( template.getBytes( ) );
 245     }
