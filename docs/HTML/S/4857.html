 117     # This long event creates a chocobone when a unit moves to where bird bones are
 118     # buried in the snow. The event is removed if chocobones were discovered on the
 119     # beach with the Merfolk.
 120     # There is a short conversation, but there are many possibilities for which units
 121     # are involved, so there are a lot of ifs. There is also a lot of animation around
 122     # the appearance of the chocobone involving a skeleton and a flightless-bird unit.
 123     [event]
 124         name=moveto
 125         id=bird_bones
 126 
 127         [filter]
 128             x,y=16,21
 129             side=1
 130         [/filter]
 131         [if]
 132             [variable]
 133                 name=unit.race
 134                 equals=bats
 135             [/variable]
 136             [or]
 137                 [variable]
 138                     name=unit.variation
 139                     equals=bat
 140                 [/variable]
 141             [/or]
 142             [then]
 143                 # Bones examined by a bat or undead bat
 144                 [message]
 145                     speaker=Ras-Tabahn
 146                     message= _ "There are some... large bones there. Claws and wings... I think."
 147                 [/message]
 148             [/then]
 149             [else]
 150                 [if]
 151                     [variable]
 152                         name=unit.race
 153                         equals=undead
 154                     [/variable]
 155                     [then]
 156                         # Bones examined by a ghost, skeleton, or ghoul.
 157                         [message]
 158                             speaker=$unit.id
 159                             message= _ "Bones. Claws and wings."
 160                         [/message]
 161                     [/then]
 162                     [else]
 163                         # Bones examined by a human or Vendraxis.
 164                         [message]
 165                             speaker=$unit.id
 166                             message= _ "There are some large bones here. I can see claws and wings."
 167                         [/message]
 168                     [/else]
 169                 [/if]
 170             [/else]
 171         [/if]
