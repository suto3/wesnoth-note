   3 local function index(x, y)
   4         -- the 2000 bias ensure that the correct x is recovered for negative y
   5         return x * 16384 + y + 2000
   6 end

   8 local invscale = 1 / 16384
   9 local function revindex(p)
  10         local x = math.floor(p * invscale)
  11         return x, p - x * 16384 - 2000
  12 end

 112         table.sort(indices)
 113         for i,p in ipairs(indices) do
 114                 local x, y = revindex(p)

 119 function methods:of_pairs(t)
 120         local values = self.values
 121 
 122         for i,v in ipairs(t) do
 123                 local value_table = {}
 124                 local x_index
 125                 local y_index
 126                 if v.x and v.y then
 127                         x_index = "x"
 128                         y_index = "y"
 129                 else
 130                         x_index = 1
 131                         y_index = 2
 132                 end
 133                 for k,val in pairs(v) do
 134                         if k ~= x_index and k ~= y_index then
 135                                 value_table[k] = val
 136                         end
 137                 end
 138                 if next(value_table) then
 139                         values[index(v[x_index], v[y_index])] = value_table
 140                 else
 141                         values[index(v[x_index], v[y_index])] = true
 142                 end
 143         end
 144 end

 146 function methods:of_wml_var(name)
 147         local values = self.values
 148         for i = 0, wesnoth.get_variable(name .. ".length") - 1 do
 149                 local t = wesnoth.get_variable(string.format("%s[%d]", name, i))
 150                 local x, y = t.x, t.y
 151                 t.x, t.y = nil, nil
 152                 values[index(x, y)] = next(t) and t or true
 153         end

 156 function methods:of_triples(t)

 176 function methods:to_wml_var(name)
 177         local i = 0
 178         wesnoth.set_variable(name)
 179         self:stable_iter(function(x, y, v)

 181                         wesnoth.set_variable(string.format("%s[%d]", name, i), v)
 182                 end
 183                 wesnoth.set_variable(string.format("%s[%d].x", name, i), x)
 184                 wesnoth.set_variable(string.format("%s[%d].y", name, i), y)
 185                 i = i + 1

 195 function methods:random()
 196         -- Select a random hex from the hexes in the location set
 197         -- This seems "inelegant", but I can't come up with another way without creating an extra array
 198         -- Return -1, -1 if empty
 199         local r = wesnoth.random(self:size())
 200         local i, xr, yr = 1, -1, -1
 201         self:iter( function(x, y, v)
 202                 if (i == r) then xr, yr = x, y end
 203                 i = i + 1
 204         end)
 205         return xr, yr
 206 end
