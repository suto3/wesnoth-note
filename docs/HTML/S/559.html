  49         template <typename... T>
  50         auto zip(const T&... containers) -> boost::iterator_range<boost::zip_iterator<decltype(boost::make_tuple(std::begin(containers)...))>>
  51         {
  52                 auto zip_begin = boost::make_zip_iterator(boost::make_tuple(std::begin(containers)...));
  53                 auto zip_end = boost::make_zip_iterator(boost::make_tuple(std::end(containers)...));
  54                 return boost::make_iterator_range(zip_begin, zip_end);
  55         }

 105 static inline tree_view_node& add_name_tree_node(tree_view_node& header_node, const std::string& type, const t_string& label, const t_string& tooltip = "")
 106 {
 107         /* Note: We have to pass data instead of just doing 'child_label.set_label(label)' below
 108          * because the tree_view_node::add_child needs to have the correct size of the
 109          * node child widgets for its internal size calculations.
 110          * Same is true for 'use_markup'
 111          */
 112         auto& child_node = header_node.add_child(type, { { "name",{ { "label", label },{ "use_markup", "true" } } } });
 113         auto& child_label = find_widget<styled_widget>(&child_node, "name", true);
 114 
 115         child_label.set_tooltip(tooltip);
 116         return child_node;
 117 }

 119 static inline std::string get_hp_tooltip(const utils::string_map& res, const std::function<int (const std::string&, bool)>& get)
 120 {
 121         std::ostringstream tooltip;
 122 
 123         std::set<std::string> resistances_table;
 124 
 125         bool att_def_diff = false;
 126         for(const utils::string_map::value_type &resist : res) {
 127                 std::ostringstream line;
 128                 line << translation::gettext(resist.first.c_str()) << ": ";
 129 
 130                 // Some units have different resistances when attacking or defending.
 131                 const int res_att = 100 - get(resist.first, true);
 132                 const int res_def = 100 - get(resist.first, false);
 133 
 134                 if(res_att == res_def) {
 135                         line << "<span color='" << unit_helper::resistance_color(res_def) << "'>" << utils::signed_percent(res_def) << "</span>";
 136                 } else {
 137                         line << "<span color='" << unit_helper::resistance_color(res_att) << "'>" << utils::signed_percent(res_att) << "</span>" << "/"
 138                              << "<span color='" << unit_helper::resistance_color(res_def) << "'>" << utils::signed_percent(res_def) << "</span>";
 139                         att_def_diff = true;
 140                 }
 141 
 142                 resistances_table.insert(line.str());
 143         }
 144 
 145         tooltip << "<big>" << _("Resistances: ") << "</big>";
 146         if(att_def_diff) {
 147                 tooltip << _("(Att / Def)");
 148         }
 149 
 150         for(const std::string &line : resistances_table) {
 151                 tooltip << '\n' << font::unicode_bullet << " " << line;
 152         }
 153 
 154         return tooltip.str();
 155 }

 157 static inline std::string get_mp_tooltip(int total_movement, std::function<int (t_translation::terrain_code)> get)
 158 {
 159         std::ostringstream tooltip;
 160         tooltip << "<big>" << _("Movement Costs:") << "</big>";
 161 
 162         ter_data_cache tdata = help::load_terrain_types_data();
 163 
 164         if(!tdata) {
 165                 return "";
 166         }
 167 
 168         for(t_translation::terrain_code terrain : preferences::encountered_terrains()) {
 169                 if(terrain == t_translation::FOGGED || terrain == t_translation::VOID_TERRAIN || t_translation::terrain_matches(terrain, t_translation::ALL_OFF_MAP)) {
 170                         continue;
 171                 }
 172 
 173                 const terrain_type& info = tdata->get_terrain_info(terrain);
 174                 if(info.union_type().size() == 1 && info.union_type()[0] == info.number() && info.is_nonnull()) {
 175                         const std::string& name = info.name();
 176                         const int moves = get(terrain);
 177 
 178                         tooltip << '\n' << font::unicode_bullet << " " << name << ": ";
 179 
 180                         // movement  -  range: 1 .. 5, movetype::UNREACHABLE=impassable
 181                         const bool cannot_move = moves > total_movement;
 182 
 183                         std::string color;
 184                         if(cannot_move) {
 185                                 // cannot move in this terrain
 186                                 color = "red";
 187                         } else if(moves > 1) {
 188                                 color = "yellow";
 189                         } else {
 190                                 color = "white";
 191                         }
 192 
 193                         tooltip << "<span color='" << color << "'>";
 194 
 195                         // A 5 MP margin; if the movement costs go above the unit's max moves + 5, we replace it with dashes.
 196                         if(cannot_move && (moves > total_movement + 5)) {
 197                                 tooltip << font::unicode_figure_dash;
 198                         } else {
 199                                 tooltip << moves;
 200                         }
 201 
 202                         tooltip << "</span>";
 203                 }
 204         }
 205 
 206         return tooltip.str();
 207 }

 213 template<typename T>
 214 void unit_preview_pane::print_attack_details(T attacks, tree_view_node& parent_node)
 215 {
 216         if(attacks.empty()) {
 217                 return;
 218         }
 219 
 220         auto& header_node = add_name_tree_node(
 221                 parent_node,
 222                 "header",
 223                 "<b>" + _("Attacks") + "</b>"
 224         );
 225 
 226         for(const auto& a : attacks) {
 227 
 228                 auto& subsection = add_name_tree_node(
 229                         header_node,
 230                         "item",
 231                         (formatter() << font::span_color(font::unit_type_color) << a.damage() << font::weapon_numbers_sep << a.num_attacks() << " " << a.name() << "</span>").str()
 232                 );
 233 
 234                 add_name_tree_node(
 235                         subsection,
 236                         "item",
 237                         (formatter() << font::span_color(font::weapon_details_color) << a.range() << font::weapon_details_sep << a.type() << "</span>").str()
 238                 );
 239 
 240                 for(const auto& pair : a.special_tooltips()) {
 241                         add_name_tree_node(
 242                                 subsection,
 243                                 "item",
 244                                 (formatter() << font::span_color(font::weapon_details_color) << pair.first << "</span>").str(),
 245                                 (formatter() << "<span size='x-large'>" << pair.first << "</span>" << "\n" << pair.second).str()
 246                         );
 247                 }
 248         }
 249 }

 251 void unit_preview_pane::set_displayed_type(const unit_type& type)
 252 {
 253         // Sets the current type id for the profile button callback to use
 254         current_type_ = type.id();
 255 
 256         if(icon_type_) {
 257                 std::string mods;
 258 
 259                 if(resources::controller) {
 260                         mods = "~RC(" + type.flag_rgb() + ">" +
 261                                  team::get_side_color_index(resources::controller->current_side())
 262                                  + ")";
 263                 }
 264 
 265                 mods += "~SCALE_INTO_SHARP(144,144)" + image_mods_;
 266 
 267                 icon_type_->set_label((type.icon().empty() ? type.image() : type.icon()) + mods);
 268         }
 269 
 270         if(label_name_) {
 271                 label_name_->set_label("<big>" + type.type_name() + "</big>");
 272                 label_name_->set_use_markup(true);
 273         }
 274 
 275         if(label_level_) {
 276                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(type.level())}});
 277 
 278                 label_level_->set_label("<b>" + l_str + "</b>");
 279                 label_level_->set_use_markup(true);
 280         }
 281 
 282         if(label_race_) {
 283                 label_race_ ->set_label(type.race()->name(type.genders().front()));
 284         }
 285 
 286         if(icon_race_) {
 287                 icon_race_->set_label("icons/unit-groups/race_" + type.race_id() + "_30.png");
 288         }
 289 
 290         if(icon_alignment_) {
 291                 const std::string& alignment_name = type.alignment().to_string();
 292 
 293                 icon_alignment_->set_label("icons/alignments/alignment_" + alignment_name + "_30.png");
 294                 icon_alignment_->set_tooltip(unit_type::alignment_description(
 295                         type.alignment(),
 296                         type.genders().front()));
 297         }
 298 
 299         if(label_details_) {
 300                 std::stringstream str;
 301 
 302                 str << "<span size='large'> </span>" << "\n";
 303 
 304                 str << font::span_color(font::unit_type_color) << type.type_name() << "</span>" << "\n";
 305 
 306                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(type.level())}});
 307                 str << l_str << "\n";
 308 
 309                 str << type.alignment() << "\n";
 310 
 311                 str << "\n"; // Leave a blank line where traits would be
 312 
 313                 str <<  _("HP: ") << type.hitpoints() << "\n";
 314 
 315                 str << _("XP: ") << type.experience_needed(true);
 316 
 317                 label_details_->set_label(str.str());
 318                 label_details_->set_use_markup(true);
 319         }
 320 
 321         if(tree_details_) {
 322 
 323                 tree_details_->clear();
 324                 tree_details_->add_node("hp_xp_mp", {
 325                         { "hp",{
 326                                 { "label", (formatter() << "<small>" << "<span color='#21e100'>" << "<b>" << _("HP: ") << "</b>" << type.hitpoints() << "</span>" << " | </small>").str() },
 327                                 { "use_markup", "true" },
 328                                 { "tooltip", get_hp_tooltip(type.movement_type().get_resistances().damage_table(), [&type](const std::string& dt, bool is_attacker) { return type.resistance_against(dt, is_attacker); }) }
 329                         } },
 330                         { "xp",{
 331                                 { "label", (formatter() << "<small>" << "<span color='#00a0e1'>" << "<b>" << _("XP: ") << "</b>" << type.experience_needed() << "</span>" << " | </small>").str() },
 332                                 { "use_markup", "true" },
 333                                 { "tooltip", (formatter() << _("Experience Modifier: ") << unit_experience_accelerator::get_acceleration() << '%').str() }
 334                         } },
 335                         { "mp",{
 336                                 { "label", (formatter() << "<small>" << "<b>" << _("MP: ") << "</b>" << type.movement() << "</small>").str() },
 337                                 { "use_markup", "true" },
 338                                 { "tooltip", get_mp_tooltip(type.movement(), [&type](t_translation::terrain_code terrain) { return type.movement_type().movement_cost(terrain); }) }
 339                         } },
 340                 });
 341 
 342                 // Print trait details
 343                 {
 344                         tree_view_node* header_node = nullptr;
 345 
 346                         for(const auto& tr : type.possible_traits()) {
 347                                 t_string name = tr[type.genders().front() == unit_race::FEMALE ? "female_name" : "male_name"];
 348                                 if(tr["availability"] != "musthave" || name.empty()) {
 349                                         continue;
 350                                 }
 351 
 352                                 if(header_node == nullptr) {
 353                                         header_node = &add_name_tree_node(
 354                                                 tree_details_->get_root_node(),
 355                                                 "header",
 356                                                 "<b>" + _("Traits") + "</b>"
 357                                         );
 358                                 }
 359 
 360                                 add_name_tree_node(
 361                                         *header_node,
 362                                         "item",
 363                                         name
 364                                 );
 365                         }
 366                 }
 367 
 368                 // Print ability details
 369                 if(!type.abilities().empty()) {
 370 
 371                         auto& header_node = add_name_tree_node(
 372                                 tree_details_->get_root_node(),
 373                                 "header",
 374                                 "<b>" + _("Abilities") + "</b>"
 375                         );
 376 
 377                         for(const auto& ab : zip(type.abilities() , type.ability_tooltips())) {
 378                                 add_name_tree_node(
 379                                         header_node,
 380                                         "item",
 381                                         boost::get<0>(ab),
 382                                         (formatter() << "<span size='x-large'>" << boost::get<0>(ab) << "</span>\n" << boost::get<1>(ab)).str()
 383                                 );
 384                         }
 385                 }
 386 
 387                 print_attack_details(type.attacks(), tree_details_->get_root_node());
 388         }

 391 void unit_preview_pane::set_displayed_unit(const unit& u)
 392 {
 393         // Sets the current type id for the profile button callback to use
 394         current_type_ = u.type_id();
 395 
 396         if(icon_type_) {
 397                 std::string mods = u.image_mods();
 398 
 399                 if(u.can_recruit()) {
 400                         mods += "~BLIT(" + unit::leader_crown() + ")";
 401                 }
 402 
 403                 for(const std::string& overlay : u.overlays()) {
 404                         mods += "~BLIT(" + overlay + ")";
 405                 }
 406 
 407                 mods += "~SCALE_INTO_SHARP(144,144)" + image_mods_;
 408 
 409                 icon_type_->set_label(u.absolute_image() + mods);
 410         }
 411 
 412         if(label_name_) {
 413                 std::string name;
 414                 if(!u.name().empty()) {
 415                         name = "<span size='large'>" + u.name() + "</span>" + "\n" + "<small>" + font::span_color(font::unit_type_color) + u.type_name() + "</span></small>";
 416                 } else {
 417                         name = "<span size='large'>" + u.type_name() + "</span>\n";
 418                 }
 419 
 420                 label_name_->set_label(name);
 421                 label_name_->set_use_markup(true);
 422         }
 423 
 424         if(label_level_) {
 425                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(u.level())}});
 426 
 427                 label_level_->set_label("<b>" + l_str + "</b>");
 428                 label_level_->set_use_markup(true);
 429         }
 430 
 431         if(label_race_) {
 432                 label_race_->set_label(u.race()->name(u.gender()));
 433         }
 434 
 435         if(icon_race_) {
 436                 icon_race_->set_label("icons/unit-groups/race_" + u.race()->id() + "_30.png");
 437         }
 438 
 439         if(icon_alignment_) {
 440                 const std::string& alignment_name = u.alignment().to_string();
 441 
 442                 icon_alignment_->set_label("icons/alignments/alignment_" + alignment_name + "_30.png");
 443                 icon_alignment_->set_tooltip(unit_type::alignment_description(
 444                         u.alignment(),
 445                         u.gender()));
 446         }
 447 
 448         if(label_details_) {
 449                 std::stringstream str;
 450 
 451                 const std::string name = "<span size='large'>" + (!u.name().empty() ? u.name() : " ") + "</span>";
 452                 str << name << "\n";
 453 
 454                 str << font::span_color(font::unit_type_color) << u.type_name() << "</span>" << "\n";
 455 
 456                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(u.level())}});
 457                 str << l_str << "\n";
 458 
 459                 str << u.alignment() << "\n";
 460 
 461                 str << utils::join(u.trait_names(), ", ") << "\n";
 462 
 463                 str << font::span_color(u.hp_color())
 464                         << _("HP: ") << u.hitpoints() << "/" << u.max_hitpoints() << "</span>" << "\n";
 465 
 466                 str << font::span_color(u.xp_color())
 467                         << _("XP: ") << u.experience() << "/" << u.max_experience() << "</span>";
 468 
 469                 label_details_->set_label(str.str());
 470                 label_details_->set_use_markup(true);
 471         }
 472 
 473         if(tree_details_) {
 474                 tree_details_->clear();
 475                 tree_details_->add_node("hp_xp_mp", {
 476                         { "hp",{
 477                                 { "label", (formatter() << "<small>" << font::span_color(u.hp_color()) << "<b>" << _("HP: ") << "</b>" << u.hitpoints() << "/" << u.max_hitpoints() << "</span>" << " | </small>").str() },
 478                                 { "use_markup", "true" },
 479                                 { "tooltip", get_hp_tooltip(u.get_base_resistances(), [&u](const std::string& dt, bool is_attacker) { return u.resistance_against(dt, is_attacker, u.get_location()); }) }
 480                         } },
 481                         { "xp",{
 482                                 { "label", (formatter() << "<small>" << font::span_color(u.xp_color()) << "<b>" << _("XP: ") << "</b>" << u.experience() << "/" << u.max_experience() << "</span>" << " | </small>").str() },
 483                                 { "use_markup", "true" },
 484                                 { "tooltip", (formatter() << _("Experience Modifier: ") << unit_experience_accelerator::get_acceleration() << '%').str() }
 485                         } },
 486                         { "mp",{
 487                                 { "label", (formatter() << "<small>" << "<b>" << _("MP: ") << "</b>" << u.movement_left() << "/" << u.total_movement() << "</small>").str() },
 488                                 { "use_markup", "true" },
 489                                 { "tooltip", get_mp_tooltip(u.total_movement(), [&u](t_translation::terrain_code terrain) { return u.movement_cost(terrain); }) }
 490                         } },
 491                 });
 492 
 493                 if(!u.trait_names().empty()) {
 494                         auto& header_node = add_name_tree_node(
 495                                 tree_details_->get_root_node(),
 496                                 "header",
 497                                 "<b>" + _("Traits") + "</b>"
 498                         );
 499 
 500                         assert(u.trait_names().size() == u.trait_descriptions().size());
 501                         for (size_t i = 0; i < u.trait_names().size(); ++i) {
 502                                 add_name_tree_node(
 503                                         header_node,
 504                                         "item",
 505                                         u.trait_names()[i],
 506                                         u.trait_descriptions()[i]
 507                                 );
 508                         }
 509                 }
 510 
 511                 if(!u.get_ability_list().empty()) {
 512                         auto& header_node = add_name_tree_node(
 513                                 tree_details_->get_root_node(),
 514                                 "header",
 515                                 "<b>" + _("Abilities") + "</b>"
 516                         );
 517 
 518                         for(const auto& ab : u.ability_tooltips()) {
 519                                 add_name_tree_node(
 520                                         header_node,
 521                                         "item",
 522                                         std::get<1>(ab),
 523                                         std::get<2>(ab)
 524                                 );
 525                         }
 526                 }
 527                 print_attack_details(u.attacks(), tree_details_->get_root_node());
 528         }
 529 }
