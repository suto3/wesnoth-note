 236 static path get_dir(const path &dirpath)
 237 {
 238         bool is_dir = is_directory_internal(dirpath);
 239         if (!is_dir) {
 240                 error_code ec;
 241                 bfs::create_directory(dirpath, ec);
 242                 if (ec) {
 243                         ERR_FS << "Failed to create directory " << dirpath.string() << ": " << ec.message() << '\n';
 244                 }
 245                 // This is probably redundant
 246                 is_dir = is_directory_internal(dirpath);
 247         }
 248         if (!is_dir) {
 249                 ERR_FS << "Could not open or create directory " << dirpath.string() << '\n';
 250                 return std::string();
 251         }
 252 
 253         return dirpath;

 255 static bool create_directory_if_missing(const path &dirpath)
 256 {
 257         error_code ec;
 258         bfs::file_status fs = bfs::status(dirpath, ec);
 259         if (error_except_not_found(ec)) {
 260                 ERR_FS << "Failed to retrieve file status for " << dirpath.string() << ": " << ec.message() << '\n';
 261                 return false;
 262         } else if (bfs::is_directory(fs)) {
 263                 DBG_FS << "directory " << dirpath.string() << " exists, not creating\n";
 264                 return true;
 265         } else if (bfs::exists(fs)) {
 266                 ERR_FS << "cannot create directory " << dirpath.string() << "; file exists\n";
 267                 return false;
 268         }
 269 
 270         bool created = bfs::create_directory(dirpath, ec);
 271         if (ec) {
 272                 ERR_FS << "Failed to create directory " << dirpath.string() << ": " << ec.message() << '\n';
 273         }
 274         return created;
 275 }
 276 static bool create_directory_if_missing_recursive(const path& dirpath)
 277 {
 278         DBG_FS << "creating recursive directory: " << dirpath.string() << '\n';
 279 
 280         if (dirpath.empty())
 281                 return false;
 282         error_code ec;
 283         bfs::file_status fs = bfs::status(dirpath);
 284         if (error_except_not_found(ec)) {
 285                 ERR_FS << "Failed to retrieve file status for " << dirpath.string() << ": " << ec.message() << '\n';
 286                 return false;
 287         } else if (bfs::is_directory(fs)) {
 288                 return true;
 289         } else if (bfs::exists(fs)) {
 290                 return false;
 291         }
 292 
 293         if (!dirpath.has_parent_path() || create_directory_if_missing_recursive(dirpath.parent_path())) {
 294                 return create_directory_if_missing(dirpath);
 295         } else {
 296                 ERR_FS << "Could not create parents to " << dirpath.string() << '\n';
 297                 return false;
 298         }
 299 }

 301 void get_files_in_dir(const std::string &dir,
 302                       std::vector<std::string>* files,
 303                       std::vector<std::string>* dirs,
 304                       file_name_option mode,
 305                       file_filter_option filter,
 306                       file_reorder_option reorder,
 307                       file_tree_checksum* checksum) {
 308         if(path(dir).is_relative() && !game_config::path.empty()) {
 309                 path absolute_dir(game_config::path);
 310                 absolute_dir /= dir;
 311                 if(is_directory_internal(absolute_dir)) {
 312                         get_files_in_dir(absolute_dir.string(), files, dirs, mode, filter, reorder, checksum);
 313                         return;
 314                 }
 315         }
 316 
 317         const path dirpath(dir);
 318 
 319         if (reorder == DO_REORDER) {
 320                 LOG_FS << "searching for _main.cfg in directory " << dir << '\n';
 321                 const path maincfg = dirpath / maincfg_filename;
 322 
 323                 if (file_exists(maincfg)) {
 324                         LOG_FS << "_main.cfg found : " << maincfg << '\n';
 325                         push_if_exists(files, maincfg, mode == ENTIRE_FILE_PATH);
 326                         return;
 327                 }
 328         }
 329 
 330         error_code ec;
 331         bfs::directory_iterator di(dirpath, ec);
 332         bfs::directory_iterator end;
 333         if (ec) {
 334                 // Probably not a directory, let the caller deal with it.
 335                 return;
 336         }
 337         for(; di != end; ++di) {
 338                 bfs::file_status st = di->status(ec);
 339                 if (ec) {
 340                         LOG_FS << "Failed to get file status of " << di->path().string() << ": " << ec.message() << '\n';
 341                         continue;
 342                 }
 343                 if (st.type() == bfs::regular_file) {
 344                         {
 345                                 std::string basename = di->path().filename().string();
 346                                 if (filter == SKIP_PBL_FILES && looks_like_pbl(basename))
 347                                         continue;
 348                                 if(!basename.empty() && basename[0] == '.' )
 349                                         continue;
 350                         }
 351                         push_if_exists(files, di->path(), mode == ENTIRE_FILE_PATH);
 352 
 353                         if (checksum != nullptr) {
 354                                 std::time_t mtime = bfs::last_write_time(di->path(), ec);
 355                                 if (ec) {
 356                                         LOG_FS << "Failed to read modification time of " << di->path().string() << ": " << ec.message() << '\n';
 357                                 } else if (mtime > checksum->modified) {
 358                                         checksum->modified = mtime;
 359                                 }
 360 
 361                                 uintmax_t size = bfs::file_size(di->path(), ec);
 362                                 if (ec) {
 363                                         LOG_FS << "Failed to read filesize of " << di->path().string() << ": " << ec.message() << '\n';
 364                                 } else {
 365                                         checksum->sum_size += size;
 366                                 }
 367                                 checksum->nfiles++;
 368                         }
 369                 } else if (st.type() == bfs::directory_file) {
 370                         std::string basename = di->path().filename().string();
 371 
 372                         if(!basename.empty() && basename[0] == '.' )
 373                                 continue;
 374                         if (filter == SKIP_MEDIA_DIR
 375                                 && (basename == "images" || basename == "sounds"))
 376                                 continue;
 377 
 378                         const path inner_main(di->path() / maincfg_filename);
 379                         bfs::file_status main_st = bfs::status(inner_main, ec);
 380                         if (error_except_not_found(ec)) {
 381                                 LOG_FS << "Failed to get file status of " << inner_main.string() << ": " << ec.message() << '\n';
 382                         } else if (reorder == DO_REORDER && main_st.type() == bfs::regular_file) {
 383                                 LOG_FS << "_main.cfg found : " << (mode == ENTIRE_FILE_PATH ? inner_main.string() : inner_main.filename().string()) << '\n';
 384                                 push_if_exists(files, inner_main, mode == ENTIRE_FILE_PATH);
 385                         } else {
 386                                 push_if_exists(dirs, di->path(), mode == ENTIRE_FILE_PATH);
 387                         }
 388                 }
 389         }
 390 
 391         if (files != nullptr)
 392                 std::sort(files->begin(),files->end());
 393 
 394         if (dirs != nullptr)
 395                 std::sort(dirs->begin(),dirs->end());
 396 
 397         if (files != nullptr && reorder == DO_REORDER) {
 398                 // move finalcfg_filename, if present, to the end of the vector
 399                 for (unsigned int i = 0; i < files->size(); i++) {
 400                         if (ends_with((*files)[i], "/" + finalcfg_filename)) {
 401                                 files->push_back((*files)[i]);
 402                                 files->erase(files->begin()+i);
 403                                 break;
 404                         }
 405                 }
 406                 // move initialcfg_filename, if present, to the beginning of the vector
 407                 int foundit = -1;
 408                 for (unsigned int i = 0; i < files->size(); i++)
 409                         if (ends_with((*files)[i], "/" + initialcfg_filename)) {
 410                                 foundit = i;
 411                                 break;
 412                         }
 413                 if (foundit > 0) {
 414                         std::string initialcfg = (*files)[foundit];
 415                         for (unsigned int i = foundit; i > 0; i--)
 416                                 (*files)[i] = (*files)[i-1];
 417                         (*files)[0] = initialcfg;
 418                 }
 419         }
 420 }

 427 std::string get_next_filename(const std::string& name, const std::string& extension)
 428 {
 429         std::string next_filename;
 430         int counter = 0;
 431 
 432         do {
 433                 std::stringstream filename;
 434 
 435                 filename << name;
 436                 filename.width(3);
 437                 filename.fill('0');
 438                 filename.setf(std::ios_base::right);
 439                 filename << counter << extension;
 440                 counter++;
 441                 next_filename = filename.str();
 442         } while(file_exists(next_filename) && counter < 1000);
 443         return next_filename;
 444 }

 503 void set_user_data_dir(std::string newprefdir)
 504 {
 505 #ifdef PREFERENCES_DIR
 506         if (newprefdir.empty()) newprefdir = PREFERENCES_DIR;
 507 #endif
 508 
 509 #ifdef _WIN32
 510         if(newprefdir.size() > 2 && newprefdir[1] == ':') {
 511                 //allow absolute path override
 512                 user_data_dir = newprefdir;
 513         } else if(is_path_relative_to_cwd(newprefdir)) {
 514                 // Custom directory relative to workdir (for portable installs, etc.)
 515                 user_data_dir = get_cwd() + "/" + newprefdir;
 516         } else {
 517                 if(newprefdir.empty()) {
 518                         newprefdir = "Wesnoth" + get_version_path_suffix();
 519                 }
 520 
 521                 wchar_t docs_path[MAX_PATH];
 522 
 523                 HRESULT res = SHGetFolderPathW(nullptr,
 524                                                                            CSIDL_PERSONAL | CSIDL_FLAG_CREATE, nullptr,
 525                                                                            SHGFP_TYPE_CURRENT,
 526                                                                            docs_path);
 527                 if(res != S_OK) {
 528                         //
 529                         // Crummy fallback path full of pain and suffering.
 530                         //
 531                         ERR_FS << "Could not determine path to user's Documents folder! ("
 532                                    << std::hex << "0x" << res << std::dec << ") "
 533                                    << "User config/data directories may be unavailable for "
 534                                    << "this session. Please report this as a bug.\n";
 535                         user_data_dir = path(get_cwd()) / newprefdir;
 536                 } else {
 537                         path games_path = path(docs_path) / "My Games";
 538                         create_directory_if_missing(games_path);
 539 
 540                         user_data_dir = games_path / newprefdir;
 541                 }
 542         }
 543 
 544 #else /*_WIN32*/
 545 
 546         std::string backupprefdir = ".wesnoth" + get_version_path_suffix();
 547 
 548 #ifdef _X11
 549         const char *home_str = getenv("HOME");
 550 
 551         if (newprefdir.empty()) {
 552                 char const *xdg_data = getenv("XDG_DATA_HOME");
 553                 if (!xdg_data || xdg_data[0] == '\0') {
 554                         if (!home_str) {
 555                                 newprefdir = backupprefdir;
 556                                 goto other;
 557                         }
 558                         user_data_dir = home_str;
 559                         user_data_dir /= ".local/share";
 560                 } else user_data_dir = xdg_data;
 561                 user_data_dir /= "wesnoth";
 562                 user_data_dir /= get_version_path_suffix();
 563         } else {
 564                 other:
 565                 path home = home_str ? home_str : ".";
 566 
 567                 if (newprefdir[0] == '/')
 568                         user_data_dir = newprefdir;
 569                 else
 570                         user_data_dir = home / newprefdir;
 571         }
 572 #else
 573         if (newprefdir.empty()) newprefdir = backupprefdir;
 574 
 575         const char* home_str = getenv("HOME");
 576         path home = home_str ? home_str : ".";
 577 
 578         if (newprefdir[0] == '/')
 579                 user_data_dir = newprefdir;
 580         else
 581                 user_data_dir = home / newprefdir;
 582 #endif
 583 
 584 #endif /*_WIN32*/
 585         setup_user_data_dir();
 586         user_data_dir = normalize_path(user_data_dir.string(), true, true);
 587 }

 820 // Throws io_exception if an error occurs
 821 void write_file(const std::string& fname, const std::string& data)
 822 {
 823         scoped_ostream os = ostream_file(fname);
 824         os->exceptions(std::ios_base::goodbit);
 825 
 826         const size_t block_size = 4096;
 827         char buf[block_size];
 828 
 829         for(size_t i = 0; i < data.size(); i += block_size) {
 830                 const size_t bytes = std::min<size_t>(block_size,data.size() - i);
 831                 std::copy(data.begin() + i, data.begin() + i + bytes,buf);
 832 
 833                 os->write(buf, bytes);
 834                 if (os->bad())
 835                         throw io_exception("Error writing to file: '" + fname + "'");
 836         }
 837 }

 953 bool is_root(const std::string& path)
 954 {
 955 #ifndef _WIN32
 956         error_code ec;
 957         const bfs::path& p = bfs::canonical(path, ec);
 958         return ec ? false : !p.has_parent_path();
 959 #else
 960         //
 961         // Boost.Filesystem is completely unreliable when it comes to detecting
 962         // whether a path refers to a drive's root directory on Windows, so we are
 963         // forced to take an alternative approach here. Instead of hand-parsing
 964         // strings we'll just call a graphical shell service.
 965         //
 966         // There are several poorly-documented ways to refer to a drive in Windows by
 967         // escaping the filesystem namespace using \\.\, \\?\, and \??\. We're just
 968         // going to ignore those here, which may yield unexpected results in places
 969         // such as the file dialog. This function really shouldn't be used for
 970         // security validation anyway, and there are virtually infinite ways to name
 971         // a drive's root using the NT object namespace so it's pretty pointless to
 972         // try to catch those there.
 973         //
 974         // (And no, shlwapi.dll's PathIsRoot() doesn't recognize \\.\C:\, \\?\C:\, or
 975         // \??\C:\ as roots either.)
 976         //
 977         // More generally, do NOT use this code in security-sensitive applications.
 978         //
 979         // See also: <https://googleprojectzero.blogspot.com/2016/02/the-definitive-guide-on-win32-to-nt.html>
 980         //
 981         const std::wstring& wpath = bfs::path{path}.make_preferred().wstring();
 982         return PathIsRootW(wpath.c_str()) == TRUE;
 983 #endif
 984 }

1156 std::string get_binary_file_location(const std::string& type, const std::string& filename)
1157 {
1158         // We define ".." as "remove everything before" this is needed becasue
1159         // on the one hand allowing ".." would be a security risk but
1160         // especialy for terrains the c++ engine puts a hardcoded "terrain/" before filename
1161         // and there would be no way to "escape" from "terrain/" otherwise. This is not the
1162         // best solution but we cannot remove it without another solution (subtypes maybe?).
1163 
1164         // using 'for' instead 'if' to allow putting delcaration and check into the brackets
1165         for(std::string::size_type pos = filename.rfind("../"); pos != std::string::npos;)
1166                 return get_binary_file_location(type, filename.substr(pos + 3));
1167 
1168         if (!is_legal_file(filename))
1169                 return std::string();
1170 
1171         for(const std::string &bp : get_binary_paths(type))
1172         {
1173                 path bpath(bp);
1174                 bpath /= filename;
1175                 DBG_FS << "  checking '" << bp << "'\n";
1176                 if (file_exists(bpath)) {
1177                         DBG_FS << "  found at '" << bpath.string() << "'\n";
1178                         return bpath.string();
1179                 }
1180         }
1181 
1182         DBG_FS << "  not found\n";
1183         return std::string();
1184 }

1240 static path subtract_path(const path &full, const path &prefix)
1241 {
1242         path::iterator fi = full.begin()
1243                      , fe = full.end()
1244                      , pi = prefix.begin()
1245                      , pe = prefix.end();
1246         while (fi != fe && pi != pe && *fi == *pi) {
1247                 ++fi;
1248                 ++pi;
1249         }
1250         path rest;
1251         if (pi == pe)
1252                 while (fi != fe) {
1253                         rest /= *fi;
1254                         ++fi;
1255                 }
1256         return rest;
1257 }
