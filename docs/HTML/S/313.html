 194 terrain_type::terrain_type(const terrain_type& base, const terrain_type& overlay) :
 195         icon_image_(),

 197         minimap_image_overlay_(overlay.minimap_image_),
 198         editor_image_(base.editor_image_ + "~BLIT(" + overlay.editor_image_ +")"),
 199         id_(base.id_+"^"+overlay.id_),
 200         name_(overlay.name_),
 201         editor_name_((base.editor_name_.empty() ? base.name_ : base.editor_name_) + " / " + (overlay.editor_name_.empty() ? overlay.name_ : overlay.editor_name_)),
 202         description_(overlay.description()),
 203         help_topic_text_(),
 204         number_(t_translation::terrain_code(base.number_.base, overlay.number_.overlay)),
 205         mvt_type_(overlay.mvt_type_),
 206         vision_type_(overlay.vision_type_),
 207         def_type_(overlay.def_type_),
 208         union_type_(),
 209         height_adjust_(base.height_adjust_),
 210         height_adjust_set_(base.height_adjust_set_),
 211         submerge_(base.submerge_),
 212         submerge_set_(base.submerge_set_),

 214         max_light_(std::max(base.max_light_, overlay.max_light_)),
 215         min_light_(std::min(base.min_light_, overlay.min_light_)),
 216         heals_(std::max<int>(base.heals_, overlay.heals_)),
 217         income_description_(),
 218         income_description_ally_(),

 220         income_description_own_(),
 221         editor_group_(),

 224         keep_(base.keep_ || overlay.keep_),
 225         overlay_(false),
 226         combined_(true),
 227         editor_default_base_(),

 229         hide_in_editor_(base.hide_in_editor_ || overlay.hide_in_editor_)

 248

 250         union_type_.insert( union_type_.end(), def_type_.begin(), def_type_.end() );
 251         union_type_.insert( union_type_.end(), vision_type_.begin(), vision_type_.end() );
 252 
 253         // remove + and -
 254         union_type_.erase(std::remove(union_type_.begin(), union_type_.end(),

 256 
 257         union_type_.erase(std::remove(union_type_.begin(), union_type_.end(),

 259 
 260         // remove doubles
 261         std::sort(union_type_.begin(),union_type_.end());
 262         union_type_.erase(std::unique(union_type_.begin(), union_type_.end()), union_type_.end());

 265

 276                 income_description_enemy_ = overlay.income_description_enemy_;

 313 void create_terrain_maps(const config::const_child_itors &cfgs,
 314                          t_translation::ter_list& terrain_list,
 315                          std::map<t_translation::terrain_code, terrain_type>& letter_to_terrain)
 316 {
 317         for (const config &terrain_data : cfgs)
 318         {
 319                 terrain_type terrain(terrain_data);
 320                 DBG_G << "create_terrain_maps: " << terrain.number() << " "
 321                         << terrain.id() << " " << terrain.name() << " : " << terrain.editor_group() << "\n";
 322 
 323                 std::pair<std::map<t_translation::terrain_code, terrain_type>::iterator, bool> res;
 324                 res = letter_to_terrain.emplace(terrain.number(), terrain);
 325                 if (!res.second) {
 326                         terrain_type& curr = res.first->second;
 327                         if(terrain == curr) {
 328                                 LOG_G << "Merging terrain " << terrain.number()
 329                                         << ": " << terrain.id() << " (" << terrain.name() << ")\n";
 330                                 std::vector<std::string> eg1 = utils::split(curr.editor_group());
 331                                 std::vector<std::string> eg2 = utils::split(terrain.editor_group());
 332                                 std::set<std::string> egs;
 333                                 bool clean_merge = true;
 334                                 for (std::string& t : eg1) {
 335                                         clean_merge &= egs.insert(t).second;
 336                                 }
 337                                 for (std::string& t : eg2) {
 338                                         clean_merge &= egs.insert(t).second;
 339                                 }
 340 
 341                                 std::string joined = utils::join(egs);
 342                                 curr.set_editor_group(joined);
 343                                 if(clean_merge) {
 344                                         LOG_G << "Editor groups merged to: " << joined << "\n";
 345                                 } else {
 346                                         LOG_G << "Merged terrain " << terrain.number()
 347                                         << ": " << terrain.id() << " (" << terrain.name() << ") "
 348                                         << "with duplicate editor groups [" << terrain.editor_group() << "] "
 349                                         << "and [" << curr.editor_group() << "]\n";
 350                                 }
 351                         } else {
 352                                 ERR_G << "Duplicate terrain code definition found for " << terrain.number() << "\n"
 353                                         << "Failed to add terrain " << terrain.id() << " (" << terrain.name() << ") "
 354                                         << "[" << terrain.editor_group() << "]" << "\n"
 355                                         << "which conflicts with  " << curr.id() << " (" << curr.name() << ") "
 356                                         << "[" << curr.editor_group() << "]" << "\n\n";
 357                         }
 358                 } else {
 359                         terrain_list.push_back(terrain.number());
 360                 }
 361         }
 362 }

 364 void merge_alias_lists(t_translation::ter_list& first, const t_translation::ter_list& second)
 365 {
 366         // Insert second vector into first when the terrain _ref^base is encountered
 367 
 368         bool revert = (first.front() == t_translation::MINUS ? true : false);
 369         t_translation::ter_list::iterator i;
 370 
 371         for(i = first.begin(); i != first.end(); ++i) {
 372                 if(*i == t_translation::PLUS) {
 373                         revert = false;
 374                         continue;
 375                 } else if(*i == t_translation::MINUS) {
 376                         revert = true;
 377                         continue;
 378                 }
 379 
 380                 if(*i == t_translation::BASE) {
 381                         t_translation::ter_list::iterator insert_it = first.erase(i);
 382                         //if we are in reverse mode, insert PLUS before and MINUS after the base list
 383             //so calculation of base aliases will work normal
 384                         if(revert) {
 385 //                              insert_it = first.insert(insert_it, t_translation::PLUS);
 386 //                              insert_it++;
 387                                 insert_it = first.insert(insert_it, t_translation::MINUS);
 388                         }
 389                         else {
 390                 //else insert PLUS after the base aliases to restore previous "reverse state"
 391                                 insert_it =  first.insert(insert_it, t_translation::PLUS);
 392                         }
 393 
 394                         first.insert(insert_it, second.begin(), second.end());
 395 
 396                         break;
 397                 }
 398         }
