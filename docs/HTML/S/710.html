  26 inline void
  27 blend_image(const surface& src, tblend_functor functor)
  28 {
  29         for(Uint32 color = 0x00FF0000; color != 0x00000000; color >>= 8) {
  30                 for(int i = 0xf; i < 0x100; i += 0x10) {
  31                         const surface dst = blend_surface(src, i / 255., color);
  32                         if(functor) {
  33                                 functor(dst, i, color);
  34                         }
  35                 }
  36         }
  37 }

  39 inline std::string
  40 blend_get_filename(std::string root, const Uint8 amount, const Uint32 color)
  41 {
  42         // The name of the file is
  43         // A the amount of blended [0..256) as hex.
  44         // C the color to blend with as hex.
  45         char filename[] = "AA_CCCCCCCC.png";
  46         snprintf(filename, sizeof(filename), "%02X_%08X.png", amount, color);
  47 
  48         return root + filename;
  49 }
