   8 function ca_fast_combat_leader:evaluation(cfg, data)
   9     -- Some parts of this are very similar to ca_fast_combat.lua.
  10     -- However, for speed reasons we really want this to be a separate CA, so
  11     -- that the (more expensive) calculations for keeping the leader safe only
  12     -- get done once, when all other Fast MAI CAs are entirely done.
  13 
  14     leader_weight = (cfg and cfg.leader_weight) or 2
  15     leader_attack_max_units = (cfg and cfg.leader_attack_max_units) or 3
  16     leader_additional_threat = (cfg and cfg.leader_additional_threat) or 1
  17 
  18     data.move_cache = { turn = wesnoth.current.turn }
  19     data.gamedata = FAU.gamedata_setup()
  20 
  21     local filter_own = H.get_child(cfg, "filter")
  22     local filter_enemy = H.get_child(cfg, "filter_second")
  23 
  24     local enemies, leader
  25     if (not filter_own) and (not filter_enemy) then
  26         local attacks_aspect = ai.aspects.attacks
  27         for _,unit in ipairs(attacks_aspect.own) do
  28             if unit.canrecruit and (unit.attacks_left > 0) and (#unit.attacks > 0) then
  29                 leader = unit
  30                 break
  31             end
  32         end
  33         if (not leader) then return 0 end
  34         enemies = attacks_aspect.enemy
  35     else
  36         leader = AH.get_live_units(
  37             FAU.build_attack_filter("leader", filter_own)
  38         )[1]
  39         if (not leader) or (leader.attacks_left == 0) or (#leader.attacks == 0) then return 0 end
  40         enemies = AH.get_live_units(
  41             FAU.build_attack_filter("enemy", filter_enemy)
  42         )
  43     end
  44 
  45     local enemy_map = LS.create()
  46     for _,e in ipairs(enemies) do
  47         enemy_map:insert(e.x, e.y)
  48     end
  49 
  50     -- Exclude hidden enemies, except if attack_hidden_enemies=yes is set in [micro_ai] tag
  51     local viewing_side = wesnoth.current.side
  52     if (not cfg.attack_hidden_enemies) then
  53         local hidden_enemies = AH.get_live_units {
  54             { "filter_side", { { "enemy_of", { side = wesnoth.current.side } } } },
  55             { "filter_vision", { side = wesnoth.current.side, visible = 'no' } }
  56         }
  57 
  58         for _,e in ipairs(hidden_enemies) do
  59             enemy_map:remove(e.x, e.y)
  60         end
  61     else
  62         viewing_side = 0
  63     end
  64 
  65     local aggression = ai.aspects.aggression
  66     if (aggression > 1) then aggression = 1 end
  67     local own_value_weight = 1. - aggression
  68 
  69     -- Get the locations to be avoided
  70     local avoid_map = FAU.get_avoid_map(cfg)
  71 
  72     -- Enemy power and number maps
  73     -- Currently, the power is simply the summed hitpoints of all enemies that
  74     -- can get to a hex
  75     local enemies = AH.get_live_units {
  76         { "filter_side", { { "enemy_of", { side = wesnoth.current.side } } } }
  77     }
  78 
  79     local enemy_power_map = LS.create()
  80     local enemy_number_map = LS.create()
  81 
  82     for _,enemy in ipairs(enemies) do
  83         -- Only need to consider enemies that are close enough
  84         if (wesnoth.map.distance_between(leader.x, leader.y, enemy.x, enemy.y) <= (enemy.max_moves + leader.max_moves + 1)) then
  85             enemy_power = enemy.hitpoints
  86 
  87             local old_moves = enemy.moves
  88             enemy.moves = enemy.max_moves
  89             local reach = wesnoth.find_reach(enemy)
  90             enemy.moves = old_moves
  91 
  92             for _,loc in ipairs(reach) do
  93                 enemy_power_map:insert(
  94                     loc[1], loc[2],
  95                     (enemy_power_map:get(loc[1], loc[2]) or 0) + enemy_power
  96                 )
  97                 enemy_number_map:insert(
  98                     loc[1], loc[2],
  99                     (enemy_number_map:get(loc[1], loc[2]) or 0) + 1
 100                 )
 101             end
 102         end
 103     end
 104 
 105     local leader_info = FAU.get_unit_info(leader, data.gamedata)
 106     local leader_copy = FAU.get_unit_copy(leader.id, data.gamedata)
 107 
 108     -- If threatened_leader_fights=yes, check out the current threat (power only,
 109     -- not units) on the AI leader
 110     local leader_current_threat = 0
 111     if cfg and cfg.threatened_leader_fights then
 112         for xa,ya in H.adjacent_tiles(leader.x, leader.y) do
 113             local enemy_power = enemy_power_map:get(xa, ya) or 0
 114             if (enemy_power > leader_current_threat) then
 115                 leader_current_threat = enemy_power
 116             end
 117         end
 118     end
 119 
 120     local attacks = AH.get_attacks({ leader }, { include_occupied = cfg.include_occupied_attack_hexes, viewing_side = viewing_side })
 121 
 122     if (#attacks > 0) then
 123         local max_rating, best_target, best_dst = -9e99
 124         for _,attack in ipairs(attacks) do
 125             if enemy_map:get(attack.target.x, attack.target.y)
 126                 and (not avoid_map:get(attack.dst.x, attack.dst.y))
 127             then
 128                 -- First check if the threat against the leader at this hex
 129                 -- is acceptable
 130                 local acceptable_attack = true
 131                 for xa,ya in H.adjacent_tiles(attack.dst.x, attack.dst.y) do
 132                     local enemy_power = enemy_power_map:get(xa, ya) or 0
 133                     local enemy_number = enemy_number_map:get(xa, ya) or 0
 134 
 135                     -- A threat is considered acceptable, if it is within the
 136                     -- limits given for HP and units, or if it is not more than the
 137                     -- threat on the leader at the current position times
 138                     -- leader_additional_threat (the latter only if
 139                     -- threatened_leader_fights=yes is set, otherwise
 140                     -- leader_current_threat is zero)
 141                     if (enemy_power > leader_current_threat * leader_additional_threat) then
 142                         if (enemy_power * leader_weight > leader.hitpoints)
 143                             or (enemy_number > leader_attack_max_units)
 144                         then
 145                             acceptable_attack = false
 146                             break
 147                         end
 148                     end
 149                 end
 150 
 151                 if acceptable_attack then
 152                     local target = wesnoth.get_unit(attack.target.x, attack.target.y)
 153                     local target_info = FAU.get_unit_info(target, data.gamedata)
 154 
 155                     local att_stat, def_stat = FAU.battle_outcome(
 156                         leader_copy, target, { attack.dst.x, attack.dst.y },
 157                         leader_info, target_info, data.gamedata, data.move_cache
 158                     )
 159 
 160                     local rating, attacker_rating, defender_rating, extra_rating = FAU.attack_rating(
 161                         { leader_info }, target_info, { { attack.dst.x, attack.dst.y } },
 162                         { att_stat }, def_stat, data.gamedata,
 163                         {
 164                             own_value_weight = own_value_weight,
 165                             leader_weight = cfg.leader_weight
 166                         }
 167                     )
 168 
 169                     acceptable_attack = FAU.is_acceptable_attack(attacker_rating, defender_rating, own_value_weight)
 170 
 171                     if acceptable_attack and (rating > max_rating) then
 172                         max_rating, best_target, best_dst = rating, target, attack.dst
 173                     end
 174                 end
 175             end
 176         end
 177 
 178         if best_target then
 179             data.leader = leader
 180             data.fast_target, data.fast_dst = best_target, best_dst
 181             return cfg.ca_score
 182         end
 183     end
 184 
 185     return 0
 186 end
