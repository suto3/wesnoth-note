 133 void menu_button::signal_handler_left_button_click(const event::ui_event event, bool& handled)
 134 {
 135         assert(get_window());
 136         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 137 
 138         sound::play_UI_sound(settings::sound_button_click);
 139 
 140         // If a button has a retval do the default handling.
 141         dialogs::drop_down_menu droplist(this->get_rectangle(), this->values_, this->selected_, this->get_use_markup(), this->keep_open_,
 142                 nullptr);
 143 
 144         droplist_ = &droplist;
 145 
 146         if(droplist.show(get_window()->video())) {
 147                 droplist_ = nullptr;
 148 
 149                 const int selected = droplist.selected_item();
 150 
 151                 // Saftey check. If the user clicks a selection in the dropdown and moves their mouse away too
 152                 // quickly, selected_ could be set to -1. This returns in that case, preventing crashes.
 153                 if(selected < 0) {
 154                         return;
 155                 }
 156 
 157                 selected_ = selected;
 158 
 159                 this->set_label(values_[selected_]["label"]);
 160 
 161                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 162 
 163                 if(callback_state_change_) {
 164                         callback_state_change_(*this);
 165                 }
 166 
 167                 if(retval_ != 0) {
 168                         if(window* window = get_window()) {
 169                                 window->set_retval(retval_);
 170                                 return;
 171                         }
 172                 }
 173         }
 174 
 175         droplist_ = nullptr;
 176 
 177         handled = true;
 178 }

 180 void menu_button::set_values(const std::vector<::config>& values, int selected)
 181 {
 182         assert(static_cast<size_t>(selected) < values.size());
 183         assert(static_cast<size_t>(selected_) < values_.size());
 184 
 185         if(values[selected]["label"] != values_[selected_]["label"]) {
 186                 set_is_dirty(true);
 187         }
 188 
 189         values_ = values;
 190         selected_ = selected;
 191 
 192         set_label(values_[selected_]["label"]);
 193 }

 199 
 200         if(selected != selected_) {
 201                 set_is_dirty(true);
 202         }
 203 
 204         selected_ = selected;
 205 
 206         set_label(values_[selected_]["label"]);
 207 }

 314         init_control(widget);
 315 
 316         widget->set_retval(get_retval(retval_id_, retval_, id));
 317         if(!options_.empty()) {
