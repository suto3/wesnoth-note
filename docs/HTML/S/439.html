  41         void click_callback(window& window, bool keep_open, bool&, bool&, point coordinate)
  42         {
  43                 listbox& list = find_widget<listbox>(&window, "list", true);
  44 
  45                 /* Disregard clicks on scrollbars and toggle buttons so the dropdown menu can be scrolled or have an embedded
  46                  * toggle button selected without the menu closing.
  47                  *
  48                  * This works since this click_callback function is called before widgets' left-button-up handlers.
  49                  *
  50                  * Additionally, this is done before row deselection so selecting/deselecting a toggle button doesn't also leave
  51                  * the list with no row visually selected. Oddly, the visial deselection doesn't seem to cause any crashes, and
  52                  * the previously selected row is reselected when the menu is opened again. Still, it's odd to see your selection
  53                  * vanish.
  54                  */
  55                 if(list.vertical_scrollbar()->get_state() == scrollbar_base::PRESSED) {
  56                         return;
  57                 }
  58 
  59                 if(dynamic_cast<toggle_button*>(window.find_at(coordinate, true)) != nullptr) {
  60                         return;
  61                 }
  62 
  63                 /* FIXME: This dialog uses a listbox with 'has_minimum = false'. This allows a listbox to have 0 or 1 selections,
  64                  * and selecting the same entry toggles that entry's state (ie, if it was selected, it will be deselected). Because
  65                  * of this, selecting the same entry in the dropdown list essentially sets the list's selected row to -1, causing problems.
  66                  *
  67                  * In order to work around this, we first manually deselect the selected entry here. This handler is called *before*
  68                  * the listbox's click handler, and as such the selected item will remain toggled on when the click handler fires.
  69                  */
  70                 const int sel = list.get_selected_row();
  71                 if(sel >= 0) {
  72                         list.select_row(sel, false);
  73                 }
  74 
  75                 SDL_Rect rect = window.get_rectangle();
  76                 if(!sdl::point_in_rect(coordinate, rect)) {
  77                         window.set_retval(window::CANCEL);
  78                 } else if(!keep_open) {
  79                         window.set_retval(window::OK);
  80                 }
  81         }

  83         void callback_flip_embedded_toggle(window& window)
  84         {
  85                 listbox& list = find_widget<listbox>(&window, "list", true);
  86 
  87                 /* If the currently selected row has a toggle button, toggle it.
  88                  * Note this cannot be handled in click_callback since at that point the new row selection has not registered,
  89                  * meaning the currently selected row's button is toggled.
  90                  */
  91                 grid* row_grid = list.get_row_grid(list.get_selected_row());
  92                 if(toggle_button* checkbox = find_widget<toggle_button>(row_grid, "checkbox", false, false)) {
  93                         checkbox->set_value_bool(!checkbox->get_value_bool());
  94                 }
  95         }

  97         void resize_callback(window& window)
  98         {
  99                 window.set_retval(window::CANCEL);
 100                 window.close();

 104 void drop_down_menu::pre_show(window& window)
 105 {
 106         window_ = &window;
 107 
 108         window.set_variable("button_x", wfl::variant(button_pos_.x));
 109         window.set_variable("button_y", wfl::variant(button_pos_.y));
 110         window.set_variable("button_w", wfl::variant(button_pos_.w));
 111         window.set_variable("button_h", wfl::variant(button_pos_.h));
 112 
 113         listbox& list = find_widget<listbox>(&window, "list", true);
 114 
 115         std::map<std::string, string_map> data;
 116         string_map item;
 117 
 118         for(const auto& entry : items_) {
 119                 data.clear();
 120 
 121                 item["label"] = entry["icon"];
 122                 data.emplace("icon", item);
 123 
 124                 if(!entry.has_attribute("image")) {
 125                         item["label"] = entry["label"];
 126                         item["use_markup"] = utils::bool_string(use_markup_);
 127                         data.emplace("label", item);
 128                 }
 129 
 130                 if(entry.has_attribute("details")) {
 131                         item["label"] = entry["details"];
 132                         item["use_markup"] = utils::bool_string(use_markup_);
 133                         data.emplace("details", item);
 134                 }
 135 
 136                 grid& new_row = list.add_row(data);
 137 
 138                 // Set the tooltip on the whole panel
 139                 find_widget<toggle_panel>(&new_row, "panel", false).set_tooltip(entry["tooltip"]);
 140 
 141                 if(entry.has_attribute("image")) {
 142                         image* img = new image();
 143                         img->set_definition("default");
 144                         img->set_label(entry["image"]);
 145 
 146                         grid* mi_grid = dynamic_cast<grid*>(new_row.find("menu_item", false));
 147                         if(mi_grid) {
 148                                 mi_grid->swap_child("label", img, false);
 149                         }
 150                 }
 151 
 152                 if(entry.has_attribute("checkbox")) {
 153                         toggle_button* checkbox = new toggle_button();
 154                         checkbox->set_definition("default");
 155                         checkbox->set_id("checkbox");
 156                         checkbox->set_value_bool(entry["checkbox"].to_bool(false));
 157 
 158                         if(callback_toggle_state_change_ != nullptr) {
 159                                 checkbox->set_callback_state_change(std::bind(callback_toggle_state_change_));
 160                         }
 161 
 162                         grid* mi_grid = dynamic_cast<grid*>(new_row.find("menu_item", false));
 163                         if(mi_grid) {
 164                                 mi_grid->swap_child("icon", checkbox, false);
 165                         }
 166                 }
 167         }
 168 
 169         if(selected_item_ >= 0 && unsigned(selected_item_) < list.get_item_count()) {
 170                 list.select_row(selected_item_);
 171         }
 172 
 173         window.keyboard_capture(&list);
 174 
 175         // Dismiss on click outside the window
 176         window.connect_signal<event::SDL_LEFT_BUTTON_UP>(
 177                 std::bind(&click_callback, std::ref(window), keep_open_, _3, _4, _5), event::dispatcher::front_child);
 178 
 179         // Handle embedded button toggling.
 180         // For some reason this works as a listbox value callback but don't ask me why.
 181         // -vultraz 2/17/17
 182         list.set_callback_value_change(std::bind(&callback_flip_embedded_toggle, std::ref(window)));
 183 
 184         // Dismiss on resize
 185         window.connect_signal<event::SDL_VIDEO_RESIZE>(std::bind(&resize_callback, std::ref(window)), event::dispatcher::front_child);
 186 }

 188 void drop_down_menu::post_show(window& window)
 189 {
 190         listbox& list = find_widget<listbox>(&window, "list", true);
 191 
 192         selected_item_ = list.get_selected_row();
 193 
 194         window_ = nullptr;
 195 }

 197 boost::dynamic_bitset<> drop_down_menu::get_toggle_states() const
 198 {
 199         assert(window_ != nullptr);
 200 
 201         const listbox& list = find_widget<const listbox>(window_, "list", true);
 202 
 203         boost::dynamic_bitset<> states;
 204 
 205         for(unsigned i = 0; i < list.get_item_count(); ++i) {
 206                 const grid* row_grid = list.get_row_grid(i);
 207 
 208                 if(const toggle_button* checkbox = find_widget<const toggle_button>(row_grid, "checkbox", false, false)) {
 209                         states.push_back(checkbox->get_value_bool());
 210                 } else {
 211                         states.push_back(false);
 212                 }
 213         }
 214 
 215         return states;
 216 }
