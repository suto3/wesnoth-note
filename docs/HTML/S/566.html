  63 /**
  64  * Swaps an item in a grid for another one.*/
  65 void swap_grid(grid* g,
  66                            grid* content_grid,
  67                            widget* widget,
  68                            const std::string& id)
  69 {
  70         assert(content_grid);
  71         assert(widget);
  72 
  73         // Make sure the new child has same id.
  74         widget->set_id(id);
  75 
  76         // Get the container containing the wanted widget.
  77         grid* parent_grid = nullptr;
  78         if(g) {
  79                 parent_grid = find_widget<grid>(g, id, false, false);
  80         }
  81         if(!parent_grid) {
  82                 parent_grid = find_widget<grid>(content_grid, id, true, false);
  83         }
  84         parent_grid = dynamic_cast<grid*>(parent_grid->parent());
  85         assert(parent_grid);
  86 
  87         // Replace the child.
  88         auto old = parent_grid->swap_child(id, widget, false);
  89         assert(old);
  90 }

 119 void stacked_widget::select_layer_impl(std::function<bool(unsigned int i)> display_condition)
 120 {
 121         const unsigned int num_layers = get_layer_count();
 122 
 123         // Deselect all layers except the chosen ones.
 124         for(unsigned int i = 0; i < num_layers; ++i) {
 125                 const bool selected = display_condition(i);
 126 
 127                 /* Selecting a previously selected item will deselect it, regardless of the what is passed to
 128                  * select_item. This causes issues if this function is called when all layers are visible (for
 129                  * example, initialization). For layers other than the chosen one, this is the desired behavior.
 130                  * However the chosen layer could *also* be deselected undesirably due to the conditions outlined
 131                  * above, and as this widget's generator does not stipulate a minimum selection, it's possible to
 132                  * end up with no layers visible at all.
 133                  *
 134                  * This works around that by performing no selection unless necessary to change states.
 135                  */
 136                 if(generator_->is_selected(i) != selected) {
 137                         generator_->select_item(i, selected);
 138                 }
 139         }
 140 
 141         // If we already have our chosen layers, exit.
 142         if(selected_layer_ >= 0) {
 143                 return;
 144         }
 145 
 146         // Else, re-show all layers.
 147         for(unsigned int i = 0; i < num_layers; ++i) {
 148                 /* By design, only the last selected item will receive events even if multiple items are visible
 149                  * and said item is not at the top of the stack. If this point is reached, all layers have already
 150                  * been hidden by the loop above, so the last layer selected will be the top-most one, as desired.
 151                  */
 152                 generator_->select_item(i, true);
 153         }
 154 }

 167 void stacked_widget::select_layer(const int layer)
 168 {
 169         update_selected_layer_index(layer);
 170 
 171         select_layer_impl([this](unsigned int i)
 172         {
 173                 return i == static_cast<unsigned int>(selected_layer_);
 174         });
 175 }

 177 void stacked_widget::select_layers(const boost::dynamic_bitset<>& mask)
 178 {
 179         assert(mask.size() == get_layer_count());
 180 
 181         select_layer_impl([&](unsigned int i)
 182         {
 183                 if(mask[i]) {
 184                         update_selected_layer_index(i);
 185                 }
 186 
 187                 return mask[i];
 188         });
 189 }

 216  * == Stacked widget ==
 217  *
 218  * A stacked widget holds several widgets on top of each other. This can be used
 219  * for various effects; add an optional overlay to an image, stack it with a
 220  * spacer to force a minimum size of a widget. The latter is handy to avoid

 222  *
 223  * A stacked widget has no states.

 270 builder_stacked_widget::builder_stacked_widget(const config& real_cfg)
 271         : builder_styled_widget(real_cfg), stack()
 272 {
 273         const config& cfg = real_cfg.has_child("stack") ? real_cfg.child("stack") : real_cfg;
 274         if(&cfg != &real_cfg) {
 275                 lg::wml_error() << "Stacked widgets no longer require a [stack] tag. Instead, place [layer] tags directly in the widget definition.\n";
 276         }
 277         VALIDATE(cfg.has_child("layer"), _("No stack layers defined."));
 278         for(const auto & layer : cfg.child_range("layer"))
 279         {
 280                 stack.emplace_back(std::make_shared<builder_grid>(layer));
 281         }
 282 }

 284 widget* builder_stacked_widget::build() const
 285 {
 286         stacked_widget* widget = new stacked_widget();
 287 
 288         init_control(widget);
 289 
 290         DBG_GUI_G << "Window builder: placed stacked widget '" << id
 291                           << "' with definition '" << definition << "'.\n";
 292 
 293         std::shared_ptr<const stacked_widget_definition::resolution>
 294         conf = std::static_pointer_cast<const stacked_widget_definition::resolution>(
 295                                         widget->config());
 296         assert(conf);
 297 
 298         widget->init_grid(conf->grid);
 299 
 300         widget->finalize(stack);
 301 
 302         return widget;
