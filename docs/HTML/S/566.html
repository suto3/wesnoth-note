  86 void stacked_widget::select_layer_impl(std::function<bool(unsigned int i)> display_condition)
  87 {
  88         const unsigned int num_layers = get_layer_count();
  89 
  90         // Deselect all layers except the chosen ones.
  91         for(unsigned int i = 0; i < num_layers; ++i) {
  92                 const bool selected = display_condition(i);
  93 
  94                 /* Selecting a previously selected item will deselect it, regardless of the what is passed to
  95                  * select_item. This causes issues if this function is called when all layers are visible (for
  96                  * example, initialization). For layers other than the chosen one, this is the desired behavior.
  97                  * However the chosen layer could *also* be deselected undesirably due to the conditions outlined
  98                  * above, and as this widget's generator does not stipulate a minimum selection, it's possible to
  99                  * end up with no layers visible at all.
 100                  *
 101                  * This works around that by performing no selection unless necessary to change states.
 102                  */
 103                 if(generator_->is_selected(i) != selected) {
 104                         generator_->select_item(i, selected);
 105                 }
 106         }
 107 
 108         // If we already have our chosen layers, exit.
 109         if(selected_layer_ >= 0) {
 110                 return;
 111         }
 112 
 113         // Else, re-show all layers.
 114         for(unsigned int i = 0; i < num_layers; ++i) {
 115                 /* By design, only the last selected item will receive events even if multiple items are visible
 116                  * and said item is not at the top of the stack. If this point is reached, all layers have already
 117                  * been hidden by the loop above, so the last layer selected will be the top-most one, as desired.
 118                  */
 119                 generator_->select_item(i, true);
 120         }
 121 }

 134 void stacked_widget::select_layer(const int layer)
 135 {
 136         update_selected_layer_index(layer);
 137 
 138         select_layer_impl([this](unsigned int i)
 139         {
 140                 return i == static_cast<unsigned int>(selected_layer_);
 141         });
 142 }
 143 
 144 void stacked_widget::select_layers(const boost::dynamic_bitset<>& mask)
 145 {
 146         assert(mask.size() == get_layer_count());
 147 
 148         select_layer_impl([&](unsigned int i)
 149         {
 150                 if(mask[i]) {
 151                         update_selected_layer_index(i);
 152                 }
 153 
 154                 return mask[i];
 155         });
 156 }

 183  * == Stacked widget ==
 184  *
 185  * A stacked widget holds several widgets on top of each other. This can be used
 186  * for various effects; add an optional overlay to an image, stack it with a
 187  * spacer to force a minimum size of a widget. The latter is handy to avoid
 188  * making a separate definition for a single instance with a fixed size.
 189  *
 190  * A stacked widget has no states.
 191  * @begin{parent}{name="gui/"}
 192  * @begin{tag}{name="stacked_widget_definition"}{min=0}{max=-1}{super="generic/widget_definition"}
 193  * @begin{tag}{name="resolution"}{min=0}{max=-1}{super="generic/widget_definition/resolution"}

 196  * @end{tag}{name="stacked_widget_definition"}
 197  * @end{parent}{name="gui/"}

 237 builder_stacked_widget::builder_stacked_widget(const config& real_cfg)
 238         : builder_styled_widget(real_cfg), stack()
 239 {
 240         const config& cfg = real_cfg.has_child("stack") ? real_cfg.child("stack") : real_cfg;
 241         if(&cfg != &real_cfg) {
 242                 lg::wml_error() << "Stacked widgets no longer require a [stack] tag. Instead, place [layer] tags directly in the widget definition.\n";
 243         }
 244         VALIDATE(cfg.has_child("layer"), _("No stack layers defined."));
 245         for(const auto & layer : cfg.child_range("layer"))
 246         {
 247                 stack.emplace_back(std::make_shared<builder_grid>(layer));
 248         }
 249 }

 251 widget* builder_stacked_widget::build() const
 252 {
 253         stacked_widget* widget = new stacked_widget();
 254 
 255         init_control(widget);
 256 
 257         DBG_GUI_G << "Window builder: placed stacked widget '" << id
 258                           << "' with definition '" << definition << "'.\n";
 259 
 260         std::shared_ptr<const stacked_widget_definition::resolution>
 261         conf = std::static_pointer_cast<const stacked_widget_definition::resolution>(
 262                                         widget->config());
 263         assert(conf);
 264 
 265         widget->init_grid(conf->grid);
 266 
 267         widget->finalize(stack);
 268 
 269         return widget;
 270 }
