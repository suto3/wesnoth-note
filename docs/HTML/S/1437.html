 167             arguments.add( "--nospellcheck" ); //$NON-NLS-1$
 168         }

 253     /**
 254      * Runs the specified WMLTools as a workspace job
 255      * 
 256      * @param tool
 257      *        The tool to run.
 258      *        Currently only the following tools are supported: WMLLINT,
 259      *        WMLSCOPE, WMLINDENT
 260      * @param targetPath
 261      *        If this is not null if will use the targetpath as the
 262      *        argument for launching the tool
 263      */
 264     public static void runWMLToolAsWorkspaceJob( final Tools tool,
 265         final String targetPath )
 266     {
 267         // TODO: remove/rework this hackish method.
 268         if( tool == Tools.WESNOTH_ADDON_MANAGER ) {
 269             return;
 270         }
 271 
 272         IEditorReference[] editors = WorkspaceUtils.getWorkbenchWindow( )
 273             .getPages( )[0].getEditorReferences( );
 274 
 275         for( IEditorReference editor: editors ) {
 276             if( editor.isDirty( ) ) {
 277                 editor.getEditor( false ).doSave( null );
 278             }
 279         }
 280 
 281         if( targetPath == null && WorkspaceUtils.getSelectedFile( ) != null ) {
 282             EditorUtils.openEditor( WorkspaceUtils.getSelectedFile( ), false );
 283         }
 284         final String toolName = tool.toString( );
 285 
 286         WorkspaceJob job = new WorkspaceJob( Messages.WMLTools_28 + toolName ) {
 287             private ExternalToolInvoker toolInvoker_  = null;
 288             private AtomicInteger       workReporter_ = new AtomicInteger( );
 289 
 290             @Override
 291             protected void canceling( )
 292             {
 293                 toolInvoker_.kill( true );
 294                 super.canceling( );
 295             }
 296 
 297             @Override
 298             public IStatus runInWorkspace( final IProgressMonitor monitor )
 299             {
 300                 monitor.beginTask( toolName, 1050 );
 301                 MessageConsole console = GUIUtils.createConsole( toolName
 302                     + Messages.WMLTools_29, null, true );
 303                 OutputStream[] stdout = new OutputStream[] { console
 304                     .newMessageStream( ) };
 305                 OutputStream[] stderr = new OutputStream[] { console
 306                     .newMessageStream( ) };
 307 
 308                 String location;
 309                 IResource resource = null;
 310                 String installName = Preferences.getDefaultInstallName( );
 311 
 312                 IFile selFile = WorkspaceUtils.getSelectedFile( );
 313                 if( targetPath != null ) {
 314                     location = targetPath;
 315                 }
 316                 else {
 317                     if( selFile != null ) {
 318                         location = selFile.getLocation( ).toOSString( );
 319                         resource = selFile;
 320                     }
 321                     else {
 322                         resource = WorkspaceUtils.getSelectedContainer( );
 323                         location = resource.getLocation( ).toOSString( );
 324                     }
 325 
 326                     installName = WesnothInstallsUtils
 327                         .getInstallNameForResource( resource );
 328                 }
 329 
 330                 switch( tool ) {
 331                     case WMLINDENT:
 332                         if( selFile != null && targetPath == null ) {
 333                             String stdin = EditorUtils.getEditorDocument( )
 334                                 .get( );
 335                             // don't output to stdout as we will put that in
 336                             // the editor
 337                             toolInvoker_ = WMLTools.runWMLIndent(
 338                                 installName, null, stdin, false, null,
 339                                 stdout );
 340                         }
 341                         else {
 342                             toolInvoker_ = WMLTools.runWMLIndent(
 343                                 installName, location, null, false,
 344                                 stdout, stderr );
 345                         }
 346                         break;
 347                     case WMLLINT:
 348                         toolInvoker_ = WMLTools.runWMLLint( installName,
 349                             location, true, false, stdout, stderr );
 350                         break;
 351                     case WMLSCOPE:
 352                         toolInvoker_ = WMLTools.runWMLScope( installName,
 353                             location, false, stdout, stderr );
 354                         break;
 355                 }
 356                 if( toolInvoker_ == null ) {
 357                     return Status.CANCEL_STATUS;
 358                 }
 359 
 360                 monitor.worked( 50 );
 361                 // need to fill up to '1000' worked
 362                 // we will add 1 for each 2 lines of output (for each file)
 363                 Thread stdoutWatcher = new Thread( new Runnable( ) {
 364                     @Override
 365                     public void run( )
 366                     {
 367                         int nr;
 368                         while( toolInvoker_.readOutputLine( ) != null ) {
 369                             nr = workReporter_.incrementAndGet( );
 370                             if( nr % 2 == 0 ) {
 371                                 synchronized( monitor ) {
 372                                     monitor.worked( 1 );
 373                                 }
 374                             }
 375                         }
 376                     }
 377                 } );
 378                 Thread stderrWatcher = new Thread( new Runnable( ) {
 379                     @Override
 380                     public void run( )
 381                     {
 382                         int nr;
 383                         while( toolInvoker_.readErrorLine( ) != null ) {
 384                             nr = workReporter_.incrementAndGet( );
 385                             if( nr % 2 == 0 ) {
 386                                 synchronized( monitor ) {
 387                                     monitor.worked( 1 );
 388                                 }
 389                             }
 390                         }
 391                     }
 392                 } );
 393                 stderrWatcher.start( );
 394                 stdoutWatcher.start( );
 395                 toolInvoker_.waitForTool( );
 396                 if( tool == Tools.WMLINDENT && selFile != null
 397                     && targetPath == null ) {
 398                     EditorUtils.replaceEditorText( toolInvoker_
 399                         .getOutputContent( ) );
 400                 }
 401 
 402                 if( tool == Tools.WMLSCOPE ) {
 403                     if( resource != null ) {
 404                         ResourceUtils.deleteMarkers( resource,
 405                             Constants.MARKER_WMLSCOPE );
 406                     }
 407                     parseAndAddMarkers( toolInvoker_.getOutputContent( ),
 408                         Constants.MARKER_WMLSCOPE );
 409                 }
 410                 else if( tool == Tools.WMLLINT ) {
 411                     if( resource != null ) {
 412                         ResourceUtils.deleteMarkers( resource,
 413                             Constants.MARKER_WMLLINT );
 414                     }
 415                     parseAndAddMarkers( toolInvoker_.getOutputContent( ),
 416                         Constants.MARKER_WMLLINT );
 417                 }
 418 
 419                 monitor.worked( 50 );
 420                 monitor.done( );
 421                 return Status.OK_STATUS;
 422             }
 423         };
 424         job.schedule( );
 425     }

 506     /**
 507      * Parses the output from the WMLTool and adds markers accordingly
 508      * 
 509      * @param output
 510      *        The output to parse and add the markers from.
 511      * @param markerType
 512      *        The type of markers to add from the parsed tokens.
 513      */
 514     public static void parseAndAddMarkers( String output, String markerType )
 515     {
 516         String[] lines = StringUtils.getLines( output );
 517         boolean startMD5 = false;
 518 
 519         for( String line: lines ) {
 520             if( line.startsWith( "#" ) || //$NON-NLS-1$
 521                 line.matches( "^[\\t ]*$" ) || //$NON-NLS-1$
 522                 line.startsWith( "wmllint:" ) ) {
 523                 continue;
 524             }
 525             if( line.startsWith( "%%" ) ) //$NON-NLS-1$
 526             {
 527                 startMD5 = ! startMD5;
 528                 continue;
 529             }
 530             // skip parsing collisions
 531             if( startMD5 ) {
 532                 continue;
 533             }
 534 
 535             ResourceUtils.addMarkerForLine( line, markerType );
 536         }
 537     }

 539     /**
 540      * Checks if the specified wmlTool is existing an can
 541      * be runned (python path is valid aswell)
 542      * 
 543      * @param paths
 544      *        The paths for current install
 545      * @param wmlTool
 546      *        The wml tool to check
 547      * @return True if tool is ok to be used
 548      */
 549     public static boolean checkWMLTool( Paths paths, String wmlTool )
 550     {
 551         String pythonPath = Preferences.getString( Preferences.PYTHON_PATH );
 552         if( pythonPath.isEmpty( )
 553             || ( pythonPath.matches( "^.*(/|\\\\).*$" ) && ! ResourceUtils
 554                 .isValidFilePath( Preferences
 555                     .getString( Preferences.PYTHON_PATH ) ) ) ) {
 556             GUIUtils.showWarnMessageBox( Messages.WMLTools_42 );
 557             return false;
 558         }
 559 
 560         if( wmlTool != null ) {
 561             if( paths.getWMLToolsDir( ).equals( "" ) ) //$NON-NLS-1$
 562             {
 563                 GUIUtils.showWarnMessageBox( Messages.WMLTools_45 );
 564                 return false;
 565             }
 566 
 567             File wmlToolFile = new File( paths.getWMLToolsDir( ) + wmlTool );
 568 
 569             if( ! wmlToolFile.exists( ) ) {
 570                 GUIUtils.showErrorMessageBox( String.format(
 571                     Messages.WMLTools_47, wmlToolFile ) );
 572                 return false;
 573             }
 574 
 575             return true;
 576         }
 577 
 578         return false;
 579     }

 581     /**
 582      * Runs a specified python script with the specified arguments
 583      * (the call returns immediately)
 584      * 
 585      * @param arguments
 586      *        the arguments of the "python" executable.
 587      *        The first argument should be the script file name
 588      * @param stdin
 589      *        A string that will be written to stdin of the python script
 590      * @param stdout
 591      *        An array of streams where to write the stdout from the script.
 592      *        A non null array implies 'stdoutMonitoring' true.
 593      * @param stderr
 594      *        An array of streams where to write the stderr from the script
 595      *        A non null array implies 'stderrMonitoring' true.
 596      * @param stderrMonitoring
 597      *        True to start stderr monitoring on tool
 598      * @param stdoutMonitoring
 599      *        True to start stdout monitoring on tool
 600      * @return An {@link ExternalToolInvoker} instance that launched the
 601      *         specified python script.
 602      */
 603     public static ExternalToolInvoker runPythonScript(
 604         List< String > arguments, String stdin, boolean stderrMonitoring,
 605         boolean stdoutMonitoring, final OutputStream[] stdout,
 606         final OutputStream[] stderr )
 607     {
 608         final ExternalToolInvoker pyscript = new ExternalToolInvoker(
 609             Preferences.getString( Preferences.PYTHON_PATH ), arguments );
 610 
 611         pyscript.runTool( );
 612         if( stderrMonitoring == true || ( stderr != null && stderr.length > 0 ) ) {
 613             pyscript.startErrorMonitor( stderr );
 614         }
 615         if( stdoutMonitoring == true || ( stdout != null && stdout.length > 0 ) ) {
 616             pyscript.startOutputMonitor( stdout );
 617         }
 618         if( stdin != null ) {
 619             try {
 620                 BufferedWriter stdinStream = new BufferedWriter(
 621                     new OutputStreamWriter( pyscript.getStdin( ) ) );
 622                 stdinStream.write( stdin );
 623                 stdinStream.close( );
 624             } catch( IOException e ) {
 625                 Logger.getInstance( ).logException( e );
 626             }
 627         }
 628         return pyscript;
 629     }
