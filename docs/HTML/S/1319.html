 276 # Plural Idle 1:
 277 #  When argument 1 or 2 is expected
 278 #  Here we expect " ' or a [ wich will say us what kind of string we will add.
 279 class LuaPlIdle1:
 280     def __init__(self):
 281         self.regex = None
 282         self.iffail = None
 283         
 284     def run(self, xline, lineno, match):
 285         status = pywmlx.state.machine._pending_luastring.plural.status
 286         realmatch = re.match(r'\s*(\S)', xline)
 287         if realmatch:
 288             _nextstate = 'lua_idle'
 289             if realmatch.group(1) == '"':
 290                 _nextstate = 'lua_pl01'
 291             elif realmatch.group(1) == "'":
 292                 _nextstate = 'lua_pl02'
 293             elif realmatch.group(1) == '[':
 294                 _nextstate = 'lua_pl03'
 295             elif realmatch.group(1) == ')':
 296                 if status == 'wait_string':
 297                     finfo = pywmlx.nodemanip.fileref + ":" + str(lineno)
 298                     errmsg = ("first argument of function '_ (...)' must be a "
 299                               "string")
 300                     wmlerr(finfo, errmsg)
 301                 else:
 302                     # if parenthesis is closed rightly after first argument
 303                     # this is not a plural form. So we set 
 304                     # PendingLuaString.plural to None
 305                     if status == 'wait_plural':
 306                         pywmlx.state.machine._pending_luastring.plural = None
 307                     xline = xline [ realmatch.end(): ]
 308                     return (xline, 'lua_idle')
 309             else:
 310                 # this should never happen. But if the first value after _ (
 311                 # does not start with a string marker, than wmlxgettext will
 312                 # finish with an error
 313                 finfo = pywmlx.nodemanip.fileref + ":" + str(lineno)
 314                 errmsg = "first argument of function '_ (...)' must be a string"
 315                 wmlerr(finfo, errmsg)
 316             xline = xline [ realmatch.end(): ]
 317             xline = realmatch.group(1) + xline
 318             return (xline, _nextstate)
 319                 
 320         else:
 321             # You can find a new line rigthly after _ (. In this case, we will
 322             # continue to search a string into the next line
 323             return (None, 'lua_plidle1')

 374         # 'lua_bug' is not an actual state. The regexp here should always match.

 380         status = pywmlx.state.machine._pending_luastring.plural.status
 381         if status == 'wait_string':
 382             _nextstate = 'lua_plidle2'
 383             # we must fix values for luatype (wich is used later by 
 384             #                                 PendingLuaString.store() )
 385             # and ismultiline

 387             if match.group(2) == '':
 388                 pywmlx.state.machine._pending_luastring.ismultiline = True
 389                 _nextstate = 'lua_pl10'
 390                 xline = None
 391             else:
 392                 xline = xline [ match.end(): ]
 393                 pywmlx.state.machine._pending_luastring.plural.status = (
 394                     'wait_plural'
 395                 )
 396             # write first line of 'string' on PendingLuaString
 397             pywmlx.state.machine._pending_luastring.luastring = match.group(1)
 398             return (xline, _nextstate)
 399         else:
 400             _nextstate = 'lua_plidle3'
 401             # we must fix values for plural.pluraltype
 402             # and plural.ismultiline
 403             pywmlx.state.machine._pending_luastring.plural.pluraltype = 1
 404             if match.group(2) == '':
 405                 pywmlx.state.machine._pending_luastring.plural.ismultiline = True
 406                 _nextstate = 'lua_pl10'
 407                 xline = None
 408             else:
 409                 xline = xline [ match.end(): ]
 410                 pywmlx.state.machine._pending_luastring.plural.status = (
 411                     'wait_close'
 412                 )

 680         self.iffail = None

 683         rx_str = ( r'function\s+([a-zA-Z0-9_.]+)|' +

 697             pywmlx.state.machine._pending_wmlstring.store()
 698             pywmlx.state.machine._pending_wmlstring = None
 699         return (xline, 'lua_idle')
