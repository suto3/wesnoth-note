 101 def checksentence(mystring, finfo, *, islua=False):
 102     m = re.match(r'\s*$', mystring)
 103     if m:
 104         wmlwarn(finfo, "found an empty translatable message")
 105         return 1
 106     elif warnall() and not islua:
 107         if "}" in mystring:
 108             wmsg = ("found a translatable string containing a WML macro. "
 109                     " Translation for this string will NEVER work")
 110             wmlwarn(finfo, wmsg)
 111             return 2
 112         else:
 113             return 0
 114     else:
 115         return 0

 119 class PendingPlural:
 120     def __init__(self):
 121         self.string = ''
 122         # status values:
 123         #    'wait_string'    --> rightly after _ ( when we need to know
 124         #                         wich string type we will manage
 125         #    'wait_plural'    --> after first argument. Search for plural or
 126         #                         close parenthesis
 127         #    'wait_close'     --> expect close parenthesis
 128         self.status = 'wait_string'
 129         self.pluraltype = 0
 130         self.numequals = 0
 131         self.ismultiline = False
 132     
 133     def addline(self, value, isfirstline=False):
 134         if self.pluraltype != 3:
 135             value = re.sub('\\\s*$', '', value)
 136         else:
 137             value = value.replace('\\', r'\\')
 138         if isfirstline:
 139             self.string = value
 140         else:
 141             self.string = self.string + '\n' + value
 142     
 143     def convert(self):
 144         if self.pluraltype == 2:
 145             self.string = re.sub(r"\\\'", r"'", self.string)
 146         if self.pluraltype != 3 and self.pluraltype!=0:
 147             self.string = re.sub(r'(?<!\\)"', r'\"', self.string)
 148         if self.pluraltype == 3:
 149             self.string = self.string.replace('"', r'\"')
 150         if self.ismultiline:    
 151             lf = r'\\n"' + '\n"'
 152             self.string = re.sub(r'(\n\r|\r\n|[\n\r])', 
 153                                 lf, self.string)
 154             self.string = '""\n"' + self.string + '"'
 155         if not self.ismultiline:
 156             self.string = '"' + self.string + '"'
 157         return PoCommentedStringPL(self.string, ismultiline=self.ismultiline)

 161 class PendingLuaString:
 162     def __init__(self, lineno, luatype, luastring, ismultiline, 
 163                  istranslatable, numequals=0, plural=None):
 164         self.lineno = lineno
 165         self.luatype = luatype
 166         self.luastring = ''
 167         self.ismultiline = ismultiline
 168         self.istranslatable = istranslatable
 169         self.numequals = numequals
 170         if luatype != 'lua_plural':
 171             self.addline(luastring, True)
 172         self.plural = plural
 173     
 174     def addline(self, value, isfirstline=False):
 175         if self.luatype != 'luastr3':
 176             value = re.sub('\\\s*$', '', value)
 177         else:
 178             value = value.replace('\\', r'\\')
 179         if isfirstline:
 180             self.luastring = value
 181         else:
 182             self.luastring = self.luastring + '\n' + value
 183     
 184     # this function is used by store, when translating lua pending plural into 
 185     # PoCommentedString.plural
 186     def storePlural(self):
 187         if self.plural is None:
 188             return None
 189         else:
 190             return self.plural.convert()
 191     
 192     def store(self):
 193         global _pending_addedinfo
 194         global _pending_overrideinfo
 195         global _linenosub
 196         if checkdomain() and self.istranslatable:
 197             _linenosub += 1
 198             finfo = pywmlx.nodemanip.fileref + ":" + str(self.lineno)
 199             fileno = pywmlx.nodemanip.fileno
 200             errcode = checksentence(self.luastring, finfo, islua=True)
 201             if errcode != 1:
 202                 # when errcode is equal to 1, the translatable string is empty
 203                 # so, using "if errcode != 1" 
 204                 # we will add the translatable string ONLY if it is NOT empty
 205                 if self.luatype == 'luastr2':
 206                     self.luastring = re.sub(r"\\\'", r"'", self.luastring)
 207                 if self.luatype != 'luastr3':
 208                     self.luastring = re.sub(r'(?<!\\)"', r'\"', self.luastring)
 209                 if self.luatype == 'luastr3':
 210                     self.luastring = self.luastring.replace('"', r'\"')
 211                 loc_wmlinfos = []
 212                 loc_addedinfos = None
 213                 if _pending_overrideinfo is not None:
 214                     loc_wmlinfos.append(_pending_overrideinfo)
 215                 if (_pending_luafuncname is not None and 
 216                         _pending_overrideinfo is None):
 217                     winf = '[lua]: ' + _pending_luafuncname
 218                     loc_wmlinfos.append(winf)
 219                 if _pending_addedinfo is None:
 220                     loc_addedinfos = []
 221                 if _pending_addedinfo is not None:
 222                     loc_addedinfos = _pending_addedinfo
 223                 loc_posentence = _dictionary.get(self.luastring)
 224                 if loc_posentence is None:
 225                     _dictionary[self.luastring] = PoCommentedString(
 226                                 self.luastring, 
 227                                 orderid=(fileno, self.lineno, _linenosub),
 228                                 ismultiline=self.ismultiline,
 229                                 wmlinfos=loc_wmlinfos, finfos=[finfo],
 230                                 addedinfos=loc_addedinfos,
 231                                 plural=self.storePlural() )
 232                 else:
 233                     loc_posentence.update_with_commented_string(
 234                            PoCommentedString(
 235                                 self.luastring, 
 236                                 orderid=(fileno, self.lineno, _linenosub),
 237                                 ismultiline=self.ismultiline,
 238                                 wmlinfos=loc_wmlinfos, finfos=[finfo],
 239                                 addedinfos=loc_addedinfos,
 240                                 plural=self.storePlural()
 241                     ) )
 242         # finally PendingLuaString.store() will clear pendinginfos,
 243         # in any case (even if the pending string is not translatable)
 244         _pending_overrideinfo = None
 245         _pending_addedinfo = None

 249 class PendingWmlString:
 250     def __init__(self, lineno, wmlstring, ismultiline, istranslatable):
 251         self.lineno = lineno
 252         self.wmlstring = wmlstring.replace('\\', r'\\')
 253         self.ismultiline = ismultiline
 254         self.istranslatable = istranslatable
 255     
 256     def addline(self, value):
 257         self.wmlstring = self.wmlstring + '\n' + value.replace('\\', r'\\')
 258     
 259     def store(self):
 260         global _pending_addedinfo 
 261         global _pending_overrideinfo
 262         global _linenosub
 263         global _pending_winfotype
 264         if _pending_winfotype is not None:
 265             if self.ismultiline is False and self.istranslatable is False:
 266                 winf = _pending_winfotype + '=' + self.wmlstring
 267                 pywmlx.nodemanip.addWmlInfo(winf)
 268             _pending_winfotype = None
 269         if checkdomain() and self.istranslatable:
 270             finfo = pywmlx.nodemanip.fileref + ":" + str(self.lineno)
 271             errcode = checksentence(self.wmlstring, finfo, islua=False)
 272             if errcode != 1:
 273                 # when errcode is equal to 1, the translatable string is empty
 274                 # so, using "if errcode != 1" 
 275                 # we will add the translatable string ONLY if it is NOT empty
 276                 _linenosub += 1
 277                 self.wmlstring = re.sub('""', r'\"', self.wmlstring)
 278                 pywmlx.nodemanip.addNodeSentence(self.wmlstring, 
 279                                              ismultiline=self.ismultiline, 
 280                                              lineno=self.lineno, 
 281                                              lineno_sub=_linenosub,
 282                                              override=_pending_overrideinfo, 
 283                                              addition=_pending_addedinfo)
 284         _pending_overrideinfo = None
 285         _pending_addedinfo = None

 318 def run(*, filebuf, fileref, fileno, startstate, waitwml=True):
 319     global _states
 320     global _current_lineno
 321     global _linenosub
 322     global _waitwml
 323     global _currentdomain
 324     global _dictionary
 325     global _pending_luafuncname
 326     global _on_luatag
 327     _pending_luafuncname = None
 328     _on_luatag = False
 329     # cs is "current state"
 330     cs = _states.get(startstate)
 331     cs_debug = startstate
 332     _current_lineno = 0
 333     _linenosub = 0
 334     _waitwml = waitwml
 335     _currentdomain = _initialdomain
 336     pywmlx.nodemanip.newfile(fileref, fileno)
 337     # debug_cs = startstate
 338     for xline in filebuf:
 339         xline = xline.strip('\n\r')
 340         _current_lineno += 1
 341         # on new line, debug file will write another marker
 342         if _debugmode:
 343             print('@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@',
 344                   file=_fdebug)
 345         while xline is not None:
 346             # print debug infos (if debugmode is on)
 347             if _debugmode:
 348                 lno = '%05d' % _current_lineno
 349                 print('------------------------------------------------------',
 350                       file=_fdebug)
 351                 print('LINE', lno, '|', xline, file=_fdebug)
 352             # action number is used to know what function we should run    
 353             action = 0
 354             v = None
 355             m = None
 356             if cs.regex is None:
 357                 # action = 1 --> execute state.run
 358                 action = 1
 359                 if _debugmode:
 360                     print('ALWAYS-RUN x', cs_debug, file=_fdebug)
 361             else:
 362                 # m is match
 363                 m = re.match(cs.regex, xline)
 364                 if m:
 365                     # action = 1 --> execute state.run
 366                     action = 1
 367                     if _debugmode:
 368                         print('RUN state  \\', cs_debug, file=_fdebug)
 369                 else:
 370                     # action = 2 --> change to the state pointed by 
 371                     #                state.iffail
 372                     action = 2
 373                     if _debugmode:
 374                         print('FAIL state |', cs_debug, file=_fdebug)
 375             if action == 1:
 376                 # xline, ns: xline --> override xline with new value
 377                 #            ns --> value of next state
 378                 xline, ns = cs.run(xline, _current_lineno, m)
 379                 cs_debug = ns
 380                 cs = _states.get(ns)
 381             else:
 382                 cs_debug = cs.iffail
 383                 cs = _states.get(cs.iffail)
 384         # end while xline
 385     # end for xline
 386     pywmlx.nodemanip.closefile(_dictionary, _current_lineno)
