  42 #pragma once
  43 
  44 #include <array>
  45 #include <cstdint>
  46 
  47 class MD5 {
  48 
  49 public:
  50 // methods for controlled operation:
  51   MD5              ();  // simple initializer
  52   void  update     (const uint8_t* input, const uint32_t input_length);
  53   void  finalize   ();
  54 
  55 // methods to acquire finalized result
  56   std::array<uint8_t, 16> raw_digest ();  // digest as a 16-byte binary array
  57 
  58 
  59 private:
  60 
  61 // first, some types:
  62   typedef uint32_t uint4;
  63   typedef uint16_t uint2;
  64   typedef uint8_t  uint1;
  65 
  66 // next, the private data:
  67   uint4 state[4];
  68   uint4 count[2];     // number of *bits*, mod 2^64
  69   uint1 buffer[64];   // input buffer
  70   uint1 digest[16];
  71   uint1 finalized;
  72 
  73 // last, the private methods, mostly static:
  74   void init             ();                 // called by all constructors
  75   void transform        (const uint1 buffer[64]); // does the real update work.  Note
  76                                                   // that length is implied to be 64.
  77 
  78   static void encode    (uint1 *dest, uint4 *src, uint4 length);
  79   static void decode    (uint4 *dest, const uint1 *src, uint4 length);
  80 
  81   static inline uint4  rotate_left (uint4 x, uint4 n);
  82   static inline uint4  F           (uint4 x, uint4 y, uint4 z);
  83   static inline uint4  G           (uint4 x, uint4 y, uint4 z);
  84   static inline uint4  H           (uint4 x, uint4 y, uint4 z);
  85   static inline uint4  I           (uint4 x, uint4 y, uint4 z);
  86   static inline void   FF  (uint4& a, uint4 b, uint4 c, uint4 d, uint4 x,
  87                             uint4 s, uint4 ac);
  88   static inline void   GG  (uint4& a, uint4 b, uint4 c, uint4 d, uint4 x,
  89                             uint4 s, uint4 ac);
  90   static inline void   HH  (uint4& a, uint4 b, uint4 c, uint4 d, uint4 x,
  91                             uint4 s, uint4 ac);
  92   static inline void   II  (uint4& a, uint4 b, uint4 c, uint4 d, uint4 x,
  93                             uint4 s, uint4 ac);
  94 
  95 };
