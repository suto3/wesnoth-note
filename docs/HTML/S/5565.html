  28 #
  29 # This test should be failing although you might think it succeeds.
  30 # The test asserts that if filter_location for a menu_item is false,
  31 # it should return false if wesnoth attempts to fire it. However this
  32 # isn't the case, filter_location is normally checked before the
  33 # attempt to fire it.
  34 #
  35 {GENERIC_UNIT_TEST "test_wml_menu_items_2" (
  36     [event]
  37         name = start
  38 
  39         [set_menu_item]
  40             id=test2
  41             description="spawn a unit"
  42             [filter_location]
  43                 [not]
  44                     [have_unit]
  45                         x=$x1
  46                         y=$y1
  47                     [/have_unit]
  48                 [/not]
  49             [/filter_location]
  50             [command]
  51                 {UNIT 2 "Orcish Grunt" $x1 $y1 ()}
  52             [/command]
  53         [/set_menu_item]
  54 
  55         [lua]
  56             code = << r = wesnoth.fire_wml_menu_item("test2", 3, 3)
  57                                   r = r and not wesnoth.fire_wml_menu_item("test2", 3, 3)
  58                                   r = r and wesnoth.fire_wml_menu_item("test2", 4, 4)
  59                                   r = r and not wesnoth.fire_wml_menu_item("test2", 4, 4)
  60                                   wesnoth.set_variable("result", r) >>
  61         [/lua]
  62 
  63         {RETURN {VARIABLE_CONDITIONAL result boolean_equals true}}
  64     [/event]
