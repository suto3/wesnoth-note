  67 inline std::string concat_path(const std::string& a, const std::string& b)
  68 {
  69         //
  70         // As of Boost 1.61, normalize_path() displays unusual behavior when passing
  71         // it paths with extra path separators (e.g. //opt becomes
  72         // //opt/home/shadowm/src/wesnoth, where the extraneous sequence is probably
  73         // the current working dir), so avoid leaving those around.
  74         //
  75         // TODO: Maybe handle this corner case in filesystem::normalize_path()
  76         //       instead, really.
  77         //
  78         if((a.empty() || !fs::is_path_sep(a.back())) && (b.empty() || !fs::is_path_sep(b.front()))) {
  79                 return a + fs::path_separator() + b;
  80         } else {
  81                 return a + b;
  82         }
  83 }

  85 inline std::string filesystem_root()
  86 {
  87         // TODO: Multiple drives support (may require cooperation from the caller).
  88         return std::string(1, fs::path_separator());
  89 }

  91 inline void isort_dir_entries(std::vector<std::string>& entries)
  92 {
  93         // Yes, this uses Wesnoth's locale and not the filesystem/OS locale. Yes, this
  94         // isn't ideal. No, we don't really need to worry about it. It's just a
  95         // cosmetic procedure anyway.
  96         std::sort(entries.begin(), entries.end(),
  97                           [](const std::string& a, const std::string& b) { return translation::icompare(a, b) < 0; });
  98 }

 127 std::string file_dialog::path() const
 128 {
 129         const std::string& dir_norm = fs::normalize_path(current_dir_, true);
 130 
 131         if(current_entry_.empty() || current_entry_ == CURRENT_DIR) {
 132                 return dir_norm;
 133         } else if(current_entry_ == PARENT_DIR) {
 134                 return fs::directory_name(dir_norm);
 135         }
 136 
 137         return concat_path(dir_norm, current_entry_);
 138 }

 140 file_dialog& file_dialog::set_path(const std::string& value)
 141 {
 142         if(value.empty()) {
 143                 current_dir_ = filesystem_root();
 144         }
 145 
 146         const std::string& norm = fs::normalize_path(value, true);
 147 
 148         if(fs::is_directory(norm)) {
 149                 current_dir_ = norm;
 150         } else {
 151                 current_dir_ = fs::nearest_extant_parent(norm);
 152                 if(current_dir_.empty()) {
 153                         current_dir_ = filesystem_root();
 154                 }
 155                 // The file may or may not exist. We'll find out eventually when setting up
 156                 // the dialog.
 157                 current_entry_ = fs::base_name(norm);
 158         }
 159 
 160         return *this;
 161 }

 279 bool file_dialog::on_exit(window& window)
 280 {
 281         if(window.get_retval() == FILE_DIALOG_ITEM_RETVAL) {
 282                 // Attempting to exit by double clicking items -- only proceeds if the item
 283                 // was a file.
 284                 if(process_fileview_submit(window)) {
 285                         window.set_retval(window::OK, false);
 286                         return true;
 287                 } else {
 288                         return false;
 289                 }
 290         }
 291 
 292 
 293         if(window.get_retval() == window::OK) {
 294                 // Attempting to exit by pressing Enter/clicking OK -- only proceeds if the
 295                 // textbox was not altered by the user to point to a different directory.
 296                 return process_textbox_submit(window);
 297         }
 298 
 299         return true;
 300 }

 322 bool file_dialog::process_submit_common(window& window, const std::string& name)
 323 {
 324         const auto stype = register_new_selection(name);
 325 
 326         //DBG_FILEDLG << "current_dir_=" << current_dir_ << "  current_entry_=" << current_entry_ << '\n';
 327 
 328         if(is_selection_type_acceptable(stype)) {
 329                 return save_mode_ ? confirm_overwrite(window, stype) : true;
 330         }
 331 
 332         switch(stype) {
 333                 case SELECTION_IS_DIR:
 334                         // TODO: Adapt for implementing directory selection mode.
 335                         sync_bookmarks_bar(window);
 336                         refresh_fileview(window);
 337                         break;
 338                 case SELECTION_PARENT_NOT_FOUND:
 339                         // We get here in save mode or not. Use the file creation language only in
 340                         // save mode.
 341                         if(save_mode_) {
 342                                 show_transient_error_message(window.video(), vgettext("The file or folder $path cannot be created.", {{"path", name}}));
 343                                 break;
 344                         }
 345                         FALLTHROUGH;
 346                 case SELECTION_NOT_FOUND:
 347                         // We only get here if we aren't in save mode.
 348                         show_transient_error_message(window.video(), vgettext("The file or folder $path does not exist.", {{"path", name}}));
 349                         break;
 350                 case SELECTION_IS_FILE:
 351                         // TODO: Adapt for implementing directory selection mode.
 352                 default:
 353                         assert(false && "Unimplemented selection mode or semantics");
 354         }
 355 
 356         return false;
 357 }

 373 std::string file_dialog::get_filelist_selection(listbox& filelist)
 374 {
 375         const int row = filelist.get_selected_row();
 376 
 377         if(row == -1) {
 378                 // Shouldn't happen...
 379                 return "";
 380         }
 381 
 382         const bool i_am_root = fs::is_root(current_dir_);
 383 
 384         if(row == 0 && !i_am_root) {
 385                 return PARENT_DIR;
 386         } else {
 387                 size_t n = i_am_root ? row : row - 1;
 388 
 389                 if(n < dir_subdirs_.size()) {
 390                         return dir_subdirs_[n];
 391                 } else {
 392                         n -= dir_subdirs_.size();
 393 
 394                         if(n < dir_files_.size()) {
 395                                 return dir_files_[n];
 396                         } else {
 397                                 assert(false && "File list selection is out of range!");
 398                         }
 399                 }
 400         }
 401 
 402         return "";
 403 }

 405 file_dialog::SELECTION_TYPE file_dialog::register_new_selection(const std::string& name)
 406 {
 407         std::string new_path, new_parent;
 408 
 409         if(fs::is_relative(name)) {
 410                 // On Windows, \ represents a path relative to the root of the process'
 411                 // current working drive specified by the current working dir, so we get
 412                 // here. This makes it the only platform where is_relative() and is_root()
 413                 // aren't mutually exclusive.
 414                 if(fs::is_root(name)) {
 415                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is relative to a root resource\n";
 416                         // Using the browsed dir's root drive instead of the cwd's makes the most
 417                         // sense for users.
 418                         new_parent = fs::root_name(current_dir_);
 419                         new_path = fs::normalize_path(concat_path(new_parent, name), true, true);
 420                 } else {
 421                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' seems relative\n";
 422                         new_parent = current_dir_;
 423                         new_path = fs::normalize_path(concat_path(current_dir_, name), true, true);
 424                 }
 425         } else {
 426                 DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' seems absolute\n";
 427                 new_parent = fs::directory_name(name);
 428                 new_path = fs::normalize_path(name, true, true);
 429                 DBG_FILEDLG << "register_new_selection(): new selection is " << new_path << '\n';
 430         }
 431 
 432         if(!new_path.empty()) {
 433                 if(fs::is_directory(new_path)) {
 434                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is a directory: " << new_path << '\n';
 435                         current_dir_ = new_path;
 436                         current_entry_.clear();
 437                         return SELECTION_IS_DIR;
 438                 } else if(fs::file_exists(new_path)) {
 439                         // FIXME: Perhaps redundant since the three-params call to normalize_path()
 440                         //        above necessarily validates existence.
 441                         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' is a file, symbolic link, or special: " << new_path << '\n';
 442                         current_dir_ = fs::directory_name(new_path);
 443                         current_entry_ = fs::base_name(new_path);
 444                         return SELECTION_IS_FILE;
 445                 }
 446         }
 447 
 448         // The path does not exist, at least not entirely. See if the parent does
 449         // (in save mode non-existent files are accepted as long as the parent dir
 450         // exists).
 451         const std::string& absolute_parent = fs::normalize_path(new_parent, true, true);
 452         if(!absolute_parent.empty()) {
 453                 DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' does not exist or is not accessible, but parent exists\n";
 454                 current_dir_ = absolute_parent;
 455                 current_entry_ = fs::base_name(name);
 456                 return SELECTION_NOT_FOUND;
 457         }
 458 
 459         DBG_FILEDLG << "register_new_selection(): new selection '" << name << "' does not exist or is not accessible\n";
 460         return SELECTION_PARENT_NOT_FOUND;
 461 }

 463 void file_dialog::set_input_text(text_box& t, const std::string& value)
 464 {
 465         if(value.empty()) {
 466                 clear_input_text(t);
 467                 return;
 468         }
 469 
 470         t.set_value(value);
 471 
 472         const size_t vallen = t.get_length();
 473         const size_t extlen = utf8::size(extension_);
 474 
 475         if(save_mode_ && extlen && vallen > extlen) {
 476                 // Highlight everything but the extension if it matches
 477                 if(value.substr(vallen - extlen) == extension_) {
 478                         t.set_selection(0, vallen - extlen);
 479                 }
 480         }
 481 }

 493 void file_dialog::refresh_fileview(window& window)
 494 {
 495         cursor::setter cur{cursor::WAIT};
 496 
 497         dir_files_.clear();
 498         dir_subdirs_.clear();
 499 
 500         // TODO: Need to detect and handle cases where we don't have search permission
 501         //       on current_dir_, otherwise things may get weird.
 502         filesystem::get_files_in_dir(current_dir_, &dir_files_, &dir_subdirs_, filesystem::FILE_NAME_ONLY);
 503         isort_dir_entries(dir_files_);
 504         isort_dir_entries(dir_subdirs_);
 505 
 506         //
 507         // Clear and refill the filelist box.
 508         //
 509 
 510         listbox& filelist = find_widget<listbox>(&window, "filelist", false);
 511         button& rm_button = find_widget<button>(&window, "delete_file", false);
 512 
 513         filelist.clear();
 514 
 515         // Parent entry
 516         if(!fs::is_root(current_dir_)) {
 517                 // label_parent may not necessarily be always ".." in the future, so push
 518                 // with check_selection = false and check the selection ourselves here.
 519                 push_fileview_row(filelist, label_parent, icon_parent, false);
 520                 if(current_entry_ == PARENT_DIR || current_entry_.empty()) {
 521                         filelist.select_row(0, true);
 522                         rm_button.set_active(false);
 523                 } else {
 524                         rm_button.set_active(true);
 525                 }
 526         }
 527 
 528         for(const auto& dir : dir_subdirs_) {
 529                 push_fileview_row(filelist, dir, icon_dir);
 530         }
 531 
 532         for(const auto& file : dir_files_) {
 533                 push_fileview_row(filelist, file, icon_file);
 534         }
 535 
 536         find_widget<styled_widget>(&window, "current_dir", false).set_label(current_dir_);
 537         set_input_text(find_widget<text_box>(&window, "filename", false), current_entry_);
 538 
 539         on_row_selected(window);
 540 }

 542 void file_dialog::push_fileview_row(listbox& filelist, const std::string& name, const std::string& icon, bool check_selection)
 543 {
 544         // TODO: Hopefully some day GUI2 will allow us to make labels be ellipsized
 545         //       dynamically at layout/rendering time.
 546         std::string label = name;
 547         utils::ellipsis_truncate(label, FILE_DIALOG_MAX_ENTRY_LENGTH);
 548 
 549         std::map<std::string, string_map> data;
 550         data["icon"]["label"] = icon;
 551         data["file"]["label"] = label;
 552 
 553         grid& last_grid = filelist.add_row(data);
 554         const unsigned last_pos = filelist.get_item_count() - 1;
 555 
 556         //
 557         // Crummy hack around the lack of an option to hook into row double click
 558         // events for all rows using the GUI2 listbox API. Assign a special retval to
 559         // each row that triggers a special check during dialog exit.
 560         //
 561         find_widget<toggle_panel>(&last_grid, "item_panel", false)
 562                         .set_retval(FILE_DIALOG_ITEM_RETVAL);
 563 
 564         if(check_selection && name == current_entry_) {
 565                 filelist.select_row(last_pos, true);
 566         }
 567 }

 569 void file_dialog::sync_bookmarks_bar(window& window)
 570 {
 571         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 572 
 573         // Internal state has normalized path delimiters but dot entries aren't
 574         // resolved after callers call set_path(), so compare against a canonical
 575         // version. The bookmark paths are already canonical, though.
 576         const std::string& canon_current_dir = fs::normalize_path(current_dir_, true, true);
 577 
 578         // Go backwards so we can match user-defined bookmarks first (otherwise it may
 579         // become impossible for the user to delete them if they match any of the
 580         // predefined paths).
 581         auto it = std::find(bookmark_paths_.rbegin(), bookmark_paths_.rend(), canon_current_dir);
 582 
 583         if(it == bookmark_paths_.rend()) {
 584                 if(current_bookmark_ >= 0) {
 585                         bookmarks_bar.select_row(unsigned(current_bookmark_), false);
 586                 }
 587                 current_bookmark_ = -1;
 588         } else {
 589                 const int new_selection = int(std::distance(bookmark_paths_.begin(), it.base()) - 1);
 590                 if(new_selection != current_bookmark_) {
 591                         assert(unsigned(new_selection) < bookmarks_bar.get_item_count());
 592                         if(current_bookmark_ >= 0) {
 593                                 bookmarks_bar.select_row(unsigned(current_bookmark_), false);
 594                         }
 595                         bookmarks_bar.select_row(unsigned(new_selection), true);
 596                         current_bookmark_ = new_selection;
 597                 }
 598         }
 599 
 600         // Update bookmark edit controls.
 601         button& del_button = find_widget<button>(&window, "remove_bookmark", false);
 602 
 603         if(user_bookmarks_begin_ == -1) {
 604                 del_button.set_active(false);
 605         } else {
 606                 del_button.set_active(current_bookmark_ >= user_bookmarks_begin_);
 607         }

 610 void file_dialog::on_row_selected(window& window)
 611 {
 612         listbox& filelist = find_widget<listbox>(&window, "filelist", false);
 613         text_box& file_textbox = find_widget<text_box>(&window, "filename", false);
 614         button& rm_button = find_widget<button>(&window, "delete_file", false);
 615 
 616         // Don't use register_new_selection() here, we don't want any parsing to be
 617         // performed at this point.
 618         current_entry_ = get_filelist_selection(filelist);
 619 
 620         // Clear the textbox when selecting ..
 621         if(current_entry_ != PARENT_DIR) {
 622                 set_input_text(file_textbox, current_entry_);
 623                 rm_button.set_active(true);
 624         } else {
 625                 clear_input_text(file_textbox);
 626                 rm_button.set_active(false);
 627         }
 628 
 629         // Need to do this every time so that input can still be sent to the
 630         // textbox without clicking on it.
 631         window.keyboard_capture(&file_textbox);
 632 }

 634 void file_dialog::on_bookmark_selected(window& window)
 635 {
 636         // Don't let us steal the focus from the primary widgets.
 637         text_box& file_textbox = find_widget<text_box>(&window, "filename", false);
 638         window.keyboard_capture(&file_textbox);
 639 
 640         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 641         const int new_selection = bookmarks_bar.get_selected_row();
 642 
 643         if(new_selection < 0) {
 644                 if(current_bookmark_ >= 0) {
 645                         // Don't allow the user to unselect the selected bookmark. That wouldn't
 646                         // make any sense.
 647                         bookmarks_bar.select_row(unsigned(current_bookmark_));
 648                 }
 649 
 650                 return;
 651         }
 652 
 653         assert(unsigned(new_selection) < bookmark_paths_.size());
 654         current_bookmark_ = new_selection;
 655         set_path(bookmark_paths_[new_selection]);
 656         refresh_fileview(window);
 657 
 658         // Update bookmark edit controls.
 659         button& del_button = find_widget<button>(&window, "remove_bookmark", false);
 660         del_button.set_active(user_bookmarks_begin_ >= 0
 661                                                   && current_bookmark_ >= user_bookmarks_begin_);
 662 }

 664 void file_dialog::on_bookmark_add_cmd(window& window)
 665 {
 666         const std::string& default_label = fs::base_name(current_dir_);
 667 
 668         std::string label = default_label;
 669 
 670         const bool confirm = bookmark_create::execute(label, window.video());
 671         if(!confirm) {
 672                 return;
 673         }
 674 
 675         if(label.empty()) {
 676                 label = default_label;
 677         }
 678 
 679         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 680 
 681         desktop::add_user_bookmark(label, current_dir_);
 682         bookmark_paths_.push_back(current_dir_);
 683         const unsigned top_bookmark = bookmark_paths_.size() - 1;
 684 
 685         if(user_bookmarks_begin_ == -1) {
 686                 user_bookmarks_begin_ = top_bookmark;
 687         }
 688 
 689         std::map<std::string, string_map> data;
 690         data["bookmark"]["label"] = label;
 691         bookmarks_bar.add_row(data);
 692 
 693         current_bookmark_ = -1;
 694 
 695         sync_bookmarks_bar(window);
 696 }

 698 void file_dialog::on_bookmark_del_cmd(window& window)
 699 {
 700         assert(user_bookmarks_begin_ >= 0
 701                    && current_bookmark_ >= 0
 702                    && current_bookmark_ >= user_bookmarks_begin_
 703                    && current_bookmark_ < int(bookmark_paths_.size()));
 704 
 705         listbox& bookmarks_bar = find_widget<listbox>(&window, "bookmarks", false);
 706         desktop::remove_user_bookmark(current_bookmark_ - user_bookmarks_begin_);
 707         bookmark_paths_.erase(bookmark_paths_.begin() + current_bookmark_);
 708         bookmarks_bar.remove_row(current_bookmark_);
 709 
 710         current_bookmark_ = -1;
 711 
 712         sync_bookmarks_bar(window);
 713 }

 732 void file_dialog::on_file_delete_cmd(window& window)
 733 {
 734         if(current_entry_.empty()) {
 735                 return;
 736         }
 737 
 738         const std::string& selection = concat_path(current_dir_, current_entry_);
 739         const bool is_dir = fs::is_directory(selection);
 740 
 741         const std::string& message = (is_dir
 742                         ? _("The following folder and its contents will be permanently deleted:")
 743                         : _("The following file will be permanently deleted:"))
 744                         + "\n\n" + selection + "\n\n" + _("Do you wish to continue?");
 745 
 746         if(gui2::show_message(window.video(), _("Confirm"), message, message::yes_no_buttons) == gui2::window::CANCEL) {
 747                 return;
 748         }
 749 
 750         const bool result = is_dir
 751                         ? fs::delete_directory(selection)
 752                         : fs::delete_file(selection);
 753 
 754         if(!result) {
 755                 show_transient_error_message(window.video(),
 756                                 vgettext("Could not delete $path|. Make sure you have the appropriate permissions to write to this location.",
 757                                                  {{"path", selection}}));
 758         } else {
 759                 refresh_fileview(window);
 760         }
 761 }
