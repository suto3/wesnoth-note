  78 void story_viewer::pre_show(window& window)
  79 {
  80         window.set_enter_disabled(true);
  81 
  82         // Special callback handle key presses
  83         connect_signal_pre_key_press(window, std::bind(&story_viewer::key_press_callback, this, std::ref(window), _5));
  84 
  85         connect_signal_mouse_left_click(find_widget<button>(&window, "next", false),
  86                 std::bind(&story_viewer::nav_button_callback, this, std::ref(window), DIR_FORWARD));
  87 
  88         connect_signal_mouse_left_click(find_widget<button>(&window, "back", false),
  89                 std::bind(&story_viewer::nav_button_callback, this, std::ref(window), DIR_BACKWARDS));

 102 void story_viewer::display_part(window& window)
 103 {
 104         static const int VOICE_SOUND_SOURCE_ID = 255;
 105         // Update Back button state. Doing this here so it gets called in pre_show too.
 106         find_widget<button>(&window, "back", false).set_active(part_index_ != 0);
 107 
 108         //
 109         // Music and sound
 110         //
 111         if(!current_part_->music().empty()) {
 112                 config music_config;
 113                 music_config["name"] = current_part_->music();
 114                 music_config["ms_after"] = 2000;
 115                 music_config["immediate"] = true;
 116 
 117                 sound::play_music_config(music_config);
 118         }
 119 
 120         if(!current_part_->sound().empty()) {
 121                 sound::play_sound(current_part_->sound());
 122         }
 123 
 124         sound::stop_sound(VOICE_SOUND_SOURCE_ID);
 125         if(!current_part_->voice().empty()) {
 126                 sound::play_sound_positioned(current_part_->voice(), VOICE_SOUND_SOURCE_ID, 0, 0);
 127         }
 128 
 129         config cfg, image;
 130 
 131         //
 132         // Background images
 133         //
 134         bool has_background = false;
 135         config* base_layer = nullptr;
 136 
 137         for(const auto& layer : current_part_->get_background_layers()) {
 138                 has_background |= !layer.file().empty();
 139 
 140                 const bool preserve_ratio = layer.keep_aspect_ratio();
 141 
 142                 // By default, no scaling will be applied.
 143                 std::string width_formula  = "(image_original_width)";
 144                 std::string height_formula = "(image_original_height)";
 145 
 146                 if(layer.scale_horizontally() && preserve_ratio) {
 147                         height_formula = "(min((image_original_height * width  / image_original_width), height))";
 148                 } else if(layer.scale_vertically()) {
 149                         height_formula = "(height)";
 150                 }
 151 
 152                 if(layer.scale_vertically() && preserve_ratio) {
 153                         width_formula  = "(min((image_original_width  * height / image_original_height), width))";
 154                 } else if(layer.scale_horizontally()) {
 155                         width_formula  = "(width)";
 156                 }
 157 
 158                 // Background layers are always centered.
 159                 image["x"] = "(max(pos, 0) where pos = (width  / 2 - image_width  / 2))";
 160                 image["y"] = "(max(pos, 0) where pos = (height / 2 - image_height / 2))";
 161                 image["w"] = width_formula;
 162                 image["h"] = height_formula;
 163                 image["name"] = layer.file();
 164 
 165                 config& layer_image = cfg.add_child("image", image);
 166 
 167                 if(base_layer == nullptr || layer.is_base_layer()) {
 168                         base_layer = &layer_image;
 169                 }
 170         }
 171 
 172         canvas& window_canvas = window.get_canvas(0);
 173 
 174         /* In order to avoid manually loading the image and calculating the scaling factor, we instead
 175          * delegate the task of setting the necessary variables to the canvas once the calculations
 176          * have been made internally.
 177          *
 178          * This sets the necessary values with the data for "this" image when its drawn. If no base
 179          * layer was found (which would be the case if no backgrounds were provided at all), simply set
 180          * some sane defaults directly.
 181          */
 182         if(base_layer != nullptr) {
 183                 (*base_layer)["actions"] = R"((
 184                         [
 185                                 set_var('base_scale_x', as_decimal(image_width)  / as_decimal(image_original_width)),
 186                                 set_var('base_scale_y', as_decimal(image_height) / as_decimal(image_original_height)),
 187                                 set_var('base_origin', loc(clip_x, clip_y))
 188                         ]
 189                 ))";
 190         } else {
 191                 window_canvas.set_variable("base_scale_x", wfl::variant(1));
 192                 window_canvas.set_variable("base_scale_y", wfl::variant(1));
 193                 window_canvas.set_variable("base_origin",  wfl::variant(std::make_shared<wfl::location_callable>(map_location::ZERO())));
 194         }
 195 
 196         cfg.add_child("image", get_title_area_decor_config());
 197 
 198         window_canvas.set_cfg(cfg);
 199 
 200         // Needed to make the background redraw correctly.
 201         window_canvas.set_is_dirty(true);
 202         window.set_is_dirty(true);
 203 
 204         //
 205         // Title
 206         //
 207         label& title_label = find_widget<label>(&window, "title", false);
 208 
 209         std::string title_text = current_part_->title();
 210         bool showing_title;
 211 
 212         if(current_part_->show_title() && !title_text.empty()) {
 213                 showing_title = true;
 214 
 215                 PangoAlignment title_text_alignment = decode_text_alignment(current_part_->title_text_alignment());
 216 
 217                 title_label.set_visible(widget::visibility::visible);
 218                 title_label.set_text_alignment(title_text_alignment);
 219                 title_label.set_label(title_text);
 220         } else {
 221                 showing_title = false;
 222 
 223                 title_label.set_visible(widget::visibility::invisible);
 224         }
 225 
 226         //
 227         // Story text
 228         //
 229         stacked_widget& text_stack = find_widget<stacked_widget>(&window, "text_and_control_stack", false);
 230 
 231         std::string new_panel_mode;
 232 
 233         switch(current_part_->story_text_location()) {
 234                 case storyscreen::part::BLOCK_TOP:
 235                         new_panel_mode = "top";
 236                         break;
 237                 case storyscreen::part::BLOCK_MIDDLE:
 238                         new_panel_mode = "center";
 239                         break;
 240                 case storyscreen::part::BLOCK_BOTTOM:
 241                         new_panel_mode = "bottom";
 242                         break;
 243         }
 244 
 245         text_stack.set_vertical_alignment(new_panel_mode);
 246 
 247         /* Set the panel mode control variables.
 248          *
 249          * We use get_layer_grid here to ensure the widget is always found regardless of
 250          * whether the background is visible or not.
 251          */
 252         canvas& panel_canvas = find_widget<panel>(text_stack.get_layer_grid(LAYER_BACKGROUND), "text_panel", false).get_canvas(0);
 253 
 254         panel_canvas.set_variable("panel_position", wfl::variant(new_panel_mode));
 255         panel_canvas.set_variable("title_present", wfl::variant(static_cast<int>(showing_title))); // cast to 0/1
 256 
 257         const std::string& part_text = current_part_->text();
 258 
 259         if(part_text.empty() || !has_background) {
 260                 // No text or no background for this part, hide the background layer.
 261                 text_stack.select_layer(LAYER_TEXT);
 262         } else if(text_stack.current_layer() != -1)  {
 263                 // If the background layer was previously hidden, re-show it.
 264                 text_stack.select_layer(-1);
 265         }
 266 
 267         // Convert the story part text alignment types into the Pango equivalents
 268         PangoAlignment story_text_alignment = decode_text_alignment(current_part_->story_text_alignment());
 269 
 270         scroll_label& text_label = find_widget<scroll_label>(&window, "part_text", false);
 271 
 272         text_label.set_text_alignment(story_text_alignment);
 273         text_label.set_text_alpha(0);
 274         text_label.set_label(part_text);
 275 
 276         begin_fade_draw(true);
 277 
 278         //
 279         // Floating images (handle this last)
 280         //
 281         const auto& floating_images = current_part_->get_floating_images();
 282 
 283         // If we have images to draw, draw the first one now. A new non-repeating timer is added
 284         // after every draw to schedule the next one after the specified interval.
 285         //
 286         // TODO: in the old GUI1 dialog, floating images delayed the appearance of the story panel until
 287         //       drawing was finished. Might be worth looking into restoring that.
 288         if(!floating_images.empty()) {
 289                 draw_floating_image(window, floating_images.begin(), part_index_);
 290         }
 291 }

 293 void story_viewer::draw_floating_image(window& window, floating_image_list::const_iterator image_iter, int this_part_index)
 294 {
 295         const auto& images = current_part_->get_floating_images();
 296 
 297         // If the current part has changed or we're out of images to draw, exit the draw loop.
 298         if((this_part_index != part_index_) || (image_iter == images.end())) {
 299                 timer_id_ = 0;
 300                 return;
 301         }
 302 
 303         const auto& floating_image = *image_iter;
 304 
 305         std::ostringstream x_ss;
 306         std::ostringstream y_ss;
 307 
 308         // Floating images are scaled by the same factor as the background.
 309         x_ss << "(trunc(fi_ref_x * base_scale_x) + base_origin.x";
 310         y_ss << "(trunc(fi_ref_y * base_scale_y) + base_origin.y";
 311 
 312         if(floating_image.centered()) {
 313                 x_ss << " - (image_original_width  / 2)";
 314                 y_ss << " - (image_original_height / 2)";
 315         }
 316 
 317         x_ss << " where fi_ref_x = " << floating_image.ref_x() << ")";
 318         y_ss << " where fi_ref_y = " << floating_image.ref_y() << ")";
 319 
 320         config cfg, image;
 321 
 322         image["x"] = x_ss.str();
 323         image["y"] = y_ss.str();
 324         image["w"] = floating_image.autoscale() ? "(width)"  : "(image_width)";
 325         image["h"] = floating_image.autoscale() ? "(height)" : "(image_height)";
 326         image["name"] = floating_image.file();
 327 
 328         // TODO: implement handling of the tiling options.
 329         //image["resize_mode"] = "tile_centered"
 330 
 331         cfg.add_child("image", image);
 332 
 333         canvas& window_canvas = window.get_canvas(0);
 334 
 335         // Needed to make the background redraw correctly.
 336         window_canvas.append_cfg(cfg);
 337         window_canvas.set_is_dirty(true);
 338 
 339         window.set_is_dirty(true);
 340 
 341         ++image_iter;
 342 
 343         // If a delay is specified, schedule the next image draw. This *must* be a non-repeating timer!
 344         // Else draw the next image immediately.
 345         const unsigned int draw_delay = floating_image.display_delay();
 346 
 347         if(draw_delay != 0) {
 348                 timer_id_ = add_timer(draw_delay,
 349                         std::bind(&story_viewer::draw_floating_image, this, std::ref(window), image_iter, this_part_index), false);
 350         } else {
 351                 draw_floating_image(window, image_iter, this_part_index);
 352         }
 353 }

 355 void story_viewer::nav_button_callback(window& window, NAV_DIRECTION direction)
 356 {
 357         // If a button is pressed while fading in, abort and set alpha to full opaque.
 358         if(fade_state_ == FADING_IN) {
 359                 halt_fade_draw();
 360 
 361                 // Only set full alpha if Forward was pressed.
 362                 if(direction == DIR_FORWARD) {
 363                         find_widget<scroll_label>(&window, "part_text", false).set_text_alpha(ALPHA_OPAQUE);
 364                         flag_stack_as_dirty(window);
 365                         return;
 366                 }
 367         }
 368 
 369         // If a button is pressed while fading out, skip and show next part.
 370         if(fade_state_ == FADING_OUT) {
 371                 display_part(window);
 372                 return;
 373         }
 374 
 375         assert(fade_state_ == NOT_FADING);
 376 
 377         part_index_ = (direction == DIR_FORWARD ? part_index_ + 1 : part_index_ -1);
 378 
 379         // If we've viewed all the parts, close the dialog.
 380         if(part_index_ >= controller_.max_parts()) {
 381                 window.close();
 382                 return;
 383         }
 384 
 385         if(part_index_ < 0) {
 386                 part_index_ = 0;
 387         }
 388 
 389         update_current_part_ptr();
 390 
 391         begin_fade_draw(false);
 392 }

 394 void story_viewer::key_press_callback(window& window, const SDL_Keycode key)
 395 {
 396         const bool next_keydown =
 397                    key == SDLK_SPACE
 398                 || key == SDLK_RETURN
 399                 || key == SDLK_KP_ENTER
 400                 || key == SDLK_RIGHT;
 401 
 402         const bool back_keydown =
 403                    key == SDLK_BACKSPACE
 404                 || key == SDLK_LEFT;
 405 
 406         if(next_keydown) {
 407                 nav_button_callback(window, DIR_FORWARD);
 408         } else if(back_keydown) {
 409                 nav_button_callback(window, DIR_BACKWARDS);
 410         }
 411 }

 418 void story_viewer::begin_fade_draw(bool fade_in)
 419 {
 420         set_next_draw();

 422         fade_step_ = fade_in ? 0 : 10;
 423         fade_state_ = fade_in ? FADING_IN : FADING_OUT;

 426 void story_viewer::halt_fade_draw()

 433 void story_viewer::draw_callback(window& window)
 434 {
 435         if(next_draw_ && SDL_GetTicks() < next_draw_) {
 436                 return;
 437         }
 438 
 439         if(fade_state_ == NOT_FADING) {
 440                 return;
 441         }
 442 
 443         // If we've faded fully in...
 444         if(fade_state_ == FADING_IN && fade_step_ > 10) {
 445                 halt_fade_draw();
 446                 return;
 447         }
 448 
 449         // If we've faded fully out...
 450         if(fade_state_ == FADING_OUT && fade_step_ < 0) {
 451                 halt_fade_draw();
 452 
 453                 display_part(window);
 454                 return;
 455         }
 456 
 457         unsigned short new_alpha = utils::clamp<short>(fade_step_ * 25.5, 0, ALPHA_OPAQUE);
 458         find_widget<scroll_label>(&window, "part_text", false).set_text_alpha(new_alpha);
 459 
 460         // The text stack also needs to be marked dirty so the background panel redraws correctly.
 461         flag_stack_as_dirty(window);
 462 
 463         if(fade_state_ == FADING_IN) {
 464                 fade_step_ ++;
 465         } else if(fade_state_ == FADING_OUT) {
 466                 fade_step_ --;
 467         }
 468 
 469         set_next_draw();
 470 }
