  70 std::pair<unit_map::unit_iterator, bool> unit_map::add(const map_location &l, const unit &u) {
  71         self_check();
  72         unit_ptr p = unit_ptr (new unit(u)); //TODO: should this instead take a shared pointer to a unit, rather than make a copy?
  73         p->set_location(l);
  74         std::pair<unit_map::unit_iterator, bool> res( insert(p) );

  79 std::pair<unit_map::unit_iterator, bool> unit_map::move(const map_location &src, const map_location &dst) {
  80         self_check();
  81         DBG_NG << "Unit map: Moving unit from " << src << " to " << dst << "\n";
  82 
  83         //Find the unit at the src location
  84         lmap::iterator i = lmap_.find(src);
  85         if(i == lmap_.end()) { return std::make_pair(make_unit_iterator(i), false);}
  86 
  87         umap::iterator uit(i->second);
  88 
  89         if(src == dst){ return std::make_pair(make_unit_iterator(uit), true);}
  90 
  91         //Fail if there is no unit to move
  92         unit_ptr p = uit->second.unit;
  93         if(!p){ return std::make_pair(make_unit_iterator(uit), false);}
  94 
  95         p->set_location(dst);
  96 
  97         lmap_.erase(i);
  98 
  99         std::pair<lmap::iterator,bool> res = lmap_.emplace(dst, uit);
 100 
 101         //Fail and don't move if the destination is already occupied
 102         if(res.second == false) {
 103                 p->set_location(src);
 104                 lmap_.emplace(src, uit);
 105                 return std::make_pair(make_unit_iterator(uit), false);
 106         }
 107 
 108         self_check();
 109 
 110         return std::make_pair(make_unit_iterator(uit), true);
 111 }

 114 /** Inserts the unit pointed to by @a p into the unit_map.
 115 
 116 It needs to succeed on the insertion to the umap and to the lmap
 117 otherwise all operations are reverted.
 118 1. Construct a unit_pod
 119 2. Try insertion into the umap
 120 3. Try insertion in the lmap and remove the umap entry on failure
 121 
 122 The one oddity is that to facilitate non-invalidating iterators the list
 123 sometimes has nullptr pointers which should be used when they correspond
 124 to uids previously used.
 125  */
 126 std::pair<unit_map::unit_iterator, bool> unit_map::insert(unit_ptr p) {
 127         self_check();
 128         assert(p);
 129 
 130         size_t unit_id = p->underlying_id();
 131         const map_location &loc = p->get_location();
 132 
 133         if (!loc.valid()) {
 134                 ERR_NG << "Trying to add " << p->name()
 135                         << " - " << p->id() << " at an invalid location; Discarding.\n";
 136                 return std::make_pair(make_unit_iterator(umap_.end()), false);
 137         }
 138 
 139         unit_pod upod;
 140         upod.unit = p ;
 141 
 142         DBG_NG << "Adding unit " << p->underlying_id() << " - " << p->id()
 143                 << " to location: (" << loc << ")\n";
 144 
 145         std::pair<umap::iterator, bool> uinsert = umap_.emplace(unit_id, upod);
 146 
 147         if (! uinsert.second) {
 148                 //If the pod is empty reinsert the unit in the same list element
 149                 if (!uinsert.first->second.unit) {
 150                         unit_pod &opod = uinsert.first->second;
 151                         opod.unit = p ;
 152                         assert(opod.ref_count != 0);
 153                 } else {
 154                         unit_ptr q = uinsert.first->second.unit;
 155                         ERR_NG << "Trying to add " << p->name()
 156                                    << " - " << p->id() << " - " << p->underlying_id()
 157                                    << " ("  << loc << ") over " << q->name()
 158                                    << " - " << q->id() << " - " << q->underlying_id()
 159                                    << " ("  << q->get_location()
 160                                    << ").\n";
 161 
 162                         p->clone(false);
 163                         ERR_NG << "The new unit was assigned underlying_id="
 164                                    << p->underlying_id()
 165                                    << " to prevent duplicate id conflicts.\n";
 166 
 167                         uinsert = umap_.emplace(p->underlying_id(), upod);
 168                         int guard(0);
 169                         while (!uinsert.second && (++guard < 1e6) ) {
 170                                 if(guard % 10 == 9){
 171                                         ERR_NG << "\n\nPlease Report this error to https://gna.org/bugs/index.php?18591 "
 172                                                 "\nIn addition to the standard details of operating system and wesnoth version "
 173                                                 "and how it happened, please answer the following questions "
 174                                                 "\n 1. Were you playing multi-player?"
 175                                                 "\n 2. Did you start/restart/reload the game/scenario?"
 176                                                 "\nThank you for your help in fixing this bug.\n";
 177                                 }
 178                                 p->clone(false);
 179                                 uinsert = umap_.emplace(p->underlying_id(), upod); }
 180                         if (!uinsert.second) {
 181                                 throw game::error("One million collisions in unit_map"); }
 182                 }
 183         }
 184 
 185         std::pair<lmap::iterator,bool> linsert = lmap_.emplace(loc, uinsert.first);
 186 
 187         //Fail if the location is occupied
 188         if(! linsert.second) {
 189                 if(upod.ref_count == 0) {
 190                         //Undo a virgin insertion
 191                         umap_.erase(uinsert.first);
 192                 } else {
 193                         //undo a reinsertion
 194                         uinsert.first->second.unit.reset();
 195                 }
 196                 DBG_NG << "Trying to add " << p->name()
 197                            << " - " << p->id() << " at location ("<<loc <<"); Occupied  by "
 198                            <<(linsert.first->second->second).unit->name()<< " - " << linsert.first->second->second.unit->id() <<"\n";
 199 
 200                 return std::make_pair(make_unit_iterator(umap_.end()), false);
 201         }
 202 
 203         self_check();
 204         return std::make_pair( make_unit_iterator( uinsert.first ), true);
 205 }

 216 size_t unit_map::num_iters() const  {
 217         ///Add up number of extant iterators
 218         size_t num_iters(0);
 219         umap::const_iterator ui = umap_.begin();
 220         umap::const_iterator uend = umap_.end();
 221         for( ; ui != uend ; ++ui){
 222                 if(ui->second.ref_count < 0) {
 223                         //Somewhere, someone generated 2^31 iterators to this unit
 224                         bool a_reference_counter_overflowed(false);
 225                         assert(a_reference_counter_overflowed);
 226                 }
 227                 num_iters += ui->second.ref_count;
 228         }
 229 
 230         return num_iters;
 231 }

 233 void unit_map::clear(bool force) {
 234         assert(force  || (num_iters() == 0));
 235 
 236         for (umap::iterator i = umap_.begin(); i != umap_.end(); ++i) {
 237                 if (is_valid(i)) {
 238                         DBG_NG << "Delete unit " << i->second.unit->underlying_id() << "\n";
 239                         i->second.unit.reset();
 240                 }
 241         }
 242 
 243         lmap_.clear();
 244         umap_.clear();

 247 unit_ptr unit_map::extract(const map_location &loc) {
 248         self_check();
 249         lmap::iterator i = lmap_.find(loc);
 250         if (i == lmap_.end()) { return unit_ptr(); }
 251 
 252         umap::iterator uit(i->second);
 253 
 254         unit_ptr u = uit->second.unit;
 255         size_t uid( u->underlying_id() );
 256 
 257         DBG_NG << "Extract unit " << uid << " - " << u->id()
 258                         << " from location: (" << loc << ")\n";
 259 
 260         assert(uit->first == uit->second.unit->underlying_id());
 261         if(uit->second.ref_count == 0){
 262                 umap_.erase(uit);
 263         } else {
 264                 //Soft extraction keeps the old lit item if any iterators reference it
 265                 uit->second.unit.reset();
 266         }
 267 
 268         lmap_.erase(i);
 269         self_check();
 270 
 271         return u;
 272 }

 275 size_t unit_map::erase(const map_location &loc) {
 276         self_check();
 277         unit_ptr u = extract(loc);
 278         if (!u) return 0;
 279         u.reset();
 280         return 1;

 304 unit_map::unit_iterator unit_map::find_first_leader(int side) {
 305         unit_map::iterator i = begin(), i_end = end();
 306         unit_map::iterator first_leader = end();
 307 
 308         for (; i != i_end; ++i) {
 309                 if (i->side() == side && i->can_recruit()){
 310                         if ((first_leader == end()) || (i->underlying_id() < first_leader->underlying_id()) )
 311                                 first_leader = i;
 312                 }
 313         }
 314         return first_leader;

 335 bool unit_map::self_check() const {
 336         bool good(true);
 337 
 338         umap::const_iterator uit(umap_.begin());
 339         for(; uit != umap_.end(); ++uit){
 340                 if(uit->second.ref_count < 0){
 341                         good=false;
 342                         ERR_NG << "unit_map pod ref_count <0 is " << uit->second.ref_count<< std::endl;
 343                 }
 344                 if(uit->second.unit){
 345                         uit->second.unit->id(); //crash if bad pointer
 346                 }
 347                 if(uit->first <= 0){
 348                         good=false;
 349                         ERR_NG << "unit_map umap uid <=0 is " << uit->first << std::endl;
 350                 }
 351                 if(!uit->second.unit && uit->second.ref_count == 0 ){
 352                         good=false;
 353                         ERR_NG << "unit_map umap unit==nullptr when refcount == 0" << std::endl;
 354                 }
 355                 if(uit->second.unit && uit->second.unit->underlying_id() != uit->first){
 356                         good=false;
 357                         ERR_NG << "unit_map umap uid("<<uit->first<<") != underlying_id()["<< uit->second.unit->underlying_id()<< "]" << std::endl;
 358                 }
 359         }
 360         lmap::const_iterator locit(lmap_.begin());
 361         for(; locit != lmap_.end(); ++locit){
 362                 if(locit->second == umap_.end() ){
 363                         good=false;
 364                         ERR_NG << "unit_map lmap element == umap_.end() "<< std::endl;
 365                 }
 366                 if(locit->first != locit->second->second.unit->get_location()){
 367                         good=false;
 368                         ERR_NG << "unit_map lmap location != unit->get_location() " << std::endl;
 369                 }
 370         }
 371         //assert(good);
 372         return good;
 373 }

 377 bool unit_map::has_unit(const unit * const u) const
 378 {
 379         assert(u);
 380 
 381         for(const umap::value_type& item : umap_) {
 382                 if(item.second.unit.get() == u) {
 383                         return true;
 384                 }
 385         }
 386         return false;
