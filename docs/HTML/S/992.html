  63 unit_race::unit_race(const config& cfg) :
  64                 cfg_(cfg),
  65                 id_(cfg["id"]),
  66                 icon_(cfg["editor_icon"]),
  67                 plural_name_(cfg["plural_name"].t_str()),
  68                 description_(cfg["description"].t_str()),
  69                 ntraits_(cfg["num_traits"]),
  70                 traits_(cfg.child_range("trait")),
  71                 topics_(cfg.child_range("topic")),
  72                 global_traits_(!cfg["ignore_global_traits"].to_bool()),
  73                 undead_variation_(cfg["undead_variation"])
  74 
  75 {
  76         if (id_.empty()) {
  77                 lg::wml_error() << "[race] '" << cfg["name"] << "' is missing an id field.";
  78         }
  79         if (plural_name_.empty()) {
  80                 lg::wml_error() << "[race] '" << cfg["name"] << "' is missing a plural_name field.";
  81                 plural_name_ = (cfg["name"]);
  82         }
  83 
  84         // use "name" if "male_name" or "female_name" aren't available
  85         name_[MALE] = cfg["male_name"];
  86         if(name_[MALE].empty()) {
  87                 name_[MALE] = (cfg["name"]);
  88         }
  89         name_[FEMALE] = cfg["female_name"];
  90         if(name_[FEMALE].empty()) {
  91                 name_[FEMALE] = (cfg["name"]);
  92         }
  93 
  94         name_generator_factory generator_factory = name_generator_factory(cfg, {"male", "female"});
  95 
  96         for(int i=MALE; i<NUM_GENDERS; i++) {
  97                 GENDER gender = static_cast<GENDER>(i);
  98                 name_generator_[i] = generator_factory.get_name_generator(gender_string(gender));
  99         }
 100 }

 130 const std::string& gender_string(unit_race::GENDER gender) {
 131         switch(gender) {
 132         case unit_race::FEMALE:
 133                 return unit_race::s_female;
 134         default:
 135         case unit_race::MALE:
 136                 return unit_race::s_male;
 137         }
 138 }
 139 
 140 unit_race::GENDER string_gender(const std::string& str, unit_race::GENDER def) {
 141         if ( str == unit_race::s_male ) {
 142                 return unit_race::MALE;
 143         } else if ( str == unit_race::s_female ) {
 144                 return unit_race::FEMALE;
 145         }
 146         return def;
 147 }
