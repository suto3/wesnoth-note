 195 def resistance_rating_color_class(resistance):
 196     """Return a color class adequate for the provided unit resistance value."""
 197     if resistance < 0:
 198         return 'red'
 199     elif resistance <= 20:
 200         return 'yellow'
 201     elif resistance <= 40:
 202         return 'olive'
 203     else:
 204         return 'green'

 217 def mvtcost_rating_color_class(str_mvtcost, str_moves):
 218     """Return a color class adequate for the provided movement cost value."""

 245 class Translation:
 246     def __init__(self, localedir, langcode):
 247         self.catalog = {}
 248         self.localedir = localedir
 249         self.langcode = langcode
 250         class Dummy:
 251             def gettext(self, msgid):
 252                 if not msgid:
 253                     return ""
 254                 caret = msgid.find("^")
 255                 if caret < 0:
 256                     return msgid
 257                 return msgid[caret + 1:]
 258         self.dummy = Dummy()
 259 
 260     def translate(self, string, textdomain):
 261         if textdomain not in self.catalog:
 262             try:
 263                 self.catalog[textdomain] = gettext.translation(
 264                     textdomain, self.localedir, [self.langcode])
 265                 self.catalog[textdomain].add_fallback(self.dummy)
 266             except IOError:
 267                 self.catalog[textdomain] = self.dummy
 268             except AttributeError:
 269                 self.catalog[textdomain] = self.dummy
 270             except IndexError:
 271                 # not sure why, but this happens within the
 272                 # gettext.translation call sometimes
 273                 self.catalog[textdomain] = self.dummy
 274             except ValueError:
 275                 self.catalog[textdomain] = self.dummy
 276         return self.catalog[textdomain].gettext(string)

 360     def analyze_units(self, grouper, add_parents):
 361         """
 362         This takes all units belonging to a campaign, then groups them either
 363         by race or faction, and creates an advancements tree out of it.
 364         """
 365 
 366         # Build an advancement tree forest of all units.
 367         forest = self.forest = helpers.UnitForest()
 368         units_added = {}
 369         for uid, u in list(self.wesnoth.unit_lookup.items()):
 370             if u.hidden:
 371                 continue
 372             if grouper.unitfilter(u):
 373                 forest.add_node(helpers.UnitNode(u))
 374                 units_added[uid] = u
 375 
 376         #print("    %d/%d units" % (len(units_added), len(self.wesnoth.unit_lookup)))
 377 
 378         # Always add any child units, even if they have been filtered out..
 379         while units_added:
 380             new_units_added = {}
 381             for uid, u in list(units_added.items()):
 382                 for auid in u.advance:
 383                     if not auid in forest.lookup:
 384                         try:
 385                             au = self.wesnoth.unit_lookup[auid]
 386                         except KeyError:
 387                             error_message(
 388                                 "Warning: Unit %s not found as advancement of %s\n" %
 389                                 (auid, repr(uid)))
 390                             continue
 391                         forest.add_node(helpers.UnitNode(au))
 392                         new_units_added[auid] = au
 393             units_added = new_units_added
 394 
 395         if add_parents:
 396             # Also add parent units
 397             added = True
 398             while added:
 399                 added = False
 400                 for uid, u in list(self.wesnoth.unit_lookup.items()):
 401                     if uid in forest.lookup:
 402                         continue
 403                     for auid in u.advance:
 404                         if auid in forest.lookup:
 405                             forest.add_node(helpers.UnitNode(u))
 406                             added = True
 407                             break
 408 
 409         forest.update()
 410 
 411         # Partition trees by race/faction of first unit.
 412         groups = {}
 413         breadth = 0
 414 
 415         for tree in list(forest.trees.values()):
 416             u = tree.unit
 417             ugroups = grouper.groups(u)
 418 
 419             for group in ugroups:
 420                 groups[group] = groups.get(group, []) + [tree]
 421                 breadth += tree.breadth
 422 
 423         thelist = list(groups.keys())
 424         thelist.sort(key=lambda x: grouper.group_name(x))
 425 
 426         rows_count = breadth + len(thelist)
 427         # Create empty grid.
 428         rows = []
 429         for j in range(rows_count):
 430             column = []
 431             for i in range(6):
 432                 column.append((1, 1, None))
 433             rows.append(column)
 434 
 435         # Sort advancement trees by name of first unit and place into the grid.
 436         def by_name(t):
 437             x = T(t.unit, "name")
 438             return "" if x is None else x
 439 
 440         def grid_place(nodes, x):
 441             nodes.sort(key=by_name)
 442             for node in nodes:
 443                 level = max(0, min(5, node.unit.level))
 444                 rows[x][level] = (1, node.breadth, node)
 445                 for i in range(1, node.breadth):
 446                     rows[x + i][level] = (0, 0, node)
 447                 grid_place(node.children, x)
 448                 x += node.breadth
 449             return x
 450 
 451         x = 0
 452         for group in thelist:
 453             node = helpers.GroupNode(group)
 454             node.name = grouper.group_name(group)
 455 
 456             rows[x][0] = (6, 1, node)
 457             for i in range(1, 6):
 458                 rows[x][i] = (0, 0, None)
 459             nodes = groups[group]
 460             x += 1
 461             x = grid_place(nodes, x)
 462 
 463         self.unitgrid = rows
 464 
 465         return len(forest.lookup)

 736     def get_abilities(self, u):
 737         anames = []
 738         already = {}
 739         for abilities in u.get_all(tag="abilities"):
 740             try:
 741                 c = abilities.get_all()
 742             except AttributeError:
 743                 c = []
 744             for ability in c:
 745                 try:
 746                     id = ability.get_text_val("id")

 748                     error_message("Error: Ignoring ability " + ability.debug())
 749                     continue
 750                 if id in already:
 751                     continue
 752                 already[id] = True
 753                 name = T(ability, "name")
 754                 if not name:
 755                     name = ability.name.decode("utf8")

 757                 # are normally hidden in the game and used to implement more
 758                 # complex ones.
 759                 if name:
 760                     anames.append(name)

 763     def get_recursive_attacks(self, this_unit):
 764 
 765         def copy_attributes(copy_from, copy_to):
 766             for c in copy_from.data:
 767                 if isinstance(c, wmlparser3.AttributeNode):
 768                     copy_to.data.append(c)
 769 
 770         # Use attacks of base_units as base, if we have one.
 771         base_unit = self.wesnoth.get_base_unit(this_unit)
 772         attacks = []
 773         if base_unit:
 774             attacks = copy.deepcopy(self.get_recursive_attacks(base_unit))
 775 
 776         base_attacks_count = len(attacks)
 777         for i, attack in enumerate(this_unit.get_all(tag="attack")):
 778             # Attack merging is order based.
 779             if i < base_attacks_count:
 780                 copy_attributes(attack, attacks[i])
 781             else:
 782                 attacks.append(attack)
 783 
 784         return attacks

1422 def html_postprocess_file(filename, isocode, batchlist):

1425     ehtml = ""
1426 
1427     cids = [[], []]
1428     for addon in batchlist:
1429         for campaign in addon.get("campaigns", []):
1430             if campaign["units"] == "?" or campaign["units"] <= 0:
1431                 continue
1432             lang = isocode if addon["name"] == "mainline" else "en_US"
1433             c = addon["name"], campaign["id"], campaign["translations"].get(
1434                 lang, campaign["name"]), lang
1435             if addon["name"] == "mainline":
1436                 cids[0].append(c)
1437             else:
1438                 cids[1].append(c)

1445             url = cleanurl("../../%s/%s/%s.html" % (addon, lang, cname))
1446             chtml += '<li><a title="%s" href="%s" role="menuitem">%s</a></li>\n' % (
1447                 cleantext(campname), url, cleantext(campname, quote=False))
1448         if i == 0 and cids[1]:
1449             chtml += '</ul><ul>'
1450 
1451     eids = [[], []]
1452     for addon in batchlist:
1453         for era in addon.get("eras", []):

1455                 continue
1456             lang = isocode if addon["name"] == "mainline" else "en_US"
1457             e = addon["name"], era["id"], era["translations"].get(
1458                 lang, era["name"]), lang
1459             if addon["name"] == "mainline":
1460                 eids[0].append(e)
1461             else:
1462                 eids[1].append(e)
1463     for i in range(2):
1464         eras = eids[i]
1465         eras.sort(key=lambda x: x[2])
1466 
1467         for era in eras:
1468             addon, eid, eraname, lang = era
1469             url = cleanurl("../../%s/%s/%s.html" % (addon, lang, eid))
1470             ehtml += '<li><a title="%s" href="%s" role="menuitem">%s</a></li>\n' % (
1471                 cleantext(eraname), url, cleantext(eraname, quote=False))
