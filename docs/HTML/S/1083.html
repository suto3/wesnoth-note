  96 void config_cache::write_file(std::string file_path, const preproc_map& defines)
  97 {
  98         if(defines.empty()) {
  99                 if(filesystem::file_exists(file_path)) {
 100                         filesystem::delete_directory(file_path);
 101                 }
 102                 return;
 103         }
 104 
 105         filesystem::scoped_ostream stream = filesystem::ostream_file(file_path);
 106         config_writer writer(*stream, true, game_config::cache_compression_level);
 107 
 108         // Write all defines to stream.
 109         for(const preproc_map::value_type& define : defines) {
 110                 define.second.write(writer, define.first);
 111         }
 112 }

 137 void config_cache::read_cache(const std::string& file_path, config& cfg)
 138 {
 139         static const std::string extension = ".gz";
 140 
 141         std::stringstream defines_string;
 142         defines_string << file_path;
 143 
 144         bool is_valid = true;
 145 
 146         for(const preproc_map::value_type& d : defines_map_) {
 147                 //
 148                 // Only WESNOTH_VERSION is allowed to be non-empty.
 149                 //
 150                 if((!d.second.value.empty() || !d.second.arguments.empty()) &&
 151                    d.first != "WESNOTH_VERSION")
 152                 {
 153                         is_valid = false;
 154                         ERR_CACHE << "Invalid preprocessor define: " << d.first << '\n';
 155                         break;
 156                 }
 157 
 158                 defines_string << " " << d.first;
 159         }
 160 
 161         // Do cache check only if define map is valid and
 162         // caching is allowed.
 163         const std::string& cache_path = filesystem::get_cache_dir();
 164 
 165         if(is_valid && !cache_path.empty()) {
 166                 // Use a hash for a shorter display of the defines.
 167                 const std::string fname = cache_path + "/" +
 168                                                                   cache_file_prefix_ +
 169                                                                   utils::sha1(defines_string.str()).hex_digest();
 170                 const std::string fname_checksum = fname + ".checksum" + extension;
 171 
 172                 filesystem::file_tree_checksum dir_checksum;
 173 
 174                 if(!force_valid_cache_ && !fake_invalid_cache_) {
 175                         try {
 176                                 if(filesystem::file_exists(fname_checksum)) {
 177                                         config checksum_cfg;
 178 
 179                                         DBG_CACHE << "Reading checksum: " << fname_checksum << "\n";
 180                                         read_file(fname_checksum, checksum_cfg);
 181 
 182                                         dir_checksum = filesystem::file_tree_checksum(checksum_cfg);
 183                                 }
 184                         } catch(config::error&) {
 185                                 ERR_CACHE << "cache checksum is corrupt" << std::endl;
 186                         } catch(filesystem::io_exception&) {
 187                                 ERR_CACHE << "error reading cache checksum" << std::endl;
 188                         }
 189                 }
 190 
 191                 if(force_valid_cache_) {
 192                         LOG_CACHE << "skipping cache validation (forced)\n";
 193                 }
 194 
 195                 if(filesystem::file_exists(fname + extension) && (force_valid_cache_ || (dir_checksum == filesystem::data_tree_checksum()))) {
 196                         LOG_CACHE << "found valid cache at '" << fname << extension << "' with defines_map " << defines_string.str() << "\n";
 197                         log_scope("read cache");
 198 
 199                         try {
 200                                 read_file(fname + extension,cfg);
 201                                 const std::string define_file = fname + ".define" + extension;
 202 
 203                                 if(filesystem::file_exists(define_file)) {
 204                                         config_cache_transaction::instance().add_define_file(define_file);
 205                                 }
 206 
 207                                 return;
 208                         } catch(config::error& e) {
 209                                 ERR_CACHE << "cache " << fname << extension << " is corrupt. Loading from files: "<< e.message << std::endl;
 210                         } catch(filesystem::io_exception&) {
 211                                 ERR_CACHE << "error reading cache " << fname << extension << ". Loading from files" << std::endl;
 212                         } catch (boost::iostreams::gzip_error& e) {
 213                                 //read_file -> ... -> read_gz can throw this exception.
 214                                 ERR_CACHE << "cache " << fname << extension << " is corrupt. Error code: " << e.error() << std::endl;
 215                         }
 216                 }
 217 
 218                 LOG_CACHE << "no valid cache found. Writing cache to '" << fname << extension << " with defines_map "<< defines_string.str() << "'\n";
 219 
 220                 // Now we need queued defines so read them to memory
 221                 read_defines_queue();
 222 
 223                 preproc_map copy_map(make_copy_map());
 224 
 225                 read_configs(file_path, cfg, copy_map);
 226                 add_defines_map_diff(copy_map);
 227 
 228                 try {
 229                         write_file(fname + extension, cfg);
 230                         write_file(fname + ".define" + extension, copy_map);
 231 
 232                         config checksum_cfg;
 233 
 234                         filesystem::data_tree_checksum().write(checksum_cfg);
 235                         write_file(fname_checksum, checksum_cfg);
 236                 } catch(filesystem::io_exception&) {
 237                         ERR_CACHE << "could not write to cache '" << fname << "'" << std::endl;
 238                 }
 239 
 240                 return;
 241         }
 242 
 243         LOG_CACHE << "Loading plain config instead of cache\n";
 244 
 245         preproc_map copy_map(make_copy_map());
 246         read_configs(file_path, cfg, copy_map);
 247         add_defines_map_diff(copy_map);
 248 }

 250 void config_cache::read_defines_file(const std::string& file_path)
 251 {
 252         config cfg;
 253         read_file(file_path, cfg);
 254 
 255         DBG_CACHE << "Reading cached defines from: " << file_path << "\n";
 256 
 257         // use static preproc_define::read_pair(config) to make a object
 258         // and pass that object config_cache_transaction::insert_to_active method
 259         for(const config::any_child &value : cfg.all_children_range()) {
 260                 config_cache_transaction::instance().insert_to_active(
 261                         preproc_define::read_pair(value.cfg));
 262         }
 263 }

 332 bool config_cache::clean_cache()
 333 {
 334         std::vector<std::string> files, dirs;
 335         filesystem::get_files_in_dir(filesystem::get_cache_dir(), &files, &dirs, filesystem::ENTIRE_FILE_PATH);
 336 
 337         LOG_CACHE << "clean_cache(): " << files.size() << " files, "
 338                           << dirs.size() << " dirs to check\n";
 339 
 340         const std::string& exclude_current = cache_file_prefix_ + "*";
 341 
 342         bool status = true;
 343 
 344         status &= delete_cache_files(files, exclude_current);
 345         status &= delete_cache_files(dirs, exclude_current);
 346 
 347         LOG_CACHE << "clean_cache(): done\n";
 348 
 349         return status;
 350 }

 369 bool config_cache::delete_cache_files(const std::vector<std::string>& paths,
 370                                                                           const std::string& exclude_pattern)
 371 {
 372         const bool delete_everything = exclude_pattern.empty();
 373         bool status = true;
 374 
 375         for(const std::string& file_path : paths)
 376         {
 377                 if(!delete_everything) {
 378                         const std::string& fn = filesystem::base_name(file_path);
 379 
 380                         if(utils::wildcard_string_match(fn, exclude_pattern)) {
 381                                 LOG_CACHE << "delete_cache_files(): skipping " << file_path
 382                                                   << " excluded by '" << exclude_pattern << "'\n";
 383                                 continue;
 384                         }
 385                 }
 386 
 387                 LOG_CACHE << "delete_cache_files(): deleting " << file_path << '\n';
 388                 if(!filesystem::delete_directory(file_path)) {
 389                         ERR_CACHE << "delete_cache_files(): could not delete "
 390                                           << file_path << '\n';
 391                         status = false;
 392                 }
 393         }
 394 
 395         return status;
 396 }

 446 bool compare_define(const preproc_map::value_type& a, const preproc_map::value_type& b)
 447 {
 448         if(a.first < b.first) {
 449                 return true;
 450         }
 451 
 452         if(b.first < a.first) {
 453                 return false;
 454         }
 455 
 456         if(a.second < b.second) {
 457                 return true;
 458         }
 459 
 460         return false;
 461 }

 465 void config_cache_transaction::add_defines_map_diff(preproc_map& new_map)
 466 {
 467         if(get_state() == ACTIVE) {
 468                 preproc_map temp;
 469                 std::set_difference(new_map.begin(),
 470                                 new_map.end(),
 471                                 active_map_.begin(),
 472                                 active_map_.end(),
 473                                 std::insert_iterator<preproc_map>(temp,temp.begin()),
 474                                 &compare_define);
 475 
 476                 for(const preproc_map::value_type &def : temp) {
 477                         insert_to_active(def);
 478                 }
 479 
 480                 temp.swap(new_map);
 481         } else if (get_state() == LOCKED) {
 482                 new_map.clear();
 483         }
 484 }
