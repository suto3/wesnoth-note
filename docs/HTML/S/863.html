  31 /// This is a variant on a list that never invalidates iterators (unless, of
  32 /// course, the list ceases to exist). In particular, an erase() will only flag
  33 /// an element for removal from the list. Flagged elements are ignored by list
  34 /// functions, but they still exist from the perspective of iterators that had
  35 /// pointed (and still point) to the element.
  36 ///
  37 /// Assignment is incompatible with the goal of preserving iterators, so it is
  38 /// not implemented (declared private though, to emphasize that this is
  39 /// intentional).
  40 ///
  41 /// The insert() and erase() members are static, as they do not require a
  42 /// reference to the list.
  43 ///
  44 /// Instead of being undefined, ++end() is equal to begin(), and --begin() is
  45 /// equal to end().
  46 template <class Data>
  47 class  smart_list

  49         /// Nodes in the smart list.
  50         struct node_t
  51         {
  52                 /// Default constructor. This is for creating the root of a list.
  53                 node_t() : dat_ptr(nullptr), ref_count(1), next(this), prev(this)
  54                 {}
  55                 /// Initialized constructor. This is for creating a node in a list.
  56                 explicit node_t(const Data & d) : dat_ptr(new Data(d)), ref_count(1), next(nullptr), prev(nullptr)
  57                 {}
  58                 /// Destructor.
  59                 ~node_t();
  60 
  61                 /// The data of the node. This is nullptr for a list's root.
  62                 Data * const dat_ptr;
  63                 /// ref_count counts 1 for the list and 2 for each iterator pointing to
  64                 /// this node. (So nodes are flagged for deletion when ref_count is even.)
  65                 mutable size_t ref_count;
  66                 // List linking.
  67                 node_t * next;
  68                 node_t * prev;

  70         private: // Disallowed (and unneeded) functions (declared but not implemented).
  71                 /// No assignments.
  72                 node_t & operator=(const node_t &);
  73                 /// No copying.
  74                 node_t(const node_t & that);

 408 template <class Data>
 409 inline typename smart_list<Data>::iterator smart_list<Data>::erase

 412         flag(*pos.ptr_); // We know *pos cannot get deleted yet because pos points to it.

 414 }
 415 /** Erase a range of nodes. */
 416 template <class Data>
 417 inline typename smart_list<Data>::iterator smart_list<Data>::erase
 418         (const iterator & start, const iterator & stop)
 419 {
 420         // Loop through all nodes from start to stop.
 421         // We cannot rely on iterators because *stop might be flagged.
 422         node_t * node_ptr = start.ptr_;
 423         while ( node_ptr != stop.ptr_  &&  iterator::derefable(node_ptr) ) {
 424                 flag(*node_ptr);
 425                 node_ptr = check_erase(node_ptr);
 426         }
 427 
 428         // node_ptr is more reliable than stop because of the derefable() condition.
 429         return iterator(node_ptr);
 430 }

 432 /**
 433  * Resize the list.
 434  * This does not count flagged nodes.
 435  */
 436 template <class Data>
 437 inline void smart_list<Data>::resize(size_type n, const value_type & d)
 438 {
 439         size_type count = 0;
 440         iterator it = begin();
 441         iterator _end = end();
 442 
 443         // See how our current size compares to n.
 444         // (Not calling size(), since that would do at least as much work.)
 445         while ( it != _end  &&  count < n ) {
 446                 ++it;
 447                 ++count;
 448         }
 449 
 450         if ( count < n )
 451                 // We need more nodes.
 452                 insert(_end, n - count, d);
 453         else if ( it != _end )
 454                 // We need to remove nodes.
 455                 erase(it, _end);
 456 }

 512 /** Remove nodes equal (under @a p) to their immediate predecessor. */
 513 template <class Data>
 514 template <class BinaryPredicate>
 515 inline void smart_list<Data>::unique(const BinaryPredicate p)
 516 {
 517         // Initial state.
 518         iterator _end = end();
 519         iterator cur = begin();
 520         iterator prev = cur;
 521         // Look at the second node, making the first node "previous".
 522         if ( cur != _end )
 523                 ++cur;
 524 
 525         // Look for elements equal to their predecessors.
 526         while ( cur != _end )
 527                 if ( p(*prev, *cur) )
 528                         // Duplicate. Remove it.
 529                         cur = erase(cur);
 530                 else
 531                         // Update the tracking of our previous element.
 532                         prev = cur++;
 533 }

 535 /**
 536  * Merge @a L into *this, using @a p to determine order.
 537  * If both lists are sorted under @a p, then so is the merged list.
 538  * This is stable; equivalent nodes retain their order, with nodes of *this
 539  * considered to come before elements of @a L.
 540  */
 541 template <class Data>
 542 template <class BinaryPredicate>
 543 inline void smart_list<Data>::merge(smart_list & L, const BinaryPredicate & p)
 544 {
 545         // Merging a list into itself is a no-op, not a duplication.
 546         if ( this == &L )
 547                 return;
 548 
 549         node_t * dest = root_.next;
 550         node_t * source = L.root_.next;
 551 
 552         // Basic merge loop. Continues until one of the lists is depleted.
 553         while ( iterator::derefable(dest)  &&  iterator::derefable(source) ) {
 554                 if ( p(*(source->dat_ptr), *(dest->dat_ptr)) ) {
 555                         // We found something to merge in. See if we can merge multiple
 556                         // nodes at once.
 557                         node_t * end_merge = source->next;
 558                         while ( iterator::derefable(end_merge)  &&  p(*(end_merge->dat_ptr), *(dest->dat_ptr)) )
 559                                 end_merge = end_merge->next;
 560                         // Now end_merge is one past the nodes to merge.
 561                         // Move the nodes over.
 562                         splice(dest, *source, *(end_merge->prev));
 563                         // Advance the source.
 564                         source = end_merge;
 565                 }
 566                 else
 567                         // Advance the destination.
 568                         dest = dest->next;
 569         }
 570 
 571         // Append any remaining nodes from L.
 572         if ( iterator::derefable(source) ) // hence, we made it to our end.
 573                 splice(&root_, *source, *(L.root_.prev));
 574 }

 576 /**
 577  * Sort *this, using the order given by @a p.
 578  * This is stable; equivalent nodes retain their order.
 579  * This is an efficient sort, O(n lg n).
 580  */
 581 template <class Data>
 582 template <class BinaryPredicate>
 583 inline void smart_list<Data>::sort(const BinaryPredicate & p)
 584 {
 585         // We'll leverage merge() to do a merge sort.
 586 
 587         // First, get the real size of the list, so we can allocate memory before
 588         // altering *this (exception safety).
 589         size_type count = 0;
 590         for ( node_t * node_ptr = root_.next; iterator::derefable(node_ptr); node_ptr = node_ptr->next )
 591                 ++count;
 592         // Abort on trivial cases.
 593         if ( count < 2 )
 594                 return;
 595 
 596         // Split *this into 1-length pieces.
 597         std::vector<smart_list> sorter(count); // No memory allocations after this point.
 598         for ( size_type i = 0; i < count; ++i )
 599                 sorter[i].splice(&(sorter[i].root_), *(root_.next), *(root_.next));
 600 
 601         // At the start of each iteration, step will be the distance between lists
 602         // containing data. (There will be O(lg n) iterations.)
 603         for ( size_type step = 1; step < count; step *= 2 )
 604                 // Merge consecutive data-bearing lists. Summing over all iterations of
 605                 // this (inner) loop, there will be O(n) comparisons made.
 606                 for ( size_type i = 0; i + step < count; i += 2*step )
 607                         sorter[i].merge(sorter[i+step], p);
 608 
 609         // The entire (sorted) list is now in the first element of the vector.
 610         swap(sorter[0]);
 611 }

 614 /**
 615  * Low-level insertion.
 616  * The insertion will occur before @a pos. Pass &root_ to append to the list.
 617  */
 618 template <class Data>
 619 inline typename smart_list<Data>::node_t * smart_list<Data>::insert
 620         (node_t * const pos, const value_type & d)
 621 {
 622         node_t * new_node = new node_t(d);
 623         // The new node is essentially a 1-node list.
 624         link(pos, *new_node, *new_node);
 625         // Return a pointer to the new node.
 626         return new_node;
 627 }
 628 /**
 629  * Low-level erasure.
 630  * The node will only be erased if its ref_count is 0. (So the caller must
 631  * remove their reference before calling this.)
 632  * @returns the next node in the list (could be a flagged node).
 633  */
 634 template <class Data>
 635 inline typename smart_list<Data>::node_t * smart_list<Data>::check_erase
 636         (node_t * const pos)
 637 {
 638         // Sanity check.
 639         if ( !iterator::derefable(pos) )
 640                 return nullptr;
 641 
 642         // Remember our successor.
 643         node_t * ret_val = pos->next;
 644 
 645         // Can we actually erase?
 646         if ( pos->ref_count == 0 ) {
 647                 // Disconnect *pos from the list.
 648                 unlink(*pos, *pos);
 649                 // Now we can delete.
 650                 delete pos;
 651         }
 652 
 653         return ret_val;
 654 }

 656 /**
 657  * Assuming the nodes from @a begin_link to @a end_link are linked, they will
 658  * be inserted into *this before @a pos. (Pass &root_ to append to the list.)
 659  * Note that *end_link is included in the insert; this is not exactly
 660  * analogous to a range of iterators.
 661  * This is a constant-time operation; reference counts are not updated.
 662  */
 663 template <class Data>
 664 inline void smart_list<Data>::link(node_t * const pos, node_t & begin_link, node_t & end_link)
 665 {
 666         // Link the new nodes into the list.
 667         begin_link.prev = pos->prev;
 668         end_link.next   = pos;
 669 
 670         // Link the list to the new nodes.
 671         begin_link.prev->next = &begin_link;
 672         end_link.next->prev   = &end_link;
 673 }
 674 /**
 675  * Assuming @a begin_unlink and @a end_unlink are nodes from *this, with
 676  * @a end_unlink a successor of @a begin_unlink, that chain of nodes will
 677  * be removed from *this.
 678  * Ownership of the removed chain is transferred to the caller, who becomes
 679  * repsonsible for not leaving the nodes in limbo.
 680  * This is a constant-time operation; reference counts are not updated.
 681  */
 682 template <class Data>
 683 inline void smart_list<Data>::unlink(node_t & begin_unlink, node_t & end_unlink)
 684 {
 685         // Disconnect the list from the nodes.
 686         begin_unlink.prev->next = end_unlink.next;
 687         end_unlink.next->prev = begin_unlink.prev;
 688 
 689         // Disconnect the nodes from the list. This leaves the nodes in limbo.
 690         end_unlink.next = nullptr;
 691         begin_unlink.prev = nullptr;
 692 }
 693 
 694 /**
 695  * Low-level splice of the chain from @a b to @a e (including b and e)
 696  * to the spot before @a pos.
 697  */
 698 template <class Data>
 699 inline void smart_list<Data>::splice(node_t * const pos, node_t & b, node_t & e)
 700 {
 701         // Remove from the old list.
 702         unlink(b, e);
 703         // Splice into the new list.
 704         link(pos, b, e);
 705 }

 708 /**
 709  * Equality, if lists contain equal elements in the same order.
 710  */
 711 template <class Data>
 712 inline bool operator==(const smart_list<Data> & a, const smart_list<Data> & b)
 713 {
 714         typename smart_list<Data>::const_iterator end_a = a.end();
 715         typename smart_list<Data>::const_iterator end_b = b.end();
 716         typename smart_list<Data>::const_iterator it_a = a.begin();
 717         typename smart_list<Data>::const_iterator it_b = b.begin();
 718 
 719         for ( ; it_a != end_a  &&  it_b != end_b; ++it_a, ++it_b )
 720                 if ( *it_a != *it_b )
 721                         // mismatch
 722                         return false;
 723 
 724         // All comparisons were equal, so they are the same if we compared everything.
 725         return it_a == end_a  &&  it_b == end_b;
 726 }
 727 /**
 728  * Lexicographical order.
 729  */
 730 template <class Data>
 731 inline bool operator<(const smart_list<Data> & a, const smart_list<Data> & b)
 732 {
 733         typename smart_list<Data>::const_iterator end_a = a.end();
 734         typename smart_list<Data>::const_iterator end_b = b.end();
 735         typename smart_list<Data>::const_iterator it_a = a.begin();
 736         typename smart_list<Data>::const_iterator it_b = b.begin();
 737 
 738         for ( ; it_a != end_a  &&  it_b != end_b; ++it_a, ++it_b )
 739                 if ( *it_a < *it_b )
 740                         // a is less than b.
 741                         return true;
 742                 else if ( *it_b < *it_a )
 743                         // b is less than a.
 744                         return false;
 745 
 746         // All comparisons were equal, so a is less than b if a is shorter.
 747         return it_b != end_b;
 748 }

 751 /* ** smart_list::node_t ** */
 752 
 753 template <class Data>
 754 inline smart_list<Data>::node_t::~node_t()
 755 {
 756         // Some safety checks.
 757         if ( dat_ptr == nullptr )
 758                 // Root node: make sure there are no lingering iterators to the list.
 759                 assert(next == this);
 760         else {
 761                 // Normal node: make sure we are not still in a list.
 762                 assert(next == nullptr  &&  prev == nullptr);
 763                 // Make sure no iterators point to us.
 764                 assert(ref_count == 0);
 765         }
 766 
 767         delete dat_ptr;
 768 }

 773 /**
 774  * Advances our pointer to an unflagged node, possibly in the reverse direction.
 775  * This will advance at least one step, and will not stop on a flagged node.
 776  * In addition, this takes care of updating reference counts.
 777  * (This is the code shared by increments and decrements.)
 778  */
 779 template <class Data>
 780 template <class Value, bool Reversed>
 781 inline void smart_list<Data>::iterator_base<Value, Reversed>::advance(bool reverse)
 782 {
 783         node_t * old_ptr = ptr_;
 784 
 785         // Guarantee a change.
 786         inc(reverse);
 787         // Skip as necessary.
 788         skip_flagged(reverse);
 789 
 790         // Update reference counts.
 791         refer();
 792         unref(old_ptr);
 793 }
 794 
 795 /** Make sure we are not pointing to a flagged node. */
 796 template <class Data>
 797 template <class Value, bool Reversed>
 798 inline void smart_list<Data>::iterator_base<Value, Reversed>::skip_flagged
 799         (bool reverse)
 800 {
 801         while ( derefable()  &&  smart_list<Data>::flagged(*ptr_) )
 802                 inc(reverse);
 803 }
 804 
 805 /** Add a reference to that to which we point. */
 806 template <class Data>
 807 template <class Value, bool Reversed>
 808 inline void smart_list<Data>::iterator_base<Value, Reversed>::refer() const
 809 {
 810         if ( derefable() )
 811                 ptr_->ref_count += 2;
 812 }
 813 
 814 /**
 815  * Remove a reference.
 816  * May delete old_ptr. So call this after updating ptr_ to a new value.
 817  */
 818 template <class Data>
 819 template <class Value, bool Reversed>
 820 inline void smart_list<Data>::iterator_base<Value, Reversed>::unref(node_t * old_ptr)
 821 {
 822         if ( derefable(old_ptr) ) {
 823                 // Remove a reference.
 824                 old_ptr->ref_count -= 2;
 825                 smart_list<Data>::check_erase(old_ptr);
 826         }
 827 }
