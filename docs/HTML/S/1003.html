  91 /**
  92  * Returns whether or not *this matches the given @a filter, ignoring the
  93  * complexities introduced by [and], [or], and [not].
  94  */
  95 static bool matches_simple_filter(const attack_type & attack, const config & filter)
  96 {
  97         const std::vector<std::string>& filter_range = utils::split(filter["range"]);
  98         const std::string& filter_damage = filter["damage"];
  99         const std::string& filter_attacks = filter["number"];
 100         const std::string& filter_accuracy = filter["accuracy"];
 101         const std::string& filter_parry = filter["parry"];
 102         const std::string& filter_movement = filter["movement_used"];
 103         const std::vector<std::string> filter_name = utils::split(filter["name"]);
 104         const std::vector<std::string> filter_type = utils::split(filter["type"]);
 105         const std::string filter_special = filter["special"];
 106         const std::string filter_special_active = filter["special_active"];
 107         const std::string filter_formula = filter["formula"];
 108 
 109         if ( !filter_range.empty() && std::find(filter_range.begin(), filter_range.end(), attack.range()) == filter_range.end() )
 110                 return false;
 111 
 112         if ( !filter_damage.empty() && !in_ranges(attack.damage(), utils::parse_ranges(filter_damage)) )
 113                 return false;
 114 
 115         if (!filter_attacks.empty() && !in_ranges(attack.num_attacks(), utils::parse_ranges(filter_attacks)))
 116                 return false;
 117 
 118         if (!filter_accuracy.empty() && !in_ranges(attack.accuracy(), utils::parse_ranges(filter_accuracy)))
 119                 return false;
 120 
 121         if (!filter_parry.empty() && !in_ranges(attack.parry(), utils::parse_ranges(filter_parry)))
 122                 return false;
 123 
 124         if (!filter_movement.empty() && !in_ranges(attack.movement_used(), utils::parse_ranges(filter_movement)))
 125                 return false;
 126 
 127         if ( !filter_name.empty() && std::find(filter_name.begin(), filter_name.end(), attack.id()) == filter_name.end() )
 128                 return false;
 129 
 130         if ( !filter_type.empty() && std::find(filter_type.begin(), filter_type.end(), attack.type()) == filter_type.end() )
 131                 return false;
 132 
 133         if ( !filter_special.empty() && !attack.get_special_bool(filter_special, true) )
 134                 return false;
 135 
 136         if ( !filter_special_active.empty() && !attack.get_special_bool(filter_special_active, false) )
 137                 return false;
 138 
 139         if (!filter_formula.empty()) {
 140                 try {
 141                         const wfl::attack_type_callable callable(attack);
 142                         const wfl::formula form(filter_formula, new wfl::gamestate_function_symbol_table);
 143                         if(!form.evaluate(callable).as_bool()) {
 144                                 return false;
 145                         }
 146                 } catch(wfl::formula_error& e) {
 147                         lg::wml_error() << "Formula error in weapon filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 148                         // Formulae with syntax errors match nothing
 149                         return false;
 150                 }
 151         }
 152 
 153         // Passed all tests.
 154         return true;
 155 }

 157 /**
 158  * Returns whether or not *this matches the given @a filter.
 159  */
 160 bool attack_type::matches_filter(const config& filter) const
 161 {
 162         // Handle the basic filter.
 163         bool matches = matches_simple_filter(*this, filter);
 164 
 165         // Handle [and], [or], and [not] with in-order precedence
 166         for (const config::any_child &condition : filter.all_children_range() )
 167         {
 168                 // Handle [and]
 169                 if ( condition.key == "and" )
 170                         matches = matches && matches_filter(condition.cfg);
 171 
 172                 // Handle [or]
 173                 else if ( condition.key == "or" )
 174                         matches = matches || matches_filter(condition.cfg);
 175 
 176                 // Handle [not]
 177                 else if ( condition.key == "not" )
 178                         matches = matches && !matches_filter(condition.cfg);
 179         }
 180 
 181         return matches;
 182 }

 184 /**
 185  * Modifies *this using the specifications in @a cfg, but only if *this matches
 186  * @a cfg viewed as a filter.
 187  *
 188  * @returns whether or not @c this matched the @a cfg as a filter.
 189  */
 190 bool attack_type::apply_modification(const config& cfg)
 191 {
 192         if( !matches_filter(cfg) )
 193                 return false;
 194 
 195         const std::string& set_name = cfg["set_name"];
 196         const t_string& set_desc = cfg["set_description"];
 197         const std::string& set_type = cfg["set_type"];
 198         const std::string& set_icon = cfg["set_icon"];
 199         const std::string& del_specials = cfg["remove_specials"];
 200         const config &set_specials = cfg.child("set_specials");
 201         const std::string& increase_damage = cfg["increase_damage"];
 202         const std::string& set_damage = cfg["set_damage"];
 203         const std::string& increase_attacks = cfg["increase_attacks"];
 204         const std::string& set_attacks = cfg["set_attacks"];
 205         const std::string& set_attack_weight = cfg["attack_weight"];
 206         const std::string& set_defense_weight = cfg["defense_weight"];
 207         const std::string& increase_accuracy = cfg["increase_accuracy"];
 208         const std::string& set_accuracy = cfg["set_accuracy"];
 209         const std::string& increase_parry = cfg["increase_parry"];
 210         const std::string& set_parry = cfg["set_parry"];
 211         const std::string& increase_movement = cfg["increase_movement_used"];
 212         const std::string& set_movement = cfg["set_movement_used"];
 213         // NB: If you add something here that requires a description,
 214         // it needs to be added to describe_modification as well.
 215 
 216         if(set_name.empty() == false) {
 217                 id_ = set_name;
 218         }
 219 
 220         if(set_desc.empty() == false) {
 221                 description_ = set_desc;
 222         }
 223 
 224         if(set_type.empty() == false) {
 225                 type_ = set_type;
 226         }
 227 
 228         if(set_icon.empty() == false) {
 229                 icon_ = set_icon;
 230         }
 231 
 232         if(del_specials.empty() == false) {
 233                 const std::vector<std::string>& dsl = utils::split(del_specials);
 234                 config new_specials;
 235                 for (const config::any_child &vp : specials_.all_children_range()) {
 236                         std::vector<std::string>::const_iterator found_id =
 237                                 std::find(dsl.begin(), dsl.end(), vp.cfg["id"].str());
 238                         if (found_id == dsl.end()) {
 239                                 new_specials.add_child(vp.key, vp.cfg);
 240                         }
 241                 }
 242                 specials_ = new_specials;
 243         }
 244 
 245         if (set_specials) {
 246                 const std::string &mode = set_specials["mode"];
 247                 if (mode != "append") {
 248                         specials_.clear();
 249                 }
 250                 for (const config::any_child &value : set_specials.all_children_range()) {
 251                         specials_.add_child(value.key, value.cfg);
 252                 }
 253         }
 254 
 255         if(set_damage.empty() == false) {
 256                 damage_ = std::stoi(set_damage);
 257                 if (damage_ < 0) {
 258                         damage_ = 0;
 259                 }
 260         }
 261 
 262         if(increase_damage.empty() == false) {
 263                 damage_ = utils::apply_modifier(damage_, increase_damage, 0);
 264                 if (damage_ < 0) {
 265                         damage_ = 0;
 266                 }
 267         }
 268 
 269         if(set_attacks.empty() == false) {
 270                 num_attacks_ = std::stoi(set_attacks);
 271                 if (num_attacks_ < 0) {
 272                         num_attacks_ = 0;
 273                 }
 274 
 275         }
 276 
 277         if(increase_attacks.empty() == false) {
 278                 num_attacks_ = utils::apply_modifier(num_attacks_, increase_attacks, 1);
 279         }
 280 
 281         if(set_accuracy.empty() == false) {
 282                 accuracy_ = std::stoi(set_accuracy);
 283         }
 284 
 285         if(increase_accuracy.empty() == false) {
 286                 accuracy_ = utils::apply_modifier(accuracy_, increase_accuracy, 1);
 287         }
 288 
 289         if(set_parry.empty() == false) {
 290                 parry_ = std::stoi(set_parry);
 291         }
 292 
 293         if(increase_parry.empty() == false) {
 294                 parry_ = utils::apply_modifier(parry_, increase_parry, 1);
 295         }
 296 
 297         if(set_movement.empty() == false) {
 298                 movement_used_ = std::stoi(set_movement);
 299         }
 300 
 301         if(increase_movement.empty() == false) {
 302                 movement_used_ = utils::apply_modifier(movement_used_, increase_movement, 1);
 303         }
 304 
 305         if(set_attack_weight.empty() == false) {
 306                 attack_weight_ = lexical_cast_default<double>(set_attack_weight,1.0);
 307         }
 308 
 309         if(set_defense_weight.empty() == false) {
 310                 defense_weight_ = lexical_cast_default<double>(set_defense_weight,1.0);
 311         }
 312 
 313         return true;
 314 }

 316 /**

 318  * made. This can be used to get a description of the modification(s) specified
 319  * by @a cfg (if *this matches cfg as a filter).
 320  *
 321  * If *description is provided, it will be set to a (translated) description
 322  * of the modification(s) applied (currently only changes to the number of
 323  * strikes, damage, accuracy, and parry are included in this description).
 324  *
 325  * @returns whether or not @c this matched the @a cfg as a filter.
 326  */

 330                 return false;
 331 
 332         // Did the caller want the description?
 333         if(description != nullptr) {
 334                 const std::string& increase_damage = cfg["increase_damage"];
 335                 const std::string& set_damage = cfg["set_damage"];

 344 
 345                 std::vector<t_string> desc;

 347                 if(!increase_damage.empty()) {
