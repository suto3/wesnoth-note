  48 SDL_Rect intersect_rects(SDL_Rect const &rect1, SDL_Rect const &rect2)
  49 {
  50         SDL_Rect res;
  51         res.x = std::max<int>(rect1.x, rect2.x);
  52         res.y = std::max<int>(rect1.y, rect2.y);
  53         int w = std::min<int>(rect1.x + rect1.w, rect2.x + rect2.w) - res.x;
  54         int h = std::min<int>(rect1.y + rect1.h, rect2.y + rect2.h) - res.y;
  55         if (w <= 0 || h <= 0) return empty_rect;
  56         res.w = w;
  57         res.h = h;
  58         return res;
  59 }

  61 SDL_Rect union_rects(SDL_Rect const &rect1, SDL_Rect const &rect2)
  62 {
  63         if (rect1.w == 0 || rect1.h == 0) return rect2;
  64         if (rect2.w == 0 || rect2.h == 0) return rect1;
  65         SDL_Rect res;
  66         res.x = std::min<int>(rect1.x, rect2.x);
  67         res.y = std::min<int>(rect1.y, rect2.y);
  68         res.w = std::max<int>(rect1.x + rect1.w, rect2.x + rect2.w) - res.x;
  69         res.h = std::max<int>(rect1.y + rect1.h, rect2.y + rect2.h) - res.y;
  70         return res;
  71 }

  73 void fill_rect_alpha(SDL_Rect &rect, Uint32 color, Uint8 alpha, surface target)
  74 {
  75         if(alpha == SDL_ALPHA_OPAQUE) {
  76                 sdl::fill_rect(target,&rect,color);
  77                 return;
  78         } else if(alpha == SDL_ALPHA_TRANSPARENT) {
  79                 return;
  80         }
  81 
  82         surface tmp(create_compatible_surface(target,rect.w,rect.h));
  83         if(tmp == nullptr) {
  84                 return;
  85         }
  86 
  87         SDL_SetSurfaceBlendMode (tmp, SDL_BLENDMODE_BLEND);
  88 
  89         SDL_Rect r {0,0,rect.w,rect.h};
  90         sdl::fill_rect(tmp,&r,color);
  91         adjust_surface_alpha(tmp, alpha);
  92         sdl_blit(tmp,nullptr,target,&rect);
  93 }
