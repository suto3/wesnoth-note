  71         double road_path_calculator::cost(const map_location& loc, const double /*so_far*/) const
  72         {
  73                 ++calls;
  74                 if(loc.x < 0 || loc.y < 0 || loc.x >= map_.w || loc.y >= map_.h) {
  75 
  76                         return (pathfind::cost_calculator::getNoPathValue());
  77                 }
  78 
  79                 // We multiply the cost by a random amount,
  80                 // depending upon how 'windy' the road should be.
  81                 // If windiness is 1, that will mean that the cost is always genuine,
  82                 // and so the road always takes the shortest path.
  83                 // If windiness is greater than 1, we sometimes over-report costs
  84                 // for some segments, to make the road wind a little.
  85 
  86                 double windiness = 1.0;
  87 
  88                 if(windiness_ > 1) {
  89                         // modified pseudo_random taken from builder.cpp
  90                         unsigned int a = (loc.x + 92872973) ^ 918273;
  91                         unsigned int b = (loc.y + 1672517) ^ 128123;
  92                         unsigned int c = a*b + a + b + seed_;
  93                         unsigned int random = c*c;
  94                         // this is just "big random number modulo windiness_"
  95                         // but avoid the "modulo by a low number (like 2)"
  96                         // because it can increase arithmetic patterns
  97                         int noise = random % (windiness_ * 137) / 137;
  98                         windiness += noise;
  99                 }
 100 
 101                 const t_translation::terrain_code c = map_[loc.x][loc.y];
 102                 const std::map<t_translation::terrain_code, double>::const_iterator itor = cache_.find(c);
 103                 if(itor != cache_.end()) {
 104                         return itor->second*windiness;
 105                 }
 106 
 107                 static std::string terrain;
 108                 terrain = t_translation::write_terrain_code(c);
 109                 double res = getNoPathValue();
 110                 if(const config &child = cfg_.find_child("road_cost", "terrain", terrain)) {
 111                         res = child["cost"].to_double();
 112                 }
 113 
 114                 cache_.emplace(c, res);
 115                 return windiness*res;
 116         }

 241 /**
 242  * Generate a height-map.
 243  *
 244  * Basically we generate a lot of hills, each hill being centered at a certain
 245  * point, with a certain radius - being a half sphere.  Hills are combined
 246  * additively to form a bumpy surface.  The size of each hill varies randomly
 247  * from 1-hill_size.  We generate 'iterations' hills in total.  The range of
 248  * heights is normalized to 0-1000.  'island_size' controls whether or not the
 249  * map should tend toward an island shape, and if so, how large the island
 250  * should be.  Hills with centers that are more than 'island_size' away from
 251  * the center of the map will be inverted (i.e. be valleys).  'island_size' as
 252  * 0 indicates no island.
 253  */
 254 height_map default_map_generator_job::generate_height_map(size_t width, size_t height, size_t iterations, size_t hill_size, size_t island_size, size_t island_off_center)
 255 {
 256         height_map res(width, std::vector<int>(height,0));
 257 
 258         size_t center_x = width/2;
 259         size_t center_y = height/2;
 260 
 261         LOG_NG << "off-centering...\n";
 262 
 263         if(island_off_center != 0) {
 264                 switch(rng_()%4) {
 265                 case 0:
 266                         center_x += island_off_center;
 267                         break;
 268                 case 1:
 269                         center_y += island_off_center;
 270                         break;
 271                 case 2:
 272                         if(center_x < island_off_center) {
 273                                 center_x = 0;
 274                         } else {
 275                                 center_x -= island_off_center;
 276                         }
 277                         break;
 278                 case 3:
 279                         if(center_y < island_off_center) {
 280                                 center_y = 0;
 281                         } else {
 282                                 center_y -= island_off_center;
 283                         }
 284                         break;
 285                 }
 286         }
 287 
 288         for(size_t i = 0; i != iterations; ++i) {
 289 
 290                 // (x1,y1) is the location of the hill,
 291                 // and 'radius' is the radius of the hill.
 292                 // We iterate over all points, (x2,y2).
 293                 // The formula for the amount the height is increased by is:
 294                 // radius - sqrt((x2-x1)^2 + (y2-y1)^2) with negative values ignored.
 295                 //
 296                 // Rather than iterate over every single point, we can reduce the points
 297                 // to a rectangle that contains all the positive values for this formula --
 298                 // the rectangle is given by min_x,max_x,min_y,max_y.
 299 
 300                 // Is this a negative hill? (i.e. a valley)
 301                 bool is_valley = false;
 302 
 303                 int x1 = island_size > 0 ? center_x - island_size + (rng_()%(island_size*2)) : int(rng_()%width);
 304                 int y1 = island_size > 0 ? center_y - island_size + (rng_()%(island_size*2)) : int(rng_()%height);
 305 
 306                 // We have to check whether this is actually a valley
 307                 if(island_size != 0) {
 308                         const size_t diffx = std::abs(x1 - int(center_x));
 309                         const size_t diffy = std::abs(y1 - int(center_y));
 310                         const size_t dist = size_t(std::sqrt(double(diffx*diffx + diffy*diffy)));
 311                         is_valley = dist > island_size;
 312                 }
 313 
 314                 const int radius = rng_()%hill_size + 1;
 315 
 316                 const int min_x = x1 - radius > 0 ? x1 - radius : 0;
 317                 const int max_x = x1 + radius < static_cast<long>(res.size()) ? x1 + radius : res.size();
 318                 const int min_y = y1 - radius > 0 ? y1 - radius : 0;
 319                 const int max_y = y1 + radius < static_cast<long>(res.front().size()) ? y1 + radius : res.front().size();
 320 
 321                 for(int x2 = min_x; x2 < max_x; ++x2) {
 322                         for(int y2 = min_y; y2 < max_y; ++y2) {
 323                                 const int xdiff = (x2-x1);
 324                                 const int ydiff = (y2-y1);
 325 
 326                                 const int hill_height = radius - int(std::sqrt(double(xdiff*xdiff + ydiff*ydiff)));
 327 
 328                                 if(hill_height > 0) {
 329                                         if(is_valley) {
 330                                                 if(hill_height > res[x2][y2]) {
 331                                                         res[x2][y2] = 0;
 332                                                 } else {
 333                                                         res[x2][y2] -= hill_height;
 334                                                 }
 335                                         } else {
 336                                                 res[x2][y2] += hill_height;
 337                                         }
 338                                 }
 339                         }
 340                 }
 341         }
 342 
 343         // Find the highest and lowest points on the map for normalization:
 344         int heighest = 0, lowest = 100000, x;
 345         for(x = 0; size_t(x) != res.size(); ++x) {
 346                 for(int y = 0; size_t(y) != res[x].size(); ++y) {
 347                         if(res[x][y] > heighest) {
 348                                 heighest = res[x][y];
 349                         }
 350 
 351                         if(res[x][y] < lowest) {
 352                                 lowest = res[x][y];
 353                         }
 354                 }
 355         }
 356 
 357         // Normalize the heights to the range 0-1000:
 358         heighest -= lowest;
 359         for(x = 0; size_t(x) != res.size(); ++x) {
 360                 for(int y = 0; size_t(y) != res[x].size(); ++y) {
 361                         res[x][y] -= lowest;
 362                         res[x][y] *= 1000;
 363                         if(heighest != 0) {
 364                                 res[x][y] /= heighest;
 365                         }
 366                 }
 367         }
 368 
 369         return res;
 370 }

 372 /**
 373  * Generate a lake.
 374  *
 375  * It will create water at (x,y), and then have 'lake_fall_off' % chance to
 376  * make another water tile in each of the directions n,s,e,w.  In each of the
 377  * directions it does make another water tile, it will have 'lake_fall_off'/2 %
 378  * chance to make another water tile in each of the directions. This will
 379  * continue recursively.
 380  */
 381 bool default_map_generator_job::generate_lake(terrain_map& terrain, int x, int y, int lake_fall_off, std::set<map_location>& locs_touched)
 382 {
 383         if(x < 0 || y < 0 || x >= terrain.w || y >= terrain.h || lake_fall_off < 0) {
 384                 return false;
 385         }
 386         //we checked for this eariler.
 387         unsigned int ulake_fall_off = lake_fall_off;
 388         terrain[x][y] = t_translation::SHALLOW_WATER;
 389         locs_touched.insert(map_location(x,y));
 390 
 391         if((rng_()%100) < ulake_fall_off) {
 392                 generate_lake(terrain,x+1,y,lake_fall_off/2,locs_touched);
 393         }
 394 
 395         if((rng_()%100) < ulake_fall_off) {
 396                 generate_lake(terrain,x-1,y,lake_fall_off/2,locs_touched);
 397         }
 398 
 399         if((rng_()%100) < ulake_fall_off) {
 400                 generate_lake(terrain,x,y+1,lake_fall_off/2,locs_touched);
 401         }
 402 
 403         if((rng_()%100) < ulake_fall_off) {
 404                 generate_lake(terrain,x,y-1,lake_fall_off/2,locs_touched);
 405         }
 406 
 407         return true;
 408 }

 410 /**
 411  * River generation.
 412  *
 413  * Rivers have a source, and then keep on flowing until they meet another body
 414  * of water, which they flow into, or until they reach the edge of the map.
 415  * Rivers will always flow downhill, except that they can flow a maximum of
 416  * 'river_uphill' uphill.  This is to represent the water eroding the higher
 417  * ground lower.
 418  *
 419  * Every possible path for a river will be attempted, in random order, and the
 420  * first river path that can be found that makes the river flow into another
 421  * body of water or off the map will be used.
 422  *
 423  * If no path can be found, then the river's generation will be aborted, and
 424  * false will be returned.  true is returned if the river is generated
 425  * successfully.
 426  */
 427 
 428 bool default_map_generator_job::generate_river_internal(const height_map& heights,
 429         terrain_map& terrain, int x, int y, std::vector<map_location>& river,
 430         std::set<map_location>& seen_locations, int river_uphill)
 431 {
 432         const bool on_map = x >= 0 && y >= 0 &&
 433                 x < static_cast<long>(heights.size()) &&
 434                 y < static_cast<long>(heights.back().size());
 435 
 436         if(on_map && !river.empty() && heights[x][y] >
 437                         heights[river.back().x][river.back().y] + river_uphill) {
 438 
 439                 return false;
 440         }
 441 
 442         // If we're at the end of the river
 443         if(!on_map || terrain[x][y] == t_translation::SHALLOW_WATER ||
 444                         terrain[x][y] == t_translation::DEEP_WATER) {
 445 
 446                 LOG_NG << "generating river...\n";
 447 
 448                 // Generate the river
 449                 for(auto i : river) {
 450                         terrain[i.x][i.y] = t_translation::SHALLOW_WATER;
 451                 }
 452 
 453                 LOG_NG << "done generating river\n";
 454 
 455                 return true;
 456         }
 457 
 458         map_location current_loc(x,y);
 459         map_location adj[6];
 460         get_adjacent_tiles(current_loc,adj);
 461         std::shuffle(std::begin(adj), std::end(adj), rng_);
 462 
 463         // Mark that we have attempted from this map_location
 464         seen_locations.insert(current_loc);
 465         river.push_back(current_loc);
 466         for(const map_location& loc : adj) {
 467                 if(seen_locations.count(loc) == 0) {
 468                         const bool res = generate_river_internal(heights,terrain,loc.x,loc.y,river,seen_locations,river_uphill);
 469                         if(res) {
 470                                 return true;
 471                         }
 472 
 473                 }
 474         }
 475 
 476         river.pop_back();
 477 
 478         return false;
 479 }

 511 /** Function which, given the map will output it in a valid format. */
 512 static std::string output_map(const terrain_map& terrain,
 513                 t_translation::starting_positions& starting_positions)
 514 {
 515         // Remember that we only want the middle 1/9th of the map.
 516         // All other segments of the map are there only to give
 517         // the important middle part some context.
 518         // We also have a border so also adjust for that.
 519         const size_t begin_x = terrain.w / 3 - gamemap::default_border ;
 520         const size_t end_x = terrain.w * 2 / 3 + gamemap::default_border;
 521         const size_t begin_y = terrain.h / 3 - gamemap::default_border;
 522         const size_t end_y = terrain.h * 2 / 3 + gamemap::default_border;
 523 
 524         terrain_map map(end_x - begin_x, end_y - begin_y);
 525         for(size_t y = begin_y; y != end_y; ++y) {
 526                 for(size_t x = begin_x; x != end_x; ++x) {
 527                         map[x - begin_x][y - begin_y] = terrain[x][y];
 528                 }
 529         }
 530 
 531         // Since the map has been resized,
 532         // the starting locations also need to be fixed
 533         for (auto it = starting_positions.left.begin(); it != starting_positions.left.end(); ++it) {
 534                 starting_positions.left.modify_data(it, [=](t_translation::coordinate&  pos) { pos.x -= begin_x; pos.y -= begin_y; });
 535         }
 536         return t_translation::write_game_map(map, starting_positions);
 537 }

 539 static int rank_castle_location(int x, int y, const is_valid_terrain& valid_terrain, int min_x, int max_x, int min_y, int max_y, size_t min_distance, const std::vector<map_location>& other_castles, int highest_ranking)
 540 {
 541         const map_location loc(x,y);
 542 
 543         size_t avg_distance = 0, lowest_distance = 1000;
 544 
 545         for(std::vector<map_location>::const_iterator c = other_castles.begin(); c != other_castles.end(); ++c) {
 546                 const size_t distance = distance_between(loc,*c);
 547                 if(distance < 6) {
 548                         return 0;
 549                 }
 550 
 551                 if(distance < lowest_distance) {
 552                         lowest_distance = distance;
 553                 }
 554 
 555                 if(distance < min_distance) {
 556                         avg_distance = 0;
 557                         return -1;
 558                 }
 559 
 560                 avg_distance += distance;
 561         }
 562 
 563         if(!other_castles.empty()) {
 564                 avg_distance /= other_castles.size();
 565         }
 566 
 567         for(int i = x-1; i <= x+1; ++i) {
 568                 for(int j = y-1; j <= y+1; ++j) {
 569                         if(!valid_terrain(i,j)) {
 570                                 return 0;
 571                         }
 572                 }
 573         }
 574 
 575         const int x_from_border = std::min<int>(x - min_x,max_x - x);
 576         const int y_from_border = std::min<int>(y - min_y,max_y - y);
 577 
 578         const int border_ranking = min_distance - std::min<int>(x_from_border,y_from_border) + min_distance - x_from_border - y_from_border;
 579 
 580         int current_ranking = border_ranking*2 + avg_distance*10 + lowest_distance*10;
 581         static const int num_nearby_locations = 11*11;
 582 
 583         const int max_possible_ranking = current_ranking + num_nearby_locations;
 584 
 585         if(max_possible_ranking < highest_ranking) {
 586                 return current_ranking;
 587         }
 588 
 589         int surrounding_ranking = 0;
 590 
 591         for(int xpos = x-5; xpos <= x+5; ++xpos) {
 592                 for(int ypos = y-5; ypos <= y+5; ++ypos) {
 593                         if(valid_terrain(xpos,ypos)) {
 594                                 ++surrounding_ranking;
 595                         }
 596                 }
 597         }
 598 
 599         return surrounding_ranking + current_ranking;
 600 }

 608         const map_location loc(x,y);
 609         std::set<map_location> locs;
 610         get_tiles_radius(loc,radius,locs);
 611         map_location best_loc;
 612         int best_rating = 0;
 613         for(auto i : locs) {
 614                 if(i.x < 0 || i.y < 0 || i.x >= map.w ||
 615                                 i.y >= map.h) {
 616 
 617                         continue;
 618                 }
 619 
 620                 const t_translation::terrain_code t = map[i.x][i.y];
 621                 const std::string str = t_translation::write_terrain_code(t);
 622                 if(const config &child = cfg.find_child("village", "terrain", str)) {
 623                         tcode_list_cache::iterator l = adj_liked_cache.find(t);
 624                         t_translation::ter_list *adjacent_liked;
 625                         if(l != adj_liked_cache.end()) {
 626                                 adjacent_liked = &(l->second);
 627                         } else {
 628                                 adj_liked_cache[t] = t_translation::read_list(child["adjacent_liked"]);
 629                                 adjacent_liked = &(adj_liked_cache[t]);
 630                         }
 631 
 632                         int rating = child["rating"];
 633                         map_location adj[6];
 634                         get_adjacent_tiles(map_location(i.x,i.y),adj);
 635                         for(size_t n = 0; n != 6; ++n) {
 636                                 if(adj[n].x < 0 || adj[n].y < 0 ||
 637                                                 adj[n].x >= map.w ||
 638                                                 adj[n].y >= map.h) {
 639 
 640                                         continue;
 641                                 }
 642 
 643                                 const t_translation::terrain_code t2 = map[adj[n].x][adj[n].y];
 644                                 rating += std::count(adjacent_liked->begin(),adjacent_liked->end(),t2);
 645                         }
 646 
 647                         if(rating > best_rating) {
 648                                 best_loc = map_location(i.x,i.y);
 649                                 best_rating = rating;
 650                         }
 651                 }
 652         }
 653 
 654         return best_loc;
 655 }

 657 // "flood fill" a tile name to adjacent tiles of certain terrain
 658 static void flood_name(const map_location& start, const std::string& name, std::map<map_location,std::string>& tile_names,
 659         const t_translation::ter_match& tile_types, const terrain_map& terrain,
 660         unsigned width, unsigned height,
 661         size_t label_count, std::map<map_location,std::string>* labels, const std::string& full_name) {
 662         map_location adj[6];
 663         get_adjacent_tiles(start,adj);
 664         size_t n;
 665         //if adjacent tiles are tiles and unnamed, name them
 666         for(n = 0; n < 6; n++) {
 667                 //we do not care for tiles outside the middle part
 668                 //cast to unsigned to skip x < 0 || y < 0 as well.
 669                 if(unsigned(adj[n].x) >= width / 3 || unsigned(adj[n].y) >= height / 3) {
 670                         continue;
 671                 }
 672 
 673                 const t_translation::terrain_code terr = terrain[adj[n].x + (width / 3)][adj[n].y + (height / 3)];
 674                 const map_location loc(adj[n].x, adj[n].y);
 675                 if((t_translation::terrain_matches(terr, tile_types)) && (tile_names.find(loc) == tile_names.end())) {
 676                         tile_names.emplace(loc, name);
 677                         //labeling decision: this is result of trial and error on what looks best in game
 678                         if(label_count % 6 == 0) { //ensure that labels do not occur more often than every 6 recursions
 679                                 labels->emplace(loc, full_name);
 680                                 label_count++; //ensure that no adjacent tiles get labeled
 681                         }
 682                         flood_name(adj[n], name, tile_names, tile_types, terrain, width, height, label_count++, labels, full_name);
 683                 }
 684         }
 685 }

 687 std::string default_map_generator_job::default_generate_map(generator_data data, std::map<map_location,std::string>* labels, const config& cfg)
 688 {
 689         log_scope("map generation");
 690 
 691         // Odd widths are nasty
 692         VALIDATE(is_even(data.width), _("Random maps with an odd width aren't supported."));
 693 
 694         // Try to find configuration for castles
 695         const config& castle_config = cfg.child("castle");
 696         if(!castle_config) {
 697                 LOG_NG << "Could not find castle configuration\n";
 698                 return std::string();
 699         }
 700 
 701         int ticks = SDL_GetTicks();
 702 
 703         // We want to generate a map that is 9 times bigger than the actual size desired.
 704         // Only the middle part of the map will be used, but the rest is so that the map we
 705         // end up using can have a context (e.g. rivers flowing from out of the map into the map,
 706         // same for roads, etc.)
 707         data.width  *= 3;
 708         data.height *= 3;
 709 
 710         config naming = game_config_.child("naming");
 711 
 712         if(cfg.has_child("naming")) {
 713                 naming.append_attributes(cfg.child("naming"));
 714         }
 715 
 716         // If the [naming] child is empty, we cannot provide good names.
 717         std::map<map_location,std::string>* misc_labels = naming.empty() ? nullptr : labels;
 718 
 719         std::shared_ptr<name_generator>
 720                 base_name_generator, river_name_generator, lake_name_generator,
 721                 road_name_generator, bridge_name_generator, mountain_name_generator,
 722                 forest_name_generator, swamp_name_generator;
 723 
 724         if(misc_labels != nullptr) {
 725                 name_generator_factory base_generator_factory{ naming, {"male", "base", "bridge", "road", "river", "forest", "lake", "mountain", "swamp"} };
 726 
 727                 naming.get_old_attribute("base_names", "male_names", "[naming]male_names= is deprecated, use base_names= instead");
 728                 //Due to the attribute detection feature of the factory we also support male_name_generator= but keep it undocumented.
 729 
 730                 base_name_generator = base_generator_factory.get_name_generator( (naming.has_attribute("base_names") || naming.has_attribute("base_name_generator")) ? "base" : "male" );
 731                 river_name_generator    = base_generator_factory.get_name_generator("river");
 732                 lake_name_generator     = base_generator_factory.get_name_generator("lake");
 733                 road_name_generator     = base_generator_factory.get_name_generator("road");
 734                 bridge_name_generator   = base_generator_factory.get_name_generator("bridge");
 735                 mountain_name_generator = base_generator_factory.get_name_generator("mountain");
 736                 forest_name_generator   = base_generator_factory.get_name_generator("forest");
 737                 swamp_name_generator    = base_generator_factory.get_name_generator("swamp");
 738         }
 739 
 740         // Generate the height of everything.
 741         const height_map heights = generate_height_map(data.width, data.height, data.iterations, data.hill_size, data.island_size, data.island_off_center);
 742 
 743         LOG_NG << "Done generating height map. " << (SDL_GetTicks() - ticks) << " ticks elapsed" << "\n";
 744         ticks = SDL_GetTicks();
 745 
 746         // Find out what the 'flatland' on this map is, i.e. grassland.
 747         std::string flatland = cfg["default_flatland"];
 748         if(flatland.empty()) {
 749                 flatland = t_translation::write_terrain_code(t_translation::GRASS_LAND);
 750         }
 751 
 752         const t_translation::terrain_code grassland = t_translation::read_terrain_code(flatland);
 753 
 754         std::vector<terrain_height_mapper> height_conversion;
 755         for(const config& h : cfg.child_range("height")) {
 756                 height_conversion.push_back(terrain_height_mapper(h));
 757         }
 758 
 759         terrain_map terrain(data.width, data.height, grassland);
 760         for(size_t x = 0; x != heights.size(); ++x) {
 761                 for(size_t y = 0; y != heights[x].size(); ++y) {
 762                         for(auto i : height_conversion) {
 763                                 if(i.convert_terrain(heights[x][y])) {
 764                                         terrain[x][y] = i.convert_to();
 765                                         break;
 766                                 }
 767                         }
 768                 }
 769         }
 770 
 771         t_translation::starting_positions starting_positions;
 772         LOG_NG << output_map(terrain, starting_positions);
 773         LOG_NG << "Placed landforms. " << (SDL_GetTicks() - ticks) << " ticks elapsed" << "\n";
 774         ticks = SDL_GetTicks();
 775 
 776         /* Now that we have our basic set of flatland/hills/mountains/water,
 777          * we can place lakes and rivers on the map.
 778          * All rivers are sourced at a lake.
 779          * Lakes must be in high land - at least 'min_lake_height'.
 780          * (Note that terrain below a certain altitude may be made into bodies of water
 781          *  in the code above - i.e. 'sea', but these are not considered 'lakes',
 782          *  because they are not sources of rivers).
 783          *
 784          * We attempt to place 'max_lakes' lakes.
 785          * Each lake will be placed at a random location, if that random location meets theminimum
 786          * terrain requirements for a lake. We will also attempt to source a river from each lake.
 787          */
 788         std::set<map_location> lake_locs;
 789 
 790         std::map<map_location, std::string> river_names, lake_names, road_names, bridge_names, mountain_names, forest_names, swamp_names;
 791 
 792         const size_t nlakes = data.max_lakes > 0 ? (rng_()%data.max_lakes) : 0;
 793         for(size_t lake = 0; lake != nlakes; ++lake) {
 794                 for(int tries = 0; tries != 100; ++tries) {
 795                         const int x = rng_()%data.width;
 796                         const int y = rng_()%data.height;
 797 
 798                         if(heights[x][y] <= cfg["min_lake_height"].to_int()) {
 799                                 continue;
 800                         }
 801 
 802                         std::vector<map_location> river = generate_river(heights, terrain, x, y, cfg["river_frequency"]);
 803 
 804                         if(!river.empty() && misc_labels != nullptr) {
 805                                 const std::string base_name = base_name_generator->generate();
 806                                 const std::string& name = river_name_generator->generate({{"base",  base_name}});
 807                                 LOG_NG << "Named river '" << name << "'\n";
 808 
 809                                 size_t name_frequency = 20;
 810                                 for(std::vector<map_location>::const_iterator r = river.begin(); r != river.end(); ++r) {
 811                                         const map_location loc(r->x-data.width/3,r->y-data.height/3);
 812 
 813                                         if(((r - river.begin())%name_frequency) == name_frequency/2) {
 814                                                 misc_labels->emplace(loc, name);
 815                                         }
 816 
 817                                         river_names.emplace(loc, base_name);
 818                                 }
 819                         }
 820 
 821                         LOG_NG << "Generating lake...\n";
 822 
 823                         std::set<map_location> locs;
 824                         if(generate_lake(terrain, x, y, cfg["lake_size"], locs) && misc_labels != nullptr) {
 825                                 bool touches_other_lake = false;
 826 
 827                                 std::string base_name = base_name_generator->generate();
 828                                 const std::string& name = lake_name_generator->generate({{"base",  base_name}});
 829 
 830                                 // Only generate a name if the lake hasn't touched any other lakes,
 831                                 // so that we don't end up with one big lake with multiple names.
 832                                 for(auto i : locs) {
 833                                         if(lake_locs.count(i) != 0) {
 834                                                 touches_other_lake = true;
 835 
 836                                                 // Reassign the name of this lake to be the same as the other lake
 837                                                 const map_location loc(i.x-data.width/3,i.y-data.height/3);
 838                                                 const std::map<map_location,std::string>::const_iterator other_name = lake_names.find(loc);
 839                                                 if(other_name != lake_names.end()) {
 840                                                         base_name = other_name->second;
 841                                                 }
 842                                         }
 843 
 844                                         lake_locs.insert(i);
 845                                 }
 846 
 847                                 if(!touches_other_lake) {
 848                                         const map_location loc(x-data.width/3,y-data.height/3);
 849                                         misc_labels->erase(loc);
 850                                         misc_labels->emplace(loc, name);
 851                                 }
 852 
 853                                 for(auto i : locs) {
 854                                         const map_location loc(i.x-data.width/3,i.y-data.height/3);
 855                                         lake_names.emplace(loc, base_name);
 856                                 }
 857                         }
 858 
 859                         break;
 860                 }
 861         }
 862 
 863         LOG_NG << "Generated rivers. " << (SDL_GetTicks() - ticks) << " ticks elapsed" << "\n";
 864         ticks = SDL_GetTicks();
 865 
 866         const size_t default_dimensions = 40*40*9;
 867 
 868         /*
 869          * Convert grassland terrain to other types of flat terrain.
 870          *
 871          * We generate a 'temperature map' which uses the height generation
 872          * algorithm to generate the temperature levels all over the map.  Then we
 873          * can use a combination of height and terrain to divide terrain up into
 874          * more interesting types than the default.
 875          */
 876         const height_map temperature_map = generate_height_map(data.width,data.height,
 877                 cfg["temperature_iterations"].to_int() * data.width * data.height / default_dimensions,
 878                 cfg["temperature_size"], 0, 0);
 879 
 880         LOG_NG << "Generated temperature map. " << (SDL_GetTicks() - ticks) << " ticks elapsed" << "\n";
 881         ticks = SDL_GetTicks();
 882 
 883         std::vector<terrain_converter> converters;
 884         for(const config& cv : cfg.child_range("convert")) {
 885                 converters.emplace_back(cv);
 886         }
 887 
 888         LOG_NG << "Created terrain converters. " << (SDL_GetTicks() - ticks) << " ticks elapsed" << "\n";
 889         ticks = SDL_GetTicks();
 890 
 891         // Iterate over every flatland tile, and determine what type of flatland it is, based on our [convert] tags.
 892         for(int x = 0; x != data.width; ++x) {
 893                 for(int y = 0; y != data.height; ++y) {
 894                         for(auto i : converters) {
 895                                 if(i.convert_terrain(terrain[x][y],heights[x][y],temperature_map[x][y])) {
 896                                         terrain[x][y] = i.convert_to();
 897                                         break;
 898                                 }
 899                         }
 900                 }
 901         }
 902 
 903         LOG_NG << "Placing castles...\n";
 904 
 905         /*
 906          * Castle configuration tag contains a 'valid_terrain' attribute which is a
 907          * list of terrains that the castle may appear on.
 908          */
 909         const t_translation::ter_list list =
 910                 t_translation::read_list(castle_config["valid_terrain"]);
 911 
 912         const is_valid_terrain terrain_tester(terrain, list);
 913 
 914         /*
 915          * Attempt to place castles at random.
 916          *
 917          * After they are placed, we run a sanity check to make sure no two castles
 918          * are closer than 'min_distance' hexes apart, and that they appear on a
 919          * terrain listed in 'valid_terrain'.
 920          *
 921          * If not, we attempt to place them again.
 922          */
 923         std::vector<map_location> castles;
 924         std::set<map_location> failed_locs;
 925 
 926         for(int player = 0; player != data.nplayers; ++player) {
 927                 LOG_NG << "placing castle for " << player << "\n";
 928                 lg::scope_logger inner_scope_logging_object__(lg::general(), "placing castle");
 929                 const int min_x = data.width/3 + 3;
 930                 const int min_y = data.height/3 + 3;
 931                 const int max_x = (data.width/3)*2 - 4;
 932                 const int max_y = (data.height/3)*2 - 4;
 933                 int min_distance = castle_config["min_distance"];
 934 
 935                 map_location best_loc;
 936                 int best_ranking = 0;
 937                 for(int x = min_x; x != max_x; ++x) {
 938                         for(int y = min_y; y != max_y; ++y) {
 939                                 const map_location loc(x,y);
 940                                 if(failed_locs.count(loc)) {
 941                                         continue;
 942                                 }
 943 
 944                                 const int ranking = rank_castle_location(x, y, terrain_tester, min_x, max_x, min_y, max_y, min_distance, castles, best_ranking);
 945                                 if(ranking <= 0) {
 946                                         failed_locs.insert(loc);
 947                                 }
 948 
 949                                 if(ranking > best_ranking) {
 950                                         best_ranking = ranking;
 951                                         best_loc = loc;
 952                                 }
 953                         }
 954                 }
 955 
 956                 if(best_ranking == 0) {
 957                         ERR_NG << "No castle location found, aborting." << std::endl;
 958                         const std::string error = _("No valid castle location found. Too many or too few mountain hexes? (please check the 'max hill size' parameter)");
 959                         throw mapgen_exception(error);
 960                 }
 961 
 962                 assert(std::find(castles.begin(), castles.end(), best_loc) == castles.end());
 963                 castles.push_back(best_loc);
 964 
 965                 // Make sure the location can't get a second castle.
 966                 failed_locs.insert(best_loc);
 967         }
 968 
 969         LOG_NG << "Placing roads...\n";
 970         ticks = SDL_GetTicks();
 971 
 972         // Place roads.
 973         // We select two tiles at random locations on the borders of the map
 974         // and try to build roads between them.
 975         int nroads = cfg["roads"];
 976         if(data.link_castles) {
 977                 nroads += castles.size()*castles.size();
 978         }
 979 
 980         std::set<map_location> bridges;
 981 
 982         road_path_calculator calc(terrain, cfg, rng_());
 983         for(int road = 0; road != nroads; ++road) {
 984                 lg::scope_logger another_inner_scope_logging_object__(lg::general(), "creating road");
 985 
 986                 /*
 987                  * We want the locations to be on the portion of the map we're actually
 988                  * going to use, since roads on other parts of the map won't have any
 989                  * influence, and doing it like this will be quicker.
 990                  */
 991                 map_location src = random_point_at_side(data.width/3 + 2,data.height/3 + 2);
 992                 map_location dst = random_point_at_side(data.width/3 + 2,data.height/3 + 2);
 993 
 994                 src.x += data.width/3 - 1;
 995                 src.y += data.height/3 - 1;
 996                 dst.x += data.width/3 - 1;
 997                 dst.y += data.height/3 - 1;
 998 
 999                 if(data.link_castles && road < int(castles.size() * castles.size())) {
1000                         const size_t src_castle = road/castles.size();
1001                         const size_t dst_castle = road%castles.size();
1002                         if(src_castle >= dst_castle) {
1003                                 continue;
1004                         }
1005 
1006                         src = castles[src_castle];
1007                         dst = castles[dst_castle];
1008                 } else if(src.x == dst.x || src.y == dst.y) {
1009                         // If the road isn't very interesting (on the same border), don't draw it.
1010                         continue;
1011                 }
1012 
1013                 if(calc.cost(src, 0.0) >= 1000.0 || calc.cost(dst, 0.0) >= 1000.0) {
1014                         continue;
1015                 }
1016 
1017                 // Search a path out for the road
1018                 pathfind::plain_route rt = pathfind::a_star_search(src, dst, 10000.0, calc, data.width, data.height);
1019 
1020                 const std::string& road_base_name = misc_labels != nullptr
1021                         ? base_name_generator->generate()
1022                         : "";
1023                 const std::string& road_name = misc_labels != nullptr
1024                         ? road_name_generator->generate({{"base", road_base_name}})
1025                         : "";
1026                 const int name_frequency = 20;
1027                 int name_count = 0;
1028 
1029                 bool on_bridge = false;
1030 
1031                 // Draw the road.
1032                 // If the search failed, rt.steps will simply be empty.
1033                 for(std::vector<map_location>::const_iterator step = rt.steps.begin();
1034                                 step != rt.steps.end(); ++step) {
1035 
1036                         const int x = step->x;
1037                         const int y = step->y;
1038 
1039                         if(x < 0 || y < 0 || x >= static_cast<long>(data.width) || y >= static_cast<long>(data.height)) {
1040                                 continue;
1041                         }
1042 
1043                         // Find the configuration which tells us what to convert this tile to, to make it into a road.
1044                         const config& child = cfg.find_child("road_cost", "terrain", t_translation::write_terrain_code(terrain[x][y]));
1045                         if(child.empty()){
1046                                 continue;
1047                         }
1048 
1049                         /* Convert to bridge means that we want to convert depending on the direction of the road.
1050                          * Typically it will be in a format like convert_to_bridge = \,|,/
1051                          * '|' will be used if the road is going north-south
1052                          * '/' will be used if the road is going south west-north east
1053                          * '\' will be used if the road is going south east-north west
1054                          * The terrain will be left unchanged otherwise (if there is no clear direction).
1055                          */
1056                         const std::string& convert_to_bridge = child["convert_to_bridge"];
1057                         if(!convert_to_bridge.empty()) {
1058                                 if(step == rt.steps.begin() || step+1 == rt.steps.end()) {
1059                                         continue;
1060                                 }
1061 
1062                                 const map_location& last = *(step-1);
1063                                 const map_location& next = *(step+1);
1064 
1065                                 map_location adj[6];
1066                                 get_adjacent_tiles(*step,adj);
1067 
1068                                 int direction = -1;
1069 
1070                                 // If we are going north-south
1071                                 if((last == adj[0] && next == adj[3]) || (last == adj[3] && next == adj[0])) {
1072                                         direction = 0;
1073                                 }
1074 
1075                                 // If we are going south west-north east
1076                                 else if((last == adj[1] && next == adj[4]) || (last == adj[4] && next == adj[1])) {
1077                                         direction = 1;
1078                                 }
1079 
1080                                 // If we are going south east-north west
1081                                 else if((last == adj[2] && next == adj[5]) || (last == adj[5] && next == adj[2])) {
1082                                         direction = 2;
1083                                 }
1084 
1085                                 if(misc_labels != nullptr && !on_bridge) {
1086                                         on_bridge = true;
1087                                         std::string bridge_base_name = base_name_generator->generate();
1088                                         const std::string& name = bridge_name_generator->generate({{"base",  bridge_base_name}});
1089                                         const map_location loc(x - data.width / 3, y-data.height/3);
1090                                         misc_labels->emplace(loc, name);
1091                                         bridge_names.emplace(loc, bridge_base_name); //add to use for village naming
1092                                         bridges.insert(loc);
1093                                 }
1094 
1095                                 if(direction != -1) {
1096                                         const std::vector<std::string> items = utils::split(convert_to_bridge);
1097                                         if(size_t(direction) < items.size() && !items[direction].empty()) {
1098                                                 terrain[x][y] = t_translation::read_terrain_code(items[direction]);
1099                                         }
1100 
1101                                         continue;
1102                                 }
1103                         } else {
1104                                 on_bridge = false;
1105                         }
1106 
1107                         // Just a plain terrain substitution for a road
1108                         const std::string& convert_to = child["convert_to"];
1109                         if(!convert_to.empty()) {
1110                                 const t_translation::terrain_code letter = t_translation::read_terrain_code(convert_to);
1111                                 if(misc_labels != nullptr && terrain[x][y] != letter && name_count++ == name_frequency && !on_bridge) {
1112                                         misc_labels->emplace(map_location(x - data.width / 3, y - data.height / 3), road_name);
1113                                         name_count = 0;
1114                                 }
1115 
1116                                 terrain[x][y] = letter;
1117                                 if(misc_labels != nullptr) {
1118                                         const map_location loc(x - data.width / 3, y - data.height / 3); //add to use for village naming
1119                                         if(road_base_name != "")
1120                                                 road_names.emplace(loc, road_base_name);
1121                                 }
1122                         }
1123                 }
1124         }
1125 
1126         // Now that road drawing is done, we can plonk down the castles.
1127         for(std::vector<map_location>::const_iterator c = castles.begin(); c != castles.end(); ++c) {
1128                 if(!c->valid()) {
1129                         continue;
1130                 }
1131 
1132                 const int x = c->x;
1133                 const int y = c->y;
1134                 const int player = c - castles.begin() + 1;
1135                 const t_translation::coordinate coord(x, y);
1136                 starting_positions.insert(t_translation::starting_positions::value_type(std::to_string(player), coord));
1137                 terrain[x][y] = t_translation::HUMAN_KEEP;
1138
