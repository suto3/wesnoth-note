 117 void user_info::update_state(int selected_game_id,
 118                                                          const room_info* current_room /*= nullptr*/)
 119 {
 120         if(game_id != 0) {
 121                 if(game_id == selected_game_id) {
 122                         state = SEL_GAME;
 123                 } else {
 124                         state = GAME;
 125                 }
 126         } else {
 127                 if(current_room != nullptr && current_room->is_member(name)) {
 128                         state = SEL_ROOM;
 129                 } else {
 130                         state = LOBBY;
 131                 }
 132         }
 133         update_relation();
 134 }

 136 void user_info::update_relation()
 137 {
 138         if(name == preferences::login()) {
 139                 relation = ME;
 140         } else if(preferences::is_ignored(name)) {
 141                 relation = IGNORED;
 142         } else if(preferences::is_friend(name)) {
 143                 relation = FRIEND;
 144         } else {
 145                 relation = NEUTRAL;
 146         }

 149 // Returns an abbreviated form of the provided string - ie, 'Ageless Era' should become 'AE'
 150 static std::string make_short_name(const std::string& long_name)
 151 {
 152         if(long_name.empty()) {
 153                 return "";
 154         }
 155 
 156         size_t pos = 0;
 157 
 158         std::stringstream ss;
 159         ss << long_name[pos];
 160 
 161         while(pos < long_name.size()) {
 162                 pos = long_name.find(' ', pos + 1);
 163 
 164                 if(pos <= long_name.size() - 2) {
 165                         ss << long_name[pos + 1];
 166                 }
 167         }
 168 
 169         return ss.str();
 170 }

 172 game_info::game_info(const config& game, const config& game_config, const std::vector<std::string>& installed_addons)
 173         : id(game["id"])
 174         , map_data(game["map_data"])
 175         , name(game["name"])
 176         , scenario()
 177         , remote_scenario(false)
 178         , map_info()
 179         , map_size_info()
 180         , era()
 181         , era_short()
 182         , gold(game["mp_village_gold"])
 183         , support(game["mp_village_support"])
 184         , xp(game["experience_modifier"].str() + "%")
 185         , vision()
 186         , status()
 187         , time_limit()
 188         , vacant_slots(lexical_cast_default<int>(game["slots"], 0)) // Can't use to_int() here.
 189         , current_turn(0)
 190         , reloaded(game["savegame"].to_bool())
 191         , started(false)
 192         , fog(game["mp_fog"].to_bool())
 193         , shroud(game["mp_shroud"].to_bool())
 194         , observers(game["observer"].to_bool(true))
 195         , shuffle_sides(game["shuffle_sides"].to_bool(true))
 196         , use_map_settings(game["mp_use_map_settings"].to_bool())
 197         , registered_users_only(game["registered_users_only"].to_bool())
 198         , verified(true)
 199         , password_required(game["password"].to_bool())
 200         , have_era(true)
 201         , have_all_mods(true)
 202         , has_friends(false)
 203         , has_ignored(false)
 204         , display_status(NEW)
 205         , required_addons()
 206         , addons_outcome(SATISFIED)
 207 {
 208         const auto parse_requirements = [&](const config& c, const std::string& id_key) {
 209                 if(c.has_attribute(id_key)) {
 210                         if(std::find(installed_addons.begin(), installed_addons.end(), c[id_key].str()) == installed_addons.end()) {
 211                                 required_addon r;
 212                                 r.addon_id = c[id_key].str();
 213                                 r.outcome = NEED_DOWNLOAD;
 214 
 215                                 r.message = vgettext("Missing addon: $id", {{"id", c[id_key].str()}});
 216                                 required_addons.push_back(r);
 217                                 if(addons_outcome == SATISFIED) {
 218                                         addons_outcome = NEED_DOWNLOAD;
 219                                 }
 220                         }
 221                 }
 222         };
 223 
 224         for(const config& addon : game.child_range("addon")) {
 225                 parse_requirements(addon, "id");
 226         }
 227 
 228         /*
 229          * Modifications have a different format than addons. The id and addon_id are keys sent by the
 230          * server, so we have to parse them separately here and add them to the required_addons vector.
 231          */
 232         for(const config& mod : game.child_range("modification")) {
 233                 parse_requirements(mod, "addon_id");
 234         }
 235 
 236         std::string turn = game["turn"];
 237         if(!game["mp_era"].empty()) {
 238                 const config& era_cfg = game_config.find_child("era", "id", game["mp_era"]);
 239                 if(era_cfg) {
 240                         era = era_cfg["name"].str();
 241                         era_short = era_cfg["short_name"].str();
 242                         if(era_short.empty()) {
 243                                 era_short = make_short_name(era);
 244                         }
 245 
 246                         ADDON_REQ result = check_addon_version_compatibility(era_cfg, game);
 247                         addons_outcome = std::max(addons_outcome, result); // Elevate to most severe error level encountered so far
 248                 } else {
 249                         have_era = !game["require_era"].to_bool(true);
 250                         era = vgettext("Unknown era: $era_id", {{"era_id", game["mp_era_addon_id"].str()}});
 251                         era_short = make_short_name(era);
 252                         verified = false;
 253 
 254                         addons_outcome = NEED_DOWNLOAD;
 255                 }
 256         } else {
 257                 era = _("Unknown era");
 258                 era_short = "??";
 259                 verified = false;
 260         }
 261 
 262         std::stringstream info_stream;
 263         info_stream << era;
 264 
 265         if(!game.child_or_empty("modification").empty()) {
 266                 for(const config& cfg : game.child_range("modification")) {
 267                         if(const config& mod = game_config.find_child("modification", "id", cfg["id"])) {
 268                                 mod_info += (mod_info.empty() ? "" : ", ") + mod["name"].str();
 269 
 270                                 if(cfg["require_modification"].to_bool(false)) {
 271                                         ADDON_REQ result = check_addon_version_compatibility(mod, game);
 272                                         addons_outcome = std::max(addons_outcome, result); // Elevate to most severe error level encountered so far
 273                                 }
 274                         } else {
 275                                 mod_info += (mod_info.empty() ? "" : ", ") + cfg["addon_id"].str();
 276 
 277                                 if(cfg["require_modification"].to_bool(false)) {
 278                                         have_all_mods = false;
 279                                         mod_info += " " + _("(missing)");
 280 
 281                                         addons_outcome = NEED_DOWNLOAD;
 282                                 }
 283                         }
 284                 }
 285         }
 286 
 287         if(map_data.empty()) {
 288                 map_data = filesystem::read_map(game["mp_scenario"]);
 289         }
 290 
 291         if(map_data.empty()) {
 292                 info_stream << "  ????";
 293         } else {
 294                 try {
 295                         gamemap map(std::make_shared<terrain_type_data>(game_config), map_data);
 296                         std::ostringstream msi;
 297                         msi << map.w() << font::unicode_multiplication_sign << map.h();
 298                         map_size_info = msi.str();
 299                         info_stream << "  " + map_size_info;
 300                 } catch(incorrect_map_format_error& e) {
 301                         ERR_CF << "illegal map: " << e.message << std::endl;
 302                         verified = false;
 303                 } catch(wml_exception& e) {
 304                         ERR_CF << "map could not be loaded: " << e.dev_message << '\n';
 305                         verified = false;
 306                 }
 307         }
 308         info_stream << " ";
 309 
 310         //
 311         // Check scenarios and campaigns
 312         //
 313         if(!game["mp_scenario"].empty() && game["mp_campaign"].empty()) {
 314                 // Check if it's a multiplayer scenario
 315                 const config* level_cfg = &game_config.find_child("multiplayer", "id", game["mp_scenario"]);
 316 
 317                 // Check if it's a user map
 318                 if(!*level_cfg) {
 319                         level_cfg = &game_config.find_child("generic_multiplayer", "id", game["mp_scenario"]);
 320                 }
 321 
 322                 if(*level_cfg) {
 323                         scenario = formatter() << "<b>" << _("(S)") << "</b>" << " " << (*level_cfg)["name"].str();
 324                         info_stream << scenario;
 325 
 326                         // Reloaded games do not match the original scenario hash, so it makes no sense
 327                         // to test them, since they always would appear as remote scenarios
 328                         if(!reloaded) {
 329                                 if(const config& hashes = game_config.child("multiplayer_hashes")) {
 330                                         std::string hash = game["hash"];
 331                                         bool hash_found = false;
 332                                         for(const auto & i : hashes.attribute_range()) {
 333                                                 if(i.first == game["mp_scenario"] && i.second == hash) {
 334                                                         hash_found = true;
 335                                                         break;
 336                                                 }
 337                                         }
 338 
 339                                         if(!hash_found) {
 340                                                 remote_scenario = true;
 341                                                 info_stream << "  ";
 342                                                 info_stream << _("Remote scenario");
 343                                                 verified = false;
 344                                         }
 345                                 }
 346                         }
 347 
 348                         if((*level_cfg)["require_scenario"].to_bool(false)) {
 349                                 ADDON_REQ result = check_addon_version_compatibility((*level_cfg), game);
 350                                 addons_outcome = std::max(addons_outcome, result); // Elevate to most severe error level encountered so far
 351                         }
 352                 } else {
 353                         scenario = vgettext("Unknown scenario: $scenario_id", {{"scenario_id", game["mp_scenario_name"].str()}});
 354                         info_stream << scenario;
 355                         verified = false;
 356                 }
 357         } else if(!game["mp_campaign"].empty()) {
 358                 if(const config& level_cfg = game_config.find_child("campaign", "id", game["mp_campaign"])) {
 359                         std::stringstream campaign_text;
 360                         campaign_text
 361                                 << "<b>" << _("(C)") << "</b>" << " "
 362                                 << level_cfg["name"] << "  "
 363                                 << game["mp_scenario_name"];
 364 
 365                         // Difficulty
 366                         config difficulties = gui2::dialogs::generate_difficulty_config(level_cfg);
 367                         for(const config& difficulty : difficulties.child_range("difficulty")) {
 368                                 if(difficulty["define"] == game["difficulty_define"]) {
 369                                         campaign_text << "  " << difficulty["description"];
 370 
 371                                         break;
 372                                 }
 373                         }
 374 
 375                         scenario = campaign_text.str();
 376                         info_stream << campaign_text.rdbuf();
 377 
 378                         // TODO: should we have this?
 379                         //if(game["require_scenario"].to_bool(false)) {
 380                                 ADDON_REQ result = check_addon_version_compatibility(level_cfg, game);
 381                                 addons_outcome = std::max(addons_outcome, result); // Elevate to most severe error level encountered so far
 382                         //}
 383                 } else {
 384                         scenario = vgettext("Unknown campaign: $campaign_id", {{"campaign_id", game["mp_campaign"].str()}});
 385                         info_stream << scenario;
 386                         verified = false;
 387                 }
 388         } else {
 389                 scenario = _("Unknown scenario");
 390                 info_stream << scenario;
 391                 verified = false;
 392         }
 393 
 394         // Remove any newlines that might have been in game titles
 395         boost::replace_all(scenario, "\n", " " + font::unicode_em_dash + " ");
 396 
 397         if(reloaded) {
 398                 info_stream << "  ";
 399                 info_stream << _("Reloaded game");
 400                 verified = false;
 401         }
 402 
 403         if(!turn.empty()) {
 404                 started = true;
 405                 int index = turn.find_first_of('/');
 406                 if(index > -1) {
 407                         const std::string current_turn_string = turn.substr(0, index);
 408                         current_turn = lexical_cast<unsigned int>(current_turn_string);
 409                 }
 410                 status = _("Turn") + " " + turn;
 411         } else {
 412                 started = false;
 413                 if(vacant_slots > 0) {
 414                         status = _n("Vacant Slot:", "Vacant Slots:", vacant_slots) + " " + game["slots"];
 415                 }
 416         }
 417 
 418         if(fog) {
 419                 vision = _("Fog");
 420                 if(shroud) {
 421                         vision += "/";
 422                         vision += _("Shroud");
 423                 }
 424         } else if(shroud) {
 425                 vision = _("Shroud");
 426         } else {
 427                 vision = _("none");
 428         }
 429 
 430         if(game["mp_countdown"].to_bool()) {
 431                 time_limit = formatter()
 432                         << game["mp_countdown_init_time"].str() << "+"
 433                         << game["mp_countdown_turn_bonus"].str() << "/"
 434                         << game["mp_countdown_action_bonus"].str();
 435         }
 436 
 437         map_info = info_stream.str();
 438 }

 440 game_info::ADDON_REQ game_info::check_addon_version_compatibility(const config& local_item, const config& game)
 441 {
 442         if(!local_item.has_attribute("addon_id") || !local_item.has_attribute("addon_version")) {
 443                 return SATISFIED;
 444         }
 445 
 446         if(const config& game_req = game.find_child("addon", "id", local_item["addon_id"])) {
 447                 required_addon r {local_item["addon_id"].str(), SATISFIED, ""};
 448 
 449                 // Local version
 450                 const version_info local_ver(local_item["addon_version"].str());
 451                 version_info local_min_ver(local_item.has_attribute("addon_min_version") ? local_item["addon_min_version"] : local_item["addon_version"]);
 452 
 453                 // If the UMC didn't specify last compatible version, assume no backwards compatibility.
 454                 // Also apply some sanity checking regarding min version; if the min ver doens't make sense, ignore it.
 455                 local_min_ver = std::min(local_min_ver, local_ver);
 456 
 457                 // Remote version
 458                 const version_info remote_ver(game_req["version"].str());
 459                 version_info remote_min_ver(game_req.has_attribute("min_version") ? game_req["min_version"] : game_req["version"]);
 460 
 461                 remote_min_ver = std::min(remote_min_ver, remote_ver);
 462 
 463                 // Check if the host is too out of date to play.
 464                 if(local_min_ver > remote_ver) {
 465                         r.outcome = CANNOT_SATISFY;
 466 
 467                         // TODO: Figure out how to ask the add-on manager for the user-friendly name of this add-on.
 468                         r.message = vgettext("The host's version of <i>$addon</i> is incompatible. They have version <b>$host_ver</b> while you have version <b>$local_ver</b>.", {
 469                                 {"addon",     r.addon_id},
 470                                 {"host_ver",  remote_ver.str()},
 471                                 {"local_ver", local_ver.str()}
 472                         });
 473 
 474                         required_addons.push_back(r);
 475                         return r.outcome;
 476                 }
 477 
 478                 // Check if our version is too out of date to play.
 479                 if(remote_min_ver > local_ver) {
 480                         r.outcome = NEED_DOWNLOAD;
 481 
 482                         // TODO: Figure out how to ask the add-on manager for the user-friendly name of this add-on.
 483                         r.message = vgettext("Your version of <i>$addon</i> is incompatible. You have version <b>$local_ver</b> while the host has version <b>$host_ver</b>.", {
 484                                 {"addon", r.addon_id},
 485                                 {"host_ver", remote_ver.str()},
 486                                 {"local_ver", local_ver.str()}
 487                         });
 488 
 489                         required_addons.push_back(r);
 490                         return r.outcome;
 491                 }
 492         }
 493 
 494         return SATISFIED;
 495 }

 524 bool game_info::match_string_filter(const std::string& filter) const
 525 {
 526         const std::string& s1 = map_info;
 527         const std::string& s2 = name;
 528         return std::search(s1.begin(), s1.end(), filter.begin(), filter.end(),
 529                         chars_equal_insensitive) != s1.end()
 530             || std::search(s2.begin(), s2.end(), filter.begin(), filter.end(),
 531                         chars_equal_insensitive) != s2.end();
 532 }
