  26 /** Small struct to store and manipulate ToD color adjusts. */
  27 // This is a color delta, so do not replace with color_t!
  28 struct tod_color {
  29         explicit tod_color(int red = 0, int green = 0, int blue = 0)
  30                 : r(utils::clamp(red, -255, 255))
  31                 , g(utils::clamp(green, -255, 255))
  32                 , b(utils::clamp(blue, -255, 255))
  33         {}
  34         bool operator==(const tod_color& o) const {
  35                 return r == o.r && g == o.g && b == o.b;
  36         }
  37         bool is_zero() const {
  38                 return r == 0 && g == 0 && b == 0;
  39         }
  40         bool operator!=(const tod_color& o) const {
  41                 return !operator==(o);
  42         }
  43         tod_color operator+(const tod_color& o) const {
  44                 return tod_color(r + o.r, g + o.g, b + o.b);
  45         }
  46 
  47         int r,g,b;
  48 };

  73                         && image == o.image
  74                         && name == o.name
  75                         && id == o.id
  76                         && image_mask == o.image_mask
  77                         //&& color == o.color
  78                         && sounds == o.sounds;
  79         }
