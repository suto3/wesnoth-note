  56 std::string addon_list::colorify_addon_state_string(const std::string& str, ADDON_STATUS state, bool verbose)
  57 {
  58         color_t colorname = font::NORMAL_COLOR;
  59 
  60         switch(state) {
  61         case ADDON_NONE:
  62                 if(!verbose) {
  63                         return str;
  64                 }
  65                 colorname = font::weapon_details_color;
  66                 break;
  67         case ADDON_INSTALLED:
  68         case ADDON_INSTALLED_LOCAL_ONLY:
  69         case ADDON_NOT_TRACKED:
  70                 colorname = font::GOOD_COLOR;
  71                 break;
  72         case ADDON_INSTALLED_UPGRADABLE:
  73                 colorname = font::YELLOW_COLOR;
  74                 break;
  75         case ADDON_INSTALLED_OUTDATED:
  76                 colorname = color_outdated;
  77                 break;
  78         case ADDON_INSTALLED_BROKEN:
  79                 colorname = font::BAD_COLOR;
  80                 break;
  81         default:
  82                 colorname = font::GRAY_COLOR;
  83                 break;
  84         }
  85 
  86         return font::span_color(colorname) + str + "</span>";
  87 }

  89 std::string addon_list::describe_status(const addon_tracking_info& info)
  90 {
  91         std::string tc, tx;
  92 
  93         switch(info.state) {
  94         case ADDON_NONE:
  95                 tx = info.can_publish ? _("addon_state^Published, not installed") : _("addon_state^Not installed");
  96                 break;
  97 
  98         case ADDON_INSTALLED:
  99         case ADDON_NOT_TRACKED:
 100                 // Consider add-ons without version information as installed
 101                 // for the main display. Their Description info should elaborate
 102                 // on their status.
 103                 tx = info.can_publish ? _("addon_state^Published") : _("addon_state^Installed");
 104                 break;
 105         case ADDON_INSTALLED_LOCAL_ONLY:
 106                 tx = info.can_publish ? _("addon_state^Ready to publish") : _("addon_state^Installed, not ready to publish");
 107                 break;
 108 
 109         case ADDON_INSTALLED_UPGRADABLE:
 110                 tx = info.can_publish ? _("addon_state^Published, upgradable") : _("addon_state^Installed, upgradable");
 111                 break;
 112 
 113         case ADDON_INSTALLED_OUTDATED:
 114                 tx = info.can_publish ? _("addon_state^Published, outdated on server") : _("addon_state^Installed, outdated on server");
 115                 break;
 116 
 117         case ADDON_INSTALLED_BROKEN:
 118                 tx = info.can_publish ? _("addon_state^Published, broken") : _("addon_state^Installed, broken");
 119                 break;
 120 
 121         default:
 122                 tx = _("addon_state^Unknown");
 123         }
 124 
 125         return colorify_addon_state_string(tx, info.state, true);
 126 }

 128 void addon_list::set_addons(const addons_list& addons)
 129 {
 130         listbox& list = get_listbox();
 131         list.clear();
 132 
 133         addon_vector_.clear();
 134 
 135         for(const auto& a : addons) {
 136                 const addon_info& addon = a.second;
 137                 addon_tracking_info tracking_info = get_addon_tracking_info(addon);
 138 
 139                 addon_vector_.push_back(&addon);
 140 
 141                 std::map<std::string, string_map> data;
 142                 string_map item;
 143 
 144                 if(!tracking_info.can_publish) {
 145                         item["label"] = addon.display_icon();
 146                         data.emplace("icon", item);
 147 
 148                         item["label"] = addon.display_title();
 149                         data.emplace("name", item);
 150                 } else {
 151                         item["label"] = addon.display_icon() + "~BLIT(icons/icon-addon-publish.png)";
 152                         data.emplace("icon", item);
 153 
 154                         const std::string publish_name = formatter()
 155                                 << "<span color='#00ff00'>" // GOOD_COLOR
 156                                 << addon.display_title()
 157                                 << "</span>";
 158 
 159                         item["label"] = publish_name;
 160                         data.emplace("name", item);
 161                 }
 162 
 163                 item["label"] = describe_status(tracking_info);
 164                 data.emplace("installation_status", item);
 165 
 166                 // If the addon is upgradable or ourdated on server, we display the two relevant
 167                 // versions directly in the list for convenience.
 168                 const bool special_version_display =
 169                         tracking_info.state == ADDON_INSTALLED_UPGRADABLE ||
 170                         tracking_info.state == ADDON_INSTALLED_OUTDATED;
 171 
 172                 std::ostringstream ss;
 173                 if(special_version_display) {
 174                         ss << tracking_info.installed_version.str() << "\n";
 175                 }
 176 
 177                 ss << addon.version.str();
 178 
 179                 if(special_version_display) {
 180                         ss.str(colorify_addon_state_string(ss.str(), tracking_info.state, false));
 181                 }
 182 
 183                 item["label"] = ss.str();
 184                 data.emplace("version", item);
 185 
 186                 item["label"] = addon.author;
 187                 data.emplace("author", item);
 188 
 189                 item["label"] = size_display_string(addon.size);
 190                 data.emplace("size", item);
 191 
 192                 item["label"] = std::to_string(addon.downloads);
 193                 data.emplace("downloads", item);
 194 
 195                 item["label"] = addon.display_type();
 196                 data.emplace("type", item);
 197 
 198                 grid* row_grid = &list.add_row(data);
 199 
 200                 // Set special retval for the toggle panels
 201                 find_widget<toggle_panel>(row_grid, "list_panel", false).set_retval(DEFAULT_ACTION_RETVAL);
 202 
 203                 grid* control_grid = find_widget<grid>(row_grid, "single_install_buttons", false, false);
 204                 if(!control_grid) {
 205                         continue;
 206                 }
 207                 stacked_widget& install_update_stack = find_widget<stacked_widget>(row_grid, "install_update_stack", false);
 208 
 209                 if(!tracking_info.can_publish) {
 210                         const bool is_updatable = tracking_info.state == ADDON_INSTALLED_UPGRADABLE;
 211                         const bool is_installed =
 212                                 tracking_info.state == ADDON_INSTALLED || tracking_info.state == ADDON_INSTALLED_UPGRADABLE;
 213 
 214                         install_update_stack.select_layer(static_cast<int>(is_updatable));
 215 
 216                         if(!is_updatable) {
 217                                 find_widget<button>(control_grid, "single_install", false).set_active(!is_installed);
 218                                 if(install_function_ != nullptr) {
 219                                         gui2::event::connect_signal_mouse_left_click(
 220                                                 find_widget<button>(control_grid, "single_install", false),
 221                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 222                                         {
 223                                                 install_function_(addon);
 224                                                 handled = true;
 225                                                 halt = true;
 226                                         });
 227                                 }
 228                         } else {
 229                                 find_widget<button>(control_grid, "single_update", false).set_active(true);
 230                                 if(update_function_ != nullptr) {
 231                                         gui2::event::connect_signal_mouse_left_click(
 232                                                 find_widget<button>(control_grid, "single_update", false),
 233                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 234                                         {
 235                                                 update_function_(addon);
 236                                                 handled = true;
 237                                                 halt = true;
 238                                         });
 239                                 }
 240                         }
 241 
 242                         if(is_installed) {
 243                                 if(uninstall_function_ != nullptr) {
 244                                         gui2::event::connect_signal_mouse_left_click(
 245                                                 find_widget<button>(control_grid, "single_uninstall", false),
 246                                                 [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 247                                         {
 248                                                 uninstall_function_(addon);
 249                                                 handled = true;
 250                                                 halt = true;
 251                                         });
 252                                 }
 253                         }
 254 
 255                         find_widget<button>(control_grid, "single_uninstall", false).set_active(is_installed);
 256 
 257                         find_widget<grid>(control_grid, "single_install_buttons", false).set_visible(install_buttons_visibility_);
 258                         find_widget<label>(row_grid, "installation_status", false).set_visible(install_status_visibility_);
 259                 } else {
 260                         const bool is_updatable = tracking_info.state == ADDON_INSTALLED_OUTDATED;
 261                         const bool can_delete = !addon.local_only;
 262 
 263                         button& install_button = find_widget<button>(control_grid, "single_install", false);
 264                         button& update_button = find_widget<button>(control_grid, "single_update", false);
 265                         button& uninstall_button = find_widget<button>(control_grid, "single_uninstall", false);
 266 
 267                         install_button.set_active(true);
 268                         update_button.set_active(true);
 269                         uninstall_button.set_active(can_delete);
 270 
 271                         if(true) {
 272                                 gui2::event::connect_signal_mouse_left_click(
 273                                         install_button,
 274                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 275                                 {
 276                                         publish_function_(addon);
 277                                         handled = true;
 278                                         halt = true;
 279                                 });
 280 
 281                                 install_button.set_tooltip(_("Publish add-on"));
 282                         }
 283 
 284                         if(is_updatable) {
 285                                 gui2::event::connect_signal_mouse_left_click(
 286                                         update_button,
 287                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 288                                 {
 289                                         publish_function_(addon);
 290                                         handled = true;
 291                                         halt = true;
 292                                 });
 293 
 294                                 update_button.set_tooltip(_("Send new version to server"));
 295                         }
 296 
 297                         if(can_delete) {
 298                                 gui2::event::connect_signal_mouse_left_click(
 299                                         uninstall_button,
 300                                         [this, addon](gui2::event::dispatcher&, const gui2::event::ui_event, bool& handled, bool& halt)
 301                                 {
 302                                         delete_function_(addon);
 303                                         handled = true;
 304                                         halt = true;
 305                                 });
 306 
 307                                 uninstall_button.set_tooltip(_("Delete add-on from server"));
 308                         }
 309 
 310                         install_update_stack.select_layer(static_cast<int>(is_updatable));
 311                 }
 312         }
 313 
 314         select_first_addon();
 315 }

 317 const addon_info* addon_list::get_selected_addon() const
 318 {
 319         const listbox& list = find_widget<const listbox>(&get_grid(), "addons", false);
 320 
 321         const int index = list.get_selected_row();
 322         if(index == -1 || index >= static_cast<int>(addon_vector_.size())) {
 323                 return nullptr;
 324         }
 325 
 326         return addon_vector_.at(index);
 327 }

 339 void addon_list::select_addon(const std::string& id)
 340 {
 341         listbox& list = get_listbox();
 342 
 343         const addon_info& info = **std::find_if(addon_vector_.begin(), addon_vector_.end(),
 344                 [&id](const addon_info* a)
 345         {
 346                 return a->id == id;
 347         });
 348 
 349         for(unsigned int i = 0u; i < list.get_item_count(); ++i)
 350         {
 351                 grid* row = list.get_row_grid(i);
 352                 const label& name_label = find_widget<label>(row, "name", false);
 353                 if(name_label.get_label().base_str() == info.display_title())
 354                 {
 355                         list.select_row(i);
 356                 }
 357         }
 358 }

 372 void addon_list::finalize_setup()
 373 {
 374         listbox& list = get_listbox();
 375 
 376         list.register_sorting_option(0, [this](const int i) { return addon_vector_[i]->title; });
 377         list.register_sorting_option(1, [this](const int i) { return addon_vector_[i]->author; });
 378         list.register_sorting_option(2, [this](const int i) { return addon_vector_[i]->size; });
 379         list.register_sorting_option(3, [this](const int i) { return addon_vector_[i]->downloads; });
 380         list.register_sorting_option(4, [this](const int i) { return addon_vector_[i]->display_type(); });
 381 
 382         auto order = std::make_pair(0, gui2::listbox::SORT_ASCENDING);
 383         list.set_active_sorting_option(order);
 384 }

 386 void addon_list::select_first_addon()
 387 {
 388         if(addon_vector_.empty()) {
 389                 // Happens in the dialog unit test.
 390                 return;
 391         }
 392 
 393         const addon_info* first_addon = addon_vector_.at(0);
 394         for(const addon_info* a : addon_vector_) {
 395                 if(a->display_title().compare(first_addon->display_title()) < 0) {
 396                         first_addon = a;
 397                 }
 398         }
 399         select_addon(first_addon->id);
 400 }

 426 static widget::visibility parse_visibility(const std::string& str)
 427 {
 428         if(str == "visible") {
 429                 return widget::visibility::visible;
 430         } else if(str == "hidden") {
 431                 return widget::visibility::hidden;
 432         } else if(str == "invisible") {
 433                 return widget::visibility::invisible;
 434         } else {
 435                 FAIL("Invalid visibility value");
 436         }
 437 }
