 179 void configure_engine::write_parameters()
 180 {
 181         config& scenario = this->state_.get_starting_pos();
 182         const mp_game_settings& params = this->state_.mp_settings();
 183 
 184         if (params.random_start_time) {
 185                 if (!tod_manager::is_start_ToD(scenario["random_start_time"])) {
 186                         scenario["random_start_time"] = true;
 187                 }
 188         }
 189         else {
 190                 scenario["random_start_time"] = false;
 191         }
 192         scenario["experience_modifier"] = params.xp_modifier;
 193         scenario["turns"] = params.num_turns;
 194 
 195         for (config& side : scenario.child_range("side"))
 196         {
 197                 side["fog"] = params.fog_game;
 198                 side["shroud"] = params.shroud_game;
 199                 if (!params.use_map_settings) {
 200                         side["village_gold"] = params.village_gold;
 201                         side["village_support"] = params.village_support;
 202                 }
 203         }
 204 }
