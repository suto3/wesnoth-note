  38 repeating_button::repeating_button()
  39         : styled_widget(COUNT), clickable_item(), state_(ENABLED), repeat_timer_(0)
  40 {
  41         connect_signal<event::MOUSE_ENTER>(std::bind(
  42                         &repeating_button::signal_handler_mouse_enter, this, _2, _3));
  43         connect_signal<event::MOUSE_LEAVE>(std::bind(
  44                         &repeating_button::signal_handler_mouse_leave, this, _2, _3));
  45 
  46         connect_signal<event::LEFT_BUTTON_DOWN>(std::bind(
  47                         &repeating_button::signal_handler_left_button_down, this, _2, _3));
  48         connect_signal<event::LEFT_BUTTON_UP>(std::bind(
  49                         &repeating_button::signal_handler_left_button_up, this, _2, _3));
  50 }

  88 void repeating_button::set_state(const state_t state)
  89 {
  90         if(state != state_) {
  91                 state_ = state;
  92                 set_is_dirty(true);
  93 
  94                 if(state_ == DISABLED && repeat_timer_) {
  95                         remove_timer(repeat_timer_);
  96                         repeat_timer_ = 0;
  97                 }
  98         }
  99 }

 125 void
 126 repeating_button::signal_handler_left_button_down(const event::ui_event event,
 127                                                                                                    bool& handled)
 128 {
 129         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 130 
 131         // If the timer isn't set it's the initial down event.
 132         if(!repeat_timer_) {
 133 
 134                 // mimic the old gui and only play the sound once.
 135                 sound::play_UI_sound(settings::sound_button_click);
 136 
 137                 window* window = get_window();
 138                 if(window) {
 139                         repeat_timer_ = add_timer(settings::repeat_button_repeat_time,
 140                                                                           [this, window](unsigned int) {
 141                                                                                         window->fire(event::LEFT_BUTTON_DOWN, *this);
 142                                                                           },true);
 143 
 144                         window->mouse_capture();
 145                 }
 146 
 147                 set_state(PRESSED);
 148         }
 149 
 150         handled = true;
 151 }
 152 
 153 void repeating_button::signal_handler_left_button_up(const event::ui_event event,
 154                                                                                                           bool& handled)
 155 {
 156         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 157 
 158         if(repeat_timer_) {
 159                 remove_timer(repeat_timer_);
 160                 repeat_timer_ = 0;
 161         }
 162 
 163         if(get_active()) {
 164                 set_state(FOCUSED);
 165         }
 166         handled = true;
 167 }
