 213 void modify_grid_with_data(grid* grid, const std::map<std::string, string_map>& map)
 214 {
 215         for(const auto& v : map) {
 216                 const std::string& key = v.first;
 217                 const string_map& strmap = v.second;
 218 
 219                 widget* w = grid->find(key, false);
 220                 if(!w) {
 221                         continue;
 222                 }
 223 
 224                 styled_widget* c = dynamic_cast<styled_widget*>(w);
 225                 if(!c) {
 226                         continue;
 227                 }
 228 
 229                 for(const auto & vv : strmap) {
 230                         if(vv.first == "label") {
 231                                 c->set_label(vv.second);
 232                         } else if(vv.first == "tooltip") {
 233                                 c->set_tooltip(vv.second);
 234                         }
 235                 }
 236         }

 253 void mp_lobby::update_gamelist()
 254 {
 255         SCOPE_LB;
 256         gamelistbox_->clear();
 257         gamelist_id_at_row_.clear();
 258         lobby_info_.make_games_vector();
 259 
 260         int select_row = -1;
 261         for(unsigned i = 0; i < lobby_info_.games().size(); ++i) {
 262                 const mp::game_info& game = *lobby_info_.games()[i];
 263 
 264                 if(game.id == selected_game_id_) {
 265                         select_row = i;
 266                 }
 267 
 268                 gamelist_id_at_row_.push_back(game.id);
 269                 LOG_LB << "Adding game to listbox (1)" << game.id << "\n";
 270                 grid* grid = &gamelistbox_->add_row(make_game_row_data(game));
 271 
 272                 adjust_game_row_contents(game, gamelistbox_->get_item_count() - 1, grid);
 273         }
 274 
 275         if(select_row >= 0 && select_row != gamelistbox_->get_selected_row()) {
 276                 gamelistbox_->select_row(select_row);
 277         }
 278 
 279         update_selected_game();
 280         gamelist_dirty_ = false;
 281         last_gamelist_update_ = SDL_GetTicks();
 282         lobby_info_.sync_games_display_status();
 283         lobby_info_.apply_game_filter();
 284         update_gamelist_header();
 285         gamelistbox_->set_row_shown(lobby_info_.games_visibility());
 286 }

 288 void mp_lobby::update_gamelist_diff()
 289 {
 290         SCOPE_LB;
 291         lobby_info_.make_games_vector();
 292         int select_row = -1;
 293         unsigned list_i = 0;
 294         int list_rows_deleted = 0;
 295 
 296         std::vector<int> next_gamelist_id_at_row;
 297         for(unsigned i = 0; i < lobby_info_.games().size(); ++i) {
 298                 const mp::game_info& game = *lobby_info_.games()[i];
 299 
 300                 if(game.display_status == mp::game_info::NEW) {
 301                         LOG_LB << "Adding game to listbox " << game.id << "\n";
 302 
 303                         if(list_i != gamelistbox_->get_item_count()) {
 304                                 gamelistbox_->add_row(make_game_row_data(game), list_i);
 305                                 DBG_LB << "Added a game listbox row not at the end" << list_i
 306                                            << " " << gamelistbox_->get_item_count() << "\n";
 307                                 list_rows_deleted--;
 308                         } else {
 309                                 gamelistbox_->add_row(make_game_row_data(game));
 310                         }
 311 
 312                         grid* grid = gamelistbox_->get_row_grid(gamelistbox_->get_item_count() - 1);
 313                         adjust_game_row_contents(game, gamelistbox_->get_item_count() - 1, grid);
 314 
 315                         list_i++;
 316                         next_gamelist_id_at_row.push_back(game.id);
 317                 } else {
 318                         if(list_i >= gamelistbox_->get_item_count()) {
 319                                 ERR_LB << "Ran out of listbox items -- triggering a full refresh\n";
 320                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 321                                 return;
 322                         }
 323 
 324                         if(list_i + list_rows_deleted >= gamelist_id_at_row_.size()) {
 325                                 ERR_LB << "gamelist_id_at_row_ overflow! " << list_i << " + "
 326                                            << list_rows_deleted
 327                                            << " >= " << gamelist_id_at_row_.size()
 328                                            << " -- triggering a full refresh\n";
 329                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 330                                 return;
 331                         }
 332 
 333                         int listbox_game_id = gamelist_id_at_row_[list_i + list_rows_deleted];
 334                         if(game.id != listbox_game_id) {
 335                                 ERR_LB << "Listbox game id does not match expected id "
 336                                            << listbox_game_id << " " << game.id << " (row " << list_i << ")\n";
 337                                 wesnothd_connection_.send_data(config("refresh_lobby"));
 338                                 return;
 339                         }
 340 
 341                         if(game.display_status == mp::game_info::UPDATED) {
 342                                 LOG_LB << "Modifying game in listbox " << game.id << " (row " << list_i << ")\n";
 343                                 grid* grid = gamelistbox_->get_row_grid(list_i);
 344                                 modify_grid_with_data(grid, make_game_row_data(game));
 345                                 adjust_game_row_contents(game, list_i, grid);
 346                                 ++list_i;
 347                                 next_gamelist_id_at_row.push_back(game.id);
 348                         } else if(game.display_status == mp::game_info::DELETED) {
 349                                 LOG_LB << "Deleting game from listbox " << game.id << " (row "
 350                                            << list_i << ")\n";
 351                                 gamelistbox_->remove_row(list_i);
 352                                 ++list_rows_deleted;
 353                         } else {
 354                                 // clean
 355                                 LOG_LB << "Clean game in listbox " << game.id << " (row " << list_i << ")\n";
 356                                 next_gamelist_id_at_row.push_back(game.id);
 357                                 ++list_i;
 358                         }
 359                 }
 360         }
 361 
 362         for(unsigned i = 0; i < next_gamelist_id_at_row.size(); ++i) {
 363                 if(next_gamelist_id_at_row[i] == selected_game_id_) {
 364                         select_row = i;
 365                 }
 366         }
 367 
 368         next_gamelist_id_at_row.swap(gamelist_id_at_row_);
 369         if(select_row >= static_cast<int>(gamelistbox_->get_item_count())) {
 370                 ERR_LB << "Would select a row beyond the listbox" << select_row << " "
 371                            << gamelistbox_->get_item_count() << "\n";
 372                 select_row = gamelistbox_->get_item_count() - 1;
 373         }
 374 
 375         if(select_row >= 0 && select_row != gamelistbox_->get_selected_row()) {
 376                 gamelistbox_->select_row(select_row);
 377         }
 378 
 379         update_selected_game();
 380         gamelist_dirty_ = false;
 381         last_gamelist_update_ = SDL_GetTicks();
 382         lobby_info_.sync_games_display_status();
 383         lobby_info_.apply_game_filter();
 384         update_gamelist_header();
 385         gamelistbox_->set_row_shown(lobby_info_.games_visibility());
 386 }

 400 std::map<std::string, string_map> mp_lobby::make_game_row_data(const mp::game_info& game)
 401 {
 402         std::map<std::string, string_map> data;
 403 
 404         std::string color_string;
 405         if(game.vacant_slots > 0) {
 406                 color_string = (game.reloaded || game.started) ? "yellow" : "green";
 407         } else {
 408                 color_string = game.observers ? "#ddd" : "red";
 409         }
 410 
 411         if(!game.have_era && (game.vacant_slots > 0 || game.observers)) {
 412                 color_string = "#444";
 413         }
 414 
 415         add_label_data(data, "status", colorize(game.status, color_string));
 416         add_label_data(data, "name",   colorize(game.name, color_string));
 417 
 418         add_label_data(data, "era",             colorize(game.era, "#a69275"));
 419         add_label_data(data, "era_short",       game.era_short);
 420         add_label_data(data, "mods",            colorize(game.mod_info, "#a69275"));
 421         add_label_data(data, "map_info",        game.map_info);
 422         add_label_data(data, "scenario",        game.scenario);
 423         add_label_data(data, "map_size_text",   game.map_size_info);
 424         add_label_data(data, "time_limit",      game.time_limit);
 425         add_label_data(data, "gold_text",       game.gold);
 426         add_label_data(data, "xp_text",         game.xp);
 427         add_label_data(data, "vision_text",     game.vision);
 428         add_label_data(data, "time_limit_text", game.time_limit);
 429         add_label_data(data, "status",          game.status);
 430 
 431         if(game.observers) {
 432                 add_label_data(data, "observer_icon", "misc/eye.png");
 433                 add_tooltip_data(data, "observer_icon", _("Observers allowed"));
 434         } else {
 435                 add_label_data(data, "observer_icon", "misc/no_observer.png");
 436                 add_tooltip_data(data, "observer_icon", _("Observers not allowed"));
 437         }
 438 
 439         std::string vision_icon;
 440         if(game.fog) {
 441                 vision_icon = game.shroud ? "misc/vision-fog-shroud.png" : "misc/vision-fog.png";
 442         } else {
 443                 vision_icon = game.shroud ? "misc/vision-shroud.png" : "misc/vision-none.png";
 444         }
 445 
 446         add_label_data(data, "vision_icon", vision_icon);
 447         add_tooltip_data(data, "vision_icon", game.vision);
 448         return data;
 449 }

 485

 488                                 *observe_button,
 489                                 std::bind(&mp_lobby::observe_global_button_callback,
 490                                                         this,
 491                                                         std::ref(*window_)));
 492                 observe_button->set_active(game.can_observe());
 493         }
 494 
 495         if(minimap* map = dynamic_cast<minimap*>(grid->find("minimap", false))) {
 496                 map->set_config(&game_config_);
 497                 map->set_map_data(game.map_data);
 498         }
 499 }

 513 void mp_lobby::update_playerlist()
 514 {
 515         if(delay_playerlist_update_) {
 516                 return;
 517         }
 518 
 519         SCOPE_LB;
 520         DBG_LB << "Playerlist update: " << lobby_info_.users().size() << "\n";
 521         lobby_info_.update_user_statuses(selected_game_id_, chatbox_->active_window_room());
 522         lobby_info_.sort_users(player_list_.sort_by_name->get_value_bool(),
 523                                                    player_list_.sort_by_relation->get_value_bool());
 524 
 525         bool lobby = false;
 526         if(mp::room_info* ri = chatbox_->active_window_room()) {
 527                 if(ri->name() == "lobby") {
 528                         lobby = true;
 529                 }
 530         }
 531 
 532         assert(player_list_.active_game.tree);
 533         assert(player_list_.active_room.tree);
 534         assert(player_list_.other_games.tree);
 535         assert(player_list_.other_rooms.tree);
 536 
 537         player_list_.active_game.tree->clear();
 538         player_list_.active_room.tree->clear();
 539         player_list_.other_games.tree->clear();
 540         player_list_.other_rooms.tree->clear();
 541 
 542         for(auto userptr : lobby_info_.users_sorted()) {
 543                 mp::user_info& user = *userptr;

 801 void mp_lobby::network_handler()
 802 {
 803         try {
 804                 config data;
 805                 if (wesnothd_connection_.receive_data(data)) {
 806                         process_network_data(data);
 807                 }
 808         } catch (wesnothd_error& e) {
 809                 LOG_LB << "caught wesnothd_error in network_handler: " << e.message << "\n";
 810                 throw;
 811         }
 812 
 813         if(gamelist_dirty_ && !delay_gamelist_update_ && (SDL_GetTicks() - last_gamelist_update_ > game_config::lobby_refresh)) {
 814                 if(gamelist_diff_update_) {
 815                         update_gamelist_diff();
 816                 } else {
 817                         update_gamelist();
 818                         gamelist_diff_update_ = true;
 819                 }
 820         }
 821 
 822         if(player_list_dirty_) {
 823                 update_gamelist_filter();
 824                 update_playerlist();
 825         }
 826 }

 828 void mp_lobby::process_network_data(const config& data)
 829 {
 830         if(const config& error = data.child("error")) {
 831                 throw wesnothd_error(error["message"]);
 832         } else if(data.child("gamelist")) {
 833                 process_gamelist(data);
 834         } else if(const config& gamelist_diff = data.child("gamelist_diff")) {
 835                 process_gamelist_diff(gamelist_diff);
 836         }
 837 
 838         chatbox_->process_network_data(data);
 839 }

 849 void mp_lobby::process_gamelist_diff(const config& data)
 850 {
 851         if(lobby_info_.process_gamelist_diff(data)) {
 852                 DBG_LB << "Received gamelist diff\n";
 853                 gamelist_dirty_ = true;
 854         } else {
 855                 ERR_LB << "process_gamelist_diff failed!" << std::endl;
 856                 wesnothd_connection_.send_data(config("refresh_lobby"));
 857         }
 858         const int joined = data.child_count("insert_child");
 859         const int left = data.child_count("remove_child");
 860         if(joined > 0 || left > 0) {
 861                 if(left > joined) {
 862                         do_notify(mp::NOTIFY_LOBBY_QUIT);
 863                 } else {
 864                         do_notify(mp::NOTIFY_LOBBY_JOIN);
 865                 }
 866         }

 891 static bool handle_addon_requirements_gui(CVideo& v, const std::vector<mp::game_info::required_addon>& reqs, mp::game_info::ADDON_REQ addon_outcome)
 892 {
 893         if(addon_outcome == mp::game_info::CANNOT_SATISFY) {
 894                 std::string e_title = _("Incompatible User-made Content.");
 895                 std::string err_msg = _("This game cannot be joined because the host has out-of-date add-ons that are incompatible with your version. You might wish to suggest that the host's add-ons be updated.");
 896 
 897                 err_msg +="\n\n";
 898                 err_msg += _("Details:");
 899                 err_msg += "\n";
 900 
 901                 for(const mp::game_info::required_addon & a : reqs) {
 902                         if (a.outcome == mp::game_info::CANNOT_SATISFY) {
 903                                 err_msg += font::unicode_bullet + " " + a.message + "\n";
 904                         }
 905                 }
 906                 gui2::show_message(v, e_title, err_msg, message::auto_close);
 907 
 908                 return false;
 909         } else if(addon_outcome == mp::game_info::NEED_DOWNLOAD) {
 910                 std::string e_title = _("Missing User-made Content.");
 911                 std::string err_msg = _("This game requires one or more user-made addons to be installed or updated in order to join.\nDo you want to try to install them?");
 912 
 913                 err_msg +="\n\n";
 914                 err_msg += _("Details:");
 915                 err_msg += "\n";
 916 
 917                 std::vector<std::string> needs_download;
 918                 for(const mp::game_info::required_addon & a : reqs) {
 919                         if(a.outcome == mp::game_info::NEED_DOWNLOAD) {
 920                                 err_msg += font::unicode_bullet + " " + a.message + "\n";
 921 
 922                                 needs_download.push_back(a.addon_id);
 923                         }
 924                 }
 925 
 926                 assert(needs_download.size() > 0);
 927 
 928                 if(gui2::show_message(v, e_title, err_msg, message::yes_no_buttons, true) == gui2::window::OK) {
 929                         // Begin download session
 930                         ad_hoc_addon_fetch_session(v, needs_download);
 931 
 932                         return true;
 933                 }
 934         }
 935 
 936         return false;
 937 }

 939 bool mp_lobby::do_game_join(int idx, bool observe)
 940 {
 941         if(idx < 0 || idx >= static_cast<int>(lobby_info_.games().size())) {
 942                 ERR_LB << "Requested join/observe of a game with index out of range: "
 943                            << idx << ", games size is " << lobby_info_.games().size() << "\n";
 944                 return false;
 945         }
 946         const mp::game_info& game = *lobby_info_.games()[idx];
 947         if(observe) {
 948                 if(!game.can_observe()) {
 949                         ERR_LB << "Requested observe of a game with observers disabled" << std::endl;
 950                         return false;
 951                 }
 952         } else {
 953                 if(!game.can_join()) {
 954                         ERR_LB << "Requested join to a game with no vacant slots" << std::endl;
 955                         return false;
 956                 }
 957         }
 958 
 959         // Prompt user to download this game's required addons if its requirements have not been met
 960         if(game.addons_outcome != mp::game_info::SATISFIED) {
 961                 if(game.required_addons.empty()) {
 962                         gui2::show_error_message(window_->video(), _("Something is wrong with the addon version check database supporting the multiplayer lobby. Please report this at http://bugs.wesnoth.org."));
 963                         return false;
 964                 }
 965 
 966                 if(!handle_addon_requirements_gui(window_->video(), game.required_addons, game.addons_outcome)) {
 967                         return false;
 968                 } else {
 969                         // Addons have been downloaded, so the game_config and installed addons list need to be reloaded.
 970                         // The lobby is closed and reopened.
 971                         window_->set_retval(RELOAD_CONFIG);
 972                         return false;
 973                 }
 974         }
 975 
 976         config response;
 977         config& join = response.add_child("join");
 978         join["id"] = std::to_string(game.id);
 979         join["observe"] = observe;
 980         if(join && !observe && game.password_required) {
 981                 std::string password;
 982                 if(!gui2::dialogs::mp_join_game_password_prompt::execute(password, window_->video())) {
 983                         return false;
 984                 }
 985 
 986                 join["password"] = password;
 987         }
 988 
 989         wesnothd_connection_.send_data(response);
 990         joined_game_id_ = game.id;
 991         return true;
 992 }
