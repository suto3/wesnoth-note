  49 sub send_data {
  50         my $doc = shift @_;
  51         return unless @_;
  52         print "SENDING: {{{" . &wml::write_text($doc) . "}}}\n";
  53         foreach my $socket (@_) {
  54                 my $sock = $socks_map{$socket};
  55                 my $packet = &wml::write_binary($outgoing_schemas{$sock}, $doc) . chr 0;
  56                 my $header = pack('N',length $packet);
  57                 $packet = $header . $packet;
  58                 my $res = syswrite($sock, $packet, length $packet);
  59                 &socket_disconnected($sock) unless $res == length $packet;
  60         }
  61 }

  80         $read_set->add($sock);

  82         $incoming_schemas{$sock} = [];

  87         $flags |= O_NONBLOCK;

 245 sub handshake($) {
 246         my ($sock) = @_;
 247         my $buf = '';
 248         my $res = sysread($sock, $buf, 4);
 249         if($res != 4) {
 250                 &socket_disconnected($sock);
 251         } else {
 252                 my $buf = pack('N', $current_connection_num);
 253                 $connection_nums{$sock} = $current_connection_num++;
 254                 my $res = syswrite($sock, $buf, 4);
 255                 if($res != 4) {
 256                         &socket_disconnected($sock);
 257                 } else {
 258                         delete $pending_handshake{$sock};
 259                         &send_data($login_response, $sock);
 260                 }
 261         }
 262 }
