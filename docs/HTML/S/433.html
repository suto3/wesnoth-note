 117 void tooltip::pre_show(window& window)
 118 {
 119         find_widget<styled_widget>(&window, "label", false).set_label(message_);
 120 
 121         window.set_variable("mouse_x", wfl::variant(mouse_.x));
 122         window.set_variable("mouse_y", wfl::variant(mouse_.y));
 123 
 124         window.set_variable("source_x", wfl::variant(source_rect_.x));
 125         window.set_variable("source_y", wfl::variant(source_rect_.y));
 126         window.set_variable("source_w", wfl::variant(source_rect_.w));
 127         window.set_variable("source_h", wfl::variant(source_rect_.h));
 128 }
 129 
 130 const std::string& tooltip::window_id() const
 131 {
 132         return window_id_;
 133 }

 138 static tooltip& tip()
 139 {
 140         /*
 141          * Allocating a static tip object causes a segmentation fault when Wesnoth
 142          * terminates. So instead create an object on the heap and never free it.
 143          */
 144         static tooltip* t = new tooltip();
 145         return *t;
 146 }
 147 
 148 void show(CVideo& video,
 149                   const std::string& window_id,
 150                   const t_string& message,
 151                   const point& mouse,
 152                   const SDL_Rect& source_rect)
 153 {
 154         /*
 155          * For now allow invalid tip names, might turn them to invalid wml messages
 156          * later on.
 157          */
 158         tooltip& t = tip();
 159         t.set_window_id(window_id);
 160         t.set_message(message);
 161         t.set_mouse(mouse);
 162         t.set_source_rect(source_rect);
 163         try
 164         {
 165                 t.show(video);
 166         }
 167         catch(window_builder_invalid_id&)
 168         {
 169                 ERR_CFG << "Tip with the requested id '" << window_id
 170                                 << "' doesn't exist, fall back to the default.\n";
 171                 t.set_window_id("tooltip_large");
 172                 try
 173                 {
 174                         t.show(video);
 175                 }
 176                 catch(window_builder_invalid_id&)
 177                 {
 178                         ERR_CFG << "Default tooltip doesn't exist, no message shown."
 179                                         << std::endl;
 180                 }
 181         }
 182 }

 184 void remove()
