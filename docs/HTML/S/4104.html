 107     def get_all(self, **kw):
 108         """
 109         This gets all child tags or child attributes of the tag.
 110         For example:
 111 
 112         [unit]
 113             name=A
 114             name=B
 115             [attack]
 116             [/attack]
 117             [attack]
 118             [/attack]
 119         [/unit]
 120 
 121         unit.get_all(att = "name")
 122         will return two nodes for "name=A" and "name=B"
 123 
 124         unit.get_all(tag = "attack")
 125         will return two nodes for the two [attack] tags.
 126 
 127         unit.get_all()
 128         will return 4 nodes for all 4 sub-elements.
 129 
 130         unit.get_all(att = "")
 131         Will return the two attribute nodes.
 132 
 133         unit.get_all(tag = "")
 134         Will return the two tag nodes.
 135 
 136         If no elements are found an empty list is returned.
 137         """
 138         if len(kw) == 1 and "tag" in kw and kw["tag"]:
 139             return self.speedy_tags.get(kw["tag"], [])
 140 
 141         r = []
 142         for sub in self.data:
 143             ok = True
 144             for k, v in kw.items():
 145                 if k == "tag":
 146                     if not isinstance(sub, TagNode): ok = False
 147                     elif v != "" and sub.name != v: ok = False
 148                 elif k == "att":
 149                     if not isinstance(sub, AttributeNode): ok = False
 150                     elif v != "" and sub.name != v: ok = False
 151             if ok:
 152                 r.append(sub)
 153         return r

 274     def parse_line_without_commands(self, line):
 275         """
 276         Once the .plain commands are handled WML lines are passed to
 277         this.
 278         """
 279         if not line: return
 280 
 281         if line.strip():
 282             self.skip_newlines_after_plus = False
 283 
 284         if self.in_tag:
 285             self.handle_tag(line)
 286             return
 287 
 288         if self.in_arrows:
 289             arrows = line.find('>>')
 290             if arrows >= 0:
 291                 self.in_arrows = False
 292                 self.temp_string += line[:arrows]
 293                 self.temp_string_node = StringNode(self.temp_string)
 294                 self.temp_string = ""
 295                 self.temp_key_nodes[self.commas].value.append(
 296                     self.temp_string_node)
 297                 self.in_arrows = False
 298                 self.parse_line_without_commands(line[arrows + 2:])
 299             else:
 300                 self.temp_string += line
 301             return
 302 
 303         quote = line.find('"')
 304 
 305         if not self.in_string:
 306             arrows = line.find('<<')
 307             if arrows >= 0 and (quote < 0 or quote > arrows):
 308                 self.parse_line_without_commands(line[:arrows])
 309                 self.in_arrows = True
 310                 self.parse_line_without_commands(line[arrows + 2:])
 311                 return
 312 
 313         if quote >= 0:
 314             if self.in_string:
 315                 # double quote
 316                 if quote < len(line) - 1 and line[quote + 1] == '"':
 317                     self.temp_string += line[:quote + 1]
 318                     self.parse_line_without_commands(line[quote + 2:])
 319                     return
 320                 self.temp_string += line[:quote]
 321                 self.temp_string_node = StringNode(self.temp_string)
 322                 if self.translatable:
 323                     self.temp_string_node.textdomain = self.textdomain
 324                     self.translatable = False
 325                 self.temp_string = ""
 326                 if not self.temp_key_nodes:
 327                     raise WMLError(self, "Unexpected string value.")
 328 
 329                 self.temp_key_nodes[self.commas].value.append(
 330                     self.temp_string_node)
 331 
 332                 self.in_string = False
 333                 self.parse_line_without_commands(line[quote + 1:])
 334             else:
 335                 self.parse_outside_strings(line[:quote])
 336                 self.in_string = True
 337                 self.parse_line_without_commands(line[quote + 1:])
 338         else:
 339             if self.in_string:
 340                 self.temp_string += line
 341             else:
 342                 self.parse_outside_strings(line)

 344     def parse_outside_strings(self, line):
 345         """
 346         Parse a WML fragment outside of strings.
 347         """
 348         if not line: return
 349         if not self.temp_key_nodes:
 350             line = line.lstrip()
 351             if not line: return
 352 
 353             if line.startswith("#textdomain "):
 354                 self.textdomain = line[12:].strip()
 355                 return
 356 
 357             # Is it a tag?
 358             if line[0] == "[":
 359                 self.handle_tag(line)
 360             # No tag, must be an attribute.
 361             else:
 362                 self.handle_attribute(line)
 363         else:
 364             for i, segment in enumerate(line.split("+")):
 365                 segment = segment.lstrip(" \t")
 366 
 367                 if i > 0:
 368                     # If the last segment is empty (there was a plus sign
 369                     # at the end) we need to skip newlines.
 370                     self.skip_newlines_after_plus = not segment.strip()
 371 
 372                 if not segment: continue
 373 
 374                 if segment.rstrip() == '_':
 375                     self.translatable = True
 376                     segment = segment[1:].lstrip(" ")
 377                     if not segment: continue
 378                 self.handle_value(segment)

 381     def handle_tag(self, line):
 382         end = line.find("]")
 383         if end < 0:
 384             if line.endswith("\n"):
 385                 raise WMLError(self, "Expected closing bracket.")
 386             self.in_tag += line
 387             return
 388         tag = (self.in_tag + line[:end])[1:]
 389         self.in_tag = ""
 390         if tag[0] == "/":
 391             self.parent_node = self.parent_node[:-1]
 392         else:
 393             node = TagNode(tag, location=(self.line_in_file, self.chunk_start))
 394             if self.parent_node:
 395                 self.parent_node[-1].append(node)
 396             self.parent_node.append(node)
 397         self.parse_outside_strings(line[end + 1:])
 398 
 399     def handle_attribute(self, line):
 400         assign = line.find("=")
 401         remainder = None
 402         if assign >= 0:
 403             remainder = line[assign + 1:]
 404             line = line[:assign]
 405 
 406         self.commas = 0
 407         self.temp_key_nodes = []
 408         for att in line.split(","):
 409             att = att.strip()
 410             node = AttributeNode(att, location=(self.line_in_file, self.chunk_start))
 411             self.temp_key_nodes.append(node)
 412             if self.parent_node:
 413                 self.parent_node[-1].append(node)
 414 
 415         if remainder:
 416             self.parse_outside_strings(remainder)

 418     def handle_value(self, segment):
 419         def add_text(segment):
 420             segment = segment.rstrip()
 421             if not segment: return
 422             n = len(self.temp_key_nodes)
 423             maxsplit = n - self.commas - 1
 424             if maxsplit < 0: maxsplit = 0
 425             for subsegment in segment.split(",", maxsplit):
 426                 self.temp_string += subsegment.strip()
 427                 self.temp_string_node = StringNode(self.temp_string)
 428                 self.temp_string = ""
 429                 self.temp_key_nodes[self.commas].value.append(
 430                     self.temp_string_node)
 431                 if self.commas < n - 1:
 432                     self.commas += 1
 433 
 434         # Finish assignment on newline, except if there is a
 435         # plus sign before the newline.
 436         add_text(segment)
 437         if segment[-1] == "\n" and not self.skip_newlines_after_plus:
 438             self.temp_key_nodes = []

 440     def parse(self):
 441         """
 442         Parse preprocessed WML into a tree of tags and attributes.
 443         """
 444 
 445         # parsing state
 446         self.temp_string = ""
 447         self.temp_string_node = None
 448         self.commas = 0
 449         self.temp_key_nodes = []
 450         self.in_string = False
 451         self.in_arrows = False
 452         self.textdomain = "wesnoth"
 453         self.translatable = False
 454         self.root = RootNode()
 455         self.parent_node = [self.root]
 456         self.skip_newlines_after_plus = False
 457         self.in_tag = ""
 458 
 459         command_marker_byte = chr(254)
 460 
 461         input = self.preprocessed
 462         if not input: input = self.path
 463 
 464         for rawline in open(input, "rb"):
 465             compos = rawline.find(command_marker_byte)
 466             self.parser_line += 1
 467             # Everything from chr(254) to newline is the command.
 468             if compos != 0:
 469                 self.line_in_file += 1
 470             if compos >= 0:
 471                 self.parse_line_without_commands(rawline[:compos])
 472                 self.handle_command(rawline[compos + 1:-1])
 473             else:
 474                 self.parse_line_without_commands(rawline)
 475 
 476         if self.keep_temp_dir is None and self.temp_dir:
 477             if self.verbose:
 478                 print("removing " + self.temp_dir)
 479             shutil.rmtree(self.temp_dir, ignore_errors=True)
