  14 function ca_fast_move:execution(cfg)
  15     local move_cost_factor = cfg.move_cost_factor or 2.0
  16     if (move_cost_factor < 1.1) then move_cost_factor = 1.1 end
  17 
  18     -- Get the locations to be avoided
  19     local avoid_map = FAU.get_avoid_map(cfg)
  20 
  21     local all_units_MP = AH.get_units_with_moves { side = wesnoth.current.side, canrecruit = 'no' }
  22     local units = {}
  23     for _,unit in ipairs(all_units_MP) do
  24         if (not unit.status.guardian) then table.insert(units, unit) end
  25     end
  26 
  27     local leader = wesnoth.get_units { side = wesnoth.current.side, canrecruit = 'yes' }[1]
  28 
  29     local goals = {}
  30 
  31     -- Villages get added first, so that (hopefully, scouts and faster units will go for them first)
  32     local village_value = ai.aspects.village_value
  33     if leader and (village_value > 0) then
  34         local villages = wesnoth.get_villages()
  35 
  36         -- Eliminate villages in avoid_map and those owned by an allied side
  37         -- Also remove unowned villages if the AI has no leader
  38         for i = #villages,1,-1 do
  39             if avoid_map:get(villages[i][1], villages[i][2]) then
  40                 table.remove(villages, i)
  41             else
  42                 local owner = wesnoth.get_village_owner(villages[i][1], villages[i][2])
  43                 if owner and (not wesnoth.is_enemy(owner, wesnoth.current.side)) then
  44                     table.remove(villages, i)
  45                 elseif (not leader) and (not owner) then
  46                     table.remove(villages, i)
  47                 end
  48             end
  49         end
  50 
  51         -- Add rating that is sum of inverse distances from all other villages
  52         for i = 1,#villages-1 do
  53             local v1 = villages[i]
  54             for j = i+1,#villages do
  55                 local v2 = villages[j]
  56 
  57                 local dist = M.distance_between(v1[1], v1[2], v2[1], v2[2])
  58                 dist = math.ceil(dist / 5.)  -- In discrete steps of 5 hexes
  59 
  60                 v1.rating = (v1.rating or 0) + 1. / dist
  61                 v2.rating = (v2.rating or 0) + 1. / dist
  62             end
  63         end
  64 
  65         -- Multiply by distance from side leader
  66         for _,village in ipairs(villages) do
  67             local dist = 1  -- Just in case there is no leader
  68             dist = math.ceil(dist / 5.)  -- In discrete steps of 5 hexes
  69             if leader then
  70                 dist = M.distance_between(village[1], village[2], leader.x, leader.y)
  71             end
  72 
  73             village.rating = (village.rating or 1.) * 1. / dist
  74         end
  75 
  76         -- Now we figure out how many villages we want to go for
  77         local max_villages = math.floor(#units / 4. * village_value)
  78 
  79         local villages_to_get = math.min(max_villages, #villages)
  80 
  81         for _ = 1,villages_to_get do
  82             -- Sort villages by rating (highest rating last, because that makes table.remove faster)
  83             table.sort(villages, function(a, b) return (a.rating < b.rating) end)
  84 
  85             local x,y = villages[#villages][1], villages[#villages][2]
  86             table.insert(goals, { x = x, y = y, is_village = true })
  87             table.remove(villages)
  88 
  89             -- Now re-rate the villages, want those farthest from the last selected one first
  90             -- Need to check whether this was the last village
  91             if (#villages > 1) then
  92                 local base_rating = villages[#villages].rating
  93 
  94                 for _,village in ipairs(villages) do
  95                     village.rating = village.rating / base_rating
  96                         * M.distance_between(x, y, village[1], village[2])
  97                 end
  98             end
  99         end
 100     end
 101 
 102     -- Now add enemy leaders to the goals
 103     local enemy_leaders = AH.get_attackable_enemies { canrecruit = 'yes' }
 104 
 105     -- Sort enemy leaders by distance to AI leader
 106     if leader then
 107         table.sort(enemy_leaders, function(a, b)
 108             local dist_a = M.distance_between(leader.x, leader.y, a.x, a.y)
 109             local dist_b = m.distance_between(leader.x, leader.y, b.x, b.y)
 110             return (dist_a < dist_b)
 111         end)
 112     end
 113 
 114     for i_el,enemy_leader in ipairs(enemy_leaders) do
 115         if (not avoid_map:get(enemy_leader.x, enemy_leader.y)) then
 116             local goal = { x = enemy_leader.x, y = enemy_leader.y }
 117             table.insert(goals, goal)
 118         end
 119     end
 120 
 121     -- Putting information about all the units into the goals
 122     -- This is a one-time expense up front, but generally not too bad
 123     for _,goal in ipairs(goals) do
 124         -- Insert information about the units
 125         for i_u,unit in ipairs(units) do
 126             local dist = M.distance_between(unit.x, unit.y, goal.x, goal.y)
 127 
 128             goal[i_u] = {
 129                 dist = dist / unit.max_moves,
 130                 rating = - math.ceil(dist / unit.max_moves) - dist / 10.,  -- Combination of fastest and closest unit
 131                 i_unit = i_u
 132             }
 133         end
 134         table.sort(goal, function(a, b) return (a.rating > b.rating) end)
 135     end
 136 
 137     local keep_moving, next_goal = true, 0
 138     if (not goals[1]) then keep_moving = false end
 139     while keep_moving do
 140         keep_moving = false
 141 
 142         next_goal = next_goal + 1
 143         if (next_goal > #goals) then next_goal = 1 end
 144         local goal = goals[next_goal]
 145 
 146         local max_rating, best_unit_info = -9e99
 147         for _,unit_info in ipairs(goal) do
 148             if (not unit_info.cost) then
 149                 local _,cost =
 150                     AH.find_path_with_shroud(
 151                         units[unit_info.i_unit],
 152                         goal.x, goal.y,
 153                         { ignore_units = true }
 154                     )
 155                 cost = cost / units[unit_info.i_unit].max_moves
 156                 unit_info.cost = cost
 157             end
 158 
 159             if (unit_info.cost < unit_info.dist * move_cost_factor) then
 160                 best_unit_info = unit_info
 161                 break
 162             elseif (unit_info.cost < 1000) then
 163                 local rating = - unit_info.cost
 164                 if (rating > max_rating) then
 165                     max_rating, best_unit_info = rating, unit_info
 166                 end
 167             end
 168         end
 169 
 170         if best_unit_info then
 171             local unit = units[best_unit_info.i_unit]
 172 
 173             -- We now want the hex that is 2 steps beyond the next hop for the unit
 174             -- on its way toward the goal, ignoring any unit along the way
 175             local path = AH.find_path_with_shroud(unit, goal.x, goal.y, { ignore_units = true })
 176 
 177             -- Use current unit position as default
 178             local short_goal, index = { unit.x, unit.y }, 1
 179 
 180             for i = 2,#path do
 181                 local _, sub_cost = AH.find_path_with_shroud(unit, path[i][1], path[i][2], { ignore_units = true })
 182 
 183                 if (sub_cost <= unit.moves) then
 184                     short_goal, index = path[i], i
 185                 else
 186                     break
 187                 end
 188             end
 189 
 190             -- Now go two hexes past that, if possible
 191             if path[index + 2] then
 192                 short_goal = path[index + 2]
 193             elseif path[index + 1] then
 194                 short_goal = path[index + 1]
 195             end
 196 
 197             -- Finally find the best move for this unit
 198             local reach = wesnoth.find_reach(unit)

 201             local max_rating, best_hex = -9e99
 202             for _,loc in ipairs(reach) do
 203                 if (not avoid_map:get(loc[1], loc[2])) then
 204                     local rating = -M.distance_between(loc[1], loc[2], short_goal[1], short_goal[2])
 205                     local other_rating = -M.distance_between(loc[1], loc[2], goal.x, goal.y) / 10.
 206                     rating = rating + other_rating
 207 
 208                     local unit_in_way
 209                     if (rating > max_rating) then
 210                         unit_in_way = wesnoth.get_unit(loc[1], loc[2])
 211                         if (unit_in_way == unit) or (not AH.is_visible_unit(wesnoth.current.side, unit_in_way)) then
 212                             unit_in_way = nil
 213                         end
 214 
 215                         if unit_in_way and (unit_in_way.side == unit.side) then
 216                             local reach = AH.get_reachable_unocc(unit_in_way)
 217                             if (reach:size() > 1) then
 218                                 unit_in_way = nil
 219                                 rating = rating - 0.01
 220                                 other_rating = other_rating - 0.01
 221                             end
 222                         end
 223                     end
 224 
 225                     if (not unit_in_way) then
 226                         if cfg.dungeon_mode then
 227                             table.insert(pre_ratings, {
 228                                 rating = rating,
 229                                 other_rating = other_rating,
 230                                 x = loc[1], y = loc[2]
 231                             })
 232                         else
 233                             if (rating > max_rating) then
 234                                 max_rating, best_hex = rating, { loc[1], loc[2] }
 235                             end
 236                         end
 237                     end
 238                 end
 239             end
 240 
 241             -- If this is dungeon mode, we need another level of analysis, calculating
 242             -- the move cost from the target hex to the short goal hex, not just the distance
 243             if cfg.dungeon_mode then
 244                 table.sort(pre_ratings, function(a,b) return (a.rating > b.rating) end)
 245 
 246                 wesnoth.extract_unit(unit)
 247                 local old_x, old_y = unit.x, unit.y
 248 
 249                 local max_rating = -9e99
 250                 for _,pre_rating in ipairs(pre_ratings) do
 251                     -- If pre_rating is worse than the full rating, we are done because the
 252                     -- move cost can never be less than the distance, so we cannot possibly do
 253                     -- better than the pre-rating
 254                     if (pre_rating.rating <= max_rating) then break end
 255 
 256                     unit.x, unit.y = pre_rating.x, pre_rating.y
 257                     local _,cost = AH.find_path_with_shroud(unit, short_goal[1], short_goal[2])
 258 
 259                     local rating = - cost + pre_rating.other_rating
 260 
 261                     if (rating > max_rating) then
 262                         max_rating, best_hex = rating, { pre_rating.x, pre_rating.y }
 263                     end
 264                 end
 265 
 266                 unit.x, unit.y = old_x, old_y
 267                 wesnoth.put_unit(unit)
 268             end
 269 
 270             if best_hex then
 271                 local dx, dy = goal.x - best_hex[1], goal.y - best_hex[2]
 272                 local move_result = AH.robust_move_and_attack(ai, unit, best_hex, nil, { dx = dx, dy = dy })
 273                 -- If something unexpected happened, return and reconsider
 274                 if (not move_result.ok) then return end
 275             end
 276 
 277             -- Also remove this unit from all the tables; using table.remove is fine here
 278             for _,tmp_goal in ipairs(goals) do
 279                 for i_info,tmp_info in ipairs(tmp_goal) do
 280                     if (tmp_info.i_unit == best_unit_info.i_unit) then
 281                         table.remove(tmp_goal, i_info)
 282                         break
 283                     end
 284                 end
 285             end
 286 
 287             -- Finally, if this was a village, remove the goal (we only do one unit per village
 288             if goal.is_village then
 289                 table.remove(goals, next_goal)
 290                 next_goal = next_goal - 1
 291             end
 292         else
 293             table.remove(goals, next_goal)
 294             next_goal = next_goal - 1
 295         end
 296 
 297         for _,goal in ipairs(goals) do
 298             if goal[1] then
 299                 keep_moving = true
 300                 break
 301             end
 302         end
 303     end
