  96 bool section_is_referenced(const std::string &section_id, const config &cfg)
  97 {
  98         if (const config &toplevel = cfg.child("toplevel"))
  99         {
 100                 const std::vector<std::string> toplevel_refs
 101                         = utils::quoted_split(toplevel["sections"]);
 102                 if (std::find(toplevel_refs.begin(), toplevel_refs.end(), section_id)
 103                         != toplevel_refs.end()) {
 104                         return true;
 105                 }
 106         }
 107 
 108         for (const config &section : cfg.child_range("section"))
 109         {
 110                 const std::vector<std::string> sections_refd
 111                         = utils::quoted_split(section["sections"]);
 112                 if (std::find(sections_refd.begin(), sections_refd.end(), section_id)
 113                         != sections_refd.end()) {
 114                         return true;
 115                 }
 116         }
 117         return false;

 144 void parse_config_internal(const config *help_cfg, const config *section_cfg,
 145                                                    section &sec, int level)
 146 {
 147         if (level > max_section_level) {
 148                 std::cerr << "Maximum section depth has been reached. Maybe circular dependency?"
 149                                   << std::endl;
 150         }
 151         else if (section_cfg != nullptr) {
 152                 const std::vector<std::string> sections = utils::quoted_split((*section_cfg)["sections"]);
 153                 sec.level = level;
 154                 std::string id = level == 0 ? "toplevel" : (*section_cfg)["id"].str();
 155                 if (level != 0) {
 156                         if (!is_valid_id(id)) {
 157                                 std::stringstream ss;
 158                                 ss << "Invalid ID, used for internal purpose: '" << id << "'";
 159                                 throw parse_error(ss.str());
 160                         }
 161                 }
 162                 std::string title = level == 0 ? "" : (*section_cfg)["title"].str();
 163                 sec.id = id;
 164                 sec.title = title;
 165                 std::vector<std::string>::const_iterator it;
 166                 // Find all child sections.
 167                 for (it = sections.begin(); it != sections.end(); ++it) {
 168                         if (const config &child_cfg = help_cfg->find_child("section", "id", *it))
 169                         {
 170                                 section child_section;
 171                                 parse_config_internal(help_cfg, &child_cfg, child_section, level + 1);
 172                                 sec.add_section(child_section);
 173                         }
 174                         else {
 175                                 std::stringstream ss;
 176                                 ss << "Help-section '" << *it << "' referenced from '"
 177                                    << id << "' but could not be found.";
 178                                 throw parse_error(ss.str());
 179                         }
 180                 }
 181 
 182                 generate_sections(help_cfg, (*section_cfg)["sections_generator"], sec, level);
 183                 //TODO: harmonize topics/sections sorting
 184                 if ((*section_cfg)["sort_sections"] == "yes") {
 185                         std::sort(sec.sections.begin(),sec.sections.end(), section_less());
 186                 }
 187 
 188                 bool sort_topics = false;
 189                 bool sort_generated = true;
 190 
 191                 if ((*section_cfg)["sort_topics"] == "yes") {
 192                   sort_topics = true;
 193                   sort_generated = false;
 194                 } else if ((*section_cfg)["sort_topics"] == "no") {
 195                   sort_topics = false;
 196                   sort_generated = false;
 197                 } else if ((*section_cfg)["sort_topics"] == "generated") {
 198                   sort_topics = false;
 199                   sort_generated = true;
 200                 } else if (!(*section_cfg)["sort_topics"].empty()) {
 201                   std::stringstream ss;
 202                   ss << "Invalid sort option: '" << (*section_cfg)["sort_topics"] << "'";
 203                   throw parse_error(ss.str());
 204                 }
 205 
 206                 std::vector<topic> generated_topics =
 207                 generate_topics(sort_generated,(*section_cfg)["generator"]);
 208 
 209                 const std::vector<std::string> topics_id = utils::quoted_split((*section_cfg)["topics"]);
 210                 std::vector<topic> topics;
 211 
 212                 // Find all topics in this section.
 213                 for (it = topics_id.begin(); it != topics_id.end(); ++it) {
 214                         if (const config &topic_cfg = help_cfg->find_child("topic", "id", *it))
 215                         {
 216                                 std::string text = topic_cfg["text"];
 217                                 text += generate_topic_text(topic_cfg["generator"], help_cfg, sec, generated_topics);
 218                                 topic child_topic(topic_cfg["title"], topic_cfg["id"], text);
 219                                 if (!is_valid_id(child_topic.id)) {
 220                                         std::stringstream ss;
 221                                         ss << "Invalid ID, used for internal purpose: '" << id << "'";
 222                                         throw parse_error(ss.str());
 223                                 }
 224                                 topics.push_back(child_topic);
 225                         }
 226                         else {
 227                                 std::stringstream ss;
 228                                 ss << "Help-topic '" << *it << "' referenced from '" << id
 229                                    << "' but could not be found." << std::endl;
 230                                 throw parse_error(ss.str());
 231                         }
 232                 }
 233 
 234                 if (sort_topics) {
 235                         std::sort(topics.begin(),topics.end(), title_less());
 236                         std::sort(generated_topics.begin(),
 237                           generated_topics.end(), title_less());
 238                         std::merge(generated_topics.begin(),
 239                           generated_topics.end(),topics.begin(),topics.end()
 240                           ,std::back_inserter(sec.topics),title_less());
 241                 }
 242                 else {
 243                         sec.topics.insert(sec.topics.end(),
 244                                 topics.begin(), topics.end());
 245                         sec.topics.insert(sec.topics.end(),
 246                                 generated_topics.begin(), generated_topics.end());
 247                 }
 248         }
 249 }

 261 std::vector<topic> generate_topics(const bool sort_generated,const std::string &generator)
 262 {
 263         std::vector<topic> res;
 264         if (generator == "") {
 265                 return res;
 266         }
 267 
 268         if (generator == "abilities") {
 269                 res = generate_ability_topics(sort_generated);
 270         } else if (generator == "weapon_specials") {
 271                 res = generate_weapon_special_topics(sort_generated);
 272         } else if (generator == "time_of_days") {
 273                 res = generate_time_of_day_topics(sort_generated);
 274         } else if (generator == "traits") {
 275                 res = generate_trait_topics(sort_generated);
 276         } else {
 277                 std::vector<std::string> parts = utils::split(generator, ':', utils::STRIP_SPACES);
 278                 if (parts.size() > 1 && parts[0] == "units") {
 279                         res = generate_unit_topics(sort_generated, parts[1]);
 280                 } else if (parts[0] == "era" && parts.size()>1) {
 281                         res = generate_era_topics(sort_generated, parts[1]);
 282                 } else {
 283                         WRN_HP << "Found a topic generator that I didn't recognize: " << generator << "\n";
 284                 }
 285         }
 286 
 287         return res;
 288 }

 295                 generate_terrain_sections(help_cfg, sec, level);

 297                 DBG_HP << "Generating eras...\n";
 298                 generate_era_sections(help_cfg, sec, level);
 299         } else  {
 300                 std::vector<std::string> parts = utils::split(generator, ':', utils::STRIP_SPACES);
 301                 if (parts.size() > 1 && parts[0] == "units") {
 302                         generate_unit_sections(help_cfg, sec, level, true, parts[1]);
 303                 } else if (generator.size() > 0) {
 304                         WRN_HP << "Found a section generator that I didn't recognize: " << generator << "\n";
 305                 }
 306         }

 309 std::string generate_topic_text(const std::string &generator, const config *help_cfg, const section &sec, const std::vector<topic>& generated_topics)
 310 {
 311         std::string empty_string = "";
 312         if (generator == "") {
 313                 return empty_string;
 314         } else if (generator == "about") {
 315                 return generate_about_text();
 316         } else {
 317                 std::vector<std::string> parts = utils::split(generator, ':');
 318                 if (parts.size() > 1 && parts[0] == "contents") {
 319                         if (parts[1] == "generated") {
 320                                 return generate_contents_links(sec, generated_topics);
 321                         } else {
 322                                 return generate_contents_links(parts[1], help_cfg);
 323                         }
 324                 }
 325         }
 326         return empty_string;
 327 }

 482         }
 483 
 484         if (sort_generated)

 486         return topics;

 490 std::vector<topic> generate_ability_topics(const bool sort_generated)
 491 {
 492         std::vector<topic> topics;
 493         std::map<t_string, std::string> ability_description;
 494         std::map<t_string, std::set<std::string, string_less> > ability_units;
 495         // Look through all the unit types, check if a unit of this type
 496         // should have a full description, if so, add this units abilities
 497         // for display. We do not want to show abilities that the user has
 498         // not encountered yet.
 499         for (const unit_type_data::unit_type_map::value_type &type_mapping : unit_types.types())
 500         {
 501                 const unit_type &type = type_mapping.second;
 502                 if (description_type(type) == FULL_DESCRIPTION) {
 503 
 504                         std::vector<t_string> const* abil_vecs[2];
 505                         abil_vecs[0] = &type.abilities();
 506                         abil_vecs[1] = &type.adv_abilities();
 507 
 508                         std::vector<t_string> const* desc_vecs[2];
 509                         desc_vecs[0] = &type.ability_tooltips();
 510                         desc_vecs[1] = &type.adv_ability_tooltips();
 511 
 512                         for(int i=0; i<2; ++i) {
 513                                 std::vector<t_string> const& abil_vec = *abil_vecs[i];
 514                                 std::vector<t_string> const& desc_vec = *desc_vecs[i];
 515                                 for(size_t j=0; j < abil_vec.size(); ++j) {
 516                                         t_string const& abil_name = abil_vec[j];
 517                                         const std::string abil_desc =
 518                                                 j >= desc_vec.size() ? "" : desc_vec[j].str();
 519 
 520                                         ability_description.emplace(abil_name, abil_desc);
 521 
 522                                         if (!type.hide_help()) {
 523                                                 //add a link in the list of units with this ability
 524                                                 std::string type_name = type.type_name();
 525                                                 std::string ref_id = unit_prefix +  type.id();
 526                                                 //we put the translated name at the beginning of the hyperlink,
 527                                                 //so the automatic alphabetic sorting of std::set can use it
 528                                                 std::string link = make_link(type_name, ref_id);
 529                                                 ability_units[abil_name].insert(link);
 530                                         }
 531                                 }
 532                         }
 533                 }

1182 std::vector<std::string> parse_text(const std::string &text)
1183 {
1184         std::vector<std::string> res;
1185         bool last_char_escape = false;
1186         const char escape_char = '\\';
1187         std::stringstream ss;
1188         size_t pos;
1189         enum { ELEMENT_NAME, OTHER } state = OTHER;
1190         for (pos = 0; pos < text.size(); ++pos) {
1191                 const char c = text[pos];
1192                 if (c == escape_char && !last_char_escape) {
1193                         last_char_escape = true;
1194                 }
1195                 else {
1196                         if (state == OTHER) {
1197                                 if (c == '<') {
1198                                         if (last_char_escape) {
1199                                                 ss << c;
1200                                         }
1201                                         else {
1202                                                 res.push_back(ss.str());
1203                                                 ss.str("");
1204                                                 state = ELEMENT_NAME;
1205                                         }
1206                                 }
1207                                 else {
1208                                         ss << c;
1209                                 }
1210                         }
1211                         else if (state == ELEMENT_NAME) {
1212                                 if (c == '/') {
1213                                         std::string msg = "Erroneous / in element name.";
1214                                         throw parse_error(msg);
1215                                 }
1216                                 else if (c == '>') {
1217                                         // End of this name.
1218                                         std::stringstream s;
1219                                         const std::string element_name = ss.str();
1220                                         ss.str("");
1221                                         s << "</" << element_name << ">";
1222                                         const std::string end_element_name = s.str();
1223                                         size_t end_pos = text.find(end_element_name, pos);
1224                                         if (end_pos == std::string::npos) {
1225                                                 std::stringstream msg;
1226                                                 msg << "Unterminated element: " << element_name;
1227                                                 throw parse_error(msg.str());
1228                                         }
1229                                         s.str("");
1230                                         const std::string contents = text.substr(pos + 1, end_pos - pos - 1);
1231                                         const std::string element = convert_to_wml(element_name, contents);
1232                                         res.push_back(element);
1233                                         pos = end_pos + end_element_name.size() - 1;
1234                                         state = OTHER;
1235                                 }
1236                                 else {
1237                                         ss << c;
1238                                 }
1239                         }
1240                         last_char_escape = false;
1241                 }
1242         }
1243         if (state == ELEMENT_NAME) {
1244                 std::stringstream msg;
1245                 msg << "Element '" << ss.str() << "' continues through end of string.";
1246                 throw parse_error(msg.str());
1247         }
1248         if (ss.str() != "") {
1249                 // Add the last string.
1250                 res.push_back(ss.str());
1251         }
1252         return res;
1253 }

1255 std::string convert_to_wml(const std::string &element_name, const std::string &contents)
1256 {
1257         std::stringstream ss;
1258         bool in_quotes = false;
1259         bool last_char_escape = false;
1260         const char escape_char = '\\';
1261         std::vector<std::string> attributes;
1262         // Find the different attributes.
1263         // No checks are made for the equal sign or something like that.
1264         // Attributes are just separated by spaces or newlines.
1265         // Attributes that contain spaces must be in single quotes.
1266         for (size_t pos = 0; pos < contents.size(); ++pos) {
1267                 const char c = contents[pos];
1268                 if (c == escape_char && !last_char_escape) {
1269                         last_char_escape = true;
1270                 }
1271                 else {
1272                         if (c == '\'' && !last_char_escape) {
1273                                 ss << '"';
1274                                 in_quotes = !in_quotes;
1275                         }
1276                         else if ((c == ' ' || c == '\n') && !last_char_escape && !in_quotes) {
1277                                 // Space or newline, end of attribute.
1278                                 attributes.push_back(ss.str());
1279                                 ss.str("");
1280                         }
1281                         else {
1282                                 ss << c;
1283                         }
1284                         last_char_escape = false;
1285                 }
1286         }
1287         if (in_quotes) {
1288                 std::stringstream msg;
1289                 msg << "Unterminated single quote after: '" << ss.str() << "'";
1290                 throw parse_error(msg.str());
1291         }
1292         if (ss.str() != "") {
1293                 attributes.push_back(ss.str());
1294         }
1295         ss.str("");
1296         // Create the WML.
1297         ss << "[" << element_name << "]\n";
1298         for (std::vector<std::string>::const_iterator it = attributes.begin();
1299                  it != attributes.end(); ++it) {
1300                 ss << *it << "\n";
1301         }
1302         ss << "[/" << element_name << "]\n";
1303         return ss.str();
1304 }

1357 std::string get_first_word(const std::string &s)
1358 {
1359         size_t first_word_start = s.find_first_not_of(' ');
1360         if (first_word_start == std::string::npos) {
1361                 return s;
1362         }
1363         size_t first_word_end = s.find_first_of(" \n", first_word_start);
1364         if( first_word_end == first_word_start ) {
1365                 // This word is '\n'.
1366                 first_word_end = first_word_start+1;
1367         }
1368 
1369         //if no gap(' ' or '\n') found, test if it is CJK character
1370         std::string re = s.substr(0, first_word_end);
1371 
1372         utf8::iterator ch(re);
1373         if (ch == utf8::iterator::end(re))
1374                 return re;
1375 
1376         ucs4::char_t firstchar = *ch;
1377         if (font::is_cjk_char(firstchar)) {
1378                 re = unicode_cast<utf8::string>(firstchar);
1379         }
1380         return re;

1494 std::string generate_table(const table_spec &tab, const unsigned int spacing)
1495 {
1496         table_spec::const_iterator row_it;
1497         std::vector<std::pair<std::string, unsigned> >::const_iterator col_it;
1498         unsigned int num_cols = 0;
1499         for (row_it = tab.begin(); row_it != tab.end(); ++row_it) {
1500                 if (row_it->size() > num_cols) {
1501                         num_cols = row_it->size();
1502                 }
1503         }
1504         std::vector<unsigned int> col_widths(num_cols, 0);
1505         // Calculate the width of all columns, including spacing.
1506         for (row_it = tab.begin(); row_it != tab.end(); ++row_it) {
1507                 unsigned int col = 0;
1508                 for (col_it = row_it->begin(); col_it != row_it->end(); ++col_it) {
1509                         if (col_widths[col] < col_it->second + spacing) {
1510                                 col_widths[col] = col_it->second + spacing;
1511                         }
1512                         ++col;
1513                 }
1514         }
1515         std::vector<unsigned int> col_starts(num_cols);
1516         // Calculate the starting positions of all columns
1517         for (unsigned int i = 0; i < num_cols; ++i) {
1518                 unsigned int this_col_start = 0;
1519                 for (unsigned int j = 0; j < i; ++j) {
1520                         this_col_start += col_widths[j];
1521                 }
1522                 col_starts[i] = this_col_start;
1523         }
1524         std::stringstream ss;
1525         for (row_it = tab.begin(); row_it != tab.end(); ++row_it) {
1526                 unsigned int col = 0;
1527                 for (col_it = row_it->begin(); col_it != row_it->end(); ++col_it) {
1528                         ss << jump_to(col_starts[col]) << col_it->first;
1529                         ++col;
1530                 }
1531                 ss << "\n";
1532         }
1533         return ss.str();
1534 }
