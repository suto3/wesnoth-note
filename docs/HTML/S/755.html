  38 version_info::version_info(const std::string& str)
  39         : nums_(3,0)
  40         , special_("")
  41         , special_separator_('\0')
  42 {
  43         std::string v = str;
  44         boost::trim(v);
  45 
  46         if(v.empty())
  47                 return;
  48 
  49         //
  50         // The breakpoint is where the "special" version component begins.
  51         // For 1.1.2a it would at the index of the char 'a'. For 1.1.4+dev it is at '+'.
  52         //
  53         // For 1.5.2 it is at npos.
  54         //
  55         const std::string::size_type breakpoint_pos = v.find_first_not_of(".0123456789");
  56         std::string left_side;
  57         if(breakpoint_pos != std::string::npos) {
  58                 const std::string right_side = v.substr(breakpoint_pos);
  59                 assert(right_side.empty() == false);
  60 
  61                 if((right_side[0] >= 'A' && right_side[0] <= 'Z') || (right_side[0] >= 'a' && right_side[0] <= 'z')) {
  62                         special_separator_ = '\0';
  63                         special_ = right_side;
  64                 }
  65                 else {
  66                         special_separator_ = right_side[0];
  67                         if(right_side.size() > 1) {
  68                                 special_ = right_side.substr(1);
  69                         }
  70                 }
  71 
  72                 left_side = v.substr(0, breakpoint_pos);
  73         }
  74         else {
  75                 left_side = v;
  76         }
  77 
  78         const std::vector<std::string> components = utils::split(left_side, '.');
  79         const size_t s = components.size();
  80         if(s == 0) {
  81                 return;
  82         }
  83         else if(s > 3) {
  84                 nums_.resize(s, 0);
  85         }
  86 
  87         for(size_t i = 0; (i < s); ++i) {
  88                 nums_[i] = lexical_cast_default<unsigned int>(components[i]);
  89         }
  90 }

  92 std::string version_info::str() const
  93 {
  94         const size_t s = nums_.size();
  95 
  96         std::ostringstream o;
  97         for(size_t k = 0; k < s; ++k) {
  98                 o << nums_[k];
  99 
 100                 if(s != 1+k) {
 101                         o << '.';
 102                 }
 103         }
 104 
 105         if(! special_.empty()) {
 106                 if(special_separator_ != '\0') {
 107                         o << special_separator_;
 108                 }
 109 
 110                 o << special_;
 111         }
 112 
 113         return o.str();
 114 }

 145         enum COMP_TYPE {
 146                 EQUAL,
 147                 NOT_EQUAL,
 148                 LT, GT
 149         };

 151         /*
 152                    x         >          y
 153         x0.x1.x2.x3.[...].xN > y0.y1.y2.y3.[...].yN iff
 154 
 155         x0 > y0 || (x0 == y0 && (x1 > y1 || (x1 == y1 && (x2 > y2 || (x2 >= y2 ||
 156 
 157         */
 158         template<typename _Toperator, typename _Tfallback_operator>
 159         bool recursive_order_operation(const std::vector<unsigned int>& l, const std::vector<unsigned int>& r, size_t k)
 160         {
 161                 if(k >= l.size() || k >= r.size()) {
 162                         return false;
 163                 }
 164 
 165                 const unsigned int& lvalue = l[k];
 166                 const unsigned int& rvalue = r[k];
 167 
 168                 _Toperator o;
 169                 _Tfallback_operator fallback_o;
 170 
 171                 bool ret = o(lvalue, rvalue);
 172                 if((!ret) && fallback_o(lvalue, rvalue)) {
 173                         ret = recursive_order_operation<_Toperator, _Tfallback_operator>(l,r,++k);
 174                 }
 175                 return ret;
 176         }

 178 #ifdef _MSC_VER
 179 #pragma warning (push)
 180 #pragma warning (disable: 4706)
 181 #endif
 182         bool version_numbers_comparison_internal(const version_info& l, const version_info& r, COMP_TYPE o)
 183         {
 184                 std::vector<unsigned int> lc = l.components();
 185                 std::vector<unsigned int> rc = r.components();
 186 
 187                 const size_t lsize = lc.size();
 188                 const size_t rsize = rc.size();
 189                 const size_t csize = std::max(lsize, rsize);
 190 
 191                 // make compatible, missing items default to zero
 192                 if(lsize < csize) lc.resize(csize, 0);
 193                 if(rsize < csize) rc.resize(csize, 0);
 194 
 195                 bool result = true;
 196 
 197                 const std::vector<unsigned int>& lcc = lc;
 198                 const std::vector<unsigned int>& rcc = rc;
 199 
 200                 switch(o)
 201                 {
 202                         case EQUAL: case NOT_EQUAL: {
 203                                 for(size_t i = 0; i < csize; ++i) {
 204                                         const unsigned int& lvalue = lc[i];
 205                                         const unsigned int& rvalue = rc[i];
 206                                         if(o == NOT_EQUAL) {
 207                                                 if((result = (lvalue != rvalue))) {
 208 #ifdef _MSC_VER
 209 #pragma warning (pop)
 210 #endif
 211                                                         return true;
 212                                                 }
 213                                                 continue;
 214                                         } else {
 215                                                 result = result && lvalue == rvalue;
 216                                                 if(!result) {
 217                                                         break;
 218                                                 }
 219                                         }
 220                                 }
 221                                 break;
 222                         }
 223                         case LT:
 224                                 result = recursive_order_operation<std::less<unsigned int>, std::equal_to<unsigned int> >(lcc, rcc, 0);
 225                                 break;
 226                         case GT:
 227                                 result = recursive_order_operation<std::greater<unsigned int>, std::equal_to<unsigned int> >(lcc, rcc, 0);
 228                                 break;
 229                         default:
 230                                 assert(0 == 1);
 231                                 break;
 232                 }
 233                 return result;
 234         }

 278 VERSION_COMP_OP parse_version_op(const std::string& op_str)
 279 {
 280         if(op_str == "==") {
 281                 return OP_EQUAL;
 282         } else if(op_str == "!=") {
 283                 return OP_NOT_EQUAL;
 284         } else if(op_str == "<") {
 285                 return OP_LESS;
 286         } else if(op_str == "<=") {
 287                 return OP_LESS_OR_EQUAL;
 288         } else if(op_str == ">") {
 289                 return OP_GREATER;
 290         } else if(op_str == ">=") {
 291                 return OP_GREATER_OR_EQUAL;
 292         }
 293 
 294         return OP_INVALID;
 295 }

 297 bool do_version_check(const version_info& a, VERSION_COMP_OP op, const version_info& b)
 298 {
 299         switch(op) {
 300         case OP_EQUAL:
 301                 return a == b;
 302         case OP_NOT_EQUAL:
 303                 return a != b;
 304         case OP_LESS:
 305                 return a < b;
 306         case OP_LESS_OR_EQUAL:
 307                 return a <= b;
 308         case OP_GREATER:
 309                 return a > b;
 310         case OP_GREATER_OR_EQUAL:
 311                 return a >= b;
 312         default:
 313                 ;
 314         }
 315 
 316         return false;
 317 }
