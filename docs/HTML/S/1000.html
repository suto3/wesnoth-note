  75 {
  76         if(accuracy_ == 0 && parry_ == 0) {

  78         }
  79

  90 /**
  91  * Returns whether or not *this matches the given @a filter, ignoring the
  92  * complexities introduced by [and], [or], and [not].
  93  */
  94 static bool matches_simple_filter(const attack_type & attack, const config & filter)
  95 {
  96         const std::vector<std::string>& filter_range = utils::split(filter["range"]);
  97         const std::string& filter_damage = filter["damage"];
  98         const std::string& filter_attacks = filter["number"];
  99         const std::string& filter_accuracy = filter["accuracy"];
 100         const std::string& filter_parry = filter["parry"];
 101         const std::string& filter_movement = filter["movement_used"];
 102         const std::vector<std::string> filter_name = utils::split(filter["name"]);
 103         const std::vector<std::string> filter_type = utils::split(filter["type"]);
 104         const std::string filter_special = filter["special"];
 105         const std::string filter_special_active = filter["special_active"];
 106         const std::string filter_formula = filter["formula"];
 107 
 108         if ( !filter_range.empty() && std::find(filter_range.begin(), filter_range.end(), attack.range()) == filter_range.end() )
 109                 return false;
 110 
 111         if ( !filter_damage.empty() && !in_ranges(attack.damage(), utils::parse_ranges(filter_damage)) )
 112                 return false;
 113 
 114         if (!filter_attacks.empty() && !in_ranges(attack.num_attacks(), utils::parse_ranges(filter_attacks)))
 115                 return false;
 116 
 117         if (!filter_accuracy.empty() && !in_ranges(attack.accuracy(), utils::parse_ranges(filter_accuracy)))
 118                 return false;
 119 
 120         if (!filter_parry.empty() && !in_ranges(attack.parry(), utils::parse_ranges(filter_parry)))
 121                 return false;
 122 
 123         if (!filter_movement.empty() && !in_ranges(attack.movement_used(), utils::parse_ranges(filter_movement)))
 124                 return false;
 125 
 126         if ( !filter_name.empty() && std::find(filter_name.begin(), filter_name.end(), attack.id()) == filter_name.end() )
 127                 return false;
 128 
 129         if ( !filter_type.empty() && std::find(filter_type.begin(), filter_type.end(), attack.type()) == filter_type.end() )
 130                 return false;
 131 
 132         if ( !filter_special.empty() && !attack.get_special_bool(filter_special, true) )
 133                 return false;
 134 
 135         if ( !filter_special_active.empty() && !attack.get_special_bool(filter_special_active, false) )
 136                 return false;
 137 
 138         if (!filter_formula.empty()) {
 139                 try {
 140                         const wfl::attack_type_callable callable(attack);
 141                         const wfl::formula form(filter_formula);
 142                         if(!form.evaluate(callable).as_bool()) {
 143                                 return false;
 144                         }
 145                 } catch(wfl::formula_error& e) {
 146                         lg::wml_error() << "Formula error in weapon filter: " << e.type << " at " << e.filename << ':' << e.line << ")\n";
 147                         // Formulae with syntax errors match nothing
 148                         return false;
 149                 }
 150         }
 151 
 152         // Passed all tests.
 153         return true;
 154 }

 156 /**
 157  * Returns whether or not *this matches the given @a filter.
 158  */
 159 bool attack_type::matches_filter(const config& filter) const
 160 {
 161         // Handle the basic filter.
 162         bool matches = matches_simple_filter(*this, filter);
 163 
 164         // Handle [and], [or], and [not] with in-order precedence
 165         for (const config::any_child &condition : filter.all_children_range() )
 166         {
 167                 // Handle [and]
 168                 if ( condition.key == "and" )
 169                         matches = matches && matches_filter(condition.cfg);
 170 
 171                 // Handle [or]
 172                 else if ( condition.key == "or" )
 173                         matches = matches || matches_filter(condition.cfg);
 174 
 175                 // Handle [not]
 176                 else if ( condition.key == "not" )
 177                         matches = matches && !matches_filter(condition.cfg);
 178         }
 179 
 180         return matches;
 181 }

 183 /**
 184  * Modifies *this using the specifications in @a cfg, but only if *this matches
 185  * @a cfg viewed as a filter.
 186  *
 187  * @returns whether or not @c this matched the @a cfg as a filter.
 188  */
 189 bool attack_type::apply_modification(const config& cfg)
 190 {
 191         if( !matches_filter(cfg) )
 192                 return false;
 193 
 194         const std::string& set_name = cfg["set_name"];
 195         const t_string& set_desc = cfg["set_description"];
 196         const std::string& set_type = cfg["set_type"];
 197         const std::string& set_icon = cfg["set_icon"];
 198         const std::string& del_specials = cfg["remove_specials"];
 199         const config &set_specials = cfg.child("set_specials");
 200         const std::string& increase_damage = cfg["increase_damage"];
 201         const std::string& set_damage = cfg["set_damage"];
 202         const std::string& increase_attacks = cfg["increase_attacks"];
 203         const std::string& set_attacks = cfg["set_attacks"];
 204         const std::string& set_attack_weight = cfg["attack_weight"];
 205         const std::string& set_defense_weight = cfg["defense_weight"];
 206         const std::string& increase_accuracy = cfg["increase_accuracy"];
 207         const std::string& set_accuracy = cfg["set_accuracy"];
 208         const std::string& increase_parry = cfg["increase_parry"];
 209         const std::string& set_parry = cfg["set_parry"];
 210         const std::string& increase_movement = cfg["increase_movement_used"];
 211         const std::string& set_movement = cfg["set_movement_used"];
 212         // NB: If you add something here that requires a description,
 213         // it needs to be added to describe_modification as well.
 214 
 215         if(set_name.empty() == false) {
 216                 id_ = set_name;
 217         }
 218 
 219         if(set_desc.empty() == false) {
 220                 description_ = set_desc;
 221         }
 222 
 223         if(set_type.empty() == false) {
 224                 type_ = set_type;
 225         }
 226 
 227         if(set_icon.empty() == false) {
 228                 icon_ = set_icon;
 229         }
 230 
 231         if(del_specials.empty() == false) {
 232                 const std::vector<std::string>& dsl = utils::split(del_specials);
 233                 config new_specials;
 234                 for (const config::any_child &vp : specials_.all_children_range()) {
 235                         std::vector<std::string>::const_iterator found_id =
 236                                 std::find(dsl.begin(), dsl.end(), vp.cfg["id"].str());
 237                         if (found_id == dsl.end()) {
 238                                 new_specials.add_child(vp.key, vp.cfg);
 239                         }
 240                 }
 241                 specials_ = new_specials;
 242         }
 243 
 244         if (set_specials) {
 245                 const std::string &mode = set_specials["mode"];
 246                 if (mode != "append") {
 247                         specials_.clear();
 248                 }
 249                 for (const config::any_child &value : set_specials.all_children_range()) {
 250                         specials_.add_child(value.key, value.cfg);
 251                 }
 252         }
 253 
 254         if(set_damage.empty() == false) {
 255                 damage_ = std::stoi(set_damage);
 256                 if (damage_ < 0) {
 257                         damage_ = 0;
 258                 }
 259         }
 260 
 261         if(increase_damage.empty() == false) {
 262                 damage_ = utils::apply_modifier(damage_, increase_damage, 0);
 263                 if (damage_ < 0) {
 264                         damage_ = 0;
 265                 }
 266         }
 267 
 268         if(set_attacks.empty() == false) {
 269                 num_attacks_ = std::stoi(set_attacks);
 270                 if (num_attacks_ < 0) {
 271                         num_attacks_ = 0;
 272                 }
 273 
 274         }
 275 
 276         if(increase_attacks.empty() == false) {
 277                 num_attacks_ = utils::apply_modifier(num_attacks_, increase_attacks, 1);
 278         }
 279 
 280         if(set_accuracy.empty() == false) {
 281                 accuracy_ = std::stoi(set_accuracy);
 282         }
 283 
 284         if(increase_accuracy.empty() == false) {
 285                 accuracy_ = utils::apply_modifier(accuracy_, increase_accuracy, 1);
 286         }
 287 
 288         if(set_parry.empty() == false) {
 289                 parry_ = std::stoi(set_parry);
 290         }
 291 
 292         if(increase_parry.empty() == false) {
 293                 parry_ = utils::apply_modifier(parry_, increase_parry, 1);
 294         }
 295 
 296         if(set_movement.empty() == false) {
 297                 movement_used_ = std::stoi(set_movement);
 298         }
 299 
 300         if(increase_movement.empty() == false) {
 301                 movement_used_ = utils::apply_modifier(movement_used_, increase_movement, 1);
 302         }
 303 
 304         if(set_attack_weight.empty() == false) {
 305                 attack_weight_ = lexical_cast_default<double>(set_attack_weight,1.0);
 306         }
 307 
 308         if(set_defense_weight.empty() == false) {
 309                 defense_weight_ = lexical_cast_default<double>(set_defense_weight,1.0);
 310         }
 311 
 312         return true;
 313 }

 315 /**
 316  * Trimmed down version of apply_modification(), with no modifications actually
 317  * made. This can be used to get a description of the modification(s) specified
 318  * by @a cfg (if *this matches cfg as a filter).
 319  *
 320  * If *description is provided, it will be set to a (translated) description
 321  * of the modification(s) applied (currently only changes to the number of
 322  * strikes, damage, accuracy, and parry are included in this description).
 323  *
 324  * @returns whether or not @c this matched the @a cfg as a filter.
 325  */
 326 bool attack_type::describe_modification(const config& cfg,std::string* description)
 327 {
 328         if( !matches_filter(cfg) )
 329                 return false;
 330 
 331         // Did the caller want the description?
 332         if(description != nullptr) {
 333                 const std::string& increase_damage = cfg["increase_damage"];
 334                 const std::string& set_damage = cfg["set_damage"];
 335                 const std::string& increase_attacks = cfg["increase_attacks"];
 336                 const std::string& set_attacks = cfg["set_attacks"];
 337                 const std::string& increase_accuracy = cfg["increase_accuracy"];

 339                 const std::string& increase_parry = cfg["increase_parry"];
 340                 const std::string& set_parry = cfg["set_parry"];
 341                 const std::string& increase_movement = cfg["increase_movement_used"];
 342                 const std::string& set_movement = cfg["set_movement_used"];
 343 
 344                 std::vector<t_string> desc;
 345 
 346                 if(!increase_damage.empty()) {
 347                         desc.emplace_back(VNGETTEXT(
 348                                 // TRANSLATORS: Current value for WML code increase_damage, documented in https://wiki.wesnoth.org/EffectWML
