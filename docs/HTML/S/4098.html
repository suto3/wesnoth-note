  28 def expand_square_braces(path):
  29     """Expands the square brackets notation to normal file names.
  30 Yields the expanded file names, or the original path itself
  31 if no expansion could be performed"""
  32     if "[" not in path: # clearly, no expansion can be done in this case
  33         yield path
  34     else:
  35         substitutions = [] # a matrix that will hold all the substitutions to be performed
  36 
  37         for i, match in enumerate(re.finditer(r"\[(.*?)\]", path)): # stop on the first closed square braces, to allow multiple substitutions
  38             substitutions.append([]) # a new list which will host all the substitutions for the current expansion
  39             for token in match.group(1).split(","):
  40                 match_mult = re.match("(.+)\*(\d+)", token) # repeat syntax, eg. [melee*3]
  41                 if match_mult:
  42                     substitutions[i].extend([Substitution(match_mult.group(1), match.start(0), match.end(0))] *
  43                                          int(match_mult.group(2)))
  44                     continue
  45                 match_range = re.match("(\d+)~(\d+)", token) # range syntax, eg [1~4]
  46                 if match_range:
  47                     before, after = int(match_range.group(1)), int(match_range.group(2))
  48                     incr = 1 if before <= after else -1 # to allow iterating in reversed order, eg. [4~1]
  49                     substitutions[i].extend([Substitution(str(n), match.start(0), match.end(0)) for n in range(before, after + incr, incr)])
  50                     continue
  51                 substitutions[i].append(Substitution(token, match.start(0), match.end(0))) # no operator found
  52 
  53         # the purpose is to have all the subs "increasing" simultaneously
  54         # in other words, if a string has two square braces blocks, like [1~3] and [melee*2,ranged]
  55         # we want to expand them as (1,melee), (2,melee), (3,ranged)
  56         # for this reason we need to transpose the matrix
  57         substitutions = zip(*substitutions)
  58 
  59         for sub_array in substitutions:
  60             new_string = path
  61             for sub in reversed(sub_array): # to avoid creating "holes" in the strings
  62                 yield new_string[:sub.start] + sub.sub + new_string[sub.end:] # these are the expanded strings

 103 def comma_split(csstring, list=None, strip="r"):
 104     "Split a comma-separated string, and append the entries to a list if specified."
 105     vallist = [x.lstrip() for x in csstring.split(",") if x.lstrip()]
 106     # strip=: utils::split will remove trailing whitespace from items in comma-
 107     # separated lists but the wml-tags.lua split function only removes leading
 108     # whitespace. So two flags are offered to change default behavior: one to
 109     # lstrip() only, the other to warn about trailing whitespace.
 110     if 'w' in strip:
 111         for item in vallist:
 112             if re.search('\s$', item):
 113                 print('Trailing whitespace may be problematic: "%s" in "%s"' % (item, csstring))
 114     if 'l' not in strip:
 115         vallist = [x.rstrip() for x in vallist]
 116     if list is not None:
 117         list.extend(vallist)
 118     else:
 119         return vallist

 121 def parse_attribute(line):
 122     "Parse a WML key-value pair from a line."
 123     if '=' not in line or line.find("#") > -1 and line.find("#") < line.find("="):
 124         return None
 125     where = line.find("=")
 126     leader = line[:where]
 127     after = line[where+1:]
 128     after = after.lstrip()
 129     if re.search("\s#", after):
 130         where = len(re.split("\s+#", after)[0])
 131         value = after[:where]
 132         comment = after[where:]
 133     else:
 134         value = after.rstrip()
 135         comment = ""
 136     # Return four fields: stripped key, part of line before value,
 137     # value, trailing whitespace and comment.
 138     return (leader.strip(), leader+"=", string_strip(value), comment)

 255 def parse_macroref(start, line):
 256     brackdepth = parendepth = 0
 257     instring = False
 258     args = []
 259     arg = ""
 260     for i in range(start, len(line)):
 261         if instring:
 262             if line[i] == '"':
 263                 instring = False
 264             arg += line[i]
 265         elif line[i] == '"':
 266             instring = not instring
 267             arg += line[i]
 268         elif line[i] == "{":
 269             if brackdepth > 0:
 270                 arg += line[i]
 271             brackdepth += 1
 272         elif line[i] == "}":
 273             brackdepth -= 1
 274             if brackdepth == 0:
 275                 if not line[i-1].isspace():
 276                     arg = arg.strip()
 277                     if arg.startswith('"') and arg.endswith('"'):
 278                         arg = arg[1:-1].strip()
 279                     args.append(arg)
 280                     arg = ""
 281                 break
 282             else:
 283                 arg += line[i]
 284         elif line[i] == "(":
 285             parendepth += 1
 286         elif line[i] == ")":
 287             parendepth -= 1
 288         elif not line[i-1].isspace() and \
 289              line[i].isspace() and \
 290              brackdepth == 1 and \
 291              parendepth == 0:
 292             arg = arg.strip()
 293             if arg.startswith('"') and arg.endswith('"'):
 294                 arg = arg[1:-1].strip()
 295             args.append(arg)
 296             arg = ""
 297         elif not line[i].isspace() or parendepth > 0:
 298             arg += line[i]
 299     return (args, brackdepth, parendepth)

 301 def formaltype(f):
 302     # Deduce the expected type of the formal
 303     if f.startswith("_"):
 304         f = f[1:]
 305     if f == "SIDE" or f.endswith("_SIDE") or re.match("SIDE[0-9]", f):
 306         ftype = "side"
 307     elif f in ("SIDE", "X", "Y", "RED", "GREEN", "BLUE", "TURN", "PROB", "LAYER", "TIME", "DURATION") or f.endswith("NUMBER") or f.endswith("AMOUNT") or f.endswith("COST") or f.endswith("RADIUS") or f.endswith("_X") or f.endswith("_Y") or f.endswith("_INCREMENT") or f.endswith("_FACTOR") or f.endswith("_TIME") or f.endswith("_SIZE"):
 308         ftype = "numeric"
 309     elif f.endswith("PERCENTAGE"):
 310         ftype = "percentage"
 311     elif f in ("POSITION",) or f.endswith("_POSITION") or f == "BASE":
 312         ftype = "position"
 313     elif f.endswith("_SPAN"):
 314         ftype = "span"
 315     elif f == "SIDES" or f.endswith("_SIDES"):
 316         ftype = "alliance"
 317     elif f in ("RANGE",):
 318         ftype = "range"
 319     elif f in ("ALIGN",):
 320         ftype = "alignment"
 321     elif f in ("TYPES"):
 322         ftype = "types"
 323     elif f.startswith("ADJACENT") or f.startswith("TERRAINLIST") or f == "RESTRICTING":
 324         ftype = "terrain_pattern"
 325     elif f.startswith("TERRAIN") or f.endswith("TERRAIN"):
 326         ftype = "terrain_code"
 327     elif f in ("NAME", "NAMESPACE", "VAR", "IMAGESTEM", "ID", "FLAG", "BUILDER") or f.endswith("_NAME") or f.endswith("_ID") or f.endswith("_VAR") or f.endswith("_OVERLAY"):
 328         ftype = "name"
 329     elif f in ("ID_STRING", "NAME_STRING", "DESCRIPTION", "IPF"):
 330         ftype = "optional_string"
 331     elif f in ("STRING", "TYPE", "TEXT") or f.endswith("_STRING") or f.endswith("_TYPE") or f.endswith("_TEXT"):
 332         ftype = "string"
 333     elif f.endswith("IMAGE") or f == "PROFILE":
 334         ftype = "image"
 335     elif f.endswith("MUSIC",) or f.endswith("SOUND"):
 336         ftype = "sound"
 337     elif f.endswith("FILTER",):
 338         ftype = "filter"
 339     elif f == "WML" or f.endswith("_WML"):
 340         ftype = "wml"
 341     elif f in ("AFFIX", "POSTFIX", "ROTATION") or f.endswith("AFFIX"):
 342         ftype = "affix"
 343     # The regexp case avoids complaints about some wacky terrain macros.
 344     elif f.endswith("VALUE") or re.match("[ARS][0-9]", f):
 345         ftype = "any"
 346     else:
 347         ftype = None
 348     return ftype

 350 def actualtype(a):
 351     if a is None:
 352         return None
 353     # Deduce the type of the actual
 354     if a.isdigit() or a.startswith("-") and a[1:].isdigit():
 355         atype = "numeric"
 356     elif re.match(r"0\.[0-9]+\Z", a):
 357         atype = "percentage"
 358     elif re.match(r"-?[0-9]+,-?[0-9]+\Z", a):
 359         atype = "position"
 360     elif re.match(r"([0-9]+\-[0-9]+,?|[0-9]+,?)+\Z", a):
 361         atype = "span"
 362     elif a in ("melee", "ranged"):
 363         atype = "range"
 364     elif a in ("lawful", "neutral", "chaotic", "liminal"):
 365         atype = "alignment"
 366     elif a.startswith("{") or a.endswith("}") or a.startswith("$"):
 367         atype = None # Can't tell -- it's a macro expansion
 368     elif re.match(image_reference, a) or a == "unit_image":
 369         atype = "image"
 370     elif re.match(r"(\*|[A-Z][a-z]+)\^([A-Z][a-z\\|/]+\Z)?", a):
 371         atype = "terrain_code"
 372     elif a.endswith(".wav") or a.endswith(".ogg"):
 373         atype = "sound"
 374     elif a.startswith('"') and a.endswith('"') or (a.startswith("_") and a[1] not in string.ascii_lowercase):
 375         atype = "stringliteral"
 376     elif "=" in a:
 377         atype = "filter"
 378     elif re.match(r"[A-Z][a-z][a-z]?\Z", a):
 379         atype = "shortname"
 380     elif a == "":
 381         atype = "empty"
 382     elif not ' ' in a:
 383         atype = "name"
 384     else:
 385         atype = "string"
 386     return atype

 388 def argmatch(formals, actuals):
 389     if len(formals) != len(actuals):
 390         return False
 391     for (f, a) in zip(formals, actuals):
 392         # Here's the compatibility logic.  First, we catch the situations
 393         # in which a more restricted actual type matches a more general
 394         # formal one.  Then we have a fallback rule checking for type
 395         # equality or wildcarding.
 396         ftype = formaltype(f)
 397         atype = actualtype(a)
 398         if ftype == "any":
 399             pass
 400         elif (atype == "numeric" or a == "global") and ftype == "side":
 401             pass
 402         elif atype in ("filter", "empty") and ftype == "wml":
 403             pass
 404         elif atype in ("numeric", "position") and ftype == "span":
 405             pass
 406         elif atype in ("shortname", "name", "empty", "stringliteral") and ftype == "affix":
 407             pass
 408         elif atype in ("shortname", "name", "stringliteral") and ftype == "string":
 409             pass
 410         elif atype in ("shortname", "name", "string", "stringliteral", "empty") and ftype == "optional_string":
 411             pass
 412         elif atype in ("shortname",) and ftype == "terrain_code":
 413             pass
 414         elif atype in ("numeric", "position", "span", "empty") and ftype == "alliance":
 415             pass
 416         elif atype in ("terrain_code", "shortname", "name") and ftype == "terrain_pattern":
 417             pass
 418         elif atype in ("string", "shortname", "name") and ftype == "types":
 419             pass
 420         elif atype in ("numeric", "percentage") and ftype == "percentage":
 421             pass
 422         elif atype == "range" and ftype == "name":
 423             pass
 424         elif atype != ftype and ftype is not None and atype is not None:
 425             return False
 426     return True

 435         self.namespace = namespace

 446     def dump_references(self):
 447         "Dump all known references to this definition."
 448         for (file, refs) in self.references.items():
 449             print("    %s: %s" % (file, repr([x[0] for x in refs])[1:-1]))

 451     def __eq__(self, other):
 452         return self.filename == other.filename and self.lineno == other.lineno
 453 
 454     def __gt__(self, other):
 455         # Major sort by file, minor by line number.  This presumes that the
 456         # files correspond to coherent topics and gives us control of the
 457         # sequence.
 458         if self.filename == other.filename:
 459             return self.lineno > other.lineno
 460         else:
 461             return self.filename > other.filename

 463     def mismatches(self):
 464         copy = Reference(self.namespace, self.filename, self.lineno, self.docstring, self.args)
 465         copy.undef = self.undef
 466         for filename in self.references:
 467             mis = [(ln,a) for (ln,a) in self.references[filename] if a is not None and not argmatch(self.args, a)]
 468             if mis:
 469                 copy.references[filename] = mis
 470         return copy
 471     def __str__(self):
 472         if self.lineno:
 473             return '"%s", line %d' % (self.filename, self.lineno)
 474         else:
 475             return self.filename

 482     def mark_matching_resources(self, pattern, fn, n):
 483         "Mark all definitions matching a specified pattern with a reference."
 484         pattern = pattern.replace("+", r"\+")
 485         pattern = os.sep + pattern + "$"
 486         if os.sep == "\\":

 492             return None

 494         for trial in self.fileref:

 496                 key = trial
 497                 self.fileref[key].append(fn, n)

 499     def visible_from(self, defn, fn, n):
 500         "Is specified definition visible from the specified file and line?"
 501         if isinstance(defn, str):
 502             defn = self.fileref[defn]
 503         if defn.undef is not None:
 504             # Local macros are only visible in the file where they were defined
 505             return defn.filename == fn and n >= defn.lineno and n <= defn.undef
 506         if self.exports(defn.namespace):
 507             # Macros and resources in subtrees with export=yes are global
 508             return True
 509         elif not self.filelist.neighbors(defn.filename, fn):
 510             # Otherwise, must be in the same subtree.
 511             return False

 514             # This doesn't match the actual preprocessor semantics.
 515             # It means any macro without an undef is visible anywhere in the
 516             # same argument directory.
 517             #
 518             # We can't do better than this without a lot of hairy graph-
 519             # coloring logic to simulate include path interpretation.
 520             # If that logic ever gets built, it will go here.

 522     def scan_for_definitions(self, namespace, filename):
 523         ignoreflag = False
 524         conditionalsflag = False
 525         with codecs.open(filename, "r", "utf8") as dfp:
 526             state = "outside"
 527             latch_unit = in_base_unit = in_theme = False
 528             for (n, line) in enumerate(dfp):
 529                 if self.warnlevel > 1:
 530                     print(repr(line)[1:-1])
 531                 if line.strip().startswith("#textdomain"):
 532                     continue
 533                 m = re.search("# *wmlscope: warnlevel ([0-9]*)", line)
 534                 if m:
 535                     self.warnlevel = int(m.group(1))
 536                     print('"%s", line %d: warnlevel set to %d (definition-gathering pass)' \
 537                          % (filename, n+1, self.warnlevel))
 538                     continue
 539                 m = re.search("# *wmlscope: set *([^=]*)=(.*)", line)
 540                 if m:
 541                     prop = m.group(1).strip()
 542                     value = m.group(2).strip()
 543                     if namespace not in self.properties:
 544                         self.properties[namespace] = {}
 545                     self.properties[namespace][prop] = value
 546                 m = re.search("# *wmlscope: prune (.*)", line)
 547                 if m:
 548                     name = m.group(1)
 549                     if self.warnlevel >= 2:
 550                         print('"%s", line %d: pruning definitions of %s' \
 551                               % (filename, n+1, name ))
 552                     if name not in self.xref:
 553                         print("wmlscope: can't prune undefined macro %s" % name, file=sys.stderr)
 554                     else:
 555                         self.xref[name] = self.xref[name][:1]
 556                     continue
 557                 if "# wmlscope: start conditionals" in line:
 558                     if self.warnlevel > 1:
 559                         print('"%s", line %d: starting conditionals' \
 560                               % (filename, n+1))
 561                     conditionalsflag = True
 562                 elif "# wmlscope: stop conditionals" in line:
 563                     if self.warnlevel > 1:
 564                         print('"%s", line %d: stopping conditionals' \
 565                               % (filename, n+1))
 566                     conditionalsflag = False
 567                 if "# wmlscope: start ignoring" in line:
 568                     if self.warnlevel > 1:
 569                         print('"%s", line %d: starting ignoring (definition pass)' \
 570                               % (filename, n+1))
 571                     ignoreflag = True
 572                 elif "# wmlscope: stop ignoring" in line:
 573                     if self.warnlevel > 1:
 574                         print('"%s", line %d: stopping ignoring (definition pass)' \
 575                               % (filename, n+1))
 576                     ignoreflag = False
 577                 elif ignoreflag:
 578                     continue
 579                 if line.strip().startswith("#define"):
 580                     tokens = line.split()
 581                     if len(tokens) < 2:
 582                         print('"%s", line %d: malformed #define' \
 583                               % (filename, n+1), file=sys.stderr)
 584                     else:
 585                         name = tokens[1]
 586                         here = Reference(namespace, filename, n+1, line, args=tokens[2:])
 587                         here.hash = hashlib.md5()
 588                         here.docstring = line.lstrip()[8:] # Strip off #define_
 589                         state = "macro_header"
 590                     continue
 591                 elif state != 'outside' and line.strip().endswith("#enddef"):
 592                     here.hash.update(line.encode("utf8"))
 593                     here.hash = here.hash.digest()
 594                     if name in self.xref:
 595                         for defn in self.xref[name]:
 596                             if not self.visible_from(defn, filename, n+1):
 597                                 continue
 598                             elif conditionalsflag:
 599                                 continue
 600                             elif defn.hash != here.hash:
 601                                 print("%s: overrides different %s definition at %s" \
 602                                         % (here, name, defn), file=sys.stderr)
 603                             elif self.warnlevel > 0:
 604                                 print("%s: duplicates %s definition at %s" \
 605                                         % (here, name, defn), file=sys.stderr)
 606                     if name not in self.xref:
 607                         self.xref[name] = []
 608                     self.xref[name].append(here)
 609                     state = "outside"
 610                 elif state == "macro_header" and line.strip() and line.strip()[0] != "#":
 611                     state = "macro_body"
 612                 if state == "macro_header":
 613                     # Ignore macro header commends that are pragmas
 614                     if "wmlscope" not in line and "wmllint:" not in line:
 615                         here.docstring += line.lstrip()[1:]
 616                 if state in ("macro_header", "macro_body"):
 617                     here.hash.update(line.encode("utf8"))
 618                 elif line.strip().startswith("#undef"):
 619                     tokens = line.split()
 620                     name = tokens[1]
 621                     if name in self.xref and self.xref[name]:
 622                         self.xref[name][-1].undef = n+1
 623                     else:
 624                         print("%s: unbalanced #undef on %s" \
 625                               % (Reference(namespace, filename, n+1), name))
 626                 if state == 'outside':
 627                     if '[unit_type]' in line:
 628                         latch_unit = True
 629                     elif '[/unit_type]' in line:
 630                         latch_unit = False
 631                     elif '[base_unit]' in line:
 632                         in_base_unit = True
 633                     elif '[/base_unit]' in line:
 634                         in_base_unit = False
 635                     elif '[theme]' in line:
 636                         in_theme = True
 637                     elif '[/theme]' in line:
 638                         in_theme = False
 639                     elif latch_unit and not in_base_unit and not in_theme and "id" in line:
 640                         m = CrossRef.tag_parse.search(line)
 641                         if m and m.group(1) == "id":
 642                             uid = m.group(2)
 643                             if uid not in self.unit_ids:
 644                                 self.unit_ids[uid] = []
 645                             self.unit_ids[uid].append(Reference(namespace, filename, n+1))
 646                             latch_unit= False
 647     def __init__(self, dirpath=[], exclude="", warnlevel=0, progress=False):
 648         "Build cross-reference object from the specified filelist."
 649         self.filelist = Forest(dirpath, exclude)
 650         self.dirpath = [x for x in dirpath if not re.search(exclude, x)]
 651         self.warnlevel = warnlevel
 652         self.xref = {}
 653         self.fileref = {}
 654         self.noxref = False
 655         self.properties = {}
 656         self.unit_ids = {}
 657         all_in = []
 658         if self.warnlevel >=2 or progress:
 659             print("*** Beginning definition-gathering pass...")
 660         for (namespace, filename) in self.filelist.generator():
 661             all_in.append((namespace, filename))
 662             if self.warnlevel > 1:
 663                 print(filename + ":")
 664             if progress:
 665                 print(filename)
 666             if isresource(filename):
 667                 self.fileref[filename] = Reference(namespace, filename)
 668             elif iswml(filename):
 669                 # It's a WML file, scan for macro definitions
 670                 self.scan_for_definitions(namespace, filename)
 671             elif filename.endswith(".def"):
 672                 # It's a list of names to be considered defined
 673                 self.noxref = True
 674                 with codecs.open(filename, "r", "utf8") as dfp:
 675                     for line in dfp:
 676                         self.xref[line.strip()] = True
 677         # Next, decorate definitions with all references from the filelist.
 678         self.unresolved = []
 679         self.missing = []
 680         formals = []
 681         state = "outside"
 682         if self.warnlevel >=2 or progress:
 683             print("*** Beginning reference-gathering pass...")
 684         for (ns, fn) in all_in:
 685             if progress:
 686                 print(filename)
 687             if iswml(fn):
 688                 with codecs.open(fn, "r", "utf8") as rfp:
 689                     attack_name = None
 690                     beneath = 0
 691                     ignoreflag = False
 692                     for (n, line) in enumerate(rfp):
 693                         if line.strip().startswith("#define"):
 694                             formals = line.strip().split()[2:]
 695                         elif line.startswith("#enddef"):
 696                             formals = []
 697                         comment = ""
 698                         if '#' in line:
 699                             if "# wmlscope: start ignoring" in line:
 700                                 if self.warnlevel > 1:
 701                                     print('"%s", line %d: starting ignoring (reference pass)' \
 702                                           % (fn, n+1))
 703                                 ignoreflag = True
 704                             elif "# wmlscope: stop ignoring" in line:
 705                                 if self.warnlevel > 1:
 706                                     print('"%s", line %d: stopping ignoring (reference pass)' \
 707                                           % (fn, n+1))
 708                                 ignoreflag = False
 709                             m = re.search("# *wmlscope: self.warnlevel ([0-9]*)", line)
 710                             if m:
 711                                 self.warnlevel = int(m.group(1))
 712                                 print('"%s", line %d: self.warnlevel set to %d (reference-gathering pass)' \
 713                                      % (fn, n+1, self.warnlevel))
 714                                 continue
 715                             fields = line.split('#')
 716                             line = fields[0]
 717                             if len(fields) > 1:
 718                                 comment = fields[1]
 719                         if ignoreflag or not line:
 720                             continue
 721                         # Find references to macros
 722                         for match in re.finditer(CrossRef.macro_reference, line):
 723                             name = match.group(1)
 724                             candidates = []
 725                             if self.warnlevel >=2:
 726                                 print('"%s", line %d: seeking definition of %s' \
 727                                       % (fn, n+1, name))
 728                             if name in formals:
 729                                 continue
 730                             elif name in self.xref:
 731                                 # Count the number of actual arguments.
 732                                 # Set args to None if the call doesn't
 733                                 # close on this line
 734                                 (args, brackdepth, parendepth) = parse_macroref(match.start(0), line)
 735                                 if brackdepth > 0 or parendepth > 0:
 736                                     args = None
 737                                 else:
 738                                     args.pop(0)
 739                                 #if args:
 740                                 #    print('"%s", line %d: args of %s is %s' \
 741                                 #          % (fn, n+1, name, args))
 742                                 # Figure out which macros might resolve this
 743                                 for defn in self.xref[name]:
 744                                     if self.visible_from(defn, fn, n+1):
 745                                         defn.append(fn, n+1, args)
 746                                         candidates.append(str(defn))
 747                                 if len(candidates) > 1:
 748                                     print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns, fn, n), name, "; ".join(candidates)))
 749                             if len(candidates) == 0:
 750                                 self.unresolved.append((name,Reference(ns,fn,n+1)))
 751                         # Don't be fooled by HTML image references in help strings.
 752                         if "<img>" in line:
 753                             continue
 754                         # Find references to resource files
 755                         for match in re.finditer(CrossRef.file_reference, line):
 756                             for name in expand_square_braces(match.group(0)):
 757                                 # Catches maps that look like macro names.
 758                                 if (name.endswith(".map") or name.endswith(".mask")) and name[0] == '{':
 759                                     name = name[1:]
 760                                 if os.sep == "\\":
 761                                     name = name.replace("/", "\\")
 762                                 key = None
 763                                 # If name is already in our resource list, it's easy.
 764                                 if name in self.fileref and self.visible_from(name, fn, n):
 765                                     self.fileref[name].append(fn, n+1)
 766                                     continue
 767                                 # If the name contains substitutable parts, count
 768                                 # it as a reference to everything the substitutions
 769                                 # could potentially match.
 770                                 elif '{' in name or '@' in name:
 771                                     pattern = re.sub(r"(\{[^}]*\}|@R0|@V)", '.*', name)
 772                                     key = self.mark_matching_resources(pattern, fn,n+1)
 773                                     if key:
 774                                         self.fileref[key].append(fn, n+1)
 775                                 else:
 776                                     candidates = []
 777                                     for trial in self.fileref:
 778                                         if trial.endswith(os.sep + name) and self.visible_from(trial, fn, n):
 779                                             key = trial
 780                                             self.fileref[trial].append(fn, n+1)
 781                                             candidates.append(trial)
 782                                     if len(candidates) > 1:
 783                                         print("%s: more than one resource matching %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 784                                 if not key:
 785                                     self.missing.append((name, Reference(ns,fn,n+1)))
 786                         # Notice implicit references through attacks
 787                         if state == "outside":
 788                             if "[attack]" in line:
 789                                 beneath = 0
 790                                 attack_name = default_icon = None
 791                                 have_icon = False
 792                             elif "name=" in line and not "no-icon" in comment:
 793                                 attack_name = line[line.find("name=")+5:].strip()
 794                                 default_icon = os.path.join("attacks", attack_name  + ".png")
 795                             elif "icon=" in line and beneath == 0:
 796                                 have_icon = True
 797                             elif "[/attack]" in line:
 798                                 if attack_name and not have_icon:
 799                                     candidates = []
 800                                     key = None
 801                                     for trial in self.fileref:
 802                                         if trial.endswith(os.sep + default_icon) and self.visible_from(trial, fn, n):
 803                                             key = trial
 804                                             self.fileref[trial].append(fn, n+1)
 805                                             candidates.append(trial)
 806                                     if len(candidates) > 1:
 807                                         print("%s: more than one definition of %s is visible here (%s)." % (Reference(ns,fn, n), name, ", ".join(candidates)))
 808                                 if not key:
 809                                     self.missing.append((default_icon, Reference(ns,fn,n+1)))
 810                             elif line.strip().startswith("[/"):
 811                                 beneath -= 1
 812                             elif line.strip().startswith("["):
 813                                 beneath += 1
 814         # Check whether each namespace has a defined export property
 815         for namespace in self.dirpath:
 816             if namespace not in self.properties or "export" not in self.properties[namespace]:
 817                 print("warning: %s has no export property" % namespace)

 854 class Translation(dict):
 855     "Parses a po file to create a translation dictionary."
 856     def __init__(self, textdomain, isocode, topdir=""):
 857         self.textdomain = textdomain
 858         self.isocode = isocode
 859         self.gettext = {}
 860         if self.isocode != "C":
 861             isocode2 = isocode[:isocode.rfind("_")]
 862             for code in [isocode, isocode2]:
 863                 fn = "po/%s/%s.po" % (textdomain, code)
 864                 if topdir: fn = os.path.join(topdir, fn)
 865                 try:
 866                     f = file(fn)
 867                     break
 868                 except IOError:
 869                     pass
 870             else:
 871                 raise TranslationError(textdomain, self.isocode)
 872 
 873             expect = False
 874             fuzzy = "#, fuzzy\n"
 875             gettext = f.read().decode("utf8")
 876             matches = re.compile("""(msgid|msgstr)((\s*".*?")+)""").finditer(gettext)
 877             msgid = ""
 878             for match in matches:
 879                 text = "".join(re.compile('"(.*?)"').findall(match.group(2)))
 880                 if match.group(1) == "msgid":
 881                     msgid = text.replace("\\n", "\n")
 882                     expect = gettext[match.start(1) - len(fuzzy):match.start(1)] != fuzzy
 883                 elif expect:
 884                     self.gettext[msgid] = text.replace("\\n", "\n")
 885     def get(self, key, dflt):
 886         if self.isocode == "C":
 887             if key:
 888                 return key[key.find("^") + 1:]
 889             return "?"
 890         else:
 891             t = self.gettext.get(key, dflt)
 892             if not t:
 893                 if key:
 894                     return key[key.find("^") + 1:]
 895                 return "?"
 896             return t
 897     def __getitem__(self, key):
 898         if self.isocode == "C":
 899             return key
 900         else:
 901             return self.gettext[key]
 902     def __contains__(self, key):
 903         if self.isocode == "C":
 904             return True
 905         else:
 906             return key in self.gettext

 908 class Translations:
 909     "Wraps around Translation to support multiple languages and domains."
 910     def __init__(self, topdir = ""):
 911         self.translations = {}
 912         self.topdir = topdir
 913     def get(self, textdomain, isocode, key, default):
 914         t = (textdomain, isocode)
 915         if not t in self.translations:
 916             try:
 917                 self.translations[t] = Translation(textdomain, isocode, self.topdir)
 918             except TranslationError as e:
 919                 print(str(e), file=sys.stderr)
 920                 self.translations[t] = Translation(textdomain, "C", self.topdir)
 921         result = self.translations[t].get(key, default)
