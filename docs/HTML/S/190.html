  44 unit* lua_unit::get()
  45 {
  46         if (ptr) return ptr.get();
  47         if (c_ptr) return c_ptr;
  48         if (side) {
  49                 return resources::gameboard->get_team(side).recall_list().find_if_matches_underlying_id(uid).get();
  50         }
  51         unit_map::unit_iterator ui = resources::gameboard->units().find(uid);
  52         if (!ui.valid()) return nullptr;
  53         return ui.get_shared_ptr().get(); //&*ui would not be legal, must get new shared_ptr by copy ctor because the unit_map itself is holding a boost shared pointer.

  55 unit_ptr lua_unit::get_shared()
  56 {
  57         if (ptr) return ptr;
  58         if (side) {
  59                 return resources::gameboard->get_team(side).recall_list().find_if_matches_underlying_id(uid);
  60         }
  61         unit_map::unit_iterator ui = resources::gameboard->units().find(uid);
  62         if (!ui.valid()) return unit_ptr();
  63         return ui.get_shared_ptr(); //&*ui would not be legal, must get new shared_ptr by copy ctor because the unit_map itself is holding a boost shared pointer.

  66 // Having this function here not only simplifies other code, it allows us to move
  67 // pointers around from one structure to another.
  68 // This makes bare pointer->map in particular about 2 orders of magnitude faster,
  69 // as benchmarked from Lua code.
  70 bool lua_unit::put_map(const map_location &loc)
  71 {
  72         if (ptr) {
  73                 std::pair<unit_map::unit_iterator, bool> res = resources::gameboard->units().replace(loc, ptr);
  74                 if (res.second) {
  75                         ptr.reset();
  76                         uid = res.first->underlying_id();
  77                 } else {
  78                         ERR_LUA << "Could not move unit " << ptr->underlying_id() << " onto map location " << loc << '\n';
  79                         return false;
  80                 }
  81         } else if (side) { // recall list
  82                 unit_ptr it = resources::gameboard->get_team(side).recall_list().extract_if_matches_underlying_id(uid);
  83                 if (it) {
  84                         side = 0;
  85                         // uid may be changed by unit_map on insertion
  86                         uid = resources::gameboard->units().replace(loc, it).first->underlying_id();
  87                 } else {
  88                         ERR_LUA << "Could not find unit " << uid << " on recall list of side " << side << '\n';
  89                         return false;
  90                 }
  91         } else { // on map
  92                 unit_map::unit_iterator ui = resources::gameboard->units().find(uid);
  93                 if (ui != resources::gameboard->units().end()) {
  94                         map_location from = ui->get_location();
  95                         if (from != loc) { // This check is redundant in current usage
  96                                 resources::gameboard->units().erase(loc);
  97                                 resources::gameboard->units().move(from, loc);
  98                         }
  99                         // No need to change our contents
 100                 } else {
 101                         ERR_LUA << "Could not find unit " << uid << " on the map" << std::endl;
 102                         return false;
 103                 }
 104         }
 105         return true;
 106 }

 120 static lua_unit* internal_get_unit(lua_State *L, int index, bool only_on_map, int& error)
 121 {
 122         error = LU_OK;
 123         if(!luaW_isunit(L, index)) {
 124                 error = LU_NOT_UNIT;
 125                 return nullptr;
 126         }
 127         lua_unit* lu = static_cast<lua_unit*>(lua_touserdata(L, index));
 128         if(only_on_map && !lu->on_map()) {
 129                 error = LU_NOT_ON_MAP;
 130         }
 131         if(!lu->get()) {
 132                 error = LU_NOT_VALID;
 133         }
 134         return lu;
 135 }

 236 /**
 237  * Gets some data on a unit (__index metamethod).
 238  * - Arg 1: full userdata containing the unit id.
 239  * - Arg 2: string containing the name of the property.
 240  * - Ret 1: something containing the attribute.
 241  */
 242 static int impl_unit_get(lua_State *L)
 243 {
 244         lua_unit *lu = static_cast<lua_unit *>(lua_touserdata(L, 1));
 245         char const *m = luaL_checkstring(L, 2);
 246         const unit* pu = lu->get();
 247 
 248         if(strcmp(m, "valid") == 0) {
 249                 if(!pu) {
 250                         return 0;
 251                 }
 252                 if(lu->on_map()) {
 253                         lua_pushstring(L, "map");
 254                 } else if(lu->on_recall_list()) {
 255                         lua_pushstring(L, "recall");
 256                 } else {
 257                         lua_pushstring(L, "private");
 258                 }
 259                 return 1;
 260         }
 261 
 262         if(!pu) {
 263                 return luaL_argerror(L, 1, "unknown unit");
 264         }
 265 
 266         unit const &u = *pu;
 267 
 268         // Find the corresponding attribute.
 269         return_int_attrib("x", u.get_location().wml_x());
 270         return_int_attrib("y", u.get_location().wml_y());
 271         if(strcmp(m, "loc") == 0) {
 272                 lua_pushinteger(L, u.get_location().wml_x());
 273                 lua_pushinteger(L, u.get_location().wml_y());
 274                 return 2;
 275         }
 276         return_int_attrib("side", u.side());
 277         return_string_attrib("id", u.id());
 278         return_string_attrib("type", u.type_id());
 279         return_string_attrib("image_mods", u.effect_image_mods());
 280         return_string_attrib("usage", u.usage());
 281         return_int_attrib("hitpoints", u.hitpoints());
 282         return_int_attrib("max_hitpoints", u.max_hitpoints());
 283         return_int_attrib("experience", u.experience());
 284         return_int_attrib("max_experience", u.max_experience());
 285         return_int_attrib("recall_cost", u.recall_cost());
 286         return_int_attrib("moves", u.movement_left());
 287         return_int_attrib("max_moves", u.total_movement());
 288         return_int_attrib("max_attacks", u.max_attacks());
 289         return_int_attrib("attacks_left", u.attacks_left());
 290         return_tstring_attrib("name", u.name());
 291         return_bool_attrib("canrecruit", u.can_recruit());
 292         return_int_attrib("level", u.level());
 293         return_int_attrib("cost", u.cost());
 294 
 295         return_vector_string_attrib("extra_recruit", u.recruits());
 296         return_vector_string_attrib("advances_to", u.advances_to());
 297 
 298         if(strcmp(m, "alignment") == 0) {
 299                 lua_push(L, u.alignment());
 300                 return 1;
 301         }
 302 
 303         if(strcmp(m, "upkeep") == 0) {
 304                 unit::upkeep_t upkeep = u.upkeep_raw();
 305 
 306                 // Need to keep these seperate in order to ensure an int value is always used if applicable.
 307                 if(int* v = boost::get<int>(&upkeep)) {
 308                         lua_push(L, *v);
 309                 } else {
 310                         const std::string type = boost::apply_visitor(unit::upkeep_type_visitor(), upkeep);
 311                         lua_push(L, type);
 312                 }
 313 
 314                 return 1;
 315         }
 316         if(strcmp(m, "advancements") == 0) {
 317                 lua_push(L, u.modification_advancements());
 318                 return 1;
 319         }
 320         if(strcmp(m, "overlays") == 0) {
 321                 lua_push(L, u.overlays());
 322                 return 1;
 323         }
 324         if(strcmp(m, "traits") == 0) {
 325                 lua_push(L, u.get_traits_list());
 326                 return 1;
 327         }
 328         if(strcmp(m, "abilities") == 0) {
 329                 lua_push(L, u.get_ability_list());
 330                 return 1;
 331         }
 332         if(strcmp(m, "status") == 0) {
 333                 lua_createtable(L, 1, 0);
 334                 lua_pushvalue(L, 1);
 335                 lua_rawseti(L, -2, 1);
 336                 luaL_setmetatable(L, ustatusKey);
 337                 return 1;
 338         }
 339         if(strcmp(m, "variables") == 0) {
 340                 lua_createtable(L, 1, 0);
 341                 lua_pushvalue(L, 1);
 342                 lua_rawseti(L, -2, 1);
 343                 luaL_setmetatable(L, unitvarKey);
 344                 return 1;
 345         }
 346         if(strcmp(m, "attacks") == 0) {
 347                 push_unit_attacks_table(L, 1);
 348                 return 1;
 349         }
 350         return_vector_string_attrib("animations", u.anim_comp().get_flags());
 351         return_cfg_attrib("recall_filter", cfg = u.recall_filter());
 352         return_bool_attrib("hidden", u.get_hidden());
 353         return_bool_attrib("petrified", u.incapacitated());
 354         return_bool_attrib("resting", u.resting());
 355         return_string_attrib("role", u.get_role());
 356         return_string_attrib("race", u.race()->id());
 357         return_string_attrib("gender", gender_string(u.gender()));
 358         return_string_attrib("variation", u.variation());
 359         return_bool_attrib("zoc", u.get_emit_zoc());
 360         return_string_attrib("facing", map_location::write_direction(u.facing()));
 361         return_string_attrib("portrait", u.big_profile() == u.absolute_image() ? u.absolute_image() + u.image_mods() : u.big_profile());
 362         return_cfg_attrib("__cfg", u.write(cfg); u.get_location().write(cfg));
 363 
 364         if(luaW_getmetafield(L, 1, m)) {
 365                 return 1;
 366         }
 367         return 0;
 368 }

 370 /**
 371  * Sets some data on a unit (__newindex metamethod).
 372  * - Arg 1: full userdata containing the unit id.
 373  * - Arg 2: string containing the name of the property.
 374  * - Arg 3: something containing the attribute.

 376 static int impl_unit_set(lua_State *L)
 377 {
 378         lua_unit *lu = static_cast<lua_unit *>(lua_touserdata(L, 1));
 379         char const *m = luaL_checkstring(L, 2);
 380         unit* pu = lu->get();
 381         if (!pu) return luaL_argerror(L, 1, "unknown unit");
 382         unit &u = *pu;
 383 
 384         // Find the corresponding attribute.
 385         //modify_int_attrib_check_range("side", u.set_side(value), 1, static_cast<int>(teams().size())); TODO: Figure out if this is a good idea, to refer to teams() and make this depend on having a gamestate
 386         modify_int_attrib("side", u.set_side(value));
 387         modify_int_attrib("moves", u.set_movement(value));
 388         modify_int_attrib("hitpoints", u.set_hitpoints(value));
 389         modify_int_attrib("experience", u.set_experience(value));
 390         modify_int_attrib("recall_cost", u.set_recall_cost(value));
 391         modify_int_attrib("attacks_left", u.set_attacks(value));
 392         modify_int_attrib("level", u.set_level(value));
 393         modify_bool_attrib("resting", u.set_resting(value));
 394         modify_tstring_attrib("name", u.set_name(value));
 395         modify_string_attrib("role", u.set_role(value));
 396         modify_string_attrib("facing", u.set_facing(map_location::parse_direction(value)));
 397         modify_bool_attrib("hidden", u.set_hidden(value));
 398         modify_bool_attrib("zoc", u.set_emit_zoc(value));
 399         modify_bool_attrib("canrecruit", u.set_can_recruit(value));
 400 
 401         modify_vector_string_attrib("extra_recruit", u.set_recruits(value));
 402         modify_vector_string_attrib("advances_to", u.set_advances_to(value));
 403         if(strcmp(m, "alignment") == 0) {
 404                 u.set_alignment(lua_check<unit_type::ALIGNMENT>(L, 3));
 405                 return 0;
 406         }
 407 
 408         if(strcmp(m, "advancements") == 0) {
 409                 u.set_advancements(lua_check<std::vector<config> >(L, 3));
 410                 return 0;
 411         }
 412 
 413         if(strcmp(m, "upkeep") == 0) {
 414                 if(lua_isnumber(L, 3)) {
 415                         u.set_upkeep(luaL_checkinteger(L, 3));
 416                         return 0;
 417                 }
 418                 const char* v = luaL_checkstring(L, 3);
 419                 if((strcmp(v, "loyal") == 0) || (strcmp(v, "free") == 0)) {
 420                         u.set_upkeep(unit::upkeep_loyal());
 421                 } else if(strcmp(v, "full") == 0) {
 422                         u.set_upkeep(unit::upkeep_full());
 423                 } else {
 424                         std::string err_msg = "unknown upkeep value of unit: ";
 425                         err_msg += v;
 426                         return luaL_argerror(L, 2, err_msg.c_str());
 427                 }
 428                 return 0;
 429         }
 430         if(!lu->on_map()) {
 431                 map_location loc = u.get_location();
 432                 modify_int_attrib("x", loc.set_wml_x(value); u.set_location(loc));
 433                 modify_int_attrib("y", loc.set_wml_y(value); u.set_location(loc));
 434         }
 435 
 436         std::string err_msg = "unknown modifiable property of unit: ";
 437         err_msg += m;
 438         return luaL_argerror(L, 2, err_msg.c_str());
 439 }
