  79 wesnoth.wml_actions["for"] = function(cfg)
  80         if helper.child_count(cfg, "do") == 0 then
  81                 helper.wml_error "[for] does not contain any [do] tags"
  82         end
  83         
  84         local loop_lim = {}
  85         local first
  86         if cfg.array ~= nil then
  87                 if cfg.reverse then
  88                         first = wesnoth.get_variable(cfg.array .. ".length") - 1
  89                         loop_lim.last = 0
  90                         loop_lim.step = -1
  91                 else
  92                         first = 0
  93                         loop_lim.last = '$($' .. cfg.array .. ".length - 1)"
  94                         loop_lim.step = 1
  95                 end
  96         else
  97                 -- Get a literal config to fetch end and step;
  98                 -- this done is to delay expansion of variables
  99                 local cfg_lit = helper.literal(cfg)
 100                 first = cfg.start or 0
 101                 loop_lim.last = cfg_lit["end"] or first
 102                 if cfg.step then loop_lim.step = cfg_lit.step end
 103         end
 104         loop_lim = wesnoth.tovconfig(loop_lim)
 105         if loop_lim.step == 0 then -- Sanity check
 106                 helper.wml_error("[for] has a step of 0!")
 107         end
 108         if loop_lim.step ~= nil then
 109                 if (first < loop_lim.last and loop_lim.step <= 0)
 110                                 or (first > loop_lim.last and loop_lim.step >= 0) then
 111                         -- Sanity check: If they specify something like start,end,step=1,4,-1
 112                         -- then we do nothing
 113                         return
 114                 end
 115         end
 116         local i_var = cfg.variable or "i"
 117         local save_i = utils.start_var_scope(i_var)
 118         wesnoth.set_variable(i_var, first)
 119         local function loop_condition()
 120                 local sentinel = loop_lim.last
 121                 if loop_lim.step then
 122                         sentinel = sentinel + loop_lim.step
 123                         if loop_lim.step > 0 then
 124                                 return wesnoth.get_variable(i_var) < sentinel
 125                         else
 126                                 return wesnoth.get_variable(i_var) > sentinel
 127                         end
 128                 elseif loop_lim.last < first then
 129                         sentinel = sentinel - 1
 130                         return wesnoth.get_variable(i_var) > sentinel
 131                 else
 132                         sentinel = sentinel + 1
 133                         return wesnoth.get_variable(i_var) < sentinel
 134                 end
 135         end
 136         while loop_condition() do
 137                 for do_child in helper.child_range( cfg, "do" ) do
 138                         local action = utils.handle_event_commands(do_child, "loop")
 139                         if action == "break" then
 140                                 utils.set_exiting("none")
 141                                 goto exit
 142                         elseif action == "continue" then
 143                                 utils.set_exiting("none")
 144                                 break
 145                         elseif action ~= "none" then
 146                                 goto exit
 147                         end
 148                 end
 149                 wesnoth.set_variable(i_var, wesnoth.get_variable(i_var) + loop_lim.step)
 150         end
 151         ::exit::
 152         utils.end_var_scope(i_var, save_i)
 153 end

 177 function wml_actions.foreach(cfg)
 178         if helper.child_count(cfg, "do") == 0 then
 179                 helper.wml_error "[foreach] does not contain any [do] tags"
 180         end
 181         
 182         local array_name = cfg.array or helper.wml_error "[foreach] missing required array= attribute"
 183         local array = helper.get_variable_array(array_name)
 184         if #array == 0 then return end -- empty and scalars unwanted
 185         local item_name = cfg.variable or "this_item"
 186         local this_item = utils.start_var_scope(item_name) -- if this_item is already set
 187         local i_name = cfg.index_var or "i"
 188         local i = utils.start_var_scope(i_name) -- if i is already set
 189         local array_length = wesnoth.get_variable(array_name .. ".length")
 190         
 191         for index, value in ipairs(array) do
 192                 -- Some protection against external modification
 193                 -- It's not perfect, though - it'd be nice if *any* change could be detected
 194                 if array_length ~= wesnoth.get_variable(array_name .. ".length") then
 195                         helper.wml_error("WML array length changed during [foreach] iteration")
 196                 end
 197                 wesnoth.set_variable(item_name, value)
 198                 -- set index variable
 199                 wesnoth.set_variable(i_name, index-1) -- here -1, because of WML array
 200                 -- perform actions
 201                 for do_child in helper.child_range(cfg, "do") do
 202                         local action = utils.handle_event_commands(do_child, "loop")
 203                         if action == "break" then
 204                                 utils.set_exiting("none")
 205                                 goto exit
 206                         elseif action == "continue" then
 207                                 utils.set_exiting("none")
 208                                 break
 209                         elseif action ~= "none" then
 210                                 goto exit
 211                         end
 212                 end
 213                 -- set back the content, in case the author made some modifications
 214                 if not cfg.readonly then
 215                         array[index] = wesnoth.get_variable(item_name)
 216                 end
 217         end
 218         ::exit::
 219         
 220         -- house cleaning
 221         utils.end_var_scope(item_name, this_item)
 222         utils.end_var_scope(i_name, i)
 223         
 224         --[[
 225                 This forces the readonly key to be taken literally.
 226                 
 227                 If readonly=yes, then this line guarantees that the array
 228                 is unchanged after the [foreach] loop ends.
 229                 
 230                 If readonly=no, then this line updates the array with any
 231                 changes the user has applied through the $this_item
 232                 variable (or whatever variable was given in item_var).
 233                 
 234                 Note that altering the array via indexing (with the index_var)
 235                 is not supported; any such changes will be reverted by this line.
 236         ]]
 237         helper.set_variable_array(array_name, array)
 238 end

 240 function wml_actions.switch(cfg)
 241         local var_value = wesnoth.get_variable(cfg.variable)
 242         local found = false
 243 
 244         -- Execute all the [case]s where the value matches.
 245         for v in helper.child_range(cfg, "case") do
 246                 for w in utils.split(v.value) do
 247                         if w == tostring(var_value) then 
 248                                 local action = utils.handle_event_commands(v, "switch")
 249                                 found = true
 250                                 if action ~= "none" then goto exit end
 251                                 break
 252                         end
 253                 end
 254         end
 255         ::exit::
 256 
 257         -- Otherwise execute [else] statements.
 258         if not found then
 259                 for v in helper.child_range(cfg, "else") do
 260                         local action = utils.handle_event_commands(v, "switch")
 261                         if action ~= "none" then break end
 262                 end
 263         end
 264 end
