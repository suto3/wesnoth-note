   2 wmliterator.py -- Python routines for navigating a Battle For Wesnoth WML tree
   3 Author: Sapient (Patrick Parker), 2007
   4 
   5 Purpose:
   6  The WmlIterator class can be used to analyze and search the structure of WML
   7  files non-invasively (i.e. preserving existing line structure), and its main
   8  use is to determine when a transformation of deprecated content needs to take
   9  place. (I wrote it was because wmllint was trying to do a lot of things with
  10  regular expressions which really required a more algorithmic approach. Also,
  11  wmllint was often inconsistent with correct handling of comments and values
  12  inside strings.)
  13 
  14 Limitations:
  15  The WmlIterator does not attempt to expand macros, only to recognize them as
  16  another level of nesting. Also, the handling of multiple assignment syntax
  17  is somewhat limited (for similar reasons). Adding seamless support for these
  18  would be ideal, but it presents a design challenge since the iteration is
  19  supposed to be non-invasive. Thus, the current behavior is considered good
  20  enough for now.
  21 """

 108 class WmlIterator(object):
 109     """Return an iterable WML navigation object.
 110     Initialize with a list of lines or a file; if the the line list is
 111     empty and the filename is specified, lines will be read from the file.
 112 
 113     Note: if changes are made to lines while iterating, this may produce
 114     unexpected results. In such case, seek() to the line number of a
 115     scope behind where changes were made.
 116 Important Attributes:
 117     lines - this is an internal list of all the physical lines
 118     scopes - this is an internal list of all open scopes (as iterators)
 119              note: when retrieving an iterator from this list, always
 120              use a copy to perform seek() or next(), and not the original
 121     element - the wml tag, key, or macro name for this logical line
 122               (in complex cases, this may be a tuple of elements...
 123               see parseElements for list of possible values)
 124     text - the exact text of this logical line, as it appears in the
 125            original source, and ending with a newline
 126            note: the logical line also includes multi-line quoted strings
 127     span - the number of physical lines in this logical line:
 128            always 1, unless text contains a multi-line quoted string
 129     lineno - a zero-based line index marking where this text begins
 130     """
 131     def __init__(self, lines=None, filename=None, begin=-1):
 132         "Initialize a new WmlIterator."
 133         self.fname = filename
 134         if lines is None:
 135             lines = []
 136             if filename:
 137                 try:
 138                     with codecs.open(self.fname, "r", "utf8") as ifp:
 139                         lines = ifp.readlines()
 140                 except Exception:
 141                     self.printError('error opening file')
 142         self.lines = lines
 143         self.reset()
 144         self.seek(begin)

 146     def parseQuotes(self, lines):
 147         """Return the line or multiline text if a quote spans multiple lines"""
 148         text = lines[self.lineno]
 149         span = 1
 150         begincomment = text.find('#')
 151         if begincomment < 0:
 152             begincomment = None
 153         beginquote = text[:begincomment].find('<<')
 154         while beginquote >= 0:
 155             endquote = -1
 156             beginofend = beginquote+2
 157             while endquote < 0:
 158                 endquote = text.find('>>', beginofend)
 159                 if endquote < 0:
 160                     if self.lineno + span >= len(lines):
 161                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 162                         return text, span
 163                     beginofend = len(text)
 164                     text += lines[self.lineno + span]
 165                     span += 1
 166             begincomment = text.find('#', endquote+2)
 167             if begincomment < 0:
 168                 begincomment = None
 169             beginquote = text[:begincomment].find('<<', endquote+2)
 170         beginquote = text[:begincomment].find('"')
 171         while beginquote >= 0:
 172             endquote = -1
 173             beginofend = beginquote+1
 174             while endquote < 0:
 175                 endquote = text.find('"', beginofend)
 176                 if endquote < 0:
 177                     if self.lineno + span >= len(lines):
 178                         self.printError('reached EOF due to unterminated string at line', self.lineno+1)
 179                         return text, span
 180                     beginofend = len(text)
 181                     text += lines[self.lineno + span]
 182                     span += 1
 183             begincomment = text.find('#', endquote+1)
 184             if begincomment < 0:
 185                 begincomment = None
 186             beginquote = text[:begincomment].find('"', endquote+1)
 187         return text, span

 189     def closeScope(self, scopes, closerElement):
 190         """Close the most recently opened scope. Return false if not enough scopes.
 191         note: directives close all the way back to the last open directive
 192         non-directives cannot close a directive and will no-op in that case."""
 193         try:
 194             if isDirective(closerElement):
 195                 while not isDirective(scopes.pop()):
 196                     pass
 197             elif (closerElement==closeMacroType):
 198                 elem = ''
 199                 while not elem.startswith('{'):
 200                     closed = scopes.pop()
 201                     elem = closed
 202                     if isinstance(closed, WmlIterator):
 203                         elem = closed.element
 204                     if isDirective(elem):
 205                         self.printScopeError(closerElement)
 206                         scopes.append(closed) # to reduce additional errors (hopefully)
 207                         return True
 208             elif not isDirective(scopes[-1]):
 209                 closed = scopes.pop()
 210                 elem = closed
 211                 if isinstance(closed, WmlIterator):
 212                     elem = closed.element
 213                 if (elem.startswith('{') and closerElement != closeMacroType):
 214                     scopes.append(closed)
 215                 elif (isOpener(elem) and closerElement != '[/'+elem[1:]
 216                 and '+'+closerElement != elem[1]+'[/'+elem[2:]):
 217                     self.printError('reached', closerElement, 'at line', self.lineno+1, 'before closing scope', elem)
 218                     scopes.append(closed) # to reduce additional errors (hopefully)
 219             return True
 220         except IndexError:
 221             return False

 223     def parseElements(self, text):
 224         """Remove any closed scopes, return a list of element names
 225         and list of new unclosed scopes
 226     Element Types:
 227         tags: one of "[tag_name]" or "[/tag_name]"
 228             [tag_name] - opens a scope
 229             [/tag_name] - closes a scope
 230         keys: either "key=" or ("key1=", "key2=") for multi-assignment
 231             key= - does not affect the scope
 232             key1,key2= - multi-assignment returns multiple elements
 233         directives: one of "#ifdef", "#ifndef", "#ifhave", "#ifnhave", "#ifver", "#ifnver", "#else", "#endif", "#define", "#enddef"
 234             #ifdef - opens a scope
 235             #ifndef - opens a scope
 236             #ifhave - opens a scope
 237             #ifnhave - opens a scope
 238             #ifver - opens a scope
 239             #ifnver - opens a scope
 240             #else - closes a scope, also opens a new scope
 241             #endif - closes a scope
 242             #define - opens a scope
 243             #enddef - closes a scope
 244         macro calls: "{MACRO_NAME}"
 245             {MACRO_NAME - opens a scope
 246             } - closes a scope
 247         """
 248         elements = [] #(elementType, sortPos, scopeDelta)
 249         # first remove any lua strings
 250         beginquote = text.find('<<')
 251         while beginquote >= 0:
 252             endquote = text.find('>>', beginquote+2)
 253             if endquote < 0:
 254                 text = text[:beginquote]
 255                 beginquote = -1 #terminate loop
 256             else:
 257                 text = text[:beginquote] + text[endquote+2:]
 258                 beginquote = text.find('<<')
 259         # remove any quoted strings
 260         beginquote = text.find('"')
 261         while beginquote >= 0:
 262             endquote = text.find('"', beginquote+1)
 263             if endquote < 0:
 264                 text = text[:beginquote]
 265                 beginquote = -1 #terminate loop
 266             else:
 267                 text = text[:beginquote] + text[endquote+1:]
 268                 beginquote = text.find('"')
 269         # next remove any comments
 270         text = text.lstrip()
 271         commentSearch = 1
 272         if text.startswith('#ifdef'):
 273             return (['#ifdef'],)*2
 274         elif text.startswith('#ifndef'):
 275             return (['#ifndef'],)*2
 276         elif text.startswith('#ifhave'):
 277             return (['#ifhave'],)*2
 278         elif text.startswith('#ifnhave'):
 279             return (['#ifnhave'],)*2
 280         elif text.startswith('#ifver'):
 281             return (['#ifver'],)*2
 282         elif text.startswith('#ifnver'):
 283             return (['#ifnver'],)*2
 284         elif text.startswith('#else'):
 285             if not self.closeScope(self.scopes, '#else'):
 286                 self.printScopeError('#else')
 287             return (['#else'],)*2
 288         elif text.startswith('#endif'):
 289             if not self.closeScope(self.scopes, '#endif'):
 290                 self.printScopeError('#endif')
 291                 return ['#endif'], []
 292         elif text.startswith('#define'):
 293             return (['#define'],)*2
 294         elif text.find('#enddef') >= 0:
 295             elements.append(('#enddef', text.find('#enddef'), -1))
 296         elif text.startswith('#po:') or text.startswith('# po:'):
 297             elements.append(("#po", 0, 0))
 298         else:
 299             commentSearch = 0
 300         begincomment = text.find('#', commentSearch)
 301         if begincomment >= 0:
 302             text = text[:begincomment]
 303         #now find elements in a loop
 304         for m in tagPattern.finditer(text):
 305             delta = 1
 306             if isCloser(m.group(2)):
 307                 delta = -1
 308             elements.append((m.group(2), m.start(), delta))
 309         for m in keyPattern.finditer(text):
 310             for i, k in enumerate(keySplit.split(m.group(0))):
 311                 if k:
 312                     elements.append((k+'=', m.start()+i, 0))
 313         for m in macroOpenPattern.finditer(text):
 314             elements.append((m.group(1), m.start(), 1))
 315         for m in macroClosePattern.finditer(text):
 316             elements.append((closeMacroType, m.start(), -1))
 317         #sort by start position
 318         elements.sort(key=lambda x:x[1])
 319         resultElements = []
 320         openedScopes = []
 321         for elem, sortPos, scopeDelta in elements:
 322             while scopeDelta < 0:
 323                 if not(self.closeScope(openedScopes, elem)\
 324                         or self.closeScope(self.scopes, elem)):
 325                     self.printScopeError(elem)
 326                 scopeDelta += 1
 327             while scopeDelta > 0:
 328                 openedScopes.append(elem)
 329                 scopeDelta -= 1
 330             resultElements.append(elem)
 331         return resultElements, openedScopes

 359     def seek(self, lineno, clearEnd=True):
 360         """Move the iterator to a specific line number"""
 361         if clearEnd:
 362             self.endScope = None
 363         if lineno < self.lineno:
 364             for scope in reversed(self.scopes):
 365                 # if moving backwards, try to re-use a scope iterator
 366                 if scope.lineno <= lineno:
 367                     # copy the scope iterator's state to self
 368                     self.__dict__ = dict(scope.__dict__)
 369                     self.scopes = scope.scopes[:]
 370                     self.nextScopes = scope.nextScopes[:]
 371                     break
 372             else:
 373                 # moving backwards past all scopes forces a reset
 374                 self.reset()
 375         while self.lineno + self.span - 1 < lineno:
 376             self.next()
 377         return self

 410     def next(self):
 411         """Move the iterator to the next line number
 412         note: May raise StopIteration"""
 413         if not self.hasNext():
 414             if self.scopes:
 415                 self.printError("reached EOF with open scopes", self.scopes)
 416             raise StopIteration
 417         self.lineno = self.lineno + self.span
 418         self.text, self.span = self.parseQuotes(self.lines)
 419         self.scopes.extend(self.nextScopes)
 420         self.element, nextScopes = self.parseElements(self.text)
 421         self.nextScopes = []
 422         for elem in nextScopes:
 423         # remember scopes by storing a copy of the iterator
 424             copyItor = self.copy()
 425             copyItor.element = elem
 426             self.nextScopes.append(copyItor)
 427             copyItor.nextScopes.append(copyItor)
 428         if(len(self.element) == 1):
 429             # currently we only wish to handle simple single assignment syntax
 430             self.element = self.element[0]
 431         if self.endScope is not None and not self.scopes.count(self.endScope):
 432             raise StopIteration
 433         return self

 461     def printError(nav, *misc):
 462         """Print error associated with a given file; avoid printing duplicates"""
 463         if nav.fname:
 464             silenceValue = ' '.join(map(str, misc))
 465             if nav.fname not in silenceErrors:
 466                 print(nav.fname, file=sys.stderr)
 467                 silenceErrors[nav.fname] = set()
 468             elif silenceValue in silenceErrors[nav.fname]:
 469                 return # do not print a duplicate error for this file
 470             silenceErrors[nav.fname].add(silenceValue)
 471         print('wmliterator:', end=" ", file=sys.stderr)
 472         for item in misc:
 473             print(item, end=" ", file=sys.stderr)
 474         print("", file=sys.stderr) #terminate line
