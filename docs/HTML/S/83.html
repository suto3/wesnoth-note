  86 config_attribute_value &config_attribute_value::operator=(long long v)
  87 {
  88         if ( v > 0 )
  89                 // We can store this unsigned.
  90                 return *this = static_cast<unsigned long long>(v);
  91 
  92         if ( v >= INT_MIN )
  93                 // We can store this as an int.
  94                 return *this = static_cast<int>(v);
  95 
  96         // Getting to this point should be rare. (Currently, getting here means
  97         // something like there was so much draining in a campaign that the
  98         // total damage taken is not only negative, but so negative that an
  99         // int cannot hold the value.) So rare that it is not worth precise
 100         // treatment; just use a double.
 101         value_ = static_cast<double>(v);
 102         return *this;
 103 }

 115 config_attribute_value &config_attribute_value::operator=(double v)
 116 {
 117         // Try to store integers in other types.
 118         if ( v > 0.0 ) {
 119                 // Convert to unsigned and pass this off to that assignment operator.
 120                 unsigned long long ull = static_cast<unsigned long long>(v);
 121                 if ( static_cast<double>(ull) == v )
 122                         return *this = ull;
 123         }
 124         else {
 125                 // Convert to integer and pass this off to that assignment operator.
 126                 int i = static_cast<int>(v);
 127                 if ( static_cast<double>(i) == v )
 128                         return *this = i;
 129         }
 130 
 131         // If we get here, this does in fact get stored as a double.
 132         value_ = v;
 133         return *this;
 134 }

 136 namespace {
 137         /**
 138          * Attempts to convert @a source to the template type.
 139          * This is to avoid "overzealous reinterpretations of certain WML strings as
 140          * numeric types" (c.f. bug #19201).
 141          * @returns true if the conversion was successful and the source string
 142          *          can be reobtained by streaming the result.
 143          */
 144         template<typename To>
 145         bool from_string_verify(const std::string & source, To & res)
 146         {
 147                 // Check 1: convertable to the target type.
 148                 std::istringstream in_str(source);
 149                 if ( !(in_str >> res) )
 150                         return false;
 151 
 152                 // Check 2: convertable back to the same string.
 153                 std::ostringstream out_str;
 154                 out_str << res;
 155                 return out_str.str() == source;
 156         }
 157 }
 158 config_attribute_value &config_attribute_value::operator=(const std::string &v)
 159 {
 160         // Handle some special strings.
 161         if (v.empty()) { value_ = v; return *this; }
 162         if ( v == s_yes )   { value_ = yes_no(true);  return *this; }
 163         if ( v == s_no )    { value_ = yes_no(false); return *this; }
 164         if ( v == s_true )  { value_ = true_false(true);  return *this; }
 165         if ( v == s_false ) { value_ = true_false(false); return *this; }
 166 
 167         // Attempt to convert to a number.
 168         char *eptr;
 169         double d = strtod(v.c_str(), &eptr);
 170         if ( *eptr == '\0' ) {
 171                 // Possibly a number. See what type it should be stored in.
 172                 // (All conversions will be from the string since the largest integer
 173                 // type could have more precision than a double.)
 174                 if ( d > 0.0 ) {
 175                         // The largest type for positive integers is unsigned long long.
 176                         unsigned long long ull = 0;
 177                         if ( from_string_verify<unsigned long long>(v, ull) )
 178                                 return *this = ull;
 179                 }
 180                 else {
 181                         // The largest (variant) type for negative integers is int.
 182                         int i = 0;
 183                         if ( from_string_verify<int>(v, i) )
 184                                 return *this = i;
 185                 }
 186 
 187                 // This does not look like an integer, so it should be a double.
 188                 // However, make sure it can convert back to the same string (in
 189                 // case this is a string that just looks like a numeric value).
 190                 std::ostringstream tester;
 191                 tester << d;
 192                 if ( tester.str() == v ) {
 193                         value_ = d;
 194                         return *this;
 195                 }
 196         }
 197 
 198         // No conversion possible. Store the string.
 199         value_ = v;
 200         return *this;
 201 }

 211 bool config_attribute_value::to_bool(bool def) const
 212 {
 213         if ( const yes_no *p = boost::get<const yes_no>(&value_) )
 214                 return *p;
 215         if ( const true_false *p = boost::get<const true_false>(&value_) )
 216                 return *p;
 217 
 218         // No other types are ever recognized as boolean.
 219         return def;

 223         /// Visitor for converting a variant to a numeric type (T).
 224         template <typename T>
 225         class attribute_numeric_visitor : public boost::static_visitor<T>
 226         {
 227         public:
 228                 // Constructor stores the default value.
 229                 attribute_numeric_visitor(T def) : def_(def) {}
 230 
 231                 T operator()(boost::blank const &) const { return def_; }
 232                 T operator()(bool)                 const { return def_; }
 233                 T operator()(int i)                const { return static_cast<T>(i); }
 234                 T operator()(unsigned long long u) const { return static_cast<T>(u); }
 235                 T operator()(double d)             const { return static_cast<T>(d); }
 236                 T operator()(const std::string& s) const { return lexical_cast_default<T>(s, def_); }
 237                 T operator()(t_string const &)     const { return def_; }
 238 
 239         private:
 240                 const T def_;
 241         };

 321 /// Visitor handling equality checks.
 322 class config_attribute_value::equality_visitor
 323         : public boost::static_visitor<bool>
 324 {
 325 public:
 326         // Most generic: not equal.
 327         template <typename T, typename U>
 328         bool operator()(const T &, const U &) const { return false; }
 329 
 330         // Same types are comparable and might be equal.
 331         template <typename T>
 332         bool operator()(const T & lhs, const T & rhs) const { return lhs == rhs; }
 333 
 334         // Boolean values can be compared.
 335         bool operator()(const true_false & lhs, const yes_no & rhs) const { return bool(lhs) == bool(rhs); }
 336         bool operator()(const yes_no & lhs, const true_false & rhs) const { return bool(lhs) == bool(rhs); }
 337 };
