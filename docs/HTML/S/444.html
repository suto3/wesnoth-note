  89 void campaign_selection::campaign_selected(window& window)
  90 {
  91         tree_view& tree = find_widget<tree_view>(&window, "campaign_tree", false);
  92 
  93         if(tree.empty()) {
  94                 return;
  95         }
  96 
  97         assert(tree.selected_item());
  98         if(tree.selected_item()->id() != "") {
  99                 auto iter = std::find(page_ids_.begin(), page_ids_.end(), tree.selected_item()->id());
 100                 const int choice = iter - page_ids_.begin();
 101                 if(iter == page_ids_.end()) {
 102                         return;
 103                 }
 104 
 105                 multi_page& pages = find_widget<multi_page>(&window, "campaign_details", false);
 106                 pages.select_page(choice);
 107                 engine_.set_current_level(choice);
 108         }
 109 
 110 }

 112 void campaign_selection::sort_campaigns(window& window, campaign_selection::CAMPAIGN_ORDER order, bool ascending)
 113 {
 114         using level_ptr = ng::create_engine::level_ptr;
 115         auto levels = engine_.get_levels_by_type_unfiltered(ng::level::TYPE::SP_CAMPAIGN);
 116         switch(order) {
 117                 case RANK: // Already sorted by rank
 118                         if(!ascending) {
 119                                 // This'll actually never happen, but who knows if that'll ever change...
 120                                 std::reverse(levels.begin(), levels.end());
 121                         }
 122                         break;
 123                 case DATE:
 124                         std::sort(levels.begin(), levels.end(), [ascending](const level_ptr& a, const level_ptr& b) {
 125                                 auto cpn_a = std::dynamic_pointer_cast<ng::campaign>(a), cpn_b = std::dynamic_pointer_cast<ng::campaign>(b);
 126                                 if(cpn_b == nullptr) return cpn_a != nullptr;
 127                                 if(cpn_a == nullptr) return false;
 128                                 return ascending ? cpn_a->dates().first < cpn_b->dates().first : cpn_a->dates().first > cpn_b->dates().first;
 129                         });
 130                         break;
 131                 case NAME:
 132                         std::sort(levels.begin(), levels.end(), [ascending](const level_ptr& a, const level_ptr& b) {
 133                                 int cmp = translation::icompare(a->name(), b->name());
 134                                 return ascending ? cmp < 0 : cmp > 0;
 135                         });
 136                         break;
 137         }
 138 
 139         tree_view& tree = find_widget<tree_view>(&window, "campaign_tree", false);
 140         // Remember which campaign was selected...
 141         std::string was_selected = tree.selected_item()->id();
 142         tree.clear();
 143         for(const auto& level : levels) {
 144                 add_campaign_to_tree(window, level->data());
 145         }
 146 
 147         if(!was_selected.empty()) {
 148                 tree_view_node& node = find_widget<tree_view_node>(&window, was_selected, false);
 149                 node.select_node();
 150         }
 151 }

 153 void campaign_selection::toggle_sorting_selection(window& window, CAMPAIGN_ORDER order) {
 154         static bool force = false;
 155         if(force) {
 156                 return;
 157         }
 158         if(current_sorting_ == order) {
 159                 if(currently_sorted_asc_) {
 160                         currently_sorted_asc_ = false;
 161                 } else {
 162                         currently_sorted_asc_ = true;
 163                         current_sorting_ = RANK;
 164                 }
 165         } else if(current_sorting_ == RANK) {
 166                 currently_sorted_asc_ = true;
 167                 current_sorting_ = order;
 168         } else {
 169                 currently_sorted_asc_ = true;
 170                 current_sorting_ = order;
 171                 force = true;
 172                 if(order == NAME) {
 173                         toggle_button& sort_time = find_widget<toggle_button>(&window, "sort_time", false);
 174                         sort_time.set_value(0);
 175                 } else if(order == DATE) {
 176                         toggle_button& sort_name = find_widget<toggle_button>(&window, "sort_name", false);
 177                         sort_name.set_value(0);
 178                 }
 179                 force = false;
 180         }
 181         sort_campaigns(window, current_sorting_, currently_sorted_asc_);
 182 }

 184 void campaign_selection::pre_show(window& window)
 185 {
 186         /***** Setup campaign tree. *****/
 187         tree_view& tree = find_widget<tree_view>(&window, "campaign_tree", false);
 188 
 189         tree.set_selection_change_callback(
 190                 std::bind(&campaign_selection::campaign_selected, this, std::ref(window)));
 191 
 192         toggle_button& sort_name = find_widget<toggle_button>(&window, "sort_name", false);
 193         toggle_button& sort_time = find_widget<toggle_button>(&window, "sort_time", false);
 194 
 195         connect_signal_notify_modified(sort_name,
 196                 std::bind(&campaign_selection::toggle_sorting_selection, this, std::ref(window), NAME));
 197 
 198         connect_signal_notify_modified(sort_time,
 199                 std::bind(&campaign_selection::toggle_sorting_selection, this, std::ref(window), DATE));
 200 
 201         window.keyboard_capture(&tree);
 202 
 203         /***** Setup campaign details. *****/
 204         multi_page& pages = find_widget<multi_page>(&window, "campaign_details", false);
 205 
 206         for(const auto & level : engine_.get_levels_by_type_unfiltered(ng::level::TYPE::SP_CAMPAIGN)) {
 207                 const config& campaign = level->data();
 208 
 209                 /*** Add tree item ***/
 210                 add_campaign_to_tree(window, campaign);
 211 
 212                 /*** Add detail item ***/
 213                 std::map<std::string, string_map> data;
 214                 string_map item;
 215 
 216                 item["label"] = campaign["description"];
 217                 item["use_markup"] = "true";
 218 
 219                 if(!campaign["description_alignment"].empty()) {
 220                         item["text_alignment"] = campaign["description_alignment"];
 221                 }
 222 
 223                 data.emplace("description", item);
 224 
 225                 item["label"] = campaign["image"];
 226                 data.emplace("image", item);
 227 
 228                 pages.add_page(data);
 229                 page_ids_.push_back(campaign["id"]);
 230         }
 231 
 232         //
 233         // Set up Mods selection dropdown
 234         //
 235         multimenu_button& mods_menu = find_widget<multimenu_button>(&window, "mods_menu", false);
 236 
 237         if(!engine_.get_const_extras_by_type(ng::create_engine::MOD).empty()) {
 238 
 239                 std::vector<config> mod_menu_values;
 240                 std::vector<std::string> enabled = engine_.active_mods();
 241 
 242                 for(const auto& mod : engine_.get_const_extras_by_type(ng::create_engine::MOD)) {
 243                         const bool active = std::find(enabled.begin(), enabled.end(), mod->id) != enabled.end();
 244 
 245                         mod_menu_values.emplace_back(config {"label", mod->name, "checkbox", active});
 246 
 247                         mod_states_.push_back(active);
 248                 }
 249 
 250                 mods_menu.set_values(mod_menu_values);
 251                 mods_menu.select_options(mod_states_);
 252                 mods_menu.set_callback_toggle_state_change(std::bind(&campaign_selection::mod_toggled, this, std::ref(window)));
 253         } else {
 254                 mods_menu.set_active(false);
 255                 mods_menu.set_label(_("None"));
 256         }
 257 
 258         campaign_selected(window);
 259 }

 261 void campaign_selection::add_campaign_to_tree(window& window, const config& campaign)
 262 {
 263         tree_view& tree = find_widget<tree_view>(&window, "campaign_tree", false);
 264         std::map<std::string, string_map> data;
 265         string_map item;
 266 
 267         item["label"] = campaign["icon"];
 268         data.emplace("icon", item);
 269 
 270         item["label"] = campaign["name"];
 271         data.emplace("name", item);
 272 
 273         if(campaign["completed_on_hardest"].to_bool()) {
 274                 item["label"] = game_config::images::victory_laurel_hardest;
 275         } else if(campaign["completed"].to_bool()) {
 276                 if(campaign["completed_difficulties"] == campaign.child_or_empty("difficulty")["define"]) {
 277                         item["label"] = game_config::images::victory_laurel_easy;
 278                 } else {
 279                         item["label"] = game_config::images::victory_laurel;
 280                 }
 281         } else {
 282                 item.clear();
 283         }
 284 
 285         data.emplace("victory", item);
 286 
 287         tree.add_node("campaign", data).set_id(campaign["id"]);
 288 }

 290 void campaign_selection::post_show(window& window)
 291 {
 292         tree_view& tree = find_widget<tree_view>(&window, "campaign_tree", false);
 293 
 294         if(tree.empty()) {
 295                 return;
 296         }
 297 
 298         assert(tree.selected_item());
 299         if(tree.selected_item()->id() != "") {
 300                 auto iter = std::find(page_ids_.begin(), page_ids_.end(), tree.selected_item()->id());
 301                 if(iter != page_ids_.end()) {
 302                         choice_ = iter - page_ids_.begin();
 303                 }
 304         }
 305 
 306         deterministic_ = find_widget<toggle_button>(&window, "checkbox_deterministic", false).get_value_bool();
 307 
 308         preferences::set_modifications(engine_.active_mods(), false);
 309 }

 311 void campaign_selection::mod_toggled(window& window)
 312 {
 313         boost::dynamic_bitset<> new_mod_states = find_widget<multimenu_button>(&window, "mods_menu", false).get_toggle_states();
 314 
 315         // Get a mask of any mods that were toggled, regardless of new state
 316         mod_states_ = mod_states_ ^ new_mod_states;
 317 
 318         for(unsigned i = 0; i < mod_states_.size(); i++) {
 319                 if(mod_states_[i]) {
 320                         engine_.set_current_mod_index(i);
 321                         engine_.toggle_current_mod();
 322                 }
 323         }
 324 
 325         // Save the full toggle states for next time
 326         mod_states_ = new_mod_states;
 327 }
