  42         String file = mapDialog.open( );

  44         if( file == null ) {
  45             return;
  46         }
  47 
  48         try {
  49             File source = new File( file );
  50             File target = new File( WorkspaceUtils.getSelectedFolder( )
  51                 .getLocation( ).toOSString( )
  52                 + IPath.SEPARATOR + source.getName( ) );
  53 
  54             if( target.exists( ) ) {
  55                 if( GUIUtils.showMessageBox( Messages.MapUtils_4,
  56                     SWT.ICON_QUESTION | SWT.YES | SWT.NO ) == SWT.NO ) {
  57                     return;
  58                 }
  59             }
  60 
  61             ResourceUtils.copyTo( source, target );
  62             WorkspaceUtils.getSelectedFolder( ).refreshLocal(
  63                 IResource.DEPTH_INFINITE, null );
  64         } catch( Exception e ) {
  65             Logger.getInstance( ).logException( e );
  66         }
  67     }
