   1 /*
   2  * Popup menu implementation for the Wesnoth units database
   3  */
   4 var wmlunits_menu_setup = (function() {
   5     var menus = [];
   6     var visibleMenu = false;
   7 
   8     function toggleMenu(event, menu)
   9     {
  10         hideMenus(menu);
  11 
  12         if (!menu.style.display || menu.style.display === "none") {
  13             menu.style.display = "block";
  14             visibleMenu = true;
  15         } else {
  16             menu.style.display = "none";
  17             visibleMenu = false;
  18         }
  19     }
  20 
  21     function hideMenus(skipMenu)
  22     {
  23         if (!visibleMenu) {
  24             return;
  25         }
  26 
  27         for (var i = 0; i < menus.length; ++i) {
  28             if (menus[i] !== skipMenu) {
  29                 menus[i].style.display = "none";
  30             }
  31         }
  32     }
  33 
  34     function isMenuBox(e)
  35     {
  36         return e.className.search(/\bpopupmenu\b/) != -1;
  37     }
  38 
  39     function isNavBar(e)
  40     {
  41         return e.className.search(/\bnavbar\b/) != -1;
  42     }
  43 
  44     var navItems = document.getElementsByClassName("popupcontainer");
  45 
  46     // Set event handlers for individual menu triggers.
  47     for (var i = 0; i < navItems.length; ++i) {
  48         var navItem = navItems[i],
  49             menu = navItem.getElementsByClassName("popupmenu")[0];
  50 
  51         menus.push(menu);
  52 
  53         var id = menu.id,
  54             a = navItem.getElementsByClassName("popuptrigger")[0];
  55 
  56         a.addEventListener("click", function(event) {
  57             event.preventDefault();
  58             event.stopPropagation();
  59             toggleMenu(event, event.target.nextElementSibling);
  60         }, false);
  61     }
  62 
  63     // Dismiss all visible menus on click outside them.
  64     document.addEventListener("click", function(event) {
  65         if (!visibleMenu) {
  66             return;
  67         }
  68 
  69         var parent = event.target;
  70         while(parent && !isMenuBox(parent) && !isNavBar(parent)) {
  71             parent = parent.parentElement;
  72         }
  73 
  74         if (!parent || !isMenuBox(parent)) {
  75             hideMenus();
  76         }
  77     }, false);
  78 });
