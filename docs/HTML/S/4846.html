 624 # Finds the nearest hex to (X,Y) within MAX_DISTANCE that matches the FILTER, and stores
 625 # the location in VARIABLE with x, y, and terrain. If there is more than one location
 626 # found, VARIABLE will be an array.
 627 #define NEAREST_HEX X Y MAX_DISTANCE FILTER VAR_NAME
 628     [set_variable]
 629         name=distance
 630         value=0
 631     [/set_variable]
 632     [set_variable]
 633         name=location_found
 634         value=no
 635     [/set_variable]
 636     [clear_variable]
 637         name={VAR_NAME}
 638     [/clear_variable]
 639 
 640     # Store any matching locations within a radius of "distance". If nothing got stored,
 641     # add one to "distance" and try again. Do this until a location is found, or
 642     # "distance" goes over the MAX_DISTANCE:
 643     [while]
 644         [variable]
 645             name={VAR_NAME}.length
 646             less_than=1
 647         [/variable]
 648         [and]
 649             [variable]
 650                 name=distance
 651                 less_than_equal_to={MAX_DISTANCE}
 652             [/variable]
 653         [/and]
 654 
 655         [do]
 656             [store_locations]
 657                 {FILTER}
 658                 [and]
 659                     x={X}
 660                     y={Y}
 661                     radius=$distance
 662                 [/and]
 663                 variable={VAR_NAME}
 664             [/store_locations]
 665 
 666             [if]
 667                 [variable]
 668                     name={VAR_NAME}.length
 669                     greater_than=0
 670                 [/variable]
 671                 [then]
 672                     [set_variable]
 673                         name=location_found
 674                         value=yes
 675                     [/set_variable]
 676                 [/then]
 677             [/if]
 678 
 679             [set_variable]
 680                 name=distance
 681                 add=1
 682             [/set_variable]
 683         [/do]
 684     [/while]
 685 
 686     [clear_variable]
 687         name=location_found, distance
 688     [/clear_variable]
 689 #enddef

 776 # This event turns a necromancer into a lich. It can be activated by an event or from
 777 # a right-click menu option. It needs $unit to be set or it won't do anything.
 778 #define TURN_INTO_A_LICH
 779     [event]
 780         name=lichify
 781         first_time_only=no
 782 
 783         # This is just to get a more descriptive name for the "unit" variable.
 784         [set_variables]
 785             name=stored_necromancer
 786             mode=replace
 787             to_variable=unit
 788         [/set_variables]
 789 
 790         # The necromancer may have AMLAd and had its experience reset to 0. Now, as a
 791         # lich, there is a fourth level available for the unit, and all that experience
 792         # should be applied towards the level-up instead of simply being lost. We will
 793         # "unwind" each AMLA and calculate how much it cost, so we can give back those
 794         # experience points.
 795         [set_variable]
 796             name=num_amlas
 797             # This gives us the number of AMLAs that the unit has had:
 798             value=$stored_necromancer.modifications.advancement.length
 799         [/set_variable]
 800         [set_variable]
 801             # The experience that will be used for the new lich will be at least
 802             # the necromancer's current actual experience.
 803             name=experience_gained_after_max_level
 804             value=$stored_necromancer.experience
 805         [/set_variable]
 806         [set_variable]
 807             # Each AMLA took 20% more experience, so we will need to reduce this
 808             # value as we unwind them. The variable starts out storing the amount
 809             # needed for the *next* AMLA, so we will need to reduce it one step as
 810             # the first operation.
 811             name=experience_for_one_amla
 812             value=$stored_necromancer.max_experience
 813         [/set_variable]
 814 
 815         [while]
 816             [variable]
 817                 name=num_amlas
 818                 greater_than=0
 819             [/variable]
 820             [do]
 821                 [set_variable]
 822                     # Each AMLA increased the experience needed by 20%, which
 823                     # is the same as multiplying by 1.2. We reverse that here.
 824                     name=experience_for_one_amla
 825                     divide=1.2
 826                 [/set_variable]
 827                 [set_variable]
 828                     # Round it off in case it's a float:
 829                     name=experience_for_one_amla
 830                     round=0
 831                 [/set_variable]
 832                 [set_variable]
 833                     # Add the experience that was needed for the previous AMLA
 834                     # to the amount that will be given to the lich:
 835                     name=experience_gained_after_max_level
 836                     add=$experience_for_one_amla
 837                 [/set_variable]
 838                 [set_variable]
 839                     name=num_amlas
 840                     sub=1
 841                 [/set_variable]
 842             [/do]
 843         [/while]
 844 
 845         # It's nice if Ardonna looks different as an ancient lich, and since there is
 846         # another sprite available, we will use it. She needs a different type though:
 847         [if]
 848             [variable]
 849                 name=stored_necromancer.id
 850                 equals=Ardonna
 851             [/variable]
 852             [then]
 853                 [set_variable]
 854                     name=lich_type
 855                     value=Lich SotA
 856                 [/set_variable]
 857             [/then]
 858             [else]
 859                 [set_variable]
 860                     name=lich_type
 861                     value=Lich
 862                 [/set_variable]
 863             [/else]
 864         [/if]
 865 
 866         # We are going to create a new unit instead of transforming the old one.
 867         # Using the old unit would leave its AMLAs in effect, changing the hitpoints
 868         # and experience needed to level up.
 869         [unit]
 870             side=1
 871             type=$lich_type
 872             x=$stored_necromancer.x
 873             y=$stored_necromancer.y
 874             id=$stored_necromancer.id
 875             name=$stored_necromancer.name
 876             facing=$stored_necromancer.facing
 877             moves=$stored_necromancer.moves
 878             overlays=$stored_necromancer.overlays
 879             hitpoints=$stored_necromancer.hitpoints
 880             canrecruit=$stored_necromancer.canrecruit
 881             attacks_left=$stored_necromancer.attacks_left
 882             experience=$experience_gained_after_max_level
 883 
 884             # Copy the original traits into the new unit:
 885             [modifications]
 886                 [insert_tag]
 887                     name=trait
 888                     variable=stored_necromancer.modifications.trait[0]
 889                 [/insert_tag]
 890                 [insert_tag]
 891                     name=trait
 892                     variable=stored_necromancer.modifications.trait[1]
 893                 [/insert_tag]
 894             [/modifications]
 895 
 896             to_variable=new_lich  # Don't create the unit yet. Just store it.
 897         [/unit]
 898 
 899         # Hitpoints are transferred to the new unit so that turning into a lich doesn't
 900         # heal it. However, this could result in the lich having more HP than its
 901         # maximum. If that is the case, we will reset it to the maximum.
 902         [if]
 903             [variable]
 904                 name=new_lich.hitpoints
 905                 greater_than=$new_lich.max_hitpoints
 906             [/variable]
 907             [then]
 908                 [set_variable]
 909                     name=new_lich.hitpoints
 910                     value=$new_lich.max_hitpoints
 911                 [/set_variable]
 912             [/then]
 913         [/if]
 914 
 915         # The leaders are changed in a scripted event at the end of a scenario, so
 916         # they need to be handled differently than normal units:
 917         [if]
 918             [variable]
 919                 name=stored_necromancer.id
 920                 not_equals=Ardonna
 921             [/variable]
 922             [variable]
 923                 name=stored_necromancer.id
 924                 not_equals=Ras-Tabahn
 925             [/variable]
 926             [then]
 927                 [unstore_unit]
 928                     # The unit will be automatically advanced to Ancient Lich at
 929                     # this point if its experience is sufficient. Theoretically,
 930                     # it could even AMLA.
 931                     variable=new_lich
 932                     text= _ "Lich"  # This text goes by quickly, so it must be short.
 933                     red,green,blue=220,0,220  # dark-magic purple
 934                     animate=yes
 935                 [/unstore_unit]
 936             [/then]
 937             [else]
 938                 # It's unlikely, but one of the leaders could be at a lower
 939                 # level, and therefore have fewer hitpoints than the same unit
 940                 # as a lich. We will set the hitpoints to the maximum in case
 941                 # that has happened.
 942                 [set_variable]
 943                     name=new_lich.hitpoints
 944                     value=$new_lich.max_hitpoints
 945                 [/set_variable]
 946 
 947                 # The leaders are changed to liches in between scenarios, so if
 948                 # we are working with one of them, the new unit needs to get
 949                 # sent to the recall list.
 950                 [set_variable]
 951                     name=new_lich.x
 952                     value=recall
 953                 [/set_variable]
 954                 [set_variable]
 955                     name=new_lich.y
 956                     value=recall
 957                 [/set_variable]
 958                 [unstore_unit]
 959                     variable=new_lich
 960                 [/unstore_unit]
 961             [/else]
 962         [/if]
 963 
 964         [clear_variable]
 965             name=stored_necromancer, new_lich, lich_type, num_amlas, experience_for_one_amla, experience_gained_after_max_level
 966         [/clear_variable]
 967     [/event]
 968 #enddef
