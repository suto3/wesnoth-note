 109 void editor_resize_map::pre_show(window& window)
 110 {
 111         slider& height = find_widget<slider>(&window, "height", false);
 112         connect_signal_notify_modified(
 113                         height,
 114                         std::bind(&editor_resize_map::update_expand_direction,
 115                                                 this,
 116                                                 std::ref(window)));
 117 
 118         slider& width = find_widget<slider>(&window, "width", false);
 119         connect_signal_notify_modified(
 120                         width,
 121                         std::bind(&editor_resize_map::update_expand_direction,
 122                                                 this,
 123                                                 std::ref(window)));
 124 
 125         std::string name_prefix = "expand";
 126         for(int i = 0; i < 9; ++i) {
 127                 std::string name = name_prefix + std::to_string(i);
 128                 direction_buttons_[i]
 129                                 = find_widget<toggle_button>(&window, name, false, true);
 130 
 131                 direction_buttons_[i]->set_callback_state_change(
 132                                 dialog_callback<editor_resize_map,
 133                                                                 &editor_resize_map::update_expand_direction>);
 134         }
 135         direction_buttons_[0]->set_value(true);
 136         update_expand_direction(window);
 137 }

 139 /**
 140  * Convert a coordinate on a 3  by 3 grid to an index, return 9 for out of
 141  * bounds
 142  */
 143 static int resize_grid_xy_to_idx(const int x, const int y)
 144 {
 145         if(x < 0 || x > 2 || y < 0 || y > 2) {
 146                 return 9;
 147         } else {
 148                 return y * 3 + x;
 149         }
 150 }

 160 void editor_resize_map::update_expand_direction(window& window)
 161 {
 162         for(int i = 0; i < 9; ++i) {
 163                 if(direction_buttons_[i]->get_value()
 164                    && static_cast<int>(expand_direction_) != i) {
 165 
 166                         expand_direction_ = static_cast<EXPAND_DIRECTION>(i);
 167                         break;
 168                 }
 169         }
 170         for(int i = 0; i < static_cast<int>(expand_direction_); ++i) {
 171                 direction_buttons_[i]->set_value(false);
 172                 set_direction_icon(i, "none");
 173         }
 174         direction_buttons_[expand_direction_]->set_value(true);
 175         for(int i = expand_direction_ + 1; i < 9; ++i) {
 176                 direction_buttons_[i]->set_value(false);
 177                 set_direction_icon(i, "none");
 178         }
 179 
 180         int xdiff = width_->get_widget_value(window) - old_width_;
 181         int ydiff = height_->get_widget_value(window) - old_height_;
 182         int x = static_cast<int>(expand_direction_) % 3;
 183         int y = static_cast<int>(expand_direction_) / 3;
 184         set_direction_icon(expand_direction_, "center");
 185         if(xdiff != 0) {
 186                 int left = resize_grid_xy_to_idx(x - 1, y);
 187                 int right = resize_grid_xy_to_idx(x + 1, y);
 188                 if(xdiff < 0)
 189                         std::swap(left, right);
 190                 set_direction_icon(left, "left");
 191                 set_direction_icon(right, "right");
 192         }
 193         if(ydiff != 0) {
 194                 int top = resize_grid_xy_to_idx(x, y - 1);
 195                 int bottom = resize_grid_xy_to_idx(x, y + 1);
 196                 if(ydiff < 0)
 197                         std::swap(top, bottom);
 198                 set_direction_icon(top, "up");
 199                 set_direction_icon(bottom, "down");
 200         }
 201         if(xdiff < 0 || ydiff < 0 || (xdiff > 0 && ydiff > 0)) {
 202                 int nw = resize_grid_xy_to_idx(x - 1, y - 1);
 203                 int ne = resize_grid_xy_to_idx(x + 1, y - 1);
 204                 int sw = resize_grid_xy_to_idx(x - 1, y + 1);
 205                 int se = resize_grid_xy_to_idx(x + 1, y + 1);
 206                 if(xdiff < 0 || ydiff < 0) {
 207                         std::swap(nw, se);
 208                         std::swap(ne, sw);
 209                 }
 210                 set_direction_icon(nw, "topleft");
 211                 set_direction_icon(ne, "topright");
 212                 set_direction_icon(sw, "bottomleft");
 213                 set_direction_icon(se, "bottomright");
 214         }
 215 }
