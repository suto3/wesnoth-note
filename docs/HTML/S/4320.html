  21 function ca_coward:execution(cfg)
  22     local coward = get_coward(cfg)
  23     local reach = wesnoth.find_reach(coward)
  24 
  25     local filter_second =
  26         H.get_child(cfg, "filter_second")
  27         or { { "filter_side", { { "enemy_of", { side = wesnoth.current.side } } } } }
  28     local enemies = AH.get_live_units {
  29         { "and", filter_second },
  30         { "filter_location", { x = coward.x, y = coward.y, radius = cfg.distance } },
  31         { "filter_vision", { side = wesnoth.current.side, visible = 'yes' } }
  32     }
  33 
  34     -- If no enemies are close: keep unit from doing anything and exit

  36         AH.checked_stopunit_all(ai, coward)
  37         return
  38     end
  39 
  40     for i,hex in ipairs(reach) do
  41         -- Only consider unoccupied hexes
  42         local unit_in_way = wesnoth.get_unit(hex[1], hex[2])
  43         if (not AH.is_visible_unit(wesnoth.current.side, unit_in_way))
  44             or (unit_in_way == coward)
  45         then

  47             for _,enemy in ipairs(enemies) do
  48                 local dist = wesnoth.map.distance_between(hex[1], hex[2], enemy.x, enemy.y)
  49                 rating = rating - 1 / dist^2
  50             end
  51 
  52             -- Store this weighting in the third field of each 'reach' element
  53             reach[i][3] = rating
  54         else
  55             reach[i][3] = -9e99
  56         end
  57     end
  58 
  59     -- Select those within factor 2 of the maximum (note: ratings are negative)
  60     table.sort(reach, function(a, b) return a[3] > b[3] end )
  61     local best_pos = AH.filter(reach, function(tmp) return tmp[3] > reach[1][3] * 2 end)
  62 
  63     -- Now take 'seek' and 'avoid' into account
  64     for i,pos in ipairs(best_pos) do
  65         -- Weighting based on distance from 'seek' and 'avoid'
  66         local dist_seek = AH.generalized_distance(pos[1], pos[2], cfg.seek_x, cfg.seek_y)
  67         local dist_avoid = AH.generalized_distance(pos[1], pos[2], cfg.avoid_x, cfg.avoid_y)
  68         local rating = 1 / (dist_seek + 1) - 1 / (dist_avoid + 1)^2 * 0.75
  69 
  70         best_pos[i][4] = rating
  71     end
  72 
  73     -- Select all those that have the maximum score
  74     table.sort(best_pos, function(a, b) return a[4] > b[4] end)
  75     local best_overall = AH.filter(best_pos, function(tmp) return tmp[4] == best_pos[1][4] end)
  76 
  77     -- As final step, if there are more than one remaining locations,
  78     -- we take the one with the minimum score in the distance-from-enemy criterion
  79     local max_rating, best_hex = -9e99
  80     for _,pos in ipairs(best_overall) do
  81         if (pos[3] > max_rating) then
  82             max_rating, best_hex = pos[3], pos
  83         end
  84     end
  85 
  86     AH.movefull_stopunit(ai, coward, best_hex[1], best_hex[2])
  87     if (not coward) or (not coward.valid) then return end
  88 
  89     -- If 'attack_if_trapped' is set, the coward attacks the weakest unit it ends up next to
  90     if cfg.attack_if_trapped then

  92         for xa,ya in H.adjacent_tiles(coward.x, coward.y) do
  93             local target = wesnoth.get_unit(xa, ya)
  94             if target and wesnoth.is_enemy(coward.side, target.side) then
  95                 local rating = - target.hitpoints
  96 
  97                 if (rating > max_rating) then
  98                     max_rating, best_target = rating, target
  99                 end
 100             end
 101         end
 102 
 103         if best_target then
 104             AH.checked_attack(ai, coward, best_target)
 105             if (not coward) or (not coward.valid) then return end
 106         end
 107     end

 109     AH.checked_stopunit_all(ai, coward)
 110 end
