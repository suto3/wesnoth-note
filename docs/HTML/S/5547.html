   2 local helper = wesnoth.require("helper")
   3 local _ = wesnoth.textdomain 'wesnoth-help'
   4 local T = helper.set_wml_tag_metatable {}
   5 local on_event = wesnoth.require("on_event")
   6 
   7 -- The feeding event code
   8 on_event("die", function()
   9         local ec = wesnoth.current.event_context
  10         local u_killer = wesnoth.get_unit(ec.x2, ec.y2)
  11         local u_victim = wesnoth.get_unit(ec.x1, ec.y1)
  12 
  13         if not u_killer or not u_killer:matches { ability = "feeding" } then
  14                 return
  15         end
  16         if not u_victim or u_victim:matches { status = "unplagueable" } then
  17                 return
  18         end
  19         local u_killer_cfg = u_killer.__cfg
  20         for i,v in ipairs(helper.get_child(u_killer_cfg, "modifications"))do
  21                 if v[1] == "object" and v[2].feeding == true then
  22                         local effect = helper.get_child(v[2], "effect")
  23                         effect.increase_total = effect.increase_total + 1
  24                         u_killer_cfg.max_hitpoints = u_killer_cfg.max_hitpoints + 1
  25                         u_killer_cfg.hitpoints = u_killer_cfg.hitpoints + 1
  26                         wesnoth.create_unit(u_killer_cfg):to_map()
  27                         wesnoth.float_label(ec.x2, ec.y2, _ "+1 max HP", "0,255,0")
  28                         return
  29                 end
  30         end
  31         -- reaching this point means that the unit didn't have the feedng object yet.
  32         u_killer:add_modification("object", {
  33                 feeding = true,
  34                 T.effect {
  35                         apply_to = "hitpoints",
  36                         increase_total = 1,
  37                 },
  38         })
  39         u_killer.hitpoints = u_killer.hitpoints + 1
  40         wesnoth.float_label(ec.x2, ec.y2, _ "+1 max HP", "0,255,0")
  41 end)
