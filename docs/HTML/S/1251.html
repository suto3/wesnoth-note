  38 void resolve_negative_value(int size, int& val)
  39 {
  40         if(val < 0) {
  41                 val = size + val;

  44         // val is still < 0? We don't accept!

  52 /**
  53  * Parses a ']' terminated string.
  54  * This is a important optimization of lexical_cast_default
  55  */
  56 int parse_index(const char* index_str)
  57 {
  58         char* endptr;
  59         int res = strtol(index_str, &endptr, 10);
  60 
  61         if(*endptr != ']' || res > static_cast<int>(game_config::max_loop) || endptr == index_str) {
  62                 throw invalid_variablename_exception();
  63         }
  64 
  65         return res;
  66 }

  72 /**
  73  * Visitor base class.
  74  *
  75  * This provides the interface for the main functions each visitor can implement. The default implementation of
  76  * each function simply throws @ref invalid_variablename_exception.
  77  *
  78  * This class also provides two type aliases corresponding to the function return value and argument types.
  79  * Note that visitors shouldn't inherit from this directly and instead use the @ref info_visitor and
  80  * @ref info_visitor_const wrappers, since both fully specify the parameter type.
  81  *
  82  * @tparam R                   Return value type.
  83  * @tparam P                   Argument type.
  84  */
  85 template<typename R, typename P>
  86 class info_visitor_base
  87 {
  88 public:
  89         using result_t = R;
  90         using param_t = P&;
  91 
  92 #define DEFAULTHANDLER(name)                                                                                           \
  93         result_t name(param_t) const                                                                                       \
  94         {                                                                                                                  \
  95                 throw invalid_variablename_exception();                                                                        \
  96         }
  97 
  98         /** For use if the variable name was previously empty. This can only happen during calculate_value. */
  99         DEFAULTHANDLER(from_start)
 100 
 101         /** For use if the variable ended with a .somename. */
 102         DEFAULTHANDLER(from_named)
 103 
 104         /** For use if the variable ended with .somename[someindex]. */
 105         DEFAULTHANDLER(from_indexed)
 106 
 107         /** For use if the variable is a readonly value (.somename.length). */
 108         DEFAULTHANDLER(from_temporary)
 109 
 110 #undef DEFAULTHANDLER
 111 };

 121 class get_variable_key_visitor : public info_visitor<V, void>
 122 {
 123 public:
 124         // Import typedefs from base class.
 125         using param_t = typename get_variable_key_visitor::param_t;
 126 
 127         get_variable_key_visitor(const std::string& key)
 128                 : key_(key)
 129         {
 130                 if(!config::valid_id(key_)) {
 131                         throw invalid_variablename_exception();
 132                 }
 133         }
 134 
 135         void from_named(param_t state) const
 136         {
 137                 if(key_ == "length") {
 138                         state.temp_val_ = state.child_->child_count(state.key_);
 139                         state.type_ = state_temporary;
 140                         return;
 141                 }

 143                 return do_from_config(V::get_child_at(*state.child_, state.key_, 0), state);
 144         }
 145 
 146         void from_start(param_t state) const
 147         {
 148                 return do_from_config(*state.child_, state);
 149         }
 150 
 151         void from_indexed(param_t state) const
 152         {
 153                 // We do not support aaa[0].length
 154                 return do_from_config(V::get_child_at(*state.child_, state.key_, state.index_), state);
 155         }
 156 
 157 private:
 158         void do_from_config(maybe_const_t<config, V>& cfg, param_t state) const
 159         {
 160                 state.type_ = state_named;
 161                 state.key_ = key_;
 162                 state.child_ = &cfg;
 163         }

 386 /**
 387  * Replaces the child in [startindex, endindex) with 'source'
 388  * 'insert' and 'append' are subcases of this.
 389  */
 390 class replace_range_h
 391 {
 392 public:
 393         typedef config::child_itors result_t;
 394         replace_range_h(std::vector<config>& source)
 395                 : datasource_(source)
 396         {
 397         }
 398 
 399         result_t operator()(config& child, const std::string& key, int startindex, int endindex) const
 400         {
 401                 assert(endindex - startindex >= 0);
 402                 if(endindex > 0) {
 403                         // NOTE: currently this is only called from as_range_visitor_base<vi_policy_create>
 404                         // Based on that assumption we use vi_policy_create::get_child_at here instead of making this
 405             // a class template.
 406                         vi_policy_create::get_child_at(child, key, endindex - 1);
 407                 }
 408 
 409                 int size_diff = datasource_.size() - (endindex - startindex);
 410 
 411                 // remove configs first
 412                 while(size_diff < 0) {
 413                         child.remove_child(key, startindex);
 414                         ++size_diff;
 415                 }
 416 
 417                 size_t index = 0;
 418                 for(index = 0; index < static_cast<size_t>(size_diff); ++index) {
 419                         child.add_child_at(key, config(), startindex + index).swap(datasource_[index]);
 420                 }
 421 
 422                 for(; index < datasource_.size(); ++index) {
 423                         child.child(key, startindex + index).swap(datasource_[index]);
 424                 }
 425 
 426                 return get_child_range(child, key, startindex, datasource_.size());
 427         }
 428 
 429 private:
 430         std::vector<config>& datasource_;
 431 };

 466 class merge_range_h
 467 {
 468 public:
 469         typedef void result_t;
 470         merge_range_h(std::vector<config>& source)
 471                 : datasource_(source)
 472         {
 473         }
 474 
 475         void operator()(config& child, const std::string& key, int startindex, int /*endindex*/) const
 476         {
 477                 // The merge_with function only accepts configs so we convert vector -> config.
 478                 config datatemp;
 479 
 480                 // Add emtpy config to 'shift' the merge to startindex
 481                 for(int index = 0; index < startindex; ++index) {
 482                         datatemp.add_child(key);
 483                 }
 484 
 485                 // move datasource_ -> datatemp
 486                 for(size_t index = 0; index < datasource_.size(); ++index) {
 487                         datatemp.add_child(key).swap(datasource_[index]);
 488                 }
 489 
 490                 child.merge_with(datatemp);
 491         }
 492 
 493 private:
 494         std::vector<config>& datasource_;
