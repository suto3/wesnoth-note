 420 #define AI_SAVE_GOLD ACTIVE BEGIN END
 421     # Enable AI recruitment gold saving.
 422     # ACTIVE: turn from which on gold saving will be active
 423     # BEGIN: begin gold saving when unit cost ratio rises above this value
 424     # END: end gold saving when unit cost ratio drops below this value
 425     # This goes directly into the [ai] tag.
 426 
 427     [aspect]
 428         id=recruitment_save_gold
 429         [facet]
 430             [value]
 431                 active={ACTIVE}
 432                 begin={BEGIN}
 433                 end={END}
 434             [/value]
 435         [/facet]
 436     [/aspect]
 437 #enddef
 438

 442     # This goes directly into the [ai] tag.
