  27 local function ensure_config(cfg)
  28         if type(cfg) == 'table' then
  29                 return true
  30         end
  31         if type(cfg) == 'userdata' then
  32                 if getmetatable(cfg) == 'wml object' then return true end
  33                 error("Expected a table or wml object but got " .. getmetatable(cfg), 3)
  34         else
  35                 error("Expected a table or wml object but got " .. type(cfg), 3)
  36         end
  37         return false
  38 end

  40 --! Returns the first subtag of @a cfg with the given @a name.
  41 --! If @a id is not nil, the "id" attribute of the subtag has to match too.
  42 --! The function also returns the index of the subtag in the array.
  43 function wml.get_child(cfg, name, id)
  44         ensure_config(cfg)

  78 --! Returns an iterator over all the subtags of @a cfg with the given @a name.
  79 function wml.child_range(cfg, tag)
  80         ensure_config(cfg)
  81         local iter, state, i = ipairs(cfg)
  82         local function f(s)
  83                 local c
  84                 repeat
  85                         i,c = iter(s,i)
  86                         if not c then return end
  87                 until c[1] == tag
  88                 return c[2]
  89         end
  90         return f, state

 175 function variable_mt.__index(t, k)
 176         local i = tonumber(k)
 177         if i then
 178                 k = t.__varname .. '[' .. i .. ']'
 179         else

 213 local function resolve_variable_context(ctx, err_hint)
 214         if ctx == nil then
 215                 return {get = wesnoth.get_variable, set = wesnoth.set_variable}
 216         elseif type(ctx) == 'number' and ctx > 0 and ctx <= #wesnoth.sides then
 217                 return resolve_variable_context(wesnoth.sides[ctx])
 218         elseif type(ctx) == 'string' then
 219                 -- TODO: Treat it as a namespace for a global (persistent) variable
 220                 -- (Need Lua API for accessing them first, though.)
 221         elseif getmetatable(ctx) == "unit" then
 222                 return {
 223                         get = function(path) return ctx.variables[path] end,
 224                         set = function(path, val) ctx.variables[path] = val end,
 225                 }
 226         elseif getmetatable(ctx) == "side" then
 227                 return {
 228                         get = function(path) return wesnoth.get_side_variable(ctx.side, path) end,
 229                         set = function(path, val) wesnoth.set_side_variable(ctx.side, path, val) end,
 230                 }
 231         elseif getmetatable(ctx) == "unit variables" or getmetatable(ctx) == "side variables" then
 232                 return {
 233                         get = function(path) return ctx[path] end,
 234                         set = function(path, val) ctx[path] = val end,
 235                 }
 236         end
 237         error(string.format("Invalid context for %s: expected nil, side, or unit", err_hint), 3)
 238 end

 240 --! Fetches all the WML container variables with name @a var.
 241 --! @returns a table containing all the variables (starting at index 1).
 242 function wml.variable.get_array(var, context)
 243         context = resolve_variable_context(context, "get_variable_array")
 244         local result = {}
 245         for i = 1, context.get(var .. ".length") do
 246                 result[i] = context.get(string.format("%s[%d]", var, i - 1))
 247         end
 248         return result
 249 end
