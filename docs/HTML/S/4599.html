 163     # Puts bats in the southern chasm. Random bats will be chosen from
 164     # the BAT_TYPES_VALUE. Chasm hexes are found rather than hard-coded
 165     # so they can be moved or changed on the map without affecting
 166     # this macro. (As long as they don't go too far.) Besides, it's
 167     # easy to do.
 168 #define BATS_EXIT_CAVE BAT_TYPES_VALUE
 169     #find chasm hexes, and put them in an array
 170     [store_locations]
 171         variable=chasm_hexes
 172         terrain=Qx* #Chasm
 173         [and]
 174             x=10
 175             y=21
 176             radius=5
 177         [/and]
 178     [/store_locations]
 179 
 180     # Put a bat into each chasm hex.
 181     [foreach]
 182         array=chasm_hexes
 183         [do]
 184             {RANDOM ({BAT_TYPES_VALUE})}
 185             [unit]
 186                 type=$random
 187                 x=$this_item.x
 188                 y=$this_item.y
 189                 side=2
 190                 animate=yes  # The animation fades in, which looks good. If the animation is ever changed to something else, this might have to be changed.
 191             [/unit]
 192         [/do]
 193     [/foreach]
 194 
 195     {CLEAR_VARIABLE chasm_hexes}
 196 #enddef
