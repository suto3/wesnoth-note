  14 /**
  15         Some information about savefiles:
  16         A saveile can contain:
  17          * General information (toplevel atributes, [multiplayer])
  18             This is present in all savefiles
  19          * [statistics]
  20             This is present in all savefiles but it's not handled by playcampaign/play_controller/saved_game.
  21                 It's handled by savegame.cpp
  22          * [snapshot]
  23             If a savegame was saved during a scenario this contains a snapshot of the game at the point when it was saved.
  24          * [carryover_sides_start]
  25             At start-of-scenrio saves this contains data from the previous scenario that was preserved
  26          * [carryover_sides]
  27             In savefile made during the game, this tag contains data from [carryover_sides_start] that was not used in the current scenario but should be saved for a next scenario
  28          * [replay_start]
  29             A snapshot made very early to replay the game from.
  30          * [replay]
  31             A record of game actions that was made between the creation of [replay_start] and [snapshot].
  32 
  33 
  34 
  35         The following types of savegames are known:
  36          * Start of scenario savefiles
  37             These files only contain general information, statistics, and [carryover_sides_start]
  38                 When these saves are loaded, the scenario data is loaded form the game config using the next_scenario attribute from [carryover_sides_start]
  39          * Expanded Start of scenario savefiles
  40             Similar to normal Start-of-scenario savefiles, but the also contain a [scenario] that contains the scenario data.
  41                 This type is only used internally and usually doesn't get written to the disk.
  42          * Ingame savefile
  43             These files contain general information, statistics, [snapshot], [replay], [replay_start], [snapshot], [carryover_sides]
  44                 These files don't contain a [carryover_sides_start] because both starting points ([replay_start] and [snapshot])
  45                 were made after [carryover_sides_start] was merged into the scenario.
  46          * Replay savefiles
  47             Like a Ingame save made during linger mode, but without the [snapshot]
  48 */

 166 void saved_game::set_defaults()
 167 {
 168         const bool is_loaded_game = this->starting_pos_type_ != STARTINGPOS_SCENARIO;
 169         const bool is_multiplayer_tag = classification().get_tagname() == "multiplayer";
 170         static const std::vector<std::string> team_defaults {
 171                 "carryover_percentage",
 172                 "carryover_add",
 173         };
 174         for(config& side : starting_pos_.child_range("side"))
 175         {
 176                 // Set save_id default value directly after loading to its default to prevent different default behaviour in mp_connect code and sp code.
 177                 if(side["save_id"].empty())
 178                 {
 179                         side["save_id"] = side["id"];
 180                 }
 181                 if(!is_multiplayer_tag && side["side_name"].blank())
 182                 {
 183                         side["side_name"] = side["name"];
 184                 }
 185                 if(!is_loaded_game && !side["current_player"].empty())
 186                 {
 187                         ERR_NG << "Removed invalid 'current_player' attribute from [side] while loading a scenario. Consider using 'side_name' instead\n";
 188                         side["current_player"] = config::attribute_value();
 189                 }
 190                 // Set some team specific values to their defaults specified in scenario
 191                 for(const std::string& att_name : team_defaults)
 192                 {
 193                         const config::attribute_value* scenario_value = starting_pos_.get(att_name);
 194                         config::attribute_value& team_value = side[att_name];
 195                         if(scenario_value && team_value.empty())
 196                         {
 197                                 team_value = *scenario_value;
 198                         }
 199                 }
 200         }
 201 }

 203 void saved_game::expand_scenario()
 204 {
 205         if(this->starting_pos_type_ == STARTINGPOS_NONE && !has_carryover_expanded_)
 206         {
 207                 game_config_manager::get()->load_game_config_for_game(this->classification());
 208                 const config& game_config = game_config_manager::get()->game_config();
 209                 const config& scenario = game_config.find_child(classification().get_tagname(), "id", carryover_["next_scenario"]);
 210                 if(scenario)
 211                 {
 212                         this->starting_pos_type_ = STARTINGPOS_SCENARIO;
 213                         this->starting_pos_ = scenario;
 214                         // A hash has to be generated using an unmodified scenario data.
 215                         mp_settings_.hash = scenario.hash();
 216 
 217                         // Add addon_id information if it exists.
 218                         if (!scenario["addon_id"].empty() && scenario["require_scenario"].to_bool(false)) {
 219                                 mp_settings_.update_addon_requirements(config {"id",scenario["addon_id"], "version", scenario["addon_version"], "min_version", scenario["addon_min_version"]});
 220                         }
 221 
 222                         update_label();
 223                         set_defaults();
 224                 }
 225                 else
 226                 {
 227                         this->starting_pos_type_ = STARTINGPOS_INVALID;
 228                         this->starting_pos_ = config();
 229                 }
 230         }
 231 }

 262 void saved_game::load_mod(const std::string& type, const std::string& id)
 263 {
 264         if(const config& cfg = game_config_manager::get()->
 265                 game_config().find_child(type, "id", id))
 266         {
 267                 // Note the addon_id if this mod is required to play the game in mp
 268                 std::string require_attr = "require_" + type;
 269                 bool require_default = (type == "era"); // By default, eras have "require_era = true", and mods have "require_modification = false"
 270                 if (!cfg["addon_id"].empty() && cfg[require_attr].to_bool(require_default)) {
 271                         mp_settings_.update_addon_requirements(config {"id",cfg["addon_id"], "version", cfg["addon_version"], "min_version", cfg["addon_min_version"]});
 272                 }
 273 
 274                 // Copy events
 275                 for(const config& modevent : cfg.child_range("event"))
 276                 {
 277                         if(modevent["enable_if"].empty() || variable_to_bool(carryover_.child_or_empty("variables"), modevent["enable_if"]))
 278                         {
 279                                 this->starting_pos_.add_child("event", modevent);
 280                         }
 281                 }
 282                 // Copy lua
 283                 for(const config& modlua : cfg.child_range("lua"))
 284                 {
 285                         this->starting_pos_.add_child("lua", modlua);
 286                 }
 287                 // Copy load_resource
 288                 for(const config& load_resource : cfg.child_range("load_resource"))
 289                 {
 290                         this->starting_pos_.add_child("load_resource", load_resource);
 291                 }
 292         }
 293         else
 294         {
 295                 //TODO: A user message instead?
 296                 ERR_NG << "Couldn't find [" << type<< "] with id=" << id <<std::endl;
 297         }
 298 }

 300 // Gets the ids of the mp_era and modifications which were set to be active, then fetches these configs from the game_config and copies their [event] and [lua] to the starting_pos_.
 301 // At this time, also collect the addon_id attributes which appeared in them and put this list in the addon_ids attribute of the mp_settings.
 302 void saved_game::expand_mp_events()
 303 {
 304         expand_scenario();
 305         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !this->starting_pos_["has_mod_events"].to_bool(false))
 306         {
 307                 std::vector<modevents_entry> mods;
 308                 std::set<std::string> loaded_resources;
 309                 boost::copy( mp_settings_.active_mods
 310                         | boost::adaptors::transformed(modevents_entry_for("modification"))
 311                         , std::back_inserter(mods) );
 312                 if(mp_settings_.mp_era != "") //We don't want the error message below if there is no era (= if this is a sp game)
 313                 { mods.emplace_back("era", mp_settings_.mp_era); }
 314                 if(classification_.campaign != "")
 315                 { mods.emplace_back("campaign", classification_.campaign); }
 316 
 317                 // In the first iteration mod contains no [resource]s in all other iterations, mods contains only [resource]s
 318                 do {
 319                         for(modevents_entry& mod : mods)
 320                         {
 321                                 load_mod(mod.type, mod.id);
 322                         }
 323                         mods.clear();
 324                         for(const config& cfg : starting_pos_.child_range("load_resource"))
 325                         {
 326                                 if(loaded_resources.find(cfg["id"].str()) == loaded_resources.end()) {
 327                                         mods.emplace_back("resource", cfg["id"].str());
 328                                         loaded_resources.insert(cfg["id"].str());
 329                                 }
 330                         }
 331                         starting_pos_.clear_children("load_resource");
 332                 } while(!mods.empty());
 333                 this->starting_pos_["has_mod_events"] = true;
 334         }
 335 }

 337 void saved_game::expand_mp_options()
 338 {
 339         if(starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_)
 340         {
 341                 std::vector<modevents_entry> mods;
 342 
 343                 boost::copy( mp_settings_.active_mods
 344                         | boost::adaptors::transformed(modevents_entry_for("modification"))
 345                         , std::back_inserter(mods) );
 346                 mods.emplace_back("era", mp_settings_.mp_era);
 347                 mods.emplace_back("multiplayer", get_scenario_id());
 348                 mods.emplace_back("campaign", classification().campaign);
 349 
 350                 config& variables = carryover_.child_or_add("variables");
 351                 for(modevents_entry& mod : mods)
 352                 {
 353                         if(const config& cfg = this->mp_settings().options.find_child(mod.type, "id", mod.id))
 354                         {
 355                                 for(const config& option : cfg.child_range("option"))
 356                                 {
 357                                         try
 358                                         {
 359                                                 variable_access_create(option["id"], variables).as_scalar() = option["value"];
 360                                         }
 361                                         catch(const invalid_variablename_exception&)
 362                                         {
 363                                                 ERR_NG << "variable " << option["id"] << "cannot be set to " << option["value"] << std::endl;
 364                                         }
 365                                 }
 366                         }
 367                         else
 368                         {
 369                                 LOG_NG << "Couldn't find [" << mod.type<< "] with id=" << mod.id << " for [option]s" << std::endl;
 370                         }
 371                 }
 372         }
 373 }

 375 void saved_game::expand_random_scenario()
 376 {
 377         expand_scenario();
 378         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO)
 379         {
 380                 // If the entire scenario should be randomly generated
 381                 if(!starting_pos_["scenario_generation"].empty())
 382                 {
 383                         LOG_NG << "randomly generating scenario...\n";
 384                         const cursor::setter cursor_setter(cursor::WAIT);
 385 
 386                         config scenario_new = random_generate_scenario(starting_pos_["scenario_generation"],
 387                                 starting_pos_.child("generator"));
 388                         //Preserve "story" form the scenario toplevel.
 389                         for(config& story : starting_pos_.child_range("story"))
 390                         {
 391                                 scenario_new.add_child("story", story);
 392                         }
 393                         scenario_new["id"] = starting_pos_["id"];
 394                         starting_pos_ = scenario_new;
 395                         update_label();
 396                         set_defaults();
 397                 }
 398 
 399                 // If no map_data is provided, try to load the specified file directly
 400                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_file"].empty()) {
 401                         starting_pos_["map_data"] = filesystem::read_map(starting_pos_["map_file"]);
 402                 }
 403 
 404                 // If the map should be randomly generated
 405                 // We dont want that we accidentally to this twice so we check for starting_pos_["map_data"].empty()
 406                 if(starting_pos_["map_data"].empty() && !starting_pos_["map_generation"].empty()) {
 407                         LOG_NG << "randomly generating map...\n";
 408                         const cursor::setter cursor_setter(cursor::WAIT);
 409 
 410                         starting_pos_["map_data"] = random_generate_map(
 411                                 starting_pos_["map_generation"], starting_pos_.child("generator"));
 412                 }
 413         }
 414 }

 416 void saved_game::expand_carryover()
 417 {
 418         expand_scenario();
 419         if(this->starting_pos_type_ == STARTINGPOS_SCENARIO && !has_carryover_expanded_)
 420         {
 421                 carryover_info sides(carryover_);
 422 
 423                 sides.transfer_to(get_starting_pos());
 424                 for(config& side_cfg : get_starting_pos().child_range("side"))
 425                 {
 426                         sides.transfer_all_to(side_cfg);
 427                 }
 428 
 429                 carryover_ = sides.to_config();
 430                 has_carryover_expanded_ = true;
 431         }
 432 }

 466 const config& saved_game::get_replay_starting_pos()
 467 {
 468         if(!replay_start_.empty())
 469         {
 470                 return replay_start_;
 471         }
 472         if(!has_carryover_expanded_)
 473         {
 474                 //Try to load the scenario form game config or from [scenario] if there is no [replay_start]
 475                 expand_scenario();
 476                 expand_carryover();
 477         }
 478         if(starting_pos_type_ == STARTINGPOS_SCENARIO)

 480                 return starting_pos_;
 481         }
 482         return this->replay_start_.child("some_non_existet_invalid");

 485 void saved_game::convert_to_start_save()
 486 {
 487         assert(starting_pos_type_ == STARTINGPOS_SNAPSHOT);
 488         carryover_info sides(starting_pos_, true);
 489         sides.merge_old_carryover(carryover_info(carryover_));
 490         sides.rng().rotate_random();
 491         carryover_ = sides.to_config();
 492         has_carryover_expanded_ = false;
 493         replay_data_ = replay_recorder_base();
 494         replay_start_ = config();
 495         remove_snapshot();
 496 }

 521 std::string saved_game::get_scenario_id()

 555 void saved_game::cancel_orders()
 556 {
 557         for(config &side : this->starting_pos_.child_range("side"))
 558         {
 559                 // for humans "goto_x/y" is used for multi-turn-moves
 560                 // for the ai "goto_x/y" is a way for wml to order the ai to move a unit to a certain place.
 561                 // we want to cancel human order but not to break wml.
 562                 if (side["controller"] != "human" && side["controller"] != "network") continue;
 563                 for(config &unit : side.child_range("unit"))
 564                 {
 565                         unit["goto_x"] = -999;
 566                         unit["goto_y"] = -999;
 567                 }
 568         }
 569 }

 600 void saved_game::set_data(config& cfg)
 601 {
 602         log_scope("read_game");
 603 
 604         if(config & caryover_sides = cfg.child("carryover_sides"))
 605         {
 606                 carryover_.swap(caryover_sides);
 607                 has_carryover_expanded_ = true;
 608         }
 609         else if(config & caryover_sides_start = cfg.child("carryover_sides_start"))
 610         {
 611                 carryover_.swap(caryover_sides_start);
 612                 has_carryover_expanded_ = false;
 613         }
 614         else
 615         {
 616                 carryover_ = config();
 617                 has_carryover_expanded_ = false;
 618         }
 619 
 620         if(config & replay_start = cfg.child("replay_start"))
 621         {
 622                 replay_start_.swap(replay_start);
 623         }
 624         else
 625         {
 626                 replay_start_ = config();
 627         }
 628         replay_data_ = replay_recorder_base();
 629         //Serversided replays can contain multiple [replay]
 630         for(config& replay : cfg.child_range("replay"))
 631         {
 632                 replay_data_.append_config(replay);
 633         }
 634         replay_data_.set_to_end();
 635         if(config& snapshot = cfg.child("snapshot"))
 636         {
 637                 this->starting_pos_type_ = STARTINGPOS_SNAPSHOT;
 638                 this->starting_pos_.swap(snapshot);
 639         }
 640         else if(config& scenario = cfg.child("scenario"))
 641         {
 642                 this->starting_pos_type_ = STARTINGPOS_SCENARIO;
 643                 this->starting_pos_.swap(scenario);
 644         }
 645         else
 646         {
 647                 this->starting_pos_type_ = STARTINGPOS_NONE;
 648                 this->starting_pos_ = config();
 649         }
 650 
 651         LOG_NG << "scenario: '" << carryover_["next_scenario"].str() << "'\n";
 652 
 653         if (const config &stats = cfg.child("statistics")) {
 654                 statistics::fresh_stats();
 655                 statistics::read_stats(stats);
 656         }
 657 
 658         classification_ = game_classification(cfg);
 659         mp_settings_ = mp_game_settings(cfg.child_or_empty("multiplayer"));
 660         cfg.clear();
 661 }
 662 
 663 void saved_game::clear()
 664 {
 665         carryover_.clear();
 666         classification_ = game_classification();
 667         has_carryover_expanded_ = false;
 668         mp_settings_ = mp_game_settings();
 669         replay_data_.swap(replay_recorder_base());
 670         replay_start_.clear();
 671         starting_pos_.clear();
 672         starting_pos_type_ = STARTINGPOS_NONE;
 673 }
