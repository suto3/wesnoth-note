  49 def getIndent(itemText, subitemText):
  50     item = re.compile('^( |\t)*').search(itemText).group()
  51     subitem = re.compile('^( |\t)*').search(subitemText).group()
  52     if item == '' or subitem == '':
  53         return subitem
  54     if item[0] == '\t' and subitem[0] == '\t':
  55         return (len(subitem) - len(item)) * '\t'
  56     if item[0] == ' ' and subitem[0] == ' ':
  57         return (len(subitem) - len(item)) * ' '
  58     return '\t'

 145     def parseScenario (self, scenarioContent):
 146         self.addScenarioData(scenarioContent)
 147         sideDetail = side_block.search(scenarioContent, 0)
 148         while sideDetail != None:
 149             self.addSideData(sideDetail.group())
 150             self.addAiData(sideDetail.group())
 151             searchStart = sideDetail.end()
 152             sideDetail = side_block.search(scenarioContent, searchStart)
 153         self.updateAi()

 191 def parseAll(dirName, fileList):
 192     scenarioListIndex = 0
 193     scenarioList = wikiScenarioList()
 194     for fileName in fileList:
 195         if os.path.splitext(fileName)[1] != '.cfg':
 196             continue
 197         if os.path.isdir(os.path.join(dirName, fileName)):
 198             continue
 199         f = file(os.path.join(dirName, fileName))
 200         fileContent = f.read()
 201         f.close()
 202         searchStart = 0
 203         scenario = scenario_block.match(fileContent, searchStart)
 204         while scenario !=  None:
 205             scenarioList.addScenario(wikiScenario())
 206             scenarioList[scenarioListIndex].parseScenario(scenario.group(0))
 207             searchStart = scenario.end()
 208             scenario = scenario_block.search(fileContent, searchStart)
 209             scenarioListIndex += 1
 210         updated_file = fileContent
 211         for scenarioItem in scenarioList:
 212             if scenarioItem.full_description != scenarioItem.updated_description:
 213                 updated_file = updated_file.replace(scenarioItem.full_description, scenarioItem.updated_description)
 214         if updated_file != fileContent:
 215             (basename_out, ext_out) = os.path.splitext(fileName)
 216             basename_out = basename_out + suffix + ext_out
 217             f = file(basename_out,'w')
 218             f.write(updated_file)
 219             f.close()
