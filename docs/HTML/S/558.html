  49         template <typename... T>
  50         auto zip(const T&... containers) -> boost::iterator_range<boost::zip_iterator<decltype(boost::make_tuple(std::begin(containers)...))>>
  51         {
  52                 auto zip_begin = boost::make_zip_iterator(boost::make_tuple(std::begin(containers)...));
  53                 auto zip_end = boost::make_zip_iterator(boost::make_tuple(std::end(containers)...));
  54                 return boost::make_iterator_range(zip_begin, zip_end);
  55         }

  89 static inline tree_view_node& add_name_tree_node(tree_view_node& header_node, const std::string& type, const t_string& label, const t_string& tooltip = "")
  90 {
  91         /* Note: We have to pass data instead of just doing 'child_label.set_label(label)' below
  92          * because the tree_view_node::add_child needs to have the correct size of the
  93          * node child widgets for its internal size calculations.
  94          * Same is true for 'use_markup'
  95          */
  96         auto& child_node = header_node.add_child(type, { { "name",{ { "label", label },{ "use_markup", "true" } } } });
  97         auto& child_label = find_widget<styled_widget>(&child_node, "name", true);
  98 
  99         child_label.set_tooltip(tooltip);
 100         return child_node;
 101 }

 103 static inline std::string get_hp_tooltip(const utils::string_map& res, const std::function<int (const std::string&, bool)>& get)
 104 {
 105         std::ostringstream tooltip;
 106 
 107         std::set<std::string> resistances_table;
 108 
 109         bool att_def_diff = false;
 110         for(const utils::string_map::value_type &resist : res) {
 111                 std::ostringstream line;
 112                 line << translation::gettext(resist.first.c_str()) << ": ";
 113 
 114                 // Some units have different resistances when attacking or defending.
 115                 const int res_att = 100 - get(resist.first, true);
 116                 const int res_def = 100 - get(resist.first, false);
 117 
 118                 if(res_att == res_def) {
 119                         line << "<span color='" << unit_helper::resistance_color(res_def) << "'>" << utils::signed_percent(res_def) << "</span>";
 120                 } else {
 121                         line << "<span color='" << unit_helper::resistance_color(res_att) << "'>" << utils::signed_percent(res_att) << "</span>" << "/"
 122                              << "<span color='" << unit_helper::resistance_color(res_def) << "'>" << utils::signed_percent(res_def) << "</span>";
 123                         att_def_diff = true;
 124                 }
 125 
 126                 resistances_table.insert(line.str());
 127         }
 128 
 129         tooltip << "<big>" << _("Resistances: ") << "</big>";
 130         if(att_def_diff) {
 131                 tooltip << _("(Att / Def)");
 132         }
 133 
 134         for(const std::string &line : resistances_table) {
 135                 tooltip << '\n' << font::unicode_bullet << " " << line;
 136         }
 137 
 138         return tooltip.str();
 139 }

 141 static inline std::string get_mp_tooltip(int total_movement, std::function<int (t_translation::terrain_code)> get)
 142 {
 143         std::ostringstream tooltip;
 144         tooltip << "<big>" << _("Movement Costs:") << "</big>";
 145 
 146         ter_data_cache tdata = help::load_terrain_types_data();
 147 
 148         if(!tdata) {
 149                 return "";
 150         }
 151 
 152         for(t_translation::terrain_code terrain : preferences::encountered_terrains()) {
 153                 if(terrain == t_translation::FOGGED || terrain == t_translation::VOID_TERRAIN || t_translation::terrain_matches(terrain, t_translation::ALL_OFF_MAP)) {
 154                         continue;
 155                 }
 156 
 157                 const terrain_type& info = tdata->get_terrain_info(terrain);
 158                 if(info.union_type().size() == 1 && info.union_type()[0] == info.number() && info.is_nonnull()) {
 159                         const std::string& name = info.name();
 160                         const int moves = get(terrain);
 161 
 162                         tooltip << '\n' << font::unicode_bullet << " " << name << ": ";
 163 
 164                         // movement  -  range: 1 .. 5, movetype::UNREACHABLE=impassable
 165                         const bool cannot_move = moves > total_movement;
 166 
 167                         std::string color;
 168                         if(cannot_move) {
 169                                 // cannot move in this terrain
 170                                 color = "red";
 171                         } else if(moves > 1) {
 172                                 color = "yellow";
 173                         } else {
 174                                 color = "white";
 175                         }
 176 
 177                         tooltip << "<span color='" << color << "'>";
 178 
 179                         // A 5 MP margin; if the movement costs go above the unit's max moves + 5, we replace it with dashes.
 180                         if(cannot_move && (moves > total_movement + 5)) {
 181                                 tooltip << font::unicode_figure_dash;
 182                         } else {
 183                                 tooltip << moves;
 184                         }
 185 
 186                         tooltip << "</span>";
 187                 }
 188         }
 189 
 190         return tooltip.str();
 191 }

 197 template<typename T>
 198 void unit_preview_pane::print_attack_details(T attacks, tree_view_node& parent_node)
 199 {
 200         if(attacks.empty()) {
 201                 return;
 202         }
 203 
 204         auto& header_node = add_name_tree_node(
 205                 parent_node,
 206                 "header",
 207                 "<b>" + _("Attacks") + "</b>"
 208         );
 209 
 210         for(const auto& a : attacks) {
 211 
 212                 auto& subsection = add_name_tree_node(
 213                         header_node,
 214                         "item",
 215                         (formatter() << "<span color='#f5e6c1'>" << a.damage() << font::weapon_numbers_sep << a.num_attacks() << " " << a.name() << "</span>").str()
 216                 );
 217 
 218                 add_name_tree_node(
 219                         subsection,
 220                         "item",
 221                         (formatter() << "<span color='#a69275'>" << a.range() << font::weapon_details_sep << a.type() << "</span>").str()
 222                 );
 223 
 224                 for(const auto& pair : a.special_tooltips()) {
 225                         add_name_tree_node(
 226                                 subsection,
 227                                 "item",
 228                                 (formatter() << font::span_color(font::weapon_details_color) << pair.first << "</span>").str(),
 229                                 (formatter() << "<span size='x-large'>" << pair.first << "</span>" << "\n" << pair.second).str()
 230                         );
 231                 }
 232         }
 233 }

 235 void unit_preview_pane::set_displayed_type(const unit_type& type)
 236 {
 237         // Sets the current type id for the profile button callback to use
 238         current_type_ = type.id();
 239 
 240         if(icon_type_) {
 241                 std::string mods;
 242 
 243                 if(resources::controller) {
 244                         mods = "~RC(" + type.flag_rgb() + ">" +
 245                                  team::get_side_color_index(resources::controller->current_side())
 246                                  + ")";
 247                 }
 248 
 249                 mods += "~SCALE_INTO_SHARP(144,144)" + image_mods_;
 250 
 251                 icon_type_->set_label((type.icon().empty() ? type.image() : type.icon()) + mods);
 252         }
 253 
 254         if(label_name_) {
 255                 label_name_->set_label("<big>" + type.type_name() + "</big>");
 256                 label_name_->set_use_markup(true);
 257         }
 258 
 259         if(label_level_) {
 260                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(type.level())}});
 261 
 262                 label_level_->set_label("<b>" + l_str + "</b>");
 263                 label_level_->set_use_markup(true);
 264         }
 265 
 266         if(label_race_) {
 267                 label_race_ ->set_label(type.race()->name(type.genders().front()));
 268         }
 269 
 270         if(icon_race_) {
 271                 icon_race_->set_label("icons/unit-groups/race_" + type.race_id() + "_30.png");
 272         }
 273 
 274         if(icon_alignment_) {
 275                 const std::string& alignment_name = type.alignment().to_string();
 276 
 277                 icon_alignment_->set_label("icons/alignments/alignment_" + alignment_name + "_30.png");
 278                 icon_alignment_->set_tooltip(unit_type::alignment_description(
 279                         type.alignment(),
 280                         type.genders().front()));
 281         }
 282 
 283         if(label_details_) {
 284                 std::stringstream str;
 285 
 286                 str << "<span size='large'> </span>" << "\n";
 287 
 288                 str << "<span color='#a69275'>" << type.type_name() << "</span>" << "\n";
 289 
 290                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(type.level())}});
 291                 str << l_str << "\n";
 292 
 293                 str << type.alignment() << "\n";
 294 
 295                 str << "\n"; // Leave a blank line where traits would be
 296 
 297                 str <<  _("HP: ") << type.hitpoints() << "\n";
 298 
 299                 str << _("XP: ") << type.experience_needed(true);
 300 
 301                 label_details_->set_label(str.str());
 302                 label_details_->set_use_markup(true);
 303         }
 304 
 305         if(tree_details_) {
 306 
 307                 tree_details_->clear();
 308                 tree_details_->add_node("hp_xp_mp", {
 309                         { "hp",{
 310                                 { "label", (formatter() << "<small>" << "<span color='#21e100'>" << "<b>" << _("HP: ") << "</b>" << type.hitpoints() << "</span>" << " | </small>").str() },
 311                                 { "use_markup", "true" },
 312                                 { "tooltip", get_hp_tooltip(type.movement_type().get_resistances().damage_table(), [&type](const std::string& dt, bool is_attacker) { return type.resistance_against(dt, is_attacker); }) }
 313                         } },
 314                         { "xp",{
 315                                 { "label", (formatter() << "<small>" << "<span color='#00a0e1'>" << "<b>" << _("XP: ") << "</b>" << type.experience_needed() << "</span>" << " | </small>").str() },
 316                                 { "use_markup", "true" },
 317                                 { "tooltip", (formatter() << _("Experience Modifier: ") << unit_experience_accelerator::get_acceleration() << '%').str() }
 318                         } },
 319                         { "mp",{
 320                                 { "label", (formatter() << "<small>" << "<b>" << _("MP: ") << "</b>" << type.movement() << "</small>").str() },
 321                                 { "use_markup", "true" },
 322                                 { "tooltip", get_mp_tooltip(type.movement(), [&type](t_translation::terrain_code terrain) { return type.movement_type().movement_cost(terrain); }) }
 323                         } },
 324                 });
 325 
 326                 // Print trait details
 327                 {
 328                         tree_view_node* header_node = nullptr;
 329 
 330                         for(const auto& tr : type.possible_traits()) {
 331                                 t_string name = tr[type.genders().front() == unit_race::FEMALE ? "female_name" : "male_name"];
 332                                 if(tr["availability"] != "musthave" || name.empty()) {
 333                                         continue;
 334                                 }
 335 
 336                                 if(header_node == nullptr) {
 337                                         header_node = &add_name_tree_node(
 338                                                 tree_details_->get_root_node(),
 339                                                 "header",
 340                                                 "<b>" + _("Traits") + "</b>"
 341                                         );
 342                                 }
 343 
 344                                 add_name_tree_node(
 345                                         *header_node,
 346                                         "item",
 347                                         name
 348                                 );
 349                         }
 350                 }
 351 
 352                 // Print ability details
 353                 if(!type.abilities().empty()) {
 354 
 355                         auto& header_node = add_name_tree_node(
 356                                 tree_details_->get_root_node(),
 357                                 "header",
 358                                 "<b>" + _("Abilities") + "</b>"
 359                         );
 360 
 361                         for(const auto& ab : zip(type.abilities() , type.ability_tooltips())) {
 362                                 add_name_tree_node(
 363                                         header_node,
 364                                         "item",
 365                                         boost::get<0>(ab),
 366                                         (formatter() << "<span size='x-large'>" << boost::get<0>(ab) << "</span>\n" << boost::get<1>(ab)).str()
 367                                 );
 368                         }
 369                 }
 370 
 371                 print_attack_details(type.attacks(), tree_details_->get_root_node());
 372         }

 375 void unit_preview_pane::set_displayed_unit(const unit& u)
 376 {
 377         // Sets the current type id for the profile button callback to use
 378         current_type_ = u.type_id();
 379 
 380         if(icon_type_) {
 381                 std::string mods = u.image_mods();
 382 
 383                 if(u.can_recruit()) {
 384                         mods += "~BLIT(" + unit::leader_crown() + ")";
 385                 }
 386 
 387                 for(const std::string& overlay : u.overlays()) {
 388                         mods += "~BLIT(" + overlay + ")";
 389                 }
 390 
 391                 mods += "~SCALE_INTO_SHARP(144,144)" + image_mods_;
 392 
 393                 icon_type_->set_label(u.absolute_image() + mods);
 394         }
 395 
 396         if(label_name_) {
 397                 std::string name;
 398                 if(!u.name().empty()) {
 399                         name = "<span size='large'>" + u.name() + "</span>" + "\n" + "<small><span color='#a69275'>" + u.type_name() + "</span></small>";
 400                 } else {
 401                         name = "<span size='large'>" + u.type_name() + "</span>\n";
 402                 }
 403 
 404                 label_name_->set_label(name);
 405                 label_name_->set_use_markup(true);
 406         }
 407 
 408         if(label_level_) {
 409                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(u.level())}});
 410 
 411                 label_level_->set_label("<b>" + l_str + "</b>");
 412                 label_level_->set_use_markup(true);
 413         }
 414 
 415         if(label_race_) {
 416                 label_race_->set_label(u.race()->name(u.gender()));
 417         }
 418 
 419         if(icon_race_) {
 420                 icon_race_->set_label("icons/unit-groups/race_" + u.race()->id() + "_30.png");
 421         }
 422 
 423         if(icon_alignment_) {
 424                 const std::string& alignment_name = u.alignment().to_string();
 425 
 426                 icon_alignment_->set_label("icons/alignments/alignment_" + alignment_name + "_30.png");
 427                 icon_alignment_->set_tooltip(unit_type::alignment_description(
 428                         u.alignment(),
 429                         u.gender()));
 430         }
 431 
 432         if(label_details_) {
 433                 std::stringstream str;
 434 
 435                 const std::string name = "<span size='large'>" + (!u.name().empty() ? u.name() : " ") + "</span>";
 436                 str << name << "\n";
 437 
 438                 str << "<span color='#a69275'>" << u.type_name() << "</span>" << "\n";
 439 
 440                 std::string l_str = vgettext("Lvl $lvl", {{"lvl", std::to_string(u.level())}});
 441                 str << l_str << "\n";
 442 
 443                 str << u.alignment() << "\n";
 444 
 445                 str << utils::join(u.trait_names(), ", ") << "\n";
 446 
 447                 str << font::span_color(u.hp_color())
 448                         << _("HP: ") << u.hitpoints() << "/" << u.max_hitpoints() << "</span>" << "\n";
 449 
 450                 str << font::span_color(u.xp_color())
 451                         << _("XP: ") << u.experience() << "/" << u.max_experience() << "</span>";
 452 
 453                 label_details_->set_label(str.str());
 454                 label_details_->set_use_markup(true);
 455         }
 456 
 457         if(tree_details_) {
 458                 tree_details_->clear();
 459                 tree_details_->add_node("hp_xp_mp", {
 460                         { "hp",{
 461                                 { "label", (formatter() << "<small>" << font::span_color(u.hp_color()) << "<b>" << _("HP: ") << "</b>" << u.hitpoints() << "/" << u.max_hitpoints() << "</span>" << " | </small>").str() },
 462                                 { "use_markup", "true" },
 463                                 { "tooltip", get_hp_tooltip(u.get_base_resistances(), [&u](const std::string& dt, bool is_attacker) { return u.resistance_against(dt, is_attacker, u.get_location()); }) }
 464                         } },
 465                         { "xp",{
 466                                 { "label", (formatter() << "<small>" << font::span_color(u.xp_color()) << "<b>" << _("XP: ") << "</b>" << u.experience() << "/" << u.max_experience() << "</span>" << " | </small>").str() },
 467                                 { "use_markup", "true" },
 468                                 { "tooltip", (formatter() << _("Experience Modifier: ") << unit_experience_accelerator::get_acceleration() << '%').str() }
 469                         } },
 470                         { "mp",{
 471                                 { "label", (formatter() << "<small>" << "<b>" << _("MP: ") << "</b>" << u.movement_left() << "/" << u.total_movement() << "</small>").str() },
 472                                 { "use_markup", "true" },
 473                                 { "tooltip", get_mp_tooltip(u.total_movement(), [&u](t_translation::terrain_code terrain) { return u.movement_cost(terrain); }) }
 474                         } },
 475                 });
 476 
 477                 if(!u.trait_names().empty()) {
 478                         auto& header_node = add_name_tree_node(
 479                                 tree_details_->get_root_node(),
 480                                 "header",
 481                                 "<b>" + _("Traits") + "</b>"
 482                         );
 483 
 484                         assert(u.trait_names().size() == u.trait_descriptions().size());
 485                         for (size_t i = 0; i < u.trait_names().size(); ++i) {
 486                                 add_name_tree_node(
 487                                         header_node,
 488                                         "item",
 489                                         u.trait_names()[i],
 490                                         u.trait_descriptions()[i]
 491                                 );
 492                         }
 493                 }
 494 
 495                 if(!u.get_ability_list().empty()) {
 496                         auto& header_node = add_name_tree_node(
 497                                 tree_details_->get_root_node(),
 498                                 "header",
 499                                 "<b>" + _("Abilities") + "</b>"
 500                         );
 501 
 502                         for(const auto& ab : u.ability_tooltips()) {
 503                                 add_name_tree_node(
 504                                         header_node,
 505                                         "item",
 506                                         std::get<1>(ab),
 507                                         std::get<2>(ab)
 508                                 );
 509                         }
 510                 }
 511                 print_attack_details(u.attacks(), tree_details_->get_root_node());
 512         }
 513 }
