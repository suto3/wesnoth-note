 134 unsigned pango_text::insert_text(const unsigned offset, const std::string& text)
 135 {
 136         if (text.empty() || length_ == maximum_length_) {
 137                 return 0;
 138         }
 139 
 140         // do we really need that assert? utf8::insert will just append in this case, which seems fine
 141         assert(offset <= length_);
 142 
 143         unsigned len = utf8::size(text);
 144         if (length_ + len > maximum_length_) {
 145                 len = maximum_length_ - length_;
 146         }
 147         const utf8::string insert = text.substr(0, utf8::index(text, len));
 148         utf8::string tmp = text_;
 149         this->set_text(utf8::insert(tmp, offset, insert), false);
 150         // report back how many characters were actually inserted (e.g. to move the cursor selection)
 151         return len;
 152 }

 165 gui2::point pango_text::get_cursor_position(
 166                 const unsigned column, const unsigned line) const
 167 {
 168         this->recalculate();
 169 
 170         // First we need to determine the byte offset, if more routines need it it
 171         // would be a good idea to make it a separate function.
 172         p_itor itor{layout_};
 173 
 174         // Go the wanted line.
 175         if(line != 0) {
 176                 if(pango_layout_get_line_count(layout_) >= static_cast<int>(line)) {
 177                         return gui2::point(0, 0);
 178                 }
 179 
 180                 for(size_t i = 0; i < line; ++i) {
 181                         pango_layout_iter_next_line(itor);
 182                 }
 183         }
 184 
 185         // Go the wanted column.
 186         for(size_t i = 0; i < column; ++i) {
 187                 if(!pango_layout_iter_next_char(itor)) {
 188                         // It seems that the documentation is wrong and causes and off by
 189                         // one error... the result should be false if already at the end of
 190                         // the data when started.
 191                         if(i + 1 == column) {
 192                                 break;
 193                         }
 194                         // We are beyond data.
 195                         return gui2::point(0, 0);
 196                 }
 197         }
 198 
 199         // Get the byte offset
 200         const int offset = pango_layout_iter_get_index(itor);
 201 
 202         // Convert the byte offset in a position.
 203         PangoRectangle rect;
 204         pango_layout_get_cursor_pos(layout_, offset, &rect, nullptr);
 205 
 206         return gui2::point(PANGO_PIXELS(rect.x), PANGO_PIXELS(rect.y));
 207 }

 209 std::string pango_text::get_token(const gui2::point & position, const char * delim) const
 210 {
 211         this->recalculate();
 212 
 213         // Get the index of the character.
 214         int index, trailing;
 215         if (!pango_layout_xy_to_index(layout_, position.x * PANGO_SCALE,
 216                 position.y * PANGO_SCALE, &index, &trailing)) {
 217                 return "";
 218         }
 219 
 220         std::string txt = pango_layout_get_text(layout_);
 221 
 222         std::string d(delim);
 223 
 224         if (index < 0 || (static_cast<size_t>(index) >= txt.size()) || d.find(txt.at(index)) != std::string::npos) {
 225                 return ""; // if the index is out of bounds, or the index character is a delimiter, return nothing
 226         }
 227 
 228         size_t l = index;
 229         while (l > 0 && (d.find(txt.at(l-1)) == std::string::npos)) {
 230                 --l;
 231         }
 232 
 233         size_t r = index + 1;
 234         while (r < txt.size() && (d.find(txt.at(r)) == std::string::npos)) {
 235                 ++r;
 236         }
 237 
 238         return txt.substr(l,r-l);
 239 }

 256 gui2::point pango_text::get_column_line(const gui2::point& position) const
 257 {
 258         this->recalculate();
 259 
 260         // Get the index of the character.
 261         int index, trailing;
 262         pango_layout_xy_to_index(layout_, position.x * PANGO_SCALE,
 263                 position.y * PANGO_SCALE, &index, &trailing);
 264 
 265         // Extract the line and the offset in pixels in that line.
 266         int line, offset;
 267         pango_layout_index_to_line_x(layout_, index, trailing, &line, &offset);
 268         offset = PANGO_PIXELS(offset);
 269 
 270         // Now convert this offset to a column, this way is a bit hacky but haven't
 271         // found a better solution yet.
 272 
 273         /**
 274          * @todo There's still a bug left. When you select a text which is in the
 275          * ellipses on the right side the text gets reformatted with ellipses on
 276          * the left and the selected character is not the one under the cursor.
 277          * Other widget toolkits don't show ellipses and have no indication more
 278          * text is available. Haven't found what the best thing to do would be.
 279          * Until that time leave it as is.
 280          */
 281         for(size_t i = 0; ; ++i) {
 282                 const int pos = this->get_cursor_position(i, line).x;
 283 
 284                 if(pos == offset) {
 285                         return  gui2::point(i, line);
 286                 }
 287         }

 290 bool pango_text::set_text(const std::string& text, const bool markedup)
 291 {
 292         if(markedup != markedup_text_ || text != text_) {
 293                 // assert(layout_);
 294 
 295                 const ucs4::string wide = unicode_cast<ucs4::string>(text);
 296                 const std::string narrow = unicode_cast<utf8::string>(wide);
 297                 if(text != narrow) {
 298                         ERR_GUI_L << "pango_text::" << __func__
 299                                         << " text '" << text
 300                                         << "' contains invalid utf-8, trimmed the invalid parts.\n";
 301                 }
 302                 if(markedup) {
 303                         if(!this->set_markup(narrow)) {
 304                                 return false;
 305                         }
 306                 } else {
 307                         /*
 308                          * pango_layout_set_text after pango_layout_set_markup might
 309                          * leave the layout in an undefined state regarding markup so
 310                          * clear it unconditionally.
 311                          */
 312                         pango_layout_set_attributes(layout_, nullptr);
 313                         pango_layout_set_text(layout_, narrow.c_str(), narrow.size());
 314                 }
 315                 text_ = narrow;
 316                 length_ = wide.size();
 317                 markedup_text_ = markedup;
 318                 calculation_dirty_ = true;
 319                 surface_dirty_ = true;
 320         }
 321 
 322         return true;
 323 }

 493 void pango_text::recalculate(const bool force) const
 494 {
 495         if(calculation_dirty_ || force) {
 496                 // assert(layout_);
 497 
 498                 calculation_dirty_ = false;
 499                 surface_dirty_ = true;
 500 
 501                 p_font font{get_font_families(font_class_), font_size_, font_style_};
 502                 pango_layout_set_font_description(layout_, font.get());
 503 
 504                 if(font_style_ & pango_text::STYLE_UNDERLINE) {
 505                         PangoAttrList *attribute_list = pango_attr_list_new();
 506                         pango_attr_list_insert(attribute_list
 507                                         , pango_attr_underline_new(PANGO_UNDERLINE_SINGLE));
 508 
 509                         pango_layout_set_attributes (layout_, attribute_list);
 510                         pango_attr_list_unref(attribute_list);
 511                 }
 512 
 513                 int maximum_width = 0;
 514                 if(characters_per_line_ != 0) {
 515                         PangoFont* f = pango_font_map_load_font(
 516                                           pango_cairo_font_map_get_default()
 517                                         , context_
 518                                         , font.get());
 519 
 520                         PangoFontMetrics* m = pango_font_get_metrics(f, nullptr);
 521 
 522                         int w = pango_font_metrics_get_approximate_char_width(m);
 523                         w *= characters_per_line_;
 524 
 525                         maximum_width = ceil(pango_units_to_double(w));
 526                 } else {
 527                         maximum_width = maximum_width_;
 528                 }
 529 
 530                 if(maximum_width_ != -1) {
 531                         maximum_width = std::min(maximum_width, maximum_width_);
 532                 }
 533 
 534                 /*
 535                  * See set_maximum_width for some more background info as well.
 536                  * In order to fix the problem first set a width which seems to render
 537                  * correctly then lower it to fit. For the campaigns the 4 does "the
 538                  * right thing" for the terrain labels it should use the value 0 to set
 539                  * the ellipse properly. Need to see whether this is a bug in pango or
 540                  * a bug in my understanding of the pango api.
 541                  */
 542                 int hack = 4;
 543                 do {
 544                         pango_layout_set_width(layout_, maximum_width == -1
 545                                         ? -1
 546                                         : (maximum_width + hack) * PANGO_SCALE);
 547                         pango_layout_get_pixel_extents(layout_, nullptr, &rect_);
 548 
 549                         DBG_GUI_L << "pango_text::" << __func__
 550                                         << " text '" << gui2::debug_truncate(text_)
 551                                         << "' maximum_width " << maximum_width
 552                                         << " hack " << hack
 553                                         << " width " << rect_.x + rect_.width
 554                                         << ".\n";
 555 
 556                         --hack;
 557                 } while(maximum_width != -1
 558                                 && hack >= 0 && rect_.x + rect_.width > maximum_width);
 559 
 560                 DBG_GUI_L << "pango_text::" << __func__
 561                                 << " text '" << gui2::debug_truncate(text_)
 562                                 << "' font_size " << font_size_
 563                                 << " markedup_text " << markedup_text_
 564                                 << " font_style " << std::hex << font_style_ << std::dec
 565                                 << " maximum_width " << maximum_width
 566                                 << " maximum_height " << maximum_height_
 567                                 << " result " <<  rect_
 568                                 << ".\n";
 569                 if(maximum_width != -1 && rect_.x + rect_.width > maximum_width) {
 570                         DBG_GUI_L << "pango_text::" << __func__
 571                                         << " text '" << gui2::debug_truncate(text_)
 572                                         << " ' width " << rect_.x + rect_.width
 573                                         << " greater as the wanted maximum of " << maximum_width
 574                                         << ".\n";
 575                 }
 576         }
 577 }

 579 /***
 580  * Inverse table
 581  *
 582  * Holds a high-precision inverse for each number i, that is, a number x such that x * i / 256 is close to 255.
 583  */
 584 struct inverse_table
 585 {
 586         unsigned values[256];
 587 
 588         inverse_table()
 589         {
 590                 values[0] = 0;
 591                 for (int i = 1; i < 256; ++i) {
 592                         values[i] = (255 * 256) / i;
 593                 }
 594         }
 595 
 596         unsigned operator[](Uint8 i) const { return values[i]; }
 597 };
 598 
 599 static const inverse_table inverse_table_;
 600 
 601 /***
 602  * Helper function for un-premultiplying alpha
 603  * Div should be the high-precision inverse for the alpha value.
 604  */
 605 static void unpremultiply(Uint8 & value, const unsigned div) {
 606         unsigned temp = (value * div) / 256u;
 607         // Note: It's always the case that alpha * div < 256 if div is the inverse
 608         // for alpha, so if cairo is computing premultiplied alpha by rounding down,
 609         // this min is not necessary. However, if cairo generates illegal output,
 610         // the min may be selected.
 611         // It's probably not worth removing the min, since branch prediction will
 612         // make it essentially free if one of the branches is never actually
 613         // selected.
 614         value = std::min(255u, temp);
 615 }

 617 /**
 618  * Converts from cairo-format ARGB32 premultiplied alpha to plain alpha.
 619  * @param c a uint32 representing the color
 620  */
 621 static void from_cairo_format(Uint32 & c)
 622 {
 623         Uint8 a = (c >> 24) & 0xff;
 624         Uint8 r = (c >> 16) & 0xff;
 625         Uint8 g = (c >> 8) & 0xff;
 626         Uint8 b = c & 0xff;
 627 
 628         const unsigned div = inverse_table_[a];
 629         unpremultiply(r, div);
 630         unpremultiply(g, div);
 631         unpremultiply(b, div);
 632 
 633         c = (static_cast<Uint32>(a) << 24) | (static_cast<Uint32>(r) << 16) | (static_cast<Uint32>(g) << 8) | static_cast<Uint32>(b);
 634 }

 636 void pango_text::rerender(const bool force) const
 637 {
 638         if(surface_dirty_ || force) {
 639                 // assert(layout_);
 640 
 641                 this->recalculate(force);
 642                 surface_dirty_ = false;
 643 
 644                 // TODO: This looks broken to me. If rect_.x is negative, shouldn't that increase width?
 645                 // I think it maybe should be
 646                 // int width  = (-rect_.x) + rect_.width;
 647                 int width  = rect_.x + rect_.width;
 648                 int height = rect_.y + rect_.height;
 649                 if(maximum_width_  > 0) { width  = std::min(width, maximum_width_); }
 650                 if(maximum_height_ > 0) { height = std::min(height, maximum_height_); }
 651 
 652                 cairo_format_t format = CAIRO_FORMAT_ARGB32;
 653                 const unsigned stride = cairo_format_stride_for_width(format, width);
 654 
 655                 this->create_surface_buffer(stride * height);
 656 
 657                 if (!surface_buffer_.size()) {
 658                         // surface_.assign(nullptr);
 659                         surface_.assign(create_neutral_surface(0, 0));
 660                         return;
 661                 }
 662 
 663                 cairo_surface_t* cairo_surface = cairo_image_surface_create_for_data(&surface_buffer_[0], format, width, height, stride);
 664                 cairo_t* cr = cairo_create(cairo_surface);
 665 
 666                 /* set color (used for foreground). */
 667                 cairo_set_source_rgba(cr,
 668                         foreground_color_.r / 256.0,
 669                         foreground_color_.g / 256.0,
 670                         foreground_color_.b / 256.0,
 671                         foreground_color_.a / 256.0
 672                 );
 673 
 674                 pango_cairo_show_layout(cr, layout_);
 675 
 676                 static_assert(sizeof(Uint32) == 4, "Something is wrong with our typedefs");
 677 
 678                 // The cairo surface is in CAIRO_FORMAT_ARGB32 which uses
 679                 // pre-multiplied alpha. SDL doesn't use that so the pixels need to be
 680                 // decoded again.
 681                 Uint32 * pixels = reinterpret_cast<Uint32 *>(&surface_buffer_[0]);
 682 
 683                 for(int y = 0; y < height; ++y) {
 684                         for(int x = 0; x < width; ++x) {
 685                                 from_cairo_format(pixels[y * width + x]);
 686                         }
 687                 }
 688 
 689                 surface_.assign(SDL_CreateRGBSurfaceFrom(
 690                         &surface_buffer_[0], width, height, 32, stride, 0x00FF0000, 0x0000FF00, 0x000000FF, 0xFF000000));
 691                 cairo_destroy(cr);
 692                 cairo_surface_destroy(cairo_surface);
 693         }
 694 }
 695 
 696 void pango_text::create_surface_buffer(const size_t size) const
 697 {
 698         // clear old buffer
 699         surface_.assign(nullptr);
 700 
 701         surface_buffer_.resize(size);
 702 
 703         // Not sure why this is needed, perhaps SDL assumes it?
 704         for (auto & c : surface_buffer_) { c = 0; }
 705 }

 707 bool pango_text::set_markup(const std::string & text) {
 708         return this->set_markup_helper(link_aware_ ? this->format_link_tokens(text) : text);
 709 }
 710 
 711 std::string pango_text::format_link_tokens(const std::string & text) const {
 712         std::string delim = " \n\r\t";
 713         // Tokenize according to these delimiters, and stream the results of `handle_token` on each token to get the new text.
 714 
 715         std::string result;
 716 
 717         int last_delim = -1;
 718         for (size_t index = 0; index < text.size(); ++index) {
 719                 if (delim.find(text.at(index)) != std::string::npos) {
 720                         // want to include chars from range since last token, dont want to include any delimiters
 721                         result += this->handle_token(text.substr(last_delim + 1, index - last_delim - 1));
 722                         result += text.at(index);
 723                         last_delim = index;
 724                 }
 725         }
 726         if (last_delim < static_cast<int>(text.size()) - 1) {
 727                 result += this->handle_token(text.substr(last_delim + 1, text.size() - last_delim - 1));
 728         }
 729 
 730         return result;
 731 }
 732 
 733 std::string pango_text::handle_token(const std::string & token) const
 734 {
 735         if (looks_like_url(token)) {
 736                 return format_as_link(token, link_color_.to_hex_string());
 737         } else {
 738                 return token;
 739         }
 740 }

 742 bool pango_text::set_markup_helper(const std::string& text)
 743 {
 744         if(pango_parse_markup(text.c_str(), text.size()
 745                         , 0, nullptr, nullptr, nullptr, nullptr)) {
 746 
 747                 /* Markup is valid so set it. */
 748                 pango_layout_set_markup(layout_, text.c_str(), text.size());
 749                 return true;
 750         }
 751 
 752         /*
 753          * The markup is invalid. Try to recover.
 754          *
 755          * The pango engine tested seems to accept stray single quotes ' and
 756          * double quotes ". Stray ampersands & seem to give troubles.
 757          * So only try to recover from broken ampersands, by simply replacing them
 758          * with the escaped version.
 759          */
 760         std::string semi_escaped{semi_escape_text(text)};
 761 
 762         /*
 763          * If at least one ampersand is replaced the semi-escaped string
 764          * is longer than the original. If this isn't the case then the
 765          * markup wasn't (only) broken by ampersands in the first place.
 766          */
 767         if(text.size() == semi_escaped.size()
 768                         || !pango_parse_markup(semi_escaped.c_str(), semi_escaped.size()
 769                                 , 0, nullptr, nullptr, nullptr, nullptr)) {
 770 
 771                 /* Fixing the ampersands didn't work. */
 772                 ERR_GUI_L << "pango_text::" << __func__
 773                                 << " text '" << text
 774                                 << "' has broken markup, set to normal text.\n";
 775 
 776                 this->set_text(_("The text contains invalid markup: ") + text, false);
 777                 return false;
 778         }
 779 
 780         /* Replacement worked, still warn the user about the error. */
 781         WRN_GUI_L << "pango_text::" << __func__
 782                         << " text '" << text
 783                         << "' has unescaped ampersands '&', escaped them.\n";
 784 
 785         pango_layout_set_markup(layout_, semi_escaped.c_str(), semi_escaped.size());
 786         return true;
 787 }
