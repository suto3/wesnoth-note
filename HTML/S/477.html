  79 void select_orb_colors::setup_orb_group(const std::string& base_id, bool& shown, const std::string& initial, window& window, bool connect)
  80 {
  81         std::string prefix = "orb_" + base_id + "_";
  82         toggle_button& toggle = find_widget<toggle_button>(&window, prefix + "show", false);
  83         toggle.set_value_bool(shown);
  84         if(connect) {
  85                 connect_signal_mouse_left_click(toggle, std::bind(
  86                         &select_orb_colors::handle_toggle_click,
  87                         this,
  88                         std::ref(shown)
  89                 ));
  90         }
  91 
  92         grid& selection = find_widget<grid>(&window, prefix + "selection", false);
  93         group<std::string>& group = groups_[base_id];
  94 
  95         using iteration::walker_base;
  96         std::unique_ptr<walker_base> iter(selection.create_walker());
  97         while(!iter->at_end(walker_base::child)) {
  98                 widget* next = iter->get(walker_base::child);
  99                 if(toggle_button* button = dynamic_cast<toggle_button*>(next)) {
 100                         const std::string& id = button->id();
 101                         group.add_member(button, id.substr(prefix.size()));
 102                 }
 103                 iter->next(walker_base::child);
 104         }
 105         group.set_member_states(initial);
 106 }

 114 {

 117         show_moved_ = game_config::show_moved_orb;

 119         show_enemy_ = game_config::show_enemy_orb;
 120 
 121         setup_orb_group("unmoved", show_unmoved_, game_config::colors::unmoved_orb_color, window, false);
 122         setup_orb_group("partial", show_partial_, game_config::colors::partial_orb_color, window, false);
 123         setup_orb_group("moved", show_moved_, game_config::colors::moved_orb_color, window, false);
 124         setup_orb_group("ally", show_ally_, game_config::colors::ally_orb_color, window, false);

 126 }
