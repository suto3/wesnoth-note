 110         std::string get_data_paged(int which_page)
 111         {
 112                 if(size_t(which_page) >= pages.size()) {
 113                         return "";
 114                 }
 115                 return data.substr(pages[which_page].first, pages[which_page].second);
 116         }
 117 
 118         void clear_data()
 119         {
 120                 data.clear();
 121         }
 122 
 123         void set_data(const std::string& new_data)
 124         {
 125                 data = new_data;
 126                 repaginate();
 127         }
 128 
 129         int count_pages()
 130         {
 131                 return std::max<int>(pages.size(), 1);
 132         }
 133 
 134 private:
 135         void repaginate()
 136         {
 137                 pages.clear();
 138                 size_t start = 0;
 139                 while(start < data.size()) {
 140                         size_t end = data.find_last_of('\n', start + max_inspect_win_len);
 141                         if(end == std::string::npos) {
 142                                 end = data.size() - 1;
 143                         }
 144                         int len = end - start + 1;
 145                         pages.emplace_back(start, len);
 146                         start += len;
 147                 }
 148         }
 149         static const unsigned int max_inspect_win_len = 20000;
 150         std::string data;
 151         std::vector<std::pair<size_t,int>> pages;
 152 };

 323         void handle_stuff_list_item_clicked(widget& tree)
 324         {
 325                 tree_view_node* selected = dynamic_cast<tree_view&>(tree).selected_item();
 326                 callbacks[selected->describe_path()](*selected);
 327 
 328                 // We recursively fold, but non-recursively unfold.
 329                 // This is because only one node on a level should be open at any given time.
 330                 // Furthermore, there's no need to remember that a subnode was open once the parent is closed.
 331                 if(!selected->is_root_node()) {
 332                         for(auto& node : selected->parent_node().children()) {
 333                                 if(&node != selected) {
 334                                         node.fold(true);
 335                                 }
 336                         }
 337 
 338                         selected->unfold();
 339                 }
 340 
 341                 view_.update(model_);
 342         }

 367         template<typename T>
 368         std::shared_ptr<T> get_controller()
 369         {
 370                 for(auto& c : controllers) {
 371                         if(std::shared_ptr<T> p = std::dynamic_pointer_cast<T>(c)) {
 372                                 return p;
 373                         }
 374                 }

 780 void team_mode_controller::show_units(tree_view_node&, int side)
 781 {
 782         std::ostringstream s;
 783         for(unit_map::const_iterator i = dc().units().begin(); i != dc().units().end();
 784                 ++i) {
 785                 if(i->side() != side) {
 786                         continue;
 787                 }
 788                 s << '(' << i->get_location() << ") ";
 789                 if(i->can_recruit()) {
 790                         s << "LEADER ";
 791                 }
 792 
 793                 s << "\nid=\"" << i->id() << "\" (" << i->type_id() << ")\n"
 794                   << "L" << i->level() << "; " << i->experience() << '/'
 795                   << i->max_experience() << " XP; " << i->hitpoints() << '/'
 796                   << i->max_hitpoints() << " HP\n";
 797                 for(const auto & str : i->get_traits_list())
 798                 {
 799                         s << "\t" << str << std::endl;
 800                 }
 801                 s << std::endl;
 802         }
 803         model().set_data(s.str());
 804 }
