   9 local function custom_cost(x, y, unit, enemy_map, enemy_attack_map, multiplier)
  10     local terrain = wesnoth.get_terrain(x, y)
  11     local move_cost = wesnoth.unit_movement_cost(unit, terrain)
  12 
  13     move_cost = move_cost + (enemy_map:get(x,y) or 0)
  14     move_cost = move_cost + (enemy_attack_map.units:get(x,y) or 0) * multiplier
  15 
  16     return move_cost
  17 end

  21 function ca_goto:evaluation(cfg, data)
  22     -- If cfg.release_all_units_at_goal is set, check whether the goal has
  23     -- already been reached, in which case we do not do anything
  24     if MAISD.get_mai_self_data(data, cfg.ai_id, "release_all") then
  25         return 0
  26     end
  27 
  28     -- For convenience, we check for locations here, and just pass that to the exec function
  29     -- This is mostly to make the unique_goals option easier
  30     local width, height = wesnoth.get_map_size()
  31     local all_locs = wesnoth.get_locations {
  32         x = '1-' .. width,
  33         y = '1-' .. height,
  34         { "and", H.get_child(cfg, "filter_location") }
  35     }
  36     if (#all_locs == 0) then return 0 end
  37 
  38     -- If 'unique_goals' is set, check whether there are locations left to go to.
  39     -- This does not have to be a persistent variable
  40     local locs = {}
  41     if cfg.unique_goals then
  42         -- First, some cleanup of previous turn data
  43         local str = 'goal_taken_' .. (wesnoth.current.turn - 1)
  44         local old_goals = MAISD.get_mai_self_data(data, cfg.ai_id)
  45         for goal,_ in pairs(old_goals) do
  46             if string.find(goal, str) then
  47                 old_goals[goal] = nil  -- This also removes it from data
  48             end
  49         end
  50 
  51         -- Now on to the current turn
  52         for _,loc in ipairs(all_locs) do
  53             local str = 'goal_taken_' .. wesnoth.current.turn  .. '_' .. loc[1] .. '_' .. loc[2]
  54             if (not MAISD.get_mai_self_data(data, cfg.ai_id, str)) then
  55                 table.insert(locs, loc)
  56             end
  57         end
  58     else
  59         locs = all_locs
  60     end
  61     if (not locs[1]) then return 0 end
  62 
  63     local all_units = AH.get_units_with_moves {
  64         side = wesnoth.current.side,
  65         { "and", H.get_child(cfg, "filter") }
  66     }
  67 
  68     local units = {}
  69     if cfg.release_unit_at_goal then
  70         for _,unit in ipairs(all_units) do
  71             if (not MAIUV.get_mai_unit_variables(unit, cfg.ai_id, "release")) then
  72                 table.insert(units, unit)
  73             end
  74         end
  75     else
  76         units = all_units
  77     end
  78     if (not units[1]) then return 0 end
  79 
  80     -- Now store units and locs, so that we don't need to duplicate this in the exec function
  81     GO_units, GO_locs = units, locs
  82 
  83     return cfg.ca_score
  84 end

  86 function ca_goto:execution(cfg, data)
  87     local units, locs = GO_units, GO_locs
  88 
  89     local enemy_map, enemy_attack_map
  90     if cfg.avoid_enemies then
  91         if (type(cfg.avoid_enemies) ~= 'number') then
  92             H.wml_error("Goto AI avoid_enemies= requires a number as argument")
  93         elseif (cfg.avoid_enemies <= 0) then
  94             H.wml_error("Goto AI avoid_enemies= argument must be >0")
  95         end
  96 
  97         local enemies = AH.get_visible_units(wesnoth.current.side, {
  98             { "filter_side", { { "enemy_of", { side = wesnoth.current.side } } } },
  99         })
 100         local live_enemies = {}
 101         for _,enemy in ipairs(enemies) do
 102             if (not enemy.status.petrified) then
 103                 table.insert(live_enemies, enemy)
 104             end
 105         end
 106 
 107         enemy_map = LS.create()
 108         for _,enemy in ipairs(enemies) do
 109             enemy_map:insert(enemy.x, enemy.y, (enemy_map:get(enemy.x, enemy.y) or 0) + 1000)
 110         end
 111         for _,enemy in ipairs(live_enemies) do
 112             for xa,ya in H.adjacent_tiles(enemy.x, enemy.y) do
 113                 enemy_map:insert(xa, ya, (enemy_map:get(xa, ya) or 0) + 10)
 114             end
 115         end
 116 
 117         enemy_attack_map = BC.get_attack_map(live_enemies)
 118     end
 119 
 120     local max_rating, closest_hex, best_path, best_unit = -9e99
 121     for _,unit in ipairs(units) do
 122         for _,loc in ipairs(locs) do
 123             -- If cfg.use_straight_line is set, we simply find the closest
 124             -- hex to the goal that the unit can get to
 125             if cfg.use_straight_line then
 126                 local hex, _, rating = AH.find_best_move(unit, function(x, y)
 127                     local r = -M.distance_between(x, y, loc[1], loc[2])
 128                     -- Also add distance from unit as very small rating component
 129                     -- This is mostly here to keep unit in place when no better hexes are available
 130                     r = r - M.distance_between(x, y, unit.x, unit.y) / 1000.
 131                     return r
 132                 end, { no_random = true })
 133 
 134                 if (rating > max_rating) then
 135                     max_rating = rating
 136                     closest_hex, best_unit = hex, unit
 137                 end
 138             else  -- Otherwise find the best path to take
 139                 local path, cost
 140                 if cfg.avoid_enemies then
 141                     path, cost = wesnoth.find_path(unit, loc[1], loc[2],
 142                         function(x, y, current_cost)
 143                             return custom_cost(x, y, unit, enemy_map, enemy_attack_map, cfg.avoid_enemies)
 144                         end
 145                     )
 146                 else
 147                     local enemy_at_goal
 148                     if cfg.ignore_enemy_at_goal then
 149                         enemy_at_goal = wesnoth.get_unit(loc[1], loc[2])
 150                         if enemy_at_goal and wesnoth.is_enemy(wesnoth.current.side, enemy_at_goal.side) then
 151                              wesnoth.extract_unit(enemy_at_goal)
 152                         else
 153                             enemy_at_goal = nil
 154                         end
 155                     end
 156                     path, cost = AH.find_path_with_shroud(unit, loc[1], loc[2], { ignore_units = cfg.ignore_units })
 157                     if enemy_at_goal then
 158                         wesnoth.put_unit(enemy_at_goal)
 159                         --- Give massive penalty for this goal hex
 160                         cost = cost + 100
 161                     end
 162                 end
 163 
 164                 -- Make all hexes within the unit's current MP equivalent
 165                 if (cost <= unit.moves) then cost = 0 end
 166 
 167                 local rating = - cost
 168 
 169                 -- Add a small penalty for hexes occupied by an allied unit
 170                 local unit_in_way = wesnoth.get_unit(loc[1], loc[2])
 171                 if unit_in_way and (unit_in_way ~= unit) then
 172                     rating = rating - 0.01
 173                 end
 174 
 175                 if (rating > max_rating) then
 176                     max_rating = rating
 177                     closest_hex, best_unit, best_path = loc, unit, path
 178                 end
 179             end
 180         end
 181     end
 182 
 183     -- If 'unique_goals' is set, mark this location as being taken
 184     if cfg.unique_goals then
 185         local str = 'goal_taken_' .. wesnoth.current.turn  .. '_' .. closest_hex[1] .. '_' .. closest_hex[2]
 186         local tmp_table = {}
 187         tmp_table[str] = true
 188         MAISD.insert_mai_self_data(data, cfg.ai_id, tmp_table)
 189     end
 190 
 191     -- If any of the non-standard path finding options were used,
 192     -- we need to pick the farthest reachable hex along that path
 193     -- For simplicity, we do it for all types of pathfinding here,
 194     -- rather than using ai_helper.next_hop for standard pathfinding
 195     -- Also, straight-line does not produce a path, so we do that first
 196     if not best_path then
 197         best_path = AH.find_path_with_shroud(best_unit, closest_hex[1], closest_hex[2])
 198     end
 199 
 200     -- Now go through the hexes along that path, use normal path finding
 201     closest_hex = best_path[1]
 202     for i = 2,#best_path do
 203         local sub_path, sub_cost = AH.find_path_with_shroud(best_unit, best_path[i][1], best_path[i][2], cfg)
 204         if sub_cost <= best_unit.moves then
 205             local unit_in_way = wesnoth.get_unit(best_path[i][1], best_path[i][2])
 206             if (not AH.is_visible_unit(wesnoth.current.side, unit_in_way)) then
 207                 closest_hex = best_path[i]
 208             end
 209         else
 210             break
 211         end
 212     end
 213 
 214     if closest_hex then
 215         AH.checked_move_full(ai, best_unit, closest_hex[1], closest_hex[2])
 216     else
 217         AH.checked_stopunit_moves(ai, best_unit)
 218     end
 219 
 220     if (not best_unit) or (not best_unit.valid) then return end
 221 
 222     -- If release_unit_at_goal= or release_all_units_at_goal= key is set:
 223     -- Check if the unit made it to one of the goal hexes
 224     -- This needs to be done for the original goal hexes, not by checking the SLF again,
 225     -- as that might have changed based on the new situation on the map
 226     if cfg.release_unit_at_goal or cfg.release_all_units_at_goal then
 227         local is_unit_at_goal = false
 228         for _,loc in ipairs(locs) do
 229             if (best_unit.x == loc[1]) and (best_unit.y == loc[2]) then
 230                 is_unit_at_goal = true
 231                 break
 232             end
 233         end
 234 
 235         if is_unit_at_goal then
 236             if cfg.release_unit_at_goal then
 237                 MAIUV.set_mai_unit_variables(best_unit, cfg.ai_id, { release = true })
 238             end
 239 
 240             if cfg.release_all_units_at_goal then
 241                 MAISD.insert_mai_self_data(data, cfg.ai_id, { release_all = true })
 242             end
 243         end
 244     end
 245 
 246     GO_units, GO_locs = nil, nil
 247 end
 248 
 249 return ca_goto
