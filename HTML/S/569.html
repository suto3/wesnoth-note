 117 /**
 118  * The interval between draw events.
 119  *
 120  * When the window is shown this value is set, the callback function always
 121  * uses this value instead of the parameter send, that way the window can stop
 122  * drawing when it wants.
 123  */
 124 static int draw_interval = 0;
 125 
 126 /**
 127  * SDL_AddTimer() callback for the draw event.
 128  *
 129  * When this callback is called it pushes a new draw event in the event queue.
 130  *
 131  * @returns                       The new timer interval, 0 to stop.
 132  */
 133 static Uint32 draw_timer(Uint32, void*)
 134 {
 135         //      DBG_GUI_E << "Pushing draw event in queue.\n";
 136 
 137         SDL_Event event;
 138         SDL_UserEvent data;
 139 
 140         data.type = DRAW_EVENT;
 141         data.code = 0;
 142         data.data1 = nullptr;
 143         data.data2 = nullptr;
 144 
 145         event.type = DRAW_EVENT;
 146         event.user = data;
 147 
 148         SDL_PushEvent(&event);
 149         return draw_interval;
 150 }
 151 
 152 /**
 153  * SDL_AddTimer() callback for delay_event.
 154  *
 155  * @param event                   The event to push in the event queue.
 156  *
 157  * @return                        The new timer interval (always 0).
 158  */
 159 static Uint32 delay_event_callback(const Uint32, void* event)
 160 {
 161         SDL_PushEvent(static_cast<SDL_Event*>(event));
 162         delete static_cast<SDL_Event*>(event);
 163         return 0;
 164 }
 165 
 166 /**
 167  * Allows an event to be delayed a certain amount of time.
 168  *
 169  * @note the delay is the minimum time, after the time has passed the event
 170  * will be pushed in the SDL event queue, so it might delay more.
 171  *
 172  * @param event                   The event to delay.
 173  * @param delay                   The number of ms to delay the event.
 174  */
 175 static void delay_event(const SDL_Event& event, const Uint32 delay)
 176 {
 177         SDL_AddTimer(delay, delay_event_callback, new SDL_Event(event));
 178 }

 421 window::~window()
 422 {
 423         /*
 424          * We need to delete our children here instead of waiting for the grid to
 425          * automatically do it. The reason is when the grid deletes its children
 426          * they will try to unregister them self from the linked widget list. At
 427          * this point the member of window are destroyed and we enter UB. (For
 428          * some reason the bug didn't trigger on g++ but it does on MSVC.
 429          */
 430         for(unsigned row = 0; row < get_grid().get_rows(); ++row) {
 431                 for(unsigned col = 0; col < get_grid().get_cols(); ++col) {
 432                         get_grid().remove_child(row, col);
 433                 }
 434         }
 435 
 436         /*
 437          * The tip needs to be closed if the window closes and the window is
 438          * not a tip. If we don't do that the tip will unrender in the next
 439          * window and cause drawing glitches.
 440          * Another issue is that on smallgui and an MP game the tooltip not
 441          * unrendered properly can capture the mouse and make playing impossible.
 442          */
 443         if(show_mode_ == modal) {
 444                 dialogs::tip::remove();
 445         }
 446 
 447         manager::instance().remove(*this);
 448 
 449 #ifdef DEBUG_WINDOW_LAYOUT_GRAPHS
 450 
 451         delete debug_layout_;
 452 
 453 #endif
 454 }

 461 void window::update_screen_size()
 462 {
 463         // Only if we're the toplevel window we need to update the size, otherwise
 464         // it's done in the resize event.
 465         if(draw_interval == 0) {
 466                 const SDL_Rect rect = screen_area();
 467                 settings::screen_width = rect.w;
 468                 settings::screen_height = rect.h;
 469 
 470                 settings::gamemap_width = settings::screen_width;
 471                 settings::gamemap_height = settings::screen_height;
 472 
 473                 display* display = display::get_singleton();
 474                 if(display) {
 475                         const SDL_Rect rect_gm = display->map_outside_area();
 476 
 477                         if(rect_gm.w && rect_gm.h) {
 478                                 settings::gamemap_width = rect_gm.w;
 479                                 settings::gamemap_height = rect_gm.h;
 480                                 settings::gamemap_x_offset = rect_gm.x;
 481                         }
 482                 }
 483         }
 484 }

 553 int window::show(const bool restore, const unsigned auto_close_timeout)
 554 {
 555         /*
 556          * Removes the old tip if one shown. The show_tip doesn't remove
 557          * the tip, since it's the tip.
 558          */
 559         dialogs::tip::remove();
 560 
 561         show_mode_ = modal;
 562         restore_ = restore;
 563 
 564         /**
 565          * Helper class to set and restore the drawing interval.
 566          *
 567          * We need to make sure we restore the value when the function ends, be it
 568          * normally or due to an exception.
 569          */
 570         class draw_interval_setter
 571         {
 572         public:
 573                 draw_interval_setter() : interval_(draw_interval)
 574                 {
 575                         if(interval_ == 0) {
 576                                 draw_interval = 20;
 577                                 SDL_AddTimer(draw_interval, draw_timer, nullptr);
 578 
 579                                 // There might be some time between creation and showing so
 580                                 // reupdate the sizes.
 581                                 update_screen_size();
 582                         }
 583                 }
 584 
 585                 ~draw_interval_setter()
 586                 {
 587                         draw_interval = interval_;
 588                 }
 589 
 590         private:
 591                 int interval_;
 592         };
 593 
 594         log_scope2(log_gui_draw, LOG_SCOPE_HEADER);
 595 
 596         generate_dot_file("show", SHOW);
 597 
 598         assert(status_ == NEW);
 599 
 600         draw_interval_setter draw_interval_setter;
 601 
 602         /*
 603          * Before show has been called, some functions might have done some testing
 604          * on the window and called layout, which can give glitches. So
 605          * reinvalidate the window to avoid those glitches.
 606          */
 607         invalidate_layout();
 608         suspend_drawing_ = false;
 609 
 610         if(auto_close_timeout) {
 611                 // Make sure we're drawn before we try to close ourselves, which can
 612                 // happen if the timeout is small.
 613                 draw();
 614 
 615                 SDL_Event event;
 616                 SDL_UserEvent data;
 617 
 618                 data.type = CLOSE_WINDOW_EVENT;
 619                 data.code = manager::instance().get_id(*this);
 620                 data.data1 = nullptr;
 621                 data.data2 = nullptr;
 622 
 623                 event.type = CLOSE_WINDOW_EVENT;
 624                 event.user = data;
 625 
 626                 delay_event(event, auto_close_timeout);
 627         }
 628 
 629 
 630         try
 631         {
 632                 // Start our loop drawing will happen here as well.
 633                 bool mouse_button_state_initialized = false;
 634                 for(status_ = SHOWING; status_ != CLOSED;) {
 635                         // process installed callback if valid, to allow e.g. network
 636                         // polling
 637                         events::pump();
 638 
 639                         if(!mouse_button_state_initialized) {
 640                                 /*
 641                                  * The state must be initialize when showing the dialogue.
 642                                  * However when initialized before this point there were random
 643                                  * errors. This only happened when the 'click' was done fast; a
 644                                  * slower click worked properly.
 645                                  *
 646                                  * So it seems the events need to be processed before SDL can
 647                                  * return the proper button state. When initializing here all
 648                                  * works fine.
 649                                  */
 650                                 mouse_button_state_ = SDL_GetMouseState(nullptr, nullptr);
 651                                 mouse_button_state_initialized = true;
 652                         }
 653 
 654                         if(status_ == REQUEST_CLOSE) {
 655                                 status_ = exit_hook_(*this) ? CLOSED : SHOWING;
 656                         }
 657 
 658                         // Add a delay so we don't keep spinning if there's no event.
 659                         SDL_Delay(10);
 660                 }
 661         }
 662         catch(...)
 663         {
 664                 /**
 665                  * @todo Clean up the code duplication.
 666                  *
 667                  * In the future the restoring shouldn't be needed so the duplication
 668                  * doesn't hurt too much but keep this todo as a reminder.
 669                  */
 670                 suspend_drawing_ = true;
 671 
 672                 // restore area
 673                 if(restore_) {
 674                         SDL_Rect rect = get_rectangle();
 675                         sdl_blit(restorer_, 0, video_.getSurface(), &rect);
 676                         font::undraw_floating_labels(video_.getSurface());
 677                 }
 678                 throw;
 679         }
 680 
 681         suspend_drawing_ = true;
 682 
 683         // restore area
 684         if(restore_) {
 685                 SDL_Rect rect = get_rectangle();
 686                 sdl_blit(restorer_, 0, video_.getSurface(), &rect);
 687                 font::undraw_floating_labels(video_.getSurface());
 688         }
 689 
 690         return retval_;
 691 }

 693 void window::draw()
 694 {
 695         /***** ***** ***** ***** Init ***** ***** ***** *****/
 696         // Prohibited from drawing?
 697         if(suspend_drawing_) {
 698                 return;
 699         }
 700 
 701         surface& frame_buffer = video_.getSurface();
 702 
 703         /***** ***** Layout and get dirty list ***** *****/
 704         if(need_layout_) {
 705                 // Restore old surface. In the future this phase will not be needed
 706                 // since all will be redrawn when needed with dirty rects. Since that
 707                 // doesn't work yet we need to undraw the window.
 708                 if(restore_ && restorer_) {
 709                         SDL_Rect rect = get_rectangle();
 710                         sdl_blit(restorer_, 0, frame_buffer, &rect);
 711                 }
 712 
 713                 layout();
 714 
 715                 // Get new surface for restoring
 716                 SDL_Rect rect = get_rectangle();
 717 
 718                 // We want the labels underneath the window so draw them and use them
 719                 // as restore point.
 720                 if(is_toplevel_) {
 721                         font::draw_floating_labels(frame_buffer);
 722                 }
 723 
 724                 if(restore_) {
 725                         restorer_ = get_surface_portion(frame_buffer, rect);
 726                 }
 727 
 728                 // Need full redraw so only set ourselves dirty.
 729                 dirty_list_.emplace_back(1, this);
 730         } else {
 731 
 732                 // Let widgets update themselves, which might dirty some things.
 733                 layout_children();
 734 
 735                 // Now find the widgets that are dirty.
 736                 std::vector<widget*> call_stack;
 737                 if(!new_widgets) {
 738                         populate_dirty_list(*this, call_stack);
 739                 } else {
 740                         /* Force to update and redraw the entire screen */
 741                         dirty_list_.clear();
 742                         dirty_list_.emplace_back(1, this);
 743                 }
 744         }
 745 
 746         if (dirty_list_.empty()) {
 747                 return;
 748         }
 749 
 750         for(auto & item : dirty_list_)
 751         {
 752 
 753                 assert(!item.empty());
 754 
 755                 const SDL_Rect dirty_rect
 756                                 = new_widgets ? screen_area()
 757                                                           : item.back()->get_dirty_rectangle();
 758 
 759 // For testing we disable the clipping rect and force the entire screen to
 760 // update. This way an item rendered at the wrong place is directly visible.
 761 #if 0
 762                 dirty_list_.clear();
 763                 dirty_list_.emplace_back(1, this);
 764 #else
 765                 clip_rect_setter clip(frame_buffer, &dirty_rect);
 766 #endif
 767 
 768                 /*
 769                  * The actual update routine does the following:
 770                  * - Restore the background.
 771                  *
 772                  * - draw [begin, end) the back ground of all widgets.
 773                  *
 774                  * - draw the children of the last item in the list, if this item is
 775                  *   a container it's children get a full redraw. If it's not a
 776                  *   container nothing happens.
 777                  *
 778                  * - draw [rbegin, rend) the fore ground of all widgets. For items
 779                  *   which have two layers eg window or panel it draws the foreground
 780                  *   layer. For other widgets it's a nop.
 781                  *
 782                  * Before drawing there needs to be determined whether a dirty widget
 783                  * really needs to be redrawn. If the widget doesn't need to be
 784                  * redrawing either being not visibility::visible or has status
 785                  * widget::redraw_action::none. If it's not drawn it's still set not
 786                  * dirty to avoid it keep getting on the dirty list.
 787                  */
 788 
 789                 for(std::vector<widget*>::iterator itor = item.begin();
 790                         itor != item.end();
 791                         ++itor) {
 792 
 793                         if((**itor).get_visible() != widget::visibility::visible
 794                            || (**itor).get_drawing_action()
 795                                   == widget::redraw_action::none) {
 796 
 797                                 for(std::vector<widget*>::iterator citor = itor;
 798                                         citor != item.end();
 799                                         ++citor) {
 800 
 801                                         (**citor).set_is_dirty(false);
 802                                 }
 803 
 804                                 item.erase(itor, item.end());
 805                                 break;
 806                         }
 807                 }
 808 
 809                 // Restore.
 810                 if(restore_) {
 811                         SDL_Rect rect = get_rectangle();
 812                         sdl_blit(restorer_, 0, frame_buffer, &rect);
 813                 }
 814 
 815                 // Background.
 816                 for(std::vector<widget*>::iterator itor = item.begin();
 817                         itor != item.end();
 818                         ++itor) {
 819 
 820                         (**itor).draw_background(frame_buffer, 0, 0);
 821                 }
 822 
 823                 // Children.
 824                 if(!item.empty()) {
 825                         item.back()->draw_children(frame_buffer, 0, 0);
 826                 }
 827 
 828                 // Foreground.
 829                 for(std::vector<widget*>::reverse_iterator ritor = item.rbegin();
 830                         ritor != item.rend();
 831                         ++ritor) {
 832 
 833                         (**ritor).draw_foreground(frame_buffer, 0, 0);
 834                         (**ritor).set_is_dirty(false);
 835                 }
 836         }
 837 
 838         dirty_list_.clear();
 839 
 840         std::vector<widget*> call_stack;
 841         populate_dirty_list(*this, call_stack);
 842         assert(dirty_list_.empty());
 843 }

 945 void window::layout()
 946 {
 947         /***** Initialize. *****/
 948 
 949         std::shared_ptr<const window_definition::resolution>
 950         conf = std::static_pointer_cast<const window_definition::resolution>(
 951                         config());
 952         assert(conf);
 953 
 954         log_scope2(log_gui_layout, LOG_SCOPE_HEADER);
 955 
 956         const point mouse = get_mouse_position();
 957         variables_.add("mouse_x", wfl::variant(mouse.x));
 958         variables_.add("mouse_y", wfl::variant(mouse.y));
 959         variables_.add("window_width", wfl::variant(0));
 960         variables_.add("window_height", wfl::variant(0));
 961         variables_.add("size_request_mode", wfl::variant("maximum"));
 962         get_screen_size_variables(variables_);
 963 
 964         const int maximum_width = automatic_placement_ ? maximum_width_
 965                         ? std::min(maximum_width_, settings::screen_width)
 966                         : settings::screen_width
 967                         : w_(variables_, &functions_);
 968 
 969         const int maximum_height = automatic_placement_ ? maximum_height_
 970                         ? std::min(maximum_height_, settings::screen_height)
 971                         : settings::screen_height
 972                         : h_(variables_, &functions_);
 973 
 974         /***** Handle click dismiss status. *****/
 975         button* click_dismiss_button = nullptr;
 976         if((click_dismiss_button
 977                 = find_widget<button>(this, "click_dismiss", false, false))) {
 978 
 979                 click_dismiss_button->set_visible(widget::visibility::invisible);
 980         }
 981         if(click_dismiss_) {
 982                 button* btn = find_widget<button>(this, "ok", false, false);
 983                 if(btn) {
 984                         btn->set_visible(widget::visibility::invisible);
 985                         click_dismiss_button = btn;
 986                 }
 987                 VALIDATE(click_dismiss_button,
 988                                  _("Click dismiss needs a 'click_dismiss' or 'ok' button."));
 989         }
 990 
 991         /***** Layout. *****/
 992         layout_initialize(true);
 993         generate_dot_file("layout_initialize", LAYOUT);
 994 
 995         layout_linked_widgets();
 996 
 997         try
 998         {
 999                 window_implementation::layout(*this, maximum_width, maximum_height);
1000         }
1001         catch(layout_exception_resize_failed&)
1002         {
1003 
1004                 /** @todo implement the scrollbars on the window. */
1005 
1006                 std::stringstream sstr;
1007                 sstr << __FILE__ << ":" << __LINE__ << " in function '" << __func__
1008                          << "' found the following problem: Failed to size window;"
1009                          << " wanted size " << get_best_size() << " available size "
1010                          << maximum_width << ',' << maximum_height << " screen size "
1011                          << settings::screen_width << ',' << settings::screen_height << '.';
1012 
1013                 throw wml_exception(_("Failed to show a dialog, "
1014                                                            "which doesn't fit on the screen."),
1015                                                          sstr.str());
1016         }
1017 
1018         /****** Validate click dismiss status. *****/
1019         if(click_dismiss_ && disable_click_dismiss()) {
1020                 assert(click_dismiss_button);
1021                 click_dismiss_button->set_visible(widget::visibility::visible);
1022 
1023                 connect_signal_mouse_left_click(
1024                                 *click_dismiss_button,
1025                                 std::bind(&window::set_retval, this, OK, true));
1026 
1027                 layout_initialize(true);
1028                 generate_dot_file("layout_initialize", LAYOUT);
1029 
1030                 layout_linked_widgets();
1031 
1032                 try
1033                 {
1034                         window_implementation::layout(
1035                                         *this, maximum_width, maximum_height);
1036                 }
1037                 catch(layout_exception_resize_failed&)
1038                 {
1039 
1040                         /** @todo implement the scrollbars on the window. */
1041 
1042                         std::stringstream sstr;
1043                         sstr << __FILE__ << ":" << __LINE__ << " in function '" << __func__
1044                                  << "' found the following problem: Failed to size window;"
1045                                  << " wanted size " << get_best_size() << " available size "
1046                                  << maximum_width << ',' << maximum_height << " screen size "
1047                                  << settings::screen_width << ',' << settings::screen_height
1048                                  << '.';
1049 
1050                         throw wml_exception(_("Failed to show a dialog, "
1051                                                                    "which doesn't fit on the screen."),
1052                                                                  sstr.str());
1053                 }
1054         }
1055 
1056         /***** Get the best location for the window *****/
1057         point size = get_best_size();
1058 
1059         assert(size.x <= maximum_width && size.y <= maximum_height);
1060 
1061         point origin(0, 0);
1062 
1063         if(automatic_placement_) {
1064 
1065                 switch(horizontal_placement_) {
1066                         case grid::HORIZONTAL_ALIGN_LEFT:
1067                                 // Do nothing
1068                                 break;
1069                         case grid::HORIZONTAL_ALIGN_CENTER:
1070                                 origin.x = (settings::screen_width - size.x) / 2;
1071                                 break;
1072                         case grid::HORIZONTAL_ALIGN_RIGHT:
1073                                 origin.x = settings::screen_width - size.x;
1074                                 break;
1075                         default:
1076                                 assert(false);
1077                 }
1078                 switch(vertical_placement_) {
1079                         case grid::VERTICAL_ALIGN_TOP:
1080                                 // Do nothing
1081                                 break;
1082                         case grid::VERTICAL_ALIGN_CENTER:
1083                                 origin.y = (settings::screen_height - size.y) / 2;
1084                                 break;
1085                         case grid::VERTICAL_ALIGN_BOTTOM:
1086                                 origin.y = settings::screen_height - size.y;
1087                                 break;
1088                         default:
1089                                 assert(false);
1090                 }
1091         } else {
1092 
1093                 variables_.add("window_width", wfl::variant(size.x));
1094                 variables_.add("window_height", wfl::variant(size.y));
1095 
1096                 while(reevaluate_best_size_(variables_, &functions_)) {
1097                         layout_initialize(true);
1098 
1099                         window_implementation::layout(*this,
1100                                                                                    w_(variables_, &functions_),
1101                                                                                    h_(variables_, &functions_));
1102 
1103                         size = get_best_size();
1104                         variables_.add("window_width", wfl::variant(size.x));
1105                         variables_.add("window_height", wfl::variant(size.y));
1106                 }
1107 
1108                 variables_.add("size_request_mode", wfl::variant("size"));
1109 
1110                 size.x = w_(variables_, &functions_);
1111                 size.y = h_(variables_, &functions_);
1112 
1113                 variables_.add("window_width", wfl::variant(size.x));
1114                 variables_.add("window_height", wfl::variant(size.y));
1115 
1116                 origin.x = x_(variables_, &functions_);
1117                 origin.y = y_(variables_, &functions_);
1118         }
1119 
1120         /***** Set the window size *****/
1121         place(origin, size);
1122 
1123         generate_dot_file("layout_finished", LAYOUT);
1124         need_layout_ = false;
1125 
1126         event::init_mouse_location();
1127 }

1154                         linked_size.second.height = max_size.y;
1155                 }

1158                 for(auto widget : linked_size.second.widgets)
1159                 {
1160 
1161                         point size = widget->get_best_size();
1162 
1163                         if(linked_size.second.width != -1) {
1164                                 size.x = max_size.x;
1165                         }

1167                                 size.y = max_size.y;

1248                                                                         const unsigned maximum_height)
1249 {
1250         log_scope2(log_gui_layout, LOG_IMPL_SCOPE_HEADER);
1251 
1252         /*
1253          * For now we return the status, need to test later whether this can
1254          * entirely be converted to an exception based system as in 'promised' on
1255          * the algorithm page.
1256          */
1257 
1258         try
1259         {
1260                 point size = window.get_best_size();
1261 
1262                 DBG_GUI_L << LOG_IMPL_HEADER << " best size : " << size
1263                                   << " maximum size : " << maximum_width << ','
1264                                   << maximum_height << ".\n";
1265                 if(size.x <= static_cast<int>(maximum_width)
1266                    && size.y <= static_cast<int>(maximum_height)) {
1267 
1268                         DBG_GUI_L << LOG_IMPL_HEADER << " Result: Fits, nothing to do.\n";
1269                         return;
1270                 }
1271 
1272                 if(size.x > static_cast<int>(maximum_width)) {
1273                         window.reduce_width(maximum_width);
1274 
1275                         size = window.get_best_size();
1276                         if(size.x > static_cast<int>(maximum_width)) {
1277                                 DBG_GUI_L << LOG_IMPL_HEADER << " Result: Resize width failed."
1278                                                   << " Wanted width " << maximum_width
1279                                                   << " resulting width " << size.x << ".\n";
1280                                 throw layout_exception_width_resize_failed();
1281                         }
1282                         DBG_GUI_L << LOG_IMPL_HEADER
1283                                           << " Status: Resize width succeeded.\n";
1284                 }
1285 
1286                 if(size.y > static_cast<int>(maximum_height)) {
1287                         window.reduce_height(maximum_height);
1288 
1289                         size = window.get_best_size();
1290                         if(size.y > static_cast<int>(maximum_height)) {
1291                                 DBG_GUI_L << LOG_IMPL_HEADER << " Result: Resize height failed."
1292                                                   << " Wanted height " << maximum_height
1293                                                   << " resulting height " << size.y << ".\n";
1294                                 throw layout_exception_height_resize_failed();
1295                         }
1296                         DBG_GUI_L << LOG_IMPL_HEADER
1297                                           << " Status: Resize height succeeded.\n";
1298                 }
1299 
1300                 assert(size.x <= static_cast<int>(maximum_width)
1301                            && size.y <= static_cast<int>(maximum_height));
1302 
1303 
1304                 DBG_GUI_L << LOG_IMPL_HEADER << " Result: Resizing succeeded.\n";
1305                 return;
1306         }
1307         catch(layout_exception_width_modified&)
1308         {
1309                 DBG_GUI_L << LOG_IMPL_HEADER
1310                                   << " Status: Width has been modified, rerun.\n";
1311 
1312                 window.layout_initialize(false);
1313                 window.layout_linked_widgets();
1314                 layout(window, maximum_width, maximum_height);
1315                 return;
1316         }

1419                         } else {

1424                                         if(iter == tab_order.end()) {

1427                                 }
1428                         }
1429                 } while(!is_active(*iter));

1525 /**
1526  * @page layout_algorithm Layout algorithm
1527  *
1528  * @section introduction Introduction
1529  *
1530  * This page describes how the layout engine for the dialogs works. First
1531  * a global overview of some terms used in this document.
1532  *
1533  * - @ref gui2::widget "Widget"; Any item which can be used in the widget
1534  *   toolkit. Not all widgets are visible. In general widgets cannot be
1535  *   sized directly, but this is controlled by a window. A widget has an
1536  *   internal size cache and if the value in the cache is not equal to 0,0
1537  *   that value is its best size. This value gets set when the widget can
1538  *   honor a resize request.  It will be set with the value which honors
1539  *   the request.
1540  *
1541  * - @ref gui2::grid "Grid"; A grid is an invisible container which holds
1542  *   one or more widgets.  Several widgets have a grid in them to hold
1543  *   multiple widgets eg panels and windows.
1544  *
1545  * - @ref gui2::grid::child "Grid cell"; Every widget which is in a grid is
1546  *   put in a grid cell. These cells also hold the information about the gaps
1547  *   between widgets the behavior on growing etc. All grid cells must have a
1548  *   widget inside them.
1549  *
1550  * - @ref gui2::window "Window"; A window is a top level item which has a
1551  *   grid with its children. The window handles the sizing of the window and
1552  *   makes sure everything fits.
1553  *
1554  * - @ref gui2::window::linked_size "Shared size group"; A shared size
1555  *   group is a number of widgets which share width and or height. These
1556  *   widgets are handled separately in the layout algorithm. All grid cells
1557  *   width such a widget will get the same height and or width and these
1558  *   widgets won't be resized when there's not enough size. To be sure that
1559  *   these widgets don't cause trouble for the layout algorithm, they must be
1560  *   in a container with scrollbars so there will always be a way to properly
1561  *   layout them. The engine must enforce this restriction so the shared
1562  *   layout property must be set by the engine after validation.
1563  *
1564  * - All visible grid cells; A grid cell is visible when the widget inside
1565  *   of it doesn't have the state visibility::invisible. Widgets which have the
1566  *   state @ref visibility::hidden are sized properly since when they become
1567  *   @ref visibility::visible the layout shouldn't be invalidated. A grid cell
1568  *   that's invisible has size 0,0.
1569  *
1570  * - All resizable grid cells; A grid cell is resizable under the following
1571  *   conditions:
1572  *   - The widget is visibility::visible.
1573  *   - The widget is not in a shared size group.
1574  *
1575  * There are two layout algorithms with a different purpose.
1576  *
1577  * - The Window algorithm; this algorithm's goal is it to make sure all grid
1578  *   cells fit in the window. Sizing the grid cells depends on the widget
1579  *   size as well, but this algorithm only sizes the grid cells and doesn't
1580  *   handle the widgets inside them.
1581  *
1582  * - The Grid algorithm; after the Window algorithm made sure that all grid
1583  *   cells fit this algorithm makes sure the widgets are put in the optimal
1584  *   state in their grid cell.
1585  *
1586  * @section layout_algorithm_window Window
1587  *
1588  * Here is the algorithm used to layout the window:
1589  *
1590  * - Perform a full initialization
1591  *   (@ref gui2::widget::layout_initialize (full_initialization = true)):
1592  *   - Clear the internal best size cache for all widgets.
1593  *   - For widgets with scrollbars hide them unless the
1594  *     @ref gui2::scrollbar_container::scrollbar_mode "scrollbar_mode" is
1595  *     ALWAYS_VISIBLE or AUTO_VISIBLE.
1596  * - Handle shared sizes:
1597  *   - Height and width:
1598  *     - Get the best size for all widgets that share height and width.
1599  *     - Set the maximum of width and height as best size for all these
1600  *       widgets.
1601  *   - Width only:
1602  *     - Get the best width for all widgets which share their width.
1603  *     - Set the maximum width for all widgets, but keep their own height.
1604  *   - Height only:
1605  *     - Get the best height for all widgets which share their height.
1606  *     - Set the maximum height for all widgets, but keep their own width.
1607  * - Start layout loop:
1608  *   - Get best size.
1609  *   - If width <= maximum_width && height <= maximum_height we're done.
1610  *   - If width > maximum_width, optimize the width:
1611  *     - For every grid cell in a grid row there will be a resize request
1612  *       (@ref gui2::grid::reduce_width):
1613  *       - Sort the widgets in the row on the resize priority.
1614  *         - Loop through this priority queue until the row fits
1615  *           - If priority != 0 try to share the extra width else all
1616  *             widgets are tried to reduce the full size.
1617  *           - Try to shrink the widgets by either wrapping or using a
1618  *             scrollbar (@ref gui2::widget::request_reduce_width).
1619  *           - If the row fits in the wanted width this row is done.
1620  *           - Else try the next priority.
1621  *         - All priorities done and the width still doesn't fit.
1622  *         - Loop through this priority queue until the row fits.
1623  *           - If priority != 0:
1624  *             - try to share the extra width
1625  *           -Else:
1626  *             - All widgets are tried to reduce the full size.
1627  *           - Try to shrink the widgets by sizing them smaller as really
1628  *             wanted (@ref gui2::widget::demand_reduce_width).
1629  *             For labels, buttons etc. they get ellipsized.
1630  *           - If the row fits in the wanted width this row is done.
1631  *           - Else try the next priority.
1632  *         - All priorities done and the width still doesn't fit.
1633  *         - Throw a layout width doesn't fit exception.
1634  *   - If height > maximum_height, optimize the height
1635  *       (@ref gui2::grid::reduce_height):
1636  *     - For every grid cell in a grid column there will be a resize request:
1637  *       - Sort the widgets in the column on the resize priority.
1638  *         - Loop through this priority queue until the column fits:
1639  *           - If priority != 0 try to share the extra height else all
1640  *              widgets are tried to reduce the full size.
1641  *           - Try to shrink the widgets by using a scrollbar
1642  *             (@ref gui2::widget::request_reduce_height).
1643  *             - If succeeded for a widget the width is influenced and the
1644  *               width might be invalid.
1645  *             - Throw a width modified exception.
1646  *           - If the column fits in the wanted height this column is done.
1647  *           - Else try the next priority.
1648  *         - All priorities done and the height still doesn't fit.
1649  *         - Loop through this priority queue until the column fits.
1650  *           - If priority != 0 try to share the extra height else all
1651  *             widgets are tried to reduce the full size.
1652  *           - Try to shrink the widgets by sizing them smaller as really
1653  *             wanted (@ref gui2::widget::demand_reduce_width).
1654  *             For labels, buttons etc. they get ellipsized .
1655  *           - If the column fits in the wanted height this column is done.
1656  *           - Else try the next priority.
1657  *         - All priorities done and the height still doesn't fit.
1658  *         - Throw a layout height doesn't fit exception.
1659  * - End layout loop.
1660  *
1661  * - Catch @ref gui2::layout_exception_width_modified "width modified":
1662  *   - Goto relayout.
1663  *
1664  * - Catch
1665  *   @ref gui2::layout_exception_width_resize_failed "width resize failed":
1666  *   - If the window has a horizontal scrollbar which isn't shown but can be
1667  *     shown.
1668  *     - Show the scrollbar.
1669  *     - goto relayout.
1670  *   - Else show a layout failure message.
1671  *
1672  * - Catch
1673  *   @ref gui2::layout_exception_height_resize_failed "height resize failed":
1674  *   - If the window has a vertical scrollbar which isn't shown but can be
1675  *     shown:
1676  *     - Show the scrollbar.
1677  *     - goto relayout.
1678  *   - Else:
1679  *     - show a layout failure message.
1680  *
1681  * - Relayout:
1682  *   - Initialize all widgets
1683  *     (@ref gui2::widget::layout_initialize (full_initialization = false))
1684  *   - Handle shared sizes, since the reinitialization resets that state.
1685  *   - Goto start layout loop.
1686  *
1687  * @section grid Grid
1688  *
1689  * This section will be documented later.
1690  */
