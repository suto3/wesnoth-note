   1 #version 130
   2 #define EFFECT_FLIP 1
   3 #define EFFECT_FLOP 2
   4 #define EFFECT_GRAYSCALE 4
   5 
   6 in vec4 frag_color_mod;
   7 in vec4 frag_draw_color;
   8 in vec2 frag_texture_pos;
   9 in float frag_submerge;
  10 in vec2 frag_overlay_pos;
  11 flat in int frag_effects;
  12 
  13 uniform sampler2D tex;
  14 uniform sampler2D overlay;
  15 
  16 void main()
  17 {
  18         vec4 texture_color = texture2D(tex, frag_texture_pos);
  19         vec4 overlay_color = texture2D(overlay, frag_overlay_pos);
  20 
  21         if ((frag_effects & EFFECT_GRAYSCALE) != 0) {
  22                 float rgb = texture_color.r * 0.299 + texture_color.g * 0.587
  23                                         + texture_color.b * 0.114;
  24                 texture_color = vec4(rgb, rgb, rgb, texture_color.a);
  25         }
  26         float submerge_alpha = 0;
  27         if (frag_texture_pos.y > 1.0 - frag_submerge) {
  28                 submerge_alpha = 0.7 + (frag_texture_pos.y - (1 - frag_submerge))
  29                                 / frag_submerge * 0.3;
  30         }
  31         vec4 submerge_mod = vec4(0, 0, 0, submerge_alpha);
  32         vec4 final_base_color = texture_color + frag_draw_color + frag_color_mod - submerge_mod;
  33         gl_FragColor = mix(final_base_color, overlay_color, overlay_color.a);
  34 }
