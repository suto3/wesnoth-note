  30 bool enter_create_mode(CVideo& video, const config& game_config, saved_game& state, jump_to_campaign_info jump_to_campaign)
  31 {
  32         bool configure_canceled = false;
  33 
  34         do {
  35                 ng::create_engine create_eng(video, state);
  36 
  37                 create_eng.set_current_level_type(ng::level::TYPE::SP_CAMPAIGN);
  38 
  39                 const std::vector<ng::create_engine::level_ptr> campaigns =
  40                         create_eng.get_levels_by_type_unfiltered(ng::level::TYPE::SP_CAMPAIGN);
  41 
  42                 if(campaigns.empty()) {
  43                         gui2::show_error_message(video, _("No campaigns are available."));
  44                         return false;
  45                 }
  46 
  47                 std::string random_mode = "";
  48 
  49                 // No campaign selected from command line
  50                 if(jump_to_campaign.campaign_id_.empty()) {
  51                         gui2::dialogs::campaign_selection dlg(create_eng);
  52 
  53                         try {
  54                                 dlg.show(video);
  55                         } catch(wml_exception& e) {
  56                                 e.show(video);
  57                                 return false;
  58                         }
  59 
  60                         if(dlg.get_retval() != gui2::window::OK) {
  61                                 return false;
  62                         }
  63 
  64                         if(dlg.get_deterministic()) {
  65                                 random_mode = "deterministic";
  66                         }
  67                 } else {
  68                         // Don't reset the campaign_id_ so we can know
  69                         // if we should quit the game or return to the main menu
  70 
  71                         // Checking for valid campaign name
  72                         const auto campaign = std::find_if(campaigns.begin(), campaigns.end(), [&jump_to_campaign](ng::create_engine::level_ptr level) {
  73                                 return level->data()["id"] == jump_to_campaign.campaign_id_;
  74                         });
  75 
  76                         // Didn't find a campaign with that id
  77                         if(campaign == campaigns.end()) {
  78                                 ERR_NG << "No such campaign id to jump to: [" << jump_to_campaign.campaign_id_ << "]" << std::endl;
  79                                 return false;
  80                         }
  81 
  82                         create_eng.set_current_level(campaign - campaigns.begin());
  83                 }
  84 
  85                 state.classification().random_mode = random_mode;
  86 
  87                 const std::string selected_difficulty = create_eng.select_campaign_difficulty(jump_to_campaign.difficulty_);
  88 
  89                 if(selected_difficulty == "FAIL") return false;
  90                 if(selected_difficulty == "CANCEL") {
  91                         if(!jump_to_campaign.campaign_id_.empty()) {
  92                                 jump_to_campaign.campaign_id_ = "";
  93                         }
  94 
  95                         // Canceled difficulty dialog, relaunch the campaign selection dialog
  96                         return enter_create_mode(video, game_config, state, jump_to_campaign);
  97                 }
  98 
  99                 create_eng.prepare_for_era_and_mods();
 100                 create_eng.prepare_for_campaign(selected_difficulty);
 101 
 102                 if(!jump_to_campaign.scenario_id_.empty()) {
 103                         state.set_carryover_sides_start(
 104                                 config_of("next_scenario", jump_to_campaign.scenario_id_)
 105                         );
 106                 }
 107 
 108                 if(!state.valid()) {
 109                         ERR_NG << "Cannot load scenario with id=" << state.get_scenario_id() << std::endl;
 110                         return false;
 111                 }
 112 
 113                 configure_canceled = !enter_configure_mode(video, game_config_manager::get()->game_config(), state, create_eng);
 114 
 115         } while (configure_canceled);
 116 
 117         return true;
 118 }
