  47 unit_recruit::unit_recruit(std::vector<const unit_type*>& recruit_list, team& team)
  48         : recruit_list_(recruit_list)
  49         , team_(team)
  50         , selected_index_(0)
  51 {
  52         // Ensure the recruit list is sorted by name
  53         std::sort(recruit_list_.begin(), recruit_list_.end(), [](const unit_type* t1, const unit_type* t2) {
  54                 return t1->type_name().str() < t2->type_name().str();
  55         });
  56 }
  57 
  58 static std::string can_afford_unit(const std::string& text, const bool can_afford)
  59 {
  60         return can_afford ? text : "<span color='#ff0000'>" + text + "</span>";
  61 }

  63 void unit_recruit::pre_show(window& window)
  64 {
  65         listbox& list = find_widget<listbox>(&window, "recruit_list", false);
  66 
  67 #ifdef GUI2_EXPERIMENTAL_LISTBOX
  68         connect_signal_notify_modified(*list,
  69                 std::bind(&unit_recruit::list_item_clicked,
  70                 *this,
  71                 std::ref(window)));
  72 #else
  73         list.set_callback_value_change(
  74                 dialog_callback<unit_recruit, &unit_recruit::list_item_clicked>);
  75 #endif
  76 
  77         window.keyboard_capture(&list);
  78 
  79         connect_signal_mouse_left_click(
  80                 find_widget<button>(&window, "show_help", false),
  81                 std::bind(&unit_recruit::show_help, this, std::ref(window)));
  82 
  83         for(const auto& recruit : recruit_list_)
  84         {
  85                 std::map<std::string, string_map> row_data;
  86                 string_map column;
  87 
  88                 std::string     image_string = recruit->image() + "~RC(" + recruit->flag_rgb() + ">"
  89                         + team_.color() + ")";
  90 
  91                 int wb_gold = 0;
  92                 if(resources::controller) {
  93                         if(const std::shared_ptr<wb::manager>& whiteb = resources::controller->get_whiteboard()) {
  94                                 wb::future_map future; // So gold takes into account planned spending
  95                                 wb_gold = whiteb->get_spent_gold_for(team_.side());
  96                         }
  97                 }
  98 
  99                 const bool can_afford = recruit->cost() <= team_.gold() - wb_gold;
 100 
 101                 const std::string cost_string = std::to_string(recruit->cost());
 102 
 103                 column["use_markup"] = "true";
 104 
 105                 column["label"] = image_string;
 106                 row_data.emplace("unit_image", column);
 107 
 108                 column["label"] = can_afford_unit(recruit->type_name(), can_afford);
 109                 row_data.emplace("unit_type", column);
 110 
 111                 column["label"] = can_afford_unit(cost_string, can_afford);
 112                 row_data.emplace("unit_cost", column);
 113 
 114                 list.add_row(row_data);
 115         }
 116 
 117         list_item_clicked(window);
 118 }
 119 
 120 void unit_recruit::list_item_clicked(window& window)
 121 {
 122         const int selected_row
 123                 = find_widget<listbox>(&window, "recruit_list", false).get_selected_row();
 124 
 125         if(selected_row == -1) {
 126                 return;
 127         }
 128 
 129         find_widget<unit_preview_pane>(&window, "recruit_details", false)
 130                 .set_displayed_type(*recruit_list_[selected_row]);
 131 }

 136 }

 138 void unit_recruit::post_show(window& window)
 139 {
 140         if(get_retval() == window::OK) {
 141                 selected_index_ = find_widget<listbox>(&window, "recruit_list", false)
 142                         .get_selected_row();
 143         }
