  71 plugins_manager::STATUS plugins_manager::get_status(size_t idx) {
  72         if (idx < plugins_.size()) {
  73                 if (!plugins_[idx].thread) {
  74                         return plugins_manager::STATUS::NONE;
  75                 } else {
  76                         return plugins_[idx].thread->is_running() ? plugins_manager::STATUS::RUNNING : plugins_manager::STATUS::STOPPED;
  77                 }
  78         }
  79         throw std::runtime_error("index out of bounds");
  80 }

 100 void plugins_manager::start_plugin(size_t idx)
 101 {
 102         DBG_PLG << "start_plugin[" << idx <<"]\n";
 103         if (idx < plugins_.size()) {
 104                 if (!plugins_[idx].thread) {
 105                         DBG_PLG << "creating thread[" << idx << "]\n";
 106                         plugins_[idx].thread.reset(plugins_[idx].is_file ?
 107                                                 kernel_->load_script_from_file(plugins_[idx].source) : kernel_->load_script_from_string(plugins_[idx].source));
 108                         DBG_PLG << "finished [" << idx << "], status = '" << plugins_[idx].thread->status() << "'\n";
 109                 } else {
 110                         DBG_PLG << "thread already exists, skipping\n";
 111                 }
 112                 return ;
 113         }
 114         throw std::runtime_error("index out of bounds");
 115 }

 143 void plugins_manager::notify_event(const std::string & name, const config & data)
 144 {
 145         event evt;

 157 void plugins_manager::play_slice(const plugins_context & ctxt)
 158 {
 159         if (playing_) {
 160                 *playing_ = false;      //this is to ensure "reentrancy" -- any previous calls to this function that never returned
 161                                         //and looped back into the plugins system, should be halted and their later requests discarded
 162                                         //this is to ensure the semantics that if a plugins context is left, then any pending requests
 163                                         //are discarded to prevent them from being executed at an improper time
 164         }
 165         playing_ = std::make_shared<bool> (true);
 166         std::shared_ptr<bool> local = playing_; //make a local copy of the pointer on the stack
 167 
 168         for (size_t idx = 0; idx < size(); ++idx)
 169         {
 170                 DBG_PLG << "play_slice[" << idx << "] ... \n";
 171                 if (plugins_[idx].thread && plugins_[idx].thread->is_running()) {
 172                         DBG_PLG << "is running...";
 173                         if (!*local) {                  //check playing_ before each call to be sure that we should still continue
 174                                 DBG_PLG << "aborting\n";
 175                                 return;
 176                         }
 177 
 178                         std::vector<event> input = plugins_[idx].queue; //empty the queue to a temporary variable
 179                         plugins_[idx].queue = std::vector<event>();
 180 
 181                         //application_lua_kernel::requests_list requests =
 182                         std::vector<std::function<bool(void)> > requests =
 183                                 plugins_[idx].thread->run_script(ctxt, input);
 184 
 185                         DBG_PLG << "thread returned " << requests.size() << " requests\n";
 186 
 187                         for (size_t j = 0; j < requests.size(); ++j) {
 188                                 if (!*local) return;            //check playing_ before each call to be sure that we should still continue
 189                                 if (!requests[j]()) {
 190                                         *local = false;
 191                                         return ; //call the function but if it returns false (error) then stop
 192                                 }
 193                         }
 194 
 195                         DBG_PLG << "play_slice[" << idx << "] finished.\n";
 196                 } else if (!plugins_[idx].thread) {
 197                         DBG_PLG << "thread ["<< idx << "] not created\n";
 198                 } else {
 199                         DBG_PLG << "thread ["<< idx << "] not running\n";
 200                 }
 201         }
 202         *local = false;
 203 }
