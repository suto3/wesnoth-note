   1 #textdomain wesnoth
   2 # This file contains macros useful for debugging and testing purposes.
   3 
   4 #define DEBUG_MSG MESSAGE_TEXT
   5     # Emit a debug message.  Meant to be overridden with no-op definition
   6     # of the same name for production use.
   7     [message]
   8         speaker=narrator
   9         message={MESSAGE_TEXT}
  10         image=wesnoth-icon.png
  11     [/message]
  12 #enddef
  13 
  14 #define DEBUG TEXT
  15     # Logs a WML debug message to consol and the in-game chat.
  16     # It does nothing unless the 'wml' log domain is set to debug
  17     # level at application startup (usually with --log-debug=wml),
  18     # or in-game with the :log command (e.g. :log debug wml).
  19     #
  20     # Example:
  21     # ! {DEBUG "inconsistency: $unit.id is not loyal!"}
  22     [wml_message]
  23         logger=debug
  24         message={TEXT}
  25     [/wml_message]
  26 #enddef
  27 
  28 #define LOG TEXT
  29     # Logs a WML message to console and the in-game chat. It does
  30     # nothing unless the 'wml' log domain is set to log level at
  31     # application startup (usually with --log-info=wml), or in-game
  32     # with the :log command (e.g. :log info wml).
  33     #
  34     # Example:
  35     # ! {LOG "player moved to forbidden location"}
  36     [wml_message]
  37         logger=log
  38         message={TEXT}
  39     [/wml_message]
  40 #enddef
  41 
  42 #define WARNING TEXT
  43     # Logs a WML warning to console and the in-game chat. It does
  44     # nothing unless the 'wml' log domain is set to warning level at
  45     # application startup (usually with --log-warning=wml), or in-game
  46     # with the :log command (e.g. :log warning wml).
  47     #
  48     # Example:
  49     # ! {WARNING "malformed unit data"}
  50     [wml_message]
  51         logger=warn
  52         message={TEXT}
  53     [/wml_message]
  54 #enddef
  55 
  56 #define ERROR TEXT
  57     # Logs a WML error to console and the in-game chat.
  58     #
  59     # Example:
  60     # ! {ERROR "could not respawn hero at 2,4"}
  61     [wml_message]
  62         logger=err
  63         message={TEXT}
  64     [/wml_message]
  65 #enddef
  66 
  67 #define TIMED_ACTIONS TITLE_TEXT CONTENT_WML
  68     # Measure (in milliseconds) the time arbitrary event WML takes to
  69     # execute.
  70     #
  71     # Example:
  72     #! [event]
  73     #!     name=start
  74     #!
  75     #!     {TIME_ACTIONS "orc renaming" (
  76     #!         {MODIFY_UNIT race=orc name ( _ "Azir")}
  77     #!     )}
  78     #! [/event]
  79     [set_variable]
  80         name=TIME_ACTIONS_time_begin
  81         time=stamp
  82     [/set_variable]
  83 
  84     {CONTENT_WML}
  85 
  86     [set_variable]
  87         name=TIME_ACTIONS_time_end
  88         time=stamp
  89     [/set_variable]
  90 
  91     [message]
  92         speaker=narrator
  93         caption={TITLE_TEXT}
  94         message="$($TIME_ACTIONS_time_end - $TIME_ACTIONS_time_begin) milliseconds."
  95         image=wesnoth-icon.png
  96     [/message]
  97 
  98     [clear_variable]
  99         name=TIME_ACTIONS_time_begin,TIME_ACTIONS_time_end
 100     [/clear_variable]
 101 #enddef
