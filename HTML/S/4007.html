   7 #define TELEPORT_UNIT FILTER X Y
   8     # Teleports a unit matching FILTER to X,Y
   9     #
  10     # For example, teleport player 3's leader to 4,5
  11     #! {TELEPORT_UNIT (
  12     #!  side=3
  13     #!  canrecruit=yes
  14     #! ) 4 5}
  15     [teleport]
  16         [filter]
  17             {FILTER}
  18         [/filter]
  19         x={X}
  20         y={Y}
  21     [/teleport]
  22     [redraw]
  23     [/redraw]
  24 #enddef
  25 
  26 #define TELEPORT_TILE OLD_X OLD_Y NEW_X NEW_Y
  27     # Teleports a unit on tile OLD_X,OLD_Y to NEW_X,NEW_Y
  28     #
  29     # For example, teleport any unit thats currently on 1,1 to 4,5
  30     #! {TELEPORT_TILE 1 1 4 5}
  31     [teleport]
  32         [filter]
  33             x={OLD_X}
  34             y={OLD_Y}
  35         [/filter]
  36         x={NEW_X}
  37         y={NEW_Y}
  38     [/teleport]
  39     [redraw]
  40     [/redraw]
  41 #enddef

 202 #define TELEPORT_EFFECT_OBJECT_ID _ID_
 203     # Creates and assigns a temporary object which applies the generalized silver mage
 204     # teleport animations to the unit with the matching ID
 205     #
 206     # For example, to apply teleport animations to the unit 'Bob':
 207     #!{TELEPORT_EFFECT_OBJECT_ID Bob}
 208 
 209     [object]
 210         silent=yes
 211         duration=turn
 212 
 213         [filter]
 214             id={_ID_}
 215         [/filter]
 216 
 217         [effect]
 218             apply_to=new_animation
 219 
 220             [animation]
 221                 apply_to=pre_teleport
 222                 {TELEPORT_OUT_ANIMATION}
 223             [/animation]
 224 
 225             [animation]
 226                 apply_to=post_teleport
 227                 {TELEPORT_IN_ANIMATION}
 228             [/animation]
 229         [/effect]
 230     [/object]
 231 #enddef
