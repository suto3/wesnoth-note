  83 std::string parse_wrapped_credentials_field(const std::string& raw)
  84 {
  85         if(raw.empty() || raw == EMPTY_WRAPPED_STRING) {
  86                 // empty (wrapped or not)
  87                 return raw;
  88         } else if(raw.length() < 2 || raw[0] != WRAP_CHAR || raw[raw.length() - 1] != WRAP_CHAR ) {
  89                 // malformed/not wrapped (shouldn't happen)
  90                 ERR_CFG << "malformed user credentials (did you manually edit the preferences file?)" << std::endl;
  91                 return raw;
  92         }
  93 
  94         return raw.substr(1, raw.length() - 2);
  95 }

 116         set_music_volume(music_volume());
 117         set_sound_volume(sound_volume());
 118 
 119         set_show_haloes(preferences::get("show_haloes", true));
 120         if (!preferences::get("remember_timer_settings", false)) {
 121                 preferences::erase("mp_countdown_init_time");
 122                 preferences::erase("mp_countdown_reservoir_time");
 123                 preferences::erase("mp_countdown_turn_bonus");
 124                 preferences::erase("mp_countdown_action_bonus");
 125         }
 126 
 127         /*
 128         completed_campaigns = "A,B,C"
 129         [completed_campaigns]
 130                 [campaign]
 131                         name = "A"
 132                         difficulty_levels = "EASY,MEDIUM"
 133                 [/campaign]
 134         [/completed_campaigns]
 135         */
 136         for (const std::string &c : utils::split(preferences::get("completed_campaigns"))) {
 137                 completed_campaigns[c]; // create the elements
 138         }
 139         if (const config &ccc = preferences::get_child("completed_campaigns")) {
 140                 for (const config &cc : ccc.child_range("campaign")) {
 141                         std::set<std::string> &d = completed_campaigns[cc["name"]];
 142                         std::vector<std::string> nd = utils::split(cc["difficulty_levels"]);
 143                         std::copy(nd.begin(), nd.end(), std::inserter(d, d.begin()));
 144                 }
 145         }
 146 
 147         const std::vector<std::string> v (utils::split(preferences::get("encountered_units")));
 148         encountered_units_set.insert(v.begin(), v.end());
 149 
 150         const t_translation::ter_list terrain (t_translation::read_list(preferences::get("encountered_terrain_list")));
 151         encountered_terrains_set.insert(terrain.begin(), terrain.end());
 152 
 153         if (const config &history = preferences::get_child("history"))
 154         {
 155 /* Structure of the history
 156         [history]
 157                 [history_id]
 158                         [line]
 159                                 message = foobar
 160                         [/line]
 161 */
 162                 for (const config::any_child &h : history.all_children_range())
 163                 {
 164                         for (const config &l : h.cfg.child_range("line")) {
 165                                 history_map[h.key].push_back(l["message"]);
 166                         }
 167                 }
 168         }

 170         //network::ping_timeout = get_ping_timeout();

 173 manager::~manager()
 174 {
 175         config campaigns;
 176         typedef const std::pair<std::string, std::set<std::string> > cc_elem;
 177         for (cc_elem &elem : completed_campaigns) {
 178                 config cmp;
 179                 cmp["name"] = elem.first;
 180                 cmp["difficulty_levels"] = utils::join(elem.second);
 181                 campaigns.add_child("campaign", cmp);
 182         }
 183         preferences::set_child("completed_campaigns", campaigns);
 184         std::vector<std::string> v (encountered_units_set.begin(), encountered_units_set.end());
 185         preferences::set("encountered_units", utils::join(v));
 186         t_translation::ter_list terrain (encountered_terrains_set.begin(), encountered_terrains_set.end());
 187         preferences::set("encountered_terrain_list", t_translation::write_list(terrain));
 188 
 189 /* Structure of the history
 190         [history]
 191                 [history_id]
 192                         [line]
 193                                 message = foobar
 194                         [/line]
 195 */
 196         config history;
 197         typedef std::pair<std::string, std::vector<std::string> > hack;
 198         for (const hack& history_id : history_map) {
 199 
 200                 config history_id_cfg; // [history_id]
 201                 for (const std::string& line : history_id.second) {
 202                         config cfg; // [line]
 203 
 204                         cfg["message"] = line;
 205                         history_id_cfg.add_child("line", cfg);
 206                 }
 207 
 208                 history.add_child(history_id.first, history_id_cfg);
 209         }
 210         preferences::set_child("history", history);
 211 
 212         history_map.clear();
 213         encountered_units_set.clear();
 214         encountered_terrains_set.clear();
 215 }

 281 preferences::acquaintance* add_acquaintance(const std::string& nick, const std::string& mode, const std::string& notes)
 282 {
 283         if(!utils::isvalid_wildcard(nick)) {
 284                 return nullptr;
 285         }
 286 
 287         preferences::acquaintance new_entry(nick, mode, notes);
 288 
 289         acquaintances_map::iterator iter;
 290         bool success;
 291 
 292         std::tie(iter, success) = acquaintances.emplace(nick, new_entry);
 293 
 294         if(!success) {
 295                 iter->second = new_entry;
 296         }
 297 
 298         save_acquaintances();
 299 
 300         return &iter->second;
 301 }

 303 bool remove_acquaintance(const std::string& nick) {
 304         std::map<std::string, acquaintance>::iterator i = acquaintances.find(nick);
 305 
 306         //nick might include the notes, depending on how we're removing
 307         if(i == acquaintances.end()) {
 308                 size_t pos = nick.find_first_of(' ');
 309 
 310                 if(pos != std::string::npos) {
 311                         i = acquaintances.find(nick.substr(0, pos));
 312                 }
 313         }
 314 
 315         if(i == acquaintances.end()) {
 316                 return false;
 317         }
 318 
 319         acquaintances.erase(i);
 320         save_acquaintances();
 321 
 322         return true;
 323 }

 364 bool parse_should_show_lobby_join(const std::string &sender, const std::string &message)
 365 {
 366         // If it's actually not a lobby join or leave message return true (show it).
 367         if (sender != "server") return true;
 368         std::string::size_type pos = message.find(" has logged into the lobby");
 369         if (pos == std::string::npos){
 370                 pos = message.find(" has disconnected");
 371                 if (pos == std::string::npos) return true;
 372         }
 373         int lj = lobby_joins();
 374         if (lj == SHOW_NONE) return false;
 375         if (lj == SHOW_ALL) return true;
 376         return is_friend(message.substr(0, pos));
 377 }

 480 static std::string get_system_username()
 481 {
 482         std::string res;
 483 #ifdef _WIN32
 484         wchar_t buffer[300];
 485         DWORD size = 300;
 486         if(GetUserNameW(buffer,&size)) {
 487                 //size includes a terminating null character.
 488                 assert(size > 0);
 489                 res = unicode_cast<utf8::string>(boost::iterator_range<wchar_t*>(buffer, buffer + size - 1));
 490         }
 491 #else
 492         if(char* const login = getenv("USER")) {
 493                 res = login;
 494         }
 495 #endif
 496         return res;
 497 }

 499 std::string login()
 500 {
 501         const std::string res = preferences::get("login");
 502         if(res.empty() || res == EMPTY_WRAPPED_STRING) {
 503                 const std::string& login = get_system_username();
 504                 if(!login.empty()) {
 505                         return login;
 506                 }
 507 
 508                 if(res.empty()) {
 509                         return _("player");
 510                 }
 511         }
 512 
 513         if(!wrap_login()) {
 514                 return res;
 515         } else {
 516                 return parse_wrapped_credentials_field(res);
 517         }
 518 }

1020 compression::format save_compression_format()
1021 {
1022         const std::string& choice =
1023                 preferences::get("compress_saves");
1024 
1025         // "yes" was used in 1.11.7 and earlier; the compress_saves
1026         // option used to be a toggle for gzip in those versions.
1027         if(choice.empty() || choice == "gzip" || choice == "yes") {
1028                 return compression::GZIP;
1029         } else if(choice == "bzip2") {
1030                 return compression::BZIP2;
1031         } else if(choice == "none" || choice == "no") { // see above
1032                 return compression::NONE;
1033         } /*else*/
1034 
1035         // In case the preferences file was created by a later version
1036         // supporting some algorithm we don't; although why would anyone
1037         // playing a game need more algorithms, really...
1038         return compression::GZIP;
1039 }

1137 void encounter_recruitable_units(const std::vector<team>& teams){
1138         for (std::vector<team>::const_iterator help_team_it = teams.begin();
1139                 help_team_it != teams.end(); ++help_team_it) {
1140                 help_team_it->log_recruitable();
1141                 encountered_units_set.insert(help_team_it->recruits().begin(), help_team_it->recruits().end());
1142         }
1143 }

1146         for (unit_map::const_iterator help_unit_it = units.begin();
1147                  help_unit_it != units.end(); ++help_unit_it) {
1148                 encountered_units_set.insert(help_unit_it->type_id());
1149         }
1150 }

1153         for (const team& t : teams) {

1155                         encountered_units_set.insert(u->type_id());
1156                 }
1157         }

1160 void encounter_map_terrain(const gamemap& map)
1161 {

1163                 const t_translation::terrain_code terrain = map.get_terrain(loc);
1164                 preferences::encountered_terrains().insert(terrain);
1165                 for (t_translation::terrain_code t : map.underlying_union_terrain(loc)) {
1166                         preferences::encountered_terrains().insert(t);
1167                 }
1168         });
