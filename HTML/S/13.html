  23 lua_map_generator::lua_map_generator(const config & cfg)
  24         : id_(cfg["id"])
  25         , config_name_(cfg["config_name"])
  26         , user_config_(cfg["user_config"])
  27         , create_map_(cfg["create_map"])
  28         , create_scenario_(cfg["create_scenario"])
  29         , lk_()
  30         , generator_data_(cfg)
  31 {
  32         const char* required[] {"id", "config_name", "create_map"};
  33         for (std::string req : required) {
  34                 if (!cfg.has_attribute(req)) {
  35                         if(req == "create_map" && cfg.has_attribute("create_scenario")) {
  36                                 // One of these is required, but not both
  37                                 continue;
  38                         }
  39                         std::string msg = "Error when constructing a lua map generator -- missing a required attribute '";
  40                         msg += req + "'\n";
  41                         msg += "Config was '" + cfg.debug() + "'";
  42                         throw mapgen_exception(msg);
  43                 }
  44         }
  45 }

  53                 msg += "The generator was: " + config_name_ + "\n";
