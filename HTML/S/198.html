  69 static int impl_music_get(lua_State* L) {
  70         if(lua_isnumber(L, 2)) {
  71                 push_track(L, lua_tointeger(L, 2) - 1);
  72                 return 1;
  73         }
  74         const char* m = luaL_checkstring(L, 2);
  75         if(strcmp(m, "current") == 0) {
  76                 push_track(L, sound::get_current_track());
  77                 return 1;
  78         }
  79         if(strcmp(m, "current_i") == 0) {
  80                 size_t i = sound::get_current_track();
  81                 if(i == sound::get_num_tracks()) {
  82                         lua_pushnil(L);
  83                 } else {
  84                         lua_pushinteger(L, i + 1);
  85                 }
  86                 return 1;
  87         }
  88         if(strcmp(m, "all") == 0) {
  89                 config playlist;
  90                 sound::write_music_play_list(playlist);
  91                 const auto& range = playlist.child_range("music");
  92                 std::vector<config> tracks(range.begin(), range.end());
  93                 lua_push(L, tracks);
  94                 return 1;
  95         }
  96         // This calculation reverses the one used in [volume] to get back the relative volume level.
  97         // (Which is the same calculation that's duplicated in impl_music_set.)
  98         return_float_attrib("volume", sound::get_music_volume() * 100.0f / preferences::music_volume());
  99         return luaW_getmetafield(L, 1, m);
 100 }

 102 static int impl_music_set(lua_State* L) {
 103         if(lua_isnumber(L, 2)) {
 104                 unsigned int i = lua_tointeger(L, 2) - 1;
 105                 config cfg;
 106                 if(lua_isnil(L, 3)) {
 107                         if(i < sound::get_num_tracks()) {
 108                                 sound::remove_track(i);
 109                         }
 110                 } else if(luaW_toconfig(L, 3, cfg)) {
 111                         // Don't allow play_once=yes
 112                         if(cfg["play_once"]) {
 113                                 return luaL_argerror(L, 3, "For play_once, use wesnoth.music_list.play instead");
 114                         }
 115                         if(i < sound::get_num_tracks()) {
 116                                 sound::play_music_config(cfg);
 117                         } else {
 118                                 // Remove the track at that index and add the new one in its place
 119                                 // It's a little inefficient though...
 120                                 sound::remove_track(i);
 121                                 sound::play_music_config(cfg, i);
 122                         }
 123                 } else {
 124                         lua_music_track& track = *get_track(L, 3);
 125                         if(i < sound::get_num_tracks()) {
 126                                 sound::set_track(i, track.operator->());
 127                         } else {
 128                                 track->write(cfg, true);
 129                                 sound::play_music_config(cfg);
 130                         }
 131                 }
 132                 return 0;
 133         }
 134         const char* m = luaL_checkstring(L, 2);
 135         modify_float_attrib_check_range("volume", sound::set_music_volume(value * preferences::music_volume() / 100.0f), 0.0, 100.0);
 136         modify_int_attrib_check_range("current_i", sound::play_track(value - 1), 1, static_cast<int>(sound::get_num_tracks()));
 137         return 0;
 138 }

 158 static int intf_music_add(lua_State* L) {
 159         int index = -1;
 160         if(lua_isinteger(L, 1)) {
 161                 index = lua_tointeger(L, 1);
 162                 lua_remove(L, 1);
 163         }
 164         config cfg = config_of
 165                 ("name", luaL_checkstring(L, 1))
 166                 ("append", true);
 167         bool found_ms_before = false, found_ms_after = false, found_imm = false;
 168         for(int i = 2; i <= lua_gettop(L); i++) {
 169                 if(lua_isboolean(L, i)) {
 170                         if(found_imm) {
 171                                 return luaL_argerror(L, i, "only one boolean argument may be passed");
 172                         } else {
 173                                 cfg["immediate"] = luaW_toboolean(L, i);
 174                         }
 175                 } else if(lua_isnumber(L, i)) {
 176                         if(found_ms_after) {
 177                                 return luaL_argerror(L, i, "only two integer arguments may be passed");
 178                         } else if(found_ms_before) {
 179                                 cfg["ms_after"] = lua_tointeger(L, i);
 180                                 found_ms_after = true;
 181                         } else {
 182                                 cfg["ms_before"] = lua_tointeger(L, i);
 183                                 found_ms_before = true;
 184                         }
 185                 } else {
 186                         return luaL_argerror(L, i, "unrecognized argument");
 187                 }
 188         }
 189         sound::play_music_config(cfg, index);
 190         return 0;
 191 }

 198 static int intf_music_remove(lua_State* L) {
 199         // Use a non-standard comparator to ensure iteration in descending order
 200         std::set<int, std::greater<int>> to_remove;
 201         for(int i = 1; i <= lua_gettop(L); i++) {
 202                 to_remove.insert(luaL_checkinteger(L, i));
 203         }
 204         for(int i : to_remove) {
 205                 sound::remove_track(i);
 206         }
 207         return 0;
 208 }

 215 static int impl_track_get(lua_State* L) {
 216         lua_music_track& track = *get_track(L, 1);
 217         const char* m = luaL_checkstring(L, 2);
 218         return_bool_attrib("valid", track.valid());
 219         if(!track.valid()) {
 220                 return luaL_error(L, "Tried to access member of track that is no longer valid.");
 221         }
 222         return_bool_attrib("append", track->append());
 223         return_bool_attrib("shuffle", track->shuffle());
 224         return_bool_attrib("immediate", track->immediate());
 225         return_bool_attrib("once", track->play_once());
 226         return_int_attrib("ms_before", track->ms_before());
 227         return_int_attrib("ms_after", track->ms_after());
 228         return_string_attrib("name", track->id());
 229         return_string_attrib("title", track->title());
 230         return luaW_getmetafield(L, 1, m);
 231 }

 243 static int impl_track_eq(lua_State* L) {
 244         lua_music_track* a = get_track(L, 1);
 245         lua_music_track* b = get_track(L, 2);
 246         if(!a || !b) {
 247                 // This implies that one argument is not a music track, though I suspect this is dead code...?
 248                 // Does Lua ever call this if the arguments are not of the same type?
 249                 lua_pushboolean(L, false);
 250                 return 1;
 251         }
 252         if(!a->valid() && !b->valid()) {
 253                 lua_pushboolean(L, true);
 254                 return 1;
 255         }
 256         if(a->valid() && b->valid()) {
 257                 lua_music_track& lhs = *a;
 258                 lua_music_track& rhs = *b;
 259                 lua_pushboolean(L, lhs->id() == rhs->id() && lhs->shuffle() == rhs->shuffle() && lhs->play_once() == rhs->play_once() && lhs->ms_before() == rhs->ms_before() && lhs->ms_after() == rhs->ms_after());
 260                 return 1;
 261         }
 262         lua_pushboolean(L, false);
 263         return 1;
 264 }

 283                 };
 284                 luaL_setfuncs(L, pl_callbacks, 0);
 285                 lua_pushstring(L, "music playlist");
 286                 lua_setfield(L, -2, "__metatable");
 287                 lua_setmetatable(L, -2);
 288                 lua_setfield(L, -2, "music_list");

 297                         { "__eq", impl_track_eq },
 298                         { nullptr, nullptr },
 299                 };
 300                 luaL_setfuncs(L, track_callbacks, 0);
 301                 lua_pushstring(L, Track);
 302                 lua_setfield(L, -2, "__metatable");
