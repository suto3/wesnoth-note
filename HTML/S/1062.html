 107         class pump_manager {
 108         public:
 109                 pump_manager(pump_impl & );
 110                 ~pump_manager();
 111 
 112                 /// Allows iteration through the queued events.
 113                 queued_event & next() { return queue_[pumped_count_++]; }
 114                 /// Indicates the iteration is over.
 115                 bool done() const { return pumped_count_ >= queue_.size(); }
 116 
 117                 unsigned count() {
 118                         return impl_.instance_count;
 119                 }
 120 
 121         private:
 122                 pump_impl & impl_;
 123                 int x1_, x2_, y1_, y2_;
 124                 /// Tracks the events to process.
 125                 /// This isolates these events from any events that might be generated
 126                 /// during the processing.
 127                 std::vector<queued_event> queue_;
 128                 /// Tracks how many events have been processed.
 129                 size_t pumped_count_;
 130         };

 135         pump_manager::pump_manager(pump_impl & impl) :
 136                 impl_(impl),
 137                 x1_(resources::gamedata->get_variable("x1")),
 138                 x2_(resources::gamedata->get_variable("x2")),
 139                 y1_(resources::gamedata->get_variable("y1")),
 140                 y2_(resources::gamedata->get_variable("y2")),
 141                 queue_(), // Filled later with a swap().
 142                 pumped_count_(0)
 143         {
 144                 queue_.swap(impl_.events_queue);
 145                 ++impl_.instance_count;
 146         }
 147 
 148         pump_manager::~pump_manager() {
 149                 --impl_.instance_count;
 150 
 151                 // Not sure what the correct thing to do is here. In princple,
 152                 // discarding all events (i.e. clearing events_queue) seems like
 153                 // the right thing to do in the face of an exception. However, the
 154                 // previous functionality preserved the queue, so for now we will
 155                 // restore it.
 156                 if ( !done() ) {
 157                         // The remainig events get inserted at the beginning of events_queue.
 158                         std::vector<queued_event> temp;
 159                         impl_.events_queue.swap(temp);
 160                         impl_.events_queue.insert(impl_.events_queue.end(), queue_.begin() + pumped_count_, queue_.end());
 161                         impl_.events_queue.insert(impl_.events_queue.end(), temp.begin(), temp.end());
 162                 }
 163 
 164                 // Restore the old values of the game variables.
 165                 resources::gamedata->get_variable("y2") = y2_;
 166                 resources::gamedata->get_variable("y1") = y1_;
 167                 resources::gamedata->get_variable("x2") = x2_;
 168                 resources::gamedata->get_variable("x1") = x1_;
 169         }

 172         /**
 173          * Returns true iff the given event passes all its filters.
 174          */
 175         bool wml_event_pump::filter_event(const event_handler& handler, const queued_event& ev)
 176         {
 177                 const unit_map& units = resources::gameboard->units();
 178                 unit_map::const_iterator unit1 = units.find(ev.loc1);
 179                 unit_map::const_iterator unit2 = units.find(ev.loc2);
 180                 vconfig filters(handler.get_config());
 181 
 182                 for (const vconfig &condition : filters.get_children("filter_condition"))
 183                 {
 184                         if (!conditional_passed(condition)) {
 185                                 return false;
 186                         }
 187                 }
 188 
 189                 for (const vconfig &f : filters.get_children("filter_side"))
 190                 {
 191                         side_filter ssf(f, &resources::controller->gamestate());
 192                         if ( !ssf.match(resources::controller->current_side()) )
 193                                 return false;
 194                 }
 195 
 196                 for (const vconfig &f : filters.get_children("filter"))
 197                 {
 198                         if ( !ev.loc1.matches_unit_filter(unit1, f) ) {
 199                                 return false;
 200                         }
 201                 }
 202 
 203                 vconfig::child_list special_filters = filters.get_children("filter_attack");
 204                 bool special_matches = special_filters.empty();
 205                 if ( !special_matches  &&  unit1 != units.end() )
 206                 {
 207                         const bool matches_unit = ev.loc1.matches_unit(unit1);
 208                         const config & attack = ev.data.child("first");
 209                         for (const vconfig &f : special_filters)
 210                         {
 211                                 if ( f.empty() )
 212                                         special_matches = true;
 213                                 else if ( !matches_unit )
 214                                         return false;
 215 
 216                                 special_matches = special_matches ||
 217                                                       matches_special_filter(attack, f);
 218                         }
 219                 }
 220                 if(!special_matches) {
 221                         return false;
 222                 }
 223 
 224                 for (const vconfig &f : filters.get_children("filter_second"))
 225                 {
 226                         if ( !ev.loc2.matches_unit_filter(unit2, f) ) {
 227                                 return false;
 228                         }
 229                 }
 230 
 231                 special_filters = filters.get_children("filter_second_attack");
 232                 special_matches = special_filters.empty();
 233                 if ( !special_matches  &&  unit2 != units.end() )
 234                 {
 235                         const bool matches_unit = ev.loc2.matches_unit(unit2);
 236                         const config & attack = ev.data.child("second");
 237                         for (const vconfig &f : special_filters)
 238                         {
 239                                 if ( f.empty() )
 240                                         special_matches = true;
 241                                 else if ( !matches_unit )
 242                                         return false;
 243 
 244                                 special_matches = special_matches ||
 245                                                       matches_special_filter(attack, f);
 246                         }
 247                 }
 248                 if(!special_matches) {
 249                         return false;
 250                 }
 251 
 252                 // All filters passed.
 253                 return true;
 254         }

 256         /**
 257          * Processes an event through a single event handler.
 258          * This includes checking event filters, but not checking that the event
 259          * name matches.
 260          *
 261          * @param[in,out]  handler_p  The handler to offer the event to.
 262          *                            This may be reset during processing.
 263          * @param[in]      ev         The event information.
 264          *
 265          * @returns true if the game state changed.
 266          */
 267         bool wml_event_pump::process_event(handler_ptr& handler_p, const queued_event& ev)
 268         {
 269                 DBG_EH << "processing event " << ev.name << " with id=" << ev.id << "\n";
 270 
 271                 // We currently never pass a null pointer to this function, but to
 272                 // guard against future modifications:
 273                 if ( !handler_p )
 274                         return false;
 275 
 276                 unit_map& units = resources::gameboard->units();
 277                 scoped_xy_unit first_unit("unit", ev.loc1, units);
 278                 scoped_xy_unit second_unit("second_unit", ev.loc2, units);
 279                 scoped_weapon_info first_weapon("weapon", ev.data.child("first"));
 280                 scoped_weapon_info second_weapon("second_weapon", ev.data.child("second"));
 281 
 282                 if ( !filter_event(*handler_p, ev) )
 283                         return false;
 284 
 285                 // The event hasn't been filtered out, so execute the handler.
 286                 ++impl_->internal_wml_tracking;
 287                 context::scoped evc(impl_->contexts_);
 288                 assert(resources::lua_kernel != nullptr);
 289                 handler_p->handle_event(ev, handler_p, *resources::lua_kernel);
 290                 // NOTE: handler_p may be null at this point!
 291 
 292                 if(ev.name == "select") {
 293                         resources::gamedata->last_selected = ev.loc1;
 294                 }
 295                 if(resources::screen != nullptr) {
 296                         resources::screen->maybe_rebuild();
 297                 }
 298                 return context_mutated();
 299         }

 301         /**
 302          * Helper function for show_wml_messages(), which gathers
 303          * the messages from a stringstream.
 304          */
 305         void wml_event_pump::fill_wml_messages_map(std::map<std::string, int>& msg_map, std::stringstream& source)
 306         {
 307                 while(true) {
 308                         std::string msg;
 309                         std::getline(source, msg);
 310 
 311                         if(source.eof()) {
 312                                 break;
 313                         }
 314 
 315                         if(msg == "") {
 316                                 continue;
 317                         }
 318 
 319                         if(msg_map.find(msg) == msg_map.end()) {
 320                                 msg_map[msg] = 1;
 321                         } else {
 322                                 msg_map[msg]++;
 323                         }
 324                 }
 325                 // Make sure the eof flag is cleared otherwise no new messages are shown
 326                 source.clear();
 327         }

 329         /**
 330          * Shows a summary of messages/errors generated so far by WML.
 331          * Identical messages are shown once, with (between parentheses)
 332          * the number of times that message was encountered.
 333          * The order in which the messages are shown does not need
 334          * to be the order in which these messages are encountered.
 335          * Messages are also written to std::cerr if to_cerr is true.
 336          */
 337         void wml_event_pump::show_wml_messages(std::stringstream& source, const std::string & caption,
 338                                bool to_cerr)
 339         {
 340                 // Get all unique messages in messages,
 341                 // with the number of encounters for these messages
 342                 std::map<std::string, int> messages;
 343                 fill_wml_messages_map(messages, source);
 344 
 345                 // Show the messages collected
 346                 for(std::map<std::string, int>::const_iterator itor = messages.begin();
 347                     itor != messages.end(); ++itor )
 348                 {
 349                         std::stringstream msg;
 350                         msg << itor->first;
 351                         if(itor->second > 1) {
 352                                 msg << " (" << itor->second << ")";
 353                         }
 354 
 355                         resources::screen->get_chat_manager().add_chat_message(time(nullptr), caption, 0, msg.str(),
 356                                         events::chat_handler::MESSAGE_PUBLIC, false);
 357                         if ( to_cerr )
 358                                 std::cerr << caption << ": " << msg.str() << '\n';
 359                 }
 360         }

 494 bool wml_event_pump::operator()()
 495 {
 496         // Quick aborts:
 497         if(resources::screen == nullptr)
 498                 return false;
 499         assert(resources::lua_kernel != nullptr);
 500         if ( impl_->events_queue.empty() ) {
 501                 DBG_EH << "Processing queued events, but none found.\n";
 502                 return false;
 503         }
 504         if(impl_->instance_count >= game_config::max_loop) {
 505                 ERR_NG << "game_events pump waiting to process new events because "
 506                        << "recursion level would exceed maximum: " << game_config::max_loop << '\n';
 507                 return false;
 508         }
 509         if(!lg::debug().dont_log("event_handler")) {
 510                 std::stringstream ss;
 511                 for(const queued_event& ev : impl_->events_queue) {
 512                         ss << "name=" << ev.name << ", " << "id=" << ev.id << "; ";
 513                 }
 514                 DBG_EH << "processing queued events: " << ss.str() << "\n";
 515         }
 516 
 517         const size_t old_wml_track = impl_->internal_wml_tracking;
 518         // Ensure the whiteboard doesn't attempt to build its future unit map
 519         // while events are being processed.
 520         wb::real_map real_unit_map;
 521 
 522         pump_manager pump_instance(*impl_);
 523         context::scoped evc(impl_->contexts_, false);
 524         // Loop through the events we need to process.
 525         while ( !pump_instance.done() )
 526         {
 527                 queued_event & ev = pump_instance.next();
 528 
 529                 if( ev.name.empty() && ev.id.empty() )
 530                         continue;
 531 
 532                 const std::string& event_name = ev.name;
 533                 const std::string& event_id   = ev.id;
 534 
 535                 // Clear the unit cache, since the best clearing time is hard to figure out
 536                 // due to status changes by WML. Every event will flush the cache.
 537                 unit::clear_status_caches();
 538 
 539                 { // Block for context::scoped
 540                         context::scoped inner_evc(impl_->contexts_, false);
 541                         if ( resources::lua_kernel->run_event(ev) ) {
 542                                 ++impl_->internal_wml_tracking;
 543                         }
 544                 }
 545 
 546                 assert(impl_->my_manager);
 547 
 548                 resources::gamedata->get_variable("x1") = ev.loc1.filter_loc().wml_x();
 549                 resources::gamedata->get_variable("y1") = ev.loc1.filter_loc().wml_y();
 550                 resources::gamedata->get_variable("x2") = ev.loc2.filter_loc().wml_x();
 551                 resources::gamedata->get_variable("y2") = ev.loc2.filter_loc().wml_y();
 552 
 553                 if ( event_id.empty() ) {
 554 
 555                         // Initialize an iteration over event handlers matching this event.
 556                         manager::iteration handler_iter(event_name, *impl_->my_manager);
 557 
 558                         // While there is a potential handler for this event name.
 559                         while ( handler_ptr cur_handler = *handler_iter ) {
 560                                 DBG_EH << "processing event " << event_name << " with id="<<
 561                                 cur_handler->get_config()["id"] << "\n";
 562                                 // Let this handler process our event.
 563                                 process_event(cur_handler, ev);
 564                                 // NOTE: cur_handler may be null at this point!
 565 
 566                                 ++handler_iter;
 567                         }
 568 
 569                 }
 570                 else {
 571 
 572                         //Get the handler directly via ID
 573                         handler_ptr cur_handler = impl_->my_manager->get_event_handler_by_id( event_id );
 574 
 575                         if( cur_handler ) {
 576                                 DBG_EH << "processing event " << event_name << " with id="<<
 577                                 cur_handler->get_config()["id"] << "\n";
 578                                 process_event(cur_handler, ev);
 579                         }
 580                 }
 581 
 582                 // Flush messages when finished iterating over event_handlers.
 583                 flush_messages();
 584         }
 585 
 586         if ( old_wml_track != impl_->internal_wml_tracking )
 587                 // Notify the whiteboard of any event.
 588                 // This is used to track when moves, recruits, etc. happen.
 589                 resources::whiteboard->on_gamestate_change();
 590 
 591         return context_mutated();
 592 }

 604 /**
 605  * This function can be used to detect when no WML/Lua has been executed.
 606  *
 607  * If two calls to this function return the same value, then one can
 608  * assume that the usual game mechanics have been followed, and code does
 609  * not have to account for all the things WML/Lua can do. If the return
 610  * values are different, then something unusual might have happened between
 611  * those calls.
 612  *
 613  * This is not intended as a precise metric. Rather, it is motivated by
 614  * how large the number of fired WML events is, compared to the (typical)
 615  * number of WML event handlers. It is intended for code that can benefit
 616  * from caching some aspect of the game state and that cannot rely on
 617  * [allow_undo] not being used when that state changes.
 618  */
 619 size_t wml_event_pump::wml_tracking()
 620 {
 621         return impl_->internal_wml_tracking;
 622 }
