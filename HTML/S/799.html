 123 std::string read_map(const std::string& name)
 124 {
 125         std::string res;
 126         std::string map_location = get_wml_location("maps/" + name);
 127         if(!map_location.empty()) {
 128                 res = read_file(map_location);
 129         }
 130 
 131         if (res.empty()) {
 132                 res = read_file(get_user_data_dir() + "/editor/maps/" + name);
 133         }
 134 
 135         return res;
 136 }

 138 static void get_file_tree_checksum_internal(const std::string& path, file_tree_checksum& res)
 139 {
 140 
 141         std::vector<std::string> dirs;
 142         get_files_in_dir(path,nullptr,&dirs, ENTIRE_FILE_PATH, SKIP_MEDIA_DIR, DONT_REORDER, &res);
 143 
 144         for(std::vector<std::string>::const_iterator j = dirs.begin(); j != dirs.end(); ++j) {
 145                 get_file_tree_checksum_internal(*j,res);
 146         }
 147 }

 149 const file_tree_checksum& data_tree_checksum(bool reset)
 150 {
 151         static file_tree_checksum checksum;
 152         if (reset)
 153                 checksum.reset();
 154         if(checksum.nfiles == 0) {
 155                 get_file_tree_checksum_internal("data/",checksum);
 156                 get_file_tree_checksum_internal(get_user_data_dir() + "/data/",checksum);
 157                 LOG_FS << "calculated data tree checksum: "
 158                            << checksum.nfiles << " files; "
 159                            << checksum.sum_size << " bytes" << std::endl;
 160         }
 161 
 162         return checksum;
 163 }
