  34 local function attack_filter(which, filter, is_leader)
  35     if (is_leader == nil) then is_leader = false end
  36     if (which == 'leader') then
  37         which = 'own'
  38         is_leader = true
  39     end
  40     if (which == 'own') then
  41         return {
  42             side = wesnoth.current.side,
  43             canrecruit = is_leader,
  44             { "and", filter or {} }
  45         }
  46     elseif (which == 'enemy') then
  47         return {
  48             T.filter_side { T.enemy_of { side = wesnoth.current.side } },
  49             { "and", filter or {} }
  50         }
  51     else
  52         return filter
  53     end

  58 function ca_fast_attack_utils.gamedata_setup()
  59     -- Keep game data in a table for faster access.
  60     -- This is currently re-done on every move. Could be optimized by only
  61     -- updating changes, if this is found to be critical (likely not needed).
  62 
  63     local gamedata = {}
  64 
  65     local village_map = {}
  66     for _,village in ipairs(wesnoth.get_villages()) do
  67         if (not village_map[village[1]]) then village_map[village[1]] = {} end
  68         village_map[village[1]][village[2]] = { owner = wesnoth.get_village_owner(village[1], village[2]) }
  69     end
  70     gamedata.village_map = village_map
  71 
  72     -- Only uses one leader per side right now, but only used for finding direction
  73     -- of move -> sufficient for this.
  74     gamedata.leaders = {}
  75     for _,unit_proxy in ipairs(AH.get_live_units { canrecruit = 'yes' }) do
  76         gamedata.leaders[unit_proxy.side] = { unit_proxy.x, unit_proxy.y, id = unit_proxy.id }
  77     end
  78 
  79     -- Tables that will only be populated as needed:
  80     gamedata.unit_copies = {}
  81     gamedata.unit_infos = {}
  82     gamedata.defense_maps = {}
  83 
  84     return gamedata
  85 end

  87 function ca_fast_attack_utils.single_unit_info(unit_proxy)
  88     -- Collects unit information from proxy unit table @unit_proxy into a Lua table
  89     -- so that it is accessible faster.
  90     -- Note: Even accessing the directly readable fields of a unit proxy table
  91     -- is slower than reading from a Lua table; not even talking about unit_proxy.__cfg.
  92     --
  93     -- Important: this is slow, so it should only be called as needed,
  94     -- but it does need to be redone after each move, as it contains
  95     -- information like HP and XP (or the unit might have level up or been changed
  96     -- in an event).
  97     -- Difference from the grunt rush version: also include x and y
  98 
  99     local unit_cfg = unit_proxy.__cfg
 100 
 101     local single_unit_info = {
 102         id = unit_proxy.id,
 103         canrecruit = unit_proxy.canrecruit,
 104         side = unit_proxy.side,
 105 
 106         x = unit_proxy.x,
 107         y = unit_proxy.y,
 108 
 109         hitpoints = unit_proxy.hitpoints,
 110         max_hitpoints = unit_proxy.max_hitpoints,
 111         experience = unit_proxy.experience,
 112         max_experience = unit_proxy.max_experience,
 113 
 114         alignment = unit_cfg.alignment,
 115         tod_bonus = AH.get_unit_time_of_day_bonus(unit_cfg.alignment, wesnoth.get_time_of_day().lawful_bonus),
 116         cost = unit_cfg.cost,
 117         level = unit_cfg.level
 118     }
 119 
 120     -- Include the ability type, such as: hides, heals, regenerate, skirmisher (set up as 'hides = true')
 121     local abilities = H.get_child(unit_proxy.__cfg, "abilities")
 122     if abilities then
 123         for _,ability in ipairs(abilities) do
 124             single_unit_info[ability[1]] = true
 125         end
 126     end
 127 
 128     -- Information about the attacks indexed by weapon number,
 129     -- including specials (e.g. 'poison = true')
 130     single_unit_info.attacks = {}
 131     for attack in H.child_range(unit_cfg, 'attack') do
 132         -- Extract information for specials; we do this first because some
 133         -- custom special might have the same name as one of the default scalar fields
 134         local a = {}
 135         for special in H.child_range(attack, 'specials') do
 136             for _,sp in ipairs(special) do
 137                 if (sp[1] == 'damage') then  -- this is 'backstab'
 138                     if (sp[2].id == 'backstab') then
 139                         a.backstab = true
 140                     else
 141                         if (sp[2].id == 'charge') then a.charge = true end
 142                     end
 143                 else
 144                     -- magical, marksman
 145                     if (sp[1] == 'chance_to_hit') then
 146                         a[sp[2].id] = true
 147                     else
 148                         a[sp[1]] = true
 149                     end
 150                 end
 151             end
 152         end
 153 
 154         -- Now extract the scalar (string and number) values from attack
 155         for k,v in pairs(attack) do
 156             if (type(v) == 'number') or (type(v) == 'string') then
 157                 a[k] = v
 158             end
 159         end
 160 
 161         table.insert(single_unit_info.attacks, a)
 162     end
 163 
 164     -- Resistances to the 6 default attack types
 165     local attack_types = { "arcane", "blade", "cold", "fire", "impact", "pierce" }
 166     single_unit_info.resistances = {}
 167     for _,attack_type in ipairs(attack_types) do
 168         single_unit_info.resistances[attack_type] = wesnoth.unit_resistance(unit_proxy, attack_type) / 100.
 169     end
 170 
 171     return single_unit_info
 172 end

 197 function ca_fast_attack_utils.get_unit_defense(unit_copy, x, y, defense_maps)
 198     -- Get the terrain defense of a unit as a factor (that is, e.g. 0.40 rather than 40)
 199     -- The result is stored (or accessed, if it exists) in @defense_maps
 200     --
 201     -- Inputs:
 202     -- @unit_copy: private copy of the unit (proxy table works too, but is slower)
 203     -- @x, @y: the location for which to calculate the unit's terrain defense
 204     -- @defense_maps: table in which to cache the results. Note: this is NOT an optional input
 205     --
 206     -- Sample structure of defense_maps:
 207     --   defense_maps['Vanak'][19][4] = 0.6
 208 
 209     if (not defense_maps[unit_copy.id]) then defense_maps[unit_copy.id] = {} end
 210     if (not defense_maps[unit_copy.id][x]) then defense_maps[unit_copy.id][x] = {} end
 211 
 212     if (not defense_maps[unit_copy.id][x][y]) then
 213         local defense = (100. - wesnoth.unit_defense(unit_copy, wesnoth.get_terrain(x, y))) / 100.
 214         defense_maps[unit_copy.id][x][y] = { defense = defense }
 215     end
 216 
 217     return defense_maps[unit_copy.id][x][y].defense
 218 end

 220 function ca_fast_attack_utils.is_acceptable_attack(damage_taken, damage_done, own_value_weight)
 221     -- Evaluate whether an attack is acceptable, based on the damage taken/done ratio
 222     --
 223     -- Inputs:
 224     -- @damage_taken, @damage_done: should be in gold units as returned by ca_fast_attack_utils.attack_rating
 225     --   This could be either the attacker (for taken) and defender (for done) rating of a single attack (combo)
 226     --   or the overall attack (for done) and counter attack rating (for taken)
 227     -- @own_value_weight (optional): value for the minimum ratio of done/taken that is acceptable
 228 
 229     own_value_weight = own_value_weight or 0.6 -- equivalent to aggression = 0.4 (default mainline value)
 230 
 231     -- Otherwise it depends on whether the numbers are positive or negative
 232     -- Negative damage means that one or several of the units are likely to level up
 233     if (damage_taken < 0) and (damage_done < 0) then
 234         return (damage_done / damage_taken) >= own_value_weight
 235     end
 236 
 237     if (damage_taken <= 0) then damage_taken = 0.001 end
 238     if (damage_done <= 0) then damage_done = 0.001 end
 239 
 240     return (damage_done / damage_taken) >= own_value_weight
 241 end

 243 function ca_fast_attack_utils.damage_rating_unit(attacker_info, defender_info, att_stat, def_stat, is_village, cfg)
 244     -- Calculate the rating for the damage received by a single attacker on a defender.
 245     -- The attack att_stat both for the attacker and the defender need to be precalculated for this.
 246     -- Unit information is passed in unit_infos format, rather than as unit proxy tables for speed reasons.
 247     -- Note: this is _only_ the damage rating for the attacker, not both units
 248     -- Note: damage is damage TO the attacker, not damage done BY the attacker
 249     --
 250     -- Input parameters:
 251     --  @attacker_info, @defender_info: unit_info tables produced by ca_fast_gamestate_utils.single_unit_info()
 252     --  @att_stat, @def_stat: attack statistics for the two units
 253     --  @is_village: whether the hex from which the attacker attacks is a village
 254     --    Set to nil or false if not, to anything if it is a village (does not have to be a boolean)
 255     --
 256     -- Optional parameters:
 257     --  @cfg: the optional different weights listed right below
 258     --        Note: this is currently not used in the Fast MAI, but kept in as a hook for potential upgrades
 259 
 260     local leader_weight = (cfg and cfg.leader_weight) or 2.
 261     local xp_weight = (cfg and cfg.xp_weight) or 1.
 262     local level_weight = (cfg and cfg.level_weight) or 1.
 263 
 264     local damage = attacker_info.hitpoints - att_stat.average_hp
 265 
 266     -- Count poisoned as additional 8 HP damage times probability of being poisoned
 267     if (att_stat.poisoned ~= 0) then
 268         damage = damage + 8 * (att_stat.poisoned - att_stat.hp_chance[0])
 269     end
 270     -- Count slowed as additional 4 HP damage times probability of being slowed
 271     if (att_stat.slowed ~= 0) then
 272         damage = damage + 4 * (att_stat.slowed - att_stat.hp_chance[0])
 273     end
 274 
 275     -- If attack is from a village, we count that as an 8 HP bonus
 276     if is_village then
 277         damage = damage - 8.
 278     -- Otherwise only: if attacker can regenerate, this is an 8 HP bonus
 279     elseif attacker_info.regenerate then
 280         damage = damage - 8.
 281     end
 282 
 283     if (damage < 0) then damage = 0 end
 284 
 285     -- Fractional damage (= fractional value of the attacker)
 286     local fractional_damage = damage / attacker_info.max_hitpoints
 287 
 288     -- Additionally, subtract the chance to die, in order to (de)emphasize units that might die
 289     fractional_damage = fractional_damage + att_stat.hp_chance[0]
 290 
 291     -- In addition, potentially leveling up in this attack is a huge bonus.
 292     -- we reduce the fractions damage by the chance of it happening multiplied
 293     -- by the chance of not dying itself.
 294     -- Note: this can make the fractional damage negative (as it should)
 295     local defender_level = defender_info.level
 296     if (defender_level == 0) then defender_level = 0.5 end  -- L0 units
 297 
 298     local level_bonus = 0.
 299     if (attacker_info.max_experience - attacker_info.experience <= defender_level) then
 300         level_bonus = 1. - att_stat.hp_chance[0]
 301     elseif (attacker_info.max_experience - attacker_info.experience <= defender_level * 8) then
 302         level_bonus = (1. - att_stat.hp_chance[0]) * def_stat.hp_chance[0]
 303     end
 304 
 305     fractional_damage = fractional_damage - level_bonus * level_weight
 306 
 307     -- Now convert this into gold-equivalent value
 308     local value = attacker_info.cost
 309 
 310     -- If this is the side leader, make damage to it much more important
 311     if attacker_info.canrecruit then
 312         value = value * leader_weight
 313     end
 314 
 315     -- Being closer to leveling makes the attacker more valuable
 316     -- TODO: consider using a more precise measure here
 317     local xp_bonus = attacker_info.experience / attacker_info.max_experience
 318     value = value * (1. + xp_bonus * xp_weight)
 319 
 320     local rating = fractional_damage * value
 321 
 322     return rating
 323 end

 325 function ca_fast_attack_utils.attack_rating(attacker_infos, defender_info, dsts, att_stats, def_stat, gamedata, cfg)
 326     -- Returns a common (but configurable) rating for attacks of one or several attackers against one defender
 327     --
 328     -- Inputs:
 329     --  @attackers_infos: input array of attacker unit_info tables (must be an array even for single unit attacks)
 330     --  @defender_info: defender unit_info
 331     --  @dsts: array of attack locations in form { x, y } (must be an array even for single unit attacks)
 332     --  @att_stats: array of the attack stats of the attack combination(!) of the attackers
 333     --    (must be an array even for single unit attacks)
 334     --  @def_stat: the combat stats of the defender after facing the combination of the attackers
 335     --  @gamedata: table with the game state as produced by ca_fast_gamestate_utils.gamedata()
 336     --
 337     -- Optional inputs:
 338     --  @cfg: the different weights listed right below
 339     --        Note: this is currently not used in the Fast MAI, but kept in as a hook for potential upgrades
 340     --
 341     -- Returns:
 342     --   - Overall rating for the attack or attack combo
 343     --   - Attacker rating: the sum of all the attacker damage ratings
 344     --   - Defender rating: the combined defender damage rating
 345     --   - Extra rating: additional ratings that do not directly describe damage
 346     --       This should be used to help decide which attack to pick,
 347     --       but not for, e.g., evaluating counter attacks (which should be entirely damage based)
 348     --   Note: rating = defender_rating - attacker_rating * own_value_weight + extra_rating
 349 
 350     -- Set up the config parameters for the rating
 351     local defender_starting_damage_weight = (cfg and cfg.defender_starting_damage_weight) or 0.33
 352     local defense_weight = (cfg and cfg.defense_weight) or 0.1
 353     local distance_leader_weight = (cfg and cfg.distance_leader_weight) or 0.002
 354     local occupied_hex_penalty = (cfg and cfg.occupied_hex_penalty) or 0.001
 355     local own_value_weight = (cfg and cfg.own_value_weight) or 1.0
 356 
 357     local attacker_rating = 0
 358     for i,attacker_info in ipairs(attacker_infos) do
 359         local attacker_on_village = gamedata.village_map[dsts[i][1]] and gamedata.village_map[dsts[i][1]][dsts[i][2]]
 360         attacker_rating = attacker_rating + ca_fast_attack_utils.damage_rating_unit(
 361             attacker_info, defender_info, att_stats[i], def_stat, attacker_on_village, cfg
 362         )
 363     end
 364 
 365     -- attacker_info is passed only to figure out whether the attacker might level up
 366     -- TODO: This is only works for the first attacker in a combo at the moment
 367     local defender_x, defender_y = defender_info.x, defender_info.y
 368     local defender_on_village = gamedata.village_map[defender_x] and gamedata.village_map[defender_x][defender_y]
 369     local defender_rating = ca_fast_attack_utils.damage_rating_unit(
 370         defender_info, attacker_infos[1], def_stat, att_stats[1], defender_on_village, cfg
 371     )
 372 
 373     -- Now we add some extra ratings. They are positive for attacks that should be preferred
 374     -- and expressed in fraction of the defender maximum hitpoints
 375     -- They should be used to help decide which attack to pick all else being equal,
 376     -- but not for, e.g., evaluating counter attacks (which should be entirely damage based)
 377     local extra_rating = 0.
 378 
 379     -- Prefer to attack already damaged enemies
 380     local defender_starting_damage_fraction = defender_info.max_hitpoints - defender_info.hitpoints
 381     extra_rating = extra_rating + defender_starting_damage_fraction * defender_starting_damage_weight
 382 
 383     -- If defender is on a village, add a bonus rating (we want to get rid of those preferentially)
 384     -- This is in addition to the damage bonus already included above (but as an extra rating)
 385     if defender_on_village then
 386         extra_rating = extra_rating + 10.
 387     end
 388 
 389     -- Normalize so that it is in fraction of defender max_hitpoints
 390     extra_rating = extra_rating / defender_info.max_hitpoints
 391 
 392     -- We don't need a bonus for good terrain for the attacker, as that is covered in the damage calculation
 393     -- However, we add a small bonus for good terrain defense of the _defender_ on the _attack_ hexes
 394     -- This is in order to take good terrain away from defender on its next move
 395     local defense_rating = 0.
 396     for _,dst in ipairs(dsts) do
 397         defense_rating = defense_rating + ca_fast_attack_utils.get_unit_defense(
 398             ca_fast_attack_utils.get_unit_copy(defender_info.id, gamedata),
 399             dst[1], dst[2],
 400             gamedata.defense_maps
 401         )
 402     end
 403     defense_rating = defense_rating / #dsts * defense_weight
 404 
 405     extra_rating = extra_rating + defense_rating
 406 
 407     -- Get a very small bonus for hexes in between defender and AI leader
 408     -- 'relative_distances' is larger for attack hexes closer to the side leader (possible values: -1 .. 1)
 409     if gamedata.leaders[attacker_infos[1].side] then
 410         local leader_x, leader_y = gamedata.leaders[attacker_infos[1].side][1], gamedata.leaders[attacker_infos[1].side][2]
 411 
 412         local rel_dist_rating = 0.
 413         for _,dst in ipairs(dsts) do
 414             local relative_distance =
 415                 M.distance_between(defender_x, defender_y, leader_x, leader_y)
 416                 - M.distance_between(dst[1], dst[2], leader_x, leader_y)
 417             rel_dist_rating = rel_dist_rating + relative_distance
 418         end
 419         rel_dist_rating = rel_dist_rating / #dsts * distance_leader_weight
 420 
 421         extra_rating = extra_rating + rel_dist_rating
 422     end
 423 
 424     -- Finally add up and apply factor of own unit weight to defender unit weight
 425     -- This is a number equivalent to 'aggression' in the default AI (but not numerically equal)
 426     local rating = defender_rating - attacker_rating * own_value_weight + extra_rating
 427 
 428     return rating, attacker_rating, defender_rating, extra_rating
 429 end

 431 function ca_fast_attack_utils.battle_outcome(attacker_copy, defender_proxy, dst, attacker_info, defender_info, gamedata, move_cache)
 432     -- Calculate the stats of a combat by @attacker_copy vs. @defender_proxy at location @dst
 433     -- We use wesnoth.simulate_combat for this, but cache results when possible
 434     -- Inputs:
 435     -- @attacker_copy: private unit copy of the attacker (must be a copy, does not work with the proxy table)
 436     -- @defender_proxy: defender proxy table (must be a unit proxy table on the map, does not work with a copy)
 437     -- @dst: location from which the attacker will attack in form { x, y }
 438     -- @attacker_info, @defender_info: unit info for the two units (needed in addition to the units
 439     --   themselves in order to speed things up)
 440     --  @gamedata: table with the game state as produced by ca_fast_gamestate_utils.gamedata()
 441     --  @move_cache: for caching data *for this move only*, needs to be cleared after a gamestate change
 442 
 443     local defender_defense = ca_fast_attack_utils.get_unit_defense(defender_proxy, defender_proxy.x, defender_proxy.y, gamedata.defense_maps)
 444     local attacker_defense = ca_fast_attack_utils.get_unit_defense(attacker_copy, dst[1], dst[2], gamedata.defense_maps)
 445 
 446     if move_cache[attacker_info.id]
 447         and move_cache[attacker_info.id][defender_info.id]
 448         and move_cache[attacker_info.id][defender_info.id][attacker_defense]
 449         and move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense]
 450         and move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints]
 451         and move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints][defender_info.hitpoints]
 452     then
 453         return move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints][defender_info.hitpoints].att_stat,
 454             move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints][defender_info.hitpoints].def_stat
 455     end
 456 
 457     local old_x, old_y = attacker_copy.x, attacker_copy.y
 458     attacker_copy.x, attacker_copy.y = dst[1], dst[2]
 459     local tmp_att_stat, tmp_def_stat = wesnoth.simulate_combat(attacker_copy, defender_proxy)
 460     attacker_copy.x, attacker_copy.y = old_x, old_y
 461 
 462     -- Extract only those hp_chances that are non-zero (except for hp_chance[0]
 463     -- which is always needed). This slows down this step a little, but significantly speeds
 464     -- up attack combination calculations
 465     local att_stat = {
 466         hp_chance = {},
 467         average_hp = tmp_att_stat.average_hp,
 468         poisoned = tmp_att_stat.poisoned,
 469         slowed = tmp_att_stat.slowed
 470     }
 471 
 472     att_stat.hp_chance[0] = tmp_att_stat.hp_chance[0]
 473     for i = 1,#tmp_att_stat.hp_chance do
 474         if (tmp_att_stat.hp_chance[i] ~= 0) then
 475             att_stat.hp_chance[i] = tmp_att_stat.hp_chance[i]
 476         end
 477     end
 478 
 479     local def_stat = {
 480         hp_chance = {},
 481         average_hp = tmp_def_stat.average_hp,
 482         poisoned = tmp_def_stat.poisoned,
 483         slowed = tmp_def_stat.slowed
 484     }
 485 
 486     def_stat.hp_chance[0] = tmp_def_stat.hp_chance[0]
 487     for i = 1,#tmp_def_stat.hp_chance do
 488         if (tmp_def_stat.hp_chance[i] ~= 0) then
 489             def_stat.hp_chance[i] = tmp_def_stat.hp_chance[i]
 490         end
 491     end
 492 
 493     if (not move_cache[attacker_info.id]) then
 494         move_cache[attacker_info.id] = {}
 495     end
 496     if (not move_cache[attacker_info.id][defender_info.id]) then
 497         move_cache[attacker_info.id][defender_info.id] = {}
 498     end
 499     if (not move_cache[attacker_info.id][defender_info.id][attacker_defense]) then
 500         move_cache[attacker_info.id][defender_info.id][attacker_defense] = {}
 501     end
 502     if (not move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense]) then
 503         move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense] = {}
 504     end
 505     if (not move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints]) then
 506         move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints] = {}
 507     end
 508 
 509     move_cache[attacker_info.id][defender_info.id][attacker_defense][defender_defense][attacker_info.hitpoints][defender_info.hitpoints]
 510         = { att_stat = att_stat, def_stat = def_stat }
 511 
 512     return att_stat, def_stat
 513 end
