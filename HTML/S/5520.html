 170 local function remove_ssf_info_from(cfg)
 171         cfg.side = nil
 172         cfg.team_name = nil
 173         cfg.silent = nil -- Not technically SSF info, but still unwanted
 174         for i, v in ipairs(cfg) do
 175                 if v[1] == "has_unit" or v[1] == "enemy_of" or v[1] == "allied_with" then
 176                         table.remove(cfg, i)
 177                 end
 178         end

 181 function wml_actions.objectives(cfg)
 182         if cfg.delayed_variable_substitution ~= true then
 183                 cfg = helper.parsed(cfg)
 184         end
 185 
 186         local sides = wesnoth.get_sides(cfg)
 187         local silent = cfg.silent
 188 
 189         local objectives = generate_objectives(cfg)
 190         local function set_objectives(sides, save)
 191                 local cfg2 = helper.literal(cfg)
 192                 remove_ssf_info_from(cfg2)
 193                 for i, team in ipairs(sides) do
 194                         if save then scenario_objectives[team.side] = cfg2 end
 195                         team.objectives = objectives
 196                         team.objectives_changed = not silent
 197                 end
 198         end
 199         if #sides == #wesnoth.sides or #sides == 0 then
 200                 scenario_objectives[0] = helper.literal(cfg)
 201                 remove_ssf_info_from(scenario_objectives[0])
 202                 set_objectives(wesnoth.sides)
 203         else
 204                 set_objectives(sides, true)
 205         end
 206 end

 208 local function maybe_parsed(cfg)
 209         if cfg == nil then return nil end
 210         if cfg.delayed_variable_substitution == true then
 211                 return wesnoth.tovconfig(cfg)
 212         end
 213         return cfg
 214 end

 216 function wml_actions.show_objectives(cfg)
 217         local cfg0 = maybe_parsed(scenario_objectives[0])
 218         local function local_show_objectives(sides)
 219                 local objectives0 = cfg0 and generate_objectives(cfg0)
 220                 for i, team in ipairs(sides) do
 221                         cfg = maybe_parsed(scenario_objectives[team.side])
 222                         local objectives = (cfg and generate_objectives(cfg)) or objectives0
 223                         if objectives then team.objectives = objectives end
 224                         team.objectives_changed = true
 225                 end
 226         end
 227         local sides = wesnoth.get_sides(cfg)
 228         if #sides == 0 then
 229                 local_show_objectives(wesnoth.sides)
 230         else
 231                 local_show_objectives(sides)
 232         end
 233 end
