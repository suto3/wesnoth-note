  76 loading_screen::loading_screen(std::function<void()> f)
  77         : window_(nullptr)
  78         , timer_id_(0)
  79         , animation_counter_(0)
  80         , work_(f)
  81         , worker_()
  82         , cursor_setter_()
  83         , progress_stage_label_(nullptr)
  84         , animation_label_(nullptr)
  85         , current_stage_(nullptr)
  86         , visible_stages_()
  87         , animation_stages_()
  88         , current_visible_stage_()
  89         , is_worker_running_(false)
  90 {
  91         for (const auto& pair : stages) {
  92                 visible_stages_[pair.first] = t_string(pair.second, "wesnoth-lib") + "...";
  93         }
  94         for (int i = 0; i != 20; ++i) {
  95                 std::string s(20, ' ');
  96                 s[i] = '.';
  97                 animation_stages_.push_back(s);
  98         }
  99         current_visible_stage_ = visible_stages_.end();
 100         current_load = this;
 101 }

 117 void loading_screen::pre_show(window& window)
 118 {
 119         if (work_) {
 120                 worker_.reset(new boost::thread([this]() {
 121                         is_worker_running_ = true;
 122                         try {
 123                                 work_();
 124                         } catch(...) {
 125                                 //TODO: guard this with a mutex.
 126                                 exception_ = std::current_exception();
 127                         }
 128                         is_worker_running_ = false;
 129                 }));
 130         }
 131         timer_id_ = add_timer(100, std::bind(&loading_screen::timer_callback, this, std::ref(window)), true);
 132         cursor_setter_.reset(new cursor::setter(cursor::WAIT));
 133         progress_stage_label_ = &find_widget<label>(&window, "status", false);
 134         animation_label_ = &find_widget<label>(&window, "test_animation", false);
 135 
 136         window.set_enter_disabled(true);
 137         window.set_escape_disabled(true);
 138 }

 140 void loading_screen::post_show(window& /*window*/)

 144         cursor_setter_.reset();

 147 void loading_screen::progress(const char* stage)
 148 {
 149         if(!current_load) {
 150                 return;
 151         }
 152         if(stage) {
 153                 current_load->current_stage_
 154 #if defined(_MSC_VER) && _MSC_VER < 1900
 155                         = stage;

 157                         .store(stage, std::memory_order_release);
 158 #endif
 159         }

 164 void loading_screen::timer_callback(window& window)
 165 {
 166         if (!work_ || !worker_ || worker_->timed_join(boost::posix_time::milliseconds(0))) {
 167                 if (exception_) {
 168                         clear_timer();
 169                         std::rethrow_exception(exception_);
 170                 }
 171                 window.close();
 172         }
 173         if (!work_) {
 174                 return;
 175         }
 176         const char* stage = current_stage_
 177 #if defined(_MSC_VER) && _MSC_VER < 1900
 178                 ;
 179 #else
 180                 .load(std::memory_order_acquire);
 181 #endif
 182         if (stage && (current_visible_stage_ == visible_stages_.end() || stage != current_visible_stage_->first))
 183         {
 184                 auto iter = visible_stages_.find(stage);
 185                 if(iter == visible_stages_.end()) {
 186                         WRN_LS << "Stage ID '" << stage << "' missing description." << std::endl;
 187                         return;
 188                 }
 189                 current_visible_stage_ = iter;
 190                 progress_stage_label_->set_label(iter->second);
 191         }
 192         ++animation_counter_;
 193         if (animation_counter_ % 2 == 0) {
 194                 animation_label_->set_label(animation_stages_[(animation_counter_ / 2) % animation_stages_.size()]);
 195         }
 196 }

 198 loading_screen::~loading_screen()
 199 {
 200         if (is_worker_running_) {
 201                 // The worker thread is running, exit the application to prevent memory corruption.
 202                 // TODO: this is still not optimal, the main problem is that this code here assumes
 203                 // that this happened becasue the window was closed which is not necessarily the case
 204                 // (other possibilities migth be a 'dialog doesn't fit on screen' exception casued by resizing the window)
 205 
 206                 // Another approach migth be to add exit points ( boost::this_thread::interruption_point() ) to the worker
 207                 // functions (filesystem.cpp config parsing code etc. ) and then use that to end the thread faster.
 208                 std::exit(0);
 209         }
 210         clear_timer();
 211         close();
 212         current_load = nullptr;
 213 }

 215 void loading_screen::display(CVideo& video, std::function<void()> f)
 216 {
 217         const bool use_loadingscreen_animation = !preferences::disable_loadingscreen_animation();
 218         if (current_load || video.faked()) {
 219                 f();
 220         }
 221         else if(use_loadingscreen_animation) {
 222                 loading_screen(f).show(video);
 223         }
 224         else {
 225                 loading_screen(std::function<void()>()).show(video);
 226                 f();
 227         }
 228 }
 229 void loading_screen::clear_timer()
 230 {
 231         if (timer_id_ != 0) {
 232                 remove_timer(timer_id_);
 233                 timer_id_ = 0;
 234         }
