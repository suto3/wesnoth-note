  47 /**
  48  * Build the event name associated with the given menu item id.
  49  * This is a separate function so it can be easily shared by multiple
  50  * constructors.
  51  */
  52 inline std::string make_item_name(const std::string& id)
  53 {
  54         return std::string("menu item") + (id.empty() ? "" : ' ' + id);

 141 bool wml_menu_item::can_show(const map_location& hex, const game_data& data, filter_context& filter_con) const
 142 {
 143         // Failing the [show_if] tag means no show.
 144         if(!show_if_.empty() && !conditional_passed(show_if_)) {
 145                 return false;
 146         }
 147 
 148         // Failing the [fiter_location] tag means no show.
 149         if(!filter_location_.empty() && !terrain_filter(filter_location_, &filter_con)(hex)) {
 150                 return false;
 151         }
 152 
 153         // Failing to have a required selection means no show.
 154         if(needs_select_ && !data.last_selected.valid()) {
 155                 return false;
 156         }
 157 
 158         // Passed all tests.
 159         return true;
 160 }

 162 void wml_menu_item::fire_event(const map_location& event_hex, const game_data& data) const
 163 {
 164         if(!this->is_synced()) {
 165                 // It is possible to for example show a help menu during a [delay] of a synced event.
 166                 set_scontext_unsynced leave_synced_context;
 167                 assert(resources::game_events != nullptr);
 168                 resources::game_events->pump().fire(event_name_, event_hex);
 169                 return;
 170         }
 171 
 172         const map_location& last_select = data.last_selected;
 173 
 174         // No new player-issued commands allowed while this is firing.
 175         const events::command_disabler disable_commands;
 176 
 177         // instead of adding a second "select" event like it was done before, we just fire the select event again, and this
 178         // time in a synced context.
 179         // note that there couldn't be a user choice during the last "select" event because it didn't run in a synced
 180         // context.
 181         if(needs_select_ && last_select.valid()) {
 182                 synced_context::run_and_throw(
 183                         "fire_event", replay_helper::get_event(event_name_, event_hex, &last_select));
 184         } else {
 185                 synced_context::run_in_synced_context_if_not_already(
 186                         "fire_event", replay_helper::get_event(event_name_, event_hex, nullptr));
 187         }
 188 }

 190 void wml_menu_item::finish_handler() const
 191 {
 192         if(!command_.empty()) {
 193                 assert(resources::game_events);
 194                 resources::game_events->remove_event_handler(command_["id"]);
 195         }
 196 
 197         // Hotkey support
 198         if(use_hotkey_) {
 199                 hotkey::remove_wml_hotkey(hotkey_id_);
 200         }
 201 }

 203 void wml_menu_item::init_handler() const
 204 {
 205         // If this menu item has a [command], add a handler for it.
 206         if(!command_.empty()) {
 207                 assert(resources::game_events);
 208                 resources::game_events->add_event_handler(command_, true);
 209         }
 210 
 211         // Hotkey support
 212         if(use_hotkey_) {
 213                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 214         }

 217 void wml_menu_item::to_config(config& cfg) const
 218 {
 219         cfg["id"] = item_id_;
 220         cfg["image"] = image_;
 221         cfg["description"] = description_;
 222         cfg["needs_select"] = needs_select_;
 223         cfg["synced"] = is_synced_;
 224 
 225         if(use_hotkey_ && use_wml_menu_) {
 226                 cfg["use_hotkey"] = true;
 227         }
 228 
 229         if(use_hotkey_ && !use_wml_menu_) {
 230                 cfg["use_hotkey"] = "only";
 231         }
 232 
 233         if(!use_hotkey_ && use_wml_menu_) {
 234                 cfg["use_hotkey"] = false;
 235         }
 236 
 237         if(!use_hotkey_ && !use_wml_menu_) {
 238                 ERR_NG << "Bad data: wml_menu_item with both use_wml_menu and use_hotkey set to false is not supposed to be "
 239                                   "possible.";
 240                 cfg["use_hotkey"] = false;
 241         }
 242 
 243         if(!show_if_.empty()) {
 244                 cfg.add_child("show_if", show_if_.get_config());
 245         }
 246 
 247         if(!filter_location_.empty()) {
 248                 cfg.add_child("filter_location", filter_location_.get_config());
 249         }
 250 
 251         if(!command_.empty()) {
 252                 cfg.add_child("command", command_);
 253         }
 254 
 255         if(!default_hotkey_.empty()) {
 256                 cfg.add_child("default_hotkey", default_hotkey_);
 257         }
 258 }

 260 void wml_menu_item::update(const vconfig& vcfg)
 261 {
 262         const bool old_use_hotkey = use_hotkey_;
 263         // Tracks whether or not the hotkey has been updated.
 264         bool hotkey_updated = false;
 265 
 266         if(vcfg.has_attribute("image")) {
 267                 image_ = vcfg["image"].str();
 268         }
 269 
 270         if(vcfg.has_attribute("description")) {
 271                 description_ = vcfg["description"].t_str();
 272                 hotkey_updated = true;
 273         }
 274 
 275         if(vcfg.has_attribute("needs_select")) {
 276                 ERR_NG << "needs_select= is deprecated\n";
 277                 needs_select_ = vcfg["needs_select"].to_bool();
 278         }
 279 
 280         if(vcfg.has_attribute("synced")) {
 281                 is_synced_ = vcfg["synced"].to_bool(true);
 282         }
 283 
 284         if(const vconfig& child = vcfg.child("show_if")) {
 285                 show_if_ = child;
 286                 show_if_.make_safe();
 287         }
 288 
 289         if(const vconfig& child = vcfg.child("filter_location")) {
 290                 filter_location_ = child;
 291                 filter_location_.make_safe();
 292         }
 293 
 294         if(const vconfig& child = vcfg.child("default_hotkey")) {
 295                 default_hotkey_ = child.get_parsed_config();
 296                 hotkey_updated = true;
 297         }
 298 
 299         if(vcfg.has_attribute("use_hotkey")) {
 300                 const config::attribute_value& use_hotkey_av = vcfg["use_hotkey"];
 301 
 302                 use_hotkey_ = use_hotkey_av.to_bool(true);
 303                 use_wml_menu_ = use_hotkey_av.str() != "only";
 304         }
 305 
 306         if(const vconfig& cmd = vcfg.child("command")) {
 307                 const bool delayed = cmd["delayed_variable_substitution"].to_bool(true);
 308                 update_command(delayed ? cmd.get_config() : cmd.get_parsed_config());
 309         }
 310 
 311         // Update the registered hotkey?
 312 
 313         if(use_hotkey_ && !old_use_hotkey) {
 314                 // The hotkey needs to be enabled.
 315                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 316 
 317         } else if(use_hotkey_ && hotkey_updated) {
 318                 // The hotkey needs to be updated.
 319                 hotkey::add_wml_hotkey(hotkey_id_, description_, default_hotkey_);
 320 
 321         } else if(!use_hotkey_ && old_use_hotkey) {
 322                 // The hotkey needs to be disabled.
 323                 hotkey::remove_wml_hotkey(hotkey_id_);
 324         }
 325 }

 327 void wml_menu_item::update_command(const config& new_command)
 328 {
 329         // If there is an old command, remove it from the event handlers.
 330         if(!command_.empty()) {
 331                 assert(resources::game_events);
 332                 manager::iteration iter(event_name_, *resources::game_events);
 333                 while(handler_ptr hand = *iter) {
 334                         if(hand->is_menu_item()) {
 335                                 LOG_NG << "Removing command for " << event_name_ << ".\n";
 336                                 resources::game_events->remove_event_handler(command_["id"].str());
 337                         }
 338                         ++iter;
 339                 }
 340         }
 341 
 342         // Update our stored command.
 343         if(new_command.empty()) {
 344                 command_.clear();
 345         } else {
 346                 command_ = new_command;
 347 
 348                 // Set some fields required by event processing.
 349                 config::attribute_value& event_id = command_["id"];
 350                 if(event_id.empty() && !item_id_.empty()) {
 351                         event_id = item_id_;
 352                 }
 353 
 354                 command_["name"] = event_name_;
 355                 command_["first_time_only"] = false;
 356 
 357                 // Register the event.
 358                 LOG_NG << "Setting command for " << event_name_ << " to:\n" << command_;
 359                 assert(resources::game_events);
 360                 resources::game_events->add_event_handler(command_, true);
 361         }
 362 }
