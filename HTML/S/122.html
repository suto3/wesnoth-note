  83 /**
  84  * Splits a string based either on a separator, except then the text appears within specified parenthesis.
  85  *
  86  * If the separator is "0" (default), it splits a string into an odd number of parts:
  87  * - The part before the first '(',
  88  * - the part between the first '('
  89  * - and the matching right ')', etc ...
  90  * and the remainder of the string.
  91  *
  92  * Note that one can use the same character for both the left and right parenthesis, which usually makes
  93  * the most sense for this function.
  94  *
  95  * Note that this will find the first matching char in the left string and match against the corresponding
  96  * char in the right string. A correctly processed string should return a vector with an odd number of
  97  * elements. Empty elements are never removed as they are placeholders, hence REMOVE EMPTY only works for
  98  * the separator split.

 100  * INPUT:   ("a(b)c{d}e(f{g})h", 0, "({", ")}")
 101  * RETURNS: {"a", "b", "c", "d", "e", "f{g}", "h"}
 102  */

 110 /**
 111  * Similar to parenthetical_split, but also expands embedded square brackets.
 112  *
 113  * Notes:
 114  * - The Separator must be specified and number of entries in each square bracket must match in each section.
 115  * - Leading zeros are preserved if specified between square brackets.
 116  * - An asterisk as in [a*n] indicates to expand 'a' n times
 117  *
 118  * This is useful for expanding animation WML code.
 119  *
 120  * Examples:
 121  *
 122  * INPUT:   ("a[1-3](1,[5,6,7]),b[8,9]", ",")
 123  * RETURNS: {"a1(1,5)", "a2(1,6)", "a3(1,7)", "b8", "b9"}
 124  *
 125  * INPUT:   ("abc[07-10]")
 126  * RETURNS: {"abc07", "abc08", "abc09", "abc10"}
 127  *
 128  * INPUT:   ("a[1,2]b[3-4]:c[5,6]")
 129  * RETURNS: {"a1b3:c5", "a2b4:c6"}
 130  *
 131  * INPUT:   ("abc[3,1].png")
 132  * RETURNS: {"abc3.png", "abc2.png", "abc1.png"}
 133  *
 134  * INPUT:   ("abc[de,xyz]")
 135  * RETURNS: {"abcde", "abcxyz"}
 136  *
 137  * INPUT:   ("abc[1*3]")

 147 /**
 148  * Generates a new string joining container items in a list.
 149  *
 150  * @param v A container with elements.
 151  * @param s List delimiter.
 152  */
 153 template <typename T>
 154 std::string join(const T& v, const std::string& s = ",")
 155 {
 156         std::stringstream str;
 157 
 158         for(typename T::const_iterator i = v.begin(); i != v.end(); ++i) {
 159                 str << *i;
 160                 if(std::next(i) != v.end()) {
 161                         str << s;
 162                 }
 163         }
 164 
 165         return str.str();
 166 }

 168 template <typename T>
 169 std::string join_map(
 170         const T& v,
 171         const std::string& major = ",",
 172         const std::string& minor = ":")
 173 {
 174         std::stringstream str;
 175 
 176         for(typename T::const_iterator i = v.begin(); i != v.end(); ++i) {
 177                 str << i->first << minor << i->second;
 178                 if(std::next(i) != v.end()) {
 179                         str << major;
 180                 }
 181         }
 182 
 183         return str.str();
 184 }

 186 /**
 187  * Generates a new string containing a bullet list.
 188  *
 189  * List items are preceded by the indentation blanks, a bullet string and
 190  * another blank; all but the last item are followed by a newline.
 191  *
 192  * @param v A container with elements.
 193  * @param indent Number of indentation blanks.
 194  * @param bullet The leading bullet string.
 195  */
 196 template<typename T>
 197 std::string bullet_list(const T& v, size_t indent = 4, const std::string& bullet = font::unicode_bullet)
 198 {
 199         std::ostringstream str;
 200 
 201         for(typename T::const_iterator i = v.begin(); i != v.end(); ++i) {
 202                 if(i != v.begin()) {
 203                         str << '\n';
 204                 }
 205 
 206                 str << std::string(indent, ' ') << bullet << ' ' << *i;
 207         }
 208 
 209         return str.str();
 210 }
