 134 void menu_button::signal_handler_left_button_click(const event::ui_event event, bool& handled)
 135 {
 136         assert(get_window());
 137         DBG_GUI_E << LOG_HEADER << ' ' << event << ".\n";
 138 
 139         sound::play_UI_sound(settings::sound_button_click);
 140 
 141         // If a button has a retval do the default handling.
 142         dialogs::drop_down_menu droplist(this->get_rectangle(), this->values_, this->selected_, this->get_use_markup(), this->keep_open_,
 143                 nullptr);
 144 
 145         droplist_ = &droplist;
 146 
 147         if(droplist.show(get_window()->video())) {
 148                 droplist_ = nullptr;
 149 
 150                 const int selected = droplist.selected_item();
 151 
 152                 // Saftey check. If the user clicks a selection in the dropdown and moves their mouse away too
 153                 // quickly, selected_ could be set to -1. This returns in that case, preventing crashes.
 154                 if(selected < 0) {
 155                         return;
 156                 }
 157 
 158                 selected_ = selected;
 159 
 160                 this->set_label(values_[selected_]["label"]);
 161 
 162                 fire(event::NOTIFY_MODIFIED, *this, nullptr);
 163 
 164                 if(callback_state_change_) {
 165                         callback_state_change_(*this);
 166                 }
 167 
 168                 if(retval_ != 0) {
 169                         if(window* window = get_window()) {
 170                                 window->set_retval(retval_);
 171                                 return;
 172                         }
 173                 }
 174         }
 175 
 176         droplist_ = nullptr;
 177 
 178         handled = true;
 179 }

 181 void menu_button::set_values(const std::vector<::config>& values, int selected)
 182 {
 183         assert(static_cast<size_t>(selected) < values.size());
 184         assert(static_cast<size_t>(selected_) < values_.size());
 185 
 186         if(values[selected]["label"] != values_[selected_]["label"]) {
 187                 set_is_dirty(true);
 188         }
 189 
 190         values_ = values;
 191         selected_ = selected;
 192 
 193         set_label(values_[selected_]["label"]);
 194 }

 200 
 201         if(selected != selected_) {

 203         }
 204 
 205         selected_ = selected;
 206 
 207         set_label(values_[selected_]["label"]);
 208 }
