  77 /**
  78  * Small helper to keep a resource (boolean) locked.
  79  *
  80  * Some of the event handling routines can't be called recursively, this due to
  81  * the fact that they are attached to the pre queue and when the forward an
  82  * event the pre queue event gets triggered recursively causing infinite
  83  * recursion.
  84  *
  85  * To prevent that those functions check the lock and exit when the lock is
  86  * held otherwise grab the lock here.
  87  */
  88 class resource_locker
  89 {
  90 public:
  91         resource_locker(bool& locked) : locked_(locked)
  92         {
  93                 assert(!locked_);
  94                 locked_ = true;
  95         }
  96 
  97         ~resource_locker()
  98         {
  99                 assert(locked_);
 100                 locked_ = false;
 101         }

 160 void mouse_motion::signal_handler_sdl_mouse_motion(const event::ui_event event,
 161                                                                                                         bool& handled,
 162                                                                                                         const point& coordinate)
 163 {
 164         if(signal_handler_sdl_mouse_motion_entered_) {
 165                 return;
 166         }
 167         resource_locker lock(signal_handler_sdl_mouse_motion_entered_);
 168 
 169         DBG_GUI_E << LOG_HEADER << event << ".\n";
 170 
 171         if(mouse_captured_) {
 172                 assert(mouse_focus_);
 173                 if(!owner_.fire(event, *mouse_focus_, coordinate)) {
 174                         mouse_hover(mouse_focus_, coordinate);
 175                 }
 176         } else {
 177                 widget* mouse_over = owner_.find_at(coordinate, true);
 178                 while(mouse_over && !mouse_over->can_mouse_focus() && mouse_over->parent()) {
 179                         mouse_over = mouse_over->parent();
 180                 }
 181                 if(mouse_over) {
 182                         DBG_GUI_E << LOG_HEADER << "Firing: " << event << ".\n";
 183                         if(owner_.fire(event, *mouse_over, coordinate)) {
 184                                 return;
 185                         }
 186                 }
 187 
 188                 if(!mouse_focus_ && mouse_over) {
 189                         mouse_enter(mouse_over);
 190                 } else if(mouse_focus_ && !mouse_over) {
 191                         mouse_leave();
 192                 } else if(mouse_focus_ && mouse_focus_ == mouse_over) {
 193                         mouse_hover(mouse_over, coordinate);
 194                 } else if(mouse_focus_ && mouse_over) {
 195                         // moved from one widget to the next
 196                         mouse_leave();
 197                         mouse_enter(mouse_over);
 198                 } else {
 199                         assert(!mouse_focus_ && !mouse_over);
 200                 }
 201         }
 202         handled = true;
 203 }

 278 void mouse_motion::show_tooltip()
 279 {
 280         DBG_GUI_E << LOG_HEADER << "Firing: " << event::SHOW_TOOLTIP << ".\n";
 281 
 282         if(!hover_widget_) {
 283                 // See mouse_motion::stop_hover_timer.
 284                 ERR_GUI_E << LOG_HEADER << event::SHOW_TOOLTIP
 285                                   << " bailing out, no hover widget.\n";
 286                 return;
 287         }
 288 
 289         /*
 290          * Ignore the result of the event, always mark the tooltip as shown. If
 291          * there was no handler, there is no reason to assume there will be one
 292          * next time.
 293          */
 294         owner_.fire(SHOW_TOOLTIP, *hover_widget_, hover_position_);
 295 
 296         hover_shown_ = true;
 297 
 298         hover_timer_ = 0;
 299         hover_widget_ = nullptr;
 300         hover_position_ = point();
 301 }

 319 void mouse_motion::start_hover_timer(widget* widget, const point& coordinate)
 320 {
 321         assert(widget);
 322         stop_hover_timer();
 323 
 324         if(hover_shown_ || !widget->wants_mouse_hover()) {
 325                 return;
 326         }
 327 
 328         DBG_GUI_E << LOG_HEADER << "Start hover timer for widget '" << widget->id()
 329                           << "' at address " << widget << ".\n";
 330 
 331         hover_timer_
 332                         = add_timer(50, std::bind(&mouse_motion::show_tooltip, this));
 333 
 334         if(hover_timer_) {
 335                 hover_widget_ = widget;
 336                 hover_position_ = coordinate;
 337         } else {
 338                 ERR_GUI_E << LOG_HEADER << "Failed to add hover timer." << std::endl;
 339         }
 340 }

 404 template<typename T>
 405 void mouse_button<T>::signal_handler_sdl_button_down(const event::ui_event event, bool& handled,
 406                 const point& coordinate)
 407 {
 408         if(signal_handler_sdl_button_down_entered_) {
 409                 return;
 410         }
 411         resource_locker lock(signal_handler_sdl_button_down_entered_);
 412 
 413         DBG_GUI_E << LOG_HEADER << event << ".\n";
 414 
 415         if(is_down_) {
 416 #ifdef GUI2_SHOW_UNHANDLED_EVENT_WARNINGS
 417                 WRN_GUI_E << LOG_HEADER << event
 418                                   << ". The mouse button is already down, "
 419                                   << "we missed an event.\n";
 420 #endif
 421                 return;
 422         }
 423         is_down_ = true;
 424 
 425         if(mouse_captured_) {
 426                 assert(mouse_focus_);
 427                 focus_ = mouse_focus_;
 428                 DBG_GUI_E << LOG_HEADER << "Firing: " << T::sdl_button_down_event << ".\n";
 429                 if(!owner_.fire(T::sdl_button_down_event, *focus_, coordinate)) {
 430                         DBG_GUI_E << LOG_HEADER << "Firing: " << T::button_down_event << ".\n";
 431                         owner_.fire(T::button_down_event, *mouse_focus_);
 432                 }
 433         } else {
 434                 widget* mouse_over = owner_.find_at(coordinate, true);
 435                 if(!mouse_over) {
 436                         return;
 437                 }
 438 
 439                 if(mouse_over != mouse_focus_) {
 440 #ifdef GUI2_SHOW_UNHANDLED_EVENT_WARNINGS
 441                         WRN_GUI_E << LOG_HEADER << ". Mouse down on non focused widget "
 442                                           << "and mouse not captured, we missed events.\n";
 443 #endif
 444                         mouse_focus_ = mouse_over;
 445                 }
 446 
 447                 focus_ = mouse_over;
 448                 DBG_GUI_E << LOG_HEADER << "Firing: " << T::sdl_button_down_event << ".\n";
 449                 if(!owner_.fire(T::sdl_button_down_event, *focus_, coordinate)) {
 450                         DBG_GUI_E << LOG_HEADER << "Firing: " << T::button_down_event << ".\n";
 451                         owner_.fire(T::button_down_event, *focus_);
 452                 }
 453         }
 454         handled = true;
 455 }

 457 template<typename T>
 458 void mouse_button<T>::signal_handler_sdl_button_up(const event::ui_event event, bool& handled,
 459                 const point& coordinate)
 460 {
 461         if(signal_handler_sdl_button_up_entered_) {
 462                 return;
 463         }
 464         resource_locker lock(signal_handler_sdl_button_up_entered_);
 465 
 466         DBG_GUI_E << LOG_HEADER << event << ".\n";
 467 
 468         if(!is_down_) {
 469                 WRN_GUI_E << LOG_HEADER << event
 470                                   << ". The mouse button is already up, we missed an event.\n";
 471                 return;
 472         }
 473         is_down_ = false;
 474 
 475         if(focus_) {
 476                 DBG_GUI_E << LOG_HEADER << "Firing: " << T::sdl_button_up_event << ".\n";
 477                 if(!owner_.fire(T::sdl_button_up_event, *focus_, coordinate)) {
 478                         DBG_GUI_E << LOG_HEADER << "Firing: " << T::button_up_event << ".\n";
 479                         owner_.fire(T::button_up_event, *focus_);
 480                 }
 481         }
 482 
 483         widget* mouse_over = owner_.find_at(coordinate, true);
 484         if(mouse_captured_) {
 485                 const unsigned mask = SDL_BUTTON_LMASK | SDL_BUTTON_MMASK
 486                                                           | SDL_BUTTON_RMASK;
 487 
 488                 if((SDL_GetMouseState(nullptr, nullptr) & mask) == 0) {
 489                         mouse_captured_ = false;
 490                 }
 491 
 492                 if(mouse_focus_ == mouse_over) {
 493                         mouse_button_click(mouse_focus_);
 494                 } else if(!mouse_captured_) {
 495                         mouse_leave();
 496 
 497                         if(mouse_over) {
 498                                 mouse_enter(mouse_over);
 499                         }
 500                 }
 501         } else if(focus_ && focus_ == mouse_over) {
 502                 mouse_button_click(focus_);
 503         }
 504 
 505         focus_ = nullptr;
 506         handled = true;
 507 }

 527                 last_clicked_widget_ = widget;

 637 void distributor::signal_handler_sdl_key_down(const SDL_Keycode key,
 638                                                                                            const SDL_Keymod modifier,
 639                                                                                            const utf8::string& unicode)
 640 {
 641         /** @todo Test whether recursion protection is needed. */
 642 
 643         DBG_GUI_E << LOG_HEADER << event::SDL_KEY_DOWN << ".\n";
 644 
 645         if(keyboard_focus_) {
 646                 // Attempt to cast to styled_widget, to avoid sending events if the
 647                 // widget is disabled. If the cast fails, we assume the widget
 648                 // is enabled and ready to receive events.
 649                 styled_widget* control = dynamic_cast<styled_widget*>(keyboard_focus_);
 650                 if(!control || control->get_active()) {
 651                         DBG_GUI_E << LOG_HEADER << "Firing: " << event::SDL_KEY_DOWN
 652                                           << ".\n";
 653                         if(owner_.fire(event::SDL_KEY_DOWN,
 654                                                    *keyboard_focus_,
 655                                                    key,
 656                                                    modifier,
 657                                                    unicode)) {
 658                                 return;
 659                         }
 660                 }
 661         }
 662 
 663         for(std::vector<widget*>::reverse_iterator ritor
 664                 = keyboard_focus_chain_.rbegin();
 665                 ritor != keyboard_focus_chain_.rend();
 666                 ++ritor) {
 667 
 668                 if(*ritor == keyboard_focus_) {
 669                         continue;
 670                 }
 671 
 672                 if(*ritor == &owner_) {
 673                         /**
 674                          * @todo Make sure we're not in the event chain.
 675                          *
 676                          * No idea why we're here, but needs to be fixed, otherwise we keep
 677                          * calling this function recursively upon unhandled events...
 678                          *
 679                          * Probably added to make sure the window can grab the events and
 680                          * handle + block them when needed, this is no longer needed with
 681                          * the chain.
 682                          */
 683                         continue;
 684                 }
 685 
 686                 // Attempt to cast to styled_widget, to avoid sending events if the
 687                 // widget is disabled. If the cast fails, we assume the widget
 688                 // is enabled and ready to receive events.
 689                 styled_widget* control = dynamic_cast<styled_widget*>(keyboard_focus_);
 690                 if(control != nullptr && !control->get_active()) {
 691                         continue;
 692                 }

 694                 DBG_GUI_E << LOG_HEADER << "Firing: " << event::SDL_KEY_DOWN << ".\n";
 695                 if(owner_.fire(event::SDL_KEY_DOWN, **ritor, key, modifier, unicode)) {
 696 
 697                         return;
 698                 }
 699         }
