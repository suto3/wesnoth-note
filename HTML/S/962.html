  33 /**
  34  * Used to set and unset scoped defines to preproc_map
  35  *
  36  * This is preferred form to set defines that aren't global
  37  **/
  38 template<typename T>
  39 class scoped_preproc_define_internal
  40 {
  41 public:
  42         scoped_preproc_define_internal(const scoped_preproc_define_internal&) = delete;
  43         scoped_preproc_define_internal& operator=(const scoped_preproc_define_internal&) = delete;
  44 
  45         /**
  46          * Adds normal preproc define.
  47          *
  48          * @param name name of preproc define to add.
  49          * @param add true if we should add this.
  50          */
  51         scoped_preproc_define_internal(const std::string& name, bool add = true) : name_(name), add_(add)
  52         {
  53                 if(add_) {
  54                         T::instance().add_define(name_);
  55                 }
  56         }
  57 
  58         /**
  59          * This removes preproc define from cacher
  60          */
  61         ~scoped_preproc_define_internal()
  62         {
  63                 if(add_) {
  64                         T::instance().remove_define(name_);
  65                 }
  66         }
  67 
  68 protected:
  69         //
  70         // Protected to let test code access
  71         //
  72 
  73         std::string name_;
  74         bool add_;
  75 
  76 };

 189 /**
 190  * Used to share macros between cache objects
 191  * You have to create transaction object to load all
 192  * macros to memory and share them subsequent cache loads.
 193  * If transaction is locked all stored macros are still
 194  * available but new macros aren't added.
 195  **/
 196 class config_cache_transaction
 197 {
 198 public:
 199         config_cache_transaction(const config_cache_transaction&) = delete;
 200         config_cache_transaction& operator=(const config_cache_transaction&) = delete;
 201 
 202         config_cache_transaction();
 203         ~config_cache_transaction();
 204 
 205         /**
 206          * Lock the transaction so no more macros are added
 207          */
 208         void lock();
 209 
 210         /**
 211          * Used to let std::for_each insert new defines to active_map
 212          * map to active
 213          */
 214         void insert_to_active(const preproc_map::value_type& def);
 215 
 216         enum state
 217         {
 218                 FREE,
 219                 NEW,
 220                 ACTIVE,
 221                 LOCKED
 222         };
 223 
 224 private:
 225         friend class config_cache;
 226         friend class fake_transaction;
 227 
 228         static state state_;
 229         static config_cache_transaction* active_;
 230 
 231         std::vector<std::string> define_filenames_;
 232         preproc_map active_map_;
 233 
 234         static state get_state()
 235         {
 236                 return state_;
 237         }
 238 
 239         static bool is_active()
 240         {
 241                 return active_ != 0;
 242         }
 243 
 244         static config_cache_transaction& instance()
 245         {
 246                 assert(active_);
 247                 return *active_;
 248         }
 249 
 250         const std::vector<std::string>& get_define_files() const;
 251 
 252         void add_define_file(const std::string& file);
 253 
 254         preproc_map& get_active_map(const preproc_map& defines_map);
 255 
 256         void add_defines_map_diff(preproc_map& defines_map);
 257 };
