 214     # Quenoth Youths have a non-null advances_to= in order to prevent them from
 215     # having purple xp bars (until all advancements have been gained); these
 216     # events make sure that they've nullified before advancement (to prevent the
 217     # unit type from appearing in the advancement menu) and restored afterwards.
 218     [event]
 219         name=pre advance
 220         first_time_only=no
 221 
 222         [filter]
 223             type=Quenoth Youth,Quenoth Youth 2,Quenoth Youth 3
 224         [/filter]
 225 
 226         {VARIABLE unit.advances_to null}
 227 
 228         [unstore_unit]
 229             variable=unit
 230             find_vacant=no
 231             advance=no
 232         [/unstore_unit]
 233     [/event]
 234     [event]
 235         name=post advance
 236         first_time_only=no
 237 
 238         [filter]
 239             type=Quenoth Youth 3
 240             
 241             [not]
 242                 [filter_wml]
 243                     [modifications]
 244                         [advancement]
 245                             exclude_amla=warrior_2_1,warrior_2_2
 246                         [/advancement]
 247                         [advancement]
 248                             exclude_amla=hunter_2_1,hunter_2_2
 249                         [/advancement]
 250                         [advancement]
 251                             exclude_amla=leader_2_1,leader_2_2
 252                         [/advancement]
 253                     [/modifications]
 254                 [/filter_wml]
 255             [/not]
 256         [/filter]
 257 
 258         {VARIABLE unit.advances_to "Quenoth Youth 3"}
 259 
 260         [unstore_unit]
 261             variable=unit
 262             find_vacant=no
 263             advance=no
 264         [/unstore_unit]
 265     [/event]

 267     # On recruit, this switches the higher-recruit-cost units to the normal
 268     # types, so that the different cost variations don't need to be taken into
 269     # account for example in unit filters.
 270 #define DISCARD_RECRUIT_COST_HELPER_UNITS TYPE
 271     [event]
 272         name=recruit
 273         first_time_only=no
 274 
 275         [filter]
 276             race=elf
 277         [/filter]
 278 
 279         [filter_condition]
 280             [variable]
 281                 name=unit.type
 282                 contains={TYPE}
 283             [/variable]
 284 
 285             [variable]
 286                 name=unit.type
 287                 not_equals={TYPE}
 288             [/variable]
 289         [/filter_condition]
 290 
 291         {VARIABLE unit.type {TYPE}}
 292 
 293         [unstore_unit]
 294             variable=unit
 295             find_vacant=no
 296         [/unstore_unit]
 297     [/event]
 298 #enddef
