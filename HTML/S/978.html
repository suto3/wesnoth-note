  35 bool is_neutral(const surface& surf)
  36 {
  37         return (surf->format->BytesPerPixel == 4 &&
  38                         surf->format->Rmask == SDL_RED_MASK &&
  39                         (surf->format->Amask | SDL_ALPHA_MASK) == SDL_ALPHA_MASK);

  92         assert(SDL_ALPHA_TRANSPARENT==0);

  95                 return nullptr;

  98                 return surf;

 101 
 102         surface dst(create_neutral_surface(w, surf->h));

 121                         const Uint32 pixel = src_pixels [y * src->w];
 122                         for(unsigned x = 0; x < w; ++x) {

 254 // NOTE: Don't pass this function 0 scaling arguments.
 255 surface scale_surface(const surface &surf, int w, int h)
 256 {
 257         // Since SDL version 1.1.5 0 is transparent, before 255 was transparent.
 258         assert(SDL_ALPHA_TRANSPARENT==0);
 259 
 260         if(surf == nullptr)
 261                 return nullptr;
 262 
 263         if(w == surf->w && h == surf->h) {
 264                 return surf;
 265         }
 266         assert(w >= 0);
 267         assert(h >= 0);
 268 
 269         surface dst(create_neutral_surface(w,h));
 270 
 271         if (w == 0 || h ==0) {
 272                 std::cerr << "Create an empty image\n";
 273                 return dst;
 274         }
 275 
 276         surface src(make_neutral_surface(surf));
 277         // Now both surfaces are always in the "neutral" pixel format
 278 
 279         if(src == nullptr || dst == nullptr) {
 280                 std::cerr << "Could not create surface to scale onto\n";
 281                 return nullptr;
 282         }
 283 
 284         {
 285                 const_surface_lock src_lock(src);
 286                 surface_lock dst_lock(dst);
 287 
 288                 const Uint32* const src_pixels = src_lock.pixels();
 289                 Uint32* const dst_pixels = dst_lock.pixels();
 290 
 291                 fixed_t xratio = fxpdiv(surf->w,w);
 292                 fixed_t yratio = fxpdiv(surf->h,h);
 293 
 294                 fixed_t ysrc = ftofxp(0.0);
 295                 for(int ydst = 0; ydst != h; ++ydst, ysrc += yratio) {
 296                         fixed_t xsrc = ftofxp(0.0);
 297                         for(int xdst = 0; xdst != w; ++xdst, xsrc += xratio) {
 298                                 const int xsrcint = fxptoi(xsrc);
 299                                 const int ysrcint = fxptoi(ysrc);
 300 
 301                                 const Uint32* const src_word = src_pixels + ysrcint*src->w + xsrcint;
 302                                 Uint32* const dst_word = dst_pixels +    ydst*dst->w + xdst;
 303                                 const int dx = (xsrcint + 1 < src->w) ? 1 : 0;
 304                                 const int dy = (ysrcint + 1 < src->h) ? src->w : 0;
 305 
 306                                 Uint8 r,g,b,a;
 307                                 Uint32 rr,gg,bb,aa, temp;
 308 
 309                                 Uint32 pix[4], bilin[4];
 310 
 311                                 // This next part is the fixed point
 312                                 // equivalent of "take everything to
 313                                 // the right of the decimal point."
 314                                 // These fundamental weights decide
 315                                 // the contributions from various
 316                                 // input pixels. The labels assume
 317                                 // that the upper left corner of the
 318                                 // screen ("northeast") is 0,0 but the
 319                                 // code should still be consistent if
 320                                 // the graphics origin is actually
 321                                 // somewhere else.
 322                                 //
 323                                 // That is, the bilin array holds the
 324                                 // "geometric" weights. I.E. If I'm scaling
 325                                 // a 2 x 2 block a 10 x 10 block, then for
 326                                 // pixel (2,2) of ouptut, the upper left
 327                                 // pixel should be 10:1 more influential than
 328                                 // the upper right, and also 10:1 more influential
 329                                 // than lower left, and 100:1 more influential
 330                                 // than lower right.
 331 
 332                                 const fixed_t e = 0x000000FF & xsrc;
 333                                 const fixed_t s = 0x000000FF & ysrc;
 334                                 const fixed_t n = 0xFF - s;
 335                                 // Not called "w" to avoid hiding a function parameter
 336                                 // (would cause a compiler warning in MSVC2015 with /W4)
 337                                 const fixed_t we = 0xFF - e;
 338 
 339                                 pix[0] = *src_word;              // northwest
 340                                 pix[1] = *(src_word + dx);       // northeast
 341                                 pix[2] = *(src_word + dy);       // southwest
 342                                 pix[3] = *(src_word + dx + dy);  // southeast
 343 
 344                                 bilin[0] = n*we;
 345                                 bilin[1] = n*e;
 346                                 bilin[2] = s*we;
 347                                 bilin[3] = s*e;
 348 
 349                                 int loc;
 350                                 rr = bb = gg = aa = 0;
 351                                 for (loc=0; loc<4; loc++) {
 352                                   a = pix[loc] >> 24;
 353                                   r = pix[loc] >> 16;
 354                                   g = pix[loc] >> 8;
 355                                   b = pix[loc] >> 0;
 356 
 357                                   //We also have to implement weighting by alpha for the RGB components
 358                                   //If a unit has some parts solid and some parts translucent,
 359                                   //i.e. a red cloak but a dark shadow, then when we scale in
 360                                   //the shadow shouldn't appear to become red at the edges.
 361                                   //This part also smoothly interpolates between alpha=0 being
 362                                   //transparent and having no contribution, vs being opaque.
 363                                   temp = (a * bilin[loc]);
 364                                   rr += r * temp;
 365                                   gg += g * temp;
 366                                   bb += b * temp;
 367                                   aa += temp;
 368                                 }
 369 
 370                                 a = aa >> (16); // we average the alphas, they don't get weighted by any other factor besides bilin
 371                                 if (a != 0) {
 372                                         rr /= a;        // finish alpha weighting: divide by sum of alphas
 373                                         gg /= a;
 374                                         bb /= a;
 375                                 }
 376                                 r = rr >> (16); // now shift over by 16 for the bilin part
 377                                 g = gg >> (16);
 378                                 b = bb >> (16);
 379                                 *dst_word = (a << 24) + (r << 16) + (g << 8) + b;
 380                         }
 381                 }
 382         }
 383 
 384         return dst;
 385 }

 387 surface scale_surface_sharp(const surface& surf, int w, int h)
 388 {
 389         // Since SDL version 1.1.5 0 is transparent, before 255 was transparent.
 390         assert(SDL_ALPHA_TRANSPARENT==0);
 391 
 392         if(surf == nullptr)
 393                 return nullptr;
 394 
 395         if(w == surf->w && h == surf->h) {
 396                 return surf;
 397         }
 398         assert(w >= 0);
 399         assert(h >= 0);
 400 
 401         surface dst(create_neutral_surface(w,h));
 402 
 403         if (w == 0 || h ==0) {
 404                 std::cerr << "Create an empty image\n";
 405                 return dst;
 406         }
 407 
 408         surface src(make_neutral_surface(surf));
 409         // Now both surfaces are always in the "neutral" pixel format
 410 
 411         if(src == nullptr || dst == nullptr) {
 412                 std::cerr << "Could not create surface to scale onto\n";
 413                 return nullptr;
 414         }
 415 
 416         {
 417                 const_surface_lock src_lock(src);
 418                 surface_lock dst_lock(dst);
 419 
 420                 const Uint32* const src_pixels = src_lock.pixels();
 421                 Uint32* const dst_pixels = dst_lock.pixels();
 422 
 423                 float xratio = static_cast<float>(surf->w) / w;
 424                 float yratio = static_cast<float>(surf->h) / h;
 425 
 426                 float ysrc = 0.0f;
 427                 for(int ydst = 0; ydst != h; ++ydst, ysrc += yratio) {
 428                         float xsrc = 0.0f;
 429                         for(int xdst = 0; xdst != w; ++xdst, xsrc += xratio) {
 430                                 float red = 0.0f, green = 0.0f, blue = 0.0f, alpha = 0.0f;
 431 
 432                                 float summation = 0.0f;
 433 
 434                                 // We now have a rectangle, (xsrc,ysrc,xratio,yratio)
 435                                 // which we want to derive the pixel from
 436                                 for(float xloc = xsrc; xloc < xsrc+xratio; xloc += 1) {
 437                                         const float xsize = std::min<float>(std::floor(xloc+1)-xloc,xsrc+xratio-xloc);
 438 
 439                                         for(float yloc = ysrc; yloc < ysrc+yratio; yloc += 1) {
 440                                                 const int xsrcint = std::max<int>(0,std::min<int>(src->w-1,static_cast<int>(xsrc)));
 441                                                 const int ysrcint = std::max<int>(0,std::min<int>(src->h-1,static_cast<int>(ysrc)));
 442                                                 const float ysize = std::min<float>(std::floor(yloc+1)-yloc,ysrc+yratio-yloc);
 443 
 444                                                 Uint8 r,g,b,a;
 445 
 446                                                 SDL_GetRGBA(src_pixels[ysrcint*src->w + xsrcint],src->format,&r,&g,&b,&a);
 447                                                 float value = xsize * ysize;
 448                                                 summation += value;
 449                                                 if (!a) continue;
 450                                                 value *= a;
 451                                                 alpha += value;
 452                                                 red += r * value;
 453                                                 green += g * value;
 454                                                 blue += b * value;
 455                                         }
 456                                 }
 457 
 458                                 if (alpha != 0) {
 459                                         red = red / alpha + 0.5f;
 460                                         green = green / alpha + 0.5f;
 461                                         blue = blue / alpha + 0.5f;
 462                                         alpha = alpha / summation + 0.5f;
 463                                 }
 464 
 465                                 dst_pixels[ydst*dst->w + xdst] = SDL_MapRGBA(
 466                                 dst->format
 467                                 , static_cast<uint8_t>(red)
 468                                 , static_cast<uint8_t>(green)
 469                                 , static_cast<uint8_t>(blue)
 470                                 , static_cast<uint8_t>(alpha));
 471                         }
 472 
 473                 }
 474         }
 475 
 476         return dst;
 477 }

 480 surface tile_surface(const surface& surf, int w, int h, bool centered)
 481 {
 482         if (surf->w == w && surf->h == h) {
 483                 return surf;
 484         }
 485 
 486         surface dest(create_neutral_surface(w, h));
 487         surface src(make_neutral_surface(surf));
 488 
 489         if (src == nullptr || dest == nullptr) {
 490                 std::cerr << "failed to make neutral surface\n";
 491                 return nullptr;
 492         }
 493 
 494         {
 495                 const_surface_lock srclock(src);
 496                 surface_lock destlock(dest);
 497 
 498                 const Uint32* srcpixels = srclock.pixels();
 499                 Uint32* destpixels = destlock.pixels();
 500 
 501                 const int& sw = src->w;
 502                 const int& sh = src->h;
 503 
 504                 const int xoff = centered ? (w - sw) / 2 : 0;
 505                 const int yoff = centered ? (h - sh) / 2 : 0;
 506 
 507                 for (int i = 0; i<w*h; ++i) {
 508                         int x = ((i % w) - xoff);
 509                         int y = ((i / w) - yoff);
 510 
 511                         while ((x += sw) < 0) { /* DO NOTHING */ }
 512                         while ((y += sh) < 0) { /* DO NOTHING */ }
 513 
 514                         const int sx = x % sw;
 515                         const int sy = y % sh;
 516 
 517                         destpixels[i] = srcpixels[sy*sw + sx];
 518                 }
 519         }
 520 
 521         return dest;
 522 }

 524 surface adjust_surface_color(const surface &surf, int red, int green, int blue)
 525 {
 526         if(surf == nullptr)
 527                 return nullptr;
 528 
 529         if((red == 0 && green == 0 && blue == 0)) {
 530                 surface temp = surf; // TODO: remove temp surface
 531                 return temp;
 532         }
 533 
 534         surface nsurf(make_neutral_surface(surf));
 535 
 536         if(nsurf == nullptr) {
 537                 std::cerr << "failed to make neutral surface\n";
 538                 return nullptr;
 539         }
 540 
 541         {
 542                 surface_lock lock(nsurf);
 543                 Uint32* beg = lock.pixels();
 544                 Uint32* end = beg + nsurf->w*surf->h;
 545 
 546                 while(beg != end) {
 547                         Uint8 alpha = (*beg) >> 24;
 548 
 549                         if(alpha) {
 550                                 Uint8 r, g, b;
 551                                 r = (*beg) >> 16;
 552                                 g = (*beg) >> 8;
 553                                 b = (*beg) >> 0;
 554 
 555                                 r = std::max<int>(0,std::min<int>(255,int(r)+red));
 556                                 g = std::max<int>(0,std::min<int>(255,int(g)+green));
 557                                 b = std::max<int>(0,std::min<int>(255,int(b)+blue));
 558 
 559                                 *beg = (alpha << 24) + (r << 16) + (g << 8) + b;
 560                         }
 561 
 562                         ++beg;
 563                 }
 564         }
 565 
 566         return nsurf;
 567 }

 569 surface greyscale_image(const surface &surf)
 570 {
 571         if(surf == nullptr)
 572                 return nullptr;
 573 
 574         surface nsurf(make_neutral_surface(surf));
 575         if(nsurf == nullptr) {
 576                 std::cerr << "failed to make neutral surface\n";

 581                 surface_lock lock(nsurf);
 582                 Uint32* beg = lock.pixels();
 583                 Uint32* end = beg + nsurf->w*surf->h;
 584 
 585                 while(beg != end) {
 586                         Uint8 alpha = (*beg) >> 24;
 587 
 588                         if(alpha) {
 589                                 Uint8 r, g, b;
 590                                 r = (*beg) >> 16;
 591                                 g = (*beg) >> 8;
 592                                 b = (*beg);
 593                                 //const Uint8 avg = (red+green+blue)/3;
 594 
 595                                 // Use the correct formula for RGB to grayscale conversion.
 596                                 // Ok, this is no big deal :)
 597                                 // The correct formula being:
 598                                 // gray=0.299red+0.587green+0.114blue
 599                                 const Uint8 avg = static_cast<Uint8>((
 600                                         77  * static_cast<Uint16>(r) +
 601                                         150 * static_cast<Uint16>(g) +
 602                                         29  * static_cast<Uint16>(b)  ) / 256);
 603 
 604                                 *beg = (alpha << 24) | (avg << 16) | (avg << 8) | avg;
 605                         }
 606 
 607                         ++beg;
 608                 }
 609         }
 610 
 611         return nsurf;
 612 }

 615 {
 616         if(surf == nullptr)
 617                 return nullptr;
 618

 634                                 Uint8 r, g, b, result;

 636                                 g = (*beg) >> 8;

 645                         }
 646

 648                 }
 649         }

1324 void blur_surface(surface& surf, SDL_Rect rect, int depth)
1325 {
1326         if(surf == nullptr) {
1327                 return;
1328         }
1329 
1330         const int max_blur = 256;
1331         if(depth > max_blur) {
1332                 depth = max_blur;
1333         }
1334 
1335         Uint32 queue[max_blur];
1336         const Uint32* end_queue = queue + max_blur;
1337 
1338         const Uint32 ff = 0xff;
1339 
1340         const unsigned pixel_offset = rect.y * surf->w + rect.x;
1341 
1342         surface_lock lock(surf);
1343         for(int y = 0; y < rect.h; ++y) {
1344                 const Uint32* front = &queue[0];
1345                 Uint32* back = &queue[0];
1346                 Uint32 red = 0, green = 0, blue = 0, avg = 0;
1347                 Uint32* p = lock.pixels() + pixel_offset + y * surf->w;
1348                 for(int x = 0; x <= depth && x < rect.w; ++x, ++p) {
1349                         red += ((*p) >> 16)&0xFF;
1350                         green += ((*p) >> 8)&0xFF;
1351                         blue += (*p)&0xFF;
1352                         ++avg;
1353                         *back++ = *p;
1354                         if(back == end_queue) {
1355                                 back = &queue[0];
1356                         }
1357                 }

1360                 for(int x = 0; x < rect.w; ++x, ++p) {
1361                         *p = 0xFF000000
1362                                         | (std::min(red/avg,ff) << 16)
1363                                         | (std::min(green/avg,ff) << 8)
1364                                         | std::min(blue/avg,ff);
1365 
1366                         if(x >= depth) {
1367                                 red -= ((*front) >> 16)&0xFF;
1368                                 green -= ((*front) >> 8)&0xFF;
1369                                 blue -= *front&0xFF;
1370                                 --avg;
1371                                 ++front;
1372                                 if(front == end_queue) {
1373                                         front = &queue[0];
1374                                 }
1375                         }

1663 /* Simplified RotSprite algorithm.
1664  * http://en.wikipedia.org/wiki/Image_scaling#RotSprite
1665  * Lifted from: http://github.com/salmonmoose/SpriteRotator
1666  * 1) Zoom the source image by a certain factor.
1667  * 2) Scan the zoomed source image at every step=offset and put it in the result. */
1668 surface rotate_any_surface(const surface& surf, float angle, int zoom, int offset)
1669 {
1670         int src_w, src_h, dst_w, dst_h;
1671         float min_x, min_y, sine, cosine;
1672         {
1673                 float max_x, max_y;
1674                 // convert angle to radiant (angle * 2 * PI) / 360
1675                 const float radians = angle * boost::math::constants::pi<float>() / 180;
1676                 cosine = cos(radians);
1677                 sine   = sin(radians);
1678                 // calculate the size of the dst image
1679                 src_w = surf->w * zoom;
1680                 src_h = surf->h * zoom;
1681                 /* See http://en.wikipedia.org/wiki/Rotation_(mathematics) */
1682                 const float point_1x = src_h * -sine;
1683                 const float point_1y = src_h * cosine;
1684                 const float point_2x = src_w * cosine - src_h * sine;
1685                 const float point_2y = src_h * cosine + src_w * sine;
1686                 const float point_3x = src_w * cosine;
1687                 const float point_3y = src_w * sine;
1688                 /* After the rotation, the new image has different dimensions.
1689                  * E.g.: The maximum height equals the former diagonal in case the angle is 45, 135, 225 or 315 degree.
1690                  * See http://en.wikipedia.org/wiki/File:Rotation_illustration2.svg to get the idea. */
1691                 min_x = std::min(0.0F, std::min(point_1x, std::min(point_2x, point_3x)));
1692                 min_y = std::min(0.0F, std::min(point_1y, std::min(point_2y, point_3y)));
1693                 max_x = (angle >  90 && angle < 180) ? 0 : std::max(point_1x, std::max(point_2x, point_3x));
1694                 max_y = (angle > 180 && angle < 270) ? 0 : std::max(point_1y, std::max(point_2y, point_3y));
1695                 dst_w = static_cast<int>(ceil(std::abs(max_x) - min_x)) / zoom;
1696                 dst_h = static_cast<int>(ceil(std::abs(max_y) - min_y)) / zoom;
1697         }
1698         surface dst(create_neutral_surface(dst_w, dst_h));
1699         {
1700                 surface_lock dst_lock(dst);
1701                 const surface src = scale_surface(surf, src_w, src_h);
1702                 const_surface_lock src_lock(src);
1703                 const float scale =   1.f / zoom;
1704                 const int   max_x = dst_w * zoom;
1705                 const int   max_y = dst_h * zoom;
1706                 /* Loop through the zoomed src image,
1707                  * take every pixel in steps with offset distance and place it in the dst image. */
1708                 for (int x = 0; x < max_x; x += offset)
1709                         for (int y = 0; y < max_y; y += offset) {
1710                                 // calculate the src pixel that fits in the dst
1711                                 const float source_x = (x + min_x)*cosine + (y + min_y)*sine;
1712                                 const float source_y = (y + min_y)*cosine - (x + min_x)*sine;
1713                                 // if the pixel exists on the src surface
1714                                 if (source_x >= 0 && source_x < src_w
1715                                                 && source_y >= 0 && source_y < src_h)
1716                                         // get it from the src surface and place it on the dst surface
1717                                         put_pixel(dst, dst_lock, x*scale , y*scale, // multiply with scale
1718                                                         get_pixel(src, src_lock, source_x, source_y));
1719                         }
1720         }
1721 
1722         return dst;
1723 }

1725 void put_pixel(const surface& surf, surface_lock& surf_lock, int x, int y, Uint32 pixel)
1726 {
1727         const int bpp = surf->format->BytesPerPixel;
1728         /* dst is the address to the pixel we want to set */
1729         Uint8* const dst = reinterpret_cast<Uint8*>(surf_lock.pixels()) + y * surf->pitch + x * bpp;
1730         switch (bpp) {
1731         case 1:
1732                 *dst = pixel;
1733                 break;
1734         case 2:
1735                 *reinterpret_cast<Uint16*>(dst) = pixel;
1736                 break;
1737         case 3:
1738                 if (SDL_BYTEORDER == SDL_BIG_ENDIAN) {
1739                         dst[0] = (pixel >> 16) & 0xff;
1740                         dst[1] = (pixel >> 8) & 0xff;
1741                         dst[2] = pixel & 0xff;
1742                 } else {
1743                         dst[0] = pixel & 0xff;
1744                         dst[1] = (pixel >> 8) & 0xff;
1745                         dst[2] = (pixel >> 16) & 0xff;
1746                 }
1747                 break;
1748         case 4:
1749                 *reinterpret_cast<Uint32*>(dst) = pixel;
1750                 break;
1751         default:
1752                 break;
1753         }
1754 }

1756 Uint32 get_pixel(const surface& surf, const const_surface_lock& surf_lock, int x, int y)
1757 {
1758         const int bpp = surf->format->BytesPerPixel;
1759         /* p is the address to the pixel we want to retrieve */
1760         const Uint8* const src = reinterpret_cast<const Uint8*>(surf_lock.pixels()) + y * surf->pitch + x * bpp;
1761         switch (bpp) {
1762         case 1:
1763                 return *src;
1764         case 2:
1765                 return *reinterpret_cast<const Uint16*>(src);
1766         case 3:
1767                 if (SDL_BYTEORDER == SDL_BIG_ENDIAN)
1768                         return src[0] << 16 | src[1] << 8 | src[2];
1769                 else
1770                         return src[0] | src[1] << 8 | src[2] << 16;
1771                 break;
1772         case 4:
1773                 return *reinterpret_cast<const Uint32*>(src);
1774         }
1775         return 0;

1970                 }
1971                 if (src_rect.y < 0) {
1972             if (src_rect.y + src_rect.h <= 0 || src_rect.y + dst_rect.h <= 0 )
1973                 return;
1974                         dst_rect.y -= src_rect.y;
1975                         dst_rect.h += src_rect.y;
1976                         src_rect.h += src_rect.y;
1977                         src_rect.y = 0;
1978                 }
1979                 if (src_rect.x + src_rect.w > src->w) {
1980             if (src_rect.x >= src->w)
1981                 return;
1982                         src_rect.w = src->w - src_rect.x;
1983                 }
1984                 if (src_rect.y + src_rect.h > src->h) {
1985             if (src_rect.y >= src->h)
1986                 return;
1987                         src_rect.h = src->h - src_rect.y;
1988                 }
1989         }
1990 
1991         assert(dst_rect.x >= 0);
1992         assert(dst_rect.y >= 0);
1993 
1994         // Get the blit size limits.
1995         const unsigned width = std::min(src_rect.w, dst_rect.w);
1996         const unsigned height = std::min(src_rect.h, dst_rect.h);
1997 
1998         {
1999                 // Extra scoping used for the surface_lock.
2000                 const_surface_lock src_lock(src);
2001                 surface_lock dst_lock(dst);
2002 
2003                 const Uint32* const src_pixels = src_lock.pixels();
2004                 Uint32* dst_pixels = dst_lock.pixels();
2005 
2006                 for(unsigned y = 0; y < height; ++y) {
2007                         for(unsigned x = 0; x < width; ++x) {
2008 
2009                                 // We need to do the blitting using some optimizations
2010                                 // if the src is fully transparent we can ignore this pixel
2011                                 // if the src is fully opaque we can overwrite the destination with this pixel
2012                                 // if the destination is fully transparent we replace us with the source
2013                                 //
2014                                 // We do these optimizations between the extraction of the variables
2015                                 // to avoid creating variables not used (it might save us some cycles).
2016 
2017                                 const int src_offset = (y + src_rect.y) * src->w + (x + src_rect.x);
2018                                 assert(src_offset < src->w * src->h);
2019                                 const Uint32 src_pixel = src_pixels[src_offset];
2020                                 const Uint8 src_a = (src_pixel & 0xFF000000) >> 24;
2021 
2022                                 if(!src_a) {
2023                                         // Fully transparent source, ignore
2024                                         continue;
2025                                 }
2026 
2027                                 const ptrdiff_t dst_offset = (y + dst_rect.y) * dst->w + (x + dst_rect.x);
2028                                 assert(dst_offset < dst->w * dst->h);
2029                                 if(src_a == 255) {
2030                                         // Fully opaque source, copy
2031                                         dst_pixels[dst_offset] = src_pixel;
2032                                         continue;
2033                                 }
2034 
2035                                 const Uint32 dst_pixel = dst_pixels[dst_offset];
2036                                 Uint8 dst_a = (dst_pixel & 0xFF000000) >> 24;
2037 
2038                                 if(!dst_a) {
2039                                         // Fully transparent destination, copy
2040                                         dst_pixels[dst_offset] = src_pixel;
2041                                         continue;
2042                                 }
2043 
2044                                 const Uint8 src_r = (src_pixel & 0x00FF0000) >> 16;
2045                                 const Uint8 src_g = (src_pixel & 0x0000FF00) >> 8;
2046                                 const Uint8 src_b = src_pixel & 0x000000FF;
2047 
2048                                 Uint8 dst_r = (dst_pixel & 0x00FF0000) >> 16;
2049                                 Uint8 dst_g = (dst_pixel & 0x0000FF00) >> 8;
2050                                 Uint8 dst_b = dst_pixel & 0x000000FF;
2051 
2052                                 if(dst_a == 255) {
2053 
2054                                         // Destination fully opaque blend the source.
2055                                         dst_r = (((src_r - dst_r) * src_a) >> 8 ) + dst_r;
2056                                         dst_g = (((src_g - dst_g) * src_a) >> 8 ) + dst_g;
2057                                         dst_b = (((src_b - dst_b) * src_a) >> 8 ) + dst_b;
2058 
2059                                 } else {
2060 
2061                                         // Destination and source party transparent.
2062 
2063                                         // acquired the data now do the blitting
2064                                         const unsigned tmp_a = 255 - src_a;
2065 
2066                                         const unsigned tmp_r = 1 + (src_r * src_a) + (dst_r * tmp_a);
2067                                         dst_r = (tmp_r + (tmp_r >> 8)) >> 8;
2068 
2069                                         const unsigned tmp_g = 1 + (src_g * src_a) + (dst_g * tmp_a);
2070                                         dst_g = (tmp_g + (tmp_g >> 8)) >> 8;
2071 
2072                                         const unsigned tmp_b = 1 + (src_b * src_a) + (dst_b * tmp_a);
2073                                         dst_b = (tmp_b + (tmp_b >> 8)) >> 8;
2074 
2075                                         dst_a += (((255 - dst_a) * src_a) >> 8);
2076                                 }
2077 
2078                                 dst_pixels[dst_offset] = (dst_a << 24) | (dst_r << 16) | (dst_g << 8) | (dst_b);
2079 
2080                         }
2081                 }
2082         }
2083 }
