  68 void show_unit_description(CVideo& video, const unit_type &t)
  69 {
  70         std::string var_id = t.get_cfg()["variation_id"].str();
  71         if (var_id.empty())
  72                 var_id = t.get_cfg()["variation_name"].str();
  73         bool hide_help = t.hide_help();
  74         bool use_variation = false;
  75         if (!var_id.empty()) {
  76                 const unit_type *parent = unit_types.find(t.id());
  77                 assert(parent);
  78                 if (hide_help) {
  79                         hide_help = parent->hide_help();
  80                 } else {
  81                         use_variation = true;
  82                 }
  83         }
  84 
  85         if (use_variation)
  86                 help::show_variation_help(video, t.id(), var_id, hide_help);
  87         else
  88                 help::show_unit_help(video, t.id(), t.show_variations_in_help(), hide_help);
  89 }

 152 /**
 153  * Open a help dialog using a toplevel other than the default.
 154  *
 155  * This allows for complete customization of the contents, although not in a
 156  * very easy way.
 157  */
 158 void show_help(CVideo& video, const section &toplevel_sec,
 159                            const std::string& show_topic,
 160                            int xloc, int yloc)
 161 {
 162         const events::event_context dialog_events_context;
 163         const gui::dialog_manager manager;
 164 
 165         CVideo& screen = video;
 166         const surface& scr = screen.getSurface();
 167 
 168         const int width  = std::min<int>(font::relative_size(1200), scr->w - font::relative_size(20));
 169         const int height = std::min<int>(font::relative_size(850), scr->h - font::relative_size(150));
 170         const int left_padding = font::relative_size(10);
 171         const int right_padding = font::relative_size(10);
 172         const int top_padding = font::relative_size(10);
 173         const int bot_padding = font::relative_size(10);
 174 
 175         // If not both locations were supplied, put the dialog in the middle
 176         // of the screen.
 177         if (yloc <= -1 || xloc <= -1) {
 178                 xloc = scr->w / 2 - width / 2;
 179                 yloc = scr->h / 2 - height / 2;
 180         }
 181         std::vector<gui::button*> buttons_ptr;
 182         gui::button close_button_(video, _("Close"));
 183         buttons_ptr.push_back(&close_button_);
 184 
 185         gui::dialog_frame f(video, _("The Battle for Wesnoth Help"), gui::dialog_frame::default_style,
 186                                          true, &buttons_ptr);
 187         f.layout(xloc, yloc, width, height);
 188         f.draw();
 189 
 190     // Find all unit_types that have not been constructed yet and fill in the information
 191     // needed to create the help topics
 192         unit_types.build_all(unit_type::HELP_INDEXED);
 193 
 194         if (preferences::encountered_units().size() != size_t(last_num_encountered_units) ||
 195             preferences::encountered_terrains().size() != size_t(last_num_encountered_terrains) ||
 196             last_debug_state != game_config::debug ||
 197                 last_num_encountered_units < 0) {
 198                 // More units or terrains encountered, update the contents.
 199                 last_num_encountered_units = preferences::encountered_units().size();
 200                 last_num_encountered_terrains = preferences::encountered_terrains().size();
 201                 last_debug_state = game_config::debug;
 202                 generate_contents();
 203         }
 204         try {
 205                 help_browser hb(video, toplevel_sec);
 206                 hb.set_location(xloc + left_padding, yloc + top_padding);
 207                 hb.set_width(width - left_padding - right_padding);
 208                 hb.set_height(height - top_padding - bot_padding);
 209                 if (show_topic != "") {
 210                         hb.show_topic(show_topic);
 211                 }
 212                 else {
 213                         hb.show_topic(default_show_topic);
 214                 }
 215                 hb.set_dirty(true);
 216                 events::raise_draw_event();
 217                 CKey key;
 218                 for (;;) {
 219                         events::pump();
 220                         events::raise_process_event();
 221                         f.draw();
 222                         events::raise_draw_event();
 223                         if (key[SDLK_ESCAPE]) {
 224                                 // Escape quits from the dialog.
 225                                 return;
 226                         }
 227                         for (std::vector<gui::button*>::iterator button_it = buttons_ptr.begin();
 228                                  button_it != buttons_ptr.end(); ++button_it) {
 229                                 if ((*button_it)->pressed()) {
 230                                         // There is only one button, close.
 231                                         return;
 232                                 }
 233                         }
 234                         video.flip();
 235                         CVideo::delay(10);
 236                 }
 237         }
 238         catch (parse_error& e) {
 239                 std::stringstream msg;
 240                 msg << _("Parse error when parsing help text: ") << "'" << e.message << "'";
 241                 gui2::show_transient_message(video, "", msg.str());
 242         }
 243 }
