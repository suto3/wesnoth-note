  55 void trigger_full_redraw() {
  56         SDL_Event event;
  57         event.type = SDL_WINDOWEVENT;
  58         event.window.event = SDL_WINDOWEVENT_RESIZED;
  59         event.window.data1 = (*frameBuffer).h;
  60         event.window.data2 = (*frameBuffer).w;
  61 
  62         for(const auto& layer : draw_layers) {
  63                 layer->handle_window_event(event);
  64         }
  65 
  66         SDL_Event drawEvent;
  67         SDL_UserEvent data;
  68 
  69         data.type = DRAW_ALL_EVENT;
  70         data.code = 0;
  71         data.data1 = nullptr;
  72         data.data2 = nullptr;
  73 
  74         drawEvent.type = DRAW_ALL_EVENT;
  75         drawEvent.user = data;
  76         SDL_FlushEvent(DRAW_ALL_EVENT);
  77         SDL_PushEvent(&drawEvent);
  78 }

 204 void CVideo::init_window()
 205 {
 206         // Position
 207         const int x = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 208         const int y = preferences::fullscreen() ? SDL_WINDOWPOS_UNDEFINED : SDL_WINDOWPOS_CENTERED;
 209 
 210         // Dimensions
 211         const int w = preferences::resolution().first;
 212         const int h = preferences::resolution().second;
 213 
 214         // Video flags
 215         int video_flags = 0;
 216 
 217         // Add any more default flags here
 218         video_flags |= SDL_WINDOW_RESIZABLE;
 219 
 220         if(preferences::fullscreen()) {
 221                 video_flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
 222         } else if(preferences::maximized()) {
 223                 video_flags |= SDL_WINDOW_MAXIMIZED;
 224         }
 225 
 226         // Initialize window
 227         window.reset(new sdl::window("", x, y, w, h, video_flags, SDL_RENDERER_SOFTWARE));
 228 
 229         std::cerr << "Setting mode to " << w << "x" << h << std::endl;
 230 
 231         window->set_minimum_size(
 232                 preferences::min_window_width,
 233                 preferences::min_window_height
 234         );
 235 
 236         event_handler_.join_global();
 237 
 238         update_framebuffer();
 239         if(frameBuffer) {
 240                 image::set_pixel_format(frameBuffer->format);
 241         }
 242 }

 244 void CVideo::setMode(int x, int y, const MODE_EVENT mode)
 245 {
 246         assert(window);
 247         if(fake_screen_) return;
 248 
 249         switch(mode) {
 250                 case TO_FULLSCREEN:
 251                         window->full_screen();
 252                         break;
 253 
 254                 case TO_WINDOWED:
 255                         window->to_window();
 256                         window->restore();
 257                         break;
 258 
 259                 case TO_MAXIMIZED_WINDOW:
 260                         window->to_window();
 261                         window->maximize();
 262                         break;
 263 
 264                 case TO_RES:
 265                         window->restore();
 266                         window->set_size(x, y);
 267                         window->center();
 268                         break;
 269         }
 270 
 271         update_framebuffer();
 272         if(frameBuffer) {
 273                 image::set_pixel_format(frameBuffer->format);
 274         }
 275 }

 336 std::vector<std::pair<int, int>> CVideo::get_available_resolutions(const bool include_current)
 337 {
 338         std::vector<std::pair<int, int>> result;
 339 
 340         if(!window) {
 341                 return result;
 342         }
 343 
 344         const int modes = SDL_GetNumDisplayModes(window->get_display_index());
 345         if(modes <= 0) {
 346                 std::cerr << "No modes supported\n";
 347                 return result;
 348         }
 349 
 350         const std::pair<int,int> min_res = std::make_pair(preferences::min_window_width, preferences::min_window_height);
 351 
 352         SDL_DisplayMode mode;
 353         for (int i = 0; i < modes; ++i) {
 354                 if(SDL_GetDisplayMode(0, i, &mode) == 0) {
 355                         if(mode.w >= min_res.first && mode.h >= min_res.second)
 356                                 result.emplace_back(mode.w, mode.h);
 357                 }
 358         }
 359 
 360         if(std::find(result.begin(), result.end(), min_res) == result.end()) {
 361                 result.push_back(min_res);
 362         }
 363 
 364         if(include_current) {
 365                 result.push_back(current_resolution());
 366         }
 367 
 368         std::sort(result.begin(), result.end());
 369         result.erase(std::unique(result.begin(), result.end()), result.end());
 370 
 371         return result;
 372 }

 388 int CVideo::set_help_string(const std::string& str)
 389 {
 390         font::remove_floating_label(help_string_);
 391 
 392         const color_t color { 0, 0, 0, 0xbb };
 393 
 394         int size = font::SIZE_LARGE;
 395 
 396         while(size > 0) {
 397                 if(font::line_width(str, size) > getx()) {
 398                         size--;
 399                 } else {
 400                         break;
 401                 }
 402         }
 403 
 404         const int border = 5;
 405 
 406         font::floating_label flabel(str);
 407         flabel.set_font_size(size);
 408         flabel.set_position(getx()/2, gety());
 409         flabel.set_bg_color(color);
 410         flabel.set_border_size(border);
 411 
 412         help_string_ = font::add_floating_label(flabel);
 413 
 414         const SDL_Rect& rect = font::get_floating_label_rect(help_string_);
 415         font::move_floating_label(help_string_,0.0,-double(rect.h));
 416         return help_string_;
 417 }

 432 void CVideo::set_fullscreen(bool ison)
 433 {
 434 
 435         if(window && isFullScreen() != ison) {
 436 
 437                 const std::pair<int,int>& res = preferences::resolution();
 438 
 439                 MODE_EVENT mode;
 440 
 441                 if(ison) {
 442                         mode = TO_FULLSCREEN;
 443                 } else {
 444                         mode = preferences::maximized() ? TO_MAXIMIZED_WINDOW : TO_WINDOWED;
 445                 }
 446 
 447                 setMode(res.first, res.second, mode);
 448 
 449                 if(display::get_singleton()) {
 450                         display::get_singleton()->redraw_everything();
 451                 }
 452         }
 453 
 454         // Change the config value.
 455         preferences::_set_fullscreen(ison);
 456 }

 463 void CVideo::set_resolution(const unsigned width, const unsigned height)
 464 {
 465         if(static_cast<unsigned int>(current_resolution().first)  == width &&
 466                 static_cast<unsigned int>(current_resolution().second) == height) {
 467 
 468                 return;
 469         }
 470

 473         if(display::get_singleton()) {
 474                 display::get_singleton()->redraw_everything();
 475         }
 476 
 477         // Change the config values.
 478         preferences::_set_resolution(std::make_pair(width, height));
 479         preferences::_set_maximized(false);
