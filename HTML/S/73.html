 111 void editor_controller::init_tods(const config& game_config)
 112 {
 113         for (const config &schedule : game_config.child_range("editor_times")) {
 114 
 115                 const std::string& schedule_id = schedule["id"];
 116                 const std::string& schedule_name = schedule["name"];
 117                 if (schedule_id.empty()) {
 118                         ERR_ED << "Missing ID attribute in a TOD Schedule." << std::endl;
 119                         continue;
 120                 }
 121 
 122                 tods_map::iterator times = tods_.find(schedule_id);
 123                 if (times == tods_.end()) {
 124                         std::pair<tods_map::iterator, bool> new_times =
 125                                         tods_.insert( std::pair<std::string, std::pair<std::string, std::vector<time_of_day> > >
 126                         (schedule_id, std::pair<std::string, std::vector<time_of_day> >(schedule_name, std::vector<time_of_day>())) );
 127                         times = new_times.first;
 128                 } else {
 129                         ERR_ED << "Duplicate TOD Schedule identifiers." << std::endl;
 130                         continue;
 131                 }
 132 
 133                 for (const config &time : schedule.child_range("time")) {
 134                         times->second.second.emplace_back(time);
 135                 }
 136 
 137         }
 138 
 139         if (tods_.empty()) {
 140                 ERR_ED << "No editor time-of-day defined" << std::endl;
 141         }
 142 }

 144 void editor_controller::init_music(const config& game_config)
 145 {
 146         const std::string tag_name = "editor_music";
 147         if (!game_config.has_child(tag_name))
 148                 ERR_ED << "No editor music defined" << std::endl;
 149         else {
 150                 for (const config& editor_music : game_config.child_range(tag_name)) {
 151                         for (const config& music : editor_music.child_range("music")) {
 152                                 sound::music_track track(music);
 153                                 if (track.file_path().empty())
 154                                         WRN_ED << "Music track " << track.id() << " not found." << std::endl;
 155                                 else
 156                                         music_tracks_.emplace_back(music);
 157                         }
 158                 }
 159         }
 160 }

 201 {
 202         std::string modified;
 203         size_t amount = context_manager_->modified_maps(modified);
 204 
 205         std::string message;
 206         if (amount == 0) {
 207                 message = _("Do you really want to quit?");
 208         } else if (amount == 1) {
 209                 message = _("Do you really want to quit? Changes to this map since the last save will be lost.");
 210         } else {
 211                 message = _("Do you really want to quit? The following maps were modified and all changes since the last save will be lost:");
 212                 message += "\n" + modified;
 213         }
 214         return quit_confirmation::show_prompt(message);

 524         case HOTKEY_MINIMAP_CODING_UNIT:
 525                 return (preferences::minimap_movement_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 526         case HOTKEY_MINIMAP_CODING_TERRAIN:
 527                 return (preferences::minimap_terrain_coding()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 528         case HOTKEY_MINIMAP_DRAW_UNITS:
 529                 return (preferences::minimap_draw_units()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 530         case HOTKEY_MINIMAP_DRAW_TERRAIN:
 531                 return (preferences::minimap_draw_terrain()) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 532         case HOTKEY_ZOOM_DEFAULT:
 533                 return (gui_->get_zoom_factor() == 1.0) ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 534 
 535         case HOTKEY_NULL:
 536                 switch (active_menu_) {
 537                 case editor::MAP:
 538                         return index == context_manager_->current_context_index()
 539                                         ? ACTION_SELECTED : ACTION_DESELECTED;
 540                 case editor::LOAD_MRU:
 541                         return ACTION_STATELESS;
 542                 case editor::PALETTE:
 543                         return ACTION_STATELESS;
 544                 case editor::AREA:
 545                         return index == get_current_map_context().get_active_area()
 546                                         ? ACTION_SELECTED : ACTION_DESELECTED;
 547                 case editor::SIDE:
 548                         return static_cast<size_t>(index) == gui_->playing_team()
 549                                         ? ACTION_SELECTED : ACTION_DESELECTED;
 550                 case editor::TIME:
 551                         return index == get_current_map_context().get_time_manager()->get_current_time()
 552                                         ? ACTION_SELECTED : ACTION_DESELECTED;
 553                 case editor::LOCAL_TIME:
 554                         return index == get_current_map_context().get_time_manager()->get_current_area_time(
 555                                         get_current_map_context().get_active_area())
 556                                         ? ACTION_SELECTED : ACTION_DESELECTED;
 557                 case editor::MUSIC:
 558                         return get_current_map_context().is_in_playlist(music_tracks_[index].id())
 559                                         ? ACTION_ON : ACTION_OFF;
 560                 case editor::SCHEDULE:
 561                         {
 562                                 tods_map::const_iterator it = tods_.begin();
 563                                 std::advance(it, index);
 564                                 const std::vector<time_of_day>& times1 = it->second.second;
 565                                 const std::vector<time_of_day>& times2 = get_current_map_context().get_time_manager()->times();
 566                                 return (times1 == times2) ? ACTION_SELECTED : ACTION_DESELECTED;
 567                         }
 568                 case editor::LOCAL_SCHEDULE:
 569                         {
 570                                 tods_map::const_iterator it = tods_.begin();
 571                                 std::advance(it, index);
 572                                 const std::vector<time_of_day>& times1 = it->second.second;
 573                                 int active_area = get_current_map_context().get_active_area();
 574                                 const std::vector<time_of_day>& times2 = get_current_map_context().get_time_manager()->times(active_area);
 575                                 return (times1 == times2) ? ACTION_SELECTED : ACTION_DESELECTED;
 576                         }
 577                 case editor::UNIT_FACING:
 578                         {
 579                                 unit_map::const_unit_iterator un = get_current_map_context().get_units().find(gui_->mouseover_hex());
 580                                 assert(un != get_current_map_context().get_units().end());
 581                                 return un->facing() == index ? ACTION_SELECTED : ACTION_DESELECTED;
 582                         }
 583                 }
 584                 return ACTION_ON;
 585                 default:
 586                         return command_executor::get_action_state(command, index);
 587         }

1009 
1010         std::vector<config> items;
1011         for(const auto& c : items_arg) {
1012                 const std::string& id = c["id"];
1013                 const hotkey::hotkey_command& command = hotkey::get_hotkey_command(id);
1014 
1015                 if((can_execute_command(command) && (!context_menu || in_context_menu(command.id)))
1016                         || command.id == hotkey::HOTKEY_NULL)
1017                 {
1018                         items.emplace_back(config_of("id", id));
1019                 }
1020         }
1021 
1022         // No point in showing an empty menu.
1023         if(items.empty()) {
1024                 return;
1025         }
1026

1028         const std::string& first_id = items.front()["id"];
1029

1031                 active_menu_ = editor::LOAD_MRU;

1052                 context_manager_->expand_areas_menu(items, 0);

1054

1056                 active_menu_ = editor::TIME;
1057                 context_manager_->expand_time_menu(items, 0);
1058         }
1059

1066                 active_menu_ = editor::UNIT_FACING;
1067                 auto pos = items.erase(items.begin());
1068                 int dir = 0;
1069                 std::generate_n(std::inserter<std::vector<config>>(items, pos), int(map_location::NDIRECTIONS), [&dir]() -> config {

1254 void editor_controller::mouse_motion(int x, int y, const bool /*browse*/,
1255                 bool update, map_location /*new_loc*/)
1256 {
1257         if (mouse_handler_base::mouse_motion_default(x, y, update)) return;
1258         map_location hex_clicked = gui().hex_clicked_on(x, y);
1259         if (context_manager_->get_map().on_board_with_border(drag_from_hex_) && is_dragging()) {
1260                 editor_action* a = nullptr;
1261                 bool partial = false;
1262                 editor_action* last_undo = get_current_map_context().last_undo_action();
1263                 if (dragging_left_ && (SDL_GetMouseState(nullptr, nullptr) & SDL_BUTTON(1)) != 0) {
1264                         if (!context_manager_->get_map().on_board_with_border(hex_clicked)) return;
1265                         a = get_mouse_action().drag_left(*gui_, x, y, partial, last_undo);
1266                 } else if (dragging_right_ && (SDL_GetMouseState(nullptr, nullptr) & SDL_BUTTON(3)) != 0) {
1267                         if (!context_manager_->get_map().on_board_with_border(hex_clicked)) return;
1268                         a = get_mouse_action().drag_right(*gui_, x, y, partial, last_undo);
1269                 }
1270                 //Partial means that the mouse action has modified the
1271                 //last undo action and the controller shouldn't add
1272                 //anything to the undo stack (hence a different perform_ call)
1273                 if (a != nullptr) {
1274                         const std::unique_ptr<editor_action> aa(a);
1275                         if (partial) {
1276                                 get_current_map_context().perform_partial_action(*a);
1277                         } else {
1278                                 get_current_map_context().perform_action(*a);
1279                         }
1280                         context_manager_->refresh_after_action(true);
1281                 }
1282         } else {
1283                 get_mouse_action().move(*gui_, hex_clicked);
1284         }
1285         gui().highlight_hex(hex_clicked);
1286 }
