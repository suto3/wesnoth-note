 298 static config unit_status(reports::context & rc, const unit* u)
 299 {
 300         if (!u) return config();
 301         config res;
 302         map_location displayed_unit_hex = rc.screen().displayed_unit_hex();
 303         if (rc.map().on_board(displayed_unit_hex) && u->invisible(displayed_unit_hex, rc.dc())) {
 304                 add_status(res, "misc/invisible.png", N_("invisible: "),
 305                         N_("This unit is invisible. It cannot be seen or attacked by enemy units."));
 306         }
 307         if (u->get_state(unit::STATE_SLOWED)) {
 308                 add_status(res, "misc/slowed.png", N_("slowed: "),
 309                         N_("This unit has been slowed. It will only deal half its normal damage when attacking and its movement cost is doubled."));
 310         }
 311         if (u->get_state(unit::STATE_POISONED)) {
 312                 add_status(res, "misc/poisoned.png", N_("poisoned: "),
 313                         N_("This unit is poisoned. It will lose 8 HP every turn until it can seek a cure to the poison in a village or from a friendly unit with the ‘cures’ ability.\n\nUnits cannot be killed by poison alone. The poison will not reduce it below 1 HP."));
 314         }
 315         if (u->get_state(unit::STATE_PETRIFIED)) {
 316                 add_status(res, "misc/petrified.png", N_("petrified: "),
 317                         N_("This unit has been petrified. It may not move or attack."));
 318         }
 319         return res;
 320 }

 365 static config unit_abilities(const unit* u)
 366 {
 367         if (!u) return config();
 368         config res;
 369 
 370         boost::dynamic_bitset<> active;
 371         const std::vector<std::tuple<t_string,t_string,t_string> > &abilities = u->ability_tooltips(&active);
 372         const size_t abilities_size = abilities.size();
 373         for ( size_t i = 0; i != abilities_size; ++i )
 374         {
 375                 // Aliases for readability:
 376                 const std::string &base_name = std::get<0>(abilities[i]).base_str();
 377                 const t_string &display_name = std::get<1>(abilities[i]);
 378                 const t_string &description  = std::get<2>(abilities[i]);
 379 
 380                 std::ostringstream str, tooltip;
 381 
 382                 if ( active[i] )
 383                         str << display_name;
 384                 else
 385                         str << span_color(font::inactive_ability_color) << display_name << naps;
 386                 if ( i + 1 != abilities_size )
 387                         str << ", ";
 388 
 389                 tooltip << _("Ability: ") << "<b>" << display_name << "</b>";
 390                 if ( !active[i] )
 391                         tooltip << "<i>" << _(" (inactive)") << "</i>";
 392                 tooltip << '\n' << description;
 393 
 394                 add_text(res, str.str(), tooltip.str(), "ability_" + base_name);
 395         }
 396         return res;
 397 }

 410 static config unit_hp(reports::context& rc, const unit* u)
 411 {
 412         if (!u) return config();
 413         std::ostringstream str, tooltip;
 414         str << span_color(u->hp_color()) << u->hitpoints()
 415                 << '/' << u->max_hitpoints() << naps;
 416 
 417         std::set<std::string> resistances_table;
 418 
 419         bool att_def_diff = false;
 420         map_location displayed_unit_hex = rc.screen().displayed_unit_hex();
 421         for (const utils::string_map::value_type &resist : u->get_base_resistances())
 422         {
 423                 std::ostringstream line;
 424                 line << translation::gettext(resist.first.c_str()) << ": ";
 425                 // Some units have different resistances when attacking or defending.
 426                 int res_att = 100 - u->resistance_against(resist.first, true, displayed_unit_hex);
 427                 int res_def = 100 - u->resistance_against(resist.first, false, displayed_unit_hex);
 428                 const std::string def_color = unit_helper::resistance_color(res_def);
 429                 if (res_att == res_def) {
 430                         line << "<span foreground=\"" << def_color << "\">" << utils::signed_percent(res_def)
 431                         << naps << '\n';
 432                 } else {
 433                         const std::string att_color = unit_helper::resistance_color(res_att);
 434                         line << "<span foreground=\"" << att_color << "\">" << utils::signed_percent(res_att)
 435                         << naps << "/"
 436                         << "<span foreground=\"" << def_color << "\">" << utils::signed_percent(res_def)
 437                         << naps << '\n';
 438                         att_def_diff = true;
 439                 }
 440                 resistances_table.insert(line.str());
 441         }
 442 
 443         tooltip << _("Resistances: ");
 444         if (att_def_diff)
 445                 tooltip << _("(Att / Def)");
 446         tooltip << '\n';
 447         for (const std::string &line : resistances_table) {
 448                 tooltip << line;
 449         }
 450         return text_report(str.str(), tooltip.str());
 451 }

 506 static config unit_defense(reports::context & rc, const unit* u, const map_location& displayed_unit_hex)
 507 {
 508         if(!u) {
 509                 return config();
 510         }
 511 
 512         std::ostringstream str, tooltip;
 513         const gamemap &map = rc.map();
 514         if(!rc.map().on_board(displayed_unit_hex)) {
 515                 return config();
 516         }
 517 
 518         const t_translation::terrain_code &terrain = map[displayed_unit_hex];
 519         int def = 100 - u->defense_modifier(terrain);
 520         color_t color = game_config::red_to_green(def);
 521         str << span_color(color) << def << '%' << naps;
 522         tooltip << _("Terrain: ") << "<b>" << map.get_terrain_info(terrain).description() << "</b>\n";
 523 
 524         const t_translation::ter_list &underlyings = map.underlying_def_terrain(terrain);
 525         if (underlyings.size() != 1 || underlyings.front() != terrain)
 526         {
 527                 bool revert = false;
 528                 for (const t_translation::terrain_code &t : underlyings)
 529                 {
 530                         if (t == t_translation::MINUS) {
 531                                 revert = true;
 532                         } else if (t == t_translation::PLUS) {
 533                                 revert = false;
 534                         } else {
 535                                 int t_def = 100 - u->defense_modifier(t);
 536                                 color_t t_color = game_config::red_to_green(t_def);
 537                                 tooltip << '\t' << map.get_terrain_info(t).description() << ": "
 538                                         << span_color(t_color) << t_def << '%' << naps
 539                                         << (revert ? _("maximum^max.") : _("minimum^min.")) << '\n';
 540                         }
 541                 }
 542         }
 543 
 544         tooltip << "<b>" << _("Defense: ") << span_color(color)  << def << '%' << naps << "</b>";
 545         return text_report(str.str(), tooltip.str());
 546 }

 587                 movement_frac = double(u->movement_left()) / std::max<int>(1, u->total_movement());
 588                 if (movement_frac > 1.0)
 589                         movement_frac = 1.0;
 590         }
 591 
 592         std::set<t_translation::terrain_code>::const_iterator terrain_it =
 593                                 preferences::encountered_terrains().begin();

 599                 if (terrain == t_translation::FOGGED || terrain == t_translation::VOID_TERRAIN || t_translation::terrain_matches(terrain, t_translation::ALL_OFF_MAP))

 648 static int attack_info(reports::context & rc, const attack_type &at, config &res, const unit &u, const map_location &displayed_unit_hex)
 649 {
 650         std::ostringstream str, tooltip;
 651 
 652         at.set_specials_context(displayed_unit_hex, u.side() == rc.screen().playing_side());
 653         int base_damage = at.damage();
 654         int specials_damage = at.modified_damage(false);
 655         int damage_multiplier = 100;
 656         int tod_bonus = combat_modifier(rc.units(), rc.map(), displayed_unit_hex, u.alignment(), u.is_fearless());
 657         damage_multiplier += tod_bonus;
 658         int leader_bonus = under_leadership(rc.units(), displayed_unit_hex).first;
 659         if (leader_bonus != 0)
 660                 damage_multiplier += leader_bonus;
 661 
 662         bool slowed = u.get_state(unit::STATE_SLOWED);
 663         int damage_divisor = slowed ? 20000 : 10000;
 664         // Assume no specific resistance (i.e. multiply by 100).
 665         int damage = round_damage(specials_damage, damage_multiplier * 100, damage_divisor);
 666 
 667         // Hit points are used to calculate swarm, so they need to be bounded.
 668         unsigned max_hp = u.max_hitpoints();
 669         unsigned cur_hp = std::min<unsigned>(std::max(0, u.hitpoints()), max_hp);
 670 
 671         unsigned base_attacks = at.num_attacks();
 672         unsigned min_attacks, max_attacks;
 673         at.modified_attacks(false, min_attacks, max_attacks);
 674         unsigned num_attacks = swarm_blows(min_attacks, max_attacks, cur_hp, max_hp);
 675 
 676         color_t dmg_color = font::weapon_color;
 677         if ( damage > specials_damage )
 678                 dmg_color = font::good_dmg_color;
 679         else if ( damage < specials_damage )
 680                 dmg_color = font::bad_dmg_color;
 681 
 682         str << span_color(dmg_color) << "  " << damage << naps << span_color(font::weapon_color)
 683                 << font::weapon_numbers_sep << num_attacks << ' ' << at.name()
 684                 << "</span>\n";
 685         tooltip << _("Weapon: ") << "<b>" << at.name() << "</b>\n"
 686                 << _("Damage: ") << "<b>" << damage << "</b>\n";
 687 
 688         if ( tod_bonus || leader_bonus || slowed || specials_damage != base_damage )
 689         {
 690                 tooltip << '\t' << _("Base damage: ") << base_damage << '\n';
 691                 if ( specials_damage != base_damage ) {
 692                         tooltip << '\t' << _("With specials: ") << specials_damage << '\n';
 693                 }
 694                 if (tod_bonus) {
 695                         tooltip << '\t' << _("Time of day: ")
 696                                 << utils::signed_percent(tod_bonus) << '\n';
 697                 }
 698                 if (leader_bonus) {
 699                         tooltip << '\t' << _("Leadership: ")
 700                                 << utils::signed_percent(leader_bonus) << '\n';
 701                 }
 702                 if (slowed) {
 703                         tooltip << '\t' << _("Slowed: ") << "/ 2" << '\n';
 704                 }
 705         }
 706 
 707         tooltip << _("Attacks: ") << "<b>" << num_attacks << "</b>\n";
 708         if ( max_attacks != min_attacks  &&  cur_hp != max_hp ) {
 709                 if ( max_attacks < min_attacks ) {
 710                         // "Reverse swarm"
 711                         tooltip << '\t' << _("Max swarm bonus: ") << (min_attacks-max_attacks) << '\n';
 712                         tooltip << '\t' << _("Swarm: ") << "* "<< (100 - cur_hp*100/max_hp) << "%\n";
 713                         tooltip << '\t' << _("Base attacks: ") << '+' << base_attacks << '\n';
 714                         // The specials line will not necessarily match up with how the
 715                         // specials are calculated, but for an unusual case, simple brevity
 716                         // trumps complexities.
 717                         if ( max_attacks != base_attacks ) {
 718                                 int attack_diff = int(max_attacks) - int(base_attacks);
 719                                 tooltip << '\t' << _("Specials: ") << utils::signed_value(attack_diff) << '\n';
 720                         }
 721                 }
 722                 else {
 723                         // Regular swarm
 724                         tooltip << '\t' << _("Base attacks: ") << base_attacks << '\n';
 725                         if ( max_attacks != base_attacks ) {
 726                                 tooltip << '\t' << _("With specials: ") << max_attacks << '\n';
 727                         }
 728                         if ( min_attacks != 0 ) {
 729                                 tooltip << '\t' << _("Subject to swarm: ") << (max_attacks-min_attacks) << '\n';
 730                         }
 731                         tooltip << '\t' << _("Swarm: ") << "* "<< (cur_hp*100/max_hp) << "%\n";
 732                 }
 733         }
 734         else if ( num_attacks != base_attacks ) {
 735                 tooltip << '\t' << _("Base attacks: ") << base_attacks << '\n';
 736                 tooltip << '\t' << _("With specials: ") << num_attacks << '\n';
 737         }
 738 
 739         add_text(res, flush(str), flush(tooltip));
 740 
 741         std::string range = string_table["range_" + at.range()];
 742         std::string lang_type = string_table["type_" + at.type()];
 743 
 744         str << span_color(font::weapon_details_color) << "  " << "  "
 745                 << range << font::weapon_details_sep
 746                 << lang_type << "</span>\n";
 747 
 748         tooltip << _("Weapon range: ") << "<b>" << range << "</b>\n"
 749                 << _("Damage type: ")  << "<b>" << lang_type << "</b>\n"
 750                 << _("Damage versus: ") << '\n';
 751 
 752         // Show this weapon damage and resistance against all the different units.
 753         // We want weak resistances (= good damage) first.
 754         std::map<int, std::set<std::string>, std::greater<int> > resistances;
 755         std::set<std::string> seen_types;
 756         const team &unit_team = rc.dc().get_team(u.side());
 757         const team &viewing_team = rc.teams()[rc.screen().viewing_team()];
 758         for (const unit &enemy : rc.units())
 759         {
 760                 if (enemy.incapacitated()) //we can't attack statues so don't display them in this tooltip
 761                         continue;
 762                 if (!unit_team.is_enemy(enemy.side()))
 763                         continue;
 764                 const map_location &loc = enemy.get_location();
 765                 if (viewing_team.fogged(loc) ||
 766                     (viewing_team.is_enemy(enemy.side()) && enemy.invisible(loc, rc.dc())))
 767                         continue;
 768                 bool new_type = seen_types.insert(enemy.type_id()).second;
 769                 if (new_type) {
 770                         int resistance = enemy.resistance_against(at, false, loc);
 771                         resistances[resistance].insert(enemy.type_name());
 772                 }
 773         }
 774 
 775         typedef std::pair<int, std::set<std::string> > resist_units;
 776         for (const resist_units &resist : resistances) {
 777                 int damage_with_resistance = round_damage(specials_damage, damage_multiplier * resist.first, damage_divisor);
 778                 tooltip << "<b>" << damage_with_resistance << "</b>  "
 779                         << "<i>(" << utils::signed_percent(resist.first-100) << ")</i> : "
 780                         << utils::join(resist.second, ", ") << '\n';
 781         }
 782         add_text(res, flush(str), flush(tooltip));
 783 
 784         const std::string &accuracy_parry = at.accuracy_parry_description();
 785         if (!accuracy_parry.empty())
 786         {
 787                 str << span_color(font::weapon_details_color)
 788                         << "  " << accuracy_parry << "</span>\n";
 789                 int accuracy = at.accuracy();
 790                 if (accuracy) {
 791                         tooltip << _("Accuracy:") << "<b>"
 792                                 << utils::signed_percent(accuracy) << "</b>\n";
 793                 }
 794                 int parry = at.parry();
 795                 if (parry) {
 796                         tooltip << _("Parry:") << "<b>"
 797                                 << utils::signed_percent(parry) << "</b>\n";
 798                         }
 799                 add_text(res, flush(str), flush(tooltip));
 800         }
 801 
 802         at.set_specials_context_for_listing();
 803         boost::dynamic_bitset<> active;
 804         const std::vector<std::pair<t_string, t_string> > &specials = at.special_tooltips(&active);
 805         const size_t specials_size = specials.size();
 806         for ( size_t i = 0; i != specials_size; ++i )
 807         {
 808                 // Aliases for readability:
 809                 const t_string &name = specials[i].first;
 810                 const t_string &description = specials[i].second;
 811                 const color_t &details_color = active[i] ? font::weapon_details_color :
 812                                                              font::inactive_details_color;
 813 
 814                 str << span_color(details_color) << "  " << "  " << name << naps << '\n';
 815                 std::string help_page = "weaponspecial_" + name.base_str();
 816                 tooltip << _("Weapon special: ") << "<b>" << name << "</b>";
 817                 if ( !active[i] )
 818                         tooltip << "<i>" << _(" (inactive)") << "</i>";
 819                 tooltip << '\n' << description;
 820 
 821                 add_text(res, flush(str), flush(tooltip), help_page);
 822         }
 823         return damage;
 824 }

 844 static config unit_weapons(reports::context & rc, const unit *attacker, const map_location &attacker_pos, const unit *defender, bool show_attacker)
 845 {
 846         if (!attacker || !defender) return config();
 847 
 848         const unit* u = show_attacker ? attacker : defender;
 849         const map_location unit_loc = show_attacker ? attacker_pos : defender->get_location();
 850 
 851         std::ostringstream str, tooltip;
 852         config res;
 853 
 854         std::vector<battle_context> weapons;
 855         for (unsigned int i = 0; i < attacker->attacks().size(); i++) {
 856                 // skip weapons with attack_weight=0
 857                 if (attacker->attacks()[i].attack_weight() > 0) {
 858                         battle_context weapon(rc.units(), attacker_pos, defender->get_location(), i, -1, 0.0, nullptr, attacker);
 859                         weapons.push_back(weapon);
 860                 }
 861         }
 862 
 863         for (const battle_context& weapon : weapons) {
 864 
 865                 // Predict the battle outcome.
 866                 combatant attacker_combatant(weapon.get_attacker_stats());
 867                 combatant defender_combatant(weapon.get_defender_stats());
 868                 attacker_combatant.fight(defender_combatant);
 869 
 870                 const battle_context_unit_stats& context_unit_stats =
 871                                 show_attacker ? weapon.get_attacker_stats() : weapon.get_defender_stats();
 872 
 873                 int total_damage = 0;
 874                 int base_damage = 0;
 875                 int num_blows = 0;
 876                 int chance_to_hit = 0;
 877                 t_string weapon_name = _("None");
 878 
 879                 color_t dmg_color = font::weapon_color;
 880                 if (context_unit_stats.weapon) {
 881                         base_damage = attack_info(rc, *context_unit_stats.weapon, res, *u, unit_loc);
 882                         total_damage = context_unit_stats.damage;
 883                         num_blows = context_unit_stats.num_blows;
 884                         chance_to_hit = context_unit_stats.chance_to_hit;
 885                         weapon_name = context_unit_stats.weapon->name();
 886 
 887                         if ( total_damage > base_damage )
 888                                 dmg_color = font::good_dmg_color;
 889                         else if ( total_damage < base_damage )
 890                                 dmg_color = font::bad_dmg_color;
 891                 } else {
 892                         str << span_color(font::weapon_color) << weapon_name << naps << "\n";
 893                         tooltip << _("Weapon: ") << "<b>" << weapon_name << "</b>\n"
 894                                 << _("Damage: ") << "<b>" << "0" << "</b>\n";
 895                 }
 896 
 897                 color_t chance_color = game_config::red_to_green(chance_to_hit);
 898 
 899                 // Total damage.
 900                 str << "  " << span_color(dmg_color) << total_damage << naps << span_color(font::weapon_color)
 901                         << font::unicode_en_dash << num_blows
 902                         << " (" << span_color(chance_color) << chance_to_hit << "%" << naps << ")"
 903                         << naps << "\n";
 904 
 905                 tooltip << _("Weapon: ") << "<b>" << weapon_name << "</b>\n"
 906                                 << _("Total damage") << "<b>" << total_damage << "</b>\n";
 907 
 908                 // Create the hitpoints distribution.
 909                 std::vector<std::pair<int, double> > hp_prob_vector;
 910 
 911                 // First, we sort the probabilities in ascending order.
 912                 std::vector<std::pair<double, int> > prob_hp_vector;
 913                 int i;
 914 
 915                 combatant* c = show_attacker ? &attacker_combatant : &defender_combatant;
 916 
 917                 for(i = 0; i < static_cast<int>(c->hp_dist.size()); i++) {
 918                         double prob = c->hp_dist[i];
 919 
 920                         // We keep only values above 0.1%.
 921                         if(prob > 0.001)
 922                                 prob_hp_vector.emplace_back(prob, i);
 923                 }
 924 
 925                 std::sort(prob_hp_vector.begin(), prob_hp_vector.end());
 926 
 927                 //TODO fendrin -- make that dynamically
 928                 int max_hp_distrib_rows_ = 10;
 929 
 930                 // We store a few of the highest probability hitpoint values.
 931                 int nb_elem = std::min<int>(max_hp_distrib_rows_, prob_hp_vector.size());
 932 
 933                 for(i = prob_hp_vector.size() - nb_elem;
 934                                 i < static_cast<int>(prob_hp_vector.size()); i++) {
 935 
 936                         hp_prob_vector.emplace_back(prob_hp_vector[i].second, prob_hp_vector[i].first);
 937                 }
 938 
 939                 // Then, we sort the hitpoint values in ascending order.
 940                 std::sort(hp_prob_vector.begin(), hp_prob_vector.end());
 941                 // And reverse the order. Might be doable in a better manor.
 942                 std::reverse(hp_prob_vector.begin(), hp_prob_vector.end());
 943 
 944                 for(i = 0; i < static_cast<int>(hp_prob_vector.size()); i++) {
 945 
 946                         int hp = hp_prob_vector[i].first;
 947                         double prob = hp_prob_vector[i].second;
 948                         color_t prob_color = game_config::blue_to_white(prob * 100.0, true);
 949 
 950                         str             << span_color(font::weapon_details_color) << "  " << "  "
 951                                         << span_color(u->hp_color(hp)) << format_hp(hp) << naps
 952                                         << " " << font::weapon_numbers_sep << " "
 953                                         << span_color(prob_color) << format_prob(prob) << naps
 954                                         << naps << "\n";
 955                 }
 956 
 957                 add_text(res, flush(str), flush(tooltip));
 958         }
 959         return res;
 960 }
