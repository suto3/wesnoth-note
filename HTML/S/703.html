 195         template<class T>
 196         void test_resolutions(const resolution_list& resolutions)
 197         {
 198                 for(const resolution& resolution : resolutions) {
 199                         CVideo& video = test_utils::get_fake_display(resolution.first, resolution.second).video();
 200 
 201                         dialog_tester<T> ctor;
 202                         const std::unique_ptr<modal_dialog> dlg(ctor.create());
 203                         BOOST_REQUIRE_MESSAGE(dlg.get(), "Failed to create a dialog.");
 204 
 205                         const std::string id = unit_test_mark_as_tested(*(dlg.get()));
 206 
 207                         std::string exception;
 208                         try {
 209                                 dlg->show(video, 1);
 210                         } catch(gui2::layout_exception_width_modified&) {
 211                                 exception = "gui2::layout_exception_width_modified";
 212                         } catch(gui2::layout_exception_width_resize_failed&) {
 213                                 exception = "gui2::layout_exception_width_resize_failed";
 214                         } catch(gui2::layout_exception_height_resize_failed&) {
 215                                 exception = "gui2::layout_exception_height_resize_failed";
 216                         } catch(wml_exception& e) {
 217                                 exception = e.dev_message;
 218                         } catch(std::exception& e) {
 219                                 exception = e.what();
 220                         } catch(...) {
 221                                 exception = "unknown";
 222                         }
 223                         BOOST_CHECK_MESSAGE(exception.empty(),
 224                                         "Test for '" << id
 225                                         << "' Failed\nnew widgets = " << gui2::new_widgets
 226                                         << " resolution = " << resolution.first
 227                                         << 'x' << resolution.second
 228                                         << "\nException caught: " << exception << '.');
 229                 }
