 211 void game::perform_controller_tweaks() {
 212         const simple_wml::node::child_list & sides = get_sides_list();
 213 
 214         DBG_GAME << "****\n Performing controller tweaks. sides = " << std::endl;
 215         DBG_GAME << debug_sides_info() << std::endl;
 216         DBG_GAME << "****" << std::endl;
 217 
 218         update_side_data(); // Necessary to read the level_ and get sides_, etc. updated to match
 219 
 220         for(simple_wml::node::child_list::const_iterator s = sides.begin(); s != sides.end(); ++s) {
 221                 if ((**s)["controller"] != "null") {
 222                         const size_t side_index = s - sides.begin();
 223                         if (sides_[side_index] == 0) {
 224                                 sides_[side_index] = owner_;
 225                                 std::stringstream msg;
 226                                 msg << "Side "  << side_index + 1 << " had no controller during controller tweaks! The host was assigned control.";
 227                                 LOG_GAME << msg.str() << " (game id: " << id_ << ")\n";
 228                                 send_and_record_server_message(msg.str());
 229                         }
 230 
 231                         const auto user = player_connections_.find(sides_[side_index]);
 232                         std::string user_name = "null (server missing user)";
 233                         if (user == player_connections_.end()) {
 234                                 missing_user(user->socket(), __func__);
 235                         } else {
 236                                 user_name = username(user->socket());
 237                         }
 238 
 239                         // Issue change_controller command, transfering this side to its owner with proper name and controller.
 240                         // Ensures that what the server now thinks is true is effected on all of the clients.
 241                         //
 242                         // We use the "player_left" field as follows. Normally change_controller sends one message to the owner,
 243                         // and one message to everyone else. In case that a player drops, the owner is gone and should not get
 244                         // a message, instead the host gets a [side_drop] message.
 245                         //
 246                         // In the server controller tweaks, we want to avoid sending controller change messages to the host.
 247                         // Doing this has the negative consequence that all of the AI side names are given the owners name.
 248                         // Therefore, if the side belongs to the host, we pass player_left = true, otherwise player_left = false.
 249                         change_controller(side_index, sides_[side_index], user_name , sides_[side_index] == owner_);
 250 
 251                         //next line change controller types found in level_ to be what is appropriate for an observer at game start.
 252                         (*s)->set_attr("is_local", "no");
 253 
 254                         if (sides_[side_index] == 0) {
 255                                 std::stringstream msg;
 256                                 msg << "Side " << side_index + 1 << " had no controller AFTER controller tweaks! Ruh Roh!";
 257                                 LOG_GAME << msg.str() << " (game id: " << id_ << ")\n";
 258                         }
 259                 }
 260         }
 261 
 262         update_side_data(); // this is the last time that update_side_data will actually run, as now the game will start and started_ will be true.
 263 
 264         //TODO: Does it matter that the server is telling the host to change a bunch of sides?
 265         //According to playturn.cpp, the host should ignore all such messages. Still might be better
 266         //not to send them at all, although not if it complicates the server code.
 267 }

 271         const simple_wml::node::child_list & sides = get_sides_list();
 272         DBG_GAME << "****\n Starting game. sides = " << std::endl;
 273         DBG_GAME << debug_sides_info() << std::endl;
 274         DBG_GAME << "****" << std::endl;
 275 
 276 
 277         started_ = true;
 278         // Prevent inserting empty keys when reading.
 279         const simple_wml::node& multiplayer = get_multiplayer(level_.root());
 280 
 281         const bool save = multiplayer["savegame"].to_bool();
 282         LOG_GAME << client_address(starter) << "\t"
 283                 << player_connections_.find(starter)->name() << "\t" << "started"
 284                 << (save ? " reloaded" : "") << " game:\t\"" << name_ << "\" (" << id_
 285                 // << ") with: " << list_users(players_, __func__) << ". Settings: map: " << s["id"]
 286                 << ") with: " << list_users(players_, __func__) << ". Settings: map: " << multiplayer["mp_scenario"]
 287                 // << "\tera: "       << (s.child("era") ? (*s.child("era"))["id"] : "")
 288                 << "\tera: "       << multiplayer["mp_era"]
 289                 << "\tXP: "        << multiplayer["experience_modifier"]
 290                 << "\tGPV: "       << multiplayer["mp_village_gold"]
 291                 << "\tfog: "       << multiplayer["mp_fog"]
 292                 << "\tshroud: "    << multiplayer["mp_shroud"]
 293                 << "\tobservers: " << multiplayer["observer"]
 294                 << "\tshuffle: "   << multiplayer["shuffle_sides"]
 295                 << "\ttimer: "     << multiplayer["mp_countdown"]
 296                 << (multiplayer["mp_countdown"].to_bool() ?
 297                         "\treservoir time: " + multiplayer["mp_countdown_reservoir_time"].to_string() +
 298                         "\tinit time: "      + multiplayer["mp_countdown_init_time"].to_string() +
 299                         "\taction bonus: "   + multiplayer["mp_countdown_action_bonus"].to_string() +
 300                         "\tturn bonus: "     + multiplayer["mp_countdown_turn_bonus"].to_string() : "")
 301                 << "\n";
 302 
 303         for(simple_wml::node::child_list::const_iterator s = sides.begin(); s != sides.end(); ++s) {
 304                 if ((**s)["controller"] != "null") {
 305                         const size_t side_index = s - sides.begin();
 306                         if(side_index >= sides_.size()) {
 307                                 continue;
 308                         }
 309                         if (sides_[side_index] == 0) {
 310                                 std::stringstream msg;
 311                                 msg << "Side "  << side_index + 1 << " has no controller but should! The host needs to assign control for the game to proceed past that side's turn.";
 312                                 LOG_GAME << msg.str() << " (game id: " << id_ << ")\n";
 313                                 send_and_record_server_message(msg.str());
 314                         }
 315                 }
 316         }
 317 
 318         DBG_GAME << "Number of sides: " << nsides_ << "\n";
 319         int turn = 1;
 320         int side = 0;
 321         // Savegames have a snapshot that tells us which side starts.
 322         if (const simple_wml::node* snapshot = level_.root().child("snapshot")) {
 323                 turn = lexical_cast_default<int>((*snapshot)["turn_at"], 1);
 324                 side = lexical_cast_default<int>((*snapshot)["playing_team"], 0);
 325                 LOG_GAME << "Reload from turn: " << turn
 326                         << ". Current side is: " << side + 1 << ".\n";
 327         }
 328         end_turn_ = (turn - 1) * nsides_ + side - 1;
 329         num_turns_ = lexical_cast_default<int>((*starting_pos( level_.root()))["turns"], -1);
 330 
 331         end_turn();
 332         clear_history();
 333         // Send [observer] tags for all observers that are already in the game.
 334         send_observerjoins();
 335 }

 346 bool game::send_taken_side(simple_wml::document& cfg, const simple_wml::node::child_list::const_iterator side) const
 347 {
 348         const size_t side_index = (**side)["side"].to_int() - 1;
 349         //negative values are casted (int -> size_t) to very high values to this check will fail for them too.
 350         if (side_index >= sides_.size()) return false;

 352         // We expect that the host will really use our proposed side number. (He could do different...)
 353         cfg.root().set_attr_dup("side", (**side)["side"]);
 354

 356         send_to_player(owner_, cfg);
 357         return true;

 360 bool game::take_side(const socket_ptr user)
 361 {
 362         DBG_GAME << "take_side...\n";
 363 
 364         if (started_) return false;
 365 
 366         simple_wml::document cfg;
 367         cfg.root().set_attr_dup("name", player_connections_.find(user)->name().c_str());
 368 
 369         //FIXME: It the client code (multiplayer.wait.cpp) the host code (connect_engine.cpp) and the server code (this file)
 370         //       Has this code to figure out a fitting side for new players, this is clearly too much.
 371         // Check if we can figure out a fitting side.
 372         const simple_wml::node::child_list& sides = get_sides_list();
 373         for(simple_wml::node::child_list::const_iterator side = sides.begin(); side != sides.end(); ++side) {
 374                 if(((**side)["controller"] == "human" || (**side)["controller"] == "reserved")
 375                                 && (**side)["current_player"] == player_connections_.find(user)->name().c_str())
 376                 {
 377                         if (send_taken_side(cfg, side)) return true;
 378                 }
 379         }
 380         // If there was no fitting side just take the first available.
 381         for(simple_wml::node::child_list::const_iterator side = sides.begin(); side != sides.end(); ++side) {
 382                 if((**side)["controller"] == "human") {
 383                         if (send_taken_side(cfg, side)) return true;
 384                 }
 385         }
 386         DBG_GAME << "take_side: there are no more sides available\n";
 387         //if we get here we couldn't find a side to take
 388         return false;

 399 void game::update_side_data()
 400 {
 401         //added by iceiceice: since level_ will now reflect how an observer
 402         //views the replay start position and not the current position, the sides_, side_controllers_,
 403         //players_ info should not be updated from the level_ after the game has started.
 404         //controller changes are now stored in the history, so an observer that joins will get up to
 405         //date that way.
 406         if (started_) return;
 407 
 408         DBG_GAME << "update_side_data...\n";
 409         DBG_GAME << debug_player_info();
 410         // Remember everyone that is in the game.
 411         const user_vector users = all_game_users();
 412         players_.clear();
 413         observers_.clear();
 414         reset_sides();
 415         const simple_wml::node::child_list& level_sides = get_sides_list();
 416 
 417         // For each user:
 418         // * Find the username.
 419         // * Find the side this username corresponds to.
 420         for (user_vector::const_iterator user = users.begin(); user != users.end(); ++user) {
 421                 auto iter = player_connections_.find(*user);
 422                 if (iter == player_connections_.end()) {
 423                         missing_user(*user, __func__);
 424                         continue;
 425                 }
 426 
 427                 bool side_found = false;
 428                 for (simple_wml::node::child_list::const_iterator side = level_sides.begin();
 429                                 side != level_sides.end(); ++side)
 430                 {
 431                         const size_t side_index = side - level_sides.begin();
 432                         if (side_index >= sides_.size()
 433                                         || sides_[side_index] != 0) continue;
 434 
 435                         const simple_wml::string_span& player_id = (**side)["player_id"];
 436                         const simple_wml::string_span& controller = (**side)["controller"];
 437                         if ( player_id == iter->info().name().c_str()) {
 438                                 if(controller != "human" && controller != "ai") {
 439                                         //we found invalid [side] data. Some message would be cool.
 440                                         continue;
 441                                 }
 442                                 side_controllers_[side_index].parse(controller);
 443                                 sides_[side_index] = *user;
 444                                 side_found = true;
 445                         }
 446                         else if (*user == owner_ && (controller == "null")) {
 447                                 //the *user == owner_ check has no effect,
 448                                 //it's just an optimisation so that we only do this once.
 449                                 side_controllers_[side_index].parse(controller);
 450                         }
 451                 }
 452                 if (side_found) {
 453                         players_.push_back(*user);
 454                         iter->info().set_status(player::PLAYING);
 455                 } else {
 456                         observers_.push_back(*user);
 457                         iter->info().set_status(player::OBSERVING);
 458                 }
 459         }
 460         DBG_GAME << debug_player_info();
 461 }

 463 void game::transfer_side_control(const socket_ptr sock, const simple_wml::node& cfg) {
 464         DBG_GAME << "transfer_side_control...\n";
 465         if (!is_player(sock) && sock != owner_) {
 466                 send_server_message("You cannot change controllers: not a player.", sock);
 467                 return;
 468         }
 469 
 470         // Check the side number.
 471         const unsigned int side_num = cfg["side"].to_int();
 472         if(side_num < 1 || side_num > sides_.size()) {
 473                 std::ostringstream msg;
 474                 msg << "The side number has to be between 1 and "
 475                     << sides_.size() << ".";
 476                 send_server_message(msg.str(), sock);
 477                 return;
 478         }
 479 
 480         if (side_num > get_sides_list().size()) {
 481                 send_server_message("Invalid side number.", sock);
 482                 return;
 483         }
 484 
 485         const simple_wml::string_span& newplayer_name = cfg["player"];
 486         const socket_ptr old_player = sides_[side_num - 1];
 487         const auto oldplayer = player_connections_.find(old_player);
 488         if (oldplayer == player_connections_.end()) missing_user(old_player, __func__);
 489         const std::string old_player_name = username(old_player);
 490 
 491         // Not supported anymore.
 492         if (newplayer_name.empty()) {
 493                 std::stringstream msg;
 494                 msg << "Recived invalid [change_controller] with no player= attribute specified";
 495                 DBG_GAME << msg.str() << "\n";
 496                 send_server_message(msg.str(), sock);
 497                 return;
 498         }
 499 
 500         // Check if the sender actually owns the side he gives away or is the host.
 501         if (!(sock == old_player || sock == owner_)) {
 502                 std::stringstream msg;
 503                 msg << "You can't give away side " << side_num << ". It's controlled by '"
 504                         << old_player_name << "' not you.";
 505                 DBG_GAME << msg.str() << "\n";
 506                 send_server_message(msg.str(), sock);
 507                 return;
 508         }
 509         //find the player that is passed control
 510         socket_ptr newplayer = find_user(newplayer_name);
 511 
 512         // Is he in this game?
 513         if (player_connections_.find(newplayer) == player_connections_.end() || !is_member(newplayer)) {
 514                 send_server_message(newplayer_name.to_string() + " is not in this game", sock);
 515                 return;
 516         }
 517 
 518         if (newplayer == old_player) {
 519                 std::stringstream msg;
 520                 msg << "That's already " << newplayer_name << "'s side, silly.";
 521                 send_server_message(msg.str(), sock);
 522                 return;
 523         }
 524         sides_[side_num - 1] = socket_ptr();
 525         // If the old player lost his last side, make him an observer.
 526         if (std::find(sides_.begin(), sides_.end(), old_player) == sides_.end()
 527         && is_player(old_player)) {
 528                 observers_.push_back(old_player);
 529                 player_connections_.find(old_player)->info().set_status(player::OBSERVING);
 530                 players_.erase(std::remove(players_.begin(), players_.end(), old_player), players_.end());
 531                 // Tell others that the player becomes an observer.
 532                 send_and_record_server_message(old_player_name + " becomes an observer.");
 533                 // Update the client side observer list for everyone except old player.
 534                 simple_wml::document observer_join;
 535                 observer_join.root().add_child("observer").set_attr_dup("name", old_player_name.c_str());
 536                 send_data(observer_join, old_player);
 537         }
 538         change_controller(side_num - 1, newplayer, player_connections_.find(newplayer)->info().name(), false);
 539 
 540         // If we gave the new side to an observer add him to players_.
 541         if (is_observer(newplayer)) {
 542                 players_.push_back(newplayer);
 543                 player_connections_.find(newplayer)->info().set_status(player::PLAYING);
 544                 observers_.erase(std::remove(observers_.begin(), observers_.end(), newplayer), observers_.end());
 545                 // Send everyone but the new player the observer_quit message.
 546                 send_observerquit(newplayer);
 547         }
 548 }

 550 void game::change_controller(const size_t side_index,
 551                 const socket_ptr sock,
 552                 const std::string& player_name,
 553                 const bool player_left)
 554 {
 555         DBG_GAME << __func__ << "...\n";
 556 
 557         const std::string& side = lexical_cast_default<std::string, size_t>(side_index + 1);
 558         sides_[side_index] = sock;
 559 
 560         if (player_left && side_controllers_[side_index] == CONTROLLER::AI) {
 561                 // Automatic AI side transfer.
 562         } else {
 563                 if (started_) {
 564                         send_and_record_server_message(player_name + " takes control of side " + side + ".");
 565                 }
 566         }
 567 
 568         simple_wml::document response;
 569         simple_wml::node& change = response.root().add_child("change_controller");
 570 
 571         change.set_attr("side", side.c_str());
 572         change.set_attr("player", player_name.c_str());
 573

 575         change.set_attr("controller", side_controllers_[side_index].to_cstring());
 576         change.set_attr("is_local", "no");
 577 
 578         send_data(response, sock);
 579         if (started_) { //this is added instead of the if (started_) {...} below
 580                 //the purpose of these records is so that observers, replay viewers, get controller updates correctly
 581                 record_data(response.clone());
 582         }
 583 
 584         // Tell the new player that he controls this side now.
 585         // Just don't send it when the player left the game. (The host gets the
 586         // side_drop already.)
 587         if (!player_left) {
 588                 change.set_attr("is_local", "yes");
 589                 send_to_player(sock, response);
 590         }
 591 }

 845 bool game::is_legal_command(const simple_wml::node& command, const socket_ptr user)
 846 {
 847         const bool is_player = this->is_player(user);
 848         const bool is_host = user == owner_;
 849         const bool is_current = is_current_player(user);
 850 
 851         if(command.has_attr("from_side")) {
 852                 const size_t from_side_index = command["from_side"].to_int() - 1;
 853                 if(command["from_side"] == "server") {
 854                         //Someone pretends to be the server...
 855                         return false;
 856                 }
 857                 if(from_side_index >= sides_.size() || sides_[from_side_index] != user) {
 858                         return false;
 859                 }
 860         }
 861 
 862         if(is_current) return true;
 863         // Only single commands allowed.
 864         // NOTE: some non-dependent commands like move,attack.. might contain a [checkup] tag after their first data.
 865         // But those packages are only sended by the currently active player which we check above.
 866         if (!command.one_child()) return false;
 867         // Chatting is never an illegal command.
 868         if (command.child("speak")) return true;
 869         if (is_player && command.has_attr("dependent") && command.has_attr("from_side"))
 870                 //AKA it's generated by get_user_input for example [global_variable]
 871         {
 872                 return true;
 873         }
 874         if ((is_player || is_host)
 875         && (command.child("label")
 876                 || command.child("clear_labels")
 877                 || command.child("rename")
 878                 || command.child("countdown_update")
 879                 ))
 880         {
 881                 return true;
 882         }
 883         return false;
 884 }

 886 bool game::process_turn(simple_wml::document& data, const socket_ptr user) {
 887         //DBG_GAME << "processing commands: '" << cfg << "'\n";
 888         if (!started_) return false;
 889         simple_wml::node* const turn = data.root().child("turn");
 890         bool turn_ended = false;
 891         // Any private 'speak' commands must be repackaged separate
 892         // to other commands, and re-sent, since they should only go
 893         // to some clients.
 894         bool repackage = false;
 895         int index = 0;
 896         std::vector<int> marked;
 897         const simple_wml::node::child_list& commands = turn->children("command");
 898         simple_wml::node::child_list::const_iterator command;
 899         for (command = commands.begin(); command != commands.end(); ++command) {
 900                 DBG_GAME << "game " << id_ << " recieved ["  << (**command).first_child() << "] from player '" << username(user) << "'(" << user << ") during turn " << end_turn_ << "\n";
 901                 if (!is_legal_command(**command, user)) {
 902                         LOG_GAME << "ILLEGAL COMMAND in game: " << id_ << " ((("
 903                                 << simple_wml::node_to_string(**command) << ")))\n";
 904                         std::stringstream msg;
 905                         msg << "Removing illegal command '" << (**command).first_child().to_string()
 906                                 << "' from: " << username(user)
 907                                 << ". Current player is: "
 908                                 << username(current_player())
 909                                 << " (" << end_turn_ + 1 << "/" << nsides_ << ").";
 910                         LOG_GAME << msg.str() << " (socket: " << current_player()
 911                                 << ") (game id: " << id_ << ")\n";
 912                         send_and_record_server_message(msg.str());
 913 
 914                         marked.push_back(index - marked.size());
 915                 } else if ((**command).child("speak")) {
 916                         simple_wml::node& speak = *(**command).child("speak");
 917                         if (speak["to_sides"] != "" || is_muted_observer(user)) {
 918                                 DBG_GAME << "repackaging..." << std::endl;
 919                                 repackage = true;
 920                         }
 921 
 922                         const simple_wml::string_span& msg = speak["message"];
 923                         chat_message::truncate_message(msg, speak);
 924 
 925                         // Force the description to be correct,
 926                         // to prevent spoofing of messages.
 927                         speak.set_attr_dup("id", player_connections_.find(user)->info().name().c_str());
 928                         // Also check the side for players.
 929                         if (is_player(user)) {
 930                                 const size_t side_index = speak["side"].to_int() - 1;
 931                                 if (side_index >= sides_.size()
 932                                 || sides_[side_index] != user) {
 933                                         if (user == current_player()) {
 934                                                 speak.set_attr_dup("side", lexical_cast_default<std::string>(current_side() + 1).c_str());
 935                                         } else {
 936                                                 const side_vector::const_iterator s =
 937                                                                 std::find(sides_.begin(), sides_.end(), user);
 938                                                 speak.set_attr_dup("side", lexical_cast_default<std::string>(s - sides_.begin() + 1).c_str());
 939                                         }
 940                                 }
 941                         }
 942                 }
 943                 else if (is_current_player(user) && (**command).child("end_turn")) {
 944                         turn_ended = end_turn();
 945                 }
 946                 ++index;
 947         }
 948         for(std::vector<int>::const_iterator j = marked.begin(); j != marked.end(); ++j) {
 949                 turn->remove_child("command",*j);
 950         }
 951         if (turn->no_children()) {
 952                 return false;
 953         }
 954         if (!repackage) {
 955                 record_data(data.clone());
 956                 send_data(data, user, "game replay");
 957                 return turn_ended;
 958         }
 959         for (command = commands.begin(); command != commands.end(); ++command) {
 960                 simple_wml::node* const speak = (**command).child("speak");
 961                 if (speak == nullptr) {
 962                         simple_wml::document* mdata = new simple_wml::document;
 963                         simple_wml::node& mturn = mdata->root().add_child("turn");
 964                         (**command).copy_into(mturn.add_child("command"));
 965                         send_data(*mdata, user, "game replay");
 966                         record_data(mdata);
 967                         continue;
 968                 }
 969                 const simple_wml::string_span& to_sides = (*speak)["to_sides"];
 970                 // Anyone can send to the observer team.
 971                 if (is_muted_observer(user) && to_sides != game_config::observer_team_name.c_str()) {
 972                         send_server_message("You have been muted, others can't see your message!", user);
 973                         continue;
 974                 }
 975 
 976                 std::unique_ptr<simple_wml::document> message(new simple_wml::document);
 977                 simple_wml::node& message_turn = message->root().add_child("turn");
 978                 simple_wml::node& message_turn_command = message_turn.add_child("command");
 979                 speak->copy_into(message_turn_command.add_child("speak"));
 980                 if (to_sides == "") {
 981                         send_data(*message, user, "game message");
 982                         record_data(message.release());
 983                 } else if (to_sides == game_config::observer_team_name) {
 984                         send_to_players(*message, observers_, user);
 985                         record_data(message.release());
 986                 } else {
 987                         send_data_sides(*message, to_sides, user, "game message");
 988                 }
 989         }
 990         return turn_ended;
 991 }

1011 void game::handle_controller_choice(const simple_wml::node& req)
1012 {
1013         const size_t side_index = req["side"].to_int() - 1;
1014         CONTROLLER new_controller;
1015         CONTROLLER old_controller;
1016         if(!new_controller.parse(req["new_controller"])) {
1017                 send_and_record_server_message("Could not handle [request_choice] [change_controller] with invalid controller '" + req["new_controller"].to_string() + "'");
1018                 return;
1019         }
1020         if(!old_controller.parse(req["old_controller"])) {
1021                 send_and_record_server_message("Could not handle [request_choice] [change_controller] with invalid controller '" + req["old_controller"].to_string() + "'");
1022                 return;
1023         }
1024         if(old_controller != this->side_controllers_[side_index]) {
1025                 send_and_record_server_message("Found unexpected old_controller= '" + old_controller.to_string() + "' in [request_choice] [change_controller]");
1026         }
1027         if(side_index >= sides_.size()) {
1028                 send_and_record_server_message("Could not handle [request_choice] [change_controller] with invalid side '" + req["side"].to_string() + "'");
1029                 return;
1030         }
1031         const bool was_null = this->side_controllers_[side_index] == CONTROLLER::EMPTY;
1032         const bool becomes_null = new_controller == CONTROLLER::EMPTY;
1033         if(was_null) {
1034                 assert(sides_[side_index] == 0);
1035                 sides_[side_index] = current_player();
1036         }
1037         if(becomes_null) {
1038                 sides_[side_index] = socket_ptr();
1039         }
1040         side_controllers_[side_index] = new_controller;
1041 
1042         simple_wml::document* mdata = new simple_wml::document;
1043         simple_wml::node& turn = mdata->root().add_child("turn");
1044         simple_wml::node& command = turn.add_child("command");
1045         simple_wml::node& change_controller_wml = command.add_child("change_controller_wml");
1046         change_controller_wml.set_attr("controller", new_controller.to_cstring());
1047         change_controller_wml.set_attr("is_local", "yes");
1048         command.set_attr("from_side", "server");
1049         command.set_attr("dependent", "yes");
1050         if(sides_[side_index] != 0) {
1051                 //calling send_to_one to 0 connect causes the package to be sended to all clients.
1052                 send_to_player(sides_[side_index], *mdata);
1053         }
1054         change_controller_wml.set_attr("is_local", "no");
1055         send_data(*mdata, sides_[side_index], "game replay");
1056         record_data(mdata);
1057 }

1059 void game::handle_choice(const simple_wml::node& data, const socket_ptr user)
1060 {
1061         // note, that during end turn events, it's side=1 for the server but side= side_count() on the clients.
1062 
1063         // Otherwise we allow observers to cause OOS for the playing clients by sending
1064         // server choice requests based on incompatible local changes. To solve this we block
1065         // server choice requests from observers.
1066         if(!started_) {
1067                 return;
1068         }
1069         if (user != owner_ && !is_player(user)) {
1070                 return;
1071         }
1072         int request_id = lexical_cast_default<int>(data["request_id"], -10);
1073 
1074         if(request_id <= last_choice_request_id_) {
1075                 // We gave already an anwer to this request.
1076                 return;
1077         }
1078         DBG_GAME << "answering seed request " << request_id << " by player " << player_connections_.find(user)->info().name() << "(" << user << ")" << std::endl;
1079         last_choice_request_id_ = request_id;
1080 
1081         if(const simple_wml::node* rand = data.child("random_seed")) {
1082                 handle_random_choice(*rand);
1083         }
1084         else if(const simple_wml::node* ccw = data.child("change_controller_wml")) {
1085                 handle_controller_choice(*ccw);
1086         }
1087         else {
1088                 send_and_record_server_message("Found unknown server choice request: [" + data.first_child().to_string() + "]");
1089         }
1090

1242

1246         observers_.erase(std::remove(observers_.begin(), observers_.end(), player), observers_.end());
1247         const bool game_ended = players_.empty() || (host && !started_);
1248 
1249         socket_ptr user = player;
1250

1252                 << ((game_ended && !(observer && destruct))
1253                         ? (started_ ? "\tended" : "\taborted") : "\thas left")
1254                 << " game:\t\"" << name_ << "\" (" << id_ << ")"
1255                 << (game_ended && started_ && !(observer && destruct) ? " at turn: "
1256                         + lexical_cast_default<std::string,size_t>(current_turn())
1257                         + " with reason: '" + termination_reason() + "'" : "")
1258                 << (observer ? " as an observer" : "")
1259                 << (disconnect ? " and disconnected" : "")

1262                 send_server_message_to_all(player_connections_.find(user)->info().name() + " ended the game.", player);
1263         }
1264         if (game_ended || destruct) return game_ended;
1265 
1266         // Don't mark_available() since the player got already removed from the
1267         // games_and_users_list_.
1268         if (!disconnect) {
1269                 player_connections_.find(user)->info().mark_available();
1270         }
1271         if (observer) {
1272                 send_observerquit(user);

1274                 send_and_record_server_message(player_connections_.find(user)->info().name()
1275                                 + (disconnect ? " has disconnected." : " has left the game."), player);
1276         }
1277         // If the player was host choose a new one.
1278         if (host) {
1279                 owner_ = players_.front();
1280                 notify_new_host();
1281         }
1282 
1283         bool ai_transfer = false;

1292                 // Check whether the host is actually a player and make him one if not.

1295                         player_connections_.find(owner_)->info().set_status(player::PLAYING);
1296                         observers_.erase(std::remove(observers_.begin(), observers_.end(), owner_), observers_.end());
1297                         players_.push_back(owner_);
1298                         send_observerquit(owner_);
1299                 }
1300 
1301                 //send the host a notification of removal of this side
1302                 const std::string side_drop = lexical_cast_default<std::string, size_t>(side_index + 1);
1303                 simple_wml::document drop;
1304                 drop.root().set_attr("side_drop", side_drop.c_str());
1305                 drop.root().set_attr("controller", side_controllers_[side_index].to_cstring());
1306 
1307                 DBG_GAME << "*** sending side drop: \n" << drop.output() << std::endl;
1308 
1309                 send_to_player(owner_, drop);
1310         }
1311         if (ai_transfer) send_and_record_server_message("AI sides transferred to host.");
1312 
1313         DBG_GAME << debug_player_info();
1314 
1315         send_user_list(player);
1316         return false;
