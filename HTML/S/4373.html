   6 -- Evaluation process:
   7 --
   8 -- Find all enemy units that could be caused to level up by an attack
   9 --  - If only units that would cause them to level up can attack, CA score = 100,010.
  10 --    This means the attack will be done before the default AI attacks, so that AI
  11 --    units do not get used otherwise by the default AI.
  12 --  - If units that would not cause a leveling can also attack, CA score = 99,990,
  13 --    meaning we see whether the default AI attacks that unit with one of those first.
  14 -- We also check whether it is possible to move an own unit out of the way
  15 --
  16 -- Attack rating:
  17 -- 0. If the CTD (chance to die) of the AI unit is larger than the value of
  18 --    aggression for the side, do not do the attack
  19 -- 1. Otherwise, if the attack might result in a kill, do that preferentially:
  20 --     rating = CTD of defender - CTD of attacker
  21 -- 2. Otherwise, if the enemy is poisoned, do not attack (wait for it
  22 --    weaken and attack on a later turn)
  23 -- 3. Otherwise, calculate damage done to enemy (as if it were not leveling) and
  24 --    own unit, expressed in partial loss of unit value (gold) and minimize both.
  25 --    Damage to enemy is minimized because we want to level it with the weakest AI unit,
  26 --    so that we can follow up with stronger units. In addition, use of poison or
  27 --    slow attacks is strongly discouraged. See code for exact equations.

  31 function ca_attack_highxp:evaluation(cfg, data)
  32     -- Note: (most of) the code below is set up to maximize speed. Do not
  33     -- "simplify" this unless you understand exactly what that means
  34 
  35     local attacks_aspect = ai.aspects.attacks
  36     local max_unit_level = 0
  37     local units = {}
  38     for _,unit in ipairs(attacks_aspect.own) do
  39         if (unit.attacks_left > 0) and (#unit.attacks > 0) and (not unit.canrecruit) then
  40             table.insert(units, unit)
  41 
  42             local level = unit.level
  43             if (level > max_unit_level) then
  44                 max_unit_level = level
  45             end
  46         end
  47     end
  48     if (not units[1]) then return 0 end
  49 
  50     -- Mark enemies as potential targets if they are close enough to an AI unit
  51     -- that could trigger them leveling up; this is not a sufficient criterion,
  52     -- but it is much faster than path finding, so it is done for preselection.
  53     local target_infos = {}
  54     for i_t,enemy in ipairs(attacks_aspect.enemy) do
  55         if AH.is_attackable_enemy(enemy) then
  56             local XP_to_levelup = enemy.max_experience - enemy.experience
  57             if (max_unit_level >= XP_to_levelup) then
  58                 local potential_target = false
  59                 local ind_attackers, ind_other_units = {}, {}
  60                 for i_u,unit in ipairs(units) do
  61                     if (M.distance_between(enemy.x, enemy.y, unit.x, unit.y) <= unit.moves + 1) then
  62                         if (unit.level >= XP_to_levelup) then
  63                             potential_target = true
  64                             table.insert(ind_attackers, i_u)
  65                         else
  66                             table.insert(ind_other_units, i_u)
  67                         end
  68                     end
  69                 end
  70 
  71                 if potential_target then
  72                     local target_info = {
  73                         ind_target = i_t,
  74                         ind_attackers = ind_attackers,
  75                         ind_other_units = ind_other_units
  76                     }
  77                     table.insert(target_infos, target_info)
  78                 end
  79             end
  80         end
  81     end
  82     if (not target_infos[1]) then return 0 end
  83 
  84     -- The following location sets are used so that we at most need to call
  85     -- find_reach() and wesnoth.copy_unit() once per unit
  86     local reaches = LS.create()
  87     local attacker_copies = LS.create()
  88 
  89     local aggression = ai.aspects.aggression
  90     local avoid_map = LS.of_pairs(ai.aspects.avoid)
  91     local max_ca_score, max_rating, best_attack = 0, 0
  92     for _,target_info in ipairs(target_infos) do
  93         local target = attacks_aspect.enemy[target_info.ind_target]
  94         local can_force_level = {}
  95         local attack_hexes = LS.create()
  96         for xa,ya in H.adjacent_tiles(target.x, target.y) do
  97             if (not avoid_map:get(xa, ya)) then
  98                 local unit_in_way = wesnoth.get_unit(xa, ya)
  99 
 100                 if AH.is_visible_unit(wesnoth.current.side, unit_in_way) then
 101                     if (unit_in_way.side == wesnoth.current.side) then
 102                         local uiw_reach
 103                         if reaches:get(unit_in_way.x, unit_in_way.y) then
 104                             uiw_reach = reaches:get(unit_in_way.x, unit_in_way.y)
 105                         else
 106                             uiw_reach = wesnoth.find_reach(unit_in_way)
 107                             reaches:insert(unit_in_way.x, unit_in_way.y, uiw_reach)
 108                         end
 109 
 110                         -- Check whether the unit to move out of the way has an unoccupied hex to move to.
 111                         -- We do not deal with cases where a unit can move out of the way for a
 112                         -- unit that is moving out of the way of the initial unit (etc.).
 113                         local can_move = false
 114                         for _,uiw_loc in ipairs(uiw_reach) do
 115                             -- Unit in the way of the unit in the way
 116                             local uiw_uiw = wesnoth.get_unit(uiw_loc[1], uiw_loc[2])
 117                             if (not AH.is_visible_unit(wesnoth.current.side, uiw_uiw)) then
 118                                 can_move = true
 119                                 break
 120                             end
 121                         end
 122                         if (not can_move) then
 123                             -- Keep this case as the unit in the way might be a potential attacker
 124                             attack_hexes:insert(xa, ya, unit_in_way.id)
 125                         else
 126                             attack_hexes:insert(xa, ya, 'can_move_away')
 127                         end
 128                     end
 129                 else
 130                     attack_hexes:insert(xa, ya, 'empty')
 131                 end
 132             end
 133         end
 134 
 135         attack_hexes:iter(function(xa, ya, occupied)
 136             for _,i_a in ipairs(target_info.ind_attackers) do
 137                 local attacker = units[i_a]
 138                 if (occupied == 'empty') or (occupied == 'can_move_away') then
 139                     -- If the hex is not blocked, check all potential attackers
 140                     local reach
 141                     if reaches:get(attacker.x, attacker.y) then
 142                         reach = reaches:get(attacker.x, attacker.y)
 143                     else
 144                         reach = wesnoth.find_reach(attacker)
 145                         reaches:insert(attacker.x, attacker.y, reach)
 146                     end
 147 
 148                     for _,loc in ipairs(reach) do
 149                         if (loc[1] == xa) and (loc[2] == ya) then
 150                             local tmp = {
 151                                 ind_attacker = i_a,
 152                                 dst = { x = xa, y = ya },
 153                                 src = { x = attacker.x, y = attacker.y }
 154                             }
 155                             table.insert(can_force_level, tmp)
 156                             break
 157                         end
 158                     end
 159                 else
 160                     -- If hex is blocked by own units, check whether this unit
 161                     -- is one of the potential attackers
 162                     if (attacker.id == occupied) then
 163                         local tmp = {
 164                             ind_attacker = i_a,
 165                             dst = { x = xa, y = ya },
 166                             src = { x = attacker.x, y = attacker.y }
 167                         }
 168 
 169                         table.insert(can_force_level, tmp)
 170                     end
 171                 end
 172             end
 173         end)
 174 
 175         -- If a leveling attack is possible, check whether any of the other
 176         -- units (those with too low a level to force leveling up) can get there
 177         local ca_score = 100010
 178 
 179         attack_hexes:iter(function(xa, ya, occupied)
 180             if (ca_score == 100010) then  -- cannot break out of the iteration with goto
 181                 for _,i_u in ipairs(target_info.ind_other_units) do
 182                     local unit = units[i_u]
 183                     if (occupied == 'empty') or (occupied == 'can_move_away') then
 184                         -- If the hex is not blocked, check if unit can get there
 185                         local reach
 186                         if reaches:get(unit.x, unit.y) then
 187                             reach = reaches:get(unit.x, unit.y)
 188                         else
 189                             reach = wesnoth.find_reach(unit)
 190                             reaches:insert(unit.x, unit.y, reach)
 191                         end
 192 
 193                         for _,loc in ipairs(reach) do
 194                             if (loc[1] == xa) and (loc[2] == ya) then
 195                                 ca_score = 99990
 196                                 goto found_unit
 197                             end
 198                         end
 199                     else
 200                         -- If hex is blocked by own units, check whether this unit
 201                         -- is one of the potential attackers
 202                         if (unit.id == occupied) then
 203                             ca_score = 99990
 204                             goto found_unit
 205                         end
 206                     end
 207                 end
 208                 -- It is sufficient to find one unit that can get to any attack hex
 209                 ::found_unit::
 210             end
 211         end)
 212 
 213         if (ca_score >= max_ca_score) then
 214             for _,attack_info in ipairs(can_force_level) do
 215                 local attacker = units[attack_info.ind_attacker]
 216                 local attacker_copy
 217                 if attacker_copies:get(attacker.x, attacker.y) then
 218                     attacker_copy = attacker_copies:get(attacker.x, attacker.y)
 219                 else
 220                     attacker_copy = wesnoth.copy_unit(attacker)
 221                     attacker_copies:insert(attacker.x, attacker.y, attacker_copy)
 222                 end
 223 
 224                 attacker_copy.x = attack_info.dst.x
 225                 attacker_copy.y = attack_info.dst.y
 226 
 227                 -- Choose the attacker that would do the *least* damage.
 228                 -- We want the damage distribution here as if the target were not to level up
 229                 -- the chance to die is the same in either case
 230                 local old_experience = target.experience
 231                 target.experience = 0
 232                 local att_stats, def_stats, att_weapon = wesnoth.simulate_combat(attacker_copy, target)
 233                 target.experience = old_experience
 234 
 235                 local rating = -1000
 236                 if (att_stats.hp_chance[0] <= aggression) then
 237                     if (def_stats.hp_chance[0] > 0) then
 238                         rating = 5000 + def_stats.hp_chance[0] - att_stats.hp_chance[0]
 239                     elseif target.status.poisoned then
 240                         rating = -1002
 241                     else
 242                         rating = 1000
 243 
 244                         local enemy_value_loss = (target.hitpoints - def_stats.average_hp) / target.max_hitpoints
 245                         enemy_value_loss = enemy_value_loss * wesnoth.unit_types[target.type].cost
 246 
 247                         -- We want the _least_ damage to the enemy, so the minus sign is no typo!
 248                         rating = rating - enemy_value_loss
 249 
 250                         local own_value_loss = (attacker_copy.hitpoints - att_stats.average_hp) / attacker_copy.max_hitpoints
 251                         own_value_loss = own_value_loss + att_stats.hp_chance[0]
 252                         own_value_loss = own_value_loss * wesnoth.unit_types[attacker_copy.type].cost
 253 
 254                         rating = rating - own_value_loss
 255 
 256                         -- Strongly discourage poison or slow attacks
 257                         if att_weapon.poisons or att_weapon.slows then
 258                             rating = rating - 100
 259                         end
 260 
 261                         -- Minor penalty if the attack hex is occupied
 262                         if (attack_hexes:get(attack_info.dst.x, attack_info.dst.y) == 'can_move_away') then
 263                             rating = rating - 0.001
 264                         end
 265                     end
 266                 end
 267 
 268                 if (rating > max_rating)
 269                     or ((rating > 0) and (ca_score > max_ca_score))
 270                 then
 271                     max_rating = rating
 272                     max_ca_score = ca_score
 273                     best_attack = attack_info
 274                     best_attack.target = { x = target.x, y = target.y }
 275                     best_attack.ca_score = ca_score
 276                     -- Also need to save weapon number because attack simulation above
 277                     -- is for a different situation than the actual units on the map.
 278                     -- +1 because of difference between Lua and C++ indices
 279                     best_attack.attack_num = att_weapon.attack_num + 1
 280                 end
 281             end
 282         end
 283 
 284     end
 285 
 286     if best_attack then
 287         data.XP_attack = best_attack
 288     end
 289 
 290     return max_ca_score
 291 end

 293 function ca_attack_highxp:execution(cfg, data)
 294     AH.robust_move_and_attack(ai, data.XP_attack.src, data.XP_attack.dst, data.XP_attack.target, { weapon = data.XP_attack.attack_num })
 295     data.XP_attack = nil
 296 end
