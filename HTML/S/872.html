  36 /**
  37  * Returns base + increment, but will not increase base above max_sum, nor
  38  * decrease it below min_sum.
  39  * (If base is already beyond the applicable limit, base will be returned.)
  40  */
  41 inline int bounded_add(int base, int increment, int max_sum, int min_sum=0) {
  42         if ( increment >= 0 )
  43                 return std::min(base+increment, std::max(base, max_sum));
  44         else
  45                 return std::max(base+increment, std::min(base, min_sum));
  46 }

  53 /**
  54  *  round (base_damage * bonus / divisor) to the closest integer,
  55  *  but up or down towards base_damage
  56  */
  57 inline int round_damage(int base_damage, int bonus, int divisor) {
  58         if (base_damage==0) return 0;
  59         const int rounding = divisor / 2 - (bonus < divisor || divisor==1 ? 0 : 1);
  60         return std::max<int>(1, (base_damage * bonus + rounding) / divisor);
  61 }

 122 /**
 123  * Returns the quantity of `1` bits in `n` — i.e., `n`’s population count.
 124  *
 125  * Algorithm adapted from:
 126  * <https://graphics.stanford.edu/~seander/bithacks.html#CountBitsSetKernighan>
 127  *
 128  * This algorithm was chosen for relative simplicity, not for speed.
 129  *
 130  * @tparam N The type of `n`. This should be a fundamental integer type no
 131  * greater than `UINT_MAX` bits in width; if it is not, the return value is
 132  * undefined.
 133  *
 134  * @param n An integer upon which to operate.
 135  *
 136  * @returns the quantity of `1` bits in `n`, if `N` is a fundamental integer
 137  * type.
 138  */
 139 template<typename N>
 140 inline unsigned int count_ones(N n) {
 141         unsigned int r = 0;
 142         while (n) {
 143                 n &= n-1;
 144                 ++r;
 145         }
 146         return r;
 147 }

 228 /**
 229  * Returns the quantity of leading `0` bits in `n` — i.e., the quantity of
 230  * bits in `n`, minus the 1-based bit index of the most significant `1` bit in
 231  * `n`, or minus 0 if `n` is 0.
 232  *
 233  * @tparam N The type of `n`. This should be a fundamental integer type that
 234  *  (a) is not wider than `unsigned long long int` (the GCC
 235  *   count-leading-zeros built-in functions are defined for `unsigned int`,
 236  *   `unsigned long int`, and `unsigned long long int`), and
 237  *  (b) is no greater than `INT_MAX` bits in width (the GCC built-in functions
 238  *   return instances of type `int`);
 239  * if `N` does not satisfy these constraints, the return value is undefined.
 240  *
 241  * @param n An integer upon which to operate.
 242  *
 243  * @returns the quantity of leading `0` bits in `n`, if `N` satisfies the
 244  * above constraints.
 245  *
 246  * @see count_leading_ones()
 247  */
 248 template<typename N>
 249 inline unsigned int count_leading_zeros(N n) {
 250 #if defined(__GNUC__) || defined(__clang__)
 251         // GCC’s `__builtin_clz` returns an undefined value when called with 0
 252         // as argument.
 253         // [<http://gcc.gnu.org/onlinedocs/gcc/Other-Builtins.html>]
 254         if (n == 0) {
 255                 // Return the quantity of zero bits in `n` rather than
 256                 // returning that undefined value.
 257                 return static_cast<unsigned int>(bit_width(n));
 258         }
 259 #endif
 260         // Dispatch, on the static type of `n`, to one of the
 261         // `count_leading_zeros_impl` functions.
 262         return count_leading_zeros_impl(n, bit_width(n));
 263         // The second argument to `count_leading_zeros_impl` specifies the
 264         // width, in bits, of `n`.
 265         //
 266         // This is necessary because `n` may be widened (or, alas, shrunk),
 267         // and thus the information of `n`’s true width may be lost.
 268         //
 269         // At least, this *was* necessary before there were so many overloads
 270         // of `count_leading_zeros_impl`, but I’ve kept it anyway as an extra
 271         // precautionary measure, that will (I hope) be optimized out.
 272         //
 273         // To be clear, `n` would only be shrunk in cases noted above as
 274         // having an undefined result.
 275 }

 277 /**
 278  * Returns the quantity of leading `1` bits in `n` — i.e., the quantity of
 279  * bits in `n`, minus the 1-based bit index of the most significant `0` bit in
 280  * `n`, or minus 0 if `n` contains no `0` bits.
 281  *
 282  * @tparam N The type of `n`. This should be a fundamental integer type that
 283  *  (a) is not wider than `unsigned long long int`, and
 284  *  (b) is no greater than `INT_MAX` bits in width;
 285  * if `N` does not satisfy these constraints, the return value is undefined.
 286  *
 287  * @param n An integer upon which to operate.
 288  *
 289  * @returns the quantity of leading `1` bits in `n`, if `N` satisfies the
 290  * above constraints.
 291  *
 292  * @see count_leading_zeros()
 293  */
 294 template<typename N>
 295 inline unsigned int count_leading_ones(N n) {
 296         // Explicitly specify the type for which to instantiate
 297         // `count_leading_zeros` in case `~n` is of a different type.
 298         return count_leading_zeros<N>(~n);
 299 }
