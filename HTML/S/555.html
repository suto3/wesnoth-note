  60         /** The way to handle the showing or hiding of the scrollbar. */
  61         enum scrollbar_mode {
  62                 /**
  63                  * The scrollbar is always shown, whether needed or not.
  64                  */
  65                 ALWAYS_VISIBLE,
  66 
  67                 /**
  68                  * The scrollbar is never shown even notwhen needed. There's also no space
  69                  * reserved for the scrollbar.
  70                  */
  71                 ALWAYS_INVISIBLE,
  72 
  73                 /**
  74                  * The scrollbar is shown when the number of items is larger as the visible items.
  75                  * The space for the scrollbar is always reserved, just in case it's needed after
  76                  * the initial sizing (due to adding items).
  77                  */
  78                 AUTO_VISIBLE,
  79 
  80                 /**
  81                  * Like AUTO_VISIBLE, but when not needed upon the initial layout phase, the bars
  82                  * are not shown and no space is reserved for them. (The algorithm hides them by
  83                  * default.
  84                  */
  85                 AUTO_VISIBLE_FIRST_RUN,
  86         };

 271         /**
 272          * Notification if the content of a child needs a resize.
 273          *
 274          * When a resize is required the container first can try to handle it
 275          * itself. If it can't honor the request the function will call @ref
 276          * window::invalidate_layout().
 277          *
 278          * @note Calling this function on a widget with size == (0, 0) results
 279          * false but doesn't call invalidate_layout, the engine expects to be in
 280          * build up phase with the layout already invalidated.
 281          *
 282          * @param force_sizing        If the contents fit do we want to force a
 283          *                            resize? This is needed in the MP lobby since
 284          *                            items might not be properly placed yet.
 285          *                            (The listboxes with the player info need it.)
 286          *
 287          * @returns                   True if the resize is handled, false
 288          *                            otherwise.
 289          */
 290         bool content_resize_request(const bool force_sizing = false);
 291 
 292         /**
 293          * Request from the content to modify the size of the container.
 294          *
 295          * When the wanted resize fails the function will call @ref
 296          * window::invalidate_layout().
 297          *
 298          *
 299          * @note Calling this function on a widget with size == (0, 0) results
 300          * false but doesn't call invalidate_layout, the engine expects to be in
 301          * build up phase with the layout already invalidated.
 302          *
 303          * @note If @ref window::get_need_layout() is true the function returns
 304          * false and doesn't try to fit the contents since a layout phase will be
 305          * triggered anyway.
 306          *
 307          * @note This function might replace the @ref content_resize_request above.
 308          *
 309          * @param width_modification  The wanted modification to the width:
 310          *                            * negative values reduce width.
 311          *                            * zero leave width as is.
 312          *                            * positive values increase width.
 313          * @param height_modification The wanted modification to the height:
 314          *                            * negative values reduce height.
 315          *                            * zero leave height as is.
 316          *                            * positive values increase height.
 317          * @param width_modification_pos
 318          *                            The position where the additional content was
 319          *                            inserted/removed, defaults to -1 whcih means
 320          *                            'at end'
 321          * @param height_modification_po
 322          *                            The position where the additional content was
 323          *                            inserted/removed, defaults to -1 whcih means
 324          *                            'at end'
 325          *
 326          * @returns                   True is wanted modification is accepted false
 327          *                            otherwise.
 328          */
 329         bool content_resize_request(const int width_modification,
 330                                                                 const int height_modification,
 331                                                                 const int width_modification_pos = -1,
 332                                                                 const int height_modification_pos = -1);
