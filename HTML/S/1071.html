  35         /// The game event manager loads the scenario configuration object,
  36         /// and ensures that events are handled according to the
  37         /// scenario configuration for its lifetime.
  38         ///
  39         /// Thus, a manager object should be created when a scenario is played,
  40         /// and destroyed at the end of the scenario.
  41         /// If a second manager object is created before destroying the previous
  42         /// one, the game will crash with an assertion failure.
  43         class manager {
  44         public:
  45                 /// This class is similar to an input iterator through event handlers,
  46                 /// except each instance knows its own end (determined when constructed).
  47                 /// Subsequent dereferences are not guaranteed to return the same element,
  48                 /// so it is important to assign a dereference to a variable if you want
  49                 /// to use it more than once. On the other hand, a dereference will not
  50                 /// return a null pointer until the end of the iteration is reached (and
  51                 /// this is how to detect the end of the iteration).
  52                 ///
  53                 /// For simplicity, this class is neither assignable nor equality
  54                 /// comparable nor default constructable, and there is no postincrement.
  55                 /// Typedefs are also skipped.
  56                 class iteration
  57                 {
  58                 public:
  59                         /// Event-specific constructor.
  60                         explicit iteration(const std::string & event_name, manager &);
  61 
  62                         // Increment:
  63                         iteration & operator++();
  64                         // Dereference:
  65                         handler_ptr operator*();
  66 
  67                 private: // functions
  68                         /// Gets the index from a pointer, capped at end_.
  69                         handler_vec::size_type ptr_index(const handler_ptr & ptr) const
  70                         { return !bool(ptr) ? end_ : std::min(ptr->index(), end_); }
  71 
  72                 private: // data
  73                         /// The fixed-name event handlers for this iteration.
  74                         const handler_list & main_list_;
  75                         /// The varying-name event handlers for this iteration.
  76                         const handler_list & var_list_;
  77                         /// The event name for this iteration.
  78                         const std::string event_name_;
  79                         /// The end of this iteration. We intentionally exclude handlers
  80                         /// added after *this is constructed.
  81                         const handler_vec::size_type end_;
  82 
  83                         /// Set to true upon dereferencing.
  84                         bool current_is_known_;
  85                         /// true if the most recent dereference was taken from main_list_.
  86                         bool main_is_current_;
  87                         /// The current (or next) element from main_list_.
  88                         handler_list::iterator main_it_;
  89                         /// The current (or next) element from var_list_.
  90                         handler_list::iterator var_it_;
  91 
  92                         game_data * gamedata_;
  93                 };
