  69 recall::recall(config const& cfg, bool hidden)
  70         : action(cfg,hidden)
  71         , temp_unit_()
  72         , recall_hex_(cfg.child("recall_hex_")["x"],cfg.child("recall_hex_")["y"], wml_loc())
  73         , fake_unit_()
  74 {
  75         // Construct and validate temp_unit_
  76         size_t underlying_id = cfg["temp_unit_"];
  77         for(const unit_const_ptr & recall_unit : resources::gameboard->teams().at(team_index()).recall_list())
  78         {
  79                 if(recall_unit->underlying_id()==underlying_id)
  80                 {
  81                         temp_unit_.reset(new class unit(*recall_unit)); //TODO: is it necessary to make a copy?
  82                         break;
  83                 }
  84         }
  85         if(!temp_unit_.get()) {
  86                 throw action::ctor_err("recall: Invalid underlying_id");
  87         }
  88 
  89         fake_unit_.reset(unit_ptr(new class unit(*temp_unit_))); //makes copy of temp_unit_
  90 
  91         this->init();
  92 }

 115 void recall::execute(bool& success, bool& complete)
 116 {
 117         team & current_team = resources::gameboard->teams().at(team_index());
 118 
 119         assert(valid());
 120         assert(temp_unit_.get());
 121         temporary_unit_hider const raii(*fake_unit_);
 122         //Give back the spent gold so we don't get "not enough gold" message
 123         int cost = current_team.recall_cost();
 124         if (temp_unit_->recall_cost() > -1) {
 125                 cost=temp_unit_->recall_cost();
 126         }
 127         current_team.get_side_actions()->change_gold_spent_by(-cost);
 128         bool const result = synced_context::run_and_throw("recall",
 129                 replay_helper::get_recall(temp_unit_->id(), recall_hex_, map_location::null_location()),
 130                 true,
 131                 true,
 132                 synced_context::ignore_error_function);
 133 
 134         if (!result) {
 135                 current_team.get_side_actions()->change_gold_spent_by(cost);
 136         }
 137         success = complete = result;
 138 }

 140 void recall::apply_temp_modifier(unit_map& unit_map)
 141 {
 142         assert(valid());
 143         temp_unit_->set_location(recall_hex_);
 144 
 145         DBG_WB << "Inserting future recall " << temp_unit_->name() << " [" << temp_unit_->id()
 146                         << "] at position " << temp_unit_->get_location() << ".\n";
 147 
 148         //temporarily remove unit from recall list
 149         unit_ptr it = resources::gameboard->teams().at(team_index()).recall_list().extract_if_matches_id(temp_unit_->id());
 150         assert(it);
 151 
 152         //Add cost to money spent on recruits.
 153         int cost = resources::gameboard->teams().at(team_index()).recall_cost();
 154         if (it->recall_cost() > -1) {
 155                 cost = it->recall_cost();
 156         }
 157 
 158         // Temporarily insert unit into unit_map
 159         //unit map takes ownership of temp_unit
 160         unit_map.insert(temp_unit_);
 161 
 162         resources::gameboard->teams().at(team_index()).get_side_actions()->change_gold_spent_by(cost);
 163         // Update gold in top bar
 164         resources::screen->invalidate_game_status();
 165 }

 167 void recall::remove_temp_modifier(unit_map& unit_map)
 168 {
 169         temp_unit_ = unit_map.extract(recall_hex_);
 170         assert(temp_unit_.get());
 171 
 172         //Put unit back into recall list
 173         resources::gameboard->teams().at(team_index()).recall_list().add(temp_unit_);
 174 }

 176 void recall::draw_hex(map_location const& hex)
 177 {
 178         if (hex == recall_hex_)
 179         {
 180                 const double x_offset = 0.5;
 181                 const double y_offset = 0.7;
 182                 //position 0,0 in the hex is the upper left corner
 183                 std::stringstream number_text;
 184                 unit &it = *get_unit();
 185                 int cost = statistics::un_recall_unit_cost(it);
 186                 if (cost < 0) {
 187                         number_text << font::unicode_minus << resources::gameboard->teams().at(team_index()).recall_cost();
 188                 }
 189                 else {
 190                         number_text << font::unicode_minus << cost;
 191                 }
 192                 size_t font_size = 16;
 193                 color_t color {255, 0, 0}; //red
 194                 resources::screen->draw_text_in_hex(hex, display::LAYER_ACTIONS_NUMBERING,
 195                                                 number_text.str(), font_size, color, x_offset, y_offset);
 196         }
 197 }

 204 action::error recall::check_validity() const
 205 {
 206         //Check that destination hex is still free
 207         if(resources::gameboard->units().find(recall_hex_) != resources::gameboard->units().end()) {
 208                 return LOCATION_OCCUPIED;
 209         }
 210         //Check that unit to recall is still in side's recall list
 211         if( !resources::gameboard->teams()[team_index()].recall_list().find_if_matches_id(temp_unit_->id()) ) {
 212                 return UNIT_UNAVAILABLE;
 213         }
 214         //Check that there is still enough gold to recall this unit
 215         if(resources::gameboard->teams()[team_index()].recall_cost() > resources::gameboard->teams()[team_index()].gold()) {
 216                 return NOT_ENOUGH_GOLD;
 217         }
 218         //Check that there is a leader available to recall this unit
 219         if(!find_recruiter(team_index(),get_recall_hex())) {
 220                 return NO_LEADER;
 221         }
 222 
 223         return OK;
 224 }
