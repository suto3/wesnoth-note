 275 local function resolve_variable_context(ctx, err_hint)
 276         if ctx == nil then
 277                 return {get = wesnoth.get_variable, set = wesnoth.set_variable}
 278         elseif type(ctx) == 'number' and ctx > 0 and ctx <= #wesnoth.sides then
 279                 return resolve_variable_context(wesnoth.sides[ctx])
 280         elseif type(ctx) == 'string' then
 281                 -- TODO: Treat it as a namespace for a global (persistent) variable
 282                 -- (Need Lua API for accessing them first, though.)
 283         elseif getmetatable(ctx) == "unit" then
 284                 return {
 285                         get = function(path) return ctx.variables[path] end,
 286                         set = function(path, val) ctx.variables[path] = val end,
 287                 }
 288         elseif getmetatable(ctx) == "side" then
 289                 return {
 290                         get = function(path) return wesnoth.get_side_variable(ctx.side, path) end,
 291                         set = function(path, val) wesnoth.set_side_variable(ctx.side, path, val) end,
 292                 }
 293         elseif getmetatable(ctx) == "unit variables" or getmetatable(ctx) == "side variables" then
 294                 return {
 295                         get = function(path) return ctx[path] end,
 296                         set = function(path, val) ctx[path] = val end,
 297                 }
 298         end
 299         error(string.format("Invalid context for %s: expected nil, side, or unit", err_hint), 3)
 300 end

 334 --! Displays a WML message box with attributes from table @attr and options
 335 --! from table @options.
 336 --! @return the index of the selected option.
 337 --! @code
 338 --! local result = helper.get_user_choice({ speaker = "narrator" },
 339 --!     { "Choice 1", "Choice 2" })
 340 --! @endcode
 341 function helper.get_user_choice(attr, options)
 342         local result = 0
 343         function wesnoth.__user_choice_helper(i)
 344                 result = i
 345         end
 346         local msg = {}
 347         for k,v in pairs(attr) do
 348                 msg[k] = attr[k]
 349         end
 350         for k,v in ipairs(options) do
 351                 table.insert(msg, { "option", { message = v,
 352                         { "command", { { "lua", {

 354                         }}}}}})
 355         end
 356         wml_actions.message(msg)
 357         wesnoth.__user_choice_helper = nil

 372 --! Returns an iterator over adjacent locations that can be used in a for-in loop.
 373 -- Not deprecated because, unlike wesnoth.map.get_adjacent_tiles,
 374 -- this verifies that the locations are on the map.
 375 function helper.adjacent_tiles(x, y, with_borders)
 376         local x1,y1,x2,y2,b = 1,1,wesnoth.get_map_size()
 377         if with_borders then
 378                 x1 = x1 - b
 379                 y1 = y1 - b
 380                 x2 = x2 + b
 381                 y2 = y2 + b
 382         end
 383         local adj = {wesnoth.map.get_adjacent_tiles(x, y)}
 384         local i = 0
 385         return function()
 386                 while i < #adj do
 387                         i = i + 1
 388                         local u, v = adj[i][1], adj[i][2]
 389                         if u >= x1 and u <= x2 and v >= y1 and v <= y2 then
 390                                 return u, v
 391                         end
 392                 end
 393                 return nil
 394         end
 395 end

 429 function helper.rand (possible_values, random_func)
 430         random_func = random_func or wesnoth.random
 431         assert(type(possible_values) == "table" or type(possible_values) == "string", string.format("helper.rand expects a string or table as parameter, got %s instead", type(possible_values)))
 432 
 433         local items = {}
 434         local num_choices = 0
 435 
 436         if type(possible_values) == "string" then
 437                 -- split on commas
 438                 for word in possible_values:gmatch("[^,]+") do
 439                         -- does the word contain two dots? If yes, that's a range
 440                         local dots_start, dots_end = word:find("%.%.")
 441                         if dots_start then
 442                                 -- split on the dots if so and cast as numbers
 443                                 local low = tonumber(word:sub(1, dots_start-1))
 444                                 local high = tonumber(word:sub(dots_end+1))
 445                                 -- perhaps someone passed a string as part of the range, intercept the issue
 446                                 if not (low and high) then
 447                                         wesnoth.message("Malformed range: " .. possible_values)
 448                                         table.insert(items, word)
 449                                         num_choices = num_choices + 1
 450                                 else
 451                                         if low > high then
 452                                                 -- low is greater than high, swap them
 453                                                 low, high = high, low
 454                                         end
 455 
 456                                         -- if both ends represent the same number, then just use that number
 457                                         if low == high then
 458                                                 table.insert(items, low)
 459                                                 num_choices = num_choices + 1
 460                                         else
 461                                                 -- insert a table representing the range
 462                                                 table.insert(items, {low, high})
 463                                                 -- how many items does the range contain? Increase difference by 1 because we include both ends
 464                                                 num_choices = num_choices + (high - low) + 1
 465                                         end
 466                                 end
 467                         else
 468                                 -- handle as a string
 469                                 table.insert(items, word)
 470                                 num_choices = num_choices + 1
 471                         end
 472                 end
 473         else
 474                 num_choices = #possible_values
 475                 items = possible_values
 476                 -- We need to parse ranges separately anyway
 477                 for i, val in ipairs(possible_values) do
 478                         if type(val) == "table" then
 479                                 assert(#val == 2 and type(val[1]) == "number" and type(val[2]) == "number", "Malformed range for helper.rand")
 480                                 if val[1] > val[2] then
 481                                         val = {val[2], val[1]}
 482                                 end
 483                                 num_choices = num_choices + (val[2] - val[1])
 484                         end
 485                 end
 486         end
 487 
 488         local idx = random_func(1, num_choices)
 489 
 490         for i, item in ipairs(items) do
 491                 if type(item) == "table" then -- that's a range
 492                         local elems = item[2] - item[1] + 1 -- amount of elements in the range, both ends included
 493                         if elems >= idx then
 494                                 return item[1] + elems - idx
 495                         else
 496                                 idx = idx - elems
 497                         end
 498                 else -- that's a single element
 499                         idx = idx - 1
 500                         if idx == 0 then
 501                                 return item
 502                         end
 503                 end
 504         end
 505 
 506         return nil
 507 end

 532 function helper.shuffle( t, random_func)
 533         random_func = random_func or wesnoth.random
 534         -- since tables are passed by reference, this is an in-place shuffle
 535         -- it uses the Fisher-Yates algorithm, also known as Knuth shuffle
 536         assert( type( t ) == "table", string.format( "helper.shuffle expects a table as parameter, got %s instead", type( t ) ) )
 537         local length = #t
 538         for index = length, 2, -1 do
 539                 local random = random_func( 1, index )
 540                 t[index], t[random] = t[random], t[index]
 541         end
 542 end
