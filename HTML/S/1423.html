 110     /**
 111      * Parses the config. The resulted config will be available in
 112      * {@link #getParsedConfig()}
 113      */
 114     public void parse( )
 115     {
 116         if( root_ == null ) {
 117             return;
 118         }
 119 
 120         WMLTag currentTag = null;
 121         String currentTagName = "";
 122         String textdomain = "";
 123 
 124         // clear previous parsed info
 125         config_.getWMLTags( ).clear( );
 126         config_.getEvents( ).clear( );
 127         defines_.clear( );
 128 
 129 
 130         TreeIterator< EObject > itor = root_.eAllContents( );
 131         // nothing to parse!
 132         if( ! itor.hasNext( ) ) {
 133             return;
 134         }
 135 
 136         while( itor.hasNext( ) ) {
 137             EObject object = itor.next( );
 138 
 139             if( object instanceof WMLTag ) {
 140                 currentTag = ( WMLTag ) object;
 141                 currentTagName = currentTag.getName( );
 142 
 143                 if( currentTagName != null ) {
 144                     if( currentTagName.equals( "scenario" ) ) {
 145                         config_.IsScenario = true;
 146                     }
 147                     else if( currentTagName.equals( "campaign" ) ) {
 148                         config_.IsCampaign = true;
 149                     }
 150                     else if( currentTagName.equals( "preproc_define" ) ) {
 151                         addNewDefine( currentTag );
 152                     }
 153                 }
 154             }
 155             else if( object instanceof WMLKey ) {
 156                 if( currentTag != null && currentTagName != null ) {
 157                     WMLKey key = ( WMLKey ) object;
 158                     String keyName = key.getName( );
 159 
 160                     if( keyName.equals( "id" ) ) {
 161                         if( currentTagName.equals( "scenario" ) ) {
 162                             config_.ScenarioId = WMLUtils.getKeyValue( key
 163                                 .getValues( ) );
 164                         }
 165                         else if( currentTagName.equals( "campaign" ) ) {
 166                             config_.CampaignId = WMLUtils.getKeyValue( key
 167                                 .getValues( ) );
 168                         }
 169                     }
 170                     else if( keyName.equals( "name" ) ) {
 171                         if( currentTagName.equals( "set_variable" )
 172                             || currentTagName.equals( "set_variables" ) ) {
 173                             handleSetVariable( object );
 174                         }
 175                         else if( currentTagName.equals( "clear_variable" )
 176                             || currentTagName.equals( "clear_variables" ) ) {
 177                             handleUnsetVariable( object );
 178                         }
 179                         else if( currentTagName.equals( "event" ) ) {
 180                             String eventName = key.getValue( );
 181 
 182                             if( eventName.charAt( 0 ) == '"' ) {
 183                                 eventName = eventName.substring( 1 );
 184                             }
 185 
 186                             if( eventName.charAt( eventName.length( ) - 1 ) == '"' ) {
 187                                 eventName = eventName.substring( 0,
 188                                     eventName.length( ) - 1 );
 189                             }
 190 
 191                             config_.getEvents( ).add( eventName );
 192                         }
 193                     }
 194                 }
 195             }
 196             else if( object instanceof WMLMacroCall ) {
 197                 WMLMacroCall macroCall = ( WMLMacroCall ) object;
 198                 String macroCallName = macroCall.getName( );
 199                 if( macroCallName.equals( "VARIABLE" ) ) {
 200                     handleSetVariable( object );
 201                 }
 202                 else if( macroCallName.equals( "CLEAR_VARIABLE" ) ) {
 203                     handleUnsetVariable( object );
 204                 }
 205             }
 206             else if( object instanceof WMLLuaCode ) {
 207                 SimpleLuaParser luaParser = new SimpleLuaParser(
 208                     currentFileLocation_,
 209                     NodeModelUtils.getNode( object ).getStartLine( ),
 210                     ( ( WMLLuaCode ) object ).getValue( ) );
 211                 luaParser.parse( );
 212 
 213                 config_.getWMLTags( ).putAll( luaParser.getTags( ) );
 214             }
 215             else if( object instanceof WMLMacroDefine ) {
 216                 addNewDefine( ( WMLMacroDefine ) object, textdomain );
 217             }
 218             else if( object instanceof WMLTextdomain ) {
 219                 textdomain = ( ( WMLTextdomain ) object ).getName( );
 220                 if( textdomain == null ) {
 221                     textdomain = "";
 222                 }
 223             }
 224         }
 225     }

 227     /**
 228      * Creates a new define from a macro define
 229      * 
 230      * @param object
 231      */
 232     private void addNewDefine( WMLMacroDefine object, String textdomain )
 233     {
 234         ICompositeNode node = NodeModelUtils.getNode( object );
 235         if( node == null ) {
 236             return;
 237         }
 238 
 239         try {
 240             String defineHeader = object.getName( ).replaceAll( "\\r|\\n|\\t", "" );
 241             int firstSpaceIndex = defineHeader.indexOf( ' ' ) + 1;
 242             int defineNameEndIndex = defineHeader.indexOf( ' ', firstSpaceIndex + 1 );
 243 
 244             String name = defineHeader.substring( firstSpaceIndex, defineNameEndIndex );
 245 
 246             int linenum = node.getTotalStartLine( );
 247             String location = currentFileLocation_;
 248 
 249             String[] splittedArgs = defineHeader.substring( defineNameEndIndex + 1 ).split( " " );
 250             List< String > args = new ArrayList< String >( );
 251             for( String arg: splittedArgs ) {
 252                 args.add( arg );
 253             }
 254 
 255             StringBuffer value = new StringBuffer( );
 256             for( WMLValuedExpression expression: object.getExpressions( ) ) {
 257                 ICompositeNode expressionNode = NodeModelUtils.getNode( expression );
 258                 if( expressionNode != null ) {
 259                     value.append( expressionNode.getText( ) );
 260                 }
 261             }
 262 
 263             Define define = new Define( name, value.toString( ), textdomain, linenum, location, args );
 264             defines_.put( name, define );
 265         } catch( Exception e ) {
 266             // some formatting exceptions. We don't care 'bout em.
 267         }
 268     }

 270     /**
 271      * Creates a new define from the specified definition tag
 272      * and adds it to the list of defines
 273      * 
 274      * @param currentTag
 275      *        The tag that contains the define definition
 276      */
 277     private void addNewDefine( WMLTag currentTag )
 278     {
 279         String defineName = "";
 280         String location = "";
 281         int linenum = 0;
 282         String textdomain = "";
 283         String value = "";
 284         List< String > args = new ArrayList< String >( );
 285 
 286         // parse general define information
 287         for( WMLKey key: currentTag.getWMLKeys( ) ) {
 288             String keyName = key.getName( );
 289             if( keyName.equals( "name" ) ) {
 290                 defineName = getUnquotedString( key.getValue( ) );
 291             }
 292             else if( keyName.equals( "value" ) ) {
 293                 value = getUnquotedString( key.getValue( ) );
 294             }
 295             else if( keyName.equals( "textdomain" ) ) {
 296                 textdomain = getUnquotedString( key.getValue( ) );
 297             }
 298             else if( keyName.equals( "linenum" ) ) {
 299                 linenum = Integer.valueOf( getUnquotedString( key.getValue( ) ) );
 300             }
 301             else if( keyName.equals( "location" ) ) {
 302                 location = getUnquotedString( key.getValue( ) );
 303             }
 304         }
 305 
 306         // parse define arguments
 307         for( WMLTag arg: currentTag.getWMLTags( ) ) {
 308             for( WMLKey key: arg.getWMLKeys( ) ) {
 309                 if( key.getName( ).equals( "name" ) ) {
 310                     args.add( getUnquotedString( key.getValue( ) ) );
 311                 }
 312             }
 313         }
 314 
 315         defines_.put( defineName, new Define( defineName, value, textdomain,
 316             linenum, location, args ) );
 317     }

 319     private static String getUnquotedString( String str )
 320     {
 321         int startIndex = 0;
 322         int endIndex = str.length( );
 323 
 324         if( str.charAt( 0 ) == '"' ) {
 325             startIndex = 1;
 326         }
 327         if( str.charAt( endIndex - 1 ) == '"' ) {
 328             endIndex--;
 329         }
 330 
 331         return str.substring( startIndex, endIndex );
 332     }

 334     private String getVariableNameByContext( EObject context )
 335     {
 336         String variableName = null;
 337 
 338         if( context instanceof WMLKey ) {
 339             variableName = WMLUtils.getKeyValue( ( ( WMLKey ) context )
 340                 .getValues( ) );
 341         }
 342         else if( context instanceof WMLMacroCall ) {
 343             WMLMacroCall macro = ( WMLMacroCall ) context;
 344             if( macro.getParameters( ).size( ) > 0 ) {
 345                 variableName = WMLUtils.toString( macro.getParameters( )
 346 
 347                     .get( 0 ) );
 348             }
 349         }
 350 
 351         return variableName;
 352     }

 354     private void handleSetVariable( EObject context )
 355     {
 356         if( projectCache_ == null ) {
 357             return;
 358         }
 359 
 360         String variableName = getVariableNameByContext( context );
 361 
 362         if( variableName == null ) {
 363             Logger.getInstance( ).logWarn(
 364                 "setVariable: couldn't get variable name from context:"
 365                     + context );
 366         }
 367 
 368         WMLVariable variable = projectCache_.getVariables( ).get( variableName );
 369         if( variable == null ) {
 370             variable = new WMLVariable( variableName );
 371             projectCache_.getVariables( ).put( variableName, variable );
 372         }
 373 
 374         int nodeOffset = NodeModelUtils.getNode( context ).getTotalOffset( );
 375         for( Scope scope: variable.getScopes( ) ) {
 376             if( scope.contains( dependencyIndex_, nodeOffset ) ) {
 377                 return; // nothing to do
 378             }
 379         }
 380 
 381         // couldn't find any scope. Add a new one then.
 382         variable.getScopes( ).add( new Scope( dependencyIndex_, nodeOffset ) );
 383     }
 384 
 385     private void handleUnsetVariable( EObject context )
 386     {
 387         if( projectCache_ == null ) {
 388             return;
 389         }
 390 
 391         String variableName = getVariableNameByContext( context );
 392         if( variableName == null ) {
 393             Logger.getInstance( ).logWarn(
 394                 "unsetVariable: couldn't get variable name from context:"
 395                     + context );
 396         }
 397 
 398         WMLVariable variable = projectCache_.getVariables( ).get( variableName );
 399         if( variable == null ) {
 400             return;
 401         }
 402 
 403         int nodeOffset = NodeModelUtils.getNode( context ).getTotalOffset( );
 404 
 405         // get the first containing scope, and modify its end index/offset
 406 
 407         for( Scope scope: variable.getScopes( ) ) {
 408             if( scope.contains( dependencyIndex_, nodeOffset ) ) {
 409 
 410                 scope.EndIndex = dependencyIndex_;
 411                 scope.EndOffset = nodeOffset;
 412 
 413                 return;
 414             }
 415         }
 416     }
