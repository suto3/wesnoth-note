   9 function ca_wolves_multipacks_attack:evaluation(cfg)
  10     -- If wolves have attacks left, call this CA
  11     -- It will be disabled by being black-listed, so as to avoid
  12     -- having to do the full attack evaluation for every single move evaluation
  13 
  14     local wolves = AH.get_units_with_attacks {
  15         side = wesnoth.current.side,
  16         type = cfg.type or "Wolf"
  17     }
  18 
  19     if wolves[1] then return cfg.ca_score end
  20     return 0
  21 end

  23 function ca_wolves_multipacks_attack:execution(cfg)
  24     local packs = WMPF.assign_packs(cfg)
  25 
  26     -- Attacks are dealt with on a pack by pack basis
  27     -- and we want all wolves in a pack to move first, before going on to the next pack
  28     for pack_number,pack in pairs(packs) do
  29         local keep_attacking_this_pack = true
  30         local pack_has_attacked = false
  31 
  32         -- This repeats until all wolves in a pack have attacked, or none can attack any more
  33         while keep_attacking_this_pack do
  34             local wolves, attacks = {}, {}
  35             for _,pack_wolf in ipairs(pack) do
  36                 -- Wolf might have moved in previous attack -> use id to identify it
  37                 local wolf = wesnoth.get_units { id = pack_wolf.id }[1]
  38                 if wolf and (wolf.attacks_left > 0) then table.insert(wolves, wolf) end
  39             end
  40 
  41             local all_attacks = {}
  42             if wolves[1] then all_attacks = AH.get_attacks(wolves, { simulate_combat = true }) end
  43 
  44             -- Eliminate targets that would split up the wolves by more than 3 hexes
  45             local attacks = {}
  46             for _,attack in ipairs(all_attacks) do
  47                 local attack_splits_pack = false
  48                 for _,wolf in ipairs(wolves) do
  49                     local nh = AH.next_hop(wolf, attack.dst.x, attack.dst.y)
  50                     local dist = M.distance_between(nh[1], nh[2], attack.dst.x, attack.dst.y)
  51                     if (dist > 3) then
  52                         attack_splits_pack = true
  53                         break
  54                     end
  55                 end
  56                 if (not attack_splits_pack) then
  57                     table.insert(attacks, attack)
  58                 end
  59             end
  60 
  61             if attacks[1] then
  62                 -- Figure out how many different wolves can reach each target, and on how many hexes
  63                 -- The target with the largest value for the smaller of these two numbers is chosen
  64                 -- This is not an exact method, but good enough and much faster than simulating combat
  65                 local attack_map_wolves, attack_map_hexes = {}, {}
  66                 for _,attack in ipairs(attacks) do
  67                     -- Number different wolves
  68                     local att_xy = attack.src.x + attack.src.y * 1000
  69                     local def_xy = attack.target.x + attack.target.y * 1000
  70                     if (not attack_map_wolves[def_xy]) then attack_map_wolves[def_xy] = {} end
  71                     attack_map_wolves[def_xy][att_xy] = 1
  72 
  73                     -- Number different hexes
  74                     if (not attack_map_hexes[def_xy]) then attack_map_hexes[def_xy] = {} end
  75                     attack_map_hexes[def_xy][attack.dst.x + attack.dst.y * 1000] = 1
  76                 end
  77 
  78                 -- Find which target can be attacked by the most units, from the most hexes; and rate by fewest HP if equal
  79                 local max_rating, best_target = -9e99
  80                 for attack_ind,attack in pairs(attack_map_wolves) do
  81                     local number_wolves, number_hexes = 0, 0
  82                     for _,w in pairs(attack) do number_wolves = number_wolves + 1 end
  83                     for _,h in pairs(attack_map_hexes[attack_ind]) do number_hexes = number_hexes + 1 end
  84                     local rating = math.min(number_wolves, number_hexes)
  85 
  86                     local target = wesnoth.get_unit(attack_ind % 1000, math.floor(attack_ind / 1000))
  87                     rating = rating - target.hitpoints / 100.
  88 
  89                     -- Also, any target sitting next to a wolf of the same pack that has
  90                     -- no attacks left is priority targeted (in order to stick with
  91                     -- the same target for all wolves of the pack)
  92                     for xa,ya in H.adjacent_tiles(target.x, target.y) do
  93                         local adj_unit = wesnoth.get_unit(xa, ya)
  94                         if adj_unit then
  95                             local unit_pack_number = MAIUV.get_mai_unit_variables(adj_unit, cfg.ai_id, "pack_number")
  96                             if (unit_pack_number == pack_number)
  97                                and (adj_unit.side == wesnoth.current.side)
  98                                and (adj_unit.attacks_left == 0)
  99                             then
 100                                 rating = rating + 10
 101                             end
 102                         end
 103                     end
 104 
 105                     if rating > max_rating then
 106                         max_rating, best_target = rating, target
 107                     end
 108                 end
 109 
 110                 -- Now we know the best target and need to attack
 111                 -- This is done on a wolf-by-wolf basis, the outside while loop taking care of
 112                 -- the next wolf in the pack on subsequent iterations
 113                 local max_rating, best_attack = -9e99
 114                 for _,attack in ipairs(attacks) do
 115                     if (attack.target.x == best_target.x) and (attack.target.y == best_target.y) then
 116                         local rating = attack.att_stats.average_hp / 2. - attack.def_stats.average_hp
 117                         if (rating > max_rating) then
 118                             max_rating, best_attack = rating, attack
 119                         end
 120                     end
 121                 end
 122 
 123                 if cfg.show_pack_number then
 124                     WMPF.clear_label(best_attack.src.x, best_attack.src.y)
 125                 end
 126 
 127                 local attacker = wesnoth.get_unit(best_attack.src.x, best_attack.src.y)
 128                 local defender = wesnoth.get_unit(best_attack.target.x, best_attack.target.y)
 129 
 130                 AH.robust_move_and_attack(ai, attacker, best_attack.dst, defender)
 131 
 132                 if cfg.show_pack_number then
 133                     if attacker and attacker.valid then
 134                         if cfg.show_pack_number then WMPF.put_label(attacker.x, attacker.y, pack_number) end
 135                     end
 136                     if (not defender) or (not defender.valid) then
 137                         WMPF.clear_label(best_attack.target.x, best_attack.target.y)
 138                     end
 139                 end
 140 
 141                 pack_has_attacked = true
 142             else
 143                 keep_attacking_this_pack = false
 144             end
 145         end
 146 
 147         -- Finally, if any of the wolves in this pack did attack, move the rest of the pack in close
 148         if pack_has_attacked then
 149             local wolves_moves, wolves_no_moves = {}, {}
 150             for _,pack_wolf in ipairs(pack) do
 151                 -- Wolf might have moved in previous attack -> use id to identify it
 152                 local wolf = wesnoth.get_units { id = pack_wolf.id }[1]
 153                 if wolf then
 154                     if (wolf.moves > 0) then
 155                         table.insert(wolves_moves, wolf)
 156                     else
 157                         table.insert(wolves_no_moves, wolf)
 158                     end
 159                 end
 160             end
 161 
 162             -- If we have both wolves that have moved and those that have not moved,
 163             -- move the latter toward the former
 164             if wolves_moves[1] and wolves_no_moves[1] then
 165                 for _,wolf_moves in ipairs(wolves_moves) do
 166                     local best_hex = AH.find_best_move(wolf_moves, function(x, y)
 167                         local rating = 0
 168                         for _,wolf_no_moves in ipairs(wolves_no_moves) do
 169                             rating = rating - M.distance_between(x, y, wolf_no_moves.x, wolf_no_moves.y)
 170                         end
 171                         return rating
 172                     end)
 173 
 174                     if cfg.show_pack_number then
 175                         WMPF.clear_label(wolf_moves.x, wolf_moves.y)
 176                     end
 177 
 178                     AH.movefull_stopunit(ai, wolf_moves, best_hex)
 179 
 180                     if cfg.show_pack_number and wolf_moves and wolf_moves.valid then
 181                         WMPF.put_label(wolf_moves.x, wolf_moves.y, pack_number)
 182                     end
 183                 end
 184             end
 185         end
 186     end
 187 end

 189 return ca_wolves_multipacks_attack
