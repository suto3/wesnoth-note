   1 #textdomain wesnoth
   2 # Utility macros for manipulating map terrain and overlays.
   3 
   4 # These don't depend on any other macros.  Please don't change this.
   5 # ! in comments is used in generating HTML documentation, ignore it otherwise.
   6 
   7 #define MODIFY_TERRAIN TERRAIN X_SPAN Y_SPAN
   8     # Changes the terrain at a given list of coordinates
   9     #
  10     # For example, we could make 14,15 and 14,16 grassland:
  11     #! {MODIFY_TERRAIN Gg (14,14) (15,16)}
  12     [terrain]
  13         terrain={TERRAIN}
  14         x={X_SPAN}
  15         y={Y_SPAN}
  16     [/terrain]
  17 #enddef
  18 
  19 #define MODIFY_TERRAIN_MASK X Y MASK_VALUE RULES_WML
  20     # Changes the terrain for a given area
  21     [terrain_mask]
  22         x={X}
  23         y={Y}
  24         mask={MASK_VALUE}
  25         {RULES_WML}
  26     [/terrain_mask]
  27 #enddef
  28 
  29 #define SCATTER_IMAGE FILTER NUMBER IMAGE
  30     # Place NUMBER copies of the IMAGE on map hexes matching FILTER.
  31     #
  32     # This call will scatter 20 copies of a pine-tree graphic over grassland:
  33     #! {SCATTER_IMAGE (terrain=Gg) 20 scenery/pine1.png}
  34     [random_placement]
  35         num_items={NUMBER}
  36         variable=random_placement_location
  37         allow_less=yes
  38         [filter_location]
  39             {FILTER}
  40         [/filter_location]
  41         [command]
  42             [item]
  43                 image={IMAGE}
  44                 x,y=$random_placement_location.x,$random_placement_location.y
  45             [/item]
  46         [/command]
  47     [/random_placement]
  48     {CLEAR_VARIABLE random_placement_location}
  49 #enddef
  50 
  51 #define SCATTER_EMBELLISHMENTS TERRAINLIST EMBELLISHMENT_NAME PERCENTAGE
  52     # Adds the given embellishment to the given percentage of the given terrain
  53     # on the map.
  54     #
  55     # For example, this will add flowers to 5% of all grassland:
  56     #! {SCATTER_EMBELLISHMENTS G* ^Efm 5}
  57     [random_placement]
  58         num_items="size * ({PERCENTAGE} / 100)"
  59         variable=random_placement_location
  60         allow_less=yes
  61         [filter_location]
  62             terrain={TERRAINLIST}
  63         [/filter_location]
  64         [command]
  65             [terrain]
  66                 x,y=$random_placement_location.x,$random_placement_location.y
  67                 terrain={EMBELLISHMENT_NAME}
  68                 layer=overlay
  69             [/terrain]
  70         [/command]
  71     [/random_placement]
  72     {CLEAR_VARIABLE random_placement_location}
  73 #enddef
