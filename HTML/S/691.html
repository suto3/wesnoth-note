 105 config mp_game_settings::to_config() const
 106 {
 107         config cfg;
 108 
 109         cfg["scenario"] = name;
 110         cfg["hash"] = hash;
 111         cfg["mp_era"] = mp_era;
 112         cfg["mp_era_addon_id"] = mp_era_addon_id;
 113         cfg["mp_scenario"] = mp_scenario;
 114         cfg["mp_scenario_name"] = mp_scenario_name;
 115         cfg["mp_campaign"] = mp_campaign;
 116         cfg["active_mods"] = utils::join(active_mods, ",");
 117         cfg["side_users"] = utils::join_map(side_users);
 118         cfg["show_connect"] = show_connect;
 119         cfg["experience_modifier"] = xp_modifier;
 120         cfg["mp_countdown"] = mp_countdown;
 121         cfg["mp_countdown_init_time"] = mp_countdown_init_time;
 122         cfg["mp_countdown_turn_bonus"] = mp_countdown_turn_bonus;
 123         cfg["mp_countdown_reservoir_time"] = mp_countdown_reservoir_time;
 124         cfg["mp_countdown_action_bonus"] = mp_countdown_action_bonus;
 125         cfg["mp_num_turns"] = num_turns;
 126         cfg["mp_village_gold"] = village_gold;
 127         cfg["mp_village_support"] = village_support;
 128         cfg["mp_fog"] = fog_game;
 129         cfg["mp_shroud"] = shroud_game;
 130         cfg["mp_use_map_settings"] = use_map_settings;
 131         cfg["mp_random_start_time"] = random_start_time;
 132         cfg["observer"] = allow_observers;
 133         cfg["registered_users_only"] = registered_users_only;
 134         cfg["shuffle_sides"] = shuffle_sides;
 135         cfg["random_faction_mode"] = random_faction_mode;
 136         cfg["savegame"] = saved_game;
 137         cfg.add_child("options", options);
 138 
 139         for(auto& p : addons) {
 140                 config & c = cfg.add_child("addon");
 141                 p.second.write(c);
 142                 c["id"] = p.first;
 143         }
 144 
 145         return cfg;
 146 }

 148 mp_game_settings::addon_version_info::addon_version_info(const config & cfg)

 169 void mp_game_settings::update_addon_requirements(const config & cfg) {
 170         if (cfg["id"].empty()) {
 171                 WRN_NG << "Tried to add add-on metadata to a game, missing mandatory id field... skipping.\n" << cfg.debug() << "\n";
 172                 return;
 173         }
 174 
 175         mp_game_settings::addon_version_info new_data(cfg);
 176 
 177         // Check if this add-on already has an entry as a dependency for this scenario. If so, try to reconcile their version info,
 178         // by taking the larger of the min versions. The version should be the same for all WML from the same add-on...
 179         std::map<std::string, addon_version_info>::iterator it = addons.find(cfg["id"].str());
 180         if (it != addons.end()) {
 181                 addon_version_info & addon = it->second;
 182 
 183                 if (new_data.version) {
 184                         if (!addon.version || (*addon.version != *new_data.version)) {
 185                                 WRN_NG << "Addon version data mismatch -- not all local WML has same version of '" << cfg["id"].str() << "' addon.\n";
 186                         }
 187                 }
 188                 if (addon.version && !new_data.version) {
 189                         WRN_NG << "Addon version data mismatch -- not all local WML has same version of '" << cfg["id"].str() << "' addon.\n";
 190                 }
 191                 if (new_data.min_version) {
 192                         if (!addon.min_version || (*new_data.min_version > *addon.min_version)) {
 193                                 addon.min_version = *new_data.min_version;
 194                         }
 195                 }
 196         } else {
 197                 // Didn't find this addon-id in the map, so make a new entry.
 198                 addons.emplace(cfg["id"].str(), new_data);
 199         }
 200 }
