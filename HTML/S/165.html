  47 char* uncompress_buffer(const string_span& input, string_span* span)
  48 {
  49         int nalloc = input.size();
  50         int state = 0;
  51         try {
  52                 std::istringstream stream(std::string(input.begin(), input.end()));
  53                 state = 1;
  54                 boost::iostreams::filtering_stream<boost::iostreams::input> filter;
  55                 state = 2;
  56                 if (!span->empty() && *span->begin() == 'B') {
  57                         filter.push(boost::iostreams::bzip2_decompressor());
  58                 } else {
  59                         filter.push(boost::iostreams::gzip_decompressor());
  60                 }
  61                 filter.push(stream);
  62                 state = 3;
  63 
  64                 const size_t chunk_size = input.size() * 10;
  65                 nalloc = chunk_size;
  66                 std::vector<char> buf(chunk_size);
  67                 state = 4;
  68                 size_t len = 0;
  69                 size_t pos = 0;
  70                 while(filter.good() && (len = filter.read(&buf[pos], chunk_size).gcount()) == chunk_size) {
  71                         if(pos + chunk_size > 40000000) {
  72                                 throw error("WML document exceeds 40MB limit");
  73                         }
  74 
  75                         pos += len;
  76                         buf.resize(pos + chunk_size);
  77                         len = 0;
  78                 }
  79 
  80                 if(!filter.eof() && !filter.good()) {
  81                         throw error("failed to uncompress");
  82                 }
  83 
  84                 pos += len;
  85                 state = 5;
  86                 nalloc = pos;
  87 
  88                 buf.resize(pos);
  89                 state = 6;
  90 
  91                 char* small_out = new char[pos+1];
  92                 memcpy(small_out, &buf[0], pos);
  93                 state = 7;
  94 
  95                 small_out[pos] = 0;
  96 
  97                 *span = string_span(small_out, pos);
  98                 state = 8;
  99                 return small_out;
 100         } catch (std::bad_alloc& e) {
 101                 ERR_SWML << "ERROR: bad_alloc caught in uncompress_buffer() state "
 102                 << state << " alloc bytes " << nalloc << " with input: '"
 103                 << input << "' " << e.what() << std::endl;
 104                 throw error("Bad allocation request in uncompress_buffer().");
 105         }
 106 }

 108 char* compress_buffer(const char* input, string_span* span, bool bzip2)
 109 {
 110         int nalloc = strlen(input);
 111         int state = 0;
 112         try {
 113                 std::string in(input);
 114                 state = 1;
 115                 std::istringstream istream(in);
 116                 state = 2;
 117                 boost::iostreams::filtering_stream<boost::iostreams::output> filter;
 118                 state = 3;
 119                 if (bzip2) {
 120                         filter.push(boost::iostreams::bzip2_compressor());
 121                 } else {
 122                         filter.push(boost::iostreams::gzip_compressor());
 123                 }
 124                 state = 4;
 125                 nalloc = in.size()*2 + 80;
 126                 std::vector<char> buf(nalloc);
 127                 boost::iostreams::array_sink out(&buf[0], buf.size());
 128                 filter.push(boost::iostreams::counter());
 129                 filter.push(out);
 130 
 131                 state = 5;
 132 
 133                 boost::iostreams::copy(istream, filter, buf.size());
 134                 const int len = filter.component<boost::iostreams::counter>(1)->characters();
 135                 assert(len < 128*1024*1024);
 136                 if((!filter.eof() && !filter.good()) || len == static_cast<int>(buf.size())) {
 137                         throw error("failed to compress");
 138                 }
 139                 state = 6;
 140                 nalloc = len;
 141 
 142                 buf.resize(len);
 143                 state = 7;
 144 
 145                 char* small_out = new char[len];
 146                 memcpy(small_out, &buf[0], len);
 147                 state = 8;
 148 
 149                 *span = string_span(small_out, len);
 150                 assert(*small_out == (bzip2 ? 'B' : 31));
 151                 state = 9;
 152                 return small_out;
 153         } catch (std::bad_alloc& e) {
 154                 ERR_SWML << "ERROR: bad_alloc caught in compress_buffer() state "
 155                 << state << " alloc bytes " << nalloc << " with input: '"
 156                 << input << "' " << e.what() << std::endl;
 157                 throw error("Bad allocation request in compress_buffer().");
 158         }
 159 }

 226 node::node(document& doc, node* parent, const char** str, int depth) :
 227         doc_(&doc),
 228         attr_(),
 229         parent_(parent),
 230         children_(),
 231         ordered_children_(),
 232         output_cache_()
 233 {
 234         if(depth >= 1000) {
 235                 throw error("elements nested too deep");
 236         }
 237 
 238         const char*& s = *str;
 239 
 240         const char* const begin = s;
 241         while(*s) {
 242                 switch(*s) {
 243                 case '[': {
 244                         if(s[1] == '/') {
 245                                 output_cache_ = string_span(begin, s - begin);
 246                                 s = strchr(s, ']');
 247                                 if(s == nullptr) {
 248                                         throw error("end element unterminated");
 249                                 }
 250 
 251                                 ++s;
 252                                 return;
 253                         }
 254 
 255                         ++s;
 256                         const char* end = strchr(s, ']');
 257                         if(end == nullptr) {
 258                                 throw error("unterminated element");
 259                         }
 260 
 261                         const int list_index = get_children(string_span(s, end - s));
 262                         check_ordered_children();
 263 
 264                         s = end + 1;
 265 
 266                         children_[list_index].second.push_back(new node(doc, this, str, depth+1));
 267                         ordered_children_.emplace_back(list_index, children_[list_index].second.size() - 1);
 268                         check_ordered_children();
 269 
 270                         break;
 271                 }
 272                 case ' ':
 273                 case '\t':
 274                 case '\n':
 275                         ++s;
 276                         break;
 277                 case '#':
 278                         s = strchr(s, '\n');
 279                         if(s == nullptr) {
 280                                 throw error("did not find newline after '#'");
 281                         }
 282                         break;
 283                 default: {
 284                         const char* end = strchr(s, '=');
 285                         if(end == nullptr) {
 286                                 ERR_SWML << "attribute: " << s << std::endl;
 287                                 throw error("did not find '=' after attribute");
 288                         }
 289 
 290                         string_span name(s, end - s);
 291                         s = end + 1;
 292                         if(*s == '_') {
 293                                 s = strchr(s, '"');
 294                                 if(s == nullptr) {
 295                                         throw error("did not find '\"' after '_'");
 296                                 }
 297                         }
 298 
 299                         if (*s != '"') {
 300                                 end = strchr(s, '\n');
 301                                 if (!end) {
 302                                         ERR_SWML << "ATTR: '" << name << "' (((" << s << ")))" << std::endl;
 303                                         throw error("did not find end of attribute");
 304                                 }
 305                                 if (memchr(s, '"', end - s))
 306                                         throw error("found stray quotes in unquoted value");
 307                                 goto read_attribute;
 308                         }
 309                         end = s;
 310                         for(;;)
 311                         {
 312                                 // Read until the first single double quote.
 313                                 while((end = strchr(end+1, '"')) && end[1] == '"') {
 314 #ifdef _MSC_VER
 315 #pragma warning (pop)
 316 #endif
 317                                         ++end;
 318                                 }
 319                                 if(end == nullptr)
 320                                         throw error("did not find end of attribute");
 321 
 322                                 // Stop if newline.
 323                                 const char *endline = end + 1;
 324                                 while (*endline == ' ') ++endline;
 325                                 if (*endline == '\n') break;
 326 
 327                                 // Read concatenation marker.
 328                                 if (*(endline++) != '+')
 329                                         throw error("did not find newline after end of attribute");
 330                                 if (*(endline++) != '\n')
 331                                         throw error("did not find newline after '+'");
 332 
 333                                 // Read textdomain marker.
 334                                 if (*endline == '#') {
 335                                         endline = strchr(endline + 1, '\n');
 336                                         if (!endline)
 337                                                 throw error("did not find newline after '#'");
 338                                         ++endline;
 339                                 }
 340 
 341                                 // Read indentation and start of string.
 342                                 while (*endline == '\t') ++endline;
 343                                 if (*endline == '_') ++endline;
 344                                 if (*endline != '"')
 345                                         throw error("did not find quotes after '+'");
 346                                 end = endline;
 347                         }
 348 
 349                         ++s;
 350 
 351                         read_attribute:
 352                         string_span value(s, end - s);
 353                         if(attr_.empty() == false && !(attr_.back().first < name)) {
 354                                 ERR_SWML << "attributes: '" << attr_.back().first << "' < '" << name << "'" << std::endl;
 355                                 throw error("attributes not in order");
 356                         }
 357 
 358                         s = end + 1;
 359 
 360                         attr_.emplace_back(name, value);
 361                 }
 362                 }
 363         }
 364 
 365         output_cache_ = string_span(begin, s - begin);
 366         check_ordered_children();
 367 }

 483 void node::remove_child(const string_span& name, size_t index)
 484 {
 485         set_dirty();
 486 
 487         //if we don't already have a vector for this item we don't want to add one.
 488         child_map::iterator itor = find_in_map(children_, name);
 489         if(itor == children_.end()) {
 490                 return;
 491         }
 492 
 493         child_list& list = itor->second;
 494         if(index >= list.size()) {
 495                 return;
 496         }
 497 
 498         remove_ordered_child(itor - children_.begin(), index);
 499 
 500         debug_delete(list[index]);
 501         list.erase(list.begin() + index);
 502 
 503         if(list.empty()) {
 504                 remove_ordered_child_list(itor - children_.begin());
 505                 children_.erase(itor);
 506         }
 507 }

 509 void node::insert_ordered_child(int child_map_index, int child_list_index)
 510 {
 511         bool inserted = false;
 512         std::vector<node_pos>::iterator i = ordered_children_.begin();
 513         while(i != ordered_children_.end()) {
 514                 if(i->child_map_index == child_map_index && i->child_list_index > child_list_index) {
 515                         i->child_list_index++;
 516                 } else if(i->child_map_index == child_map_index && i->child_list_index == child_list_index) {
 517                         inserted = true;
 518                         i->child_list_index++;
 519                         i = ordered_children_.insert(i, node_pos(child_map_index, child_list_index));
 520                         ++i;
 521                 }
 522 
 523                 ++i;
 524         }
 525 
 526         if(!inserted) {
 527                 ordered_children_.emplace_back(child_map_index, child_list_index);
 528         }
 529 }

 531 void node::remove_ordered_child(int child_map_index, int child_list_index)
 532 {
 533         int erase_count = 0;
 534         std::vector<node_pos>::iterator i = ordered_children_.begin();
 535         while(i != ordered_children_.end()) {
 536                 if(i->child_map_index == child_map_index && i->child_list_index == child_list_index) {
 537                         i = ordered_children_.erase(i);
 538                         ++erase_count;
 539                 } else {
 540                         if(i->child_map_index == child_map_index && i->child_list_index > child_list_index) {
 541                                 i->child_list_index--;
 542                         }
 543                         ++i;
 544                 }
 545         }
 546 
 547         assert(erase_count == 1);

 704         check_ordered_children();
 705         if(output_cache_.empty() == false) {
 706                 return output_cache_.size();
 707         }
 708 
 709         int res = 0;
 710         for(attribute_list::const_iterator i = attr_.begin(); i != attr_.end(); ++i) {
 711                 res += i->first.size() + i->second.size() + 4;
 712         }
 713 
 714         size_t count_children = 0;
 715         for(child_map::const_iterator i = children_.begin(); i != children_.end(); ++i) {
 716                 for(child_list::const_iterator j = i->second.begin(); j != i->second.end(); ++j) {
 717                         res += i->first.size()*2 + 7;
 718                         res += (*j)->output_size();
 719                         ++count_children;
 720                 }
 721         }
 722 
 723         assert(count_children == ordered_children_.size());
 724 
 725         return res;
 726 }

 728 void node::shift_buffers(ptrdiff_t offset)
 729 {
 730         if(!output_cache_.empty()) {
 731                 output_cache_ = string_span(output_cache_.begin() + offset, output_cache_.size());
 732         }
 733 
 734         for(std::vector<attribute>::iterator i = attr_.begin(); i != attr_.end(); ++i) {
 735                 i->first = string_span(i->first.begin() + offset, i->first.size());
 736                 i->second = string_span(i->second.begin() + offset, i->second.size());
 737         }
 738 
 739         for(child_map::iterator i = children_.begin(); i != children_.end(); ++i) {
 740                 string_span& key = i->first;
 741                 key = string_span(key.begin() + offset, key.size());
 742                 for(child_list::iterator j = i->second.begin(); j != i->second.end(); ++j) {
 743                         (*j)->shift_buffers(offset);
 744                 }
 745         }
 746 }

 748 void node::output(char*& buf, CACHE_STATUS cache_status)
 749 {
 750         if(output_cache_.empty() == false) {
 751                 memcpy(buf, output_cache_.begin(), output_cache_.size());
 752                 if(cache_status == REFRESH_CACHE) {
 753                         shift_buffers(buf - output_cache_.begin());
 754                 }
 755                 buf += output_cache_.size();
 756                 return;
 757         }
 758 
 759         char* begin = buf;
 760 
 761         for(std::vector<attribute>::iterator i = attr_.begin(); i != attr_.end(); ++i) {
 762                 memcpy(buf, i->first.begin(), i->first.size());
 763                 i->first = string_span(buf, i->first.size());
 764                 buf += i->first.size();
 765                 *buf++ = '=';
 766                 *buf++ = '"';
 767                 memcpy(buf, i->second.begin(), i->second.size());
 768                 i->second = string_span(buf, i->second.size());
 769                 buf += i->second.size();
 770                 *buf++ = '"';
 771                 *buf++ = '\n';
 772         }
 773 
 774         for(std::vector<node_pos>::const_iterator i = ordered_children_.begin();
 775             i != ordered_children_.end(); ++i) {
 776                 assert(i->child_map_index < children_.size());
 777                 assert(i->child_list_index < children_[i->child_map_index].second.size());
 778                 string_span& attr = children_[i->child_map_index].first;
 779                 *buf++ = '[';
 780                 memcpy(buf, attr.begin(), attr.size());
 781                 attr = string_span(buf, attr.size());
 782                 buf += attr.size();
 783                 *buf++ = ']';
 784                 *buf++ = '\n';
 785                 children_[i->child_map_index].second[i->child_list_index]->output(buf, cache_status);
 786                 *buf++ = '[';
 787                 *buf++ = '/';
 788                 memcpy(buf, attr.begin(), attr.size());
 789                 buf += attr.size();
 790                 *buf++ = ']';
 791                 *buf++ = '\n';
 792         }
 793 
 794         if(cache_status == REFRESH_CACHE) {
 795                 output_cache_ = string_span(begin, buf - begin);
 796         }
 797 }

 799 std::string node_to_string(const node& n)
 800 {
 801         //calling output with status=DO_NOT_MODIFY_CACHE really doesn't modify the
 802         //node, so we can do it safely
 803         node& mutable_node = const_cast<node&>(n);
 804         std::vector<char> v(mutable_node.output_size());
 805         char* ptr = &v[0];
 806         mutable_node.output(ptr, node::DO_NOT_MODIFY_CACHE);
 807         assert(ptr == &v[0] + v.size());
 808         return std::string(v.begin(), v.end());
 809 }

 811 void node::copy_into(node& n) const
 812 {
 813         n.set_dirty();
 814         for(attribute_list::const_iterator i = attr_.begin(); i != attr_.end(); ++i) {
 815                 char* key = i->first.duplicate();
 816                 char* value = i->second.duplicate();
 817                 n.doc_->take_ownership_of_buffer(key);
 818                 n.doc_->take_ownership_of_buffer(value);
 819                 n.set_attr(key, value);
 820         }
 821 
 822         for(std::vector<node_pos>::const_iterator i = ordered_children_.begin();
 823             i != ordered_children_.end(); ++i) {
 824                 assert(i->child_map_index < children_.size());
 825                 assert(i->child_list_index < children_[i->child_map_index].second.size());
 826                 char* buf = children_[i->child_map_index].first.duplicate();
 827                 n.doc_->take_ownership_of_buffer(buf);
 828                 children_[i->child_map_index].second[i->child_list_index]->copy_into(n.add_child(buf));
 829         }
 830 }

 832 void node::apply_diff(const node& diff)
 833 {
 834         set_dirty();
 835         const node* inserts = diff.child("insert");
 836         if(inserts != nullptr) {
 837                 for(attribute_list::const_iterator i = inserts->attr_.begin(); i != inserts->attr_.end(); ++i) {
 838                         char* name = i->first.duplicate();
 839                         char* value = i->second.duplicate();
 840                         set_attr(name, value);
 841                         doc_->take_ownership_of_buffer(name);
 842                         doc_->take_ownership_of_buffer(value);
 843                 }
 844         }
 845 
 846         const node* deletes = diff.child("delete");
 847         if(deletes != nullptr) {
 848                 for(attribute_list::const_iterator i = deletes->attr_.begin(); i != deletes->attr_.end(); ++i) {
 849                         std::pair<attribute_list::iterator,
 850                           attribute_list::iterator> range = std::equal_range(attr_.begin(), attr_.end(), i->first, string_span_pair_comparer());
 851                         if(range.first != range.second) {
 852                                 attr_.erase(range.first);
 853                         }
 854                 }
 855         }
 856 
 857         const child_list& child_changes = diff.children("change_child");
 858         for(child_list::const_iterator i = child_changes.begin(); i != child_changes.end(); ++i) {
 859                 const size_t index = (**i)["index"].to_int();
 860                 for(child_map::const_iterator j = (*i)->children_.begin(); j != (*i)->children_.end(); ++j) {
 861                         const string_span& name = j->first;
 862                         for(child_list::const_iterator k = j->second.begin(); k != j->second.end(); ++k) {
 863                                 child_map::iterator itor = find_in_map(children_, name);
 864                                 if(itor != children_.end()) {
 865                                         if(index < itor->second.size()) {
 866                                                 itor->second[index]->apply_diff(**k);
 867                                         }
 868                                 }
 869                         }
 870                 }
 871         }
 872 
 873         const child_list& child_inserts = diff.children("insert_child");
 874         for(child_list::const_iterator i = child_inserts.begin(); i != child_inserts.end(); ++i) {
 875                 const size_t index = (**i)["index"].to_int();
 876                 for(child_map::const_iterator j = (*i)->children_.begin(); j != (*i)->children_.end(); ++j) {
 877                         const string_span& name = j->first;
 878                         for(child_list::const_iterator k = j->second.begin(); k != j->second.end(); ++k) {
 879                                 char* buf = name.duplicate();
 880                                 doc_->take_ownership_of_buffer(buf);
 881                                 (*k)->copy_into(add_child_at(buf, index));
 882                         }
 883                 }
 884         }
 885 
 886         const child_list& child_deletes = diff.children("delete_child");
 887         for(child_list::const_iterator i = child_deletes.begin(); i != child_deletes.end(); ++i) {
 888                 const size_t index = (**i)["index"].to_int();
 889                 for(child_map::const_iterator j = (*i)->children_.begin(); j != (*i)->children_.end(); ++j) {
 890                         if(j->second.empty()) {
 891                                 continue;
 892                         }
 893 
 894                         const string_span& name = j->first;
 895                         remove_child(name, index);
 896                 }
 897         }
 898 }

1033 const char* document::output()
1034 {
1035         if(output_ && (!root_ || root_->is_dirty() == false)) {
1036                 return output_;
1037         }
1038         if(!root_) {
1039                 assert(compressed_buf_.empty() == false);
1040                 string_span uncompressed_buf;
1041                 buffers_.push_back(uncompress_buffer(compressed_buf_, &uncompressed_buf));
1042                 output_ = uncompressed_buf.begin();
1043                 return output_;
1044         }
1045 
1046         //we're dirty, so the compressed buf must also be dirty; clear it.
1047         compressed_buf_ = string_span();
1048 
1049         std::vector<char*> bufs;
1050         bufs.swap(buffers_);
1051 
1052         const int buf_size = root_->output_size() + 1;
1053         char* buf;
1054         try {
1055                 buf = new char[buf_size];
1056         } catch (std::bad_alloc& e) {
1057                 ERR_SWML << "ERROR: Trying to allocate " << buf_size << " bytes. "
1058                 << e.what() << std::endl;
1059                 throw error("Bad allocation request in output().");
1060         }
1061         buffers_.push_back(buf);
1062         output_ = buf;
1063 
1064         root_->output(buf, node::REFRESH_CACHE);
1065         *buf++ = 0;
1066         assert(buf == output_ + buf_size);
1067 
1068         for(std::vector<char*>::iterator i = bufs.begin(); i != bufs.end(); ++i) {
1069                 delete [] *i;
1070         }
1071 
1072         bufs.clear();
1073 
1074         return output_;
1075 }

1077 string_span document::output_compressed(bool bzip2)
1078 {
1079         if(compressed_buf_.empty() == false &&
1080            (root_ == nullptr || root_->is_dirty() == false)) {
1081                 assert(*compressed_buf_.begin() == (bzip2 ? 'B' : 31));
1082                 return compressed_buf_;
1083         }

1086         assert(*compressed_buf_.begin() == (bzip2 ? 'B' : 31));
1087

1091 void document::compress()
1092 {

1094         debug_delete(root_);
1095         root_ = nullptr;
1096         output_ = nullptr;
1097         std::vector<char*> new_buffers;
1098         for(std::vector<char*>::iterator i = buffers_.begin(); i != buffers_.end(); ++i) {
1099                 if(*i != compressed_buf_.begin()) {
1100                         delete [] *i;
1101                 } else {
1102                         new_buffers.push_back(*i);
1103                 }
1104         }
1105 
1106         buffers_.swap(new_buffers);
1107         assert(buffers_.size() == 1);

1186 std::string document::stats()
1187 {
1188         std::ostringstream s;
1189         int ndocs = 0;
1190         int ncompressed = 0;
1191         int compressed_size = 0;
1192         int ntext = 0;
1193         int text_size = 0;
1194         int nbuffers = 0;
1195         int nnodes = 0;
1196         int ndirty = 0;
1197         int nattributes = 0;
1198         for(document* d = head_doc; d != nullptr; d = d->next_) {
1199                 ndocs++;
1200                 nbuffers += d->buffers_.size();
1201 
1202                 if(d->compressed_buf_.is_null() == false) {
1203                         ++ncompressed;
1204                         compressed_size += d->compressed_buf_.size();
1205                 }
1206 
1207                 if(d->output_) {
1208                         ++ntext;
1209                         text_size += strlen(d->output_);
1210                 }
1211 
1212                 if(d->root_) {
1213                         nnodes += 1 + d->root_->nchildren();
1214                         nattributes += d->root_->nattributes_recursive();
1215                 }
1216 
1217                 if(d->root_ && d->root_->is_dirty()) {
1218                         ++ndirty;
1219                 }
1220         }
1221 
1222         const int nodes_alloc = nnodes*(sizeof(node) + 12);
1223         const int attr_alloc = nattributes*(sizeof(string_span)*2);
1224         const int total_alloc = compressed_size + text_size + nodes_alloc + attr_alloc;
1225 
1226         s << "WML documents: " << ndocs << "\n"
1227           << "Dirty: " << ndirty << "\n"
1228           << "With compression: " << ncompressed << " (" << compressed_size
1229           << " bytes)\n"
1230           << "With text: " << ntext << " (" << text_size
1231           << " bytes)\n"
1232           << "Nodes: " << nnodes << " (" << nodes_alloc << " bytes)\n"
1233           << "Attr: " << nattributes << " (" << attr_alloc << " bytes)\n"
1234           << "Buffers: " << nbuffers << "\n"
1235           << "Total allocation: " << total_alloc << " bytes\n";
1236 
1237         return s.str();
1238 }
