  82 unsigned toggle_panel::num_states() const
  83 {

 142 SDL_Rect toggle_panel::get_client_rect() const
 143 {
 144         std::shared_ptr<const toggle_panel_definition::resolution> conf
 145                         = std::static_pointer_cast<const toggle_panel_definition::
 146                                                                                                   resolution>(config());
 147         assert(conf);
 148 
 149         SDL_Rect result = get_rectangle();
 150         result.x += conf->left_border;
 151         result.y += conf->top_border;
 152         result.w -= conf->left_border + conf->right_border;
 153         result.h -= conf->top_border + conf->bottom_border;
 154 
 155         return result;
 156 }

 315 /*WIKI
 316  * @page = GUIWidgetDefinitionWML

 318  *
 319  * == Toggle panel ==
 320  *
 321  * @begin{parent}{name="gui/"}
 322  * @begin{tag}{name="oggle_panel_definition"}{min=0}{max=-1}{super="generic/widget_definition"}
 323  * The definition of a toggle panel. A toggle panel is like a toggle button, but
 324  * instead of being a button it's a panel. This means it can hold multiple child
 325  * items.
 326  *
 327  * @begin{tag}{name="resolution"}{min=0}{max=-1}{super=generic/widget_definition/resolution}
 328  * The resolution for a toggle panel also contains the following keys:
 329  * @begin{table}{config}
 330  *     top_border & unsigned & 0 &     The size which isn't used for the client
 331  *                                   area. $
 332  *     bottom_border & unsigned & 0 &  The size which isn't used for the client
 333  *                                   area. $
 334  *     left_border & unsigned & 0 &    The size which isn't used for the client
 335  *                                   area. $
 336  *     right_border & unsigned & 0 &   The size which isn't used for the client
 337  *                                   area. $
 338  * @end{table}
 339  *
 340  * The following states exist:
 341  * * state_enabled, the panel is enabled and not selected.
 342  * * state_disabled, the panel is disabled and not selected.
 343  * * state_focused, the mouse is over the panel and not selected.
 344  *
 345  * * state_enabled_selected, the panel is enabled and selected.
 346  * * state_disabled_selected, the panel is disabled and selected.
 347  * * state_focused_selected, the mouse is over the panel and selected.
 348  * @begin{tag}{name="state_enabled"}{min=0}{max=1}{super="generic/state"}
 349  * @end{tag}{name="state_enabled"}
 350  * @begin{tag}{name="state_disabled"}{min=0}{max=1}{super="generic/state"}
 351  * @end{tag}{name="state_disabled"}
 352  * @begin{tag}{name="state_focused"}{min=0}{max=1}{super="generic/state"}
 353  * @end{tag}{name="state_focused"}
 354  * @begin{tag}{name="state_enabled_selected"}{min=0}{max=1}{super="generic/state"}
 355  * @end{tag}{name="state_enabled_selected"}
 356  * @begin{tag}{name="state_disabled_selected"}{min=0}{max=1}{super="generic/state"}
 357  * @end{tag}{name="state_disabled_selected"}
 358  * @begin{tag}{name="state_focused_selected"}{min=0}{max=1}{super="generic/state"}
 359  * @end{tag}{name="state_focused_selected"}
 360  * @end{tag}{name="resolution"}
 361  * @end{tag}{name="oggle_panel_definition"}
 362  * @end{parent}{name="gui/"}
 363  */
 364 toggle_panel_definition::resolution::resolution(const config& cfg)
 365         : resolution_definition(cfg)
 366         , top_border(cfg["top_border"])
 367         , bottom_border(cfg["bottom_border"])
 368         , left_border(cfg["left_border"])
 369         , right_border(cfg["right_border"])
 370 {
 371         // Note the order should be the same as the enum state_t in toggle_panel.hpp.
 372         for(const auto& c : cfg.child_range("state"))
 373         {
 374                 state.emplace_back(c.child("enabled"));
 375                 state.emplace_back(c.child("disabled"));
 376                 state.emplace_back(c.child("focused"));
 377         }
 378 }
