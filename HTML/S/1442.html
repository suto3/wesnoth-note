  48      * Returns the number of 'character' characters in the target string
  49      * 
  50      * @param target
  51      *        The String to perform the count on.
  52      * @param character
  53      *        The character to count.

  56     public static int countOf( String target, char character )
  57     {
  58         if( target == null ) {
  59             return 0;
  60         }
  61 
  62         if( target.indexOf( character ) == - 1 ) {
  63             return 0;
  64         }
  65 
  66         int cnt = 0;
  67         String tmpString = target;
  68         while( tmpString.contains( new String( new char[] { character } ) ) ) {
  69             ++cnt;
  70             tmpString = tmpString
  71                 .substring( tmpString.indexOf( character ) + 1 );
  72         }
  73         return cnt;

  76     /**
  77      * Removes all consecutive aparitions of a character in the specified string
  78      * so that only one appearance remains in each past duplications of that
  79      * string
  80      * 
  81      * @param target
  82      *        the string to process
  83      * @param character
  84      *        the character to remove
  85      * @param removeTrailing
  86      *        removes or not the trailing 'character' characters
  87      * @param removePreceding
  88      *        removes or not the preceding 'character' characters
  89      * @return The processed string.
  90      */
  91     public static String removeIncorrectCharacters( String target,
  92         char character, boolean removeTrailing, boolean removePreceding )
  93     {
  94         if( target == null ) {
  95             return ""; //$NON-NLS-1$
  96         }
  97 
  98         StringBuilder resString = new StringBuilder( );
  99 
 100         for( int i = 0; i < target.length( ); i++ ) {
 101             // pass over successive repetitions:
 102             // abbbac will become: abac
 103             if( i > 0 && ( target.charAt( i ) == target.charAt( i - 1 ) ) ) {
 104                 continue;
 105             }
 106 
 107             if( target.charAt( i ) == character
 108                 && ( ( removeTrailing && i == target.length( ) ) || ( removePreceding && i == 0 ) ) ) {
 109                 continue;
 110             }
 111 
 112             resString.append( target.charAt( i ) );
 113         }
 114         return resString.toString( );
 115     }

 153     /**
 154      * Trims any quotes " of the specified string
 155      * 
 156      * @param string
 157      *        The string to trim
 158      * @return A new string with the quotes trimmed
 159      */
 160     public static String trimQuotes( String string )
 161     {
 162         if( isNullOrEmpty( string ) ) {
 163             return "";
 164         }
 165 
 166         while( string.charAt( 0 ) == '"' ) {
 167             string = string.substring( 1 );
 168         }
 169 
 170         int strSize = string.length( ) - 1;
 171         while( strSize >= 0 && string.charAt( strSize ) == '"' ) {
 172             string = string.substring( 0, strSize );
 173             --strSize;
 174         }
 175 
 176         return string;
 177     }

 179     /**
 180      * Normalizes the path given by the string, removing repeated separators
 181      * and replacing them by '|'
 182      * 
 183      * @param path
 184      *        the string that represents the path to be normalized
 185      * @return The normalized string.
 186      */
 187     public static String normalizePath( String path )
 188     {
 189         if( path == null ) {
 190             return ""; //$NON-NLS-1$
 191         }
 192 
 193         String str = StringUtils.trimEndPathSeparators( path );
 194         StringBuilder sb = new StringBuilder( str.length( ) );
 195 
 196         // normalize strings - remove repeating separators and replace them by |
 197         for( int i = 0, sw = 0; i < str.length( ); i++ ) {
 198             if( str.charAt( i ) == '/' || str.charAt( i ) == '\\' ) {
 199                 if( sw == 0 ) // we haven't added already the | token
 200                 {
 201                     sb.append( '|' );
 202                     sw = 1;
 203                 }
 204                 continue;
 205             }
 206             sb.append( str.charAt( i ) );
 207             sw = 0; // reset the switch when meeting non-separators
 208         }
 209         return sb.toString( );
 210     }

 234         Matcher matcher = pattern.matcher( string );
 235         String indent = ""; //$NON-NLS-1$
 236         if( matcher.find( ) ) {
 237             int end = matcher.end( );
 238             indent = string.substring( matcher.start( ), end );
 239         }
 240 
 241         String[] tmpTarget = getLines( target );
 242         for( int i = 1; i < tmpTarget.length; i++ ) {
 243             tmpTarget[i] = indent + tmpTarget[i];
 244         }

 296     {
 297         List< String > groupList = new ArrayList< String >( );
 298         try {
 299             Pattern regex = Pattern
 300                 .compile( regexStr, Pattern.CASE_INSENSITIVE );
 301             Matcher regexMatcher = regex.matcher( targetString );
 302             while( regexMatcher.find( ) ) {
 303                 groupList.add( regexMatcher.group( ) );
