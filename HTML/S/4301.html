   7 local function hunter_attack_weakest_adj_enemy(ai, hunter)
   8     -- Attack the enemy with the fewest hitpoints adjacent to 'hunter', if there is one
   9     -- Returns status of the attack:
  10     --   'attacked': if a unit was attacked
  11     --   'killed': if a unit was killed
  12     --   'no_attack': if no unit was attacked
  13 
  14     if (hunter.attacks_left == 0) then return 'no_attack' end
  15 
  16     local min_hp, target = 9e99
  17     for xa,ya in H.adjacent_tiles(hunter.x, hunter.y) do
  18         local enemy = wesnoth.get_unit(xa, ya)
  19         if AH.is_attackable_enemy(enemy) then
  20             if (enemy.hitpoints < min_hp) then
  21                 min_hp, target = enemy.hitpoints, enemy
  22             end
  23         end
  24     end
  25 
  26     if target then
  27         AH.checked_attack(ai, hunter, target)
  28         if target and target.valid then
  29             return 'attacked'
  30         else
  31             return 'killed'
  32         end
  33     end
  34 
  35     return 'no_attack'
  36 end

  54 function ca_hunter:execution(cfg)
  55     -- Hunter does a random wander in area given by @cfg.hunting_ground until it finds
  56     -- and kills an enemy unit, then retreats to position given by @cfg.home_x/y
  57     -- for @cfg.rest_turns turns, or until fully healed
  58 
  59     local hunter = get_hunter(cfg)
  60     local hunter_vars = MAIUV.get_mai_unit_variables(hunter, cfg.ai_id)
  61 
  62     -- If hunting_status is not set for the hunter -> default behavior -> random wander
  63     if (not hunter_vars.hunting_status) then
  64         -- Hunter gets a new goal if none exist or on any move with 10% random chance
  65         local rand = math.random(10)
  66         if (not hunter_vars.goal_x) or (rand == 1) then
  67             -- 'locs' includes border hexes, but that does not matter here
  68             locs = AH.get_passable_locations((H.get_child(cfg, "filter_location") or {}), hunter)
  69             local rand = math.random(#locs)
  70 
  71             hunter_vars.goal_x, hunter_vars.goal_y = locs[rand][1], locs[rand][2]
  72             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
  73         end
  74 
  75         local reach_map = AH.get_reachable_unocc(hunter)
  76 
  77         -- Now find the one of these hexes that is closest to the goal
  78         local max_rating, best_hex = -9e99
  79         reach_map:iter( function(x, y, v)
  80             -- Distance from goal is first rating
  81             local rating = -M.distance_between(x, y, hunter_vars.goal_x, hunter_vars.goal_y)
  82 
  83             -- Huge rating bonus if this is next to an enemy
  84             local enemy_hp = 500
  85             for xa,ya in H.adjacent_tiles(x, y) do
  86                 local enemy = wesnoth.get_unit(xa, ya)
  87                 if AH.is_attackable_enemy(enemy) then
  88                     if (enemy.hitpoints < enemy_hp) then enemy_hp = enemy.hitpoints end
  89                 end
  90             end
  91             rating = rating + 500 - enemy_hp  -- prefer attack on weakest enemy
  92 
  93             if (rating > max_rating) then
  94                 max_rating, best_hex = rating, { x, y }
  95             end
  96         end)
  97 
  98         if (best_hex[1] ~= hunter.x) or (best_hex[2] ~= hunter.y) then
  99             AH.checked_move(ai, hunter, best_hex[1], best_hex[2])  -- partial move only
 100             if (not hunter) or (not hunter.valid) then return end
 101         else  -- If hunter did not move, we need to stop it (also delete the goal)
 102             AH.checked_stopunit_moves(ai, hunter)
 103             if (not hunter) or (not hunter.valid) then return end
 104             hunter_vars.goal_x, hunter_vars.goal_y = nil, nil
 105             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 106         end
 107 
 108         -- If this gets the hunter to the goal, we delete the goal
 109         if (hunter.x == hunter_vars.goal_x) and (hunter.y == hunter_vars.goal_y) then
 110             hunter_vars.goal_x, hunter_vars.goal_y = nil, nil
 111             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 112         end
 113 
 114         -- Finally, if the hunter ended up next to enemies, attack the weakest of those
 115         local attack_status = hunter_attack_weakest_adj_enemy(ai, hunter)
 116 
 117         -- If the enemy was killed, hunter returns home
 118         if hunter and hunter.valid and (attack_status == 'killed') then
 119             hunter_vars.goal_x, hunter_vars.goal_y = nil, nil
 120             hunter_vars.hunting_status = 'returning'
 121             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 122 
 123             if cfg.show_messages then
 124                 W.message { speaker = hunter.id, message = 'Now that I have eaten, I will go back home.' }
 125             end
 126         end
 127 
 128         return
 129     end
 130 
 131     -- If we got here, this means the hunter is either returning, or resting
 132     if (hunter_vars.hunting_status == 'returning') then
 133         goto_x, goto_y = wesnoth.find_vacant_tile(cfg.home_x, cfg.home_y, hunter)
 134 
 135         local next_hop = AH.next_hop(hunter, goto_x, goto_y)
 136         if next_hop then
 137             AH.movefull_stopunit(ai, hunter, next_hop)
 138             if (not hunter) or (not hunter.valid) then return end
 139 
 140             -- If there's an enemy on the 'home' hex and we got right next to it, attack that enemy
 141             if (M.distance_between(cfg.home_x, cfg.home_y, hunter.x, hunter.y) == 1) then
 142                 local enemy = wesnoth.get_unit(cfg.home_x, cfg.home_y)
 143                 if AH.is_attackable_enemy(enemy) then
 144                     if cfg.show_messages then
 145                         W.message { speaker = hunter.id, message = 'Get out of my home!' }
 146                     end
 147 
 148                     AH.checked_attack(ai, hunter, enemy)
 149                     if (not hunter) or (not hunter.valid) then return end
 150                 end
 151             end
 152         end
 153 
 154         -- We also attack the weakest adjacent enemy, if still possible
 155         hunter_attack_weakest_adj_enemy(ai, hunter)
 156         if (not hunter) or (not hunter.valid) then return end
 157 
 158         -- If the hunter got home, start the resting counter
 159         if (hunter.x == cfg.home_x) and (hunter.y == cfg.home_y) then
 160             hunter_vars.hunting_status = 'resting'
 161             hunter_vars.resting_until = wesnoth.current.turn + (cfg.rest_turns or 3)
 162             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 163 
 164             if cfg.show_messages then
 165                 W.message { speaker = hunter.id, message = 'I made it home - resting now until the end of Turn ' .. hunter_vars.resting_until .. ' or until fully healed.' }
 166             end
 167         end
 168 
 169         return
 170     end
 171 
 172     -- If we got here, the only remaining action should be resting
 173     if (hunter_vars.hunting_status == 'resting') then
 174         AH.checked_stopunit_moves(ai, hunter)
 175         if (not hunter) or (not hunter.valid) then return end
 176 
 177         -- We also attack the weakest adjacent enemy
 178         hunter_attack_weakest_adj_enemy(ai, hunter)
 179         if (not hunter) or (not hunter.valid) then return end
 180 
 181         -- If this is the last turn of resting, we remove the status and turn variable
 182         if (hunter.hitpoints >= hunter.max_hitpoints) and (hunter_vars.resting_until <= wesnoth.current.turn) then
 183             hunter_vars.hunting_status = nil
 184             hunter_vars.resting_until = nil
 185             MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 186 
 187             if cfg.show_messages then
 188                 W.message { speaker = hunter.id, message = 'I am done resting. It is time to go hunting again next turn.' }
 189             end
 190         end
 191         return
 192     end
 193 
 194     -- In principle we should never get here, but just in case something got changed in WML:
 195     -- Reset variable, so that hunter goes hunting on next turn
 196     hunter_vars.hunting_status = nil
 197     MAIUV.set_mai_unit_variables(hunter, cfg.ai_id, hunter_vars)
 198 end
