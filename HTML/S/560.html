  70 int tree_view::remove_node(tree_view_node* node)
  71 {
  72         assert(node && node != root_node_ && node->parent_node_);
  73         const point node_size = node->get_size();
  74 
  75         tree_view_node::node_children_vector& siblings = node->parent_node_->children_;
  76 
  77         auto node_itor = std::find(siblings.begin(), siblings.end(), *node);
  78         assert(node_itor != siblings.end());
  79 
  80         const int position = node_itor - siblings.begin();
  81 
  82         siblings.erase(node_itor);
  83 
  84         if(get_size() != point()) {
  85                 // Don't shrink the width, need to think about a good algorithm to do so.
  86                 resize_content(0, -node_size.y);
  87         }
  88 
  89         return position;
  90 }

 124 void tree_view::resize_content(const int width_modification,
 125                                                                 const int height_modification,
 126                                                 const int width__modification_pos,
 127                                                 const int height_modification_pos)
 128 {
 129         DBG_GUI_L << LOG_HEADER << " current size " << content_grid()->get_size()
 130                           << " width_modification " << width_modification
 131                           << " height_modification " << height_modification << ".\n";
 132 
 133         if(content_resize_request(width_modification, height_modification, width__modification_pos, height_modification_pos)) {
 134 
 135                 // Calculate new size.
 136                 point size = content_grid()->get_size();
 137                 size.x += width_modification;
 138                 size.y += height_modification;
 139 
 140                 // Set new size.
 141                 content_grid()->set_size(size);
 142 
 143                 // Set status.
 144                 need_layout_ = true;
 145                 // If the content grows assume it "overwrites" the old content.
 146                 if(width_modification < 0 || height_modification < 0) {
 147                         set_is_dirty(true);
 148                 }
 149                 horizontal_scrollbar_moved();
 150                 DBG_GUI_L << LOG_HEADER << " succeeded.\n";
 151         } else {
 152                 DBG_GUI_L << LOG_HEADER << " failed.\n";
 153         }
 154 }

 156 void tree_view::layout_children(const bool force)
 157 {
 158         assert(root_node_ && content_grid());
 159 
 160         if(need_layout_ || force) {
 161                 root_node_->place(indentation_step_size_,
 162                                                   get_origin(),
 163                                                   content_grid()->get_size().x);
 164                 root_node_->set_visible_rectangle(content_visible_area_);
 165 
 166                 need_layout_ = false;
 167                 horizontal_scrollbar_moved();
 168         }
 169 }

 198 template<tree_view_node* (tree_view_node::*func) ()>
 199 tree_view_node* tree_view::get_next_node()
 200 {
 201         tree_view_node* selected = selected_item();
 202         if(!selected) {
 203                 return nullptr;
 204         }
 205         tree_view_node* visible = selected->get_last_visible_parent_node();
 206         if(visible != selected) {
 207                 return visible;
 208         }
 209         return (selected->*func)();
 210 }

 212 template<tree_view_node* (tree_view_node::*func) ()>
 213 bool tree_view::handle_up_down_arrow()
 214 {
 215         if(tree_view_node* next = get_next_node<func>())
 216         {
 217                 next->select_node();
 218                 SDL_Rect visible = content_visible_area();
 219                 SDL_Rect rect = next->get_grid().get_rectangle();
 220                 visible.y = rect.y;// - content_grid()->get_y();
 221                 visible.h = rect.h;
 222                 show_content_rect(visible);
 223                 return true;
 224         }
 225         return false;
 226 }

 249 void tree_view::handle_key_left_arrow(SDL_Keymod modifier, bool& handled)
 250 {
 251         tree_view_node* selected = selected_item();
 252         if(!selected || selected->is_folded()) {
 253                 scrollbar_container::handle_key_left_arrow(modifier, handled);
 254                 return;
 255         }
 256         selected->fold();
 257         handled = true;
 258 }

 260 void tree_view::handle_key_right_arrow(SDL_Keymod modifier, bool& handled)
 261 {
 262         tree_view_node* selected = selected_item();
 263         if(!selected || !selected->is_folded()) {
 264                 scrollbar_container::handle_key_right_arrow(modifier, handled);
 265                 return;

 267         selected->unfold();

 377 builder_tree_view::builder_tree_view(const config& cfg)
 378         : builder_styled_widget(cfg)
 379         , vertical_scrollbar_mode(
 380                           get_scrollbar_mode(cfg["vertical_scrollbar_mode"]))
 381         , horizontal_scrollbar_mode(
 382                           get_scrollbar_mode(cfg["horizontal_scrollbar_mode"]))
 383         , indentation_step_size(cfg["indentation_step_size"])
 384         , nodes()
 385 {
 386 
 387         for(const auto & node : cfg.child_range("node"))

 410         DBG_GUI_G << "Window builder: placed tree_view '" << id
