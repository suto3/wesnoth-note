 121         void warn_unknown_attribute(const config::const_attr_itors& cfg)
 122         {
 123                 config::const_attribute_iterator cur = cfg.begin();
 124                 config::const_attribute_iterator end = cfg.end();
 125 
 126                 const std::string* cur_known = std::begin(internalized_attrs);
 127                 const std::string* end_known = std::end(internalized_attrs);
 128 
 129                 while(cur_known != end_known) {
 130                         if(cur == end) {
 131                                 return;
 132                         }
 133                         int comp = cur->first.compare(*cur_known);
 134                         if(comp < 0) {
 135                                 WRN_UT << "Unknown attribute '" << cur->first << "' discarded." << std::endl;
 136                                 ++cur;
 137                         }
 138                         else if(comp == 0) {
 139                                 ++cur;
 140                                 ++cur_known;
 141                         }
 142                         else {
 143                                 ++cur_known;
 144                         }
 145                 }
 146 
 147                 while(cur != end) {
 148                         WRN_UT << "Unknown attribute '" << cur->first << "' discarded." << std::endl;
 149                         ++cur;
 150                 }
 151         }

 165 void intrusive_ptr_add_ref(const unit* u)
 166 {
 167         assert(u->ref_count_ >= 0);
 168         // the next code line is to notice possible wrongly initialized units.
 169         // The 100000 is picked rather randomly. If you are in the situation
 170         // that you can actually have more then 100000 intrusive_ptr to one unit
 171         // or if you are sure that the refcounting system works
 172         // then feel free to remove the next line
 173         assert(u->ref_count_ < 100000);
 174         LOG_UT << "Adding a reference to a unit: id = " << u->id() << ", uid = " << u->underlying_id() << ", refcount = " << u->ref_count() << " ptr:" << u << std::endl;
 175         if(u->ref_count_ == 0) {
 176                 LOG_UT << "Freshly constructed" << std::endl;
 177         }
 178         ++(u->ref_count_);
 179 }

 181 void intrusive_ptr_release(const unit* u)
 182 {
 183         assert(u->ref_count_ >= 1);
 184         assert(u->ref_count_ < 100000); //See comment in intrusive_ptr_add_ref
 185         LOG_UT << "Removing a reference to a unit: id = " << u->id() << ", uid = " << u->underlying_id() << ", refcount = " << u->ref_count() << " ptr:" << u << std::endl;
 186         if(--(u->ref_count_) == 0)
 187         {
 188                 LOG_UT << "Deleting a unit: id = " << u->id() << ", uid = " << u->underlying_id() << std::endl;
 189                 delete u;
 190         }

 197 static const unit_type& get_unit_type(const std::string& type_id)
 198 {
 199         if(type_id.empty()) {
 200                 throw unit_type::error("creating unit with an empty type field");
 201         }
 202         std::string new_id = type_id;
 203         unit_type::check_id(new_id);
 204         const unit_type* i = unit_types.find(new_id);
 205         if(!i) throw unit_type::error("unknown unit type: " + type_id);
 206         return *i;

 813 void unit::generate_traits(bool must_have_only)
 814 {
 815         LOG_UT << "Generating a trait for unit type " << type().log_id() << " with must_have_only " << must_have_only  << std::endl;
 816         const unit_type& u_type = type();
 817 
 818         // Calculate the unit's traits
 819         config::const_child_itors current_traits = modifications_.child_range("trait");
 820         std::vector<const config*> candidate_traits;
 821 
 822         for(const config& t : u_type.possible_traits()) {
 823                 // Skip the trait if the unit already has it.
 824                 const std::string& tid = t["id"];
 825                 bool already = false;
 826                 for(const config& mod : current_traits) {
 827                         if(mod["id"] == tid) {
 828                                 already = true;
 829                                 break;
 830                         }
 831                 }
 832 
 833                 if(already) {
 834                         continue;
 835                 }
 836 
 837                 // Add the trait if it is mandatory.
 838                 const std::string& avl = t["availability"];
 839                 if(avl == "musthave") {
 840                         modifications_.add_child("trait", t);
 841                         current_traits = modifications_.child_range("trait");
 842                         continue;
 843                 }
 844 
 845                 // The trait is still available, mark it as a candidate for randomizing.
 846                 // For leaders, only traits with availability "any" are considered.
 847                 if(!must_have_only && (!can_recruit() || avl == "any")) {
 848                         candidate_traits.push_back(&t);
 849                 }
 850         }
 851 
 852         if(must_have_only) return;
 853 
 854         // Now randomly fill out to the number of traits required or until
 855         // there aren't any more traits.
 856         int nb_traits = current_traits.size();
 857         int max_traits = u_type.num_traits();
 858         for(; nb_traits < max_traits && !candidate_traits.empty(); ++nb_traits)
 859         {
 860                 int num = randomness::generator->get_random_int(0,candidate_traits.size()-1);
 861                 modifications_.add_child("trait", *candidate_traits[num]);
 862                 candidate_traits.erase(candidate_traits.begin() + num);
 863         }
 864 
 865         // Once random traits are added, don't do it again.
 866         // Such as when restoring a saved character.
 867         random_traits_ = false;
 868 }

 884 /**
 885  * Advances this unit to the specified type.
 886  * Experience is left unchanged.
 887  * Current hit point total is left unchanged unless it would violate max HP.
 888  * Assumes gender_ and variation_ are set to their correct values.
 889  */
 890 void unit::advance_to(const unit_type& u_type, bool use_traits)
 891 {
 892         // For reference, the type before this advancement.
 893         const unit_type& old_type = type();
 894         // Adjust the new type for gender and variation.
 895         const unit_type& new_type = u_type.get_gender_unit_type(gender_).get_variation(variation_);
 896 
 897         // Reset the scalar values first
 898         trait_names_.clear();
 899         trait_descriptions_.clear(),
 900         is_fearless_ = false;
 901         is_healthy_ = false;
 902 
 903         // Clear modification-related caches
 904         modification_descriptions_.clear();
 905 
 906         // build unit type ready to create units. Not sure if needed.
 907         new_type.get_cfg_for_units();
 908 
 909         if(!new_type.usage().empty()) {
 910                  set_usage(new_type.usage());
 911         }
 912 
 913         set_image_halo(new_type.halo());
 914         if(!new_type.ellipse().empty()) {
 915                 set_image_ellipse(new_type.ellipse());
 916         }
 917 
 918         generate_name_ &= new_type.generate_name();
 919         abilities_ = new_type.abilities_cfg();
 920         advancements_.clear();
 921 
 922         for(const config& advancement : new_type.advancements()) {
 923                 advancements_.push_back(new config(advancement));
 924         }
 925 
 926         // If unit has specific profile, remember it and keep it after advancing
 927         if(small_profile_.empty() || small_profile_ == old_type.small_profile()) {
 928                 small_profile_ = new_type.small_profile();
 929         }
 930 
 931         if(profile_.empty() || profile_ == old_type.big_profile()) {
 932                 profile_ = new_type.big_profile();
 933         }
 934         // NOTE: There should be no need to access old_cfg (or new_cfg) after this
 935         //       line. Particularly since the swap might have affected old_cfg.
 936 
 937         advances_to_ = new_type.advances_to();
 938 
 939         race_ = new_type.race();
 940         type_ = &new_type;
 941         type_name_ = new_type.type_name();
 942         description_ = new_type.unit_description();
 943         undead_variation_ = new_type.undead_variation();
 944         max_experience_ = new_type.experience_needed(false);
 945         level_ = new_type.level();
 946         recall_cost_ = new_type.recall_cost();
 947 
 948         /* Need to add a check to see if the unit's old cost is equal
 949         to the unit's old unit_type cost first.  If it is change the cost
 950         otherwise keep the old cost. */
 951         if(old_type.recall_cost() == recall_cost_) {
 952                 recall_cost_ = new_type.recall_cost();
 953         }
 954 
 955         alignment_ = new_type.alignment();
 956         max_hit_points_ = new_type.hitpoints();
 957         hp_bar_scaling_ = new_type.hp_bar_scaling();
 958         xp_bar_scaling_ = new_type.xp_bar_scaling();
 959         max_movement_ = new_type.movement();
 960         vision_ = new_type.vision(true);
 961         jamming_ = new_type.jamming();
 962         movement_type_ = new_type.movement_type();
 963         emit_zoc_ = new_type.has_zoc();
 964         attacks_.clear();
 965         std::transform(new_type.attacks().begin(), new_type.attacks().end(), std::back_inserter(attacks_), [](const attack_type& atk) {
 966                 return std::make_shared<attack_type>(atk);
 967         });
 968         unit_value_ = new_type.cost();
 969 
 970         max_attacks_ = new_type.max_attacks();
 971 
 972         flag_rgb_ = new_type.flag_rgb();
 973 
 974         anim_comp_->reset_after_advance(&new_type);
 975 
 976         if(random_traits_) {
 977                 generate_traits(!use_traits);
 978         } else {
 979                 // This will add any "musthave" traits to the new unit that it doesn't already have.
 980                 // This covers the Dark Sorcerer advancing to Lich and gaining the "undead" trait,
 981                 // but random and/or optional traits are not added,
 982                 // and neither are inappropriate traits removed.
 983                 generate_traits(true);
 984         }
 985 
 986         // Apply modifications etc, refresh the unit.
 987         // This needs to be after type and gender are fixed,
 988         // since there can be filters on the modifications
 989         // that may result in different effects after the advancement.
 990         apply_modifications();
 991 
 992         // Now that modifications are done modifying traits, check if poison should
 993         // be cleared.
 994         if(get_state("unpoisonable")) {
 995                 set_state(STATE_POISONED, false);
 996         }
 997 
 998         // Now that modifications are done modifying the maximum hit points,
 999         // enforce this maximum.
1000         if(hit_points_ > max_hit_points_) {
1001                 hit_points_ = max_hit_points_;
1002         }
1003 
1004         // In case the unit carries EventWML, apply it now
1005         if(resources::game_events) {
1006                 resources::game_events->add_events(new_type.events(), new_type.id());
1007         }
1008 }

1182 void unit::expire_modifications(const std::string& duration)
1183 {
1184         // If any modifications expire, then we will need to rebuild the unit.
1185         const unit_type* rebuild_from = nullptr;
1186 
1187         // Loop through all types of modifications.
1188         for(const auto& mod_name : ModificationTypes) {
1189                 // Loop through all modifications of this type.
1190                 // Looping in reverse since we may delete the current modification.
1191                 for(int j = modifications_.child_count(mod_name)-1; j >= 0; --j)
1192                 {
1193                         const config& mod = modifications_.child(mod_name, j);
1194 
1195                         if(mod_duration_match(mod["duration"], duration)) {
1196                                 // If removing this mod means reverting the unit's type:
1197                                 if(const config::attribute_value* v = mod.get("prev_type")) {
1198                                         rebuild_from = &get_unit_type(v->str());
1199                                 }
