 100     /**
 101      * Parses the schema
 102      * 
 103      * @param force
 104      *        True to force parsing the schema, skipping the existing cache
 105      * @param schemaPath
 106      *        The path to the 'schema.cfg' file
 107      */
 108     public void parseSchemaFile( boolean force, String schemaPath )
 109     {
 110         if( parsingDone_ && ! force ) {
 111             Logger.getInstance( ).log( "schema not parsed since there is already in cache." ); //$NON-NLS-1$
 112             return;
 113         }
 114 
 115         parsingDone_ = false;
 116         if( force ) {
 117             primitives_.clear( );
 118             tags_.clear( );
 119         }
 120 
 121         Logger.getInstance( ).log(
 122             "parsing schema " + ( force == true ? "forced": "" ) ); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
 123         File schemaFile = new File( schemaPath );
 124         String res = ResourceUtils.getFileContents( schemaFile );
 125         String[] lines = StringUtils.getLines( res );
 126         Stack< String > tagStack = new Stack< String >( );
 127 
 128         WMLTag currentTag = null;
 129         for( int index = 0; index < lines.length; index++ ) {
 130             String line = lines[index];
 131             // skip comments and empty lines
 132             if( StringUtils.startsWith( line, "#" )
 133                 || line.matches( "^[\t ]*$" ) ) {
 134                 continue;
 135             }
 136 
 137             if( StringUtils.startsWith( line, "[" ) ) //$NON-NLS-1$
 138             {
 139                 // closing tag
 140                 if( line.charAt( line.indexOf( "[" ) + 1 ) == '/' ) //$NON-NLS-1$
 141                 {
 142                     // propagate the 'needsexpanding' property to upper levels
 143                     boolean expand = false;
 144                     if( ! tagStack.isEmpty( )
 145                         && tags_.containsKey( tagStack.peek( ) ) ) {
 146                         expand = tags_.get( tagStack.peek( ) )
 147                             .is_NeedingExpansion( );
 148                     }
 149 
 150                     tagStack.pop( );
 151 
 152                     if( ! tagStack.isEmpty( )
 153                         && tags_.containsKey( tagStack.peek( ) )
 154                         && expand == true ) {
 155                         tags_.get( tagStack.peek( ) ).set_NeedingExpansion(
 156                             expand );
 157                     }
 158                 }
 159                 // opening tag
 160                 else {
 161                     String tagName = line.substring(
 162                         line.indexOf( "[" ) + 1, line.indexOf( "]" ) ); //$NON-NLS-1$ //$NON-NLS-2$
 163                     String simpleTagName = tagName;
 164                     String extendedTagName = ""; //$NON-NLS-1$
 165                     if( tagName.split( ":" ).length > 1 ) //$NON-NLS-1$
 166                     {
 167                         simpleTagName = tagName.split( ":" )[0]; //$NON-NLS-1$
 168                         extendedTagName = tagName.split( ":" )[1]; //$NON-NLS-1$
 169                     }
 170                     tagStack.push( simpleTagName );
 171 
 172                     if( ! tagName.equals( "description" ) ) //$NON-NLS-1$
 173                     {
 174                         if( tags_.containsKey( simpleTagName ) ) {
 175                             // this tags was already refered in the schema
 176                             // before they were declared
 177                             currentTag = tags_.get( simpleTagName );
 178                             currentTag.set_InhertedTagName( extendedTagName );
 179                             currentTag.set_NeedingExpansion( ! extendedTagName
 180                                 .isEmpty( ) );
 181                         }
 182                         else {
 183                             WMLTag tag = WmlFactory2.eINSTANCE.createWMLTag(
 184                                 simpleTagName, extendedTagName );
 185                             tag.set_NeedingExpansion( ! extendedTagName
 186                                 .isEmpty( ) );
 187                             currentTag = tag;
 188                             tags_.put( simpleTagName, tag );
 189                         }
 190                     }
 191                 }
 192             }
 193             else {
 194                 // top level - primitives defined
 195                 if( tagStack.peek( ).equals( "schema" ) ) //$NON-NLS-1$
 196                 {
 197                     String[] tokens = line.split( "=" ); //$NON-NLS-1$
 198                     if( tokens.length != 2 ) {
 199                         Logger.getInstance( ).logError(
 200                             "Error. invalid primitive on line :" + index ); //$NON-NLS-1$
 201                         continue;
 202                     }
 203 
 204                     primitives_.put( tokens[0].trim( ),
 205                         StringUtils.trimQuotes( tokens[1].trim( ) ) );
 206                 }
 207                 else if( tagStack.peek( ).equals( "description" ) ) //$NON-NLS-1$
 208                 {
 209                     String[] tokens = line.trim( ).split( "=" ); //$NON-NLS-1$
 210                     StringBuilder value = new StringBuilder( );
 211 
 212                     // this *should* happen only in [description]
 213                     // multi-line string
 214                     if( StringUtils.countOf( tokens[1], '"' ) % 2 != 0 ) {
 215                         value.append( tokens[1] + "\n" ); //$NON-NLS-1$
 216                         ++index;
 217                         while( StringUtils.countOf( lines[index], '"' ) % 2 == 0
 218                             && ! StringUtils.startsWith( lines[index], "#" ) && //$NON-NLS-1$
 219                             index < lines.length ) {
 220                             value.append( lines[index] + "\n" ); //$NON-NLS-1$
 221                             ++index;
 222                         }
 223                         value.append( lines[index] );
 224                     }
 225                     else {
 226                         value.append( tokens[1] );
 227                     }
 228 
 229                     // get rid of the quotes
 230                     if( value.length( ) >= 2 ) {
 231                         value = new StringBuilder( value.substring( 1,
 232                             value.length( ) - 1 ) );
 233                     }
 234 
 235                     if( currentTag != null ) {
 236                         currentTag.set_Description( value.toString( ) );
 237                     }
 238                     else {
 239                         System.out.println( "Tag shouldn't have been null!" ); //$NON-NLS-1$
 240                     }
 241                 }
 242                 else {
 243                     String tmpLine = line.trim( );
 244                     if( line.contains( "#" ) ) {
 245                         tmpLine = line
 246                             .substring( 0, line.lastIndexOf( "#" ) ).trim( ); //$NON-NLS-1$
 247                     }
 248                     String[] tokens = tmpLine.split( "=" ); //$NON-NLS-1$
 249 
 250                     if( tokens.length != 2 ) {
 251                         Logger.getInstance( ).logError(
 252                             "Error. invalid attribute on line :" + index ); //$NON-NLS-1$
 253                         continue;
 254                     }
 255 
 256                     String[] value = tokens[1].substring( 1,
 257                         tokens[1].length( ) - 1 ).split( " " ); //$NON-NLS-1$
 258                     if( value.length != 2 ) {
 259                         Logger
 260                             .getInstance( )
 261                             .logError(
 262                                 "Error. invalid attribute value on line:" + index ); //$NON-NLS-1$
 263                         continue;
 264                     }
 265 
 266                     if( currentTag != null ) {
 267                         if( tokens[0].startsWith( "_" ) ) // reference to another tag //$NON-NLS-1$
 268                         {
 269                             WMLTag targetTag = null;
 270                             targetTag = tags_.get( value[1] );
 271 
 272                             // tag wasn't created yet
 273                             if( targetTag == null ) {
 274                                 targetTag = WmlFactory2.eINSTANCE
 275                                     .createWMLTag( value[1], "",
 276                                         getCardinality( value[0] ) );
 277                                 tags_.put( value[1], targetTag );
 278                             }
 279 
 280                             currentTag.getExpressions( ).add( targetTag );
 281                         }
 282                         else {
 283                             if( primitives_.get( value[1] ) == null ) {
 284                                 Logger
 285                                     .getInstance( )
 286                                     .logError(
 287                                         "Undefined primitive type in schema.cfg for: " + value[1] ); //$NON-NLS-1$
 288                             }
 289 
 290                             currentTag.getExpressions( ).add(
 291                                 WmlFactory2.eINSTANCE.createWMLKey(
 292                                     tokens[0],
 293                                     primitives_.get( value[1] ),
 294                                     getCardinality( value[0] ),
 295                                     value[1].equals( "tstring" ) ) );
 296                         }
 297                     }
 298                     else {
 299                         // System.err.println("can't find entry for: " +
 300                         // tagStack.peek());
 301                     }
 302                 }
 303             }
 304         }
 305 
 306         for( WMLTag tag: tags_.values( ) ) {
 307             sortChildren( tag );
 308         }
 309 
 310         for( WMLTag tag: tags_.values( ) ) {
 311             expandTag( tag );
 312         }
 313 
 314         Logger.getInstance( ).log( "parsing done" ); //$NON-NLS-1$
 315         parsingDone_ = true;
 316     }

 318     /**
 319      * Expands the tags that need to (the ones based on inheritance)
 320      */
 321     private void expandTag( WMLTag tag )
 322     {
 323         if( tag.is_NeedingExpansion( ) ) {
 324             tag.set_NeedingExpansion( false );
 325 
 326             for( WMLTag subTag: tag.getWMLTags( ) ) {
 327                 expandTag( subTag );
 328             }
 329 
 330             WMLTag inhertedTag = tags_.get( tag.get_InhertedTagName( ) );
 331             if( inhertedTag != null ) {
 332                 tag.getExpressions( ).addAll( inhertedTag.getExpressions( ) );
 333             }
 334         }
 335     }

 337     /**
 338      * Sorts all tag's children by using the cardinality comparator
 339      * 
 340      * @param tag
 341      *        The tag to whom to sort the children
 342      */
 343     private void sortChildren( WMLTag tag )
 344     {
 345         WMLExpression[] expressions = tag.getExpressions( )
 346             .toArray( new WMLExpression[tag.getExpressions( ).size( )] );
 347         Arrays.sort( expressions, new CardinalityComparator( ) );
 348         tag.getExpressions( ).clear( );
 349 
 350         for( WMLExpression expression: expressions ) {
 351             tag.getExpressions( ).add( expression );
 352         }
 353 
 354         for( WMLTag subTag: tag.getWMLTags( ) ) {
 355             sortChildren( subTag );
 356         }
 357     }

 369     /**
 370      * Returns the cardinality as a character of the specified value
 371      * required = 1
 372      * optional = ?
 373      * repeated = *
 374      * forbidden = -
 375      * 
 376      * @param value
 377      *        The value the parse
 378      * @return The char representation of the cardinality
 379      */
 380     public char getCardinality( String value )
 381     {
 382         if( value.equals( "required" ) ) {
 383             return '1';
 384         }
 385         else if( value.equals( "optional" ) ) {
 386             return '?';
 387         }
 388         else if( value.equals( "repeated" ) ) {
 389             return '*';
 390         }
 391         else if( value.equals( "forbidden" ) ) {
 392             return '-';
 393         }
 394         return 'a';
 395     }

 397     /**
 398      * A WML Expression comparator that sorts just after required cardinality.
 399      * That is, after the sort the required wmlexpressions will be first
 400      */
 401     public static class CardinalityComparator implements
 402         Comparator< WMLExpression >, Serializable
 403     {
 404         private static final long serialVersionUID = 6103884038547449868L;
 405 
 406         @Override
 407         public int compare( WMLExpression o1, WMLExpression o2 )
 408         {
 409             char o1_card = o1.get_Cardinality( );
 410             char o2_card = o2.get_Cardinality( );
 411 
 412             if( o1_card == '1' && o2_card != '1' ) {
 413                 return - 1;
 414             }
 415             else if( o2_card == '1' && o1_card != '1' ) {
 416                 return 1;
 417             }
 418 
 419             return 0;
 420         }
 421     }
