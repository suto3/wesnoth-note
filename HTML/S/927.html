  59 void save_index_class::rebuild(const std::string& name, const time_t& modified) {
  60         log_scope("load_summary_from_file");
  61         config& summary = data(name);
  62         try {
  63                 config full;
  64                 std::string dummy;
  65                 read_save_file(name, full, &dummy);
  66                 extract_summary_from_config(full, summary);
  67         } catch(game::load_game_failed&) {
  68                 summary["corrupt"] = true;
  69                 }
  70         summary["mod_time"] = std::to_string(static_cast<int>(modified));
  71         write_save_index();
  72 }

  84 config& save_index_class::get(const std::string& name) {
  85         config& result = data(name);
  86         time_t m = modified_[name];
  87         config::attribute_value& mod_time = result["mod_time"];
  88         if (mod_time.empty() || static_cast<time_t>(mod_time.to_int()) != m) {
  89                 rebuild(name, m);
  90         }
  91         return result;
  92 }

  94 void save_index_class::write_save_index() {
  95         log_scope("write_save_index()");
  96         try {
  97                 filesystem::scoped_ostream stream = filesystem::ostream_file(filesystem::get_save_index_file());
  98                 if (preferences::save_compression_format() != compression::NONE) {
  99                         // TODO: maybe allow writing this using bz2 too?

 105                 ERR_SAVE << "error writing to save index file: '" << e.what() << "'" << std::endl;

 126 config& save_index_class::data() {
 127         if(loaded_ == false) {
 128                 try {
 129                         filesystem::scoped_istream stream = filesystem::istream_file(filesystem::get_save_index_file());
 130                         try {
 131                                 read_gz(data_, *stream);
 132                         } catch (boost::iostreams::gzip_error&) {
 133                                 stream->seekg(0);
 134                                 read(data_, *stream);
 135                         }
 136                 } catch(filesystem::io_exception& e) {
 137                         ERR_SAVE << "error reading save index: '" << e.what() << "'" << std::endl;
 138                 } catch(config::error& e) {
 139                         ERR_SAVE << "error parsing save index config file:\n" << e.message << std::endl;
 140                         data_.clear();
 141                 }
 142                 loaded_ = true;
 143         }
 144         return data_;
 145 }

 149 class filename_filter {

 151         filename_filter(const std::string& filter) : filter_(filter) {
 152         }
 153         bool operator()(const std::string& filename) const {
 154                 return filename.end() == std::search(filename.begin(), filename.end(),

 161 /** Get a list of available saves. */
 162 std::vector<save_info> get_saves_list(const std::string* dir, const std::string* filter)
 163 {
 164         create_save_info creator(dir);
 165 
 166         std::vector<std::string> filenames;
 167         filesystem::get_files_in_dir(creator.dir,&filenames);
 168 
 169         if (filter) {
 170 
 171                 // Replace the spaces in the filter
 172                 std::string filter_replaced(filter->begin(), filter->end());
 173                 replace_space2underbar(filter_replaced);
 174 
 175                 filenames.erase(std::remove_if(filenames.begin(), filenames.end(),
 176                                                filename_filter(filter_replaced)),
 177                                 filenames.end());
 178         }
 179 
 180         std::vector<save_info> result;
 181         std::transform(filenames.begin(), filenames.end(),
 182                        std::back_inserter(result), creator);
 183         std::sort(result.begin(),result.end(),save_info_less_time());
 184         return result;
 185 }

 192 std::string save_info::format_time_local() const

 194         if(tm* tm_l = localtime(&modified())) {
 195                 const std::string format = preferences::use_twelve_hour_clock_format() ? _("%a %b %d %I:%M %p %Y") : _("%a %b %d %H:%M %Y");

 209 bool save_info_less_time::operator() (const save_info& a, const save_info& b) const {
 210         if (a.modified() > b.modified()) {
 211                 return true;
 212         } else if (a.modified() < b.modified()) {
 213                 return false;
 214                 // Special funky case; for files created in the same second,
 215                 // a replay file sorts less than a non-replay file.  Prevents
 216                 // a timing-dependent bug where it may look like, at the end
 217                 // of a scenario, the replay and the autosave for the next
 218                 // scenario are displayed in the wrong order.
 219         } else if (a.name().find(_(" replay"))==std::string::npos && b.name().find(_(" replay"))!=std::string::npos) {
 220                 return true;
 221         } else if (a.name().find(_(" replay"))!=std::string::npos && b.name().find(_(" replay"))==std::string::npos) {
 222                 return false;
 223         } else {
 224                 return  a.name() > b.name();
 225         }
 226 }

 228 static filesystem::scoped_istream find_save_file(const std::string &name, const std::string &alt_name, const std::vector<std::string> &suffixes) {
 229         for (const std::string &suf : suffixes) {
 230                 filesystem::scoped_istream file_stream = filesystem::istream_file(filesystem::get_saves_dir() + "/" + name + suf);
 231                 if (file_stream->fail()) {
 232                         file_stream = filesystem::istream_file(filesystem::get_saves_dir() + "/" + alt_name + suf);
 233                 }
 234                 if (!file_stream->fail()) {
 235                         return file_stream;
 236                 }
 237         }
 238         LOG_SAVE << "Could not open supplied filename '" << name << "'\n";
 239         throw game::load_game_failed();
 240 }

 242 void read_save_file(const std::string& name, config& cfg, std::string* error_log)
 243 {
 244         std::string modified_name = name;
 245         replace_space2underbar(modified_name);
 246 
 247         static const std::vector<std::string> suffixes {"", ".gz", ".bz2"};
 248         filesystem::scoped_istream file_stream = find_save_file(modified_name, name, suffixes);
 249 
 250         cfg.clear();
 251         try{
 252                 /*
 253                  * Test the modified name, since it might use a .gz
 254                  * file even when not requested.
 255                  */
 256                 if(filesystem::is_gzip_file(modified_name)) {
 257                         read_gz(cfg, *file_stream);
 258                 } else if(filesystem::is_bzip2_file(modified_name)) {
 259                         read_bz2(cfg, *file_stream);
 260                 } else {
 261                         read(cfg, *file_stream);
 262                 }
 263         } catch(const std::ios_base::failure& e) {
 264                 LOG_SAVE << e.what();
 265                 if(error_log) {
 266                         *error_log += e.what();
 267                 }
 268                 throw game::load_game_failed();
 269         } catch(const config::error &err) {
 270                 LOG_SAVE << err.message;
 271                 if(error_log) {
 272                         *error_log += err.message;
 273                 }
 274                 throw game::load_game_failed();
 275         }
 276 
 277         if(cfg.empty()) {
 278                 LOG_SAVE << "Could not parse file data into config\n";
 279                 throw game::load_game_failed();
 280         }
 281 }

 283 void remove_old_auto_saves(const int autosavemax, const int infinite_auto_saves)
 284 {
 285         const std::string auto_save = _("Auto-Save");
 286         int countdown = autosavemax;
 287         if (countdown == infinite_auto_saves)
 288                 return;
 289 
 290         std::vector<save_info> games = get_saves_list(nullptr, &auto_save);
 291         for (std::vector<save_info>::iterator i = games.begin(); i != games.end(); ++i) {
 292                 if (countdown-- <= 0) {
 293                         LOG_SAVE << "Deleting savegame '" << i->name() << "'\n";
 294                         delete_game(i->name());
 295                 }
 296         }
 297 }

 326 void extract_summary_from_config(config& cfg_save, config& cfg_summary)
 327 {
 328         const config& cfg_snapshot = cfg_save.child("snapshot");
 329         //Servergenerated replays contain [scenario] and no [replay_start]
 330         const config& cfg_replay_start = cfg_save.child("replay_start") ? cfg_save.child("replay_start") : cfg_save.child("scenario") ;
 331 
 332         const config& cfg_replay = cfg_save.child("replay");
 333         const bool has_replay = cfg_replay && !cfg_replay.empty();
 334         const bool has_snapshot = cfg_snapshot && cfg_snapshot.has_child("side");
 335 
 336         cfg_summary["replay"] = has_replay;
 337         cfg_summary["snapshot"] = has_snapshot;
 338 
 339         cfg_summary["label"] = cfg_save["label"];
 340         cfg_summary["campaign_type"] = cfg_save["campaign_type"];
 341 
 342         if(cfg_save.has_child("carryover_sides_start")){
 343                 cfg_summary["scenario"] = cfg_save.child("carryover_sides_start")["next_scenario"];
 344         } else {
 345                 cfg_summary["scenario"] = cfg_save["scenario"];
 346         }
 347 
 348         cfg_summary["difficulty"] = cfg_save["difficulty"];
 349         cfg_summary["random_mode"] = cfg_save["random_mode"];
 350 
 351         cfg_summary["campaign"] = cfg_save["campaign"];
 352         cfg_summary["version"] = cfg_save["version"];
 353         cfg_summary["corrupt"] = "";
 354 
 355         if(has_snapshot) {
 356                 cfg_summary["turn"] = cfg_snapshot["turn_at"];
 357                 if(cfg_snapshot["turns"] != "-1") {
 358                         cfg_summary["turn"] = cfg_summary["turn"].str() + "/" + cfg_snapshot["turns"].str();
 359                 }
 360         }
 361 
 362         // Ensure we don't get duplicate [leader] tags
 363         cfg_summary.clear_children("leader");
 364 
 365         // Find the human leaders so we can display their icons and names in the load menu.
 366         config leader_config;
 367 
 368         bool shrouded = false;
 369 
 370         if(const config& snapshot = *(has_snapshot ? &cfg_snapshot : &cfg_replay_start)) {
 371                 for(const config &side : snapshot.child_range("side")) {
 372                         std::string leader;
 373                         std::string leader_image;
 374                         std::string leader_image_tc_modifier;
 375                         std::string leader_name;
 376                         int gold = side["gold"];
 377                         int units = 0, recall_units = 0;
 378 
 379                         if(side["controller"] != team::CONTROLLER::enum_to_string(team::CONTROLLER::HUMAN)) {
 380                                 continue;
 381                         }
 382 
 383                         if(side["shroud"].to_bool()) {
 384                                 shrouded = true;
 385                         }
 386 
 387                         for(const config &u : side.child_range("unit")) {
 388                                 if(u.has_attribute("x") && u.has_attribute("y")) {
 389                                         units++;
 390                                 } else {
 391                                         recall_units++;
 392                                 }
 393 
 394                                 // Only take the first leader
 395                                 if(!leader.empty() || !u["canrecruit"].to_bool()) {
 396                                         continue;
 397                                 }
 398 
 399                                 // Don't count it among the troops
 400                                 units--;
 401                                 leader = u["id"].str();
 402                                 leader_name = u["name"].str();
 403                                 leader_image = u["image"].str();
 404                                 leader_image_tc_modifier = "~RC(" + u["flag_rgb"].str() + ">" + u["side"].str() + ")";
 405                         }
 406 
 407                         // We need a binary path-independent path to the leader image here so it can be displayed
 408                         // for campaign-specific units even when the campaign isn't loaded yet.
 409                         std::string leader_image_path = filesystem::get_independent_image_path(leader_image);
 410 
 411                         // If the image path was found, we append the leader TC modifier. If it's not (such as in
 412                         // the case where the binary path hasn't been loaded yet, perhaps due to save_index being
 413                         // deleted), the unaltered image path is used and will be parsed by get_independent_image_path
 414                         // at runtime.
 415                         if(!leader_image_path.empty()) {
 416                                 leader_image_path += leader_image_tc_modifier;
 417 
 418                                 leader_image = leader_image_path;
 419                         }
 420 
 421                         leader_config["leader"] = leader;
 422                         leader_config["leader_name"] = leader_name;
 423                         leader_config["leader_image"] = leader_image;
 424                         leader_config["leader_image_tc_modifier"] = leader_image_tc_modifier;
 425                         leader_config["gold"] = gold;
 426                         leader_config["units"] = units;
 427                         leader_config["recall_units"] = recall_units;
 428 
 429                         cfg_summary.add_child("leader", leader_config);
 430                 }
 431         }
 432 
 433         if(!shrouded) {
 434                 if(has_snapshot) {
 435                         if (!cfg_snapshot.find_child("side", "shroud", "yes") && cfg_snapshot.has_attribute("map_data")) {
 436                                 cfg_summary["map_data"] = cfg_snapshot["map_data"].str();
 437                         } else {
 438                                 ERR_SAVE << "Not saving map because there is shroud" << std::endl;
 439                         }
 440                 } else if(has_replay) {
 441                         if (!cfg_replay_start.find_child("side","shroud","yes") && cfg_replay_start.has_attribute("map_data")) {
 442                                 cfg_summary["map_data"] = cfg_replay_start["map_data"];
 443                         } else {
 444                                 ERR_SAVE << "Not saving map because there is shroud" << std::endl;
 445                         }
 446                 }
 447         }
 448 }
