  58         /**
  59          * Utility to standardize the event names used in by_name_.
  60          * This means stripping leading and trailing spaces, and converting internal
  61          * spaces to underscores.
  62          */
  63         std::string event_handlers::standardize_name(const std::string & name)
  64         {
  65                 std::string retval = name;
  66 
  67                 // Trim leading and trailing spaces.
  68                 boost::trim(retval);
  69 
  70                 // Replace internal spaces with underscores.
  71                 boost::replace_all(retval, " ", "_");
  72 
  73                 return retval;
  74         }

  76         /**
  77          * Read-only access to the handlers with fixed event names, by event name.
  78          */
  79         const handler_list & event_handlers::get(const std::string & name) const
  80         {
  81                 // Empty list for the "not found" case.
  82                 static const handler_list empty_list;
  83 
  84                 // Look for the name in the name map.
  85                 map_t::const_iterator find_it = by_name_.find(standardize_name(name));
  86                 return find_it == by_name_.end() ? empty_list : find_it->second;
  87         }

  89         /**
  90          * Adds an event handler.
  91          * An event with a nonempty ID will not be added if an event with that
  92          * ID already exists.
  93          */
  94         void event_handlers::add_event_handler(const config & cfg, manager & man, bool is_menu_item)
  95         {
  96                 const std::string name = cfg["name"];
  97                 std::string id = cfg["id"];
  98 
  99                 if(!id.empty()) {
 100                         // Ignore this handler if there is already one with this ID.
 101                         id_map_t::iterator find_it = id_map_.find(id);
 102                         if ( find_it != id_map_.end()  &&  !find_it->second.expired() ) {
 103                                 DBG_EH << "ignoring event handler for name='" << name
 104                                        << "' with id '" << id << "'\n";
 105                                 return;
 106                         }
 107                 }
 108 
 109                 // Create a new handler.
 110                 DBG_EH << "inserting event handler for name=" << name <<
 111                         " with id=" << id << "\n";
 112                 handler_ptr new_handler(new event_handler(cfg, is_menu_item, active_.size(), man));
 113                 active_.push_back(new_handler);
 114 
 115                 // File by name.
 116                 if ( utils::might_contain_variables(name) )
 117                         dynamic_.push_back(new_handler);
 118                 else {
 119                         std::vector<std::string> name_list = utils::split(name);
 120                         for (const std::string & single_name : name_list)
 121                                 by_name_[standardize_name(single_name)].push_back(new_handler);
 122                 }
 123                 // File by ID.
 124                 if ( !id.empty() )
 125                         id_map_[id] = new_handler;
 126 
 127                 log_handlers();
 128         }

 130         /**
 131          * Removes an event handler, identified by its ID.
 132          * Events with empty IDs cannot be removed.
 133          */
 134         void event_handlers::remove_event_handler(const std::string& id)
 135         {
 136                 if ( id.empty() )
 137                         return;
 138 
 139                 DBG_EH << "removing event handler with id " << id << "\n";
 140 
 141                 // Find the existing handler with this ID.
 142                 id_map_t::iterator find_it = id_map_.find(id);
 143                 if ( find_it != id_map_.end() ) {
 144                         handler_ptr handler = find_it->second.lock();
 145                         // Remove handler.
 146                         if ( handler )
 147                                 handler->disable();
 148                         id_map_.erase(find_it); // Do this even if the lock failed.
 149                         // The index by name will self-adjust later. No need to adjust it now.
 150                 }
 151 
 152                 log_handlers();
 153         }

 156                 id_map_t::iterator find_it = id_map_.find(id);
 157                 if ( find_it != id_map_.end()  &&  !find_it->second.expired() ) {
 158                         return handler_ptr( find_it->second );
