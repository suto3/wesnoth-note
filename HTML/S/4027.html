  45 """""""""""""
  46 " Litterals "
  47 """""""""""""
  48 " A number may be decimal and negative (no space allowed between -, . and \d)
  49 " A string may start and end with a " or with << and >>
  50 " If a string is between double quote, the special sequences "" and $| may be inserted in it
  51 " The boolean constants are yes, no, true and false
  52 syn match wesmlBoolean display /\<\(yes\|no\|true\|false\)\>/ contained
  53 syn match wesmlNumber "-\?\d\+\(\.\d\+\)\?" contained
  54 syn match wesmlQStringEsc display +""+ contained
  55 syn region wesmlQString display start=/"\ze[^"]/ skip=/""/ end=/"/ contains=wesmlQStringEsc,@Spell,wesmlLuaQ,wesmlStringSpecial,wesmlStringVariable,wesmlFormula contained
  56 syn region wesmlCString display start=/<<\@<=/ end=/>>/ contains=@Spell,wesmlLuaC,wesmlStringSpecial,wesmlStringVariable,wesmlFormula contained
  57 syn match wesmlUnderscore display /\<_\>\(\s*\("\|<<\)\)\@=/ contained
  58 syn match wesmlEString display +""+ contained
  59 syn match wesmlStringSpecial display +$|+ contained

  70 """"""""""""""""
  71 " Preprocessor "
  72 """"""""""""""""
  73 " Preprocessor directives start with a # except the include directive {}
  74 syn match wesmlPPDArgument display /\s\+.\+$/ contained contains=wesmlComment
  75 syn match wesmlPPDIfverArgument display /\s\+.\+$/ contained contains=wesmlPPDOperator,wesmlVerNumber,wesmlComment
  76 syn match wesmlPPDOperator display "\(==\|!=\|<=\?\|>=\?\)" contained
  77 syn match wesmlVerNumber display "\d\+\(\.\d\+\)*\(-svn\)\?" contained
  78 syn match wesmlPPDNullary display /^\s*\zs#\s*enddef\>/
  79 syn match wesmlPPDNullaryTest display /^\s*\zs#\s*\(else\|endif\)\>/
  80 syn match wesmlPPDNary display /^\s*\zs#\s*\(define\|undef\|textdomain\|line\)\>\s\+\S/ contains=wesmlPPDArgument
  81 syn match wesmlPPDNaryTest display /^\s*\zs#\s*\(ifdef\|ifndef\|ifhave\|ifnhave\)\>\s\+\S/ contains=wesmlPPDArgument
  82 syn match wesmlPPDIfver display /^\s*\zs#\s*\(ifver\|ifnver\)\>\s\+\S/ contains=wesmlPPDIfverArgument
  83 syn match wesmlIncludeMark display +{\@<=\(\~\|\./\|/\)\?+ contained
  84 syn region wesmlInclude display start=/{/ end=/}/ contains=wesmlMacroName,wesmlNumber,wesmlVariable,wesmlIdentifier,wesmlCString,wesmlQString,wesmlEString,wesmlBoolean,wesmlFormula
  85 syn match wesmlMacroName display /{\@<=[^ }]\+/ contained contains=wesmlIncludeMark

 100 " Match a line: key,...=value,...
 101 syn match wesmlAttribute display /^\s*[^{#]\+=[^#]*/ contains=wesmlKey,wesmlEqual,wesmlValue

 110 """"""""""""
 111 " Variable "
 112 """"""""""""
 113 " An identifier must start with \h and continue with \w.
 114 " It can be indexed with []
 115 " A member can be selected with .
 116 " If the variable is in a string, it must end with a char other than \w[]$. FIXME not sure how wesnoth parser is working on this one, this set is certainly not convering all situations.
 117 syn match wesmlIdentifier display /\(\h\w*\(\[[^\]]\+\]\)\?\)\(\.\h\w*\(\[[^\]]\+\]\)\?\)*/ contained contains=wesmlBracket,wesmlDot,wesmlMacroName,wesmlBoolean,wesmlUnderscore
 118 syn match wesmlVariable display /\([\]a-zA-Z0-9_]\@<=\.\|\$\)\(\h\w*\(\[[^\]]\+\]\)\?\)\(\.\h\w*\(\[[^\]]\+\]\)\?\)*/ contained contains=wesmlBracket,wesmlDot
 119 syn region wesmlBracket display start=/\[/ end=/\]/ contained contains=wesmlNumber,wesmlVariable,wesmlInclude
 120 syn match wesmlDot display /\./ contained
 121 syn match wesmlStringVariable display /$\h[a-zA-Z0-9_\[\]$\.]*|\?/ contains=wesmlVariable,wesmlStringVariableEnd contained
 122 syn match wesmlStringVariableEnd display "|" contained
