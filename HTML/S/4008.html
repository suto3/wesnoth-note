 219 #define STANDARD_IDLE_FILTER
 220     # Poisoned units should have no idle animation.
 221     [filter]
 222         [not]
 223             status=poisoned
 224         [/not]
 225     [/filter]
 226     # Units next to enemies should not have idle animations
 227     [filter]
 228         [filter_adjacent]
 229             is_enemy="yes"
 230             count=0
 231         [/filter_adjacent]
 232     [/filter]
 233 #enddef
