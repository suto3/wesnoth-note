 118 void custom_tod::pre_show(window& window)
 119 {
 120         static std::map<std::string, tod_attribute_getter> metadata_stuff {
 121                 {"image", tod_getter_image},
 122                 {"mask",  tod_getter_mask },
 123                 {"sound", tod_getter_sound}
 124         };
 125 
 126         window.add_to_tab_order(find_widget<text_box>(&window, "tod_name", false, false));
 127         window.add_to_tab_order(find_widget<text_box>(&window, "tod_id", false, false));
 128 
 129         for(const auto& data : metadata_stuff) {
 130                 find_widget<text_box>(&window, "path_" + data.first, false).set_active(false);
 131 
 132                 button& copy_w = find_widget<button>(&window, "copy_" + data.first, false);
 133 
 134                 connect_signal_mouse_left_click(copy_w,
 135                         std::bind(&custom_tod::copy_to_clipboard_callback, this, data.second));
 136 
 137                 if(!desktop::clipboard::available()) {
 138                         copy_w.set_active(false);
 139                         copy_w.set_tooltip(_("Clipboard support not found, contact your packager"));
 140                 }
 141         }
 142 
 143         connect_signal_mouse_left_click(
 144                         find_widget<button>(&window, "browse_image", false),
 145                         std::bind(&custom_tod::select_file<tod_getter_image>, this, std::ref(window), "data/core/images/misc"));
 146 
 147         connect_signal_mouse_left_click(
 148                         find_widget<button>(&window, "browse_mask", false),
 149                         std::bind(&custom_tod::select_file<tod_getter_mask>,  this, std::ref(window), "data/core/images"));
 150 
 151         connect_signal_mouse_left_click(
 152                         find_widget<button>(&window, "browse_sound", false),
 153                         std::bind(&custom_tod::select_file<tod_getter_sound>, this, std::ref(window), "data/core/sounds/ambient"));
 154 
 155         connect_signal_mouse_left_click(
 156                         find_widget<button>(&window, "next_tod", false),
 157                         std::bind(&custom_tod::do_next_tod, this, std::ref(window)));
 158 
 159         connect_signal_mouse_left_click(
 160                         find_widget<button>(&window, "previous_tod", false),
 161                         std::bind(&custom_tod::do_prev_tod, this, std::ref(window)));
 162 
 163         connect_signal_mouse_left_click(
 164                         find_widget<button>(&window, "new", false),
 165                         std::bind(&custom_tod::do_new_tod, this, std::ref(window)));
 166 
 167         connect_signal_mouse_left_click(
 168                         find_widget<button>(&window, "delete", false),
 169                         std::bind(&custom_tod::do_delete_tod, this, std::ref(window)));
 170 
 171         connect_signal_notify_modified(
 172                         find_widget<slider>(&window, "lawful_bonus", false),
 173                         std::bind(&custom_tod::update_lawful_bonus, this, std::ref(window)));
 174 
 175         connect_signal_notify_modified(
 176                         *(color_field_r_->get_widget()),
 177                         std::bind(&custom_tod::color_slider_callback, this, std::ref(window)));
 178 
 179         connect_signal_notify_modified(
 180                         *(color_field_g_->get_widget()),
 181                         std::bind(&custom_tod::color_slider_callback, this, std::ref(window)));
 182 
 183         connect_signal_notify_modified(
 184                         *(color_field_b_->get_widget()),
 185                         std::bind(&custom_tod::color_slider_callback, this, std::ref(window)));
 186 
 187         update_selected_tod_info(window);
 188 }

 190 template<custom_tod::string_pair(*fptr)(const time_of_day&)>
 191 void custom_tod::select_file(window& window, const std::string& default_dir)
 192 {
 193         const string_pair& data = (*fptr)(get_selected_tod());
 194 
 195         std::string fn = filesystem::base_name(data.second);
 196         std::string dn = filesystem::directory_name(fn);
 197         if(dn.empty()) {
 198                 dn = default_dir;
 199         }
 200 
 201         gui2::dialogs::file_dialog dlg;
 202 
 203         dlg.set_title(_("Choose File"))
 204            .set_ok_label(_("Select"))
 205            .set_path(dn)
 206            .set_read_only(true);
 207 
 208         if(dlg.show(window.video())) {
 209                 dn = dlg.path();
 210 
 211                 if(data.first == "image") {
 212                         times_[current_tod_].image = dn;
 213                 } else if(data.first == "mask") {
 214                         times_[current_tod_].image_mask = dn;
 215                 } else if(data.first == "sound") {
 216                         times_[current_tod_].sounds = dn;
 217                 }
 218         }
 219 
 220         update_selected_tod_info(window);
 221 }

 223 void custom_tod::do_next_tod(window& window)
 224 {
 225         current_tod_ = (current_tod_ + 1) % times_.size();
 226         update_selected_tod_info(window);
 227 }

 229 void custom_tod::do_prev_tod(window& window)
 230 {
 231         current_tod_ = (current_tod_ ? current_tod_ : times_.size()) - 1;
 232         update_selected_tod_info(window);
 233 }

 235 void custom_tod::do_new_tod(window& window)
 236 {
 237         times_.insert(times_.begin() + current_tod_, time_of_day());
 238         update_selected_tod_info(window);
 239 }
 240 
 241 void custom_tod::do_delete_tod(window& window)
 242 {
 243         assert(times_.begin() + current_tod_ < times_.end());
 244 
 245         if(times_.size() == 1) {
 246                 times_.emplace_back();
 247         } else {
 248                 times_.erase(times_.begin() + current_tod_);
 249 
 250                 if(times_.begin() + current_tod_ >= times_.end()) {
 251                         current_tod_ = times_.size() - 1;
 252                 }
 253         }
 254 
 255         update_selected_tod_info(window);
 256 }

 258 const time_of_day& custom_tod::get_selected_tod() const
 259 {
 260         try {
 261                 return times_.at(current_tod_);
 262         } catch(std::out_of_range&) {
 263                 throw std::string("Attempted to fetch a non-existant ToD!");
 264         }
 265 }

 267 void custom_tod::color_slider_callback(window& window)
 268 {
 269         time_of_day& current_tod = times_[current_tod_];

 271         current_tod.color.r = color_field_r_->get_widget_value(window);

 276 }

 278 void custom_tod::update_tod_display(window& window)
 279 {
 280         display* disp = display::get_singleton();
 281         assert(disp && "Display pointer is null!");
 282 
 283         // Prevent a floating slice of window appearing alone over the
 284         // theme UI sidebar after redrawing tiles and before we have a
 285         // chance to redraw the rest of this window.
 286         window.undraw();
 287 
 288         // NOTE: We only really want to re-render the gamemap tiles here.
 289         // Redrawing everything is a significantly more expensive task.
 290         // At this time, tiles are the only elements on which ToD tint is
 291         // meant to have a visible effect. This is very strongly tied to
 292         // the image caching mechanism.
 293         //
 294         // If this ceases to be the case in the future, you'll need to call
 295         // redraw_everything() instead.
 296 
 297         disp->update_tod(&get_selected_tod());
 298 
 299         // invalidate all tiles so they are redrawn with the new ToD tint next
 300         disp->invalidate_all();
 301 
 302         // redraw tiles
 303         disp->draw(false);
 304 
 305         // NOTE: revert to invalidate_layout if necessary to display the ToD mask image.
 306         window.set_is_dirty(true);
 307 }

 314 void custom_tod::update_selected_tod_info(window& window)
 315 {
 316         const time_of_day& current_tod = get_selected_tod();
 317 
 318         find_widget<text_box>(&window, "tod_name", false).set_value(current_tod.name);
 319         find_widget<text_box>(&window, "tod_id", false).set_value(current_tod.id);
 320 
 321         find_widget<text_box>(&window, "path_image", false).set_value(current_tod.image);
 322         find_widget<text_box>(&window, "path_mask", false).set_value(current_tod.image_mask);
 323         find_widget<text_box>(&window, "path_sound", false).set_value(current_tod.sounds);
 324 
 325         find_widget<image>(&window, "current_tod_image", false).set_image(current_tod.image);
 326         find_widget<image>(&window, "current_tod_mask", false).set_image(current_tod.image_mask);
 327 
 328         find_widget<slider>(&window, "lawful_bonus", false).set_value(current_tod.lawful_bonus);
 329 
 330         color_field_r_->set_widget_value(window, current_tod.color.r);
 331         color_field_g_->set_widget_value(window, current_tod.color.g);
 332         color_field_b_->set_widget_value(window, current_tod.color.b);
 333 
 334         const std::string new_index_str = formatter() << (current_tod_ + 1) << "/" << times_.size();
 335         find_widget<label>(&window, "tod_number", false).set_label(new_index_str);
 336 
 337         update_tod_display(window);
 338 }
