  56 void carryover::transfer_all_gold_to(config& side_cfg){
  57 
  58         int cfg_gold = side_cfg["gold"].to_int();
  59 
  60         if(side_cfg["gold"].empty()) {
  61                 cfg_gold = default_gold_qty;
  62                 side_cfg["gold"] = cfg_gold;
  63         }
  64 
  65         if(add_ && gold_ > 0){
  66                 side_cfg["gold"] = cfg_gold + gold_;
  67         }
  68         else if(gold_ > cfg_gold){
  69                 side_cfg["gold"] = gold_;
  70         }
  71         side_cfg.child_or_add("variables").swap(variables_);
  72         variables_.clear();
  73         gold_ = 0;
  74 }

 121 carryover_info::carryover_info(const config& cfg, bool from_snpashot)
 122         : carryover_sides_()
 123         , variables_(cfg.child_or_empty("variables"))
 124         , rng_(cfg)
 125         , wml_menu_items_()
 126         , next_scenario_(cfg["next_scenario"])
 127         , next_underlying_unit_id_(cfg["next_underlying_unit_id"].to_int(0))
 128 {
 129         for(const config& side : cfg.child_range("side"))
 130         {
 131                 if(side["lost"].to_bool(false) || !side["persistent"].to_bool(true))
 132                 {
 133                         //this shouldnt happen outside a snpshot.
 134                         assert(from_snpashot);
 135                         continue;
 136                 }
 137                 this->carryover_sides_.emplace_back(side);
 138         }
 139         for(const config& item : cfg.child_range("menu_item"))
 140         {
 141                 wml_menu_items_.push_back(new config(item));
 142         }
 143 }

 175 void carryover_info::transfer_all_to(config& side_cfg){
 176         if(side_cfg["save_id"].empty()){
 177                 side_cfg["save_id"] = side_cfg["id"];
 178         }
 179         std::vector<carryover>::iterator iside = std::find_if(
 180                 carryover_sides_.begin(),
 181                 carryover_sides_.end(),
 182                 save_id_equals(side_cfg["save_id"])
 183         );
 184         if(iside != carryover_sides_.end())
 185         {
 186                 iside->transfer_all_gold_to(side_cfg);
 187                 iside->transfer_all_recalls_to(side_cfg);
 188                 iside->transfer_all_recruits_to(side_cfg);
 189                 carryover_sides_.erase(iside);
 190                 return;
 191         }
 192         else
 193         {
 194                 //if no carryover was found for this side, check if starting gold is defined
 195                 if(!side_cfg.has_attribute("gold") || side_cfg["gold"].empty()){
 196                         side_cfg["gold"] = default_gold_qty;
 197                 }
 198         }
 199 }

 201 void carryover_info::transfer_to(config& level)
 202 {
 203         if(!level.has_attribute("next_underlying_unit_id"))
 204         {
 205                 level["next_underlying_unit_id"] = next_underlying_unit_id_;
 206         }
 207 
 208         //if the game has been loaded from a snapshot, the existing variables will be the current ones
 209         if(!level.has_child("variables")) {
 210                 level.add_child("variables", variables_);
 211         }
 212 
 213         config::attribute_value & seed_value = level["random_seed"];
 214         if ( seed_value.empty() ) {
 215                 seed_value = rng_.get_random_seed_str();
 216                 level["random_calls"] = rng_.get_random_calls();
 217         }
 218 
 219         if(!level.has_child("menu_item")){
 220                 for(config& item : wml_menu_items_)
 221                 {
 222                         level.add_child("menu_item").swap(item);
 223                 }
 224         }
 225 
 226         next_scenario_ = "";
 227         variables_ = config();
 228         wml_menu_items_.clear();
 229 
 230 }

 232 const config carryover_info::to_config()
 233 {
 234         config cfg;
 235         cfg["next_underlying_unit_id"] = next_underlying_unit_id_;
 236         cfg["next_scenario"] = next_scenario_;
 237 
 238         for(carryover& c : carryover_sides_) {
 239                 c.to_config(cfg);
 240         }
 241 
 242         cfg["random_seed"] = rng_.get_random_seed_str();
 243         cfg["random_calls"] = rng_.get_random_calls();
 244 
 245         cfg.add_child("variables", variables_);
 246         for(const config& item : wml_menu_items_)
 247         {
 248                 cfg.add_child("menu_item", item);

 250         return cfg;

 263 void carryover_info::merge_old_carryover(const carryover_info& old_carryover)
 264 {
 265         for(const carryover & old_side : old_carryover.carryover_sides_)
 266         {
 267                 std::vector<carryover>::iterator iside = std::find_if(
 268                         carryover_sides_.begin(),
 269                         carryover_sides_.end(),
 270                         save_id_equals(old_side.get_save_id())
 271                         );
 272                 //add the side if don't already have it.
 273                 if(iside == carryover_sides_.end())
 274                 {
 275                         this->carryover_sides_.push_back(old_side);
 276                 }
 277         }
 278 }
