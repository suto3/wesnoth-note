  49 void send(const std::string& owner, const std::string& message, type t)
  50 {
  51         sdl::window* window = CVideo::get_singleton().get_window();
  52         if(window == nullptr) {
  53                 return;
  54         }
  55 
  56         int app_state = window->get_flags();
  57 
  58         // Do not show notifications when the window is visible...
  59         if ((app_state & SDL_WINDOW_SHOWN) != 0)
  60         {
  61                 // ... and it has a focus.
  62                 if ((app_state & (SDL_WINDOW_MOUSE_FOCUS | SDL_WINDOW_INPUT_FOCUS)) != 0) {
  63                         return;
  64                 }
  65         }
  66 
  67 #ifdef HAVE_LIBDBUS
  68         dbus::send_notification(owner, message, t == CHAT);
  69 #endif
  70 
  71 #ifdef __APPLE__
  72         apple_notifications::send_notification(owner, message, t);
  73 #endif
  74 
  75 #ifdef _WIN32
  76         std::string notification_title;
  77         std::string notification_message;
  78 
  79         switch (t) {
  80                 case CHAT:
  81                         notification_title = _("Chat message");
  82                         notification_message = owner + ": " + message;
  83                         break;
  84                 case TURN_CHANGED:
  85                 case OTHER:
  86                         notification_title = owner;
  87                         notification_message = message;
  88                         break;
  89         }
  90 
  91         windows_tray_notification::show(notification_title, notification_message);
  92 #endif
  93 }
  94 #endif //end #else (defined(HAVE_LIBDBUS) || defined(HAVE_GROWL) || defined(_WIN32))
