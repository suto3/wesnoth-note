   1 # The purpose of this test is to check that units spawned on
   2 # an occupied hex are placed at the nearest available vacant hex.
   3 #
   4 # Six blue grunts are spawned, one at 3,4 and five more at adjacent
   5 # hexes, leaving the southern hex 3,5 open.
   6 # Then a red grunt is spawned at 3,4.
   7 # We try to store a unit at 3,5, assert that we find one, and that
   8 # it is red.
   9 
  10 {GENERIC_UNIT_TEST "unit_spawns_at_nearest_vacant_hex" (
  11     [event]
  12         name = start
  13         {UNIT 2 "Orcish Grunt" 3 4 ()}
  14         {UNIT 2 "Orcish Grunt" 3 3 ()}
  15         {UNIT 2 "Orcish Grunt" 4 3 ()}
  16         {UNIT 2 "Orcish Grunt" 4 4 ()}
  17         {UNIT 2 "Orcish Grunt" 2 3 ()}
  18         {UNIT 2 "Orcish Grunt" 2 4 ()}
  19 
  20         {UNIT 1 "Orcish Grunt" 3 4 ()}
  21 
  22         [store_unit]
  23             variable = test
  24             [filter]
  25                 x = 3
  26                 y = 5
  27             [/filter]
  28         [/store_unit]
  29         {ASSERT ({VARIABLE_CONDITIONAL test.length equals 1})}
  30         {RETURN ({VARIABLE_CONDITIONAL test.side equals 1})}
  31     [/event]
  32 )}
