 136 void playsingle_controller::hotkey_handler::whiteboard_toggle() {
 137         whiteboard_manager_->set_active(!whiteboard_manager_->is_active());
 138 
 139         if (whiteboard_manager_->is_active()) {
 140                 std::string hk = hotkey::get_names(hotkey::hotkey_command::get_command_by_command(hotkey::HOTKEY_WB_TOGGLE).command);
 141                 utils::string_map symbols;
 142                 symbols["hotkey"] = hk;
 143 
 144                 gui()->announce(_("Planning mode activated!") + std::string("\n") + vgettext("(press $hotkey to deactivate)", symbols), font::NORMAL_COLOR);
 145         } else {
 146                 gui()->announce(_("Planning mode deactivated!"), font::NORMAL_COLOR);
 147         }
 148         //@todo Stop printing whiteboard help in the chat once we have better documentation/help
 149         whiteboard_manager_->print_help_once();
 150 }

 185 hotkey::ACTION_STATE playsingle_controller::hotkey_handler::get_action_state(hotkey::HOTKEY_COMMAND command, int index) const
 186 {
 187         switch(command) {
 188         case hotkey::HOTKEY_WB_TOGGLE:
 189                 return whiteboard_manager_->is_active() ? hotkey::ACTION_ON : hotkey::ACTION_OFF;
 190         default:
 191                 return play_controller::hotkey_handler::get_action_state(command, index);
 192         }

 195 bool playsingle_controller::hotkey_handler::can_execute_command(const hotkey::hotkey_command& cmd, int index) const
 196 {
 197         hotkey::HOTKEY_COMMAND command = cmd.id;
 198         bool res = true;
 199         switch (command){
 200 
 201                 case hotkey::HOTKEY_WML:
 202                 {
 203                         int prefixlen = wml_menu_hotkey_prefix.length();
 204                         if(cmd.command.compare(0, prefixlen, wml_menu_hotkey_prefix) != 0) {
 205                                 return false;
 206                         }
 207 
 208                         game_events::wmi_manager::item_ptr item = gamestate().get_wml_menu_items().get_item(cmd.command.substr(prefixlen));
 209                         if(!item) {
 210                                 return false;
 211                         }
 212 
 213                         return !item->is_synced() || play_controller_.can_use_synced_wml_menu();
 214                 }
 215                 case hotkey::HOTKEY_UNIT_HOLD_POSITION:
 216                 case hotkey::HOTKEY_END_UNIT_TURN:
 217                         return !browse() && !linger() && !events::commands_disabled;
 218                 case hotkey::HOTKEY_RECRUIT:
 219                 case hotkey::HOTKEY_REPEAT_RECRUIT:
 220                 case hotkey::HOTKEY_RECALL:
 221                         return (!browse() || whiteboard_manager_->is_active()) && !linger() && !events::commands_disabled;
 222                 case hotkey::HOTKEY_ENDTURN:
 223                         return (!browse() || linger()) && !events::commands_disabled;
 224 
 225                 case hotkey::HOTKEY_DELAY_SHROUD:
 226                         return !linger()
 227                                 && (viewing_team().uses_fog() || viewing_team().uses_shroud())
 228                                 && viewing_team_is_playing()
 229                                 && viewing_team().is_local_human()
 230                                 && !events::commands_disabled;
 231                 case hotkey::HOTKEY_UPDATE_SHROUD:
 232                         return !linger()
 233                                 && viewing_team_is_playing()
 234                                 && viewing_team().is_local_human()
 235                                 && !events::commands_disabled
 236                                 && viewing_team().auto_shroud_updates() == false;
 237 
 238                 // Commands we can only do if in debug mode
 239                 case hotkey::HOTKEY_CREATE_UNIT:
 240                 case hotkey::HOTKEY_CHANGE_SIDE:
 241                 case hotkey::HOTKEY_KILL_UNIT:
 242                         return !events::commands_disabled && game_config::debug && play_controller_.get_map_const().on_board(mouse_handler_.get_last_hex());
 243 
 244                 case hotkey::HOTKEY_CLEAR_LABELS:
 245                         res = !is_observer();
 246                         break;
 247                 case hotkey::HOTKEY_LABEL_TEAM_TERRAIN:
 248                 case hotkey::HOTKEY_LABEL_TERRAIN: {
 249                         const terrain_label *label = gui()->labels().get_label(mouse_handler_.get_last_hex());
 250                         res = !events::commands_disabled && play_controller_.get_map_const().on_board(mouse_handler_.get_last_hex())
 251                                 && !gui()->shrouded(mouse_handler_.get_last_hex())
 252                                 && !is_observer()
 253                                 && (!label || !label->immutable());
 254                         break;
 255                 }
 256                 case hotkey::HOTKEY_CONTINUE_MOVE: {
 257                         if(browse() || events::commands_disabled)
 258                                 return false;
 259 
 260                         if( (menu_handler_.current_unit().valid())
 261                                 && (menu_handler_.current_unit()->move_interrupted()))
 262                                 return true;
 263                         const unit_map::const_iterator i = play_controller_.get_units_const().find(mouse_handler_.get_selected_hex());
 264                         if (!i.valid()) return false;
 265                         return i->move_interrupted();
 266                 }
 267                 case hotkey::HOTKEY_WB_TOGGLE:
 268                         return !is_observer();
 269                 case hotkey::HOTKEY_WB_EXECUTE_ACTION:
 270                 case hotkey::HOTKEY_WB_EXECUTE_ALL_ACTIONS:
 271                         return whiteboard_manager_->can_enable_execution_hotkeys();
 272                 case hotkey::HOTKEY_WB_DELETE_ACTION:
 273                         return whiteboard_manager_->can_enable_modifier_hotkeys();
 274                 case hotkey::HOTKEY_WB_BUMP_UP_ACTION:
 275                 case hotkey::HOTKEY_WB_BUMP_DOWN_ACTION:
 276                         return whiteboard_manager_->can_enable_reorder_hotkeys();
 277                 case hotkey::HOTKEY_WB_SUPPOSE_DEAD:
 278                 {
 279                         //@todo re-enable this once we figure out a decent UI for suppose_dead
 280                         return false;
 281                 }
 282 
 283                 case hotkey::HOTKEY_REPLAY_STOP:
 284                 case hotkey::HOTKEY_REPLAY_PLAY:
 285                 case hotkey::HOTKEY_REPLAY_NEXT_TURN:
 286                 case hotkey::HOTKEY_REPLAY_NEXT_SIDE:
 287                 case hotkey::HOTKEY_REPLAY_NEXT_MOVE:
 288                 case hotkey::HOTKEY_REPLAY_SKIP_ANIMATION:
 289                 case hotkey::HOTKEY_REPLAY_SHOW_EVERYTHING:
 290                 case hotkey::HOTKEY_REPLAY_SHOW_EACH:
 291                 case hotkey::HOTKEY_REPLAY_SHOW_TEAM1:
 292                 case hotkey::HOTKEY_REPLAY_RESET:
 293                         return playsingle_controller_.get_replay_controller() && playsingle_controller_.get_replay_controller()->can_execute_command(cmd, index);
 294                 case hotkey::HOTKEY_REPLAY_EXIT:
 295                         return playsingle_controller_.is_replay();
 296                 default: return play_controller::hotkey_handler::can_execute_command(cmd, index);
 297         }
 298         return res;
 299 }

 301 void playsingle_controller::hotkey_handler::load_autosave(const std::string& filename)
 302 {
 303         if(playsingle_controller_.is_networked_mp())
 304         {
 305                 config savegame;
 306                 std::string error_log;
 307                 savegame::read_save_file(filename, savegame, &error_log);
 308 
 309                 if(!error_log.empty() || savegame.child_or_empty("snapshot")["replay_pos"].to_int(-1) < 0 ) {
 310                         gui2::show_error_message(play_controller_.get_display().video(),
 311                                 _("The file you have tried to load is corrupt: '") +
 312                                 error_log);
 313                         return;
 314                 }
 315                 std::shared_ptr<config> res(new config(savegame.child_or_empty("snapshot")));
 316                 throw reset_gamestate_exception(res, true);
 317         }
 318         else
 319         {
 320                 play_controller::hotkey_handler::load_autosave(filename);
 321         }
 322 }
