  68 void clean_saves(const std::string& label)
  69 {
  70         std::vector<save_info> games = get_saves_list();
  71         std::string prefix = label + "-" + _("Auto-Save");
  72         LOG_SAVE << "Cleaning saves with prefix '" << prefix << "'\n";
  73         for (std::vector<save_info>::iterator i = games.begin(); i != games.end(); ++i) {
  74                 if (i->name().compare(0, prefix.length(), prefix) == 0) {
  75                         LOG_SAVE << "Deleting savegame '" << i->name() << "'\n";
  76                         delete_game(i->name());
  77                 }

 119 // Called only by play_controller to handle in-game attempts to load. Instead of returning true,
 120 // throws a "load_game_exception" to signal a resulting load game request.
 121 bool loadgame::load_game_ingame()
 122 {
 123         if (video_.faked()) {
 124                 return false;
 125         }
 126 
 127         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 128                 return false;
 129         }
 130 
 131         if(load_data_.filename.empty()) {
 132                 return false;
 133         }
 134 
 135         if(load_data_.select_difficulty) {
 136                 if(!show_difficulty_dialog()) {
 137                         return false;
 138                 }
 139         }
 140 
 141         load_data_.show_replay |= is_replay_save(load_data_.summary);
 142 
 143         // Confirm the integrity of the file before throwing the exception.
 144         // Use the summary in the save_index for this.
 145         const config & summary = save_index_manager.get(load_data_.filename);
 146 
 147         if (summary["corrupt"].to_bool(false)) {
 148                 gui2::show_error_message(video_,
 149                                 _("The file you have tried to load is corrupt: '"));
 150                 return false;
 151         }
 152 
 153         if (!loadgame::check_version_compatibility(summary["version"].str(), video_)) {
 154                 return false;
 155         }
 156 
 157         throw load_game_exception(std::move(load_data_));
 158 }

 160 bool loadgame::load_game()
 161 {
 162         bool skip_version_check = true;
 163 
 164         if(load_data_.filename.empty()){
 165                 if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 166                         return false;
 167                 }
 168                 skip_version_check = false;
 169                 load_data_.show_replay |= is_replay_save(load_data_.summary);
 170         }
 171 
 172         if(load_data_.filename.empty()) {
 173                 return false;
 174         }
 175 
 176         if(load_data_.select_difficulty) {
 177                 if(!show_difficulty_dialog()) {
 178                         return false;
 179                 }
 180         }
 181 
 182         std::string error_log;
 183         read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 184 
 185         convert_old_saves(load_data_.load_config);
 186 
 187         if(!error_log.empty()) {
 188         try {
 189                     gui2::show_error_message(video_,
 190                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 191                                     error_log);
 192         } catch (utf8::invalid_utf8_exception&) {
 193                     gui2::show_error_message(video_,
 194                                     _("Warning: The file you have tried to load is corrupt. Loading anyway.\n") +
 195                     std::string("(UTF-8 ERROR)"));
 196         }
 197         }
 198 
 199         if (!load_data_.difficulty.empty()){
 200                 load_data_.load_config["difficulty"] = load_data_.difficulty;
 201         }
 202         // read classification to for loading the game_config config object.
 203         gamestate_.classification() = game_classification(load_data_.load_config);
 204 
 205         if (skip_version_check) {
 206                 return true;
 207         }
 208 
 209         return check_version_compatibility();
 210 }

 217 bool loadgame::check_version_compatibility(const version_info & save_version, CVideo & video)
 218 {
 219         if (save_version == game_config::version) {
 220                 return true;
 221         }
 222 
 223         const version_info &wesnoth_version = game_config::wesnoth_version;
 224 
 225         // Even minor version numbers indicate stable releases which are
 226         // compatible with each other.
 227         if (wesnoth_version.minor_version() % 2 == 0 &&
 228             wesnoth_version.major_version() == save_version.major_version() &&
 229             wesnoth_version.minor_version() == save_version.minor_version())
 230         {
 231                 return true;
 232         }
 233 
 234         // Do not load if too old. If either the savegame or the current
 235         // game has the version 'test', load. This 'test' version is never
 236         // supposed to occur, except when Soliton is testing MP servers.
 237         if (save_version < game_config::min_savegame_version &&
 238             save_version != game_config::test_version &&
 239             wesnoth_version != game_config::test_version)
 240         {
 241                 const std::string message = _("This save is from an old, unsupported version ($version_number|) and cannot be loaded.");
 242                 utils::string_map symbols;
 243                 symbols["version_number"] = save_version.str();
 244                 gui2::show_error_message(video, utils::interpolate_variables_into_string(message, &symbols));
 245                 return false;
 246         }
 247 
 248         if(preferences::confirm_load_save_from_different_version()) {
 249                 const std::string message = _("This save is from a different version of the game ($version_number|). Do you wish to try to load it?");
 250                 utils::string_map symbols;
 251                 symbols["version_number"] = save_version.str();
 252                 const int res = gui2::show_message(video, _("Load Game"), utils::interpolate_variables_into_string(message, &symbols),
 253                         gui2::dialogs::message::yes_no_buttons);
 254                 return res == gui2::window::OK;
 255         }
 256 
 257         return true;
 258 }

 265 bool loadgame::load_multiplayer_game()
 266 {
 267         if(!gui2::dialogs::game_load::execute(game_config_, load_data_, video_)) {
 268                 return false;
 269         }
 270 
 271 
 272         load_data_.show_replay |= is_replay_save(load_data_.summary);
 273         if(load_data_.filename.empty()) {
 274                 return false;
 275         }
 276 
 277         // read_save_file needs to be called before we can verify the classification so the data has
 278         // been populated. Since we do that, we report any errors in that process first.
 279         std::string error_log;
 280         {
 281                 cursor::setter cur(cursor::WAIT);
 282                 log_scope("load_game");
 283 
 284                 read_save_file(load_data_.filename, load_data_.load_config, &error_log);
 285                 copy_era(load_data_.load_config);
 286         }
 287 
 288         if(!error_log.empty()) {
 289                 gui2::show_error_message(video_,
 290                                 _("The file you have tried to load is corrupt: '") +
 291                                 error_log);
 292                 return false;
 293         }
 294 
 295         if(is_replay_save(load_data_.summary)) {
 296                 gui2::show_transient_message(video_, _("Load Game"), _("Replays are not supported in multiplayer mode."));
 297                 return false;
 298         }
 299 
 300         // We want to verify the game classification before setting the data, so we don't check on
 301         // gamestate_.classification() and instead construct a game_classification object manually.
 302         if(game_classification(load_data_.load_config).campaign_type != game_classification::CAMPAIGN_TYPE::MULTIPLAYER) {
 303                 gui2::show_transient_error_message(video_, _("This is not a multiplayer save."));
 304                 return false;
 305         }
 306 
 307         set_gamestate();
 308 
 309         return check_version_compatibility();
 310 }

 437 bool savegame::save_game(CVideo* video, const std::string& filename)
 438 {
 439 
 440         try {
 441                 Uint32 start, end;
 442                 start = SDL_GetTicks();
 443 
 444                 if (filename_ == "")
 445                         filename_ = filename;
 446 
 447                 before_save();
 448 
 449                 write_game_to_disk(filename_);
 450                 if (resources::persist != nullptr) {
 451                         resources::persist->end_transaction();
 452                         resources::persist ->start_transaction();
 453                 }
 454 
 455                 end = SDL_GetTicks();
 456                 LOG_SAVE << "Milliseconds to save " << filename_ << ": " << end - start << std::endl;
 457 
 458                 if (video != nullptr && show_confirmation_)
 459                         gui2::show_transient_message(*video, _("Saved"), _("The game has been saved."));
 460                 return true;
 461         } catch(game::save_game_failed& e) {
 462                 ERR_SAVE << error_message_ << e.message << std::endl;
 463                 if (video != nullptr){
 464                         gui2::show_error_message(*video, error_message_ + e.message);
 465                         //do not bother retrying, since the user can just try to save the game again
 466                         //maybe show a yes-no dialog for "disable autosaves now"?
 467                 }
 468 
 469                 return false;
 470         };
 471 }

 473 void savegame::write_game_to_disk(const std::string& filename)
 474 {
 475         LOG_SAVE << "savegame::save_game" << std::endl;
 476 
 477         filename_ = filename;
 478         filename_ += compression::format_extension(compress_saves_);
 479 
 480         std::stringstream ss;
 481         {
 482                 config_writer out(ss, compress_saves_);
 483                 write_game(out);
 484                 finish_save_game(out);
 485         }
 486         filesystem::scoped_ostream os(open_save_game(filename_));
 487         (*os) << ss.str();
 488 
 489         if (!os->good()) {
 490                 throw game::save_game_failed(_("Could not write to file"));
 491         }
 492 }

 644 static void convert_old_saves_1_11_0(config& cfg)
 645 {
 646         if(!cfg.has_child("snapshot")){
 647                 return;
 648         }
 649 
 650         const config& snapshot = cfg.child("snapshot");
 651         const config& replay_start = cfg.child("replay_start");
 652         const config& replay = cfg.child("replay");
 653 
 654         if(!cfg.has_child("carryover_sides") && !cfg.has_child("carryover_sides_start")){
 655                 config carryover;
 656                 //copy rng and menu items from toplevel to new carryover_sides
 657                 carryover["random_seed"] = cfg["random_seed"];
 658                 carryover["random_calls"] = cfg["random_calls"];
 659                 for(const config& menu_item : cfg.child_range("menu_item")) {
 660                         carryover.add_child("menu_item", menu_item);
 661                 }
 662                 carryover["difficulty"] = cfg["difficulty"];
 663                 carryover["random_mode"] = cfg["random_mode"];
 664                 //the scenario to be played is always stored as next_scenario in carryover_sides_start
 665                 carryover["next_scenario"] = cfg["scenario"];
 666 
 667                 config carryover_start = carryover;
 668 
 669                 //copy sides from either snapshot or replay_start to new carryover_sides
 670                 if(!snapshot.empty()){
 671                         for(const config& side : snapshot.child_range("side")) {
 672                                 carryover.add_child("side", side);
 673                         }
 674                         //for compatibility with old savegames that use player instead of side
 675                         for(const config& side : snapshot.child_range("player")) {
 676                                 carryover.add_child("side", side);
 677                         }
 678                         //save the sides from replay_start in carryover_sides_start
 679                         for(const config& side : replay_start.child_range("side")) {
 680                                 carryover_start.add_child("side", side);
 681                         }
 682                         //for compatibility with old savegames that use player instead of side
 683                         for(const config& side : replay_start.child_range("player")) {
 684                                 carryover_start.add_child("side", side);
 685                         }
 686                 } else if (!replay_start.empty()){
 687                         for(const config& side : replay_start.child_range("side")) {
 688                                 carryover.add_child("side", side);
 689                                 carryover_start.add_child("side", side);
 690                         }
 691                         //for compatibility with old savegames that use player instead of side
 692                         for(const config& side : replay_start.child_range("player")) {
 693                                 carryover.add_child("side", side);
 694                                 carryover_start.add_child("side", side);
 695                         }
 696                 }
 697 
 698                 //get variables according to old hierarchy and copy them to new carryover_sides
 699                 if(!snapshot.empty()){
 700                         if(const config& variables_from_snapshot = snapshot.child("variables")){
 701                                 carryover.add_child("variables", variables_from_snapshot);
 702                                 carryover_start.add_child("variables", replay_start.child_or_empty("variables"));
 703                         } else if (const config& variables_from_cfg = cfg.child("variables")){
 704                                 carryover.add_child("variables", variables_from_cfg);
 705                                 carryover_start.add_child("variables", variables_from_cfg);
 706                         }
 707                 } else if (!replay_start.empty()){
 708                         if(const config& variables = replay_start.child("variables")){
 709                                 carryover.add_child("variables", variables);
 710                                 carryover_start.add_child("variables", variables);
 711                         }
 712                 } else {
 713                         carryover.add_child("variables", cfg.child("variables"));
 714                         carryover_start.add_child("variables", cfg.child("variables"));
 715                 }
 716 
 717                 cfg.add_child("carryover_sides", carryover);
 718                 cfg.add_child("carryover_sides_start", carryover_start);
 719         }
 720 
 721         //if replay and snapshot are empty we've got a start of scenario save and don't want replay_start either
 722         if(replay.empty() && snapshot.empty()){
 723                 LOG_RG<<"removing replay_start \n";
 724                 cfg.clear_children("replay_start");
 725         }
 726 
 727         //remove empty replay or snapshot so type of save can be detected more easily
 728         if(replay.empty()){
 729                 LOG_RG<<"removing replay \n";
 730                 cfg.clear_children("replay");
 731         }
 732 
 733         if(snapshot.empty()){
 734                 LOG_RG<<"removing snapshot \n";
 735                 cfg.clear_children("snapshot");
 736         }
 737 }

 739 static void convert_old_saves_1_13_0(config& cfg)
 740 {
 741         if(config& carryover_sides_start = cfg.child("carryover_sides_start"))
 742         {
 743                 if(!carryover_sides_start.has_attribute("next_underlying_unit_id"))
 744                 {
 745                         carryover_sides_start["next_underlying_unit_id"] = cfg["next_underlying_unit_id"];
 746                 }
 747         }
 748         if(cfg.child_or_empty("snapshot").empty())
 749         {
 750                 cfg.clear_children("snapshot");
 751         }
 752         if(cfg.child_or_empty("replay_start").empty())
 753         {
 754                 cfg.clear_children("replay_start");
 755         }
 756         if(config& snapshot = cfg.child("snapshot"))
 757         {
 758                 //make [end_level] -> [end_level_data] since its alo called [end_level_data] in the carryover.
 759                 if(config& end_level = cfg.child("end_level") )
 760                 {
 761                         snapshot.add_child("end_level_data", end_level);
 762                         snapshot.clear_children("end_level");
 763                 }
 764                 //if we have a snapshot then we already applied carryover so there is no reason to keep this data.
 765                 if(cfg.has_child("carryover_sides_start"))
 766                 {
 767                         cfg.clear_children("carryover_sides_start");
 768                 }
 769         }
 770         if(!cfg.has_child("snapshot") && !cfg.has_child("replay_start"))
 771         {
 772                 cfg.clear_children("carryover_sides");
 773         }
 774         //This code is needed because for example otherwise it won't find the (empty) era
 775         if(!cfg.has_child("multiplayer")) {
 776                 cfg.add_child("multiplayer", config_of
 777                         ("mp_era", "era_blank")
 778                         ("show_connect", false)
 779                         ("show_configure", false)
 780                         ("mp_use_map_settings", true)
 781                 );
 782         }
 783 }

 805                                                 modifications.add_child("advancement", advancement);
 806                                         }
 807                                         modifications.clear_children("advance");
 808                                 }
 809                         }
 810                 }
 811         }
 812         for(config& snapshot : cfg.child_range("snapshot")) {
 813                 if (snapshot.has_attribute("used_items")) {
 814                         config used_items;
 815                         for(const std::string& item : utils::split(snapshot["used_items"])) {
 816                                 used_items[item] = true;
 817                         }
 818                         snapshot.remove_attribute("used_items");
 819                         snapshot.add_child("used_items", used_items);
 820                 }
 821         }

 824 void convert_old_saves(config& cfg)
 825 {
 826         version_info loaded_version(cfg["version"]);
 827         if(loaded_version < version_info("1.12.0"))
 828         {
 829                 convert_old_saves_1_11_0(cfg);
 830         }
 831         // '<= version_info("1.13.0")' doesn't work
 832         //because version_info cannot handle 1.13.0-dev versions correctly.

 834         {
 835                 convert_old_saves_1_13_0(cfg);
 836         }
 837         if(loaded_version <= version_info("1.13.1"))
 838         {
 839                 convert_old_saves_1_13_1(cfg);
 840         }
 841         LOG_RG<<"cfg after conversion "<<cfg<<"\n";
 842 }
