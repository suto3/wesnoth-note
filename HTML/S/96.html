 361 void command_executor::show_menu(const std::vector<config>& items_arg, int xloc, int yloc, bool /*context_menu*/, display& gui)
 362 {
 363         std::vector<config> items = items_arg;
 364         if (items.empty()) return;
 365 
 366         get_menu_images(gui, items);
 367 
 368         int res = -1;
 369         {
 370                 SDL_Rect pos {xloc, yloc, 1, 1};
 371                 gui2::dialogs::drop_down_menu mmenu(pos, items, -1, true, false); // TODO: last value should be variable
 372                 mmenu.show(gui.video());
 373                 if(mmenu.get_retval() == gui2::window::OK) {
 374                         res = mmenu.selected_item();
 375                 }
 376         } // This will kill the dialog.
 377         if (res < 0 || size_t(res) >= items.size()) return;
 378 
 379         const theme::menu* submenu = gui.get_theme().get_menu_item(items[res]["id"]);
 380         if (submenu) {
 381                 int y,x;
 382                 SDL_GetMouseState(&x,&y);
 383                 this->show_menu(submenu->items(), x, y, submenu->is_context(), gui);
 384         } else {
 385                 const hotkey::hotkey_command& cmd = hotkey::get_hotkey_command(items[res]["id"]);
 386                 hotkey::execute_command(cmd,this,res);
 387                 set_button_state();
 388         }
 389 }

 391 void command_executor::execute_action(const std::vector<std::string>& items_arg, int /*xloc*/, int /*yloc*/, bool /*context_menu*/, display&)
 392 {
 393         std::vector<std::string> items = items_arg;
 394         if (items.empty()) {
 395                 return;
 396         }
 397 
 398         std::vector<std::string>::iterator i = items.begin();
 399         while(i != items.end()) {
 400                 const hotkey_command &command = hotkey::get_hotkey_command(*i);
 401                 if (can_execute_command(command)) {
 402                         hotkey::execute_command(command, this);
 403                         set_button_state();
 404                 }
 405                 ++i;
 406         }
 407 }

 409 std::string command_executor::get_menu_image(display& disp, const std::string& command, int index) const {
 410 
 411         // TODO: Find a way to do away with the fugly special markup
 412         if(command[0] == '&') {
 413                 size_t n = command.find_first_of('=');
 414                 if(n != std::string::npos)
 415                         return command.substr(1, n - 1);
 416         }
 417 
 418         const std::string base_image_name = "icons/action/" + command + "_25.png";
 419         const std::string pressed_image_name = "icons/action/" + command + "_25-pressed.png";
 420 
 421         const hotkey::HOTKEY_COMMAND hk = hotkey::get_id(command);
 422         const hotkey::ACTION_STATE state = get_action_state(hk, index);
 423 
 424         const theme::menu* menu = disp.get_theme().get_menu_item(command);
 425         if (menu) {
 426                 return "icons/arrows/short_arrow_right_25.png~CROP(3,3,18,18)"; // TODO should not be hardcoded
 427         }
 428 
 429         if (filesystem::file_exists(game_config::path + "/images/" + base_image_name)) {
 430                 switch (state) {
 431                         case ACTION_ON:
 432                         case ACTION_SELECTED:
 433                                 return pressed_image_name + "~CROP(3,3,18,18)";
 434                         default:
 435                                 return base_image_name + "~CROP(3,3,18,18)";
 436                 }
 437         }
 438 
 439         switch (get_action_state(hk, index)) {
 440                 case ACTION_ON:
 441                         return game_config::images::checked_menu;
 442                 case ACTION_OFF:
 443                         return game_config::images::unchecked_menu;
 444                 case ACTION_SELECTED:
 445                         return game_config::images::selected_menu;
 446                 case ACTION_DESELECTED:
 447                         return game_config::images::deselected_menu;
 448                 default: return get_action_image(hk, index);
 449         }
 450 }

 452 void command_executor::get_menu_images(display& disp, std::vector<config>& items)
 453 {
 454         for(size_t i = 0; i < items.size(); ++i) {
 455                 config& item = items[i];
 456 
 457                 const std::string& item_id = item["id"];
 458                 const hotkey::HOTKEY_COMMAND hk = hotkey::get_id(item_id);
 459 
 460                 //see if this menu item has an associated image
 461                 std::string img(get_menu_image(disp, item_id, i));
 462                 if (img.empty() == false) {
 463                         item["icon"] = img;
 464                 }
 465 
 466                 const theme::menu* menu = disp.get_theme().get_menu_item(item_id);
 467                 if(menu) {
 468                         item["label"] = menu->title();
 469                 } else if(hk != hotkey::HOTKEY_NULL) {
 470                         std::string desc = hotkey::get_description(item_id);
 471                         if(hk == HOTKEY_ENDTURN) {
 472                                 const theme::action *b = disp.get_theme().get_action_item("button-endturn");
 473                                 if (b) {
 474                                         desc = b->title();
 475                                 }
 476                         }
 477 
 478                         item["label"] = desc;
 479                         item["details"] = hotkey::get_names(item_id);
 480                 }
 481         }
 482 }

 486 void basic_handler::handle_event(const SDL_Event& event)
 487 {
 488 
 489         switch (event.type) {
 490         case SDL_KEYDOWN:
 491         case SDL_KEYUP:
 492                 // If we're in a dialog we only want to handle items that are explicitly
 493                 // handled by the executor.
 494                 // If we're not in a dialog we can call the regular key event handler.
 495                 if (!gui::in_dialog()) {
 496                         key_event(event,exec_);
 497                 } else if (exec_ != nullptr) {
 498                         event_execute(event,exec_);
 499                 }
 500                 break;
 501         case SDL_JOYBUTTONDOWN:
 502         case SDL_JOYBUTTONUP:
 503                 if (!gui::in_dialog()) {
 504                         jbutton_event(event,exec_);
 505                 } else if (exec_ != nullptr) {
 506                         event_execute(event,exec_);
 507                 }
 508                 break;
 509         case SDL_MOUSEBUTTONDOWN:
 510         case SDL_MOUSEBUTTONUP:
 511                 if (!gui::in_dialog()) {
 512                         mbutton_event(event,exec_);
 513                 } else if (exec_ != nullptr) {
 514                         event_execute(event,exec_);
 515                 }
 516                 break;
 517         }
 518 }

 541 static void event_execute( const SDL_Event& event, command_executor* executor)
 542 {
 543         if (!executor) return;
 544         const hotkey_ptr hk = get_hotkey(event);
 545         if (!hk->active() || hk->is_disabled()) {
 546                 return;
 547         }
 548 
 549         bool press = event.type == SDL_KEYDOWN ||
 550                         event.type == SDL_JOYBUTTONDOWN ||
 551                         event.type == SDL_MOUSEBUTTONDOWN ||
 552                         event.type == SDL_TEXTINPUT;
 553 
 554         execute_command(hotkey::get_hotkey_command(hk->get_command()), executor, -1, press);
 555         executor->set_button_state();
 556 }

 558 void execute_command(const hotkey_command& command, command_executor* executor, int index, bool press)
 559 {
 560         if (executor != nullptr) {
 561                 if (!executor->can_execute_command(command, index)
 562                                 || executor->execute_command(command, index, press)) {
 563                         return;
 564                 }
 565         }
 566 
 567         if (!press) {
 568                 return; // none of the commands here respond to a key release
 569     }
 570 
 571         switch (command.id) {
 572 
 573                 case HOTKEY_MINIMAP_DRAW_TERRAIN:
 574                         preferences::toggle_minimap_draw_terrain();
 575                         executor->recalculate_minimap();
 576                         break;
 577                 case HOTKEY_MINIMAP_CODING_TERRAIN:
 578                         preferences::toggle_minimap_terrain_coding();
 579                         executor->recalculate_minimap();
 580                         break;
 581                 case HOTKEY_MINIMAP_CODING_UNIT:
 582                         preferences::toggle_minimap_movement_coding();
 583                         executor->recalculate_minimap();
 584                         break;
 585                 case HOTKEY_MINIMAP_DRAW_UNITS:
 586                         preferences::toggle_minimap_draw_units();
 587                         executor->recalculate_minimap();
 588                         break;
 589                 case HOTKEY_MINIMAP_DRAW_VILLAGES:
 590                         preferences::toggle_minimap_draw_villages();
 591                         executor->recalculate_minimap();
 592                         break;
 593                 case HOTKEY_FULLSCREEN:
 594                         executor->get_video().set_fullscreen(!preferences::fullscreen());
 595                         break;
 596                 case HOTKEY_SCREENSHOT:
 597                         make_screenshot(_("Screenshot"), executor->get_video(), &::screenshot);
 598                         break;
 599                 case HOTKEY_ANIMATE_MAP:
 600                         preferences::set_animate_map(!preferences::animate_map());
 601                         break;
 602                 case HOTKEY_MOUSE_SCROLL:
 603                         preferences::enable_mouse_scroll(!preferences::mouse_scroll_enabled());
 604                         break;
 605                 case HOTKEY_MUTE:
 606                         {
 607                                 // look if both is not playing
 608                                 static struct before_muted_s
 609                                 {
 610                                         bool playing_sound,playing_music;
 611                                         before_muted_s() : playing_sound(false),playing_music(false){}
 612                                 } before_muted;
 613                                 if (preferences::music_on() || preferences::sound_on())
 614                                 {
 615                                         // then remember settings and mute both
 616                                         before_muted.playing_sound = preferences::sound_on();
 617                                         before_muted.playing_music = preferences::music_on();
 618                                         preferences::set_sound(false);
 619                                         preferences::set_music(false);
 620                                 }
 621                                 else
 622                                 {
 623                                         // then set settings before mute
 624                                         preferences::set_sound(before_muted.playing_sound);
 625                                         preferences::set_music(before_muted.playing_music);
 626                                 }
 627                         }
 628                         break;
 629                 default:
 630                         DBG_G << "command_executor: unknown command number " << command.id << ", ignoring.\n";
 631                         break;
 632         }
 633 }

 635 void command_executor_default::set_button_state()
 636 {
 637         display& disp = get_display();
 638         for (const theme::menu& menu : disp.get_theme().menus()) {
 639 
 640                 std::shared_ptr<gui::button> button = disp.find_menu_button(menu.get_id());
 641                 if (!button) continue;
 642                 bool enabled = false;
 643                 for (const auto& command : menu.items()) {
 644 
 645                         const hotkey::hotkey_command& command_obj = hotkey::get_hotkey_command(command["id"]);
 646                         bool can_execute = can_execute_command(command_obj);
 647                         if (can_execute) {
 648                                 enabled = true;
 649                                 break;
 650                         }
 651                 }
 652                 button->enable(enabled);
 653         }
 654 
 655         for (const theme::action& action : disp.get_theme().actions()) {
 656 
 657                 std::shared_ptr<gui::button> button = disp.find_action_button(action.get_id());
 658                 if (!button) continue;
 659                 bool enabled = false;
 660                 int i = 0;
 661                 for (const std::string& command : action.items()) {
 662 
 663                         const hotkey::hotkey_command& command_obj = hotkey::get_hotkey_command(command);
 664                         std::string tooltip = action.tooltip(i);
 665                         if (filesystem::file_exists(game_config::path + "/images/icons/action/" + command + "_25.png" ))
 666                                 button->set_overlay("icons/action/" + command);
 667                         if (!tooltip.empty())
 668                                 button->set_tooltip_string(tooltip);
 669 
 670                         bool can_execute = can_execute_command(command_obj);
 671                         i++;
 672                         if (!can_execute) continue;
 673                         enabled = true;
 674 
 675                         ACTION_STATE state = get_action_state(command_obj.id, -1);
 676                         switch (state) {
 677                         case ACTION_SELECTED:
 678                         case ACTION_ON:
 679                                 button->set_check(true);
 680                                 break;
 681                         case ACTION_OFF:
