   1 #textdomain wesnoth
   2 # This file contains shortcuts for common WML events (such as prestart, side
   3 # turn, and such), which can be used to write events faster and in less space.
   4 #
   5 # It is recommended that you only use these if you're confident you could write
   6 # the expanded form as well; these are mostly intended as shortcuts for
   7 # experienced WML authors.
   8 
   9 # These don't depend on any other macros.  Please don't change this.
  10 # ! in comments is used in generating HTML documentation, ignore it otherwise.
  11 
  12 #define CALL_FUNCTION EVENT_NAME PARAMETER_WML
  13     # This will fire an event with a set of parameters, for example:
  14     #! {CALL_FUNCTION my_event a,b,c=1,2,3}
  15     #
  16     # That example would fire an event with name "my_event"
  17     # and inside that event, $param.c will be equal to "3"
  18     # This will hold true even if the event is fired recursively
  19     [set_variables]
  20         mode=insert
  21         name=param[0]
  22         [value]
  23             {PARAMETER_WML}
  24         [/value]
  25     [/set_variables]
  26     [fire_event]
  27         name={EVENT_NAME}
  28         [primary_unit]
  29             x=$x1
  30             y=$y1
  31         [/primary_unit]
  32         [secondary_unit]
  33             x=$x2
  34             y=$y2
  35         [/secondary_unit]
  36     [/fire_event]
  37     [clear_variable]
  38         name=param[0]
  39     [/clear_variable]
  40 #enddef
  41 
  42 #define NO_INTERRUPT_NO_UNDO
  43     # This can be used in mid-movement events (enter_hex and exit_hex) to cause
  44     # no interruption of movement, yet prevent the move from being undone.
  45     [allow_undo]
  46     [/allow_undo]
  47     [event]
  48         name=moveto
  49     [/event]
  50 #enddef
