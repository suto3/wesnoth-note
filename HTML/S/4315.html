  19 function wolves_multipacks_functions.assign_packs(cfg)
  20     -- Assign the pack numbers to each wolf. Keeps numbers of existing packs
  21     -- (unless pack size is down to one). Pack number is stored in wolf unit variables
  22     -- Also returns a table with the packs (locations and id's of each wolf in a pack)
  23 
  24     local pack_size = cfg.pack_size or 3
  25 
  26     local wolves = wesnoth.get_units { side = wesnoth.current.side, type = cfg.type or "Wolf" }
  27     local packs = {}
  28 
  29     -- Find wolves that already have a pack number assigned
  30     for _,w in ipairs(wolves) do
  31         local pack = MAIUV.get_mai_unit_variables(w, cfg.ai_id, "pack_number")
  32         if pack then
  33             if (not packs[pack]) then packs[pack] = {} end
  34             table.insert(packs[pack], { x = w.x, y = w.y, id = w.id })
  35         end
  36     end
  37 
  38     -- Remove packs of one
  39     -- Do not change numbers of existing packs -> pack numbers might not be consecutive afterward
  40     for pack_number,pack in pairs(packs) do
  41         if (#pack == 1) then
  42             local wolf = wesnoth.get_unit(pack[1].x, pack[1].y)
  43             MAIUV.delete_mai_unit_variables(wolf, cfg.ai_id)
  44             packs[pack_number] = nil
  45         end
  46     end
  47 
  48     -- Find wolves that are not in a pack (new ones or those removed above)
  49     local nopack_wolves = {}
  50     for _,w in ipairs(wolves) do
  51         local pack_number = MAIUV.get_mai_unit_variables(w, cfg.ai_id, "pack_number")
  52         if (not pack_number) then
  53             table.insert(nopack_wolves, w)
  54         end
  55     end
  56 
  57     -- Now assign the nopack wolves to packs
  58     -- First, go through packs that have less than pack_size members
  59     for pack_number,pack in pairs(packs) do
  60         if (#pack < pack_size) then
  61             local min_dist, best_wolf, best_ind = 9e99
  62             for ind,wolf in ipairs(nopack_wolves) do
  63                 -- Criterion is distance from the first two wolves of the pack
  64                 local dist1 = M.distance_between(wolf.x, wolf.y, pack[1].x, pack[1].y)
  65                 local dist2 = M.distance_between(wolf.x, wolf.y, pack[2].x, pack[2].y)
  66                 if (dist1 + dist2 < min_dist) then
  67                     min_dist = dist1 + dist2
  68                     best_wolf, best_ind = wolf, ind
  69                 end
  70             end
  71             if best_wolf then
  72                 table.insert(packs[pack_number], { x = best_wolf.x, y = best_wolf.y, id = best_wolf.id })
  73                 MAIUV.set_mai_unit_variables(best_wolf, cfg.ai_id, { pack_number = pack_number })
  74                 table.remove(nopack_wolves, best_ind)
  75             end
  76         end
  77     end
  78 
  79     -- Second, group remaining single wolves
  80     -- At the beginning of the scenario, this means all wolves
  81     while (#nopack_wolves > 0) do
  82         -- Find the first available pack number
  83         new_pack_number = 1
  84         while packs[new_pack_number] do new_pack_number = new_pack_number + 1 end
  85 
  86         -- If there are <=pack_size wolves left, that's the pack
  87         if (#nopack_wolves <= pack_size) then
  88             packs[new_pack_number] = {}
  89             for _,w in ipairs(nopack_wolves) do
  90                 table.insert(packs[new_pack_number], { x = w.x, y = w.y, id = w.id })
  91                 MAIUV.set_mai_unit_variables(w, cfg.ai_id, { pack_number = new_pack_number })
  92             end
  93             break
  94         end
  95 
  96         -- If more than pack_size wolves left, find those that are closest together
  97         -- They form the next pack
  98         local new_pack_wolves = {}
  99         while (#new_pack_wolves < pack_size) do
 100             local min_dist, best_wolf, best_wolf_ind = 9e99
 101             for ind,nopack_wolf in ipairs(nopack_wolves) do
 102                 local dist = 0
 103                 for _,pack_wolf in ipairs(new_pack_wolves) do
 104                     dist = dist + M.distance_between(nopack_wolf.x, nopack_wolf.y, pack_wolf.x, pack_wolf.y)
 105                 end
 106                 if dist < min_dist then
 107                     min_dist, best_wolf, best_wolf_ind = dist, nopack_wolf, ind
 108                 end
 109             end
 110             table.insert(new_pack_wolves, best_wolf)
 111             table.remove(nopack_wolves, best_wolf_ind)
 112         end
 113 
 114         -- Now insert the best pack into that 'packs' array
 115         packs[new_pack_number] = {}
 116         for ind = 1,pack_size do
 117             table.insert(
 118                 packs[new_pack_number],
 119                 { x = new_pack_wolves[ind].x, y = new_pack_wolves[ind].y, id = new_pack_wolves[ind].id }
 120             )
 121             MAIUV.set_mai_unit_variables(new_pack_wolves[ind], cfg.ai_id, { pack_number = new_pack_number })
 122         end
 123     end
 124 
 125     -- Put labels out there for all wolves
 126     if cfg.show_pack_number then
 127         for pack_number,pack in pairs(packs) do
 128             for _,wolf in ipairs(pack) do
 129                 wolves_multipacks_functions.put_label(wolf.x, wolf.y, pack_number)
 130             end
 131         end
 132     end
 133 
 134     return packs
 135 end
