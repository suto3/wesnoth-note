  77         class Dummy:
  78             def gettext(self, x):
  79                 if not x: return ""

  81                 if caret < 0: return x
  82                 return x[caret + 1:]

  85     def translate(self, string, textdomain):
  86         if textdomain not in self.catalog:
  87             try:
  88                 self.catalog[textdomain] = gettext.translation(
  89                     textdomain, self.localedir, [self.langcode])
  90                 self.catalog[textdomain].add_fallback(self.dummy)
  91             except IOError:
  92                 self.catalog[textdomain] = self.dummy
  93             except AttributeError:
  94                 self.catalog[textdomain] = self.dummy
  95             except IndexError:
  96                 # not sure why, but this happens within the
  97                 # gettext.translation call sometimes
  98                 self.catalog[textdomain] = self.dummy
  99             except ValueError:
 100                 self.catalog[textdomain] = self.dummy
 101 
 102         r = self.catalog[textdomain].gettext(string)

 181     def analyze_units(self, grouper, add_parents):
 182         """
 183         This takes all units belonging to a campaign, then groups them either
 184         by race or faction, and creates an advancements tree out of it.
 185         """
 186 
 187         # Build an advancement tree forest of all units.
 188         forest = self.forest = helpers.UnitForest()
 189         units_added = {}
 190         for uid, u in list(self.wesnoth.unit_lookup.items()):
 191             if u.hidden: continue
 192             if grouper.unitfilter(u):
 193                 forest.add_node(helpers.UnitNode(u))
 194                 units_added[uid] = u
 195 
 196         #print("    %d/%d units" % (len(units_added), len(self.wesnoth.unit_lookup)))
 197 
 198         # Always add any child units, even if they have been filtered out..
 199         while units_added:
 200             new_units_added = {}
 201             for uid, u in list(units_added.items()):
 202                 for auid in u.advance:
 203                     if not auid in units_added:
 204                         try:
 205                             au = self.wesnoth.unit_lookup[auid]
 206                         except KeyError:
 207                             error_message(
 208                                 "Warning: Unit %s not found as advancement of %s\n" %
 209                                 (auid, repr(uid)))
 210                             continue
 211                         forest.add_node(helpers.UnitNode(au))
 212                         new_units_added[auid] = au
 213             units_added = new_units_added
 214 
 215         if add_parents:
 216             # Also add parent units
 217             added = True
 218             while added:
 219                 added = False
 220                 for uid, u in list(self.wesnoth.unit_lookup.items()):
 221                     if uid in forest.lookup: continue
 222                     for auid in u.advance:
 223                         if auid in forest.lookup:
 224                             forest.add_node(helpers.UnitNode(u))
 225                             added = True
 226                             break
 227 
 228         forest.update()
 229 
 230         # Partition trees by race/faction of first unit.
 231         groups = {}
 232         breadth = 0
 233 
 234         for tree in list(forest.trees.values()):
 235             u = tree.unit
 236             ugroups = grouper.groups(u)
 237 
 238             for group in ugroups:
 239                 groups[group] = groups.get(group, []) + [tree]
 240                 breadth += tree.breadth
 241 
 242         thelist = list(groups.keys())
 243         thelist.sort(key = lambda x: grouper.group_name(x))
 244 
 245         rows_count = breadth + len(thelist)
 246         # Create empty grid.
 247         rows = []
 248         for j in range(rows_count):
 249             column = []
 250             for i in range(6):
 251                 column.append((1, 1, None))
 252             rows.append(column)
 253 
 254         # Sort advancement trees by name of first unit and place into the grid.
 255         def by_name(t):
 256             x = T(t.unit, "name")
 257             if x is None: return ""
 258             return x
 259 
 260         def grid_place(nodes, x):
 261             nodes.sort(key = by_name)
 262             for node in nodes:
 263                 level = node.unit.level
 264                 if level < 0: level = 0
 265                 if level > 5: level = 5
 266                 rows[x][level] = (1, node.breadth, node)
 267                 for i in range(1, node.breadth):
 268                     rows[x + i][level] = (0, 0, node)
 269                 grid_place(node.children, x)
 270                 x += node.breadth
 271             return x
 272 
 273         x = 0
 274         for group in thelist:
 275             node = helpers.GroupNode(group)
 276             node.name = grouper.group_name(group)
 277 
 278             rows[x][0] = (6, 1, node)
 279             for i in range(1, 6):
 280                 rows[x][i] = (0, 0, None)
 281             nodes = groups[group]
 282             x += 1
 283             x = grid_place(nodes, x)
 284 
 285         self.unitgrid = rows
 286         
 287         return len(forest.lookup)

 535     def get_recursive_attacks(self, this_unit):
 536 
 537         def copy_attributes(copy_from, copy_to):
 538             for c in copy_from.data:
 539                 if isinstance(c, wmlparser3.AttributeNode):
 540                     copy_to.data.append(c)
 541 
 542         # Use attacks of base_units as base, if we have one.
 543         base_unit = self.wesnoth.get_base_unit(this_unit)
 544         attacks = []
 545         if base_unit:
 546             attacks = copy.deepcopy(self.get_recursive_attacks(base_unit))
 547 
 548         base_attacks_count = len(attacks)
 549         for i, attack in enumerate(this_unit.get_all(tag="attack")):
 550             # Attack merging is order based.
 551             if i < base_attacks_count:
 552                 copy_attributes(attack, attacks[i])
 553             else:
 554                 attacks.append(attack)
 555 
 556         return attacks
