  84 base_manager::base_manager()
  85 {
  86         event_handler_.join_global();
  87 
  88         try{
  89 #ifdef DEFAULT_PREFS_PATH
  90                 filesystem::scoped_istream stream = filesystem::istream_file(filesystem::get_default_prefs_file(),false);
  91                 read(prefs, *stream);
  92 
  93                 config user_prefs;
  94                 stream = filesystem::istream_file(filesystem::get_prefs_file());
  95                 read(user_prefs, *stream);
  96 
  97                 prefs.merge_with(user_prefs);
  98 #else
  99                 filesystem::scoped_istream stream = filesystem::istream_file(filesystem::get_prefs_file(),false);
 100                 read(prefs, *stream);
 101 #endif
 102         } catch(const config::error& e) {
 103                 ERR_CFG << "Error loading preference, message: "
 104                                 << e.what()
 105                                 << std::endl;
 106         }
 107 }

 123 /*
 124  * Hook for setting window state variables on window resize and maximize
 125  * events. Since there is no fullscreen window event, that setter is called
 126  * from the CVideo function instead.
 127  */
 128 void prefs_event_handler::handle_window_event(const SDL_Event& event)
 129 {
 130 
 131         // Saftey check to make sure this is a window event
 132         if (event.type != SDL_WINDOWEVENT) return;
 133 
 134         switch(event.window.event) {
 135         case SDL_WINDOWEVENT_RESIZED:
 136                 _set_resolution(std::make_pair(event.window.data1,event.window.data2));
 137 
 138                 break;
 139 
 140         case SDL_WINDOWEVENT_MAXIMIZED:
 141                 _set_maximized(true);
 142 
 143                 break;
 144 
 145         case SDL_WINDOWEVENT_RESTORED:
 146                 _set_maximized(fullscreen() || false);
 147 
 148                 break;
 149         }
 150 }

 152 void write_preferences()
 153 {
 154 #ifndef _WIN32
 155     bool prefs_file_existed = access(filesystem::get_prefs_file().c_str(), F_OK) == 0;
 156 #endif
 157 
 158         try {
 159                 filesystem::scoped_ostream prefs_file = filesystem::ostream_file(filesystem::get_prefs_file());
 160                 write(*prefs_file, prefs);
 161         } catch(filesystem::io_exception&) {
 162                 ERR_FS << "error writing to preferences file '" << filesystem::get_prefs_file() << "'" << std::endl;
 163         }
 164 
 165 #ifndef _WIN32
 166     if(!prefs_file_existed) {
 167 
 168         if(chmod(filesystem::get_prefs_file().c_str(), 0600) == -1) {
 169                         ERR_FS << "error setting permissions of preferences file '" << filesystem::get_prefs_file() << "'" << std::endl;
 170         }
 171 
 172     }
 173 #endif
 174 }

 289         if (color.substr(0,4) == "orb_") {

 291                         return color.substr(5);
 292                 } else {
 293                         return color.substr(4);
 294                 }
 295         }
 296         return color;

 369 std::pair<int,int> resolution()
 370 {
 371         const std::string& x = prefs["xresolution"], y = prefs["yresolution"];
 372 
 373         if (!x.empty() && !y.empty()) {
 374                 try {
 375                         return std::make_pair(
 376                                 std::max(std::stoi(x), min_window_width),
 377                                 std::max(std::stoi(y), min_window_height));
 378                 } catch(std::invalid_argument) {}
 379         }
 380 
 381         return std::pair<int,int>(def_window_width, def_window_height);
 382 }

 526                 const char* buf_size = "4096";
 527         #else
 528                 const char* buf_size = "1024";
 529         #endif
 530

 535         preferences::set("sound_buffer_size", new_size);

 605 bool set_turn_bell(bool ison)
 606 {
 607         if(!turn_bell() && ison) {
 608                 preferences::set("turn_bell", true);
 609                 if(!music_on() && !sound_on() && !UI_sound_on()) {
 610                         if(!sound::init_sound()) {
 611                                 preferences::set("turn_bell", false);
 612                                 return false;
 613                         }
 614                 }
 615         } else if(turn_bell() && !ison) {
 616                 preferences::set("turn_bell", false);
 617                 sound::stop_bell();
 618                 if(!music_on() && !sound_on() && !UI_sound_on())
 619                         sound::close_sound();
 620         }
 621         return true;
 622 }

 658 bool set_sound(bool ison) {
 659         if(!sound_on() && ison) {
 660                 preferences::set("sound", true);
 661                 if(!music_on() && !turn_bell() && !UI_sound_on()) {
 662                         if(!sound::init_sound()) {
 663                                 preferences::set("sound", false);
 664                                 return false;
 665                         }
 666                 }
 667         } else if(sound_on() && !ison) {
 668                 preferences::set("sound", false);
 669                 sound::stop_sound();
 670                 if(!music_on() && !turn_bell() && !UI_sound_on())
 671                         sound::close_sound();
 672         }
 673         return true;
 674 }
